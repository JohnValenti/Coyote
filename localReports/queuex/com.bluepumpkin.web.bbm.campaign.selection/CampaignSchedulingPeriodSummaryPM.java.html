package com.bluepumpkin.web.bbm.campaign.selection;

import java.util.ArrayList;

import java.util.Collections;

import com.bluepumpkin.common.datatypes.BasicVO;
import com.bluepumpkin.common.l10n.RegionalFormatBundleKey;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;

/**
 * Title:        CampaignSchedulingPeriodSummaryPM
 * Description:  Displays the expanded Campaign/Scheduling Period selection list
 * Copyright:    Copyright (c) 2009
 * Company:      Verint Systems, Inc.
 * @author       Dustin Stanley
 */
@SuppressWarnings("serial")
public class CampaignSchedulingPeriodSummaryPM extends BasicVOSummaryPM {

	private ResourceBundle m_bundle;
	private String m_selectedCampaignIDs;
	private String m_selectedSchedulingPeriodIDs;
	
	public CampaignSchedulingPeriodSummaryPM(RequestContext context) {
		super(context, "campaign_sp_summary");
		
		m_bundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);
	}

	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, CampaignSchedulingPeriodSummaryPM.class);
	}
	
	protected void initContentTitle() {
		m_contentTitlePC.setPageTitle(i18n(m_bundle, BbmWebBundleKey.CAMPAIGN_SP_SUMMARY));
	}
	
	protected Collection fetchDataList() throws Exception {
		return CampaignModelHandler.getAllCampaignsAndAssociatedSchedulingPeriods(m_context);
	}

	protected Object getRootData() {
		return CampaignSchedulingPeriodTreePC.ROOT_OBJ;
	}

	protected MultiColumnNodeData toNodeData(BasicVO object) {
		if (object==null) return null;

		DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData();
		if (object.getClass() == Campaign.class) {		//handle Campaign object
			Campaign campaign = (Campaign)object;
			if (campaign.equals(CampaignSchedulingPeriodTreePC.ROOT_OBJ)) return null;			//Don't create Node Data for the root object (not a real campaign)
			nodeData.setID(CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX + campaign.getID().toString());
			//Name
			nodeData.add(campaign.getName());
			//Description
			nodeData.add(campaign.getDescription());
			//Time Zone
			nodeData.add(getLocalizer().formatTimeZone(campaign.getTimeZone().getID()));
			//Week Start Day
			nodeData.add(campaign.getWeekStartDay(getLocalizer()));
			//Day Boundary
			nodeData.add(campaign.getDayBoundary(getLocalizer()));	
			
			nodeData.setIcon(campaign.getIcon());		
			nodeData.getNodeAttributes().put(IS_DELETEABLE, Boolean.toString(isAuthorizedToDelete()));
		} else {	//handle Scheduling Period object
			SchedulingPeriod sp = (SchedulingPeriod)object;
			nodeData.setID(CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX + sp.getID().toString());
			//Name
			nodeData.add(getLocalizer().formatDate(sp.getStartTimeLocal(), RegionalFormatBundleKey.DATE_FORMAT) + 
				" - " + getLocalizer().formatDate(sp.getEndTimeLocal(), RegionalFormatBundleKey.DATE_FORMAT));
			//Description
			nodeData.add("");
			//Time Zone
			nodeData.add("");
			//Week Start Day
			nodeData.add("");
			//Day Boundary
			nodeData.add("");

			nodeData.setIcon(sp.getIcon());		
			nodeData.getNodeAttributes().put(IS_DELETEABLE, Boolean.toString(isAuthorizedToDelete()));
		}
		
		return nodeData;
	}

	protected void initHeader() {
		TableHeaderPC header = m_treePC.getHeader();
		header.addColumnHeaderData(null, i18n(m_bundle, BbmWebBundleKey.CAMPAIGN_SP_NAME));
		header.addColumnHeaderData(null, i18n(m_bundle, BbmWebBundleKey.CAMPAIGN_SP_DESCRIPTION));
		header.addColumnHeaderData(null, i18n(m_bundle, BbmWebBundleKey.CAMPAIGN_LABEL_TIME_ZONE));
		header.addColumnHeaderData(null, i18n(m_bundle, BbmWebBundleKey.CAMPAIGN_SP_WEEK_START_DAY));
		header.addColumnHeaderData(null, i18n(m_bundle, BbmWebBundleKey.CAMPAIGN_SP_DAY_BOUNDARY));
	}
	
	protected void initToolbar() {
		String addCampaignLabel = i18n(m_bundle, BbmWebBundleKey.CAMPAIGN_CREATE);
		String addSPLabel = i18n(m_bundle, BbmWebBundleKey.SP_CREATE);
		String editSPLabel = i18n(m_bundle, BbmWebBundleKey.SP_EDIT);
		String deleteLabel = i18n(m_common_bundle, CommonWebBundleKey.TOOLBAR_DELETE);

		m_toolbar.addSubmitTextButton(PageAction.ADD, addCampaignLabel, isAuthorizedToConfigure());
		m_toolbar.addSubmitTextButton(PageAction.ADD, addSPLabel, isAuthorizedToConfigure());
		m_toolbar.addSubmitTextButton(PageAction.EDIT, editSPLabel, isAuthorizedToConfigure());
		m_toolbar.addDeleteTextButton(PageAction.DELETE, deleteLabel, isAuthorizedToDelete());
	}
	
	protected boolean isAuthorizedToConfigure() {
		return true;
	}

	private boolean isAuthorizedToDelete() {
		return true;
	}
	
	public String getHtmlDisplay() {
		return super.getHtmlDisplay();
	}
	
	@Override
	protected void initSelectableItems() {
		List dataList = new ArrayList(m_dataList);
		Collections.sort(dataList, new CampaignSchedulingPeriodComparator(getLocalizer()));
		TreeNodeConverter converter = new SummaryTreeNodeConverter();
		DualTypeTreeNodeHelper treeNodeHelper = new DualTypeTreeNodeHelper(dataList, getRootData(), converter,
				Campaign.class, SchedulingPeriod.class);
		m_treePC.setTreeModel(dataList, treeNodeHelper);

		// Use New Organization ID if available
		String newID = retrieveNewID(m_context);
		if (!StringUtil.isEmpty(newID)) m_selectedID = newID;

		// Set Selected Row
		String[] selectedRow = TreeModelBuilder.getSelectedRowIndex(m_treePC, m_selectedID);
		m_treePC.setPreSelectRows(selectedRow);
	}
	
	//Getters and Setters
	public String getSelectedCampaignIDs() { return m_selectedCampaignIDs; }	
	public void setSelectedCampaignIDs(String selectedCampaignIDs) { m_selectedCampaignIDs = selectedCampaignIDs; }
	
	public String getSelectedSchedulingPeriodIDs() { return m_selectedSchedulingPeriodIDs; }	
	public void setSelectedSchedulingPeriodIDs(String selectedSchedulingPeriodIDs) { m_selectedSchedulingPeriodIDs = selectedSchedulingPeriodIDs; }	
}
