package com.bluepumpkin.web.bbm.campaign.sp;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

/*
 * The modal dialog which is shown by EmployeeSchedulingPeriodSelectorPC. 
 * See EmployeeSchedulingPeriodSelectorPC for details
 */
public class EmployeeSchedulingPeriodSelectorPopupPM extends SchedulingPeriodSelectorPopupPM {

	private static final long serialVersionUID = -7126048442020659224L;

	private String selectedEmployeeIDs;

	public EmployeeSchedulingPeriodSelectorPopupPM(RequestContext context) {
		super(context);
	}


	public static PageModel getInstance(RequestContext context) {
		return getInstance(context, EmployeeSchedulingPeriodSelectorPopupPM.class);
	}

	@Override
	protected String getRequestHandler() {
		return EmployeeSchedulingPeriodSelectorPC.EMPLOYEE_SP_SELECTOR_HANDLER;
	}

	@Override
	protected Collection<SchedulingPeriod> getSchedulingPeriods(Date date, int nNumberToGet) throws RemoteException, BbmException {
		return CampaignModelHandler.getCommonSchedulingPeriods(m_context, getEmployeeIds(), getCampaignID(), date, nNumberToGet);

	}

	private Collection<ID> getEmployeeIds() {
		if (selectedEmployeeIDs != null && selectedEmployeeIDs.isEmpty()) {
			return Collections.emptyList();
		}
		return StringUtil.parseIDStringToCollection(selectedEmployeeIDs, "|");
	}


	public void setSelectedEmployees(String selected) {
		//this is a | separated list of employee IDs. 
		//See EmployeeSchedulingPeriodSelectorPC.addSelectorPopupArg for details on why this is not comma separated values
		selectedEmployeeIDs = selected;
	}

	@Override
	public String getRequiredHTML() {
		return super.getRequiredHTML()
		//store the selected employees. Needed since changing the campaign will cause this page to reload
				+ HtmlUtil.makeHiddenInput(EmployeeSchedulingPeriodSelectorPC.FIELD_NAME_SELECTED_EMPLOYEES, selectedEmployeeIDs);
	}


}
