package com.verint.web.fs.campaign;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.campaign.ejb.CampaignManager;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.core.base.MultiUserException;

import java.rmi.RemoteException;
import java.util.*;

public class SPQueueSaver {

	private final CampaignManager campaignManager;
	private final CombinedLeastTimeIntervalCalculator combinedLeastTimeIntervalCalculator;
	private Map<SPQueue, List<Integer>> leastTimeIntervalsForCombinedSPQueues;

	public SPQueueSaver(CampaignManager campaignManager, CombinedLeastTimeIntervalCalculator combinedLeastTimeIntervalCalculator) {
		this.campaignManager = campaignManager;
		this.combinedLeastTimeIntervalCalculator = combinedLeastTimeIntervalCalculator;
	}

	public void save(CampaignSPQueueslListPM pageModel) throws MultiUserException, RemoteException, BbmUpdateException,
		BbmFinderException {
		Map<SPQueue, Collection<ID>> skillSPQueueMap = pageModel.getSkillForSPQueues();
		leastTimeIntervalsForCombinedSPQueues = new HashMap<>();
		for (SPQueue spqueue : skillSPQueueMap.keySet()) {
			Collection<ID> skillIDs = skillSPQueueMap.get(spqueue);
			spqueue.setSkills(skillIDs);
			campaignManager.updateSPQueue(spqueue);
			mapQueueLeastTimeIntervalToCombinedQueue(spqueue);
		}

		if (pageModel.getCombinedSPQueue() != null) {
			campaignManager.updateSPQueue(pageModel.getCombinedSPQueue());
		} else {
			updateCombinedQueuesLeastTimeIntervals();
		}
	}

	private void mapQueueLeastTimeIntervalToCombinedQueue(SPQueue spqueue) throws BbmFinderException, RemoteException {
		SPQueue combinedQueue = campaignManager.getCombinedSPQueue(spqueue.getSpID(), spqueue.getMediaID());
		List<Integer> combinedQueueLTIs = getLeastTimeIntervalsForCombinedQueue(combinedQueue);
		combinedQueueLTIs.add(spqueue.getLeastTimeInterval());
	}

	private List<Integer> getLeastTimeIntervalsForCombinedQueue(SPQueue combinedQueue) {
		if (leastTimeIntervalsForCombinedSPQueues.get(combinedQueue) == null) {
			leastTimeIntervalsForCombinedSPQueues.put(combinedQueue, new ArrayList<>());
		}
		return leastTimeIntervalsForCombinedSPQueues.get(combinedQueue);
	}

	private void updateCombinedQueuesLeastTimeIntervals() throws MultiUserException, RemoteException, BbmUpdateException {
		for (SPQueue currentCombinedQueue : leastTimeIntervalsForCombinedSPQueues.keySet()) {
			int combinedLeastTimeInterval = combinedLeastTimeIntervalCalculator.
				getCombinedLeastTimeInterval(leastTimeIntervalsForCombinedSPQueues.get(currentCombinedQueue));
			currentCombinedQueue.setLeastTimeInterval(combinedLeastTimeInterval);
			campaignManager.updateSPQueue(currentCombinedQueue);
		}
	}
}
