package com.bluepumpkin.web.am.adherence.exceptioncategory;

import java.text.MessageFormat;
import java.util.*;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.am.rtaaexceptioncategory.model.RTAAExceptionCategory;
import com.bluepumpkin.ejb.bbm.workresource.OrganizationScopeManagerProxy;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.bbm.organization.OrgWorkpaneListPageModel;
import com.bluepumpkin.web.am.adherence.exceptioncategory.*;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.pagemodel.PageModel;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;

/**
 * Title:        ExceptionCategoryListPageModel
 * Description:  Display the list of Exception Categories
 * Copyright:    Copyright (c) 2006
 * Company:      Witness Systems, inc
 * @author       Swati Tewari
 * @version 1.0
 */
public class ExceptionCategoryListPageModel extends OrgWorkpaneListPageModel {
	private static final Category log = Log.initCategory(ExceptionCategoryListPageModel.class.getName());
	private Collection m_exceptionCategoryList;
	private ResourceBundle m_uifBundle;
	private Map m_orgMap = null;

	public static final String NAME_SORT = "NAME_SORT";
	public static final String DESCRIPTION_SORT = "DESCRIPTION_SORT";
	public static final String OWNER_ORG_SORT = "OWNER_ORG_SORT";

	/**
	 * Constructor
	 */
	public ExceptionCategoryListPageModel(RequestContext context) {
		super(context);

		setName("exceptionCategoryList");
		m_uifBundle = m_localizer.getBundle(UIFWebBundleKey.BUNDLE_NAME);
	}

	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static PageModel getInstance(RequestContext context) {
		return getInstance(context, ExceptionCategoryListPageModel.class);
	}

	/**
	 * Loads all the Exception category value objects for the selected Organization ID
	 *
	 * @return true if load was successful
	 */
	private boolean loadExceptionCategories() {
		try {
			m_exceptionCategoryList = ExceptionCategoryModelHandler.getExceptionCategories(m_context, 
					OrganizationModelHandler.getSelfAndParentsOrganizationIDs(m_context, getSelectedIDObject()));
			if(m_exceptionCategoryList != null && m_exceptionCategoryList.size() > 0) {
				HashSet orgIds = new HashSet(m_exceptionCategoryList.size());
				for(Iterator it = m_exceptionCategoryList.iterator(); it.hasNext();) {
					RTAAExceptionCategory exceptionCategory = (RTAAExceptionCategory)it.next();
					
					if(!orgIds.contains(exceptionCategory.getOrganizationID())) {
						orgIds.add(exceptionCategory.getOrganizationID());
					}
				}
				
				m_orgMap = OrganizationModelHandler.buildOrganizationMap(m_context, orgIds);
			}
			return true;
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
			return false;
		}
	}


	/**
	 * Initialize List
	 */
	protected void initList() {
		super.initList();

		// Setup Headers
		TableHeaderPC header = m_list.getHeader();
		header.setPartialSortable(true);
		header.setDefaultSortColumn(NAME_SORT);
		header.addColumnHeaderData(NAME_SORT, i18n(m_uifBundle, UIFWebBundleKey.NAME), true);
		header.addColumnHeaderData(DESCRIPTION_SORT, i18n(m_uifBundle, UIFWebBundleKey.DESCRIPTION), true);
		header.addColumnHeaderData(OWNER_ORG_SORT,	i18n(m_bbmBundle, BbmWebBundleKey.LIST_HEADER_OWNER_ORG), false);

		// Setup basic table property
		m_list.setIsAutoSortEnabled(true);
		m_list.setMultiRowSelectable(false);
	}

	/**
	 * Initialize toolbar
	 */
	protected void initToolbar() {
		String addLabel = "Create New Category";//i18n(m_bbmBundle, BbmWebBundleKey.ORG_HOLIDAY_CREATE_NEW_ACTION);
		String editLabel = "Edit category";//i18n(m_bbmBundle, BbmWebBundleKey.ORG_HOLIDAY_EDIT_ACTION);
		String delLabel = "Delete Category";//i18n(m_bbmBundle, BbmWebBundleKey.ORG_HOLIDAY_DELETE_ACTION);

		useDefaultToolbar(false, addLabel, editLabel, delLabel);
	}

	/**
	 * Create the list model
	 */
	protected void createListModel() {
		if (!loadExceptionCategories()) return;

		ArrayList listModel = new ArrayList();
		for(Iterator it = m_exceptionCategoryList.iterator(); it.hasNext(); ) {
			RTAAExceptionCategory exceptionCategory = (RTAAExceptionCategory)it.next();

			// Create the row data
			String rowID = exceptionCategory.getID().toString();
			DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData(rowID);
			row_data.add(exceptionCategory.getName());
			row_data.add(exceptionCategory.getDescription());
			row_data.add(m_orgMap.get(exceptionCategory.getOrganizationID()));

			// Add node Attributes
			row_data.getNodeAttributes().put(IS_EDITABLE, isRowEditable(exceptionCategory.getOrganizationID()));
			
			// Add row data to list model
			listModel.add(row_data);

			// Check if row should be selected, assuming only one should be selected
			if (OrganizationModelHandler.UNREASONABLE_ORG_ID.equals(m_selectedRowOrgID)) {
				if(this.isRowSelected(exceptionCategory.getID())) {
					m_list.addSelectedRowID(rowID);
					m_selectedRowOrgID = exceptionCategory.getOrganizationID();
				}
			}
		} //end for
		
		m_list.setListModel(listModel);
	}

	/**
	 * Returns true if the list support sorting.
	 * This method should be override by derived classes who support sorting
	 */
	protected boolean isSortSupported() {
		return true;
	}

	/**
	 * Return the Selected Holiday IDs
	 */
	public ID[] getSelectedHolidayIDs() {
		return m_performActionOnIDs;
	}

	/**
	 * Get entity name. Used for title.
	 */
	protected String getTitleEntityType() {
		return "Exception Categories";//i18n(m_bbmBundle, BbmWebBundleKey.ORG_HOLIDAY_LABEL);
	}

	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return "org_exceptioncategory";
	}

	/**
	 * Return true if Current User is authorized to configure specified Organization
	 */
	protected boolean isAuthorizedToView(ID orgID) {
		return m_context.getUser().isAuthorized(PrivilegeKeys.CORE_VIEWHOLIDAYS,
				OrganizationScopeManagerProxy.getScopeID(orgID));
	}

	/**
	 * Return true if Current User is authorized to configure specified Organization
	 */
	protected boolean isAuthorizedToConfigure(ID orgID) {
		return m_context.getUser().isAuthorized(PrivilegeKeys.CORE_CONFIGUREHOLIDAYS,
				OrganizationScopeManagerProxy.getScopeID(orgID));
	}

	/**
	 * Return localized message indicating that current user is not authorized
	 * to view activities in the selected Organization.
	 */
	protected String getNotAuthorizedToViewMsg() {
		return MessageFormat.format(
			i18n(m_bbmBundle, BbmWebBundleKey.NOT_AUTHORIZED_TO_VIEW_ENTITY_IN_SELECTED_ORG),
			new Object[] { i18n(m_bbmBundle, BbmWebBundleKey.BBM_ORG_GEN_HOLIDAYS) } );
	}
}
