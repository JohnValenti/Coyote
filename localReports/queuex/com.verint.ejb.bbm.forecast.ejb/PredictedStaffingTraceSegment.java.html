package com.verint.ejb.bbm.forecast.ejb;

import com.bluepumpkin.ejb.bbm.timeseries.model.PredictedAllocationTimeSeries;
import com.bluepumpkin.ejb.bbm.timeseries.model.PredictedAllocationTimeSeriesFieldInfo;

/**
 * Trace segment wrapper for predicted staffing data.
 */
public class PredictedStaffingTraceSegment extends AbstractTraceSegment<PredictedAllocationTimeSeries> {

	public PredictedStaffingTraceSegment(PredictedAllocationTimeSeries valueObject) {
		super(
				valueObject, 
				PredictedAllocationTimeSeriesFieldInfo.Column.DESTINATIONSPQUEUEID.getIndex(),
				PredictedAllocationTimeSeriesFieldInfo.Column.TIMESERIESDATE.getIndex());
	}

	/**
	 * Returns predicted staffing time series data.
	 */
	@Override
	public double[] getTraceData() {
		return m_valueObject.getStaffSeries();
	}
}
