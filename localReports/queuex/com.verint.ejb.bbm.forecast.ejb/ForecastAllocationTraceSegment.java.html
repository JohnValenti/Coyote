package com.verint.ejb.bbm.forecast.ejb;

import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastAllocationTimeSeries;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastAllocationTimeSeriesFieldInfo;

/**
 * Trace segment wrapper for forecast allocation data.
 */
public class ForecastAllocationTraceSegment extends AbstractTraceSegment<ForecastAllocationTimeSeries> {

	public ForecastAllocationTraceSegment(ForecastAllocationTimeSeries valueObject) {
		super(
				valueObject,
				ForecastAllocationTimeSeriesFieldInfo.Column.DESTINATIONSPQUEUEID.getIndex(),
				ForecastAllocationTimeSeriesFieldInfo.Column.DATETIME.getIndex());
	}

	/**
	 * Returns forecast allocation time series data.
	 */
	@Override
	public double[] getTraceData() {
		return m_valueObject.getAllocationTimeSeries();
	}
}
