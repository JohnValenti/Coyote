package com.bluepumpkin.ejb.bbm.activity;

import java.rmi.RemoteException;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.activity.ejb.ActivityManager;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbBundleKey;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.ejb.bbm.copy.CopyableObjectPlugin;
import com.witness.ejb.bbm.copy.ejb.CopyDAO;
import com.witness.ejb.bbm.copy.model.CopyResult;
import com.witness.ejb.core.security.PrivilegeKeys;

/**
 * @author dsu
 * @version 1.0
 * @Title: ActiviyCopyPlugin
 * @Description:
 * @Copyright: Copyright (c) Aug 15, 2006
 * @Company: Witness Systems
 */
public class ActivityCategoryCopyPlugin extends CopyableObjectPlugin {

  /**
   *
   */
  private static final long serialVersionUID = 1L;

  public CopyResult copy(ID idActivityCategory, ID idOrg, boolean bForce, CopyDAO pDAO, String sCopyPrefix)
      throws BbmFinderException, BbmCreateException {
    try {
      ActivityManager pActivityManager = WfmManagerFactory.getActivityManager();
      // get the activity category to be copied
      ActivityCategory pCat = pActivityManager.findActivityCategoryById(idActivityCategory);

      // rename and copy the activity cat
      ActivityCopyUtil.createCopyActivityCategory(pActivityManager, pCat, idOrg, pDAO, sCopyPrefix);

      // return CopyResult COPIED
      return new CopyResult();
    } catch (BbmEJBCreateException e) {
      throw new BbmCreateException(e, "could not get Activities");
    } catch (RemoteException e) {
      throw new BbmCreateException(e, "could not get Activities");
    }
  }

  /*
   * (non-Javadoc)
   *
   * @see
   * com.witness.ejb.bbm.copy.CopyableObjectPlugin#isMultiTargetCopyAllowed()
   */
  public boolean isMultiTargetCopyAllowed() {
    return false;
  }

  /*
   * (non-Javadoc)
   *
   * @see
   * com.witness.ejb.bbm.GenericObjectPlugin#convertType(com.bluepumpkin.common
   * .localization.Localizer, boolean)
   */
  public String convertType(Localizer localizer, boolean isPlural) {
    String rbKey = isPlural ? BbmEjbBundleKey.OBJECT_ACTIVITY_TYPE_PLURAL : BbmEjbBundleKey.OBJECT_ACTIVITY_TYPE;
    return localizer.i18n(BbmEjbBundleKey.BUNDLE_NAME, rbKey);
  }

  /*
   * (non-Javadoc)
   *
   * @see com.witness.ejb.bbm.GenericObjectPlugin#getModificationPrivilegeID()
   */
  public ID getModificationPrivilegeID() {
    return PrivilegeKeys.CORE_VIEWACTIVITIES_ID;
  }
}
