package com.verint.wfm.ws.entityImport.campaign;

import java.util.List;

import com.verint.wfm.ws.entityImport.EntityNotPersistedInfo;

/**
 * For each campaign not persisted by createOrUpdateCampaigns(), one of these will be in the list returned This is
 * returned in lieu of an exception, because we always want to continue on to all the campaigns
 * 
 * 
 * @see EntityImportService.createOrUpdateCampaigns()
 * 
 */
public class CampaignNotPersistedInfo extends EntityNotPersistedInfo<String> {
	private static final long serialVersionUID = 1L;

	public CampaignNotPersistedInfo() {
		super();
	}

	public CampaignNotPersistedInfo(String campaignExternalSourceId) {
		super(campaignExternalSourceId);
	}

	// -----------------------------------------
	public String getCampaignExternalSourceId() {
		return super.getEntityExternalSourceId();
	}

	/**
	 * Dummy here on server, but needed for SOAP stub
	 */
	public void setCampaignExternalSourceId(String campaignExternalSourceId) {
		// do nothing
	}

	public void addCampaignServerExceptionMessage(String exception) {
		getMessages()
				.add(String.format("Exception is %s. The server could not persist campaign with external id %s.",
						exception,
						getExternalIdSummary()));
	}

	public void addExternalIdMissingMessage(String campaignName) {
		getMessages().add(String.format("External id was missing for campaign with name %s.", campaignName));
	}

	public void addExternalIdLengthExceededMessage(int actualLength, int lengthLimit) {
		getMessages().add(
				String.format("The external id %s length %d exceeded the limit of %d.", getExternalIdSummary(),
						actualLength, lengthLimit));
	}

	public void addNameMissingMessage() {
		getMessages().add(String.format("Name was missing for campaign with external id %s.", getExternalIdSummary()));
	}

	public void addNameLengthExceededMessage(int actualLength, int lengthLimit) {
		getMessages().add(
				String.format("For campaign with external id %s the name length %d exceeded the limit of %d.",
						getExternalIdSummary(), actualLength, lengthLimit));
	}
	
	public void addNameHasInvalidCharactersMessage(String invalidChars) {
		getMessages()
				.add(String
						.format("The characters %s are not allowed in the campaign name. The campaign %s has invalid characters in the name.",
								invalidChars, getExternalIdSummary()));
	}

	public void addNameNotUniqueMessage(String ambiguousCampaignWFOId, String name) {
		getMessages().add(
				String.format(
						"For campaign with external id %s, name %s conflicted with another campaign with wfo sid %d.",
						getExternalIdSummary(), name, ambiguousCampaignWFOId));
	}

	public void addWeekStartMessage(int weekStart) {
		getMessages().add(
				String.format("For campaign with external id %s, week start %d is not between 1 and 7.",
						getExternalIdSummary(), weekStart));
	}

	public String getWeekStartUpdateForbiddenMessage(String newWeekStart, String oldWeekStart) {
		return String.format(
				"For campaign with external id %s, week start may not be changed from %s to %s because the campaign contains scheduling periods",
				getExternalIdSummary(), oldWeekStart, newWeekStart);
	}

	public void addTimeZoneMissingMessage() {
		getMessages().add(
				String.format("For campaign with external id %s, time Zone was missing.", getExternalIdSummary()));
	}
	
	public void addTimeZoneInvalidMessage(String timezoneId) {
		getMessages().add(
				String.format("For campaign with external id %s, Time Zone specified %s is not valid/supported.", getExternalIdSummary(), timezoneId));
	}

	public String getTimeZoneUpdateForbiddenMessage(String newTimeZone, String oldTimeZone) {
		return String.format(
				"For campaign with external id %s, time zone may not be changed from %s to %s because the campaign contains " +
						"scheduling periods",
				getExternalIdSummary(), oldTimeZone, newTimeZone);
	}

	public void addNoSuchOrganizatonExternalIdMessage(String organizationExternalId) {
		getMessages()
				.add(String
						.format("For campaign with external id %s, the organization specified to have external id %s does not exist.",
								getExternalIdSummary(), organizationExternalId));
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public List<String> getMessages() {
		return super.getMessages();
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public void setMessages(List<String> messages) {
		// Do nothing
	}

}
