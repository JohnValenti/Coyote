package com.bluepumpkin.web.bbm.activity.tree;

import java.util.*;

import javax.swing.tree.*;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.bbm.activity.ActivityTypeModelHandler;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.bluepumpkin.web.core.pagecomponent.tree.PopupTreePC;
import com.witness.web.uif.system.RequestContext;

import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;

/**
 * Title:        ActivityTypeTreePC
 * Description:  Activity Type Tree Page Component
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author       Alvin Cham
 * @version 1.0
 */
public class ActivityTypeTreePC extends PopupTreePC {
	/**
	 * Constructor
	 */
	public ActivityTypeTreePC(RequestContext context) {
		super(context);
		setBundle(m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME));
		// Set default Header
		setHeaderTitle(i18n(m_bundle, BbmWebBundleKey.BBM_ORG_ACT_CATEGORIES));
		setName("ActivityTypeTree");
	}

	/**
	 * Retrieve all viewable activities.
	 */
	protected Collection retrieveTreeObjects() throws Exception {
		return ActivityTypeModelHandler.getAllActivityTypesVisibleToUser(m_context, getUser());
	}

	/**
	 *  Re-visit: may only restrict to superuser.
	 */
	protected boolean isAuthorizedToViewObject(ID id) {
		return true;  // since we only returned the activity types visible to user.
	}

	/**
	 * Return Input UI as a HTML code
	 */
	protected String makeInputUI(String type, Object treeObject, String onClickJS) {
		ActivityCategory actType = (ActivityCategory)treeObject;
		String id = actType.getID().toString();
		boolean isChecked = isSelectedField(actType.getID());

		String nodeID = (id!=null)? id :"";
		String label = actType.getName();
		String miscAttrib = isChecked?"checked":"";

		if (!isAuthorizedToViewObject(new ID(id))) {
 			miscAttrib += " disabled";
 		}

		Object[] args = new Object[] {type, CHECKED_ID_FN, nodeID, "", label, onClickJS, miscAttrib};
		return INPUT_MF.format(args);
	}

	/**
	 * Build Tree Model
	 */
	public TreeModel buildTree() {
		Collection cActivityTypes = getTreeObjects();
		ActivityCategory rootElement = new ActivityCategory();
		rootElement.setName("rootActivityType");
		rootElement.setID(new ID(ActivityCategory.SYSTEM_DEFINED));
		MutableTreeNode rootNode = toTreeNode(rootElement);

		DefaultTreeModel treeModel = new DefaultTreeModel(rootNode);
		for (Iterator actIter = cActivityTypes.iterator(); actIter.hasNext();) {
			ActivityCategory actType = (ActivityCategory)actIter.next();

			MutableTreeNode actNode = (MutableTreeNode)toTreeNode(actType);
			treeModel.insertNodeInto(actNode, rootNode, rootNode.getChildCount());
		}

		return treeModel;
	}

	/**
	 * Convert Tree Object to MutableTreeNode
	 */
	protected MutableTreeNode toTreeNode(Object treeObject) {
		ActivityCategory act = (ActivityCategory)treeObject;
		if (act==null) return null;

		// Create Node Data
		String actIDStr = act.getID().toString();
		DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(actIDStr);

		if (isInputSelectorDisplayed()) {
			String inputType = isMultiRowSelectable()?"checkbox":"radio";
			String inputUI = makeInputUI(inputType, act, getOnClickJS());
			nodeData.add(inputUI, "width:5px");
		}
		nodeData.add(act.getName());
		return new DefaultMutableTreeNode(nodeData);
	}
}
