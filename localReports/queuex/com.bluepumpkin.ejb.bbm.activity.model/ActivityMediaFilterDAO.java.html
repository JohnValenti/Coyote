package com.bluepumpkin.ejb.bbm.activity.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.Filter.Filter;
import com.bluepumpkin.ejb.bbm.Filter.FilterCondition;
import com.bluepumpkin.ejb.bbm.Filter.FilterDAO;
import com.bluepumpkin.ejb.bbm.dao.DAOUtil;

public class ActivityMediaFilterDAO extends FilterDAO {

	public ActivityMediaFilterDAO(Filter objActivityMediaFilter) {
		super(objActivityMediaFilter);
	}

	public FilterDAO createFilterDAO(Filter objFilter) {
		return new ActivityMediaFilterDAO(objFilter);
	}

	public void processFilterObject() {
		FilterCondition fc = ((ActivityMediaFilter) m_objFilter).m_fcParentID;
		if (fc == null)
			return;

		ID id = (ID) fc.m_objValue;

		Jdmo dmo = null;

		try {
			dmo = new Jdmo(false);
			StringBuffer strFilterText = DAOUtil.getChildOrganizationsBuffer(id, dmo);
			String strTemp = (strFilterText == null ? "NULL" : new String(strFilterText));
			m_objFilter.setUserDefinedProperty("PID", "ID", ActivityMediaFilter.OP_IN_LIST, strTemp);
		} catch (Exception e) {
			e.getMessage();
		} finally {
			if (dmo != null) {
				dmo.cleanUp();
				dmo = null;
			}
		}
	}

}
