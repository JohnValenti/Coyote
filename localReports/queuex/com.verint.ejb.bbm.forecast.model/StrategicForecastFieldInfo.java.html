package com.verint.ejb.bbm.forecast.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

/**
 * Provides DB field information for the StrategicForecast table.  This table's
 * primary key is composed of {@code TIME} and {@code QUEUEID}.  This table
 * does not use {@code ID} for its primary key, nor is the {@code ID} field
 * even necessarily unique, so it should not be used to retrieve records.
 */
public class StrategicForecastFieldInfo extends FieldInfo {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;

	public static enum Column {
		/** {@code TIME} and {@code QUEUEID} together form the primary key for this table. */
		TIME(new FieldInfoEntry("TIME", PrimitiveTypes.DATETIME)),
		/** {@code TIME} and {@code QUEUEID} together form the primary key for this table. */
		QUEUEID(new FieldInfoEntry("QUEUEID", PrimitiveTypes.ID)),
		/** Warning! {@code ID} is not a primary key in this table, nor is it necessarily unique.  It is basically useless. */
		ID(new FieldInfoEntry("ID", PrimitiveTypes.ID)),
		AHT(new FieldInfoEntry("AHT", PrimitiveTypes.INT)),
		CALLVOLUME(new FieldInfoEntry("CALLVOLUME", PrimitiveTypes.INT)),
		;
		private int index;
		private FieldInfoEntry fie;
		private Column(FieldInfoEntry fie) {
			this.fie = fie;
		}
		public FieldInfoEntry getFieldInfoEntry() {
			return this.fie;
		}
		public int getIndex() {
			return index;
		}
		static {
			int ctr = 0;
			for (Column fie : Column.values()) fie.index = ctr++;
		}
	}
	
	private static final FieldInfoEntry[] fieldInfoEntries = new FieldInfoEntry[Column.values().length];
	static {
		for (Column c : Column.values()) {
			fieldInfoEntries[c.getIndex()] = c.getFieldInfoEntry();
		}
	}
	
	@Override
	public FieldInfoEntry[] getFieldInfo() {
		return fieldInfoEntries;
	}

	/**
	 * For lack of a better alternative we return the {@code ID} field.  The
	 * {@code ID} field is not necessarily unique in the StrategicForecast table
	 * so it should not be used to retrieve records.  Use {@code TIME} and
	 * {@code QUEUEID} instead.
	 */
	@Override
	public int getIDFieldIndex() {
		return Column.ID.getIndex();
	}

	@Override
	public String getTableName() {
		return "STRATEGICFORECAST";
	}

}
