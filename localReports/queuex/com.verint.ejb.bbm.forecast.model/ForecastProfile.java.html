package com.verint.ejb.bbm.forecast.model;

import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectRich;

public class ForecastProfile extends ValueObjectRich {

	// meta data information
	private static ForecastProfileFieldInfo m_fieldInfo = new ForecastProfileFieldInfo();
	
	private static final long serialVersionUID = -5040370361636359132L;
	
	public ForecastProfile() {
		super();
		setIsModified(false);
	}

	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	public ID getID() {
		return getFieldValueID(ForecastProfileFieldInfo.FORECASTPROFILE_SID);
	}
	
	public ID getStringID() {
		return getFieldValueID(ForecastProfileFieldInfo.FORECASTPROFILE_ID);
	}
	
	public void setID(ID id) {
		setFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_SID, id);
	}
	public ID getForecastProfileID() {
		return getFieldValueID(ForecastProfileFieldInfo.FORECASTPROFILE_ID);
	}	
	
	public String getName() {
		return getFormattedFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_NAME);
	}
	
	public void setName(String name) { 
		setFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_NAME, name); 
	}
	
	public boolean isModified() {
		return this.getFieldValueBoolean(ForecastProfileFieldInfo.FORECASTPROFILE_ISMODIFIED);
	}
	
	public void setIsModified(boolean modified) { 
		setFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_ISMODIFIED, modified); 
	}
	
	public String getDescription() {
		return this.getFormattedFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_DESCRIPTION);
	}
	
	public void setDescription(String description) {
		setFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_DESCRIPTION, description);
	}
	
	public Date getCreatedDate() {
		return getFieldValueDate(ForecastProfileFieldInfo.FORECASTPROFILE_CREATED);
	}
	public void setCreatedDate(Date createdDate) {
		setFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_CREATED, createdDate);
	}
	
	public String getCreatedBy() {

	}
	
	public void setCreatedBy(String createdBy) {
		setFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_CREATEDBY, createdBy);
	}
	
	public Date getModifiedDate() {
		return getFieldValueDate(ForecastProfileFieldInfo.FORECASTPROFILE_MODIFIED);
	}
	
	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_MODIFIED, modifiedDate);
	}
	
	public String getModifiedBy() {
		return getFormattedFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_MODIFIEDBY);
	}
	
	public void setModifiedBy(String modifiedBy) {
		setFieldValue(ForecastProfileFieldInfo.FORECASTPROFILE_MODIFIEDBY, modifiedBy);

		return getFieldValueID(ForecastProfileFieldInfo.FORECASTPROFILE_SPQUEUEID);

		if (!spQueueID.isInt()) throw new IllegalArgumentException("SPQueueID must be an int-based ID");

		removeRelatedObjects(ForecastProfileFieldInfo.PROFILECOMPONENT_RELATIONSHIP_KEY, profileComponentsToRemove);

		setRelatedObjects(ForecastProfileFieldInfo.PROFILECOMPONENT_RELATIONSHIP_KEY, profileComponents);

	/**
	 * Returns a cloned copy of this ForecastProfile with the ID set to null.
	 */
	public ForecastProfile cloneForecastProfileValueFields() throws CloneNotSupportedException {
		ForecastProfile clonedProfile = (ForecastProfile)super.clone();
		clonedProfile.setFieldNull(ForecastProfileFieldInfo.FORECASTPROFILE_ID);
		clonedProfile.setFieldNull(ForecastProfileFieldInfo.FORECASTPROFILE_SID);
		return clonedProfile;
	}
	
	/**
	 * Compares the sp queue and start date of this ForecastProfile with another to see if they are equivalent.
	 */
	public boolean areSPQueuesAndStartDatesEqual(ForecastProfile profileToCompare) {
		if ((getSPQueueID() == null && profileToCompare.getSPQueueID() == null ||
				(getSPQueueID() != null && getSPQueueID().equals(profileToCompare.getSPQueueID()))) &&
			(getStartDate() == null && profileToCompare.getStartDate() == null ||
					(getStartDate() != null && getStartDate().equals(profileToCompare.getStartDate())))) {
			return true;
		}	
		return false;
	}
}
