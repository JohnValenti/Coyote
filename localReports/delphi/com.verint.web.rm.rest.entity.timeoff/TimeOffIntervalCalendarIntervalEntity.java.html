package com.verint.web.rm.rest.entity.timeoff;

import java.util.ArrayList;
import java.util.List;

import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendar;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendarIncrement;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendarInterval;
import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rest.util.FormatUtils;

@XmlRootElement(name = "interval")
public class TimeOffIntervalCalendarIntervalEntity extends AbstractWFOEntity {

	private final float allocatedHours;
	private final float approvedHours;
	private final float availableHours;
	private final String endDate;
	private final List<Integer> increments;
	private final float pendingHours;
	private final String startDate;

	public TimeOffIntervalCalendarIntervalEntity(TOIntervalCalendar toic, TOIntervalCalendarInterval interval, boolean isTOPoolView) {
		boolean forEmployee = !isTOPoolView;
		allocatedHours = interval.getAllocatedHours();
		approvedHours = interval.getApprovedHours(forEmployee);
		availableHours = interval.getAvailableHours(forEmployee);
		endDate = FormatUtils.dateToISO8601SecPrecisionString(interval.getEndTime());
		pendingHours = interval.getPendingHours(forEmployee);
		startDate = FormatUtils.dateToISO8601SecPrecisionString(interval.getStartTime());

		List<TOIntervalCalendarIncrement> toicIncrements = toic.getIncrements();
		int startIndex = toic.getIndex(interval.getStartTime());
		int endIndexExclusive = toic.getIndex(interval.getEndTime());

		increments = new ArrayList<Integer>(endIndexExclusive - startIndex);
		for (int index = startIndex; index < endIndexExclusive; index++) {
			TOIntervalCalendarIncrement toicIncrement = toicIncrements.get(index);
			increments.add(toicIncrement.getType());
		}
	}

	public float getAllocatedHours() {
		return allocatedHours;
	}

	public float getApprovedHours() {
		return approvedHours;
	}

	public float getAvailableHours() {
		return availableHours;
	}

	public String getEndDate() {
		return endDate;
	}

	public List<Integer> getIncrements() {
		return increments;
	}

	public float getPendingHours() {
		return pendingHours;
	}

	public String getStartDate() {
		return startDate;
	}
}
