package com.bluepumpkin.ejb.rm.notification.model;

import java.io.Serializable;

import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.bluepumpkin.common.datatypes.ID;

/**
 * This class encapsulates all the necessary details to send a request status
 * change notification (notiification when the request is approved, denied,
 * pending etc).
 * <p/>
 * This class lives in BBM, rather than RM, because of deserialization problems
 * in the Notification Handler message driven bean. When an instance of this
 * class, if defined in RM, is passed by RM to BBM, BBM's notification handler
 * MDB genearates an "Unable to find class for RequestNotifiationDetail"
 * exception. Adding RmEjbTierUtils.jar to notificationEjb.jar's manifest class
 * path fixes this.
 * <p/>
 * See class
 * {@link com.bluepumpkin.ejb.bbm.notifyrules.rmrulecheckers.RequestStatusChangeRuleChecker
 * RequestStatusChangeRuleChecker} for more details.
 * <p/>
 * Title: Blue Pumpkin Software Request Management Module Copyright: Copyright
 * (c) 2001-2004 Company: Blue Pumpkin Software, inc
 *
 * @author Raja Rajendran
 */
public class RequestNotificationDetail implements Serializable {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private String m_newRequestStatus;
	private RequestAggregate m_reqAgg;

	private ID employeeID;

	/**
	 * @return
	 */
	public RequestAggregate getReqAgg() {
		return m_reqAgg;
	}

	/**
	 * @return
	 */
	public String getNewRequestStatus() {
		return m_newRequestStatus;
	}

	/**
	 * @param reqAgg
	 * @param newReqStatus
	 *            the new status to which the given request was transitioned to.
	 */
	public RequestNotificationDetail(RequestAggregate reqAgg, String newReqStatus) {
		m_newRequestStatus = newReqStatus;
		m_reqAgg = reqAgg;
	}

	public RequestNotificationDetail(RequestAggregate reqAgg, String newReqStatus, ID empID) {
		this.employeeID = empID;
		m_newRequestStatus = newReqStatus;
		m_reqAgg = reqAgg;
	}

	public ID getEmployeeID() {
		return employeeID;
	}

	public void setEmployeeID(ID employeeID) {
		this.employeeID = employeeID;
	}
}
