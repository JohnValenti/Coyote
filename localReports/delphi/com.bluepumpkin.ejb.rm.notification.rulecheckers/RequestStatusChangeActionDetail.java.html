package com.bluepumpkin.ejb.rm.notification.rulecheckers;

import java.util.HashMap;
import java.util.Map;

import javax.xml.parsers.DocumentBuilder;

import org.w3c.dom.Document;
import org.w3c.dom.Element;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.notifyrules.model.ActionDetail;
import com.bluepumpkin.ejb.rm.notification.util.NotificationUtil;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.witness.ejb.core.security.UserManager;
import com.witness.ejb.core.security.model.User;

public class RequestStatusChangeActionDetail extends ActionDetail {
	DocumentBuilder m_parser = null;
	RequestAggregate m_reqAgg = null;
	Localizer m_localizer = null;
	Map m_empIDToTZMap = null;
	String m_requestStatus = null;
	UserManager m_userManager = null;
	
	public RequestStatusChangeActionDetail() {	}
	
	public RequestStatusChangeActionDetail (DocumentBuilder parser, RequestAggregate reqAgg, Localizer localizer,
			Map empIDToTZMap, String requestStatus, UserManager userManager) {
		m_parser = parser;
		m_reqAgg= reqAgg;
		m_localizer = localizer;
		m_empIDToTZMap = empIDToTZMap;
		m_requestStatus = requestStatus;
		m_userManager = userManager;
		
	}
	
	public Object clone() throws CloneNotSupportedException {
		RequestStatusChangeActionDetail cloned = (RequestStatusChangeActionDetail)super.clone();
		
		cloned.m_parser = this.m_parser;
		cloned.m_reqAgg = this.m_reqAgg;
		cloned.m_localizer = this.m_localizer;
		cloned.m_empIDToTZMap = this.m_empIDToTZMap;
		cloned.m_requestStatus = this.m_requestStatus;
		cloned.m_userManager= this.m_userManager;
		
		return cloned;
	}
	
	public ActionDetail getNewInstance() {
		return new RequestStatusChangeActionDetail();
	}

	/**
	 * should return xml string for the specific superviosr
	 * @param supervisorID
	 * @return
	 */
	public String getCustomizedSupervisorMessage(ID supervisorID) throws Exception {		 
		//create header and add to that
		Document doc = m_parser.newDocument();
		Element existingActionDetailForSupervisor = doc.createElement("requeststatuschange");
		doc.appendChild(existingActionDetailForSupervisor);
		
		Element thisDetail = NotificationUtil.getXMLLocalizedFrag(existingActionDetailForSupervisor, m_reqAgg, null, m_localizer, 
           	m_empIDToTZMap, m_requestStatus);
		
		User supervisor = m_userManager.getUserByEmployeeID(supervisorID);
		if (supervisor != null && NotificationUtil.isAuthorized(supervisor, m_reqAgg)) {
			   // append employee's action detail to supervisor's (with supervisor's privileges)	        				   
			   NotificationUtil.createAndAppendXMLLocalizedFragForRequestURL(thisDetail, m_reqAgg, true);
		   } 
		
		return xmlToText(existingActionDetailForSupervisor);
	}
	
	/**
	 * should return xml string for this employee who has role
	 * @param roleID
	 * @param empIDHasRoles
	 * @return
	 */
	public String getCustomizedRoleMessage(ID roleID, ID empIDHasRole)  throws Exception {
		return getCustomizedSupervisorMessage(empIDHasRole);
	}
	
	/**
	 * should return xml string for this user id
	 * @param userid
	 * @return
	 */
	public String getCustomizedUserIDMessage(ID workresourceID)  throws Exception {
		return getCustomizedSupervisorMessage(workresourceID);
	}
	
	
	public String getCustomizedSpecialTargetMessage(String specialTarget)  throws Exception {
		Document doc = m_parser.newDocument();
		Element existingActionDetailForSupervisor = doc.createElement("requeststatuschange");
		doc.appendChild(existingActionDetailForSupervisor);
		
		Element thisDetail = NotificationUtil.getXMLLocalizedFrag(existingActionDetailForSupervisor, m_reqAgg, null, m_localizer, 
           	m_empIDToTZMap, m_requestStatus);
		             		
		return xmlToText(existingActionDetailForSupervisor);
	}

}
