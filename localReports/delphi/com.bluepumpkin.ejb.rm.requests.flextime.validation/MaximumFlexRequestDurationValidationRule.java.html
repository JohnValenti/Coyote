package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;

public class MaximumFlexRequestDurationValidationRule extends FlexRequestDurationValidationRule {
	public static final String MAX_FLEX_REQUEST_DURATION_VALUE = "MAX_FLEX_REQUEST_DURATION_VALUE";

	public static final String MAX_FLEX_REQUEST_DURATION_UNIT = "MAX_FLEX_REQUEST_DURATION_UNIT";

	public static final ID RULE_ID = new ID(-194063);

	@Override
	protected void init(TORequest toRequest) throws Exception {
		//only one TOC per flex request
		this.toChoice = (TOChoice)toRequest.getRequestChoiceList().iterator().next();
		this.orgID = getEmployeeOrgId(toRequest.getEmployeeID(), toChoice);

		@SuppressWarnings("rawtypes")
		Map ruleParams = getValidationRuleManager().getValidationRuleParams(orgID, RULE_ID);
		this.duration = ((Integer) ruleParams.get(MAX_FLEX_REQUEST_DURATION_VALUE)).intValue();
		this.unit = ((Integer) ruleParams.get(MAX_FLEX_REQUEST_DURATION_UNIT)).intValue();
	}

	@Override
	protected boolean isOverLimit(long durationLimit, long accountedSeconds) {
		return accountedSeconds > durationLimit;
	}

	@Override
	protected String getClassName() {
		return MaximumFlexRequestDurationValidationRule.class.getName();
	}

	@Override
	protected String getBundleKey() {
		return RmEjbBundleKey.FLEX_REQ_CHOICE_MAX_DURATION;
	}
}
