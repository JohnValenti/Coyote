/**
 * This class implement the validation rule: Maximum shift duration allowed ___
 * minutes per "Organization Day" or "24 Hour Period"
 *
 */
public class MaxShiftDurationOfFlexRequestPerDayRule implements Validator {
	private static final String CLASS_NAME = MaxShiftDurationOfFlexRequestPerDayRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	public static final String FT_MAX_SHIFT_DURATION_VALUE = "FT_MAX_SHIFT_DURATION_VALUE";
	public static final String FT_MAX_SHIFT_DURATION_UNIT = "FT_MAX_SHIFT_DURATION_UNIT";
	public static final int MAX_SHIFT_DURATION_UNIT_ORGHOO = 0;
	public static final int MAX_SHIFT_DURATION_UNIT_24HOURS = 1;
	public static final ID RULE_ID = new ID(-194074);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod("validate", validatable));
		}

		TORequest toRequest = (TORequest) validatable;
		ID orgID = toRequest.getCache().getOrg().getID();

		@SuppressWarnings("rawtypes")
		Map ruleParams = getValidationRuleManager().getValidationRuleParams(orgID, RULE_ID);
		int maxDuration = ((Integer) ruleParams.get(FT_MAX_SHIFT_DURATION_VALUE)).intValue();
		int unit = ((Integer) ruleParams.get(FT_MAX_SHIFT_DURATION_UNIT)).intValue();

		ValidationResult result;
		if (MAX_SHIFT_DURATION_UNIT_ORGHOO == unit) {
			result = verifyOrgDayShiftDuration(toRequest, maxDuration);
		} else {
			result = verify24HourShiftDuration(toRequest, maxDuration);
		}

		return result;
	}

	protected ValidationResult verifyOrgDayShiftDuration(TORequest toRequest, int maxDuration) throws Exception {
		ValidationResult result = null;
		FTValidationCache ftCache = getFTValidationCache(toRequest);
		TimeZone orgTZ = getTimeZone(toRequest);
		int orgDayBoundaryOffset = getOrg(toRequest).getDayBoundaryOffset();

		for (FlexRequestMakeup makeup : toRequest.getFlexRequestMakeupList()) {
			TimeRange orgDayBoundaryTimeRange = getOrgDayBoundaryWithOffsetTimeRange(makeup.getShiftStartTime(), orgTZ,
					orgDayBoundaryOffset);
			int shiftDuration = getPaidDurationOfShiftsInTimeRange(ftCache, toRequest, makeup, orgDayBoundaryTimeRange,
					orgDayBoundaryTimeRange);
			int unpaidDurationFromCalendarEvents = getUnpaidDurationFromUnpaidCalendarEvents(toRequest, makeup);

			if ((shiftDuration - unpaidDurationFromCalendarEvents) > maxDuration) {
				result = ValidationUtil.setSoftValidationResult(toRequest,
						RmEjbBundleKey.FLEX_REQ_MAX_SHIFT_DURATION_EXCEED_ORG_DAY, (shiftDuration - unpaidDurationFromCalendarEvents),
						String.valueOf(maxDuration), CLASS_NAME);
				break;
			}
		}

		return result;
	}

	protected ValidationResult verify24HourShiftDuration(TORequest toRequest, int maxDuration) throws Exception {
		FTValidationCache ftCache = getFTValidationCache(toRequest);
		ValidationResult result = null;
		TimeZone orgTZ = getTimeZone(toRequest);
		Calendar startCal = Calendar.getInstance(orgTZ);
		Calendar iterCal = Calendar.getInstance(orgTZ);

		for (FlexRequestMakeup makeup : toRequest.getFlexRequestMakeupList()) {
			TimeRange _24HrsTimeRange = get24HoursTimeRange(makeup.getShiftStartTime(), makeup.getShiftEndTime(),
					orgTZ);

			int unpaidDurationFromCalendarEvents = getUnpaidDurationFromUnpaidCalendarEvents(toRequest, makeup);

			Collection<ShiftAssignment> shiftAssignments = getPublishedShiftAssignments(ftCache, _24HrsTimeRange);
			// create custom shift from makeup
			CustShiftReq custShiftReq = ftCache.createCustShiftReqFromMakeup(makeup, toRequest.getAggregatedRequest());
			CustShiftValidationCache cusShiftCache = new CustShiftValidationCache(custShiftReq);

			startCal.setTime(getPeriodStartDate(shiftAssignments, custShiftReq));
			startCal.add(Calendar.HOUR, -24);
			iterCal.setTime(startCal.getTime());
			iterCal.add(Calendar.HOUR, 24);
			Date endPeriod = getPeriodEndDate(shiftAssignments, custShiftReq);

			while (startCal.getTime().before(endPeriod)) {
				int shiftDuration = 0;
				for (ShiftAssignment sAssgn : shiftAssignments) {
					shiftDuration += cusShiftCache.getPaidDurationForSegment(sAssgn, startCal.getTime(),
							iterCal.getTime());
				}

				int durationOfMakeups = getMakeupDurationWithInTimeRange(makeup, _24HrsTimeRange);

				if ((shiftDuration + durationOfMakeups - unpaidDurationFromCalendarEvents) > maxDuration) {
					result = ValidationUtil.setSoftValidationResult(toRequest,
							RmEjbBundleKey.FLEX_REQ_MAX_SHIFT_DURATION_EXCEED_24HRS, (shiftDuration + durationOfMakeups - unpaidDurationFromCalendarEvents),
							String.valueOf(maxDuration), CLASS_NAME);
					break;
				}
				startCal.add(Calendar.MINUTE, 15);
				iterCal.add(Calendar.MINUTE, 15);
			}

			if (result != null) {
				break;
			}
		}

		return result;
	}

	private Date getPeriodEndDate(Collection<ShiftAssignment> saColl, CustShiftReq custShiftReq) {
		for (ShiftAssignment sAssgn : saColl) {
			if (sAssgn.getStartTime().after(custShiftReq.getStartTime())
					&& sAssgn.getStartTime().before(custShiftReq.getEndTime())) {
				return sAssgn.getEndTime();
			}
		}
		return custShiftReq.getEndTime();
	}

	private Date getPeriodStartDate(Collection<ShiftAssignment> saColl, CustShiftReq custShiftReq) {
		for (ShiftAssignment sAssgn : saColl) {
			if (sAssgn.getEndTime().after(custShiftReq.getStartTime())
					&& sAssgn.getEndTime().before(custShiftReq.getEndTime())) {
				return sAssgn.getStartTime();
			}
		}
		return custShiftReq.getStartTime();
	}

	/**
	 * Get the PAID durations of Shifts in the org boundary time range
	 *
	 * @param ftCache
	 * @param toRequest
	 * @param makeup
	 * @param intervalTimeRange
	 * @param orgDayBoundaryTimeRange
	 * @return paid duration in MINUTES
	 * @throws Exception
	 */
	private int getPaidDurationOfShiftsInTimeRange(FTValidationCache ftCache, TORequest toRequest,
			FlexRequestMakeup makeup, TimeRange intervalTimeRange, TimeRange orgDayBoundaryTimeRange) throws Exception {
		// create custom shift from makeup
		CustShiftReq custShiftReq = ftCache.createCustShiftReqFromMakeup(makeup, toRequest.getAggregatedRequest());
		// Create custom shift Cache
		CustShiftValidationCache cusShiftCache = new CustShiftValidationCache(custShiftReq);
		Collection<ShiftAssignment> shiftAssignments = getPublishedShiftAssignments(ftCache, intervalTimeRange);

		int durationOfShifts = getPaidDurationOfShiftsInTimeRange(cusShiftCache, shiftAssignments,
				orgDayBoundaryTimeRange);
		int durationOfMakeups = getMakeupDurationWithInTimeRange(makeup, orgDayBoundaryTimeRange);
		return durationOfShifts + durationOfMakeups;
	}

	/**
	 * Get PAID duration in MINUTES of shifts within the org time range
	 *
	 * @param cusShiftCache
	 * @param shiftAssignments
	 * @param orgDayBoundaryTimeRange
	 * @return PAID duration in MINUTES of shifts within the org time range
	 * @throws BbmException
	 * @throws RemoteException
	 * @throws BbmFinderException
	 * @throws BbmObjectNotFoundException
	 */
	private int getPaidDurationOfShiftsInTimeRange(CustShiftValidationCache cusShiftCache,
			Collection<ShiftAssignment> shiftAssignments, TimeRange orgDayBoundaryTimeRange)
			throws BbmObjectNotFoundException, BbmFinderException, RemoteException, BbmException {
		int paidDuration = 0;
		for (ShiftAssignment shiftAssignment : shiftAssignments) {
			if (TimePeriodUtil.overlap(shiftAssignment.getStartTime(), shiftAssignment.getEndTime(),
					orgDayBoundaryTimeRange.getStartDate(), orgDayBoundaryTimeRange.getEndDate())) {

				paidDuration += cusShiftCache.getPaidDurationForSegment(shiftAssignment,
						orgDayBoundaryTimeRange.getStartDate(), orgDayBoundaryTimeRange.getEndDate());
			}
		}

		return paidDuration;
	}

	/**
	 * Get makeup duration in MINUTES within a time range
	 *
	 * @param makeup
	 * @param timeRange
	 * @return duration in MINUTES
	 */
	private int getMakeupDurationWithInTimeRange(FlexRequestMakeup makeup, TimeRange timeRange) {
		double makeupDurationInTimeRange = 0;

		// The Before Extension
		TimeRange extBeforeTimeRange = makeup.getExtBeforeTimeRange();
		if (extBeforeTimeRange != null) {
			makeupDurationInTimeRange = timeRange.getOverlapDuration(extBeforeTimeRange);
		}

		// The After Extension
		TimeRange extAfterTimeRange = makeup.getExtAfterTimeRange();
		if (extAfterTimeRange != null) {
			makeupDurationInTimeRange += timeRange.getOverlapDuration(extAfterTimeRange);
		}

		return (int) makeupDurationInTimeRange / (1000 * 60);
	}

	private TimeRange getOrgDayBoundaryWithOffsetTimeRange(Date shiftStartTime, TimeZone orgTZ,
			int orgDayBoundaryOffset) {
		Calendar startCal = Calendar.getInstance(orgTZ);
		startCal.setTime(shiftStartTime);
		startCal.set(Calendar.HOUR_OF_DAY, 0);
		startCal.set(Calendar.MINUTE, 0);
		startCal.set(Calendar.SECOND, 0);
		startCal.add(Calendar.MINUTE, orgDayBoundaryOffset);

		Calendar endCal = Calendar.getInstance(orgTZ);
		endCal.setTime(startCal.getTime());
		endCal.add(Calendar.HOUR, 24);
		return new TimeRange(startCal.getTime(), endCal.getTime());
	}

	private TimeRange getOrg24HoursTimeRange(Date shiftStartTime, TimeZone orgTZ) {
		Calendar startCal = Calendar.getInstance(orgTZ);
		startCal.setTime(shiftStartTime);
		startCal.set(Calendar.HOUR_OF_DAY, 0);
		startCal.set(Calendar.MINUTE, 0);
		startCal.set(Calendar.SECOND, 0);
		Date orgDayStart = startCal.getTime();
		startCal.add(Calendar.HOUR, 24);
		return new TimeRange(orgDayStart, startCal.getTime());
	}

	private TimeRange get24HoursTimeRange(Date shiftStartTime, Date endStartTime, TimeZone orgTZ) {
		Calendar startCal = Calendar.getInstance(orgTZ);
		startCal.setTime(shiftStartTime);
		startCal.set(Calendar.HOUR_OF_DAY, 0);
		startCal.set(Calendar.MINUTE, 0);
		startCal.set(Calendar.SECOND, 0);
		startCal.add(Calendar.HOUR, -24);
		Calendar endCal = Calendar.getInstance(orgTZ);
		endCal.setTime(endStartTime);
		endCal.add(Calendar.HOUR, 24);
		return new TimeRange(startCal.getTime(), endCal.getTime());
	}

	private Collection<CalendarEventAssignment> getCalendarEventsWithinShift(TimeRange shiftTimeRange,
			Collection<CalendarEventAssignment> eventsInShiftRange) {
		List<CalendarEventAssignment> overlappedEvents = new ArrayList<CalendarEventAssignment>();

		TimeRange eventTimeRange = new TimeRange(new Date(), new Date());
		for (CalendarEventAssignment event : eventsInShiftRange) {
			eventTimeRange.setStartDate(event.getStartTime());
			eventTimeRange.setEndDate(event.getEndTime());
			if (shiftTimeRange.getOverlapInterval(eventTimeRange) != null && !eventTimeRange.isInstant()) {
				overlappedEvents.add(event);
			}
		}

		return overlappedEvents;
	}

	private int getUnpaidDurationFromUnpaidCalendarEvents(TORequest tor, FlexRequestMakeup makeup) throws Exception {
		TimeRange shiftTimeRange = new TimeRange(
				DateUtil.minOfTwoDates(makeup.getStartTime(), makeup.getShiftStartTime()),
				DateUtil.maxOfTwoDates(makeup.getEndTime(), makeup.getShiftEndTime()));
		Collection<CalendarEventAssignment> eventsInShiftTimeRange = getPublishedCalendarEvents(tor, shiftTimeRange);
		Collection<CalendarEventAssignment> shiftEventsWithinShift = getCalendarEventsWithinShift(shiftTimeRange,
				eventsInShiftTimeRange);

		return getUnpaidDurationFromUnpaidCalendarEvents(tor, makeup, shiftEventsWithinShift);
	}

	private int getUnpaidDurationFromUnpaidCalendarEvents(TORequest tor, FlexRequestMakeup makeup,
			Collection<CalendarEventAssignment> shiftEventsWithinShiff) throws Exception {
		int unpaidDuration = 0;

		TimeRange beforeExtTimeRange = makeup.getExtBeforeTimeRange();
		TimeRange afterExtTimeRange = makeup.getExtAfterTimeRange();
		//Main shift time range, exclude the 2 Exts
		TimeRange mainShiftTimeRange = new TimeRange(makeup.getShiftStartTime(), makeup.getShiftEndTime());

		for (CalendarEventAssignment event : shiftEventsWithinShiff) {
			if (isUnpaidCalendarEvent(tor, event)) {

				TimeRange eventTimeRange = new TimeRange(event.getStartTime(), event.getEndTime());
				//to exclude the before gap, check overlapping with the Before Ext
				if(beforeExtTimeRange != null && eventTimeRange.getOverlapInterval(beforeExtTimeRange) != null) {
					long overlappedInMili = eventTimeRange.getOverlapDuration(beforeExtTimeRange);
					unpaidDuration += overlappedInMili / (1000 * 60);
				}

				//to exclude the after gap, check overlapping with the After Ext
				if(afterExtTimeRange != null && eventTimeRange.getOverlapInterval(afterExtTimeRange) != null) {
					long overlappedInMili = eventTimeRange.getOverlapDuration(afterExtTimeRange);
					unpaidDuration += overlappedInMili / (1000 * 60);
				}

				//check overlapping with the main shift time range
				if(eventTimeRange.getOverlapInterval(mainShiftTimeRange) != null) {
					long overlappedInMili = eventTimeRange.getOverlapDuration(mainShiftTimeRange);
					unpaidDuration += overlappedInMili / (1000 * 60);
				}
			}
		}

		return unpaidDuration;
	}

	private boolean isUnpaidCalendarEvent(TORequest tor, CalendarEventAssignment event) throws Exception {
		if (event.getEventType() == Event.EVENT_TYPE_TIME_OFF
				|| event.getEventType() == Event.EVENT_TYPE_UNAVAILABILITY) {
			return true;
		} else if (event.getEventType() == Event.EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT) {
			Activity activity = getActivityByID(tor, event.getActivityID());
			if (activity != null && !activity.isPaid()) {
				return true;
			}
		}

		return false;
	}

	protected ShiftAssignment getNewShiftAssignToBeCreated(CustShiftValidationCache cusShiftCache,
			CustShiftReq custShiftReq) throws Exception {
		return cusShiftCache.getNewShiftAssignToBeCreated(custShiftReq, cusShiftCache.getScheduleAccessManager());
	}

	protected Collection<ShiftAssignment> getPublishedShiftAssignments(FTValidationCache cache, TimeRange duration)
			throws Exception {
		return cache.getPublishedShiftAssignments(duration);
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}

	protected ValidationRuleManager getValidationRuleManager() throws BbmCreateException {
		return RmManagerFactory.getInstance().getValidationRuleManager();
	}

	protected TimeZone getTimeZone(TORequest tor) throws Exception {
		return getFTValidationCache(tor).getOrg().getTimeZone();
	}

	protected Organization getOrg(TORequest tor) throws Exception {
		return getFTValidationCache(tor).getOrg();
	}

	protected Collection<CalendarEventAssignment> getPublishedCalendarEvents(TORequest tor, TimeRange eventsDuration)
			throws Exception {
		return getFTValidationCache(tor).getPublishedCalendarEvents(eventsDuration);
	}

	public Activity getActivityByID(TORequest tor, ID activityID) throws Exception {
		return getFTValidationCache(tor).getActivityByID(activityID);
	}
