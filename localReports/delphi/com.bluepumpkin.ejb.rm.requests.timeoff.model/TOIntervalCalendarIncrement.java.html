package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.util.Date;

public class TOIntervalCalendarIncrement {

	private final Date date;
	private TOIntervalCalendarInterval interval;
	private boolean isNonDstDayTransition; // NOSONAR
	private int type = TOCalendarDayData.NO_TO_HOURS_AVAILABLE_DAY_TYPE;

	// ================================================================================
	// CONSTRUCTION
	// ================================================================================

	public TOIntervalCalendarIncrement(Date date) {
		this.date = date;
	}

	public TOIntervalCalendarIncrement(Date date, TOIntervalCalendarInterval interval) {
		this(date);

		this.interval = interval;
	}

	// ================================================================================
	// PROPERTIES
	// ================================================================================

	public Date getDate() {
		return date;
	}

	public TOIntervalCalendarInterval getInterval() {
		return interval;
	}

	public void setInterval(TOIntervalCalendarInterval interval) {
		this.interval = interval;
	}

	/**
	 * Gets whether this increment marks the DST transition point for a non DST day.
	 * Example: 11/5 2016 @ 1 am EST would be set to true because 11/6 2016 @ 1am EST is the DST Transition point.
	 */
	public boolean isNonDstDayTransition() {
		return this.isNonDstDayTransition;
	}

	/**
	 * Sets whether this increment marks the DST transition point for a non DST day.
	 * Example: 11/5 2016 @ 1 am EST would be set to true because 11/6 2016 @ 1am EST is the DST Transition point.
	 */
	public void setIsNonDstDayTransition(boolean isNonDstDayTransition) {
		this.isNonDstDayTransition = isNonDstDayTransition;
	}

	public int getType() {
		return type;
	}

	public void setTypeIfHigherPriority(int type) {
		if (type > this.type) {
			this.type = type;
		}
	}

	// ================================================================================
	// INTERFACE
	// ================================================================================

	@Override
	public String toString() {
		return String.format("\"%s, %s\"", this.date, this.interval);
	}
}
