package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.util.Comparator;

public final class TOChoiceStartComparator implements Comparator<TOChoice> {

	public static final TOChoiceStartComparator INSTANCE = new TOChoiceStartComparator();

	private TOChoiceStartComparator() {

	}

	@Override
	public int compare(TOChoice o1, TOChoice o2) {
		long s1 = o1.getStartDate().getTime();
		long s2 = o2.getStartDate().getTime();

		if (s1 > s2) {
			return 1;
		} else if (s2 > s1) {
			return -1;
		}

		return 0;
	}



}
