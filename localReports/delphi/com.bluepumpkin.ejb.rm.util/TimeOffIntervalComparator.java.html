package com.bluepumpkin.ejb.rm.util;

import java.util.Comparator;
import java.util.Date;

import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOIntervalAllocation;

public class TimeOffIntervalComparator implements Comparator<TOIntervalAllocation> {
	int sortDirection=0;// 0: ASC; 1: DSC
	public TimeOffIntervalComparator(int direction) {
		sortDirection = direction;
	}
	@Override
	public int compare(TOIntervalAllocation obj1, TOIntervalAllocation obj2) {
			Date startTime1 = obj1.getStartTime();
			Date startTime2 = obj2.getStartTime();
			Date endTime1= obj1.getEndTime();
			Date endTime2= obj2.getEndTime();
			
			if(sortDirection ==0) {
				if(startTime1.compareTo(startTime2)==0) {
					return endTime1.compareTo(endTime2);
				} else {
					return startTime1.compareTo(startTime2);
				}
			} else {
				if(startTime1.compareTo(startTime2)==0) {
					return endTime2.compareTo(endTime1);
				} else {
					return startTime2.compareTo(startTime1);
				}
				
			}
	}

}
