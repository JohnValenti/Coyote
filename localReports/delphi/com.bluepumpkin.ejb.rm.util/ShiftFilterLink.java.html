	  protected boolean isScheduleValid(BiddableSchedule vo)
	  {
		  //Check the Availability criteria
	  	  if( this.m_bidSize != null && !this.m_bidSize.isValidSize(vo.getNumOfAvailableInstances()) )
		      return false;

		  //Check the WorkPattern ID criteria
		  if( this.m_templateID != null  )
		  {
			  ID id = vo.getOptMethods().getEmployee().getEmployeeTemplateID();	
			  if( id == null || !this.m_templateID.equals(id) )
			      return false;
		  }

		  List saList = vo.getOptMethods().getShiftAssignments();
		  if( saList == null || saList.size() <= 0 )
			  return false;

		  int shiftIdCount = 0;
		  int shiftLengthCount = 0;
		  int shiftStartCount = 0;
		  int shiftEndCount = 0;
		  
		  int numberOfShifts = saList.size();
		  
		  int totSchedLen = 0;
		  int totPaidSchedLen = 0;
		 
		  Iterator saIter = saList.iterator();
		  while( saIter.hasNext() )
		  {
			  ShiftAssignment sa = (ShiftAssignment)saIter.next();
			  
			  int paidShiftLen = RequestUtil.computePaidShiftLength(sa); 
			  totSchedLen += getDisplayDuration(sa);
			  totPaidSchedLen += paidShiftLen;
			  
			  //Keep a count of all matching shift criteria
			  
			  if( this.m_shiftID != null )
			  {
				  ID id = sa.getShiftID();
				  if( id != null && this.m_shiftID.equals(id) )
				      shiftIdCount++;
			  }

			  if( this.m_shiftStartDate != null )
			  {
				  if( this.m_shiftStartDate.isValidDate(getDisplayStartTime(sa)) )
				      shiftStartCount++;
			  }

			  if( this.m_shiftEndDate != null )
			  {
				  if( this.m_shiftEndDate.isValidDate(sa.getEndTime()) )
				  	shiftEndCount++;
			  }

			  if( this.m_shiftLen != null )
			  {
				  int shiftLength = this.m_shiftPaidOnly ? paidShiftLen : getDisplayDuration(sa);
				  if( this.m_shiftLen.isValidSize(shiftLength) )
					  shiftLengthCount++;
			  }
		  }

		  //Check the Schedule Length criteria
		  if( this.m_scheduleLen != null )
		  {
			  int schedLength = this.m_schedulePaidOnly ? totPaidSchedLen : totSchedLen;
			  if( !this.m_scheduleLen.isValidSize(schedLength) )
				  return false;
		  }

		  boolean found = true;
		  
		  //Check the Shift Length criteria
		  if( this.m_shiftLen != null )
		  {
		      if ( this.m_shiftLen.getMatchAllShifts() )
				  found = ( shiftLengthCount == numberOfShifts );
		      else
		          found = ( shiftLengthCount > 0 );

		      if (found == false)
			      return false;
		  }
		  
		  //Check the Shift Start criteria
		  if( this.m_shiftStartDate != null )
		  {
		      if ( this.m_shiftStartDate.getMatchAllShifts() )
				  found = ( shiftStartCount == numberOfShifts );
		      else
		          found = ( shiftStartCount > 0 );
		      
			  if (found == false)
			      return false;
		  }
		  
		  //Check the Shift End criteria
		  if( this.m_shiftEndDate != null )
		  {
		      if ( this.m_shiftEndDate.getMatchAllShifts() )
				  found = ( shiftEndCount == numberOfShifts );
		      else
		          found = ( shiftEndCount > 0 );
		      
			  if (found == false)
			      return false;
		  }
		  
		  //Check the Shift ID criteria
		  if( this.m_shiftID != null )
		  {
		      if ( this.getMatchAllShifts() )
				  found = ( shiftIdCount == numberOfShifts );
		      else
		          found = ( shiftIdCount > 0 );
		      
			  if (found == false)
			      return false;
		  }
	  
		  //If we made it this far, we've passed all the criteria
		  return true;
	  }

    /**
     * Get the display start time of a ShiftAssignment.
     * <P>
     *  @return  Date - event start time
     */
     public static Date getDisplayStartTime(ShiftAssignment sa)
     {
         Date startTime = sa.getFieldValueDate(ShiftAssignmentFields.STARTTIME);
    
         //Hack to treat initial NO_ACTIVITY events as not part of the shift
         Collection shiftEventAssignments = sa.getChildren();
         if (shiftEventAssignments != null && !shiftEventAssignments.isEmpty())
         {
             Iterator it = shiftEventAssignments.iterator();
             while (it.hasNext())
             {
                 ShiftEventAssignment sea = (ShiftEventAssignment)it.next();
                 if (sea.getActivityID().toInt() == Activity.NO_ACTIVITY &&
                            sea.getStartTime().equals(sa.getStartTime())){
                     startTime = sea.getEndTime();
                     break;
                 }
             }
         }
         return startTime;
     }

    /**
     * Get the display duration of a ShiftAssignment.
     * <P>
     *  @return int - duration (in minutes) of the event
     */
    public static int getDisplayDuration(ShiftAssignment sa)
    {
        int duration = sa.getFieldValueInt(ShiftAssignmentFields.DURATION);
    
        //Hack to treat initial NO_ACTIVITY events as not part of the shift
        Collection shiftEventAssignments = sa.getChildren();
        if (shiftEventAssignments != null && !shiftEventAssignments.isEmpty())
        {
            Iterator it = shiftEventAssignments.iterator();
            while (it.hasNext())
            {
                ShiftEventAssignment sea = (ShiftEventAssignment)it.next();
                if (sea.getActivityID().toInt() == Activity.NO_ACTIVITY &&
                          sea.getStartTime().equals(sa.getStartTime())){
                    duration = duration - sea.getDuration();
                    break;
                }
            }
        }
        return duration;
    }
