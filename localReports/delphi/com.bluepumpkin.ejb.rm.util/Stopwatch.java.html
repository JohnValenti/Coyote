package com.bluepumpkin.ejb.rm.util;

/**
 * Helper class used to measure elapsed time. 
 * Modeled after the .net Stopwatch class
 */
public class Stopwatch {
	long start;
	long elapsed;
	boolean isRunning;

	//One millisecond = 1000000 nanoseconds
	private static final long MILLI_TO_NANO = 1000000L;
	private static final long SECOND_TO_NANO = MILLI_TO_NANO * 1000L;

	public Stopwatch() {
		reset();
	}

	public void start() {
		start = System.nanoTime();
		isRunning = true;
	}

	public void stop() {
		if (isRunning) {
			elapsed = System.nanoTime() - start;
			isRunning = false;
		}
	}

	public void reset() {
		start = 0;
		elapsed = 0;
		isRunning = false;
	}

	public void restart() {
		reset();
		start();
	}

	public boolean hasElapsedTime() {
		return elapsedNanoTime() > 0L;
	}

	@Override
	public String toString() {
		return nanoDurationToString(elapsedNanoTime());
	}

	public long elapsedNanoTime() {
		if (isRunning) {
			return System.nanoTime() - start;
		}
		return elapsed;
	}


	static String nanoDurationToString(long durationNanoSeconds) {

		long milliseconds = (durationNanoSeconds / MILLI_TO_NANO) % 1000L;
		long seconds = (durationNanoSeconds / SECOND_TO_NANO) % 60;
		long minutes = durationNanoSeconds / (SECOND_TO_NANO * 60L);

		return String.format("%02d:%02d.%03d", minutes, seconds, milliseconds);
	}

	public static Stopwatch startNew() {
		Stopwatch sw = new Stopwatch();
		sw.start();
		return sw;
	}

}
