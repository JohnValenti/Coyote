package com.verint.web.rm.rest.entity.settings;

import javax.xml.bind.annotation.XmlRootElement;

import com.verint.web.rest.entity.AbstractWFOEntity;

@XmlRootElement(name = "orgSettings")
public class OrganizationSettingsEntity extends AbstractWFOEntity {

	private boolean allowEscalation;
	private CustShiftSettingsEntity custShift = new CustShiftSettingsEntity();
	private ShiftBiddingSettingsEntity shiftBidding = new ShiftBiddingSettingsEntity();
	private ShiftSwapSettingsEntity shiftSwap = new ShiftSwapSettingsEntity();
	private TimeOffSettingsEntity timeOff = new TimeOffSettingsEntity();

	public OrganizationSettingsEntity() { // NOSONAR
	}

	public boolean isAllowEscalation() {
		return allowEscalation;
	}

	public void setAllowEscalation(boolean allowEscalation) {
		this.allowEscalation = allowEscalation;
	}

	public CustShiftSettingsEntity getCustShift() {
		return custShift;
	}

	public void setCustShift(CustShiftSettingsEntity custShift) {
		this.custShift = custShift;
	}

	public ShiftBiddingSettingsEntity getShiftBidding() {
		return shiftBidding;
	}

	public void setShiftBidding(ShiftBiddingSettingsEntity shiftBidding) {
		this.shiftBidding = shiftBidding;
	}

	public ShiftSwapSettingsEntity getShiftSwap() {
		return shiftSwap;
	}

	public void setShiftSwap(ShiftSwapSettingsEntity shiftSwap) {
		this.shiftSwap = shiftSwap;
	}

	public TimeOffSettingsEntity getTimeOff() {
		return timeOff;
	}

	public void setTimeOff(TimeOffSettingsEntity timeOff) {
		this.timeOff = timeOff;
	}

}
