package com.bluepumpkin.ejb.rm.util.interval;

import java.util.ArrayList;
import java.util.List;

public class PackIntervals {

	private PackIntervals() {

	}

	/**
	 * Packs overlapping intervals into one. The caller provided pack function is used to pack overlapping intervals. 
	 * Intervals for which canPack returns false are not packed, but output as is. 
	 * The input is assumed to be sorted ascending on start. 
	 * 
	 * <pre>
	 *  Input List:    |----- 1 -----|   |-------2-------|    |-------4--------|          |-------5--------|--------7 (canPack=false)------|----8--|----9----|
	 *                                      |----------------3---------------------|            |-----6----------------|
	 *                                      
	 *                                      
	 *                                                        
	 *  Output List:   |----- 1 -----|   |------ Pack 2,3,4 ------------- ---------|      |--------5,6-----------------|                   |------8,9--------|
	 *  
	 *                                                                                                     |--------7 (canPack=false)------|
	 * </pre>
	 * 
	 */


	public static <T extends Interval> List<T> pack(Iterable<T> intervals, PackIntervalOperations<T> operations) {
		List<T> list = new ArrayList<T>();
		int accumulatorIndex = -1;
		for (T interval : intervals) {

			if (interval == null) {
				throw new IllegalArgumentException("The  input collection for pack contains a null object");
			}

			if (accumulatorIndex < 0) {
				accumulatorIndex = 0;
				list.add(accumulatorIndex, interval);
				continue;
			}
			
			accumulatorIndex = packAccumulate(list, accumulatorIndex, interval, operations);
		}
		return list;
	}

	private static <T extends Interval> int packAccumulate(List<T> list, int accumulatorIndex, T interval,
			PackIntervalOperations<T> operations) {

		T accumulator = list.get(accumulatorIndex);

		if (interval.start() <= accumulator.end()) {
			//interval overlaps with the accumulator

			if (operations.canPack(accumulator, interval)) {
				//pack the interval into the accumulator
				accumulator = operations.pack(accumulator, interval);
				list.set(accumulatorIndex, accumulator);
				return accumulatorIndex;
			}
			//interval overlaps, but should not be packed, just add the interval as is without packing
			list.add(interval);
			return accumulatorIndex;
		}

		//the new interval starts after the current accumulator ends, so make that the new accumulator
		int newAccumulatorIndex = list.size();
		list.add(newAccumulatorIndex, interval);
		return newAccumulatorIndex;
	}

}
