package com.bluepumpkin.web.rm.timeoffbid;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Iterator;
import java.util.List;
import java.util.ResourceBundle;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBid;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;
import com.bluepumpkin.web.rm.Log;
import com.witness.web.uif.pagecomponent.DefaultNodeData;
import com.witness.web.uif.pagecomponent.list.SingleColListPC;
import com.witness.web.uif.pagecomponent.selection.SelectablePC;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.pagemodel.SelectionpanePageModel;
import com.witness.web.uif.system.RequestContext;

/**
 * File:         TimeOffBidSelectionPM
 * Description:  Page model for the selection pane with the list of time off bids
 * Copyright:    Copyright (c) 2016
 * Company:      Verint System, inc
 * @author       Truc Pham
 */
public class TimeOffBidSelectionPM extends SelectionpanePageModel {
	/**
	 *
	 */
	private static final long serialVersionUID = 1L;
	private static final Category LOGGER = Log.initCategory(TimeOffBidSelectionPM.class.getName());
	private static String FORM_ACTION = "time_off_bid_selection"; // NOSONAR
	
	private Collection<TimeOffBid> timeOffBids = Collections.emptyList();
	private SingleColListPC timeOffBidListPC;
	private ResourceBundle bbmBundle;
	private boolean isPageEnabled = true;

	/** Constructor */
	public TimeOffBidSelectionPM(RequestContext context) {
		super(context);
		bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);
		// Create List
		timeOffBidListPC = new SingleColListPC(context);
		addChildComponent(timeOffBidListPC);
		setIsMultiSelectEnabled(false);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static TimeOffBidSelectionPM getInstance(RequestContext context) throws Exception { // NOSONAR
		return (TimeOffBidSelectionPM)getInstance(context, TimeOffBidSelectionPM.class);
	}

	/**
	 * Initialize Model with Data
	 */
	@SuppressWarnings("unchecked")
	@Override
	protected void initializeModel() {
		if (m_context.getUser().getEmployeeID() == null) {
			isPageEnabled = false;
			return;
		}
		if (m_isInitialized) {
			return;
		} else {
			m_isInitialized = true;
		}
		try {
			Collection<ID> orgIDs =OrganizationModelHandler.getAllOrgIdsInUserScope(m_context,m_context.getUser());
			String nameSortDirection = (String) m_context.getAttribute(RequestContext.SESSION_SCOPE,TimeOffBidSettingsRH.BID_NAME_SORT_DIRECTION);
			if(!StringUtil.isEmpty(nameSortDirection)) {
				timeOffBids = TimeOffBidModelHandler.getTimeOffBidsForOrgIDs(orgIDs,true,nameSortDirection.equals(TableHeaderPC.SORT_ASCENDING));
			} else {
				timeOffBids = TimeOffBidModelHandler.getTimeOffBidsForOrgIDs(orgIDs);
			}
		}  catch (Exception ex) {
			handleUnableToLoadDataError(LOGGER, ex);
		} finally {
			setHeaderTitle(i18n(bbmBundle,BbmWebBundleKey.Name));
			initToolbar();
			initList();
		}
	}
	@Override
	public void setHeaderTitle(String header) {
		super.setHeaderTitle(i18n(bbmBundle,BbmWebBundleKey.Name));
	}


	@SuppressWarnings({ "rawtypes", "unchecked" })
	protected void initList() {
		if (!isPageEnabled||timeOffBids.isEmpty()) {
			return;
		}

		int index=0;
		List preselected = new ArrayList();
		Collection listModel = new ArrayList(timeOffBids.size());
		for(Iterator it=timeOffBids.iterator(); it.hasNext(); ) {
			TimeOffBid bid = (TimeOffBid)it.next();
			String rowID = bid.getID().toString();
			String name = bid.getName();
			DefaultNodeData node = new DefaultNodeData(rowID, name, "");
			listModel.add( node );

			if( rowID.equals(getSelectedID()) ) {
				preselected.add( "r" + index);
			}
			index++;
		}
		if ( !preselected.isEmpty()) {
			timeOffBidListPC.setSelectedIndexes(preselected);
		}

		timeOffBidListPC.setSubmitOnSelect(true);
		timeOffBidListPC.setListModel(listModel);
	}

	/**
	 * Initialize Toolbar
	 */
	@Override
	protected void initToolbar() {
			useDefaultToolbar(false, false);
	}

	/**
	 * Return the Selectable Component
	 */
	@Override
	public SelectablePC getSelectablePC() {
		return timeOffBidListPC;
	}

	/**
	 * Return URL for HTML Form Action
	 */
	@Override
	public String getFormAction() {
		return FORM_ACTION;
	}
}
