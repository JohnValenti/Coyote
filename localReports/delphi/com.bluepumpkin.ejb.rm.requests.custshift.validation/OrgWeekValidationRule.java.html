/**
 * <li> Rule from UI: Agent complies with a weekly hours rule.
 * <p/>
 * For week where shift is being requested, verify rule hour constraints.
 * <p/>
 * <p/>
 * For tentative approvals: processing logic same as non tentative requests as
 * it uses only the published schedule for calculation of rule hours
 * <p/>
 * <p> Algorithm:
 * <p/>
 * <li> Validation rule to ensure that the employee does not violate the rule hours per week allocated to him/her.
 * <ul>
 * <li>Find the start of the week which contains the shift this employee will get after approval.
 * <li> get the schedule information for this employee for this week
 * <li> calculate the work hours for this week plus the shift hours gained
 * <p/>
 * <li> compare the week work hours with employee rule hours.
 * <li> validation result in case of failure should have information about the employee, work hours, rule limits
 * <ul>
 */
public abstract class OrgWeekValidationRule implements Validator {

	// obtain instance for logging
	public static String m_className = OrgWeekValidationRule.class.getName();

	public static Category m_cat = Log.initCategory(m_className);

	protected CustShiftValidationCache m_vc;

	protected WorkResourceManager m_wrm = null;

	protected EmpWorkRuleManager m_ewrm = null;

	protected boolean m_isDifferentWeek;

	public OrgWeekValidationRule() {
	};
	private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

	public ValidationResult validate(Validatable validatable) throws Exception {
		m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");
		String methodName = "validate";
		m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));

		try {
			if (m_ewrm == null) m_ewrm = WfmManagerFactory.getEmpWorkRuleManager();
			ValidationResult result = null;
			// get schedule information for employee for the entire org week during which the shift being received falls.
			CustShiftReq csr = (CustShiftReq) validatable;
			m_vc = (CustShiftValidationCache) csr.getValidationCache();
			ID empID = csr.getEmployeeID();
			Organization org = csr.getCache().getOrg();

			Date orgWeekStart = TOCalcUtil.getDateForOrgWeekStart(org, csr.getStartTime());
			Date orgWeekEnd = TOCalcUtil.getDateForOrgWeekEnd(org, csr.getStartTime());

			orgWeekStart = extendStartDays(orgWeekStart, org); 
			orgWeekEnd = extendEndDays(orgWeekEnd, org);

			Collection shiftAssnsDuringOrgWeek = ValidationUtil.getPublishedEventsForWorkResourceByType(Event.EVENT_TYPE_SHIFT_ASSIGNMENT, empID, orgWeekStart, orgWeekEnd);
			ScheduleAccessManager sam = csr.getCache().getScheduleAccessManager();
			ShiftAssignment shiftAssnToCreate = m_vc.getNewShiftAssignToBeCreated(csr, sam);
			ShiftAssignment shiftAssnToDelete = m_vc.getExistingShiftAssignToBeDeleted(csr, sam);
			if (shiftAssnToCreate == null) {
				//Unable to create shift between {0} and {1} for employee {2} for the specified custom shift request.
				result = ValidationUtil.setAndLogHardValidationResult(validatable,
				        RmEjbBundleKey.CS_CANNOT_CREATE_SHIFT,
				        RmEjbLogBundleKey.CS_CANNOT_CREATE_SHIFT,
				        new Serializable[]{csr.getStartTime(), csr.getEndTime(), m_vc.getEmployeeNameByID(csr.getEmployeeID())},
				        m_className);

				return result;
			}
			if (shiftAssnToDelete != null) {
				//remove the item from shiftAssnsDuringOrgWeek collection
				for (Iterator it = shiftAssnsDuringOrgWeek.iterator(); it.hasNext();) {
					ShiftAssignment sa = (ShiftAssignment) it.next();
					if (sa.equals(shiftAssnToDelete)) {
						it.remove();
						break;
					}
				}
			}
			//add the shiftAssnToCreate to the collection
			if (shiftAssnToCreate.getDescription()!=null)
				shiftAssnToCreate.setDescription("!VALIDATION_CHECK"+ shiftAssnToCreate.getDescription()); //GQ: we flag this shift so that we will know which daily conflicts to keep
			else
				shiftAssnToCreate.setDescription("!VALIDATION_CHECK");////GQ: we flag this shift so that we will know which daily conflicts to keep
			//TP:  Fix 122794, Change the flag to !VALIDATION_CHECK which has special character which is not allowed in Description (Comment in the UI) 
			//With that way, besides the flag, we still keep the current description. 
			shiftAssnsDuringOrgWeek.add(shiftAssnToCreate);
			return checkRuleHours(empID, orgWeekStart, orgWeekEnd, shiftAssnsDuringOrgWeek, org, validatable);
		} finally {
			m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
		}
	}

	//overidden by child classes
	public Date extendStartDays(Date startDate, Organization org) {
		return startDate;
	}

	//overidden by child classes
	public Date extendEndDays(Date endDate, Organization org) {
		return endDate;
	}

	/**
	 * This is where you check the rule given the week's assigments after swap for the empoyee.
	 *
	 * @param empID
	 * @param orgWeekStart
	 * @param orgWeekEnd
	 * @param shiftsDuringOrgWeek
	 * @param org
	 * @param validatable
	 * @return
	 * @throws Exception
	 */
	abstract protected ValidationResult checkRuleHours(ID empID, Date orgWeekStart, Date orgWeekEnd,
	                                                   Collection shiftsDuringOrgWeek, Organization org, Validatable validatable) throws Exception;
}
