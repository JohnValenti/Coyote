package com.bluepumpkin.web.rm.setup.filingrules;

public class FilingRuleListPageModel extends RmSettingsListPM { 

	/**
	 * Constructor
	 */
	public FilingRuleListPageModel(RequestContext context) {
		super(context,
				PrivilegeKeys.TOM_SETFILINGRULES_ID,
				PrivilegeKeys.SS_SETFILINGRULES_ID,
				PrivilegeKeys.SB_SETFILINGRULES_ID,
				PrivilegeKeys.CS_SETFILINGRULES_ID,
				PrivilegeKeys.FT_SETFILINGRULES_ID);

		// Enable Object Copy
		initCopyObjType(RequestFilingRule.OBJECT_TYPE_FILINGRULE_CS);
		setName("filingRuleList");
	}

	/**
	 * Return an instance of FilingRuleListPageModel ready to be rendered in JSP
	 * @param   context RequestContext
	 * @return  FilingRuleListPageModel  instance of FilingRuleListPageModel
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, FilingRuleListPageModel.class);
	}

	/**
	 * Loads all the RuleList value objects for the selected Organization ID
	 */
	@Override
	protected Collection loadSpecificRules() throws Exception {
		return FilingRuleModelHandler.getFilingRuleList(getSelectedIDObject());
	}

	/**
	 * Initialize List, sets the list header
	 */
	@Override
	protected void initList() {
		super.initList();

		// Setup Headers
		FilingRuleUtil.initListHeader(m_list.getHeader(), m_rmBundle, m_bbmBundle, true, m_localizer);

		// Setup basic table property
		m_list.setMultiRowSelectable(false);
	}

	/**
	 * Initialize toolbar
	 * Adds Add, Copy, Edit and Delete button
	 */
	@Override
	protected void initToolbar() {
		String applyRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_FILING_RULE_SAVE_ACTION);

		String addRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_FILING_RULE_CREATE_NEW_ACTION);

		String copyRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_FILING_RULE_COPY_ACTION);

		String editRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_FILING_RULE_EDIT_ACTION);

		String delRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_FILING_RULE_DELETE_ACTION);

		useDefaultToolbar(applyRuleLabel, addRuleLabel, copyRuleLabel,
				editRuleLabel, delRuleLabel);
	}

	/**
	 * Populates the list data in the model
	 */
	@Override
	protected void createListModel() {
		if (!loadRules()) {
			return;
		}

		ArrayList listModel = new ArrayList();

		if (!m_isEmpty) {
			ID selectedOrgID = getSelectedIDObject();
			boolean isTOConfigEnabled = isAuthorizedToConfigureTOM(selectedOrgID);
			boolean isFTConfigEnabled = isAuthorizedToConfigureFT(selectedOrgID);
			boolean isSSConfigEnabled = isAuthorizedToConfigureSS(selectedOrgID);
			boolean isSBConfigEnabled = isAuthorizedToConfigureSB(selectedOrgID);
			boolean isSCConfigEnabled = isAuthorizedToConfigureSC(selectedOrgID);

			for (Iterator it = m_ruleList.iterator(); it.hasNext(); ) {
				RequestFilingRule rule = (RequestFilingRule)it.next();

				// Create the row data
				String rowID = rule.getID().toString();
				String requestType = rule.getRequestType();

				boolean isTimeOff = (Request.REQUESTTYPE_TIMEOFF.equals(requestType) ||
						requestType==null);
				boolean isFlexTime = Request.REQUESTTYPE_FLEXTIME.equals(requestType);
				boolean isShiftSwap = Request.REQUESTTYPE_SHIFTSWAP.equals(requestType);
				boolean isShiftBid = Request.REQUESTTYPE_SHIFTBID.equals(requestType);
				boolean isCustomShift = Request.REQUESTTYPE_CUSTSHIFT.equals(requestType);

				boolean isConfigEnabled = isTimeOff ? isTOConfigEnabled :
						(isShiftBid ? isSBConfigEnabled : (isShiftSwap ? isSSConfigEnabled : isSCConfigEnabled));
				isConfigEnabled = isFlexTime ? isFTConfigEnabled : isConfigEnabled;

				String chkbox = FilingRuleUtil.createApplyChkBox(rowID,
						rule.getApplied(), (rule.getCanChangeApply() && isConfigEnabled));
                
				String orgName = getOrganizationName(rule.getOrganizationId());
				//get org timezone
                TimeZone tz = RmSetupUtil.getOrgTimeZone(m_context, rule.getOrganizationId());
				DefaultMultiColumnNodeData row_data = FilingRuleUtil.createRuleDataRow(
						m_context, rule, m_rmBundle, chkbox, orgName, true, m_localizer, tz);
				row_data.setSelectable(isConfigEnabled);

				// Add node Attributes, sets the
				// Also need to check if this rule belongs to this organization
				// or is inherited. Combination  would decide if the row is
				// editable.
				String sIsEditable = isRowEditable(rule.getOrganizationId());
                
				String copyObjType = isShiftSwap ? RequestFilingRule.OBJECT_TYPE_FILINGRULE_SS :
						(isShiftBid ? RequestFilingRule.OBJECT_TYPE_FILINGRULE_SB : 
							(isCustomShift ? RequestFilingRule.OBJECT_TYPE_FILINGRULE_CS : RequestFilingRule.OBJECT_TYPE_FILINGRULE_TOM));
				copyObjType = isFlexTime ? RequestFilingRule.OBJECT_TYPE_FILINGRULE_FT : copyObjType;

				FilingRuleUtil.addRowAttr(sIsEditable, isConfigEnabled,	copyObjType, row_data);

				// Add row data to list model
				listModel.add(row_data);

				// Check if row should be selected, assuming only one should be
				// selected
				if (OrganizationModelHandler.UNREASONABLE_ORG_ID.equals(m_selectedRowOrgID)) {
					if (isRowSelected(rule.getID())) {
						m_list.addSelectedRowID(rowID);
						m_selectedRowOrgID = rule.getOrganizationId();
					}
				}
			} //end for
		} else {
			// Create/Add no data msg Row
			MultiColumnNodeData row_data;
			row_data = FilingRuleUtil.createNoDataRow(m_rmBundle, true, m_localizer);
			listModel.add(row_data);
		}

		m_list.setListModel(listModel);
	}

	/**
	 * Return the Selected Filing Rule IDs
	 */
	public ID[] getSelectedFilingRuleIDs() {
		return m_performActionOnIDs;
	}

	/**
	 * Get entity name. Used for title.
	 * @return String Identifies the entity name
	 */
	@Override
	protected String getTitleEntityType() {
		return i18n(m_rmBundle, RmWebBundleKey.ORG_RM_FILING_RULE_LABEL);
	}

	/**
	 * Return URL for HTML Form Action
	 * @return 	String	Identifies the URL
	 */
	@Override
	public String getFormAction() {
		return "request_processing_filingrulelist";
	}

	@Override
	protected String getDialogHeader() {
		return i18n(m_rmBundle, RmWebBundleKey.ORG_RM_CREATE_NEW_FILING_RULE);
	}
	@Override
	protected void initDialog(RequestContext context) {
		
		if(LicenseUtil.isAdvancedRMLicense()){
			ID selectedOrgID = getSelectedIDObject();
			initDialogMenuAndAddTimeOffRules(selectedOrgID,context);
			addFlexTimeToDialogMenu(selectedOrgID,context);
			addShiftRulesToDialogMenu(selectedOrgID,context);
		} else {
			super.initDialog(context);
		}
	}
		
}
