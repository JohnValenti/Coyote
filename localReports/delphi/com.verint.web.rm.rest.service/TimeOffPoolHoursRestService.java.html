    @GET
    @Path("/get/exists/{toPoolId}/{startDate}/{endDate}")
    public Boolean checkIfDataExists(@PathParam("toPoolId") String toPoolId,
                                     @PathParam("startDate") String strStartTime,
                                     @PathParam("endDate") String strEndTime) throws WFOException
    {

        ID timeOffPoolId= RmUtils.stringToIntegerID(toPoolId);
        if(timeOffPoolId == null)
        {
            throw new BusinessWFOException("time off pool ID is null",Response.Status.CONFLICT);
        }
        Date startDate = null;
        Date endDate = null;
        if(strStartTime==null)
        {
            throw new BusinessWFOException("startDate is null", Response.Status.CONFLICT);
        }
        else
            startDate = FormatUtils.iso8601SecPrecisionStringToDate(strStartTime);

        if (strEndTime == null)
        {
            throw new BusinessWFOException("endDate is null", Response.Status.CONFLICT);
        }
        else
            endDate = FormatUtils.iso8601SecPrecisionStringToDate(strEndTime);
        Logger.debug("Start date: " + startDate.toString() + " " + "End date :" + endDate.toString());

        if(!endDate.after(startDate))
        {
            throw new BusinessWFOException("endDate is not after startDate", Response.Status.CONFLICT);
        }

        Calendar startCal = RmUtils.getCalendarInGMTFromDate(startDate);
        Calendar endCal = RmUtils.getCalendarInGMTFromDate(endDate);
        RequestContext context = getRequestContext();
        User user = getUser(request);
        return  timeOffPoolHoursFacade.checkIfTOPoolHoursExist(context,timeOffPoolId,startCal,endCal);
	}
