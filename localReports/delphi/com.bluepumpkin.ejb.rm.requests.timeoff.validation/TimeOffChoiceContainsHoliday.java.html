/**
 * @author rrajendran
 * 
 * Raises an alert if any of the TOChoices overlaps with a partial holiday.
 * In 15.1.0: It should skip the validation rule for Time Off Bid requests which are time off requests submitted during the bid period
 * 
 * <p> Note: This validation is invoked only if the detailLevel is {@link TORequest#DL_TIMEOFF_CHOICES_LENGTH DL_TIMEOFF_CHOICES_LENGTH}.
 * See {@link com.bluepumpkin.ejb.rm.requests.timeoff.ejb.TORequestManagerEJB#m_softValidatorDescMandArr m_softValidatorDescMandArr}
 * for further details.
 */
public class TimeOffChoiceContainsHoliday implements Validator {
    
    private static String m_className = TimeOffChoiceContainsHoliday.class.getName();
    private static Category m_cat = Log.initCategory(m_className);
    
    /**
     * Validate a time off request.  Ensure that the request is for time off
     * when the employee might be working.
     */
    @Override
	public ValidationResult validate(Validatable validatable) throws Exception
    {
         String methodName = "validate";
         if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		}

         TORequest tor = (TORequest)validatable;

         // Check against the hours of operation for the employee's organization
         // in the date range
         ValidationResult result = checkNonWorkingHolidays(tor);

         if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
		}

         return result;
    }

    /**
     * Check the choices in the time off request against the non-working
     * holidays specified for the organization.
     */
    protected ValidationResult checkNonWorkingHolidays(TORequest tor) throws Exception {
       ValidationResult result = null;
        boolean checkIfTimeOffBidRequest =false;
   		ID timeOffType = tor.getTimeOffType();
   		ID employeeId = tor.getEmployeeID();
   		TimeOffBid bid  = getOpenTimeOffBidForActivityIdAndEmployeId(timeOffType, employeeId, tor.getSubmittedOn());
   		boolean checkIfBidOpen = bid!=null?true:false;
   		long bidRequestStart =0;
   		long bidRequestEnd =0;
   		if(checkIfBidOpen){
   			bidRequestStart = bid.getRequestStartTime().getTime();
			bidRequestEnd = bid.getRequestEndTime().getTime();
   		}
        TimeMap paidHoldiaysTimeMap = getTOValidationCache(tor).getHolidaysNonWorkingTimeMap();

       // Loop through the choices and check that whether any of them has an overlap with a working holiday

       for (Iterator<TOChoice> it = tor.getRequestChoiceList().iterator(); it.hasNext(); ) {
         TOChoice c = it.next();
         	if(checkIfBidOpen){
	         	long choiceStart = c.getStartDate().getTime();
				long choiceEnd = c.getEndDate().getTime();
				boolean choicePeriodInBidRequestPeriod = bidRequestStart <= choiceStart && choiceEnd <= bidRequestEnd;
				if (choicePeriodInBidRequestPeriod) {
					checkIfTimeOffBidRequest = true;
				}
			}
         //Skip validation for time off bid requests , QC178433
         if (!checkIfTimeOffBidRequest&&paidHoldiaysTimeMap.onBitsOverlapRange(c.getStartDate(), c.getEndDate() )) {
             // The given range is partially overlapping with holidays.  
             result = ValidationUtil.setSoftValidationResult(c,
                 RmEjbBundleKey.TO_CHOICE_CONTAINS_HOLIDAY, c.getStartDate(), 
				 		c.getEndDate(), m_className);
         }
       } 

       return result;
    }
    protected TimeOffBid getOpenTimeOffBidForActivityIdAndEmployeId(ID timeOffType, ID employeeId, Date submittedDate) throws Exception {
		return LicenseUtil.isAdvancedRMLicense()?TimeOffBidUtil.getOpenTimeOffBidForActivityIdAndEmployeId(timeOffType, employeeId, submittedDate):null;
	}
    protected TOValidationCache getTOValidationCache(TORequest toRequest) {
		return toRequest.getCache();
	}
}
