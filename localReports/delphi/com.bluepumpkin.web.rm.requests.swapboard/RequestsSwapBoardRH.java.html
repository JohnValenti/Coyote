package com.bluepumpkin.web.rm.requests.swapboard;

import java.util.*;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.localization.Localizer;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.bluepumpkin.ejb.rm.requests.swap.posting.model.ShiftSwapPosting;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.WebException;
import com.witness.web.uif.base.WorkpaneRequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.pagemodel.WorkpanePageModel;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.l10n.RmWebLogBundleKey;

/**
 * Title:        RequestsSwapBoardRH
 * Description:  Handle Swap Board Requests
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, inc
 * @author       David Su
 * @version 	   1.0
 */
public class RequestsSwapBoardRH extends WorkpaneRequestHandler {
	private static Category m_log = Log.initCategory(RequestsSwapBoardRH.class.getName());

	/**
	 * Process Request for the User page
	 *
	 * @param context The Request Contex
	 */
	public void processCustomRequest(RequestContext context) throws WebException {
		// Process requested action
		if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		} else if (isAction(context, PageAction.DELETE)) {
			processDel(context);
		}
	}

	/**
	 * Return Page Model to perform generic action request on
	 */
	public WorkpanePageModel getPageModel(RequestContext context) {
		RequestsSwapBoardPM model = new RequestsSwapBoardPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		return model;
	}

	/**
	 * Process the Save action
	 *
	 * @param context The Request Context
	 */
	private void processSave(RequestContext context) {
		// Security check
		if (!isAuthorized(context, PrivilegeKeys.SS_READWRITETOSWAPBOARD)) return;

		// Extract Form Parameter
		SwapBoardPostingFormPopupPM model = new SwapBoardPostingFormPopupPM(context);
		model.extractParameters(context.getRequest());

		try {
			if (model.validate()) {
				ShiftSwapPosting posting = model.getShiftSwapPosting();	
				if (model.isNew()) {
					posting.setPostingDate(new Date());
					RequestsSwapBoardMH.createPosting(posting);
				} else {
					RequestsSwapBoardMH.updatePosting(posting);
				}
				model.setIsCloseAfterLoad(true);
			}
		} catch(RmHardValidationException ex) {
			m_log.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_POSTING, ex);
			Message msg = new Message(Message.ERROR_TYPE,	RmWebBundleKey.UNABLE_TO_SAVE_POSTING, RmWebBundleKey.BUNDLE_NAME);
			msg.setAssociatedMessage( getLocMsgFromException(context, ex) );
			model.addPageMessage(msg);
		} catch (Exception ex) {
			m_log.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_POSTING, ex);
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_POSTING, RmWebBundleKey.BUNDLE_NAME);
		}

		context.setPageModel(model);
	}

	/**
	 * Process the Role Save action
	 *
	 * @param context The Request Context
	 */
	private void processDel(RequestContext context) {
		// Security check
		if (!isAuthorized(context, PrivilegeKeys.SS_READWRITETOSWAPBOARD)) return;

		// Extract Form Parameter
		RequestsSwapBoardPM model = new RequestsSwapBoardPM(context);
		model.extractParameters(context.getRequest());

		if (model.validate() && model.hasPerformActionOnID()) {
			try {
				// Delete Posting
				ID postingID = model.getPerformActionOnIDs()[0];
				RequestsSwapBoardMH.deletePosting(postingID);
			} catch (Exception ex) {
				m_log.l7dError(RmWebLogBundleKey.UNABLE_TO_WITHDRAW_POSTING, ex);
				model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_WITHDRAW_POSTING, RmWebBundleKey.BUNDLE_NAME);
			}
		}

		context.setPageModel(model);
	}

	/**
	 * Return Localized Message from Exception
	 */
	protected String getLocMsgFromException(RequestContext context, RmHardValidationException ex) {
		Localizer localizer = context.getLocalizer();
		TimeZone tz = context.getViewingTimeZone();
		return ex.getLocalizedMessage(localizer, tz);
	}
}
