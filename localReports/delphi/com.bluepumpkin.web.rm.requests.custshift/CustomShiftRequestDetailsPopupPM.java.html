				CustomShiftRequestsMH.CSObjectData csObjects = CustomShiftRequestsMH.getCSObjectDataForDetailsDialog(m_context, m_customShiftRequest);

	/**
	 * Return true if the hashCode of the ShiftAssignment object matches the one in the CustShiftReq hashCode field.
	 */
	public boolean checkHashCode(ShiftAssignment sa, CustShiftReq csr)
	{
		String sHashCodeFromDB = m_customShiftRequest.getHashCodeForUndrlyngShift();
		if (sHashCodeFromDB==null)
			return false;
		
		try
		{
			int iHashCodeFromDB = Integer.parseInt(sHashCodeFromDB);			
			return (iHashCodeFromDB == sa.hashCode());
		}
		catch (NumberFormatException ex)
		{
			return false;
		}
	}

	/**
	 * Get a Pair of Dates, where the first one is the start of the currently selected day,
	 * and the second one is the end of the currently selected day. This range does NOT take
	 * the selected startTime into consideration.
	 * This range is NOT aligned with org day boundary. Should it be?
	 * 
	 * @param existingRequest - if null, we will get the current day set in the m_datePickerPC. If not null, 
	 * 		  	we will get the day from the existingRequest.
	 */
	public Pair getFromToDates()
	{
		Date date = m_customShiftRequest.getShiftStartTime();		
		Pair dates = CustomShiftUtil.getFullDayRange(date, m_view_tz);		
		return dates;
	}

	/**

	 */
	protected void initMainList() 
	{
		super.initMainList();

		String title = i18n(m_bundle, RmWebBundleKey.REQUEST_SUMMARY);
		m_mainContainerPC.setTitle(title);

		m_mainContainerPC.setSummary(i18n(m_bundle, RmWebBundleKey.REQUEST_SUMMARY_TABLE));

		//Employee Name
		m_mainContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.EMPLOYEE_NAME_LABEL), getEmployeeName(m_customShiftRequest.getEmployeeID()));

		//Sub Type
		m_mainContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.REQUEST_TYPE_LABEL), CustomShiftUtil.getRequestSubTypeString(m_customShiftRequest, m_localizer));
		
		//Start Date
		m_mainContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.CSR_REQUEST_START_DATE), CustomShiftUtil.getRequestStartDateString(m_customShiftRequest, m_localizer, m_view_tz));
		
		//End Date

	/**
	 * Initialize Shift List
	 */
	protected void initShiftList() 
	{
		m_shiftContainerPC = new FormTwoColumnPC(m_context);
		m_containerList.add(m_shiftContainerPC);
		m_shiftContainerPC.setWrapperCSSClass(CSSUtil.CLASS_BORDER_DEFAULT);
		m_shiftContainerPC.setLastRowType(FormTwoColumnPC.LAST_ROW_NONE);		
		m_shiftContainerPC.setTitle(i18n(m_bundle, RmWebBundleKey.RM_SA_FILTER_SHIFT));
		m_shiftContainerPC.setSummary(i18n(m_bundle, RmWebBundleKey.SHIFT_DETAILS_TABLE_NONEDITABLE));

		//Shift Start Date
		m_shiftContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.CSR_SHIFT_START_DATE), hasShift() ?
				CustomShiftUtil.getShiftStartDateString(m_customShiftRequest, m_localizer, m_view_tz) : "");
		
		//Shift End Date 
		m_shiftContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.CSR_SHIFT_END_DATE), hasShift() ? 
				CustomShiftUtil.getShiftEndDateString(m_customShiftRequest, m_localizer, m_view_tz) : "");

		//Duration  
		m_shiftContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.CSR_DURATION), hasShift() ? 
				CustomShiftUtil.getDurationString(m_customShiftRequest, m_localizer) : "");

		//Shift 
		m_shiftContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.RM_SA_FILTER_SHIFT), hasShift() ? 
				(m_shift==null?"":m_shift.getName()) : "");
		
		//Activity
		m_shiftContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.RM_SA_FILTER_ACTIVITY), hasShift() ?
				(m_activity==null?"":m_activity.getName()) : "");
		
		//Overtime 
		m_shiftContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.ORG_RM_FILING_RULE_OVERTIME), hasShift() ?
				(m_customShiftRequest.isOT() ? i18n(m_bundle, RmWebBundleKey.YES) : i18n(m_bundle, RmWebBundleKey.NO)) : "");
	}
	
	/**
	 * Initialize the Extension Before & After list container
	 */ 
	private void initExtensionsList() 
	{ 
		String firstTitle = i18n(m_bundle, RmWebBundleKey.CSR_EXTENSION_BEFORE_SHIFT);
		String secondTitle = i18n(m_bundle, RmWebBundleKey.CSR_EXTENSION_AFTER_SHIFT);
		m_extContainerPC = new MyFormFourColumnPC(m_context, firstTitle, secondTitle);			
		m_extContainerPC.setWrapperCSSClass(CSSUtil.CLASS_BORDER_DEFAULT);
		m_extContainerPC.setLastRowType(FormTwoColumnPC.LAST_ROW_NONE);
		m_extContainerPC.setSummary(i18n(m_bundle, RmWebBundleKey.EXTENSION_DETAILS_TABLE));

		//Before/After Type 
		m_extContainerPC.addRowData(m_localizer.i18n(m_bundle, RmWebBundleKey.CSR_TYPE), 
				hasExtBefore() ? m_extBeforeType.getName() : "", 
				m_localizer.i18n(m_bundle, RmWebBundleKey.CSR_TYPE),
				hasExtAfter() ? m_extAfterType.getName() : "");
		
		//Before/After Activity 
		m_extContainerPC.addRowData(m_localizer.i18n(m_bundle, RmWebBundleKey.RM_SA_FILTER_ACTIVITY), 
				hasExtBefore() ? m_extBeforeActivity.getName() : "", 
				m_localizer.i18n(m_bundle, RmWebBundleKey.RM_SA_FILTER_ACTIVITY),
				hasExtAfter() ? m_extAfterActivity.getName() : "");
		
		//Before/After Duration strings
		m_extContainerPC.addRowData(m_localizer.i18n(m_bundle, RmWebBundleKey.CSR_DURATION), 
				getExtBeforeDurationString(), 
				m_localizer.i18n(m_bundle, RmWebBundleKey.CSR_DURATION),
				getExtAfterDurationString());
		
		//Before/After Gap 
		m_extContainerPC.addRowData(m_localizer.i18n(m_bundle, RmWebBundleKey.CSR_GAP), 
				hasExtBefore() ? CustomShiftUtil.getExtBeforeGapString(m_customShiftRequest, m_localizer) : "", 
				m_localizer.i18n(m_bundle, RmWebBundleKey.CSR_GAP),
				hasExtAfter() ? CustomShiftUtil.getExtAfterGapString(m_customShiftRequest, m_localizer) : "");
		
		//Before/After OT 
		m_extContainerPC.addRowData(
				i18n(m_bundle, RmWebBundleKey.ORG_RM_FILING_RULE_OVERTIME), 
				hasExtBefore() ? (m_customShiftRequest.getExtBeforeIsOT()?i18n(m_bundle, RmWebBundleKey.YES) : i18n(m_bundle, RmWebBundleKey.NO)) : "",
						i18n(m_bundle, RmWebBundleKey.ORG_RM_FILING_RULE_OVERTIME),
				hasExtAfter() ? (m_customShiftRequest.getExtAfterIsOT()?i18n(m_bundle, RmWebBundleKey.YES) : i18n(m_bundle, RmWebBundleKey.NO)) : "");
		
		m_containerList.add(m_extContainerPC);
	}
