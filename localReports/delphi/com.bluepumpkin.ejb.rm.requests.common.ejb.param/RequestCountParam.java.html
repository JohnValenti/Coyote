package com.bluepumpkin.ejb.rm.requests.common.ejb.param;

import java.util.HashSet;
import java.util.Set;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;

public class RequestCountParam {
	private Set<ID> employeeIDs = new HashSet<ID>();
	private TimeRange timeRange;
	private Set<String> statuses = new HashSet<String>();
	private String requestType;

	public Set<ID> getEmployeeIDs() {
		return this.employeeIDs;
	}

	public RequestCountParam addEmployeeID(ID employeeID) {
		if (!this.employeeIDs.contains(employeeID)) {
			this.employeeIDs.add(employeeID);
		}
		return this;
	}

	public RequestCountParam addAllEmployeeIDs(Set<ID> employeeIDs) {
		this.employeeIDs.addAll(employeeIDs);
		return this;
	}

	public TimeRange getTimeRange() {
		return this.timeRange;
	}

	public RequestCountParam setTimeRange(TimeRange timeRange) {
		this.timeRange = timeRange;
		return this;
	}

	public Set<String> getStatuses() {
		return this.statuses;
	}

	public RequestCountParam addStatus(String status) {
		if (!this.statuses.contains(status)) {
			this.statuses.add(status);
		}
		return this;
	}

	public String getRequestType() {
		return this.requestType;
	}

	public RequestCountParam setRequestType(String requestType) {
		this.requestType = requestType;
		return this;
	}
}
