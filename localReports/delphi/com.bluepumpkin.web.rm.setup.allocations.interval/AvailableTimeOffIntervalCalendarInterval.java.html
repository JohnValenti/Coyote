package com.bluepumpkin.web.rm.setup.allocations.interval;

import java.text.MessageFormat;
import java.text.NumberFormat;
import java.text.ParseException;
import java.text.SimpleDateFormat;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendarInterval;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOIntervalAllocation;
import com.bluepumpkin.ejb.rm.util.RmMath;

public class AvailableTimeOffIntervalCalendarInterval {

	private String startTime;
	private String endTime;
	private String allocatedHours;

	/**
	 * Serialization constructor.
	 */
	public AvailableTimeOffIntervalCalendarInterval() { // NOSONAR
	}

	public AvailableTimeOffIntervalCalendarInterval(TOIntervalCalendarInterval interval, SimpleDateFormat dateTimeAttributeFormat,
			NumberFormat numberFormat) {
		this();

		startTime = dateTimeAttributeFormat.format(interval.getStartTime());
		endTime = dateTimeAttributeFormat.format(interval.getEndTime());
		if (interval.getAllocatedHours() != null) {
			allocatedHours = numberFormat.format(interval.getAllocatedHours());
		}
	}

	public String getAllocatedHours() {
		return allocatedHours;
	}

	public void setAllocatedHours(String allocatedHours) {
		this.allocatedHours = allocatedHours;
	}

	public String getEndTime() {
		return endTime;
	}

	public void setEndTime(String endTime) {
		this.endTime = endTime;
	}

	public String getStartTime() {
		return startTime;
	}

	public void setStartTime(String startTime) {
		this.startTime = startTime;
	}

	/**
	 * Gets a new TOIntervalAllocation based on the instance values.
	 * 
	 * @throws ParseException
	 */
	public TOIntervalAllocation toIntervalAllocation(ID poolID, SimpleDateFormat dateTimeAttributeFormat, NumberFormat numberFormat)
			throws ParseException {

		TOIntervalAllocation interval = new TOIntervalAllocation();
		if (allocatedHours != null) {
			allocatedHours = allocatedHours.trim();
		}
		if (allocatedHours.length() == 0) {
			interval.setAllocatedHours(0);
		} else {
			float hours = numberFormat.parse(allocatedHours).floatValue();
			interval.setAllocatedHours(RmMath.roundToDecimalPlace(hours, AvailableTimeOffIntervalCalendarPC.NUMBER_OF_DECIMALS));
		}
		interval.setEndTime(dateTimeAttributeFormat.parse(endTime));
		interval.setStartTime(dateTimeAttributeFormat.parse(startTime));
		interval.setTOPoolID(poolID);
		return interval;
	}

	@Override
	public String toString() {
		return MessageFormat.format("Start Time: {0}, End Time: {1}, Allocated Hours: {2}", startTime, endTime, allocatedHours);
	}
}
