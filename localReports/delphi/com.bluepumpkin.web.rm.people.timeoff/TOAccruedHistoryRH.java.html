package com.bluepumpkin.web.rm.people.timeoff;

import java.util.Collection;

import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

public class TOAccruedHistoryRH extends RequestHandler {

	public void processRequest(RequestContext context) throws Exception {
		if (isAction(context, PageAction.DELETE)) {
			processDelete(context);
		}
	}

	public void processDelete(RequestContext context) {
		TOAccruedHistoryPU model = new TOAccruedHistoryPU(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		
		Collection accruedIDs = model.getTOAccruedIDs();
		try {
			BbmManagerFactory.getWorkResourceManager(context.isInWhatIfMode())
					.deleteTOAccruedHisory(accruedIDs);
		} catch (Exception e) {
			model.addPageMessage(Message.ERROR_TYPE,
					BbmWebBundleKey.DELETE_FAILURE, BbmWebBundleKey.BUNDLE_NAME);
		}
	}

}
