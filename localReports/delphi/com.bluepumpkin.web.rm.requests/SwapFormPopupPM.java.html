	/** Creates a new instance of SwapFormPopupPM */
	public SwapFormPopupPM(RequestContext context, String formAction) {
		super(context, formAction);

		m_mainListPC = new MultiColListPC(context);
        m_mainListPC.setHasRowHeaders(false); //508
		addChildComponent(m_mainListPC);

		m_datepickerPC = new DatePickerPC(context, "dp");
		addChildComponent(m_datepickerPC);
		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
		addRequiredJavaScriptFile(RmJavaScriptFileID.SHIFT_SWAP_FORM_JS);
	}

	/**
	 * Initialize Model with Data
	 */
	protected void initializeModel() {
		if (m_isInitialized) return; else m_isInitialized = true;

		try	{
			m_orgSetting = RequestsMH.getOrgSettingForEmployee(m_context, m_empID);

			// Startday of datepicker in Shift Swap does not default to organization start date
			String orgID = this.m_orgSetting.getCurrentOrgId();
			m_weekStartDate = m_datepickerPC.getFirstDayOfWeek();
			try {
				Organization org = OrganizationModelHandler.getOrganization(m_context, new ID(orgID));
				m_weekStartDate = org.getWeekStartDate();
			} catch (Exception e) {
				m_log.warn(" First Day of the week could not be set for Shift Swap because of " + e.getMessage());
				//skip the setting of the firstday of the week
			}

			initDataPickersWithOrgWeekStartDay(m_weekStartDate);
			loadSpecificData();
		} catch (RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataError(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		} finally {
			initContentTitle();
			initContent();
			initToolbar();
		}
	}

	/**
	 * Prepare toolbar for rendering
	 */
	protected void initToolbar(){
		ToolbarUtil.addValidateSaveBtn(m_toolbar, isSaveButtonEnabled());
		if (m_isShowDoneInsteadOfCancelBtn) {
			ToolbarUtil.addCloseWindowDoneBtn(m_toolbar, true);
		} else {
			ToolbarUtil.addCloseWindowCancelBtn(m_toolbar, true);
		}
		if (!isSaveButtonEnabled() && m_isShowDoneInsteadOfCancelBtn) {
			ToolbarUtil.addResetBtn(m_toolbar, false);
		} else {
			ToolbarUtil.addResetBtn(m_toolbar, true);
		}

	}


	protected void initDataPickersWithOrgWeekStartDay(int weekStartDay) {
		m_datepickerPC.setFirstDayOfWeek(weekStartDay);
	}

	protected void setDateRangePickerStartAndEnd(DateRangePickerPC dateRangePickerPC){
		//date minutes should be factor of 15
		Calendar startCal = Calendar.getInstance(m_context.getViewingTimeZone());
		startCal.setTime(new Date());
		while ((startCal.get(Calendar.MINUTE)%15) != 0 ){
			startCal.add(Calendar.MINUTE, 1);
		}
		Calendar endCal = Calendar.getInstance(m_context.getViewingTimeZone());
		endCal.setTime(new Date());
		endCal.add(Calendar.DATE, 1);
		endCal.set(Calendar.HOUR_OF_DAY, 0);
		endCal.set(Calendar.MINUTE, 0);

		dateRangePickerPC.getStartDatePickerPC().setDate(startCal.getTime());
		dateRangePickerPC.getEndDatePickerPC().setDate(endCal.getTime());
	}

	/**
	 * Initialize Content
	 */
	protected void initContent() {
		if (!m_isAuthorized) return;

		m_mainListPC.getHeader().addColumnHeaderData("", "");
		m_mainListPC.setEnabled(false);
		m_mainListPC.setUseWrapper(false);

		ArrayList listModel = new ArrayList();
		addListContent(listModel);
		m_mainListPC.setListModel(listModel);

        m_mainListPC.setTableAttribute("summary", i18n(m_bundle, RmWebBundleKey.SWAP_TABLE_SUMMARY));
	}

	/**
	 * Add List Content
	 */
	protected abstract void addListContent(List listModel);

	/**
	 * Return new offer Expiration Date
	 */
	protected Date createNewOfferExpDate() {
		Calendar cal = Calendar.getInstance(m_view_tz, m_locale);
		DateTimeUtil.adjustToDayEnd(cal);
		return cal.getTime();
	}

	/**
	 * Create Shift Swap Type (one way / two way) Drop Down
	 */
	protected String createShiftSwapTypeDD(boolean isOnChangeRefreshPage) {
		String twoWayLabel = i18n(m_bundle, RmWebBundleKey.SWAPTYPE_TWOWAY);
		String oneWayLabel = i18n(m_bundle, RmWebBundleKey.SWAPTYPE_ONEWAY);

		if (m_orgSetting!=null && !m_orgSetting.getDisableOneWayShiftSwap()) {
			List options = new ArrayList(2);
			options.add( new StringsPair(ShiftSwapRequest.SHIFTSWAP_TYPE_TWOWAY, twoWayLabel) );
			options.add( new StringsPair(ShiftSwapRequest.SHIFTSWAP_TYPE_ONEWAY, oneWayLabel) );

			String selectedValue = getSwapType();
			String onChangeJS = isOnChangeRefreshPage?JSUtil.REFRESH_PAGE_NO_CONFIRM
					:JSUtil.ON_CHANGE;

			//return HtmlUtil.makeSelectInput(SWAPTYPE_FN, selectedValue, onChangeJS, options);

            //508 TBD: passing selectedValue instead of selectedText. SHould still work, but let's see!
            String swapTypeLabel = i18n(m_bundle, RmWebBundleKey.SWAPTYPE); //508
            return HtmlUtil.makeSelectInput(SWAPTYPE_FN, options, null, selectedValue, swapTypeLabel, onChangeJS, true);

		} else {
			return twoWayLabel + HtmlUtil.
					makeHiddenInput(SWAPTYPE_FN, ShiftSwapRequest.SHIFTSWAP_TYPE_TWOWAY);
		}
	}

	/**
	 * Return localize shift Swap Type (one way / two way)
	 */
	public String getShiftSwapTypeDisplay() {
		String swapType = getSwapType();
		String rbKey = ShiftSwapRequest.SHIFTSWAP_TYPE_ONEWAY.equals(swapType) ?
				RmWebBundleKey.SWAPTYPE_ONEWAY:
				RmWebBundleKey.SWAPTYPE_TWOWAY;

		return i18n(m_bundle, rbKey);
	}

	/**
	 * Return localize shift Swap Type (one way / two way) with Hidden Input
	 */
	protected String getShiftSwapTypeDisplayWithHiddenInput() {
		return getShiftSwapTypeDisplay() +
				HtmlUtil.makeHiddenInput(SWAPTYPE_FN, getSwapType());
	}

	/**
	 * Return true if it is a Two Way Swap
	 */
	public boolean isTwoWaySwap() {
		return ShiftSwapRequest.SHIFTSWAP_TYPE_TWOWAY.equals(getSwapType());
	}

	/**
	 * Add Negotiation Option Row
	 */
	protected void addPickupOptionRow(List listModel, boolean allowPartial) {
		Object[] args = new Object[1];
		args[0] = m_context.getSystemManagerFactory()
		  .getVerticalizationManager().getVerticalizedString(m_context, DefaultVerticalizationManager.WORKERSMALL);

		StringBuffer sb = new StringBuffer(size);
		String label_pickupEntire_v = MessageFormat.format(i18n(m_bundle, RmWebBundleKey.SWAPBOARD_PICKUP_ENTIRE_POSTING_SENTENCE), args);
		String input_toAgt = HtmlChoiceInput.makeRadioInput(ALLOWPARTIAL_FN, "false", label_pickupEntire_v, !allowPartial, JSUtil.ON_CHANGE);


		String label_pickupPartial_v = MessageFormat.format(i18n(m_bundle, RmWebBundleKey.SWAPBOARD_PICKUP_PARTIAL_POSTING_SENTENCE), args);
		String input_toMgr = HtmlChoiceInput.makeRadioInput(ALLOWPARTIAL_FN, "true", label_pickupPartial_v, allowPartial, JSUtil.ON_CHANGE);

		sb.append(input_toAgt)
				.append("<br>")
				.append(input_toMgr);


		addDataRow(sb.toString(), listModel);
	}

	/**
	 * Add Negotiation Option Row
	 */
	protected void addNegotiationOptionRow(List listModel, boolean isNegotiable,
			String rbKey_allowFeedback, String rbKey_submitToMgr) {
		String label_toAgt = i18n(m_bundle, rbKey_allowFeedback);
		String input_toAgt = HtmlChoiceInput.makeRadioInput(NEGOTIATION_FN, "true", label_toAgt, isNegotiable, JSUtil.ON_CHANGE);

		String label_toMgr = i18n(m_bundle, rbKey_submitToMgr);
		String input_toMgr = HtmlChoiceInput.makeRadioInput(NEGOTIATION_FN, "false", label_toMgr, !isNegotiable, JSUtil.ON_CHANGE);


		StringBuffer sb = new StringBuffer(size);
		sb.append(input_toAgt)
				.append("<br>")
				.append(input_toMgr);
		addDataRow(sb.toString(), listModel);
	}

	protected void addNegotiationOptionRow(List listModel, boolean isNegotiable,
			String rbKey_allowFeedback, String rbKey_submitToMgr, Object[] args) {

		String label_toAgt = i18n(m_bundle, rbKey_allowFeedback);
		String label_toAgt_v = MessageFormat.format(label_toAgt, args);
		String input_toAgt = HtmlChoiceInput.makeRadioInput(NEGOTIATION_FN, "true", label_toAgt_v, isNegotiable, JSUtil.ON_CHANGE);

		String label_toMgr = i18n(m_bundle, rbKey_submitToMgr);
		String label_toMgr_v = MessageFormat.format(label_toMgr, args);

		StringBuffer sb = new StringBuffer(size);
		sb.append(input_toAgt)
				.append("<br>")
				.append(input_toMgr);
		addDataRow(sb.toString(), listModel);
	}

	/**
	 * Initialize Date Picker
	 */
	private DatePickerPC initDatePicker(String fn, Date date, boolean isTimeEnabled) {
		m_datepickerPC.reset();
		m_datepickerPC.setFirstDayOfWeek(m_weekStartDate);
		m_datepickerPC.setDateInputFieldName(fn);
		m_datepickerPC.setIsTimeEnabled(isTimeEnabled);
		if (date!=null) m_datepickerPC.setDate(date);
		return m_datepickerPC;
	}

	/**
	 * Return Date Picker for specified params
	 */
	protected DatePickerPC getDatePicker(String fn, Date date, boolean isTimeEnabled) {
		DatePickerPC datePickerPC = initDatePicker(fn, date, isTimeEnabled);
		datePickerPC.setLowerLimit(m_nowDate); // Prevent Date in the past
		return datePickerPC;
	}

	/**
	 * Return Date Picker HTML Display for specified params
	 */
	protected String getDatePickerHTML(String fn, Date date, boolean isTimeEnabled) {
		return getDatePicker(fn, date, isTimeEnabled).getHtmlDisplay();
	}

	/**
	 * Return Date Picker HTML Display
	 */
	protected String getReadOnlyDate(String fn, Date date, boolean isTimeEnabled) {
		DatePickerPC dpPC = initDatePicker(fn, date, isTimeEnabled);
		dpPC.setInputType(DatePickerPC.INPUT_TYPE_READONLY);
		dpPC.setIsPickerEnabled(false);
		return dpPC.getHtmlDisplay();
	}

	/**
	 * Extract Date Parameter
	 */
	protected Date extractDate(String fn, boolean isTimeEnabled) {
		initDatePicker(fn, null, isTimeEnabled);
		return m_datepickerPC.extractDate(fn);
	}
