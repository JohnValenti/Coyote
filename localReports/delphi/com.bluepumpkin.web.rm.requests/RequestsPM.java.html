	/**
	 * Initialize for Display
	 */
	@Override
	public void initForDisplay() {
		if (m_isInitializedForDisplay) return;
		else {
			boolean isAuthToViewRtqs = true;
			if (Request.REQUESTTYPE_ALL.equals(m_requestType)) {
				isAuthToViewRtqs = isAuthToViewAnyRqts();
				if (isAuthToViewRtqs) {
					initTimeOffPopupWindowArgs();
					if(LicenseUtil.isAdvancedRMLicense()) {
						initFlexTimePopupWindowArgs();
					}
					initShiftSwapPopupWindowArgs();
					initShiftBidPopupWindowArgs();
					initCustomShiftPopupWindowArgs();

					initNewRequestMenu();
				}
			} else if (Request.REQUESTTYPE_TIMEOFF.equals(m_requestType)) {
				isAuthToViewRtqs = isAuthToViewTORqts();
				if (isAuthToViewRtqs) initTimeOffPopupWindowArgs();
			} else if (Request.REQUESTTYPE_FLEXTIME.equals(m_requestType)) {
				isAuthToViewRtqs = isAuthToViewFTRqts();
				if (isAuthToViewRtqs && LicenseUtil.isAdvancedRMLicense()) {
					initFlexTimePopupWindowArgs();
				}
			} else if (Request.REQUESTTYPE_SHIFTSWAP.equals(m_requestType)) {
				isAuthToViewRtqs = isAuthToViewSSRqts();
				if (isAuthToViewRtqs) initShiftSwapPopupWindowArgs();
			} else if (Request.REQUESTTYPE_SHIFTBID.equals(m_requestType)) {
				isAuthToViewRtqs = isAuthToViewSBRqts();
				if (isAuthToViewRtqs) initShiftBidPopupWindowArgs();
			} else if (Request.REQUESTTYPE_CUSTSHIFT.equals(m_requestType)) {
				isAuthToViewRtqs = isAuthToViewCSRqts();
				if (isAuthToViewRtqs) initCustomShiftPopupWindowArgs();
			}

			if (m_showFilterAndActiveComponent)
				if (!isAuthToViewRtqs) addUnauthorizedToViewMsg();
			if (!m_isPageEnabled) addPageNotEnabledMsg();

			super.initForDisplay();
		}
	}

	/**
	 * Create Icon Button
	 */
	protected String createIconButton(String src, String alt, String onClickJS)
    {
        StringBuffer sb = new StringBuffer();
		sb.append("<span class=\"itemActionWpr\"><span class=\"itemAction\">");

        if (onClickJS != null) {
            //For 508 Accessibility compliance, the action icons must be focusable links
            String img = HtmlUtil.imageTag(null, src, null, null, alt, null, null, true, "0");
            sb.append(HtmlLinkUtil.createActiveLink("", img, alt, alt, null, onClickJS));
        } else {
            HtmlUtil.imageTag(null, src, null, null, alt, null, onClickJS, true);
        }

        sb.append("</span></span>");
        return sb.toString();
	}

	/**
	 * Return true if the Action is valid for the specified Request
	 */
	protected boolean isValidAction(String action, RequestAggregate request) {
		String status = request.getRequestStatus();

		return PageAction.VIEW_DETAIL.equals(action) ||	!request.hasExpired() &&
				(
					(
					    ( RequestAuditTrail.STATUS_WAITLIST.equals(status) ||
					      RequestAuditTrail.STATUS_PENDING.equals(status) ||
					      RequestAuditTrail.STATUS_TENTATIVE.equals(status) ||
					      RequestAuditTrail.STATUS_ESCALATED.equals(status) ||
					      RequestAuditTrail.STATUS_NEGOTIATION.equals(status) )
						&&
					    ( PageAction.EDIT.equals(action) ||	PageAction.WITHDRAW.equals(action) )
					)
					||
					(
						RequestAuditTrail.STATUS_NEGOTIATION.equals(request.getRequestStatus()) &&
					    PageAction.SUBMIT_TO_MGR.equals(action) &&
					    canSubmitToManager(action, request)
					)
					||
					(
						request.getRequestType().equals(Request.REQUESTTYPE_SHIFTBID) &&
						RequestAuditTrail.STATUS_APPROVED.equals(request.getRequestStatus()) &&
						PageAction.WITHDRAW.equals(action)
					)
				);
	}

	/**
	 * Create List Item Actions
	 */
	public String createItemActions(RequestAggregate request, String rowID) {
		StringBuffer sb = new StringBuffer(200);

		sb.append("<nobr>");
		appendItemAction(request, PageAction.EDIT, rowID, sb);
		if (!request.getRequestStatus().equals(RequestAuditTrail.STATUS_APPROVED)){
			//Submit to manager of request should only occur if not approved.
			//Required to not show icon in case of submit of withdrawal
			appendItemAction(request, PageAction.SUBMIT_TO_MGR, rowID, sb);
		}

		//Note: only one or neither of the add to waitlist actions will be shown at any given time
		appendItemAction(request, RmPageAction.ADD_TO_WAITLIST_ACTION, rowID, sb);
		appendItemAction(request, RmPageAction.ADD_TO_MAN_WAITLIST_ACTION, rowID, sb);

		appendItemAction(request, PageAction.DENY, rowID, sb);
		appendItemAction(request, PageAction.ESCALATE, rowID, sb);
		appendItemAction(request, PageAction.WITHDRAW, rowID, sb);
		appendItemAction(request, PageAction.WITHDRAW_NEGOTIATION, rowID, sb);
		appendItemAction(request, PageAction.VIEW_DETAIL, rowID, sb);
		appendItemAction(request, PageAction.TENTATIVELY_APPROVE, rowID, sb);
		appendItemAction(request, PageAction.APPROVE, rowID, sb);
		sb.append("</nobr>");

        if (request.isTimeOffRequest() && request.getRequestStatus().equals(RequestAuditTrail.STATUS_APPROVED)) {
            TOWithdraw toWithdraw = ((TORequest) request).getWithdrawInfo();
            if (toWithdraw != null && RequestAuditTrail.isRequestForWithdrawalStates(toWithdraw.getRequestStatus())) {
                sb.append("<BR>");
                appendItemAction(request, PageAction.ACCEPT_WITHDRAW_ACTION, rowID, sb);
                appendItemAction(request, PageAction.REJECT_WITHDRAW_ACTION, rowID, sb);
                appendItemAction(request, PageAction.CANCEL_WITHDRAW_ACTION, rowID, sb);
            }
        }

        if (request.isShiftSwapRequest() && request.getRequestStatus().equals(RequestAuditTrail.STATUS_APPROVED)) {
            ShiftSwapWithdraw ssWithdraw = ((ShiftSwapRequest) request).getWithdrawInfo();
            if (ssWithdraw != null && RequestAuditTrail.isRequestForWithdrawalStates(ssWithdraw.getRequestStatus())) {
                sb.append("<BR>");
        		appendItemAction(request, PageAction.SUBMIT_TO_MGR, rowID, sb);
                appendItemAction(request, PageAction.ACCEPT_WITHDRAW_ACTION, rowID, sb);
                appendItemAction(request, PageAction.REJECT_WITHDRAW_ACTION, rowID, sb);
                appendItemAction(request, PageAction.CANCEL_WITHDRAW_ACTION, rowID, sb);
            }
        }
		return sb.toString();
	}

    protected boolean isAuthToProcessCancelWithdrawRequest(RequestAggregate request) {
        String rType = request.getRequestType();
        boolean result = false;
        if (Request.REQUESTTYPE_TIMEOFF.equals(rType) && ((TORequest) request).isEligibleForCancelWithdrawAction()) {
            result = isAuthToUpdateRqt(request);
        }
        if (Request.REQUESTTYPE_FLEXTIME.equals(rType) && ((TORequest) request).isEligibleForCancelWithdrawAction()) {
            result = isAuthToUpdateRqt(request);
        }
        if (Request.REQUESTTYPE_SHIFTSWAP.equals(rType) && ((ShiftSwapRequest) request).isEligibleForCancelWithdrawAction()) {
            result = isAuthToUpdateRqt(request);
        }
        return result;
    }

    //This method is overridden in implementing classes
    protected boolean isAuthToUpdateRqt(RequestAggregate request) {
        return false;
    }

	/**
	 * Return true if User is authorized to waitlist the specified Request.
	 * Both agents and managers use the same privilege for this.
	 */
	protected boolean isAuthToWaitlistRqt(RequestAggregate request) {
		String rType = request.getRequestType();

		boolean result = false;
		if (Request.REQUESTTYPE_TIMEOFF.equals(rType)) {
			ID privID = PrivilegeKeys.TOM_ADDREQUESTTOWAITLIST_ID;
			result = isTOAuthPriv(request, privID);
		}
		return result;
	}

    /**
     * Return true if User is authorize to approve specified Time Off Request
     */
	protected boolean isTOAuthPriv(RequestAggregate request, ID privID) {
		return m_user.isAuthorized(privID);
	}

	protected boolean isFTAuthPriv(RequestAggregate request, ID privID) {
		return m_user.isAuthorized(privID);
	}

        try
        {
            String pageIDSizeFetched = BbmManagerFactory.getDBConfigManager().getValue(ConfigKey.RM_PAGE_ID_SIZE_LIMIT);
            if ( pageIDSizeFetched != null )

        }
    }

    /**
     * Return the number of records shown. Mainly used by pagination control.
     * We override the parent's method so that the cached page id size, if
     * larger than the limiting page id size or less than 0, is not used.
     */
    @Override
	public int getPageIDSizeWithoutNormalizing() {
        int pageIDSize = super.getPageIDSizeWithoutNormalizing();
        if ( m_limitingPageSize > 0 && ( pageIDSize > m_limitingPageSize || pageIDSize < 0 ) ) {
            pageIDSize = m_limitingPageSize;
        }
        return pageIDSize;
