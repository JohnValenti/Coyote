package com.bluepumpkin.ejb.rm.requests.timeoff.ejb;

/**
 * Title:        Message-driven bean for handling the TO Notifications for purpose of scanning waitlists or refreshinh cached TOHoursPerDay.
 * Description:  Implementation of Notification handler message-driven bean
 * Copyright:    Copyright (c) 2006
 * Company:      Witness Systems
 * @author       Sameet Joshi July 2006
 * @version 1.0
 */

import java.util.Calendar;
import java.util.Map;
import java.util.concurrent.ConcurrentHashMap;

import javax.ejb.CreateException;
import javax.ejb.MessageDrivenBean;
import javax.ejb.MessageDrivenContext;
import javax.jms.JMSException;
import javax.jms.Message;
import javax.jms.MessageListener;
import javax.jms.ObjectMessage;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbLogBundleKey;
import com.bluepumpkin.ejb.bbm.timeoffcalculator.TONotifyMessage;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.ejb.TOHoursPerDayManager;
import com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb.TimeOffIntervalNotificationHandler;
import com.bluepumpkin.ejb.rm.requests.timeoff.waitlist.ejb.TOWaitlistManager;

public class TONotificationHandlerMDB implements MessageDrivenBean, MessageListener {

	private static final long serialVersionUID = 1L;


	private transient TOWaitlistManager toWaitlistManager;
	private transient TOHoursPerDayManager toHoursPerDayManager;

	private static final Category LOG = Log.initCategory(TONotificationHandlerMDB.class.getName());
	private static final int MSG_REDELIVERY_ATTEMPTS = 4;
	private static final Map<String, Integer> failedMsgIds = new ConcurrentHashMap<String, Integer>();

	public void ejbCreate() throws CreateException {
		try {
			toWaitlistManager = RmManagerFactory.getInstance(true).getTOWaitlistManager();
		} catch (Exception e) {
			Object[] param = new Object[]{"ejbCreate", TONotificationHandlerMDB.class.getName()};
			LOG.l7dError(BbmEjbLogBundleKey.UNKNOWN_EXCEPTION_MF, param, e);
			throw new CreateException();
		}
		try {
			toHoursPerDayManager = RmManagerFactory.getInstance(true).getTOHoursPerDayManager();
		} catch (Exception e) {
			Object[] param = new Object[]{"ejbCreate", TONotificationHandlerMDB.class.getName()};
			LOG.l7dError(BbmEjbLogBundleKey.UNKNOWN_EXCEPTION_MF, param, e);
			throw new CreateException();
		}
	}

	@Override
	public void ejbRemove() { // NOSONAR

	}

	@Override
	public void setMessageDrivenContext(MessageDrivenContext messageDrivenContext) { // NOSONAR

	}

	protected static TOWaitlistManager getTOWaitlistManager() throws BbmCreateException {
		return RmManagerFactory.getInstance(true).getTOWaitlistManager();
	}



	/**
	 * This method gets called by the application server when there is a new JMS
	 * message published to the specified topic
	 */
	@Override
	public void onMessage(Message msg) {

		if (!(msg instanceof ObjectMessage)) {
			return;
		}

		TONotifyMessage notifyMessage = null;
		String msgID = null;

		try {
			msgID = msg.getJMSMessageID();

			notifyMessage = convertToNotifyMessage(msg);

			if (notifyMessage == null) {
				return;
			}

			LOG.debug(notifyMessage);

			handleTONotifyMessage(msgID, notifyMessage);

			if (failedMsgIds.containsKey(msgID)) {
				failedMsgIds.remove(msgID);
				LOG.info("Finished processing Successfuly MSGID= (" + msgID + "): failedMsgIds size=" + failedMsgIds.size());
			}

		} catch (Exception e) {
			handleException(msg, msgID, notifyMessage, e);
		}
	}

	private TONotifyMessage convertToNotifyMessage(Message msg) throws JMSException {
		Object obj = ((ObjectMessage) msg).getObject();

		if (!(obj instanceof TONotifyMessage)) {
			Object[] param = new Object[] { msg };
			LOG.l7dError(BbmEjbLogBundleKey.NOTIFICATION_INVALID_MSG_MF, param);
			return null;
		}

		return (TONotifyMessage) (obj);

	}

	private void handleTONotifyMessage(String msgID, TONotifyMessage msgObj) throws Exception { // NOSONAR

		// trace debug data
		traceDebugData(msgID, msgObj);

		logFailedMessage(msgID, msgObj);

		logReceivedMesssage(msgID, msgObj);
		
		TimeOffIntervalNotificationHandler intervalHandler = new TimeOffIntervalNotificationHandler(toHoursPerDayManager);
		
		long timeStart = Calendar.getInstance().getTimeInMillis();

		if (msgObj.isMessageForTOHoursPerDayUpdate()) {
			toHoursPerDayManager.lockRecords(msgObj);
			toHoursPerDayManager.updateTOHoursPerDay(msgObj);
			logDurationAfterOperation(" updateTOHoursPerDay", timeStart);
		}
		
		intervalHandler.onTONotifyMessage(msgObj);

		if (msgObj.isMessageForWaitlistScan()) {
			timeStart = Calendar.getInstance().getTimeInMillis();
			toWaitlistManager.processWaitlistScan(msgObj);
			logDurationAfterOperation("processWaitlistScan", timeStart);
		}
	}

	private void logDurationAfterOperation(String message, long timeStart) {
		if (RequestUtil.isRMDebugEnabled()) {
			long milliSeconds = Calendar.getInstance().getTimeInMillis() - timeStart;
			LOG.info(String.format("%s PERF, after %s, %d Msecs", Thread.currentThread().getName(), message, milliSeconds));
		}
	}

	private void logReceivedMesssage(String msgID, TONotifyMessage msgObj) {
		if (RequestUtil.isRMDebugEnabled()) {
			LOG.info(Thread.currentThread().getName() + "***************Received  MSGID=" + msgID + " msgObj=" + msgObj);
		}
	}

	private void logFailedMessage(String msgID, TONotifyMessage msgObj) {
		if (failedMsgIds.containsKey(msgID)) {
			LOG.info("***Received for processing MSGID= (" + msgID + "): This is Attempt No " + failedMsgIds.get(msgID) + msgObj);
		}
	}

	private void traceDebugData(String msgID, TONotifyMessage msgObj) throws Exception { // NOSONAR
		if(msgObj.isEnableTraceDebugData()) {
			toHoursPerDayManager.traceDebugData(msgID, msgObj, null, 1);
		}
	}

	private void handleException(Message msg, String msgID, TONotifyMessage notifyMessage, Exception e) {

		Object[] param = new Object[] { "onMessage", TONotificationHandlerMDB.class.getName() };
		LOG.l7dError(BbmEjbLogBundleKey.UNKNOWN_EXCEPTION_MF, param, e);

		if (msgID == null) {
			LOG.error("TONotificationHandlerMDB handleException message id is null", e);
			return;
		}

		int count = 0;
		if (failedMsgIds.containsKey(msgID)) {
			count = failedMsgIds.get(msgID);
		}
		count = count + 1;
		failedMsgIds.put(msgID, count);

		safeTraceDebugData(msg, msgID, e, count);
		boolean needRetry = notifyMessage != null && notifyMessage.isRetryDelivery();

		if (count < MSG_REDELIVERY_ATTEMPTS && needRetry) {
			LOG.info("Throwing exception for redelivery  MSGID (" + msgID + ") : This is Attempt No " + failedMsgIds.get(msgID));
			throw new RmRuntimeException("REDELIVER MSGID=" + msgID);
		}

		failedMsgIds.remove(msgID);
		String errorMsg = String.format("Count exceeded maximum allowed attempts (%d); or isRetryDelivery=%s; not redelivering this MSGID (%s) any more ", MSG_REDELIVERY_ATTEMPTS, needRetry ? "Yes" : "No", msgID);
		LOG.error(errorMsg);
	}

	private void safeTraceDebugData(Message msg, String msgID, Exception e, int count) {
		try {
			toHoursPerDayManager.traceDebugData(msgID, msg, e, count);
		} catch (Exception e2) {
			LOG.error("Trace TO notification handler exception failed", e2);
		}
	}

}
