/**
 * An auction bidder can only have one shift bid request approved.  After approval of
 * the first shift bid request, all other shift bid requests (in pending or escalated status)
 * must be marked invalid.
 *
 * Title:        Blue Pumpkin Software Request Management Module
 * Copyright:    Copyright (c) 2001-2004
 * Company:      Blue Pumpkin Software, inc
 * @author       Raja Rajendran
 */
public class NoApprovedShiftBidRequestsForBidderHV implements Validator {
    private static final String m_className = NoApprovedShiftBidRequestsForBidderHV.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(RequestUtil.m_defaultTracePriority, "Validation Rule: " + m_className + " started");

        ShiftBidRequestDAO sbrDAO = null;
        try {
            ShiftBidRequest sbReq = (ShiftBidRequest) validatable;
            sbrDAO = new ShiftBidRequestDAO(ShiftBidRequest.DL_BASIC);

            // get approved requests for employee.
            Collection approvedSBReqs = sbrDAO.getRequestsUsingFilter(new FilterExpr[] {
                new FilterExpr(FilterExpr.LHS_REQUESTSTATUS, FilterExpr.OP_EQUALTO, RequestAuditTrail.STATUS_APPROVED),
                new FilterExpr(FilterExpr.LHS_BIDDERIDS, FilterExpr.OP_EQUALTO, sbReq.getShiftBidderID())
            }, ShiftBidRequest.DL_BASIC);

            // there can only be one approved request in the system for a given shift bidder.
            assert approvedSBReqs.size() <= 1:"approvedSBReqs.size() <= 1";

            // no approved requests, so rule does not fail.
            if (approvedSBReqs.size() == 0) return null;

            // rule fails since we have an approved request.
            ShiftBidRequest approvedSBReq = (ShiftBidRequest) approvedSBReqs.iterator().next();

            ValidationResult valResult = ValidationUtil.setAndLogHardValidationResult(sbReq,
            		RmEjbBundleKey.SBR_APPR_SBR_ALREADY_EXISTS,
								RmEjbLogBundleKey.SBR_APPR_SBR_ALREADY_EXISTS,
								new Serializable[]{approvedSBReq.getID()}, m_className);

            return valResult;
//        } c_atch (Exception e) {
//            t_hrow RequestUtil.createValidationExceptionWrapper(e, m_cat);
        } finally {
            if (sbrDAO != null) sbrDAO.cleanUp();
            m_cat.log(RequestUtil.m_defaultTracePriority, "Validation Rule: " + m_className + " done");
        }
    }

}
