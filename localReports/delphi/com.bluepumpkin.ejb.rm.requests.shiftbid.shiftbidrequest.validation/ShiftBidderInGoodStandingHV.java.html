package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.validation;

/**
 * <li> verify if shift bidder's end date has not expired (employee terminated or left).
 * <li> verify if shift bdider is still added to the auction.
 *
 * @author rrajendran
 *
 */
public class ShiftBidderInGoodStandingHV implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

    private final static String m_className = ShiftBidderInGoodStandingHV.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

        try {
            ShiftBidRequest sbr = (ShiftBidRequest) validatable;

            //TODO: check if this one verification is sufficient.
            Employee emp = sbr.getOptMethods().getEmployee();
            if (emp.getEndTime() != null && emp.getEndTime().before(new Date()))  {
                ValidationResult result = ValidationUtil.setAndLogHardValidationResult(sbr,
                		RmEjbBundleKey.SBR_EMP_NOT_IN_GOOD_STANDING,
										RmEjbLogBundleKey.SBR_EMP_NOT_IN_GOOD_STANDING,
										new Serializable[] {emp.getEndTime(), emp.getID()},
										m_className);
                return result;
            }

            ShiftBidder bidder = sbr.getOptMethods().getShiftBidder();
            if (!bidder.getIsAddedToAuction()) {
            	ValidationResult result = null;
            	// Verify if user is already scheduled in this auction (based on his status)
            	// and then return appropriate message
            	if(bidder.getStatus() == ShiftBidder.STATUS_UNAVAILABLE) {
            		result = ValidationUtil.setAndLogHardValidationResult(sbr,
                    		RmEjbBundleKey.SBR_BIDDER_NOT_AVAIL_TO_AUC,
    										RmEjbLogBundleKey.SBR_BIDDER_NOT_AVAIL_TO_AUC,
    										new Serializable[]{bidder.getID(), NumberFactory.newInteger(bidder.getStatus())},
    										m_className);
            	}
            	else {
            		result = ValidationUtil.setAndLogHardValidationResult(sbr,
                    		RmEjbBundleKey.SBR_BIDDER_NOT_ADDED_TO_AUC,
    										RmEjbLogBundleKey.SBR_BIDDER_NOT_ADDED_TO_AUC,
    										new Serializable[]{bidder.getID(), NumberFactory.newInteger(bidder.getStatus())},
    										m_className);
            	}
                return result;
            }

            return null;
        } finally {
           m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
        }
    }
}
