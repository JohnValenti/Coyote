package com.bluepumpkin.web.rm.requests.timeoff;

/**
 * Title:		     TimeOffRequestsRH
 * Description:  Time Off Request Request Handler

 * @version      1.0
 */
public class TimeOffRequestsRH extends RequestsRH {
	private static Category m_log = Log.initCategory(TimeOffRequestsRH.class.getName());

	/**
	 * Process Request for the User page
	 *
	 * @param context The Request Context
	 */  
	@Override
	public void processCustomRequest(RequestContext context) throws WebException {
		// Process requested action
		if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		} else {
			super.processCustomRequest(context);
		}
	}

	/**
	 * Process Save
	 */
	public void processSave(RequestContext context) {
		// Extract Form Parameter
		TimeOffRequestFormPopupPM model = new TimeOffRequestFormPopupPM(context);
		model.extractParameters(context.getRequest());

		if (model.validate()) {
			try {				
				// Security check
				TORequest request = model.getTimeOffRequest();
				ID requestID = request.getID();
				if (!RequestUtil.isAuthToUpdate(context, request)) {
					return;
				}

				if (requestID==null) {
					requestID = TimeOffRequestsMH.createTORequest(request, model.getComment());
				} else {
					TimeOffRequestsMH.updateTORequest(request, model.getComment());
					//QC 176957:When approving a time off the page selector jumps back to page 1 in request management- ESR#4663410-Cloned QC 177450
					//Fix for Edit action 
					model.setOpenerValue(AgentsRequestsPM.IS_ON_SAME_PAGE_FOR_EDIT_FN, "true");
				}
				
				model.setSavedRequestID(requestID);
			} catch(MultiUserException ex) {
				model.addPageMessage(Message.WARNING_TYPE,
						RmWebBundleKey.UNABLE_TO_UPDATE_REQUEST_DUE_TO_MULTIUSER_EXCEPTION,
						RmWebBundleKey.BUNDLE_NAME);
			} catch(RmHardValidationException ex) {
				m_log.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
				Message msg = new Message(Message.ERROR_TYPE,	RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
				msg.setAssociatedMessage( getLocMsgFromException(context, ex) );
				model.addPageMessage(msg);
			} catch(Exception ex) {
				m_log.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
				model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
			}
		}

		context.setPageModel(model);
	}

	/**
	 * Create AgentsRequestsPM or subclass
	 */
	@Override
	protected AgentsRequestsPM createAgentsRequestsPM(RequestContext context) {
		return new AgentsTimeOffRequestsPM(context);
	}
}
