	public static final String HOURS_LABEL      = "HOURS_LABEL";

	/**
	 * Constructs a schedule view page model shell.
	 */
	public MyTimeOffReportPM(RequestContext context)	{
		super(context);
		m_rmBundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);
		m_uifBundle = m_localizer.getBundle(UIFWebBundleKey.BUNDLE_NAME);
		
		m_containerList = new ContainerList(this, 4);
		m_toTakenPC = createCCPC("toTaken");
		m_toSchedPC = createCCPC("toSched");
		initDateRangePickerPC();
	}

	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, MyTimeOffReportPM.class);
	}

	/**
	 * Populate Model
	 */
	protected void loadData() throws Exception {
		if (m_empID!=null) {
			ID orgID = OrganizationModelHandler.getEmployeeOrgID(m_context, m_empID);
			if (orgID == null){
            	this.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.EMP_NO_ORG,
    					RmWebBundleKey.BUNDLE_NAME);
            	return;
            }
			//get org time zone for display
			m_orgTimeZone = RmSetupUtil.getOrgTimeZone(m_context, orgID);
			
			m_dateRangePickerPC.setTimeZone(m_orgTimeZone);
			if (m_startDate == null){
				setDates();
			}
			
			setTakenSchedColls(TimeOffRequestsMH.getTOActivities(m_empID, m_startDate, m_endDate));
		}
	}
	
	private void setDates(){ 
		if  (m_startDate == null) {
	//		init to current year
			Calendar cal = Calendar.getInstance(m_orgTimeZone);
			cal.setTime(new Date());
			cal.set(Calendar.MONTH, 0);
			cal.set(Calendar.DATE, 1);
			cal.set(Calendar.HOUR, 0);
			cal.set(Calendar.MINUTE, 0);
			cal.set(Calendar.SECOND, 0);
			m_startDate = cal.getTime();
			m_dateRangePickerPC.getStartDatePickerPC().setDate(m_startDate);
			
			cal.set(Calendar.MONTH, 11);
			cal.set(Calendar.DATE, 31);
			cal.set(Calendar.HOUR, 11);
			cal.set(Calendar.MINUTE, 59);
			cal.set(Calendar.SECOND, 59);
			m_endDate = cal.getTime();
			m_dateRangePickerPC.getEndDatePickerPC().setDate(m_endDate);
		}
		
		setDatesToProfile();
	}
	
	private void getDatesFromProfile(){
//		get from profile
		String sDateDtr = m_context.getUserProperty(RmKeys.RANGE_START_DATE);
		String eDateDtr = m_context.getUserProperty(RmKeys.RANGE_END_DATE);
		
		if (sDateDtr != null && eDateDtr != null){
			m_startDate = new Date(Long.valueOf(sDateDtr).longValue());	
			m_endDate = new Date(Long.valueOf(eDateDtr).longValue());
		} 
	}
	
	private void setDatesToProfile(){
		HashMap propMap = new HashMap(2);
		if (m_startDate != null)
			propMap.put(RmKeys.RANGE_START_DATE, String.valueOf(m_startDate.getTime()));
		if (m_endDate != null)
			propMap.put(RmKeys.RANGE_END_DATE, String.valueOf(m_endDate.getTime()));

		//set dates to the profile
		m_context.setUserProperties(propMap);
	}
	
	private void setTakenSchedColls(Collection toActivityColl) 
	throws Exception{
		m_toTakenList = new ArrayList();
		m_toSchedList = new ArrayList();
	
		//iterate over the collection and assign to taken/sched list
		for (Iterator it = toActivityColl.iterator(); it.hasNext();){
			TimeOffEventSummary toEventSummary = (TimeOffEventSummary)it.next();
			
			if (toEventSummary.getStartTime().before(new Date())){
				//add to taken
				m_toTakenList.add(toEventSummary);
			} else {
				//add to sched
				m_toSchedList.add(toEventSummary);
			}
		}
	}
	
	/**
	 * Create and add Date Range Picker as Child Component
	 */
	private void initDateRangePickerPC() {
		m_dateRangePickerPC = new DateRangePickerPC(m_context, DATE_CHOICE_FN);
		m_dateRangePickerPC.setIsTimeEnabled(false);
	
		m_dateRangePickerPC.setOnChangeJS(JSUtil.REFRESH_PAGE_NO_CONFIRM);
		m_dateRangePickerPC.setIsTodayButtonEnabled(true);
		m_dateRangePickerPC.setIsTomorrowButtonEnabled(true);
		m_dateRangePickerPC.setIsThisWeekButtonEnabled(true);
		m_dateRangePickerPC.setIsNextWeekButtonEnabled(true);
		m_dateRangePickerPC.setIsThisMonthButtonEnabled(true);
		m_dateRangePickerPC.setIsThisYearButtonEnabled(true);

		// initialize date pickers properties
		DatePickerPC startDatePC = m_dateRangePickerPC.getStartDatePickerPC();
		DatePickerPC endDatePC = m_dateRangePickerPC.getEndDatePickerPC();
		if (m_startDate != null)
			startDatePC.setDate(m_startDate);
		startDatePC.setIsDateValueOptional(false);
		
		if (endDatePC != null)
			endDatePC.setDate(m_endDate);
		endDatePC.setIsDateValueOptional(false);
		addChildComponent(m_dateRangePickerPC);
	}

	/**
	 * Create Collapsible Container PC
	 */
	private CollapsibleContainerPC createCCPC(String name) {
		CollapsibleContainerPC containerPC = new CollapsibleContainerPC(m_context);
		containerPC.setName(name);
		containerPC.setToggleable(true);
		containerPC.setStyleClass(CSSUtil.CLASS_BORDER_DEFAULT);
		containerPC.setTemplate("{0}");
		return containerPC;
	}

	/**
	 * Initiailize Container

	protected void initContainer(CollapsibleContainerPC containerPC, String label_rbKey, String content) {
		String title = i18n(m_bundle, label_rbKey);
		containerPC.setTitle(title);

		content = OPEN_WPR + content + CLOSE_WPR;
		containerPC.addComponent(content);
		m_containerList.add(containerPC);
	}

	/**
	 * Create and Add Inner List PC to Container
	 */
	private MultiColListPC addInnerListPC(CollapsibleContainerPC container) {
		MultiColListPC listPC = new MultiColListPC(m_context);
		listPC.setHeightInPixels(100);
		listPC.setEnabled(false);
		container.addComponent(listPC);
		return listPC;
	}

	/**
	 * Initialize Auto Processing Container
	 */
	protected void initTimeOffTaken() {
		String title = i18n(m_bundle, RmWebBundleKey.AGENTS_TO_REPORT_TAKEN);
		m_toTakenPC.setTitle(title);

		MultiColListPC listPC = addInnerListPC(m_toTakenPC);
		initListHeader(listPC.getHeader());
		List listModel = new ArrayList();
		if (m_toTakenList.isEmpty()) {
			// Create/Add no data msg Row
			MultiColumnNodeData row_data = createNoDataRow(RmWebBundleKey.AGENTS_TO_REPORT_NONE_TAKEN);
			listModel.add(row_data);
		} else {
			for(Iterator it=m_toTakenList.iterator(); it.hasNext(); ) {
				TimeOffEventSummary toEventSummary = (TimeOffEventSummary)it.next();
				listModel.add(getRowData(toEventSummary));
			}
		}
		listPC.setListModel(listModel);
		m_containerList.add(m_toTakenPC);
	}
	
	/**
	 * Initialize Filing Rule Container
	 */
	protected void initTimeOffScheduled() {
		String title = i18n(m_bundle, RmWebBundleKey.AGENTS_TO_REPORT_SCHED);
		m_toSchedPC.setTitle(title);

		MultiColListPC listPC = addInnerListPC(m_toSchedPC);
		initListHeader(listPC.getHeader());

		List listModel = new ArrayList();
		if (m_toSchedList.isEmpty()) {
			// Create/Add no data msg Row
			MultiColumnNodeData row_data = createNoDataRow(RmWebBundleKey.AGENTS_TO_REPORT_NONE_SCHED);
			listModel.add(row_data);
		} else {
			for(Iterator it=m_toSchedList.iterator(); it.hasNext(); ) {
				TimeOffEventSummary toEventSummary = (TimeOffEventSummary)it.next();
				listModel.add(getRowData(toEventSummary));
			}
		}
		listPC.setListModel(listModel);

		m_containerList.add(m_toSchedPC);
	}
	
	private DefaultMultiColumnNodeData getRowData(TimeOffEventSummary toEventSummary){
		DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData(toEventSummary.getActivityName());
		row_data.add(toEventSummary.getActivityName(), CSSUtil.STYLE_NORMAL_LEFT);
		row_data.add(RequestUtil.getDateDisplay(m_context, toEventSummary.getStartTime()));
		row_data.add(RequestUtil.getDateDisplay(m_context, toEventSummary.getEndTime()));
		
		long minuteLen = (toEventSummary.getEndTime().getTime()-toEventSummary.getStartTime().getTime())/(1000*60);
		row_data.add(TimeOffRequestUtil.getLengthStrFromMinutes(minuteLen, m_localizer, m_rmBundle));
		row_data.add(m_localizer.formatNumber(toEventSummary.getHoursAccounted(), 2));
		return row_data;
	}
	
	/**
	 * Create No Data Message Row
	 */
	private DefaultMultiColumnNodeData createNoDataRow(String key) {
		DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData("-1");
		int colspan = 5;
		String noDataMsg = m_localizer.i18n(m_rmBundle, key);
		row_data.add(noDataMsg, CSSUtil.STYLE_NORMAL_LEFT, colspan);
		row_data.setSelectable(false);
		return row_data;
	}

	/**
	 * Initialize List Header
	 */
	private void initListHeader(TableHeaderPC header) {
		// Setup Headers
		header.setStyleClass(CSSUtil.CLASS_INNER_HEADER_BORDER);

		header.addColumnHeaderData(ACTIVITY_LABEL,
				m_localizer.i18n(m_rmBundle,RmWebBundleKey.AGENTS_TO_REPORT_ACTIVITY));
		header.addColumnHeaderData(START_LABEL,
				m_localizer.i18n(m_rmBundle,
						RmWebBundleKey.AGENTS_TO_REPORT_START));
		header.addColumnHeaderData(END_LABEL,
				m_localizer.i18n(m_rmBundle,
						RmWebBundleKey.AGENTS_TO_REPORT_END));
		header.addColumnHeaderData(DAYS_LABEL,
				m_localizer.i18n(m_uifBundle, UIFWebBundleKey.LENGTH));
		header.addColumnHeaderData(HOURS_LABEL,
				m_localizer.i18n(m_rmBundle,
						RmWebBundleKey.AGENTS_TO_REPORT_HOURS));
	}
	
	/**
	 * Extract parameter
	 */
	public boolean extractParameters(HttpServletRequest request) {
		boolean success = super.extractParameters(request);

		m_startDate = m_dateRangePickerPC.getStartDatePickerPC().getDate();
		m_endDate = m_dateRangePickerPC.getEndDatePickerPC().getDate();
	
		if (m_startDate == null){
			getDatesFromProfile();
			if (m_startDate != null){
				m_dateRangePickerPC.getStartDatePickerPC().setDate(m_startDate);
				m_dateRangePickerPC.getEndDatePickerPC().setDate(m_endDate);
			}
		}
		setDatesToProfile();
		this.setPageDirty(false);

		return success;
	}
