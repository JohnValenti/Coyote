package com.bluepumpkin.web.rm.requests.timeoff;

import java.text.NumberFormat;
import java.util.ResourceBundle;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOCalendarDayData;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendarIncrement;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendarInterval;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;
import com.bluepumpkin.web.rm.keys.RmKeys;
import com.bluepumpkin.web.rm.keys.RmPageAction;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.util.UrlBuilder;
import com.witness.web.uif.util.html.HtmlLinkUtil;

public class TimeOffIntervalCalendarCellDetailBuilder {

	// CSS classes
	static final String CSS_NO_HOURS = "no-hours";
	static final String CSS_HOURS_AVAILABLE = "hour-available";
	static final String CSS_UNPAID_HOLIDAY = "unpaid-holiday";
	static final String CSS_NON_OPERATION = "non-operation";
	static final String CSS_PAID_HOLIDAY = "paid-holiday";
	static final String CSS_BLACKOUT = "blackout";
	static final String CSS_PENDING = "pending";
	static final String CSS_SCHEDULED = "scheduled";

	// True if we are in agent mode. False if not
	private boolean isAgentView;
	// True if the agent is viewing his TO Pool, False if the agent is viewing only his personal data
	private boolean isAgentTOPoolView;
	private ID selectedPoolID;
	private ID selectedOrgID;
	// availableOrAllocatedHoursText is available hours if we in agent mode and allocated hours if we are not in agent mode
	private String availableOrAllocatedHoursText;
	private String pendingHoursText;
	private String approvedHoursText;
	private String pendingHoursLink;
	private String approvedHoursLink;
	private final NumberFormat numberFormat;
	private final ResourceBundle rmWebBundle;
	protected Localizer localizer;
	private final String name;
	private boolean isShowPendingApproveHourLink = false;

	TimeOffIntervalCalendarCellDetailBuilder(NumberFormat nf, Localizer lz, String componentName, boolean isShowLink) {
		numberFormat = nf;
		rmWebBundle = lz.getBundle(RmWebBundleKey.BUNDLE_NAME);
		localizer = lz;
		name = componentName;
		isShowPendingApproveHourLink = isShowLink;
	}

	public void setMode(boolean isAgentView, boolean isAgentTOPoolView) {
		this.isAgentView = isAgentView;
		this.isAgentTOPoolView = isAgentTOPoolView;
	}

	public void setSelectedOrganization(ID orgID) {
		this.selectedOrgID = orgID;
	}

	public void setSelectedPool(ID topID) {
		this.selectedPoolID = topID;
	}

	public void initialize(TOIntervalCalendarInterval interval) {
		boolean forEmployee = showEmployeeData();
		Float pendingHours = interval.getPendingHours(forEmployee);
		Float approvedHours = interval.getApprovedHours(forEmployee);
		Float hours = isAgentView ? interval.getAvailableHours(forEmployee) : interval.getAllocatedHours();

		availableOrAllocatedHoursText = formatAvailableOrAllocatedHours(interval, hours);
		pendingHoursText = formatHours(pendingHours);
		approvedHoursText = formatHours(approvedHours);

		pendingHoursLink = pendingHoursText;
		approvedHoursLink = approvedHoursText;
		boolean isShowLinkForAgent =isAgentView?isShowPendingApproveHourLink:true;
		if (pendingHours != null && pendingHours > 0 && isShowLinkForAgent) {
			pendingHoursLink = getHourPendingLink(pendingHoursText, interval);
		}

		if (approvedHours != null && approvedHours > 0 &&isShowLinkForAgent) {
			approvedHoursLink = getHourScheduledLink(approvedHoursText, interval);
		}
	}

	// returns true if we are on the agent view and the agent is viewing just his personal data
	private boolean showEmployeeData() {
		return isAgentView && !this.isAgentTOPoolView;
	}

	private String formatAvailableOrAllocatedHours(TOIntervalCalendarInterval interval, Float hours) {
		if (hours != null && interval.getTOPoolID() != null) {
			return numberFormat.format(hours);
		}
		return localizer.i18n(rmWebBundle, RmWebBundleKey.ORG_RM_SETTINGS_NOT_APPLICABLE);
	}

	private String formatHours(Float hours) {
		if (hours != null) {
			return numberFormat.format(hours);
		}
		return localizer.i18n(rmWebBundle, RmWebBundleKey.ORG_RM_SETTINGS_NOT_APPLICABLE);
	}

	private String getHourPendingLink(String pendingText, TOIntervalCalendarInterval interval) {
		UrlBuilder builder = new UrlBuilder(isAgentView ? RmKeys.RH_AG_VIEW_TIMEOFF_LIST : RmKeys.RH_AG_REQUEST_TIMEOFF_LIST);
		builder.add("pageAction", RmPageAction.AGENTS_TIMEOFF_PENDING_ACTION);
		builder.add("TOPOOLID", interval.getTOPoolID());
		builder.add("orgID", selectedOrgID);
		builder.add("date", interval.getStartTime().getTime());
		builder.add("isParentView", isInOrgView());
		builder.add("isAgentCal", isAgentView);
		builder.add("isPendingTOView", "true");
		return HtmlLinkUtil.createURLLink(pendingText, builder.get(), HtmlLinkUtil.PAGE_LINK, true);
	}

	private boolean isInOrgView() {
		return !isAgentView && selectedPoolID == null;
	}

	private String getHourScheduledLink(String scheduledText, TOIntervalCalendarInterval interval) {

		// "orgID,poolID,date,hourType,parentView,agentCal"
		String popupJS = String.format("%s.handleAgentsPopup(%s,%s,%d,%s,%s,%s);",
				// Object name
				name,
				// orgID
				toJsStringArgument(selectedOrgID),
				// poolID
				toJsStringArgument(interval.getTOPoolID()),
				// date
				interval.getStartTime().getTime(),
				// hour type - Scheduled
				toJsStringArgument(RmKeys.SCHEDULED_TO_TYPE),
				// is Organization view (as opposed to time off pool view)
				toJsStringArgument(isInOrgView()),
				// is Agent view
				toJsStringArgument(isAgentView));

		return HtmlLinkUtil.createActiveLink(scheduledText, null, null, popupJS);
	}

	private static String toJsStringArgument(Object value) {
		return value == null ? "null" : String.format("'%s'", value);
	}

	public String getCellText() {
		return String.format("%s, %s, %s", availableOrAllocatedHoursText, pendingHoursLink, approvedHoursLink);
	}

	public String getCellTooltip(String dateRangeText) {
		return String.format("%s%n%s, %s, %s", dateRangeText, availableOrAllocatedHoursText, pendingHoursText, approvedHoursText);
	}

	public void appendOrganziationColorStyles(StringBuilder sb, OrganizationSetting orgSetting, String prefix) {

		String[] keys = TOCalendarDayData.DAY_TYPE_TO_SETTINGS_KEY;

		appendOrganziationColorStyle(sb, orgSetting, prefix, CSS_NO_HOURS, keys[TOCalendarDayData.NO_TO_HOURS_AVAILABLE_DAY_TYPE]);
		appendOrganziationColorStyle(sb, orgSetting, prefix, CSS_HOURS_AVAILABLE, keys[TOCalendarDayData.TO_HOURS_AVAILABLE_DAY_TYPE]);
		appendOrganziationColorStyle(sb, orgSetting, prefix, CSS_UNPAID_HOLIDAY, keys[TOCalendarDayData.UNPAID_HOLIDAY_DAY_TYPE]);
		appendOrganziationColorStyle(sb, orgSetting, prefix, CSS_NON_OPERATION, keys[TOCalendarDayData.NON_OPERATION_DAY_TYPE]);
		appendOrganziationColorStyle(sb, orgSetting, prefix, CSS_PAID_HOLIDAY, keys[TOCalendarDayData.PAID_HOLIDAY_DAY_TYPE]);
		appendOrganziationColorStyle(sb, orgSetting, prefix, CSS_BLACKOUT, keys[TOCalendarDayData.BLACKOUT_DAY_TYPE]);
		appendOrganziationColorStyle(sb, orgSetting, prefix, CSS_PENDING, keys[TOCalendarDayData.PENDING_DAY_TYPE]);
		appendOrganziationColorStyle(sb, orgSetting, prefix, CSS_SCHEDULED, keys[TOCalendarDayData.SCHEDULED_DAY_TYPE]);
	}

	private void appendOrganziationColorStyle(StringBuilder sb, OrganizationSetting orgSetting, String prefix, String cssClass, String key) {
		String color = orgSetting.getValue(key);
		if (!StringUtil.isEmptyOrWhiteSpace(color)) {
			sb.append(prefix).append(".").append(cssClass).append(" { background-color:");
			if (!color.startsWith("#")) {
				sb.append("#");
			}
			sb.append(color).append(";}");
		}
	}

	public String getCssClass(TOIntervalCalendarIncrement increment) { // NOSONAR
		switch (increment.getType()) {
		case TOCalendarDayData.TO_HOURS_AVAILABLE_DAY_TYPE:
			return CSS_HOURS_AVAILABLE;
		case TOCalendarDayData.UNPAID_HOLIDAY_DAY_TYPE:
			return CSS_UNPAID_HOLIDAY;
		case TOCalendarDayData.NON_OPERATION_DAY_TYPE:
			return CSS_NON_OPERATION;
		case TOCalendarDayData.PAID_HOLIDAY_DAY_TYPE:
			return CSS_PAID_HOLIDAY;
		case TOCalendarDayData.BLACKOUT_DAY_TYPE:
			return CSS_BLACKOUT;
		case TOCalendarDayData.PENDING_DAY_TYPE:
			return CSS_PENDING;
		case TOCalendarDayData.SCHEDULED_DAY_TYPE:
			return CSS_SCHEDULED;
		case TOCalendarDayData.NO_TO_HOURS_AVAILABLE_DAY_TYPE:
		default:
			return CSS_NO_HOURS;
		}
	}

	public String getAvailableOrAllocatedHoursText() {
		return availableOrAllocatedHoursText;
	}

	public String getPendingHoursText() {
		return pendingHoursText;
	}

	public String getScheduledHoursText() {
		return approvedHoursText;
	}

	public String getAvailableOrAllocatedLabelText() {
		return localizer.i18n(rmWebBundle, isAgentView ? RmWebBundleKey.TIMEOFF_HOURS_AVAILABLE : RmWebBundleKey.TIMEOFF_HOURS_ALLOCATED);
	}
}
