package com.bluepumpkin.ejb.rm.setup.validation.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;


/**

public class OrgValidation extends ValueObjectBase {
    
    private static FieldInfo m_fieldInfo = new OrgValidationFieldInfo();
    
    /** Default constructor
     */
    public OrgValidation() {
        super();
    }
    
    /** Gets the FieldInfo for the object's class.
     *  @return FieldInfo for the object's class.
     */
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }
    
    
    /** Return's Organization ID attribute for the class.
     *  @param ID - ID object for the Organization ID.
     */
    public ID getOrganizationID() {
        return getFieldValueID(OrgValidationFieldInfo.ORGANIZATIONVALIDATION_I_ORGANIZATIONID);
    }
    
    /** Return's attribute value for the Validation Rule ID for the class.
     *  @param ID - Validation Rule ID.
     */
    public ID getValidationRuleID() {
        return getFieldValueID(OrgValidationFieldInfo.ORGANIZATIONVALIDATION_I_VALIDATIONRULEID);
    }
    
    
    /** Return's isEnabled attribute value as boolean.
     *  @param boolean- isEnabled attribute value as boolean.
     */
    public boolean getEnabled() {
        return getFieldValueBoolean(OrgValidationFieldInfo.ORGANIZATIONVALIDATION_B_ISENABLED);
    }
    
    
    // setters
    
    
    /** Set the  pOrganizationId for the object.
     *  @param pOrganizationId - ID for the Organization.
     */
    public void setOrganizationID(ID pOrganizationId) {
        setFieldValue(OrgValidationFieldInfo.ORGANIZATIONVALIDATION_I_ORGANIZATIONID, pOrganizationId);
    }
    
    /** Set the  validationruleid attribute for the object.
     *  @param pValidationRuleId- ID for the validation rule.
     */
    public void setValidationRuleID(ID pValidationRuleId) {
        setFieldValue(OrgValidationFieldInfo.ORGANIZATIONVALIDATION_I_VALIDATIONRULEID, pValidationRuleId);
    }
    
    
    /** Set the  isEnabled attribute for the object.
     *  @param pisEnabled- isEnabled attribue value as boolean.
     */
    public void setEnabled(boolean pisEnabled) {
        setFieldValue(OrgValidationFieldInfo.ORGANIZATIONVALIDATION_B_ISENABLED, pisEnabled);
    }

	public Integer getActivityID() {
		Object val = getFieldValue(OrgValidationFieldInfo.ORGANIZATIONVALIDATION_I_ACTIVITYID);
		if (val != null) {
			return (Integer) val;
		}
		return null;
	}

	public void setActivityID(Integer activityID) {
		setFieldValue(OrgValidationFieldInfo.ORGANIZATIONVALIDATION_I_ACTIVITYID, activityID);
	}
}
