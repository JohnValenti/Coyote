public class OrgValidationFieldInfo extends FieldInfo {
    // field id constants
    public static final int ORGANIZATIONVALIDATION_I_ID = 0;
    public static final int ORGANIZATIONVALIDATION_I_ORGANIZATIONID = 1;
    public static final int ORGANIZATIONVALIDATION_I_VALIDATIONRULEID = 2;
    public static final int ORGANIZATIONVALIDATION_B_ISENABLED = 3;
    public static final int ORGANIZATIONVALIDATION_I_ACTIVITYID= 4;
    
    protected static final int ORGANIZATIONVALIDATION_NUMOFFIELDS = 5;
    
    private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[ORGANIZATIONVALIDATION_NUMOFFIELDS];
    
    private static String m_strTableName = "ORGANIZATIONVALIDATION";

    static
    {
        RequestUtil.initFieldInfoEntries(m_arrFieldInfo, OrgValidationFieldInfo.class, 
            "ORGANIZATIONVALIDATION_", ORGANIZATIONVALIDATION_NUMOFFIELDS);
    }

    public OrgValidationFieldInfo() {
        super();
    }

    // implement abstract methods in parent class
    public String getTableName() {
        return m_strTableName;
    }

    public int getIDFieldIndex() {
        return ORGANIZATIONVALIDATION_I_ID;
    }

    public FieldInfoEntry[] getFieldInfo() {
        return m_arrFieldInfo;
    }
}
