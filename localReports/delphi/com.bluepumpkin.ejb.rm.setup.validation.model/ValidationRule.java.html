	/* Returns the boolean status value for the rule indicating the rule
	 * is enabled or disabled or in the case of timeoff partially enabled (due to multiple activities selected)
	 * @return int
	 * 	int VALIDATION_PARTIALLY_ENABLED = -1;
		int VALIDATION_NOT_ENABLED = 0;
		int VALIDATION_ENABLED = 1;

	public int getEnabledState() {
		return enabledState;
	}
	
	public boolean isEnabled(){
		return enabledState == VALIDATION_ENABLED;
	}

	public boolean isPartiallyEnabled() {
		return enabledState == VALIDATION_PARTIALLY_ENABLED;
	}

	/* Returns the attribute value map for the rule
	 * @return HashMap
	 **/
	public HashMap getParamMap() {
		return m_paramMap;
	}
	
	public boolean isParamValueMultipleValue(String paramName){
		
		Integer paramValue = (Integer)getParamMap().get(paramName);
		if(paramValue != null && paramValue.intValue()==Integer.MAX_VALUE){
			return true;
		}
		return false;
	
	}
	public boolean isAnyParamValuesMultipleValue(){
		for (Object val : getParamMap().values()) {
			Integer paramValue = (Integer) val;
			if (paramValue != null && paramValue.intValue() == Integer.MAX_VALUE) {
				return true;
			}
		}
		return false;
	}

	public static boolean isRuleTimeOffType(String key) {
		return key.equals(ValidationRule.VALIDATION_AVAILABLE_TO_HRS)
				|| key.equals(ValidationRule.VALIDATION_ENOUGH_TO_HRS)
				|| key.equals(ValidationRule.VALIDATION_COMPLY_FILINGRULE)
				|| key.equals(ValidationRule.VALIDATION_AVOID_BLACKOUT)
				|| key.equals(ValidationRule.VALIDATION_AVOID_CAL_EVENTS)
				|| key.equals(ValidationRule.VALIDATION_TO_NOT_SWAPPED)
				|| key.equals(ValidationRule.VALIDATION_REQ_MIN_DURATION)
				|| key.equals(ValidationRule.VALIDATION_REQ_MAX_DURATION)
				|| key.equals(ValidationRule.VALIDATION_REQUEST_IN_PAST)
				|| key.equals(ValidationRule.VALIDATION_VTO_REDUCE_NET_STAFFING_SURPLUS)
				|| key.equals(ValidationRule.VALIDATION_VTO_COMPLY_MAX_VTO_HRS)
				|| key.equals(ValidationRule.VALIDATION_TOB_COMPLY_MAX_EMPLOYEE_APPROVAL)
				|| key.equals(ValidationRule.VALIDATION_TOB_SUBMITTED_DATE)
				|| key.equals(ValidationRule.VALIDATION_TOB_DONOT_APPROVE_OVERLAP_REQUEST);

	}
