package com.bluepumpkin.ejb.rm.setup.calendartimeoff.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

public class EmployeeReferenceScheduleAssignment implements java.io.Serializable {
	
	
	private static final long serialVersionUID = 1L;
	
	ID id;
	//Using DEID instead of the integer SID value since the clustered index on the SP table is on ID (which is called DEID in code)
	ID schedulingPeriodDEID;
	ID workResourceID;
	Date startTime;
	Date endTime;
	
	
	public ID getID() {
		return id;
	}
	public void setID(ID id) {
		this.id = id;
	}
	public ID getSchedulingPeriodDEID() {
		return schedulingPeriodDEID;
	}
	public void setSchedulingPeriodDEID(ID deid) {
		this.schedulingPeriodDEID = deid;
	}
	
	public ID getWorkResourceID() {
		return workResourceID;
	}
	public void setWorkResourceID(ID workResourceID) {
		this.workResourceID = workResourceID;
	}
	public Date getStartTime() {
		return startTime;
	}
	public void setStartTime(Date startTime) {
		this.startTime = startTime;
	}
	public Date getEndTime() {
		return endTime;
	}
	public void setEndTime(Date endTime) {
		this.endTime = endTime;
	}


}
