package com.bluepumpkin.ejb.rm.setup.calendartimeoff.model;

import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;


/**
 * Represents a BlackoutDay value
 */
public class BlackoutDay extends ValueObjectBase {

	//private static FieldInfo m_fieldInfo = new BlackoutDayFieldInfo();
	private static FieldInfo m_fieldInfo = new BlackoutDayFieldInfo();

	private static TimeZone s_gmt = TimeZone.getTimeZone("GMT");

	public BlackoutDay() {
		super();
	}

	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public ID getTOPoolID() {
		return getFieldValueID(BlackoutDayFieldInfo.BLACKOUTDAY_I_TOPOOLID);
	}


	/**
	 * The blackout date stored in DB is always 12:00AM in GMT TZ.
	 * For instance if 1/1/2004 is a blackout day for an org, then
	 * the stored timestamp is 1/1/2004 12:00AM GMT.  Note that the
	 * org's timezone (for which the blackout date applies) is not used.
	 * <p/>
	 * <p/>
	 * In other words, the blackout date is a {@link LocalDate LocalDate}
	 * set to 12:00AM.
	 * <p/>
	 * <p> Only the year, month and day fields of the date are used.
	 * The fields for hour, minutes etc are not.
	 * For instance if the date is set to 1/1/2004 12:00AM GMT, this means the blackout day
	 * applies to 1/1/2004 in the org's (for which the allocation is specified) TZ.
	 * <p/>
	 * <p> To extract the year, month and day fields for this date, always use a calendar
	 * initialized to GMT TZ.
	 *
	 * @return
	 */
	public Date getBlackoutDate() {
		return getFieldValueDate(BlackoutDayFieldInfo.BLACKOUTDAY_D_DAYDATE);
	}

	/**
	 * @param pDate See {@link #getBlackoutDate() getBlackoutDate()} for notes
	 *              about the value for this parameter.
	 */
	public LocalDate getBlackoutLocalDate() {
		Date d = getFieldValueDate(BlackoutDayFieldInfo.BLACKOUTDAY_D_DAYDATE);
		if (d != null)
			return new LocalDate(d, s_gmt);
		else
			return null;
	}

	public void setTOPoolID(ID id) {
		setFieldValue(BlackoutDayFieldInfo.BLACKOUTDAY_I_TOPOOLID, id);
	}

	/**
	 * @param pDate See {@link #getBlackoutDate() getBlackoutDate()} for notes
	 *              about the value for this parameter.
	 */
	public void setBlackoutDate(Date pDate) {
		setFieldValue(BlackoutDayFieldInfo.BLACKOUTDAY_D_DAYDATE, pDate);
	}

	/**
	 * @param pDate See {@link #getBlackoutDate() getBlackoutDate()} for notes
	 *              about the value for this parameter.
	 */
	public void setBlackoutDate(LocalDate pDate) {
		// To maintain compatibility with the original Date interface, convert
		// the input local date to a date using GMT
		if (pDate == null) {
			setFieldValue(BlackoutDayFieldInfo.BLACKOUTDAY_D_DAYDATE, (Date) null);
		} else {
			setFieldValue(BlackoutDayFieldInfo.BLACKOUTDAY_D_DAYDATE, pDate.getTime(s_gmt));
		}
	}

}
