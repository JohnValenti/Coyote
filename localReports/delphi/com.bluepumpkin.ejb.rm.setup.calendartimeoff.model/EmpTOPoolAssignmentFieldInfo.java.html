public final class EmpTOPoolAssignmentFieldInfo extends FieldInfo {

    // field id constants
    public static final int EMPTOPOOL_ID = 0;
    public static final int EMPTOPOOL_TOPOOLID = 1;
    public static final int EMPTOPOOL_WORKRESOURCEID = 2;
    public static final int EMPTOPOOL_STARTTIME = 3;
    public static final int EMPTOPOOL_ENDTIME = 4;

    private static final int EMP_TOPOOL_NUMOFFIELDS = 5;

    private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[EMP_TOPOOL_NUMOFFIELDS];
    private static String m_strTableName = "WORKRESOURCETOPOOL";

    // static initialization
    static {
      m_arrFieldInfo[EMPTOPOOL_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
      m_arrFieldInfo[EMPTOPOOL_TOPOOLID] = new FieldInfoEntry("TOPOOLID", PrimitiveTypes.ID);
      m_arrFieldInfo[EMPTOPOOL_WORKRESOURCEID] = new FieldInfoEntry("WORKRESOURCEID", PrimitiveTypes.ID);
      m_arrFieldInfo[EMPTOPOOL_STARTTIME] = new FieldInfoEntry("STARTTIME", PrimitiveTypes.DATETIME);
      m_arrFieldInfo[EMPTOPOOL_ENDTIME] = new FieldInfoEntry("ENDTIME", PrimitiveTypes.DATETIME);
    }

    public EmpTOPoolAssignmentFieldInfo() {
        super();
        setNotifiable(true);
    }

    // implement abstract methods in parent class
    public String getTableName() { return m_strTableName; }
    public int getIDFieldIndex() { return EMPTOPOOL_ID; }
    public int getParentIDFieldIndex() { return EMPTOPOOL_WORKRESOURCEID; }
    public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }
}
