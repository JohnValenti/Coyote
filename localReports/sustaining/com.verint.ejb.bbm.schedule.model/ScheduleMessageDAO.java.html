package com.verint.ejb.bbm.schedule.model;

import java.util.Collection;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.verint.ejb.bbm.schedule.model.ScheduleMessage.SchedulingAction;

/**
 * DAO to access data from the SCHEDULEMESSAGE Table
 * 
 */
public class ScheduleMessageDAO extends DAOBase<ScheduleMessage> {

	private static FieldInfo m_fieldInfo = new ScheduleMessageFieldInfo();
	
	public ScheduleMessageDAO() {
		super();
	}

	public ScheduleMessageDAO(Jdmo dmo) {
		super(dmo);
	}

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}	

	@Override
	protected ScheduleMessage createValueObject() {
		return new ScheduleMessage();
	}
	
	/**
	 * Returns the messages in the order that they are updated.
	 */
	public Collection<ScheduleMessage> getMessagesByLastUpdated() throws BbmFinderException{
		StringBuilder afterWhere = new StringBuilder();
		afterWhere.append(" order by A.");
		afterWhere.append(m_fieldInfo.getFieldName(ScheduleMessageFieldInfo.LASTUPDATED));
		return getObjects(null, afterWhere.toString());
	}

	/**
	 * Returns the scheduling action for the request IDs (if available)
	 * 
	 * @throws JdmoException
	 */
	public Map<ID, SchedulingAction> getSchedulingActionByRequestID(Collection<ID> requestIDs) throws JdmoException {
		if (requestIDs == null || requestIDs.isEmpty()) {
			return new HashMap<ID, SchedulingAction>();
		}
		// Could be converted to stored proc latter
		// select A.ACTION, A.REQUESTID from SCHEDULEMESSAGE A,
		// SCHEDULEMESAAGEREQUESTMAPPING B
		// where A.ID = B.MESSAGEID
		// AND A.REQUESTID IN (@reuestIDs
		ScheduleMessageRequestMappingFieldInfo mappingInfo = new ScheduleMessageRequestMappingFieldInfo();
		StringBuilder sb = new StringBuilder();
		// build the query
		sb.append("SELECT A.");
		sb.append(m_fieldInfo.getFieldName(ScheduleMessageFieldInfo.ACTION));
		sb.append(", B.");
		sb.append(mappingInfo.getFieldName(ScheduleMessageRequestMappingFieldInfo.REQUESTID));
		sb.append(" FROM ");
		sb.append(m_fieldInfo.getTableName());
		sb.append(" A,  ");
		sb.append(mappingInfo.getTableName());
		sb.append(" B where ");
		sb.append("A.");
		sb.append(m_fieldInfo.getFieldName(ScheduleMessageFieldInfo.ID));
		sb.append(" = B.");
		sb.append(mappingInfo.getFieldName(ScheduleMessageRequestMappingFieldInfo.MESSAGEID));
		sb.append(" AND ");
		sb.append("B.");
		sb.append(mappingInfo.getFieldName(ScheduleMessageRequestMappingFieldInfo.REQUESTID));
		sb.append(" IN ");
		sb.append(m_dmo.createInClause(requestIDs));
		JdmoRowset results = m_dmo.createRowset(sb.toString());
		Map<ID, SchedulingAction> actionByRequest = new HashMap<ID, SchedulingAction>();
		while (results.next()) {
			actionByRequest.put(
					results.getID(mappingInfo.getFieldName(ScheduleMessageRequestMappingFieldInfo.REQUESTID)),
					ScheduleMessage.getSchedulingAction(results.getInt(1)));

		}
		return actionByRequest;
	}

}
