public class FteRequirementsAppletRequestHandler extends OptimizationAppletRequestHandler {

	@Override
	protected void processAppletRequest(RequestContext context, Map request, Map response) {
		try {
			if (isAppletAction(request, FteRequirementsKeys.PROC_GET_SINGLE_QUEUE_FTEREQUIREMENTS_MODEL)) {
				getFullSingleQueueModel(context, request, response);
			} else if (isAppletAction(request, FteRequirementsKeys.PROC_GET_SINGLE_QUEUE_FTEREQUIREMENTS_TRACECUBE)) {
				getFteRequirements(context, request, response);
			} else if (isAppletAction(request, FteRequirementsKeys.PROC_GET_COMBINED_QUEUE_FTEREQUIREMENTS_MODEL)) {
				getPartialCombinedQueueModel(context, request, response);
			} else if (isAppletAction(request, FteRequirementsKeys.PROC_GET_MULTI_QUEUE_FTEREQUIREMENTS_MODEL)) {
				getPartialMultiQueueModel(context, request, response);
			} else if (isAppletAction(request, FteRequirementsKeys.PROC_SAVE_FTEREQUIREMENTS_SPQUEUE)) {
				saveFteRequirementsForSpQueue(context, request, response);
			} else if (isAppletAction(request, FteRequirementsKeys.PROC_RESTORE_FTEREQUIREMENTS_SPQUEUE)) {
				restoreFteRequirementsForSpQueue(context, request, response);
			} else {
				super.processAppletRequest(context, request, response);
			}
		} catch (Exception e) {
				handleError(context, e,
				CoreWebLogBundleKey.APPLET_PROCESS_REQUEST_ERROR,
				CoreWebBundleKey.APPLET_PROCESS_REQUEST_ERROR,
				CoreWebBundleKey.BUNDLE_NAME);
		}
	}

	/**
	 * Returns (or rather, adds to the response) a RequireTraceCube of FTE requirements for
	 * the given queue at the given dates.
	 */
	private void getFteRequirements(RequestContext context, Map request, Map response) 
    {   
        try
        {
        	ID campaignId = (ID)request.get(OptimizationKeys.PARAM_CAMPAIGN_ID);
        	ID queueId = (ID)request.get(OptimizationKeys.PARAM_QUEUE_ID);
        	ID mediaId = (ID)request.get(OptimizationKeys.PARAM_MEDIA_ID);
        	Date start = (Date)request.get(OptimizationKeys.PARAM_START_DATE);
        	Date end = (Date)request.get(OptimizationKeys.PARAM_END_DATE);
        	Date adjustedEnd = new Date(end.getTime() - 60000);  // 1 minute before.
        	
			FteRequirementsManager manager =
				WfmManagerFactory.getFteRequirementsManager(context.isInWhatIfMode());
        	RequireTraceCube fteRequirements = manager.getFteRequirementsTraceCube(campaignId, queueId,
        			mediaId, start, adjustedEnd);
        	
        	response.put(FteRequirementsKeys.RETVAL_FTEREQUIREMENTS_TRACECUBE, fteRequirements);
        }
        catch(Exception e){
        	notifyError(context, response, e);
        }
    }
	
	/**
	 * Returns a SingleFteRequirementsQueueDTO, which contains a queue, spqueue, and RequireTraceCube for
	 * the given spqueue.
	 */
	private void getFullSingleQueueModel(RequestContext context, Map request, Map response) {   
        try {
        	Response<SingleFteRequirementsQueueDTO> fullResponse = getFullResponseForSingleQueue(context, request, response);
	        response.put(FteRequirementsKeys.RETVAL_FTEREQUIREMENTS_MODEL, fullResponse);
        } catch(Exception e){
        	notifyError(context, response, e);
        }
    }
	
	/**
	 * Creates and returns a SingleFteRequirementsQueueDTO which contains a queue, spqueue, and RequireTraceCube
	 * objects for the given spqueue.  This DTO contains all the necessary data for a single queue so we 
	 * also refer to it as a "full response".
	 */
	private Response<SingleFteRequirementsQueueDTO> getFullResponseForSingleQueue(RequestContext context, Map request, Map response) throws Exception {
		SingleQueuePartialResponse partialResponse = createSingleQueuePartialResponse(context, request);
    	Date adjustedEnd = new Date(partialResponse.getSp().getEndTime().getTime() - 60000);  // 1 minute before.
		FteRequirementsManager manager =
			WfmManagerFactory.getFteRequirementsManager(context.isInWhatIfMode());
    	RequireTraceCube fteRequirements = manager.getFteRequirementsTraceCube(partialResponse.getCampaign().getID(), 
    			partialResponse.getQueue().getID(), partialResponse.getQueue().getMediaID(),
    			partialResponse.getSp().getStartTime(), adjustedEnd);
    	SingleFteRequirementsQueueDTO queueDto = new 
    	SingleFteRequirementsQueueDTO(partialResponse.getQueue(), 
    				partialResponse.getSPQueue(), 
    				fteRequirements);

    	Response<SingleFteRequirementsQueueDTO> fullResponse = 
    		new Response<SingleFteRequirementsQueueDTO>(partialResponse.getMedia(), 
    				partialResponse.getCampaign(), 
    				partialResponse.getSp(), queueDto, partialResponse.getHoursOfOperation());
    	return fullResponse;	
	}
	
	/**
	 * Returns a CombinedFteRequirementsQueueDTO.  This DTO contains the aggregated FTE Requirements for
	 * the combined queue (and not the requirements for the constituent single queues, hence the name "partial".  These
	 * will be loaded when the user clicks on a single queue).
	 */
	private void getPartialCombinedQueueModel(RequestContext context, Map request, Map response) {   
        try {
        	Response<CombinedFteRequirementsQueueDTO> fullResponse = getPartialResponseForCombinedQueue(context, request, response);
	        response.put(FteRequirementsKeys.RETVAL_FTEREQUIREMENTS_MODEL, fullResponse);
        } catch(Exception e){
        	notifyError(context, response, e);
        }
    }
	
	/**
	 * Creates and returns a MultiFteRequirementsQueueDTO.  This DTO contains the aggregated FTE Requirements for
	 * the multi-selected queues (and not the requirements for the constituent single queues, hence the name "partial".  These
	 * will be loaded when the user clicks on a single queue).
	 */
	private void getPartialMultiQueueModel(RequestContext context, Map request, Map response) {   
        try {
        	Response<MultiFteRequirementsQueueDTO> fullResponse = getPartialResponseForMultiQueue(context, request, response);
	        response.put(FteRequirementsKeys.RETVAL_FTEREQUIREMENTS_MODEL, fullResponse);
        } catch(Exception e){
        	notifyError(context, response, e);
        }
    }
	
	/**
	 * Creates and returns a partial response for a Combined FTE Requirements queue.
	 */
	private Response<CombinedFteRequirementsQueueDTO> getPartialResponseForCombinedQueue(RequestContext context, Map request, Map response)
			throws Exception {
        HashMap<ID, ID> queueMediaMap = new HashMap<ID, ID>();
        List<SingleFteRequirementsQueueDTO> singleQueueDtos = new ArrayList<SingleFteRequirementsQueueDTO>();
        ArrayList<TraceCube> constituentFteRequirementsCubes = new ArrayList<TraceCube>();
        TraceCube aggregatedFteRequirements; 
        CombinedQueuePartialResponse partialResponse = createCombinedQueuePartialResponse(context, request);
        Date adjustedEnd = new Date(partialResponse.getSp().getEndTime().getTime() - 60000);  // 1 minute before.
        	
        //If it is an unskilled SP that means that we are pulling the trace cube from combined queues only.
        //Currently, this is either the phone combined, project combined, or "combined-combined" (the combination
        //of phone and project).
		FteRequirementsManager manager =
			WfmManagerFactory.getFteRequirementsManager(context.isInWhatIfMode());
        if (!partialResponse.getSp().getSkillBased()) {
        	Collection<ID> mediaIDs = new HashSet<ID>();
        	for (int i = 0; i <= partialResponse.getQueues().size() - 1; i++) {
        		if (partialResponse.getQueues().get(i).getMediaID() != null) {
        			mediaIDs.add(partialResponse.getQueues().get(i).getMediaID());
        			queueMediaMap.put(partialResponse.getQueues().get(i).getID(), partialResponse.getQueues().get(i).getMediaID());
        		}
        		//The only reason we need to add the single queue dtos to the response is so that the model can get the media type.
        		//It needs the single queue dtos in order to construct the single queue models which will contain the media type.
        		//The single queue models will not be used to store trace data as fte requirements are always stored at the combined
        		//queue level for unskilled sps.
        		singleQueueDtos.add(new SingleFteRequirementsQueueDTO(partialResponse.getQueues().get(i), partialResponse.getSPQueues().get(i)));
        	}
        	for (ID mediaID : mediaIDs) {
    			RequireTraceCube constituentFteRequirementsCube = manager.getFteRequirementsTraceCube(partialResponse.getCampaign().getID(),
						null, mediaID, partialResponse.getSp().getStartTime(), adjustedEnd);
    			constituentFteRequirementsCubes.add(constituentFteRequirementsCube);        		
        	}
        } else {	//skill-based, use aggregation of constituent queues to build the tracecube 
            for (int i = 0; i <= partialResponse.getQueues().size() - 1; i++) {
            	queueMediaMap.put(partialResponse.getQueues().get(i).getID(), partialResponse.getQueues().get(i).getMediaID());
    			singleQueueDtos.add(new SingleFteRequirementsQueueDTO(partialResponse.getQueues().get(i), partialResponse.getSPQueues().get(i)));
    			RequireTraceCube constituentFteRequirementsCube = manager.getFteRequirementsTraceCube(partialResponse.getCampaign().getID(),
						partialResponse.getQueues().get(i).getID(), partialResponse.getQueues().get(i).getMediaID(),
						partialResponse.getSp().getStartTime(), adjustedEnd);
    			constituentFteRequirementsCubes.add(constituentFteRequirementsCube);
    		}
        }
        
        aggregatedFteRequirements =  TraceOperator.combineQueue(
        		constituentFteRequirementsCubes.toArray(new TraceCube[constituentFteRequirementsCubes.size()]),
        		true, queueMediaMap);
        TraceUtil.setUndefinedValuesToZero(aggregatedFteRequirements);

        MultiFteRequirementsQueueDTO multiQueueDto = new MultiFteRequirementsQueueDTO(singleQueueDtos, 
        		(RequireTraceCube)aggregatedFteRequirements);
        CombinedFteRequirementsQueueDTO combinedQueueDto;
        if (!partialResponse.getSp().getSkillBased()) {
        	combinedQueueDto = new CombinedFteRequirementsQueueDTO(partialResponse.getSPQueue(), 
        			(RequireTraceCube)aggregatedFteRequirements, multiQueueDto);
        } else {
        	combinedQueueDto = new CombinedFteRequirementsQueueDTO(partialResponse.getSPQueue(), multiQueueDto);
        }
        Response<CombinedFteRequirementsQueueDTO> fullResponse = 
        	new Response<CombinedFteRequirementsQueueDTO>(partialResponse.getMedia(), 
        				partialResponse.getCampaign(), 
        				partialResponse.getSp(), combinedQueueDto, partialResponse.getHoursOfOperation());
	    return fullResponse;
    }
	
	/**
	 * Creates and returns a partial response for multi-selected FTE Requirements queues.
	 */
	private Response<MultiFteRequirementsQueueDTO> getPartialResponseForMultiQueue(RequestContext context, Map request, Map response) throws Exception {
		HashMap<ID, ID> queueMediaMap = new HashMap<ID, ID>();
    	MultiQueuePartialResponse partialResponse = createMultiQueuePartialResponse(context, request);
    	Date adjustedEnd = new Date(partialResponse.getSp().getEndTime().getTime() - 60000);  // 1 minute before.
    	List<SingleFteRequirementsQueueDTO> singleQueueDtos = new ArrayList<SingleFteRequirementsQueueDTO>();
    	TraceCube[] constituentFteRequirementsCubes = new TraceCube[partialResponse.getQueues().size()]; 
		FteRequirementsManager manager =
			WfmManagerFactory.getFteRequirementsManager(context.isInWhatIfMode());
    	for (int i = 0; i <= partialResponse.getQueues().size() - 1; i++) {
    		queueMediaMap.put(partialResponse.getQueues().get(i).getID(), partialResponse.getMedia().getID());
			singleQueueDtos.add(new SingleFteRequirementsQueueDTO(partialResponse.getQueues().get(i), partialResponse.getSPQueues().get(i)));
			RequireTraceCube fteRequirementsTraceCube = manager.getFteRequirementsTraceCube(partialResponse.getCampaign().getID(),
					partialResponse.getQueues().get(i).getID(), partialResponse.getMedia().getID(),
					partialResponse.getSp().getStartTime(), adjustedEnd);
			constituentFteRequirementsCubes[i] = fteRequirementsTraceCube;
		}

    	TraceCube aggregatedFteRequirements = TraceOperator.combineQueue(constituentFteRequirementsCubes, true, queueMediaMap);
    	TraceUtil.setUndefinedValuesToZero(aggregatedFteRequirements);
    	MultiFteRequirementsQueueDTO multiQueueDto = new MultiFteRequirementsQueueDTO(singleQueueDtos, 
    			(RequireTraceCube)aggregatedFteRequirements);
    	Response<MultiFteRequirementsQueueDTO> fullResponse = 
    		new Response<MultiFteRequirementsQueueDTO>(partialResponse.getMedia(), 
    				partialResponse.getCampaign(), 
    				partialResponse.getSp(), multiQueueDto, partialResponse.getHoursOfOperation());
        return fullResponse;
	}
	
	private void saveFteRequirementsForSpQueue(RequestContext context, Map request, Map response) {
		SPQueue spQueue = (SPQueue)request.get(FteRequirementsKeys.PARAM_SP_QUEUE);
		SchedulingPeriod sp = (SchedulingPeriod)request.get(FteRequirementsKeys.PARAM_SP);
		RequireTraceCube fteRequirementsTraceCube = 
			(RequireTraceCube)request.get(FteRequirementsKeys.PARAM_FTEREQUIREMENTS_TRACECUBE);
		
		try {
			//We need to set off hours values to 0 before we save.  We need to do this because
			//otherwise the FS Client will treat them as negative FTE.  Once the FS Client is gone,
			//we can remove this line.
			TraceUtil.setUndefinedValuesToZero(fteRequirementsTraceCube);

			FteRequirementsManager manager =
				WfmManagerFactory.getFteRequirementsManager(context.isInWhatIfMode());
			manager.saveFteRequirementsForSpQueue(fteRequirementsTraceCube, sp, spQueue);
		} catch (Exception e) {
			e.printStackTrace();
			notifyError(context, response, e);
		}
	}
	
	/**
	 * Returns the FTE Requirements for the given SP Queue that are based solely on the forecast
	 * for that SP Queue (in other words, no manual modifications are taken into account).
	 */
	private void restoreFteRequirementsForSpQueue(RequestContext context, Map request, Map response) {
		SPQueue spQueue = (SPQueue)request.get(FteRequirementsKeys.PARAM_SP_QUEUE);
		SchedulingPeriod sp = (SchedulingPeriod)request.get(FteRequirementsKeys.PARAM_SP);
		
		try {
			FteRequirementsManager manager =
				WfmManagerFactory.getFteRequirementsManager(context.isInWhatIfMode());
			RequireTraceCube fteRequirements = manager.calculateUnmodifiedFteRequirementsForSPQueue(sp, spQueue);
			response.put(FteRequirementsKeys.RETVAL_FTEREQUIREMENTS_TRACECUBE, fteRequirements);
		} catch (Exception e) {
			e.printStackTrace();
			notifyError(context, response, e);
		}
	}
