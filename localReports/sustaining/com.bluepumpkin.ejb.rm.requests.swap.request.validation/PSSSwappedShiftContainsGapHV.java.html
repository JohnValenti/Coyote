package com.bluepumpkin.ejb.rm.requests.swap.request.validation;

import java.io.Serializable;
import java.util.Iterator;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.effectivity.TimePeriodUtil;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftEventAssignment;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;

import com.bluepumpkin.ejb.rm.requests.swap.request.model.ShiftSwapRequest;
import com.bluepumpkin.ejb.rm.requests.swap.shiftitem.model.ShiftSwapItem;
import com.bluepumpkin.ejb.rm.util.RmUtil;
import com.verint.ejb.wfm.WfmManagerFactory;

/**
 *
 * <b>Hard Validation: applies only for partial shift swap requests</b>
 * <p>
 *    Validates that the shifts being swapped does not contain any gap activity
 * <p>
 *    Checks for both the shifts for a two way swap
 */
public class PSSSwappedShiftContainsGapHV  implements Validator
{
    private static final String m_className = PSSSwappedShiftContainsGapHV.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);
    private ShiftSwapValidationCache m_vc = null;

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception
    {
        String methodName = "validate";
        ShiftSwapRequest ssr = (ShiftSwapRequest)validatable;

        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));
        m_vc = (ShiftSwapValidationCache) ssr.getValidationCache();

        ShiftSwapValidationCache valCache = ssr.getCache();
        ShiftSwapItem ssItem1 = (ShiftSwapItem) ssr.getShiftSwapItems().get(0);
        ShiftSwapItem ssItem2 = (ShiftSwapItem) ssr.getShiftSwapItems().get(1);
        if ((ssItem1.getIsPartial() || ssItem2.getIsPartial())) {
        	return verifyGapActivities(ssr);
        }
        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName));
        return null;
    }
    
    private ValidationResult verifyGapActivities(ShiftSwapRequest ssr) throws Exception{
    	ValidationResult result = null;
    	ShiftSwapItem ssItem1 = (ShiftSwapItem) ssr.getShiftSwapItems().get(0);
        ShiftSwapItem ssItem2 = (ShiftSwapItem) ssr.getShiftSwapItems().get(1);
        
        ScheduleAccessManager sam = WfmManagerFactory.getScheduleAccessManager();
        
        if (!ssItem1.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF))
        	result = verifyGapActivitiesForItem(ssItem1, ssr.isTentative(), sam, ssr);
        if (!ssItem2.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF))
        	result = verifyGapActivitiesForItem(ssItem2, ssr.isTentative(), sam, ssr);
        
    	return result;
    }
    
    private ValidationResult verifyGapActivitiesForItem(ShiftSwapItem ssItem, 
    		boolean tentative, ScheduleAccessManager sam, ShiftSwapRequest ssr) throws Exception{
    	ValidationResult result = null;
//    	 get the published shift assignment associated with SSItem.
        ShiftAssignment pubShift = m_vc.getShiftAssignForSSItem(ssItem, true, !ssItem.getIsPartial(), sam);
		
		// fetch the unpublished shift assignment only for non tentative requests.
		ShiftAssignment unpubShift = null;
		if ( !tentative ) { 
			unpubShift = m_vc.getShiftAssignForSSItem(ssItem, false, !ssItem.getIsPartial(), sam);
		}
    	
		//check only for overlap
    	if (pubShift != null){
    		//iterate through shift activities to look for gap
    		for (Iterator itr = pubShift.getChildObjects(0).iterator(); itr.hasNext(); ) {
    			ShiftEventAssignment shiftEventAssn = (ShiftEventAssignment) itr.next();
    			if (shiftEventAssn.getActivityID().equals(Activity.ACTIVITY_SHIFT_OVERTIME_GAP) &&
    					TimePeriodUtil.overlap(shiftEventAssn.getStartTime(), shiftEventAssn.getEndTime(),
    							ssItem.getStartDate(), ssItem.getEndDate())){
    				//throw hard validation
    				return setHardValidationResult( ssr, 
    						ssr.getValidationCache().getEmployeeNameByID(ssItem.getEmployeeID()),
    						RmEjbBundleKey.PSS_SHIFT_CONTAINS_GAP_ACTIVITY);		
    			}
    		}
    	}
    	
    	//check only for overlap
    	if (!tentative && unpubShift != null){
//    		iterate through shift activities to look for gap
    		for (Iterator itr = pubShift.getChildObjects(0).iterator(); itr.hasNext(); ) {
    			ShiftEventAssignment shiftEventAssn = (ShiftEventAssignment) itr.next();
    			if (shiftEventAssn.getActivityID().equals(Activity.ACTIVITY_SHIFT_OVERTIME_GAP) &&
    					TimePeriodUtil.overlap(shiftEventAssn.getStartTime(), shiftEventAssn.getEndTime(),
    							ssItem.getStartDate(), ssItem.getEndDate())){
    				//throw hard validation
    				return setHardValidationResult( ssr, 
    												ssr.getValidationCache().getEmployeeNameByID(ssItem.getEmployeeID()),
    												RmEjbBundleKey.PSS_SHIFT_CONTAINS_GAP_ACTIVITY);		
    			}		
    		}		
    	} 
    	
    	return result;
    }
    
    private ValidationResult setHardValidationResult(Validatable validatable, String msgKey, Serializable obj1)
    {
        return ValidationUtil.setHardValidationResult(validatable, msgKey, obj1, this.getClass().getName());
    }
}
