package com.verint.ejb.bbm.servicegoals.ejb;

import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.verint.ejb.bbm.servicegoals.model.ASATimeSeriesDataItem;
import com.verint.ejb.bbm.servicegoals.model.ServiceGoalsTimeSeriesDataItem;
import com.verint.ejb.bbm.servicegoals.model.ServiceGoalsTimeSeriesDataItemFieldInfo;
import com.verint.ejb.bbm.servicegoals.model.ServiceLevelPercentTimeSeriesDataItem;

/**
 * This is a special DAO written for bulk create/update of the service goals.
 * For the regular oiperations, use the DAO specific to the service goal type
 * 
 */
public class ServiceGoalsTimeSeriesBulkOperationsDAO extends ServiceGoalsTimeSeriesDAO<ServiceGoalsTimeSeriesDataItem> {

	private static Category m_cat = Log.initCategory(ServiceGoalsTimeSeriesBulkOperationsDAO.class.getName());
	private static FieldInfo m_fieldInfo = new ServiceGoalsTimeSeriesDataItemFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	@Override
	protected ServiceLevelPercentTimeSeriesDataItem createValueObject() {
		throw new UnsupportedOperationException(
				"This class is only for Bulk creates or updates of the Service Goal Timeseries. ");
	}

	public void bulkCreateOrUpdateSLOrASAServiceGoalTimeSeries(
			Map<SPQueue, List<ServiceGoalsTimeSeriesDataItem>> serviceGoalTimeSeriesBySPQueue)
					throws BbmUpdateException, JdmoException {
		// Finally, save the service goals, irrespective of whether they are new
		// or old
		String rootStart = "<GoalSeries>";
		String rootEnd = "</GoalSeries>";

		StringBuilder slStringBuilder = new StringBuilder();
		slStringBuilder.append(rootStart);

		for (SPQueue spQueue : serviceGoalTimeSeriesBySPQueue.keySet()) {
			String spqIdStr = spQueue.getID().toString();
			slStringBuilder.append(getGoalSeriesDataAsString(spqIdStr, serviceGoalTimeSeriesBySPQueue.get(spQueue)));
		}
		slStringBuilder.append(rootEnd);
		// call the procedure and execute this string
		saveServiceGoals(slStringBuilder.toString());

	}

	protected boolean validateAndReturnIfSLType(ServiceGoalsTimeSeriesDataItem item) throws IllegalArgumentException {
		if (item instanceof ServiceLevelPercentTimeSeriesDataItem) {
			return true;
		} else if (!(item instanceof ASATimeSeriesDataItem)) {
			m_cat.error(
					"ServiceGoalsTimeSeriesBulkOperationsDAO.validateAndReturnIfSLType: encountered service goal type other than "
							+ "Service Level or ASA.");
			throw new IllegalArgumentException(
					"ServiceGoalsTimeSeriesBulkOperationsDAO.validateAndReturnIfSLType: encountered service goal type other than "
							+ "Service Level or ASA.");
		}
		return false;
	}

	protected String getGoalSeriesDataAsString(String spqIdStr, List<ServiceGoalsTimeSeriesDataItem> items) {		
		boolean isServiceGoal = validateAndReturnIfSLType(items.get(0));
		
		String elementStart = "<G>";
		String elementEnd = "</G>";
		String seperator = ",";

		StringBuilder slStringBuilder = new StringBuilder();
		for (ServiceGoalsTimeSeriesDataItem item : items) {

			slStringBuilder.append(elementStart);
			slStringBuilder.append(spqIdStr);
			slStringBuilder.append(seperator);
			slStringBuilder.append(JdmoUtil.formatDBString(item.getTime()));
			slStringBuilder.append(seperator);
			if (isServiceGoal) {
				slStringBuilder.append(item.getValue());
				slStringBuilder.append(seperator);

			} else {
				slStringBuilder.append(seperator);
				slStringBuilder.append(item.getValue());
			}
			slStringBuilder.append(seperator);
			slStringBuilder.append(seperator);
			slStringBuilder.append(elementEnd);
		}
		return slStringBuilder.toString();
	}

	/**
	 * Calls the procedure to update the service goals. The data should be a
	 * comma-separated list of spqueueid, date, service level, asa, occupancy,
	 * max dials for each row. The data is in the format
	 * <GoalSeries><G>0000066D.003701,20160501 04:00:00,0.75,,,</S>
	 * <S>0000066D.003701,20160501 08:00:00,,0.68,,</G></GoalSeries>
	 */
	protected void saveServiceGoals(String data) throws JdmoException {
		Jdmo jdmo = new Jdmo();
		try {
			String pStmt = "BP_BF_CREATEORUPDATESERVICEGOALS";
			JdmoQuery jQuery = jdmo.createQuery(pStmt.toString(), Jdmo.STORPROC_QUERY);
			jQuery.setParString(1, data);
			JdmoRowset rs = jdmo.createRowset(jQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

		} finally {
			jdmo.cleanUp();
		}
	}

}
