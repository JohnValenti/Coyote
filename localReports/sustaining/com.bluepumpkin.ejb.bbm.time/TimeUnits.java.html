package com.bluepumpkin.ejb.bbm.time;

/**
 * This enumeration lists the possible time units which may be used to define
 * time intervals.
 */
public enum TimeUnits {

	Minute(10), Hour(20), Day(30), Week(40), Month(50);

	/** An internal representation of the relative magnitude of the time units. */
	private final int m_relativeSize;

	private TimeUnits(int relativeSize) {
		m_relativeSize = relativeSize;
	}

	/**
	 * Returns whether this TimeUnits instance is shorter than the specified
	 * TimeUnits instance.
	 * 
	 * @param other
	 * @return
	 */
	public boolean isShorterThan(TimeUnits other) {
		return this.m_relativeSize < other.m_relativeSize;
	}

	/**
	 * Returns whether this TimeUnits instance is longer than the specified
	 * TimeUnits instance.
	 * 
	 * @param other
	 * @return
	 */
	public boolean isLongerThan(TimeUnits other) {
		return this.m_relativeSize > other.m_relativeSize;
	}
}
