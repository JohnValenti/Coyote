package com.verint.web.fs.workrules.assignmentrules.question;

import java.rmi.RemoteException;
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.Pair;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.empworkrule.model.TimePeriodOfDay;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.bluepumpkin.web.bbm.activity.ActivityModelHandler;
import com.bluepumpkin.web.bbm.activity.ActivityModelHandler.ActivitySelectionProperty;
import com.bluepumpkin.web.core.setup.general.GeneralSetupModelHandler;
import com.verint.web.fs.workrules.WorkRulesModelHandler;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.question.PeriodQuestion.PeriodResponseType;
import com.verint.web.fs.workrules.assignmentrules.question.RuleTypeQuestion.RuleTypeResponseType;
import com.witness.web.uif.system.RequestContext;

/**
 * Be sure to set the ruleType. This is likely not to be immediately available when the instance is constructed. It
 * shouldn't be needed for retrieving the first question (which is the Rule Type). But it may be needed after that.
 */
public class DefaultQuestionDataProvider implements QuestionDataProvider {
	private RequestContext context;
	private ComplexWorkRule assignmentRuleBeingEdited;
	private RuleTypeResponseType ruleType;
	private PeriodResponseType periodType;
	private List<Pair<QuestionType, Response>> answeredQuestions = new ArrayList<Pair<QuestionType, Response>>();
	private Organization organization;
	private Map<ID, String> shiftNameMap;

	public DefaultQuestionDataProvider(RequestContext context, Organization org, ComplexWorkRule assignmentRuleBeingEdited, 
			Map<ID, String> shiftIDNames) {
		super();
		this.context = context;
		this.organization = org;
		this.assignmentRuleBeingEdited = assignmentRuleBeingEdited;
		shiftNameMap = shiftIDNames;
	}

	@Override
	public List<TimePeriodOfDay> getTimePeriods() throws BPException {
		return new ArrayList<TimePeriodOfDay>(WorkRulesModelHandler.getTimePeriodsOfDay(context));
	}

	@Override
	public boolean hasShifts() throws BPException {
		return shiftNameMap != null && !shiftNameMap.isEmpty();
	}

	@Override
	public ID getOrgID() {
		return organization.getID();
	}

	@Override
	public ComplexWorkRule getAssignmentRuleBeingEdited() throws BPException {
		return assignmentRuleBeingEdited;
	}

	@Override
	public RuleTypeResponseType getRuleType() throws BPException {
		return ruleType;
	}
	
	public PeriodResponseType getPeriodType() throws BPException {
		return periodType;
	}

	/**
	 * After an answer has been determined, this method should be called
	 * 
	 * @param questionType
	 * @param response
	 */
	public void setAnswer(QuestionType questionType, Response answer) {
		if (answer == null) return;
		if (QuestionType.RuleType.equals(questionType)) {
			this.ruleType = (RuleTypeResponseType) answer.getResponseType();
		}
		if (QuestionType.CalendarPeriod.equals(questionType)) {
			this.periodType = (PeriodResponseType) answer.getResponseType();
		}
		this.answeredQuestions.add(new Pair<QuestionType, Response>(questionType, answer));
	}
	
	@Override
	public List<Pair<QuestionType, Response>> getAnswers() {
		return this.answeredQuestions;
	}

	@Override
	public boolean hasShiftEvents() throws BbmFinderException {
		return ! WorkRulesModelHandler.getShiftEventsByOrgHierarchy(this.context, this.getOrgID()).isEmpty();
	}

	@Override
	public boolean hasActivities() throws BbmException {
		try {
			return ! ActivityModelHandler.getActivitiesByActivitySelectionType(context, ActivitySelectionProperty.UsedInShiftEvent,
					Collections.singletonList(getOrgID())).isEmpty();
		} catch (RemoteException e) {
			throw new BbmException(e);
		}
	}
	
	@Override
	public boolean isConsecutiveWeekendsEnabled() throws BPException {
		try {
			return "1".equals(GeneralSetupModelHandler.getGenericDbValue("ConsecutiveWeekends"));
		} catch (BPException e) {
			throw e;
		} catch (Exception e) {
			throw new BPException(e);
		}
	}

	@Override
	public Organization getOrganization() {
		return this.organization;
	}

	@Override
	public boolean isMonthlyRulesEnabled() throws BPException {
		 try {
			return WorkRulesModelHandler.isMonthlyRulesEnabled();
		} catch (BPException e) {
			throw e;
		} catch (Exception e) {
			throw new BPException(e);
		}
	}

	@Override
	public boolean isMultiWeekRulesEnabled() throws BPException {
		try {
			return WorkRulesModelHandler.isMultiWeekRulesEnabled();
		} catch (BPException e) {
			throw e;
		} catch (Exception e) {
			throw new BPException(e);
		}
	}
	
	@Override
	public Shift getShift(ID shiftID) throws BPException {
		return WorkRulesModelHandler.getShift(context, shiftID);
	}		
}
