package com.verint.web.fs.workrules.assignmentrules.question;

import java.util.ArrayList;
import java.util.Collections;
import java.util.HashMap;
import java.util.Iterator;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.TimeOfDay;
import com.bluepumpkin.ejb.bbm.empworkrule.model.TimePeriodOfDay;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit.TimeType;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule.WorkUnitCountConstraint;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseImpl;
import com.verint.web.fs.workrules.assignmentrules.ResponseParam;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamConstraint;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamType;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;

public class TimePeriodOfDayQuestion implements Question {
	private List<TimePeriodOfDay> data_timePeriods = new ArrayList<TimePeriodOfDay>();
	private ComplexWorkRule data_assignmentRule;
	private List<TimePeriodOfDayResponse> data_responses = new ArrayList<TimePeriodOfDayQuestion.TimePeriodOfDayResponse>();
	private Response data_initialAnswer;
	private boolean data_includeBetweenClauses;
	private Map<WorkUnit.TimeType, TimePeriodOfDayResponse> data_responsesByTimeType = new HashMap<WorkUnit.TimeType, TimePeriodOfDayQuestion.TimePeriodOfDayResponse>();
	private TimePeriodOfDayResponseType data_endBetweenResponseType;
	private TimePeriodOfDayResponseType data_startBetweenResponseType;
	private Map<TimePeriodOfDay, TimePeriodOfDayResponseType> data_dbResponseTypes = new HashMap<TimePeriodOfDay, TimePeriodOfDayQuestion.TimePeriodOfDayResponseType>();

	public TimePeriodOfDayQuestion(ComplexWorkRule assignmentRule, List<TimePeriodOfDay> timePeriods,
			boolean includeBetweenClauses) {
		super();
		data_assignmentRule = assignmentRule;
		data_timePeriods.addAll(timePeriods);
		data_includeBetweenClauses = includeBetweenClauses;

		constructResponseTypes();
		constructResponses();
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.TimeOfDay;
	}

	@Override
	public List<TimePeriodOfDayResponse> getResponses() {
		return data_responses;
	}

	@Override
	public Response getInitialAnswer() {
		return data_initialAnswer;
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		WorkUnit wu = assignmentRule.getWorkUnit();
		if (wu == null) {
			wu = new WorkUnit();
			assignmentRule.setWorkUnit(wu);
		}

		TimePeriodOfDayResponse tpodr = ((TimePeriodOfDayResponse) answer);
		if (tpodr.getResponseType().isDBVal()) {
			wu.setTimePeriodOfDay(tpodr.getTimePeriodOfDay());
		} else {
			assignmentRule.getWorkUnit().setTimeType(tpodr.getResponseType().getTimeType());
			WorkUnitCountConstraint wucc = tpodr.getResponseType().getWorkUnitCountConstraint();
			if (wucc != null) {
				assignmentRule.setWorkUnitCountConstraint(wucc);
			}
			Iterator<ResponseParam> paramIter = tpodr.getResponseParameters().iterator();
			TimeOfDay fromTime = (TimeOfDay)paramIter.next().getValue();
			TimeOfDay toTime = (TimeOfDay) paramIter.next().getValue();
			wu.setFromTime(fromTime);
			wu.setToTime(toTime);
		}
	}

	private void constructResponseTypes() {
		this.data_startBetweenResponseType = new TimePeriodOfDayResponseType(TimeType.StartBetween, WorkUnitCountConstraint.FairlyStartingBetween);
		this.data_endBetweenResponseType = new TimePeriodOfDayResponseType(TimeType.EndBetween, WorkUnitCountConstraint.FairlyEndingBetween);
		for (TimePeriodOfDay tpod : data_timePeriods) {
			data_dbResponseTypes.put(tpod, new TimePeriodOfDayResponseType(tpod));
		}
	}

	private void constructResponses() {
		for (TimePeriodOfDay tpod : data_timePeriods) {
			TimePeriodOfDayResponse response = new TimePeriodOfDayResponse(
					data_dbResponseTypes.get(tpod),
					Collections.<ResponseParam> emptyList(),
					tpod);
			data_responses.add(response);
		}
		if (data_includeBetweenClauses) {
			TimeOfDay fromTime = new TimeOfDay(0);
			TimeOfDay toTime = new TimeOfDay(0);
			if (data_assignmentRule != null && data_assignmentRule.getWorkUnit() != null) {
				fromTime = data_assignmentRule.getWorkUnit().getFromTime();
				toTime = data_assignmentRule.getWorkUnit().getToTime();
			}

			// Start Between x and y
			ArrayList<ResponseParam> startParams = new ArrayList<ResponseParam>();
			ResponseParam startBetweenFrom = new ResponseParam(
					ResponseParamType.TimeOfDay,
					Collections.<ResponseParamConstraint> emptyList(),
					fromTime);
			startParams.add(startBetweenFrom);
			ResponseParam startBetweenTo = new ResponseParam(
					ResponseParamType.TimeOfDay,
					Collections.<ResponseParamConstraint> emptyList(),
					toTime);
			startParams.add(startBetweenTo);
			TimePeriodOfDayResponse betweenStartingResponse = new TimePeriodOfDayResponse(
					data_startBetweenResponseType,
					startParams,
					null);
			data_responses.add(betweenStartingResponse);
			data_responsesByTimeType.put(WorkUnit.TimeType.StartBetween, betweenStartingResponse);

			// End Between x and y
			ArrayList<ResponseParam> endParams = new ArrayList<ResponseParam>();
			ResponseParam endBetweenFrom = new ResponseParam(
					ResponseParamType.TimeOfDay,
					Collections.<ResponseParamConstraint> emptyList(),
					fromTime);
			endParams.add(endBetweenFrom);
			ResponseParam endBetweenTo = new ResponseParam(
					ResponseParamType.TimeOfDay,
					Collections.<ResponseParamConstraint> emptyList(),
					toTime);
			endParams.add(endBetweenTo);
			TimePeriodOfDayResponse betweenEndingResponse = new TimePeriodOfDayResponse(
					data_endBetweenResponseType,
					endParams,
					null);
			data_responses.add(betweenEndingResponse);
			data_responsesByTimeType.put(WorkUnit.TimeType.EndBetween, betweenEndingResponse);
		}

		// find initial answer
		if (data_assignmentRule != null && data_assignmentRule.getWorkUnit() != null) {
			WorkUnit wu = data_assignmentRule.getWorkUnit();
			TimePeriodOfDay tpod = wu.getTimePeriodOfDay();
			if (tpod != null) {
				for (TimePeriodOfDayResponse response : data_responses) {
					if (tpod.equals(response.getTimePeriodOfDay())) {
						data_initialAnswer = response;
						break;
					}
				}
			} else {
				if (wu.getTimeType() != null) {
					data_initialAnswer = data_responsesByTimeType.get(wu.getTimeType());
				}
			}
		}
	}

	public static class TimePeriodOfDayResponseType implements ResponseType {
		private TimeType timeType = TimeType.DBTimePeriod;
		private TimePeriodOfDay tpod = null;
		private WorkUnitCountConstraint wucc;

		public TimePeriodOfDayResponseType(TimePeriodOfDay tpod) {
			this(TimeType.DBTimePeriod, null);
			this.tpod = tpod;
		}

		public TimePeriodOfDayResponseType(TimeType timeType, WorkUnitCountConstraint wucc) {
			this.timeType = timeType;
			this.wucc = wucc;
		}

		public TimeType getTimeType() {
			return timeType;
		}

		public boolean isDBVal() {
			return timeType == TimeType.DBTimePeriod;
		}
		
		/**
		 * May return null if the answer does not determine a specific work unit count constraint.
		 * @return
		 */
		public WorkUnitCountConstraint getWorkUnitCountConstraint() {
			return wucc;
		}

		public TimePeriodOfDay getTimePeriodOfDay() {
			return tpod;
		}

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Collections.emptyList();
		}
		
		public String getIDStr() {
			return timeType + ", tpodid: " + ( tpod == null ? "": tpod.getID().toString());
		}

		@Override
		public int hashCode() {
			final int prime = 31;
			int result = 1;
			result = prime * result + ((timeType == null) ? 0 : timeType.hashCode());
			result = prime * result + ((tpod == null) ? 0 : tpod.hashCode());
			return result;
		}

		@Override
		public boolean equals(Object obj) {
			if (this == obj)
				return true;
			if (obj == null)
				return false;
			if (getClass() != obj.getClass())
				return false;
			TimePeriodOfDayResponseType other = (TimePeriodOfDayResponseType) obj;
			if (timeType != other.timeType)
				return false;
			if (tpod == null) {
				if (other.tpod != null)
					return false;
			} else if (!tpod.equals(other.tpod))
				return false;
			return true;
		}

	}

	public static class TimePeriodOfDayResponse extends ResponseImpl<TimePeriodOfDayResponseType> {
		private TimePeriodOfDay timePeriodOfDay;

		public TimePeriodOfDayResponse(TimePeriodOfDayResponseType responseType, List<ResponseParam> parameters,
				TimePeriodOfDay timePeriodOfDay) {
			super(QuestionType.TimeOfDay, responseType, parameters);
			this.timePeriodOfDay = timePeriodOfDay;
		}

		public TimePeriodOfDay getTimePeriodOfDay() {
			return timePeriodOfDay;
		}

		@Override
		public String getIDStr() {
			return getResponseType().getIDStr();
		}

	}
}
