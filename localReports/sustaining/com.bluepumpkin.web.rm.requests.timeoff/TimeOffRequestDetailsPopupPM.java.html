package com.bluepumpkin.web.rm.requests.timeoff;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAuditTrail;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;
import com.bluepumpkin.web.rm.keys.RmKeys;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.requests.RequestDetailsPopupPM;
import com.bluepumpkin.web.rm.requests.RequestUtil;
import com.bluepumpkin.web.rm.requests.RequestsMH;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:		     TimeOffRequestDetailsPopupPM
 * Description:  Time Off Request Details
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       David Su
 * @version      1.0
 */
public class TimeOffRequestDetailsPopupPM extends RequestDetailsPopupPM {
	private MultiColListPC m_choiceListPC;

	private String m_toTypeName;
	private OrganizationSetting m_orgSetting;

	/**
	 * Constructor
	 */
	public TimeOffRequestDetailsPopupPM(RequestContext context) {
		super(context, RmKeys.RH_TIMEOFF_DETAILS);
		m_request = new TORequest(TORequest.DL_BASIC);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, TimeOffRequestDetailsPopupPM.class);
	}//getInstance

	/**
	 * Fetch Sepecific Request for given ID
	 */
	@Override
	protected RequestAggregate fetchRequest(ID requestID) throws Exception {
		return TimeOffRequestsMH.getTORequest(requestID);
	}

	/**
	 * Load Specific Data
	 */
	@Override
	protected void loadSpecifiedData() throws Exception {
		TORequest request = (TORequest)m_request;
		m_valResults = m_request.getValidationResults(false);
		m_toTypeName = TimeOffRequestsMH.getTimeOffTypeName(m_context, request);

//		check for waitlist feature
		m_orgSetting = RequestsMH.getOrgSettingForEmployee(m_context, m_empID);
	}

	/**
	 * Return Content Title Item Name
	 */
	@Override
	protected String getContentTitleItemName(){
		return i18n(m_bundle, RmWebBundleKey.REQUEST_TYPE_TIME_OFF);
	}

	/**
	 * Initialize Content
	 */
	@Override
	protected void initContent() {
		initMainList();
		initAlertList();
		initChoiceList();
		initAuditTrailList();
	}

	/**
	 * Initialize Main List
	 */
	@Override
	protected void initMainList() {
		super.initMainList();

		String title = i18n(m_bundle, RmWebBundleKey.REQUEST_SUMMARY);
		m_mainContainerPC.setTitle(title);

        m_mainContainerPC.setSummary(i18n(m_bundle, RmWebBundleKey.REQUEST_SUMMARY_TABLE));

		TORequest toRequest = (TORequest)m_request;
		m_mainContainerPC.addRow(
				i18n(m_bundle, RmWebBundleKey.EMPLOYEE_NAME_LABEL),
				getEmployeeName(toRequest.getEmployeeID()));

		m_mainContainerPC.addRow(
				i18n(m_bundle, RmWebBundleKey.TIMEOFF_TYPE_LABEL),
				m_toTypeName);

		String dbTypeRBKey = TimeOffRequestUtil.
				getDebitTypeRBKey(toRequest.getTimeOffDebitType());
		m_mainContainerPC.addRow(
				i18n(m_bundle, RmWebBundleKey.TIMEOFF_HOURS_LABEL),
				i18n(m_bundle, dbTypeRBKey));

//		add the option to add to waitlist if applicable
		if (toRequest.isWaitlistEnabled() && toRequest.isWaitlisted()) {
			m_mainContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.TIMEOFF_WAITLIST_EXPIRY_DATE),
					RequestUtil.getDateDisplay(m_context,toRequest.getWaitlistInfo().getTOWaitlistExpiryDate()));
			if (!com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil.isWaitListRankOnRqPageDisabled()) {
			m_mainContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.TIMEOFF_WAITLIST_RANK),
					TimeOffRequestsMH.getWaitlistRank(toRequest.getID()) );
			}
		}
	}


	/**
	 * Initialize Choise List
	 */
	public void initChoiceList() {
		TORequest toRequest = (TORequest)m_request;
		m_choiceListPC = createNewList();

        m_choiceListPC.setTableAttribute("summary", i18n(m_bundle, RmWebBundleKey.TO_REQUEST_POPUP_CHOICES_TABLE_SUMMARY));

		String reqStatus = toRequest.getRequestStatus();
		ID selectedChoiceID = toRequest.getApprovedChoiceID();
		boolean isSelectable=false;
		boolean isShowSelectedInfo =
				RequestAuditTrail.STATUS_APPROVED.equals(reqStatus) ||
				(RequestAuditTrail.STATUS_TENTATIVE.equals(reqStatus) && isManagerMode());
		TimeOffRequestUtil.TimeOffChoiceListPCConfig config = new TimeOffRequestUtil.TimeOffChoiceListPCConfig()
																.setIsSelectable(isSelectable)
																.setIsShowSelectedInfo(isShowSelectedInfo)
																.setIsShowAlert(true);
		TimeOffRequestUtil.initTimeOffChoiceListPC(m_context, m_choiceListPC, toRequest,
				selectedChoiceID, m_common_bundle, m_bundle, config);
	}
}
