package com.verint.web.rest.wfm.exception;

public class EntityTimeSeriesException extends EntityException {

	public EntityTimeSeriesException(String entityName, String id) {
		super(entityName, id);
	}

	@Override
	public String getTitle() {
		return "A BbmTimeSeriesException occurred on the server.  Please check the server logs for details.  Entity: " +
				getEntityName() + " ID: " + getId();
	}

	@Override
	public String getDetail() {
		return "There was a problem with a time series for entity: " + getEntityName().toString();
	}
}
