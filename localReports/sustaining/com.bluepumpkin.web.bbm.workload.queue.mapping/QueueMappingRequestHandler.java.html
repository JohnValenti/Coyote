public class QueueMappingRequestHandler extends RequestHandler {
    private static final Category log = Log.initCategory(QueueMappingRequestHandler.class.getName());
    private ResourceBundle m_bbmBundle;
    private static final String DEFAULT_PAGE = "defaultPage";
    private static final String QUEUE_GROUP_MAPPING_PAGE = "queueGroupMappingPage";
    private static final String QUEUE_GROUP_MAPPING_LIST_PAGE = "queueGroupMappingListPage";
    private static final String CONFIGURE_PRIVILEGE = PrivilegeKeys.FS_CONFIGUREQUEUES;
    private Localizer m_localizer;
    protected final String ACTION_SEARCH_DSG = "SEARCH_DSG";

    /**
     * Process Requests for the queue area
     * 
     * @throws Exception
     */
    @Override
    public void processRequest(RequestContext context) throws Exception {
        m_localizer = context.getLocalizer();
        m_bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);

        // Process requested action
        if (isAction(context, PageAction.NEXT)) {
            processNext(context);
        } else if (isAction(context, PageAction.PREVIOUS)) {
            processPrev(context);
        } else if (isAction(context, PageAction.EDIT)) {
            processListEdit(context);
        } else if (isAction(context, PageAction.DELETE)) {
            processListDelete(context);
        } else if (isAction(context, PageAction.SAVE)) {
            processSave(context);
        } else if (isAction(context, PageAction.RESET)) {
            processRevert(context);
        } else if (isAction(context, PageAction.CANCEL)) {
            processCancel(context);
        } else if (isAction(context, "CHANGEDS_ACTION")) {
            gotoForm(context);
        } else if (isAction(context, PageAction.TABLE_SORT)) {
            if (isWorkpaneMode(context)) {
                gotoForm(context);
            } else {
                final QueueMappingListPM listModel = new QueueMappingListPM(context);
                listModel.extractParameters(context.getRequest());
                listModel.setDefaultSortColumn();
                context.setPageModel(listModel);
                gotoList(context);
            }
        } else if (isAction(context, ACTION_SEARCH_DSG)) {
            searchDSG(context);
        } else {
            gotoList(context);
        }
    }

    /**
     * Processes the next button action on the pagination toolbar.
     *
     * @param context
     */
    protected void processNext(RequestContext context) {
        final QueueMappingListPM listModel = new QueueMappingListPM(context);
        context.setPageModel(listModel);
        listModel.adjustEditIndexForward();
        // Go back to the list
        this.gotoList(context);
    }

    /**
     * Processes the previous button action on the pagination toolbar.
     *
     * @param context
     */
    protected void processPrev(RequestContext context) {
        final QueueMappingListPM listModel = new QueueMappingListPM(context);
        context.setPageModel(listModel);
        listModel.adjustEditIndexBackward();
        // Go back to the list
        this.gotoList(context);
    }

    /**
     * Return true if it is in Workpane Mode
     */
    protected static boolean isWorkpaneMode(RequestContext context) {
        return "true".equals(context.getParameter(WorkpaneFieldNames.IS_WORKPANE_MODE));
    }

    private void processSave(RequestContext context) {
        if (!isAuthorized(context, CONFIGURE_PRIVILEGE)) {
            gotoList(context);
            return;
        }
        final QueueMappingPM formModel = new QueueMappingPM(context);
        formModel.extractParameters(context.getRequest());
        // ID queueID = formModel.getSelectedIDObject();

        final ID queueID = new ID(formModel.getPerformActionOnID());

        ArrayList curMappedGroups = new ArrayList();
        if (formModel.getCurGroupIDColl() != null) {
            curMappedGroups = (ArrayList) formModel.getCurGroupIDColl();
        }

        final Collection newMappedGroups = formModel.getAssignedGroupIDs();

        for (final Iterator iter = newMappedGroups.iterator(); iter.hasNext();) {
            final ID groupID = (ID) iter.next();
            if (curMappedGroups.contains(groupID)) {// no need to do anything
                iter.remove();
                curMappedGroups.remove(groupID);
            }
        }
        try {
            // unlink unselected groups
            QueueGroupMH.unLinkGroupsFromQueue(queueID, curMappedGroups);
            // link new groups
            QueueGroupMH.linkGroupsToQueue(queueID, newMappedGroups);
        } catch (final BbmCreateException ex) {
            log.l7dError(BbmWebLogBundleKey.GROUP_MAPPING_ERROR_SAVE_DUPLICATED_KEY, ex);
            formModel.addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.GROUP_MAPPING_ERROR_SAVE_DUPLICATED_KEY));
        } catch (final Exception ex) {
            log.l7dError(BbmWebLogBundleKey.GROUP_MAPPING_ERROR_SAVE, ex);
            formModel.addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.GROUP_MAPPING_ERROR_SAVE));
        }
        gotoList(context);
    }

    private void processRevert(RequestContext context) {
        gotoForm(context);
    }

    /**
     * Go to List
     */
    private void gotoList(RequestContext context) {
        setNextPage(context, QUEUE_GROUP_MAPPING_LIST_PAGE);
    }

    /**
     * Process Edit Action from list
     */
    private void processListEdit(RequestContext context) {

        System.out.println("Inside processListEdit for QueueMappingRequest");

        // Extract Parameter
        final QueueMappingListPM model = new QueueMappingListPM(context);
        model.extractParameters(context.getRequest());
        final ID[] ids = model.getSelectedQueueIDs();

        final String selectedRow = ids[0].toString();
        final QueueMappingPM formModel = new QueueMappingPM(context);
        formModel.setPerformActionOnID(selectedRow);
        // System.out.println(selectedRow);
        // formModel.setIsListMode(true);

        context.setPageModel(formModel);

        if (!StringUtil.isEmpty(selectedRow)) {
            gotoForm(context);
        }

        System.out.println("exiting processListEdit for QueueMappingRequest");
    }

    /**
     * Process delete Action from list
     */
    private void processListDelete(RequestContext context) {
        if (!isAuthorized(context, CONFIGURE_PRIVILEGE)) {
            gotoList(context);
            return;
        }
        // QueueSummaryPageModel model= new QueueSummaryPageModel(context);
        // model.extractParameters(context.getRequest());

        final QueueListPageModel model = new QueueListPageModel(context);
        model.extractParameters(context.getRequest());
        final ID[] ids = model.getSelectedQueueIDs();
        final String selectedRow = ids[0].toString();

        final ArrayList list = new ArrayList();
        list.add(new ID(selectedRow));

        if (!StringUtil.isEmpty(selectedRow)) {
            try {
                // WorkloadModelHandler.deleteObject(new
                // ID(selectedRow),BPObjectInfo.BPOBJECT_QUEUE,context.getLocaleContext());
                WorkloadModelHandler.deleteQueues(context, list);
                model.setSelectedID("");
                model.setPerformActionOnID("");
                QueueUtil.setSelectedQueueID(context, "");
            } catch (final Exception ex) {
                log.l7dError(BbmWebLogBundleKey.QUEUE_DELETE_FAILURE, ex);
                model.addPageMessage(Message.ERROR_TYPE, verticaliseMsg(BbmWebBundleKey.QUEUE_DELETE_FAILURE, context));
            }

        }

        context.setPageModel(model);
        // Go back to the list
        // System.out.println("going to DEFAULT_PAGE 2");
        setNextPage(context, DEFAULT_PAGE);
    }

    /**
     * Go to Form
     */
    private void gotoForm(RequestContext context) {

        setNextPage(context, QUEUE_GROUP_MAPPING_PAGE);
    }

    /**
     * Process Cancel Action from form
     */
    private void processCancel(RequestContext context) {
        setNextPage(context, QUEUE_GROUP_MAPPING_LIST_PAGE);
    }

    private String verticaliseMsg(String key, RequestContext context) {
        // verticalise queue
        final Object[] args = new Object[1];
        args[0] = context.getSystemManagerFactory().getVerticalizationManager().getVerticalizedString(context, DefaultVerticalizationManager.QUEUESMALL);
        return MessageFormat.format(m_localizer.i18n(m_bbmBundle, key), args);
    }

    /**
     * @param context
     * @throws Exception
     */
    private void searchDSG(RequestContext context) throws Exception {
        final Map paramMap = ParameterUtil.getParameterMap(context.getRequest());
        final String callback = context.getRequest().getParameter("callback");
        final boolean isJSONP = callback != null;
        int noRows = QueueMappingPM.DEFAULT_DSG_COUNT;

        Collection results = null;
        Collection allMappedGroupIds = null;
        final Set mappedQueueGroupId = null;
        String dsId = null;
        try {
            String queryStr = null;
            final String[] queryParamVals = (String[]) paramMap.get("q");
            final String[] dsIdParamVals = (String[]) paramMap.get("dsID");
            final String dsLimitCount = context.getRequest().getParameter(QueueMappingPM.DSG_LIMIT_COUNT_HTML_NAME);

            if ((dsLimitCount != null) && (dsLimitCount.length() > 0)) {
                noRows = Integer.parseInt(dsLimitCount);
            }

            if ((queryParamVals != null) && (queryParamVals.length > 0)) {
                queryStr = queryParamVals[0];
            }

            if ((dsIdParamVals != null) && (dsIdParamVals.length > 0)) {
                dsId = dsIdParamVals[0];
            }

            results = GroupMH.getGroupsLinkedToDataSourceID(new ID(dsId), noRows, queryStr, "order by A.NAME");

            if ((results != null) && !results.isEmpty()) {
                allMappedGroupIds = QueueGroupMH.getLinkedGroupIDs(context);
            }
        } catch (final Exception ex) {
            m_log.error("", ex);
            context.getResponse().setStatus(500);
            return;
        }

        context.getResponse().setHeader("Cache-Control", "no-cache");
        context.getResponse().setHeader("Content-Type", "application/json");
        context.getResponse().setCharacterEncoding("UTF-8");
        final PrintWriter pw = context.getResponse().getWriter();
        pw.flush();

        try {
            if (isJSONP) {
                pw.write(callback + "(");
            }

            pw.write("{ \"results\": [");
            final Localizer localizer = context.getLocalizer();
            final Iterator<Group> resultIter = results.iterator();

            while (resultIter.hasNext()) {
                final Group result = resultIter.next();
                pw.write("{\"id\":\"");
                pw.write(result.getID().toString());
                pw.write("\", \"name\":");
                pw.write(JSUtil.createJSONStringLiteral(result.getName()));
                pw.write(", \"mapstatus\":");

                pw.write(JSUtil.createJSONStringLiteral(((allMappedGroupIds != null) && allMappedGroupIds.contains(result.getID())) ? "Mapped" : "&nbsp;"));
                pw.write("}");

                if (resultIter.hasNext()) {
                    pw.write(", ");

                }
            }

            pw.write("]}");
            if (isJSONP) {
                pw.write(");");
            }

        } catch (final Exception ex) {
            log.error(ex.getMessage());
            throw new BbmFinderException(ex);
        } finally {
            pw.close();
        }

    }
