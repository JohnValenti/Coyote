package com.verint.web.rm.rest.entity.timeoff;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.Map;

import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityProperties;
import com.verint.web.rest.entity.AbstractWFOEntity;

@XmlRootElement(name = "timeoffData")
public class TimeOffDataEntity extends AbstractWFOEntity {
	protected int employeeId;
	protected int debitType;
	protected boolean canWaitList;
	protected RequestableActivities requestableActivities;

	public int getEmployeeId() {
		return employeeId;
	}

	public void setEmployeeId(int employeeId) {
		this.employeeId = employeeId;
	}

	public int getDebitType() {
		return debitType;
	}

	public void setDebitType(int debitType) {
		this.debitType = debitType;
	}

	public boolean getCanWishlist() {
		return canWaitList;
	}

	public void setCanWishlist(boolean canWaitList) {
		this.canWaitList = canWaitList;
	}

	public RequestableActivities getRequestableActivities() {
		return requestableActivities;
	}

	public void setRequestableActivities(Collection<Activity> activities, Map<ID, ActivityProperties> activityPropertiesLookup) {
		List<RequestableActivityEntity> lstRequestableActivities = new ArrayList<RequestableActivityEntity>();
		for (Activity activity : activities) {
			RequestableActivityEntity rae = new RequestableActivityEntity(activity.getID().toInt(), activity.getName());
			if (activityPropertiesLookup.containsKey(activity.getID())) {
				ActivityProperties properties = activityPropertiesLookup.get(activity.getID());
				rae.setTimeoffWithFlex(properties.getTimeoffWithFlex());
				rae.setUseDailyTOPool(properties.isUseDailyTOPool());
				rae.setUseIntervalTOPool(properties.isUseIntervalTOPool());
				rae.setPaid(activity.isPaid());
			}
			lstRequestableActivities.add(rae);
		}
		this.requestableActivities = new RequestableActivities(lstRequestableActivities);
	}

	@XmlRootElement(name = "requestableActivity")
	private class RequestableActivityEntity {
		private int activityId;
		private String name;
		private boolean paid;
		private boolean useDailyTOPool;
		private boolean useIntervalTOPool;
		private int timeoffWithFlex;

		protected RequestableActivityEntity(int activityId, String name) {
			this.activityId = activityId;
			this.name = name;
		}

		public int getActivityId() {
			return this.activityId;
		}

		public void setActivityId(int activityId) {
			this.activityId = activityId;
		}

		public String getName() {
			return name;
		}

		public void setName(String name) {
			this.name = name;
		}

		public boolean isUseDailyTOPool() {
			return useDailyTOPool;
		}

		public void setUseDailyTOPool(boolean useDailyTOPool) {
			this.useDailyTOPool = useDailyTOPool;
		}

		public boolean isUseIntervalTOPool() {
			return useIntervalTOPool;
		}

		public void setUseIntervalTOPool(boolean useIntervalTOPool) {
			this.useIntervalTOPool = useIntervalTOPool;
		}

		public int getTimeoffWithFlex() {
			return timeoffWithFlex;
		}

		public void setTimeoffWithFlex(int timeoffWithFlex) {
			this.timeoffWithFlex = timeoffWithFlex;
		}

		public boolean isPaid() {
			return paid;
		}

		public void setPaid(boolean paid) {
			this.paid = paid;
		}
	}

	@XmlRootElement(name = "requestableActivities")
	protected class RequestableActivities extends ArrayList<TimeOffDataEntity.RequestableActivityEntity>{
		public RequestableActivities(List<TimeOffDataEntity.RequestableActivityEntity> wrapped) {
			super(wrapped);
		}
	}
}
