package com.verint.web.fs.widgets;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Formatter;
import java.util.SortedSet;
import java.util.TreeSet;

import javax.servlet.http.HttpServletRequest;

import com.bluepumpkin.common.datatypes.Duration;
import com.bluepumpkin.common.datatypes.TimeOfDay;
import com.bluepumpkin.ejb.bbm.vo.HOOAssignment;
import com.verint.web.fs.workrules.WorkRulesJSFileID;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.pagecomponent.CompositePC;
import com.witness.web.uif.system.RequestContext;

/**
 * NOTE: When you utilize this picker in your page, you will also need to include
 * the StartTimesPicker.css file in your screen definition.
 * 
 * @author rfunderburg
 */
public class StartTimesPickerPC extends CompositePC {
	private static final long serialVersionUID = 1L;

	private static final int cellIntervalMinutes = 15;
	private static final int expansionIntervalRows = 15;
	private static final boolean rowExpansionEnabled = false;
	public static final String STYLE_TIME_SELECTOR ="time-selector-table";

	private ArrayList<ToggleVisiblePC> togglers = new ArrayList<ToggleVisiblePC>();

	private TreeSet<TimeOfDay> selectedStartTimes = new TreeSet<TimeOfDay>();
	private TreeSet<TimeOfDay> addedStartTimes = new TreeSet<TimeOfDay>();
	private TreeSet<TimeOfDay> removedStartTimes = new TreeSet<TimeOfDay>();
	private TreeSet<TimeOfDay> multiSelectedStartTimes = new TreeSet<TimeOfDay>();
	private int earliestStartTime = 0;
	private int latestEndTime = 1440;
	private int shiftLength = 0;
	private int dayOffset = 0;
	private int timeZoneOffset = 0;
	private boolean canShiftAssignmentsCrossDayBoundary;
	/**
	 * This variable should be used to set the Display status of the Time Picker
	 * table. // * Based on the boolean value the style=
	 * "display:none/ display:"" "would be changed
	 */
	private boolean displayTable = true;
	public StartTimesPickerPC(RequestContext context, String name, Duration timeZoneOffset, Duration dayOffset, boolean canShiftAssignmentsCrossDayBoundary) {
		super(context);
		setName(name);
		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
		setJSMediator(WorkRulesJSFileID.START_TIMES_PICKER);
		this.dayOffset = dayOffset.getDurationInMinutes();
		this.timeZoneOffset = timeZoneOffset.getDurationInMinutes();
		this.canShiftAssignmentsCrossDayBoundary = canShiftAssignmentsCrossDayBoundary;
	}

	@Override
	public void initForDisplay() {
		if (rowExpansionEnabled) {
			// Set up togglers
			int rowCtr = 0;
			for (int mins = 0; mins < 60; mins += cellIntervalMinutes ) {
				boolean isExpansionRow = (rowCtr++ % expansionIntervalRows) == 0; 
				if (isExpansionRow) {
					int tNum = togglers.size();
					ToggleVisiblePC tvpc = new ToggleVisiblePC(m_context, getName() + "_toggler" + tNum, Collections.singleton("#" + getName() + " [toggleGroup=" + tNum + "]"));
					this.addChildComponent(tvpc);
					togglers.add(tvpc);
				}
			}
		}
		super.initForDisplay();
	}

	@Override
	public String getHtmlDisplay() {
		StringBuffer html = new StringBuffer();

		// Table Start
		html.append("<table id=\""+ getName()+"_table\" ");
		if(!this.displayTable){
			html.append(" style=\"display:none\"");
		}
		html.append("class=\""+StartTimesPickerPC.STYLE_TIME_SELECTOR +"\" onselectstart=\"return false;\" ondragstart=\"return false;\">");

		// Table Headers
		html.append("<thead><tr><th></th>");
		int colCtr = 0;
		for (int mins = dayOffset + timeZoneOffset; mins < 1440 + dayOffset
				+ timeZoneOffset; mins += 60)
			html.append("<th class=\"tableHeaderDarkLeft col-label\" col=\"")
					.append(colCtr++).append("\">").append(
							m_localizer.formatTimeInMins(mins)/*
															 * new
															 * Formatter().format
															 * ("%tI %tp",
															 * mins*60000L,
															 * mins*60000L)
															 */)
					.append("</th>");
		html.append("</tr></thead>");
		
		// Table Rows
		html.append("<tbody>");
		
		boolean is24Hr = (latestEndTime - earliestStartTime == 1440);
		
		int rowCtr = 0;
		for (int mins = 0; mins < 60; mins += cellIntervalMinutes) {	//for each row iteration
			Formatter formatter = new Formatter();
			boolean isExpansionRow = (rowCtr % expansionIntervalRows) == 0;
			isExpansionRow &= rowExpansionEnabled;
			/* html.append("<tr").append(isExpansionRow ? "" : " toggleGroup='"
			 + (rowCtr / expansionIntervalRows) +
			 "' style='display:none;'").append("><th class=\"tableHeaderDarkLeft row-label\" row=\"").append(rowCtr).append("\"").append(">");*/
			boolean onLastRow = false;
			if (mins + cellIntervalMinutes >= 60) {
				onLastRow = true;
			}
			html.append("<tr><th class=\"tableHeaderDarkLeft");
			html.append(" row-label\" row=\"").append(rowCtr).append("\"").append(">");
			if (isExpansionRow) {
				html.append(togglers.remove(0).getHtmlDisplay());
			}
			html.append(formatter.format(":%02d", mins)).append("</th>");
			colCtr = 0;
			for (int hourMins = dayOffset; hourMins < 1440 + dayOffset; hourMins += 60) {	//for each column iteration
				boolean selected = selectedStartTimes.contains(new TimeOfDay(
						hourMins + mins));
				boolean multiSelected = multiSelectedStartTimes
						.contains(new TimeOfDay(hourMins + mins));
				int start = hourMins + mins;
				int end = start + this.shiftLength;
				int latestEndTimeAfterStart = this.latestEndTime;
				while (latestEndTimeAfterStart < this.earliestStartTime)
					latestEndTimeAfterStart += 1440;
				boolean isInvalid;
				if (is24Hr && canShiftAssignmentsCrossDayBoundary) {
					isInvalid = false;
				} else {
					isInvalid = (start < this.earliestStartTime || end > latestEndTimeAfterStart)
							&& (start + 1440 < this.earliestStartTime || end + 1440 > latestEndTimeAfterStart);
				}

				String className = "";

				if (selected) {
					if (isInvalid) {
						className = "selected-invalid";
					} else {
						className = "selected";
					}
				} else if (multiSelected) {
					if (isInvalid) {
						className = "multiselected-invalid";
					} else {
						className = "multiselected";
					}
				} else if (isInvalid) {
					className = "unselected-invalid";
				} else {
					className = "unselected";
				}

				html.append("<td class=\"grid-cell");
				if (hourMins + 60 >= 1440 + dayOffset) {	//If we are on the last column, append a style for the right edge of table
					html.append(" rightmost-column");
				}
					
				if (onLastRow) {	//If we are on the last row, append a style for the bottom edge of table
					html.append(" bottom-row");
				}
				
				html.append(" " + className)
						.append("\" row=\"").append(rowCtr).append("\" col=\"")
						.append(colCtr).append("\" value=\"").append(
								mins + hourMins).append("\"").append(
								selected ? "selected=\"true\"" : "").append(
								"></td>");

				colCtr++;
			}
			html.append("</tr>");
			rowCtr++;
		}
		html.append("</tbody>");

		html.append("<input type='hidden' name='" + getName() + "' value=\""
				+ convertTreeSetToString(selectedStartTimes) + "\"></input>");
		html.append("<input type='hidden' name='" + getName()
				+ "_multiSelectedStartTimes' value=\""
				+ convertTreeSetToString(multiSelectedStartTimes)
				+ "\"></input>");
		html.append("<input type='hidden' name='" + getName()
				+ "_addedStartTimes' value=\""
				+ convertTreeSetToString(addedStartTimes) + "\"></input>");
		html.append("<input type='hidden' name='" + getName()
				+ "_removedStartTimes' value=\""
				+ convertTreeSetToString(removedStartTimes) + "\"></input>");
		html.append("<input type='hidden' name='" + getName()
				+ "_earliestStartTime' value=\"" + earliestStartTime
				+ "\"></input>");
		html.append("<input type='hidden' name='" + getName()
				+ "_latestEndTime' value=\"" + latestEndTime + "\"></input>");

		// Table End
		html.append("</table>");

		return html.toString();
	}

	public String getJavaScriptInitCode() {
		StringBuffer js = new StringBuffer(super.getJavaScriptInitCode());
		// Form Reset
		js.append("\nresetJSArray[resetJSArray.length] = \"").append(getName())
				.append(".reset();\"");
		return js.toString();
	}

	@Override
	public boolean extractParameters(HttpServletRequest request) {
		boolean success = super.extractParameters(request);
		String selectedStartTimesStr = getParameter(request, getName());
		String addedStartTimesStr = getParameter(request, getName()
				+ "_addedStartTimes");
		String removedStartTimesStr = getParameter(request, getName()
				+ "_removedStartTimes");

		if (selectedStartTimesStr != null)
			parseStartTimeString(selectedStartTimes, selectedStartTimesStr);
		if (addedStartTimesStr != null)
			parseStartTimeString(addedStartTimes, addedStartTimesStr);
		if (removedStartTimesStr != null)
			parseStartTimeString(removedStartTimes, removedStartTimesStr);

		return success;
	}

	private String convertTreeSetToString(Collection<TimeOfDay> startTimes) {
		String retVal = "";
		for (TimeOfDay time : startTimes) {
			if (retVal.length() > 0)
				retVal += ",";
			retVal += time.getMinutesSinceMindnight();
		}
		return retVal;
	}

	private void parseStartTimeString(Collection<TimeOfDay> startTimes,
			String startTimesString) throws NumberFormatException {
		if (startTimesString != null && startTimesString.length() > 0) {
			while (startTimesString.endsWith(","))
				startTimesString = startTimesString.substring(0,
						startTimesString.length() - 1);
			String[] times = startTimesString.split(",");
			for (String time : times) {
				if (time.length() > 0)
					startTimes.add(new TimeOfDay(Integer.parseInt(time)));
			}
		}
	}

	public SortedSet<TimeOfDay> getStartTimes() {
		return new TreeSet<TimeOfDay>(selectedStartTimes);
	}

	public void setStartTimes(Collection<TimeOfDay> startTimes) {
		selectedStartTimes = new TreeSet<TimeOfDay>(startTimes);
	}

	public SortedSet<TimeOfDay> getAddedStartTimes() {
		return new TreeSet<TimeOfDay>(addedStartTimes);
	}

	public SortedSet<TimeOfDay> getRemovedStartTimes() {
		return new TreeSet<TimeOfDay>(removedStartTimes);
	}

	public void setMultiSelectedStartTimes(Collection<TimeOfDay> startTimes) {
		multiSelectedStartTimes = new TreeSet<TimeOfDay>(startTimes);
	}

	// Set which cells represent invalid start times based on the hours of
	// operation
	// and the maximum shift length
	public void setInvalidStartTimes(HOOAssignment hoo, int shiftLength) {
		int earliestStartTime = 1440;
		int latestEndTime = 0;

		for (short iDay = 1; iDay <= 7; iDay++) {
			if (hoo.getDayOpen(iDay) > -1
					&& hoo.getDayOpen(iDay) < earliestStartTime) {
				earliestStartTime = hoo.getDayOpen(iDay);
			}
			if (hoo.getDayClose(iDay) > latestEndTime) {
				latestEndTime = hoo.getDayClose(iDay);
			}
		}
		this.earliestStartTime = earliestStartTime;
		this.latestEndTime = latestEndTime;
		this.shiftLength = shiftLength;
	}

	public int getShiftLength() {
		return shiftLength;
	}

	public void setShiftLength(int shiftLength) {
		this.shiftLength = shiftLength;
	}

	public int getEarliestStartTime() {
		return earliestStartTime;
	}

	public int getLatestEndTime() {
		return latestEndTime;
	}

	public void setEarliestStartTime(int earliestStartTime) {
		this.earliestStartTime = earliestStartTime;
	}

	public void setLatestEndTime(int latestEndTime) {
		this.latestEndTime = latestEndTime;
	}

	public void setDisplay(boolean isDisplayOn) {
		this.displayTable = isDisplayOn;
	}

}
