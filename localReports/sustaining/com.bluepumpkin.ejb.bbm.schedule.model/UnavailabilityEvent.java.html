import com.bluepumpkin.ejb.bbm.activity.model.Event;

public class UnavailabilityEvent extends CalendarEventAssignment {
	/**
	 * Returns event type (see enumeration values in Event class). The unique
	 * key that identifies the event object is ID plus event type.
	 */
	public int getEventType() {
		return Event.EVENT_TYPE_UNAVAILABILITY;
	}

	/**
	 * <B>isPaid</B>
	 *
	 * @return boolean which indicates whether or not this is a paid event
	 */
	public boolean getPaid() {
		return false;
	}

	/**
	 * Returns a clone of this calendar event assignment object that will be
	 * used when publishing this calendar event assignment object
	 */
	public Event cloneForPublishing() {
		return super.deepCopy(new UnavailabilityEvent());
	}
}
