package com.verint.ejb.rfs.reports.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public final class RfsBranchFieldInfo extends FieldInfo {

    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// field id constants
    public static final int BRANCH_ID = 0;
    public static final int BRANCH_PARENTID = 1;
    public static final int BRANCH_ID_NAME = 2;
    public static final int BRANCH_VIEW = 3;
    public static final int BRANCH_LABEL = 4;
    
    private static final int BRANCH_NUMOFFIELDS = 5;

    private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[BRANCH_NUMOFFIELDS];
    private static String m_strTableName = "BranchViewNodes";

    // static initialization
    static {
      m_arrFieldInfo[BRANCH_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
      m_arrFieldInfo[BRANCH_PARENTID] = new FieldInfoEntry("PARENT", PrimitiveTypes.ID);
      m_arrFieldInfo[BRANCH_ID_NAME] = new FieldInfoEntry("BRANCH", PrimitiveTypes.STRING);
      m_arrFieldInfo[BRANCH_VIEW] = new FieldInfoEntry("BRANCHVIEW", PrimitiveTypes.STRING);      
      m_arrFieldInfo[BRANCH_LABEL] = new FieldInfoEntry("LABEL", PrimitiveTypes.STRING);      
    }

    public RfsBranchFieldInfo() {
        super();
        setNotifiable(true);
    }

    // implement abstract methods in parent class
    public String getTableName() { return m_strTableName; }
    public int getIDFieldIndex() { return BRANCH_ID; }
    public int getParentIDFieldIndex() { return BRANCH_PARENTID; }
    public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }
}
