package com.verint.test.utils;

import com.bluepumpkin.common.jdmo.Jdmo;

/**
 * DBUtils class created to build database connections to the underlying Impact360 database model.
 */
public class DBUtils {
	/**
	 * Retrieves the DB host from the VM arguments if given any.
	 * @return DB Host. If no VM argument is given for dbHost, then the host argument will be passed instead
	 */
	public static String getDBHost() {
		if (System.getProperty("dbHost") == null) {
			return System.getProperty("host");
		} else {
			return System.getProperty("dbHost");
		}
	}

	/**
	 * Creates a jdmo object to create queries.
	 * @param hostName Database host name
	 * @param dbName Name of database to query from
	 * @param dbUser User to access database
	 * @param dbPass Password for user to access database
	 * @return
	 */
	public static Jdmo createJdmo(String hostName, String dbName, String dbUser, String dbPass) {
		String driverName = net.sourceforge.jtds.jdbc.Driver.class.getName();
		String driverUrl = "jdbc:jtds:sqlserver://" + hostName + ":" + 1433 + "/" + dbName;
		// For Windows Authentication to test fixture
		driverUrl += ";domain=" + hostName;
		driverUrl += ";useNTLMv2=true";
		driverUrl += ";appName=WfmReportingIntegrationTest";
		return new Jdmo(driverName, driverUrl, dbUser, dbPass);
	}

	/**
	 * Retrieves the name of the database to query
	 * @return Name of database
	 */
	public static String getDBName() {
		if (System.getProperty("dbName") == null) {
			return "BPMAINDB";
		} else {
			return System.getProperty("dbName");
		}
	}

	/**
	 * Retrieves the user for the associated database
	 * @return User for the database
	 */
	public static String getDBUser() {
		if (System.getProperty("dbUser") == null) {
			return "systemadmin";
		} else {
			return System.getProperty("dbUser");
		}
	}

	/**
	 * Retrieves the password for the Database
	 * @return The password for the user to access the database
	 */
	public static String getDBPass() {
		if (System.getProperty("dbPass") == null) {
			return "!veRinT$5";
		} else {
			return System.getProperty("dbPass");
		}
	}
}
