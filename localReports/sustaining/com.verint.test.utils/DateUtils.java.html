	/**
	 * Convert provided time String like 8:00 pm to minutes since midnight.
	 * For example: 2:30 am equals to 150 minutes.
	 *
	 * @param sTime String time like 8:00 pm or 2:00 am
	 * @return minutes since midnight.
	 */
	public static Integer timeToMinutes12HoursStr(String sTime) {
		LocalTime time = LocalTime.parse(sTime, DateTimeFormatter.ofPattern("h:mm a"));
		return time.getHour() * 60 + time.getMinute();
	}

	/**
	 * Convert provided time String like 8:00 to minutes since midnight.
	 * For example: 12:30 equals to 750 minutes.
	 *
	 * @param sTime String time like 8:00 or 12:00
	 * @return minutes since midnight.
	 */
	public static Integer timeToMinutes24HoursStr(String sTime) {
		LocalTime time = LocalTime.parse(sTime, DateTimeFormatter.ofPattern("H:mm"));
		return time.getHour() * 60 + time.getMinute();
	}

	/**
	 * Convert provided time in minutes to String format (hh:mm).
	 * For example: 480 minutes converted to 8:00
	 *
	 * @param minutes which we need to convert.
	 * @return formatted time like 8:00 or 12:00
	 */
	public static String timeMinutesTo24HoursStr(int minutes) {
		int hours = minutes / 60; //since both are ints, you get an int
		int mins = minutes % 60;

		return String.format("%d:%02d", hours, mins);
	}

	public static Integer secondsToMinutes(Integer seconds){

	/**
	 * Convert provided time in minutes to String format (hh:mm AM/PM).
	 * For example: 480 minutes converted to 8:00 AM
	 * 780 minutes converted to 1:00 PM
	 *
	 * @param minutes which we need to convert.
	 * @return formatted time like 8:00 or 12:00
	 */
	public static String timeMinutesTo12HoursStr(int minutes) {
		int hours = minutes / 60; //since both are ints, you get an int
		int mins = minutes % 60;

		if (hours < 12) {
			return String.format("%d:%02d AM", hours, mins);
		} else if (hours == 12) {
			return String.format("%d:%02d PM", hours, mins);
		} else {
			hours -= 12;
			return String.format("%d:%02d PM", hours, mins);
		}
	}

	/**
	 * @param startTimesStr
	 * @return
	 */
	public static List<Integer> parseStartTimesStr(String startTimesStr) {

		List<Integer> startsInMinsList = new ArrayList<>();

		//in case of empty time string return empty list
		if (StringUtils.isBlank(startTimesStr)) {
			return startsInMinsList;
		}

		String[] startTimesList = startTimesStr.split(",\\s");
		final int minTime = 15;


		for (int i = 0; i < startTimesList.length; i++) {
			String innerStartTimes[] = startTimesList[i].split("\\s-\\s");

			if (innerStartTimes.length > 1) {
				int difference = (DateUtils.timeToMinutes12HoursStr(innerStartTimes[innerStartTimes.length - 1]) -
						DateUtils.timeToMinutes12HoursStr(innerStartTimes[0])) / 15;
				for (int j = 0; j < difference + 1; j++) {
					Integer startTime = DateUtils.timeToMinutes12HoursStr(innerStartTimes[0]) + minTime * j;
					startsInMinsList.add(startTime);
				}
			} else {
				Integer startTime = DateUtils.timeToMinutes12HoursStr(innerStartTimes[0]);
				startsInMinsList.add(startTime);
			}
		}

		return startsInMinsList;
	}

	/**
	 * Parse representation of date in string object to Date object.
	 *
	 * @param value string representation of the date.
	 * @param format string format for the date
	 * @param timeZone which time zone to use.
	 * @return
	 */
	public static Date stringToDate(String value, String format, TimeZone timeZone) {
		SimpleDateFormat formatter = new SimpleDateFormat(format);
		formatter.setTimeZone(timeZone);
		try {
			return formatter.parse(value);
		} catch (ParseException e) {
			throw new RuntimeException(e);
		}
	}

	private static Date addToDate(Date date, int calendarField, int value, Calendar cal) {
		cal.setTime(date);
		cal.add(calendarField, value);
		return cal.getTime();
	}

	public static Date createDate(int year, int month, int day) {
		Calendar cal = Calendar.getInstance(DEFAULT_TIME_ZONE);
		cal.clear();
		cal.set(year, month - 1, day);
		return cal.getTime();
	}

	public static Date createDate(int year, int month, int day, TimeZone timeZone) {
		Calendar cal = Calendar.getInstance(timeZone);

	}

	public static Date createDateWithoutTime(Date date, TimeZone timeZone) {
		Calendar cal = Calendar.getInstance(timeZone);
		cal.setTime(date);
		int year = cal.get(Calendar.YEAR);
		int month = cal.get(Calendar.MONTH);
		int day = cal.get(Calendar.DAY_OF_MONTH);
		return createDate(year, month + 1, day, timeZone);
	}

	/**
	 * This method is used by com.verint.systemtest.utils.parser.workrules.TimeBanksHtmlPageParser
	 * to parse the time bank's target hours into an integer.
	 *
	 * @param sTime
	 * @return
	 */
	public static Integer timeToMinutes(String sTime) {
		String[] totalTime = sTime.split(":");
		if (totalTime.length != 2) {
			throw new RuntimeException("Text " + sTime + " could not be parsed");
		}
		Integer hours = Integer.valueOf(totalTime[0]);
		Integer minutes = Integer.valueOf(totalTime[1]);
		return getMinutes(hours, minutes);
	}

	/**
	 * Checks that date is a start day of week
	 *
	 * @param date           Date to check
	 * @param startDayOfWeek Start day of week
	 * @param timeZone       time zone for setting date object
	 * @return true if date is a start Day Of Week
	 */
	public static boolean isDateWeekStartDay(Date date, DayOfWeek startDayOfWeek, TimeZone timeZone) {
		Calendar c = Calendar.getInstance(timeZone);
		c.setTime(date);
		DayOfWeek dayOfWeek = DayOfWeek.fromCalendarConstant(c.get(Calendar.DAY_OF_WEEK));
		return startDayOfWeek == dayOfWeek;
	}

	/**
	 * This method returns a positive or negative time value in string 24h format
	 * @param minutes
	 * @return
	 */
	public static String minutesToStr(int minutes) {
		int hours = minutes / 60; //since both are ints, you get an int
		int mins;

		if (minutes < 0) {
			mins = -minutes % 60;
		} else {
			mins = minutes % 60;
		}

		return String.format("%d:%02d", hours, mins);
	}

	public static LocalDate generateAlignmentDate(DayOfWeek weekStartDay) {
		return generateAlignmentDate(weekStartDay, 0);
	}

	public static LocalDate generateAlignmentDate(DayOfWeek weekStartDay, int addDays) {
		Calendar c = Calendar.getInstance();
		c.set(Calendar.DAY_OF_WEEK, weekStartDay.getCalendarConstant());
		c.add(Calendar.DAY_OF_YEAR, addDays);
		return new LocalDate(c);
	}
