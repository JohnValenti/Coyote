import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.campaign.model.SPEmpTemplate;
import com.bluepumpkin.ejb.bbm.campaign.model.SPEmpTemplateFieldInfo;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

public class SPEmpTemplateDAO extends DAOBase<SPEmpTemplate> {

	private static FieldInfo m_fieldInfo = new SPEmpTemplateFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public SPEmpTemplateDAO() {
		super();
	}

	public SPEmpTemplateDAO(Jdmo dmo) {
		super(dmo);
	}

	protected SPEmpTemplate createValueObject() {
		return (new SPEmpTemplate());
	}

	protected boolean allowCreateWithId() {
		return true;
	}

	public void setIDs(SPEmpTemplate spET) throws BbmCreateException {
		try {
			String stmt = "SELECT s.ID SPID, e.ID ETID FROM SP s, EMPTEMPLATE e WHERE s.SID='" + spET.getSPID() + "'"
					+ " and e.SID = '" + spET.getEmpTemplateID() + "'";
			JdmoRowset rs = m_dmo.createRowset(stmt);
			if (rs.next()) {
				spET.setEmpTemplateID(new ID(rs.getID("ETID")));
				spET.setSPID(new ID(rs.getID("SPID")));
			}
		} catch (JdmoException ex) {
			throw new BbmCreateException(ex);
		}
	}

	public void setSIDs(Collection<SPEmpTemplate> spETs) throws BbmFinderException {
		// Need to fix this to do just one call to the database.
		try {
			for (SPEmpTemplate spET : spETs) {
				String stmt = "SELECT s.SID SPSID, e.SID ETSID FROM SP s, EMPTEMPLATE e WHERE s.ID='" + spET.getSPID() + "'"
						+ " and e.ID = '" + spET.getEmpTemplateID() + "'";
				JdmoRowset rs = m_dmo.createRowset(stmt);
				if (rs.next()) {
					spET.setEmpTemplateID(new ID(rs.getID("ETSID")));
					spET.setSPID(new ID(rs.getID("SPSID")));
				}
			}
		} catch (JdmoException ex) {
			throw new BbmFinderException(ex);
		}
	}

	public Collection<SPEmpTemplate> getSPEmpTemplatesBySP(ID spid) throws BbmFinderException {
		try {
			StringBuffer strWhere = new StringBuffer(30);
			strWhere.append("A.SPID = (SELECT ID FROM SP WHERE SID = ").append(spid).append(")");
			Collection<SPEmpTemplate> objs = getObjects(new String(strWhere));
			setSIDs(objs);
			return objs;
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
	}

}
