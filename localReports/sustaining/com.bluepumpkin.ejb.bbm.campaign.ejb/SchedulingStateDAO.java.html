package com.bluepumpkin.ejb.bbm.campaign.ejb;

import java.util.Collection;
import java.util.Date;
import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingState;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingStateFieldInfo;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.dao.DAONode;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectUtil;

public class SchedulingStateDAO extends DAONode {
	private static FieldInfo m_fieldInfo = new SchedulingStateFieldInfo();
	private static Category m_cat = Log.initCategory(SchedulingStateDAO.class
			.getName());   

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	@Override
	protected DAOBase createChildDAO(int iType) {
		if (iType == 0)
			return new SchedulingStateMessageDAO(m_dmo);
		else
			return new SchedulingWarningDAO(m_dmo);
	}

	@Override
	protected ValueObjectBase createValueObject() {
		return new SchedulingState();
	}

	@Override
	public void loadChildObjects(Collection arrParents) throws BbmFinderException {
		if (arrParents == null)
			return;
		SchedulingState state = null;
		DAOBase msgDAO = createChildDAO(SchedulingStateFieldInfo.CHILD_MESSAGES);
		DAOBase warningDAO = createChildDAO(SchedulingStateFieldInfo.CHILD_WARNINGS);
		Collection msgs = null;
		Collection warnings = null;
		Collection<ID> ids = null;
		for (Iterator i = arrParents.iterator(); i.hasNext();) {
			state = (SchedulingState) i.next();
			msgs = msgDAO.getObjects("SPID=" + state.getParentID() + "and MODE='" + state.getMode() + "'");
			// load messages and delete messages
			if (msgs != null) {
				for (Iterator iMsg = msgs.iterator(); iMsg.hasNext();)
					state.fillChildObject(SchedulingStateFieldInfo.CHILD_MESSAGES, (ValueObjectBase) iMsg.next());
			}
			// load warnings and delete warnings
			warnings = warningDAO.getObjects("SPID=" + state.getParentID() + "and MODE='" + state.getMode() + "'");
			if (warnings != null) {
				for (Iterator iWarning = warnings.iterator(); iWarning.hasNext();)
					state.fillChildObject(SchedulingStateFieldInfo.CHILD_WARNINGS, (ValueObjectBase) iWarning.next());
			}

			try {
				ids = ValueObjectUtil.getIDFromObjects(msgs);
				msgDAO.deleteObjects(ids);

				// ids = ValueObjectUtil.getIDFromObjects(warnings);
				// warningDAO.deleteObjects(ids); //GQ: Now, these are deleted
				// when the SchedulingWarningDialog is closed
			} catch (BbmRemoveException e) {
				throw new BbmFinderException(e);
			}
		}
	}

	/**
	 * Gets the last updated time for all the scheduling periods in-progress
	 * 
	 * @return
	 * @throws BbmFinderException
	 *             if there is exception of any kind while getting the data from
	 *             the db <bold>Note:</bold> Always cleanup the DAO after use to
	 *             prevent connection leak
	 */
	public Map<ID, Date> getTimeLastUpdatedBySPs() throws BbmFinderException {
		Map<ID, Date> spIdVsDate = new HashMap<ID, Date>();
		JdmoRowset rs = null;
		try {
			// select SPID, max(timelastupdated) as timelastupdated from
			// SPSCHEDSTATE where mode='S' group by spid
			StringBuffer pStmt = new StringBuffer();
			pStmt.append(" SELECT ");
			pStmt.append(m_fieldInfo
					.getFieldName(SchedulingStateFieldInfo.SPID));
			pStmt.append(" , ");
			pStmt.append("MAX(");
			pStmt.append(m_fieldInfo
					.getFieldName(SchedulingStateFieldInfo.TIMELASTUPDATED));
			pStmt.append(") AS ");
			pStmt.append(m_fieldInfo
					.getFieldName(SchedulingStateFieldInfo.TIMELASTUPDATED));
			pStmt.append(" FROM ");
			pStmt.append(m_fieldInfo.getTableName());
			pStmt.append(" GROUP BY ");
			pStmt.append(m_fieldInfo
					.getFieldName(SchedulingStateFieldInfo.SPID));

			rs = m_dmo.createRowset(pStmt.toString());
			while (rs.next()) {
				spIdVsDate.put(rs.getID(1), rs.getTimestamp(2));
			}
		} catch (Exception any) {
			throw new BbmFinderException(any);
		} finally {
			if (rs != null)
				try {
					rs.close();
				} catch (JdmoException e) {
					m_cat.warn("Error closing the ResultSet", e);
				}
		}
		return spIdVsDate;
	}

	/**
	 * Gets the last updated time for the given scheduling periods if it is
	 * present, Otherwise returns null. Expects sid of the sp, an int value
	 * 
	 * @return
	 * @throws BbmFinderException
	 *             if there is exception of any kind while getting data from the
	 *             db <bold>Note:</bold> Always cleanup the DAO after use to
	 *             prevent connection leak
	 */
	public Date getTimeLastUpdated(ID spID, String mode)
			throws BbmFinderException {
		JdmoRowset rs = null;
		try {
			// select max(timelastupdated) as timelastupdated from SPSCHEDSTATE
			// where spid=<spid> and mode='S'
			StringBuffer pStmt = new StringBuffer();
			pStmt.append(" SELECT ");
			pStmt.append("MAX(");
			pStmt.append(m_fieldInfo
					.getFieldName(SchedulingStateFieldInfo.TIMELASTUPDATED));
			pStmt.append(") AS ");
			pStmt.append(m_fieldInfo
					.getFieldName(SchedulingStateFieldInfo.TIMELASTUPDATED));
			pStmt.append(" FROM ");
			pStmt.append(m_fieldInfo.getTableName());
			pStmt.append(" WHERE ");
			pStmt.append(m_fieldInfo
					.getFieldName(SchedulingStateFieldInfo.SPID));
			pStmt.append(" = ");
			JdmoUtil.asSqlLiteral(spID.toInt(), pStmt);
			pStmt.append(" AND ");
			pStmt.append(m_fieldInfo
					.getFieldName(SchedulingStateFieldInfo.MODE));
			pStmt.append("=");
			JdmoUtil.asSqlLiteral(mode, pStmt);
			
			rs = m_dmo.createRowset(pStmt.toString());
			if (rs.next()) {
				return rs.getTimestamp(1);
			}
		} catch (Exception any) {
			throw new BbmFinderException(any);
		} finally {
			if (rs != null)
				try {
					rs.close();
				} catch (JdmoException e) {
					m_cat.warn("Error closing the ResultSet", e);
				}
		}
		return null;
	}

	public void deleteWarnings(Collection<ID> ids) throws BbmFinderException {
		DAOBase warningDAO = createChildDAO(SchedulingStateFieldInfo.CHILD_WARNINGS);
		try {
			warningDAO.deleteObjects(ids);
		} catch (BbmRemoveException e) {
			throw new BbmFinderException(e);
		}
	}

}
