package com.verint.web.rest.wfm.manager;

import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.activity.ejb.ActivityManager;
import com.bluepumpkin.ejb.bbm.employeefilter.ejb.EmployeeFilter;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManager;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.verint.web.rest.wfm.resource.fterequirements.FteRequirementsManagerWrapper;
import com.verint.web.rest.wfm.resource.fterequirements.FteRequirementsManagerWrapperImpl;
import org.glassfish.hk2.utilities.binding.AbstractBinder;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.inject.Singleton;

public class EjbManagerBinder extends AbstractBinder {

	private final static Logger log = LoggerFactory.getLogger(EjbManagerBinder.class.getName());

	@Override
	protected void configure() {
		try {
			bind(WfmManagerFactory.getActivityManager()).to(ActivityManager.class);
			bind(WfmManagerFactory.getScheduleAccessManager()).to(ScheduleAccessManager.class);
			bind(BbmManagerFactory.getEmployeeFilter()).to(EmployeeFilter.class);
			bind(BbmManagerFactory.getWorkResourceManager()).to(WorkResourceManager.class);
			bind(FteRequirementsManagerWrapperImpl.class).to(FteRequirementsManagerWrapper.class).in(Singleton.class);
		} catch (BbmEJBCreateException e) {
			log.error("Exception occurred when instantiating EJBs in EjbManagerBinder: ", e);
		}
	}
}
