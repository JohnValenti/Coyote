package com.bluepumpkin.ejb.am.timecollection.util;

/**
 * Title:        Blue Pumpkin Software Activity Management
 * Description: Employee state
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author Vincent Yu
 * @version 1.0
 */

public class State implements java.io.Serializable {
  /**
   * name for storing state in the session manager object
   */
  public static final String SESSIONNAME = "am.tc.state";
  /**
   * ID for current activity
   */
  private int agentState = -1;
  /**
   * ID for current Time Entry Event
   */
  private int curTimeEntryEventId = -1;
  public State() {  }
  public State(int st, int id) { agentState = st; curTimeEntryEventId = id; }
  /**
   * accessor for AgentState variable
   */
  public int getAgentState() { return agentState; }
  /**
   * accessor for AgentState variable
   */
  public void setAgentState(int state) { agentState = state; }
  /**
   * accessor for CurTimeEntryEventId variable
   */
  public int getCurTimeEntryEventId() { return curTimeEntryEventId; }
  /**
   * accessor for CurTimeEntryEventId variable
   */
  public void setCurTimeEntryEventId(int id) { curTimeEntryEventId = id; }
  public String toString() { return "agentState: " + agentState + ", curTimeEntryEventId: " + curTimeEntryEventId; }
}
