	private Map<ID, Media> filterMediaByExcludedMediaTypes(Map<ID, Media> media)
	{
		m_excludeMediaRawParam = m_context.getParameter(PARAM_EXCLUDE_MEDIA);
		if (m_excludeMediaRawParam == null) return media;
		
		String[] mediaTypes = m_excludeMediaRawParam.split(",");
		List<ID> lookup = new ArrayList<ID>(mediaTypes.length);
		for(String mediaType : mediaTypes) { lookup.add(new ID(mediaType.trim())); }
		
		HashMap<ID, Media> filteredMedia = new HashMap<ID, Media>();
		for (ID key : media.keySet()) {
			if (!lookup.contains(key)) {
				filteredMedia.put(key, media.get(key));
			}
		}

		return filteredMedia;
	}
	
	protected void preLoadData() throws Exception 
	{
		//Needs to be synchronized because the work frame (Applet) and the queue  
		//selection frame (this class) both load at the same time.
		synchronized (m_context.getRequest().getSession())
		{	
			// Retrieve the defined media types.
			Map<ID, Media> mediaMap = WorkloadModelHandler.getMediaMap(m_context);
			mediaMap = filterMediaByExcludedMediaTypes(mediaMap);
			
			m_isDistributedCampaignEnabled = Boolean.parseBoolean(m_context.getParameter(PARAM_IS_DISTRIBUTED_CAMPAIGN_ENABLED));
			
			Collection<Media> mediaList = SortUtil.sortBasicVO(mediaMap.values(), m_localizer);
			m_SpQueueAttribFilterBoxPC.init(mediaList, m_isDistributedCampaignEnabled, m_isDistributedCampaignOnly);
			
			// Get the filter values.
			m_campaignID = m_SpQueueAttribFilterBoxPC.getCampaignID();
			ID mediaID = m_SpQueueAttribFilterBoxPC.getMediaID();
			m_schedulingPeriodID = m_SpQueueAttribFilterBoxPC.getSchedulingPeriodID();
			
			ID queueAttribID = m_SpQueueAttribFilterBoxPC.getQueueAttributeID();
			
			if(mediaID != null) {
				Media media = (Media)mediaMap.get(mediaID);
				mediaMap = new HashMap<ID, Media>();
				mediaMap.put(mediaID, media);
			}
			
			// Configure the selection page component based on the filter.
			SPQueueSelectionPC spQueuePC = (SPQueueSelectionPC)m_treePC;
			spQueuePC.setMultiRowSelectable(false);
			spQueuePC.setIsComplexIDs(true);
			//spQueuePC.setIsHideProcessQueues(true);
			
			WorkloadManager wlm = WfmManagerFactory.getWorkloadManager(m_context.isInWhatIfMode());
			Collection<ID> rootQueues = java.util.Collections.emptyList();
			if(m_schedulingPeriodID != null) {
				rootQueues = wlm.getRootLinkQueueIDs(m_schedulingPeriodID);
			}

			spQueuePC.bind(m_campaignID, m_schedulingPeriodID, queueAttribID, mediaMap, rootQueues);
			
			//NOTE: the below commented code causes a loop when the user hits "cancel" when prompted to save
			//if the page is dirty.  Other than that, this code will maintain the sync between the left and right frames.
			//Some work will be needed to prevent the looping issue if we want to include this code again.
//			if(isNewCampaignSelected(m_lastCampaignID, m_campaignID, m_isInfoSubmittedFromSelf) ||
//					isNewSchedulingPeriodSelected(m_lastSchedulingPeriodID, m_schedulingPeriodID, m_isInfoSubmittedFromSelf) ||
//					!isSelectionInScope(spQueuePC))

		} // synchronized
	}
