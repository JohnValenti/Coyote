package com.bluepumpkin.web.bbm.workload.queue.selection;

import com.bluepumpkin.web.bbm.workload.queue.selection.queueattribute.QueueAttributeQueueSelectionStrategy;

public class SPQueueSelectionStrategyFactory {
	
	public static final String FORECASTING_SELECTION_TYPE = "FORECASTING";
	public static final String SERVICE_GOALS_SELECTION_TYPE = "SERVICE_GOALS";
	public static final String FTE_REQUIREMENTS_SELECTION_TYPE = "FTE_REQUIREMENTS";
	public static final String FORECAST_ALLOCATION_SELECTION_TYPE = "FORECAST_ALLOCATION";
	public static final String CALENDAR_ALLOCATION_SELECTION_TYPE = "CALENDAR_ALLOCATION";
	public static final String QUEUEATTRIBUTE_SELECTION_TYPE = "QUEUEATTRIBUTE_SELECTION_TYPE"; 
	/**
	 * This factory returns an ISPQueueSelectionStrategy given the type specified in
	 * spQueueSelectionBehaviorType.  There are three known types: FORECASTING, SERVICE_GOALS,
	 * and FTE_REQUIREMENTS.
	 */
	public static ISPQueueSelectionStrategy getSPQueueSelectionStrategy(String spQueueSelectionBehaviorType) {
		if (FORECASTING_SELECTION_TYPE.equals(spQueueSelectionBehaviorType)) {
			return new ForecastingSPQueueSelectionStrategy();
		} else if (SERVICE_GOALS_SELECTION_TYPE.equals(spQueueSelectionBehaviorType)) {
			return new ServiceGoalsSPQueueSelectionStrategy();
		} else if (FTE_REQUIREMENTS_SELECTION_TYPE.equals(spQueueSelectionBehaviorType)) {
			return new FteRequirementsSPQueueSelectionStrategy();
		} else if (FORECAST_ALLOCATION_SELECTION_TYPE.equals(spQueueSelectionBehaviorType)) {
			return new ForecastAllocationSPQueueSelectionStrategy();
		} else if(QUEUEATTRIBUTE_SELECTION_TYPE.equals(spQueueSelectionBehaviorType)) {
			return new QueueAttributeQueueSelectionStrategy();
		} else if (CALENDAR_ALLOCATION_SELECTION_TYPE.equals(spQueueSelectionBehaviorType)) {
			return new CalendarAllocationSPQueueSelectionStrategy();
		} else
			throw new UnsupportedOperationException("SPQueueSelectionStrategyFactory: SPQueue selection type not recognized.");
	}
}
