package com.verint.wfm.ws.entityImport.schedulingPeriod;

import java.util.Collection;
import java.util.Iterator;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.dao.MapUtil.SystemType;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.verint.wfm.ws.entityImport.schedulingPeriod.model.SchedulingPeriodDTO;

class ConvertSchedulingPeriodDTOParameters {
	private final SystemType systemType;
	private final Collection<SchedulingPeriodServiceDelegate.SchedulingPeriodWithStuffWrapper> newSchedulingPeriodWrappers;
	private final Collection<SchedulingPeriodServiceDelegate.SchedulingPeriodWithStuffWrapper> oldSchedulingPeriodWrappers;
	private final List<SchedulingPeriodNotPersistedInfo> schedulingPeriodNotPersistedInfos;
	private final Map<ID, ID> queueExternalWfoIDMap;
	private final Map<ID, Campaign> wfoIDCampaignMap;
	private final Map<SchedulingPeriodServiceDelegate.SchedPeriodNaturalKey, SchedulingPeriod> naturalKeyOldSchedulingPeriodMap;
	private final Map<SchedulingPeriodServiceDelegate.SchedPeriodHOONaturalKey, CampaignHOO> naturalKeyOldSchedulingPeriodHooMap;
	private final Map<SchedulingPeriodServiceDelegate.SPQueueNaturalKey, SPQueue> naturalKeyOldSPQueueMap;
	private final Map<ID, Queue> wfoIDQueueMap;
	private final Collection<SchedulingPeriodServiceDelegate.SchedulingPeriodWithStuffWrapper> allSchedulingPeriodWrappers;
	private final Iterator<SchedulingPeriodDTO> schedulingPeriodDTOIterator;
	private final SchedulingPeriodDTO schedulingPeriodDTO;
	private final SchedulingPeriodNotPersistedInfo schedulingPeriodNotPersistedInfo;
	private final Map<String, ID> organizationIDsByNameMap;
	private final ID wfoCampaignSID;
	private SchedulingPeriodServiceDelegate.SchedulingPeriodWithStuffWrapper schedulingPeriodWithStuffWrapper;

	ConvertSchedulingPeriodDTOParameters(SystemType systemType,
			ID wfoCampaignSID,
			Iterator<SchedulingPeriodDTO> schedulingPeriodDTOIterator,
			SchedulingPeriodDTO schedulingPeriodDTO,
			SchedulingPeriodNotPersistedInfo schedulingPeriodNotPersistedInfo,
			ConvertSchedulingPeriodDTOCollectionParameters convertSchedulingPeriodDTOCollectionParameters,
			ConvertSchedulingPeriodDTOMapParameters convertSchedulingPeriodDTOMapParameters) {
		this.systemType = systemType;
		this.newSchedulingPeriodWrappers = convertSchedulingPeriodDTOCollectionParameters.getNewSchedulingPeriodWrappers();
		this.oldSchedulingPeriodWrappers = convertSchedulingPeriodDTOCollectionParameters.getOldSchedulingPeriodWrappers();
		this.schedulingPeriodNotPersistedInfos = convertSchedulingPeriodDTOCollectionParameters.getSchedulingPeriodNotPersistedInfos();
		this.queueExternalWfoIDMap = convertSchedulingPeriodDTOMapParameters.getQueueExternalWfoIDMap();
		this.wfoIDCampaignMap = convertSchedulingPeriodDTOMapParameters.getWfoIDCampaignMap();
		this.naturalKeyOldSchedulingPeriodMap = convertSchedulingPeriodDTOMapParameters.getNaturalKeyOldSchedulingPeriodMap();
		this.naturalKeyOldSchedulingPeriodHooMap = convertSchedulingPeriodDTOMapParameters.getNaturalKeyOldSchedulingPeriodHooMap();
		this.naturalKeyOldSPQueueMap = convertSchedulingPeriodDTOMapParameters.getNaturalKeyOldSPQueueMap();
		this.wfoIDQueueMap = convertSchedulingPeriodDTOMapParameters.getWfoIDQueueMap();
		this.allSchedulingPeriodWrappers = convertSchedulingPeriodDTOCollectionParameters.getAllSchedulingPeriodWrappers();
		this.schedulingPeriodDTOIterator = schedulingPeriodDTOIterator;
		this.schedulingPeriodDTO = schedulingPeriodDTO;
		this.schedulingPeriodNotPersistedInfo = schedulingPeriodNotPersistedInfo;
		this.organizationIDsByNameMap = convertSchedulingPeriodDTOMapParameters.getOrganizationIDsByNameMap();
		this.wfoCampaignSID = wfoCampaignSID;
	}

	SystemType getSystemType() {
		return systemType;
	}

	Collection<SchedulingPeriodServiceDelegate.SchedulingPeriodWithStuffWrapper> getNewSchedulingPeriodWrappers() {
		return newSchedulingPeriodWrappers;
	}

	Collection<SchedulingPeriodServiceDelegate.SchedulingPeriodWithStuffWrapper> getOldSchedulingPeriodWrappers() {
		return oldSchedulingPeriodWrappers;
	}

	List<SchedulingPeriodNotPersistedInfo> getSchedulingPeriodNotPersistedInfos() {
		return schedulingPeriodNotPersistedInfos;
	}

	Map<ID, ID> getQueueExternalWfoIDMap() {
		return queueExternalWfoIDMap;
	}

	Map<ID, Campaign> getWfoIDCampaignMap() {
		return wfoIDCampaignMap;
	}

	Map<SchedulingPeriodServiceDelegate.SchedPeriodNaturalKey, SchedulingPeriod> getNaturalKeyOldSchedulingPeriodMap() {
		return naturalKeyOldSchedulingPeriodMap;
	}

	Map<SchedulingPeriodServiceDelegate.SchedPeriodHOONaturalKey, CampaignHOO> getNaturalKeyOldSchedulingPeriodHooMap() {
		return naturalKeyOldSchedulingPeriodHooMap;
	}

	Map<SchedulingPeriodServiceDelegate.SPQueueNaturalKey, SPQueue> getNaturalKeyOldSPQueueMap() {
		return naturalKeyOldSPQueueMap;
	}

	Map<ID, Queue> getWfoIDQueueMap() {
		return wfoIDQueueMap;
	}

	Collection<SchedulingPeriodServiceDelegate.SchedulingPeriodWithStuffWrapper> getAllSchedulingPeriodWrappers() {
		return allSchedulingPeriodWrappers;
	}

	Iterator<SchedulingPeriodDTO> getSchedulingPeriodDTOIterator() {
		return schedulingPeriodDTOIterator;
	}

	SchedulingPeriodDTO getSchedulingPeriodDTO() {
		return schedulingPeriodDTO;
	}

	SchedulingPeriodNotPersistedInfo getSchedulingPeriodNotPersistedInfo() {
		return schedulingPeriodNotPersistedInfo;
	}

	Map<String, ID> getOrganizationIDsByNameMap() {
		return organizationIDsByNameMap;
	}

	ID getWfoCampaignSID() {
		return wfoCampaignSID;
	}

	Campaign getCampaign() {
		return getWfoIDCampaignMap().get(getWfoCampaignSID());
	}

	SchedulingPeriodServiceDelegate.SchedulingPeriodWithStuffWrapper getSchedulingPeriodWithStuffWrapper() {
		return schedulingPeriodWithStuffWrapper;
	}

	void setSchedulingPeriodWithStuffWrapper(SchedulingPeriodServiceDelegate.SchedulingPeriodWithStuffWrapper  schedulingPeriodWrapper) {
		schedulingPeriodWithStuffWrapper = schedulingPeriodWrapper;
	}
}
