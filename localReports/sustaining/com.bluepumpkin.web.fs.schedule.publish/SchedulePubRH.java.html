package com.bluepumpkin.web.fs.schedule.publish;

import com.bluepumpkin.web.core.base.RequestHandler;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;

/**
 * Title:       SchedulePubRH
 * Description: Schedule Publish Request Handler
 * Copyright:   Copyright (c) 2004
 * Company:     Blue Pumpkin Software, Inc
 *
 * @author David Su, dsu
 * @version 1.0
 *          Created on Aug 24, 2004, 9:21:55 AM
 */
public class SchedulePubRH extends RequestHandler {

	/**
	 * Process the request.
	 *
	 * @param context This is the request context object
	 */
	public void processRequest(RequestContext context) throws Exception {
		// Verify privilege
		if (!isAuthorized(context)) return;

		// Extract parameter
		SchedulePubPM model = new SchedulePubPM(context);
		model.extractParameters(context.getRequest());

		if (isAction(context, PageAction.PUBLISH)) {
			model.validate();
			if (!model.hasError()) {
				try {
					if (model.isPublishPast()) {
						SchedulePubMH.publishSchedForDates(context, model.getFromDatePast(), model.getToDatePast(), SchedulePubMH.PAST_TYPE);
					}

					if (model.isPublishFuture()) {
						SchedulePubMH.publishSchedForDates(context, model.getFromDateFuture(), model.getToDateFuture(), SchedulePubMH.FUTURE_TYPE);
					}
					model.addPageMessage(Message.RESPONSE_TYPE, FsWebBundleKey.SCHEDULE_PUB_SUCC, FsWebBundleKey.BUNDLE_NAME);
				} catch(Exception ex) {
					model.handleUnableToUpdateInfo(m_log, ex);
				}
			}
		}

		context.setPageModel(model);
	}

		/**
	 * Verify that current user contain the specified privilege
	 *
	 * Add default error message if not authorized
	 */
	protected boolean isAuthorized(RequestContext context) {
		boolean isAuthorized = SchedulePubPM.isAuthorized(context.getUser());

		// Create Error Message
		if (!isAuthorized) {
			Message errorMsg = new Message(Message.ERROR_TYPE,
					UIFWebBundleKey.NO_PRIVILEGE_ACCESS_DENIED,
					UIFWebBundleKey.BUNDLE_NAME);
			context.addPageMessage(errorMsg);
		}

		return isAuthorized;
	}
}
