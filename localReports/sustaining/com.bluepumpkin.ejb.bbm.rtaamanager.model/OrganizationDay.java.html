package com.bluepumpkin.ejb.bbm.rtaamanager.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class OrganizationDay extends ValueObjectBase  implements Comparable{

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// meta data info.
	private static FieldInfo m_fieldInfo = new OrganizationDayFieldInfo();
	
	public OrganizationDay() {
		super();
	}
	
	public OrganizationDay(ID id, ID idValue, Date start, Date end, boolean isOrgDay) {
		setFieldValue(OrganizationDayFieldInfo.ORGANIZATIONDAY_ID, id);
		setFieldValue(OrganizationDayFieldInfo.ORGANIZATIONDAY_STARTTIME, start);
		setFieldValue(OrganizationDayFieldInfo.ORGANIZATIONDAY_ENDTIME, end);
		if (isOrgDay)
			setFieldValue(OrganizationDayFieldInfo.ORGANIZATIONDAY_ORGANIZATIONID, idValue);
		else
			setFieldValue(OrganizationDayFieldInfo.ORGANIZATIONDAY_SPID, idValue);
	}
	
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}
	
	public ID getOrganizationID() {
		return getFieldValueID(OrganizationDayFieldInfo.ORGANIZATIONDAY_ORGANIZATIONID);
	}
	
	public ID getSPID() {
		return getFieldValueID(OrganizationDayFieldInfo.ORGANIZATIONDAY_SPID);
	}
	
	public Date getStartTime() {
		return getFieldValueDate(OrganizationDayFieldInfo.ORGANIZATIONDAY_STARTTIME);
	}
	
	public Date getEndTime() {
		return getFieldValueDate(OrganizationDayFieldInfo.ORGANIZATIONDAY_ENDTIME);
	}
	
	public int compareTo(Object o)
    {
		OrganizationDay item2 = (OrganizationDay) o;
        
        long lDateDiff = getStartTime().getTime() - item2.getStartTime().getTime();
        
        if (lDateDiff < 0 ) return -1;
        if (lDateDiff > 0) return 1;
        return 0;        
    }

}
