package com.bluepumpkin.ejb.rm.setup.calendartimeoff.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class TOIntervalAllocationFieldInfo extends FieldInfo {

	private static final long serialVersionUID = 1L;

	//field index constants
	public static final int ID = 0;
	public static final int TOPOOLID = 1;
	public static final int STARTTIME = 2;
	public static final int ENDTIME = 3;
	public static final int ALLOCATEDHOURS = 4;
	public static final int NUMOFFIELDS = 5;

	private static String TABLE_NAME = "TOINTERVALALLOCATION";

	private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[NUMOFFIELDS];
	// static initialization
	static {
		m_arrFieldInfo[ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
		m_arrFieldInfo[TOPOOLID] = new FieldInfoEntry("TOPOOLID", PrimitiveTypes.ID);
		m_arrFieldInfo[STARTTIME] = new FieldInfoEntry("STARTTIME", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[ENDTIME] = new FieldInfoEntry("ENDTIME", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[ALLOCATEDHOURS] = new FieldInfoEntry("ALLOCATEDHOURS", PrimitiveTypes.FLOAT);
	}

	@Override
	public String getTableName() {
		return TOIntervalAllocationFieldInfo.TABLE_NAME;
	}

	@Override
	public int getIDFieldIndex() {
		return ID;
	}

	@Override
	public FieldInfoEntry[] getFieldInfo() {
		return m_arrFieldInfo;
	}

}
