package com.bluepumpkin.ejb.rm.setup.calendartimeoff.model;

import java.util.Date;
import java.util.TimeZone;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.ejb.bbm.vo.*;
import com.bluepumpkin.ejb.bbm.effectivity.*;
import com.bluepumpkin.ejb.bbm.workresource.model.WRTimeProperty;

/**
 * Title:        Blue Pumpkin Software Foundation
 * Description:  EmpTOPoolAssignment holds a record in WORKRESOURCETOPOOL table
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 *
 * @author lixin zhang
 * @version 1.0
 */
public final class EmpTOPoolAssignment extends ValueObjectEffectivity implements WRTimeProperty, TimeZonePeriod, Comparable {
	// meta data information
	private static FieldInfo m_fieldInfo = new EmpTOPoolAssignmentFieldInfo();

	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	private boolean m_bOverlap = false;

	// Only used for ActivityManager applications
	protected String m_strTimeZone;
	protected int m_intDayBoundaryOffset;
	protected String m_TOPoolName;

	public EmpTOPoolAssignment() {
		super();
	}


	public ID getTOPoolID() {
		return (ID) getFieldValue(EmpTOPoolAssignmentFieldInfo.EMPTOPOOL_TOPOOLID);
	}


	public ID getWorkResourceID() {
		return (ID) getFieldValue(EmpTOPoolAssignmentFieldInfo.EMPTOPOOL_WORKRESOURCEID);
	}


	public Date getStartTime() {
		return getFieldValueDate(EmpTOPoolAssignmentFieldInfo.EMPTOPOOL_STARTTIME);
	}


	public Date getEndTime() {
		return getFieldValueDate(EmpTOPoolAssignmentFieldInfo.EMPTOPOOL_ENDTIME);
	}

	public boolean isOverlap() {
		return m_bOverlap;
	}

	public void setTOPoolID(ID id) {
		setFieldValue(EmpTOPoolAssignmentFieldInfo.EMPTOPOOL_TOPOOLID, id);
	}

	public void setWorkResourceID(ID id) {
		setFieldValue(EmpTOPoolAssignmentFieldInfo.EMPTOPOOL_WORKRESOURCEID, id);
	}

	public void setStartTime(Date dtValue) {
		setFieldValue(EmpTOPoolAssignmentFieldInfo.EMPTOPOOL_STARTTIME, dtValue);
	}

	public void setEndTime(Date dtValue) {
		setFieldValue(EmpTOPoolAssignmentFieldInfo.EMPTOPOOL_ENDTIME, dtValue);
	}

	public void setOverlap(boolean bValue) {
		m_bOverlap = bValue;
	}

	public void setTimeZone(String timeZone) { m_strTimeZone = timeZone; }

	public String getTimeZone() { return m_strTimeZone; }

	public void setTOPoolName(String name) { m_TOPoolName = name; }

	public String getTOPoolName() { return m_TOPoolName; }

	public boolean overlap(EmpTOPoolAssignment item) {
		Date start_1 = getStartTime();
		Date end_1 = getEndTime();

		Date start_2 = item.getStartTime();
		Date end_2 = item.getEndTime();

		return (end_2 == null || start_1.getTime() < end_2.getTime()) &&
		        (end_1 == null || start_2.getTime() < end_1.getTime());
	}

	public int compareTo(Object obj) {
		EmpTOPoolAssignment item = (EmpTOPoolAssignment) obj;
		Date date1 = getStartTime();
		Date date2 = item.getStartTime();
		if (date1 == null || date2 == null)
			return 0;
		if (date1.before(date2))
			return -1;
		else if (date1.after(date2))
			return 1;
		else
			return 0;
	}

	public boolean effectivityValueEqual(Object obj) {
		EmpTOPoolAssignment item = (EmpTOPoolAssignment) obj;
		return getTOPoolID().equals(item.getTOPoolID());
	}

	//*********************** used to locate local date *************************
	private LocalDate m_dtStartLocalDate, m_dtEndLocalDate;
	private TimeZone m_dtStartTZ;

	public LocalDate getStartLocalDate() { return m_dtStartLocalDate; }

	public LocalDate getEndLocalDate() { return m_dtEndLocalDate; }

	public TimeZone getStartTimeZone() { return m_dtStartTZ; }

	public void convertTimePeriodToLocal(TimeZone tz) {
		m_dtStartLocalDate = new LocalDate(getStartTime(), tz);

		Date dtEnd = getEndTime();
		m_dtEndLocalDate = (dtEnd == null ? null : new LocalDate(dtEnd, tz));

		m_dtStartTZ = tz;
	}
	//***************************************************************************

	//********************** helper function to locate a date within TOPOOL assignments

	public static EmpTOPoolAssignment getRelatedTOPoolAssignment(Collection colTOPoolssignments, Date dt) {
		// convert other effective dates,
		Iterator it = colTOPoolssignments.iterator();

		while (it.hasNext()) {
			EmpTOPoolAssignment wka = (EmpTOPoolAssignment) it.next();

			Date dtStart = wka.getStartTime();
			Date dtEnd = wka.getEndTime();

			boolean bLast = !(it.hasNext());

			if ((dt.getTime() - dtStart.getTime()) >= 0 && (dtEnd == null || (dt.getTime() - dtEnd.getTime()) < (bLast ? 1 : 0)))
				return wka;
		}

		return null;
	}

	public static EmpTOPoolAssignment getRelatedTOPoolAssignment(Collection colTOPoolssignments, LocalDate dtLocal) {
		long tmLocalCompare = dtLocal.getTime().getTime();

		Iterator it = colTOPoolssignments.iterator();
		EmpTOPoolAssignment wka, wkaNext;
		long tmCompare, tmCompareNext;

		if (it.hasNext()) {
			wka = (EmpTOPoolAssignment) it.next();

			while (it.hasNext()) {
				wkaNext = (EmpTOPoolAssignment) it.next();
				tmCompare = wka.getStartLocalDate().getTime().getTime();
				tmCompareNext = wkaNext.getStartLocalDate().getTime().getTime();

				if (tmLocalCompare >= tmCompare && tmLocalCompare < tmCompareNext)
					return wka;

				wka = wkaNext;
			}

			tmCompare = wka.getStartLocalDate().getTime().getTime();
			if (tmLocalCompare < tmCompare) return null;
			if (wka.getEndTime() == null) return wka;

			tmCompareNext = wka.getEndLocalDate().getTime().getTime();
			if (tmLocalCompare >= tmCompare && tmLocalCompare < tmCompareNext) return wka;

		}

		return null;
	}

	public TimePeriod cloneTimePeriod() {
		EmpTOPoolAssignment item = new EmpTOPoolAssignment();

		item.setID(getID());
		item.setParentID(getParentID());
		item.setTOPoolID(getTOPoolID());
		item.setStartTime(getStartTime());
		item.setEndTime(getEndTime());

		return item;
	}
}
