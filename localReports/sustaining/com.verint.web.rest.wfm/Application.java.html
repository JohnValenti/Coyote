package com.verint.web.rest.wfm;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.verint.web.rest.wfm.manager.EjbManagerBinder;
import io.katharsis.repository.annotations.JsonApiRelationshipRepository;
import io.katharsis.repository.annotations.JsonApiResourceRepository;
import io.katharsis.rs.KatharsisProperties;
import org.glassfish.hk2.utilities.binding.AbstractBinder;
import org.glassfish.jersey.process.internal.RequestScoped;
import org.glassfish.jersey.server.ResourceConfig;
import org.reflections.Reflections;

import javax.inject.Singleton;
import javax.ws.rs.ApplicationPath;
import java.util.HashSet;
import java.util.Set;

@ApplicationPath("/")
public class Application extends ResourceConfig {

	public static final String APPLICATION_URL = "http://localhost:7001/wfm-api";

	public Application() {
		property(KatharsisProperties.RESOURCE_SEARCH_PACKAGE, "com.verint.web");
		property(KatharsisProperties.RESOURCE_DEFAULT_DOMAIN, APPLICATION_URL);
		register(APIAuthenticationFilter.class);
		register(KatharsisDynamicFeature.class);
		register(APIResponseFilter.class);

		register(new EjbManagerBinder());

		register(new AbstractBinder() {
			@Override
			public void configure() {
				bindFactory(ObjectMapperFactory.class).to(ObjectMapper.class).in(Singleton.class);

				bind(HK2InterceptionService.class).to(org.glassfish.hk2.api.InterceptionService.class).in(Singleton.class);

				// lookup and bind service types with Katharsis repository annotations
				Set<Class<?>> services = lookupServices();
				for (Class serviceClazz : services) {
					bindService(serviceClazz);
				}
			}

			private void bindService(Class<?> serviceType) {
				//TODO: should services be Singletons or RequestScoped?
				bind(serviceType).to(serviceType).proxy(true).in(RequestScoped.class);
				//bind(serviceType).to(serviceType).proxy(true).in(Singleton.class);
			}
		});
	}

	private Set<Class<?>> lookupServices() {
		Set<Class<?>> services = new HashSet<Class<?>>();
		Reflections ref = new Reflections(getProperty(KatharsisProperties.RESOURCE_SEARCH_PACKAGE));
		services.addAll(ref.getTypesAnnotatedWith(JsonApiResourceRepository.class));
		services.addAll(ref.getTypesAnnotatedWith(JsonApiRelationshipRepository.class));
		return services;
	}
}
