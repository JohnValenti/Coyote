package com.verint.web.rest.wfm;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.localization.DefaultLocalizationManager;
import com.bluepumpkin.ejb.core.CoreEJBCreateException;
import com.bluepumpkin.ejb.core.CoreManagerFactory;
import com.bluepumpkin.ejb.core.profile.ejb.UserProfileSync;
import com.verint.web.rest.util.SecurityUtils;
import com.witness.ejb.core.security.UserManager;
import com.witness.ejb.core.security.model.TokenInfo;
import com.witness.ejb.core.security.model.User;
import io.katharsis.errorhandling.ErrorData;
import io.katharsis.errorhandling.exception.KatharsisMappableException;
import io.katharsis.errorhandling.mapper.KatharsisExceptionMapper;
import io.katharsis.response.BaseResponseContext;
import io.katharsis.response.HttpStatus;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.annotation.Priority;
import javax.inject.Inject;
import javax.servlet.http.HttpServletRequest;
import javax.ws.rs.Priorities;
import javax.ws.rs.WebApplicationException;
import javax.ws.rs.container.ContainerRequestContext;
import javax.ws.rs.container.ContainerRequestFilter;
import javax.ws.rs.container.PreMatching;
import java.io.IOException;
import java.rmi.RemoteException;
import java.util.TimeZone;

@PreMatching
@Priority(Priorities.AUTHENTICATION)
public class APIAuthenticationFilter extends JsonApiErrorResponse implements ContainerRequestFilter {

	private static final Logger log = LoggerFactory.getLogger(APIAuthenticationFilter.class.getName());

	@Inject
	private javax.inject.Provider<org.glassfish.jersey.internal.util.collection.Ref<HttpServletRequest>> request;

	@Override
	public void filter(ContainerRequestContext requestContext) throws WebApplicationException {
		HttpServletRequest request = this.request.get().get();

		User user = (User)request.getSession().getAttribute(SessionConstants.WFM_API_USER_KEY);

		if (user == null) {
			user = setUserForRequest(request);
		}
		try {
			if (user == null) {
				throw new NotAuthorizedException("User not authenticated!");
			} else {
				Localizer localizer = getLocalizer(request, user);
				TimeZone timeZone = getTimeZone(request, user);

				requestContext.setProperty(SessionConstants.WFM_API_USER_KEY, user);
				requestContext.setProperty(SessionConstants.WFM_LOCALIZER_KEY, localizer);
				requestContext.setProperty(SessionConstants.WFM_USER_TIME_ZONE, timeZone);
			}
		} catch (KatharsisMappableException e) {
			BaseResponseContext responseContext = new KatharsisExceptionMapper().toErrorResponse(e);
			try {
				prepareResponse(requestContext, responseContext, null);
			} catch (IOException io) {
				throw new WebApplicationException(io);
			}
		} catch (RemoteException | CoreEJBCreateException e) {
			log.error("An exception occurred when trying to initialize User data: ", e);
			throw new WebApplicationException(e);
		}
	}

	private Localizer getLocalizer(HttpServletRequest request, User user) {
		Localizer localizer = (Localizer) request.getSession().getAttribute(SessionConstants.WFM_LOCALIZER_KEY);
		if (localizer == null) {
			localizer = DefaultLocalizationManager.getLocalizer(user.getID());
			request.getSession().setAttribute(SessionConstants.WFM_LOCALIZER_KEY, localizer);
		}
		return localizer;
	}

	private TimeZone getTimeZone(HttpServletRequest request, User user) throws CoreEJBCreateException, RemoteException {
		TimeZone timeZone = (TimeZone) request.getSession().getAttribute(SessionConstants.WFM_USER_TIME_ZONE);
		if (timeZone == null) {
			UserProfileSync userProfileMgr = CoreManagerFactory.getUserProfileManager();
			String tzString = userProfileMgr.load(user.getID(), "UserTimezone");
			if (!StringUtil.isEmpty(tzString)) {
				timeZone = TimeZone.getTimeZone(tzString);
			} else {
				log.error("Error retrieving timezone for user " + user.getUserName() + ". Setting default TimeZone.");
				timeZone = TimeZone.getDefault();
			}
			request.getSession().setAttribute(SessionConstants.WFM_USER_TIME_ZONE, timeZone);
		}
		return timeZone;
	}

	/**
	 * Get authToken from request header
	 */
	private static String getAuthToken(HttpServletRequest request) {
		return request.getHeader("Impact360AuthToken");
	}

	private static User setUserForRequest(HttpServletRequest request) {
		String userName = null;

		String authToken = getAuthToken(request);
		if (!StringUtil.isEmpty(authToken)) {
			// validate token, get username
			try {
				TokenInfo tokInfo = SecurityUtils.getUserManager().validateTokenV2(authToken, true);
				if (tokInfo == null) {
					return null;
				}
				userName = tokInfo.getUserName();
			} catch (Exception e) {
				log.error("Error while validating token " + authToken + ":", e);
			}
		}

		if (!StringUtil.isEmpty(userName)) {
			// make the user the "current" user is in the session.
			try {
				UserManager manager = CoreManagerFactory.getUserManager(false);
				User user = manager.getUserByName(userName);
				request.getSession().setAttribute(SessionConstants.WFM_API_USER_KEY, user);
				return user;
			} catch (Exception ex) {
				log.error("Error while looking up user by username " + userName + ":", ex);
			}
		}

		return (User)request.getSession().getAttribute("requestContextUser");
	}

	private class NotAuthorizedException extends KatharsisMappableException {
		private static final String TITLE = "Not Authenticated Error";

		NotAuthorizedException(String message) {
			super(HttpStatus.FORBIDDEN_403, ErrorData.builder()
					.setTitle(TITLE)
					.setDetail(message)
					.setStatus(String.valueOf(HttpStatus.FORBIDDEN_403))
					.build());
		}
	}
}
