package com.verint.wfm.ws.entityImport.forecast.model;

import java.text.DateFormat;
import java.text.SimpleDateFormat;
import java.util.Calendar;
import java.util.Date;
import java.util.TimeZone;

import javax.xml.bind.annotation.XmlElement;
import javax.xml.bind.annotation.XmlSeeAlso;
import javax.xml.bind.annotation.XmlType;
import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;

import com.bluepumpkin.common.util.StringUtil;
import com.verint.wfm.ws.ServiceApplicationException;
import com.verint.wfm.ws.entityImport.schedulingPeriod.SchedulingPeriodServiceDelegate;
import com.verint.wfm.ws.util.CalendarUtil;
import com.verint.wfm.ws.util.JaxbDateSerializer;

@XmlSeeAlso({ ProjectForecastDTO.class, TimeSeriesForecastDTO.class })
@XmlType(name = "AbstractForecastDTO", propOrder = { "forecastExternalNaturalKey", "description"})
public abstract class AbstractForecastDTO {
	private static final long serialVersionUID = 1L;

	protected ForecastExternalNaturalKey m_forecastExternalNaturalKey;
	protected String m_description;

	public AbstractForecastDTO() {

	}

	public ForecastExternalNaturalKey getForecastExternalNaturalKey() {
		return m_forecastExternalNaturalKey;
	}

	public void setForecastExternalNaturalKey(ForecastExternalNaturalKey forecastExternalNaturalKey) {
		m_forecastExternalNaturalKey = forecastExternalNaturalKey;
	}

	public String getDescription() {
		return m_description;
	}

	public void setDescription(String description) {
		m_description = description;
	}

	@XmlType(name = "ForecastExternalNaturalKey", propOrder = { "campaignExternalSourceId", "startDate",
			"queueExternalSourceId" })
	public static class ForecastExternalNaturalKey {
		private String m_campaignExternalSourceId;
		private Date m_startDate;
		private String m_queueExternalSourceId;

		public ForecastExternalNaturalKey() {
		}

		public ForecastExternalNaturalKey(String campaignExternalSourceId, Date startDate,
				String queueExternalSourceId) {

			m_campaignExternalSourceId = campaignExternalSourceId;
			m_startDate = startDate;
			m_queueExternalSourceId = queueExternalSourceId;
		}

		public String getCampaignExternalSourceId() {
			return m_campaignExternalSourceId;
		}

		public void setCampaignExternalSourceId(String campaignExternalSourceId) {
			m_campaignExternalSourceId = campaignExternalSourceId;
		}

		public Calendar startCalendarWithZone(TimeZone timeZone) throws ServiceApplicationException {
			return CalendarUtil.convertMidnightDateToCalendarWithZone(m_startDate, timeZone);
		}

		@XmlElement(type = Date.class)
		@XmlJavaTypeAdapter(value = JaxbDateSerializer.class)
		public Date getStartDate() {
			return m_startDate;
		}

		public void setStartDate(Date startDate) {
			m_startDate = startDate;
		}

		public String getQueueExternalSourceId() {
			return m_queueExternalSourceId;
		}

		public void setQueueExternalSourceId(String queueExternalSourceId) {
			m_queueExternalSourceId = queueExternalSourceId;
		}

		@Override
		public int hashCode() {
			// CAT.error("hash inputs: " + m_campaignExternalSourceId + ", " + m_startDate + ", "
			// + m_queueExternalSourceId);
			return m_campaignExternalSourceId.trim().hashCode() + m_startDate.hashCode()
					+ m_queueExternalSourceId.hashCode();
		}

		@Override
		public boolean equals(Object obj) {
			ForecastExternalNaturalKey k = (ForecastExternalNaturalKey) obj;
			// CAT.error("equals this? " + m_campaignExternalSourceId.trim() + ", " + m_startDate +
			// ", "
			// + m_queueExternalSourceId + "that? " + k.m_campaignExternalSourceId + ", " +
			// k.m_startDate + ", "
			// + k.m_queueExternalSourceId);
			// CAT.error("equals? " +
			// m_campaignExternalSourceId.trim().equals(k.m_campaignExternalSourceId.trim()) +
			// "&&"
			// + m_startDate.equals(k.m_startDate) + "&&"
			// + m_queueExternalSourceId.equals(k.m_queueExternalSourceId));
			return m_campaignExternalSourceId.trim().equals(k.m_campaignExternalSourceId.trim())
					&& m_startDate.equals(k.m_startDate)
					&& m_queueExternalSourceId.equals(k.m_queueExternalSourceId);
		}

		@Override
		public String toString() {
			StringBuilder b = new StringBuilder();
			if (!StringUtil.isEmptyOrWhiteSpace(this.getCampaignExternalSourceId())) {
				b.append(" Campaign External Id: ");
				b.append(this.getCampaignExternalSourceId());
			}
			if (!StringUtil.isEmptyOrWhiteSpace(this.getQueueExternalSourceId())) {
				b.append(" Queue External Id: ");
				b.append(this.getQueueExternalSourceId());
			}
			if (m_startDate != null) {
				b.append(" StartDate relative format: ");

				Calendar zonelessCalendar = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
				zonelessCalendar.setTime(m_startDate);

				DateFormat zonelessFormat = new SimpleDateFormat(SchedulingPeriodServiceDelegate.DATE_FORMAT);
				zonelessFormat.setTimeZone(TimeZone.getTimeZone("UTC"));

				b.append(zonelessFormat.format(zonelessCalendar.getTime()));
			}
			return b.toString();
		}
	}
}
