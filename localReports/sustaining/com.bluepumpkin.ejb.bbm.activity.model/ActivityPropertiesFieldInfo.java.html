package com.bluepumpkin.ejb.bbm.activity.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class ActivityPropertiesFieldInfo extends FieldInfo {

	/**
	 *
	 */
	private static final long serialVersionUID = 1L;
	// field id constants
	public static final int ACTIVITYPROPERTIES_ID = 0;
	public static final int USEDAILYTOPOOL = 1;
	public static final int USEINTERVALTOPOOL = 2;
	public static final int FLEXTIMEOFF = 3;

	   // field types
	private static final int ACTIVITYPROPERTIES_NUMOFFIELDS = 4;

	private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[ACTIVITYPROPERTIES_NUMOFFIELDS];
	private static String m_strTableName = "ACTIVITYPROPERTIES";


   // static initialization
	static {
	  m_arrFieldInfo[ACTIVITYPROPERTIES_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
	  m_arrFieldInfo[USEDAILYTOPOOL] = new FieldInfoEntry("USEDAILYTOPOOL", PrimitiveTypes.BOOLEAN);
	  m_arrFieldInfo[USEINTERVALTOPOOL] = new FieldInfoEntry("USEINTERVALTOPOOL", PrimitiveTypes.BOOLEAN);
	  m_arrFieldInfo[FLEXTIMEOFF] = new FieldInfoEntry("FLEXTIMEOFF", PrimitiveTypes.INT);
	}

	public ActivityPropertiesFieldInfo() {
		super();
	}

	// implement abstract methods in parent class
	public String getTableName() {
		return m_strTableName;
	}

	public int getIDFieldIndex() {
		return ACTIVITYPROPERTIES_ID;
	}

	public FieldInfoEntry [] getFieldInfo() {
		return m_arrFieldInfo;
	}
}
