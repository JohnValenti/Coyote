package com.bluepumpkin.ejb.bbm.activity.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**

 */
public class ActivitySPResourceConstraint extends ValueObjectBase {

	private static final long serialVersionUID = 1L;

	private static FieldInfo m_fieldInfo = new ActivitySPResourceConstraintFieldInfo();

	public ActivitySPResourceConstraint() {
		super();
	}

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ID getSpID() {
		return getFieldValueID(ActivitySPResourceConstraintFieldInfo.SPSID);
	}

	public void setSpID(ID val) {
		setFieldValue(ActivitySPResourceConstraintFieldInfo.SPSID, val);
	}

	public ID getActivityID() {
		return getFieldValueID(ActivitySPResourceConstraintFieldInfo.ACTIVITYID);
	}

	public void setActivityID(ID val) {
		setFieldValue(ActivitySPResourceConstraintFieldInfo.ACTIVITYID, val);
	}

	public int getMinChair() {
		return getFieldValueInt(ActivitySPResourceConstraintFieldInfo.MINCHAIR);
	}

	public void setMinChair(int val) {
		setFieldValue(ActivitySPResourceConstraintFieldInfo.MINCHAIR, val);
	}

	public int getMaxChair() {
		return getFieldValueInt(ActivitySPResourceConstraintFieldInfo.MAXCHAIR);
	}

	public void setMaxChair(int val) {
		setFieldValue(ActivitySPResourceConstraintFieldInfo.MAXCHAIR, val);
	}

	public void setModified(Date modified) {
		setFieldValue(ActivitySPResourceConstraintFieldInfo.MODIFIED, modified);
	}

	public Date getModified() {
		return getFieldValueDate(ActivitySPResourceConstraintFieldInfo.MODIFIED);
	}

	public void setModifiedBy(String strValue) {
		setFieldValue(ActivitySPResourceConstraintFieldInfo.MODIFIEDBY, strValue);
	}

	public String getModifiedBy() {
		return getFormattedFieldValue(ActivitySPResourceConstraintFieldInfo.MODIFIEDBY);
	}
}
