package com.bluepumpkin.ejb.am.timecollection.model;

import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

/*
 * Title:        Blue Pumpkin Software
 * Description:  eLearning Entry
 * Copyright:    Copyright (c) 2004
 * Company:      Blue Pumpkin Software, Inc
 * Created:      Sep 13, 2004
 * @author       SSong
 * @version		 1.0
 */
public class ELEntry extends RawTimeEntry {

	public ELEntry() {}
	private Date m_EndTime;
	/**
	 * Constructor for BackEnd
	 * @param m_id
	 * @param m_activityid
	 * @param m_employeeid
	 * @param sourceID
	 * @param m_start
	 * @deprecated
	 */
	public ELEntry(ID id, ID activityID, ID employeeID, ID dataSourceID, Date startDate) {
		super(id, employeeID, activityID, null, dataSourceID, 0, null, 0, startDate, startDate.getTime());
	}
	/**
	 * Constrcutor for FrontEnd
	 * @param m_activityid
	 * @param m_employeeid
	 * @param sourceID
	 * @param m_start
	 * @deprecated
	 */
	public ELEntry(ID activityID, ID employeeID, ID dataSourceID, Date startDate) {
		super(null, employeeID, activityID, null, dataSourceID, 0, null, 0, startDate, startDate.getTime());	
	}
	
	/**
	 * Constructor for BackEnd
	 * @param m_id
	 * @param m_activityid
	 * @param m_employeeid
	 * @param sourceID
	 * @param m_start
	 * @param m_end
	 */
	public ELEntry(ID id, ID activityID, ID employeeID, ID dataSourceID, Date startDate, Date endDate) {
		super(id, employeeID, activityID, null, dataSourceID, 0, null, 0, startDate, startDate.getTime());
		m_EndTime = endDate;
	}
	/**
	 * Constrcutor for FrontEnd
	 * @param m_activityid
	 * @param m_employeeid
	 * @param sourceID
	 * @param m_start
	 * @param m_end
	 */
	public ELEntry(ID activityID, ID employeeID, ID dataSourceID, Date startDate, Date endDate) {
		super(null, employeeID, activityID, null, dataSourceID, 0, null, 0, startDate, startDate.getTime());
		m_EndTime = endDate;
	}

	public Date getEndTime() {
		return m_EndTime==null?getStartTime():m_EndTime;
	}
	
	public void setEndTime(Date endTime) {
		m_EndTime = endTime;
	}
	
	/* (non-Javadoc)
	 * @see java.io.Externalizable#readExternal(java.io.ObjectInput)
	 */
	public void readExternal(ObjectInput in)
	throws IOException, ClassNotFoundException {
		super.readExternal(in);
		m_EndTime = (Date)in.readObject();
	}

	/* (non-Javadoc)
	 * @see java.io.Externalizable#writeExternal(java.io.ObjectOutput)
	 */
	public void writeExternal(ObjectOutput out) throws IOException {
		super.writeExternal(out);
		out.writeObject(m_EndTime);
	}
}
