package com.bluepumpkin.web.bbm.people.staffingprofiles;

import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;

import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;

public abstract class WorkRulesListPM extends WorkpaneListPM {

	ResourceBundle m_bundle;
	protected int m_paginationTotalIDSize = 0;
		
	protected boolean isByOrg;
	protected Organization selectedOrg = null;
	protected Campaign selectedCampaign = null;
	protected SchedulingPeriod selectedSchedulingPeriod = null;
	private String m_selectedCampaignID;
	private String m_selectedSchedulingPeriodID;
	
	protected boolean option_showTimeZoneToggle = true;
	
	protected TimeZone optimizationTargetTimeZone;
	
	public WorkRulesListPM(RequestContext context) {
		super(context, PageModel.FRAMED_MODEL);
		m_bundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);

		isByOrg = true;

		setIsMultiRowSelectable(true);
		setIsPaginatingSelectedIDs(false);
	}
	
	/**
	 * Load up selected campaign or org and set the timezone offset
	 */
	@Override
	protected void loadData() throws Exception {
		super.loadData();

		ID selectedID = getSelectedIDObject();
		if (selectedID != null) {
			if (isByOrg) {
				selectedOrg = OrganizationModelHandler.getOrganization(m_context, selectedID);
				optimizationTargetTimeZone = selectedOrg.getTimeZone();
			}
			else {
				if (getSelectedCampaignID() == null || getSelectedCampaignID().length() == 0) {
					selectedSchedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
							m_context, new ID(getSelectedSchedulingPeriodID()));
					selectedCampaign = CampaignModelHandler.getCampaign(m_context,
							selectedSchedulingPeriod.getCampaignID());
				} else {
					selectedCampaign = CampaignModelHandler.getCampaign(m_context, new ID(getSelectedCampaignID()));
				}
				optimizationTargetTimeZone = selectedCampaign.getTimeZone();
			}
		} 
	}

	/**
	 * Initialize Component For Display
	 */
	public void initForDisplay() {
		if (m_isInitializedForDisplay)
			return;
		else {
			if (isByOrg && selectedOrg == null) {
				addPageMessage(Message.INSTRUCTION_TYPE, m_localizer.i18n(
						BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.ORGANIZATION_GENERAL_SELECT_ORG_MSG));
			} /*else if (!isByOrg && selectedSchedulingPeriod == null) {
				addPageMessage(Message.INSTRUCTION_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.PLEASE_SELECT_A_SP));
			} */
			super.initForDisplay();
		}
	}
	
	protected void initToolbar() {
		//Enable pagination
		m_toolbar.setPagination("test", this, true, JSUtil.REFRESH_PAGE);
		
		boolean canAdd = isAddActionEnabled(); 
		if (isByOrg && ! m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES_ID, new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE))) {
			canAdd = false;
		} else if (!isByOrg && ! m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES_ID)) {
			canAdd = false;
		}
		ToolbarUtil.addAddBtn(m_toolbar, canAdd);
		ToolbarUtil.addEditBtn(m_toolbar, false);
		ToolbarUtil.addDeleteBtn(m_toolbar, false);
		m_toolbar.addSubmitTextButton(WorkRulesRH.WORK_RULES_COPY_PAGE_ACTION, m_toolbar.i18nCore(UIFWebBundleKey.TOOLBAR_COPY), false);
		//ToolbarUtil.addImportBtn(m_toolbar, false);
		//ToolbarUtil.addExportBtn(m_toolbar, false);
	}
	
	protected void initContentTitle() {
		super.initContentTitle();

		// Set Org or Campaign Name
		ID selectedID = getSelectedIDObject();
		if (selectedID != null) {
			if (isByOrg && selectedOrg != null) {
				m_contentTitlePC.setItemName(selectedOrg.getName());
			} else if ( ! isByOrg && selectedCampaign != null){
				m_contentTitlePC.setItemName(selectedCampaign.getName());
			}
		}
		
		// Add time zone toggle
		if (optimizationTargetTimeZone != null && option_showTimeZoneToggle) {
			m_contentTitlePC.addActiveComponent(new TimeZoneOverrideTogglePC(m_context, m_toolbar, optimizationTargetTimeZone, false));
		}
	}
	
	//Return a collection of DefaultMultiColumnNodeData objects that are to be displayed on the paginated list
	protected List<MultiColumnNodeData> getPagedRows(List<MultiColumnNodeData> allRows) 
	{
		if (this.getEditIndex() + this.getPageIDSize() > allRows.size()) {
			return allRows.subList(this.getEditIndex(), allRows.size());
		} else {
			return allRows.subList(this.getEditIndex(), this.getEditIndex() + this.getPageIDSize());
		}
	}
	
	protected void setPaginationTotalIDSize(int totalIDSize) {
		m_paginationTotalIDSize = totalIDSize;
	}
	
	public int getPaginationTotalIDSize() {
		return m_paginationTotalIDSize;
	}
	
	protected void initPagination(List<MultiColumnNodeData> allRows) {
		//TODO: disable second sort
		SortUtil.sort(allRows, m_list.getHeader().getSortColumnIndexes(), m_list.getHeader().isAscending(), m_localizer);

		//Pagination
		if (isPaginationNavPressed() == false) {
			setEditIndex(0);
		}
		setPaginationTotalIDSize(allRows.size());
		m_list.setListModel(getPagedRows(allRows));		
	}
	
	/**
	 * Gets timeZone offset in minutes to be added to any org or campaign based time for display
	 * @return
	 */
	public static int getTimeZoneOffset(RequestContext context, TimeZone optimizationTargetTimeZone) {
		return TimeZoneOverrideUtil.getTimeZoneOffset(context, optimizationTargetTimeZone, new Date());
	}
	
	private boolean hasEditPrivileges() {
		if (isByOrg) {
			return selectedOrg != null && m_context.getUser().isAuthorized(
					PrivilegeKeys.FS_EDITWORKRULES_ID, new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE));
		}
		else {
			return m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES_ID);
		}
	}
	
	private boolean isEditable(boolean hasEditPrivs, ID orgID) {
		if (isByOrg) {
			return hasEditPrivs && selectedOrg != null && selectedOrg.getID().equals(orgID);
		} else {
			return hasEditPrivs && orgID == null;
		}		
	}
	
	//Overriding this so that cached IDs are set properly
	public void setSelectedID(String ids) {
		super.setSelectedID(ids);
		if (!isByOrg) {
			setSelectedCampaignID(CampaignSchedulingPeriodSelectionUtil.extractIDs(
					getSelectedID(), CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX));
			setSelectedSchedulingPeriodID(CampaignSchedulingPeriodSelectionUtil.extractIDs(
					getSelectedID(), CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX));
		}
	}
	
	public String getSelectedCampaignID() { return m_selectedCampaignID; }	
	public void setSelectedCampaignID(String selectedCampaignIDs) { m_selectedCampaignID = selectedCampaignIDs; }
	
	public String getSelectedSchedulingPeriodID() { return m_selectedSchedulingPeriodID; }	
	public void setSelectedSchedulingPeriodID(String selectedSchedulingPeriodIDs) { m_selectedSchedulingPeriodID = selectedSchedulingPeriodIDs; }
	
	protected void setRowAttributes(MultiColumnNodeData rowData, ID orgID) {
		HashMap<String, String> nodeAttributes = rowData.getNodeAttributes();
		boolean hasEditPrivs = hasEditPrivileges();
		boolean isEditable = isEditable(hasEditPrivs, orgID);
		nodeAttributes.put(IS_EDITABLE, String.valueOf(isEditable));
		nodeAttributes.put(IS_DELETEABLE, String.valueOf(isEditable));
		toggleBtnOnNodeSelect(rowData, WorkRulesRH.WORK_RULES_COPY_PAGE_ACTION, hasEditPrivs);
		rowData.setNodeAttributes(nodeAttributes);
	}
	
	protected void setDetailRowAttributes(MultiColumnNodeData mcnd, MultiColumnNodeData mcndDetail, ID orgID) {
		HashMap<String, String> nodeAttributes = mcndDetail.getNodeAttributes();
		nodeAttributes.put(IS_EDITABLE, mcnd.getNodeAttributes().get(IS_EDITABLE));
		toggleBtnOnNodeSelect(mcndDetail, WorkRulesRH.WORK_RULES_COPY_PAGE_ACTION, hasEditPrivileges());
		nodeAttributes.put(IS_COPYABLE, mcnd.getNodeAttributes().get(IS_COPYABLE));
		nodeAttributes.put(IS_DELETEABLE, mcnd.getNodeAttributes().get(IS_DELETEABLE));
		nodeAttributes.put("isDetail", "true");
	}
}
