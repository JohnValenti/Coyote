package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.ejb.am.lsg.model.LSGAgeBucket;

public class CFLsgAgeBucket implements Serializable {

	private static final long serialVersionUID = -3545890591568898895L;
	public float proportion;
    public Date startTime;
    public Date endTime;

    public static CFLsgAgeBucket convertToCFLSGAgeBucket(LSGAgeBucket lsgAgeBucket) {
        if (lsgAgeBucket == null) {
            return null;
        }
        CFLsgAgeBucket cfLsgAgeBucket = new CFLsgAgeBucket();
        cfLsgAgeBucket.endTime = lsgAgeBucket.getEndTime();
        cfLsgAgeBucket.startTime = lsgAgeBucket.getStartTime();
        cfLsgAgeBucket.proportion = lsgAgeBucket.getLsgAgeBucketProportion().getProportion();
        return cfLsgAgeBucket;
    }

    public static CFLsgAgeBucket[] convertToCFLSGAgeBucket(Collection<LSGAgeBucket> lsgAgeBuckets) {
        if (lsgAgeBuckets == null) {
            return null;
        }
        CFLsgAgeBucket[] cfLsgAgeBuckets = new CFLsgAgeBucket[lsgAgeBuckets.size()];
        int index = 0;
        for (LSGAgeBucket lsgAgeBucket : lsgAgeBuckets) {
            cfLsgAgeBuckets[index] = convertToCFLSGAgeBucket(lsgAgeBucket);
            index++;
        }
        return cfLsgAgeBuckets;
    }
}
