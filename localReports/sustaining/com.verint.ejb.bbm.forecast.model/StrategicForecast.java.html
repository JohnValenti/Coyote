package com.verint.ejb.bbm.forecast.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

import java.util.Date;

public class StrategicForecast extends ValueObjectBase {

	/**
	 *
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new StrategicForecastFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}


	public ID getQueueID() {
		return getFieldValueID(StrategicForecastFieldInfo.Column.QUEUEID.getIndex());
	}

	public void setQueueID(ID queueId) {
		setFieldValue(StrategicForecastFieldInfo.Column.QUEUEID.getIndex(), queueId);
	}

	public double getCallVolume() {
		return getFieldValueDouble(StrategicForecastFieldInfo.Column.CALLVOLUME.getIndex());
	}

	public void setCallVolume(double callVolume) {
		setFieldValue(StrategicForecastFieldInfo.Column.CALLVOLUME.getIndex(), callVolume);
	}

	public double getAHT() {
		return getFieldValueDouble(StrategicForecastFieldInfo.Column.AHT.getIndex());
	}

	public void setAHT(double aht) {
		setFieldValue(StrategicForecastFieldInfo.Column.AHT.getIndex(), aht);
	}

	public Date getTime() {
		return getFieldValueDate(StrategicForecastFieldInfo.Column.TIME.getIndex());
	}

	public void setTime(Date time) {
		setFieldValue(StrategicForecastFieldInfo.Column.TIME.getIndex(), time);
	}
}
