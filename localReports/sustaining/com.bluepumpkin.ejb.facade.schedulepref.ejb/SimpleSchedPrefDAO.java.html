package com.bluepumpkin.ejb.facade.schedulepref.ejb;

import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.schedulepref.ejb.SchedulePrefManager;
import com.bluepumpkin.ejb.bbm.schedulepref.model.SimpleSchedulePreference;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectUtil;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManager;
import com.bluepumpkin.ejb.facade.schedulepref.model.SimpleSchedPrefFacade;
import com.verint.ejb.wfm.WfmManagerFactory;


public class SimpleSchedPrefDAO {

    public SimpleSchedPrefDAO() {
    }

    /**
     *   get simple schedule preference for a given employee
     *
     *   @param idEmployee employee id
     *   @param dtViewDate view date
     *
     *   @return SimpleSchedPrefFacade, return null if object not found
     *
     */
    public static SimpleSchedPrefFacade getSimpleSchedulePreference(ID idEmployee, Date dtViewDate) throws Exception
    {
	 // get managers
	 WorkResourceManager wrMgr = BbmManagerFactory.getWorkResourceManager();
	 SchedulePrefManager prefMgr = WfmManagerFactory.getSchedulePrefManager(false);
	 if (dtViewDate == null) dtViewDate = new Date();

	 // to convert effective date to local date
	 Collection colWRAssignments = wrMgr.getWRAssignmentsWithTimeZone(idEmployee);

	 Collection colPrefs = prefMgr.getSimpleSchedulePreferences(idEmployee, dtViewDate, dtViewDate);
	 if (colPrefs.isEmpty()) return null;

	 SimpleSchedulePreference item = (SimpleSchedulePreference) colPrefs.iterator().next();

	 SimpleSchedPrefFacade prefFacade = new SimpleSchedPrefFacade();
	 ValueObjectUtil.copy(prefFacade, item, 0);

	 // copy child objects: start time preference
	 prefFacade.setStartOffset(item.getStartOffset());
	 prefFacade.setEndOffset(item.getEndOffset());

	 // convert effectivity date
	 prefFacade.convertEffectivityToLocalDate(colWRAssignments);

	 return prefFacade;
    }


    // construct the simple schedule preference object from facade object
    private static SimpleSchedulePreference constructSimplePreferenceFromFacade(SimpleSchedPrefFacade prefFacade)
	 throws Exception
    {
	 // get managers
	 WorkResourceManager wrMgr = BbmManagerFactory.getWorkResourceManager();

	 // for convert effective date to local date
	 ID idEmployee = prefFacade.getEmployeeID();
	 Collection colWRAssignments = wrMgr.getWRAssignmentsWithTimeZone(idEmployee);

	 // convert effectivity date
	 prefFacade.convertEffectivityToGMTDate(colWRAssignments);

	 SimpleSchedulePreference item = new SimpleSchedulePreference();
	 ValueObjectUtil.copy(item, prefFacade, 0);

	 // copy child objects: start time preference
	 item.setStartOffset(prefFacade.getStartOffset());
	 item.setEndOffset(prefFacade.getEndOffset());

	 return item;
    }


    /**
     *   update simple schedule preference
     *
     */
    public static ID updateSimpleSchedulePreference(SimpleSchedPrefFacade prefFacade) throws Exception
    {
	 SimpleSchedulePreference item = constructSimplePreferenceFromFacade(prefFacade);

	 // create the object
	 SchedulePrefManager prefMgr = WfmManagerFactory.getSchedulePrefManager(false);
	 return prefMgr.updateSimpleSchedulePreference(item);
    }

}
