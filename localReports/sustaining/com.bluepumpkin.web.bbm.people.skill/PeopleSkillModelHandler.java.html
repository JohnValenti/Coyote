package com.bluepumpkin.web.bbm.people.skill;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Date;
import java.util.HashMap;

import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.web.core.base.ModelHandler;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        PeopleSkillModelHandler
 * Description:  Handle interaction with People Facade
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       Swati Tewari
 * @version 1.0
 */
public class PeopleSkillModelHandler extends ModelHandler {

	/**
	 * Return HashMap of employee IDs to Skill Assignments
	 *
	 * @param empIDs Collection of Employee IDs
	 * @param sDate  The Start Date
	 * @param eDate  The End Date
	 */
	public static HashMap getSkillAssignments(RequestContext context, Collection empIDs, Date sDate, Date eDate)
	throws BbmException, RemoteException {
		return WfmManagerFactory.getSkillManager(context.isInWhatIfMode()).getSkillAssignments(empIDs, sDate, eDate);
	}	
}
