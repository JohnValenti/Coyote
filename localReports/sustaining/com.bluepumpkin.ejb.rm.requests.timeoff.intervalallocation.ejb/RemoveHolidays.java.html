package com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.HashSet;
import java.util.List;
import java.util.Set;
import java.util.TimeZone;
import java.util.concurrent.TimeUnit;

import com.bluepumpkin.ejb.bbm.holiday.model.Holiday;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignmentFields;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftEventAssignment;
import com.bluepumpkin.ejb.bbm.schedule.util.ScheduleUtil;
import com.bluepumpkin.ejb.rm.util.interval.Interval;
import com.bluepumpkin.ejb.rm.util.interval.IntervalComparator;
import com.bluepumpkin.ejb.rm.util.interval.MergeIntervals;
import com.bluepumpkin.ejb.rm.util.interval.SingleTypeMergeIntervalOperations;

/**
 * Removes holidays from ShiftAssignments  
 *
 */
class RemoveHolidays {

	//Adapt ShiftAssignments and Holidays to Interval
	static class ShiftHoliday implements Interval {

		final long start;
		final long end;
		final ShiftAssignment shift;
		static final long ONE_MIN = TimeUnit.MINUTES.toMillis(1);


		public ShiftHoliday(long st, long et, ShiftAssignment shiftAssignment) {
			start = st;
			end = et;
			shift = shiftAssignment;
		}
		
		public static ShiftHoliday fromShift(ShiftAssignment s) {
			return new ShiftHoliday(s.getStartTime().getTime(), s.getEndTime().getTime(), s);
		}

		public static ShiftHoliday fromHoliday(Holiday h, TimeZone orgTimeZone) {
			long start = h.getStartDate().getTime(orgTimeZone).getTime();
			//make end exclusive by adding a minute. 
			long end = h.getEndDate().getTime(orgTimeZone).getTime() + ONE_MIN;
			return new ShiftHoliday(start, end, null);
		}


		boolean isHoliday() {
			return shift == null;
		}


		@Override
		public long start() {
			return start;
		}

		@Override
		public long end() {
			return end;
		}

		public ShiftAssignment clipWithDeepCopy() {
			ShiftAssignment clone = ScheduleUtil.deepCloneShiftAssigment(shift);
			return clip(clone, start, end);
		}

		public ShiftAssignment clipIfNeeded() {
			return clip(shift, start, end);
		}

		private static ShiftAssignment clip(ShiftAssignment shift, long start, long end) {

			if (shift.getStartTime().getTime() == start && shift.getEndTime().getTime() == end) {
				//no change
				return shift;
			}

			//Clip shift
			shift.setStartTime(new Date(start));
			long duration = end - start;
			shift.setDuration((int) TimeUnit.MILLISECONDS.toMinutes(duration));

			//Clip shift events
			List<ShiftEventAssignment> events = new ArrayList<ShiftEventAssignment>(shift.getChildren());
			shift.clearChildObjectMap(ShiftAssignmentFields.CHILD_SHIFT_EVENT);

			for (ShiftEventAssignment event : events) {

				if (event.getEndTime().getTime() < start || event.getStartTime().getTime() > end) {
					//the event is now outside the truncated shift
					continue;
				}

				long origEventStart = event.getStartTime().getTime();

				if (origEventStart < start) {
					event.setStartTime(new Date(start));
					duration = TimeUnit.MINUTES.toMillis(event.getDuration()) - (start - origEventStart);
					event.setDuration((int) TimeUnit.MILLISECONDS.toMinutes(duration));
				}

				if (event.getEndTime().getTime() > end) {
					duration = end - event.getStartTime().getTime();
					event.setDuration((int) TimeUnit.MILLISECONDS.toMinutes(duration));
				}

				shift.fillChildObject(ShiftAssignmentFields.CHILD_SHIFT_EVENT, event);
			}

			return shift;
		}

	}
	
	static class RemoveHolidayMerge extends SingleTypeMergeIntervalOperations<ShiftHoliday> {

		@Override
		public ShiftHoliday merge(ShiftHoliday interval1, ShiftHoliday interval2) {
			//if we are merging it means there is a holiday so return null to remove it
			return null;
		}

		@Override
		public ShiftHoliday copy(ShiftHoliday interval, Long start, Long end) {
			return new ShiftHoliday(start, end, interval.shift);
		}

	}

	static final RemoveHolidayMerge OPS = new RemoveHolidayMerge();

	private RemoveHolidays() {

	}

	/**
	 * Removes holidays from the give shift assignments.  
	 */
	public static List<ShiftAssignment> removeHolidays(Collection<ShiftAssignment> shiftAssignments, Collection<Holiday> holidays,
			TimeZone orgTimeZone) {
		List<ShiftHoliday> shiftIntervals = convertShifts(shiftAssignments);
		List<ShiftHoliday> holidayIntervals = convertHolidays(holidays, orgTimeZone);
		
		List<ShiftHoliday> merged = MergeIntervals.mergeIntervalCollections(shiftIntervals, holidayIntervals, OPS);

		return convertShiftHoliday(merged);
	}

	//convert from ShiftHolidays back to ShiftAssignments
	private static List<ShiftAssignment> convertShiftHoliday(List<ShiftHoliday> merged) {

		Set<Long> sharedShiftStartTime = getSharedShiftStartTime(merged);

		List<ShiftAssignment> shifts = new ArrayList<ShiftAssignment>(merged.size());

		for (ShiftHoliday sh : merged) {

			if (sh.isHoliday()) {
				continue;
			}

			ShiftAssignment shift = sh.shift;
			long shiftStart = shift.getStartTime().getTime();

			if (sharedShiftStartTime.contains(shiftStart)) {
				//sh.shift is shared among multiple segments. Do a deep copy and then clip
				shift = sh.clipWithDeepCopy();
			} else {
				shift = sh.clipIfNeeded();
			}
			shifts.add(shift);
		}

		return shifts;

	}

	private static Set<Long> getSharedShiftStartTime(List<ShiftHoliday> merged) {
		Set<Long> sharedShiftStartTime = new HashSet<Long>();
		int count = merged.size() - 1;
		for (int i = 0; i < count; i++) {
			ShiftHoliday item = merged.get(i);
			ShiftHoliday nextItem = merged.get(i + 1);
			if (item.shift == nextItem.shift && item.shift != null) {
				sharedShiftStartTime.add(item.shift.getStartTime().getTime());

			}
		}
		return sharedShiftStartTime;
	}

	private static List<ShiftHoliday> convertShifts(Collection<ShiftAssignment> shifts) {
		List<ShiftHoliday> result = new ArrayList<ShiftHoliday>(shifts.size());
		for (ShiftAssignment s : shifts) {
			result.add(ShiftHoliday.fromShift(s));
		}
		return result;
	}

	private static List<ShiftHoliday> convertHolidays(Collection<Holiday> holidays, TimeZone orgTimeZone) {
		List<ShiftHoliday> result = new ArrayList<ShiftHoliday>(holidays.size());
		for (Holiday h : holidays) {
			result.add(ShiftHoliday.fromHoliday(h, orgTimeZone));
		}

		Collections.sort(result, IntervalComparator.getStartEndAscending());
		return result;
	}

}
