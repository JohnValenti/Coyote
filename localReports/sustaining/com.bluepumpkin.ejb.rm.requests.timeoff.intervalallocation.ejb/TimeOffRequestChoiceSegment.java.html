package com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

public class TimeOffRequestChoiceSegment {

	final int toRequestID;
	final int toChoiceID;
	final Date segmentStart;
	final boolean passValidation;

	public TimeOffRequestChoiceSegment(int requestID, int choiceID, Date start, boolean pass) {
		this.toRequestID = requestID;
		this.toChoiceID = choiceID;
		this.segmentStart = start;
		this.passValidation = pass;
	}


	@Override
	public boolean equals(Object o) {
		if (o == this) {
			return true;
		}

		if (!(o instanceof TimeOffRequestChoiceSegment)) {
			return false;
		}

		TimeOffRequestChoiceSegment other = (TimeOffRequestChoiceSegment) o;
		return other.toRequestID == this.toRequestID
				&& other.toChoiceID == this.toChoiceID
				&& other.segmentStart.getTime() == this.segmentStart.getTime()
				&& other.passValidation == this.passValidation;

	}

	@Override
	public int hashCode() {
		int result = 17;
		result = 31 * result + toRequestID;
		result = 31 * result + toChoiceID;
		result = 31 * result + segmentStart.hashCode();
		return result;
	}
	
	public int getRequestID(){
		return toRequestID;
	}
	public int getChoiceID(){
		return toChoiceID;
	}
	public Date getStart(){
		return segmentStart;
	}
	public boolean isPassValidation(){
		return passValidation;
	}


}
