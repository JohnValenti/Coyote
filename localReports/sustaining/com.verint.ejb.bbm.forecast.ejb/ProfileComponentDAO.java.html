package com.verint.ejb.bbm.forecast.ejb;

import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.verint.ejb.bbm.forecast.model.ProfileComponent;
import com.verint.ejb.bbm.forecast.model.ProfileComponentFieldInfo;

public class ProfileComponentDAO extends DAOBase<ProfileComponent> {
	private static FieldInfo m_fieldInfo = new ProfileComponentFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	/**
	* default constructor
	*/
	public ProfileComponentDAO(){
		super();
	}
	
	public ProfileComponentDAO(Jdmo dmo) {
		super(dmo);
	}
	
	protected ProfileComponent createValueObject() {
		return (new ProfileComponent());
	}
	
	public Collection<ProfileComponent> getComponents(ID forecastProfileId) throws BbmFinderException {
		return getObjectsByParentID(forecastProfileId);
	}
}
