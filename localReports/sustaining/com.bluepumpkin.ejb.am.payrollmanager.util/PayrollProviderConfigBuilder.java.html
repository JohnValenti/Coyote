		DocumentBuilder parser = docBuilderFactory.newDocumentBuilder();

	private static String retrieve(Node node) {
		int type = node.getNodeType();
		switch (type) {
			case Node.ELEMENT_NODE: {
				NodeList children = node.getChildNodes();
				if (children != null) {
				int len = children.getLength();
				for (int i = 0; i < len; i++) {
					String result = retrieve(children.item(i));
					if (result != null) {
						return result;
					}
				}
				return null;
				}
			}
			case Node.TEXT_NODE: {
				String result = node.getNodeValue();
				return result.trim();
			}
		}
		return null;
	}

	public static List parse(String location, ID dsppID) throws Exception{
		DocumentBuilderFactory docBuilderFactory = DocumentBuilderFactory.newInstance();
		// Set Validation to false to work around XML parser issue
		docBuilderFactory.setValidating(true);
		DocumentBuilder parser = docBuilderFactory.newDocumentBuilder();
		parser.setErrorHandler(new PayrollProviderConfigBuilder());
		Document doc = parser.parse(location);
		Element root = doc.getDocumentElement();
//		root.normalize();
		String version = null;
		NodeList nl = doc.getElementsByTagName("FileVersion");
		if (nl.getLength() == 1)
			version = retrieve(nl.item(0));
		nl = doc.getElementsByTagName("PayrollCalculatedEarningField");
		ArrayList array = new ArrayList();
		HashMap map = new HashMap();
		if (nl.getLength() != 0) {
			for (int i = nl.getLength(); --i>=0; ) {
				PayrollExportConfig pec = buildCalculatedEarningType(nl.item(i));
				pec.setVersion(version);
				pec.setPayrollProviderDataSourceID(dsppID);
				String fName = pec.getFieldName();
				Integer counter = (Integer)map.get(fName);
				int counterNum = counter==null?0:counter.intValue();
				if (counter==null)
					map.put(fName, new Integer(counterNum));
				else {
					counterNum++;
					map.put(fName, new Integer(counterNum));
				}
				pec.setUniqueNum(counterNum);
				array.add(pec);
			}
		}
		nl = doc.getElementsByTagName("PayrollCalculatedField");
		if (nl.getLength() != 0) {
			for (int i = nl.getLength(); --i>=0; ) {
				PayrollExportConfig pec = buildCalculated(nl.item(i));
				pec.setVersion(version);
				pec.setPayrollProviderDataSourceID(dsppID);
				String fName = pec.getFieldName();
				Integer counter = (Integer)map.get(fName);
				int counterNum = counter==null?0:counter.intValue();
				if (counter==null)
					map.put(fName, new Integer(counterNum));
				else {
					counterNum++;
					map.put(fName, new Integer(counterNum));
				}
				pec.setUniqueNum(counterNum);
				array.add(pec);
			}
		}
		nl = doc.getElementsByTagName("PayrollUserField");
		if (nl.getLength() != 0) {
			for (int i = nl.getLength(); --i>=0; ) {
				PayrollExportConfig pec = buildUserDefined(nl.item(i));
				pec.setVersion(version);
				pec.setPayrollProviderDataSourceID(dsppID);
				String fName = pec.getFieldName();
				Integer counter = (Integer)map.get(fName);
				int counterNum = counter==null?0:counter.intValue();
				if (counter==null)
					map.put(fName, new Integer(counterNum));
				else {
					counterNum++;
					map.put(fName, new Integer(counterNum));
				}
				pec.setUniqueNum(counterNum);
				array.add(pec);
			}
		}
		return array;
	}
	private static PayrollExportConfig buildCalculatedEarningType(Node node) {
		//Get all the sub elements
		PayrollExportConfig pec = new PayrollExportConfig();
		pec.setFieldType(PayrollExportConfig.CALCULATEDEARNINGS);
		for (Node child = node.getFirstChild();
			 child != null;
			 child = child.getNextSibling())
		{
			switch (child.getNodeType()) {
				case Node.ELEMENT_NODE:
					if (child.getNodeName().equalsIgnoreCase("FieldName")) {
						pec.setFieldName(getStr(child));
					} else if (child.getNodeName().equalsIgnoreCase("FieldSN")) {
						pec.setSequence(Integer.parseInt(getStr(child)));
						pec.setIsMandatory(true);
					} else if (child.getNodeName().equalsIgnoreCase("DisplayName")) {
						pec.setDisplayName(getStr(child));
					} else
						break;
			}
		}
		return pec;
	}

	/**
	 * Get the Leaf level String
	 */
	private static String getStr(Node child) {
		Node leaf = child.getFirstChild();
		if (leaf == null)
			return null;
		if (leaf.getNodeType() == Node.TEXT_NODE)
			return leaf.getNodeValue();
		else
			return null;
	}
