package com.verint.ejb.bbm.schedule.ejb;

import javax.naming.Context;
import javax.naming.InitialContext;
import java.rmi.RemoteException;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.core.base.SessionEJBBase;
import com.bluepumpkin.ejb.core.CoreManagerFactory;
import com.verint.ejb.bbm.schedule.ejb.AutoScheduleRequestManager.SchedulingMode;
import com.verint.ejb.bbm.schedule.model.WSAutoScheduleRequest;
import com.verint.ejb.bbm.schedule.model.WSAutoScheduleRequestFieldInfo;
import com.verint.ejb.bbm.schedule.util.SchedulingRequestUtil;
import com.witness.ejb.core.base.CoreFinderException;
import com.witness.ejb.core.security.UserManager;
import com.witness.ejb.core.security.model.User;

/**
 * EJB Implementation for creating/updating WSAutoScheduleRequests
 */
@SuppressWarnings("serial")
public class AutoScheduleRequestManagerEJB extends SessionEJBBase {

	private static Category CAT = Log.initCategory(AutoScheduleRequestManagerEJB.class.getName());
	private UserManager userManager;
	private boolean isWhatIf = false;

	/**
	 * Perform one-time initialization of this EJB instance
	 */
	public void onEjbCreate() {
		try {
			// First query environment to get WIF setting from DD
			Context initialContext = new InitialContext();
			Boolean wif = (Boolean) initialContext.lookup("java:comp/env/WIF");
			if (wif != null) {
				isWhatIf = wif.booleanValue();
			}
			userManager = CoreManagerFactory.getUserManager(isWhatIf);
		} catch (Exception e) {
			handleException("onEjbCreate", e, false);
		}
	}

	public Map<ID, ID> createRequests(List<ID> spIds, SchedulingMode mode, String createdBy) throws BbmCreateException {
		Map<ID, ID> spIdVsrequestId = new HashMap<ID, ID>();
		String user = createdBy;
		if (user == null) {
			// get the superuser
			user = getSuperUser();
		}
		for (ID spID : spIds) {
			WSAutoScheduleRequest request = new WSAutoScheduleRequest();
			request.setFieldValue(WSAutoScheduleRequestFieldInfo.SPID, spID.toInt());
			request.setFieldValue(WSAutoScheduleRequestFieldInfo.MODE, mode.mode());
			request.setFieldValue(WSAutoScheduleRequestFieldInfo.SCHEDULESTATUS,
					WSAutoScheduleRequest.SCHEDULE_STATUS_NEW);
			spIdVsrequestId.put(spID, createSchedulingRequest(request, user));
		}
		return spIdVsrequestId;
	}

	public void updateSchedulingRequests(List<ID> requests, String status, String statusMessage)
			throws BbmUpdateException {
		try {
			SchedulingRequestUtil.getInstance().updateSchedulingRequests(requests, status, statusMessage);
		} catch (JdmoException e) {
			CAT.error("Error updating the status for Scheduling Requests", e);
			throw new BbmUpdateException(e);
		}
	}

	public WSAutoScheduleRequest getSchedulingRequest(ID requestID)
			throws BbmObjectNotFoundException, BbmFinderException {
		return SchedulingRequestUtil.getInstance().getSchedulingRequestWithCurrentStatus(requestID);
	}

	/**
	 * Persists the given WSAutoScheduleRequest in the database.
	 */
	public ID createSchedulingRequest(WSAutoScheduleRequest request, String modifiedBy) throws BbmCreateException {
		try {
			return SchedulingRequestUtil.getInstance().createSchedulingRequest(request, modifiedBy);
		} catch (JdmoException e) {
			CAT.error("Error inserting the scheduling request for the scheduling Period " + request.getSPID()
					+ " to scheduling queue", e);
			throw new BbmCreateException(e);
		}
	}

	private String getSuperUser() {
		try {
			User superuser = userManager.getUserByID(User.SUPER_USERID);
			return superuser.getUserName();
		} catch (CoreFinderException e) {
			CAT.warn("Could not find the super user");
		} catch (RemoteException e) {
			CAT.error("Error getting the super user");
		}
		return null;
	}

	/** override the base class to provide the appropriate logging category */
	protected Category getCategory() {
		return CAT;
	}

	{
		super.init(AutoScheduleRequestManagerEJB.class.getName());
	}
}
