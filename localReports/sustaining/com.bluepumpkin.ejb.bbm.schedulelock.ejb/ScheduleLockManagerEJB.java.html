package com.bluepumpkin.ejb.bbm.schedulelock.ejb;

import com.bluepumpkin.ejb.core.base.SessionEJBBase;

@SuppressWarnings("serial")
public class ScheduleLockManagerEJB extends SessionEJBBase {

	private static Category m_cat = Log.initCategory( ScheduleLockManagerEJB.class.getName());

	/** override the base class to provide the appropriate logging category */

	{
		return m_cat;
	}

	{
		 super.init(ScheduleLockManagerEJB.class.getName());
	}

	protected void onEjbCreate()
	{

	}


	/**
	 *  check whether able to modify an assignment to a campaign, currently used to
	 *  check whether able to modify campaign workresource assignment
	 *
	 *  @param idCampaign: campaign id
	 *  @param dtStart, dtEnd: the time period
	 *
	 *  raise MultiUserException if cann't modify
	 */
	 public void checkCampaignAssignmentCanBeModified(ID idCampaign, Date dtStart, Date dtEnd)
			throws BbmFinderException, MultiUserException
	 {
		 methodStart("checkCampaignAssignmentCanBeModified", idCampaign, dtStart, dtEnd);
		 ScheduleLockDAO dao = new ScheduleLockDAO();
		 try {
			//dao.cleanUpLock(getTimeoutMinutes());
			dao.checkCampaignAssignmentCanBeModified(idCampaign, dtStart, dtEnd);
		 }
		 catch(MultiUserException e)
		 {
			handleException(e, false);
			throw e;
		 }
		 catch(Exception e)
		 {
			handleException(e, false);
			throw new BbmFinderException(e);
		 }
		 finally
		 {
			dao.cleanUp();
			methodFinish();
		 }
	 }

	/**
	 *  check whether able to modify employee schedule
	 *
	 *  @param idLock the schedule lock current user holds
	 *  @param idEmployeeID: employee id
	 *  @param dtStart, dtEnd: the time period
	 *
	 *  raise MultiUserException if cann't modify
	 */
	 public void checkEmployeeScheduleCanBeModified(String sLock, ID employeeID, Date dtStart, Date dtEnd)
			throws BbmFinderException, MultiUserException
	{
		checkEmployeeScheduleCanBeModified(sLock,Arrays.asList(employeeID), dtStart, dtEnd);
	 }

	/**
	 *  check whether able to modify employee schedule
	 *  
	 *  Note: this doesn't apply when the employee is auto-scheduled via sp's linked organizations
	 *
	 *  @param idLock the schedule lock current user holds
	 *  @param colEmployeeIDs a collection of employee ids
	 *  @param dtStart, dtEnd: the time period
	 *
	 *  raise MultiUserException if cann't modify
	 */
	 public void checkEmployeeScheduleCanBeModified(String userName, Collection<ID> employeeIDs, Date startDate, Date endDate)
			throws BbmFinderException, MultiUserException
	 {
 		 methodStart("checkEmployeeScheduleCanBeModified", employeeIDs, startDate, endDate);
 		 
		 ScheduleLockDAO dao = new ScheduleLockDAO();
		 try {
			dao.checkEmployeeScheduleByLinkedEmployeesCanBeModified(userName, employeeIDs, startDate, endDate);
		 } catch(MultiUserException e) {
			 //QC96672 - in case of any error roll backed the whole transaction 
			handleException(e, true);
			throw e;
		 } catch(Exception e){
			 //QC96672 - in case of any error roll backed the whole transaction 
			handleException(e, true);
			throw new BbmFinderException(e); 
		 } finally {
			dao.cleanUp();
			methodFinish();
		 }
	 }

	public void checkIfEmployeeExistingInSP(ID spSID, Collection<ID> employeeIDs) throws BbmFinderException, MultiUserException
	{
		methodStart("checkIfEmployeeExistingInSP", spSID, employeeIDs);
		
		// Get a hold of scheduling period, to see if employees are linked directly or via linked organizations
		SchedulingPeriodDAO schedulingPeriodDAO = new SchedulingPeriodDAO();
		SchedulingPeriod schedulingPeriod = schedulingPeriodDAO.getSchedulingPeriodByID(spSID); 
		
		// Find out if one or more employees are not being used
		ScheduleLockDAO dao = new ScheduleLockDAO();
		try
		{
			if (schedulingPeriod.isUsingAllEmployees()){
				dao.checkIfEmployeesFoundInSPOrgs(spSID, employeeIDs);
			} else {;
				dao.checkIfEmployeesFoundInSPByLinkedEmployees(spSID, employeeIDs);				
			}
		} catch (MultiUserException e) {
			handleException(e, false);
			throw e;
		} catch (BbmFinderException e) {
			handleException(e, false);
			throw new BbmFinderException(e);
		} finally{
			dao.cleanUp();
			schedulingPeriodDAO.cleanUp();
			methodFinish();
		}
	}
}
