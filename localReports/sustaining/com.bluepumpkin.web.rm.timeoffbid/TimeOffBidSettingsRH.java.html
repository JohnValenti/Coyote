/**

public class TimeOffBidSettingsRH extends SettingRH {
	private static final Category LOGGER = Log.initCategory(TimeOffBidSettingsRH.class.getName());
	protected static final String BID_NAME_SORT_DIRECTION = "ASC";

	/**
	 * Process Request actions for the Time Off Bidding Employees List page
	 *
	 * @param context
	 *            The Request Context
	 * @throws java.rmi.RemoteException
	 */
	@Override
	public void processRequest(RequestContext context) throws Exception, WebException {
		if (isAction(context, RmPageAction.OPEN_TIMEOFF_BID)) {
			processBidStatus(context, true);
		} else if (isAction(context, RmPageAction.CLOSE_TIMEOFF_BID)) {
			processBidStatus(context, false);
		} else {
			super.processRequest(context);
			if (isAction(context, PageAction.TABLE_SORT)) {
				/*
				 * QC172593: Note: Should not use model.getSortColumn since it
				 * returns wrong result TimeOffBidSummaryPM model = new
				 * TimeOffBidSummaryPM(context); model.getSortColumn(); So, get
				 * exactly the value returned when submitting request
				 */
				String sortColumn = context.getRequest().getParameter(TableHeaderPC.SORT_COLUMN_FN);
				String sortOrder = context.getRequest().getParameter(TableHeaderPC.SORT_ORDER_FN);
				if (sortColumn.equals(BbmWebBundleKey.Name)) {
					context.setAttribute(RequestContext.SESSION_SCOPE, BID_NAME_SORT_DIRECTION, sortOrder);
				} else {
					context.removeAttribute(RequestContext.SESSION_SCOPE, BID_NAME_SORT_DIRECTION);
				}
			}
		}
	}

	/**
	 * Return New Instance of SettingPM
	 */
	@Override
	protected SettingPM newSettingPM(RequestContext context) {
		return new TimeOffBidSettingsPM(context);
	}

	@Override
	protected void processAdd(RequestContext context) {
		TimeOffBidSettingsPM model = new TimeOffBidSettingsPM(context);
		if (isAuthorized(context, PrivilegeKeys.FS_EDITTIMEOFFBIDDING)) {
			// Use same information
			model.extractParameters(context.getRequest());

			// Blank out some text fields
			model.setSelectedID(null);
			model.getTimeOffBid().setName("");
			model.getTimeOffBid().setMinValuePerChoice(0);
			model.getTimeOffBid().setMinUnitPerChoice(TimeOffBid.TIMEOFF_BID_WEEKS);
			model.getTimeOffBid().setMaxValuePerChoice(0);
			model.getTimeOffBid().setMaxUnitPerChoice(TimeOffBid.TIMEOFF_BID_WEEKS);
			model.getTimeOffBid().setMaxValueForApproval(0);
			model.getTimeOffBid().setMaxUnitForApproval(TimeOffBid.TIMEOFF_BID_WEEKS);
			model.getTimeOffBid().setIsApprovalMaxValueVariesByEmp(false);
			// Indicate it is in add mode
			model.setRefreshSelection(true);
			model.setIsAddMode(true);
			TimeOffBidSummaryPM summaryModel = new TimeOffBidSummaryPM(context);
			summaryModel.setSelectedID(null);
		}

		prepareForNextPage(context, model, FORM_PAGE);
	}

	/**
	 * Process Save Action
	 */
	@SuppressWarnings("unchecked")
	@Override
	protected void processSave(RequestContext context) {
		TimeOffBidSettingsPM model = new TimeOffBidSettingsPM(context);
		if (isAuthorized(context, PrivilegeKeys.FS_EDITTIMEOFFBIDDING)) {
			// Use same information
			model.extractParameters(context.getRequest());
			model.validate();

			if (!model.hasError()) {
				try {
					TimeOffBid timeOffBid = model.getTimeOffBid();
					timeOffBid.setCreated(new Date());
					// Get all parent orgs of selected org to validate unique
					// name in org branch
					Collection<ID> orgIds = OrganizationModelHandler.getSelfAndParentsOrganizationIDs(context, timeOffBid.getOrganizationId());
					if (model.isAddMode()) {
						timeOffBid.setStatus(TimeOffBid.STATUS_OPEN_FOR_BIDDING);
						ID newID = TimeOffBidModelHandler.createTimeOffBid(timeOffBid, orgIds);
						model.setSelectedID(newID.toString());
					} else {
						TimeOffBidModelHandler.updateTimeOffBid(timeOffBid, orgIds);

					}
					model.setRefreshSelection(true);
					model.setIsAddMode(false);
				} catch (BbmCreateDuplicateKeyException ex) {
					model.setHasError(true);
					model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_TIMEOFFBID_NAME_EXISTS, RmWebBundleKey.BUNDLE_NAME);
					LOGGER.l7dError(RmWebLogBundleKey.RM_TIMEOFF_BID_UNABLE_TO_SAVE, ex);
				} catch (BbmUpdateDuplicateKeyException ex) {
					model.setHasError(true);
					model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_TIMEOFFBID_NAME_EXISTS, RmWebBundleKey.BUNDLE_NAME);
					LOGGER.l7dError(RmWebLogBundleKey.RM_TIMEOFF_BID_UNABLE_TO_SAVE, ex);
				} catch (RmHardValidationException ex) {
					Message msg = new Message(Message.ERROR_TYPE, ex.getLocalizedMessage());
					model.setHasError(true);
					model.addPageMessage(msg);
					LOGGER.l7dError(RmWebLogBundleKey.RM_TIMEOFF_BID_UNABLE_TO_SAVE, ex);
				} catch (Exception ex) {
					Message msg = new Message(Message.ERROR_TYPE, RmWebBundleKey.RM_TIMEOFF_BID_UNABLE_TO_SAVE, RmWebBundleKey.BUNDLE_NAME);
					model.setHasError(true);
					model.addPageMessage(msg);
					LOGGER.l7dError(RmWebLogBundleKey.RM_TIMEOFF_BID_UNABLE_TO_SAVE, ex);
				}
			}
		}
		prepareForNextPage(context, model, null);
	}

	/**
	 * Process Bid Status Change
	 */
	@SuppressWarnings("unchecked")
	public void processBidStatus(RequestContext context, boolean isOpenBidAction) {
		TimeOffBidSettingsPM model = new TimeOffBidSettingsPM(context);
		model.extractParameters(context.getRequest());
		ID auctionID = model.getSelectedIDObject();
		if (auctionID != null) {
			try {
				TimeOffBid timeOffBid = model.getTimeOffBid();
				Collection<ID> orgIds = OrganizationModelHandler.getSelfAndParentsOrganizationIDs(context, timeOffBid.getOrganizationId());
				TimeOffBidModelHandler.updateTimeOffBid(timeOffBid, orgIds);
				if (isOpenBidAction) {
					timeOffBid.setStatus(TimeOffBid.STATUS_OPEN_FOR_BIDDING);
				} else {
					timeOffBid.setStatus(TimeOffBid.STATUS_CLOSED_FOR_BIDDING);
				}
				TimeOffBidModelHandler.updateTimeOffBid(timeOffBid, orgIds);
			} catch (RmHardValidationException ex) {
				Message msg = new Message(Message.ERROR_TYPE, ex.getLocalizedMessage());
				model.addPageMessage(msg);
				LOGGER.l7dError(RmWebLogBundleKey.UNABLE_TO_CHANGE_TIMEOFF_BID_STATUS, ex);
			} catch (Exception ex) {
				Message msg = new Message(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_CHANGE_TIMEOFF_BID_STATUS, RmWebBundleKey.BUNDLE_NAME);
				context.addPageMessage(msg);
				LOGGER.l7dError(RmWebLogBundleKey.UNABLE_TO_CHANGE_TIMEOFF_BID_STATUS, ex);
			}
		}
		prepareForNextPage(context, model, null);
	}

	@Override
	protected void processDelete(RequestContext context) {
		WorkpanePageModel model;
		Collection<ID> selectedIDs = new ArrayList<ID>();
		// Determine which model to use
		boolean isWorkpaneMode = isWorkpaneMode(context);
		if (isWorkpaneMode) {
			model = new TimeOffBidSettingsPM(context);
		} else {
			model = new TimeOffBidSummaryPM(context);
		}
		if (isAuthorized(context, PrivilegeKeys.FS_EDITTIMEOFFBIDDING)) {
			String strID = context.getParameter(WorkpaneFieldNames.SELECTED_ID_FN);
			selectedIDs = StringUtil.parseIDStringToCollection(strID, ",");
			if (selectedIDs!=null&& !selectedIDs.isEmpty()) {
				try {
					TimeOffBidModelHandler.deleteTimeOffBidsByIds(selectedIDs);
				} catch (BbmCreateException e) {
					Message msg = new Message(Message.ERROR_TYPE, RmWebBundleKey.RM_TIMEOFF_BID_DELETE_CONFIRM, RmWebBundleKey.BUNDLE_NAME);
					model.addPageMessage(msg);
				} catch (BbmRemoveException e) {
					Message msg = new Message(Message.ERROR_TYPE, RmWebBundleKey.RM_TIMEOFF_BID_DELETE_CONFIRM, RmWebBundleKey.BUNDLE_NAME);
					model.addPageMessage(msg);
				} catch (RemoteException e) {
					Message msg = new Message(Message.ERROR_TYPE, RmWebBundleKey.RM_TIMEOFF_BID_DELETE_CONFIRM, RmWebBundleKey.BUNDLE_NAME);
					model.addPageMessage(msg);
				}
				if (isWorkpaneMode(context)) {
					// Need to refresh Selection Pane
					((TimeOffBidSettingsPM) model).setRefreshSelection(true);
				}	
			}
		}
		prepareForNextPage(context, model, null);
	}

}
