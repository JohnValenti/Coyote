package com.verint.web.fs.dashboard.portlets.pulse;

import java.io.Serializable;
import java.util.Date;

public class SummaryData implements Serializable{
	
	/**
	 * 
	 */
	private static final long serialVersionUID = 7727479652433630064L;
	private Date date;
	private int m_forecasted;
	private int m_actual;
	private int m_required;
	
		
	public SummaryData(Date date, int forecasted, int actual, int required)
	{
		this.date = date;
		this.m_forecasted = forecasted;
		this.m_actual = actual;
		this.m_required = required;
	}
	public Date getDate() {
		return date;
	}
	public int getForecasted() {
		return m_forecasted;
	}
	public int getActual() {
		return m_actual;
	}
	public int getRequired() {
		return m_required;
	}

	@Override
	public int hashCode() {
		final int prime = 31;
		int result = 1;
		result = prime * result + ((date == null) ? 0 : date.hashCode());
		result = prime * result + m_actual;
		result = prime * result + m_forecasted;
		result = prime * result + m_required;
		return result;
	}

	@Override
	public boolean equals(Object obj) {
		if (this == obj)
			return true;
		if (obj == null)
			return false;
		if (getClass() != obj.getClass())
			return false;
		SummaryData other = (SummaryData) obj;
		if (date == null) {
			if (other.date != null)
				return false;
		} else if (!date.equals(other.date))
			return false;
		if (m_actual != other.m_actual)
			return false;
		if (m_forecasted != other.m_forecasted)
			return false;
		if (m_required != other.m_required)
			return false;
		return true;
	}
	
	
	
	
	

}
