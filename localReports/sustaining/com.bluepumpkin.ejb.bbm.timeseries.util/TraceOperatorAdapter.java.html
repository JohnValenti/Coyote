	public static HashMap getMediaForQueues(Object[] cubes, boolean isWhatIf) {
		HashMap queueMap = new HashMap();
		try {
			ArrayList arrayList = new ArrayList();
			for (int i = 0; i < cubes.length; i++) {
				TraceCube cube = (TraceCube)cubes[i];
				if (cube.getQueueID() != null && cube.getQueueID().toInt() != -1) {
					arrayList.add(cube.getQueueID());
				}
			}
			queueMap =getMediaForQueues(arrayList, isWhatIf);
		} catch (Exception e) {
			e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
		}
		return queueMap;
	}

	public static HashMap getMediaForQueues(Collection queueIDCol, boolean isWhatIf) {
		HashMap queueMap = new HashMap();

					Queue queue = (Queue) iterator.next();

			}
		} catch (Exception e) {
			e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
		}
		return queueMap;
	}
	
	/**
	 * Returns a map from queue ID to media ID for the given collection of trace cubes.
	 * 
	 * @param traceCubes
	 * @return
	 * @throws BbmFinderException
	 */
	public static Map<ID, ID> getQueueMediaMap(Collection<? extends TraceCube> traceCubes)
			throws BbmFinderException {
		Map<ID, ID> queueMediaMap = new HashMap<ID, ID>();
		Collection<ID> queueIDs = new ArrayList<ID>();
		for (TraceCube cube : traceCubes) {
			queueIDs.add(cube.getQueueID());
		}
		Collection<Queue> queues = new QueueDAO().getObjects(queueIDs);
		for (Queue queue : queues) {
			queueMediaMap.put(queue.getID(), queue.getMediaID());
		}
		return queueMediaMap;
	}

	/**
	 * Utility function to take collection of TraceCube and combine
	 *
	 * @param cubeCol
	 * @param needAllValid
	 * @return
	 * @throws BbmTimeSeriesException
	 */
	public static TraceCube combineQueue(Collection cubeCol, boolean needAllValid, boolean isWhatIf) throws BbmTimeSeriesException {
		TraceCube[] cubeArray = new TraceCube[cubeCol.size()];
		int i = 0;
		for (Iterator it = cubeCol.iterator(); it.hasNext();) {
			cubeArray[i++] = (TraceCube) it.next();
		}
		return combineQueue(cubeArray, needAllValid, isWhatIf);
	}

	public static Map<ID, TraceCube[]> combineCubeMap(Map<ID, TraceCube[]> queueCubeMap, TraceCube[] metaCubes, boolean keepCombineOnly, ID comQueID, boolean needAllValid,boolean isWhatIf) throws Exception {
		boolean isPredictCombinedAlready = false;
		TraceCube[] combinedQs = queueCubeMap.get(comQueID);
		TraceCube combinedPredictQ = null;
		if (combinedQs != null && combinedQs[2] != null) {
			combinedPredictQ = combinedQs[2];
			isPredictCombinedAlready = true;
		}

		// now combined cubes for the combine queue
		TraceCube[] cmbCubes = queueCubeMap.remove(comQueID);
		if (cmbCubes == null) {
			cmbCubes = new TraceCube[5];
		}
		for (int i = 0; i < metaCubes.length; i++) {
			if (metaCubes[i] != null) {
				List<TraceCube> realCubeList = new ArrayList<TraceCube>(queueCubeMap.size());
				for (TraceCube[] realCubes : queueCubeMap.values()) {
					if (realCubes[i] != null) {
						TraceCube toAdd = realCubes[i];
						if (i == 2) {
							if (isPredictCombinedAlready) {
								cmbCubes[i] = combinedPredictQ;
								cmbCubes[i].setQueueID(comQueID);
								break;
							}
							if (realCubes[2] instanceof AggrForecastedTraceCube) {
								toAdd = realCubes[i];
							} else {
								AggrForecastedTraceCube arftc = new AggrForecastedTraceCube(realCubes[i].getQueueID(), realCubes[i].getRawStartDate(), realCubes[i].getRawEndDate(), realCubes[i].getTraceTypes());
								arftc.setForecastTrace((ForecastTraceCube) realCubes[1]);
								arftc.setPredictTraceCube((PredictTraceCube) realCubes[i]);
								toAdd = arftc;
							}
						}
						if (i == 3) {
							if (realCubes[i] instanceof AggrRequiredTraceCube) {
								toAdd = realCubes[i];
							} else {
								// for SG, need convert to AggrRequireCube
								AggrRequiredTraceCube artc =
								        new AggrRequiredTraceCube(realCubes[i].getQueueID(), realCubes[i].getRawStartDate(), realCubes[i].getRawEndDate(), realCubes[i].getTraceTypes());
								artc.setServiceGoalTrace((ServiceGoalTraceCube) realCubes[i]);
								artc.setForecastTrace(realCubes[1]);
								toAdd = artc;
							}
						}
						if (i == 1) { //Added this part for calculating CRate.
							if (realCubes[i] instanceof AggrForecastedTraceCube) {
								toAdd = realCubes[i];
							} else {
								AggrForecastedTraceCube arftc = new AggrForecastedTraceCube(realCubes[i].getQueueID(), realCubes[i].getRawStartDate(), realCubes[i].getRawEndDate(), realCubes[i].getTraceTypes());
								arftc.setForecastTrace((ForecastTraceCube) realCubes[1]);
								arftc.setPredictTraceCube((PredictTraceCube) realCubes[2]);
								toAdd = arftc;
							}
							if (isPredictCombinedAlready && combinedPredictQ.getTraceTypes() != null) {
								short[] predictTypes = combinedPredictQ.getTraceTypes();
								short[] types = toAdd.getTraceTypes();
								short[] typesComb = new short[predictTypes.length + types.length];
								System.arraycopy(types, 0, typesComb, 0, types.length);
								System.arraycopy(predictTypes, 0, typesComb, types.length, predictTypes.length);
								toAdd.setTraceType(typesComb);
							}
						}
						realCubeList.add(toAdd);
					}
				}
				if (!realCubeList.isEmpty()) {

					cmbCubes[i] = combineQueue(realCubeList, needAllValid, isWhatIf);
					if (cmbCubes[i] != null) {
						if (i == 2) {
							//for predict, convert back
							cmbCubes[i] = ((AggrForecastedTraceCube) cmbCubes[i]).getPredictTrace();
						}
						if (i == 3) {
							// for SG, convert back
							cmbCubes[i] = ((AggrRequiredTraceCube) cmbCubes[i]).getServiceGoalTrace();
						}
						//Convert back to Forecast Cube Followed the precedent set above
						if (i == 1 && cmbCubes[i] instanceof AggrForecastedTraceCube) {
							if (isPredictCombinedAlready
							        && ((AggrForecastedTraceCube) cmbCubes[i]).getPredictTrace() != null) {
								short types[] = combinedPredictQ.getTraceTypes();
								for (int j = 0; j < types.length; j++) {
									if (combinedPredictQ.isTraceTypeComputed(types[j])) {
										combinedPredictQ.setTraceValue(types[j], ((AggrForecastedTraceCube) cmbCubes[i]).getPredictTrace().getTraceValueD(types[j]));
									}
								}
							}
							cmbCubes[i] = ((AggrForecastedTraceCube) cmbCubes[i]).getForecastTrace();
						}
						cmbCubes[i].setQueueID(comQueID);
					}
				}
			}
		}
		if (keepCombineOnly) {
			queueCubeMap.clear();
		}
		//Calculate the computed values for Trace cubes
		calcComputedValuesForAggrForcastTraceCubes(cmbCubes);
		queueCubeMap.put(comQueID, cmbCubes);
		return queueCubeMap;
	}

	public static void calcComputedValuesForAggrForcastTraceCubes(TraceCube[] queCubes, Map<ID, SPQueue> spQueueMap) throws Exception {
		if (queCubes != null) {
			ID mediaID = Media.MEDIA_ID_PHONE;
			for (TraceCube queCube : queCubes) {
				if (queCube != null) {
					ID spQueueId = queCube.getSPQueueSID();
					SPQueue spQueue = getSPQueue(spQueueId, spQueueMap);
					if (spQueue != null) {
						mediaID = spQueue.getMediaID();
						break;
					}
				}
			}
			// Predicted needs to be computed before forecasted, else we could
			// use a loop here
			if (queCubes[2] != null) {
				queCubes[2].calcComputedTraceValuesForCubes(queCubes, mediaID);
				ID spQueueId = queCubes[2].getSPQueueSID();
				SPQueue spQueue = getSPQueue(spQueueId, spQueueMap);
				computeAndScaleNetStaffing((PredictTraceCube) queCubes[2],
						(RequireTraceCube) queCubes[4], spQueue);
			}
			if (queCubes[1] != null)
				queCubes[1].calcComputedTraceValuesForCubes(queCubes, mediaID);
			if (queCubes[4] != null)
				queCubes[4].calcComputedTraceValuesForCubes(queCubes, mediaID);
		}
    }

	}
	
	public static void computeAndScaleNetStaffing(PredictTraceCube pCube, RequireTraceCube rCube, SPQueue pCubeSPQueue) throws Exception {
		boolean computeNetStaff = false;
		if (pCube == null || rCube == null || rCube.getTraceValueD(Trace.FTE) == null || pCube.getSPQueueSID()==null || pCubeSPQueue==null) return;
		for (int i = 0; i < pCube.getTraceTypes().length; i++) {
			if (pCube.getTraceTypes()[i] == Trace.NETSTAFFING) {
				computeNetStaff = true;
				break;
			}
		}
		if (!computeNetStaff) return; //Nettstaffing not requested, just return here
		int[][] ranges = null;
		if (pCubeSPQueue.getLeastTimeInterval() == SPQueue.LEAST_TIME_INTERVAL_15MINS) {
			double[] cReqFTEArray = rCube.getTraceValueD(Trace.FTE);
			double[] cForFTEArray = pCube.getTraceValueD(Trace.AFTE); //Adjusted FTE
			double[] cForSTAFFArray = pCube.getTraceValueD(Trace.STAFFING); //This is actually Adjusted Staffing, but the types are switched
			double[] value = new double[cReqFTEArray.length];
			for (int i = 0; cReqFTEArray != null && i < cReqFTEArray.length; i++) {
				if (!Trace.isValidValue(cForFTEArray[i]) || !Trace.isValidValue(cReqFTEArray[i]) || !Trace.isValidValue(cForSTAFFArray[i])) {
					value[i] = Trace.TRACENA;
				} else {
					double fStaffRatio = (cForFTEArray[i] == 0 || cForSTAFFArray[i] == 0) ? 0.0d : ((double) cForFTEArray[i] / (double) cForSTAFFArray[i]);
					if (fStaffRatio == 0) fStaffRatio = 1;
					// not a good hack but we cannot show N/A when the FSTAFF or FFTE is zero.
					// NETSTAFF = RFTE because user should be able to request OT even if FSTAFF or FFTE is zero
					value[i] = (double) (cForFTEArray[i] - cReqFTEArray[i]) / fStaffRatio;
				}
			}
			pCube.setTraceValue(Trace.NETSTAFFING, value);
		} else {
			if (pCubeSPQueue.getLeastTimeInterval() == SPQueue.LEAST_TIME_INTERVAL_HOURLY) {
				ranges = getHourlyRanges(pCube);
			}
			if (pCubeSPQueue.getLeastTimeInterval() == SPQueue.LEAST_TIME_INTERVAL_DAILY) {
				ranges = getDailyRanges(pCube);
			}
			if (pCubeSPQueue.getLeastTimeInterval() == SPQueue.LEAST_TIME_INTERVAL_WEEKLY) {
				ranges = getWeeklyRanges(pCube);
			}
			if (ranges != null) {
				int[] range=null;
				for (int i = 0; i < ranges.length; i++) {
					if(range!=null&& range[1]>ranges[i][0]){
						continue;//no point in going ahead ; this is a bug in the range
					}
					range = ranges[i];
					if(range[1] > pCube.getRawEndOffset()){
						//adjust the date to end of cube
						m_cat.info("ERROR *** date range end not part of SP: range[1]="+range[1]+ ": \t pCube.getRawEndOffset()="+pCube.getRawEndOffset()+ ": \t spQueue.getLeastTimeInterval()="+pCubeSPQueue.getLeastTimeInterval() +" \t spQ="+pCubeSPQueue.getID() +"\tST="+pCube.getRawStartDate()+"\tEN="+pCube.getRawEndDate());
						range[1] = pCube.getRawEndOffset();
					}
					double aggFFTE = TraceOperator.aggregatePeriodD(pCube, Trace.AFTE, range[0], range[1]);
					double aggRFTE = TraceOperator.aggregatePeriodD(rCube, Trace.FTE, range[0], range[1]);
					double aggFSTAFF = TraceOperator.aggregatePeriodD(pCube, Trace.ASTAFFING, range[0], range[1]);
					// (FFTE ï¿½ RFTE)/ (FFTE / FSTAFF)
					double fStaffRatio = (aggFFTE == 0.0d || aggFSTAFF == 0.0d) ? 0.0d : (double) (aggFFTE / aggFSTAFF);
					if (fStaffRatio == 0.0d) fStaffRatio = 1.0d;
					// not a good hack but we cannot show N/A when the FSTAFF or FFTE is zero.
					double netStaff = ((aggFFTE - aggRFTE) * 4) / fStaffRatio;
					double scaleVal = netStaff / ((range[1] - range[0])+1);
					double diffCarryOver = 0;
					for (int cou = range[0]; cou <= range[1]; cou++) {
						int finalVal = TraceUtil.roundDouble(scaleVal + diffCarryOver);
						diffCarryOver = (scaleVal + diffCarryOver) - (double) finalVal;
						pCube.setTraceValue(Trace.NETSTAFFING, finalVal, cou, true);
					}
				}
			}
		}
	}

	public static int[][] getWeeklyRanges(TraceCube cube) {
		Date weekEndDate = null;
		ArrayList aWeekRanges = new ArrayList();
		//Must use campaign TZ to determine SP weeks. The week ranges are not affected by the viewing TimeZone.
		Calendar cal = Calendar.getInstance(TimeZone.getTimeZone(cube.getCampaignTimeZoneID()));
		Date startDate = cube.getRawStartDate();
		Date endDate = cube.getRawEndDate();
		Date nextStartDate = startDate;
		cal.setTime(startDate);
		int startHour = cal.get(Calendar.HOUR_OF_DAY);

		do {
			startDate = nextStartDate;
			cal.setTime(startDate);
			cal.add(Calendar.DATE, 7);

			//If day boundary is 2AM, on Spring DST day, adding a week to 2AM start date will
			//result in end date at 1AM instead of 3AM. We must manually correct for this.
			if (cal.get(Calendar.HOUR_OF_DAY) < startHour)
				cal.add(Calendar.HOUR_OF_DAY, 1);
			else if (cal.get(Calendar.HOUR_OF_DAY) > startHour) //undo our hour addition for the following week
				cal.add(Calendar.HOUR_OF_DAY, -1);

			nextStartDate = cal.getTime();
			cal.add(Calendar.SECOND, -1);
			weekEndDate = cal.getTime();
			aWeekRanges.add(new TimeRange(startDate, weekEndDate));
		} while (nextStartDate.before(endDate));
		int[][] weekRanges = null;
		if (aWeekRanges != null) {
			weekRanges = new int[aWeekRanges.size()][2];
			int w = 0;
			for (Iterator it = aWeekRanges.iterator(); it.hasNext(); w++) {
				TimeRange curWeek = (TimeRange) it.next();
				weekRanges[w][0] = TraceUtil.gapOffSet(cube.getStartDate(), cube.getStartOffSet(), curWeek.getStartDate());
				weekRanges[w][1] = TraceUtil.gapOffSet(cube.getStartDate(), cube.getStartOffSet(), curWeek.getEndDate());
			}
		}
		return weekRanges;
	}

	public static int[][] getDailyRanges(TraceCube cube) {
		int numOfDays = TimeZoneUtil.numberOfDays(cube.getRawStartDate(), cube.getRawEndDate(), TimeZone.getTimeZone(cube.getCampaignTimeZoneID())); // dates are inclusive.
		int[][] returnVal = new int[numOfDays][2];
		for (int i = 0; i < numOfDays; i++) {
			int[] dayRangeArray = TraceUtil.getDailyRange(cube,TimeZone.getTimeZone( cube.getCampaignTimeZoneID()), i, true);
			returnVal[i][0] = dayRangeArray[0];
			returnVal[i][1] = dayRangeArray[1];
		}                                               
		return returnVal;
	}

	public static int[][] getHourlyRanges(TraceCube cube) {
		int endPoint = cube.getRawEndOffset();
		Date start = cube.getRawStartDate();
		Calendar cal = Calendar.getInstance(TimeZone.getTimeZone(cube.getCampaignTimeZoneID()));
		cal.setTime(start);
		int startMin = cal.get(Calendar.MINUTE);
		// Get the absolute position
		int pos = startMin / Trace.INTERVAL;
		// Need figure out how many hours
		int hours = 1;
		int totalLength = endPoint - (pos + 1);
		for (; totalLength > 0; totalLength -= 4, hours++) ;
		int begin = 0;
		int end = (3 - pos) == 0 ? 3 : (3 - pos);// if end point is already 3, then firt hour only one point
		//this is done so that we will flag periods only  upto the given time (now)
		//int cutOffPoint = TraceUtil.gapOffSet(cube.getStartDate(),(short)0, now)-1;
		// Cal each hour
		int[][] returnVal = new int[hours][2];
		for (int i = 0; i < hours; i++) {
			returnVal[i][0] = begin;
			returnVal[i][1] = end;
			begin = end + 1;
			if (begin >= endPoint)
				break;
			end += 4;
			end = (end >= endPoint ? (endPoint - 1) : end);
		}
		return returnVal;
	}

	public static void setOverUnder(PredictTraceCube pCube, RequireTraceCube rCube, SPQueue spQueue) {
		//((RFTE/AFTE)  -1 ) *100
		// (1 ï¿½ (RFTE/AFTE))    *100
		int leastCount= 1; //1 for 15 mins /4 for hourly /96 daily /4*24*7 for weekly
		double[] cReqFTEArray = rCube.getTraceValueD(Trace.FTE);
		double[] cForFTEArray = pCube.getTraceValueD(Trace.AFTE);
		int arrayLength = cReqFTEArray.length;
		double[] valuesArray = new double[arrayLength];
		for (int i = 0; cReqFTEArray != null && i < cReqFTEArray.length; i++) {
			if (!Trace.isValidValue(cForFTEArray[i]) || !Trace.isValidValue(cReqFTEArray[i])) {
				valuesArray[i] = Trace.TRACENA;
			} else {
				double fteRatio = cForFTEArray[i] == 0 ? 0.0 : ((double) cReqFTEArray[i] * 100d / (double) cForFTEArray[i]);
				if (cReqFTEArray[i] > cForFTEArray[i]) {   //UNDER
					if (!spQueue.isFTEThresholdUnderPercentage()) {
						if ((cReqFTEArray[i] - cForFTEArray[i]) > spQueue.getFTEThresholdUnder()) {
							valuesArray[i] = Trace.NETSTAFF_UNDER;
						}
					} else {
						// check for percentage
						if ((100 - fteRatio) > spQueue.getFTEThresholdUnder()) {
							valuesArray[i] = Trace.NETSTAFF_UNDER;
						}
					}
				} else if (cReqFTEArray[i] < cForFTEArray[i]) { //OVER
					if (!spQueue.isFTEThresholdOverPercentage()) {
						if ((cForFTEArray[i] - cReqFTEArray[i]) > spQueue.getFTEThresholdOver()) {
							valuesArray[i] = Trace.NETSTAFF_OVER;
						}
					} else {
						// check for percentage
						if ((fteRatio - 100) > spQueue.getFTEThresholdOver()) {
							valuesArray[i] = Trace.NETSTAFF_OVER;
						}
					}
				} else {
					valuesArray[i] = 0;
				}
			}
		}
