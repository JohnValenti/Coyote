package com.bluepumpkin.ejb.am.timecollection.jms;

import java.io.IOException;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.io.Serializable;
import java.util.ArrayList;

/**
 * Title:        Blue Pumpkin Software Activity Management
 *
 * Description:  represents JMS message from BPX with error messages
 *               for previous ChangeStateMessage
 *
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */

public class ChangeStateStatusMessage extends ChangeStateMessage {
	/** serialization/deserialization compatible with previous versions. */
	static final long serialVersionUID = 5009369884047693286L;

	private String  m_status;
	private boolean  m_successful;
	private ArrayList m_errorMessages;

	public static final int UNDETERMINED = 0;
	public static final int SUCCESSFUL = 1;
	public static final int SUCCESSFUL_WITH_WARNINGS = 2;
	public static final int FAILED = 3;

	public static class StatusError implements Serializable {
		public String messageId;
		public Object arguments[];

		public StatusError(String messageId,
						   Object arguments[]) {
			this.messageId = messageId;
			this.arguments = arguments;
		}

		public String toString() {
			StringBuffer buffer = new StringBuffer();
			buffer.append("StatusError[");
			buffer.append("\nStatusError.messageId: "+messageId);
			if (arguments != null && arguments.length != 0) {
				for (int i = 0; i < arguments.length; i++) {
					buffer.append("\nStatusError.arguments["+i+"]: "+arguments[i]);
				}
			} else
				buffer.append("\nStatusError.arguments: null");
			buffer.append("\n]");
			return buffer.toString();
		}
	}

	/**
	 *  default constructor
	 */
	public ChangeStateStatusMessage() {
	}

	/**
	 *  constructor - ChangeStateMessage
	 *
	 *  @param changeStateMessage   ChangeStateMessage to copy
	 */
	public ChangeStateStatusMessage(ChangeStateMessage changeStateMessage) {
		super(changeStateMessage);
	}

	/*
	* getters/setters
	*/
	public String getStatus() {
		return m_status;
	}
	public void setStatus( String status ) {
		this.m_status = status;
	}

	public boolean isSuccessful() {
		return m_successful;
	}
	public void setSuccessful( boolean successful ) {
		this.m_successful = successful;
	}

	public ArrayList getErrorMessages() {
		return m_errorMessages;
	}
	public void setErrorMessages( ArrayList errors ) {
		this.m_errorMessages = errors;
	}

	/**
	 * Add an error message to the list of error messages.
	 * To do: Need to figure out which message file
	 *        defines the strings.
	 *
	 * @param messageId - key into resource bundle
	 * @param arguments - array of parameters for the message.
	 */
	public void addErrorMessage(String messageId,
								Object arguments[]) {
		ArrayList errors = getErrorMessages();
		if (errors == null) {
			setErrorMessages(errors = new ArrayList(5));
		}
		errors.add(new StatusError(messageId, arguments));
	}

	/**
	* print object variables
	*/
	public String toString () {
		StringBuffer buffer = new StringBuffer();
		buffer.append(super.toString());

		buffer.append("\nChangeStateStatusMessage.m_errorMessages: "+m_errorMessages);
		buffer.append("\nChangeStateStatusMessage.m_status: "+m_status);
		buffer.append("\nChangeStateStatusMessage.m_successful: "+m_successful);


		return buffer.toString();
	}

	private void writeObject(ObjectOutputStream out) throws IOException {
		out.defaultWriteObject();		
		out.writeObject(m_status);
		out.writeBoolean(m_successful);
		out.writeObject(m_errorMessages);
	}
	
	private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
		in.defaultReadObject();
		m_status = (String)in.readObject();
		m_successful = in.readBoolean();
		m_errorMessages = (ArrayList)in.readObject();
	}
	
}
