package com.verint.web.fs.optimization.servicegoals;

import java.io.Serializable;

import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.timeseries.model.ClientServiceGoalTraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastTraceCube;
import com.verint.web.fs.optimization.CombinedQueueDTO;

/**
 * The Class CombinedServiceGoalsQueueDTO. Instances of this class represent data transfer objects for combined queue selections
 * in the context of service goals.
 */
public class CombinedServiceGoalsQueueDTO extends 
	CombinedQueueDTO<MultiServiceGoalsQueueDTO> implements Serializable {

	private static final long serialVersionUID = -7996710625956861182L;
	
	// This one is used when it is non-skill based, otherwise, the aggregation from the encapsulated
	// multi queue is used.
	private ClientServiceGoalTraceCube m_aggregatedServiceGoals;
	private ForecastTraceCube m_aggregatedForecast;

	/**
	 * Creates a new instance of CombinedServiceGoalsQueueDTO. Use this constructor when it is skill based.
	 * 
	 * @param spQueue the sp queue
	 * @param multiQueueDto the multi queue dto
	 */
	public CombinedServiceGoalsQueueDTO(SPQueue spQueue,
			MultiServiceGoalsQueueDTO multiQueueDto) {
		super(spQueue, multiQueueDto);
	}
	
	/**
	 * Creates a new instance of CombinedServiceGoalsQueueDTO. Use this constructor when it is non-skill based.
	 * 
	 * @param spQueue the sp queue
	 * @param serviceGoals the service goals
	 * @param forecast the forecast
	 * @param multiQueueDto the multi queue dto
	 */
	public CombinedServiceGoalsQueueDTO(SPQueue spQueue, ClientServiceGoalTraceCube serviceGoals,
			ForecastTraceCube forecast,
			MultiServiceGoalsQueueDTO multiQueueDto) {
		super(spQueue, multiQueueDto);
		m_aggregatedServiceGoals = serviceGoals;
		m_aggregatedForecast = forecast;
	}
	
	/**
	 * Gets the aggregated service goals for combined.
	 * 
	 * @return the service goals
	 */
	public ClientServiceGoalTraceCube getServiceGoals() {
		if (m_aggregatedServiceGoals != null) {
			return m_aggregatedServiceGoals;
		}
		return m_multiQueueDto.getServiceGoals();
	}
	
	/**
	 * Gets the aggregated forecast for combined.
	 * 
	 * @return the forecast
	 */
	public ForecastTraceCube getForecast() {
		if (m_aggregatedForecast != null) {
			return m_aggregatedForecast;
		}
		return m_multiQueueDto.getForecast();
	}
}
