package com.bluepumpkin.ejb.bbm.rtaamanager.ejb;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.HashMap;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.rtaamanager.model.OrganizationDay;
import com.bluepumpkin.ejb.bbm.activity.model.Event;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectUtil;

public class RTAAUtil {
	public final static ID ID_0 = new ID(0);
	public static final int NUM_OF_MILLI_SEC_IN_A_DAY =1000*60*60*24;
	public static final int THIRTY_DAYS = 30;
	public static int getParentEventType(Event event) {
		switch (event.getEventType())
		{
			case Event.EVENT_TYPE_TIME_ENTRY :
				return Event.EVENT_TYPE_TIME_RECORD;
			case Event.EVENT_TYPE_SHIFT_EVENT_ASSIGNMENT :
				return Event.EVENT_TYPE_SHIFT_ASSIGNMENT;
			default :
				return -1;
		}
	}


	public static Object getOverlap(HashMap hmOverlaps, ID workresourceID) {
		Object obj = hmOverlaps.get(workresourceID);
		if (obj == null)
			return null;
		ArrayList list = (ArrayList)obj;
		if (list.isEmpty())
			return null;
		else
			return list.iterator().next();
	}


	public static Collection getEventIDs(HashMap groupedEvents) {
		ArrayList ids = new ArrayList();

		Collection events = null;
		ValueObjectBase obj = null;
		for (Iterator i = groupedEvents.values().iterator(); i.hasNext(); ) {
			events = (Collection)i.next();
			if (events != null) {
				for (Iterator iterEvent = events.iterator(); iterEvent.hasNext(); ) {
					obj = (ValueObjectBase)iterEvent.next();
					ids.add(obj.getID());
				}
			}
		}

		return ids;
	}

	public static int getNumberOfValues(HashMap map) {
		int num = 0;

		for (Iterator i = map.values().iterator(); i.hasNext(); ) {
			Collection col = (Collection)i.next();
			if (col != null)
				num += col.size();
		}

		return num;
	}

	public static ID findOverlapOrgDayID(Collection orgDays,Date timespot) {
		OrganizationDay orgDay = null;
		if (orgDays == null || orgDays.isEmpty())
			return RTAAUtil.ID_0;
		for (Iterator i = orgDays.iterator(); i.hasNext(); ) {
			orgDay = (OrganizationDay)i.next();
			if (timespot.before(orgDay.getEndTime())&& !timespot.before(orgDay.getStartTime()))
				return orgDay.getID();
		}
		return RTAAUtil.ID_0;
	}

	/**
	 * Returns false for all those Transition dates which are <= 30  days apart
	 * @param currentTransitionDate
	 * @param previousTransitionDate
	 * @return
	 */
	static boolean isDSTTransitionGreaterThanMonthApart(Date currentTransitionDate, Date previousTransitionDate){

		return ((currentTransitionDate.getTime() - previousTransitionDate.getTime())/ RTAAUtil.NUM_OF_MILLI_SEC_IN_A_DAY)
				> RTAAUtil.THIRTY_DAYS;
	}
}
