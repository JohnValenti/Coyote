	public Collection getConflicts(ID idWorkResource, Organization pOrganization, LocalDate dtStartDate,
			LocalDate dtEndDate, Collection pEvents, Collection pConflicts, boolean isMonthlySp) throws WorkRuleException {
		if (!m_bInitialized) {
			throw new WorkRuleException("Work Rule has not been initialized");
		}
		
		StringBuffer sMessage = new StringBuffer();
		if (!isValid(sMessage)) {
			throw new WorkRuleException("Work Rule is InValid");
		}

		TimeZone tzOrg = pOrganization.getTimeZone();

		if (m_ePeriodType != NOPERIOD && m_ePeriodType != ROLLING_WEEK) {
			// make a call to the rule once for each period
			ArrayList pAllConflicts = new ArrayList();
			
			//exit the conflict check if the workrule is not applicable for the monthly sp
			if (isMonthlySp && !isWorkRuleApplicableOnMonthlySP(dtStartDate.getTime(tzOrg), dtEndDate.getTime(tzOrg))) {
				return pAllConflicts;
			}
			
			Collection pOverlappingPeriods = getOverlappingPeriods(dtStartDate, dtEndDate, tzOrg);
			
			for (Iterator i = pOverlappingPeriods.iterator(); i.hasNext();) {
				TimeSpan pPeriod = (TimeSpan) i.next();
				Date dtPeriodStart = pPeriod.dtLocalSpanStart.getTime(tzOrg);
				Date dtPeriodEnd = pPeriod.dtLocalSpanEnd.getTime(tzOrg);
				long lDayBoundary = pOrganization.getDayBoundaryOffset() * 60 * 1000;
				dtPeriodStart.setTime(dtPeriodStart.getTime() + lDayBoundary);
				dtPeriodEnd.setTime(dtPeriodEnd.getTime() + lDayBoundary);

				Collection pNewConflicts = getConflictsInPeriod(dtPeriodStart, dtPeriodEnd, pOrganization, pEvents,
						pConflicts);
				pAllConflicts.addAll(SpecifyConflicts(idWorkResource, pNewConflicts));
			}
			return pAllConflicts;
		} else {
			if (m_ePeriodType == ROLLING_WEEK) {
				return new ArrayList();
			}
			return SpecifyConflicts(
					idWorkResource,
					getConflictsInPeriod(m_dtStartDate.getTime(tzOrg), m_dtEndDate.getTime(tzOrg), pOrganization, pEvents,
							pConflicts));
		}
	}

	protected boolean isWorkRuleApplicableOnMonthlySP(Date dtStart, Date dtEnd) {
		float daysInSP = Math.round((dtEnd.getTime() - dtStart.getTime()) / MILLISECONDS_IN_DAY);

		if (m_ePeriodType == CalendarPeriod.CalendarUnit.Month28Day.getDBValue()) {
			return isPeriodMatchesDaysInMonthlySP(daysInSP, CalendarPeriod.CalendarUnit.Month28Day.getNumDays());
		} else if (m_ePeriodType == CalendarPeriod.CalendarUnit.Month29Day.getDBValue()) {
			return isPeriodMatchesDaysInMonthlySP(daysInSP, CalendarPeriod.CalendarUnit.Month29Day.getNumDays());
		} else if (m_ePeriodType == CalendarPeriod.CalendarUnit.Month30Day.getDBValue()) {
			return isPeriodMatchesDaysInMonthlySP(daysInSP, CalendarPeriod.CalendarUnit.Month30Day.getNumDays());
		} else if (m_ePeriodType == CalendarPeriod.CalendarUnit.Month31Day.getDBValue()) {
			return isPeriodMatchesDaysInMonthlySP(daysInSP, CalendarPeriod.CalendarUnit.Month31Day.getNumDays());
		}
		
		return true;
	}
	
	protected boolean isPeriodMatchesDaysInMonthlySP(float daysInMonthlySP, float daysInPeriod) {
		if (daysInMonthlySP == daysInPeriod) {
			return true;
		}
		return false;
	}
	
	private Collection getConflictsInPeriod(Date dtStartDate, Date dtEndDate, Organization pOrganization,
			Collection pEvents, Collection pConflicts) throws WorkRuleException {

		// switch to proper built in WorkRule
		if (m_pExternalWorkRule != null) {
			return m_pExternalWorkRule.getWorkRuleConflicts(m_pWorkRuleParameters, m_nPriority, pOrganization,
					m_ePeriodOverlap, dtStartDate, dtEndDate, pEvents, pConflicts);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDMAXHOURS)) {
			return InternalRules.getMaxHoursConflicts(m_pWorkRuleParameters, m_nPriority, pOrganization, m_ePeriodOverlap,
					dtStartDate, dtEndDate, pEvents, pConflicts);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDMINPAIDHOURS)) {
			return InternalRules.getMinPaidHoursConflicts(m_pWorkRuleParameters, m_nPriority, pOrganization,
					m_ePeriodOverlap, dtStartDate, dtEndDate, pEvents, pConflicts);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDTIMEOFDAY)) {
			return InternalRules.getTimeOfDayConflicts(m_pWorkRuleParameters, m_nPriority, pOrganization, m_ePeriodOverlap,
					dtStartDate, dtEndDate, pEvents, pConflicts);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDWEEKDAY)) {
			return InternalRules.getWeekDayConflicts(m_pWorkRuleParameters, m_nPriority, pOrganization, m_ePeriodOverlap,
					dtStartDate, dtEndDate, pEvents, pConflicts);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDCONSECUTIVEDAY)) {
			return InternalRules.getConsecutiveDaysConflicts(m_pWorkRuleParameters, m_nPriority, pOrganization,
					m_ePeriodOverlap, dtStartDate, dtEndDate, pEvents, pConflicts);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDWORK)) {
			return InternalRules.getMustWorkConflicts(m_pWorkRuleParameters, m_nPriority, pOrganization, m_ePeriodOverlap,
					dtStartDate, dtEndDate, pEvents, pConflicts, this);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDHAVEOFF)) {
			return InternalRules.getMustHaveOffConflicts(m_pWorkRuleParameters, m_nPriority, pOrganization,
					m_ePeriodOverlap, dtStartDate, dtEndDate, pEvents, pConflicts, this);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDFAIRNESS)) {
			return new ArrayList(); // note multi employee rules are not
									// validated yet
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDCONSECTIVE)) {
			return InternalRules.getConsecutiveConflicts(m_pWorkRuleParameters, m_nPriority, pOrganization,
					m_ePeriodOverlap, dtStartDate, dtEndDate, pEvents, pConflicts);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDMINSPACING)) {
			return InternalRules.getSpacingConflicts(m_pWorkRuleParameters, m_ePeriodOverlap, dtStartDate, dtEndDate,
					pEvents);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDCROSSDAY)) {
			return InternalRules.getCrossDayConflicts(pOrganization, m_ePeriodOverlap, dtStartDate, dtEndDate, pEvents);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDTEAMS)) {
			return new ArrayList(); // note multi employee rules are not
									// validated yet
		} else {
			throw new WorkRuleException("unknown Rule Template");
		}
	}

	private Collection SpecifyConflicts(ID idWorkResource, Collection pConflicts) {

		for (Iterator i = pConflicts.iterator(); i.hasNext();) {
			CommonConflict pException = (CommonConflict) i.next();
			pException.setWorkResourceID(idWorkResource);
			if (m_idEarningType != null) {
				pException.setType(Alert.PAYRULE);
			} else {
				pException.setType(Alert.WORKRULE);
			}
			if (m_dPayModification > 0.0) {
				pException.setAdjustment(m_dPayModification);
			}
			String sDescription = m_sName;
			pException.setDescription(sDescription);
			pException.setPriority(m_nPriority);
			pException.setEarningType(m_idEarningType);
			pException.SetWorkRuleID(m_idWorkRule);
		}
		return pConflicts;
	}

	public LocalDate expandStartDate(LocalDate dtStartDate, LocalDate dtEndDate) throws WorkRuleException {

		if (m_ePeriodType != NOPERIOD && m_ePeriodType != ROLLING_WEEK) {
			Collection pOverlappingPeriods = getOverlappingPeriods(dtStartDate, dtEndDate, null);
			for (Iterator i = pOverlappingPeriods.iterator(); i.hasNext();) {
				TimeSpan pPeriod = (TimeSpan) i.next();
				if (pPeriod.dtLocalSpanStart.getTime().before(dtStartDate.getTime())) {
					return pPeriod.dtLocalSpanStart;
				}
			}
			return dtStartDate;
		}

		if (m_pExternalWorkRule != null) {
			int nLookBack = m_pExternalWorkRule.getLookBack(m_pWorkRuleParameters);
			dtStartDate.add(Calendar.DAY_OF_YEAR, -1 * nLookBack);
			return dtStartDate;
		}

		if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDCONSECUTIVEDAY)) {
			// the consecutive day rule needs look back if it doesn't have a
			// period;
			Iterator w = m_pWorkRuleParameters.iterator();
			int nMinConsecutiveDays = ((WorkRuleParameter) w.next()).getIntValue();
			int nMaxConsecutiveDays = ((WorkRuleParameter) w.next()).getIntValue();
			dtStartDate.add(Calendar.DAY_OF_YEAR, -1 * nMaxConsecutiveDays);
			return dtStartDate;
		}

		if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDMINSPACING)) {
			// the spacing rule needs 1 day look back
			dtStartDate.add(Calendar.DAY_OF_YEAR, -1);
			return dtStartDate;
		}

		return dtStartDate;
	}

	public Pair expandWindow(LocalDate dtStartDate, LocalDate dtEndDate) throws WorkRuleException {

		if (m_ePeriodType != NOPERIOD && m_ePeriodType != ROLLING_WEEK) {
			Collection pOverlappingPeriods = getOverlappingPeriods(dtStartDate, dtEndDate, null);
			for (Iterator i = pOverlappingPeriods.iterator(); i.hasNext();) {
				TimeSpan pPeriod = (TimeSpan) i.next();
				if (pPeriod.dtLocalSpanStart.getTime().before(dtStartDate.getTime())) {
					dtStartDate = pPeriod.dtLocalSpanStart;
				}
				if (pPeriod.dtLocalSpanEnd.getTime().after(dtEndDate.getTime())) {
					dtEndDate = pPeriod.dtLocalSpanEnd;
				}
			}
			return new Pair(dtStartDate, dtEndDate);
		}

		if (m_pExternalWorkRule != null) {
			int nLookBack = m_pExternalWorkRule.getLookBack(m_pWorkRuleParameters);
			int nLookForward = m_pExternalWorkRule.getLookForward(m_pWorkRuleParameters);
			LocalDate dtExpandedStartDate = new LocalDate(dtStartDate);
			dtExpandedStartDate.add(Calendar.DAY_OF_YEAR, -1 * nLookBack);
			LocalDate dtExpandedEndDate = new LocalDate(dtEndDate);
			dtExpandedStartDate.add(Calendar.DAY_OF_YEAR, nLookForward);

			return new Pair(dtExpandedStartDate, dtExpandedStartDate);
		}

		if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDCONSECUTIVEDAY)) {
			// the consecutive day rule needs look back if it doesn't have a
			// period;
			Iterator w = m_pWorkRuleParameters.iterator();
			int nMinConsecutiveDays = ((WorkRuleParameter) w.next()).getIntValue();
			int nMaxConsecutiveDays = ((WorkRuleParameter) w.next()).getIntValue();
			LocalDate dtExpandedStartDate = new LocalDate(dtStartDate);
			dtExpandedStartDate.add(Calendar.DAY_OF_YEAR, -1 * nMaxConsecutiveDays);
			return new Pair(dtExpandedStartDate, dtEndDate);
		}
		return new Pair(dtStartDate, dtEndDate);
	}

	public boolean isValid(Locale locale, StringBuffer sMessage) throws WorkRuleException {
		// return an error string if invalid, returns null
		ResourceBundle rs = ResourceBundle.getBundle(BbmEjbBundleKey.BUNDLE, locale);

		if (!m_dtEndDate.getTime().after(m_dtStartDate.getTime())) {
			sMessage.append(rs.getString(BbmEjbBundleKey.WORKRULE_END_DATE_AFTER_START));
			return false;
		}
		if (m_pExternalWorkRule != null) {
			return m_pExternalWorkRule.isValid(m_pWorkRuleParameters, m_ePeriodType, m_nPeriodLength, locale, sMessage);
		}
		if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDMAXHOURS)) {
			return InternalRules.isMaxHoursValid(m_pWorkRuleParameters, m_ePeriodType, m_nPeriodLength, locale, sMessage);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDMINPAIDHOURS)) {
			return InternalRules
					.isMinPaidHoursValid(m_pWorkRuleParameters, m_ePeriodType, m_nPeriodLength, locale, sMessage);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDTIMEOFDAY)) {
			return InternalRules.isTimeOfDayValid(m_pWorkRuleParameters, m_ePeriodType, m_nPeriodLength, locale, sMessage);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDWEEKDAY)) {
			return InternalRules.isWeekDayValid(m_pWorkRuleParameters, m_ePeriodType, m_nPeriodLength, locale, sMessage);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDCONSECUTIVEDAY)) {
			return InternalRules.isConsecutiveDaysValid(m_pWorkRuleParameters, m_ePeriodType, m_nPeriodLength, locale,
					sMessage);
		}

		else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDWORK)) {
			return InternalRules.isWorkValid(m_pWorkRuleParameters, m_ePeriodType, m_nPeriodLength, locale, sMessage);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDHAVEOFF)) {
			return InternalRules.isHaveOffValid(m_pWorkRuleParameters, m_ePeriodType, m_nPeriodLength, locale, sMessage);
		} else if (m_idWorkRuleTemplate.equals(WorkRuleTemplate.IDCONSECTIVE)) {
			return InternalRules.isConsecutiveValid(m_pWorkRuleParameters, m_ePeriodType, m_nPeriodLength, locale, sMessage);
		} else {
			return true;// throw new WorkRuleException("unknown Rule Template");
		}
	}

	private Collection getOverlappingPeriods(LocalDate dtStartDate, LocalDate dtEndDate, TimeZone tzOrg) {
		// returns a collection of LocalDate TimeSpans for rules with periods
		// each Period is a collection with 2 Date objects (start and end)
		if (tzOrg == null) {
			tzOrg = TimeZoneUtil.GMT_TIMEZONE;
		}
		if (m_adtPeriods == null) {
			m_adtPeriods = new ArrayList();
			if (dtEndDate.getTime().before(m_dtStartDate.getTime()) || dtStartDate.getTime().after(m_dtEndDate.getTime())) {
				return m_adtPeriods; // return an empty collection
			}

			Date startInOrgTimeZone = TimeZoneUtil.getMidNightDate(m_dtStartDate.getTime(tzOrg), tzOrg);
			LocalDate dtPeriodStart = new LocalDate(startInOrgTimeZone, tzOrg);
			LocalDate dtPeriodEnd = new LocalDate(startInOrgTimeZone, tzOrg);

			while (true) {
				if (m_ePeriodType == WORKDAY) {
					dtPeriodEnd.add(Calendar.DAY_OF_MONTH, m_nPeriodLength);
				} else if (m_ePeriodType == WORKWEEK) {
					dtPeriodEnd.add(Calendar.DAY_OF_MONTH, m_nPeriodLength * 7);
				} else if (m_ePeriodType == MONTH
						|| // QC95801
						m_ePeriodType == CalendarPeriod.CalendarUnit.Month28Day.getDBValue()
						|| m_ePeriodType == CalendarPeriod.CalendarUnit.Month29Day.getDBValue()
						|| m_ePeriodType == CalendarPeriod.CalendarUnit.Month30Day.getDBValue()
						|| m_ePeriodType == CalendarPeriod.CalendarUnit.Month31Day.getDBValue()) {
					dtPeriodEnd.add(Calendar.MONTH, m_nPeriodLength);
				} else if (m_ePeriodType == YEAR) {
					dtPeriodEnd.add(Calendar.YEAR, m_nPeriodLength);
				}

				if (dtPeriodStart.getTime().after(dtEndDate.getTime())
						|| dtPeriodStart.getTime().after(m_dtEndDate.getTime())) {
					break;
				}

				if (dtPeriodStart.getTime().before(dtEndDate.getTime())
						&& dtPeriodEnd.getTime().after(dtStartDate.getTime())) {
					// period overlaps pay period, add to Collection
					// System.out.println("*********overlapping period "
					// +dtPeriodStart.getTime() + " " + dtPeriodEnd.getTime());
					TimeSpan pTimeSpan = new TimeSpan();
					pTimeSpan.dtLocalSpanStart = dtPeriodStart;
					pTimeSpan.dtLocalSpanEnd = dtPeriodEnd;
					/*
					 * Incorrect conflict warning for assignment rule violation
					 * - ESR#4353724 should not use rule effectivity end to
					 * control the rule checking period. rule effecitivity start
					 * and end decide when the rule will apply rule checking
					 * period is up to alignmnet date and the period type(month
					 * or week), can not be cut by effectivity end here. if
					 * (pTimeSpan
					 * .dtLocalSpanEnd.getTime().after(m_dtEndDate.getTime()))
					 * pTimeSpan.dtLocalSpanEnd = m_dtEndDate; //truncate period
					 * if rule ends early
					 */
					m_adtPeriods.add(pTimeSpan);
				}
				dtPeriodStart = new LocalDate(dtPeriodEnd);
				dtPeriodEnd = new LocalDate(dtPeriodEnd);
			}
		}

		return m_adtPeriods;
	}

	@Override
	/**
	 * Claims that two WorkRules are equal if their priorities
	 * are equal.
	 * <p/>
	 * FIXME: This must have been convenient for somebody once, but it's likely
	 * to cause some pretty weird bugs for people doing things that ought to
	 * be reasonable.S

	public boolean equals(Object o) {
		if (this == o) {
			return true;
		}

		if (!(o instanceof WorkRule)) {
			return false;
		}

		WorkRule n = (WorkRule) o;
		return n.getPriority() == m_nPriority;
	}

		return 31 * 17 + m_nPriority;
	}

		if (m_nPriority < n.getPriority()) {
			return -1;
		} else if (m_nPriority > n.getPriority()) {
			return 1;
		} else {
			return 0;
		}
	}

	private ID getWorkRuleTemplateID(ComplexWorkRule cRule) {
		if (cRule.isOffRule()) {
			return WorkRuleTemplate.IDHAVEOFF;
		} else {
			return WorkRuleTemplate.IDWORK;
		}
	}

	private Collection getWorkRuleParameters(ComplexWorkRule cRule, Collection shifts) throws WorkRuleException {
		ArrayList params = new ArrayList();
		// get rule parameters from complex work rule attributes
		int nClause = getClauseValue(cRule);
		int nOrder = 1;
		int nWorkUnitType = cRule.getWorkUnit().getWorkUnitType().getDBValue();
		WorkRuleParameter param = new WorkRuleParameter(nOrder++, WorkRuleParameter.CLAUSE, getClauseValue(cRule));
		params.add(param);

		// nMin, nMax,
		params.add(new WorkRuleParameter(nOrder++, WorkRuleParameter.INT, cRule.getMinWorkUnitCount()));
		params.add(new WorkRuleParameter(nOrder++, WorkRuleParameter.INT, cRule.getMaxWorkUnitCount()));
		if (nClause == 1) {
			// nElement, WORKUNITTYPE Employee's time period unit of work is
			// day(1), day of week (2), partial weekend (4), full weekend (6)
			params.add(new WorkRuleParameter(nOrder++, WorkRuleParameter.ELEMENT, nWorkUnitType));
			// nPartOfDay
			params.add(new WorkRuleParameter(nOrder++, WorkRuleParameter.PARTOFDAY, cRule.getWorkUnit()
					.getTimeOfDayConstant()));
		}
		if (nClause == 2) {
			// nWeekDay, nPartOfDay
			params.add(new WorkRuleParameter(nOrder++, WorkRuleParameter.WEEKDAY, cRule.getWorkUnit().getDayOfWeek()
					.get0SundayCount()));
			params.add(new WorkRuleParameter(nOrder++, WorkRuleParameter.PARTOFDAY, cRule.getWorkUnit()
					.getTimeOfDayConstant()));
		}
		if (nClause == 3) {
			// idShift
			ID shiftID = cRule.getWorkUnit().getShiftID();
			if (shifts != null) {
				Shift shift = null;
				for (Iterator i = shifts.iterator(); i.hasNext();) {
					shift = (Shift) i.next();
					if (shift.getDEID().equals(shiftID)) {
						shiftID = shift.getID(); // convert from deid to sid
						break;
					}
				}
			}
			params.add(new WorkRuleParameter(nOrder++, WorkRuleParameter.SHIFT, shiftID));
		}

		return params;
	}

	private int getClauseValue(ComplexWorkRule cRule) {
		WorkUnitType workUnitType = cRule.getWorkUnit().getWorkUnitType();
		// Employee's time period unit of work is a shift (0), day(1), day of
		// week (2), partial weekend (5), full weekend (4) or hour(3)
		if (workUnitType == WorkUnitType.Shift) {
			return 3;
		}
		if (workUnitType == WorkUnitType.DayOfWeek) {
			return 2;
		}
		return 1;
	}

	public boolean isFairnessRule() {
		return this.m_complexWorkRule.getWorkUnitCountConstraint() == WorkUnitCountConstraint.Fairly
				|| this.m_complexWorkRule.getWorkUnitCountConstraint() == WorkUnitCountConstraint.FairlyStartingBetween
				|| this.m_complexWorkRule.getWorkUnitCountConstraint() == WorkUnitCountConstraint.FairlyEndingBetween;
	}
