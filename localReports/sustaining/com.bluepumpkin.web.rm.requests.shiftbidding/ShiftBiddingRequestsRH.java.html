package com.bluepumpkin.web.rm.requests.shiftbidding;

import com.witness.web.uif.system.RequestContext;
import com.bluepumpkin.web.rm.requests.RequestsRH;
import com.bluepumpkin.web.rm.keys.RmPageAction;

/**
 * File:         ShiftBiddingRequestsRH
 * Description:  Request Handler for the list of shift bidding requests
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       pbosin, David Su
 * @version      1.0
 * Created on July 29, 2002, 1:22 PM
 */
public class ShiftBiddingRequestsRH extends RequestsRH {

	/**
	 * Process Request
	 *
	 * @param context The Request Context
	 */
	public void processRequest(RequestContext context) throws Exception {
		// Process requested action
		if (isAction(context, RmPageAction.GO_TO_MY_BID_OPTION)) {
			processGoToMyBidOptions(context);
		} else {
			setNextScreenName(context, "MYSCHEDULE_SHIFTBIDDING_REQUESTS");
			super.processRequest(context);
		}
	}
}
