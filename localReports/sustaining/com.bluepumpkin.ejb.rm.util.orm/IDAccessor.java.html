package com.bluepumpkin.ejb.rm.util.orm;

import java.lang.reflect.InvocationTargetException;
import java.lang.reflect.Method;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;

class IDAccessor extends BaseAccessor {

	IDAccessor(Method m) {
		super(m);
	}

	@Override
	void setProperty(Object target, Object value) throws IllegalAccessException, InvocationTargetException {
		ID id = null;
		if (value instanceof String) {
			id = ID.fromString((String) value);
		} else if (value instanceof Integer) {
			id = ID.fromInt((Integer) value);
		} else if (value instanceof Long) {
			id = ID.fromLong((Long) value);
		} else if (value instanceof ID) {
			id = (ID) value;
		} else if (value != null) {
			throw new RmRuntimeException("ID is being set with a non compatible type " + value.getClass());
		}

		super.setProperty(target, id);
	}

}
