package com.bluepumpkin.ejb.rm.util.orm;

import java.lang.reflect.InvocationTargetException;
import java.lang.reflect.Method;

class BaseAccessor {

	Method method;

	BaseAccessor(Method m) {
		method = m;
	}

	void setProperty(Object target, Object value) throws IllegalAccessException, InvocationTargetException {
		method.invoke(target, value);
	}

	Object getProperty(Object target) throws IllegalAccessException, InvocationTargetException {
		return method.invoke(target);
	}

	Class<?> getGetterReturnType() {
		return method.getReturnType();
	}
}
