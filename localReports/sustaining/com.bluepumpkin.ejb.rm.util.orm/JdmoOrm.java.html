package com.bluepumpkin.ejb.rm.util.orm;

import java.util.Collection;
import java.util.List;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;
import com.bluepumpkin.ejb.rm.util.JdmoPreparedStatement;

public class JdmoOrm {

	private JdmoOrm() {

	}

	//Read from database
	public static <T> List<T> toObjects(Class<T> entityClass, String sql, Object... parameters) {
		Jdmo jdmo = null;
		try {
			jdmo = new Jdmo();
			return toObjects(entityClass, jdmo, sql, parameters);
		} finally {
			if (jdmo != null) {
				jdmo.cleanUp();
			}
		}
	}

	public static <T> List<T> toObjects(Class<T> entityClass, Jdmo dmo, String sql, Object... parameters) {
		try {
			JdmoRowset rs = JdmoPreparedStatement.createRowset(dmo, sql, parameters);
			return toObjects(entityClass, rs);
		} catch (JdmoException e) {
			throw new RmRuntimeException("In JdmoOrm.toObjects the call to createRowSet failed", e);
		}
	}

	public static <T> List<T> toObjects(Class<T> entityClass, JdmoRowset rs) {
		return JdmoOrmRead.toObjects(entityClass, rs);
	}

	//Write to database
	public static <T> void insert(Class<T> entityClass, Collection<T> objects) {
		insert(entityClass, objects, null);
	}

	public static <T> void insert(Class<T> entityClass, Collection<T> objects, WriteOptions options) {
		Jdmo jdmo = null;
		try {
			jdmo = new Jdmo();
			insert(jdmo, entityClass, objects, options);
		} finally {
			if (jdmo != null) {
				jdmo.cleanUp();
			}
		}
	}

	public static <T> void insert(Jdmo dmo, Class<T> entityClass, Collection<T> objects) {
		insert(dmo, entityClass, objects, null);
	}

	public static <T> void insert(Jdmo dmo, Class<T> entityClass, Collection<T> objects, WriteOptions options) {
		JdmoOrmWrite<T> writer = new JdmoOrmWrite<T>(entityClass, options);
		String tableName = getTableName(options);
		writer.insert(dmo, tableName, objects);
	}

	public static <T> void insertToTempTable(Jdmo dmo, Class<T> entityClass, Collection<T> objects, WriteOptions options) {
		JdmoOrmWrite<T> writer = new JdmoOrmWrite<T>(entityClass, options);
		String tempTableName = getTableName(options);
		writer.insertIntoTempTable(dmo, tempTableName, objects);
	}



	/*
	 * Merges objects which have an ID property into the database. The ID is assumed to be integer based
	 * For objects without a valid ID an ID is created using BP_NEXTKEY and the table name. 
	 * Then the objects are inserted into a temp table and merged with the destination table using a merge on the ID column 
	 */

	public static <T> void merge(Class<T> entityClass, Collection<T> objects) {
		merge(entityClass, objects, null);
	}

	public static <T> void merge(Class<T> entityClass, Collection<T> objects, WriteOptions options) {
		Jdmo jdmo = null;
		try {
			jdmo = new Jdmo();
			merge(jdmo, entityClass, objects, options);
		} finally {
			if (jdmo != null) {
				jdmo.cleanUp();
			}
		}
	}

	public static <T> void merge(Jdmo dmo, Class<T> entityClass, Collection<T> objects) {
		merge(dmo, entityClass, objects, null);
	}

	public static <T> void merge(Jdmo dmo, Class<T> entityClass, Collection<T> objects, WriteOptions options) {
		JdmoOrmWrite<T> writer = new JdmoOrmWrite<T>(entityClass, options);
		String tableName = getTableName(options);
		writer.merge(dmo, tableName, objects);
	}

	private static String getTableName(WriteOptions options) {
		return options == null ? null : options.tableName;
	}

}
