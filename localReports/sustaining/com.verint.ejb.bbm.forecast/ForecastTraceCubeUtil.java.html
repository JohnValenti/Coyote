package com.verint.ejb.bbm.forecast;

import java.util.Date;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmTimeSeriesException;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastTraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.workload.model.MediaType;

/**
 * A utility class to assist with the creation of ForecastTraceCubes.
 */
public class ForecastTraceCubeUtil {
	//A map which associates the trace types of the forecast data to a MediaType.  Used for creating and populating ForecastTraceCubes.
	private static final Map<MediaType, short[]> mediaTypeToMetaTraceCubeMap = new HashMap<MediaType, short[]>();
	
	static {
		mediaTypeToMetaTraceCubeMap.put(MediaType.IMMEDIATE, new short[] {Trace.CV, Trace.AHT, Trace.CV_VH});
		mediaTypeToMetaTraceCubeMap.put(MediaType.DEFERRED, new short[] {Trace.CV, Trace.AHT, Trace.CV_VH});
		mediaTypeToMetaTraceCubeMap.put(MediaType.OUTBOUND, new short[] {Trace.CRATE, Trace.AHT, Trace.RPCRATE, Trace.RPCAHT});
		mediaTypeToMetaTraceCubeMap.put(MediaType.PROJECT, new short[] {Trace.CV, Trace.AHT, Trace.CV_VH});
	}

	private ForecastTraceCubeUtil() {};
	
	/**
	 * Returns a ForecastTraceCube populated with zero values for the trace types associated with the given MediaType.
	 */
	public static ForecastTraceCube createZeroForecast(MediaType mediaType, ID queueId, Date start, Date end) throws BbmTimeSeriesException {
		short[] types = mediaTypeToMetaTraceCubeMap.get(mediaType);
		ForecastTraceCube result = new ForecastTraceCube(queueId, start, end, types);
		for (short type : types) {
			result.initTraceValue(type, 0);
		}
		return result;
	}
	
	/**
	 * Returns a ForecastTraceCube populated with zero values for the trace types associated with the MediaType
	 * of the given media ID.
	 */
	public static ForecastTraceCube createZeroForecast(ID mediaID, ID queueID, Date start, Date end) throws BbmTimeSeriesException {
		return createZeroForecast(MediaType.get(mediaID), queueID, start, end);
	}

	/**
	 * Returns a meta trace cube containing the trace types associated to the given MediaType.
	 */
	public static ForecastTraceCube getMetaTraceCubeForMediaType(MediaType mediaType) {
		return new ForecastTraceCube(mediaTypeToMetaTraceCubeMap.get(mediaType));
	}

	public static Map<ID, TraceCube> convertOutboundClientForecastIntoSavableFormat(
			Map<ID, TraceCube> forecast) throws BbmTimeSeriesException {
		Map<ID, TraceCube> ftc = new HashMap<ID, TraceCube>();
		for (Map.Entry<ID, TraceCube> entry : forecast.entrySet()) {
			ftc.put(entry.getKey(), convertOutboundClientForecastIntoSavableFormat(entry.getValue()));
		}
		return ftc;
	}

	/**
	 * Converts the trace types in a ClientForecastTraceCube (currently Trace.FORECASTING_CRATE,
	 * Trace.FORECASTING_AHT, Trace.FORECASTING_RPCRATE, Trace.FORECASTING_RPCAHT) to types suitable
	 * for being saved in the database (Trace.RATE, Trace.AHT, Trace.RPCRATE, Trace.RPCAHT)
	 */
	public static TraceCube convertOutboundClientForecastIntoSavableFormat(TraceCube forecast)
			throws BbmTimeSeriesException {
		TraceCube ftc = new ForecastTraceCube(forecast.getQueueID(), forecast.getRawStartDate(), forecast.getRawEndDate(),
				new short[] {Trace.CRATE, Trace.AHT, Trace.RPCRATE, Trace.RPCAHT});

		ftc.setTraceValue(Trace.CRATE, forecast.getTraceValueD(Trace.FORECASTING_CRATE),forecast.getRawStartDate());
		ftc.setTraceValue(Trace.AHT, forecast.getTraceValueD(Trace.FORECASTING_AHT), forecast.getRawStartDate());
		ftc.setTraceValue(Trace.RPCRATE, forecast.getTraceValueD(Trace.FORECASTING_RPCRATE), forecast.getRawStartDate());
		ftc.setTraceValue(Trace.RPCAHT, forecast.getTraceValueD(Trace.FORECASTING_RPCAHT), forecast.getRawStartDate());

		return ftc;
	}
}
