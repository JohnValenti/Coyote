package com.bluepumpkin.ejb.bbm.schedule.ejb;

/**
 * Title:        Blue Pumpkin Software Basic Business Model
 * Description:  Implements the cache whose lifetime is bounded by the duration
 *               of the EJB method call. The cache curently is used to access
 *               activities and employee names
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       Greg Fichtenholtz
 * @version 1.0
 */
import java.util.Collection;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Iterator;
import java.util.Map;
import java.util.Set;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.audit.ejb.EventAuditTrailManager;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbLogBundleKey;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManager;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeName;

class CachePerEJBMethod {
	private static ThreadLocal m_threadLocal = new ThreadLocal();

	/**
	 * it's okay to keep a reference to the EJB Beab instance here, because the
	 * lifetime of this object is limited to a single EJB method call
	 */
	private ScheduleAccessManagerEJB m_ejb;
	private HashMap<ID, EmployeeName> m_mapEmployeeNames = new HashMap<ID, EmployeeName>();
	private HashMap m_mapActivities = new HashMap();

	/**
	 * ctor - private so that only this class can create new instances
	 */
	private CachePerEJBMethod(ScheduleAccessManagerEJB ejb) {
		m_ejb = ejb;
	}

	/**
	 * This static method will be called to read/write contents of the cache
	 */
	static CachePerEJBMethod getCache() {
		return (CachePerEJBMethod) m_threadLocal.get();
	}

	/**
	 * This method HAS to be called at the beginning of each method in schedule
	 * manager EJB to initialize the cache. If it is not called, then some other
	 * pinstance of CachePerEJBMethod object will be associated with this
	 * execution thread an the client may get stale information from the cache
	 */
	static CachePerEJBMethod newCache(ScheduleAccessManagerEJB ejb) {
		CachePerEJBMethod cache = new CachePerEJBMethod(ejb);
		m_threadLocal.set(cache);
		return cache;
	}

	/* delete the cache */
	static void deleteCache() {
		m_threadLocal.set(null);
	}

	/**
	 * returns the identity (username) of the current user
	 */
	String getCallerIdentity() {
		return m_ejb.getCallerIdentify();
	}

	/**
	 * get the map of employee names given their IDs
	 */
	Map<ID, EmployeeName> getEmployeeNames(Collection<ID> listIDs) {
		Map<ID, EmployeeName> mapNames = new HashMap<ID, EmployeeName>();
		Set<ID> setNotInCache = new HashSet<ID>();

		for (Iterator<ID> it = listIDs.iterator(); it.hasNext();) {
			ID workResourceID = it.next();
			EmployeeName empName = m_mapEmployeeNames.get(workResourceID);
			if (empName == null) {
				setNotInCache.add(workResourceID);
			} else {
				mapNames.put(workResourceID, empName);
			}
		}

		try {
			/**
			 * if there are any names to be found, get them from work resource
			 * manager EJB
			 */
			if (!setNotInCache.isEmpty()) {
				HashMap mapNotInCache = m_ejb.getWorkResourceManager().getWorkResourceNamesByIDs(setNotInCache);

				/* update the cache (merge the two maps) */
				m_mapEmployeeNames.putAll(mapNotInCache);
				mapNames.putAll(mapNotInCache);
			}
		} catch (Exception e) {
			/* log the error */
			m_ejb.getCategory().l7dError(BbmEjbLogBundleKey.UNKNOWN_EXCEPTION_MF,
					new Object[] { "getEmployeeNames", "CachePerEJBMethod" }, e);
		}
		return mapNames;
	}

	/**
	 * This method will retrieve an activity given the activity ID
	 */
	Activity getActivity(ID activityID) {
		Activity activity = (Activity) m_mapActivities.get(activityID);
		try {
			if (activity == null) {
				activity = m_ejb.getActivityManager().findActivityById(activityID);
				m_mapActivities.put(activityID, activity);
			}
		} catch (Exception e) {
			/* log the error */
			m_ejb.getCategory().l7dError(BbmEjbLogBundleKey.UNKNOWN_EXCEPTION_MF,
					new Object[] { "getActivity", "CachePerEJBMethod" }, e);
		}
		return activity;
	}

	WorkResourceManager getWorkResourceManager() {
		return m_ejb.getWorkResourceManager();
	}

	EventAuditTrailManager getAuditTrailManager() {
		return m_ejb.getEventAuditTrailManager();
	}

	ScheduleEventHandler getEventHandler() {
		return m_ejb.getEventHandler();
	}
}
