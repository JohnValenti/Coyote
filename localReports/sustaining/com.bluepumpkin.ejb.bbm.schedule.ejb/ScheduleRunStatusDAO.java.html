package com.bluepumpkin.ejb.bbm.schedule.ejb;

import java.sql.Timestamp;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.schedule.model.ScheduleRunStatus;
import com.bluepumpkin.ejb.bbm.schedule.model.ScheduleRunStatusFields;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.core.base.MultiUserException;

public class ScheduleRunStatusDAO extends DAOBase<ScheduleRunStatus> {
	// meta data info.
	private static FieldInfo m_fieldInfo = new ScheduleRunStatusFields();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ScheduleRunStatusDAO() {
		super();
	}

	public ScheduleRunStatusDAO(Jdmo dmo) {
		super(dmo);
	}

	@Override
	public ScheduleRunStatus createValueObject() {
		return new ScheduleRunStatus();
	}

	/**
	 * This is an override of the base class the the SID column from SP is the
	 * FK column ID into the new table and is also the PK
	 */
	@Override
	protected boolean allowCreateWithId() {
		return true;
	}

	private boolean existScheduleRunStatus(ID sID) throws JdmoException {
		JdmoQuery query = null;
		JdmoRowset rs = null;
		String strSQL;
		try {

			strSQL = "SELECT 1 FROM SCHEDULERUNSTATUS WHERE ID = ?";
			query = m_dmo.createQuery(strSQL, Jdmo.PARAM_QUERY);
			query.setParID(1, sID);
			rs = m_dmo.createRowset(query, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			return rs.next();
		} finally {
			if (query != null) {
				query.close();
			}

			if (rs != null) {
				rs.close();
			}
		}
	}

	public void updateStatusForScheduleRun(ID sid, int status, String modifiedBy) throws BbmException {
		ScheduleRunStatus obj = null;
		boolean isExistScheduleRunStatus = false;
		try {
			isExistScheduleRunStatus = existScheduleRunStatus(sid);
		} catch (JdmoException e1) {
			throw new BbmException(e1);
		}

		try {
			if (isExistScheduleRunStatus) { // Update
				obj = getObjectByID(sid);
				obj.setRunStatus(status);
				if (modifiedBy != null) {
					// is the End-event
					obj.setModifiedBy(modifiedBy);
				}
				obj.setModifiedDate(new Timestamp(System.currentTimeMillis()));
				updateObject(obj);

			} else if (status == ScheduleRunStatus.STATUS_START) { // Only
																	// Create
																	// new row
																	// for
																	// Start-event
				obj = new ScheduleRunStatus();
				obj.setID(sid);
				obj.setRunStatus(status);
				if (modifiedBy != null) {
					obj.setModifiedBy(modifiedBy);
				}
				obj.setModifiedDate(new Timestamp(System.currentTimeMillis()));
				createObject(obj);
			}
		} catch (MultiUserException e) {
			throw new BbmException(e);
		} catch (BbmUpdateException e) {
			throw new BbmException(e);
		} catch (BbmCreateException e) {
			throw new BbmException(e);
		}
	}

}
