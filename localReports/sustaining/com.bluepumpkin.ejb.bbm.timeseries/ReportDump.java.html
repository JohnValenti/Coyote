package com.bluepumpkin.ejb.bbm.timeseries;

import java.rmi.RemoteException;

import javax.ejb.RemoveException;

import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.timeseries.ejb.TimeSeriesReportFacadeManager;

/*
 * Title:        Blue Pumpkin Software
 * Description:  ReportDump main class used by BPFX
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * Created:      Sep 30, 2003
 * @author       ssong
 * @version		 1.0
 */
public class ReportDump {
	public static void main(String[] args)
	{				
		try
		{
			TimeSeriesReportFacadeManager facadeManager = BbmManagerFactory.getTimeSeriesReportFacadeManager();			
			facadeManager.dumpTimeSeries();
			facadeManager.remove();
		}
		catch (com.bluepumpkin.ejb.bbm.BbmEJBCreateException e)
		{
			e.printStackTrace();
			System.exit(-1);
		}
		catch (RemoteException e)
		{
			e.printStackTrace();
			System.exit(-1);
		}
		catch (RemoveException e)
		{
			e.printStackTrace();
			System.exit(-1);
		}		
		catch (com.bluepumpkin.ejb.bbm.base.BbmCreateException e)
		{
			e.printStackTrace();
			System.exit(-1);
		}		
		System.exit(0);
	}
}
