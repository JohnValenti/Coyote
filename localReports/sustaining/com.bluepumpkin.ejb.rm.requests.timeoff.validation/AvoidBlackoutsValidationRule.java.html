package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.Iterator;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.timeoffcalculator.TimeMap;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOPool;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.RmUtil;
import com.bluepumpkin.ejb.rm.util.TORequestUtil;

/**
 * Rule in UI: Request avoids blackout days.
 * <p/>
 * Check that there are no blackout days during the requested period for the
 * employee's organization.
 * <p/>
 * <p/>
 * Applies equally for debit, debit-if and don't debit
 * <p/>
 * <p/>
 * Applies equally for tentative and non tentative (no unpublished schedule information used).
 */
public class AvoidBlackoutsValidationRule implements Validator {

	private static final String CLASS_NAME = AvoidBlackoutsValidationRule.class.getName();

	private static final Category LOG = Log.initCategory(CLASS_NAME);

	/**
	 * Validate a time off request.
	 */
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		TORequest tor = (TORequest) validatable;
		String methodName = "validate";
		
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		}

		if (!TORequestUtil.isUseIntervalTOPool(tor) && !TORequestUtil.isUseDailyTOPool(tor)) {
			//We shouldn't be checking this validator.
			return result;
		}
		
		// Check against blackout days
		// Check to see if the blackout days timemap is already in the cache
		TOChoice toChoice = tor.getTOChoiceForValidation();
		if (toChoice == null) {
			// Loop through the choices and check that each one contains some
			// time that is not part of a blackout day
			Iterator<TOChoice> it = tor.getRequestChoiceList().iterator();
			while (it.hasNext()) {
				toChoice = it.next();
				// The given range contains at least one blackout day.  Validation fails
				/*Do not validate Requests for past dates
								Sameet Feb 2009 V11 Feature*/
				if (isTimeOffChoiceForPastDate(tor, toChoice)) {
					LOG.debug("Requested Choice is for Past hence it is skipped;" + " Start date:" + toChoice.getStartDate() + ": enddate:" + toChoice.getEndDate());
					continue;
				}
				result = setValidationResultForBlackout(tor, toChoice);
			}
		} else {
			// validate one TO CHOICE ONLY
			// The given range contains at least one blackout day.  Validation fails
			if (isTimeOffChoiceForPastDate(tor, toChoice)) {
				LOG.debug("Requested Choice is for Past hence it is skipped;" + " Start date:" + toChoice.getStartDate() + ": enddate:" + toChoice.getEndDate());
				return null;
			}
			result = setValidationResultForBlackout(tor, toChoice);
		}
		return result;
	}

	protected boolean isTimeOffChoiceForPastDate(TORequest tor, TOChoice toChoice) {
		return TORequestUtil.isTimeOffChoiceForPastDate(tor, toChoice);
	}

	protected ValidationResult setValidationResultForBlackout(TORequest tor, TOChoice toChoice) throws Exception {
		String methodName = "setValidationResultForBlackout";
		ValidationResult result = null;
		
		// Need a Time-Off Pool to check for Blackout Days
		TOPool pTOPool = tor.getCache().getTOPool();
		if (pTOPool != null) {
			TimeMap boTm = tor.getCache().getBlackoutDaysTimeMap();
			if (boTm.onBitsOverlapRange(toChoice.getStartDate(), toChoice.getEndDate())) {
				result = ValidationUtil.setSoftValidationResult(toChoice, RmEjbBundleKey.TO_BLACKOUT_DAYS,
				        tor.getCache().getEmployeeNameByID(), toChoice.getStartDate(), toChoice.getEndDate(),
				        tor.getCache().getOrgNameForMsg(), CLASS_NAME);
			}
		}
		
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpExitMethod(methodName, result));
		}
		return result;
	}

}
