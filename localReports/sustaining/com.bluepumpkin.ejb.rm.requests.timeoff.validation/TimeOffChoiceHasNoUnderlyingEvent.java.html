	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		TORequest toReq = (TORequest) validatable;
		if (toReq.isApproved()) {
			TOChoice apprChoice = toReq.getApprovedChoice();
			CalendarEventAssignment eventPub = null;
			CalendarEventAssignment eventUnPub = null;
			try {
				eventPub = TORequestUtil.getCalEventAssignForTOChoiceAligned(toReq, apprChoice, true, null);
				if (eventPub == null) { //check if event exists in published world.

					eventUnPub = TORequestUtil.getCalEventAssignForTOChoiceAligned(toReq, apprChoice, false, null);
