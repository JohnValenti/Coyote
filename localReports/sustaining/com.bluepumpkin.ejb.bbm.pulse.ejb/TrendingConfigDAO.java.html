import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.pulse.model.TrendingConfig;
import com.bluepumpkin.ejb.bbm.pulse.model.TrendingConfigFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.core.base.MultiUserException;

/**
 * TrendingConfigDAO is not to be accessed by other objects directly
 * Use default package level restriction
 */
class TrendingConfigDAO extends DAOBase {

	private static FieldInfo m_fieldInfo = new TrendingConfigFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public TrendingConfigDAO() {
		 super();
	}

	public TrendingConfigDAO(Jdmo dmo) {
		 super(dmo);
	}

	protected ValueObjectBase createValueObject()
	{
		return (new TrendingConfig());
	}

	/** create a TrendingConfig */
	public ID createTrendingConfig(TrendingConfig objValue) throws BbmCreateException
	{
		// Convert the definition into String
		objValue.saveProcess();
		return createObject(objValue);
	}

	/**
	 *   get an SP's TrendingConfig definitions
	 *
	 *   @parameter ID, sp id
	 *
	 *   @return TrendingConfig object
	 */
	public TrendingConfig getTrendingConfig(ID spID) throws BbmFinderException
	{	
		StringBuffer strWhere = new StringBuffer(30);
		strWhere.append("A.SPID = ").append(spID);
		Collection configCol = getObjects(strWhere.toString());
		if (configCol.isEmpty())
			return null;
		return (TrendingConfig)configCol.iterator().next();
	}

	/**
	 * get one TrendingConfig object based on its ID
	 *
	 * @param ID, TrendingConfig ID
	 *
	 * @return TrendingConfig
	 */
	public TrendingConfig getTrendingConfigByID(ID id) throws BbmFinderException
	{		
		return (TrendingConfig)getObjectByID(id);
	}

	/**
	 * update one TrendingConfig object
	 * @param TrendingConfig
	 */
	public void updateTrendingConfig(TrendingConfig tv) throws MultiUserException, BbmUpdateException
	{
		if (tv != null) {
			tv.saveProcess();
			updateObject(tv);
		}
	}
}
