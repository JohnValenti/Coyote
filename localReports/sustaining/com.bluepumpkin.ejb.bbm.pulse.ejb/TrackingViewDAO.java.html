import java.util.Collection;
import java.util.Collections;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.pulse.model.TrackingView;
import com.bluepumpkin.ejb.bbm.pulse.model.TrackingViewFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**
 * TrackingViewDAO is not to be accessed by other objects directly
 * Use default package level restriction
 */
class TrackingViewDAO extends DAOBase {

	private static FieldInfo m_fieldInfo = new TrackingViewFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public TrackingViewDAO() {
		 super();
	}

	public TrackingViewDAO(Jdmo dmo) {
		 super(dmo);
	}

	protected ValueObjectBase createValueObject()
	{
		return (new TrackingView());
	}

	/** create a TrackingView */
	public ID createTrackingView(TrackingView objValue) throws BbmCreateException
	{
		// Convert the definition into String
		objValue.saveChartDef();
		return createObject(objValue);
	}

	/**
	 *   get an employee's TrackingView definitions
	 *
	 *   @parameter ID, employee id
	 *
	 *   @return a collection of TrackingView objects
	 */
	public Collection getTrackingViews(ID userID) throws BbmFinderException
	{
		if (userID == null) return Collections.EMPTY_LIST;

		StringBuffer strWhere = new StringBuffer(30);
		strWhere.append("A.BPUSERID = ").append(userID);
		strWhere.append(" or A.ISGLOBALVIEW =1");
		Collection col = getObjects(new String(strWhere));
		for (Iterator it = col.iterator(); it.hasNext(); ) {
			TrackingView tv = (TrackingView)it.next();
			// Next line is to populate the view definition
			tv.retrieveChart();
		}
		return col;
	}

	/**
	 * Return all the private (non-global) TrackingView objects which are NOT visible to the user.
	 * @param ID, user ID
	 * @return Collection, TrackingView objects
	 */
	public Collection getInvisibleTrackingViews(ID userID) throws BbmFinderException 
	{
		if (userID == null) return Collections.EMPTY_LIST;

		StringBuffer strWhere = new StringBuffer(30);
		strWhere.append("A.BPUSERID <> ").append(userID);
		strWhere.append(" and A.ISGLOBALVIEW =0");
		Collection col = getObjects(new String(strWhere));
		for (Iterator it = col.iterator(); it.hasNext(); ) {
			TrackingView tv = (TrackingView)it.next();
			// Next line is to populate the view definition
			tv.retrieveChart();
		}
		return col;
	}

	/**
	 * get one TrackView object based on its ID
	 *
	 * @param ID, TrackView ID
	 *
	 * @return TrackingView
	 */
	public TrackingView getTrackingViewByID(ID id) throws BbmFinderException
	{
		if (id == null) return null;
		TrackingView tv = (TrackingView)getObjectByID(id);
		// Next line is to populate the view definition
		tv.retrieveChart();
		return tv;
	}

	/**
	 * Application logic to prevent TrackingView  with same name in the same category
	 * 1. Global View, no model with same name
	 * 2. Employee View, no model with same name for the same employee nd global
	 * @param TrackingView
	 * @return boolean
	 */
	public boolean checkSameName(TrackingView tv) throws JdmoException {
		String name = tv.getViewName();
		if (name == null)
			return false;
		ID userID = tv.getUserID();
		String pStmt = "select ID from USERTRACEVIEW where NAME=?";
		JdmoQuery jQuery = null;
		if (!tv.isGlobal()) {
			pStmt = "select ID from USERTRACEVIEW where NAME=? and (BPUSERID=? or ISGLOBALVIEW=1)";
			jQuery = m_dmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			jQuery.setParID(2, userID);
		} else {
			jQuery = m_dmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
		}
		jQuery.setParString(1, name);
		JdmoRowset rs = m_dmo.createRowset(jQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
		if (rs.next()) {
			// during update, check if the Model with same name has the same ID
			if (tv.getID() != null) {
				ID tvID = rs.getID(1);
				if (tvID.equals(tv.getID())) {
					// same view ID, it is okay to update
					return false;
				}
			}
			return true;
		}
		return false;
	}
}
