import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.common.jdmo.*;
import com.bluepumpkin.common.localization.Localizer;
import com.witness.ejb.bbm.delete.DeleteException;
import com.witness.ejb.bbm.delete.DeleteableObjectPlugin;
import com.witness.ejb.bbm.delete.ejb.DAOUtil;
import com.witness.ejb.bbm.delete.model.*;
import com.bluepumpkin.ejb.facade.l10n.FacadeEjbBundleKey;


public class DeleteObjectiveDAO implements DeleteableObjectPlugin {
  
  /**
	 * 
	 */
	private static final long serialVersionUID = 1L;

public DeleteObjectiveDAO() {
  }
  
  /**
   * check if objective is linked to Scorecard Settings (ScorecardSetup table) by checking following link tables
   *       ScorecardSetup
   *
   * @return DeletionResult
   *    if Obj is linked to ScorecardSetup tables
   *       set ResultCode to CONFIRM_DELETE, and fill DeleteResult.m_colLinkedObjects
   *       with names of linked objective objects
   *    else
   *       set ResultCode to DELETED
   *
   */
  private DeletionResult checkLinkedObjects(ID objID, Jdmo dmo) throws Exception {
	DeletionResult dResult = new DeletionResult();
	Collection colLinkedObjects = new ArrayList();
	
	// check if link to ScorecardSetup
	StringBuffer strSQL = new StringBuffer();
	strSQL.append("select org.ID, org.NAME from ORGANIZATION org, SCORECARDSETUP st	where org.ID = st.ORGANIZATIONID AND st.DEFAULTOBJECTIVEID = ");
	strSQL.append(objID);

	DAOUtil.loadLinkedObjects(colLinkedObjects, new String(strSQL), BPObjectInfo.BPOBJECT_SCORECARDSETUP,
			FacadeEjbBundleKey.BUNDLE_NAME, FacadeEjbBundleKey.OBJECT_SCORECARDSETUP, dmo);
	
	if (colLinkedObjects.size() > 0) {
		  dResult.setResultCode(DeletionResult.CONFIRM_DELETE);
		  dResult.setLinkedObjects(colLinkedObjects);
		  return dResult;
	}
	
	// not linked to any object, marked it as deleted
	dResult.setResultCode(DeletionResult.DELETED);
	return dResult;
  }
  
  /**
   *  delete objective implements the following algorithms
   *  1) call checkLinkedObjects() to check if objective is linked to any other objects
   *  2) if objective is linked to campaign
   *        just return DeletionResult, since the deletion need user confirmation
   *     else, deleted it
   *
   */
  public DeletionResult deleteObject(ID idObjective,
		  Date dtDeleteOnDate, Localizer localizer, Object objCustomDefined)
		  throws DeleteException {
    
    Jdmo dmo = new Jdmo(false);
    try {
  	  DeletionResult dResult = checkLinkedObjects(idObjective, dmo);

	  switch (dResult.getResultCode()) {
		case DeletionResult.DELETED:
	      DAOUtil.deleteObjectCascade(idObjective, "OBJECTIVE", dmo);
		  break;
		default:
		  // do nothing if prohibit delete or need substitute
		  break;
	  }
      return dResult;
    }
    catch(Exception e) {
      throw new DeleteException(e);
    }
    finally {
      dmo.cleanUp();
    }
  }
  
  
  /**
   *  force delete objective implements the following algorithms
   *  Cascade delete it.
   *
   */
  public void forceDeleteObject(ID idObjective) throws DeleteException {
    Jdmo dmo = new Jdmo(false);
    try {
      DAOUtil.deleteObjectCascade(idObjective, "OBJECTIVE", dmo);
    }
    catch(Exception e) {
      throw new DeleteException(e);
    }
    finally {
      dmo.cleanUp();
    }
  }

	/**
	 * substitute the object with another object in object relationship
	 * 
	 * @param objID
	 *            the id of the object being substituted
	 * @param substituteObjID
	 *            the id of the substitute object
	 * @param dtSubstituteDate
	 *            the date where the substitution occurred
	 */
	public void substituteObject(ID objID, ID substituteObjID,
			Date dtSubstituteDate) throws DeleteException {
	}

	/**
	 * Converts an object type to a localized string
	 * 
	 * @param localizer The localizer
	 * @param isPlural Specify whether display is for multiple object 
	 * @return a localized object type string
	 */
	public String convertType(Localizer localizer, boolean isPlural) {
		String rbKey = isPlural ? FacadeEjbBundleKey.OBJECT_OBJECTIVE
				: FacadeEjbBundleKey.OBJECT_OBJECTIVE;
		return localizer.i18n(FacadeEjbBundleKey.BUNDLE_NAME, rbKey);
	}

}
