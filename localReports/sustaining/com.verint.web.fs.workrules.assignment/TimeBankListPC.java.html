package com.verint.web.fs.workrules.assignment;

import java.util.ArrayList;
import java.util.Calendar;
import java.util.Collection;
import java.util.Comparator;
import java.util.LinkedHashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.l10n.RegionalFormatBundleKey;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.vo.DBSortField;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.TimeBank;
import com.bluepumpkin.ejb.bbm.workrules.model.TimeBankSortField;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.OrganizationCampaignNameComparator;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.MultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.InsertableMultiColListPC;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;

public class TimeBankListPC extends InsertableMultiColListPC {

	private Collection<TimeBank> timeBanks;
	Map<ID, Organization> orgMap = null;
	
	public TimeBankListPC(RequestContext context, String name, Collection<TimeBank> timeBanks, Map<ID, Organization> orgMap) {
		super(context);

		setName(name);
		this.timeBanks = timeBanks;
		this.orgMap = orgMap;
		
		initHeader();
	}

	@Override
	public String getHtmlDisplay() {
		initRows();
		return super.getHtmlDisplay();
	}

	private void initHeader() {
		TableHeaderPC header = getHeader();
		Localizer localizer = m_context.getLocalizer();
		header.setPartialSortable(true);
		header.addColumnHeaderData(FsWebBundleKey.FS_NAME,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_NAME), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_START_DATE,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_START_DATE), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_END_DATE,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_END_DATE), true);
		header.addColumnHeaderData(FsWebBundleKey.SUMMARY_TB_TARGET_HOURS,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SUMMARY_TB_TARGET_HOURS), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_ORGANIZATION,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ORGANIZATION), true);
	}
	
	
	private void initRows() {
		Collection<MultiColumnNodeData> rows = super.getListModel();
		
		if (timeBanks != null) {
			for (TimeBank timeBank : timeBanks) {
				DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(timeBank.getID());
				
				//Name
				nodeData.add(timeBank.getName());
				
				//Time Banks are stored in the DB at GMT time and with an exclusive end date.  We need to subtract a day
				//from the end date for display purposes.
				//Start Date
				nodeData.add(m_localizer.formatDate(timeBank.getStartDate(), TimeZoneUtil.GMT_TIMEZONE, RegionalFormatBundleKey.DATE_FORMAT));			
				//End Date
				Calendar endDateDisplay = Calendar.getInstance(TimeZoneUtil.GMT_TIMEZONE);
				endDateDisplay.setTime(timeBank.getEndDate());
				endDateDisplay.add(Calendar.DAY_OF_YEAR, -1);
				nodeData.add(m_localizer.formatDate(endDateDisplay.getTime(), TimeZoneUtil.GMT_TIMEZONE, RegionalFormatBundleKey.DATE_FORMAT));
				
				//Target Hours
				nodeData.add(m_localizer.formatDurationInMins(timeBank.getTotalTargetMinutes()));
				
				//Organization
				if (timeBank.getOrganizationID() != null) {
					Organization o = orgMap.get(timeBank.getOrganizationID());
					if (o != null) {
						nodeData.add(o.getName());	
					} else {
						nodeData.add("");
					}
				} else {
					nodeData.add("");			
				}
				
				setRowAttributes(nodeData, timeBank);
				
				rows.add(nodeData);
			}
		}
	}
	
	protected void setRowAttributes(DefaultMultiColumnNodeData rowData, TimeBank timeBank) {
		rowData.setNodeAttribute("timeBankStartDate", Long.toString(timeBank.getStartDate().getTime()));
		rowData.setNodeAttribute("timeBankEndDate", Long.toString(timeBank.getEndDate().getTime()));
	}
	
	public LinkedHashMap<DBSortField, Comparator> getSortComparators() {
		LinkedHashMap<DBSortField, Comparator> comparators =
			new LinkedHashMap<DBSortField, Comparator>();
		for (TimeBankSortField sortField : getDBSortFieldIndices()) {
			switch (sortField) {
				case ID:
					comparators.put(TimeBankSortField.ID, null);
					break;
				case NAME:
					comparators.put(TimeBankSortField.NAME, null);
					break;
				case START_DATE:
					comparators.put(TimeBankSortField.START_DATE, null);
					break;
				case END_DATE:
					comparators.put(TimeBankSortField.END_DATE, null);
					break;
				case TOTAL_TARGET_HOURS:
					comparators.put(TimeBankSortField.TOTAL_TARGET_HOURS, null);
					break;
				case ORGANIZATION:
					comparators.put(TimeBankSortField.ORGANIZATION, new OrganizationCampaignNameComparator(getLocalizer()));
					break;
				default: break;
			}
		}
		
		return comparators;
	}
	
	public ArrayList<TimeBankSortField> getDBSortFieldIndices() {
		ArrayList<TimeBankSortField> retVal = new ArrayList<TimeBankSortField>();
		if (getHeader().getSortColumnIndexes() == null) {
			//Default sort index is by ID (chronological order in which object was created)
			retVal.add(TimeBankSortField.ID);
		} else {
			for (int sortField : getHeader().getSortColumnIndexes()) {
				switch (sortField) {
					case 0: 
						retVal.add(TimeBankSortField.NAME);
						break;
					case 1: 
						retVal.add(TimeBankSortField.START_DATE);
						break;
					case 3: 
						retVal.add(TimeBankSortField.END_DATE);
						break;
					case 4: 
						retVal.add(TimeBankSortField.TOTAL_TARGET_HOURS);
						break;
					case 5: 
						retVal.add(TimeBankSortField.ORGANIZATION);
						break;
					default: break;
				}
			}
		}
		return retVal;
	}
}
