package com.verint.web.fs.workrules.assignment;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.datatypes.Pair;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.web.fs.keys.FsJavaScriptFileID;
import com.witness.web.uif.pagecomponent.list.InsertableMultiColListPC;
import com.witness.web.uif.pagecomponent.picker.date.DatePickerPC;
import com.witness.web.uif.system.RequestContext;

/**
 * An InsertableMultiColListPC that handles lists of effectivity records.
 */
public class EffectivityMultiColListPC extends InsertableMultiColListPC {

	private static final String ID_METADATA_DELIMITER = "$";
	private DatePickerPC datePickerPC;
	
	public EffectivityMultiColListPC(RequestContext context) {
		super(context);
		
		setJSMediator(FsJavaScriptFileID.EFFECTIVITY_TABLE);
		datePickerPC = new DatePickerPC(context, "effectivityMultiColDatePickerPC");
	}

	/**
	 * Returns a the collection of IDs that have been removed from the list.  Does NOT return
	 * the associated date range metadata.
	 */
	public Collection<ID> getRemovedRowIDCollection() {
		Collection<String> idData = StringUtil.parseStringToCollection(m_removedRowIDs, ID_DELIMITER);
		Collection<ID> retVal = new ArrayList<ID>();
		for (String id : idData) {
			String[] metaData = StringUtil.parseStringToArray(id, ID_METADATA_DELIMITER);
			if (metaData.length > 0 && metaData[0].length() > 0) retVal.add(new ID(metaData[0]));
		}
		
		return retVal;
	}
	
	/**
	 * Returns the row IDs that have been removed along with the associated date range metadata.
	 * The date range represents the time period where the effectivity will no longer be active
	 * once the assignment is removed.
	 * Key - value object ID
	 * Value - Pair of localdates.  The first date represents the new end of the effective record.
	 * 		The second (optional) date represents the continuation of the effective record.
	 */
	public Map<ID, Pair<LocalDate, LocalDate>> getRemovedRowMetadata() {
		Map<ID, Pair<LocalDate, LocalDate>> retVal = new HashMap<ID, Pair<LocalDate, LocalDate>>();

		Collection<String> idDataCol = StringUtil.parseStringToCollection(m_removedRowIDs, ID_DELIMITER);
		for (String idData : idDataCol) {
			String[] metaData = StringUtil.parseStringToArray(idData, ID_METADATA_DELIMITER);
			switch(metaData.length) {
				case 0:	//Shouldn't happen, this means a blank value was in the ID list
					break;
				case 1:	//ID only, this means the assignment was removed for the current period
					if (metaData[0].length() > 0) {
						ID id = new ID(metaData[0]);
						retVal.put(id, null);
					}					
					break;
				case 2:	//ID + start of effective end date, this means we are ending the effective record on this date
					if (metaData[0].length() > 0) {
						ID id = new ID(metaData[0]);
						LocalDate effectiveEndDate = datePickerPC.convertStrToLocalDate(metaData[1]);
						if (effectiveEndDate != null) {
							retVal.put(id, new Pair<LocalDate, LocalDate>(effectiveEndDate, null));
						}
					}
					break;
				case 3:	//ID + start & end dates, this means we are ending the effective record on the start date and
						//reinstating it on the continuation date
					if (metaData[0].length() > 0) {
						ID id = new ID(metaData[0]);
						LocalDate effectiveEndDate = datePickerPC.convertStrToLocalDate(metaData[1]);
						LocalDate effectiveContinuationDate = datePickerPC.convertStrToLocalDate(metaData[2]);
						if (effectiveEndDate != null && effectiveContinuationDate != null) {
							retVal.put(id, new Pair<LocalDate, LocalDate>(effectiveEndDate, effectiveContinuationDate));
						}
					}
					break;
				default:	//Don't do anything if we hit an unexpected case
					break;
			}
		}
		
		return retVal;
	}
}
