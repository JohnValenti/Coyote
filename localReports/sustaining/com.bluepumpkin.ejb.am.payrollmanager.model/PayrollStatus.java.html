package com.bluepumpkin.ejb.am.payrollmanager.model;

import java.util.*;
import java.io.Serializable;

import com.bluepumpkin.common.datatypes.*;
/**
 * Title:        Blue Pumpkin Software Foundation
 * Description:  PayrollStatus, capture each payroll post/export request info
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */

public class PayrollStatus implements Serializable
{
	public static int PENDING = 0;
	public static int COMPLETED = 1;
	public static int FAILED = 2;

	private ID m_RequestID;
	// Which user initiate the request
	private ID m_UserID;
	// User name
	private String m_UserName;
	// Which PayPeriod is used to export/post
	private ID m_PayPeriodID;
	// PayPeriod start date
	private LocalDate m_PayPeriodStartDate;
	// PayPeriod end date
	private LocalDate m_PayPeriodEndDate;
	// The PayPeriodType name
	private String m_PayPeriodType;
	// The payroll is exported/posted to which DataSource
	private ID m_DataSourceID;
	// The PayrollProviderDataSource name
	private String m_DataSourceName;
	// Employees are included in this export/post request
	private Collection m_Employee;
	// Number of Employees included in this export/post request
	private int m_EmployeeNumber;
	// The file to be exported/posted
	private String m_DestinationFileLocation;
	// When the request is initiated
	private Date m_RequestTime;
	// Request is for Post/Export
	private boolean m_IsForPost;
	// The status of this request
	private int m_Status;
	// Container for the Error messages
	private Collection m_Error;

	public PayrollStatus() {
		super();
	}
	/**
	 * Constructor for creation the status into DB
	 */
	public PayrollStatus(ID userID, ID payperiodID, ID datasourceID,
						 String destination, boolean forPost,
						 Collection employees)
	{
		m_UserID = userID;
		m_PayPeriodID = payperiodID;
		m_DataSourceID = datasourceID;
		m_DestinationFileLocation = destination;
		m_IsForPost = forPost;
		m_Employee = employees;
		m_RequestTime = new Date();
		m_Status = PENDING;
	}
	/**
	 * Constructor for backend load
	 */
	public PayrollStatus(ID id, ID userID, ID payperiodID, ID datasourceID,
						 String destination, Date time, boolean forPost,
						 int status)
	{
		m_RequestID = id;
		m_UserID = userID;
		m_PayPeriodID = payperiodID;
		m_DataSourceID = datasourceID;
		m_DestinationFileLocation = destination;
		m_RequestTime = time;
		m_IsForPost = forPost;
		m_Status = status;
	}

	public void setEmployee(Collection col) { m_Employee = col; }
	public Collection getEmployee() { return m_Employee; }
	public int getEmployeeNumber() { return m_EmployeeNumber; }
	public void setEmployeeNumber(int number) { m_EmployeeNumber = number; }

	public ID getRequestID() { return m_RequestID; }
	public void setRequestID(ID id) { m_RequestID = id; }
	public ID getUserID() { return m_UserID; }
	public String getUserName() { return m_UserName; }
	public void setUserName(String name) { m_UserName = name; }

	public ID getPayPeriodID() { return m_PayPeriodID; }
	public LocalDate getPayPeriodStartDate() { return m_PayPeriodStartDate; }
	public void setPayPeriodStartDate(LocalDate date) { m_PayPeriodStartDate = date; }
	public LocalDate getPayPeriodEndDate() { return m_PayPeriodEndDate; }
	public void setPayPeriodEndDate(LocalDate date) { m_PayPeriodEndDate = date; }

	public String getPayPeriodType() { return m_PayPeriodType; }
	public void setPayPeriodType(String name) { m_PayPeriodType = name; }
	public Date getRequestTime() { return m_RequestTime; }
	public boolean isForPost() { return m_IsForPost; }
	public int getStatus() { return m_Status; }
	public String getDestination() { return m_DestinationFileLocation; }

	public ID getDataSourceID() { return m_DataSourceID; }
	public String getDataSourceName() { return m_DataSourceName; }
	public void setDataSourceName(String name) { m_DataSourceName = name; }

	public Collection getErrorMessages() { return m_Error; }
	public void setErrorMessages(Collection error) { m_Error = error; }
}
