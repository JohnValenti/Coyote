package com.bluepumpkin.ejb.am.payrollmanager.model;

import java.io.Serializable;

public class PayrollProvider implements Serializable
{
	private ID m_PayrollProviderID;
	private String m_Name;
	private String m_ConfigFileLocation;
	private String m_FormatFileLocation;
	private String m_Description;
	private boolean m_IsConfigured = false;
	public PayrollProvider() {
		super();
	}
	// Constrcutor for the backend, this value object is basically read-only
	public PayrollProvider(ID id, String name, String location1, String location2, String desc) {
		m_PayrollProviderID = id;
		m_Name = name;
		m_ConfigFileLocation = location1;
		m_FormatFileLocation = location2;
		m_Description = desc;
	}
	// Constrcutor for the backend, this value object is basically read-only
	public PayrollProvider(String name, String location, String desc) {
		m_Name = name;
		m_ConfigFileLocation = location;
	}

	public String getConfigFileLocation() { return m_ConfigFileLocation; }
/*    public void setConfigFileLocation(String location) {
		m_ConfigFileLocation = location;
	}
*/
	public String getFormatFileLocation() { return m_FormatFileLocation; }
	public String getProviderName() { return m_Name; }

	public ID getID() { return m_PayrollProviderID; }
	public void setID(ID id) { m_PayrollProviderID = id; }

	public String getDescription() { return m_Description; }
	public void setDescription(String desc) { m_Description = desc; }

	// Helper function to denote if it is configured or not
	public boolean isConfigured() { return m_IsConfigured; }
	public void setIsConfigured(boolean value) { m_IsConfigured = value; }
}
