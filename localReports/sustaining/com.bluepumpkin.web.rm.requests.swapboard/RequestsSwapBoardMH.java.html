package com.bluepumpkin.web.rm.requests.swapboard;

import java.util.*;
import java.rmi.RemoteException;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.*;
import com.bluepumpkin.ejb.core.base.MultiUserException;
import com.bluepumpkin.ejb.rm.base.*;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.requests.swap.posting.ejb.*;
import com.bluepumpkin.ejb.rm.requests.swap.posting.model.*;
import com.bluepumpkin.web.rm.requests.RequestsMH;
import com.bluepumpkin.web.bbm.employee.EmployeeModelHandler;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        RequestsSwapBoardMH
 * Description:  Model Handler for Swap Board
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author       Zhibo Wang, David Su
 * @version 1.0
 */
public class RequestsSwapBoardMH extends RequestsMH {

	/**
	 * Return Shift Swap Posting Manager
	 */
	private static ShiftSwapPostingManager getShiftSwapPostingManager() throws BbmCreateException {
		return RmManagerFactory.getInstance().getShiftSwapPostingManager();
	}

	/**
	 * Return Employee Map for specified Shift Swap Posting
	 */
	public static Map getEmployeeNameMap(RequestContext context, ShiftSwapPosting posting)
			throws RemoteException, BbmException {

		ArrayList empIDList = new ArrayList(1);
		ID empID = posting.getEmployeeID();
		empIDList.add(empID);

		return EmployeeModelHandler.getEmployeeNamesByIDs(context, empIDList);
	}

	/**
	 * Return Collection of Unique Employee IDs for specified collection of
	 * Shift Swap Postings
	 */
	public static Collection<ID> getUniqueEmpIDsForPostings(Collection<ShiftSwapPosting> postings) {
		if (postings == null || postings.isEmpty()) {
			return Collections.emptyList();
		}

		Set<ID> empIDSet = new HashSet<ID>(postings.size());
		for (ShiftSwapPosting posting : postings) {
			ID empID = posting.getEmployeeID();
			empIDSet.add(empID);
		}

		return empIDSet;
	}

	/**
	 * Return All Postings
	 */
	public static SSPostingCollection getAllPostings(ID empID, int chunkSize, boolean fetchPartial)
			throws RemoteException, BbmFinderException, BbmCreateException, RmHardValidationException {
		return getShiftSwapPostingManager().findAllShiftSwapPostings(empID, ShiftSwapPosting.DL_ALL, chunkSize, fetchPartial);
	}

	/**
	 * Return My Postings
	 */
	public static SSPostingCollection getMyPostings(ID empID, int chunkSize, boolean fetchPartial)
			throws RemoteException, BbmFinderException, BbmCreateException, RmHardValidationException {
		return getShiftSwapPostingManager().findShiftSwapPostingsByEmployee(empID, ShiftSwapPosting.DL_ALL, chunkSize, fetchPartial);
	}

	/**
	 * Return My Organization Postings 
	 */
	public static SSPostingCollection getMyOrgPostings(ID empID, int chunkSize, boolean fetchPartial)
			throws RemoteException, BbmException, RmHardValidationException {
		return getShiftSwapPostingManager().findShiftSwapPostingsForMyOrg(empID, ShiftSwapPosting.DL_ALL, chunkSize, fetchPartial);
	}

	/**
	 * Return My Campaign Postings
	 */
	public static SSPostingCollection getMyCampaignPostings(ID empID, int chunkSize, boolean fetchPartial)
			throws RemoteException, BbmFinderException, BbmCreateException, RmHardValidationException {
		return getShiftSwapPostingManager().findShiftSwapPostingsForMyCampaign(empID, ShiftSwapPosting.DL_ALL, chunkSize, fetchPartial);
	}

	/**
	 * Return Postign for specified posting IDs
	 */
	public static Collection getPostings(Collection postingIDs)
			throws RemoteException, BbmFinderException, BbmCreateException, RmHardValidationException {
		List postingIDList = new ArrayList(postingIDs);
		return getShiftSwapPostingManager().findNextShiftSwapPostings(postingIDList, ShiftSwapPosting.DL_ALL);
	}

	/**
	 * Return ShiftSwapPosting By ID
	 */
	public static ShiftSwapPosting getPostingByID(ID objId)
			throws RemoteException, BbmFinderException, BbmCreateException, RmHardValidationException {
		return getShiftSwapPostingManager().findShiftSwapPostingById(objId,  ShiftSwapPosting.DL_ALL);
	}

	/**
	 * Create ShiftSwapPosting
	 */
	public static ID createPosting(ShiftSwapPosting obj)
			throws RemoteException, BbmCreateException, RmHardValidationException {
		return getShiftSwapPostingManager().createShiftSwapPosting(obj);
	}

	/**
	 * Update ShiftSwapPosting
	 */
	public static void updatePosting(ShiftSwapPosting obj)
			throws RemoteException, BbmUpdateException, MultiUserException, BbmCreateException, RmHardValidationException {
		getShiftSwapPostingManager().updateShiftSwapPosting(obj);
	}

	/**
	 * Delete ShiftSwapPosting
	 */
	public static void deletePosting(ID objID)
			throws RemoteException, BbmRemoveException, BbmFinderException, BbmCreateException, RmHardValidationException {
		getShiftSwapPostingManager().deleteShiftSwapPosting(objID);		
	}

	/**
	 * Delete ShiftSwapPostings
	 */
	public static void deletePostings(Collection objIDs)
			throws RemoteException, BbmRemoveException, BbmFinderException, BbmCreateException, RmHardValidationException  {
		getShiftSwapPostingManager().deleteShiftSwapPostings(objIDs);
	}	
}
