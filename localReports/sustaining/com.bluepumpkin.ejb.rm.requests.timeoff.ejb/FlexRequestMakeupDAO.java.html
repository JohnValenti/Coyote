package com.bluepumpkin.ejb.rm.requests.timeoff.ejb;

import java.util.Collection;

/**
 * DAO class for FlexRequestMakeup
 */
public class FlexRequestMakeupDAO extends DAOBase<FlexRequestMakeup> {

	private static FieldInfo m_fieldInfo = new FlexRequestMakeupFieldInfo();
	private static Category m_cat = Log.initCategory(FlexRequestMakeupDAO.class.getName());

	/**
	 * Constructor
	 */
	public FlexRequestMakeupDAO() {
		super();
	}

	/**
	 * Constructor
	 * @param - Jdmo - Jdmo object instance
	 */
	public FlexRequestMakeupDAO(Jdmo dmo) {
		super(dmo);
	}

	protected FieldInfo getFieldInfo() {
		return FlexRequestMakeupDAO.m_fieldInfo;
	}

	protected Category getCategory() {
		return FlexRequestMakeupDAO.m_cat;
	}

	/**
	 * Creates the value object.
	 *
	 * @return - returns new Value object
	 * @see com.bluepumpkin.ejb.bbm.vo.ValueObjectBase
	 * @see com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexRequestMakeup
	 */
	protected FlexRequestMakeup createValueObject() {
		return (new FlexRequestMakeup());
	}

	/**
	 * Finds FlexRequestMakeup for the request id object passed as argument.
	 *
	 * @param - ID - ID object for FlexRequestMakeup.
	 * @return - FlexRequestMakeup Object.
	 * @throw BbmFinderException
	 */
	public FlexRequestMakeup getFlexRequestMakeupById(ID pFlexRequestMakeupId) throws BbmFinderException {
		return (FlexRequestMakeup) getObjectByID(pFlexRequestMakeupId);
	}

	/**
	 * Finds all flex request makeups for the Time off request id passed as argument.
	 * @param - ID - ID object for TimeOffRequest.
	 * @return - Collection of FlexRequestMakeup Object.
	 * @see FlexRequestMakeup.
	 * @throw BbmFinderException
	 */
	public Collection<FlexRequestMakeup> findFlexRequestMakeups(ID pTimeOffRequestID) throws BbmFinderException {
		return (Collection<FlexRequestMakeup>) getObjects(getWhereClauseForFindFlexRequestMakeups(pTimeOffRequestID));
	}

	/**
	 * Returns the contents of the SQL WHERE clause that is used to fetch the event templates for a given set of work
	 * resources in a given time period
	 */
	protected String getWhereClauseForFindFlexRequestMakeups(ID pTimeOffRequestID) {

		StringBuilder strSQL = new StringBuilder(200);
		strSQL.append(" WHERE ");
		strSQL.append(FlexRequestMakeupFieldInfo
				.getColumnName(FlexRequestMakeupFieldInfo.FLEXREQUESTMAKEUP_I_TIMEOFFREQUESTID));
		strSQL.append(" = ");
		strSQL.append(pTimeOffRequestID.toString());
		strSQL.append(" ORDER BY ");
		strSQL.append(FlexRequestMakeupFieldInfo.getColumnName(FlexRequestMakeupFieldInfo.FLEXREQUESTMAKEUP_D_STARTTIME))
				.append(" ");
		return strSQL.toString();
	}

	/**
	 * Create new FlexRequestMakeup the FlexRequestMakeup object passed as argument.
	 * @param - pFlexRequestMakeup - FlexRequestMakeup object for one flex request makeup day.
	 * @return - ID object for created FlexRequestMakeup.
	 * @throw BbmCreateException
	 */
	public ID createFlexRequestMakeup(FlexRequestMakeup pFlexRequestMakeup) throws Exception {
		checkFlexRequestMakeupValid(pFlexRequestMakeup);

		// Call base class method to create object
		ID findFlexRequestMakeupId = createObject(pFlexRequestMakeup);
		return findFlexRequestMakeupId;
	}

	/**
	 * Add validity checking for fields in FlexRequestMakeup 
	 * @param pFlexRequestMakeup
	 * @throws RmException
	 */
	protected void checkFlexRequestMakeupValid(FlexRequestMakeup pFlexRequestMakeup) throws RmException {
		// -- Make sure request id is specified
		if (pFlexRequestMakeup.getTORequestID() == null) {
			throw RequestUtil.createRmException(RmEjbLogBundleKey.REQUEST_ID_NULL_FOR_FLEX, m_cat);
		}

		// -- Make sure SP id is specified
		if (pFlexRequestMakeup.getSPID() == null) {
			throw RequestUtil.createRmException(RmEjbLogBundleKey.SP_ID_NULL_FOR_FLEX, m_cat);
		}

		// -- Make sure start/end times are specified and end is not before start
		Date start = pFlexRequestMakeup.getStartTime();
		Date end = pFlexRequestMakeup.getEndTime();
		if ((start == null) || (end == null) || (end.before(start))) {
			throw RequestUtil.createRmException(RmEjbLogBundleKey.INVALID_TIME_RANGE_FOR_FLEX, start, end, m_cat);
		}

		// -- Make sure shiftstart/shiftend times are specified and end is not before start
		start = pFlexRequestMakeup.getShiftStartTime();
		end = pFlexRequestMakeup.getShiftEndTime();
		if ((start == null) || (end == null) || (end.before(start))) {
			throw RequestUtil.createRmException(RmEjbLogBundleKey.INVALID_SHIFT_TIME_RANGE_FOR_FLEX, start, end, m_cat);
		}

		// -- Make sure SHIFTASSIGNMENTID is specified
		if ((pFlexRequestMakeup.isFieldNull(FlexRequestMakeupFieldInfo.FLEXREQUESTMAKEUP_I_SHIFTASSIGNMENTID))
				|| (pFlexRequestMakeup.getShiftAssignmentID() == null)) {
			throw RequestUtil.createRmException(RmEjbLogBundleKey.SHIFTASSIGNMENTID_NULL_FOR_FLEX, m_cat);
		}
	}

	/**
	 * Update database for the FlexRequestMakeup object passed as argument.
	 * @param - pFlexRequestMakeup - FlexRequestMakeup object for flex time request makeup.
	 * @throw BbmUpdateException
	 */
	public void updateFlexRequestMakeup(FlexRequestMakeup pFlexRequestMakeup) throws Exception {
		updateObject(pFlexRequestMakeup);
	}

	/**
	 * Delete row in the FLEXREQUESTMAKEUP database table for the id passed as argument.
	 * @param - pFlexRequestMakeupId - ID object for the FlexRequestMakeup row.
	 * @see FlexRequestMakeup.
	 * @throw BbmRemoveException
	 */
	public void deleteFlexRequestMakeup(ID pFlexRequestMakeupId) throws Exception {
		deleteObject(pFlexRequestMakeupId);
	}
}
