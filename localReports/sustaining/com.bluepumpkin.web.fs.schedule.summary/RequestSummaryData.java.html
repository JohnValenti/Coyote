package com.bluepumpkin.web.fs.schedule.summary;

/**
 * Title:        RequestSummaryData
 * Description:  Base Class for Request Summary Data
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, DSu
 * @version      1.0
 *
 * Created on Nov 19, 2003, 5:29:29 PM
 */
public class RequestSummaryData {
	private int m_requestsTotal;
	private int m_requestsExpired;
	private int m_requestsPending;
	private int m_requestsApproved;
	private int m_requestsDenied;
	private int m_requestsInvalid;

	/**
	 * loaded constructor
	 */
	public RequestSummaryData(
			int requestsTotal,
			int requestsExpired,
			int requestsPending,
			int requestsApproved,
			int requestsDenied,
			int requestsInvalid) {
		m_requestsTotal = requestsTotal;
		m_requestsExpired = requestsExpired;
		m_requestsPending = requestsPending;
		m_requestsApproved = requestsApproved;
		m_requestsDenied = requestsDenied;
		m_requestsInvalid = requestsInvalid;
	}

	public int getRequestsTotal(){
		return m_requestsTotal;
	}

	public int getRequestsExpired() {
		return m_requestsExpired;
	}

	public int getRequestsPending() {
		return m_requestsPending;
	}

	public int getRequestsApproved() {
		return m_requestsApproved;
	}

	public int getRequestsDenied() {
		return m_requestsDenied;
	}

	public int getRequestsInvalid() {
		return m_requestsInvalid;
	}
}
