/**
 * Time-off Accrual Calculator Popup Page
 * @author rsun
 */
public class TOAccrualCalculatorPopupPM extends PopupWindowPM {
	private static Category m_cat = Log.initCategory(TOAccrualCalculatorPopupPM.class.getName());

	private final ResourceBundle m_bundle;
	private ID m_empID = null;
	private ID m_activityID = null;
	private DatePickerPC m_asOfDatePicker = null;
    private FormTwoColumnPC m_formPC = null;
    
	private Collection m_toActivities = Collections.EMPTY_LIST;
	private TOAccrual m_toAccrual = null;
	private Calendar m_toYearStart = null;
	private TimeOffActivitySummary m_toYearSummary = null;
	private boolean m_isAgentPage = true; //the default is true to be on the safe side

	public TOAccrualCalculatorPopupPM(RequestContext context) {
		super(context);
		m_bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);
		
		m_asOfDatePicker = new DatePickerPC(m_context, "asOfDate");
		addChildComponent(m_asOfDatePicker);
		m_formPC = new FormTwoColumnPC(m_context);
        //m_formPC.setHasRowHeaders(true); //508 TBD
        
		addChildComponent(m_formPC);
	}

	/**
	 * Return Page Model Instance
	 */
	public static PageModel getInstance(RequestContext context) {
		return getInstance(context, TOAccrualCalculatorPopupPM.class);
	}

	/**
	 * Initialize Model with Data
	 */
	protected void initializeModel() {
		if (m_isInitialized) return; else m_isInitialized = true;

		try {
			loadData();
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		} finally {
			initContentTitle();
			initScreenContent();
			initToolbar();
		}
	}

	protected void loadData() throws Exception {
		if (m_empID == null)//This should never happen
			return;
		
		//Show employeeName. (For improper empID, EmployeeModelHandler will throw NotFound Exception)
		String employeeName = m_localizer.formatName(
				EmployeeModelHandler.getEmployeeNameByID(m_context, m_empID));
		getContentTitle().setItemName(employeeName);
		
		//Get m_toActivities, based on m_empID
		ID orgID = OrganizationModelHandler.getEmployeeOrgID(m_context, m_empID);
		Collection timeOffTypes = RmManagerFactory.getInstance().getTimeOffRequestManager().getTimeOffTypes(orgID);
		m_toActivities = new ArrayList();
		for (Iterator it = timeOffTypes.iterator(); it.hasNext();) {
			Activity act = (Activity) it.next();
			if (act.isTimeoff()&&act.isTimeoffWithAllotment() )
				m_toActivities.add(act);
		}
		
		if (m_activityID != null) {
			Date asOfDate = m_asOfDatePicker.getDate();
			try {
				//m_toAccrual
				m_toAccrual = ScheduleSummaryMH.calculateAccruedTO(m_empID, m_activityID, asOfDate);
				
				//m_toYearStart & m_toYearSummary
				Calendar asOfCal = Calendar.getInstance(m_context.getViewingTimeZone());
				asOfCal.setTime(asOfDate);
				m_toYearStart = ScheduleSummaryMH.getYearStart(m_context, m_empID, asOfCal,
						FsKeys.TO_YEAR_CURRENT);
				Collection toSummaryData = ScheduleSummaryMH.getTimeOffSummary(m_empID, m_toYearStart,m_activityID, m_isAgentPage);
				for (Iterator it = toSummaryData.iterator(); it.hasNext();) {
					TimeOffActivitySummary summary = (TimeOffActivitySummary) it.next();
					if (m_activityID.equals(summary.getActivityID()))
						m_toYearSummary = summary;
				}
			} catch (RmException ex) {
				this.addPageMessage(Message.ERROR_TYPE, ex);
			}
		}
	}
    
	protected void initScreenContent() {
		m_formPC.setLastRowType(FormTwoColumnPC.LAST_ROW_NONE);
		m_formPC.setCssClasses("formRowDark40", "formRowLight60");
		
		m_asOfDatePicker.setLowerLimit(new Date());//no earlier than now
        String sLabel = HtmlUtil.createLabel(i18n(m_bundle, FsWebBundleKey.TO_ACCRUED_AS_OF), m_asOfDatePicker.getDateInputID());
		m_formPC.addRow(sLabel, m_asOfDatePicker);
		
		Collection options = new ArrayList();
		for (Iterator it = m_toActivities.iterator(); it.hasNext();) {
			Activity act = (Activity) it.next();
			options.add(new StringsPair(act.getID().toString(), act.getName()));
		}
		String selected = m_activityID == null ? null : m_activityID.toString();
        sLabel = HtmlUtil.createLabel(i18n(m_bundle, FsWebBundleKey.TO_SUMMARY_TO_TYPE), "activityID");
		m_formPC.addRow(sLabel, HtmlUtil.makeSelectInput("activityID", selected, "", options) );
		
		// Show the calculation results from m_toAccrual
		if (m_toAccrual != null && validateTOAccrual(m_toAccrual)) {
			String hoursLable = " " + i18n(m_common_bundle, UIFWebBundleKey.HOURS);
			double accruedHours = m_toAccrual.getEstimatedAccrHrs();
			String hourString = m_localizer.formatNumber(accruedHours, 2) + hoursLable;
			if (accruedHours < 0d)//Show red for negative Hours
				hourString = "<font color=red>" + hourString + "</font>";
			m_formPC.addRow(i18n(m_bundle, FsWebBundleKey.ACCRUED_TIME_OFF), hourString);
			
			ResourceBundle bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);
			String accrualSchedule = null;
			SortedSet toAccrualDetailSet = m_toAccrual.getToAccrualDetailSet();
			if (toAccrualDetailSet != null && toAccrualDetailSet.size() > 0) {
				TOAccrualDetailRow row = (TOAccrualDetailRow) toAccrualDetailSet.last();
				if (row != null) {
					String rate = m_localizer.formatNumber(row.getAccrualRate(), 2) + hoursLable;
					String year = String.valueOf(row.getYear());
					String schedule = ActivityModelHandler.getAccrualSchedulePerStr(m_localizer,
							m_toAccrual.getToAccrualSchedule() );
					Object[] args = new Object[] {rate, schedule, year};
					accrualSchedule = m_localizer.i18n(m_bundle,
							FsWebBundleKey.ACCRUAL_RATE_PER_SCHEDULE, args);
				}
			}
			m_formPC.addRow(i18n(bbmBundle, BbmWebBundleKey.AccrualSchedule), accrualSchedule);
			
			String maxBalance   = null;
			String maxCarryover = null;
			if (m_toAccrual.actTOBal != null) {
				if (m_toAccrual.actTOBal.getMaxBalance()   < ActivityTimeOffBalance.MAX_BALANCE_NA)
					maxBalance   = m_localizer.formatNumber(m_toAccrual.actTOBal.getMaxBalance(),   2) + hoursLable;
				if (m_toAccrual.actTOBal.getMaxCarryOver() < ActivityTimeOffBalance.MAX_BALANCE_NA)
					maxCarryover = m_localizer.formatNumber(m_toAccrual.actTOBal.getMaxCarryOver(), 2) + hoursLable;
			}
			m_formPC.addRow(i18n(bbmBundle, BbmWebBundleKey.MaxBalance),   maxBalance);
			m_formPC.addRow(i18n(bbmBundle, BbmWebBundleKey.MaxCarryover), maxCarryover);
			
			//TimeOffActivitySummary
			if (m_toYearSummary != null) {
				TimeZone tz = m_context.getViewingTimeZone();
				Calendar endCal = Calendar.getInstance(tz);
				endCal.setTime(m_toYearStart.getTime());
				endCal.add(Calendar.YEAR,1);
				endCal.add(Calendar.MINUTE,-1);
				String endDate = m_localizer.formatDate(endCal.getTime(), tz,
						RegionalFormatBundleKey.DATE_FORMAT);
				String asOfTemplate = i18n(m_bundle, FsWebBundleKey.HOURS_AS_OF_DATE);
				String scheduledAsOfEndDate = BPMessageFormat.format(asOfTemplate,
						new Object[] {i18n(m_bundle, FsWebBundleKey.TO_SUMMARY_SCHEDULED), endDate});
				String pendingAsOfEndDate   = BPMessageFormat.format(asOfTemplate,
						new Object[] {i18n(m_bundle, FsWebBundleKey.TO_SUMMARY_PENDING),   endDate});
				
				m_formPC.addRow(scheduledAsOfEndDate,
						m_localizer.formatNumber(m_toYearSummary.getScheduled(), 2));
				m_formPC.addRow(pendingAsOfEndDate,
						m_localizer.formatNumber(m_toYearSummary.getPending(), 2));
			}
		}
	}

	public String getHtmlDisplay() {
		String html = m_formPC.getHtmlDisplay();
		
		//Display Debug information
		if (m_toAccrual != null
				&& m_cat.isDebugEnabled() && RequestUtil.isRMDebugEnabled()){
			html += m_toAccrual.toHtml();
		}
		
		return html;
	}

	protected void initContentTitle() {
		String pageTitle = i18n(m_bundle, FsWebBundleKey.TO_ACCRUAL_CALCULATOR);
		m_contentTitlePC.setPageTitle(pageTitle);
	}

	protected void initToolbar() {
		m_toolbar.addSubmitTextButton(
				PageAction.VIEW,
				i18n(m_bundle, FsWebBundleKey.CALCULATE),
				true);
		addCloseButton();
	}

	public void setEmpID(ID id) {
		this.m_empID = id;
	}

	public void setActivityID(ID id) {
		this.m_activityID = id;
	}

	protected boolean validateTOAccrual(TOAccrual toAccrual) {
		boolean success = true;
		if (toAccrual != null && !toAccrual.getValidationResults(true).isEmpty()) {
			Collection m_valResults = toAccrual.getValidationResults(true);
			for (Iterator it = m_valResults.iterator(); it.hasNext();) {
				ValidationResult vr = (ValidationResult) it.next();
				System.out.println("IS SOFT="+vr.isSoft()
						+ "MSG="+vr.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
				if (!vr.isSoft()) {
					addPageMessage(Message.ERROR_TYPE,
							vr.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
					success = false;
				}
			}
		}
		return success;
	}

	 * Set is agents Page

		m_isAgentPage = isAgentPage;
	}

	/**
	 * Return is agents Page

	}

}
