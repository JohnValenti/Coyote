package com.bluepumpkin.ejb.am.timecollection.gtcc.ejb;

import java.rmi.RemoteException;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.am.Log;
import com.bluepumpkin.ejb.core.base.SessionEJBBase;

/**
 * Title: Blue Pumpkin Software Activity Management Description: Copyright:
 * Copyright (c) 2001 Company: Blue Pumpkin Software, Inc
 * 
 * @author Sankar Nair
 * @version 1.0
 */

public class GTCCAdherenceImportManagerEJB extends SessionEJBBase {

	private final static String clsName = GTCCAdherenceImportManagerEJB.class
			.getName();
	private static Category m_cat = Log.initCategory(clsName);

	private Jdmo jdmo = null; // only used in Junit testing

	{
		super.init(clsName);
	}

	public GTCCAdherenceImportManagerEJB() {
		super();
	}

	public GTCCAdherenceImportManagerEJB(Jdmo jdmo) { // jdmo only used in
		// Junit testing
		this();
		this.jdmo = jdmo;
	}

	public ID getActivityID(String activityName) throws RemoteException {
		methodStart("getActivityID", activityName);

		GTCCImportAssistDAO dao = null;
		dao = (jdmo == null) ? new GTCCImportAssistDAO()
				: new GTCCImportAssistDAO(jdmo);
		ID activityID = dao.getActivityID(activityName);

		dao.cleanUp();

		methodFinish();
		return activityID;
	}

	protected Category getCategory() {
		return m_cat;
	}

	public Date[][] getActivityEventsForWorkResource(ID workResourceID,
			ID activityID, Date startDate, Date endDate) throws RemoteException {
		if (workResourceID == null || activityID == null || startDate == null
				|| endDate == null)
			return new Date[0][0];
		GTCCImportAssistDAO dao = null;
		dao = (jdmo == null) ? new GTCCImportAssistDAO()
				: new GTCCImportAssistDAO(jdmo);
		Date[][] dates = dao.getActivityEventsForWorkResource(workResourceID,
				activityID, startDate, endDate);

		dao.cleanUp();

		methodFinish();
		return dates;
	}
}
