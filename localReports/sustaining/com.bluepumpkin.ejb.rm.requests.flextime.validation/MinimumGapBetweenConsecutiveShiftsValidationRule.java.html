package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Calendar;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.shifts.model.ShiftsConflict;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexRequestMakeup;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.RmUtil;

public class MinimumGapBetweenConsecutiveShiftsValidationRule implements Validator  {
	public static final String MIN_GAP_BETWEEN_CONSECUTIVE_SHIFTS = "MIN_GAP_BETWEEN_CONSECUTIVE_SHIFTS";
	public static final ID RULE_ID = new ID(-194070);

	private static final String CLASS_NAME = MinimumGapBetweenConsecutiveShiftsValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {

		ValidationResult result;
		LOG.debug(RmUtil.dumpEnterMethod("validate", validatable));

		TORequest toRequest = (TORequest) validatable;

		int minTimeBetweenShift = getMinTimeBetweenShift(toRequest);

		if (minTimeBetweenShift > 0) {

			// load relevant shift assignments
			Collection<ShiftAssignment> shiftAssignments = getShiftAssignments(toRequest, minTimeBetweenShift);

			// find time ranges for to-be shift assignment events (i.e. makeup time is included)
			Collection<TimeRange> eventTimeRanges = getTimeRangesWithMakeup(shiftAssignments, toRequest);

			// check makeup shifts with time ranges
			result = checkMinimumGap(toRequest, eventTimeRanges, minTimeBetweenShift);

		} else {
			result = null;
		}

		LOG.debug(RmUtil.dumpExitMethod("validate", result));
		return result;
	}

	private Collection<TimeRange> getTimeRangesWithMakeup(Collection<ShiftAssignment> shiftAssignments, TORequest tor) {
		Collection<TimeRange> result = new ArrayList<TimeRange>();
		for (ShiftAssignment sa : shiftAssignments) {
			Date eventStart = sa.getStartTime();
			Date eventEnd = sa.getEndTime();
			TimeRange mainShift = sa.getMainShiftTimeRange();

			// if the current shift is used in makeup request, update start/end with makeup request
			for (FlexRequestMakeup flexMakeup : tor.getFlexRequestMakeupList()) {
				if (flexMakeup.getShiftStartTime().equals(mainShift.getStartDate()) && flexMakeup.getShiftEndTime().equals(mainShift.getEndDate())) {
					eventStart = Collections.min(Arrays.asList(eventStart, flexMakeup.getStartTime()));
					eventEnd = Collections.max(Arrays.asList(eventEnd, flexMakeup.getEndTime()));
					break;
				}
			}

			result.add(new TimeRange(eventStart, eventEnd));
		}
		return result;
	}

	private ValidationResult checkMinimumGap(TORequest tor, Collection<TimeRange> eventTimeRanges, int minTimeBetweenShift) throws Exception {

		Collection<ShiftsConflict> conflicts = new ArrayList<ShiftsConflict>();
		Calendar cal = Calendar.getInstance(getTimeZone(tor));
		for (FlexRequestMakeup flexMakeup : tor.getFlexRequestMakeupList()) {
			if (flexMakeup.getExtBeforeActivityID() != null) {
				cal.setTime(flexMakeup.getShiftStartTime());
				cal.add(Calendar.MINUTE, - (flexMakeup.getExtBeforeGap() + flexMakeup.getExtBeforeDuration()));
				conflicts.addAll(checkMinimumGapBefore(cal.getTime(), eventTimeRanges, minTimeBetweenShift));

			}
			if (flexMakeup.getExtAfterActivityID() != null) {
				cal.setTime(flexMakeup.getShiftEndTime());
				cal.add(Calendar.MINUTE, flexMakeup.getExtAfterGap() + flexMakeup.getExtAfterDuration());
				conflicts.addAll(checkMinimumGapAfter(cal.getTime(), eventTimeRanges, minTimeBetweenShift));
			}
		}

		ID empID = tor.getEmployeeID();
		String empName = getEmployeeName(tor, empID);
		return RequestUtil.setValidationResult(empID, empName, tor, conflicts, CLASS_NAME);

	}

	private ShiftsConflict createConflict(Date date, int gap, int minSpacingMinutes) {
		ShiftsConflict conflict = new ShiftsConflict(ShiftsConflict.ORG_MINSPACING, minSpacingMinutes - gap, date);
		conflict.setDoubleTarget(minSpacingMinutes);
		return conflict;
	}

	private Collection<ShiftsConflict> checkMinimumGapBefore(Date time, Collection<TimeRange> timeRanges, int minSpacingMinutes) {
		Collection<ShiftsConflict> conflicts = new ArrayList<ShiftsConflict>();
		for (TimeRange timeRange : timeRanges) {
			if (timeRange.getStartDate().before(time)) {
				long gap = (time.getTime() - timeRange.getEndDate().getTime()) / (1000 * 60);
				if (gap < minSpacingMinutes) {
					conflicts.add(createConflict(time, (int) gap, minSpacingMinutes));
				}
			}
		}
		return conflicts;
	}

	private Collection<ShiftsConflict> checkMinimumGapAfter(Date time, Collection<TimeRange> timeRanges, int minSpacingMinutes) {
		Collection<ShiftsConflict> conflicts = new ArrayList<ShiftsConflict>();
		for (TimeRange timeRange : timeRanges) {
			if (timeRange.getEndDate().after(time)) {
				long gap = (timeRange.getStartDate().getTime() - time.getTime()) / (1000 * 60);
				if (gap < minSpacingMinutes) {
					conflicts.add(createConflict(time, (int) gap, minSpacingMinutes));
				}
			}
		}
		return conflicts;
	}

	@SuppressWarnings("unchecked")
	private Collection<ShiftAssignment> getShiftAssignments(TORequest toRequest, int minTimeBetweenShift) throws Exception {
		Collection<TimeRange> makeupDurations = getMakeupAndGapDurations(toRequest);
		Date checkStartDate = addMinutes(Collections.min(makeupDurations).getStartDate(), getTimeZone(toRequest), - minTimeBetweenShift);
		Date checkEndDate = addMinutes(Collections.max(makeupDurations).getEndDate(), getTimeZone(toRequest), minTimeBetweenShift);
		return getPublishedShiftAssignments(toRequest, new TimeRange(checkStartDate, checkEndDate));
	}

	private Date addMinutes(Date date, TimeZone tz, int mins) {
		Calendar cal = Calendar.getInstance(tz);
		cal.setTime(date);
		cal.add(Calendar.MINUTE, mins);
		return cal.getTime();
	}

	private Collection<TimeRange> getMakeupAndGapDurations(TORequest toRequest) throws Exception {
		return getFTValidationCache(toRequest).getMakeupAndGapDurations();
	}

	private int getMinTimeBetweenShift(TORequest toRequest) throws Exception {
		return getFTValidationCache(toRequest).getOrg().getMinTimeBetweenShift();
	}

	private Collection<ShiftAssignment> getPublishedShiftAssignments(TORequest tor, TimeRange eventsDuration) throws Exception {
		return getFTValidationCache(tor).getPublishedShiftAssignments(eventsDuration);
	}

	private String getEmployeeName(TORequest tor, ID empID) throws Exception {
		return getFTValidationCache(tor).getEmployeeNameByID(empID);
	}

	private TimeZone getTimeZone(TORequest tor) throws Exception {
		return getFTValidationCache(tor).getOrg().getTimeZone();
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}

}
