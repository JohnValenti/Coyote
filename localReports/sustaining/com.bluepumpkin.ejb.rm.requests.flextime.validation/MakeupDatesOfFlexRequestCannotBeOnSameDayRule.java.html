package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.ArrayList;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexRequestMakeup;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;

/**
 * Validate the 2 makeup date of a flex request cannot be made on the same day.
 *
 */
public class MakeupDatesOfFlexRequestCannotBeOnSameDayRule implements Validator{

	private static final String CLASS_NAME = MakeupDatesOfFlexRequestCannotBeOnSameDayRule.class.getName();

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		TORequest toRequest = (TORequest)validatable;

		Date duplicatedFlexDate = getFirstDuplicatedDate(toRequest.getFlexRequestMakeupList());

		if (duplicatedFlexDate != null) {
			//Does violate the rule, use the setHardValidationResult to return the error message
			return ValidationUtil.setHardValidationResult(toRequest,
											RmEjbBundleKey.FLEX_REQ_DUPLICATE_MAKEUP_DATE,
											duplicatedFlexDate,
											CLASS_NAME);
		}

		//The TORequest passes the rule, no violation
		return null;
	}

	private Date getFirstDuplicatedDate(List<FlexRequestMakeup> flexRequestMakeupCache) {
		if (flexRequestMakeupCache == null) {
			return null;
		}

		List<Date> dateInMakeupList = new ArrayList<Date>();

		for (FlexRequestMakeup flex: flexRequestMakeupCache) {
			Date shiftStartTime = flex.getShiftStartTime();
			if (dateInMakeupList.contains(shiftStartTime)) {
				return flex.getShiftStartTime();
			}

			dateInMakeupList.add(shiftStartTime);
		}

		return null;
	}
}
