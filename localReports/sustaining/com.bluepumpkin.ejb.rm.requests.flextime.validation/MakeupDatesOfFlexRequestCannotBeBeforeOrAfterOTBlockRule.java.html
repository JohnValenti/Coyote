package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.workrules.ejb.WorkRuleManager;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexRequestMakeup;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.verint.ejb.wfm.WfmManagerFactory;

public class MakeupDatesOfFlexRequestCannotBeBeforeOrAfterOTBlockRule implements Validator {

	private static final String CLASS_NAME = MakeupDatesOfFlexRequestCannotBeBeforeOrAfterOTBlockRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	private static final String COMMA = ",";

	protected ScheduleAccessManager scheduleAccessManager = null;
	protected WorkRuleManager workRuleManager = null;

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		TORequest toRequest = (TORequest) validatable;

		List<FlexRequestMakeup> flexRequestMakeups = toRequest.getFlexRequestMakeupList();
		List<ID> shiftAssignmentIDs = getShiftAssignmentIDsFromFlexRequestMakeup(flexRequestMakeups);
		List<ShiftAssignment> shiftAssignments = getUnpublishedShiftAssignmentByIDs(shiftAssignmentIDs);
		Map<ID, ShiftAssignment> shiftAssignMap = getMapOfShiftAssignment(shiftAssignments);

		List<ViolatedDetail> vilodatedList = validateFlexRequestsAndOTBlock(flexRequestMakeups, shiftAssignMap);


		if (vilodatedList != null && !vilodatedList.isEmpty()) {

			//Does violate the rule, use the setHardValidationResult to return the error message
			return ValidationUtil.setHardValidationResult(toRequest,
											RmEjbBundleKey.FLEX_REQ_NOT_FLEX_MAKEUP_FOR_OT,
											getParamForErrorMessage(vilodatedList),
											CLASS_NAME);
		}

		//DOES NOT violate the rull, return NULL
		return null;
	}

	private String getParamForErrorMessage(List<ViolatedDetail> vilodatedList) {
		List<String> shiftNames = getShiftNamesByIDs(getShiftIDs(vilodatedList));
		StringBuilder str = new StringBuilder();

		//Need to load localized comma from this bundle RmEjbBundleKey.COMMAR_DELIMITER

		boolean isFirst=true;
		for (String s: shiftNames) {

			if (!isFirst) {
				str.append(COMMA);
			}

			str.append(s);
			isFirst = false;
		}

		return str.toString();
	}

	private List<String> getShiftNamesByIDs(List<ID> shiftIds) {
		List<String> shiftName = new ArrayList<String>();
		try {
			Collection<Shift> shifts = this.getWorkRuleManager().getShiftsByIDs(shiftIds);
			if (shifts == null) {
				return shiftName;
			}

			for (Shift s: shifts) {
				shiftName.add(s.getName());
			}

		} catch (Exception e) {
			LOG.error("MakeupDatesOfFlexRequestCannotBeBeforeOrAfterOTBlockRule.getShiftNamesByIDs(...)", e);
		}

		return shiftName;
	}

	private List<ID> getShiftIDs(List<ViolatedDetail> vilodatedList) {
		List<ID> ids = new ArrayList<ID>();
		for (ViolatedDetail v: vilodatedList) {
			ids.add(v.getShiftID());
		}

		return ids;
	}

	private List<ViolatedDetail> validateFlexRequestsAndOTBlock(List<FlexRequestMakeup> flexRequestMakeups, Map<ID, ShiftAssignment> shiftAssignMap) {
		List<ViolatedDetail> vilodatedList = new ArrayList<ViolatedDetail>();

		for (FlexRequestMakeup flex: flexRequestMakeups) {
			ShiftAssignment shiftAssign = shiftAssignMap.get(flex.getShiftAssignmentID());
			vilodatedList.addAll(validateSingleFlexRequestsAndOTBlock(flex, shiftAssign));
		}

		return vilodatedList;
	}

	private List<ViolatedDetail> validateSingleFlexRequestsAndOTBlock(FlexRequestMakeup flexRequestMakeup, ShiftAssignment shiftAssign) {
		List<ViolatedDetail> vilodatedList = new ArrayList<ViolatedDetail>();

		if (!flexRequestMakeup.getShiftAssignmentID().equals(shiftAssign.getID())
			|| flexRequestMakeup == null
			|| shiftAssign == null) {
			return vilodatedList;
		}

		//Check activity the BEFORE the OTP extension
		if (shiftAssign.getOTExtensionBeforeActivityID() != null
			&& shiftAssign.getExtensionBefore() > 0
			&& flexRequestMakeup.getExtBeforeActivityID() != null) {
			ViolatedDetail violiatedDetail = new ViolatedDetail();
			violiatedDetail.setBeforeActivityID(flexRequestMakeup.getExtBeforeActivityID())
							.setShiftAssignmentID(shiftAssign.getID())
							.setShiftID(shiftAssign.getShiftID());
			vilodatedList.add(violiatedDetail);
		}

		//Check activity the AFTER the OTP extension
		if (shiftAssign.getOTExtensionAfterActivityID() != null
			&& shiftAssign.getExtensionAfter() > 0
			&& flexRequestMakeup.getExtAfterActivityID() != null) {
			ViolatedDetail violiatedDetail = new ViolatedDetail();
			violiatedDetail.setBeforeActivityID(flexRequestMakeup.getExtAfterActivityID())
							.setShiftAssignmentID(shiftAssign.getID())
							.setShiftID(shiftAssign.getShiftID());
			vilodatedList.add(violiatedDetail);
		}

		//return a list of violated info (shift ID, befre/after activities, shiftAssignmentId)
		return vilodatedList;
	}

	private Map<ID, ShiftAssignment> getMapOfShiftAssignment(List<ShiftAssignment> shiftAssignments) {
		Map<ID, ShiftAssignment> map = new HashMap<ID, ShiftAssignment>();

		for (ShiftAssignment shiftAssg: shiftAssignments) {
			map.put(shiftAssg.getID(), shiftAssg);
		}

		return map;
	}

	private List<ID> getShiftAssignmentIDsFromFlexRequestMakeup(List<FlexRequestMakeup> flexRequestMakeup) {
		List<ID> shiftAssgsIDs = new ArrayList<ID>();

		if (flexRequestMakeup == null || flexRequestMakeup.isEmpty()) {
			return shiftAssgsIDs;
		}

		for (FlexRequestMakeup shiftAssig: flexRequestMakeup) {
			shiftAssgsIDs.add(shiftAssig.getShiftAssignmentID());
		}

		return shiftAssgsIDs;
	}

	private List<ShiftAssignment> getUnpublishedShiftAssignmentByIDs(List<ID> shiftAssignmentIds) {
		try {
			return getScheduleAccessManager().getUnpublishedShiftAssignmentByIDs(shiftAssignmentIds);
		} catch(Exception e) {
			LOG.error("MakeupDatesOfFlexRequestCannotBeBeforeOrAfterOTBlockRule.getUnpublishedShiftAssignmentByIDs(...)", e);
		}

		return new ArrayList<ShiftAssignment>();
	}


	protected ScheduleAccessManager getScheduleAccessManager() throws BbmEJBCreateException {
		if (this.scheduleAccessManager == null) {
			//not support WhatIf mode
			scheduleAccessManager = WfmManagerFactory.getScheduleAccessManager(false);
		}
		return this.scheduleAccessManager;
	}

	protected WorkRuleManager getWorkRuleManager() throws BbmEJBCreateException {
		if (this.workRuleManager == null) {
			//not support WhatIf mode
			workRuleManager = WfmManagerFactory.getWorkRuleManager();
		}
		return this.workRuleManager;
	}

	private class ViolatedDetail {

		private ID beforeActivityID;
		private ID afterActivityID;
		private ID shiftAssignmentID;
		private ID shiftID;

		public ID getBeforeActivityID() {
			return beforeActivityID;
		}

		public ViolatedDetail setBeforeActivityID(ID activityID) {
			this.beforeActivityID = activityID;
			return this;
		}

		public ID getAfterActivityID() {
			return afterActivityID;
		}

		public ViolatedDetail setAfterActivityID(ID activityID) {
			this.afterActivityID = activityID;
			return this;
		}

		public ID getShiftAssignmentID() {
			return shiftAssignmentID;
		}

		public ViolatedDetail setShiftAssignmentID(ID shiftAssignmentID) {
			this.shiftAssignmentID = shiftAssignmentID;
			return this;
		}

		public ID getShiftID() {
			return shiftID;
		}

		public ViolatedDetail setShiftID(ID shiftID) {
			this.shiftID = shiftID;
			return this;
		}
	}
}
