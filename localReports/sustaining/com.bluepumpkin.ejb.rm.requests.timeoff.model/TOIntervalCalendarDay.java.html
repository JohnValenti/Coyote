package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.BlackoutDay;

public class TOIntervalCalendarDay {
	private BlackoutDay blackoutDay;
	private Date startTime;
	private Date endTime;
	private List<TOIntervalCalendarInterval> intervals = new ArrayList<TOIntervalCalendarInterval>();

	// ================================================================================
	// CONSTRUCTION
	// ================================================================================

	public TOIntervalCalendarDay(Date startTime, Date endTime) {
		this.startTime = startTime;
		this.endTime = endTime;
	}

	public TOIntervalCalendarDay(Date startTime, Date endTime, TOIntervalCalendarInterval... intervals) {
		this.startTime = startTime;
		this.endTime = endTime;
		this.intervals.addAll(Arrays.asList(intervals));
	}

	// ================================================================================
	// INTERFACE
	// ================================================================================

	public void add(TOIntervalCalendarInterval interval) {
		intervals.add(interval);
	}

	public Date getStartTime() {
		return startTime;
	}

	public Date getEndTime() {
		return endTime;
	}

	public List<TOIntervalCalendarInterval> getIntervals() {
		return intervals;
	}

	public long getLength() {
		return endTime.getTime() - startTime.getTime();
	}

	public BlackoutDay getBlackoutDay() {
		return blackoutDay;
	}

	public void setBlackoutDay(BlackoutDay blackoutDay) {
		this.blackoutDay = blackoutDay;
	}

	public boolean isBlackoutDay() {
		return blackoutDay != null;
	}

	@Override
	public String toString() {
		return String.format("%s - %s, Intervals: %s", startTime, endTime, intervals.size());
	}
}
