package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.util.Comparator;

import com.bluepumpkin.ejb.rm.util.RmUtil;

public final class TOChoiceGroupRankComparator implements Comparator<TOChoiceGroup> {

	public static final TOChoiceGroupRankComparator INSTANCE = new TOChoiceGroupRankComparator();

	private TOChoiceGroupRankComparator() {

	}

	@Override
	public int compare(TOChoiceGroup o1, TOChoiceGroup o2) {
		return RmUtil.integerCompare(o1.getUserRank(), o2.getUserRank());
	}



}
