package com.bluepumpkin.web.rm.requests.custshift;

/**
 * Title:        MyCustomShiftRequestsPM
 * Description:  Page Model for My Custom Shift Requests
 * Copyright:    Copyright (c) 2011
 * Company:      Verint Systems, Inc
 * @author       Gonzalo Quintanar
 */
public class MyCustomShiftRequestsPM extends MyRequestsPM 
{
	private Map m_empNameMap = Collections.EMPTY_MAP;
	private Map m_empOrgMap = Collections.EMPTY_MAP;

	/**
	 * Constructs a schedule view page model shell.
	 */
	public MyCustomShiftRequestsPM(RequestContext context)	
	{
		super(context, Request.REQUESTTYPE_CUSTSHIFT);		
	}

	/**
	 * Add Shift Swap Not Enabled Message
	 */
	public static void addCSNotEnabledMsg(PageModel model) 
	{
		model.addPageMessage(Message.INFO_TYPE, 
	    RmWebBundleKey.CSR_CUSTOM_SHIFT_NOT_ENABLED, RmWebBundleKey.BUNDLE_NAME);
	}
	  
	/**
	 * Add Page Not Enabled Message
	 */
	protected void addPageNotEnabledMsg() 
	{
		MyCustomShiftRequestsPM.addCSNotEnabledMsg(this);
	}

	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static PageModel getInstance(RequestContext context) throws Exception 
	{
		return getInstance(context, MyCustomShiftRequestsPM.class);
	}

	/**
	 * Populate Model
	 */
	protected void loadData() throws Exception 
	{
		if (isAuthToViewCSRqts()) 
		{
			try 
			{
				m_orgSetting = RequestsMH.getOrgSettingForEmployee(m_context, m_empID);
				m_isPageEnabled = m_orgSetting.getEnableCustomShiftRequests() || m_orgSetting.getAllowShiftChangeRequests();

				if (m_isPageEnabled) 
				{
					m_requestList = CustomShiftRequestsMH.getRequestsByEmployee(m_empID, m_isIncludeExpired);

					Collection empIDList = CustomShiftRequestsMH.getUniqueEmpIDs(m_requestList);
					m_empNameMap = CustomShiftRequestsMH.getEmpNameMap(m_context, empIDList);
					m_empOrgMap = CustomShiftRequestsMH.getEmpOrgMap(m_context, empIDList);
				}
			}
			catch (RmHardValidationException ex) 
			{
				Message msg = handleUnableToLoadDataError(log, ex);
				msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
			}
		}
	}

	/**
	 * Initialize List Header
	 */
	protected void initHeader() 
	{
		TableHeaderPC header = m_list.getHeader();
		header.setPartialSortable(true);

		//Status column
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.STATUS, true);
		
		//Sub Type Column
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.TYPE, true);

		//Submitted column
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.SUBMITTED, true);
		
		//Campaign column
		addHeaderData(header, m_bundle, RmWebBundleKey.CAMPAIGN, true);
		
		//Start Date column
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.START_DATE, true);
		
		//End Date column
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.END_DATE, true);
		
		//Common columns
		addCommonHeaderData();
	}

	/**
	 * Create Row Data From Request
	 */
	protected MultiColumnNodeData createRowData(RequestAggregate request) 
	{
		CustShiftReq csRequest = (CustShiftReq)request;
		String rowID = request.getID().toString();
		DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData(rowID);

		//Status column
		row_data.add(createStatusIcon(request));

		//Sub Type Column
		row_data.add(CustomShiftUtil.getRequestSubTypeString(csRequest, m_localizer));

		//Submitted column
		row_data.add(createDataWrapper(csRequest.getSubmittedOn()));

		//Campaign column
		row_data.add(getCampaignName(csRequest));
		
		//Start Date column
		row_data.add(createDataWrapper(csRequest.getStartTime()));
		
		//End Date column
		row_data.add(createDataWrapper(csRequest.getEndTime()));

		//Common columns
		addCommonListData(request, row_data);
		
		return row_data;
	}
	
	protected String getCampaignName(CustShiftReq csRequest)
	{
		String name = "";
		ID spID = csRequest.getSPID();
		
		if (spID != null)
		{
			try
			{
				SchedulingPeriod sp = CampaignModelHandler.getSchedPeriod(m_context, spID);
				ID campaignID = sp.getCampaignID();
				if (campaignID != null)
				{
					Campaign campaign = CampaignModelHandler.getCampaign(m_context, campaignID);
					if (campaign != null)
						name = campaign.getName();
				}
			}
			catch (Exception ex)
			{
			}
		}		
		
		return name;
	}


	/**
	 * Add Create Request Button
	 */
	protected void addCreateRequestButton() 
	{
		boolean isBtnEnabled = m_isPageEnabled && isAuthToEditCSRqts();
		String newRequestLabel = i18n(m_bundle, RmWebBundleKey.CREATE_NEW_REQUEST);
		m_toolbar.addPopupWindowTextButton(RmPageAction.POPUP_CUSTSHIFT_REQUEST_FORM_ACTION,
				newRequestLabel, isBtnEnabled);
	}
}
