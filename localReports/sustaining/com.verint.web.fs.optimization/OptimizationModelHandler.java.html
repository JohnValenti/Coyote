package com.verint.web.fs.optimization;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.campaign.ejb.CampaignManager;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.workload.ejb.WorkloadManager;
import com.bluepumpkin.ejb.bbm.workload.model.Media;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.bluepumpkin.ejb.core.CoreEJBCreateException;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.ejb.core.base.CoreFinderException;
import com.witness.web.uif.base.ModelHandler;
import com.witness.web.uif.system.RequestContext;

public class OptimizationModelHandler extends ModelHandler {
	
	public static Media getMedia(RequestContext context, ID mediaId) throws CoreEJBCreateException, CoreFinderException, RemoteException, BbmEJBCreateException, BbmFinderException {
		if (mediaId == null) return null;
		WorkloadManager manager = WfmManagerFactory.getWorkloadManager(context.isInWhatIfMode());
		return manager.getMediaByID(mediaId);
	}
	
	public static SPQueue getSPQueue(RequestContext context, ID spQueueID) throws RemoteException, BbmException{
		CampaignManager manager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
		return manager.getSPQueue(spQueueID);
	}
	
	public static Queue getQueue(RequestContext context, ID queueID) throws RemoteException, BbmException{
		WorkloadManager manager = WfmManagerFactory.getWorkloadManager(context.isInWhatIfMode());
		return manager.getQueueByID(queueID);
	}
	
	public static Collection<SPQueue> getSPQueues(RequestContext context, Collection<ID> spQueueIDs) throws RemoteException, BbmException{
		CampaignManager manager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
		return manager.getSPQueues(spQueueIDs);
	}
	
	public static Campaign getCampaign(RequestContext context, ID campaignId) throws RemoteException, BbmException{
		return CampaignModelHandler.getCampaign(context, campaignId);
	}
	
	public static SchedulingPeriod getSP(RequestContext context, ID spId) throws RemoteException, BbmException{
		return CampaignModelHandler.getSchedulingPeriodByID(context, spId);
	}
	
	public static Collection<CampaignHOO> getCampaignHOO(RequestContext context, ID campaignId, Date start, Date end) throws Exception
	{
		// find HOO, then normalize the result
	    // for each campaign, fetch the HOO, convert to HOOPeriod, then normalize the result
		return CampaignModelHandler.getCampaignHOOAssignments(context, campaignId, start, end);
	}
}
