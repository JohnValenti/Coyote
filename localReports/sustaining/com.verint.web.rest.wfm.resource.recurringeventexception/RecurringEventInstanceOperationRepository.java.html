/**
 * This repository handles RecurringEventInstanceOperations.  These are not entities; the reason for this is that
 * Recurring Event Instances are not actually persisted in the DB.  However, event instances are generated from
 * the templates for client side usage/display (with null IDs as they do not map to a DB record).  When these instances
 * are modified in the UI, an operation containing the modification information is POSTed to this repository.  In effect,
 * this repository handles submission of an "operation" entity.
 */
@JsonApiResourceRepository(RecurringEventInstanceOperation.class)
public class RecurringEventInstanceOperationRepository extends WfoResourceRepository<String, RecurringEventInstanceOperation> {

	private static final Logger log = LoggerFactory.getLogger(RecurringEventInstanceOperationRepository.class.getName());
