package com.verint.ejb.bbm.timeseries;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.verint.ejb.bbm.timeseries.model.PredictedTimeSeries;
import com.verint.ejb.bbm.timeseries.model.PredictedTimeSeriesFieldInfo;

public class PredictedTimeSeriesDAO extends DAOBase<PredictedTimeSeries> {

	private FieldInfo m_fieldInfo = new PredictedTimeSeriesFieldInfo();

	
	public PredictedTimeSeriesDAO() {
		super();
	}
	
	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	@Override
	protected PredictedTimeSeries createValueObject() {
		return new PredictedTimeSeries();
	}

	public boolean hasTimeSeries(int spID) throws JdmoException {
		StringBuffer pStmt = new StringBuffer();
		pStmt.append("select PREDICTEDTIMESERIES.ID FROM PREDICTEDTIMESERIES, SPQUEUE, SP");
		pStmt.append(" where PREDICTEDTIMESERIES.SPQUEUEID=SPQUEUE.ID");
		pStmt.append(" and SPQUEUE.SPID=SP.ID");
		pStmt.append(" and  SP.SID=?");

		JdmoQuery jq = m_dmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
		jq.setParInt(1, spID);
		JdmoRowset rs = m_dmo.createRowset(jq, Jdmo.FORWARD_ONLY,
				Jdmo.READ_ONLY);
		boolean hasSchedules = false;
		if (rs.next())
			hasSchedules = true;
		rs.close();
		return hasSchedules;

	}

}
