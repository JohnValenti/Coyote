package com.verint.web.rest.planner.facade;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.workload.ejb.WorkloadManager;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.fs.Log;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.verint.web.rest.planner.entity.QueuesEntity;
import com.verint.web.rest.exception.checked.SystemWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.util.SecurityUtils;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.system.RequestContext;
import com.witness.ejb.core.security.model.User;


import java.util.List;
import java.util.Map;
import java.util.Date;
import java.util.ArrayList;
import java.util.Collection;

public class QueuesFacadeImpl implements IQueuesFacade {
	private static final Category logger = Log.initCategory(QueuesFacadeImpl.class.getName());

	public List<QueuesEntity> getQueues(RequestContext context, boolean bIsExport, boolean bIsLTF) throws WFOException {
		try {
			WorkloadManager wlm = getWorkloadManager(context);
			Map<Queue, String> queueHash = wlm.getQueuesLinkedToSPAndHasHistoricalData(bIsExport, bIsLTF);
			List<QueuesEntity> queuesList = buildQueueEntities(context, queueHash);
			return queuesList;
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	public List<QueuesEntity> getQueuesWithHistoryTimeseriesInThePeriod(RequestContext context,
			Date startTime, Date endTime) throws WFOException {
		try {
			WorkloadManager wlm = getWorkloadManager(context);
			Map<Queue, String> queueHash = wlm.getQueuesLinkedToSPAndHasHistoricalDataInThePeriod(startTime, endTime);
			return buildQueueEntities(context, queueHash);
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}


	private WorkloadManager getWorkloadManager(RequestContext context) throws WFOException {
		try {
			WorkloadManager workloadManager = WfmManagerFactory.getWorkloadManager(context.isInWhatIfMode());
			return workloadManager;
		} catch (BbmEJBCreateException e) {
			logger.error(e.getLocalizedMessage(), e);
			throw new SystemWFOException(e.getLocalizedMessage());
		}
	}

	private List<QueuesEntity> buildQueueEntities(RequestContext context, Map<Queue, String> queueHash) throws WFOException {
		List<QueuesEntity> queuesEntities = new ArrayList<QueuesEntity>();
		Collection<Queue> scopedQueues = getCampaignScopedQueues(context);
		User user = context.getUser();

		for (Map.Entry<Queue, String> entry : queueHash.entrySet()) {
			Queue queue = entry.getKey();
			String tz = entry.getValue();
			QueuesEntity queuesEntity = new QueuesEntity();
			queuesEntity.setQId(queue.getDEID().toString());
			queuesEntity.setName(queue.getName());
			queuesEntity.setMediaId(queue.getMediaStrID());
			queuesEntity.setOrganizationId(queue.getOrganizationID().toString());
			queuesEntity.setTimeZone(tz);

			if (getIsAuthorized(user, queue.getOrganizationID()) &&
					scopedQueues.contains(queue)) {
				queuesEntities.add(queuesEntity);
			}
		}
		return queuesEntities;
	}

	private boolean getIsAuthorized(User user, ID orgID) {
		boolean isAuthorized = false;
		if (orgID != null) {
			isAuthorized = SecurityUtils.isPrivWithOrgScope(user, PrivilegeKeys.FS_VIEWQUEUES, false, orgID);
		}
		return isAuthorized;
	}

	private Collection<Queue> getCampaignScopedQueues(RequestContext context) throws WFOException {
		try {
			Collection queues = CampaignModelHandler.getQueuesAssignedToCampaignsForUser(context, context.getUser(), null,
				null, true);
			return queues;
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}
}
