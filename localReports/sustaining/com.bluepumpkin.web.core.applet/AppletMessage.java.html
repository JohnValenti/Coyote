package com.bluepumpkin.web.core.applet;

import java.io.CharArrayWriter;
import java.io.PrintWriter;
import java.io.Serializable;

/**
 * Title:        
 * Description:  
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, dsu
 * @version      1.0
 *
 * Created on Apr 23, 2003, 6:27:04 PM
 */
public class AppletMessage implements Serializable {
	static final long serialVersionUID = -3586159700117643849L;
	
	public static final int FATAL = 0;
	public static final int ERROR = 1;
	public static final int WARNING = 2;
	public static final int INFO = 3;
	public static final int DEBUG = 4;

	private int m_type;
	private String m_msg = "";
	private String m_exMsg = "";
	private Exception m_ex;

	/**
	 * Constructor
	 *
	 * @param type - Message Type
	 * @param msg - Message
	 */
	public AppletMessage(int type, String msg) {
		m_type = type;
		m_msg = msg;
	}

	/**
	 * Constructor
	 *
	 * @param type - Message Type
	 * @param msg - Message
	 * @param ex - Associated Exception
	 */
	public AppletMessage(int type, String msg, Exception ex) {
		this(type, msg);
		m_ex = ex;
	}

	public int getType() { return m_type; }

	public String getMsg() { return m_msg; }

	public Exception getException() { return m_ex; }

	public String getExceptionMsg() { return m_exMsg; }

	/**
	 * Return specified message with Exception Message
	 */
	public String getFullMsg() { return m_msg + "\n" + m_exMsg; }

	/**
	 * Converts Stack Trace to String
	 */
	public void initExceptionMsg() {
		if (m_ex!=null) {
			CharArrayWriter caw = new CharArrayWriter();
			m_ex.printStackTrace(new PrintWriter(caw, true));
			m_exMsg = caw.toString();
		}
	}
}
