	public AppletPM(RequestContext context) {
		super(context, AppletPM.FRAMED_MODEL);

		// getting rid of borders - applet handles on its own
		setTemplateStyle("", "", "border:0px solid #7386AD;", "border-bottom:0px solid white;", "");

		// most applets handle their own toolbars and content title
		m_toolbar = null;
		m_contentTitlePC = null;

		m_appletScreen = m_context.getScreen();
		if (m_appletScreen!=null) {
			String idType = m_appletScreen.getParameterValue(AppletKeys.APPLET_ID_TYPE);
			m_isStringID = "String".equalsIgnoreCase(idType);
		}

		addRequiredJavaScriptFile(JavaScriptFileID.MEDIATOR_WORKPANE_GENERIC);
		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
		setJSMediator(WfmJavaScriptFileID.MEDIATOR_APPLET_PAGE);
	}

	/**
	 * Initialize Model with Data
	 */
	protected void initializeModel() {
		if (m_isInitialized) return; else m_isInitialized = true;

		try {
			// Initialize AppletPC
			initAppletPC();

			// Add Selected ID Field
			String selectedIDF = HtmlUtil.makeHiddenInput(SELECTED_ID_FN, getSelectedID());
			addGenericHTML(selectedIDF);
		} catch (Exception ex) {
			handleUnableToLoadDataError(m_log, ex);
		}
	}

	// This is required to make sure that the applet jar version complies with xxxx.xxxx.xxxx.xxxx
	private static String normalizeJarVerionItem(int num) {
		final int ITEM_LIMIT=4;
		String result = String.valueOf(num);
		if (result.length()>ITEM_LIMIT) {
			result = Integer.toHexString(num);
			if (result.length()>ITEM_LIMIT) {
				result = result.substring(0, ITEM_LIMIT);
			}
		}

		return result;
	}

	// This is required to make sure that the applet jar version complies with xxxx.xxxx.xxxx.xxxx
	private static String getAppletJarVersion() {
		String major = normalizeJarVerionItem(PackageInfo.VERSION.getMajor());
		String minor = normalizeJarVerionItem(PackageInfo.VERSION.getMinor());
		String branch = normalizeJarVerionItem(PackageInfo.VERSION.getBranch());
		String build = normalizeJarVerionItem(PackageInfo.VERSION.getBuild());

		StringBuffer versionSB = new StringBuffer(125);
		versionSB.append(major).append(".").append(minor).append(".").append(branch).append(".").append(build);
		return versionSB.toString();
	}

	/**
	 * Initialize AppletPC
	 * @throws Exception
	 */
	protected void initAppletPC() throws Exception {
		if (m_appletScreen==null) return;

		// Handle Applet PC
		String appletPCClass = m_appletScreen.getParameterValue(AppletKeys.APPLET_PC);
		if (!StringUtil.isEmpty(appletPCClass)) m_appletPCClass = appletPCClass;
		m_appletPC = (AppletPC)ReflectionHelperManager
				.newInstancePC(m_appletPCClass, m_context);
		m_appletPC.setName(APPLET_PC_NAME);
		addChildComponent(m_appletPC);

		// Handle Applet Class
		String appletClass = getRequiredParamValue(m_appletScreen, AppletKeys.APPLET_CLASS);
		m_appletPC.setAppletClass(appletClass);

		// Handle Applet Request Handler
		String appletRH = getRequiredParamValue(m_appletScreen, AppletKeys.APPLET_RH);
		m_appletPC.setAppletRequestHandler(appletRH);

		// Handle Applet Jar Files
		jarFileList.add(RESOURCE_JAR);
		jarFileList.add(APPLET_COMMON_JAR);

		// Handle resource bundle jar.  The specific bundles for each language are packaged
		//by the localization team, and we will load only the correct bundle for the current language. 
		String localeStr = m_locale.toString();
		if (!localeStr.startsWith("en")) {
			StringBuffer sb = new StringBuffer(50);
			sb.append("../StaticFiles_").append(localeStr).append("/")
					.append("appletbundle_").append(localeStr).append(".jar");

			jarFileList.add(sb.toString());
		}

		// Need to load applet specific jar last since there seem to be a bug in JRE that it only load
		// jars before applet specific jar.
		String appletJar = getRequiredParamValue(m_appletScreen, AppletKeys.APPLET_JAR);
		StringTokenizer stk = new StringTokenizer(appletJar, ",");
		while(stk.hasMoreTokens()) {
			jarFileList.add(stk.nextToken());
		}

		StringBuffer cacheArchiveSB = new StringBuffer(256);
		StringBuffer cacheVersionSB = new StringBuffer(256);
		boolean isNotFirstItem=false;
		String jarVersion = getAppletJarVersion();
		for(Iterator<String> it = jarFileList.iterator(); it.hasNext(); ) {
			if (isNotFirstItem) {
				cacheArchiveSB.append(",");
				cacheVersionSB.append(",");
			}

			cacheArchiveSB.append(it.next());
			cacheVersionSB.append(jarVersion);

			isNotFirstItem=true;
		}

		m_appletPC.setAppletJarFiles(cacheArchiveSB.toString());
		m_appletPC.setAppletJarFileVersions(cacheVersionSB.toString());

		// Determine whether Applet supports Printing
		String appletPrinting = m_appletScreen.getParameterValue(AppletKeys.APPLET_PRINTING);
		if ("true".equalsIgnoreCase(appletPrinting)) {
			addJSVariable(JSUtil.JS_VAR_IS_SELF_PRINTING, "true");
		}

		// Handle Applet PC JavaScript;
		String jsFileStr = m_appletScreen.getParameterValue(AppletKeys.APPLET_PC_JS);
		String[] jsFiles = StringUtil.parseStringToArray(jsFileStr, ",");
		for(int i=0; i<jsFiles.length; i++) {
			String jsFile = m_context.getStaticFileUrl(jsFiles[i]);
			m_appletPC.setAppletJS(jsFile);
		}

		// Handle Applet Model Type
		String appletModelType = m_appletScreen.getParameterValue(AppletKeys.APPLET_MODEL_TYPE);
		if (!StringUtil.isEmpty(appletModelType)) {
			int type = Integer.parseInt(appletModelType);
			setPageModelType(type);
		}
	}

	/**
	 * Return Required Parameter Value
	 *
	 * @param screen - Screen Object
	 * @param key - Paramter Key
	 * @return Parameter Value
	 * @throws RuntimeException if required parameter is missing
	 */
	private String getRequiredParamValue(Screen screen, String key)
			throws RuntimeException {
		String value = screen.getParameterValue(key);

		if (StringUtil.isEmpty(value)) {
			String msg = "Required Applet Paremeter not found: " + key;
			throw new RuntimeException(msg);
		}

		return value;
	}

	/**
	 * Set Selected IDs
	 */
	public void setSelectedID(String ids) {
		if (m_isStringID) {
			m_stringIDs = ids;
		} else {
			super.setSelectedID(ids);
		}
	}

	/**
	 * Return Selected IDs
	 */
	public String getSelectedID() {
		if (m_isStringID) {
			return m_stringIDs;
		} else {
			return super.getSelectedID();
		}
	}

    /**
     * Return the custom JavaScript code for the Page model.
     * The output of this method is added outside of the JavaScript main() function.
     * The two JS functions defined here are used in order to fix two JavaScript errors 
     * that happen on the Adherence and Pulse applet pages when clicking the Adherence 
     * or Pulse tabs while using JRE 1.4.2 and IE8. (QC#56703/QA#98201). Apparently, the 
     * applets are loading before the JavaScript main() function has completed, so the 
     * window.pageMediator object doesn't exist yet, which results in JavaScript
     * errors. The fix is to use setTimeout() to wait until the window.pageMediator
     * object is defined before trying to access it.
     */
    public String getCustomJavaScript() 
    {
        StringBuffer sb = new StringBuffer(1024);
        sb.append(super.getCustomJavaScript());
        
        sb.append("\n\n");
        sb.append("//This function will evaluate any JavaScript expression you pass-in. \n");
        sb.append("//If the expression results in an exception, the we set a timer to call this function after 50 miliseconds to try again. \n");
        sb.append("//If/when there is no exception, we return the result of the expression. \n");
        sb.append("function evalJSExpWithExceptionCheck(exp) \n");
        sb.append("{ \n");
        sb.append("    try { \n");
        sb.append("        var retVal = eval(exp); \n");
        sb.append("        return retVal; \n");
        sb.append("    } \n");
        sb.append("    catch (ex) {\n");
        sb.append("        console.log('evalJS(' + exp + ') failed: ' + ex);");
        sb.append("        setTimeout('evalJSExpWithExceptionCheck(\\'' + exp + '\\')', 1000); \n");
        sb.append("    } \n");        
        sb.append("}\n\n");
        
        sb.append("\n\n");
        sb.append("//This function will set the context dates as soon as the pageMediator is available. \n");
        sb.append("function setWorkPaneContextDatesWhenReady(startDate, endDate) \n");
        sb.append("{ \n");
        sb.append("    if (typeof(window.pageMediator) != \"undefined\") \n");
        sb.append("    { \n");
        sb.append("        window.pageMediator.setWorkPaneContextDates(startDate, endDate); \n");
        sb.append("    } \n");
        sb.append("    else \n");
        sb.append("    { \n");
        sb.append("        setTimeout('setWorkPaneContextDatesWhenReady(\\'' + startDate + '\\', \\'' + endDate + '\\')', 50); \n");
        sb.append("    } \n");        
        sb.append("}\n\n");
        
        sb.append("//This function will return the selectedIDs if they are available. Otherwise, returns 'undefined'. \n");
        sb.append("function getSelectedIDsIfDefined() \n");
        sb.append("{  \n");
		sb.append("    try {\n");
        sb.append("        var ids = ''; \n");
        sb.append("        if (typeof(top.selectionMediator) != \"undefined\") \n");
        sb.append("        { \n");
        sb.append("            return top.selectionMediator.getSelectedID(); \n");
        sb.append("        } \n");
        sb.append("        else \n");
        sb.append("        { \n");
        sb.append("            return 'undefined';\n");
        sb.append("        } \n");
		sb.append("    }\n");
		sb.append("    catch (ex) {\n");
		sb.append("        return 'undefined';\n");
		sb.append("    }\n");       
        sb.append("}\n\n");        
        
        sb.append("//This function will return the selectedIDs if they are available. Otherwise, returns 'undefined' and \n");
        sb.append("//will call waitForSelectedIDs() to refresh workpane when the selectedIDs become available. \n");
        sb.append("function getSelectedIDsWhenDefined() \n");
        sb.append("{ \n");
        sb.append("    var ids = ''; \n");
        sb.append("    if (typeof(top.selectionMediator) != \"undefined\") \n");
        sb.append("    { \n");
        sb.append("        return top.selectionMediator.getSelectedID(); \n");
        sb.append("    } \n");
        sb.append("    else \n");
        sb.append("    { \n");
        sb.append("        waitForSelectedIDs(); \n");
        sb.append("        return 'undefined';");
        sb.append("    } \n");        
        sb.append("}\n\n");
        
        sb.append("//This private function will wait until the selectedIDs are available, then refreshes the workpane with the IDs. \n");
        sb.append("function waitForSelectedIDs() \n");
        sb.append("{ \n");
        sb.append("    if (typeof(top.selectionMediator) != \"undefined\") \n");
        sb.append("    { \n");
        sb.append("        var ids = top.selectionMediator.getSelectedID(); \n");
        sb.append("        setTimeout('top.selectionMediator.submitTreeSelection()', 500); \n");
        sb.append("    } \n");
        sb.append("    else \n");
        sb.append("    { \n");
        sb.append("        setTimeout('waitForSelectedIDs()', 50); \n");
        sb.append("    } \n");        
        sb.append("}\n\n");

        return sb.toString();
    }
