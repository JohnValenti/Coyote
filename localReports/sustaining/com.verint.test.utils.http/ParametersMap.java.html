package com.verint.test.utils.http;

import com.bluepumpkin.common.datatypes.ID;
import com.gargoylesoftware.htmlunit.util.NameValuePair;
import com.verint.test.constants.ParameterKeys;

import java.util.HashMap;
import java.util.List;
import java.util.stream.Collectors;

public class ParametersMap extends HashMap<String, String> {
	public String put(String key, Integer value) {
		return super.put(key, value.toString());
	}

	public String put(String key, ID value) {
		return super.put(key, value.toString());
	}

	public String put(String key, List<ID> value) {
		if(value == null){
			return super.put(key, "");
		}else {
			return super.put(key, value.stream().map(id -> id.toString()).collect(Collectors.joining(",")));
		}
	}

	public String put(String key, Boolean value) {
		return super.put(key, value.toString());
	}

	public void putForCheckBoxPC(String inputName, boolean value) {
		if (value) {
			// inputName__input is passed only when checked
			put(inputName + "__input", "checked");
		}
		// true/false - depends on the change value status
		put(inputName + "__edited", true);
		put(inputName + "__hidden", true);
	}

	/**
	 * Construct additional Http parameters for requested parameters.
	 * For example we need to construct such parameters set for parameters shiftPatternWorkDaysTable_MQ.
	 * <p>
	 * shiftPatternWorkDaysTable_MQ__input = 0:00
	 * shiftPatternWorkDaysTable_MQ__hidden = true
	 * shiftPatternWorkDaysTable_MQ__edited = true
	 * <p>
	 * Call this method like  putInputValue("shiftPatternWorkDaysTable_MQ", "0:00", true, true)
	 *
	 * @param key
	 * @param value
	 * @param edited
	 * @param hidden
	 * @throws Exception
	 */
	public void putInputValue(final String key, String value, boolean edited, boolean hidden) throws Exception {
		if (null != value) { //sometime we need to skip __input parameter
			this.put(key.concat(ParameterKeys.KEY_SUFFIX_INPUT), value);
		}
		this.put(key.concat(ParameterKeys.KEY_SUFFIX_EDITED), edited);

		//if we need to put hidden parameter for requested value
		if (hidden) {
			this.put(key.concat(ParameterKeys.KEY_SUFFIX_HIDDEN), hidden);
		}
	}

	/**
	 * Convert Parameters Map to HtmlUnit request required parameters list.
	 *
	 * @return
	 */
	public List<NameValuePair> toWebRequestParameters() {
		List<NameValuePair> requestParameters = this.keySet().stream()
				.map(key -> new NameValuePair(key, this.get(key)))
				.collect(Collectors.toList());

		return requestParameters;
	}
}
