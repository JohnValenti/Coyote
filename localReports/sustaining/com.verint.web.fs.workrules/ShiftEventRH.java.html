package com.verint.web.fs.workrules;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        WorkRulesShiftEventsRH.java
 * Description:  Request Handler for Shift Events tab in Work Rules
 * Copyright:    Copyright (c) 2009
 * Company:      Verint Systems, Inc
 * @author       Dustin Stanley
 * @version      1.0
 */
public class ShiftEventRH extends WorkRulesRH<ShiftEventListPM, ShiftEventFormPM> {
	@Override
	public void processRequest(RequestContext context) throws Exception {
			String pageAction = getPageAction(context);
			//Remove Session Item Index when pageAction is null or refresh or is created new!!!
			if(pageAction== null||pageAction.equals(PageAction.NEW_ID)||pageAction.equals(PageAction.REFRESH)) {
				String strItemIndex = (String) context.getAttribute(RequestContext.SESSION_SCOPE, ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX);
				if (!StringUtil.isEmpty(strItemIndex))
					context.removeAttribute(RequestContext.SESSION_SCOPE, ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX);
			} else
				super.processRequest(context);
	
	}
	@Override
	protected ShiftEventListPM getListPageModel(RequestContext context) {
		// Don't getInstance yet because the pagination needs to be set before the data is loaded
		ShiftEventListPM pm = new ShiftEventListPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
	}
	
	@Override
	protected ShiftEventFormPM getFormPageModel(RequestContext context) {
		return ShiftEventFormPM.getInstance(context);
	}

	@Override
	protected void processAdd(RequestContext context, ShiftEventListPM pageModel)
			throws BPException {
		setNextPage(context, FORM_PAGE);
	}

	@Override
	protected void processCopy(RequestContext context, ShiftEventListPM pageModel)
			throws BPException {
		try {
			if (pageModel.isByOrg) {
				WorkRulesModelHandler.copyShiftEvents(context, pageModel.getPerfromActionOnIDsAsCollection(), 
						pageModel.isByOrg, pageModel.getSelectedIDObject(), pageModel.getLocalizer().getLocaleContext());
			} else {
				pageModel.loadData();			//we need to do this to initialize the campaign/sp value objects
				WorkRulesModelHandler.copyShiftEvents(context, pageModel.getPerfromActionOnIDsAsCollection(), 
						pageModel.isByOrg, pageModel.selectedCampaign.getDEID(),
						pageModel.getLocalizer().getLocaleContext());
			}
			context.setPageModel(null);
			pageModel = getListPageModel(context);		//reload the new shift events
			pageModel.addPageMessage(Message.RESPONSE_TYPE,
					FsWebBundleKey.FS_COPY_SHIFT_EVENT_SUCCESS, FsWebBundleKey.BUNDLE_NAME);
		} catch (Exception ex) {
				pageModel.addPageMessage(Message.ERROR_TYPE,
						FsWebBundleKey.FS_UNABLE_TO_COPY_SHIFT_EVENTS, FsWebBundleKey.BUNDLE_NAME);
				throw new BPException(ex);
		} finally {
			context.setPageModel(pageModel);			
		}
	}

	@Override
	protected void processDelete(RequestContext context, ShiftEventListPM pageModel) throws BPException {
		WorkRulesModelHandler.deleteShiftEvents(context, pageModel.getPerfromActionOnIDsAsCollection());
		context.setPageModel(null); // reset the page model to remove the deleted entries 
	}

	@Override
	protected void processEdit(RequestContext context, ShiftEventListPM pageModel)
			throws BPException {
		setNextPage(context, FORM_PAGE);
	}

	@Override
	protected void processExport(RequestContext context,
			ShiftEventListPM pageModel) throws BPException {
		try {
			ShiftEventExportPopupPM model = new ShiftEventExportPopupPM(context);
			model.extractParameters(context.getRequest());
			model.processExport(context);
		} catch (Exception ex) {
			throw new BPException(ex);
		}

	}

	@Override
	protected void processImport(RequestContext context,
			ShiftEventListPM pageModel) throws BPException {
		try {
			ShiftEventImportPopupPM model = new ShiftEventImportPopupPM(context);
			model.extractParameters(context.getRequest());
			model.processImport(context);
			
			//Cache the ProfilePageModel in the Request
			//model.setPageDirty(false);
			if (model.getPageAction() != null && model.getPageAction().equals(ShiftEventImportPopupPM.SUCCESS))
				model.setShowDoneButton(true);
			context.setPageModel(model);
			//setNextPage(context, "listPage");
		} catch (Exception ex) {
			throw new BPException(ex);
		}
	}

	@Override
	protected void processSave(RequestContext context, ShiftEventFormPM pageModel)
			throws BPException {
		try {
			if (pageModel.validate()) {
				if (pageModel.isNew()) {
					for (ShiftEvent se : pageModel.getShiftEvents()) {
						WorkRulesModelHandler.createShiftEvent(context, se);
					}
				} else {
					WorkRulesModelHandler.updateShiftEvents(context, pageModel.getShiftEvents());
				}
			} else {
				setNextPage(context, "formPage");
			}
		} catch (Exception ex) {
			throw new BPException(ex);
		}
	}

	@Override
	protected void processDownload(RequestContext context,
			ShiftEventListPM pageModel) throws BPException {
		// TODO Auto-generated method stub
		
	}
	@Override
	protected void processFind(RequestContext context, ShiftEventListPM pageModel)
			throws BPException {
		pageModel.extractParameters(context.getRequest());
		context.setPageModel(pageModel);
				
	}
	
	@Override
	protected void processTableSort(RequestContext context,ShiftEventListPM pageModel) {
		String sortColumn = context.getRequest().getParameter("sortColumn");
		String sortOrder = context.getRequest().getParameter("sortOrder");
		context.setUserProperty(ShiftEventListPM.SHIFTEVENTLIST_SORTCOLUMN, sortColumn, true);
		context.setUserProperty(ShiftEventListPM.SHIFTEVENTLIST_SORTORDER, sortOrder, true);
	}
	@Override
	protected void processPageLeft(RequestContext context, ShiftEventListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexBackward();	
		String strItemIndex =
				 (String)context.getAttribute(RequestContext.SESSION_SCOPE, ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex))
			context.removeAttribute(RequestContext.SESSION_SCOPE, ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX);
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE,ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX, Integer.toString(itemIndex));
	}
	
	@Override
	protected void processPageRight(RequestContext context, ShiftEventListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexForward();
		String strItemIndex =
				 (String)context.getAttribute(RequestContext.SESSION_SCOPE, ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex))
			context.removeAttribute(RequestContext.SESSION_SCOPE, ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX);
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE,ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX, Integer.toString(itemIndex));
	}
	
	@Override
	protected void processPageSelect(RequestContext context,ShiftEventListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		
		pageModel.setPagePaginationAndEditIndices(
				pageModel.getPaginationControl().getSelectedPaginationPageIndex());
		String strItemIndex =
				 (String)context.getAttribute(RequestContext.SESSION_SCOPE, ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex))
			context.removeAttribute(RequestContext.SESSION_SCOPE, ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX);
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE,ShiftEventListPM.SHIFTEVENTLIST_ITEMINDEX, Integer.toString(itemIndex));  
	}
}
