package com.verint.web.fs.workrules;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.l10n.BbmWebLogBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

public class ShiftListRH extends WorkRulesRH<ShiftListPM, ShiftFormPM> {
	private static final Category log = Log.initCategory(ShiftListRH.class.getName());

	@Override
	public void processRequest(RequestContext context) throws Exception {
		String pageAction = getPageAction(context);
		//Remove Session Item Index when pageAction is null or refresh or is created new!!!
		if(pageAction== null||pageAction.equals(PageAction.NEW_ID)||pageAction.equals(PageAction.REFRESH)) {
			String strItemIndex = (String) context.getAttribute(RequestContext.SESSION_SCOPE, ShiftListPM.SHIFTLIST_ITEMINDEX);
			if (!StringUtil.isEmpty(strItemIndex)) {
				context.removeAttribute(RequestContext.SESSION_SCOPE, ShiftListPM.SHIFTLIST_ITEMINDEX);
			}
		} else {
			super.processRequest(context);
		}
	}
	@Override
	protected ShiftListPM getListPageModel(RequestContext context) {
		
		ShiftListPM pm = new ShiftListPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
		
		// Can't getInstance yet because the pagination needs to be set up before the data is loaded
		//return ShiftListPM.getInstance(context);
	}
	
	@Override
	protected ShiftFormPM getFormPageModel(RequestContext context) {
/*
		ShiftFormPM pm = new ShiftFormPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
*/
		return ShiftFormPM.getInstance(context);
	}

	@Override
	protected void processAdd(RequestContext context, ShiftListPM pageModel)
			throws BPException {
		setNextPage(context, "formPage");
	}

	@Override
	protected void processCopy(RequestContext context, ShiftListPM pageModel)
			throws BPException {	
		try {
			if (pageModel.isByOrg) {
				WorkRulesModelHandler.copyShifts(context, pageModel.getPerfromActionOnIDsAsCollection(), 
					pageModel.isByOrg, pageModel.getSelectedIDObject(), pageModel.getLocalizer().getLocaleContext());
			} else {
				pageModel.loadData();			//we need to do this to initialize the campaign/sp value objects
				WorkRulesModelHandler.copyShifts(context, pageModel.getPerfromActionOnIDsAsCollection(), 
						pageModel.isByOrg, pageModel.selectedCampaign.getDEID(),
						pageModel.getLocalizer().getLocaleContext());
			}
			context.setPageModel(null);
			pageModel = getListPageModel(context);		//reload the new shifts
			pageModel.addPageMessage(Message.RESPONSE_TYPE,
					FsWebBundleKey.FS_COPY_SHIFTS_SUCCESS, FsWebBundleKey.BUNDLE_NAME);
		} catch(Exception ex) {
			pageModel.addPageMessage(Message.ERROR_TYPE,
					FsWebBundleKey.FS_UNABLE_TO_COPY_SHIFTS, FsWebBundleKey.BUNDLE_NAME);
			throw new BPException(ex);
		} finally {
			context.setPageModel(pageModel);
		}
	}

	@Override
	protected void processDelete(RequestContext context, ShiftListPM pageModel)
			throws BPException {
		WorkRulesModelHandler.deleteShifts(context, pageModel.getPerfromActionOnIDsAsCollection());
		context.setPageModel(null); // reset the page model to remove the deleted entries 
	}

	@Override
	protected void processEdit(RequestContext context, ShiftListPM pageModel)
			throws BPException {
		ShiftFormPM formPM = new ShiftFormPM(context);
		formPM.extractParameters(context.getRequest());
		context.setPageModel(formPM);
		setNextPage(context, "formPage");
	}

	@Override
	protected void processExport(RequestContext context, ShiftListPM pageModel)
			throws BPException {
		ShiftExportPopupPM model = new ShiftExportPopupPM(context);
		model.extractParameters(context.getRequest());
		model.processExport(context);
		context.setPageModel(model);
	}

	@Override
	protected void processDownload(RequestContext context, ShiftListPM pageModel)
			throws BPException {
		try {
			ShiftExportPopupPM model = new ShiftExportPopupPM(context);
			model.extractParameters(context.getRequest());
			boolean isSuccess = model.processExport(context);
			
			model.addPageMessage(Message.INFO_TYPE, "Export completed successfully!");
			context.setPageModel(model);
			
			if(!isSuccess) {
				setNextPage(context, "exportPage");
			}
		} catch (Exception ex) {
			log.l7dError(BbmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}
	}
	
	@Override
	protected void processImport(RequestContext context, ShiftListPM pageModel)
			throws BPException {
		try {
			ShiftImportPopupPM model = new ShiftImportPopupPM(context);
			model.extractParameters(context.getRequest());
			model.processImport(context);
			
			//Cache the ProfilePageModel in the Request
			//model.setPageDirty(false);
			if (model.getPageAction() != null && model.getPageAction().equals(ShiftImportPopupPM.SUCCESS))
				model.setShowDoneButton(true);
			context.setPageModel(model);
			//setNextPage(context, "listPage");
		} catch (Exception ex) {
			log.l7dError(BbmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
			//model.addPageMessage(Message.ERROR_TYPE, BbmWebBundleKey.UploadDataFailure, BbmWebBundleKey.BUNDLE_NAME);
		}
	}

	@Override
	protected void processSave(RequestContext context, ShiftFormPM pageModel)
			throws BPException {
		try {
			if (pageModel.validate()) {
				if (pageModel.isNew()) {
					WorkRulesModelHandler.createShift(context, pageModel.getShifts().iterator().next());
				} else {
					WorkRulesModelHandler.updateShifts(context, pageModel.getShifts());
				}
			} else {
				setNextPage(context, "formPage");
			}
		} catch (BbmException e) {
			setNextPage(context, "formPage");
			throw e;
		}
	}
	@Override
	protected void processFind(RequestContext context, ShiftListPM pageModel)
			throws BPException {
		pageModel.extractParameters(context.getRequest());
		context.setPageModel(pageModel);
	}

	@Override
	protected void processTableSort(RequestContext context,ShiftListPM pageModel) {
		String sortColumn = context.getRequest().getParameter("sortColumn");
		String sortOrder = context.getRequest().getParameter("sortOrder");
		context.setUserProperty(ShiftListPM.SHIFTLIST_SORTCOLUMN, sortColumn, true);
		context.setUserProperty(ShiftListPM.SHIFTLIST_SORTORDER, sortOrder, true);
	}
	
	@Override
	protected void processPageLeft(RequestContext context, ShiftListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexBackward();	
		String strItemIndex =
				 (String)context.getAttribute(RequestContext.SESSION_SCOPE, ShiftListPM.SHIFTLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex))
			context.removeAttribute(RequestContext.SESSION_SCOPE, ShiftListPM.SHIFTLIST_ITEMINDEX);
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE,ShiftListPM.SHIFTLIST_ITEMINDEX, Integer.toString(itemIndex));
	}
	
	@Override
	protected void processPageRight(RequestContext context, ShiftListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexForward();
		String strItemIndex =
				 (String)context.getAttribute(RequestContext.SESSION_SCOPE, ShiftListPM.SHIFTLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex))
			context.removeAttribute(RequestContext.SESSION_SCOPE, ShiftListPM.SHIFTLIST_ITEMINDEX);
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE,ShiftListPM.SHIFTLIST_ITEMINDEX, Integer.toString(itemIndex));
	}
	
	@Override
	protected void processPageSelect(RequestContext context,ShiftListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		
		pageModel.setPagePaginationAndEditIndices(
				pageModel.getPaginationControl().getSelectedPaginationPageIndex());
		String strItemIndex =
				 (String)context.getAttribute(RequestContext.SESSION_SCOPE, ShiftListPM.SHIFTLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex))
			context.removeAttribute(RequestContext.SESSION_SCOPE, ShiftListPM.SHIFTLIST_ITEMINDEX);
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE,ShiftListPM.SHIFTLIST_ITEMINDEX, Integer.toString(itemIndex));  
	}
}
