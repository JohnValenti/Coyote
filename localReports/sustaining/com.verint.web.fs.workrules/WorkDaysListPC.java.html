package com.verint.web.fs.workrules;

import java.util.Date;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPattern;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPatternShiftAttributes;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.jsp.HtmlDiv;
import com.witness.web.uif.jsp.HtmlTag;
import com.witness.web.uif.keys.ImageFileID;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.MultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.InsertableMultiColListPC;
import com.witness.web.uif.pagecomponent.table.DefaultHeaderData;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.SortUtil;

public class WorkDaysListPC extends InsertableMultiColListPC {
	private static final long serialVersionUID = 1L;

	private static final String POPUP_SHIFT = "POPUP_SHIFT";
	private static final String[] POPUP_SHIFT_ARGS = new String[] { ShiftListPopupPM.ARG_SHIFT_IDS,
			ShiftListPopupPM.ARG_TIME_ZONE, ShiftListPopupPM.ARG_TIME_ZONE_DATE };
	private ShiftPattern shiftPattern;
	private TimeZone optimizationTargetTimeZone;
	private Date timeZoneEffectiveDate;
	private int dayOfWeekStart = 1;

	public WorkDaysListPC(RequestContext context, String name, ShiftPattern shiftPattern,
			TimeZone optimizationTargetTimeZone, Date timeZoneEffectiveDate, boolean isByOrg,
			Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap) {
		super(context);
		setName(name);

		this.shiftPattern = shiftPattern;
		if (optimizationTargetTimeZone != null) {
			this.optimizationTargetTimeZone = optimizationTargetTimeZone;
		} else {
			this.optimizationTargetTimeZone = getRequestContext().getViewingTimeZone();
		}
		this.timeZoneEffectiveDate = timeZoneEffectiveDate;

		//The start day of week for the work pattern matches the start of week of its owner (either an 
		//org or a campaign).  The first day of week that is rendered in the table header should match this
		//start of week.
		dayOfWeekStart = WorkRulesUtil.getStartDayOfWeekOfWorkPatternOwner(shiftPattern, orgMap, campaignMap);

		initHeader();
		initRows(shiftPattern);

		addPopupArgs(POPUP_SHIFT, ShiftListPopupPM.FORM_ACTION, "",
				StringUtil.createDelimitedString(POPUP_SHIFT_ARGS, ","), "1000,600,ctr,ctr", true, 1);
	}
	


	private void initHeader() {
		TableHeaderPC header = getHeader();
		Localizer localizer = m_context.getLocalizer();
		DefaultHeaderData dhd = new DefaultHeaderData(FsWebBundleKey.FS_SHIFT, getLocalizer().i18n(
				FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_SHIFT));
		dhd.addImageButton(getShiftPopupImage(), getShiftPopupCaption(), getShiftPopupJS());
		dhd.setSortable(true);
		header.addColumnHeaderData(dhd);

		switch (shiftPattern.getPeriod()) {
		case Week:
			String[] dow = m_localizer.getWeekdayNames(false);
			int[] dowOrder = m_localizer.getWeekdayValues(dayOfWeekStart);
			for (int i = 0; i < dowOrder.length; i++) {
				header.addColumnHeaderData(dow[dowOrder[i]], dow[dowOrder[i]]);
			}
			break;
		case TwentyEightDayMonth:
		case TwentyNineDayMonth:
		case ThirtyDayMonth:
		case ThirtyOneDayMonth:
			for (int i = 0; i < shiftPattern.getPeriod().getDaysInPeriod().length; i++) {
				header.addColumnHeaderData(Integer.toString(shiftPattern.getPeriod().getDaysInPeriod()[i]),
						m_localizer
								.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MONTHLY_WORK_PATTERN_DAY_HEADER,
										new Object[] { m_localizer.formatNumber(shiftPattern.getPeriod()
												.getDaysInPeriod()[i]) }));
			}
			break;
		default:
		}

		header.addColumnHeaderData(FsWebBundleKey.FS_CONSISTENT_SHIFT_EVENTS_HEADER,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_CONSISTENT_SHIFT_EVENTS_HEADER));
		header.addColumnHeaderData(FsWebBundleKey.FS_MIN_CONSECUTIVE_DAYS_HEADER,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MIN_CONSECUTIVE_DAYS_HEADER));
		header.addColumnHeaderData(FsWebBundleKey.FS_MAX_CONSECUTIVE_DAYS_HEADER,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MAX_CONSECUTIVE_DAYS_HEADER));
	}

	public HtmlTag getWrapper() {
		if (m_wrapper == null)
			m_wrapper = new HtmlDiv();
		m_wrapper.setStyle("margin:20px");
		return m_wrapper;
	}

	private void initRows(ShiftPattern shiftPattern) {
		List<MultiColumnNodeData> rows = super.getListModel();
		HashMap<Shift, ShiftPatternShiftAttributes> shiftInfo = shiftPattern.getShiftPatternShiftAttributes();
		for (Shift shift : shiftInfo.keySet()) {
			DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(shift.getID());

			// Shift Name
			nodeData.add(shift.getName());

			// The shift occurrences are stored relative to the start of week.
			for (int i = 0; i < shiftPattern.getPeriod().getDaysInPeriod().length; i++) {
				nodeData.add(getDOWCellText(shiftInfo.get(shift), shiftPattern.getPeriod().getDaysInPeriod()[i]));
			}

			// Consistent Shift Activities
			nodeData.add(getLocalizer().formatYesNo(shiftInfo.get(shift).isConsistentShiftActivities()));

			// Min Consecutive Days
			nodeData.add(Integer.toString(shiftInfo.get(shift).getMinConsecutiveDay()));

			// Max Consecutive Days
			nodeData.add(Integer.toString(shiftInfo.get(shift).getMaxConsecutiveDay()));

			setRowAttributes(nodeData, shift);

			rows.add(nodeData);
		}

		SortUtil.sort(getListModel(), getHeader().getColumnIndexByName(FsWebBundleKey.FS_SHIFT), true,
				m_context.getLocalizer());
	}

	/**
	 * Gets the appropriate day of week cell for a dow index (monday = 1)
	 * 
	 * @param shiftInfo
	 * @param dow
	 *            (Monday=1 per db schema)
	 * @return
	 */
	private String getDOWCellText(ShiftPatternShiftAttributes shiftInfo, int dow) {
		if (shiftInfo.getConsistentShiftGroupMap().get(dow) == null) {
			return getLocalizer().formatYesNo(shiftInfo.getOccurrenceDays().contains(dow));
		} else {
			return getLocalizer().i18n(
					FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_CONSISTENT_START_TIMES_FORMAT,
					new Object[] { getLocalizer().formatYesNo(shiftInfo.getOccurrenceDays().contains(dow)),
							Integer.toString(shiftInfo.getConsistentShiftGroupMap().get(dow)) });
		}
	}

	private String getShiftPopupImage() {
		return ImageFileID.TOOLTIP_BUTTON;
	}

	private String getShiftPopupCaption() {
		return m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_VIEW_SHIFT_DETAILS);
	}

	private String getShiftPopupJS() {
		return "WorkRulesUtil.handleInfoPopup(this, " + getName() + "," + getHeader().getName() + ",'"
				+ ShiftListPopupPM.POPUP_ID + "','" + ShiftListPopupPM.ARG_SHIFT_ID + "','"
				+ ShiftListPopupPM.ARG_SHIFT_IDS + "','" + ShiftListPopupPM.ARG_TIME_ZONE + ","
				+ ShiftListPopupPM.ARG_TIME_ZONE_DATE + "','" + optimizationTargetTimeZone.getID() + ","
				+ timeZoneEffectiveDate.getTime() + "');";
	}

	protected void setRowAttributes(DefaultMultiColumnNodeData rowData, Shift shift) {
		rowData.setNodeAttribute(ShiftListPopupPM.ARG_SHIFT_ID, shift.getID().toString());
	}
}
