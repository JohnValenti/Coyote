		this(queueID, startDate, endDate, types, TimeZoneUtil.GMT_TIMEZONE);
	}

	/**
	 * TraceCube is a container for types of Trace, given period
	 *
	 * @param ID,      queueID
	 * @param Date,    startDate,
	 * @param Date,    endDate,
	 * @param short[], the Trace Types to be included
	 * @param TimeZone
	 */
	public AggrRequiredTraceCube(ID queueID, Date startDate, Date endDate, short[] types, TimeZone timeZone) throws BbmTimeSeriesException{
		super(queueID, startDate, endDate, null);
		checkTraceTypeSupport(types);
		m_Types = types;

		ArrayList<Short> requireTypes = new ArrayList(types.length);
		ArrayList<Short> servicegoalTypes = new ArrayList(types.length);
		for (int i=0; i<types.length; i++) {
			short type = types[i];
			if (TraceUtil.META_REQUIRE.supportType(type)) {
				requireTypes.add(type);
			} else {
				servicegoalTypes.add(type);
			}
		}
		if (!requireTypes.isEmpty()) {
			m_RTC =  new RequireTraceCube(queueID, startDate, endDate, toShortArray(requireTypes), timeZone);
		}
		if (!servicegoalTypes.isEmpty()) {
			m_SGTC = new ServiceGoalTraceCube(queueID, startDate, endDate, toShortArray(servicegoalTypes), timeZone);
		}

		
		short[] cvTypes = TraceUtil.getAggTraceType(types);
		if (cvTypes.length != types.length) {
			short[] cv = {Trace.CV, CV_VH, VAR};
			m_FTC = new ForecastTraceCube(queueID, startDate, endDate, cv, timeZone);
		}
	}

	// Base TraceCube supports all types of Trace
	protected short[] getSupportedTraceType() {	return TYPES; }

	/**
	 * Get the maximum length of TraceValue for several Trace Types
	 *
	 * @param short[], types
	 * @return int, length
	 */
	public int getMaximumLength(short[] types) {
		int length = 0;
		if (m_RTC != null) {
			length = m_RTC.getMaximumLength(types);
		}
		if (m_SGTC != null) {
			length = (m_SGTC.getMaximumLength(types) > length) ? m_SGTC.getMaximumLength(types) : length;
		}
		if (m_FTC != null && types[0] == Trace.CV) {
			length = m_FTC.getMaximumLength(types);
		}
		return length;
	}

	 * Set value without validation check, for internal routine
	 */
	public void initTraceValue(short type, int[] value, Date start) {
		TraceCube tc = getTraceCubeForType(type);
		if (tc != null) {
			if (tc instanceof ServiceGoalTraceCube)
				((ServiceGoalTraceCube) tc).initTraceValue(type, value, start);
			else {
				int offSet = TraceUtil.gapOffSet(m_StartDate, m_StartOffSet, start);
				tc.setTraceValue(type, value, offSet);
			}
		}
	}

	protected String getCubeName() { return "AggrRequiredTraceCube"; }

	public TraceCube getTraceCubeForType(short type) {
		if (TraceUtil.META_REQUIRE.supportType(type)) {
			if (m_RTC != null && m_RTC.getTraceValueD(type) != null)
				return m_RTC;
		} else {
			if (TraceUtil.META_SERVICEGOAL.supportType(type)) {
				if (m_SGTC != null && m_SGTC.getTraceValueD(type) != null)
					return m_SGTC;
			} else {
				if (getForecastTrace() != null) {
					return getForecastTrace();
				}
			}
		}
		return null;
	}

	// make a clone from given parameters
	public TraceCube newInstance(ID queueID, Date startDate, Date endDate, short[] types) throws BbmTimeSeriesException {
		return new AggrRequiredTraceCube(queueID, startDate, endDate, types);
	}

	// override default clone function, deep copy of the tracecube without queueID
	public Object clone() {
		try {
			AggrRequiredTraceCube clone = (AggrRequiredTraceCube) super.clone();
			if (getForecastTrace() != null) {
				clone.setForecastTrace((ForecastTraceCube) getForecastTrace().clone());
			}
			return clone;
		} catch (Exception e) {
			return null;
		}
	}

	// make a clone for the current TraceCube object, with queue ID set to null
	public TraceCube cloneTraceCube() throws BbmTimeSeriesException {
		AggrRequiredTraceCube clone = (AggrRequiredTraceCube) super.cloneTraceCube();
		if (getForecastTrace() != null) {
			clone.setForecastTrace(getForecastTrace().cloneTraceCube());
		}
		return clone;
	}

	public String toString() {
		StringBuffer sb = new StringBuffer(1024);

		}
		return sb.toString();
	}
