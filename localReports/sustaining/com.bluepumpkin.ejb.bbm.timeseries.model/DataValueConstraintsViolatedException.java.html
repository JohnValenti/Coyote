package com.bluepumpkin.ejb.bbm.timeseries.model;

/**
 * This exception occurs when an attempt is made to set a value of
 * a trace cube to a value that is invalid for that particular trace
 * type.
 */
public class DataValueConstraintsViolatedException extends
		IllegalArgumentException {
	private static final long serialVersionUID = 1L;
	protected final DataValueConstraints constraints;
	protected final Double value;
	
	public DataValueConstraintsViolatedException(
			DataValueConstraints constraints, Double value) {
		super();
		this.constraints = constraints;
		this.value = value;
	}

	@Override
	public String getMessage() {
		String min = "undefined";
		String max = "undefined";
		if (constraints != null) {
			min = "" + constraints.getMinValue();
			max = "" + constraints.getMaxValue();
		}
		return "Value of " + value + "violates constraints of min " + min + " and max of " + max; 
	}

	/**
	 * @returns the {@link DataValueConstraints} that was violated when setting the value
	 */
	public DataValueConstraints getConstraints() {
		return constraints;
	}

	/**
	 * @returns the value that violated the constraints when it was being set 
	 */
	public Double getValue() {
		return value;
	}
}
