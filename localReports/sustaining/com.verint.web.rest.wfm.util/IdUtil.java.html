package com.verint.web.rest.wfm.util;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.stream.StreamSupport;


public class IdUtil {

	/**
	 * Returns a collection of ID objects, with each ID backed by an Integer in the given collection.
	 */
	public static Collection<ID> fromInt(Iterable<Integer> ids) {
		List<ID> retVal = new ArrayList<>();
		for (Integer id : ids) {
			retVal.add(ID.fromInt(id));
		}
		return retVal;
	}

	public static String toString(Object idObject) {
		if (idObject == null) {
			return null;
		}
		return idObject.toString();
	}

	public static String toDelimitedString(Iterable<? extends Serializable> ids) {
		return StreamSupport.stream(ids.spliterator(), false).
				map(Serializable::toString).reduce(StringUtil.DELIM, String::concat);

//		return StringUtil.createDelimitedString(StreamSupport.stream(ids.spliterator(), false).toArray(),
//				StringUtil.DELIM);
	}
}
