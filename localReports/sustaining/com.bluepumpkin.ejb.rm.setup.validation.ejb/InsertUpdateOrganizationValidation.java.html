package com.bluepumpkin.ejb.rm.setup.validation.ejb;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoParam;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.ejb.rm.util.DAOUtil;
import com.bluepumpkin.ejb.rm.util.JdmoBatchExecute;

public class InsertUpdateOrganizationValidation {

	private static class MergeData {
		final int validationRuleID;
		final int organizationID;
		final Integer activityID;
		final boolean isEnabled;

		MergeData(int validationRuleID, int organizationID, Integer activityID, boolean isEnabled) {
			this.validationRuleID = validationRuleID;
			this.organizationID = organizationID;
			this.activityID = activityID;
			this.isEnabled = isEnabled;
		}
	}

	List<MergeData> mergeData;

	JdmoParam nullInteger;
	
	
	public void add(int validationRuleID, int organizationID, Integer activityID, boolean isEnabled) {
		if (mergeData == null) {
			mergeData = new ArrayList<MergeData>();
		}

		if (activityID == null && nullInteger == null) {
			nullInteger = new JdmoParam(null, java.sql.Types.INTEGER);
		}

		mergeData.add(new MergeData(validationRuleID, organizationID, activityID, isEnabled));
	}

	public void mergeToDB(Jdmo dmo) throws JdmoException {
		if (mergeData == null || mergeData.isEmpty()) {
			return;
		}

		createTempTable(dmo);
		JdmoQuery jdmoQuery = dmo.createQuery("RM_MERGE_ORGANIZATION_VALIDATION", Jdmo.STORPROC_QUERY_NORS);
		dmo.execute(jdmoQuery);
	}

	private void createTempTable(Jdmo dmo) throws JdmoException {

		DAOUtil.createTempTable(dmo, "mergeOrganizationValidation",
				"ValidationRuleID int, OrganizationID int,ActivityID int, IsEnabled bit ");

		JdmoBatchExecute<MergeData> batchExecute = new JdmoBatchExecute<MergeData>(dmo) {
			@Override
			public void set(Map<Integer, Object> param, MergeData value) {
				param.put(1, value.validationRuleID);
				param.put(2, value.organizationID);
				param.put(3, value.activityID == null ? nullInteger : value.activityID);
				param.put(4, value.isEnabled);
			}
		};

		batchExecute.executeBatch("insert into #mergeOrganizationValidation values (?,?,?,?)", mergeData);

	}

}
