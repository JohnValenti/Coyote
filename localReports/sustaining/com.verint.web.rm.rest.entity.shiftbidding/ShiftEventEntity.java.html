package com.verint.web.rm.rest.entity.shiftbidding;

import com.bluepumpkin.ejb.bbm.schedule.model.ShiftEventAssignment;
import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rest.util.FormatUtils;

/**
 * Represents a Shift Event.
 */
public class ShiftEventEntity extends AbstractWFOEntity {

	private final String activityID;
	private final String endDate;
	private final String startDate;

	public ShiftEventEntity(ShiftEventAssignment shiftEventAssignment) {
		setId(shiftEventAssignment.getID().toString());
		startDate = FormatUtils.dateToISO8601SecPrecisionString(shiftEventAssignment.getStartTime());
		endDate = FormatUtils.dateToISO8601SecPrecisionString(shiftEventAssignment.getEndTime());
		activityID = shiftEventAssignment.getActivityID().toString();
	}

	public String getActivityID() {
		return activityID;
	}

	public String getEndDate() {
		return endDate;
	}

	public String getStartDate() {
		return startDate;
	}
}
