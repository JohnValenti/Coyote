package com.bluepumpkin.ejb.rm.util;

import java.util.ArrayList;
import java.util.List;

public class JdmoPreparedStatementBuilder {

	StringBuilder stringBuilder;
	List<Object> parameters;
	private static final Object[] NO_PARAMS = {};

	public JdmoPreparedStatementBuilder() { // NOSONAR

	}

	public JdmoPreparedStatementBuilder(String sql) {
		add(sql);
	}

	public JdmoPreparedStatementBuilder add(String sql, Object... params) {
		initialize();
		stringBuilder.append(sql);

		for (Object p : params) {
			parameters.add(p);
		}

		return this;
	}

	private void initialize() {
		if (stringBuilder == null) {
			stringBuilder = new StringBuilder();
			parameters = new ArrayList<Object>();
		}
	}

	public String getSql() {
		return stringBuilder == null ? "" : stringBuilder.toString();
	}

	public Object[] getParameters() {
		return parameters == null ? NO_PARAMS : parameters.toArray(new Object[parameters.size()]);
	}

	@Override
	public String toString() {
		return getSql();
	}

}
