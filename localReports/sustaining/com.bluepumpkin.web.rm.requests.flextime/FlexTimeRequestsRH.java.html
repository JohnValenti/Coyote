package com.bluepumpkin.web.rm.requests.flextime;

import java.io.Writer;
import java.util.Date;

import org.json.JSONWriter;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.core.base.MultiUserException;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.l10n.RmWebLogBundleKey;
import com.bluepumpkin.web.rm.requests.AgentsRequestsPM;
import com.bluepumpkin.web.rm.requests.RequestUtil;
import com.bluepumpkin.web.rm.requests.RequestsRH;
import com.bluepumpkin.web.rm.requests.timeoff.AgentsTimeOffRequestsPM;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.WebException;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.pagecomponent.picker.date.DatePickerPC;
import com.witness.web.uif.system.RequestContext;

public class FlexTimeRequestsRH extends RequestsRH {
	private static final Category LOG = Log.initCategory(FlexTimeRequestsRH.class.getName());
	private FlexTimeRequestsMH modelHandler = new FlexTimeRequestsMH();

	private static final int INDEX_ZERO = 0;
	private static final int INDEX_ONE = 1;
	private static final int INDEX_TWO = 2;
	private static final int INDEX_THREE = 3;

	private static final String PARAM_PAGE_MODE = "pageMode";
	private static final String PARAM_REQUEST_ID = "requestID";
	
	/**
	 * Process Request for the User page
	 *
	 * @param context The Request Context
	 */
	@Override
	public void processCustomRequest(RequestContext context) throws WebException {
		// Process requested action
		if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		} else if (isAction(context, "GET_TIMEOFF_HOURS")) {
			getTimeOffHours(context);
		} else if (isAction(context, "FORMAT_MAKEUP_LENGTH")) {
			formatMakeupLength(context);
		} else if (isAction(context, PageAction.RESTORE_DEFAULTS)) {
			revert(context);
		} else {
			super.processCustomRequest(context);
		}
	}

	/**
	 * When revert is selected on the UI dialog, this method will refresh the UI
	 * as if it's new & clean dialog
	 * @param context
	 */
	private void revert(RequestContext context) {
		FlexTimeRequestFormPopupPM model = new FlexTimeRequestFormPopupPM(context);
		String requestID = context.getParameter(PARAM_REQUEST_ID);
		if (requestID != null && !requestID.isEmpty()) {
			model.setRequestID(new ID(context.getParameter(PARAM_REQUEST_ID)));
		}
		
		context.setAttribute(RequestContext.REQUEST_SCOPE, PARAM_PAGE_MODE, context.getParameter(PARAM_PAGE_MODE));
		context.setPageModel(model);
	}
	
	private void formatMakeupLength(RequestContext context) throws WebException {
		try {
			context.getResponse().setContentType(RequestContext.DEFAULT_RESPONSE_CONTENT_TYPE);
			Writer out = context.getResponse().getWriter();
			JSONWriter writer = new JSONWriter(out);
			writer
			.object()
				.key("length")
				.value(modelHandler.formatMakeupLength(context, Integer.parseInt(context.getParameter("length"))))
			.endObject();
			out.flush();
		} catch (Exception e) {
			LOG.error("Exception", e);
			throw new WebException();
		}
	}

	private void getTimeOffHours(RequestContext context) throws WebException {
		try {

			// extract the parameters empId timeOffTypeId  startDate endDate
			ID empId = ID.fromInt(Integer.parseInt(context.getParameter("empId")));
			ID timeOffTypeId = ID.fromInt(Integer.parseInt(context.getParameter("timeOffTypeId")));
			DatePickerPC datePicker = new DatePickerPC(context, null);
			datePicker.setIsTimeEnabled(true);
			Date startDate = datePicker.extractDate("startDate");
			Date endDate = datePicker.extractDate("endDate");

			String[] finalArray = modelHandler.calculateTOLengthAndAccountedHours(context, empId, timeOffTypeId, startDate, endDate, false);

			context.getResponse().setContentType(RequestContext.DEFAULT_RESPONSE_CONTENT_TYPE);
			Writer out = context.getResponse().getWriter();
			JSONWriter writer = new JSONWriter(out);
			writer.object()
				.key("TOLengthFormatted")
				.value(finalArray[INDEX_ZERO])
				.key("accountedHoursFormatted")
				.value(finalArray[INDEX_ONE])
				.key("alertsFormatted")
				.value(finalArray[INDEX_TWO])
				.key("acountedHoursInMinutes")
				.value(finalArray[INDEX_THREE])
				.endObject();
			out.flush();
		} catch (Exception e) {
			LOG.error("Exception", e);
			throw new WebException();
		}
	}

	/**
	 * Process Save
	 */
	public void processSave(RequestContext context) {
		// Extract Form Parameter
		FlexTimeRequestFormPopupPM model = new FlexTimeRequestFormPopupPM(context);
		model.extractParameters(context.getRequest());

		if (model.validate()) {
			try {
				// Security check
				TORequest request = model.getTimeOffRequest();
				ID requestID = request.getID();
				if (!RequestUtil.isAuthToUpdate(context, request)) {
					return;
				}

				if (requestID == null) {
					requestID = modelHandler.createFTRequest(request, model.getComment());
				} else {
					modelHandler.updateFTRequest(request, model.getComment());
				}

				model.setSavedRequestID(requestID);
			} catch(MultiUserException ex) {
				model.addPageMessage(Message.WARNING_TYPE,
						RmWebBundleKey.UNABLE_TO_UPDATE_REQUEST_DUE_TO_MULTIUSER_EXCEPTION,
						RmWebBundleKey.BUNDLE_NAME);
			} catch(RmHardValidationException ex) {
				LOG.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
				Message msg = new Message(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
				msg.setAssociatedMessage( getLocMsgFromException(context, ex) );
				model.addPageMessage(msg);
			} catch(BbmUpdateException ex) {
				LOG.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
				Message msg = new Message(Message.ERROR_TYPE,	RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
				msg.setAssociatedMessage( ex.getLocalizedMessage(context.getLocaleContext().getLanguageLocale()));
				model.addPageMessage(msg);
			} catch(BbmCreateException ex) {
				LOG.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
				Message msg = new Message(Message.ERROR_TYPE,	RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
				msg.setAssociatedMessage( ex.getLocalizedMessage(context.getLocaleContext().getLanguageLocale()));
				model.addPageMessage(msg);
			} catch(Exception ex) {
				LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
				model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
			}
		}

		context.setPageModel(model);
	}

	/**
	 * Create AgentsRequestsPM or subclass
	 */
	@Override
	protected AgentsRequestsPM createAgentsRequestsPM(RequestContext context) {
		return new AgentsTimeOffRequestsPM(context);
	}
}
