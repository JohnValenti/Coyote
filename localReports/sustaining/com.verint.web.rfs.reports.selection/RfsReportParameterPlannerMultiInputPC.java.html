package com.verint.web.rfs.reports.selection;

import javax.servlet.http.HttpServletRequest;

import org.w3c.dom.Node;

import com.bluepumpkin.web.core.reports.ReportControlsXml;
import com.bluepumpkin.web.core.reports.ReportParametersPCInterface;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.bluepumpkin.web.ireport.framework.RffRequest;
import com.verint.web.rfs.reports.util.RfsReportParameterUtil;
import com.witness.web.uif.pagecomponent.CompositePC;
import com.witness.web.uif.pagecomponent.form.FormInputPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

public class RfsReportParameterPlannerMultiInputPC extends CompositePC implements ReportParametersPCInterface {

	private static final long serialVersionUID = 1L;
	
	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();
	
	private FormInputPC m_minPC = null;
	private FormInputPC m_maxPC = null;
	
	public RfsReportParameterPlannerMultiInputPC(RequestContext context, ToolbarPC toolbar, Node controlNode) {
		super(context);
		setName(reportControls.getControlSingleParameterName(controlNode));
		m_minPC = new FormInputPC(context);
		m_maxPC = new FormInputPC(context);
		
		m_minPC.setInputType(FormInputPC.TYPE_INTEGER);
		addChildComponent(m_minPC);
		m_maxPC.setInputType(FormInputPC.TYPE_INTEGER);
		addChildComponent(m_maxPC);

	}
	
	
	public void initForDisplay() {
		if (m_isInitializedForDisplay)
			return;

		super.initForDisplay();
	}
	
	public String getLabel(RffRequest rff, Node controlNode) {
		
		return "";
	}

	public String getHtmlDisplay() {
		// TODO Auto-generated method stub
		return null;
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode) {
		return getHtmlDisplay(rff, controlNode, false);
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode,boolean readOnly, String param, String empCatId) {
				
		StringBuffer content = new StringBuffer();
		if(param.equals(RfsReportParameterUtil.PARAM_HEADCOUNT)) {
			m_minPC.setInputFN("LowerHeadCounts"+empCatId);
			m_minPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_BELOW"));
			m_minPC.setInputValue("0");
			m_maxPC.setInputFN("UpperHeadCounts"+empCatId);
			m_maxPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_ABOVE"));
			m_maxPC.setInputValue("1000");
		} else if(param.equals(RfsReportParameterUtil.PARAM_PERCENT_HEADCOUNT)) {
			m_minPC.setInputFN("LowerPercentHeadCounts"+empCatId);
			m_minPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_MIN"));
			m_minPC.setInputValue("0");
			m_maxPC.setInputFN("UpperPercentHeadCounts"+empCatId);
			m_maxPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_MAX"));
			m_maxPC.setInputValue("100");
		} else if(param.equals(RfsReportParameterUtil.PARAM_PERCENT_FTE)) {
			m_minPC.setInputFN("LowerPercentFTEs"+empCatId);
			m_minPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_MIN"));
			m_minPC.setInputValue("0");
			m_maxPC.setInputFN("UpperPercentFTEs"+empCatId);
			m_maxPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_MAX"));
			m_maxPC.setInputValue("100");
		} else if(param.equals(RfsReportParameterUtil.PARAM_HC_VS_ACTUAL)) {
			m_minPC.setInputFN("LessActualHeadCounts"+empCatId);
			m_minPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_BELOW"));
			m_minPC.setInputValue("1000");
			m_maxPC.setInputFN("MoreActualHeadCounts"+empCatId);
			m_maxPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_ABOVE"));
			m_maxPC.setInputValue("1000");
		} else if(param.equals(RfsReportParameterUtil.PARAM_HC_VS_BUDGET)) {
			m_minPC.setInputFN("LessBudgetedHeadCounts"+empCatId);
			m_minPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_BELOW"));
			m_minPC.setInputValue("1000");
			m_maxPC.setInputFN("MoreBudgetedHeadCounts"+empCatId);
			m_maxPC.setInputFieldDesc(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_ABOVE"));
			m_maxPC.setInputValue("1000");
		}

		if(param.equals(RfsReportParameterUtil.PARAM_HEADCOUNT) ||
				param.equals(RfsReportParameterUtil.PARAM_HC_VS_ACTUAL) ||
				param.equals(RfsReportParameterUtil.PARAM_HC_VS_BUDGET)) 
		{
			content.append(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_BELOW"));
		} else if(param.equals(RfsReportParameterUtil.PARAM_PERCENT_HEADCOUNT) ||
				param.equals(RfsReportParameterUtil.PARAM_PERCENT_FTE))
		{
			content.append(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_MIN"));
		} 
		content.append(HtmlUtil.NBSP);
		content.append(m_minPC.getHtmlDisplay());
		content.append(HtmlUtil.NBSP);
		if(param.equals(RfsReportParameterUtil.PARAM_HEADCOUNT) ||
				param.equals(RfsReportParameterUtil.PARAM_HC_VS_ACTUAL) ||
				param.equals(RfsReportParameterUtil.PARAM_HC_VS_BUDGET)) 
		{
			content.append(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_ABOVE"));
		} else if(param.equals(RfsReportParameterUtil.PARAM_PERCENT_HEADCOUNT) ||
				param.equals(RfsReportParameterUtil.PARAM_PERCENT_FTE))
		{
			content.append(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PLANNER_FILTER_MAX"));
		} 
		content.append(HtmlUtil.NBSP);
		content.append(m_maxPC.getHtmlDisplay());
		
		return content.toString();
	}

	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode) {
		
	}

		return null;
	}

		super.extractParameters(request);
		return true;
	}

	public String getHtmlDisplay(RffRequest arg0, Node arg1, boolean arg2) {
		// TODO Auto-generated method stub
		return null;
	}
}
