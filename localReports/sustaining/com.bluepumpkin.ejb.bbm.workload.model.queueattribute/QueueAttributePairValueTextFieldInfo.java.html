import com.bluepumpkin.common.datatypes.*;

import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class QueueAttributePairValueTextFieldInfo extends FieldInfo {

    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// field id constants
    public static final int QAPV_ID = 0;
    public static final int QAPV_QUEUEATTRIBUTEID = 1;
    public static final int QAPV_QUEUEID = 2;
    public static final int QAPV_VALUE = 3;
    
	public static final int QAPV_NUMOFFIELDS = 4;


    private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[QAPV_NUMOFFIELDS];
    private static String m_strTableName = "QUEUEATTRIBUTEPAIRVALUETEXT";

    // static initialization
    static {
      m_arrFieldInfo[QAPV_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
      m_arrFieldInfo[QAPV_QUEUEATTRIBUTEID] = new FieldInfoEntry("QUEUEATTRIBUTEID", PrimitiveTypes.ID);
      m_arrFieldInfo[QAPV_QUEUEID] = new FieldInfoEntry("QUEUEID", PrimitiveTypes.ID);
      m_arrFieldInfo[QAPV_VALUE] = new FieldInfoEntry("VALUE", PrimitiveTypes.STRING);
    }

    public QueueAttributePairValueTextFieldInfo() {
        super();
    }

    public int getParentIDFieldIndex() {
    	return QAPV_QUEUEID;
    }
    
    // implement abstract methods in parent class
    public String getTableName() { return m_strTableName; }
    public int getIDFieldIndex() { return QAPV_ID; }
    
    public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }
}
