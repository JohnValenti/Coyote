package com.verint.ejb.bbm.forecast;

import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.util.TraceUtil;
import com.bluepumpkin.ejb.bbm.workload.model.TimeUnit;


/**
 * The Class TraceCubeValuesConverterByTimeUnit. This is a general purpose class to convert trace cube values from and into
 * seconds. The behavior is controlled by the constructor parameters timeunit and traceTypesToConvert.
 * The converter will only attempt to convert the traces of the given type and only from and to the given time unit.
 */
public class TraceCubeValuesConverterByTimeUnit {

	private short[] m_traceTypesToConvert;
	private TimeUnit m_timeUnit;
	
	/**
	 * Instantiates a new instance of TraceCubeValuesConverterByTimeUnit.
	 * The converter will only attempt to convert the traces of the given type and only from and to the given time unit.
	 *
	 * @param traceTypesToConvert the trace types to convert values for.
	 * @param timeUnit the time unit
	 */
	public TraceCubeValuesConverterByTimeUnit(short[] traceTypesToConvert, TimeUnit timeUnit) {
		m_traceTypesToConvert = traceTypesToConvert;
		m_timeUnit = timeUnit;
	}
	
	/**
	 * Converts values in a trace cube into the encapsulated time unit in this object.
	 *
	 * @param cubeInSeconds the cube in seconds
	 */
	public void convertValues(TraceCube cubeInSeconds) {
		if (cubeInSeconds != null) {
			TraceUtil.convertSeconds(cubeInSeconds, m_traceTypesToConvert, m_timeUnit);
		}
	}
	
	/**
	 * Converts values to seconds from the time unit encapsulated in this object.
	 *
	 * @param cubeInTimeUnit the original cube in time unit
	 */
	public void convertValuesToSeconds(TraceCube cubeInTimeUnit) {
		if (cubeInTimeUnit != null) {
			TraceUtil.convertValuesToSeconds(cubeInTimeUnit, m_traceTypesToConvert, m_timeUnit);
		}
	}
}
