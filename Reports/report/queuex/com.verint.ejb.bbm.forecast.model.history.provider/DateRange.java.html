package com.verint.ejb.bbm.forecast.model.history.provider;

import java.util.Date;

class DateRange {
	public Date start;
	public Date end;

	public DateRange(Date start, Date end) {
		this.start = start;
		this.end = end;
	}

	public boolean equals(Object rangeToCompare) {
		if (this == rangeToCompare) {
			return true;
		}
		if (!(rangeToCompare instanceof DateRange)){
			return false;
		}
		DateRange other = (DateRange)rangeToCompare;
		return this.start.equals(other.start) && this.end.equals(other.end);
	}

	public int hashCode() {
		int result = 17;
		result = 37 * result + start.hashCode();
		result = 37 * result + end.hashCode();
		return result;
	}
}
