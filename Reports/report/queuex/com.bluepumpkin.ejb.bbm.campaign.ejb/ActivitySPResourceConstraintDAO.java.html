package com.bluepumpkin.ejb.bbm.campaign.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.activity.model.ActivitySPResourceConstraint;
import com.bluepumpkin.ejb.bbm.activity.model.ActivitySPResourceConstraintFieldInfo;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOEffectivity;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

import java.util.List;

public class ActivitySPResourceConstraintDAO extends DAOEffectivity {

	private static FieldInfo m_fieldInfo = new ActivitySPResourceConstraintFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ActivitySPResourceConstraintDAO() {
		super();
	}

	public ActivitySPResourceConstraintDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ValueObjectBase createValueObject() {
		return (new ActivitySPResourceConstraint());
	}

	public List<ActivitySPResourceConstraint> getActivitySPResourceConstraintsBySP(ID schedulingPeriodSID)
			throws BbmFinderException {
		return (List<ActivitySPResourceConstraint>) getObjects("SPSID = '" + schedulingPeriodSID + "'");
	}
}
