import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.campaign.model.SP;
import com.bluepumpkin.ejb.bbm.campaign.model.SPFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**
 * This DAO hits the SP table (like SchedulingPeriodDAO), but it only loads
 * fields that are not loaded by SchedulingPeriodDAO. These fields are related
 * to FTE: FTESHIFTS. FTEWORKHOURS, etc.
 */
@Deprecated
public class SPDAO {
	private static FieldInfo m_fieldInfo = new SPFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	private static final int COL_INDEX_ID = 1;

	/**
	 * default constructor
	 */
	public SPDAO() {
		super();
	}

	protected ValueObjectBase createValueObject() {
		return (new SP());
	}

	/**
	 * Update the SP object from the SP table (as opposed to the
	 * SchedulingPeriod object). We only update the following fields for now:
	 * FTESHIFTS, FTEWORKHOURS, FTEPAIDHOURS, FTEPHONEHOURS, FTEWAGE.
	 *
	 * @param spID - The SP ID
	 * @return An SP object.
	 * @throws BbmFinderException
	 */
	public static void updateRealSP(SP sp) throws BbmUpdateException {
		Jdmo jdmo = new Jdmo();
		try {
			if (sp == null || sp.getID() == null) {
				return;
			}

			String strQuery = "update SP set FTESHIFTS = " + sp.getFteShifts() + ", FTEWORKHOURS = " + sp.getFteWorkHours()
					+ ", FTEPAIDHOURS = " + sp.getFtePaidHours() + ", FTEPHONEHOURS = " + sp.getFtePhoneHours()
					+ ", FTEWAGE = " + sp.getFteWage() + " where ID = '" + sp.getId().toString() + "'";
			jdmo.executeCommand(strQuery);
		} catch (JdmoException ex) {
			throw new BbmUpdateException(ex);
		} finally {
			jdmo.cleanUp();
		}
	}

	/**
	 * Get SP by id. This is just a simplified SP object for now. It only
	 * incudes the following fields: ID, FTESHIFTS, FTEWORKHOURS, FTEPAIDHOURS,
	 * FTEPHONEHOURS, FTEWAGE
	 */
	public static SP getSPByID(ID id) throws BbmFinderException, BbmObjectNotFoundException {
		Jdmo jdmo = new Jdmo();
		SP sp = null;
		try {
			StringBuffer pStmt = new StringBuffer(
					"select ID, FTESHIFTS, FTEWORKHOURS, FTEPAIDHOURS, FTEPHONEHOURS, FTEWAGE ");
			pStmt.append("from SP where SID=?");
			JdmoQuery jq1 = jdmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			jq1.setParString(1, id.toString());
			JdmoRowset rs = jdmo.createRowset(jq1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

			if (rs.next()) {
				sp = new SP();

				sp.setId(rs.getID(COL_INDEX_ID));
				sp.setFteShifts(rs.getInt(2));
				sp.setFteWorkHours(rs.getInt(3));
				sp.setFtePaidHours(rs.getInt(4));
				sp.setFtePhoneHours(rs.getInt(5));
				sp.setFteWage(rs.getInt(6));
			}

			if (sp == null) {
				throw new BbmObjectNotFoundException(id.toString());
			}

		} catch (Exception e) {
			throw new BbmFinderException(e);
		} finally {
			jdmo.cleanUp();
		}

		return sp;
	}
}
