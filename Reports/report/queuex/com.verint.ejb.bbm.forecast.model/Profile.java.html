package com.verint.ejb.bbm.forecast.model;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeContext;
import com.bluepumpkin.ejb.bbm.time.TimeInterval;
import com.bluepumpkin.ejb.bbm.time.TimeIntervalAtTime;
import com.bluepumpkin.ejb.bbm.time.TimeIntervalFactory;
import com.bluepumpkin.ejb.bbm.time.TimeUnits;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.verint.ejb.bbm.forecast.model.history.ProfileEntry;

/**
 * This class is a model for a forecast profile composed of one or more weeks of
 * historical data combined using a weighted average according to externally
 * supplied weights for each historical week.
 */
public class Profile {
	private ForecastProfile m_forecastProfile;
	private final TimeIntervalAtTime m_interval;
	private List<ProfileListener> m_listeners = new ArrayList<ProfileListener>();
	private Collection<ProfileEntry> m_entries = new ArrayList<ProfileEntry>();
	private Queue m_queue;
	private int m_strategicVolumeWeight;
	private int m_strategicAhtWeight;

	/**
	 * To be used only when forecast profile has a start date (i.e., it is not
	 * part of a manually-saved profile).
	 *
	 * @param forecastProfile
	 * @param queue
	 * @param timeContext
	 */
	public Profile(ForecastProfile forecastProfile, Queue queue, TimeContext timeContext) {
		m_forecastProfile = forecastProfile;
		m_queue = queue;
		m_interval = createTimeIntervalForForecastProfile(forecastProfile, timeContext);
		m_entries.addAll(convertToProfileEntries(forecastProfile.getProfileComponents(), timeContext));
		m_strategicVolumeWeight = forecastProfile.getStrategicCallVolumeWeight();
		m_strategicAhtWeight = forecastProfile.getStrategicAverageHandlingTimeWeight();
	}

	/**
	 * To be used only when forecast profile has no start date (i.e., it is part
	 * of a manually-saved profile).
	 *
	 * @param forecastProfile
	 * @param interval
	 *            the time interval to be used when interpreting the profile.
	 *            Should represent the interval of the SP week.
	 * @param queue
	 * @param timeContext
	 */
	public Profile(ForecastProfile forecastProfile, TimeIntervalAtTime interval, Queue queue, TimeContext timeContext) {
		m_forecastProfile = forecastProfile;
		m_queue = queue;
		m_interval = new TimeIntervalAtTime(interval, timeContext);
		m_entries.addAll(convertToProfileEntries(forecastProfile.getProfileComponents(), timeContext));
		m_strategicVolumeWeight = forecastProfile.getStrategicCallVolumeWeight();
		m_strategicAhtWeight = forecastProfile.getStrategicAverageHandlingTimeWeight();
	}

	public Profile(Profile profileToCopy) {
		try {
			m_forecastProfile = (ForecastProfile) profileToCopy.getForecastProfile().clone();
			m_forecastProfile.setID(null);
			m_interval = profileToCopy.m_interval;
			m_queue = profileToCopy.getQueue();
			for (ProfileEntry pe : profileToCopy.getProfileEntries()) {
				m_entries.add(new ProfileEntry(pe));
			}
			m_strategicVolumeWeight = profileToCopy.m_strategicVolumeWeight;
			m_strategicAhtWeight = profileToCopy.m_strategicAhtWeight;
		} catch (CloneNotSupportedException e) {
			throw new RuntimeException(e);
		}
	}

	/**
	 * Clears the profile entry list and the underlying ForecastProfile
	 *
	 * @param fireEventsForSetProfileEntries
	 *            - true if you want setProfileEntries() to fire events, false
	 *            otherwise.
	 */
	public void clear() {
		ID spQueueID = getSPQueueID();
		ForecastProfile oldFP = m_forecastProfile;
		m_forecastProfile = new ForecastProfile();
		m_forecastProfile.setStartDate(m_interval.getStartTime());
		m_forecastProfile.setEndDate(m_interval.getEndTime());
		m_forecastProfile.setSPQueueID(spQueueID);
		this.setProfileEntries(new ArrayList<ProfileEntry>(0));
		this.setIsModified(false);
		if (oldFP.isModified()) {
			fireProfileModified(true, false);
		}
	}

	public ForecastProfile getForecastProfile() {
		return m_forecastProfile;
	}

	/**
	 * Returns the ID of the underlying forecast profile.
	 */
	public ID getID() {
		return m_forecastProfile.getID();
	}

	/**
	 * Gets the copy of the server side forecast profile object encapsulated by
	 * this profile. The ID of the copy is set to null.
	 *
	 * @return the forecast profile copy
	 * @throws CloneNotSupportedException
	 *             the clone not supported exception
	 */
	public ForecastProfile getForecastProfileCopy() throws CloneNotSupportedException {
		ForecastProfile forecastProfile = m_forecastProfile.cloneForecastProfileValueFields();
		return forecastProfile;
	}

	public Queue getQueue() {
		return m_queue;
	}

	public Collection<ProfileEntry> getProfileEntries() {
		return Collections.unmodifiableCollection(m_entries);
	}

	public void setProfileEntries(Collection<ProfileEntry> entries) {
		m_entries = new ArrayList<ProfileEntry>(entries);
		Collection<ProfileComponent> pcCol = new ArrayList<ProfileComponent>(entries.size());
		for (ProfileEntry pe : entries) {
			pcCol.add(pe.getProfileComponent());
		}
		m_forecastProfile.setProfileComponents(pcCol);
	}

	public void addProfileEntry(ProfileEntry profileEntry) {
		m_entries.add(profileEntry);
		m_forecastProfile.addProfileComponents(Collections.singleton(profileEntry.getProfileComponent()));
	}

	/**
	 * ESR#4163092: Creating Combined Forecast in WFO Suite is taking too long
	 * to complete. Fire events after adding, removing, or setting a
	 * profileEntry. The listener is CompositeProfile, which just calls
	 * updateCompositeProfileEntries() with no parameters, so we can just pass
	 * null paramters to the listerner.
	 * @param recalc - Pass true if you want the forecast to be recalculated, false otherwise.
	 */
	public void fireEventsForProfileEntryChange(boolean recalc) {
		Date start = new Date();
		fireProfileEntriesChanged(null, null,recalc);
		Date end = new Date();
		double duration = (end.getTime() - start.getTime()) / 1000.0;
		System.out.println("Profile.fireProfileEntriesChanged duration= " + duration + " sec\n");
	}

	public ID getSPQueueID() {
		return m_forecastProfile.getSPQueueID();
	}

	public boolean isModified() {
		return m_forecastProfile.isModified();
	}

	public void setIsModified(boolean modified) {
		if (modified != isModified()) {
			m_forecastProfile.setIsModified(modified);
			fireProfileModified(!modified, modified);
		}
	}

	public void removeProfileEntries(Collection<ProfileEntry> profileEntries, boolean fireEvents) {
		Collection<ProfileEntry> old = new ArrayList<ProfileEntry>(m_entries);
		m_entries.removeAll(profileEntries);
		for (ProfileEntry entry : profileEntries) {
			m_forecastProfile.removeProfileComponents(Collections.singleton(entry.getProfileComponent()));
		}
		if (fireEvents) {
			fireProfileEntriesChanged(old, new ArrayList<ProfileEntry>(m_entries),true);
		}
	}

	public void addListener(ProfileListener listener) {
		if (!m_listeners.contains(listener)) {
			m_listeners.add(listener);
		}
	}

	public void removeListener(ProfileListener listener) {
		m_listeners.remove(listener);
	}

	private void fireProfileModified(boolean oldValue, boolean newValue) {
		for (ProfileListener listener : m_listeners) {
			listener.modifiedChanged(this, oldValue, newValue);
		}
	}

	private void fireProfileEntriesChanged(Collection<ProfileEntry> oldProfileEntries, Collection<ProfileEntry> newProfileEntries,
			boolean recalc) {
		for (ProfileListener listener : m_listeners) {
			listener.profileEntriesChanged(this, oldProfileEntries, newProfileEntries, recalc);
		}
	}

	private Collection<ProfileEntry> convertToProfileEntries(Collection<ProfileComponent> profileComponents,
			TimeContext timeContext) {
		ArrayList<ProfileEntry> retVal = new ArrayList<ProfileEntry>();
		for (ProfileComponent pc : profileComponents) {
			// TODO: If the queue is null (combined), is any more descriptive
			// queue name required?
			retVal.add(new ProfileEntry(pc, timeContext, m_interval.getStartTime(),
					(m_queue != null) ? m_queue.getName() : ""));
		}
		return retVal;
	}

	public TimeIntervalAtTime getInterval() {
		return m_interval;
	}

	public Date getStartDate() {
		return m_interval.getStartTime();
	}

	/**
	 * Returns true if this profile contains a profile entry with the same
	 * historical interval as the input entry.
	 */
	public boolean containsEntry(ProfileEntry entry) {
		for (ProfileEntry pe : getProfileEntries()) {
			if (pe.getHistoryPeriod().equals(entry.getHistoryPeriod())) {
				return true;
			}
		}
		return false;
	}

	/**
	 * Returns the weight from 0 to 100 percent, inclusive, that represents the
	 * relative importance of the strategic volume forecast in the overall
	 * forecast calculation. 0 indicates that the strategic volume forecast is
	 * to be disregarded; 100 indicates that the tactical volume forecast is to
	 * be disregarded.
	 *
	 * @return
	 */
	public int getStrategicVolumeWeight() {
		return m_strategicVolumeWeight;
	}

	/**
	 * Returns the weight from 0 to 100 percent, inclusive, that represents the
	 * relative importance of the strategic AHT forecast in the overall forecast
	 * calculation. 0 indicates that the strategic AHT forecast is to be
	 * disregarded; 100 indicates that the tactical AHT forecast is to be
	 * disregarded.
	 *
	 * @return
	 */
	public int getStrategicAverageHandlingTimeWeight() {
		return m_strategicAhtWeight;
	}

	public void setStrategicVolumeWeight(int cvW) {
		if (m_strategicVolumeWeight != cvW) {
			int oldV = m_strategicVolumeWeight;
			m_strategicVolumeWeight = cvW;
			fireStrategicCVWeightChanged(oldV, cvW);
		}
	}

	public void setStrategicAHTWeight(int ahtW) {
		if (m_strategicAhtWeight != ahtW) {
			int oldV = m_strategicAhtWeight;
			m_strategicAhtWeight = ahtW;
			fireStrategicAHTWeightChanged(oldV, ahtW);
		}

	}

	private void fireStrategicCVWeightChanged(int oldValue, int newValue) {
		for (ProfileListener listener : m_listeners) {
			listener.strategicCVWeightChanged(this, oldValue, newValue);
		}
	}

	private void fireStrategicAHTWeightChanged(int oldValue, int newValue) {
		for (ProfileListener listener : m_listeners) {
			listener.strategicAHTWeightChanged(this, oldValue, newValue);
		}
	}

	/**
	 * Creates a TimeIntervalAtTime which spans the length of the given forecastProfile.  Assumes that forecastProfile
	 * has its start/end dates set.
	 */
	private static TimeIntervalAtTime createTimeIntervalForForecastProfile(ForecastProfile forecastProfile, TimeContext timeContext) {
		TimeInterval interval = new TimeInterval(TimeUnits.Day, forecastProfile.getStartDate(),
				forecastProfile.getEndDate(), timeContext);
		if (interval.getUnitCount() < 7) {
			return new TimeIntervalAtTime(forecastProfile.getStartDate(), interval);
		}
		return new TimeIntervalAtTime(forecastProfile.getStartDate(), new TimeIntervalFactory(timeContext).getOneWeek());
	}

	@Override
	public String toString() {
		return "Profile[ForecastProfileId: " + m_forecastProfile.getID() + " Interval: " + m_interval.toString() +
				" Queue: " + m_queue.getID() + "]";
	}
}
