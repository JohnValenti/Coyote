package com.verint.ejb.bbm.forecast.ejb;

import java.util.Calendar;
import java.util.Collection;
import java.util.Date;
import java.util.HashMap;
import java.util.LinkedList;
import java.util.Map;

import com.bluepumpkin.common.datatypes.Duration;
import com.bluepumpkin.common.datatypes.TimeContext;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastTimeSeries;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastTimeSeriesFieldInfo;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.util.DSTConversionUtil;

/**
 * This class contains methods to convert between a Collection of ForecastTimeSeries and a Collection
 * of ForecastTraceCubes.
 * 
 * @author rfunderburg
 *
 */
public class ForecastTimeSeriesUtil {
	private static final Duration intervalLength = Duration.fromMinutes(Trace.INTERVAL);
	
	// A map of supported types to Field Prefixes used in the FieldInfo for that type.
	// Only support AHT and CV for now. Later add support for the other trace types:
	// Trace.CRATE,	Trace.RPCRATE,	Trace.RPCAHT
	private static final Map<Short, String> supportedTypes = new HashMap<Short, String>();
	static {
		supportedTypes.put(Trace.CV, "CALLVOLUME");
		supportedTypes.put(Trace.AHT, "AHT");
		supportedTypes.put(Trace.CRATE, "CRATE");
		supportedTypes.put(Trace.RPCRATE, "RPCRATE");
		supportedTypes.put(Trace.RPCAHT, "RPCAHT");
	}
	
	/**
	 * This is a double array of FieldInfo indexes.
	 * First dimension is type. Second is the interval number during the day (0-95).
	 * 
	 * This is used so we can set the field values using ValueObjectBase.setFieldValue(index).
	 */
	private static final int[][] valueIndices = new int[Trace.MAX_TYPE][96];
	static {
		ForecastTimeSeriesFieldInfo fi = new ForecastTimeSeriesFieldInfo();
		for (short type : supportedTypes.keySet()) {
			String prefix = supportedTypes.get(type);
			valueIndices[type] = new int[96];
			for (int i = 0; i < 96; i++) {
				valueIndices[type][i] = fi.getFieldIndexByName(prefix + (i + 1));
			}
		}
	}
	
	/**
	 * Constructs ForecastTimeSeries objects that contain the trace values of the given TraceCube
	 * Only sets trace values and date. Does not set any other fields.
	 * @param traceCube
	 * @param timeContext
	 * @return
	 */
	public static Collection<ForecastTimeSeries> getForecastTimeSeriesCol(TraceCube traceCube, TimeContext timeContext) {
		Date startDate = traceCube.getRawStartDate();
		Date endDate = traceCube.getRawEndDate();
		Collection<ForecastTimeSeries> retVal = new LinkedList<ForecastTimeSeries>();
		Calendar timeCursor = Calendar.getInstance(timeContext.getTimeZone());
		timeCursor.setTime(startDate);
		while (timeCursor.getTime().before(endDate)) {
			Date currentDay = timeCursor.getTime();
			ForecastTimeSeries fts = new ForecastTimeSeries();
			fts.setDateTime(timeCursor.getTime());
			
			DSTConversionUtil.adjustCalendarForDST(timeCursor, timeContext, 1);
			
			// Subtract one millisecond, since the Trace functions are end inclusive.
			Date nextDay = new Date(timeCursor.getTime().getTime() - 1);
			
			for (short type : supportedTypes.keySet()) {
				double[] traceVals = traceCube.getTraceValueD(type, currentDay, nextDay);
				if (traceVals != null) {
					double[] vals = DSTConversionUtil.convertFromDSTAwareTimeToIdealTime(
							traceCube.getTraceValueD(type, currentDay, nextDay),
							Trace.TRACENA, 
							currentDay, 
							intervalLength, 
							timeContext);
					if (vals.length == 0) continue;
					if (vals.length != 96) throw new IllegalArgumentException("TraceCube for queue " + traceCube.getQueueID() + " contains " + vals.length + " DST-corrected values for type " + Trace.getTraceName(type) + " for day " + currentDay + ". Expected 96.");
	
					for (int i = 0; i < 96; i++) {
						fts.setFieldValue(valueIndices[type][i % 96], convertTraceOffToZero(vals[i]));
					}
				}
			}
			retVal.add(fts);
		}
		return retVal;
	}

	private static double convertTraceOffToZero(double value) {
		return value == Trace.TRACEOFF ? 0.0d : value;
	}

}
