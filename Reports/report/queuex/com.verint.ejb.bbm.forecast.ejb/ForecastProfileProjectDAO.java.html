package com.verint.ejb.bbm.forecast.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueueFieldInfo;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.dao.DAOUtil;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.verint.ejb.bbm.forecast.model.ForecastProfileProject;
import com.verint.ejb.bbm.forecast.model.ForecastProfileProjectFieldInfo;
import java.util.Collection;
import java.util.Collections;

public class ForecastProfileProjectDAO extends DAOBase<ForecastProfileProject> {

	private static Category LOG = Log.initCategory(ForecastProfileProjectDAO.class.getName());
	private static FieldInfo m_fieldInfo = new ForecastProfileProjectFieldInfo();

	public ForecastProfileProjectDAO() {
		super();
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	@Override
	protected ForecastProfileProject createValueObject() {
		return new ForecastProfileProject();
	}

	public void createProject(ForecastProfileProject project) throws BbmCreateException {
		if(project == null) {
			return;
		}
		createObject(project);
	}

	public void createProjects(Collection<ForecastProfileProject> projects) throws BbmCreateException {
		if (projects != null && !projects.isEmpty()) {
			createObjects(projects);
		}
	}

	public void deleteProject(ID projectID) throws BbmRemoveException{
		if(projectID == null) {
			return;
		}
		deleteObject(projectID);
	}

	public void deleteProjects(Collection <ID> projectIDs) throws BbmRemoveException {
		deleteObjects(projectIDs);
	}

	public Collection<ForecastProfileProject> getActiveForecastProfileProjects(Collection<ID> spQueueIDs) throws JdmoException, 
		BbmFinderException {
		try {
			if (spQueueIDs == null) {
				return Collections.emptyList();
			}

			String strWhere = " A.FORECASTPROFILEID in (select ID from FORECASTPROFILE FP where FP.SPQUEUEID in " + m_dmo.createInClause
				(DAOUtil.mapIDsToDEIDs(spQueueIDs, new SPQueueFieldInfo()).values()) + " and FP.FORECASTINSTANCEID is null)";

			return getObjects(strWhere, "ORDER BY A.QUEUEID,A.STARTTIME ASC ");
		} catch(JdmoException e) {
			LOG.error("Exception getting active forecast profile projects.", e);
			throw new BbmFinderException(e);
		}
	}

	/**
	 * Deletes all ForecastProfileProject records with the given forecast profile IDs.
	 */
	public void deleteByForecastProfileIDs(Collection<ID> forecastProfileIDs) throws BbmRemoveException{
		try {
			String where = "SELECT ID FROM " + getFieldInfo().getTableName() + " A WHERE A.FORECASTPROFILEID IN " +
				getDMO().createInClause(forecastProfileIDs);
			deleteObjects(where);
		} catch (JdmoException ex) {
			throw new BbmRemoveException(ex);
		}
	}
}
