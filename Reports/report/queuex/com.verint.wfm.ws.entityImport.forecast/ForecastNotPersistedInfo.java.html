package com.verint.wfm.ws.entityImport.forecast;

import java.util.Calendar;
import java.util.List;

import com.verint.wfm.ws.entityImport.EntityNotPersistedInfo;
import com.verint.wfm.ws.entityImport.forecast.model.AbstractForecastDTO.ForecastExternalNaturalKey;

/**
 * For each scheduling period not persisted by createOrUpdateOrganizations(), one of these will be
 * in the list returned This is returned in lieu of an exception, because we always want to continue
 * on to all the organizations
 * 
 * 
 * @see EntityImportService.createOrUpdateOrganizations()
 * 
 */
public class ForecastNotPersistedInfo extends EntityNotPersistedInfo<ForecastExternalNaturalKey> {
	private static final long serialVersionUID = 1L;

	public ForecastNotPersistedInfo() {
		super();
	}

	public ForecastNotPersistedInfo(ForecastExternalNaturalKey forecastExternalNaturalKey) {
		super(forecastExternalNaturalKey);
	}

	// -----------------------------------------
	public ForecastExternalNaturalKey getForecastExternalNaturalKey() {
		return super.getEntityExternalSourceId();
	}

	/**
	 * Dummy here on server, but needed for SOAP stub
	 */
	public void setForecastExternalNaturalKey(ForecastExternalNaturalKey forecastExternalNaturalKey) {
		// do nothing
	}

	// / -------------------------------------------------------- Add error messages from here
	public void addExternalNaturalKeyMissingMessage(String forecastDescription) {
		getMessages().add(
				String.format("External Key was missing for forecast with name %s.", forecastDescription));
	}

	public void addCampaignExternalSourceIdInKeyMissingMessage() {
		getMessages()
				.add(String
						.format("Forecast with incomplete external key %s missing the required property campaign external id.",
								getExternalIdSummary()));
	}

	public void addStartDateInKeyMissingMessage() {
		getMessages()
				.add(String
						.format("Forecast with incomplete external key %s missing the required property period start date.",
								getExternalIdSummary()));
	}

	public void addQueueExternalSourceIdInKeyMissingMessage(String campaignExternalSourceId) {
		getMessages()
				.add(String
						.format("Forecast with incomplete external key %s missing the required property queue external id.",
								getExternalIdSummary()));
	}

	public void addTimeSeriesTypesMissingMessage() {
		getMessages().add(
				String.format("Forecast with %s is missing time series types.", getExternalIdSummary()));
	}

	public void addTimeSeriesTypeInvalidMessage(String timeSeriesType) {
		getMessages().add(
				String.format("Forecast %s has an invalid time series type %s.", getExternalIdSummary(),
						timeSeriesType));
	}

	public void addNoSuchQueueExternalIdMessage() {
		getMessages().add(
				String.format("In Forecast %s the queue specified to have external id %s does not exist.",
						getExternalIdSummary(), getEntityExternalSourceId().getQueueExternalSourceId()));
	}

	public void addNoSuchCampaignExternalIdMessage() {
		getMessages().add(
				String.format("In Forecast %s the campaign specified to have external id %s does not exist.",
						getExternalIdSummary(), getEntityExternalSourceId().getCampaignExternalSourceId()));
	}

	public void addNoSuchSchedulingPeriodQueueForForecastExternalNaturalKeyMessage() {
		getMessages().add(
				String.format("There is no such scheduling period queue for forecast external key %s.",
						getExternalIdSummary()));
	}

	public void addNoSuchSchedulingPeriodForForecastExternalNaturalKeyMessage() {
		getMessages().add(
				String.format("In Forecast %s, there is no scheduling period for the campaign external"
						+ " id and start date specified.", getExternalIdSummary()));
	}

	public void addTimeSeriesLengthExceedsSchedulingPeriodDurationMessage(Calendar forecastEndDate,
			Calendar schedulingPeriodEndDate) {
		getMessages().add(
				String.format("Forecast %s has a time series stretching out to %tc exceeding the the"
						+ " scheduling period which ends at %tc.", getExternalIdSummary(), forecastEndDate,
						schedulingPeriodEndDate));
	}

	public void addTimeSeriesLengthFallBehindSchedulingPeriodDurationMessage(int timeSeriesLength,
			int daysInPeriod) {
		getMessages()
				.add(String
						.format("Forecast %s has a time series of length %d which is less than 96 times the number of days %d in the scheduling period",
								getExternalIdSummary(), timeSeriesLength, daysInPeriod));
	}

	public void addProjectForecastForNonProjectMediumMessage(String mediumName) {
		getMessages().add(
				String.format("A project forecast %s is not allowed for scheduling period queue with non project medium %s.",
						getExternalIdSummary(), mediumName));
	}
	public void addTimeSeriesForecastForProjectMediumMessage() {
		getMessages().add(
				String.format("A time series forecast %s is not allowed for scheduling period queue with project medium %s.",
						getExternalIdSummary()));
		
	}
	
	public void addUnhandledMediumMessage(String mediumName) {
		getMessages().add(
				String.format("A Forecast can't be handled for medium %s that is neither immediate, deferred or project.",
						getExternalIdSummary(), mediumName));
	}
	
	
	/**
	 * Axis bug work around
	 */
	@Override
	public List<String> getMessages() {
		return super.getMessages();
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public void setMessages(List<String> messages) {
		// Do nothing
	}
}
