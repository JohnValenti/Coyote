package com.verint.web.fs.skills;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;

public class SkillsAssignmentUtil {

	public static String getClassificationString(Localizer localizer, int classification) {
		switch (classification) {
			case 0:
				return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.EMPLOYEE_SKILL_CLASSIFICATION_PRIMARY);
			case 1:
				return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.EMPLOYEE_SKILL_CLASSIFICATION_RESERVE1);
			case 2:
				return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.EMPLOYEE_SKILL_CLASSIFICATION_RESERVE2);
		}
		return "";
	}
}
