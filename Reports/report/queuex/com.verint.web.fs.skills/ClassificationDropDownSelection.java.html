package com.verint.web.fs.skills;

import com.bluepumpkin.web.core.pagecomponent.DropDownSelectionPC;
import com.witness.web.uif.system.RequestContext;

public class ClassificationDropDownSelection extends DropDownSelectionPC<Integer> {

	public ClassificationDropDownSelection(RequestContext context, String name) {
		super(context, name);

		this.addOption("0", SkillsAssignmentUtil.getClassificationString(context.getLocalizer(), 0), 0, true);
		this.addOption("1", SkillsAssignmentUtil.getClassificationString(context.getLocalizer(), 1), 1, false);
		this.addOption("2", SkillsAssignmentUtil.getClassificationString(context.getLocalizer(), 2), 2, false);
	}
}
