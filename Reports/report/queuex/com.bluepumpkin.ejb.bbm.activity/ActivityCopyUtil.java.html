package com.bluepumpkin.ejb.bbm.activity;

import java.rmi.RemoteException;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.Pair;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.activity.ejb.ActivityManager;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategoryFilter;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityProperties;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityValidationException;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbBundleKey;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.ejb.bbm.LinkedObject;
import com.witness.ejb.bbm.ObjectInfo;
import com.witness.ejb.bbm.copy.CopyableObjectPlugin;
import com.witness.ejb.bbm.copy.ejb.CopyDAO;
import com.witness.ejb.bbm.copy.model.CopyResult;

public final class ActivityCopyUtil {

	private ActivityCopyUtil() {

	}

	public static CopyResult copyActivity(ID idActivity, ID idOrg, boolean bForce, CopyDAO pDAO, Pair pIDPair,
			String sCopyPrefix) throws BbmCreateException, BbmFinderException, ActivityValidationException {
		try {
			ActivityManager pActivityManager = WfmManagerFactory.getActivityManager(pDAO.isWhatIfMode());
			// get the activity to be copied
			Activity pActivity = pActivityManager.findActivityById(idActivity);

			// get the associated mediaids, if any
			Collection mediaIds = pActivityManager.findMediaForActivity(idActivity);

			// get the associated queues, if any
			Collection queueIds = pActivityManager.findQueueForActivity(idActivity);

			// get the linked objects (activity cats)
			ActivityCategory pCat = pActivityManager.findActivityCategoryById(pActivity.getActivityCategoryId());

			Collection cActivityCategories = pActivityManager.findOrganizationActivityCategories(idOrg,
					new ActivityCategoryFilter());
			
			// get associated activity properties
			ActivityProperties activityProperties = pActivityManager.findPropertiesForActivity(idActivity);
			activityProperties.setActivityId(idActivity);

			// see if the activity cat exists at the new org
			ID idFoundCat = null;
			for (Iterator i = cActivityCategories.iterator(); i.hasNext();) {
				ActivityCategory pCat2 = (ActivityCategory) i.next();
				if (pActivity.getActivityCategoryId().equals(pCat2.getID())) {
					idFoundCat = pCat2.getID();
					break;
				}
			}

			// if not check see if an activity cat with a similar name exists at
			// the new org
			if (idFoundCat == null) {
				for (Iterator i = cActivityCategories.iterator(); i.hasNext();) {
					ActivityCategory pCat2 = (ActivityCategory) i.next();
					if (CopyableObjectPlugin.isSubString(pCat.getName(), pCat2.getName())) {
						idFoundCat = pCat2.getID();
						break;
					}
				}
			}

			// if found rename and copy the activity, return COPIED
			if (idFoundCat != null) {
				ID idCopiedActivity = createCopyActivity(pActivityManager, pActivity, idOrg, idFoundCat, mediaIds,
						queueIds, activityProperties, pDAO, sCopyPrefix);
				pIDPair.setFirst(idActivity);
				pIDPair.setSecond(idActivity);
				return new CopyResult();
			}

			// else if bForce, perform a deep copy, return COPIED
			if (bForce) {
				ID idCopiedCat = createCopyActivityCategory(pActivityManager, pCat, idOrg, pDAO, sCopyPrefix);
				ID idCopiedActivity = createCopyActivity(pActivityManager, pActivity, idOrg, idCopiedCat, mediaIds,
						queueIds, activityProperties, pDAO, sCopyPrefix);
				pIDPair.setFirst(idActivity);
				pIDPair.setSecond(idActivity);
				return new CopyResult();
			}

			// else return CONFIRM_DEEP_COPY
			Collection cLinkedObjects = new ArrayList();
			cLinkedObjects.add(new LinkedObject(pCat.getID(), pCat.getName(), ObjectInfo.BPOBJECT_ACTIVITYCATEGORY,
					BbmEjbBundleKey.BUNDLE_NAME, BbmEjbBundleKey.OBJECT_ACTIVITY));

			return new CopyResult(CopyResult.CONFIRM_DEEP_COPY, cLinkedObjects);
		} catch (BbmEJBCreateException e) {
			throw new BbmCreateException(e, "could not get Activities");
		} catch (RemoteException e) {
			throw new BbmCreateException(e, "could not get Activities");
		}
	}

	protected static ID createCopyActivity(ActivityManager pActivityManager, Activity pActivity, ID idOrg,
			ID idActivityCategory, Collection mediaIds, Collection queueIds, ActivityProperties activityProperties,
			CopyDAO pDAO, String sCopyPrefix)
			throws RemoteException, BbmCreateException, BbmFinderException, ActivityValidationException {

		String sNewName = sCopyPrefix + pActivity.getName();
		int nName = 0;

		while (true) {
			boolean bNameIsUnique = pDAO.isNameUnique(sNewName, "ACTIVITY");

			if (bNameIsUnique) {
				break;
			}
			nName++;
			sNewName = sCopyPrefix + pActivity.getName() + " " + nName;
		}

		try {
			// We need to clone the activity before copying rather than re-use
			// the existing activity
			// because the activity objects are cached. We don't want multiple
			// entries in the
			// cache to point to the same activity object.
			Activity clonedActivity = (Activity) pActivity.clone();

			clonedActivity.setName(sNewName);
			clonedActivity.setOrganizationId(idOrg);
			clonedActivity.setActivityCategoryId(idActivityCategory);

			ID activityID = pActivityManager.createActivity(clonedActivity, mediaIds, activityProperties);
			pActivityManager.createActivityQueue(activityID, queueIds);
			return activityID;
		} catch (CloneNotSupportedException ex) {
			throw new BbmCreateException(ex);
		}
	}

	protected static ID createCopyActivityCategory(ActivityManager pActivityManager, ActivityCategory pCat, ID idOrg,
			CopyDAO pDAO, String sCopyPrefix) throws RemoteException, BbmCreateException, BbmFinderException {

		String sNewName = sCopyPrefix + pCat.getName();
		int nName = 0;

		while (true) {
			boolean bNameIsUnique = pDAO.isNameUnique(sNewName, "ACTIVITYCATEGORY");

			if (bNameIsUnique) {
				break;
			}
			nName++;
			sNewName = sCopyPrefix + pCat.getName() + " " + nName;
		}

		pCat.setName(sNewName);
		pCat.setOrganizationId(idOrg);

		return pActivityManager.createActivityCategory(pCat);
	}
}
