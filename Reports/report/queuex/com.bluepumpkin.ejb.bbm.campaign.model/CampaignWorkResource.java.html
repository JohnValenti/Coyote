import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.effectivity.TimePeriod;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectEffectivity;

public class CampaignWorkResource extends ValueObjectEffectivity {

	/**
	 *
	 */
	private static final long serialVersionUID = 1L;
	ID m_idSP = null;

	public void setSPID(ID idSP) {
		m_idSP = idSP;
	}

	public ID getSPID() {
		return m_idSP;
	}

	private static FieldInfo m_fieldInfo = new CampaignWorkResourceFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public CampaignWorkResource() {
		super();
	}

	public ID getCampaignID() {
		return getFieldValueID(CampaignWorkResourceFieldInfo.CAMPAIGNWORKRESOURCE_CAMPAIGNID);
	}

	public ID getWorkResourceID() {
		return getFieldValueID(CampaignWorkResourceFieldInfo.CAMPAIGNWORKRESOURCE_WORKRESOURCEID);
	}

	@Override
	public Date getStartTime() {
		return getFieldValueDate(CampaignWorkResourceFieldInfo.CAMPAIGNWORKRESOURCE_STARTTIME);
	}

	@Override
	public Date getEndTime() {
		return getFieldValueDate(CampaignWorkResourceFieldInfo.CAMPAIGNWORKRESOURCE_ENDTIME);
	}

	public void setCampaignID(ID id) {
		setFieldValue(CampaignWorkResourceFieldInfo.CAMPAIGNWORKRESOURCE_CAMPAIGNID, id);
	}

	public void setWorkResourceID(ID id) {
		setFieldValue(CampaignWorkResourceFieldInfo.CAMPAIGNWORKRESOURCE_WORKRESOURCEID, id);
	}

	@Override
	public void setStartTime(Date dt) {
		setFieldValue(CampaignWorkResourceFieldInfo.CAMPAIGNWORKRESOURCE_STARTTIME, dt);
	}

	@Override
	public void setEndTime(Date dt) {
		setFieldValue(CampaignWorkResourceFieldInfo.CAMPAIGNWORKRESOURCE_ENDTIME, dt);
	}

	// clone the assignment object without the ID field
	@Override
	public TimePeriod cloneTimePeriod() {
		CampaignWorkResource item = new CampaignWorkResource();

		item.setCampaignID(getCampaignID());
		item.setWorkResourceID(getWorkResourceID());
		item.setStartTime(getStartTime());
		item.setEndTime(getEndTime());

		return item;
	}

}
