import java.util.Date;

import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.ejb.bbm.effectivity.*;
import com.bluepumpkin.ejb.bbm.vo.*;

public class CampaignOrg extends ValueObjectEffectivity {
	/**
	 *
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new CampaignOrgFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public CampaignOrg() {
		super();
	}

	public ID getCampaignID() {
		return getFieldValueID(CampaignOrgFieldInfo.CAMPAIGNORG_CAMPAIGNID);
	}

	public ID getOrganizationID() {
		return getFieldValueID(CampaignOrgFieldInfo.CAMPAIGNORG_ORGANIZATIONID);
	}

	@Override
	public Date getStartTime() {
		return getFieldValueDate(CampaignOrgFieldInfo.CAMPAIGNORG_STARTTIME);
	}

	@Override
	public Date getEndTime() {
		return getFieldValueDate(CampaignOrgFieldInfo.CAMPAIGNORG_ENDTIME);
	}

	public void setCampaignID(ID id) {
		setFieldValue(CampaignOrgFieldInfo.CAMPAIGNORG_CAMPAIGNID, id);
	}

	public void setOrganizationID(ID id) {
		setFieldValue(CampaignOrgFieldInfo.CAMPAIGNORG_ORGANIZATIONID, id);
	}

	@Override
	public void setStartTime(Date dt) {
		setFieldValue(CampaignOrgFieldInfo.CAMPAIGNORG_STARTTIME, dt);
	}

	@Override
	public void setEndTime(Date dt) {
		setFieldValue(CampaignOrgFieldInfo.CAMPAIGNORG_ENDTIME, dt);
	}

	// clone the assignment object without the ID field
	@Override
	public TimePeriod cloneTimePeriod() {
		CampaignOrg item = new CampaignOrg();

		item.setCampaignID(getCampaignID());
		item.setOrganizationID(getOrganizationID());
		item.setStartTime(getStartTime());
		item.setEndTime(getEndTime());

		return item;
	}
}
