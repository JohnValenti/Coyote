	private float[] getValuesFor15MinuteIntervals(float[] hourlyValues) {
		float[] retVal = new float[96];

		if (getModelValueGranularity() == Granularity.Hourly) {
			int intervalCounter = 0;
			for (float hourlyValue : hourlyValues) {
				Arrays.fill(retVal, intervalCounter, intervalCounter + 4, hourlyValue);
				intervalCounter += 4;
			}
		} else {
			// When granularity is daily only the first value is set.
			Arrays.fill(retVal, 0, 96, hourlyValues[0]);
		}
		return retVal;
	}

	/**
	 * Order by SPID, then DayNumber
	 */
	@Override
	public int compareTo(SPShrinkage sps) {
		if (this.equals(sps)) {
			return 0;
		}

		// Not equal. Order by SPID, then DayNumber
		int retVal = this.getSPID().compareTo(sps.getSPID());
		if (retVal == 0) {
			retVal = Integer.valueOf(this.getDayNumber()).compareTo(Integer.valueOf(sps.getDayNumber()));
		}
		return retVal;
	}

	/**
	 * @return true if both the SPID and DayNumber columns are equal
	 */
	@Override
	public boolean equals(Object otherObj) {
		if (!(otherObj instanceof SPShrinkage)) {
			return false;
		}
		SPShrinkage sps = (SPShrinkage) otherObj;
		if (getSPID().equals(sps.getSPID()) && getDayNumber() == sps.getDayNumber()) {
			return true;
		}
		return false;
	}

	/**
	 * Returns a hash code derived from this shrinkage object's ID and day number.
	 */
	@Override
	public int hashCode() {
		int result = 17;
		result = 37 * result + getSPID().hashCode();
		result = 37 * result + getDayNumber();
		return result;
	}
