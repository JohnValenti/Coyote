package com.bluepumpkin.ejb.bbm.campaign.ejb;


import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;

import java.util.List;

/**
 * A container for storing the newly created scheduling periods as a result of a cloning operation.
 * Contains the cloned scheduling period object as well as any child scheduling period objects (only
 * applicable for distributed campaigns).
 */
class ClonedSchedulingPeriodResult {

	private final SchedulingPeriod clonedSchedulingPeriod;
	private final List<SchedulingPeriod> clonedChildSchedulingPeriods;

	ClonedSchedulingPeriodResult(SchedulingPeriod clonedSchedulingPeriod,
			List<SchedulingPeriod> clonedChildSchedulingPeriods) {

		this.clonedSchedulingPeriod = clonedSchedulingPeriod;
		this.clonedChildSchedulingPeriods = clonedChildSchedulingPeriods;
	}

	SchedulingPeriod getClonedSchedulingPeriod() {
		return clonedSchedulingPeriod;
	}

	List<SchedulingPeriod> getClonedChildSchedulingPeriods() {
		return clonedChildSchedulingPeriods;
	}
}
