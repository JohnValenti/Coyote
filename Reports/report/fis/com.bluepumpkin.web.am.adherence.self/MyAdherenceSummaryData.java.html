package com.bluepumpkin.web.am.adherence.self;

/**
 * Title:        MyAdherenceSummaryData
 * Description:  Data structure to hold the My Adherence Sumamry data.
 * Copyright:    Copyright (c) 2008
 * Company:      Verint Systems. Inc.
 * @author       Gonzalo Quintanar
 */

public class MyAdherenceSummaryData 
{
    /**
     * The total duration of all approved exceptions for the day.
     */
    int m_approved = 0;
    
    /**
     * The total duration of all unapproved exceptions for the day.
     */
    int m_unapproved = 0;
    
    /**
     * The total duration of all scheduled activities that do not map to the NO_ACTIVITY activity for the day.
     */
    int m_scheduled = 0;
    
    /**
     * The total duration of all unapproved exceptions during scheduled activities that do not map to the NO_ACTIVITY activity for the day.
     */
    int m_exceptions = 0;
    
    /**
     * The percentage of time adhering to In-Office schedule.
     * = (1 - (Adherence Exceptions during In-Office Hours / Scheduled In-Office Hours))
     */
    float m_percentage = 0;

    /**
     * Get the total duration of all approved exceptions for the day.
     */
    public int getApproved()
    {
        return m_approved;
    }
    
    /**
     * Get the total duration of all unapproved exceptions for the day.
     */
    public int getUnapproved()
    {
        return m_unapproved;
    }
    
    /**
     * Get the total duration of all scheduled activities that do not map to the NO_ACTIVITY activity for the day.
     */
    public int getScheduled()
    {
        return m_scheduled;
    }
    
    /**
     * Get the total duration of all unapproved exceptions during scheduled activities that do not map to the NO_ACTIVITY activity for the day.
     */
    public int getExcpetions()
    {
        return m_exceptions;
    }
    
    /**
     * Get the percentage of time adhering to In-Office schedule.
     * @return (1 - (Adherence Exceptions during In-Office Hours / Scheduled In-Office Hours)) * 100
     */
    public float getPercentage()
    {
        return m_percentage;
    }
    
    /**
     * Set the total duration of all approved exceptions for the day.
     */
    public void setApproved(int approved)
    {
        m_approved = approved;
    }
    
    /**
     * Set the total duration of all unapproved exceptions for the day.
     */
    public void setUnapproved(int unapproved)
    {
        m_unapproved = unapproved;
    }
    
    /**
     * Set the total duration of all scheduled activities that do not map to the NO_ACTIVITY activity for the day.
     */
    public void setScheduled(int scheduled)
    {
        m_scheduled = scheduled;
    }
    
    /**
     * Set the total duration of all unapproved exceptions during scheduled activities that do not map to the NO_ACTIVITY activity for the day.
     */
    public void setExceptions(int exceptions)
    {
        m_exceptions = exceptions;
    }
    
    /**
     * Set the percentage of time adhering to In-Office schedule.
     */
    public void setPercentage(float percentage)
    {
        m_percentage = percentage;
    }
}
