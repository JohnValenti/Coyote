package com.verint.web.rest.wfm.resource.calendareventassignment;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.schedule.model.BbmScheduleConflictException;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventAssignment;
import com.bluepumpkin.ejb.core.base.MultiUserException;

import java.rmi.RemoteException;

class CalendarEventAssignmentRepositoryUtil {

	private CalendarEventAssignmentRepositoryUtil() {
		throw new IllegalAccessError("Utility class");
	}

	/**
	 * Updates an object extending the CalendarEventAssignment class based on the entity.
	 *
	 * @param entity Entity object holding information to be updated
	 * @param sam    ScheduleAccessManager to access update methods for Events
	 * @param <T>    Entity holding information of what will be updated
	 * @return Updated CalendarEventAssignment as an entity.
	 * @throws BbmFinderException
	 * @throws RemoteException
	 * @throws BbmScheduleConflictException
	 * @throws MultiUserException
	 * @throws BbmUpdateException
	 */
	static <T extends CalendarEventAssignmentEntity> T updateCEAByType(T entity,
			ScheduleAccessManager sam)
			throws BbmFinderException, RemoteException, BbmScheduleConflictException, MultiUserException,
			BbmUpdateException {
		CalendarEventAssignment cea = sam.getCalendarEventAssignmentByID(ID.fromInt(entity.getID()));
		CalendarEventAssignmentEntityUtil.updateCeaAccordingToEntity(entity, cea);
		sam.updateCalendarEventAssignment(cea);
		return entity;
	}

}
