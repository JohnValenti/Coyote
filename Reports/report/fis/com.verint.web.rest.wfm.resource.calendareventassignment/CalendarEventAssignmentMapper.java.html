package com.verint.web.rest.wfm.resource.calendareventassignment;

import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventAssignment;
import com.verint.web.rest.wfm.util.MapperUtil;
import org.mapstruct.DecoratedWith;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.factory.Mappers;

import java.util.Collection;

@Mapper(uses = { MapperUtil.class })
@DecoratedWith(CalendarEventAssignmentMapperDecorator.class)
public interface CalendarEventAssignmentMapper {

	CalendarEventAssignmentMapper INSTANCE = Mappers.getMapper(CalendarEventAssignmentMapper.class);

	@Mapping(source = "linkedShiftAssignmentID", target = "shiftAssignmentID")
	@Mapping(source = "eventTemplateID", target = "calendarEventTemplateID")
	@Mapping(source = "workResourceIDs", target = "calendarEventAttendees")
	CalendarEventAssignmentEntity ceaToCeaEntity(CalendarEventAssignment source);

	Collection<CalendarEventAssignmentEntity> ceasToCeaEntities(Collection<CalendarEventAssignment> source);
}
