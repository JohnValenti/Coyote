import java.util.ArrayList;

import java.util.TimeZone;

import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;

/**
 * Sends Recurring event exception updates to RM/IFs through remote cache manager. This class hooks
 * the RecurringEventException table changes (CRUD) and sends the appropriate notifications through
 * PeopleCacheProcessor. PeopleCacheProcessor will be used by the RemoteCacheManager to send
 * Recurring event exception updates to RM/IFs.
 */
public class RecurringEventExceptionChangeListner implements DAOChangeHook {

	private static Category m_cat = Log.initCategory(RecurringEventExceptionChangeListner.class.getName());
	TimeZone tz = TimeZone.getTimeZone("GMT");

	/**
	 * Notifies RemoteCacheManager to send Recurring event exception Event creation.
	 * Will be called when RecurringEventException table has a new entry created as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.schedule.model.RecurringEventException objects that were created.
	 */
	public void OnCreateObject(ValueObjectBase[] valueObjects) throws JdmoException {
		if (m_cat.isDebugEnabled()) m_cat.debug("Recurring Event Exception Hook OnCreateObject valueObjects = " + valueObjects);
		if (valueObjects != null) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			throw new IllegalArgumentException();
		}
	}

	public void OnDeleteObject(ValueObjectBase[] valueObjects) throws JdmoException {
		//do nothing
	}

	public void OnUpdateObject(ValueObjectPairList oldAndNewObjects) throws JdmoException {
		//do nothing
	}

	public void updateTOHoursPerDayForValueObjects(ValueObjectBase[] valueObjects) {
		try {
			for (int i = 0; i < valueObjects.length; i++) {
				RecurringEventException exception = (RecurringEventException) valueObjects[i];
				notifyTOHoursPerDay(exception);
			}
		} catch (Exception e) {
			e.printStackTrace();
		}
	}

	public void notifyTOHoursPerDay(RecurringEventException exception) throws BbmUpdateException {
		Calendar rangeEndCal = Calendar.getInstance(tz);
		rangeEndCal.setTime(exception.getExceptionDate());
		rangeEndCal.add(Calendar.HOUR, 24);
		
		ArrayList empIds = new ArrayList();
		empIds.add(exception.getWorkResourceID());
		
		TONotifyMessage changeStateMessage = new TONotifyMessage(empIds,
				exception.getExceptionDate(), rangeEndCal.getTime(),
				TONotifyMessage.TO_NOTIFY_EMP_UNAVAIL_CHANGE);
		changeStateMessage.sendMessage();
		if(RequestUtil.isRMDebugEnabled())m_cat.debug("RecurringEventException LISTNER EMPID=" + exception.getWorkResourceID() + "; exception date=" + exception.getExceptionDate());

	}

	public TOHoursPerDayManager getTOHoursPerDayManager() throws BbmCreateException {
		return RmManagerFactory.getInstance(true).getTOHoursPerDayManager(null, null);
	}

}
