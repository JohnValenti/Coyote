/**
 * Sends Recurring Unavailability Event updates to RM/IFs through remote cache manager. This class hooks
 * the Recurring Unavailability Event table changes (CRUD) and sends the appropriate notifications through
 * PeopleCacheProcessor. PeopleCacheProcessor will be used by the RemoteCacheManager to send
 * Recurring Unavailability Event updates to RM/IFs.
 */
public class RecurringUnavailabilityEventChangeListner implements DAOChangeHook {

	private static Category m_cat = Log.initCategory(RecurringUnavailabilityEventChangeListner.class.getName());
	TimeZone tz = TimeZone.getTimeZone("GMT");

	/**
	 * Notifies RemoteCacheManager to send Recurring Unavailability Event creation.
	 * Will be called when Recurring Unavailability Event table has a new entry created as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.Recurring Unavailability Event.model.Recurring Unavailability Event objects that were created.
	 */
	public void OnCreateObject(ValueObjectBase[] valueObjects) throws JdmoException {
		if (m_cat.isDebugEnabled()) m_cat.debug("Recurring Unavailability Event Hook OnCreateObject valueObjects = " + valueObjects);
		if (valueObjects != null) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			throw new IllegalArgumentException();
		}
	}

	/**
	 * Notifies RemoteCacheManager to send Recurring Unavailability Event deletion.
	 * Will be called when Recurring Unavailability Event table has an entry deleted as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.Recurring Unavailability Event.model.Recurring Unavailability Event objects that were deleted.
	 */
	public void OnDeleteObject(ValueObjectBase[] valueObjects) throws JdmoException {
		// When an Recurring Unavailability Event is deleted the send notification
		if (m_cat.isDebugEnabled()) m_cat.debug("Recurring Unavailability Event Hook OnDeleteObject valueObjects = " + valueObjects);
		if (valueObjects != null && valueObjects.length > 0) {
			for (int i = 0; i < valueObjects.length; i++) {
				CalendarEventTemplate template = (CalendarEventTemplate) valueObjects[i];
				if (template.getEventType() == Event.EVENT_TYPE_UNAVAILABILITY) {
					try {

						//Passing null as start date and MAXDate as end date as the real date range is not available.
						// in future it would be good if the CalendarEventTemplateImpl has the Date range in it.
						TONotifyMessage changeStateMessage = new TONotifyMessage(template.getWorkResourceIDs(), null, RequestUtil.MAX_DATE, TONotifyMessage.TO_NOTIFY_EMP_UNAVAIL_CHANGE);
						changeStateMessage.sendMessage();
						if (RequestUtil.isRMDebugEnabled()) m_cat.debug("RecurringEventTemplate LISTNER  DELETE EVENT EMPID=" + template.getWorkResourceIDs());
					} catch (Exception e) {
						e.printStackTrace();
					}
				}
			}
		} else {
			throw new IllegalArgumentException();
		}
	}

	/**
	 * Notifies RemoteCacheManager to send Recurring Unavailability Event updates.
	 * Will be called when Recurring Unavailability Event table has been updated as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param old and current value objects com.bluepumpkin.ejb.bbm.Recurring Unavailability Event.model.Recurring Unavailability Event
	 *            will be in oldAndNewObjects.
	 */
	public void OnUpdateObject(ValueObjectPairList oldAndNewObjects) throws JdmoException {
		if (oldAndNewObjects != null && oldAndNewObjects.getCurrentValues() != null) {
			if (m_cat.isDebugEnabled()) m_cat.debug("Recurring Unavailability Event Hook OnUpdateObject sending update notification " + oldAndNewObjects.getCurrentValues());
			ValueObjectBase[] newValueObjects = oldAndNewObjects.getCurrentValues();
			ValueObjectBase[] oldValueObjects = oldAndNewObjects.getOldValues();
			for (int i = 0; i < newValueObjects.length; i++) {
				RecurringEventTemplateBase newTemplate = (RecurringEventTemplateBase) newValueObjects[i];
				RecurringEventTemplateBase oldTemplate = (RecurringEventTemplateBase) oldValueObjects[i];
				try {
					if (newTemplate.getEventType() == Event.EVENT_TYPE_UNAVAILABILITY) {
						notifyTOHoursPerDay(newTemplate);
						if (!newTemplate.getEarliestPossibleStartTime().equals(oldTemplate.getEarliestPossibleStartTime())
						        || !newTemplate.getLatestPossibleEndTime().equals(oldTemplate.getLatestPossibleEndTime())) {
							notifyTOHoursPerDay(oldTemplate);
						}
					}
				} catch (BbmUpdateException e) {
					e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
				}
			}
		} else {
			throw new IllegalArgumentException();
		}
	}

	public void updateTOHoursPerDayForValueObjects(ValueObjectBase[] valueObjects) {
		try {
			for (int i = 0; i < valueObjects.length; i++) {
				RecurringEventTemplateBase template = (RecurringEventTemplateBase) valueObjects[i];
				if (template.getEventType() == Event.EVENT_TYPE_UNAVAILABILITY) {
					notifyTOHoursPerDay(template);
				}
			}
		} catch (Exception e) {
			e.printStackTrace();
		}
	}

	public void notifyTOHoursPerDay(RecurringEventTemplateBase template) throws BbmUpdateException {

		TONotifyMessage changeStateMessage = new TONotifyMessage(template.getWorkResourceIDs(),
		        template.getEarliestPossibleStartTime(), template.getLatestPossibleEndTime(), TONotifyMessage.TO_NOTIFY_EMP_UNAVAIL_CHANGE);
		changeStateMessage.sendMessage();
		if (RequestUtil.isRMDebugEnabled()) m_cat.debug("RecurringEventTemplate LISTNER EMPID=" + template.getWorkResourceIDs() + template.getEarliestPossibleStartTime() + " : ENDDATE=" + template.getLatestPossibleEndTime());

	}

	public TOHoursPerDayManager getTOHoursPerDayManager() throws BbmCreateException {
		return RmManagerFactory.getInstance(true).getTOHoursPerDayManager(null, null);
	}

}
