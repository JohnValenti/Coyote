package com.verint.ejb.bbm.forecast.ejb;

import java.util.Collection;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBaseWithAutoIDMapping;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectUtil;
import com.bluepumpkin.ejb.bbm.workload.model.QueueFieldInfo;
import com.verint.ejb.bbm.forecast.model.ForecastProfileFieldInfo;
import com.verint.ejb.bbm.forecast.model.ForecastProfileList;
import com.verint.ejb.bbm.forecast.model.ForecastProfileListFieldInfo;

public class ForecastProfileListDAO extends DAOBaseWithAutoIDMapping<ForecastProfileList> {
	private static FieldInfo m_fieldInfo = new ForecastProfileListFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	/**
	* default constructor
	*/
	public ForecastProfileListDAO(){
		super();
	}
	
	public ForecastProfileListDAO(Jdmo dmo) {
		super(dmo);
	}
	
	protected ForecastProfileList createValueObject() {
		return (new ForecastProfileList());
	}
	
	public Collection<ForecastProfileList> getForecastProfileListsByForecastProfileID(ID forecastProfileId) throws BbmFinderException {
		String where = " " + m_fieldInfo.getFieldName(ForecastProfileListFieldInfo.FORECASTPROFILELIST_FORECASTPROFILEID) + " = " + 
		JdmoUtil.asSqlLiteral(forecastProfileId);
		return getObjects(where);
	}
	
	/**
	 * This method expects forecastProfileIds to be the forecast profile's string based Ids (the foreign key on the forecast profile list table
	 * is on the forecast profile's string ID).
	 * Returns a map of forecast profile lists, keyed on the SID of the forecast profile to which these lists belong.
	 */
	public Map<ID, Collection<ForecastProfileList>> getForecastProfileListsByForecastProfileIDs(Collection<ID> forecastProfileIds) throws BbmFinderException {
		Map<ID, Collection<ForecastProfileList>> retVal = new HashMap<ID, Collection<ForecastProfileList>>();
		try {
			String where = " " + m_fieldInfo.getFieldName(ForecastProfileListFieldInfo.FORECASTPROFILELIST_FORECASTPROFILEID) + " IN " + 
				getDMO().createInClause(forecastProfileIds);
			Collection<ForecastProfileList> lists = getObjects(where);
			retVal = ValueObjectUtil.groupByField(ForecastProfileListFieldInfo.FORECASTPROFILELIST_FORECASTPROFILEID, lists);	
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
		return retVal;
	}

	@Override
	protected Map<Integer, ? extends FieldInfo> getForeignKeyFieldInfo() {
		Map<Integer, FieldInfo> map = new HashMap<Integer, FieldInfo>();
		map.put(ForecastProfileListFieldInfo.FORECASTPROFILELIST_QUEUEID, new QueueFieldInfo());
		map.put(ForecastProfileListFieldInfo.FORECASTPROFILELIST_FORECASTPROFILEID, new ForecastProfileFieldInfo());
		return map;
	}

}
