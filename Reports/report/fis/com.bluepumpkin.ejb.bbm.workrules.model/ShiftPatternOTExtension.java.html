package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class ShiftPatternOTExtension extends ValueObjectBase {
	private static final long serialVersionUID = 1L;
	
	private static FieldInfo m_fieldInfo = new ShiftPatternOTExtensionFieldInfo();
	
	public ID getShiftPatternID() { return getFieldValueID(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_SHIFTPATTERNID); }
	public void setShiftPatternID(ID shiftPatternID) { setFieldValue(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_SHIFTPATTERNID, shiftPatternID); }
	
	public ID getShiftOTExtensionID() { return getFieldValueID(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_SHIFTOTEXTENSIONID); }
	public void setShiftOTExtensionID(ID shiftOTExtensionID) { setFieldValue(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_SHIFTOTEXTENSIONID, shiftOTExtensionID); }
	
	public ShiftPattern.StartTime getOTExtensionStartTime() { return ShiftPattern.StartTime.fromIntVal(getFieldValueInt(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_OTEXTSTARTTIMES)); }
	public void setOTExtensionStartTime(ShiftPattern.StartTime otExtensionStartTime) {
		if (otExtensionStartTime != null) {
			setFieldValue(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_OTEXTSTARTTIMES, otExtensionStartTime.getIntVal());
		}
	}
	
	public Date getCreatedDate() { return getFieldValueDate(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_CREATED); }
	public void setCreatedDate(Date createdDate) { setFieldValue(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_CREATED, createdDate); }
	
	public String getCreatedBy() { return getFormattedFieldValue(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_CREATEDBY); }
	public void setCreatedBy(String createdBy) { setFieldValue(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_CREATEDBY, createdBy); }

	public Date getModifiedDate() { return getFieldValueDate(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_MODIFIED); }
	public void setModifiedDate(Date modifiedDate) { setFieldValue(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_MODIFIED, modifiedDate); }
	
	public String getModifiedBy() { return getFormattedFieldValue(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_MODIFIEDBY); }
	public void setModifiedBy(String modifiedBy) { setFieldValue(ShiftPatternOTExtensionFieldInfo.SHIFTPATTERNOTEXTENSION_MODIFIEDBY, modifiedBy); }

	@Override
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
}
