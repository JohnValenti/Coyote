package com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb;

import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;



class EmployeeSPTimeZone {
	final int employeeID;
	final int schedulingPeriodID;
	final int organizationID;
	final String schedulingPeriodTimeZoneID;
	final Date startDate;

	EmployeeSPTimeZone(ID employeeID, ID spID, String spTimeZoneID, ID organizationID, Date start) {
		this(employeeID.toInt(), spID.toInt(), spTimeZoneID, organizationID.toInt(), start);
	}

	EmployeeSPTimeZone(int employeeID, int spID, String spTimeZoneID, int orgID, Date start) {
		this.employeeID = employeeID;
		this.schedulingPeriodID = spID;
		this.schedulingPeriodTimeZoneID = spTimeZoneID;
		this.startDate = start;
		this.organizationID = orgID;
	}

	TimeZone getSpTimeZone() {
		return TimeZone.getTimeZone(schedulingPeriodTimeZoneID);
	}

}
