package com.bluepumpkin.ejb.bbm.time;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;

/**
 * A region composed of the union of zero or more separate {@link ISimpleRegion}
 * s.
 * 
 * @param <T>
 *            the class of {@link ISimpleRegion} of which this
 *            {@code CompoundRegion} is composed
 */
public class CompoundRegion<T extends ISimpleRegion<T>> {
	private Collection<T> m_subRegions;

	/**
	 * Creates an empty {@link CompoundRegion}.
	 */
	public CompoundRegion() {
		m_subRegions = new ArrayList<T>();
	}

	/**
	 * Creates a copy of {@code source}.
	 * 
	 * @param source
	 */
	public CompoundRegion(CompoundRegion<T> source) {
		m_subRegions = new ArrayList<T>(source.m_subRegions);
	}

	/**
	 * Creates a {@link CompoundRegion} containing the specified simple regions.
	 * 
	 * @param initialArea
	 */
	public CompoundRegion(Collection<T> subRegions) {
		m_subRegions = new ArrayList<T>(subRegions);
	}

	/**
	 * Adds {@code regionToAdd} to this {@link CompoundRegion}. This may result
	 * in a {@code CompoundRegion} containing fewer, the same, or more separate
	 * areas than before, depending on the number of existing areas within this
	 * {@code CompoundRegion} that {@code regionToAdd} intersects with or
	 * contains.
	 * 
	 * @param regionToAdd
	 */
	public void add(T regionToAdd) {
		// To add a simple region, we first determine which existing simple
		// regions
		// intersect with it.
		// Those that do not intersect are unaffected and are included in the
		// result
		// unchanged. Those that are contained in the new region are discarded.
		// Those that intersect but are not contained are then added to it and
		// the
		// resulting single simple region is added to the result.
		Collection<T> newSubRegions = new ArrayList<T>();
		Collection<T> intersectAreas = new ArrayList<T>();
		for (T subRegion : m_subRegions) {
			if (regionToAdd.intersects(subRegion) || regionToAdd.adjoins(subRegion)) {
				if (!regionToAdd.contains(subRegion)) {
					intersectAreas.add(subRegion);
				}
			} else {
				newSubRegions.add(subRegion);
			}
		}
		for (T intersectArea : intersectAreas) {
			Collection<T> partialSum = regionToAdd.add(intersectArea);
			if (partialSum.size() != 1) {
				throw new IllegalStateException("Logic error: Adding two intersecting areas yielded "
						+ partialSum.size() + " areas instead of 1");
			} else {
				regionToAdd = partialSum.iterator().next();
			}
		}
		newSubRegions.add(regionToAdd);
		m_subRegions = newSubRegions;
	}

	/**
	 * Adds another {@link CompoundRegion} to this.
	 * 
	 * @param compoundRegionToAdd
	 */
	public void add(CompoundRegion<T> compoundRegionToAdd) {
		for (T subRegion : compoundRegionToAdd.m_subRegions) {
			add(subRegion);
		}
	}

	/**
	 * Subtracts {@code regionToSubtract} from this {@link CompoundRegion}.
	 * 
	 * @param regionToSubtract
	 */
	public void subtract(T regionToSubtract) {
		Collection<T> newSubRegions = new ArrayList<T>();
		for (T subRegion : m_subRegions) {
			newSubRegions.addAll(subRegion.subtract(regionToSubtract));
		}
		m_subRegions = newSubRegions;
	}

	/**
	 * Subtracts another {@link CompoundRegion} from this.
	 * 
	 * @param compoundRegionToSubtract
	 */
	public void subtract(CompoundRegion<T> compoundRegionToSubtract) {
		for (T subRegion : compoundRegionToSubtract.m_subRegions) {
			subtract(subRegion);
		}
	}

	/**
	 * Returns whether this {@link CompoundRegion} contains all of the specified
	 * {@link ISimpleRegion}.
	 * 
	 * @param simpleRegion
	 * @return
	 */
	public boolean contains(T simpleRegion) {
		for (T area : m_subRegions) {
			if (area.contains(simpleRegion)) {
				return true;
			}
		}
		return false;
	}

	/**
	 * Returns whether this {@link CompoundRegion} intersects with the specified
	 * {@link ISimpleRegion}.
	 * 
	 * @param simpleRegion
	 * @return
	 */
	public boolean intersects(T simpleRegion) {
		for (T area : m_subRegions) {
			if (area.intersects(simpleRegion)) {
				return true;
			}
		}
		return false;
	}

	/**
	 * Returns whether this {@link CompoundRegion} is empty; that is, whether it
	 * is composed of zero simple regions.
	 * 
	 * @return
	 */
	public boolean isEmpty() {
		return m_subRegions.isEmpty();
	}

	/**
	 * Returns all the constituent simple regions in this region.
	 * 
	 * @return
	 */
	public Collection<T> getSimpleRegions() {
		return Collections.unmodifiableCollection(m_subRegions);
	}
}
