package com.bluepumpkin.web.am.timecollection;

import java.util.ArrayList;
import java.util.StringTokenizer;

import com.bluepumpkin.ejb.core.profile.model.ToXmlString;

 */

public class MyTimeProfileHelper implements ToXmlString{
	private String xmlString = "";

	public MyTimeProfileHelper() {
	}

	public String toXmlString(){
		return xmlString;
	}

	public ArrayList getArrListFromString(String profileString){
		StringTokenizer historyData = new StringTokenizer(profileString,";");
		ArrayList aFullList = new ArrayList();
		while (historyData.hasMoreElements()) {
			String gotToken=historyData.nextToken();
			StringTokenizer tabToken = new StringTokenizer(gotToken,",");
			ArrayList aList = new ArrayList();
			while (tabToken.hasMoreElements()) {
				aList.add(tabToken.nextToken());
			}
			aFullList.add(aList);
		}
		return aFullList;
	}

	public String getStrFromArrList(ArrayList aList){
		//go through the  arraylist and add to the string
		StringBuffer aStringBuffer = new StringBuffer(aList.size()*3);
		for (int i=0; i<aList.size(); i++){
			ArrayList list = (ArrayList)aList.get(i);
			for (int j=0; j<list.size(); j++){
				if (j!=2){
					aStringBuffer.append(String.valueOf(list.get(j))+",");
				} else {
					aStringBuffer.append(String.valueOf(list.get(j))+";");
				}
			}
		}
		return aStringBuffer.toString();
	}
}
