package com.verint.web.fs.workrules;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.system.RequestContext;
import com.bluepumpkin.ejb.bbm.workrules.model.VTOEvent;

public class VTOEventRH extends WorkRulesRH<VTOEventListPM, VTOEventFormPM>{
	
	@Override
	protected VTOEventListPM getListPageModel(RequestContext context) {
		// Don't getInstance yet because the pagination needs to be set before the data is loaded
		VTOEventListPM pm = new VTOEventListPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
	}
	@Override
	protected VTOEventFormPM getFormPageModel(RequestContext context) {
		return VTOEventFormPM.getInstance(context);
	}
	
	@Override
	protected void processAdd(RequestContext context, VTOEventListPM pageModel)
			throws BPException {
		setNextPage(context, FORM_PAGE);
	}
	
	@Override
	protected void processCopy(RequestContext context, VTOEventListPM pageModel)
			throws BPException {
		try {
			if (pageModel.isByOrg) {
				WorkRulesModelHandler.copyVTOEvents(context, pageModel.getPerfromActionOnIDsAsCollection(), 
						pageModel.isByOrg, pageModel.getSelectedIDObject(), pageModel.getLocalizer().getLocaleContext());
			} else {
				pageModel.loadData();			//we need to do this to initialize the campaign/sp value objects
				WorkRulesModelHandler.copyVTOEvents(context, pageModel.getPerfromActionOnIDsAsCollection(), 
						pageModel.isByOrg, pageModel.selectedCampaign.getDEID(),
						pageModel.getLocalizer().getLocaleContext());
			}
			context.setPageModel(null);
			pageModel = getListPageModel(context);		//reload the new shift events
			pageModel.addPageMessage(Message.RESPONSE_TYPE,
					FsWebBundleKey.FS_COPY_SHIFT_EVENT_SUCCESS, FsWebBundleKey.BUNDLE_NAME);
		} catch (Exception ex) {
				pageModel.addPageMessage(Message.ERROR_TYPE,
						FsWebBundleKey.FS_UNABLE_TO_COPY_SHIFT_EVENTS, FsWebBundleKey.BUNDLE_NAME);
				throw new BPException(ex);
		} finally {
			context.setPageModel(pageModel);			
		}
	}
	@Override
	protected void processDelete(RequestContext context, VTOEventListPM pageModel) throws BPException {
		WorkRulesModelHandler.deleteVTOEvents(context, pageModel.getPerfromActionOnIDsAsCollection());
		context.setPageModel(null); // reset the page model to remove the deleted entries 
	}

	@Override
	protected void processEdit(RequestContext context, VTOEventListPM pageModel)
			throws BPException {
		setNextPage(context, FORM_PAGE);
	}

	@Override
	protected void processExport(RequestContext context,
			VTOEventListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processImport(RequestContext context,
			VTOEventListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}
	@Override
	protected void processSave(RequestContext context, VTOEventFormPM pageModel)
			throws BPException {

		if (pageModel.isNew()) {
			if (pageModel.validate()) {
				for (VTOEvent vto : pageModel.getVTOEvents()) {
					WorkRulesModelHandler.createVTOEvent(context, vto);
				}
			}
			else
				setNextPage(context, "formPage");
		} else {
			if (pageModel.validate()) {
				WorkRulesModelHandler.updateVTOEvents(context, pageModel.getVTOEvents());
			} else {
				setNextPage(context, "formPage");
			}
		}		
	
		
	}
	@Override
	protected void processDownload(RequestContext context,
			VTOEventListPM pageModel) throws BPException {
		// TODO Auto-generated method stub
		
	}
	
}
