package com.verint.web.fs.workrules;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.system.RequestContext;

public class ProjectRulesListRH
	extends WorkRulesRH<ProjectRulesListPM, ProjectRulesFormPM>
{
	protected ProjectRulesListPM getListPageModel(RequestContext context)
	{

		ProjectRulesListPM pm = new ProjectRulesListPM(context);
		pm.extractParameters(context.getRequest());
		return pm;

		// Can't getInstance yet because the pagination needs to be set up before the data is loaded
		//return ProjectRulesListPM.getInstance(context);
	}

	@Override
	protected ProjectRulesFormPM getFormPageModel(RequestContext context)
	{
		/*
		 ProjectRulesFormPM pm = new ProjectRulesFormPM(context);
		 pm.extractParameters(context.getRequest());
		 return pm;
		 */
		return ProjectRulesFormPM.getInstance(context);
	}

	@Override
	protected void processCopy(RequestContext context,
		ProjectRulesListPM pageModel)
		throws BPException
	{
		try {
			if (pageModel.isByOrg) {
				WorkRulesModelHandler.copyProjects(context, pageModel.getPerfromActionOnIDsAsCollection(),
					pageModel.isByOrg, pageModel.getSelectedIDObject(), pageModel.getLocalizer().getLocaleContext());
			}
			else {
				pageModel.loadData();			//we need to do this to initialize the campaign/sp value objects
				WorkRulesModelHandler.copyProjects(context, pageModel.getPerfromActionOnIDsAsCollection(),
					pageModel.isByOrg, pageModel.selectedCampaign.getDEID(),
					pageModel.getLocalizer().getLocaleContext());
			}
			context.setPageModel(null);
			pageModel = getListPageModel(context);		//reload the new shifts
			pageModel.addPageMessage(Message.RESPONSE_TYPE,
				FsWebBundleKey.FS_COPY_SHIFTS_SUCCESS, FsWebBundleKey.BUNDLE_NAME);
		}
		catch(Exception ex) {
			pageModel.addPageMessage(Message.ERROR_TYPE,
				FsWebBundleKey.FS_UNABLE_TO_COPY_SHIFTS, FsWebBundleKey.BUNDLE_NAME);
			throw new BPException(ex);
		}
		finally {
			context.setPageModel(pageModel);
		}
	}

	@Override
	protected void processDelete(RequestContext context,
		ProjectRulesListPM pageModel)
		throws BPException
	{

		WorkRulesModelHandler.deleteProjectRules(context, pageModel.getPerfromActionOnIDsAsCollection());
		//WorkRulesModelHandler.deleteShifts(context, pageModel.getPerfromActionOnIDsAsCollection());
		context.setPageModel(null); // reset the page model to remove the deleted entries 
	}

	@Override
	protected void processEdit(RequestContext context,
		ProjectRulesListPM pageModel)
		throws BPException
	{
		ProjectRulesFormPM formPM = new ProjectRulesFormPM(context);
		formPM.extractParameters(context.getRequest());
		context.setPageModel(formPM);
		setNextPage(context, "formPage");
	}

	@Override
	protected void processExport(RequestContext context,
		ProjectRulesListPM pageModel)
		throws BPException
	{
		// TODO Auto-generated method stub
	}

	@Override
	protected void processImport(RequestContext context,
		ProjectRulesListPM pageModel)
		throws BPException
	{
		// TODO Auto-generated method stub
	}

	@Override
	protected void processSave(RequestContext context,
		ProjectRulesFormPM pageModel)
		throws BPException
	{
		try {
			if (pageModel.validate()) {
				if (pageModel.isNew()) {
					WorkRulesModelHandler.createProject(context, pageModel.getProjectRule().iterator().next());
				}
				else {
					WorkRulesModelHandler.updateProjects(context, pageModel.getRules());
				}
			}
			else {
				setNextPage(context, "formPage");
			}
		}
		catch(BbmException e) {
			setNextPage(context, "formPage");
			throw e;
		}
	}
