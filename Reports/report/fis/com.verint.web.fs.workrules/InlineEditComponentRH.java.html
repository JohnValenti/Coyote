package com.verint.web.fs.workrules;

import java.io.PrintWriter;
import java.util.Collection;
import java.util.Collections;
import java.util.Map;
import java.util.Set;
import java.util.TimeZone;

import javax.servlet.http.HttpServletResponse;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftOTExtension;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPattern;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPatternPeriod;
import com.bluepumpkin.ejb.bbm.workrules.model.VTOEvent;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;
import com.bluepumpkin.web.fs.l10n.FsWebLogBundleKey;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.system.RequestContext;

public class InlineEditComponentRH extends RequestHandler {

	public final static String URL = "inline_edit_component";
	public final static String PARAM_TYPE = "type";
	public final static String PARAM_IDS = "ids";
	public final static String PARAM_ORG_ID = "orgID";
	public final static String PARAM_CAMPAIGN_ID = "campaignID";
	public final static String PARAM_TIME_ZONE_ID = "tzID";
	public final static String PARAM_SHIFT_PATTERN_PERIOD = "shiftPatternPeriod";
	public final static String OT_EXTENSION_TYPE = "shiftOTExtension";
	public final static String SHIFT_TYPE = "shift";
	public final static String VTO_EVENT_TYPE = "vtoEvent";
	public final static String ACTIVITY_TYPE = "activity";
	
	@Override
	public void processRequest(RequestContext context) throws Exception {	
		String ids = context.getParameter(PARAM_IDS);
		String type = context.getParameter(PARAM_TYPE);
		StringBuffer response = new StringBuffer();
		try {
			if (OT_EXTENSION_TYPE.equals(type)) {
				Map<ShiftOTExtension, Set<ShiftPattern.StartTime>> emptyMap = Collections.emptyMap();
				OTExtensionInlineEditPC pc_otExtension =
					new OTExtensionInlineEditPC(context, "shiftPatternOTExtensionTable", emptyMap, false, null, null, null);
				Collection<ShiftOTExtension> shiftOTExtensions = 
					WorkRulesModelHandler.getShiftOTExtensionsByIDs(context, StringUtil.parseIDStringToCollection(ids));
				for (ShiftOTExtension sote : shiftOTExtensions) {
					response.append(pc_otExtension.getRowHtml(sote));
				}
			}
			if (SHIFT_TYPE.equals(type)) {
				TimeZone optimizationTargetTimeZone = TimeZone.getTimeZone(context.getParameter(PARAM_TIME_ZONE_ID));
				Organization org = null;
				Campaign campaign = null;
				String orgIDStr = context.getParameter(PARAM_ORG_ID);
				String shiftPatternPeriodStr = context.getParameter(PARAM_SHIFT_PATTERN_PERIOD);
				
				String campaignIDStr = context.getParameter(PARAM_CAMPAIGN_ID);
				
				if (StringUtil.isEmpty(orgIDStr) == false) {
					org = OrganizationModelHandler.getOrganization(context, new ID(orgIDStr));
				}
				if (StringUtil.isEmpty(campaignIDStr) == false) {
					campaign = CampaignModelHandler.getCampaign(context, new ID(campaignIDStr));
				}
				WorkDaysInlineEditListPC pc_workDays =
					new WorkDaysInlineEditListPC(context, "shiftPatternWorkDaysTable", org, campaign, optimizationTargetTimeZone,
							ShiftPatternPeriod.fromString(shiftPatternPeriodStr));
				Collection<Shift> shifts = 
					WorkRulesModelHandler.getShiftsByIDs(context, StringUtil.parseIDStringToCollection(ids));
				response.append(pc_workDays.getRowHtml(shifts));
			}
			if (VTO_EVENT_TYPE.equals(type)) {
				Map<VTOEvent, Set<ShiftPattern.StartTime>> emptyMap = Collections.emptyMap();
				VTOEventInlineEditPC pc_vtoEventInlineEditPC =  new VTOEventInlineEditPC(context, "shiftPatternVTOEventTable", emptyMap, false, null);
				Collection<ID> idCol = StringUtil.parseIDStringToCollection(ids);
				if (idCol.size() > 0) {
					for (VTOEvent vtoEvent : WorkRulesModelHandler.getVTOEventsByIDs(context, idCol)) {
						response.append(pc_vtoEventInlineEditPC.getRowHtml(vtoEvent));
					}
				}
			}
			if (ACTIVITY_TYPE.equals(type)) {
				if (ids.length() > 0) {
					Map<ID, Activity> activityMap = WorkRulesModelHandler.getActivities(context, 
							StringUtil.parseIDStringToCollection(ids));
					CompactActivityListPC pc_activityListPC = new CompactActivityListPC(context, "activityTable", activityMap.values());
					response.append(pc_activityListPC.getRowHtml(activityMap.values()));
				}
			}
		} catch (Exception ex) {
			m_log.error(context.getLocalizer().i18n(FsWebLogBundleKey.BUNDLE_NAME, FsWebLogBundleKey.MSG_COULD_NOT_ADD_ROW_TO_TABLE), ex);
			context.getResponse().setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
			context.getResponse().getWriter().write(context.getLocalizer().i18n(FsWebLogBundleKey.BUNDLE_NAME, FsWebLogBundleKey.MSG_COULD_NOT_ADD_ROW_TO_TABLE));
		} finally {
			context.getResponse().setContentType(RequestContext.DEFAULT_RESPONSE_CONTENT_TYPE);
			PrintWriter responseWriter = context.getResponse().getWriter();		
			responseWriter.write(response.toString());
			responseWriter.close();
		}
	}

}
