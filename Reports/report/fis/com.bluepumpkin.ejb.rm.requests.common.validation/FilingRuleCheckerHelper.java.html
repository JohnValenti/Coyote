package com.bluepumpkin.ejb.rm.requests.common.validation;

import java.util.Collection;
import java.util.TimeZone;
import java.util.Iterator;
import java.util.ArrayList;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.holiday.model.Holiday;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.setup.filingrules.model.RequestFilingRule;
import com.bluepumpkin.ejb.rm.cache.CacheUtilRM;

/**
 * Helper class to create instances of FilingRuleChecker
 */
public class FilingRuleCheckerHelper {

	private static Category log = Log.initCategory(FilingRuleCheckerHelper.class.getName());

	/** Dont create instances of this class */
	private FilingRuleCheckerHelper() {
	}

	/**
	 * @see #getChecker(Organization, String, ID, TimeRange)
	 */
	public static FilingRuleChecker getChecker(ID orgId, String requestType, TimeRange range) throws Exception {
		return getChecker(orgId, requestType, null, range);
	}

	/**
	 * @see #getChecker(Organization, String, ID, TimeRange)
	 */
	public static FilingRuleChecker getChecker(Organization org, String requestType, TimeRange range) throws Exception {
		return getChecker(org, requestType, null, range);
	}

	/**
	 * @see #getChecker(Organization, String, ID, TimeRange)
	 *
	 */
	public static FilingRuleChecker getChecker(ID orgId, String requestType, ID activityId, TimeRange range)
			throws Exception {
		// get org from ID
		Organization org = ValidationUtil.getOrganizationByID(orgId);
		return getChecker(org, requestType, activityId, range);
	}

	/**
	 * Create a FilingRuleChecker. A filing rule checker is seeded with a list of enabled filing rules for the given
	 * requestType activityID and org. It is also seeded with the holidays in the given range and the dayBoundaryOffset for
	 * the given org.
	 *
	 * @param org the organization
	 * @param requestType the type of request
	 * @param activityId, an optional input for requests that have associated activities (like timeoff). This can be null.
	 * @param range the time range of interest
	 * @return a FilingRuleChecker ready to check requests matching the input specification
	 * @exception com.bluepumpkin.ejb.bbm.base.BbmFinderException when information necessary to build the checker cannot be
	 *                obtained.
	 */
	public static FilingRuleChecker getChecker(Organization org, String requestType, ID activityID, TimeRange range)
			throws Exception {
		// Get the request filing rules specified
		Collection rules = CacheUtilRM.getEnabledFilingRules(org.getID(), requestType, activityID);
		// Get the holidays for the organization during the period
		TimeZone tz = org.getTimeZone();
		Collection holidays = ValidationUtil.getHolidays(org, tz, range);
		// Build a checker with the information we've gathered.
		return new FilingRuleChecker(rules, holidays, tz, org.getDayBoundaryOffset());
	}

	public static FilingRuleChecker getChecker(Organization org, String requestType, int reqSubType, TimeRange range)
			throws Exception {
		// Get the request filing rules specified
		Collection rules = CacheUtilRM.getEnabledFilingRules(org.getID(), requestType, null);
		ArrayList applicableRules = new ArrayList();
		for (Iterator iterator = rules.iterator(); iterator.hasNext();) {
			RequestFilingRule filingRule = (RequestFilingRule) iterator.next();
			if ((filingRule.getRequestSubType() & reqSubType) > 0) {
				applicableRules.add(filingRule);
			}
		}
		// Get the holidays for the organization during the period
		TimeZone tz = org.getTimeZone();
		Collection holidays = ValidationUtil.getHolidays(org, tz, range);
		// Build a checker with the information we've gathered.
		return new FilingRuleChecker(applicableRules, holidays, tz, org.getDayBoundaryOffset());
	}

	/**
	 * Create a FilingRuleChecker for Flex Time requests. A filing rule checker is seeded with a list of enabled filing rules
	 * for the given requestType activityID and org. It is also seeded with the holidays in the given range and the
	 * dayBoundaryOffset for the given org.
	 * @param org the organization
	 * @param requestType the type of request
	 * @param activityId, an optional input for requests that have associated activities (like timeoff). This can be null.
	 * @param reqSubType Flex Time request filing rules has a reqSubType, where 0=TO before makeup, and 1= makeup before TO.
	 * @param range the time range of interest
	 * @return a FilingRuleChecker ready to check requests matching the input specification
	 * @exception com.bluepumpkin.ejb.bbm.base.BbmFinderException when information necessary to build the checker cannot be
	 *                obtained.
	 */
	public static FilingRuleChecker getChecker(Organization org, String requestType, ID activityID, int reqSubType,
			TimeRange range) throws Exception {
		// Get the request filing rules specified
		@SuppressWarnings("unchecked")
		Collection<RequestFilingRule> rules = 
				(Collection<RequestFilingRule>)CacheUtilRM.getEnabledFilingRules(org.getID(), requestType, activityID);
		// Get the holidays for the organization during the period
		TimeZone tz = org.getTimeZone();
		Collection<Holiday> holidays = ValidationUtil.getHolidays(org, tz, range);

		ArrayList<RequestFilingRule> applicableRules = new ArrayList<RequestFilingRule>();
		for (RequestFilingRule filingRule : rules) {
			if (filingRule.getRequestSubType() == reqSubType) {
				applicableRules.add(filingRule);
			}
		}

		// Build a checker with the information we've gathered.
		return new FilingRuleChecker(applicableRules, holidays, tz, org.getDayBoundaryOffset());
	}
}
