package com.bluepumpkin.ejb.rm.requests.common.validation;

import java.util.Date;

import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;

/**
 * @author rrajendran
 *
 * Ensures that the request's expiration date is after the current time.  
 */
public class RequestNotExpiredOnCreateHV implements Validator {

	private static final String m_className = RequestNotExpiredOnCreateHV.class.getName();

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	public ValidationResult validate(Validatable validatable) throws Exception {
		RequestAggregate reqAgg = (RequestAggregate) validatable;
		
		Date expirationDate = reqAgg.getExpirationDate();
		if ( !expirationDate.after(new Date()) ) {
			return ValidationUtil.setHardValidationResult(reqAgg, RmEjbBundleKey.REQ_ALREADY_EXPIRED, 
				expirationDate, m_className);
		}
		
		return null;
	}
}
