package com.verint.wfm.ws.entityImport.model;

import javax.xml.bind.annotation.XmlType;

@XmlType(name = "HoursOfOperationDTO", propOrder = { "full24Hour", "dayOpen", "dayClose" })
public class HoursOfOperationDTO {
	private static final long serialVersionUID = 1L;
	
	protected boolean m_isFull24Hour;
	protected short[] m_dayOpen = new short[8]; // We start Sunday at index 1 not 0
	protected short[] m_dayClose = new short[8];

	public boolean isFull24Hour() {
		return m_isFull24Hour;
	}

	public void setFull24Hour(boolean full24Hour) {
		m_isFull24Hour = full24Hour;
	}

	public short[] getDayOpen() {
		return m_dayOpen;
	}

	public void setDayOpen(short[] dayOpen) {
		m_dayOpen = dayOpen;
	}

	public short[] getDayClose() {
		return m_dayClose;
	}

	public void setDayClose(short[] dayClose) {
		m_dayClose = dayClose;
	}
}
