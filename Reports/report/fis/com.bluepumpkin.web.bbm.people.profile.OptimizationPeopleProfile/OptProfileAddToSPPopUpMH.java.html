package com.bluepumpkin.web.bbm.people.profile.OptimizationPeopleProfile;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Date;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.campaign.ejb.CampaignManager;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignWorkResource;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManager;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeType;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.core.base.ModelHandler;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.verint.web.wfm.WfmModelHandler;
import com.witness.web.uif.system.RequestContext;


public class OptProfileAddToSPPopUpMH extends ModelHandler {

	public static final Category LOG = Log.initCategory(OptProfileAddToSPPopUpMH.class
			.getName());

	/**
	 * Retrieve the Campaign Manager EJB remote interface
	 */
	public static CampaignManager getCampaignManager(RequestContext context) throws BbmException {
		return WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
	}

	public static void addEmployeestoSP(RequestContext context, ID spID, Collection<ID> empIDs) throws BbmCreateException, RemoteException, BbmException{
		if (empIDs == null || empIDs.isEmpty() || spID == null)
			return;

		getCampaignManager(context).linkWorkResourcesToSchedulingPeriod(spID, empIDs);
	}

	public static void removeEmployeesFromSP(RequestContext context, ID spID, ID spDEID, Collection<ID> empIDs) throws BbmCreateException, RemoteException, BbmException{
		//TODO: move method from WfmModelHandler to this class once workload code has been completely decoupled from foundation
		WfmModelHandler.removeEmployeesFromSP(context, spID, spDEID, empIDs);
	}

	public static Collection<CampaignWorkResource> getCampaignWorkResourceAssignments(RequestContext context, ID campaignID, Date startDate, Date endDate) throws BbmFinderException {
		try {
			CampaignManager cm = getCampaignManager(context);
			return cm.getCampaignWorkResourceAssignments(campaignID, startDate, endDate);
		} catch (BbmFinderException e) {
			throw e;
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
	}

	public static Collection<ID> getCampaignPoolerAssignments(RequestContext context, ID spID, Date startDate, Date endDate) throws BbmFinderException {
		try {
			CampaignManager cm = getCampaignManager(context);
			return cm.getSPPoolerAssignments(spID, startDate, endDate);
		} catch (BbmFinderException e) {
			throw e;
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
	}

	public static Map<ID, EmployeeType> getEmployeeTypesFromOrgs(Collection<ID> orgIDs, RequestContext context)
			throws BbmEJBCreateException, BbmFinderException, RemoteException {
		Map<ID, EmployeeType> empTypeMap = new HashMap<ID, EmployeeType>();

		WorkResourceManager wrMgr = BbmManagerFactory.getWorkResourceManager(context.isInWhatIfMode());
		Collection<EmployeeType> empTypes = wrMgr.getEmployeeType(orgIDs);
		for (EmployeeType empType : empTypes) {
			empTypeMap.put(empType.getSID(), empType);
		}

		return empTypeMap;
	}
}
