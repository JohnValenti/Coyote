package com.bluepumpkin.ejb.bbm.timeseries.util;

import java.util.Date;

import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;

/**
 * A service level based on the percentage of calls that will be answered within a
 * certain time interval.
 */
public class PercentCallsAnswered
	implements IServiceGoal
{
	private final TraceCube m_serviceLevelTraceCube;
	private final short m_serviceLevelTraceType;
	private final int m_secondsToAnswer;

	/**
	 * Creates a service level indicating that on average <code>percentCallsAnswered</code>% of the
	 * customers will be served within <code>secondsToAnswer</code> seconds.
	 * 
	 * @param serviceLevelTraceCube
	 * @param secondsToAnswer
	 */
	public PercentCallsAnswered(TraceCube serviceLevelTraceCube, short serviceLevelTraceType, int secondsToAnswer)
	{
		m_serviceLevelTraceCube = serviceLevelTraceCube;
		m_serviceLevelTraceType = serviceLevelTraceType;
		m_secondsToAnswer = secondsToAnswer;
	}

	/**
	 * Return the number of people required to satisfy this service level given an
	 * arrival rate and average handling time.
	 * 
	 * @param arrivalRate
	 * @param averageHandlingTime
	 * @return
	 */
	public int getFteRequirement(Date time, double arrivalRate, double averageHandlingTime)
	{
		double targetPca = m_serviceLevelTraceCube.getTraceValue(m_serviceLevelTraceType, time);
		if (targetPca <= 0) {
			return 0;
		}
		return ErlangC.getInstance().getAgentsPca(arrivalRate, averageHandlingTime, targetPca, m_secondsToAnswer);
	}
}
