package com.bluepumpkin.ejb.bbm.timeseries.util;

import java.util.Date;

import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;

/**
 * A service level based on the average length of time a caller will have to wait,
 * modeling abandonment in relation to a specified average patience of the callers.
 */
public class AverageSecondsToAnswerWithPatience implements IServiceGoal {
	private final TraceCube m_serviceLevelTraceCube;
	private final short m_serviceLevelTraceType;
	private final int m_patience;
	private final PatienceSource m_patienceSource;

	/**
	 * Creates a service level indicating that on average calls will be answered within
	 * a given number of seconds and assuming that callers will wait on average
	 * a certain number of seconds to be answered before abandoning the queue.
	 * 
	 * @param serviceLevelTraceCube the trace cube containing ASA data
	 * @param serviceLevelTraceType the trace type for the ASA data
	 * @param patience the callers' average patience in seconds
	 * @param patienceSource the source from which patience is to be derived: from
	 * the {@code patience} parameter or as a multiple of the ASA data
	 */
	public AverageSecondsToAnswerWithPatience(
			TraceCube serviceLevelTraceCube,
			short serviceLevelTraceType,
			int patience,
			PatienceSource patienceSource) {
		m_serviceLevelTraceCube = serviceLevelTraceCube;
		m_serviceLevelTraceType = serviceLevelTraceType;
		m_patience = patience;
		m_patienceSource = patienceSource;
	}

	@Override
	public int getFteRequirement(Date time, double arrivalRate, double averageHandlingTime) {
		double targetAsa = m_serviceLevelTraceCube.getTraceValue(m_serviceLevelTraceType, time);
		if (targetAsa <= 0) {
			return 0;
		}
		
		double patience = (m_patienceSource == PatienceSource.SetExplicitly) ? m_patience : 3 * targetAsa;
		if(patience ==0){
			
			patience = 3*targetAsa;
		}
		
		return ErlangWithPatience.getInstance().getAgentsAsaWithPatience(
				arrivalRate,
				averageHandlingTime,
				1. / patience,
				targetAsa);
	}

	/**
	 * Currently in an unskilled SP when Erlang A calculations are selected using the IsNewErlangC
	 * DB hack, patience is not entered as part of the service goals configuration.  Instead 3*ASA
	 * is used for patience.  This enum is used to indicate which source should be used.
	 */
	public static enum PatienceSource {
		SetExplicitly,
		DerivedFromASA
	}
}
