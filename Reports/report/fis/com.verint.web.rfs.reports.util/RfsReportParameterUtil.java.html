	@SuppressWarnings("unchecked")
	public List<StringsPair> getOptions(RequestContext context, Node paramNode, String currentValue, String pickerName, String parentId) {
		List<StringsPair> options = null;
		try {	
			options = RfsReportParametersModelHandler.getReportParamOptions(context, currentValue, pickerName, parentId);
		} catch (Exception ex) {
			
		}
		if (options == null || options.size() == 0) {
			options = reportControls.getReportParamListOptions(paramNode);
			if (options != null) {
				Localizer localizer = SessionUtil.getLocalizer(context.getLocaleContext());
				ResourceBundle bundle = ReportParametersDesignPC.getReportBundle();
				for (int i = 0; i < options.size(); i++) {
					StringsPair pair = (StringsPair) options.get(i);
					pair.setValue(localizer.i18n(bundle, pair.getValue()));
				}
			}
		}
		if (options == null || options.size() == 0) {
			return null;
		}
		return options;
	}
	
	public List<StringsPair> getOptions(RequestContext context, String pickerName, String currentValue) {
		List<StringsPair> options = null;
		try {

		} catch (Exception ex) {
			
		} finally {
			if (options != null) {
				Localizer localizer = SessionUtil.getLocalizer(context.getLocaleContext());
				ResourceBundle bundle = ReportParametersDesignPC.getReportBundle();
				for (int i = 0; i < options.size(); i++) {
					StringsPair pair = (StringsPair) options.get(i);
					pair.setValue(localizer.i18n(bundle, pair.getValue()));
				}
			}
		}

		return options;
	}
	
	public String getControlPickerName(RequestContext context, Node controlNode) {
		String pickerName = rfsReportControls.getControlPickerName(controlNode);
		String filterBy = reportControls.getControlFilterBy(controlNode);
		// Get picker used for resources, forecast/history/duration/volume
		if(RESOURCES.equals(pickerName)) {
			String resourcesType = context.getRequest().getParameter(PARAM_RESOURCETYPE);			
			if(resourcesType == null || resourcesType.equals("")) {
				resourcesType = FORECAST_RESOURCES;
			}
			if(FILTER_DURATION_FORECAST.equals(filterBy)) {
				pickerName = DURATION_FORECAST;
			} else if (FORECAST_RESOURCES.equals(resourcesType) && FILTER_VOLUME_FORECAST.equals(filterBy)) {
				pickerName = VOLUME_FORECAST;
			} else {
				pickerName = resourcesType;
			}		
		} else if(CATEGORY.equals(pickerName))  { // Get picker used for category - transaction category/resource category 
			String categoryType = context.getRequest().getParameter(PARAM_CATEGORYTYPE);
			if(categoryType == null || categoryType.equals("") || CATEGORYTYPE_RESOURCE.equals(categoryType)) {
				pickerName = RESOURCE_CATEGORY;
			} else if(CATEGORYTYPE_TRANSACTION.equals(categoryType)) {
				pickerName = TRANSACTION_CATEGORY;
			}
		} else if(FORECAST_RESOURCES.equals(pickerName) && !(filterBy == null || filterBy.equals("")))  { // Get picker used for forecast resources 
			pickerName = filterBy;
		}
		return pickerName;
	}
