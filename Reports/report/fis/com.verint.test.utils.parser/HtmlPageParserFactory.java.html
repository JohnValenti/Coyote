package com.verint.test.utils.parser;


import java.util.HashMap;
import java.util.Map;

/**
 * Factory provide singleton objects of Html Page Parsers
 */
public class HtmlPageParserFactory {

	private static Map<String, HtmlPageParser> parserMap = new HashMap<String, HtmlPageParser>();

	private HtmlPageParserFactory() {
	}

	/**
	 * Returns a singleton class of the HtmlPageParser matching the input class name.
	 * @param parserType - The fully qualified class name of the parser to return.
	 * @return - The singleton parser matching the given class name.
	 * @throws Exception
	 */
	public static HtmlPageParser getInstance(final String parserType) throws Exception {
		//is required parser already instantiated
		if (parserMap.containsKey(parserType)) {
			return parserMap.get(parserType);
		}

		//instantiate new parser
		Class<?> clazz = Class.forName(parserType);
		HtmlPageParser htmlPageParser = (HtmlPageParser) clazz.newInstance();

		parserMap.put(parserType, htmlPageParser);
		return htmlPageParser;
	}

	/**
	 * Returns a singleton class of the HtmlPageParser matching the input class name.
	 *
	 * @param parserType - The fully qualified class name of the parser to return.
	 * @param config - Additional configuration of the parser which we need to use for page parsing.
	 * @return The singleton parser matching the given class name with loaded configuration.
	 * @throws Exception
	 */
	public static HtmlPageParser getInstance(final String parserType, Map<String, Object> config) throws Exception {
		//is required parser already instantiated
		if (parserMap.containsKey(parserType)) {
			HtmlPageParser htmlPageParser = parserMap.get(parserType);
			htmlPageParser.setConfig(config);
			return htmlPageParser;
		}

		//instantiate new parser
		Class<?> clazz = Class.forName(parserType);
		HtmlPageParser htmlPageParser = (HtmlPageParser) clazz.newInstance();
		htmlPageParser.setConfig(config);

		parserMap.put(parserType, htmlPageParser);
		return htmlPageParser;
	}
}
