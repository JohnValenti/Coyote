package com.bluepumpkin.web.bbm.campaign.selection;

/**
 * Title:        CampaignSchedulingPeriodTreePC
 * Description:  Campaign / Scheduling Period Tree Page Component

public class CampaignSchedulingPeriodTreePC extends MultiColInputTreePC {
	public static Campaign ROOT_OBJ = null;
	static { ROOT_OBJ = new Campaign(); ROOT_OBJ.setID(new ID(-1)); }
	public static final String CHECKED_ID_FN = "checkedID";
	private boolean m_areCampaignsSelectable;
	private ArrayList<String> m_selectedParentNodeIDs = new ArrayList<String>();

	private boolean isAjaxMode = false;
	
	public CampaignSchedulingPeriodTreePC(RequestContext context) {
		this(context, true);
	}
	
	public CampaignSchedulingPeriodTreePC(RequestContext context, boolean areCampaignsSelectable) {
		super(context, CHECKED_ID_FN, BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.CAMPAIGN_NAME);
		setName("campaignSPTree");
		setCollapsed(true);
		m_areCampaignsSelectable = areCampaignsSelectable;

		addJSVariableAsString("isCampaignSelectable", String.valueOf(m_areCampaignsSelectable));
		try {
			DBConfigManager dbManager = BbmManagerFactory.getDBConfigManager(context.isInWhatIfMode());
			isAjaxMode = dbManager.getBooleanValue("com/bluepumpkin/web/bbm/campaign/selection/ajax");
			if (isAjaxMode) {
				addRequiredJavaScriptFile(JavaScriptFileID.UNDERSCORE);
				addRequiredJavaScriptFile(JavaScriptFileID.MUSTACHE_TEMPLATE_LIB);
				addRequiredJavaScriptFile(WfmJavaScriptFileID.CAMPAIGN_SP_SELECTION_MANAGER);
			}
		} catch (Exception ex) {
			log.error(ex);
		}
	}

	protected Object getTreeRootData() {
		return ROOT_OBJ;
	}

	protected Collection retrieveTreeObjects() throws Exception {
		if (isAjaxMode) {
			return Collections.emptyList();
		} else {
			return CampaignModelHandler.getCampaignsAndAssociatedSchedulingPeriodsForUser(m_context, m_context.getUser(), true);
		}
	}
	
	protected TreeNodeHelper getTreeNodeHelper(Collection treeObjects) {
		return new DualTypeTreeNodeHelper(treeObjects, getTreeRootData(), this,
				Campaign.class, SchedulingPeriod.class);
	}

	protected Collection<?> sortTreeObjects(Collection treeObjects) {
		List<?> resultList = (treeObjects instanceof List<?>) ? (List<?>)treeObjects : new ArrayList(treeObjects);
		Collections.sort(resultList, new CampaignSchedulingPeriodComparator(getLocalizer()));
		return resultList;
	}

	protected TreeModel buildTreeModel(Collection treeObjects) {
		Collection<?> sortedObjs = sortTreeObjects(treeObjects);
		TreeNodeHelper treeNodeHelper = getTreeNodeHelper(sortedObjs);
		return TreeModelBuilder.buildTreeModel(sortedObjs, treeNodeHelper);
	}
	
	protected void addInputUI(Object treeObject, DefaultMultiColumnNodeData nodeData) {
		String inputType = isMultiRowSelectable()?"checkbox":"radio";
		String inputUI = makeInputUI(inputType, treeObject, getOnClickJS());
		nodeData.add(inputUI, "width:1%");
	}
	
	public DefaultMutableTreeNode toTreeNode(Object treeObject) {
		if (treeObject==null) return null;

		// Create Node Data
		ID nodeID = getNodeID(treeObject);
		String nodeDisplay = getNodeDisplay(treeObject, true);
		
		String nodeIDStr = (nodeID!=null)?nodeID.toString():"";
		DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(nodeIDStr);
		
		if (m_isInputSelectorDisplayed) {
			addInputUI(treeObject, nodeData);
		}
		nodeData.add(nodeDisplay);
		nodeData.getNodeAttributes().put("notMultiSelectable", "true");
		if (!m_areCampaignsSelectable && nodeID.toString().startsWith("c")) nodeData.setSelectable(false);
		
		return new DefaultMutableTreeNode(nodeData);
	}
	
	public String getNodeNameDisplay(Object object) {
		if (object.getClass().equals(Campaign.class)) {
			return ((Campaign)object).getName();
		} else if (object.getClass().equals(SchedulingPeriod.class)) {
			SchedulingPeriod sp = (SchedulingPeriod)object;
			LocalDate endDate = new LocalDate(sp.getEndTimeLocal());
			endDate.add(Calendar.DATE, -1);
			return getLocalizer().formatDate(sp.getStartTimeLocal(), RegionalFormatBundleKey.DATE_FORMAT) + 
				" - " + getLocalizer().formatDate(endDate, RegionalFormatBundleKey.DATE_FORMAT);
		}
		
		return "";
	}
	
	public ID getNodeID(Object object) {
		if (((BasicVO)object).getID() == null) { return null; }
		if (object.getClass().equals(Campaign.class))
			return new ID(CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX + ((BasicVO)object).getID().toString());
		else {	//Scheduling period object selected, create the SP ID and campaign ID as a composite ID string
			return new ID(CampaignSchedulingPeriodSelectionUtil.buildCompositeIDString(
					((SchedulingPeriod)object).getID(), ((SchedulingPeriod)object).getCampaignID()));
		}
	}

	public ID getParentNodeID(Object object) {
		if (((BasicVO)object).getParentID() == null) { return null; }
		return new ID(CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX + ((BasicVO)object).getParentID().toString());
	}
	
	//We want to expand the selected scheduling period's parent node (which is a campaign).
	//  If this campaign is a sub campaign, we also want to expand its parent (which is a distributed campaign).
	public void setExpandedNodes(String selectedID) {
		try {
			String selectedSpID = CampaignSchedulingPeriodSelectionUtil.extractIDs(selectedID, "s");
			String selectedCampaignID = CampaignSchedulingPeriodSelectionUtil.extractIDs(selectedID, "c");
			if (StringUtil.isEmpty(selectedSpID) == false) {
				for (Object obj : m_treeObjects) {
					if (obj.getClass().equals(SchedulingPeriod.class)) {
						SchedulingPeriod sp = (SchedulingPeriod)obj;
						if (sp.getID().toString().equals(selectedSpID)) {
							m_selectedParentNodeIDs.add("c:" + sp.getCampaignID().toString());
							Campaign campaign = retrieveCampaignFromTreeObjects(sp.getCampaignID());
							if (campaign != null && campaign.getParentID() != null) {
								m_selectedParentNodeIDs.add("c:" + campaign.getParentID());
							}
						}
					}
				}
			} else if (StringUtil.isEmpty(selectedCampaignID) == false) {
				for (Object obj : m_treeObjects) {
					if (obj.getClass().equals(Campaign.class)) {
						Campaign campaign = (Campaign)obj;
						if (campaign.getID().toString().equals(selectedCampaignID) &&
								campaign.getParentID() != null) {
							m_selectedParentNodeIDs.add("c:" + campaign.getParentID());
						}
					}
				}				
			}
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		}
	}
	
	protected boolean isNodeOpen(MultiColumnNodeData rowData, boolean isCollapsed) {
		return !(isCollapsed || m_bCollapsed) ||
			m_selectedParentNodeIDs.contains(rowData.getID().toString());
	}
	
	protected boolean isChildNodeCollapsed(MultiColumnNodeData rowData) {
		return m_selectedParentNodeIDs.contains(rowData.getID().toString()) == false;
	}
	
	protected String getToggleImageControl(MutableTreeNode node, String sRowID){
		String img = m_selectedParentNodeIDs.contains(
				((MultiColumnNodeData)((DefaultMutableTreeNode)node).getUserObject()).getID())
				? m_sOpenImgPath : m_sClosedImgPath;
		String alt = m_bCollapsed ? m_sClosedImgAlt : m_sOpenedImgAlt;
		Object[] params = { getName(), sRowID, img, alt };
		return IMG_CTRL_MF.format(params);
	}
	
	private Campaign retrieveCampaignFromTreeObjects(ID campaignID) {
		for (Object obj : m_treeObjects) {
			if (obj.getClass().equals(Campaign.class)) {
				Campaign campaign = (Campaign)obj;
				if (campaign.getID().equals(campaignID)) {
					return campaign;
				}
			}
		}
		return null;
	}
}
