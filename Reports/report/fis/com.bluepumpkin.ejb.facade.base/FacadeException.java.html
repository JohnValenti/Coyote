package com.bluepumpkin.ejb.facade.base;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.ejb.facade.l10n.FacadeEjbLogBundleKey;

/**
 * Title:        Blue Pumpkin Software Facade Module
 * Description:  facade exception
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */

public class FacadeException extends BPException {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// Override BPException's default resource bundle value, must provide
        // the getBundleName() method
	protected String m_bundle = "com.bluepumpkin.ejb.facade.l10n.FacadeEjbLogBundle";
	// In some cases, the EJB will return an object back to the controller
        // via Exception So the controller doesn't store it in the session
	private Object m_content = null;
	/**
	 * a default constructor
	 */
	public FacadeException() {
		super();
	}

	/**
	 *  Construct a new BPException object.
	 *
	 *  @param e any embbed core Exception, usually wrapps a non-BPException exception
	 *  @param bundleName base resource bundle name for localized message, if null defaults to "com.bluepumpkin.common.l10n.CommonBundle"
	 *  @param msgID key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 *  @param sevCode user can provide a specific severity code, but the constructor will pick up the greatest value from passed-in exception, existing default exception severity code and this user specied value. This is to ensure the severity code is always increased.
	 *  @param args provide parameters for the localized string, specified by msgID in the resource bundle file.
	 */
	public FacadeException (Exception e, String bundleName, String msgID, int sevCode, Object[] args) {
		super(e,bundleName, msgID, sevCode, args);
		m_bundle = bundleName;
	}
	
	
	/**
	 *  Construct
	 *
	 *  @param e, any embbed facade Exception, usually wrapps a
         *            non-BPException exception
	 *  @param msgID, key to locate the localized message information
         *            from the resource bundle file, with default value
         *            as "DEFAULT"
	 *  @param sevCode, user can provide a specific severity code,
         *            but the constructor will pick up the greatest value
         *            from passed-in exception, existing default exception
         *            severity code and this user specied value.
         *            This is to ensure the severity code is always increased.
	 *  @param args, provide parameters for the localized string,
         *            specified by msgID in the resource bundle file.
	 */
	public FacadeException (Exception e, String msgID, int sevCode, Object[] args) {
		super(e, msgID, sevCode, args);
	}
	/**
	 *  Construct
	 *
	 *  @param e, any embbed facade Exception, usually wrapps a
         *            non-BPException exception
	 *  @param msgID, key to locate the localized message information
         *            from the resource bundle file, with default value
         *            as "DEFAULT"
	 *  @param sevCode, user can provide a specific severity code,
         *            but the constructor will pick up the greatest value
         *            from passed-in exception, existing default exception
         *            severity code and this user specied value.
         *            This is to ensure the severity code is always increased.
	 */
	public FacadeException(Exception e, String msgID, int sevCode) {
		this(e, msgID, sevCode, null);
	}
	/**
	 *  Construct
	 *
	 *  @param e, any embbed facade Exception, usually wrapps a
         *            non-BPException exception
	 *  @param msgID, key to locate the localized message information
         *            from the resource bundle file, with default value
         *            as "DEFAULT"
	 *  @param args, provide parameters for the localized string,
         *            specified by msgID in the resource bundle file.
	 */
	public FacadeException(Exception e, String msgID, Object[] args) {
		this(e, msgID, DEFAULT_SEVCODE, args);
	}

	/**
	 *  Construct
	 *
	 *  @param msgID, key to locate the localized message information from
         *            the resource bundle file, with default value as "DEFAULT"
	 *  @param args, provide parameters for the localized string,
         *            specified by msgID in the resource bundle file.
	 */
	public FacadeException(String msgID, Object[] args) {
		this(null, msgID, DEFAULT_SEVCODE, args);
	}
	/**
	 *  Construct
	 *
	 *  @param msgID, key to locate the localized message information
         *           from the resource bundle file, with default value
         *           as "DEFAULT"
	 *  @param sevCode, user can provide a specific severity code,
         *           but the constructor will pick up the greatest value from
         *           passed-in exception, existing default exception severity
         *           code and this user specied value.
         *           This is to ensure the severity code is always increased.
	 */
	public FacadeException(String msgID, int sevCode) {
		this(null, msgID, sevCode, null);
	}
	/**
	 *  Construct
	 *
	 *  @param msgID, key to locate the localized message information
         *            from the resource bundle file, with default value
         *            as "DEFAULT"
	 */
	public FacadeException(String msgID) {
		this(null, msgID, DEFAULT_SEVCODE, null);
	}

	public FacadeException(Exception e) {
		this(e, FacadeEjbLogBundleKey.DEFAULT, DEFAULT_SEVCODE, null);
	}

	/**
	 *  getBundleName
	 *
	 *  Override the default resource bundle name defined in the
         *  BPException, so have to provide the override method to
         *  get the runtime value
	 *  @return the default resource bundle name
	 */
	public String getBundleName() {
		return m_bundle;
	}

	/**
	 * setContent
	 */
	public void setContent(Object obj) {
		m_content = obj;
	}
	public Object getContent() {
		return m_content;
	}
}
