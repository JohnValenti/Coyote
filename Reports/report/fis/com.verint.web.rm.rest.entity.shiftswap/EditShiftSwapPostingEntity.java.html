package com.verint.web.rm.rest.entity.shiftswap;

import javax.xml.bind.annotation.XmlRootElement;

import com.verint.web.rest.entity.AbstractWFOEntity;

/**
 * Represents the parts of a ShiftSwapPosting that can be edited.
 */
@XmlRootElement(name = "posting")
public class EditShiftSwapPostingEntity extends AbstractWFOEntity {

	private boolean allowPartialPickup;
	private String comment;
	private String expirationDate;
	private boolean negotiation;

	public EditShiftSwapPostingEntity() { // NOSONAR
	}

	public boolean isAllowPartialPickup() {
		return allowPartialPickup;
	}

	public void setAllowPartialPickup(boolean allowPartialPickup) {
		this.allowPartialPickup = allowPartialPickup;
	}

	public String getComment() {
		return comment;
	}

	public void setComment(String comment) {
		this.comment = comment;
	}

	public String getExpirationDate() {
		return expirationDate;
	}

	public void setExpirationDate(String expirationDate) {
		this.expirationDate = expirationDate;
	}

	public boolean isNegotiation() {
		return negotiation;
	}

	public void setNegotiation(boolean negotiation) {
		this.negotiation = negotiation;
	}
}
