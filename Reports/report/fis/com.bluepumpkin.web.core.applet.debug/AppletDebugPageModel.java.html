package com.bluepumpkin.web.core.applet.debug;

import java.util.ResourceBundle;

import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.witness.ejb.core.security.model.User;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.contenttitle.ContentTitlePC;
import com.witness.web.uif.pagemodel.WorkpanePageModel;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        AppletDebugPageModel.java
 * Description:  Used to generate View for Applet Debug Page
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       Alvin Cham
 * @version      1.0
 *
 * Created on Feb 23, 2004, 11:13 AM
 */
public class AppletDebugPageModel extends WorkpanePageModel {
	private boolean m_isAuthorized = false;
	private boolean m_isEnabled = false;
	private ResourceBundle m_core_bundle;

	/** Creates a new instance of AppletDebugPageModel */
	public AppletDebugPageModel(RequestContext context) {
		super(context, BASIC_MODEL);
		m_core_bundle = m_localizer.getBundle(CoreWebBundleKey.BUNDLE_NAME);

		m_isAuthorized = isAuthorized(context.getUser());

		AppletDebugManager manager = AppletDebugManager.getInstance();
		m_isEnabled = manager.isEnabled(context);

		setJSMediator(JavaScriptFileID.MEDIATOR_WORKPANE_GENERIC);
	}

	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static AppletDebugPageModel getInstance(RequestContext context) {
		AppletDebugPageModel model = null;

		if (context.getPageModel() instanceof AppletDebugPageModel) {
			model = (AppletDebugPageModel)context.getPageModel();
		} else {
			model = new AppletDebugPageModel(context);
			model.extractParameters(context.getRequest());
			context.setPageModel(model);
		}

		// Make sure model is populated
		model.populateModel(context);
		return model;
	}

	/**
	 * Populate Model
	 */
	public void populateModel(RequestContext context) {
		if (m_isInitialized) return; else m_isInitialized = true;

		if (m_isAuthorized) {
			addMsg();
		}

		initToolbar();
	}

	/**
	 * Return true if current user is authorized to page model
	 */
	public static boolean isAuthorized(User user) {
		return user!=null && user.isSuperUser();
	}

	/**
	 * Add Hit Counter is Disabled Message
	 */
	private void addMsg() {
		if (isEnabled()) {
			addPageMessage(Message.WARNING_TYPE,
							CoreWebBundleKey.APPLET_DEBUG_IS_ENABLED,
							CoreWebBundleKey.BUNDLE_NAME);
		} else {
			addPageMessage(Message.WARNING_TYPE,
							CoreWebBundleKey.APPLET_DEBUG_IS_DISABLED,
							CoreWebBundleKey.BUNDLE_NAME);
		}
	}

	public boolean isEnabled() {
		return m_isEnabled;
	}

	/**
	 * Initialize toolbar
	 * This is the default toolbar, dervided classes should override this method
	 */
	protected void initToolbar() {
		if (isEnabled()) {
			String disableLabel = i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_DISABLE);
			m_toolbar.addSubmitTextButton(PageAction.DISABLE_APPLET_DEBUG, disableLabel, m_isAuthorized);
		} else {
			String enableLabel = i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_ENABLE);
			m_toolbar.addSubmitTextButton(PageAction.ENABLE_APPLET_DEBUG, enableLabel, m_isAuthorized);
		}

		String refreshLabel = i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_REFRESH);
		m_toolbar.addSubmitTextButton(PageAction.REFRESH, refreshLabel, m_isAuthorized);
	}

	/**
	 * Return the Page Title component
	 */
	public ContentTitlePC getContentTitle() {
		// Initialize Content Title
		String pageTitle = i18n(m_core_bundle, CoreWebBundleKey.APPLET_DEBUG_MODE);
		m_contentTitlePC.setPageTitle(pageTitle);

		return m_contentTitlePC;
	}


	/**
	 * Return Html Display
	 */
	public String getHtmlDisplay() {
		return "";
	}


	/** Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return "applet_debug";
	}

}
