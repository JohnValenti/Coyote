package com.verint.ejb.wfm.cache;

import java.lang.reflect.InvocationTargetException;
import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Collections;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;
import com.bluepumpkin.ejb.bbm.cache.CacheManager;
import com.verint.ejb.wfm.cache.CacheSkill;
import com.bluepumpkin.ejb.bbm.cache.CacheUtil;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.skill.model.Skill;

public class WfmCacheUtil extends CacheUtil {

	/**
	 * @return
	 */
	private static CacheLastShiftAssignment getCacheLastShiftAssignmentPub() throws IllegalArgumentException,
			SecurityException, InstantiationException, IllegalAccessException, InvocationTargetException,
			NoSuchMethodException {

		return (CacheLastShiftAssignment) checkIfThreadLocalCacheEnabledAndInstCacheIfNecess(
				CacheLastShiftAssignment.class, CacheManager.CACHENAME_CACHELASTSHIFTASSIGNMENTPUB);
	}

	public static ShiftAssignment getLastShiftAssn(ID empID, boolean isForPublished) throws BbmObjectNotFoundException,
			BbmEJBCreateException, BbmFinderException, RemoteException, IllegalArgumentException, SecurityException,
			InstantiationException, IllegalAccessException, InvocationTargetException, NoSuchMethodException {
		ShiftAssignment lastShift = null;

		// check if cache enabled.
		CacheLastShiftAssignment cacheLastShift = getCacheLastShiftAssignmentPub();
		// if enabled
		if (cacheLastShift != null) {
			lastShift = cacheLastShift.getLastShiftAssignment(empID, isForPublished);
		} else { // cache not enabled
			Map empIDToLastShiftMap = CacheLastShiftAssignment.getLastShiftAssignmentUncached(
					Collections.singleton(empID), isForPublished, null);
			lastShift = (ShiftAssignment) empIDToLastShiftMap.get(empID);
		}
		return lastShift;
	}

	/**
	 * Note: This method will <b>not work</b> if the thread local cache is not
	 * enabled.
	 * 
	 * See javadoc for {@link CacheSkill#getSkillByID(ID) getSkillByID(ID)}
	 * method.
	 * 
	 * @param skillIDs
	 * @return
	 */
	public static Skill getSkillByID(ID skillID) throws BbmFinderException, RemoteException, IllegalArgumentException,
			SecurityException, InstantiationException, IllegalAccessException, InvocationTargetException,
			NoSuchMethodException {

		CacheSkill cacheSkill = getCacheSkill();

		if (cacheSkill == null) {
			throw new RuntimeException("Cache must be enabled for this method");
		}

		return cacheSkill.getSkillByID(skillID);
	}

	/**
	 * Note: This method will <b>not work</b> if the thread local cache is not
	 * enabled.
	 * 
	 * See javadoc for {@link CacheSkill#getSkillByID(ID) getSkillByID(ID)}
	 * method.
	 * 
	 * @param skillIDs
	 * @return
	 */
	public static Collection getSkillIDsNotDeleted(Collection skillIDs) throws IllegalArgumentException,
			SecurityException, InstantiationException, IllegalAccessException, InvocationTargetException,
			NoSuchMethodException, BbmFinderException, RemoteException {

		CacheSkill cacheSkill = getCacheSkill();

		if (cacheSkill == null) {
			throw new RuntimeException("Cache must be enabled for this method");
		}

		return cacheSkill.getSkillIDsNotDeleted(skillIDs);
	}

	/**
	 * @return
	 */
	private static CacheSkill getCacheSkill() throws IllegalArgumentException, SecurityException,
			InstantiationException, IllegalAccessException, InvocationTargetException, NoSuchMethodException {
		return (CacheSkill) checkIfThreadLocalCacheEnabledAndInstCacheIfNecess(CacheSkill.class,
				CacheManager.CACHENAME_CACHESKILL);
	}
}
