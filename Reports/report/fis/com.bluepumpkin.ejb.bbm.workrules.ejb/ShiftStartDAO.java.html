package com.bluepumpkin.ejb.bbm.workrules.ejb;

import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftStart;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftStartFieldInfo;

public class ShiftStartDAO extends DAOBase<ShiftStart> {

	private static FieldInfo m_fieldInfo = new ShiftStartFieldInfo();
	
	public ShiftStartDAO() {
		super();
	}
	
	public ShiftStartDAO(Jdmo dmo) {
		super(dmo);
	}

	@Override
	protected ShiftStart createValueObject() {
		return new ShiftStart();
	}

	@Override
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	/**
	 * 
	 * @param shiftIDs - This method assumes the IDs in the input collection are DEIDs
	 * @return
	 * @throws BbmFinderException 
	 */
	public Collection<ShiftStart> getShiftStartsForShifts(Collection<ID> shiftIDs) throws BbmFinderException {
		return getObjects("SHIFTID IN (" + JdmoUtil.createIDList(shiftIDs) + ")");
	}
}
