package com.bluepumpkin.ejb.bbm.workrules.ejb;

import java.util.Date;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.dao.OneToManyRelationship;
import com.bluepumpkin.ejb.bbm.dao.RichObjectDAO;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftGroup;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftGroupFieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftGroupShift;

public class ShiftGroupDAO extends RichObjectDAO<ShiftGroup> {

	private static FieldInfo m_fieldInfo = new ShiftGroupFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	private ShiftGroupShiftDAO sgsDAO;

	public ShiftGroupDAO() {
		super();
	}

	public ShiftGroupDAO(Jdmo dmo) {
		super(dmo);
	}
	
	@Override
	protected ShiftGroup createValueObject() {
		return new ShiftGroup();
	}

	@Override
	protected void buildDBRelationships() {
		sgsDAO = new ShiftGroupShiftDAO(m_dmo);
		
		addRelationship(new OneToManyRelationship(this, sgsDAO, getFieldInfo().getIDFieldIndex(),
				sgsDAO.getFieldInfo().getFieldIndexByName("SHIFTGROUPID"),
				ShiftGroup.SHIFTGROUPSHIFT_RELATIONSHIP_KEY));	
	}
	
	public void setChildFieldsOnCreate(String relationshipKey, ShiftGroup shiftGroup, ValueObjectBase childObject) {
		ShiftGroupShift sgs = (ShiftGroupShift)childObject;
		sgs.setCreatedBy(getCallerPrincipal());
		sgs.setCreatedDate(new Date(System.currentTimeMillis()));	
	}
	
	public void setChildFieldsOnUpdate(String relationshipKey, ShiftGroup shiftGroup, ValueObjectBase childObject) {
		ShiftGroupShift sgs = (ShiftGroupShift)childObject;
		sgs.setModifiedBy(getCallerPrincipal());
		sgs.setModifiedDate(new Date(System.currentTimeMillis()));	
	}
}
