package com.verint.ejb.facade.scheduling;

import java.util.ArrayList;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.JMSMessageBase;

public class ClearScheduleRequestMessage extends JMSMessageBase {
	/**
	 * 
	 */
	private static final long serialVersionUID = -3939392854453075992L;
	private final List<ID> m_requestIDs = new ArrayList<ID>();
	
	private int m_clearAttempt = 0;
	
	private boolean m_forceClear = false;
	
	
	//keep the scheduling information
	//useful during retry
	private final ScheduleRequestParameters m_ScheduleRequestParams;
	
	public ClearScheduleRequestMessage(List requests, ScheduleRequestParameters schedRequestParameters) {
		if (requests != null) {
			m_requestIDs.addAll(requests);
		}
		m_ScheduleRequestParams = schedRequestParameters;
	}
	
	/**
	 * 
	 * @param requests
	 * @param schedulingParameters
	 * @param clearAttempt - Number of times this message was attempted to be processed.  
	 * FacadeSchedulingManagerMDB will create a new ClearScheduleRequestMessage and increment the clearAttempt on the message
	 * to indicate how many attempts have occurred to clear the requests in process.
	 */
	public ClearScheduleRequestMessage(List requests, ScheduleRequestParameters schedulingParameters, int clearAttempt)
	{
		this(requests, schedulingParameters);
		m_clearAttempt = clearAttempt;
	}
	
	
	public ClearScheduleRequestMessage(List requests, ScheduleRequestParameters schedulingParameters, boolean forceClear)
	{
		this(requests, schedulingParameters);
		m_forceClear = forceClear;
	}
	
	
	
	/**
	 * Returns the clear attempt count	
	 * @return
	 */
	public int getClearAttempt() {
		return m_clearAttempt;
	}

	/**
	 * @return the m_requestIDs
	 */
	public List<ID> getRequestsToClear() {
		return new ArrayList<ID>(m_requestIDs);
	}
	
	public ScheduleRequestParameters getScheduleRequestParameters(){
		return m_ScheduleRequestParams;
	}
	
	public boolean isForceClear()
	{
		return m_forceClear;
	}

}
