package com.bluepumpkin.ejb.bbm.workrules.timebanking.ejb;


import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.*;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.timebanking.model.TimeBankPlannerData;
import com.bluepumpkin.ejb.bbm.workrules.timebanking.model.TimeBankPlannerDataFieldInfo;

public class TimeBankPlannerDataDAO extends DAOBase {
	private static FieldInfo m_fieldInfo = new TimeBankPlannerDataFieldInfo();

	public TimeBankPlannerDataDAO() {
		super();
	}

	public TimeBankPlannerDataDAO(Jdmo dmo) {
		super(dmo);
	}

	/**
	 * Returns the ID of a timebankplannerdata given the name.
	 *
	 * @param name
	 * @return ID
	 * @throws JdmoException
	 */
	public ID getTimeBankPlannerDataIDByName(String name) throws JdmoException {
		ID timeBankPlannerDataID = null;

		final Jdmo dmo = new Jdmo();
		try {
			final StringBuffer pStmt = new StringBuffer("select ID from TIMEBANKPLANNERDATA where NAME=?");
			final JdmoQuery jq = dmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			jq.setParString(1, name);
			final JdmoRowset rs = dmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

			if (rs.next()) {
				timeBankPlannerDataID = rs.getID(1);
			}

		} finally {
			dmo.cleanUp();
		}
		return timeBankPlannerDataID;
	}

	/**
	 * Deletes timebankplannerdata given the name.
	 *
	 * @throws BbmRemoveException
	 */
	public void deleteTimeBankPlannerDataByName(String name) throws BbmRemoveException {
		final Jdmo jdmo = new Jdmo();
		if (StringUtil.isEmpty(name)) {
			return;
		}
		try {
			String sqlQuery = "DELETE FROM TIMEBANKPLANNERDATA WHERE NAME = "
					+ JdmoUtil.asSqlLiteral(name);
			jdmo.execute(sqlQuery);
		} catch (JdmoException ex) {
			throw new BbmRemoveException(ex);
		} finally {
			jdmo.cleanUp();
		}
	}

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	protected TimeBankPlannerData createValueObject() {
		return new TimeBankPlannerData();
	}

}
