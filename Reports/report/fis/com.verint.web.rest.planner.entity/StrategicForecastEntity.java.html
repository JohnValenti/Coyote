package com.verint.web.rest.planner.entity;


import com.verint.web.rest.entity.AbstractWFOEntity;

import javax.xml.bind.annotation.XmlRootElement;

@XmlRootElement(name = "strategicForecast")
public class StrategicForecastEntity extends AbstractWFOEntity {
	private String queueID;
	private String time;
	private double aht;
	private double cv;

	public String getQueueID() {
		return queueID;
	}

	public void setQueueID(String queueID) {
		this.queueID = queueID;
	}

	public String getTime() {
		return time;
	}

	/**
	 *
	 * @param time - Format yyyy-MM-dd-Thh:mm:ssZ
	 */
	public void setTime(String time) {
		this.time = time;
	}

	public double getAht() {
		return aht;
	}

	public void setAht(double aht) {
		this.aht = aht;
	}

	public double getCv() {
		return cv;
	}

	public void setCv(double cv) {
		this.cv = cv;
	}
}
