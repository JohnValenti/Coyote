package com.verint.ejb.bbm.forecast.model.history.service;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmTimeSeriesException;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.time.TimeIntervalAtTime;
import com.bluepumpkin.ejb.bbm.timeseries.model.ActualTraceCube;
import com.bluepumpkin.ejb.bbm.workload.ejb.WorkloadManager;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.bluepumpkin.ejb.bbm.workload.model.QueueList;
import com.verint.ejb.wfm.WfmManagerFactory;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Date;
import java.util.Map;


public class OutboundHistoryDataService extends AbstractHistoryDataService implements IOutboundHistoryDataService {

	private final WorkloadManager workloadManager;

	public OutboundHistoryDataService(Campaign campaign, boolean isWhatIf) throws BbmEJBCreateException {
		super(campaign, isWhatIf);
		workloadManager = WfmManagerFactory.getWorkloadManager(isWhatIf);
	}

	@Override
	public Map<ID, ActualTraceCube> getHistory(Collection<Queue> queues, TimeIntervalAtTime period)
			throws BbmTimeSeriesException {
		try {
			Date adjustedEndDate = getAdjustedEndDate(period);
			return forecastTimeSeriesManager.getOutboundHistory(
					queues, campaign.getID(), period.getStartTime(), adjustedEndDate);
		} catch (RemoteException e) {
			throw new BbmTimeSeriesException(0, e);
		} catch (BbmEJBCreateException e) {
			throw new BbmTimeSeriesException(0, e);
		} catch (BbmFinderException e) {
			throw new BbmTimeSeriesException(0, e);
		}
	}

	@Override
	public Collection<QueueList> getHistoricalOutboundCallLists(Queue queue) throws BbmFinderException {
		try {
			return workloadManager.getQueueHistoryList(queue.getID());
		} catch (RemoteException e) {
			throw new BbmFinderException(e);
		}
	}
}
