package com.verint.web.rm.rest.facade;

import java.util.HashSet;
import java.util.Set;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;

public class RequestsMetaData {
	private final Set<ID> activityIDs = new HashSet<ID>();
	private final Set<ID> shiftIDs = new HashSet<ID>();
	private final Set<ID> shiftOTExtensionIDs = new HashSet<ID>();
	private final OrganizationSetting orgSettings;

	public RequestsMetaData(OrganizationSetting orgSettings) {
		this.orgSettings = orgSettings;
	}

	public Set<ID> getActivityIDs() {
		return activityIDs;
	}

	public Set<ID> getShiftIDs() {
		return shiftIDs;
	}

	public Set<ID> getShiftOTExtensionIDs() {
		return shiftOTExtensionIDs;
	}

	public OrganizationSetting getOrgSettings() {
		return orgSettings;
	}

}
