package com.bluepumpkin.web.rm.requests.shiftbidding;

public class ShiftBidUtil {
	private static final int MAX_PREF_VALUE_FOR_FILTER = 4;
	private ShiftBidUtil() {};

	public static DateWrapper getSeniority(Employee emp, TimeZone tz, Localizer localizer) {
		Date date = emp.getStartTime();
		return new DateWrapper(date, localizer, tz, RegionalFormatBundleKey.DATE_FORMAT);
	}

	public static final String createBidPrefDDForFilter(String fieldName,
			String value, StringsPair allOption, Localizer localizer) {

		String lastOptionLbl = localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.OTHERS);
		StringsPair lastOption = new StringsPair(
				String.valueOf(RequestFilter.BID_PREF_VALUE_FIVEANDABOVE),
				lastOptionLbl
		);

		return createBidPrefDD(fieldName, value, MAX_PREF_VALUE_FOR_FILTER,
				allOption, lastOption);
	}

	public static final String createBidPrefDD(String fieldName, String value, int maxPrefValue) {
		return createBidPrefDD(fieldName, value, maxPrefValue, null, null);
	}

	public static final String createBidPrefDD(String fieldName, String value,
			int maxPrefValue, StringsPair allOption, StringsPair lastOption) {

		List<StringsPair> options = new ArrayList<StringsPair>(maxPrefValue + 1);
		if (allOption != null) {
			options.add(allOption);
		}
		for(int i=1; i<=maxPrefValue; i++) {
			String sValue = String.valueOf(i);
			options.add(	new StringsPair(sValue, sValue)	);
		}
		if (lastOption != null) {
			options.add(lastOption);
		}

		return HtmlUtil.makeSelectInput(fieldName, value, JSUtil.ON_CHANGE,	options);
	}
}
