/**
 * Handles start/end date, employee ID, and employee filter query string parameters.
 * When this handler is instantiated, it will first verify that the requiredFilterParams are present
 * in the given map of filterParams, if not present it will throw a RequiredFilterNotPresentException.
 *
 * At instantiation time, if all valid params are present, EmployeeFilter EJB will be called if an employee
 * filter name is present in order to load the employee IDs linked to that employee filter.
 */
public class DateAndEmployeeIdComboFilterParamHandler extends FilterParamHandler {

	private final Map<String, Set<String>> filterParams;
	private final EmployeeFilter ef;
	private final User user;
	private final Collection<ID> employeeIds;

	public DateAndEmployeeIdComboFilterParamHandler(EmployeeFilter ef, User user, Map<String, Set<String>> filterParams,
			Set<String> requiredFilterParamNames, String resourceName)
			throws BbmFinderException, RemoteException {
		this.filterParams = filterParams;
		this.ef = ef;
		this.user = user;
		validateFilterParams(requiredFilterParamNames, resourceName);
		this.employeeIds = loadEmployeeIds();
	}

	public Collection<ID> getEmployeeIds() {
		return employeeIds;
	}

	/**
	 * Returns a set of employee IDs using the following rules:
	 * If an "employeeIds" parameter is present, it parses the value as a comma delimited list of IDs and returns those.
	 * Otherwise, if an "employeeFilterName" parameter is present, it will parse the filter name and return a list of
	 * employee IDs associated with that filter.
	 * If neither are present, an empty collection is returned.
	 */

	public Date getStartDate() {
		Optional<Date> startDate = DateFilterParamHandler.getFilterValue(filterParams.get(START_DATE_DEFAULT_FILTER_KEY));
		if (startDate.isPresent()) {
			return startDate.get();
		}
		return null;
	}

	public Date getEndDate() {
		Optional<Date> endDate = DateFilterParamHandler.getFilterValue(filterParams.get(END_DATE_DEFAULT_FILTER_KEY));
		if (endDate.isPresent()) {
			return endDate.get();
		}
		return null;
	}

	public Optional<Collection<ID>> getEmployeeIdsParamValue() {
		return EmployeeIdsFilterParamHandler.getFilterValue(filterParams.get(EMPLOYEE_IDS_FILTER_KEY));
	}

	public Optional<String> getEmployeeFilterNameParamValue() {
		return EmployeeFilterNameFilterParamHandler.getFilterValue(filterParams.get(EMPLOYEE_FILTER_NAME_FILTER_KEY));
	}

	/**
	 * Checks that all required filter parameters are present, otherwise throws a {@link RequiredFilterNotPresentException}
	 * Required params: startDate, endDate, and one of either employeeIds or employeeFilterName
	 */
	private void validateFilterParams(Set<String> requiredFilterParamNames, String resourceName) {
		if ((!getEmployeeFilterNameParamValue().isPresent() && !getEmployeeIdsParamValue().isPresent()) ||
				getStartDate() == null || getEndDate() == null)  {
			throw new RequiredFilterNotPresentException(resourceName, Sets.newHashSet(START_DATE_DEFAULT_FILTER_KEY,
					END_DATE_DEFAULT_FILTER_KEY, "(" + EMPLOYEE_IDS_FILTER_KEY + " OR " + EMPLOYEE_FILTER_NAME_FILTER_KEY +
							")"), requiredFilterParamNames);
		}
	}
}
