package com.bluepumpkin.ejb.bbm.schedule.util.optimizeBreaks;

import java.util.*;
import com.bluepumpkin.common.datatypes.Pair;

public class Shift
{

	public Shift(Scenario scenario, String id, String name, Activity activity, Date start, TimeSpan duration, ArrayList<ShiftEvent> shiftEvents, Pair<TimeSpan, TimeSpan> shiftEventSpacing)
	{
		this.scenario = scenario;
		this.id = id;
		this.name = "Shift-" + name;
		this.activity = activity;
		this.duration = scenario.TimeSpanToInterval(duration);
		this.start = start;
		this.shiftEvents = shiftEvents;
		this.shiftEventSpacing = scenario.TimeSpanToInterval(shiftEventSpacing);
		this.maxStates = 1000;
	}

	public final boolean ScheduleShiftEvents()
	{
		Logger.Log("Scheduling work type shift events");
		Logger.Tab();
		ArrayList<ShiftEvent.Assignment> nonWorkAssignments = new ArrayList<ShiftEvent.Assignment>();
		for (int n = 0; n < 1; n++)
		{
			Logger.Log("Pass " + n);
			Logger.Tab();
			for (ShiftEvent shiftEvent : shiftEvents)
			{
				if (!shiftEvent.isWork)
				{
					nonWorkAssignments.addAll(shiftEvent.assignments);
					continue;
				}
				Collections.shuffle(shiftEvent.assignments);				
				for (ShiftEvent.Assignment a : shiftEvent.assignments)
				{
					a.Schedule();
				}
			}
			Logger.Untab();
		}

        bestScore = Double.MAX_VALUE;
        maxStates = 1000;
        statesEvaluated = 0;
		Logger.Log("placeShiftEventAssignmentsUsingDFS over non-work type shift events");
		Logger.Tab();
		assignmentsForDFS = nonWorkAssignments;
        placeShiftEventAssignmentsUsingDFS(assignmentsForDFS);
        for (ShiftEvent.Assignment a : assignmentsForDFS)
        	a.BindBest();
		Logger.Untab();
		Logger.Untab();
		return true;
	}
	public final void Initialize()
	{
		Logger.Log("Initializing shift");
		Logger.Tab();
		Logger.Log("Initializing shift events");
		Logger.Tab();
		for (ShiftEvent shiftEvent : shiftEvents)
		{
			shiftEvent.Initialize(this);
		}
		Logger.Untab();
		Logger.Log("Allocate minSpacing elbow room for shift boundaries");
		Logger.Tab();
		int elbowRoom = shiftEventSpacing.getFirst() / 2;
		int start = scenario.DateTimeToInterval(scenario.start);
		scenario.agent.SetAvailability(1, start, start + elbowRoom, false);
		int end = scenario.DateTimeToInterval(scenario.end);
		scenario.agent.SetAvailability(1, end - elbowRoom, end, false);
		Logger.Untab();
		Logger.Log("Score initial spacing of shift without breaks");
		Logger.Tab();
		int maxSpacing = shiftEventSpacing.getSecond();
		double score = Math.pow(Math.max(0, scenario.numIntervals - maxSpacing), 2);
		Logger.Log("Binding " + score + " to " + " max spacing (" + scenario.IntervalToTimeSpan(maxSpacing) + ") score: Spacing = " + scenario.IntervalToTimeSpan(duration));
		Logger.Tab();
		scenario.objectiveFunction.BindConstraint(score);
		Logger.Untab();
//		scenario.agent.AllocAvailability(0, start, end, false);
		Logger.Untab();
	}
	public final boolean LoadShiftEvent(ShiftEvent shiftEvent, Activity activity, Date starttime)
	{
		return shiftEvent.Load(this, activity, starttime);
	}

	public boolean placeShiftEventAssignmentsUsingDFS(ArrayList<ShiftEvent.Assignment> assignments)
	{
		ArrayList<ShiftEvent.Assignment> a1 = new ArrayList<ShiftEvent.Assignment>(assignments);
		double originalScore = scenario.objectiveFunction.GetScore();
		Logger.Log("placeShiftEventAssignmentsUsingDFS(" + assignments.size() + ") ObjectiveFunction = " + scenario.objectiveFunction.toString());
		Collections.shuffle(assignments);
		int nonFlexibleEventIndex = 0;
		for (ShiftEvent.Assignment assignment: assignments) {
			if (assignment.getMinCount() > 0) {
				break;
			}
			nonFlexibleEventIndex++;
		}
		if (nonFlexibleEventIndex > 0 && nonFlexibleEventIndex < assignments.size()) {
			Collections.swap(assignments, 0, nonFlexibleEventIndex);
		}

		for (ShiftEvent.Assignment assignment : assignments)
		{
			Logger.Log("Explore(" + assignments.size() + ")..." + assignment.toString());
			Logger.Tab();
			a1.remove(assignment);
			for (ShiftEvent.Binding binding : assignment.domain)
			{
				String reason = assignment.IsConsistent(binding);
				if (reason != null)
				{
					Logger.Log(binding + " is inconsistent because -- " + reason);
					continue;
				}
				Logger.Log("Trying(" + assignments.size() + ")..." + binding);
				Logger.Tab();
				assignment.Bind(binding);
				Logger.Tab();
				boolean done = placeShiftEventAssignmentsUsingDFS(a1);
				Logger.Untab();
				assignment.Unbind(binding);
				if (scenario.objectiveFunction.GetScore() != originalScore)
					Logger.Log("Reversability Error: ObjectiveFunction = " + scenario.objectiveFunction.toString()); 
				Logger.Untab();
				if (done)
				{
					Logger.Untab();
					return done;
				}
			}
			a1.add(assignment);
			Logger.Untab();

		}
		double score = scenario.objectiveFunction.GetScore();
		Logger.Log("Consider... current " + scenario.objectiveFunction.toString() + " .vs. best score = " + bestScore);
		if (score < bestScore && assignments.isEmpty())
		{
			SaveAsBest();
			bestScore = score;
		}
		return statesEvaluated++ >= maxStates;
	}
	public final void SaveAsBest()
	{
		Logger.Log("Saving state as best: Score = " + scenario.objectiveFunction.GetScore());
		Logger.Tab();
        for (ShiftEvent.Assignment a : assignmentsForDFS)
        	a.SaveAsBest();
		Logger.Untab();
	}
	public final void Log()
	{
		Logger.Log(name + ": Activity = " + activity.name + ", Start = " + start + ", End = " + scenario.IntervalToTimeSpan(duration).addTo(start) + ", ShiftEventSpacing = " + scenario.IntervalPairToTimeSpan(shiftEventSpacing));
		Logger.Tab();
		Logger.Log("Shift Events");
		Logger.Tab();
		for (ShiftEvent shiftEvent : shiftEvents)
		{
			shiftEvent.Log();
		}
		Logger.Untab();
		Logger.Untab();

	}

	private Scenario scenario;
	public String name;
	public String id;
	public Activity activity;
	private Date start;
	private int duration;
	private int statesEvaluated;
	private int maxStates;
	ArrayList<ShiftEvent.Assignment> assignmentsForDFS;	
	public ArrayList<ShiftEvent> shiftEvents;
	public Pair<Integer, Integer> shiftEventSpacing;
	// Algorithm
	private double bestScore;
}
