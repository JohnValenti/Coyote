package com.bluepumpkin.ejb.bbm.schedule.util.optimizeBreaks;
import java.util.*;

public class Queue
{
    private static class Interval
    {

        public Interval(Scenario scenario)
        {
            this.scenario = scenario;
            this.netStaffingFTE = 0;
        }
        public final void BindStaffingFTE(double d)
        {
            scenario.objectiveFunction.BindNetStaffingFTE(-Math.pow(netStaffingFTE, 2));
            netStaffingFTE += d;
            scenario.objectiveFunction.BindNetStaffingFTE(Math.pow(netStaffingFTE, 2));
        }
        public double netStaffingFTE;
        private Scenario scenario;
    }
    public Queue(Scenario scenario, String id, String name, double proficiency)
    {
        this.scenario = scenario;
        this.id = id;
        this.name = "Queue-" + name;
        this.proficiency = proficiency;
        this.intervals = new Interval[scenario.numIntervals];
        for (int n = 0; n < this.intervals.length; n++)
            this.intervals[n] = new Interval(scenario);

    }
    public final void SetNetStaffing(Date start, Date end, TimeSpan netStaffing)
    {
        Date startOriginal = start;
        Date endOriginal = end;
        if (start.before(scenario.start))
            start = scenario.start;
        if (end.before(scenario.start))
            return;
        if (start.after(scenario.end))
            return;
        if (end.after(scenario.end))
            end = scenario.end;
        double scale = (double) new TimeSpan(start, end).TotalMinutes / (double) new TimeSpan(startOriginal, endOriginal).TotalMinutes;
        if (scale != 1.0)
            netStaffing.TotalMinutes *= scale;
        Logger.Log("Set net staffing on: " + name + " from " + start + " to " + end + " -- to " + netStaffing);
        Logger.Tab();
        for (int i = scenario.DateTimeToInterval(start); i < scenario.DateTimeToInterval(end); i++)
            intervals[i].BindStaffingFTE(netStaffing.TotalMinutes / new TimeSpan(start, end).TotalMinutes);
        Logger.Untab();
    }

    public final void BindStaffingFTE(int start, int end, double magnitude)
    {
        Date dateStart = scenario.IntervalToDateTime(start);
        Logger.Log((magnitude >= 0 ? "Bind " : "Unbind ") + name + " from " + dateStart + " to " + scenario.IntervalToDateTime(end) + " by " + magnitude);
        Logger.Tab();
        LogNetStaffing("before", start, end);
        for (int i = start; i < end; i++)
        {
            Logger.Log("At " + scenario.IntervalToDateTime(i) + ": Binding StaffingFTE = " + magnitude);
            Logger.Tab();
            intervals[i].BindStaffingFTE(magnitude);
            Logger.Untab();
        }
        LogNetStaffing("after", start, end);
        Logger.Untab();
    }
    public final void Log()
    {
        Logger.Log(name);
        Logger.Tab();
        LogNetStaffing("", 0, intervals.length);
        Logger.Untab();
    }
    private final void LogNetStaffing(String label, int start, int end)
    {
        int startInterval = start;
        double totalNetStaffingFTE = intervals[start].netStaffingFTE;
        for (int i = start + 1; i < end; i++)
        {
            if (intervals[i].netStaffingFTE != intervals[i - 1].netStaffingFTE)
            {
                Logger.Log(scenario.IntervalToDateTime(startInterval) + " - " + scenario.IntervalToDateTime(i - 1) + ": NetStaffing(" + label + ") = " + new TimeSpan(0, (int)totalNetStaffingFTE, 0));
                startInterval = i;
                totalNetStaffingFTE = 0;
            }
            totalNetStaffingFTE += intervals[i].netStaffingFTE;

        }
        Logger.Log(scenario.IntervalToDateTime(startInterval) + " - " + scenario.IntervalToDateTime(end - 1) + ": NetStaffing(" + label + ") = " + new TimeSpan(0, (int)totalNetStaffingFTE, 0));
    }
    private Scenario scenario;
    public String name;
    public String id;
    public double proficiency;
    private Interval[] intervals; //  for each interval within the optimization window
}
