package com.bluepumpkin.ejb.rm.base;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.List;

public class RmAggregateException extends RuntimeException {

	private static final long serialVersionUID = 1L;
	List<Exception> exceptions; // NOSONAR

	public RmAggregateException() {
		//Default
	}


	public Collection<Exception> getExceptions() {
		return exceptions == null ? Collections.<Exception> emptyList() : exceptions;
	}

	public void addException(Exception e) {
		if (exceptions == null) {
			exceptions = new ArrayList<Exception>();
		}
		exceptions.add(e);
	}

	public boolean hasExceptions() {
		return exceptions != null;
	}

}
