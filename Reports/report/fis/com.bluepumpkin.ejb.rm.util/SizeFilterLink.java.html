package com.bluepumpkin.ejb.rm.util;

import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.BiddableSchedule;

public class SizeFilterLink extends RmFilterChain
{
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	final private static int ANY = 0;
	final private static int MIN = 1;
	final private static int MAX = 2;
	final private static int RANGE = 3;

	private int lengthType = UNDEFINED;
	private int minLength;
	private int maxLength;
	private String m_ColName; 
	private int m_colPos = -1;
	
	public void setMin( int val )
	{
		this.lengthType = MIN;
		this.minLength =  val;
	}
	
	public void setMax( int val )
	{
		this.lengthType = MAX;
		this.maxLength = val;
	}
	
	public void setRange( int min, int max )
	{
		this.lengthType = RANGE;
		this.minLength = min;
		this.maxLength = max;
	}
	
	public void setEqual( int val )
	{
		this.lengthType = ANY;
		this.minLength = val;
	}

	public void setColumnName( String name )
	{
		this.m_ColName = name;
	}

	public void setColumnPosition( int pos )
	{
		this.m_colPos = pos;
	}

	public boolean isValidSize( int len )
	{
		  switch( this.lengthType )
		  {
		  	case RANGE:
		  		if( len >= this.minLength && len <= this.maxLength )
		  			return true;
		  		break;
		  		
		  	case MAX:
		  		if( len <= this.maxLength )
		  			return true;
		  		break;

		  	case MIN:
		  		if( len >= this.minLength )
		  			return true;
		  		break;

		  	case ANY:
		  		if( len == this.minLength )
		  			return true;
		  		break;
		  }

		  return false;
	}
	
	protected boolean isScheduleValid(BiddableSchedule sched)
	{
		return true;
	}
		  
}
