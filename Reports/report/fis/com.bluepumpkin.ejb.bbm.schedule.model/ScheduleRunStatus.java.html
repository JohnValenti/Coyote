package com.bluepumpkin.ejb.bbm.schedule.model;

/**
 * Title:        Basic Business Model
 * Description:  Value object that models the status for each scheduling period run
 * Copyright:    Copyright (c) 2013
 * Company:      Verint Systems, inc
 * @author       Truc Pham
 * @version 1.0
 */
import java.util.Date;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class ScheduleRunStatus extends ValueObjectBase {
	static final long serialVersionUID = -1410362669689034873L;

	private static FieldInfo m_fieldInfo = new ScheduleRunStatusFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public static int STATUS_START = 1;
	public static int STATUS_END_OK = 2;
	public static int STATUS_START_PENDING = 3;
	public static int STATUS_END_FAIL = -1;
	public static int STATUS_END_CANCEL = -2;

	public int getRunStatus() {
		return getFieldValueInt(ScheduleRunStatusFields.RUNSTATUS);
	}

	public Date getModifiedDate() {
		return getFieldValueDate(ScheduleRunStatusFields.MODIFIED);
	}

	public String getModifiedBy() {
		return getFormattedFieldValue(ScheduleRunStatusFields.MODIFIEDBY);
	}

	public void setRunStatus(int status) {
		setFieldValue(ScheduleRunStatusFields.RUNSTATUS, status);
	}

	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(ScheduleRunStatusFields.MODIFIED, modifiedDate);
	}

	public void setModifiedBy(String modifiedBy) {
		setFieldValue(ScheduleRunStatusFields.MODIFIEDBY, modifiedBy);
	}

	public static int getStatusByString(String strStatus) {
		int status = STATUS_END_FAIL;
		if (strStatus.equals("OK")) {
			status = STATUS_END_OK;
		} else if (strStatus.equals("CANCEL")) {
			status = STATUS_END_CANCEL;
		} else if (strStatus.equals("FAIL")) {
			status = STATUS_END_FAIL;
		} else if (strStatus.equals("PENDING")) {
			status = STATUS_START_PENDING;
		}
		return status;
	}
}
