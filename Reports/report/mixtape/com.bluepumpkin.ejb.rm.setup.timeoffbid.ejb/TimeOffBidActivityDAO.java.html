package com.bluepumpkin.ejb.rm.setup.timeoffbid.ejb;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBidActivity;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBidActivityFieldInfo;

/**
 * DAO class for TimeOffBidActivityDAO
 */
public class TimeOffBidActivityDAO extends DAOBase<TimeOffBidActivity> {

	private static FieldInfo fieldInfo = new TimeOffBidActivityFieldInfo();

	private static Category LOG = Log.initCategory(TimeOffBidActivity.class.getName());

	/**
	 * Constructor
	 */
	public TimeOffBidActivityDAO() {
		super();
	}

	/**
	 * Constructor
	 *
	 * @param - Jdmo - Jdmo object instance
	 * @see Jdmo.
	 */
	public TimeOffBidActivityDAO(Jdmo dmo) {
		super(dmo);
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return fieldInfo;
	}

	protected Category getCategory() {
		return LOG;
	}

	/**
	 * Creates the value object.
	 *
	 * @return - returns new Value object
	 * @see com.bluepumpkin.ejb.bbm.vo.ValueObjectBase
	 * @see com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBid
	 */
	@Override
	protected TimeOffBidActivity createValueObject() {
		return new TimeOffBidActivity();
	}

	/**
	 * Returns a collection of activityId by Bid IDs 
	 * 
	 * @param bidId
	 * @return Collection<ID>
	 */
	public Collection<ID>  getActivityIDsByBidId(ID bidId) throws Exception {
		if(bidId==null){
			return Collections.emptyList();
		}
		Collection<ID> activityIds = new ArrayList<ID>();
		JdmoRowset rs = null;
		try {
			String pStmt = String.format("SELECT distinct ACTIVITYID from TIMEOFFBIDACTIVITY where TIMEOFFBIDID=%d",bidId.toInt());
			rs = m_dmo.createRowset(pStmt);
			while (rs.next()) {
				ID activityId = rs.getID("ACTIVITYID");
				activityIds.add(activityId);
			}
		} finally {
			if (rs != null) {
				rs.close();
			}
			
		}
		return activityIds;
		
	}
	/**
	 * Returns a hashmap which contains Bid ActivityID and ActivityName of orgIDs that logged user can see. 
	 * 
	 * @param orgIDs
	 * @return HashMap<ID,String>
	 */
	public Map<ID,String>  getActivityIDNamesInBidsByOrgIds(Collection<ID> orgIDs) throws Exception {
		if(orgIDs==null|| orgIDs.isEmpty()){
			return null;
		}
		Map<ID,String> hashNames = new HashMap<ID,String>();
		JdmoRowset rs = null;
		try {
			StringBuilder pStmt = new StringBuilder();
			pStmt = new StringBuilder();
			pStmt.append("SELECT distinct A.ID, A.NAME from ACTIVITY A"
					+ " INNER JOIN TIMEOFFBIDACTIVITY TBA on A.ID=TBA.ACTIVITYID"
					+ " INNER JOIN TIMEOFFBID TB ON TBA.TIMEOFFBIDID=TB.ID"
					+ " WHERE TB.ORGANIZATIONID in ");
			pStmt.append(m_dmo.createInClause(orgIDs));
			rs = m_dmo.createRowset(pStmt.toString());
			while (rs.next()) {
				ID activityId = rs.getID("ID");
				String activityName = rs.getString("NAME");
				hashNames.put(activityId,activityName);
			}
			
		} finally {
			if (rs != null) {
				rs.close();
			}
			
		}
		return hashNames;
		
	}
}
