package com.bluepumpkin.web.bbm.people.staffingprofiles;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Map;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.ScopeID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.CalendarPeriod;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.ejb.core.security.Scope;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.MultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.InsertableMultiColListPC;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;

public class AssignmentRuleListPC extends InsertableMultiColListPC {
	private static final long serialVersionUID = 1L;

	// calendar period
	public final int UNIT_TYPE_DAYS = 0;
	public final int UNIT_TYPE_WEEKS = 1;
	public final int UNIT_TYPE_MONTHS = 2;
	public final int UNIT_TYPE_YEARS = 3;
	public final int UNIT_TYPE_ROLLING_WEEKS = 4;

	// work unit constraint constants (matching the F&S client enumeration)
	public final int WORK_UNIT_AT_LEAST = 0;
	public final int WORK_UNIT_NO_MORE_THAN = 1;
	public final int WORK_UNIT_BETWEEN = 2;
	public final int WORK_UNIT_FAIRLY = 3;
	public final int WORK_UNIT_EXACTLY = 4;
	public final int WORK_UNIT_AUTOROTATION = 5;
	public final int WORK_UNIT_ROTATION = 6;
	public final int WORK_UNIT_MUST_START = 7;
	public final int WORK_UNIT_MUST_END = 8;
	public final int WORK_UNIT_MUST_START_BETWEEN = 9;
	public final int WORK_UNIT_MUST_END_BETWEEN = 10;
	public final int WORK_UNIT_TEAM_START = 11;
	public final int WORK_UNIT_TEAM_START_AND_END = 12;
	public final int WORK_UNIT_TEAM_START_SAME_DAYS = 13;
	public final int WORK_UNIT_TEAM_START_AND_END_SAME_DAYS = 14;
	public final int WORK_UNIT_TEAM_START_END_BREAK_SAME_DAYS = 15;
	public final int WORK_UNIT_TEAM_OVERLAP_EACH_DAY = 16;
	public final int WORK_UNIT_TEAM_OVERLAP_EACH_WORKING_DAY = 17;
	public final int WORK_UNIT_TEAM_OVERLAP_EACH_WEEK = 18;
	public final int WORK_UNIT_NUM_SEATS_AT_MOST = 19;
	public final int WORK_UNIT_NUM_SEATS_AT_LEAST = 20;
	public final int WORK_UNIT_FAIRLY_STARTING_BETWEEN = 21;
	public final int WORK_UNIT_FAIRLY_ENDING_BETWEEN = 22;
	public final int WORK_UNIT_INVALID = 23;

	// WORKUNITCOUNTCONSTRAINT Condition of the rule - 0: at least, 1: no more
	// than, 2: between, 3: a fair number, 4: exactly

	Map<ID, Organization> orgMap = null;
	Map<ID, Campaign> campaignMap = null;
	Map<ID, Activity> activityMap = null;

	public AssignmentRuleListPC(RequestContext context, Collection<ComplexWorkRule> assignmentRules,
			TimeZone optimizationTargetTimezone, Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap,
			Map<ID, Activity> activityMap) {
		super(context);

		this.orgMap = orgMap;
		this.campaignMap = campaignMap;
		this.activityMap = activityMap;

		initRows(assignmentRules);
		initHeader();
	}

	protected void initHeader() {
		TableHeaderPC header = getHeader();
		header.setPartialSortable(true);

		header.addColumnHeaderData("assignmentrule",
				m_localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.FS_ASSIGNMENTRULES), true);
		header.setDefaultSortColumns(Collections.singletonList("assignmentrule"));
	}

	private void initRows(Collection<ComplexWorkRule> rules) {
		Collection<ComplexWorkRule> profileAssignmentRules = getWorkRulesCompatibleWithProfiles(rules);
		Collection<MultiColumnNodeData> rows = super.getListModel();

		for (ComplexWorkRule rule : profileAssignmentRules) {
			if (!rule.getCalendarPeriod().getCalendarUnit().equals(CalendarPeriod.CalendarUnit.Month28Day)
					&& !rule.getCalendarPeriod().getCalendarUnit().equals(CalendarPeriod.CalendarUnit.Month29Day)
					&& !rule.getCalendarPeriod().getCalendarUnit().equals(CalendarPeriod.CalendarUnit.Month30Day)
					&& !rule.getCalendarPeriod().getCalendarUnit().equals(CalendarPeriod.CalendarUnit.Month31Day)) {
				DefaultMultiColumnNodeData mcnd = new DefaultMultiColumnNodeData(rule.getID());
				Organization o = orgMap.get(rule.getOrganizationID());
				mcnd.add(rule.getName());
				rows.add(mcnd);
			}
		}
	}

	protected boolean isAuthorized(ID ordID) {
		return (m_context.getUser().isAuthorized(PrivilegeKeys.CORE_CONFIGUREORGANIZATION,
				new ScopeID(ordID, Scope.ORG_SCOPE)) && (m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES_ID,
				new ScopeID(ordID, Scope.ORG_SCOPE))));
	}

	private Collection<ComplexWorkRule> getWorkRulesCompatibleWithProfiles(Collection<ComplexWorkRule> aRules) {
		Collection<ComplexWorkRule> profileRules = new ArrayList<ComplexWorkRule>();
		for (ComplexWorkRule rule : aRules) {

			ID idCalendarPeriod = rule.getCalendarPeriodID();
			if (idCalendarPeriod == null)
				continue;
			CalendarPeriod cp = rule.getCalendarPeriod();
			int maxSpan = getMaxSpan(cp);
			if (maxSpan > 7)
				continue;

			int workUnitConstraint = rule.getWorkUnitCountConstraint().getDBValue();
			if (workUnitConstraint == WORK_UNIT_FAIRLY || workUnitConstraint == WORK_UNIT_FAIRLY_STARTING_BETWEEN
					|| workUnitConstraint == WORK_UNIT_FAIRLY_ENDING_BETWEEN)
				continue;
			profileRules.add(rule);

		}
		return profileRules;
	}

	private int getMaxSpan(CalendarPeriod p) {
		int maxSpanDays = 7;
		int unit = p.getUnitCount();
		CalendarPeriod.CalendarUnit calendarUnit = p.getCalendarUnit();

		switch (calendarUnit) {
			case Day:
				maxSpanDays = unit;
				break;
			case Week:
			case RollingWeek:
				maxSpanDays = unit * 7;
				break;
			case Month:
				maxSpanDays = (int) ((double) unit * 30.5);
				break;
			case Year:
				maxSpanDays = unit * 365;
				break;
			default:
				break;
		}

		return maxSpanDays;
	}
}
