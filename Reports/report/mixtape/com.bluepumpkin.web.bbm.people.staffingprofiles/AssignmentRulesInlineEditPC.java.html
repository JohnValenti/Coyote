	@Override

			js.append("\n").append(getName()).append(".addButtonEnableAttribute('isDeleteable', 'removeAssignmentRule');\n");

	}

	private boolean isRowsInited = false;
	private void initRows() {
		if (isRowsInited) return;
		isRowsInited = true;
		
		Collection<MultiColumnNodeData> rows = super.getListModel();

		try {
			ArrayList<ID> list = new ArrayList<ID>();
			list.addAll(m_ComplexWorkRules.keySet());
			Collection<ComplexWorkRule> rules =  WfmManagerFactory.getWorkRuleManager().getComplexWorkRulesByIDs(list);
			int i = 0;
			for (ComplexWorkRule rule : rules ){
				DefaultMultiColumnNodeData nodeData = getNodeData(rule, m_ComplexWorkRules.get(rule.getID()), i++);
				rows.add(nodeData);
			}
		}
		catch (Exception e){
			log.debug("cant get WorkRule Manager");
		}
	}
	
	protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, ComplexWorkRule rule) {
		if (!rule.isMultiEditCommon()) mcnd.setNodeAttribute("notMultiEditCommon", String.valueOf(true));
		mcnd.setNodeAttribute(WorkpaneMultiColPM.IS_DELETEABLE, String.valueOf(true));
	}
	
	/**
	 * To be called by the request handler that dynamically returns a html row via ajax.
	 * @param VTOEvent
	 * @return
	 */
	public String getRowHtml(ComplexWorkRule rule) {
		StringBuffer html = new StringBuffer();
		super.appendHtmlRowData(html, getNodeData(rule, null, 0), 0, 0);
		return html.toString();
	}
	
	protected DefaultMultiColumnNodeData getNodeData(ComplexWorkRule rule, HashSet<EmployeeTemplate> employeeTemplates, int index) {
		DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(rule.getID());
		
		if (employeeTemplates != null) {
			boolean isMultiEdit = (employeeTemplates.size() == m_empTemplates.size());
			if (isMultiEdit == false) {
				nodeData.add(this.MULTI_VALUE_DATA + rule.getName());
			}
			else
				nodeData.add(rule.getName());
		}
		else 
			nodeData.add(rule.getName());
		
		setRowAttributes(nodeData, rule);
		return nodeData;
	}
