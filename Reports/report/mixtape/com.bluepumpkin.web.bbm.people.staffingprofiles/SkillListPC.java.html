		try {

	protected void initHeader() {

	private void initRows(Collection<Skill> skills) {
		Collection<MultiColumnNodeData> rows = super.getListModel();
		if (skills != null){
			for (Skill skill : skills) {
//				String detailRowID = getName() + "__detail" + getJSSuffix(skill);
				DefaultMultiColumnNodeData mcnd = new DefaultMultiColumnNodeData(skill.getID());
				// Name
				mcnd.add(skill.getName());
				//skill media
				//TODO: put media name
				String medianame = "";
				for (Iterator iterator = m_mediaStringPairs.iterator(); iterator.hasNext();) {
					StringsPair pair = (StringsPair) iterator.next();
					if (pair.getKey().trim().equalsIgnoreCase(skill.getMediaID().toString().trim()))
						medianame = pair.getValue();
				}
				mcnd.add(medianame);
				
				//skill description
				mcnd.add(skill.getDescription());
				
				//skill org
				mcnd.add(skill.getOrganizationName());
				rows.add(mcnd);	
			}
		} 
	}

	public LinkedHashMap<DBSortField, Comparator> getSortComparators() {
		LinkedHashMap<DBSortField, Comparator> comparators = new LinkedHashMap<DBSortField, Comparator>();
		for (SkillAssignmentSortField sortField : getDBSortFieldIndices()) {
			switch (sortField) {
				case ID:
					comparators.put(SkillAssignmentSortField.ID, null);
					break;
				case NAME:
					comparators.put(SkillAssignmentSortField.NAME, null);
					break;
				case MEDIA:
					comparators.put(SkillAssignmentSortField.MEDIA, null);
					break;
				case DESCRIPTION:
					comparators.put(SkillAssignmentSortField.DESCRIPTION, null);
					break;
				case ORGANIZATION:
					comparators.put(SkillAssignmentSortField.ORGANIZATION, null);
					break;
			}
		}
		
		return comparators;
	}
	
	public ArrayList<SkillAssignmentSortField> getDBSortFieldIndices() {
		ArrayList<SkillAssignmentSortField> retVal = new ArrayList<SkillAssignmentSortField>();
		if (getHeader().getSortColumnIndexes() == null) {
			//Default sort index is by ID (chronological order in which shift event was created)
			retVal.add(SkillAssignmentSortField.ID);
		} else {
			for (int sortField : getHeader().getSortColumnIndexes()) {
				switch (sortField) {
					case 1: 
						retVal.add(SkillAssignmentSortField.NAME);
						break;
					case 2: 
						retVal.add(SkillAssignmentSortField.MEDIA);
						break;
					case 3: 
						retVal.add(SkillAssignmentSortField.DESCRIPTION);
						break;
					case 4: 
						retVal.add(SkillAssignmentSortField.ORGANIZATION);
						break;
				}
			}
		}		
		return retVal;
	}
