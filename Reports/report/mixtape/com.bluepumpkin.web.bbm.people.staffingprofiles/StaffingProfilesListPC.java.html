	public StaffingProfilesListPC(RequestContext context, Collection<EmployeeTemplate> empTemplates,
			TimeZone optimizationTargetTimeZone, boolean isByOrg, Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap) {
		super(context);

		this.isByOrg = isByOrg;
		this.orgMap = orgMap;

		m_formInputPC = new FormInputPC(context);
		this.addChildComponent(m_formInputPC);

		initRows(empTemplates);
		initHeader();
	}

	private void initRows(Collection<EmployeeTemplate> empTemplates) {
		Collection<MultiColumnNodeData> rows = super.getListModel();

		WorkRuleManager wrm = null;
		SkillManager skillManager = null;
		Map<ID, ShiftPattern> mapIDShiftPattern = new HashMap<ID, ShiftPattern>();
		Map<ID, ComplexWorkRule> mapIDComplexWorkRule = new HashMap<ID, ComplexWorkRule>();
		Map<ID, Skill> mapIDSkill = new HashMap<ID, Skill>();
		try {
			wrm = WfmManagerFactory.getWorkRuleManager(m_context.isInWhatIfMode());
			skillManager = WfmManagerFactory.getSkillManager(m_context.isInWhatIfMode());
		} catch (Exception e) {
			log.error("An exception occurred when initializing WorkRuleManager or SkillManager for StaffingProfilesListPC", e);
		}
		for (EmployeeTemplate empTemplate : empTemplates) {

			DefaultMultiColumnNodeData rowData = new DefaultMultiColumnNodeData(empTemplate.getID());
			rowData.add(empTemplate.getName());

			String shiftPatternName = "";
			try {
				List<ID> shiftPatterns = empTemplate.getShiftPatterns();
				ID idShiftPattern = shiftPatterns.iterator().next();
				ShiftPattern sp;
				if (idShiftPattern != null) {
					sp = mapIDShiftPattern.get(idShiftPattern);
					if (sp == null) {
						sp = wrm.getShiftPatternsByIDs(Collections.singleton(idShiftPattern)).iterator().next();
						mapIDShiftPattern.put(sp.getID(), sp);
					}
					shiftPatternName = sp.getName();
				}
			} catch (Exception e) {
				log.error("An exception occurred when initializing data for StaffingProfilesListPC", e);
			}

			rowData.add(shiftPatternName);

			ID orgID = empTemplate.getOrganizationID();
			if (orgID != null) {
				Organization o = orgMap.get(empTemplate.getOrganizationID());
				rowData.add(o.getName());
			} else {
				rowData.add("local");
			}

			rowData.add(m_localizer.formatNumber(empTemplate.getWage(), 2));
			double iProficiency = empTemplate.getProficiency();
			rowData.add(m_localizer.formatNumber(iProficiency, 1));

			int iChatSession = empTemplate.getChatSessions();
			rowData.add(String.valueOf(iChatSession));

			List<ID> ids = empTemplate.getAssignmentRuleIDs();
			String rules = "";
			try {
				//check cache
				List<ID> missCache = new ArrayList<ID>(ids.size());
				for(ID id : ids) {
					if (!mapIDComplexWorkRule.containsKey(id)) {
						missCache.add(id);
					}
				}
				if (!missCache.isEmpty()) {
					Collection<ComplexWorkRule> wrs = wrm.getComplexWorkRulesByIDs(missCache);
					for (ComplexWorkRule rule : wrs) {
						mapIDComplexWorkRule.put(rule.getID(), rule);
					}
				}
				
				for (ID id : ids) {
					if (!StringUtil.isEmpty(rules)) {
						rules = rules + ",";
					}
					rules = rules + mapIDComplexWorkRule.get(id).getName();
				}
			} catch (Exception e) {
				log.error("An exception occurred when initializing data for StaffingProfilesListPC", e);
			}

			rowData.add(rules);

			List<SkillAssignment> skillAssignments = empTemplate.getSkillAssignments();
			String skills = "";
			for (SkillAssignment skillAssignment : skillAssignments) {
				ID id = skillAssignment.getSkillID();
				try {
					Skill skill = mapIDSkill.get(id);
					if (skill == null) {
						skill = skillManager.getSkillByID(id);
						mapIDSkill.put(id, skill);
					}
					if (!StringUtil.isEmpty(skills)) {
						skills = skills + ",";
					}
					skills = skills + skill.getName();
				} catch (Exception e) {
					log.error("An exception occurred when initializing data for StaffingProfilesListPC", e);
				}

			}
			rowData.add(skills);

			if (!isByOrg) {
				rowData.add("0 %");
				rowData.add(getMaxAgentRatioUI("test", 100, true));
				rowData.add("0");
				rowData.add(getMaxAgentsUI("test", 9999, true));
			}
			rows.add(rowData);
			setRowAttributes(rowData, empTemplate);

		}
	}

	protected void setRowAttributes(DefaultMultiColumnNodeData rowData, EmployeeTemplate shiftPattern) {
	}

	protected NumberWrapper getMaxAgentsUI(String rowID, int agents, boolean isEditable) {
		String sAgents = "";

		if (isEditable) {
			m_formInputPC.reset();
			m_formInputPC.setSize(5);
			m_formInputPC.setInputType(FormInputPC.TYPE_INTEGER);
			m_formInputPC.setInputValue(String.valueOf(agents));
			m_formInputPC.setLowerLimit(0);
			m_formInputPC.initForDisplay();
			sAgents = m_formInputPC.getHtmlDisplay();
		} else {
			sAgents = m_localizer.formatNumber(agents);
		}
		return new NumberWrapper(agents, sAgents);
	}

	protected NumberWrapper getMaxAgentRatioUI(String rowID, int agents, boolean isEditable) {
		String sAgents = "";

		if (isEditable) {
			m_formInputPC.reset();
			m_formInputPC.setSize(5);
			m_formInputPC.setInputType(FormInputPC.TYPE_INTEGER);
			m_formInputPC.setInputValue(String.valueOf(agents));
			m_formInputPC.setLowerLimit(0);
			m_formInputPC.initForDisplay();

			String percentString = " %";
			sAgents = m_formInputPC.getHtmlDisplay() + percentString;

		} else {
			sAgents = m_localizer.formatNumber(agents);
		}
		return new NumberWrapper(agents, sAgents);
	}
}
