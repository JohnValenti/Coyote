package com.verint.web.fs.people.selection;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.employee.EmployeeListPageModel;
import com.bluepumpkin.web.bbm.employee.EmployeeListRequestHandler;
import com.bluepumpkin.web.bbm.l10n.BbmWebLogBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.pagemodel.WorkpanePageModel;
import com.witness.web.uif.system.RequestContext;

public class CampaignSPPeopleSummaryRH extends EmployeeListRequestHandler {
	private static Category log = Log.initCategory(CampaignSPPeopleSummaryRH.class.getName());

	/** Process the browser request.
	 *	This method will determine which button was pressed on the job title form,
	 *	add, delete, or show detail, and call a method to perform that action.
	 */
	public void processCustomRequest(RequestContext context) {
		String pageAction = context.getRequest().getParameter("pageAction");
		String sResult = "refresh";
		if( pageAction != null && !pageAction.equals("")) {
			log.l7dDebug(BbmWebLogBundleKey.EMP_LIST_PAGE_ACTION_TO_PERFORM_MF, new Object[]{pageAction});
			if (pageAction.equals(EmployeeListPageModel.CHANGE_VIEW)) {
				sResult = processChangeView();
			} else if (isAction(context, "REMOVE_EMP_FROM_SP")) {
				processRemoveEmpFromSP(context);
			}

			setNextPage( context, sResult);
		}
	}

	/**
	 * Return Page Model to perform generic action request on
	 */
	public WorkpanePageModel getPageModel(RequestContext context) {
		CampaignSPPeopleSummaryPM model = new CampaignSPPeopleSummaryPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		return model;
	}
	
	private void processRemoveEmpFromSP(RequestContext context) {
		CampaignSPPeopleSummaryPM model = (CampaignSPPeopleSummaryPM)getPageModel(context);
		try {
			model.removeSelectedEmployeesFromSchedulingPeriod();	
		} catch (Exception ex) {
			model.addPageMessage(Message.ERROR_TYPE, context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_ERROR_REMOVING_EMP_FROM_SP));
		}
	}
}
