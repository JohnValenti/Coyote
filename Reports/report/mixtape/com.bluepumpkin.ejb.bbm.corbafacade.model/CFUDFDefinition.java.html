package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.workload.model.queueattribute.QueueAttributeDefintion;
import com.bluepumpkin.ejb.bbm.workload.model.queueattribute.QueueAttributeDropDown;

public class CFUDFDefinition implements Serializable {
		/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
		public ID id;	
		public String name;
		public ID orgId;
		public int type;
		public String[] dropdowns = new String[0];
	
	  public static CFUDFDefinition convert(QueueAttributeDefintion from) {
		    CFUDFDefinition to = new CFUDFDefinition();
	    	
	    	to.id = from.getID();
	    	to.name = from.getName();
	    	to.orgId = from.getOrganizationID();
	    	to.type = from.getType().toInt();
	    	ArrayList<QueueAttributeDropDown> queueDropDowns = from.getQueueAttributeDropDown();
	    	ArrayList<String> list = new ArrayList<String>();
	    	for(int i=0;i<queueDropDowns.size();i++){
	    		list.add(queueDropDowns.get(i).getValue());
	    	}
	    	
	    	if (!list.isEmpty()) {
	    		to.dropdowns = new String[list.size()];
	    		list.toArray(to.dropdowns);
	    	}	
	    	
	    	return to;
	    }
	    
		public static CFUDFDefinition[] convert(Collection from) {
			if (from == null)
				return null;
			
			CFUDFDefinition[] to = new CFUDFDefinition[from.size()];
			int ix = 0;
			for (Iterator i = from.iterator(); i.hasNext(); ix++)
				to[ix] = CFUDFDefinition.convert((QueueAttributeDefintion)i.next());
			return to;
		}

}
