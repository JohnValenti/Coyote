package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Date;

// import EJB data types
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.schedule.model.PublishingPeriod;

/*
* This class is used as a container for the PublishingPeriod value object.
* Only the C++ DE clients would use this class, note that any changes to this
* class would affect the DE cilent.
*
* Use only primitive types for member variables.
* ID, Date, String, Long, Short, Integer are okay
*/

public final class CFPublishingPeriod implements Serializable
{
  /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
public ID workResourceId;
  public Date startTime;
  public Date endTime;
  public boolean isPartialPublish;

  static public  CFPublishingPeriod toObject( PublishingPeriod period )
  {
			CFPublishingPeriod retVal = new CFPublishingPeriod();
			retVal.workResourceId = period.getWorkResourceID();
			retVal.startTime = period.getStartTime();
			retVal.endTime = period.getEndTime();
			retVal.isPartialPublish = period.isPartialPublish();

			return retVal;
  }
}
