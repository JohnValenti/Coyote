package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;
import java.util.Date;

import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeQualityScore;

public class CFEmployeeQualityScore implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public int qualityScore;
	public Date startDate;
	public Date endDate;
	
	public static CFEmployeeQualityScore convert(EmployeeQualityScore from) {
		CFEmployeeQualityScore to = new CFEmployeeQualityScore();
		
		to.qualityScore = from.getQualityScore();
		to.startDate = from.getStartTime();
		to.endDate = from.getEndTime();
		
		return to;
	}
	
	public static CFEmployeeQualityScore[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeQualityScore[] to = new CFEmployeeQualityScore[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeeQualityScore.convert((EmployeeQualityScore)i.next());
		
		return to;
	}
}
