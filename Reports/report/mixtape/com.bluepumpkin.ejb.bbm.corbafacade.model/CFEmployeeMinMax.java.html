package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;

import java.util.Collection;
import java.util.Iterator;
import java.util.Date;

import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeMinMaxMinutes;;


public class CFEmployeeMinMax implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public int minMinutes;
    public int maxMinutes;
    public int maxOTMinsPerDay;
    public int maxOTMinsPerWeek;
    public int maxVTOHoursPerDay;
    public int maxVTOHoursPerWeek;
    public Date startDate;
    public Date endDate;
    
    public static CFEmployeeMinMax convert(EmployeeMinMaxMinutes from) {
    	CFEmployeeMinMax to = new CFEmployeeMinMax();
    	
    	to.minMinutes = from.getMinMinutes();
    	to.maxMinutes = from.getMaxMinutes();
    	to.maxOTMinsPerDay = from.getMaxOTMinsPerDay();
    	to.maxOTMinsPerWeek = from.getMaxOTMinsPerWeek();
    	to.maxVTOHoursPerDay = from.getMaxVTOHoursPerDay();
    	to.maxVTOHoursPerWeek = from.getMaxVTOHoursPerWeek();
    	to.startDate = from.getStartTime();
    	to.endDate = from.getEndTime();
    	
    	return to;
    }
    
	public static CFEmployeeMinMax[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeMinMax[] to = new CFEmployeeMinMax[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeeMinMax.convert((EmployeeMinMaxMinutes)i.next());
		return to;
	}

}
