package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;

import com.bluepumpkin.ejb.bbm.vo.ValueObjectRoot;

public class CFValueObjectRoot implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public String objectVersionNumber;
	public String updateUser;
	
	public static void convert(ValueObjectRoot from, CFValueObjectRoot to) {		
		to.objectVersionNumber = from.getObjectVersionNumber();
		to.updateUser = from.getUpdateUser();
	}
	
	public static void convert(CFValueObjectRoot from, ValueObjectRoot to) {
		to.setObjectVersionNumber(from.objectVersionNumber);
		to.setUpdateUser(from.updateUser);
	}
}
