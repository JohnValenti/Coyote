package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;
import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;

public class CFEmployeeSkillHistory implements Serializable {
	
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID workResourceID = null;
	public CFEmployeeSkill[] skills = null;
	
	public static CFEmployeeSkillHistory[] convert(HashMap from) {
		if (from == null)
			return null;
		
		CFEmployeeSkillHistory[] to = new CFEmployeeSkillHistory[from.size()];
		
		Map.Entry entry = null;
		int ix = 0;
		for (Iterator i = from.entrySet().iterator(); i.hasNext(); ) {
			entry = (Map.Entry)i.next();
			CFEmployeeSkillHistory item = new CFEmployeeSkillHistory();
			item.workResourceID = (ID)entry.getKey();
			item.skills = CFEmployeeSkill.convert(CFUtil.sortDesc((Collection)entry.getValue()));
			to[ix++] = item;
		}
		
		return to;
	}
}
