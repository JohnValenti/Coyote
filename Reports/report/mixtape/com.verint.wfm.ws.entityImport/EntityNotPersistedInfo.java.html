package com.verint.wfm.ws.entityImport;

import java.util.ArrayList;
import java.util.List;

import com.bluepumpkin.common.logging.Category;
import com.verint.wfm.ws.ServiceApplicationException;

/**
 * For each entity not persisted by createOrUpdate~s(), one of these will be in the list returned. This info is returned
 * in lieu of an exception, because we always want to continue iterating
 */
public abstract class EntityNotPersistedInfo<T> {
	private static final long serialVersionUID = 1L;

	final private T m_entityExternalSourceId;
	private boolean m_isAlreadyCreatedButNoUpdate;
	final private List<String> m_messages = new ArrayList<String>();

	/**
	 * Dummy here on server, but needed for SOAP stub
	 */
	public EntityNotPersistedInfo() {
		m_entityExternalSourceId = null;
	}

	protected EntityNotPersistedInfo(T entityExternalSourceId) {
		m_entityExternalSourceId = entityExternalSourceId;
	}

	// ----------------------------------------------------------
	public T getEntityExternalSourceId() { 
		return m_entityExternalSourceId;
	}

	protected String getExternalIdSummary() {
		return (getEntityExternalSourceId() == null) ? null : getEntityExternalSourceId().toString();
	}

	public boolean isAlreadyCreatedButNoUpdate() {
		return m_isAlreadyCreatedButNoUpdate;
	}

	public void setAlreadyCreatedButNoUpdate(boolean isAlreadyCreatedButNoUpdate) {
		m_isAlreadyCreatedButNoUpdate = isAlreadyCreatedButNoUpdate;
	}

	public List<String> getMessages() {
		return m_messages;
	}

	/**
	 * Dummy here on server, but needed for SOAP stub
	 */
	public void setMessages(List<String> messages) {
		// do nothing
	}
	
	public void addServiceExceptionMessage(Exception exception,
			ConsecutiveServiceApplicationExceptionMonitor monitor)
			throws ServiceApplicationException {
		monitor.increment(exception);
		getMessages().add(
				String.format("Exception is %s. The server could not persist entity %s.", exception.getMessage(),
						getExternalIdSummary()));
	}

	public void logAndAddError(Category category, String message) {
		category.error(message);
		getMessages().add(message);
	}
}
