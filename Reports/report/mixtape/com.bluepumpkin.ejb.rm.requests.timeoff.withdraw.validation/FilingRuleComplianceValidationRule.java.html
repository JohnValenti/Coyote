/**
 * Rule in UI: Request complies with filing rules.
 * <p/>
 * Check that the rquest is submitted enough time in advance according to the
 * filing rules configured for the employee's organization.
 * <p/>
 * <p/>
 * Applies equally for debit, debit-if and dont-debit requests
 * <p/>
 * <p/>
 * Applies equally for tentative and non tentative requests.
 */
public class FilingRuleComplianceValidationRule implements Validator {

    private static final String m_className = FilingRuleComplianceValidationRule.class.getName();

    private static final Category m_cat = Log.initCategory(m_className);

    /**
     * Validate a time off request.
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        ValidationResult result = null;
        TORequest tor = (TORequest) validatable;

        String methodName = "validate";
        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));

        // Build a filing rule validation checker for this request
        ID orgID=(ID)tor.getCache().getOrgIDs().iterator().next();
        TOWithdraw toWithdraw = tor.getWithdrawInfo();
        toWithdraw.setValidationTimeRange(tor.getApprovedChoice().getStartDate(), tor.getApprovedChoice().getEndDate());
        FilingRuleChecker ckr =FilingRuleCheckerHelper.getChecker(orgID,
                                                        Request.REQUESTTYPE_TIMEOFF_WITHDRAW,
                                                        null,
                                                        toWithdraw.getValidationTimeRange());
        //tor.getCache().getFilingRuleChecker(null);
        result = setValidationResultForFilingRuleCompliance(tor, toWithdraw, ckr);
        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
        return result;
    }

    public ValidationResult setValidationResultForFilingRuleCompliance(TORequest tor, TOWithdraw toWithdraw, FilingRuleChecker ckr) throws Exception {
        String methodName = "setValidationResultForFilingRuleCompliance";
        ValidationResult result = null;
        Date fileDate = toWithdraw.getCreationDate();
        TimeRange timeRange =toWithdraw.getValidationTimeRange();
        if (!ckr.requestComplies(fileDate, timeRange.getStartDate(), timeRange.getEndDate(),
                timeRange.getStartDate())) {
            // Since sometimes it might be mystifying why a request fails, we log
            // the rule that failed the requst
            RequestFilingRule rule = ckr.getLastFailingRule();
            m_cat.debug("time off Withdraw request id:" + tor.getID() + " failed because " +
                    "toWithdraw id:" + toWithdraw.getID() + " did not pass filing rule with id:" + rule.getID());

            result = ValidationUtil.setSoftValidationResult(toWithdraw,
                    RmEjbBundleKey.APPROVED_TO_WITHDRAW_FAILED_FILING_RULE, tor.getCache().getEmployeeNameByID(),
                    timeRange.getStartDate(), timeRange.getEndDate(), tor.getCache().getOrgNameForMsg(),
                    m_className);
        }
        return result;
    }
}
