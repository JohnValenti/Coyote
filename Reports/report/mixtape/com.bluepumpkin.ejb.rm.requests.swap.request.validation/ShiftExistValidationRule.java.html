/**
 * <p> <b> Hard Validation Rule: </b>
 * Ensure that the shift items, part of the shift swap request,
 * are valid (ie. the shifts themselves still exist in the schdule for the employee
 * that the shiftswap item belongs to).
 *
 * <p>
 * for 1 way swap, ensure that one swap item has a timeoff and the other swap item has a valid shift
 * for 2 way swap, ensure both swap items have valid shifts (in published and  unpublished schedule).
 *
 * <p>
 * For tentatively approved requests, only the published schedule is checked to see if the shift
 * exists.  The unpublished schedule is not.
 */
public class ShiftExistValidationRule implements Validator
{

    private static Category m_cat = Log.initCategory( ShiftExistValidationRule.class.getName());

    private ShiftSwapValidationCache m_vc = null;

    public ShiftExistValidationRule(){};

	//used for unit testing.
	void setShiftSwapValidationCache(ShiftSwapValidationCache cache) {
		m_vc = cache;
	}

    /* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception
    {
//      try {

            String methodName = "validate";
            if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, ssr));

            m_vc = (ShiftSwapValidationCache) ssr.getValidationCache();

            List items = ssr.getShiftSwapItems();

            int itemsSize = items.size();

            ValidationResult result = validateNoOfItems(validatable, itemsSize);
            if ( result != null ) return result;

            ScheduleAccessManager sam = WfmManagerFactory.getScheduleAccessManager();

            if (ssr.isOneWay()) {
                result = validateOneWaySwapItems(validatable, items, itemsSize);
                if ( result != null ) return result;
            }

            result = validateSwapAndTimeOffSSItems(validatable, items, sam);
            if ( result != null ) return result;

            if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName));
            return null;

    private ValidationResult validateNoOfItems(Validatable validatable, int itemsSize)
    {
        // there must be at least two items for a shift swap
        // todo: should this be done in other rules as well?
        if (itemsSize < 2) {
            return setHardValidationResult( validatable, RmEjbBundleKey.SS_SHIFT_EXIST_TOO_FEW_ITEMS,
                new Serializable[] {new Integer(itemsSize)} );
        }

        return null;
    }

	// todo: none of the other rules took take into account shifttype (time-off or shift)
	//   for instance, MinMaxHoursValidationRule assumes shifttype "shift" when calculating Min/Max hours
	//     same for SamePaidHoursValidationRule. SameProficienciesValidatonRule and SameSkillsValidation
	//     rule do account for shifttype "timeoff".  how abou the others?
	// todo: validate for other rules as well: one way swaps must have only 2 items
	//   one being a "time-off" and the other being "shift"
	// todo: explain the rule below?
	/**
	 * if a one-way swap there must be exactly two items. One has
	 * shifttype = "time-off" the other has shifttype="shift"
	 * 
	 * <p>NOTE: if in the future, swaps with more than two items are
	 * supported, the definition of "one-way" will need to be modified,
	 * since there are many combinations of swapping "shifts" with "time-off".
	 * 
	 * @param validatable
	 * @param items
	 * @param itemsSize
	 * @return
	 * @throws Exception
	 */
	private ValidationResult validateOneWaySwapItems(Validatable validatable, List items, int itemsSize)
            throws Exception {
            	
        if (itemsSize != 2) {
            return setHardValidationResult(validatable, RmEjbBundleKey.SS_SHIFT_EXIST_ITEMSIZE_ONEWAY,
                new Serializable[] {new Integer(itemsSize)});
        }

        ShiftSwapItem item1 = (ShiftSwapItem)items.get(0);
        ShiftSwapItem item2 = (ShiftSwapItem)items.get(1);

        // must be different types: "shift" && "time-off"
        if ( item1.getShiftType().equals(item2.getShiftType()) ) {
            return setHardValidationResult(validatable, RmEjbBundleKey.SS_SHIFT_EXIST_ONEWAY,
                new Serializable[] { m_vc.getEmployeeNameByID(item1.getEmployeeID()),
                    item1.getShiftType(), m_vc.getEmployeeNameByID(item2.getEmployeeID()),
                    item2.getShiftType()} );
        }

        return null;
    }

    /**
	 * 
	 * iterate through the list of shiftswap items associated with this request.
	 * For each item check:
	 * <li> if type="shift", check that shift still exists in the published schedule.
	 * <li> if type="shift", check that shift still exists in the unpublished schedule (except for tentatively approved requests).
	 * <li> if type="time-off", check that shift does not exist in published schedule.
     * <li> if type="time-off", check that shift does not exist in unpublished schedule. (except for tentatively approved requests).
     * 
	 * @param validatable
	 * @param items
	 * @param sam
	 * @return
	 * @throws Exception
	 */
	ValidationResult validateSwapAndTimeOffSSItems(Validatable validatable, Collection items,
        ScheduleAccessManager sam) throws Exception {

		ShiftSwapRequest ssReq = (ShiftSwapRequest) validatable;
		boolean isTentative = ssReq.isTentative();
		        	
        for (Iterator itrItems=items.iterator(); itrItems.hasNext(); ) {
            // get the shiftswap item
            ShiftSwapItem item = (ShiftSwapItem)itrItems.next();
            // obtain swapitemtype (timeoff or shift) associated with the SSItem.
            String swapItemType = item.getShiftType();

            // get the published shift assignment associated with SSItem.
            ShiftAssignment pubShift = m_vc.getShiftAssignForSSItem(item, sam, true, !item.getIsPartial() );
			
			// fetch the unpublished shift assignment only for non tentative requests.
			ShiftAssignment unpubShift = null;
			if ( !isTentative ) { 
				unpubShift = m_vc.getShiftAssignForSSItem(item, sam, false, !item.getIsPartial());
			}

			// For ssItem type == SHIFT, an aligned shift must exist for the ssItem's time range.
            if (swapItemType.equals(ShiftSwapItem.SWAPITEMTYPE_SHIFT)) {
                // If no published shift assignment associated with SSItem or SSItem is not aligned with the shift.
                if (item.getIsPartial()){
                	//check only for overlap
                	if (pubShift == null || !TimePeriodUtil.inside(item.getStartDate(), item.getEndDate(),
                			pubShift.getStartTime(), snapTo15(pubShift.getEndTime()))){
                		return setHardValidationResult( validatable, RmEjbBundleKey.SS_PARTIAL_SHIFT_DOES_NOT_EXIST_IN_PUB,
                                new Serializable[] { m_vc.getEmployeeNameByID(item.getEmployeeID()),
                                        item.getStartDate(), item.getEndDate()} );	
                	}
                } else if (pubShift == null || !item.equalsShift(pubShift)) {
	                //We want to also get a shift that aligns with Time-off for a tentatively approved request that needs to be approved.
	                //Sameet Sept 2010 ; QC 57873
	                boolean setApprove = false;
	                if (isTentative) {
		                List itemList = ssReq.getShiftSwapItems();
		                for (Iterator itrItems1 = itemList.iterator(); itrItems1.hasNext();) {
			                ShiftSwapItem item1 = (ShiftSwapItem) itrItems1.next();
			                if (!item1.equals(item)) {
				                pubShift = ShiftSwapRequestUtil.getShiftAssignForSSItem(item1, true, !item1.getIsPartial(), false, sam, ssReq.getCache(), false);
				                if (pubShift != null) {
					                setApprove = true;
					                break;
				                }
			                }
		                }
	                }
	                ValidationResult result = setHardValidationResult(validatable, RmEjbBundleKey.SS_SHIFT_DOES_NOT_EXIST_IN_PUB,
	                        new Serializable[]{m_vc.getEmployeeNameByID(item.getEmployeeID()),
	                                           item.getStartDate(), item.getEndDate()});
	                result.setApprove(setApprove);
	                return result;
                }

				// if not tentatively approvead and (no unpublished shift assignment associated with 
				// SSItem or SSItem is not aligned with the shift)
                if (!isTentative){
                	if (item.getIsPartial()){
                    	//check only for overlap
                    	if (unpubShift == null || !TimePeriodUtil.inside(item.getStartDate(), item.getEndDate(),
                    			unpubShift.getStartTime(), snapTo15(unpubShift.getEndTime()))){
                    		return setHardValidationResult( validatable, RmEjbBundleKey.SS_SHIFT_PARTIAL_DOES_NOT_EXIST_IN_UNPUB,
                                    new Serializable[] { m_vc.getEmployeeNameByID(item.getEmployeeID()),
                                            item.getStartDate(), item.getEndDate()} );			
                    	} 
					} else if ((unpubShift == null || !item.equalsShift(unpubShift)) ) {
						return setHardValidationResult( validatable, RmEjbBundleKey.SS_SHIFT_DOES_NOT_EXIST_IN_UNPUB,
							new Serializable[] { m_vc.getEmployeeNameByID(item.getEmployeeID()),
									item.getStartDate(), item.getEndDate()} );
                	}
				}
			// For ssItem type == TIMEOFF, a shift must not exist during the ssItem's time range.
			// 
			//if SSItem swaptype is timeoff 
            } else if (swapItemType.equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF)) {
				// if a published shift exists.
				if (doesShiftIntersectWithItem(pubShift, item)) {
	                return setHardValidationResult( validatable, RmEjbBundleKey.SS_SHIFT_EXIST_IN_PUB,
	                    new Serializable[]  { m_vc.getEmployeeNameByID(item.getEmployeeID()),
	                                                    item.getStartDate(), item.getEndDate()} );
                }
                
                // if not tentatively approvead and an unpublished shift exists
				if (!isTentative && doesShiftIntersectWithItem(unpubShift, item)) {
					return setHardValidationResult( validatable, RmEjbBundleKey.SS_SHIFT_EXIST_IN_UNPUB,
						new Serializable[]  { m_vc.getEmployeeNameByID(item.getEmployeeID()),
														item.getStartDate(), item.getEndDate()} );
				}
            }
        } // end for(....) loop

        return null;
    } // end method


	private boolean doesShiftIntersectWithItem(ShiftAssignment shift, ShiftSwapItem item) throws Exception {
		if (shift == null) {
			return false;
		}

		return DateUtil.intervalsIntersect(shift.getStartTime(), shift.getEndTime(), item.getStartDate(), item.getEndDate());
	}

	private ValidationResult setHardValidationResult(Validatable validatable, String msgKey, Serializable[] parms)
    {
        return ValidationUtil.setHardValidationResult(validatable, msgKey, parms, this.getClass().getName());
    }
    
    /**
     * If a date is on a 15-min boundary, just return it. Otherwise, return 
     * the next 15-min boundary date. 
     * 5-minute / 1-minute shift workaround:
     * We use this method to allow partial shift-swaps at the end of a shift
     * where the shift doesn't end on a 15-minute boundary. In this case, the
     * request must end on a 15-minute boundary, so the user will enter an end
     * date which is greater than the shift's end date in order to cover the 
     * entire shift. But the validation checks to make sure the request lies 
     * within the shift. In order for the validation to succeed, the shift's 
     * end date must be assumed to be the next 15-minute interval.
     */
    private Date snapTo15(Date date)
    {
        if (date == null)
            return null;
        
        Calendar cal = Calendar.getInstance(TimeZoneUtil.GMT_TIMEZONE);
        cal.setTime(date);
        cal.set(Calendar.SECOND, 0);
        cal.set(Calendar.MILLISECOND, 0);
        int minute = cal.get(Calendar.MINUTE);
        
        if (minute % 15 == 0)
            return date;
        
        if (minute <= 15)
            minute = 15;
        else if (minute <= 30)
            minute = 30;
        else if (minute <= 45)
            minute = 45;
        else
        {
            minute = 0;
            cal.add(Calendar.HOUR_OF_DAY, 1);
        }        
        
        cal.set(Calendar.MINUTE, minute);
        return cal.getTime();
    }
