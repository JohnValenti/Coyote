/**
 *
 * <b>Hard Validation: applies only for tentative requests</b>
 * <p>
 *    swapped shifts still exist in the schedule (unpublished).
 *      for 2 way swap, both swapped shifts are verified.  For 1 way swap, the one swapped shift is verified.
 * <p>
 *    swapped shifts (in unpublished) haven't been modified in any way (duration of shift, position of shift).
 *      for 2 way swap, both swapped shifts are verified.  For 1 way swap, the one swapped shift is verified.
 * <p>
 *    shift locations in the schedule (unpublished) before swap are still empty
 *      (timeoff ok, calendarEvents ok, no unavailabilities or another shift assignment)
 *      for 2 way swap, both orig shifts are verified to not have any shifts for the org day.  For 1 way swap, the one orig
 *      shift is verified to not have any shifts for the org day.
 *      TODO: if shiftassign is put underneath an existing time off event, do we recalculate time off length?
 * <p>
 *    swapped shifts not yet published.
 *      for 2 way swap, both swapped shifts are verified.  For 1 way swap, the one swapped shift is verified.
 *
 */
public class TentativeRequestHardValidationRule implements Validator
{
    private static final String m_className = TentativeRequestHardValidationRule.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);

    /**

     * @param ssr

    protected ValidationResult checkIfSwappedShiftsExist(ShiftSwapRequest ssr, ScheduleAccessManager sam)
        throws Exception
    {
//        try {

    /**
     * verify that no 'events' (unavails and other shift assignments) overlap the
     *  interval represented by the ssItems before the swap.  This ensures that the 'tentative'
     *  request can be undone.
     *
     * @param ssr
     * @param sam
     * @return
     * @throws Exception
     */
    protected ValidationResult checkIfEventsExistDuringOrigShifts(ShiftSwapRequest ssr, ScheduleAccessManager sam)
        throws Exception
    {
//        try {
            ValidationResult result = null;

            // For this shift swap request, get associated shift assignments which start during the orgDay for
            // swapitem.startTime.
            ShiftAssignment[] shiftAssigns = ssr.getCache().getShiftAssignsForReqSameOrgDay(false, sam);
            List ssItemsList = ssr.getShiftSwapItems();

            int ssItemsSize = ssItemsList.size();
            for (int outIdx = 0; outIdx  < ssItemsSize; outIdx ++) {
                // shift being swapped out.
                ShiftSwapItem ssItemOut = (ShiftSwapItem)ssItemsList.get(outIdx );

                // corresponding shift being swapped in.
                int itemInIndex = (outIdx  -1 + ssItemsSize)%ssItemsSize;
                ShiftSwapItem ssItemIn = (ShiftSwapItem) ssItemsList.get(itemInIndex);

                // verify if the shift (if one exists) which starts on the same org day as ssItemOut.startTime
                // is aligned with ssItemIn.
                result = verifyNoShiftExistsOrShiftAligned(ssr, ssItemOut, shiftAssigns[outIdx ], ssItemIn);
                if (result != null) return result;

                // ensure that no unavails overlap the timeRange for ssItemOut.
                Collection unavails = ssr.getCache().getCalEventAssignsForSSItem(ssItemOut,
                    Event.EVENT_TYPE_UNAVAILABILITY, false, sam);
                result = verifyNoUnavailsExist(ssr, ssItemOut, unavails);
                if (result != null) return result;
            }

            return result;
//        } c_atch (Exception e) {
//            t_hrow RequestUtil.createValidationExceptionWrapper(e, m_cat);
//        }
    }

    protected ValidationResult checkIfSwappedShiftsPublished(ShiftSwapRequest ssr, ScheduleAccessManager sam)
        throws Exception
    {
//        try {
            ValidationResult result = null;
            // TODO: An exception thrown if a published shift not in synch with the swapped shift exists.
            // Is this valid behavior?
            ShiftAssignment swappedShiftAssigns[] = ssr.getCache().getSwappedShiftAssignsForReqSameOrgDay(true, sam);
            List ssItemsIn = ssr.getShiftSwapItems();

            int ssItemsOutSize = ssItemsIn.size();
            for (int i=0; i < ssItemsOutSize; i++)
            {
                int itemOutIndex = (i + 1) % ssItemsOutSize;
                ShiftSwapItem ssItemOut = (ShiftSwapItem) ssItemsIn.get(itemOutIndex);

                result = verifySwappedShiftNotPublished(ssr, (ShiftSwapItem)ssItemsIn.get(i),
                    swappedShiftAssigns[i], ssItemOut);
                if (result != null) return result;
            }

            return null;
//        } c_atch (Exception e) {
//            t_hrow RequestUtil.createValidationExceptionWrapper(e, m_cat);
//        }
    }

    /**
     * Is given ssItem aligned with the given shift?  To verify this, check if given ssItem.swapType == 'shift',
     * and given swappedShift is non-null.
     *
     * @param ssr
     * @param ssItem
     * @param swappedShift
     * @return
     */
    protected ValidationResult verifySwappedShiftExists(Validatable ssr, ShiftSwapItem ssItem,
        ShiftAssignment swappedShift)
    {
        ValidationResult result = null;

        if ( ssItem.isSwapTypeShift() && swappedShift == null )
        {
            result = ValidationUtil.setAndLogHardValidationResult(ssr,
								RmEjbBundleKey.SS_SHIFTNOTFOUND_FOR_SSITEM,
								RmEjbLogBundleKey.SS_SHIFTNOTFOUND_FOR_SSITEM,
                new Serializable[] {
									ssItem.getStartDate(), ssItem.getEndDate(),
									ssItem.getEmployeeID(), ssItem.getID()},
								m_className);

            return result;
        }

        return result;
    }

    protected ValidationResult verifySwappedShiftNotPublished(Validatable ssr,
        ShiftSwapItem ssItemIn, ShiftAssignment swappedInShift, ShiftSwapItem ssItemOut)
    throws Exception
    {
        ValidationResult result = null;

        // if ssItemIn represents a "shift" and 
        // a shift exists aligned with ssItemIn and
        // itemIn is not aligned with itemOut 
        if ( ssItemIn.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_SHIFT) &&
            swappedInShift != null && !shiftAlignsWithSSItem(swappedInShift, ssItemOut) )
        {
            // LOCAL: A published shift between {0} and {1} was found for employee {2}.
            result = ValidationUtil.setAndLogHardValidationResult(ssr,
                RmEjbBundleKey.SS_PUBLISHEDSHIFTFOUND_FOR_SSITEM,
								RmEjbLogBundleKey.SS_PUBLISHEDSHIFTFOUND_FOR_SSITEM,
                new Serializable[] {
									ssItemIn.getStartDate(), ssItemIn.getEndDate(),
									ssr.getValidationCache().getEmployeeNameByID(ssItemIn.getEmployeeID())},
                m_className);
            result.setApprove(true);
            return result;
        }

        return result;
    }

    /**
     * Verify if no shift assignment exists during the orgDay for ssItemOut.startTime or
     * if any exists it is aligned with ssItemIn.  A shift aligned with ssItemIn is not an
     * error to accomodate the scenario of a swap request which swaps shifts during the same day.
     *
     * @param ssr
     * @param ssItemOut
     * @param shiftStartingDuringSSItemOutOrgDay
     * @param ssItemIn
     * @return
     */
    protected ValidationResult verifyNoShiftExistsOrShiftAligned(Validatable ssr, ShiftSwapItem ssItemOut, ShiftAssignment shiftStartingDuringSSItemOutOrgDay,
        ShiftSwapItem ssItemIn)
    {
        ValidationResult result = null;

        // Note: if a shift starts during the orgDay for ssItemOut.startTime, it could be the shift received during
        // the swap.
        //
        // if a shift starts during the orgDay for ssItemOut.startTime and is not aligned with ssItemIn (ie. is not the
        // shift received after the swap), then another shift exists which starts on the same org day
        // as the original shift before the swap.
        if ( ssItemOut.isSwapTypeShift() && shiftStartingDuringSSItemOutOrgDay != null && !shiftAlignsWithSSItem(shiftStartingDuringSSItemOutOrgDay, ssItemIn))
        {
            // LOCAL: A new shift was found on the day of the original shift ({0} to {1}) in the swap request (employee ID = {0}, shift swap item ID = {3}).
            result = ValidationUtil.setAndLogHardValidationResult(ssr,
								RmEjbBundleKey.SS_SHIFTFOUND_FOR_SSITEM_AT_LOC_BEFORESWAP,
								RmEjbLogBundleKey.SS_SHIFTFOUND_FOR_SSITEM_AT_LOC_BEFORESWAP,
								new Serializable[] {
									ssItemOut.getStartDate(), ssItemOut.getEndDate(),
									ssItemOut.getEmployeeID(), ssItemOut.getID()},
								m_className);
            return result;
        }

        return result;
    }

    /**
     * Verify no unavails exist during the interval for the given ssItem.
     *
     * @param ssr
     * @param ssItem
     * @param unavails
     * @return
     */
    protected ValidationResult verifyNoUnavailsExist(Validatable ssr, ShiftSwapItem ssItem, Collection unavails)
    {
        ValidationResult result = null;

        if ( ssItem.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_SHIFT) && unavails != null && !unavails.isEmpty() )
        {
            // LOCAL: Unavailability events overlap the orginal shift location ({0, date} and {1, date} before the swap in the published schedule (employee ID = {2}, shift swap item ID = {3}).
            result = ValidationUtil.setAndLogHardValidationResult(ssr,
								RmEjbBundleKey.SS_UNAVAILSFOUND_FOR_SSITEM_AT_LOC_BEFORESWAP,
								RmEjbLogBundleKey.SS_UNAVAILSFOUND_FOR_SSITEM_AT_LOC_BEFORESWAP,
                new Serializable[]{
									ssItem.getStartDate(), ssItem.getEndDate(),
									ssItem.getEmployeeID(), ssItem.getID()},
								m_className);
            return result;
        }

        return result;
    }

    /**
     * Check if given shift is aligned with the shift swap item.
     *
     * @param shift
     * @param ssItem
     * @return
     */
    protected boolean shiftAlignsWithSSItem(ShiftAssignment shift, ShiftSwapItem ssItem)
    {
        return ( shift.getStartTime().equals(ssItem.getStartDate()) &&
            shift.getEndTime().equals(ssItem.getEndDate()) );
    }
}
