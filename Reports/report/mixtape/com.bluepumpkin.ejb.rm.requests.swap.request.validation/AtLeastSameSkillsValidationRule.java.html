/**
 * <li> Rule from UI: Both agents have ***at least*** same skills for the active campaign queue
 * 
 * <li> Ensure that the relevant skills for an employee giving up his shift matches
 *  at least skills of the employee receiving the shift during the time period of the shift involved.
 *  ### Employee receiving the shift can have more skills
 *
 * todo: talk to jason.  need to compare only relevant skills ie. relevant skills of
 * the employee whose shift is being given up must match the skills of the employee
 * receiving it during the shift duration.
 *
 * <p>
 * For 2 way swap, validaton done for both shift swap items of the swap request <br>
 * For 1 way swap, validation done only for the swap item with the shift. <br>
 *
 * <p>
 * Rule applies equally for tentative and non tentative requests as only
 * skills are verified ( no shift information is used)
 *  
 */
public class AtLeastSameSkillsValidationRule implements Validator{
    private static String m_className =  AtLeastSameSkillsValidationRule.class.getName();
    private static Category m_cat = Log.initCategory( m_className );

    private ShiftSwapValidationCache m_vc;
    
    public AtLeastSameSkillsValidationRule(){
    };
    
    public ValidationResult validate(Validatable validatable) throws Exception 
    {
        ShiftSwapRequest ssr = (ShiftSwapRequest)validatable;
        String methodName = "validate";
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, ssr));
        
        //todo: common code, refactor.
        ValidationResult valResult = null;
        
        m_vc = (ShiftSwapValidationCache) ssr.getValidationCache();
        
        // ssitems is an ordered list item(0) ... item(N-1)
        // item(A) gives its shift to item((A+1)modN)
        // item(A) receives its new shift from item((A+N-1)modN)
        List ssItems = ssr.getShiftSwapItems();
        
        int itemsSize = ssItems.size();        
        for (int i = 0; i < itemsSize; i++) {
            int inIndex = (i + itemsSize -1) % itemsSize;
            ShiftSwapItem itemIn  = (ShiftSwapItem)ssItems.get(inIndex);
            ShiftSwapItem itemOut = (ShiftSwapItem)ssItems.get(i);
            
            // no skills required for time off
            if (itemIn.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF))
                continue;
            
            // the SkillAssignments need to be compared for two employees,
            // for the time period encompassing the shift being received.
            Date itemInStart= itemIn.getStartDate(), itemInEnd= itemIn.getEndDate();
            List<ShiftSwapItem> items = new ArrayList<ShiftSwapItem>();
            items.add(itemIn);
            Collection<ID> skillsLikedToSP = m_vc.getRelevantSkillsForSchedulingPeriod(items);

            ValidationResult tempValResult  = compareSkills(ssr,
				skillsLikedToSP,
                m_vc.getSkillsForEmployeeDuringPeriod(itemIn.getEmployeeID(), itemInStart, itemInEnd), 
                m_vc.getSkillsForEmployeeDuringPeriod(itemOut.getEmployeeID(),itemInStart, itemInEnd), 
                itemOut.getEmployeeID(),
                itemInStart, 
                itemInEnd);

            // overwrite the actual validation result only if the last validation failed, ie result is 'non null'.  
            // Otherwise, a validation result set in the first iteration might be overwritten by a 'null'
            // validation result in the 2nd iteration. 
            valResult = (tempValResult != null)?tempValResult:valResult;
        }
        
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName));
        return valResult;
    }
    
    /**
     * @param skillsAssnIn the Collection of SkillAssignments from the employee
     * who is giving a shift away
     * @param skillsAssnOut the Collection of SkillAssignments for the employee
     * who is receiving a shift
     * @param idEmp the ID of the employee who is receiving a shift
     * @return a ValidationResult containing information about the employee
     * who is lacking some Skill, and the name of the Skill which is lacking.
     * @throws Exception
     */
    private ValidationResult compareSkills(Validatable validatable, Collection<ID> skillsLikedToSP, Collection skillAssnsForItemInEmp, 
        Collection skillAssnsForItemOutEmp, ID idEmp, Date itemInStart, Date itemInEnd) throws Exception {
        
        ValidationResult valResult = null;

		// if no skill assignments were found.
		if (skillAssnsForItemInEmp == null && skillAssnsForItemOutEmp == null) return null;
		
		if (skillAssnsForItemInEmp == null) skillAssnsForItemInEmp = Collections.EMPTY_LIST;
		if (skillAssnsForItemOutEmp == null) skillAssnsForItemOutEmp = Collections.EMPTY_LIST;

		// if no skill assignments were found.
		if (skillAssnsForItemInEmp.isEmpty() && skillAssnsForItemOutEmp.isEmpty()) return null;
        
        // Note: For skill assignments, SkillAssignment.ID is always 'null'.  The ID is stored
        // in SkillAssignment.SkillID.
        Map skillIDToSkillAssnOut = RequestUtil.getMapOfFieldToVOBase(skillAssnsForItemOutEmp, 
            SkillAssignmentFieldInfo.SKILLASSIGNMENT_SKILLID); 
        
        // for each SkillAssignment of exporting employee
        for (Iterator itr=skillAssnsForItemInEmp.iterator(); itr.hasNext(); ) {
            SkillAssignment skillAssnIn = (SkillAssignment)itr.next();
            
            // check if exporting employee skill assignment matches the importing employee skill assignment 
			SkillAssignment skillAssnOut = 
                (SkillAssignment) skillIDToSkillAssnOut.get(skillAssnIn.getSkillID());
            
            // if skill for exporting employee not found for importing employee
            if (skillAssnOut == null && skillsLikedToSP.contains(skillAssnIn.getSkillID())) {
                valResult = ValidationUtil.setSoftValidationResult(validatable, 
                    RmEjbBundleKey.SS_SAME_SKILLS, m_vc.getEmployeeNameByID(idEmp), 
                    ValidationUtil.getSkillName(skillAssnIn.getSkillID()), itemInStart, itemInEnd, m_className);
            }
        }
        
        return valResult;
    }
}
