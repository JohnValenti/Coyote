/**
 * <li> Rule From UI: Both agents are from the same call center.
 *
 * <li> Rule to ensure that for a 2 way swap:
 * <ul>
 *    <li> before the swap, employees belong to the same organization during the
 *         duration of their respective shifts.
 *    <li> <b>and</b> after the swap, employees belong to the same organization during
 *         the duration of their respective shifts assigned to them after the swap.
 *    <li> The organizations that the employees belong before and after the swap 
 *         must be the same.
 * </ul>
 * <code>
 *         emp1 in org1                             emp1 in org1
 *       --------------                           ---------------
 *          shift1                                    shift2
 *       --------------                           ---------------
 *            |                                        / \           
 *            |                                         |           
 *            \/                                        |
 *       --------------                           ----------------
 *          shift1                                    shift2
 *       --------------                           ----------------
 *          emp2 in org1                            emp2 in org1
 * </code>
 *
 * <p>
 * for a 2 way swap: If E1 has shift S1 and E2 has shift S2, then during S1 both
 * E1 and E2 must belong to the same organization. During S2 both E1 and E2 must 
 * belong to the same organization as well as this organization must match the 
 * organization during S1. <br>
 *
 * for a one way swap: if E1 has shift S1 being given up to E2, then during S1
 * both E1 and E2 must be in the same organization. <br>
 * 
 * <p>
 * Applies equally for tentative and non tentative requests as the processing only involves
 * organizations (call centers) and not schedules.
 */

//TODO: current implemention partial (see javadoc above). only verifies if all employees 
// in same org before swap.
public class SameCallCenterValidationRule implements Validator
{
    private static final Category m_cat = Log.initCategory( SameCallCenterValidationRule.class.getName());

	private static final String m_className = SameCallCenterValidationRule.class.getName();
    
    public SameCallCenterValidationRule(){};
    
    public ValidationResult validate(Validatable validatable)
                throws Exception 
    {
        ShiftSwapRequest ssr = (ShiftSwapRequest)validatable;
		String methodName = "validate";
        
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, ssr));
        
        //todo: refactor common code
        ValidationResult result = null;
        
        ShiftSwapValidationCache vc = (ShiftSwapValidationCache) ssr.getValidationCache();
        
        // ssItems is an ordered list item(0) ... item(N-1)
        // item(A) gives its shift to item((A+1)modN)
        // item(A) receives its new shift from item((A+N-1)modN)
        List ssItems= ssr.getShiftSwapItems();
        
        int ssItemsSize = ssItems.size();
        
        // get the Organization ID and org week start date for first item
        // each subsequent item must be same as first item
        ShiftSwapItem ssItemFirst= (ShiftSwapItem)ssItems.get(0);
        Date dtStartFirst = ssItemFirst.getStartDate();
        ID idOrgFirst = vc.getOrgIDForEmployeeDuringPeriod(ssItemFirst.getEmployeeID(),
        	dtStartFirst, ssItemFirst.getEndDate());
        
        // for each item, compare the Organization IDs
        for (int i = 1; i < ssItemsSize; i++) { //todo: ensure ssItems has at least 2 elements
            ShiftSwapItem ssItemCurr = (ShiftSwapItem)ssItems.get(i);
            
            Date dtStartCurr = ssItemCurr.getStartDate();
            ID idOrgCurr = vc.getOrgIDForEmployeeDuringPeriod(ssItemCurr.getEmployeeID(),
                                dtStartCurr, ssItemCurr.getEndDate());
            
            // todo: SameCallCenterPeriodValidationRule does the same as well
            // todo: Is checking for OrgIDs the same as checking for same call center?
            // todo: refactor common code between SameCallCenterPeriodValidationRule and
            //   this class.
            // if IDs do not match
            if (!idOrgFirst.equals(idOrgCurr)) {
                result = ValidationUtil.setSoftValidationResult(validatable, 
                	RmEjbBundleKey.SS_SAME_CALLCENTER,
                    new Serializable[] {
                        vc.getEmployeeNameByID(ssItemFirst.getEmployeeID()),
                        vc.getOrganizationByID(idOrgFirst).getName(), 
                        vc.getEmployeeNameByID(ssItemCurr.getEmployeeID()), 
                        vc.getOrganizationByID(idOrgCurr).getName()}, m_className);
            }
        }
     
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName));   
        return result;
    }
}
