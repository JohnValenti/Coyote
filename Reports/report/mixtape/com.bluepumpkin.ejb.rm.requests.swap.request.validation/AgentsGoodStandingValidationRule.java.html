import java.util.Collection;
import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;

import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.swap.request.model.ShiftSwapRequest;
import com.bluepumpkin.ejb.rm.util.RmUtil;

/**
 * Ensure that the employees participating in the shift swap belong to
 * the organization.
 * <p>
 * No distinction between one way or 2 way swaps.
 * <p>
 * Applies equally for tentative and non tentative requests as no schedule information is used.
 *
 */

public class AgentsGoodStandingValidationRule implements Validator
{
   private static final String m_className =  AgentsGoodStandingValidationRule.class.getName();

   private static Category m_cat = Log.initCategory(m_className);

   public AgentsGoodStandingValidationRule(){
   };

   public ValidationResult validate(Validatable validatable) throws Exception {
        ShiftSwapRequest ssr = (ShiftSwapRequest)validatable;

        String methodName = "validate";
        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, ssr));

        WorkResourceManager wrMgr = BbmManagerFactory.getWorkResourceManager();
        Date dtNow = new Date();

		ValidationResult result = null;
		 
        // check if each employee is in good standing.
        Collection ssrEmpIDs = ssr.getEmployees();
        for (Iterator it = ssrEmpIDs.iterator(); it.hasNext(); ) {
            ID empID = (ID) it.next();

			// get workResourceAssn for the employee.
            Collection wrAssns = wrMgr.getWorkResourceAssignments(empID, dtNow, dtNow, false);
			// if no workResAssn found
			if (wrAssns.isEmpty()) {
				ValidationCache vc = validatable.getValidationCache();

				// translate empID to empName
				// TODO: employee name must be localized.
				String employeeName = "<unknown>";
				if (empID != null) {
					employeeName = vc.getEmployeeNameByID(empID);
				}
				
				// add hard validation error.
				result = ValidationUtil.setHardValidationResult(validatable,
					RmEjbBundleKey.SS_NO_GOOD_STANDING, employeeName, m_className);
				
				break;
            }
        }

        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
        
        return result;
   }
}
