package com.verint.web.fs.widgets;

import java.util.Iterator;

import com.witness.web.uif.pagecomponent.MultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.InsertableMultiColListPC;
import com.witness.web.uif.pagecomponent.selection.SelectablePC;
import com.witness.web.uif.system.RequestContext;

public class MultiColMasterDetailListPC extends InsertableMultiColListPC {
	private static final long serialVersionUID = 1L;

	public MultiColMasterDetailListPC(RequestContext context) {
		super(context);
	}

	public MultiColMasterDetailListPC(RequestContext context,
			boolean isActiveHeader) {
		super(context, isActiveHeader);
	}

	@Override
	protected String appendRowData(StringBuffer html) {
		// Generate the Row data
		if (m_listModel != null) {
			int index=0;
			int rowNumDisplay = getRowNumberStartValue();
			for(Iterator it = m_listModel.iterator(); it.hasNext(); index++) {
				MultiColumnNodeData rowData = (MultiColumnNodeData)it.next();
				rowData.getNodeAttributes().put(SelectablePC.MULTISELECT_GROUP, "masterRow");
				appendHtmlRowData(html, rowData, rowNumDisplay, rowNumDisplay++);
				if (rowData instanceof MasterDetailRow) {
					for (MultiColumnNodeData mcnd : ((MasterDetailRow)rowData).getDetailRows()) {
						mcnd.getNodeAttributes().put(SelectablePC.MULTISELECT_GROUP, "detailRow");
						appendHtmlRowData(html, mcnd, rowNumDisplay, rowNumDisplay++);
					}
				}

				// Repeat header at specified interval
				if (isRepeatHeader(index+1) && it.hasNext()) appendHeader(html);
			}
		}

		return html.toString();
	}
	
}
