package com.verint.web.fs.widgets;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.HashSet;
import java.util.List;

import javax.servlet.http.HttpServletRequest;

import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.web.fs.keys.FsJavaScriptFileID;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.pagecomponent.CompositePC;
import com.witness.web.uif.system.RequestContext;

/**
 * This component is currently only used on the work rule employee assignment screen.  It is the dropdown
 * under the "Pattern" column of the Assignment Rule table.  There is a lot of custom logic in the javascript
 * component to handle the rather unusual behavior of this component, and as such it is probably not recommended
 * to reuse this component elsewhere.
 * 
 * NOTE: When you utilize this drop down selector in your page, you will also need to include
 * the CheckBoxDropDownSelectionPC.css file in your screen definition.
 */
public class CheckBoxDropDownSelectionPC<T> extends CompositePC {
	private static final long serialVersionUID = 1L;

	private boolean disableAutoUpdateOfInputText = false;
	private boolean isMultiEdit = false;
	private boolean enableSetAll = false;
	private String initialText = "";
	private String onChangeJS = "";
	private String initJS = "";
	private String resetJS;
	private ArrayList<String> IDs = new ArrayList<String>();
	private HashSet<String> selectedIDs =  new HashSet<String>();
	private ArrayList<String> displayValues = new ArrayList<String>();
	private HashMap<String, T> objects = new HashMap<String, T>();
	private boolean isEdited = false;
	private boolean isEnabled = true;
	
	public CheckBoxDropDownSelectionPC(RequestContext context, String name) {
		super(context);
		setName(name);
		this.addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
		this.addRequiredJavaScriptFile(JavaScriptFileID.JQUERY_POSITION);
		setJSMediator(FsJavaScriptFileID.MODULE_CHECKBOX_DROPDOWN);
		
		//Default reset JS behavior
		resetJS = getName() + ".reset();";
	}

	public boolean isMultiEdit() {
		return isMultiEdit;
	}

	public void setMultiEdit(boolean multiEdit) {
		isMultiEdit = multiEdit;
		if (isMultiEdit) {
			setInitialText(MULTI_VALUE_DATA);
		}
	}
	
	/**
	 * Normally the input text is updated based on the checked selections, however you can
	 * disable this and handle the update of the input text manually via javascript
	 * @param disableAutoUpdateOfInputText
	 */
	public void setDisableAutoUpdateOfInputText(boolean disableAutoUpdateOfInputText) {
		this.disableAutoUpdateOfInputText = disableAutoUpdateOfInputText;
	}
	
	/**
	 * Sets the initial text of the input box for the control (used in conjunction with disableAutoUpdateOfInputText)
	 * @param initialText
	 */
	public void setInitialText(String initialText) {
		this.initialText = initialText;
	}
	
	/**
	 * Enables/disables the "Select All" checkbox in the list
	 * @param enableSetAll
	 */
	public void setEnableSetAll(boolean enableSetAll) {
		this.enableSetAll = enableSetAll;
	}
	
	/**
	 * A javascript function to be executed when a checkbox in the list changes state
	 * @param onChangeJS
	 */
	public void setOnChangeJS(String onChangeJS) {
		this.onChangeJS = onChangeJS;
	}
	
	public void setInitJS(String initJS) {
		this.initJS = initJS;
	}
	
	public void addOption(String uniqueID, String displayValue, T object, boolean isSelected) {
		IDs.add(uniqueID);
		displayValues.add(displayValue);
		objects.put(uniqueID, object);
		if (isSelected){
			selectedIDs.add(uniqueID);
		}
	}
	
	public void setSelections(Collection<String> ids) {
		selectedIDs.clear();
		selectedIDs.addAll(ids);
	}
	
	public List<T> getSelections() {
		List<T> ret = new ArrayList<T>();
		for (String id : selectedIDs) {
			T o = objects.get(id);
			if (o != null) ret.add(o);
		}
		return ret;
	}
	
	@Override
	public String getHtmlDisplay() {
		String[] idArray = IDs.toArray(new String[IDs.size()]);
		String[] displayValueArray = displayValues.toArray(new String[displayValues.size()]);
		StringBuilder html = new StringBuilder();

		html.append("<input type='hidden' id='" + getEditedID() + "' name='" + getEditedName() + "' value='" + isEdited + "' />");
		html.append(getHiddenIDFieldHTML());
		html.append("<select id='" + getName() + "_select' name='" + getName() + "_select' multiple='multiple'>");
		
		for (int i = 0; i < idArray.length; i++) {
			if (selectedIDs.contains(idArray[i])) {
				html.append("<option value=\"" + idArray[i] + "\" selected>" + displayValueArray[i] + "</option>");
			} else {
				html.append("<option value=\"" + idArray[i] + "\">" + displayValueArray[i] + "</option>");
			}
		}
		html.append("</select>");
		return html.toString();
	}

	private String getHiddenIDFieldHTML() {
		StringBuffer html = new StringBuffer();
		for (String id : IDs) {
			if (html.length() > 0) html.append(",");
			html.append(id);
		}
		return "<input type='hidden' id='" + getSelectedIDsFieldName() + "' name='" +
			getSelectedIDsFieldName() + "' value='" + html.toString() + "'/>";
	}

	@Override
	public String getJavaScriptInitCode() {
		setJSMediatorArgs(
				Boolean.toString(disableAutoUpdateOfInputText) + ",'" +
				initialText + "'" + "," +
				Boolean.toString(enableSetAll) + ",'" +
				getEditedID() + "','" +
				getMultiEditIndicatorID() + "'"
				);
		StringBuffer js = new StringBuffer(super.getJavaScriptInitCode());
		onChangeJS += getName() + ".updateSelectedIDHiddenField();";
		js.append(getName() + ".change(function() {" + onChangeJS + " });\n");
		if (resetJS != null && resetJS.length() > 0) {
			js.append("resetJSArray[resetJSArray.length] = " + resetJS);
		}
		if (!isEnabled) initJS += getName() + ".setEnabled(false);\n";
		js.append(getName() + ".init(function() {" + initJS + " });\n");
		return js.toString();
	}
	
	private String getMultiEditIndicatorID() {
		return getName() + "__me";
	}
	
	private String getEditedID() {
		return getName() + "__edited_id";
	}
	
	@Override
	public boolean extractParameters(HttpServletRequest request) {
		if (!isAutoExtractParamEnabled()) return true;
		String editedStr = super.getParameter(request, getEditedName());
		if (editedStr != null) {
			this.isEdited = "true".equalsIgnoreCase(editedStr);
		}
		String checkedSelections = super.getParameter(request, getName() + "_selectedIDs");
		if (checkedSelections != null) {
			setSelections(StringUtil.parseStringToCollection(checkedSelections, ","));
		} else if (isEdited) {
			selectedIDs.clear();
		}
		return super.extractParameters(request);
	}
	
	public boolean isEdited() {
		return isEdited;
	}
	
	private String getEditedName() {
		return getName() + "__edited";
	}
	
	private String getSelectedIDsFieldName() {
		return getName() + "_selectedIDs";
	}
	
	public void clearOptions() {
		IDs.clear();
		objects.clear();
		selectedIDs.clear();
		displayValues.clear();
	}
	
	public void setEnabled(boolean enabled) {
		this.isEnabled = enabled;
	}
	
	public void setResetJS(String resetJS) {
		this.resetJS = resetJS;
	}
}
