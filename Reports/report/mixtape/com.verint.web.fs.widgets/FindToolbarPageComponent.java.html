package com.verint.web.fs.widgets;

/**
 * Title:        FindToolbarPageComponent
 * Description:  
 * Copyright:    Copyright (c) 2010
 * Company:      Verint Systems, Inc
 * @author       Dustin Stanley
 *
 * This item is similar to FindToolbarPageComponent in the emt project, except that
 * it is intended for use in a toolbarPC (hence it implements ToolbarElement).
 * 
 * It is intended to be used to search for items in a list.  It will submit the action
 * specified in the constructor which can then be used by the request handler to determine
 * if this component was used to submit the request.
 */

import java.util.ArrayList;

import javax.servlet.http.HttpServletRequest;

import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.jsp.HtmlTable;
import com.witness.web.uif.jsp.HtmlTableCell;
import com.witness.web.uif.jsp.HtmlTableRow;
import com.witness.web.uif.jsp.HtmlTag;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.PageComponent;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarElement;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarIF;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarTextButton;
import com.witness.web.uif.system.RequestContext;

public class FindToolbarPageComponent extends PageComponent implements ToolbarElement {
	private static final long serialVersionUID = 1L;
	
	private String defaultValue = "";
	private String inputValue;
    private String findFN;
    private String label;
    private String action;
    private boolean isEnabled = true;
    private boolean isVisible = true;
    private boolean isEmptySearchStringAllowed = true;

	private String emptySearchDialogPrompt;

    protected ToolbarIF toolbar;

    public FindToolbarPageComponent(RequestContext context, ToolbarIF toolbar, String inputFN,
            String action, String label) {
        super(context);
        this.findFN = inputFN;
        this.label = label;
        this.action = action;
        this.toolbar = toolbar;
        setName("findToolbarPC");
        
        this.extractParameters(context.getRequest());
        
        emptySearchDialogPrompt =
        	getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_FIND_EMPTY_STRING_DIALOG);
    }

    public FindToolbarPageComponent(RequestContext context, ToolbarIF toolbar, String inputFN, String action) {
        this(context, toolbar, inputFN, action, null);
        label = getLocalizer().i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.FILTERBOX_FIND);
    }

    /**
     * Returns the default value that will be displayed in the text field.
     * @return Returns the default value that will be dispalyed in the text field.
     */
    public String getDefaultValue() {
        return defaultValue;
    }

    /**
     * Sets the default value that will be displayed in the text field. 
     * @param defaultValue Default text that will be pre-populated
     */
    public void setDefaultValue(String defaultValue) {
        this.defaultValue = defaultValue;
    }

    /**
     * Returns the HTML text for this component.
     */
    
    public String getHtmlDisplay() {
    	return getHtmlDisplay(false);
    }
    
    public String getHtmlDisplayForContentTitle(){
    	return getHtmlDisplay(true);
    }
    
    private String getHtmlDisplay(boolean setBorderToEmpty){
        ArrayList<HtmlTableCell> rowcells = new ArrayList<HtmlTableCell>();
        ArrayList<HtmlTableRow> tableRows = new ArrayList<HtmlTableRow>();

        HtmlTableCell cell = new HtmlTableCell(label + "&nbsp;");
        cell.setWidth("150");
        cell.setStyleClass("tbarLbl");
        cell.setAlign("right");
        cell.setNowrap("nowrap");
        rowcells.add(cell);

        // Find: text entry
        if (defaultValue == null) {
            defaultValue = "";
        }
        
        StringBuffer htmlForFindControl = new StringBuffer();
        
        if (isEmptySearchStringAllowed == false) {
	        htmlForFindControl.append("<script type=\"text/javascript\" language=\"javascript1.2\">function validateInput(findFN) { ");
	        htmlForFindControl.append("var findTextBox = document.getElementById(findFN); if (findTextBox != null) { ");
	        htmlForFindControl.append("if (trim(findTextBox.value) == \"\") { alert('").append(emptySearchDialogPrompt).append("'); ");
	        htmlForFindControl.append("findTextBox.focus(); return false; } } return true; } ").append("</script>");
        }

        htmlForFindControl.append("<input type=\"text\" name=\"");
        htmlForFindControl.append(findFN);
        htmlForFindControl.append("\" value=\"" + inputValue + "\" class=\"inputText\" style=\"width:104px;\">");
        
        cell = new HtmlTableCell(htmlForFindControl.toString());
        
		cell.setValign("middle");
		rowcells.add(cell);

        // GO button, contained in a wrapper cell
        String btnLabel = getLocalizer().i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.FILTERBOX_GO);
        ToolbarTextButton goButton = new ToolbarTextButton(toolbar, action, btnLabel);
        if (isEmptySearchStringAllowed == false) {
        	goButton.setConditionJS("validateInput('" + findFN + "')");
        }
        cell = new HtmlTableCell(goButton);
        rowcells.add(cell);
        HtmlTableRow rows = new HtmlTableRow(rowcells);
        tableRows.add(rows);
        HtmlTable table = new HtmlTable(tableRows);
        if(setBorderToEmpty){
	        table.setAttribute("border", "0");
	        table.setAttribute("cellspacing", "0");
	        table.setAttribute("cellpadding", "0");
        }
        return table.toString();
    }

	@Override
	public ToolbarIF getToolbar() {
		return toolbar;
	}
	
	@Override
	public void setToolbar(ToolbarIF toolbar) {
		this.toolbar = toolbar;	
	}

	@Override
	public boolean isEnabled() {
		return isEnabled;
	}
	
	@Override
	public void setEnabled(boolean enabled) {
		isEnabled = enabled;
	}
	
	@Override
	public boolean isVisible() {
		return isVisible;
	}

	@Override
	public void setVisible(boolean visible) {
		isVisible = visible;
	}
	
	@Override
	public HtmlTag getContainer() {
		return null;
	}

	@Override
	public boolean isHoverable() {
		return false;
	}

	@Override
	public boolean isStatic() {
		return false;
	}

	@Override
	public void setHoverable(boolean hoverable) {}

	@Override
	public void setStatic(boolean bButton) {}

	@Override
	public String getJavaScriptInitCode() {
		return null;
	}
	
    public boolean isEmptySearchStringAllowed() {
		return isEmptySearchStringAllowed;
	}

	public String getEmptySearchDialogPrompt() {
		return emptySearchDialogPrompt;
	}

	public void setEmptySearchDialogPrompt(String emptySearchDialogPrompt) {
		this.emptySearchDialogPrompt = emptySearchDialogPrompt;
	}

	public void setEmptySearchStringAllowed(boolean isEmptySearchStringAllowed) {
		this.isEmptySearchStringAllowed = isEmptySearchStringAllowed;
	}
	
	public boolean extractParameters(HttpServletRequest request) {
		boolean success=super.extractParameters(request);

		inputValue = getParameter(request, findFN);
		if (inputValue == null) inputValue = defaultValue;

		return success;
	}
}
