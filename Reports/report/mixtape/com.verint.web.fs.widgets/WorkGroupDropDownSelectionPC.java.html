package com.verint.web.fs.widgets;

import com.bluepumpkin.web.core.pagecomponent.DropDownSelectionPC;
import com.witness.web.uif.system.RequestContext;

public class WorkGroupDropDownSelectionPC extends DropDownSelectionPC<String> {
	private static final long serialVersionUID = 1L;
	
	public static final String NO_SELECTION_TEXT = "--";
	public static final String NO_SELECTION_ID = "0";
	
	public WorkGroupDropDownSelectionPC(RequestContext context, String name) {
		super(context, name);

		this.addOption(NO_SELECTION_ID, NO_SELECTION_TEXT, NO_SELECTION_ID, true);
		
		this.addOption("1", context.getLocalizer().formatNumber(1), "1", false);
		this.addOption("2", context.getLocalizer().formatNumber(2), "2", false);
		this.addOption("3", context.getLocalizer().formatNumber(3), "3", false);
		this.addOption("4", context.getLocalizer().formatNumber(4), "4", false);
		this.addOption("5", context.getLocalizer().formatNumber(5), "5", false);
		this.addOption("6", context.getLocalizer().formatNumber(6), "6", false);
		this.addOption("7", context.getLocalizer().formatNumber(7), "7", false);
	}
}
