package com.verint.test.utils.parser;

import java.util.HashMap;
import java.util.Map;

//TODO Add Java Docs
//TODO File result is common solution which going to be used by Export/Import.
//TODO Accordingly we need documentation in JavaDocs
public class ExportFileParserFactory {

	private static Map<String, ExportFileParser> parserMap = new HashMap<String, ExportFileParser>();

	private ExportFileParserFactory() {}

	public static ExportFileParser getInstance(final String parserType) throws Exception {
		//is required parser already instantiated
		if (parserMap.containsKey(parserType)) {
			return parserMap.get(parserType);
		}

		//instantiate new parser
		Class<?> clazz = Class.forName(parserType);
		ExportFileParser exportFileParser = (ExportFileParser)clazz.newInstance();

		parserMap.put(parserType, exportFileParser);
		return exportFileParser;
	}
}
