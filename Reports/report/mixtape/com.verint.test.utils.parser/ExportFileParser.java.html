package com.verint.test.utils.parser;

import com.gargoylesoftware.htmlunit.Page;

import java.io.File;
import java.io.FileInputStream;
import java.io.IOException;
import java.io.InputStream;

//TODO Add Java Docs
public abstract class ExportFileParser<T extends FileResult> {

	public T parseFile(Page page, String delimiter) throws IOException {
		return parseFile(page.getWebResponse().getContentAsStream(), delimiter, getFileResult());
	}

	public T parseFile(File file, String delimiter) throws IOException {
		T result = getFileResult();
		result.setFile(file);
		return parseFile(new FileInputStream(file), delimiter, result);
	}

	protected abstract T parseFile(InputStream contentAsStream, String delimiter, T result) throws IOException;

	protected abstract T getFileResult();
}
