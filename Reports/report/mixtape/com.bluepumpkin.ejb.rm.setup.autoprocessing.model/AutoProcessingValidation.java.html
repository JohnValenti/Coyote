package com.bluepumpkin.ejb.rm.setup.autoprocessing.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;


/**
 * Title:        AutoProcessingValidation
 * Description:  Represents a AutoProcessingValidation class for value object.
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc.
 * @author       Shailendra Sharma
 * @version 	 1.0
 */

public class AutoProcessingValidation extends ValueObjectBase {

	private static FieldInfo m_fieldInfo = new AutoProcessingValidationFieldInfo();
	private String m_bundleKey;
	private String m_validator;
	public static int APPROVALSTATUS_DENY = 0;
	public static int APPROVALSTATUS_APPROVE = 1;
	public static int APPROVALSTATUS_WAITLIST = 2;
	public AutoProcessingValidation() {
		super();
	}

	protected FieldInfo getFieldInfo() { return m_fieldInfo; }


	// getters

	/**
	 * Java Resource Bundle Key. This is the key of the resource into the Java 
	 * Bundle that holds resources for the validation rule.
	 */
	public String getBundleKey() {
		return m_bundleKey;
	}

	/**
	 * Name of a Java Class that implements the FireRule interface.
	 */
	public String getValidator() {
		return m_validator;
	}

	public ID getAutoProcessingRuleID() {
		return getFieldValueID(AutoProcessingValidationFieldInfo.AUTOPROCESSINGVALIDATION_I_AUTOPROCESSINGRULEID);
	}


	public ID getValidationRuleID() {
		return getFieldValueID(AutoProcessingValidationFieldInfo.AUTOPROCESSINGVALIDATION_I_VALIDATIONRULEID);
	}

	public int getApprovalFlag() {
		return getFieldValueID(AutoProcessingValidationFieldInfo.AUTOPROCESSINGVALIDATION_I_APPROVALSTATUS).toInt();
	}

	public boolean isApprove() {
		return getApprovalFlag() == APPROVALSTATUS_APPROVE;
	}

	public boolean isDeny() {
		return getApprovalFlag() == APPROVALSTATUS_DENY;
	}

	public boolean isWailist() {
		return getApprovalFlag() == APPROVALSTATUS_WAITLIST;
	}
	// setters

	public void setBundleKey(String str) {
		m_bundleKey = str;
	}

	public void setValidator(String str) {
		m_validator = str;
	}

	public void setAutoProcessingRuleID(ID pAutoProcessingRuleId) {
		setFieldValue(AutoProcessingValidationFieldInfo.AUTOPROCESSINGVALIDATION_I_AUTOPROCESSINGRULEID, pAutoProcessingRuleId);
	}


	public void setValidationRuleId(ID pValidationRuleID) {
		setFieldValue(AutoProcessingValidationFieldInfo.AUTOPROCESSINGVALIDATION_I_VALIDATIONRULEID, pValidationRuleID);
	}

	public void setApprovalFlag(int pApproval) {
		setFieldValue(AutoProcessingValidationFieldInfo.AUTOPROCESSINGVALIDATION_I_APPROVALSTATUS, new ID(pApproval));
	}

}
