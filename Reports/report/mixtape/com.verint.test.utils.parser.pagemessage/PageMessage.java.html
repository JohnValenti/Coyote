package com.verint.test.utils.parser.pagemessage;


import com.gargoylesoftware.htmlunit.html.HtmlDivision;
import com.gargoylesoftware.htmlunit.html.HtmlTableHeaderCell;

public class PageMessage {

	public final PageMessageType pageMessageType;
	public final String pageMessageText;

	public PageMessage(HtmlDivision messageHtmlElement) {
		String messageClass = messageHtmlElement.getAttribute("class");
		if (messageClass == null) {
			this.pageMessageType = PageMessageType.UNKNOWN;
		} else if ("stdError".equals(messageClass)) {
			this.pageMessageType = PageMessageType.ERROR;
		} else if ("stdInfo".equals(messageClass)) {
			this.pageMessageType = PageMessageType.INFO;
		} else if ("stdWarning".equals(messageClass)) {
			this.pageMessageType = PageMessageType.WARNING;
		} else if ("stdResponse".equals(messageClass)) {
			this.pageMessageType = PageMessageType.RESPONSE;
		} else {
			this.pageMessageType = PageMessageType.UNKNOWN;
		}

		pageMessageText = messageHtmlElement.getTextContent();
	}

	public PageMessage(HtmlTableHeaderCell htmlTableHeaderCellElement) {
		pageMessageType = PageMessageType.ERROR;
		pageMessageText = htmlTableHeaderCellElement.getTextContent();
	}
}
