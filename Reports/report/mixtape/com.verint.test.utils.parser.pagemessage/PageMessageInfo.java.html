package com.verint.test.utils.parser.pagemessage;


import com.gargoylesoftware.htmlunit.html.DomElement;
import com.gargoylesoftware.htmlunit.html.HtmlDivision;
import com.gargoylesoftware.htmlunit.html.HtmlPage;
import com.gargoylesoftware.htmlunit.html.HtmlTableHeaderCell;

import java.util.ArrayList;
import java.util.Collection;
import java.util.stream.Collectors;

public class PageMessageInfo {

    private Collection<PageMessage> pageMessages;

    public PageMessageInfo(HtmlPage page) {
        //Sample HTML for the page message area:
        //<div id="pageMessages" class="pageMessageArea"  style="overflow:auto;display:block;" >
        //<a id="MessageBox"></a>
        //
        //<div class="stdError"><label class="hidden-label"> Error Message, </label>Unable to load data.</div>
        //<div class="stdError"><label class="hidden-label"> Error Message, </label>A valid start time is required.</div>
        //</div>

        pageMessages = new ArrayList<>();

        DomElement element = page.getElementById("pageMessages");

        if (element != null && element.hasChildNodes()) {
            for (DomElement childElement : element.getChildElements()) {
                if (childElement instanceof HtmlDivision) {
                    PageMessage m = new PageMessage((HtmlDivision) childElement);
                    if (m.pageMessageText.contains("Please see messages for problems.")) {
                        parseMessagesInTable(page);
                    } else {
                        pageMessages.add(m);
                    }
                }
            }
        }
    }

    private void parseMessagesInTable(HtmlPage page) {
        DomElement element = page.getElementById("tabler0");
        if (element != null && element.hasChildNodes()) {
            for (DomElement childElement : element.getChildElements()) {
                if (childElement instanceof HtmlTableHeaderCell) {
                    PageMessage m = new PageMessage((HtmlTableHeaderCell) childElement);
                    if (m.pageMessageText.contains("Please see messages for problems.")) {
                        parseMessagesInTable(page);
                    } else {
                        pageMessages.add(m);
                    }
                }
            }
        }
    }

    public Collection<PageMessage> getErrorMessages() {
        return pageMessages.stream()
                .filter(e -> e.pageMessageType == PageMessageType.ERROR)
                .collect(Collectors.toList());
    }

    public Collection<PageMessage> getResponseMessages() {
        return pageMessages.stream()
                .filter(e -> e.pageMessageType == PageMessageType.RESPONSE)
                .collect(Collectors.toList());
    }

    public Collection<PageMessage> getInfoMessages() {
        return pageMessages.stream()
                .filter(e -> e.pageMessageType == PageMessageType.INFO)
                .collect(Collectors.toList());
    }

    public Collection<PageMessage> getWarningMessage() {
        return pageMessages.stream()
                .filter(e -> e.pageMessageType == PageMessageType.WARNING)
                .collect(Collectors.toList());
    }
}
