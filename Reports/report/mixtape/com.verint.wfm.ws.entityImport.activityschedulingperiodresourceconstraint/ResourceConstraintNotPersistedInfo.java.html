package com.verint.wfm.ws.entityImport.activityschedulingperiodresourceconstraint;

import java.util.List;

import com.verint.wfm.ws.entityImport.EntityNotPersistedInfo;
import com.verint.wfm.ws.entityImport.activityschedulingperiodresourceconstraint.model.ResourceConstraintDTO.ResourceConstraintExternalKey;
import com.verint.wfm.ws.entityImport.schedulingPeriod.model.SchedulingPeriodDTO.SchedulingPeriodExternalNaturalKey;

public class ResourceConstraintNotPersistedInfo extends
		EntityNotPersistedInfo<ResourceConstraintExternalKey> {
	private static final long serialVersionUID = 1L;

	public ResourceConstraintNotPersistedInfo() {
		super();
	}

	public ResourceConstraintNotPersistedInfo(
			ResourceConstraintExternalKey activityAndSchedulingPeriodExternalNaturalKey) {
		super(activityAndSchedulingPeriodExternalNaturalKey);
	}

	// -----------------------------------------
	public ResourceConstraintExternalKey getActivityAndSchedulingPeriodExternalNaturalKey() {
		return super.getEntityExternalSourceId();
	}

	/**
	 * Dummy here on server, but needed for SOAP stub
	 */
	public void setActivityAndSchedulingPeriodExternalNaturalKey(
			ResourceConstraintExternalKey activityAndSchedulingPeriodExternalNaturalKey) {
		// do nothing
	}

	public void addSchedulingPeriodServerExceptionMessage(String exception) {
		getMessages()
				.add(String
						.format("Exception is %s. The server could not persist activity scheduling period resource constraint %s.",
								exception, getExternalIdSummary()));
	}

	public void addActivityNameMissingMessage() {
		getMessages()
				.add(String
						.format("For activity scheduling period resource constraint %s, the activity name is missing.",
								getExternalIdSummary()));
	}

	public void addNoSuchActivityNameMessage(String activityName) {
		getMessages()
				.add(String
						.format("For activity scheduling period resource constraint %s, the activity specified to have name %s does not exist.",
								getExternalIdSummary(), activityName));
	}

	public void addNoSuchSchedulingPeriodMessage(
			SchedulingPeriodExternalNaturalKey schedulingPeriodExternalNaturalKey) {
		getMessages()
				.add(String
						.format("For activity scheduling period resource constraint %s, the scheduling period %s does not exist.",
								getExternalIdSummary(), schedulingPeriodExternalNaturalKey));
	}

	public void addInvalidActivityMinSeatConstraintMessage(int minSeats) {
		getMessages()
				.add(String
						.format("For activity scheduling period resource constraint %s, min seats %d is not integer > 0.",
								getExternalIdSummary(), minSeats));
	}

	public void addInvalidActivityMaxSeatConstraintMessage(int maxSeats) {
		getMessages()
				.add(String
						.format("For activity scheduling period resource constraint %s, max seats %d is not integer > 0.",
								getExternalIdSummary(), maxSeats));
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public List<String> getMessages() {
		return super.getMessages();
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public void setMessages(List<String> messages) {
		// Do nothing
	}
}
