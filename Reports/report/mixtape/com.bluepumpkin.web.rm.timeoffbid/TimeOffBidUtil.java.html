package com.bluepumpkin.web.rm.timeoffbid;

import java.util.Calendar;
import java.util.Collection;
import java.util.Date;
import java.util.Iterator;
import java.util.Map;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.l10n.RegionalFormatBundleKey;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.workresource.model.Employee;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBid;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBidder;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.witness.web.uif.data.wrapper.DateWrapper;
import com.witness.web.uif.l10n.UIFWebBundleKey;

public class TimeOffBidUtil {
	public static final int DAYS = 0;
	public static final int WEEKS = 1;
	
	private TimeOffBidUtil() {
	}

	public static DateWrapper getSeniority(Employee emp, TimeZone tz, Localizer localizer) {
		Date date = emp.getStartTime();
		return new DateWrapper(date, localizer, tz, RegionalFormatBundleKey.DATE_FORMAT);
	}

	public static String formatMaxApprovalTime(TimeOffBid timeOffBid, TimeOffBidder timeOffBidder, Localizer localizer) {
		int maxPeriods = timeOffBidder.getMaxNumOfPeriods();
		int periodUnit = timeOffBidder.getPeriodUnit();
		boolean approvalMaxVaries = timeOffBid.getIsApprovalMaxValueVariesByEmp();
		String result = "";
		int one = 1;

		if (approvalMaxVaries) {
			if (maxPeriods != 0) {
				if (periodUnit == DAYS) {
					if (maxPeriods == one) {
						result = maxPeriods + " " + localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.DAY);
					} else {
						result = maxPeriods + " " + localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.DAYS);
					}
				} else {
					if (maxPeriods == one) {
						result = maxPeriods + " " + localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.WEEK);
					} else {
						result = maxPeriods + " " + localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.WEEKS);
					}
				}
			}

		} else {
			if (timeOffBid.getMaxValueForApproval() != 0) {
				if (timeOffBid.getMaxUnitForApproval() == DAYS) {
					if (timeOffBid.getMaxValueForApproval() == one) {
						result = timeOffBid.getMaxValueForApproval() + " " + localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.DAY);
					} else {
						result = timeOffBid.getMaxValueForApproval() + " " + localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.DAYS);
					}
				} else {
					if (timeOffBid.getMaxValueForApproval() == one) {
						result = timeOffBid.getMaxValueForApproval() + " " + localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.WEEK);
					} else {
						result = timeOffBid.getMaxValueForApproval() + " " + localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.WEEKS);
					}
				}
			}
		}
		return result;
	}

	public static Date getUTCDateTime() {
		Calendar cal = Calendar.getInstance();
		TimeZone tz = cal.getTimeZone();
		int offSet = tz.getRawOffset();
		int offSetHours = 0;
		int offSetMins = 0;

		if (tz.inDaylightTime(new Date())) {
			offSet += tz.getDSTSavings();
		}

		offSetHours = offSet / 1000 / 60 / 60;
		offSetMins = offSet / 1000 / 60 % 60;

		cal.add(Calendar.HOUR_OF_DAY, -offSetHours);
		cal.add(Calendar.MINUTE, -offSetMins);

		return cal.getTime();
	}

	public static String displayActivityNames(Map<ID, String> mapNames, Collection<ID> activityIds) {
		StringBuilder names = new StringBuilder();
		if (activityIds == null || mapNames == null || activityIds.isEmpty() || mapNames.isEmpty()) {
			return "";
		}
		int count = 0;
		for (Iterator<ID> it = activityIds.iterator(); it.hasNext();) {
			ID activityID = it.next();
			String activityName = mapNames.get(activityID);
			names.append(activityName);
			if (count < activityIds.size() - 1) {
				names.append(", ");
			}
			count++;
		}
		return names.toString();
	}

	public static boolean isLeapYear(int year) {
		int testYear = year;
		
		// Use -1 to test the current year.
		if (testYear == -1) {
			Calendar cal = Calendar.getInstance();
			testYear = cal.get(Calendar.YEAR);
		}

		// Test for Leap Year.
		return ((testYear % 400 == 0) || ((testYear % 4 == 0) && (testYear % 100 != 0)));
	}
}
