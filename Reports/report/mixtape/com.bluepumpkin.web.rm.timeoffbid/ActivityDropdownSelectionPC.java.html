import java.util.Collection;

import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.web.core.pagecomponent.DropDownSelectionPC;
import com.witness.web.uif.system.RequestContext;

public class ActivityDropdownSelectionPC extends DropDownSelectionPC<Activity> {
	private static final long serialVersionUID = 1L;
	public ActivityDropdownSelectionPC(RequestContext context, String name) {
		super(context, name);
	}
	public ActivityDropdownSelectionPC(RequestContext context, String name,  Collection<Activity> activities) {
		this(context, name, activities, false);
	}

	public ActivityDropdownSelectionPC(RequestContext context, String name, Collection<Activity> activities,boolean isSelected) {
		super(context, name);
		for (Activity a : activities) {
			addOption(a.getID().toString(), a.getName(), a,isSelected);
		}
	}
}
