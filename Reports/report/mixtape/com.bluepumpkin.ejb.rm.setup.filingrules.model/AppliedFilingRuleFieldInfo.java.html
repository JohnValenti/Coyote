public class AppliedFilingRuleFieldInfo extends FieldInfo {
    // field id constants
    // Id for the Applied filing rule.
    public static final int APPLIEDFILINGRULE_I_ID = 0;

    // RequestFiling Rule for the Applied filing rule.
    public static final int APPLIEDFILINGRULE_I_REQUESTFILINGRULEID = 1;

    // Applied status Request filing rule.
    public static final int APPLIEDFILINGRULE_B_ISAPPLIED = 2;

    //Organization Id for which the request filing rule belongs.
    public static final int APPLIEDFILINGRULE_I_ORGANIZATIONID = 3;

    //Number of database fields.
    protected static final int APPLIEDFILINGRULE_NUMOFFIELDS = 4;

    //FieldInfoEntry  for the class.
    private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[APPLIEDFILINGRULE_NUMOFFIELDS];

    // Database table name for the class.
    private static String m_strTableName = "APPLIEDFILINGRULE";

    static
    {
        RequestUtil.initFieldInfoEntries(m_arrFieldInfo,  AppliedFilingRuleFieldInfo.class,  
                "APPLIEDFILINGRULE_", APPLIEDFILINGRULE_NUMOFFIELDS);
    }

    /* Default constructor for the class.
    **/
    public AppliedFilingRuleFieldInfo() {
        super();
    }

    /* Returns the name of the database table.
    * @return String - the name of the database table for the Rules.
    **/
    public String getTableName() {
        return m_strTableName;
    }

    /* Returns the index of id field for the class.
    * @return int - index of id field for the class
    **/
    public int getIDFieldIndex() {
        return APPLIEDFILINGRULE_I_ID;
    }

    /* Returns the FieldInfoEntry array for the class.
    * @return FieldInfoEntry array - FieldInfoEntry array for the class
    **/
    public FieldInfoEntry[] getFieldInfo() {
        return m_arrFieldInfo;
    }
}
