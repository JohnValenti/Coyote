package com.verint.web.rm.rest.entity;

import java.util.ArrayList;
import java.util.List;

import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rest.util.FormatUtils;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.web.uif.system.RequestContext;

public class RequestEntity extends AbstractWFOEntity {
	private final String requestType;

	private final List<Entity> actions = new ArrayList<Entity>();
	private int employeeId;
	private String expiration;
	private String lastModifiedDate;
	private int statusCode;
	private StatusHistoryEntity statusHistoryEntity;
	private String submittedOn;
	private ValidationResultsEntity validationResultsEntity;

	public RequestEntity(String requestType) {
		this.requestType = requestType;
	}

	public List<Entity> getActions() {
		return actions;
	}

	public int getEmployeeId() {
		return employeeId;
	}

	public void setEmployeeId(int employeeId) {
		this.employeeId = employeeId;
	}

	public String getExpiration() {
		return expiration;
	}

	public void setExpiration(String expiration) {
		this.expiration = expiration;
	}

	public String getLastModifiedDate() {
		return lastModifiedDate;
	}

	public void setLastModifiedDate(String lastModifiedDate) {
		this.lastModifiedDate = lastModifiedDate;
	}

	public String getRequestType() {
		return requestType;
	}

	public int getStatusCode() {
		return statusCode;
	}

	public void setStatusCode(int statusCode) {
		this.statusCode = statusCode;
	}

	public StatusHistoryEntity getStatusHistory() {
		return statusHistoryEntity;
	}

	public void setStatusHistory(List<RequestStatusEntity> lstReqStatus) {
		this.statusHistoryEntity = new StatusHistoryEntity(lstReqStatus);
	}

	public String getSubmittedOn() {
		return submittedOn;
	}

	public void setSubmittedOn(String submittedOn) {
		this.submittedOn = submittedOn;
	}

	public ValidationResultsEntity getValidationResults() {
		return validationResultsEntity;
	}

	public void setValidationResults(List<ValidationResultEntity> lstValResults) {
		this.validationResultsEntity = new ValidationResultsEntity(lstValResults);
	}

	public void populate(RequestAggregate req, RequestContext context) {
		setEmployeeId(req.getEmployeeID().toInt());
		setExpiration(FormatUtils.dateToISO8601SecPrecisionString(req.getExpirationDate()));
		setId(RmUtils.getString(req.getID()));
		setLastModifiedDate(FormatUtils.dateToISO8601SecPrecisionString(req.getLastModifiedAt()));
		if (!StringUtil.isEmptyOrWhiteSpace(req.getRequestStatus())) {
			setStatusCode(RmUtils.getRequestStatusCode(req.getRequestStatus()));
		}
		setSubmittedOn(FormatUtils.dateToISO8601SecPrecisionString(req.getSubmittedOn()));
		RmUtils.setStatusHistory(this, req.getAuditTrail());
		RmUtils.setValidationResults(this, req.getValidationResults(true), context);
	}
}
