package com.verint.web.rm.rest.entity;

import javax.xml.bind.annotation.XmlRootElement;

import com.verint.web.rest.entity.AbstractWFOEntity;


@XmlRootElement(name = "result")
public class StatusEntity extends AbstractWFOEntity {
	private boolean success;

	public StatusEntity() { // NOSONAR
	}

	public StatusEntity(boolean success) {
		setSuccess(success);
	}

	public boolean isSuccess() {
		return success;
	}

	public void setSuccess(boolean success) {
		this.success = success;
	}
}
