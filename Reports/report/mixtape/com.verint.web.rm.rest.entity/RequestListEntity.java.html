package com.verint.web.rm.rest.entity;

import java.util.ArrayList;
import java.util.List;

import javax.xml.bind.annotation.XmlRootElement;

import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rm.rest.entity.shiftbidding.ActivityEntity;
import com.verint.web.rm.rest.entity.shiftbidding.AuctionEntity;

@XmlRootElement(name = "requests")
public class RequestListEntity extends AbstractWFOEntity {

	private final List<ActivityEntity> activities = new ArrayList<ActivityEntity>();
	private final List<AuctionEntity> auctions = new ArrayList<AuctionEntity>();
	private final List<RequestEntity> requests = new ArrayList<RequestEntity>();
	private final List<Entity> shifts = new ArrayList<Entity>();
	private final List<Entity> shiftOTExtensions = new ArrayList<Entity>();

	public List<ActivityEntity> getActivities() {
		return activities;
	}

	public List<AuctionEntity> getAuctions() {
		return auctions;
	}

	public List<RequestEntity> getRequests() {
		return requests;
	}

	public List<Entity> getShifts() {
		return shifts;
	}

	public List<Entity> getShiftOTExtensions() {
		return shiftOTExtensions;
	}
}
