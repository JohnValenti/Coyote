package com.verint.web.rm.rest.entity;

public class MessageEntity {
	private String[] args;
	private String message;
	private int type;
	private String key;

	public MessageEntity(String key, String[] args, String message, int type) {
		this.setArgs(args);
		this.key = key;
		this.message = message;
		this.type = type;
	}

	public String[] getArgs() {
		return args;
	}

	public void setArgs(String[] args) {
		this.args = args;
	}

	public String getMessage() {
		return message;
	}

	public void setMessage(String message) {
		this.message = message;
	}

	public int getType() {
		return type;
	}

	public void setType(int type) {
		this.type = type;
	}

	public String getKey() {
		return key;
	}

	public void setKey(String key) {
		this.key = key;
	}
}
