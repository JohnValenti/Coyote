package com.verint.web.fs.workrules.assignmentrules.question;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.verint.web.fs.workrules.assignmentrules.*;

import java.util.Collections;
import java.util.List;

public class ToleranceQuestion implements Question {

	private ComplexWorkRule data_assignmentRule;
	private ResponseImpl<ToleranceResponseType> response;

	public ToleranceQuestion(QuestionDataProvider dataProvider) throws BPException {
		this.data_assignmentRule = dataProvider.getAssignmentRuleBeingEdited();
		Integer tolerance = null;
		if (data_assignmentRule != null) {
			tolerance = new Integer((int) data_assignmentRule.getTolerance());
		}
		ResponseParam rp = new ResponseParam(
				ResponseParamType.Integer,
				Collections.emptyList(),
				tolerance);
		this.response = new ResponseImpl<ToleranceResponseType>(
				getQuestionType(),
				ToleranceResponseType.Tolerance,
				Collections.singletonList(rp));
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.Tolerance;
	}

	@Override
	public List<? extends Response> getResponses() {
		return Collections.singletonList(this.response);
	}

	@Override
	public Response getInitialAnswer() {
		return this.response;
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		assignmentRule.setTolerance((Integer) answer.getResponseParameters().iterator().next().getValue());
	}

	public enum ToleranceResponseType implements ResponseType {
		Tolerance;

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Collections.singletonList(ResponseParamType.Integer);
		}
	}

}
