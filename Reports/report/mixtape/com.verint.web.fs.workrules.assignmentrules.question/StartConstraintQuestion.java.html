package com.verint.web.fs.workrules.assignmentrules.question;

import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.verint.web.fs.workrules.assignmentrules.*;

import java.util.Collections;
import java.util.List;

public class StartConstraintQuestion implements Question {

	private List<ResponseImpl<StartConstraintResponseType>> data_responses;

	public StartConstraintQuestion() {
		super();
		this.data_responses = Collections.singletonList(new ResponseImpl<StartConstraintResponseType>(
				QuestionType.StartConstraint,
				StartConstraintResponseType.SameTime,
				Collections.emptyList()));
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.StartConstraint;
	}

	@Override
	public List<? extends Response> getResponses() {
		return data_responses;
	}

	@Override
	public Response getInitialAnswer() {
		return getResponses().iterator().next();
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		// Do Nothing
	}

	public enum StartConstraintResponseType implements ResponseType {
		SameTime;

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Collections.emptyList();
		}
	}
}
