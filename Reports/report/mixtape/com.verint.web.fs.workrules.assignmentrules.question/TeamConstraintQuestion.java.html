package com.verint.web.fs.workrules.assignmentrules.question;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.List;

import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule.WorkUnitCountConstraint;
import com.verint.web.fs.workrules.assignmentrules.IntegerConstraint;
import com.verint.web.fs.workrules.assignmentrules.ResponseImpl;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseParam;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamConstraint;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamType;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;

public class TeamConstraintQuestion implements Question {

	public class TeamConstraintResponse extends ResponseImpl<TeamConstraintResponseType> {
		public TeamConstraintResponse(TeamConstraintResponseType responseType,
				List<ResponseParam> parameters) {
			super(TeamConstraintQuestion.this.getQuestionType(), responseType, parameters);
		}
		
		public String toString() {
			return getResponseType().name();
		}
	}

	/**
	 * Each of these response types correspond to a WorkUnitCountConstraint
	 */
	public enum TeamConstraintResponseType implements ResponseType {
		TeamStart(WorkUnitCountConstraint.TeamStart),
		TeamStartSameDays(WorkUnitCountConstraint.TeamStartSameDays),
		TeamStartAndEnd(WorkUnitCountConstraint.TeamStartAndEnd),
		TeamStartAndEndSameDays(WorkUnitCountConstraint.TeamStartAndEndSameDays),
		TeamStartEndBreakSameDays(WorkUnitCountConstraint.TeamStartEndBreakSameDays),
		TeamOverlapEachDay(WorkUnitCountConstraint.TeamOverlapEachDay, ResponseParamType.Integer),
		TeamOverlapEachWorkingDay(WorkUnitCountConstraint.TeamOverlapEachWorkingDay, ResponseParamType.Integer),
		TeamOverlapEachWeek(WorkUnitCountConstraint.TeamOverlapEachWeek, ResponseParamType.Integer);

		private WorkUnitCountConstraint workUnitCountConstraint;
		private ResponseParamType[] paramTypes;

		TeamConstraintResponseType(WorkUnitCountConstraint constraintVal, ResponseParamType... paramTypes) {
			this.workUnitCountConstraint = constraintVal;
			this.paramTypes = paramTypes;
		}

		public WorkUnitCountConstraint getWorkUnitCountConstraint() {
			return this.workUnitCountConstraint;
		}
		
		public static TeamConstraintResponseType fromWorkUnitCountConstraint(WorkUnitCountConstraint wucc) {
			for (TeamConstraintResponseType trtrt : TeamConstraintResponseType.values()) {
				if (trtrt.workUnitCountConstraint.equals(wucc)) return trtrt;
			}
			return null;
		}

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Arrays.asList(paramTypes);
		}
	}

	private ComplexWorkRule data_assignmentRule;
	private List<TeamConstraintResponse> data_responses = null;

	/**
	 * Constructs a new TeamConstraintQuestion. 
	 * @param assignmentRule The Assignment Rule we are editing, or null for a new rule.
	 */
	public TeamConstraintQuestion(ComplexWorkRule assignmentRule) {
		this.data_assignmentRule = assignmentRule;
		populateResponses();
	}
	
	@Override
	public QuestionType getQuestionType() {
		return QuestionType.TeamConstraintType;
	}

	@Override
	public TeamConstraintResponse getInitialAnswer() {
		if (data_assignmentRule == null) return null;
		
		WorkUnitCountConstraint wucc = data_assignmentRule.getWorkUnitCountConstraint();
		for (TeamConstraintResponse response : data_responses) {
			if (response.getResponseType().getWorkUnitCountConstraint().equals(wucc)) return response;
		}
		
		return null;
	}
	
	/**
	 * Constructs a possible response of a given type. Will set the value of any response parameters
	 * if an assignment rule is specified.
	 * @param type
	 * @param assignmentRule to use to set up the response parameter values, or null for defaults.
	 * @return
	 */
	private TeamConstraintResponse constructResponse(TeamConstraintResponseType type, ComplexWorkRule assignmentRule) {
		if (type == null) return null;
		
		boolean useMinCountParam = false;
		switch (type) {
		case TeamOverlapEachDay:
		case TeamOverlapEachWorkingDay:
		case TeamOverlapEachWeek:
			useMinCountParam = true;
			break;
		default:
			useMinCountParam = false;
			break;
		}

		List<ResponseParam> parameters = Collections
				.<ResponseParam> emptyList();
		if (useMinCountParam) {
			List<ResponseParamConstraint> constraints = new ArrayList<ResponseParamConstraint>();
			constraints.add(new IntegerConstraint(0, 9999));
			parameters = Collections
					.<ResponseParam> singletonList(new ResponseParam(
							ResponseParamType.Integer,
							constraints ,
							assignmentRule == null ? 0 :assignmentRule.getMinWorkUnitCount()));
		}

		return new TeamConstraintResponse(type, parameters);		
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		if (!(answer instanceof TeamConstraintResponse)) throw new IllegalArgumentException("Response must be of type " + TeamConstraintResponse.class.getName());
		TeamConstraintResponse response = (TeamConstraintResponse) answer;
		WorkUnitCountConstraint wucc = response.getResponseType().getWorkUnitCountConstraint();
		assignmentRule.setWorkUnitCountConstraint(wucc);

		// Set the min and max counts
		assignmentRule.setMaxWorkUnitCount(1);
		List<? extends ResponseParam> paramList = response.getResponseParameters();
		// If there is a parameter, it represents the min count.
		// Only TeamOverlapEachDay, TeamOverlapEachWorkingDay, and TeamOverlapEachWeek have a parameter
		if (!paramList.isEmpty())
			assignmentRule.setMinWorkUnitCount((Integer)paramList.iterator().next().getValue());
	}

	@Override
	public List<TeamConstraintResponse> getResponses() {
		return Collections.unmodifiableList(data_responses);
	}

	private void populateResponses() {
		data_responses = new ArrayList<TeamConstraintQuestion.TeamConstraintResponse>();
		for (TeamConstraintResponseType trtrt : TeamConstraintResponseType.values()) {
			data_responses.add(constructResponse(trtrt, data_assignmentRule));
		}
	}

}
