package com.verint.web.fs.workrules.assignmentrules.question;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.List;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule.WorkUnitCountConstraint;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseImpl;
import com.verint.web.fs.workrules.assignmentrules.ResponseParam;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamType;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;

public class RuleTypeQuestion implements Question {
	private ComplexWorkRule data_assignmentRule;
	private List<RuleTypeResponse> data_responses = new ArrayList<RuleTypeQuestion.RuleTypeResponse>();
	private RuleTypeResponse data_initialAnswer;

	public RuleTypeQuestion(QuestionDataProvider dataProvider) throws BPException {
		super();
		this.data_assignmentRule = dataProvider.getAssignmentRuleBeingEdited();

		constructResponses();
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.RuleType;
	}

	@Override
	public List<RuleTypeResponse> getResponses() {
		return Collections.unmodifiableList(data_responses);
	}

	@Override
	public RuleTypeResponse getInitialAnswer() {
		return data_initialAnswer;
	}

	/**
	 * This question does not affect much in the Assignment Rule directly. It sets the offRule property and sets the
	 * minWorkUnitCount and maxWorkUnitCount if the answer is a number of seats.
	 */
	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		RuleTypeResponseType answerType = (RuleTypeResponseType) answer.getResponseType();
		assignmentRule.setOffRule(answerType.equals(RuleTypeResponseType.Off));
	}

	private void constructResponses() {
		RuleTypeResponse mustWork = new RuleTypeResponse(
				RuleTypeResponseType.MustWork,
				Collections.<ResponseParam> emptyList());
		data_responses.add(mustWork);

		RuleTypeResponse off = new RuleTypeResponse(RuleTypeResponseType.Off, Collections.<ResponseParam> emptyList());
		data_responses.add(off);

		RuleTypeResponse team = new RuleTypeResponse(
				RuleTypeResponseType.Team,
				Collections.<ResponseParam> emptyList());
		data_responses.add(team);

		RuleTypeResponse start = new RuleTypeResponse(
				RuleTypeResponseType.Start,
				Collections.<ResponseParam> emptyList());
		data_responses.add(start);

		// Set the selected response
		if (data_assignmentRule != null) {
			if (data_assignmentRule.isOffRule()) {
				data_initialAnswer = off;
			} else {
				data_initialAnswer = mustWork;
			}

			WorkUnitCountConstraint constraint = data_assignmentRule.getWorkUnitCountConstraint();
			switch (constraint) {
			case TeamStart:
			case TeamStartAndEnd:
			case TeamStartSameDays:
			case TeamStartAndEndSameDays:
			case TeamStartEndBreakSameDays:
			case TeamOverlapEachDay:
			case TeamOverlapEachWorkingDay:
			case TeamOverlapEachWeek:
				data_initialAnswer = team;
				break;
			case MustStart:
				data_initialAnswer = start;
				break;
			}
		}
	}

	public enum RuleTypeResponseType implements ResponseType {
		MustWork, Off, Team, Start;

		private ResponseParamType[] paramTypes;

		RuleTypeResponseType(ResponseParamType... paramTypes) {
			this.paramTypes = paramTypes;
		}

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Arrays.asList(paramTypes);
		}
	}

	public class RuleTypeResponse extends ResponseImpl<RuleTypeResponseType> {
		public RuleTypeResponse(RuleTypeResponseType responseType, List<ResponseParam> parameters) {
			super(QuestionType.RuleType, responseType, parameters);
		}

		@Override
		public String toString() {
			return getResponseType().toString();
		}
		
	}
}
