package com.verint.web.fs.workrules.assignmentrules.question;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit.WorkUnitType;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseImpl;
import com.verint.web.fs.workrules.assignmentrules.ResponseParam;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamType;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;

public class TeamMatchQuestion implements Question {
	private List<TeamMatchResponse> data_responses = new ArrayList<TeamMatchQuestion.TeamMatchResponse>();
	private ComplexWorkRule data_assignmentRule;
	private Response data_initialAnswer;

	public TeamMatchQuestion(ComplexWorkRule assignmentRule) {
		super();
		this.data_assignmentRule = assignmentRule;
		constructResponses();
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.TeamMatchType;
	}

	@Override
	public List<? extends Response> getResponses() {
		return data_responses;
	}

	@Override
	public Response getInitialAnswer() {
		return data_initialAnswer;
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		if (!(answer instanceof TeamMatchResponse)) {
			throw new IllegalArgumentException("Answer must be of type " + TeamMatchResponse.class.getName() + ". Not "
					+ answer.getClass().getName());
		}
		
		TeamMatchResponse tmr = (TeamMatchResponse) answer;
		WorkUnit wu = assignmentRule.getWorkUnit();
		if (wu == null) {
			wu = new WorkUnit();
			assignmentRule.setWorkUnit(wu);
		}
		wu.setWorkUnitType(tmr.getResponseType().getWorkUnitType());
	}

	private void constructResponses() {
		TeamMatchResponse allOther = new TeamMatchResponse(TeamMatchResponseType.AllOther);
		data_responses.add(allOther);

		TeamMatchResponse supervisor = new TeamMatchResponse(TeamMatchResponseType.Supervisor);
		data_responses.add(supervisor);

		TeamMatchResponse teamLead = new TeamMatchResponse(TeamMatchResponseType.TeamLead);
		data_responses.add(teamLead);

		// Set initial answer
		if (data_assignmentRule != null && data_assignmentRule.getWorkUnit() != null) {
			WorkUnitType wut = data_assignmentRule.getWorkUnit().getWorkUnitType();
			for (TeamMatchResponse response : data_responses) {
				if (response.getResponseType().getWorkUnitType() == wut) {
					data_initialAnswer = response;
				}
			}
		}
	}

	public enum TeamMatchResponseType implements ResponseType {
		AllOther(WorkUnitType.Teams), Supervisor(WorkUnitType.SupervisorTeams), TeamLead(WorkUnitType.LeadTeams);

		private WorkUnitType wut;

		TeamMatchResponseType(WorkUnitType wut) {
			this.wut = wut;
		}

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Collections.emptyList();
		}

		public WorkUnitType getWorkUnitType() {
			return wut;
		}
	}

	public class TeamMatchResponse extends ResponseImpl<TeamMatchResponseType> {
		public TeamMatchResponse(TeamMatchResponseType responseType) {
			super(QuestionType.TeamMatchType, responseType, Collections.<ResponseParam> emptyList());
		}
	}
}
