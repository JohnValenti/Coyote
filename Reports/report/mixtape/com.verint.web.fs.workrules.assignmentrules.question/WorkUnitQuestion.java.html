package com.verint.web.fs.workrules.assignmentrules.question;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.datatypes.DayOfWeek;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit.WorkUnitType;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.verint.web.fs.workrules.assignmentrules.*;
import com.verint.web.fs.workrules.assignmentrules.question.QuestionFactory.QuestionFactoryState;
import com.verint.web.fs.workrules.assignmentrules.question.RuleTypeQuestion.RuleTypeResponseType;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;

public class WorkUnitQuestion implements Question {

	private List<WorkUnitResponse> data_responses = new ArrayList<WorkUnitQuestion.WorkUnitResponse>();
	private WorkUnitResponse data_initialAnswer;
	private QuestionDataProvider data_provider;
	private QuestionFactoryState questionFactoryState;

	public WorkUnitQuestion(QuestionDataProvider dataProvider, QuestionFactoryState state) throws BPException {
		super();
		this.data_provider = dataProvider;
		this.questionFactoryState = state;

		constructResponses();
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.WorkUnit;
	}

	@Override
	public List<WorkUnitResponse> getResponses() {
		return data_responses;
	}

	@Override
	public Response getInitialAnswer() {
		return data_initialAnswer;
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		if (assignmentRule == null) {
			return;
		}

		int min = assignmentRule.getMinWorkUnitCount();
		int max = assignmentRule.getMaxWorkUnitCount();

		WorkUnit wu = assignmentRule.getWorkUnit();
		if (wu == null) {
			wu = new WorkUnit();
			assignmentRule.setWorkUnit(wu);
		}
		wu.setWorkUnitType(((WorkUnitResponse) answer).getResponseType().getWorkUnitType());

		// The setter methods for the counts on the assignment rule
		// will modify the underlying value based on work unit type,
		// so make sure that they still are the right values, since
		// the min and max count question probably already has set
		// its value on the assignment rule.
		assignmentRule.setMinWorkUnitCount(min);
		assignmentRule.setMaxWorkUnitCount(max);

		switch ((WorkUnitResponseType) answer.getResponseType()) {
			case Activities:
				@SuppressWarnings("unchecked")
				List<Activity> activities = (List<Activity>) answer.getResponseParameters().iterator().next().getValue();
				wu.setActivities(activities);
				break;
			case DayOfWeek:
				assignmentRule.getWorkUnit().setDayOfWeek((DayOfWeek) answer.getResponseParameters().iterator().next().getValue());
				break;
			case Shift:
				assignmentRule.getWorkUnit().setShiftID(((Shift) answer.getResponseParameters().iterator().next().getValue()).getID());
				break;
		}

	}

	private void constructResponses() throws BPException {
		// This logic was ported from C++. In Sentence.cpp, method GetWorkUnit
		// Modified per Jason to remove Min Shift Spacing, Min Shift Event Spacing, Consecutive Weekends, Dollars of Wage Within This Group

		ComplexWorkRule assignmentRule = data_provider.getAssignmentRuleBeingEdited();

		// Shift
		if (data_provider.getRuleType() == RuleTypeResponseType.MustWork && data_provider.hasShifts()) {
			Shift shift = null;
			if (assignmentRule != null && assignmentRule.getWorkUnit() != null) {
				WorkUnit wu = assignmentRule.getWorkUnit();
				if (wu.getShiftID() != null) {
					shift = data_provider.getShift(wu.getShiftID());
				}
			}

			ResponseParam shiftParam = new ResponseParam(
					ResponseParamType.Shift,
					Collections.singletonList(new OrgConstraint(data_provider.getOrgID())),
					shift);
			WorkUnitResponse shiftResponse = new WorkUnitResponse(
					WorkUnitResponseType.Shift,
					Collections.singletonList(shiftParam));
			this.data_responses.add(shiftResponse);
		}

		// Activity
		if (data_provider.getRuleType() == RuleTypeResponseType.MustWork && data_provider.hasActivities()) {
			ArrayList<Activity> selectedActivities = new ArrayList<Activity>();
			if (assignmentRule != null && assignmentRule.getWorkUnit() != null) {
				WorkUnit wu = assignmentRule.getWorkUnit();
				if (wu.getActivities() != null) {
					selectedActivities.addAll(wu.getActivities());
				}
			}
			ResponseParam activityParam = new ResponseParam(
					ResponseParamType.Activities,
					Collections.singletonList(new OrgConstraint(data_provider.getOrgID())),
					selectedActivities);
			WorkUnitResponse activities = new WorkUnitResponse(
					WorkUnitResponseType.Activities,
					Collections.singletonList(activityParam));
			this.data_responses.add(activities);
		}

		// Any Day
		WorkUnitResponse day = new WorkUnitResponse(
				WorkUnitResponseType.AnyDay,
				Collections.emptyList());
		this.data_responses.add(day);

		// Day of Week
		ResponseParam dowParam = new ResponseParam(
				ResponseParamType.DayOfWeek,
				Collections.emptyList(),
				assignmentRule == null ? DayOfWeek.Monday : assignmentRule.getWorkUnit().getDayOfWeek());
		WorkUnitResponse dayOfWeek = new WorkUnitResponse(
				WorkUnitResponseType.DayOfWeek,
				Collections.singletonList(dowParam));
		this.data_responses.add(dayOfWeek);

		if (data_provider.getRuleType() == RuleTypeResponseType.MustWork) {
			// Partial Weekend
			WorkUnitResponse partialWeekend = new WorkUnitResponse(WorkUnitResponseType.PartialWeekend);
			this.data_responses.add(partialWeekend);

			// Full Weekend
			WorkUnitResponse fullWeekend = new WorkUnitResponse(WorkUnitResponseType.FullWeekend);
			this.data_responses.add(fullWeekend);

			// Hours
			WorkUnitResponse hours = new WorkUnitResponse(WorkUnitResponseType.Hours);
			this.data_responses.add(hours);
		}

		if (data_provider.getRuleType() != RuleTypeResponseType.MustWork) {
			// Weekend
			WorkUnitResponse weekend = new WorkUnitResponse(WorkUnitResponseType.Weekend);
			this.data_responses.add(weekend);

			// Full Weeekend, labeled as partial
			WorkUnitResponse fullWeekendLabeledAsPartial = new WorkUnitResponse(
					WorkUnitResponseType.FullWeekendLabeledAsPartial);
			this.data_responses.add(fullWeekendLabeledAsPartial);
		}

		if (data_provider.getRuleType() == RuleTypeResponseType.MustWork) {
			if (!this.questionFactoryState.contains(QuestionFactory.StateCondition.RequiresTolerance)) {
				WorkUnitResponse consecutiveDay = new WorkUnitResponse(WorkUnitResponseType.ConsecutiveDay);
				this.data_responses.add(consecutiveDay);
			}
		}

		// find initial answer
		if (assignmentRule != null && assignmentRule.getWorkUnit() != null
				&& assignmentRule.getWorkUnit().getWorkUnitType() != null) {
			WorkUnitType wut = assignmentRule.getWorkUnit().getWorkUnitType();
			for (WorkUnitResponse wur : data_responses) {
				WorkUnitResponseType wurt = wur.getResponseType();
				if (wurt.getWorkUnitType().equals(wut)) {
					data_initialAnswer = wur;
					break;
				}
			}
		}
	}

	public enum WorkUnitResponseType implements ResponseType {
		Shift(WorkUnitType.Shift, ResponseParamType.Shift),
		AnyDay(WorkUnitType.AnyDay),
		DayOfWeek(WorkUnitType.DayOfWeek, ResponseParamType.DayOfWeek),
		Hours(WorkUnitType.Hours),
		FullWeekend(WorkUnitType.FullWeekend),
		FullWeekendLabeledAsPartial(WorkUnitType.FullWeekend),
		PartialWeekend(WorkUnitType.PartialWeekend),
		Weekend(WorkUnitType.Weekend),
		Rotation(WorkUnitType.Rotation),
		ConsecutiveDay(WorkUnitType.ConsecutiveDay),
		ConsecutiveDayOfWeek(WorkUnitType.ConsecutiveDayOfWeek),
		ConsecutiveShift(WorkUnitType.ConsecutiveShift),
		ConsecutiveFullWeekend(WorkUnitType.ConsecutiveFullWeekend),
		ConsecutivePartialWeekend(WorkUnitType.ConsecutivePartialWeekend),
		ConsecutiveDayOff(WorkUnitType.ConsecutiveDayOff),
		ConsecutiveDayOfWeekOff(WorkUnitType.ConsecutiveDayOfWeekOff),
		ConsecutiveFullWeekendOff(WorkUnitType.ConsecutiveFullWeekendOff),
		ConsecutivePartialWeekendOff(WorkUnitType.ConsecutivePartialWeekendOff),
		MustStart(WorkUnitType.MustStart),
		MustStartBetween(WorkUnitType.MustStartBetween),
		MustEnd(WorkUnitType.MustEnd),
		Teams(WorkUnitType.Teams),
		SupervisorTeams(WorkUnitType.SupervisorTeams),
		LeadTeams(WorkUnitType.LeadTeams),
		MustEndBetween(WorkUnitType.MustEndBetween),
		NumSeats(WorkUnitType.NumSeats),
		MinShiftSpacing(WorkUnitType.MinShiftSpacing),
		MinShiftEventSpacing(WorkUnitType.MinShiftEventSpacing),
		Dollars(WorkUnitType.Dollars),
		Activities(WorkUnitType.Activity, ResponseParamType.Activities);

		private WorkUnitType data_type;
		private ResponseParamType[] data_paramTypes;

		WorkUnitResponseType(WorkUnitType type, ResponseParamType... paramTypes) {
			this.data_type = type;
			this.data_paramTypes = paramTypes;
		}

		private WorkUnitType getWorkUnitType() {
			return data_type;
		}

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Arrays.asList(data_paramTypes);
		}
	}

	public static class WorkUnitResponse extends ResponseImpl<WorkUnitResponseType> {
		public WorkUnitResponse(WorkUnitResponseType responseType) {
			this(responseType, Collections.emptyList());
		}

		public WorkUnitResponse(WorkUnitResponseType responseType, List<ResponseParam> parameters) {
			super(QuestionType.WorkUnit, responseType, parameters);
		}

	}

}
