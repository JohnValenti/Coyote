package com.verint.web.fs.workrules.assignmentrules.question;

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit.WorkUnitType;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule.WorkUnitCountConstraint;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.verint.web.fs.workrules.assignmentrules.OrgConstraint;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseImpl;
import com.verint.web.fs.workrules.assignmentrules.ResponseParam;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamConstraint;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamType;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;

public class StartUnitQuestion implements Question {

	private ComplexWorkRule data_assignmentRule;
	private ID data_orgID;
	private List<StartUnitResponse> data_responses = new ArrayList<StartUnitResponse>();
	private StartUnitResponse data_initialAnswer;
	private QuestionDataProvider data_provider;

	public StartUnitQuestion(QuestionDataProvider dataProvider) throws BPException {
		super();
		this.data_provider = dataProvider;
		this.data_assignmentRule = dataProvider.getAssignmentRuleBeingEdited();
		this.data_orgID = dataProvider.getOrgID();

		constructResponses();
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.StartUnit;
	}

	@Override
	public List<StartUnitResponse> getResponses() {
		return data_responses;
	}

	@Override
	public Response getInitialAnswer() {
		return data_initialAnswer;
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		if (!(answer instanceof StartUnitResponse))
			throw new IllegalArgumentException("Expected answer type: " + StartUnitResponse.class.getName()
					+ ". Actual: " + answer.getClass().getName());

		assignmentRule.setWorkUnitCountConstraint(WorkUnitCountConstraint.MustStart);
		assignmentRule.setMinWorkUnitCount(0);
		assignmentRule.setMaxWorkUnitCount(1);

		StartUnitResponse sur = (StartUnitResponse) answer;
		WorkUnit wu = assignmentRule.getWorkUnit();
		if (wu == null) {
			wu = new WorkUnit();
			assignmentRule.setWorkUnit(wu);
		}

		switch (sur.getResponseType()) {
		case AnyShift:
			wu.setWorkUnitType(WorkUnitType.MustStart);
			break;
		case AnyShiftAndEvent:
			assignmentRule.setMinWorkUnitCount(1);
			wu.setWorkUnitType(WorkUnitType.MustStart);
			break;
		case Shift: {
			wu.setWorkUnitType(WorkUnitType.MustStart);
			Shift shift = (Shift) sur.getResponseParameters().iterator().next().getValue();
			wu.setShiftID(shift.getID());
			break;
		}
		case ShiftAndEvent: {
			wu.setWorkUnitType(WorkUnitType.MustStart);
			assignmentRule.setMinWorkUnitCount(1);
			Shift shift = (Shift) sur.getResponseParameters().iterator().next().getValue();
			wu.setShiftID(shift.getID());
			break;
		}
		default:
			throw new IllegalArgumentException("Unexpected answer response type: " + sur.getResponseType());
		}

	}

	private void constructResponses() throws BPException {
		OrgConstraint orgConstraint = new OrgConstraint(this.data_orgID);

		// All Shifts
		StartUnitResponse allShift = new StartUnitResponse(StartUnitResponseType.AnyShift);
		this.data_responses.add(allShift);

		// Any SHift and its Shift Events
		StartUnitResponse anyShiftAndEvent = new StartUnitResponse(StartUnitResponseType.AnyShiftAndEvent);
		this.data_responses.add(anyShiftAndEvent);

		Shift shift = null;
		if (data_provider.hasShifts()) {
			// Shift X
			if (data_assignmentRule != null) {
				WorkUnit wu = data_assignmentRule.getWorkUnit();
				if (wu != null && wu.getShiftID() != null) {
					shift = data_provider.getShift(wu.getShiftID());
				}
			}
			
			ResponseParam specificShiftRP = new ResponseParam(
					ResponseParamType.Shift,
					Collections.<ResponseParamConstraint> singletonList(orgConstraint),
					shift);
			StartUnitResponse specificShift = new StartUnitResponse(
					StartUnitResponseType.Shift,
					Collections.<ResponseParam> singletonList(specificShiftRP));
			this.data_responses.add(specificShift);
	
			// Shift X and its Shift Events
			ResponseParam specificShiftAndEventRP = new ResponseParam(
					ResponseParamType.Shift,
					Collections.<ResponseParamConstraint> singletonList(orgConstraint),
					shift);
			StartUnitResponse specificShiftAndEvent = new StartUnitResponse(
					StartUnitResponseType.ShiftAndEvent,
					Collections.<ResponseParam> singletonList(specificShiftAndEventRP));
			this.data_responses.add(specificShiftAndEvent);
		}
		
		// find initial answer.
		// This is determined by the ComplexWorkRule and WorkUnit db tables.
		// If the WorkUnit table has a ShiftID set, it is a specific shift.
		// If it has a ShiftEventID set, it is a specific shift event.
		// If the ComplexWorkRule table has a MinWorkCount of 1 it means shift (specific or general) with events.
		// A value of 0 means without events
		if (data_assignmentRule != null) {
			StartUnitResponseType answerType = null;

			if (shift != null) {
				if (data_assignmentRule.getMinWorkUnitCount() == 1) {
					answerType = StartUnitResponseType.ShiftAndEvent;
				} else {
					answerType = StartUnitResponseType.Shift;
				}
			} else {
				if (data_assignmentRule.getMinWorkUnitCount() == 1) {
					// shiftID == null && shiftEventID == null
					answerType = StartUnitResponseType.AnyShiftAndEvent;
				} else {
					answerType = StartUnitResponseType.AnyShift;
				}
			}

			for (StartUnitResponse sur : data_responses) {
				if (sur.getResponseType().equals(answerType)) {
					this.data_initialAnswer = sur;
					break;
				}
			}
		}
	}

	public enum StartUnitResponseType implements ResponseType {
		AnyShift, AnyShiftAndEvent, Shift(ResponseParamType.Shift), ShiftAndEvent(ResponseParamType.Shift);

		private List<ResponseParamType> paramTypes = new ArrayList<ResponseParamType>();

		StartUnitResponseType() {
		}

		StartUnitResponseType(ResponseParamType paramType) {
			paramTypes.add(paramType);
		}

		@Override
		public List<ResponseParamType> getParamTypes() {
			return paramTypes;
		}
	}

	private static class StartUnitResponse extends ResponseImpl<StartUnitResponseType> {
		public StartUnitResponse(StartUnitResponseType responseType) {
			super(QuestionType.StartUnit, responseType, Collections.<ResponseParam> emptyList());
		}

		public StartUnitResponse(StartUnitResponseType responseType, List<ResponseParam> parameters) {
			super(QuestionType.StartUnit, responseType, parameters);
		}
	}
}
