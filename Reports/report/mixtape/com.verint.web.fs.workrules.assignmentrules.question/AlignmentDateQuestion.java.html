package com.verint.web.fs.workrules.assignmentrules.question;

import java.util.Calendar;
import java.util.Collections;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.CalendarPeriod;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.verint.web.fs.workrules.OrgDateUtil;
import com.verint.web.fs.workrules.assignmentrules.DayOfWeekConstraint;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseImpl;
import com.verint.web.fs.workrules.assignmentrules.ResponseParam;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamConstraint;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamType;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;
import com.verint.web.fs.workrules.assignmentrules.question.PeriodQuestion.PeriodResponseType;

public class AlignmentDateQuestion implements Question {

	private ResponseImpl<AlignmentDateResponseType> data_response;
	private ComplexWorkRule data_assignmentRule;

	public AlignmentDateQuestion(QuestionDataProvider dataProvider) throws BPException {
		super();
		this.data_assignmentRule = dataProvider.getAssignmentRuleBeingEdited();
		Organization org = dataProvider.getOrganization();
		DayOfWeekConstraint dowC = null;
		if (dataProvider.getPeriodType().equals(PeriodResponseType.Week))
			dowC = new DayOfWeekConstraint(org.getWeekStartDay());
		
		// The Alignment date is stored as a date that has the time value set to zero.
		Calendar orgStart = Calendar.getInstance(org.getTimeZone());
		orgStart.setTime(OrgDateUtil.snapDateToWeekStart(org, new Date()));
		orgStart.set(Calendar.HOUR_OF_DAY, 0);
		orgStart.set(Calendar.MINUTE, 0);
		orgStart.set(Calendar.SECOND, 0);
		orgStart.set(Calendar.MILLISECOND, 0);
		
		LocalDate value = new LocalDate(orgStart.getTime(), org.getTimeZone());
		if (data_assignmentRule != null && data_assignmentRule.getCalendarPeriod() != null) {
			value = data_assignmentRule.getCalendarPeriod().getAlignmentDate();
		}
		ResponseParam rp = new ResponseParam(
				ResponseParamType.Date,
				dowC==null?Collections.<ResponseParamConstraint>emptyList():Collections.<ResponseParamConstraint> singletonList(dowC),
				value);
		data_response = new ResponseImpl<AlignmentDateResponseType>(
				QuestionType.AlignmentDate,
				AlignmentDateResponseType.AlignmentDate,
				Collections.singletonList(rp));
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.AlignmentDate;
	}

	@Override
	public List<? extends Response> getResponses() {
		return Collections.singletonList(data_response);
	}

	@Override
	public Response getInitialAnswer() {
		return data_response;
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		CalendarPeriod cp = assignmentRule.getCalendarPeriod();
		if (cp == null) {
			cp = new CalendarPeriod();
			assignmentRule.setCalendarPeriod(cp);
		}
		cp.setAlignmentDate((LocalDate)answer.getResponseParameters().iterator().next().getValue());
	}

	public enum AlignmentDateResponseType implements ResponseType {
		AlignmentDate;

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Collections.singletonList(ResponseParamType.Date);
		}

	}
}
