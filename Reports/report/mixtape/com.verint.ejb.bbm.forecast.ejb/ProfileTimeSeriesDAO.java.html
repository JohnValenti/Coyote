package com.verint.ejb.bbm.forecast.ejb;

import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.dao.DAOBaseWithAutoIDMapping;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.workload.model.QueueFieldInfo;
import com.verint.ejb.bbm.forecast.model.ForecastProfileFieldInfo;
import com.verint.ejb.bbm.forecast.model.ProfileTimeSeries;
import com.verint.ejb.bbm.forecast.model.ProfileTimeSeriesFieldInfo;
import com.verint.ejb.bbm.forecast.model.ProfileTimeSeriesFieldInfo.Column;

public class ProfileTimeSeriesDAO extends DAOBaseWithAutoIDMapping<ProfileTimeSeries> {
	private static final ProfileTimeSeriesFieldInfo fieldInfo = new ProfileTimeSeriesFieldInfo();

	public ProfileTimeSeriesDAO() {
		super();
	}

	@Override
	protected Map<Integer, ? extends FieldInfo> getForeignKeyFieldInfo() {
		Map<Integer, FieldInfo> retVal = new HashMap<Integer, FieldInfo>();
		retVal.put(Column.QueueID.getIndex(), new QueueFieldInfo());
		retVal.put(Column.ForecastProfileID.getIndex(), new ForecastProfileFieldInfo());
		return retVal;
	}

	public ProfileTimeSeriesDAO(Jdmo dmo, boolean isDmoCreated) {
		super(dmo, isDmoCreated);
	}

	public ProfileTimeSeriesDAO(Jdmo dmo) {
		super(dmo);
	}

	@Override
	protected ProfileTimeSeries createValueObject() {
		return new ProfileTimeSeries();
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return fieldInfo;
	}
}
