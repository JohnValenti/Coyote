	public Collection<ForecastProfileProject> getActiveForecastProfileProjects(Collection <ID> spQueueIDs) throws JdmoException, BbmFinderException
	{
		try
		{
			if (spQueueIDs == null) 
                return Collections.emptyList();
			
            //select * from FORECASTPROFILEPROJECT A where A.FORECASTPROFILEID in 
            //(select ID from FORECASTPROFILE FP where FP.SPQUEUEID in ('000009e7.000011') and FP.FORECASTINSTANCEID is null)
            
			StringBuffer strWhere = new StringBuffer(200);
			strWhere.append(" A.FORECASTPROFILEID in (select ID from FORECASTPROFILE FP where FP.SPQUEUEID in ");
			strWhere.append(m_dmo.createInClause(DAOUtil.mapIDsToDEIDs(spQueueIDs, new SPQueueFieldInfo()).values()));
			strWhere.append(" and FP.FORECASTINSTANCEID is null)");

			StringBuffer strOrderBy = new StringBuffer(200);
			strOrderBy.append("ORDER BY A.QUEUEID,A.STARTTIME ASC ");
			
			
			
			Collection<ForecastProfileProject> projects = getObjects(strWhere.toString(),strOrderBy.toString());
			return projects;
		}
		catch(JdmoException e)
		{
			e.printStackTrace();
			throw new BbmFinderException(e);
		}		
	}
	
	/**
	 * Deletes all ForecastProfileProject records with the given forecast profile IDs.
	 */
	public void deleteByForecastProfileIDs(Collection<ID> forecastProfileIDs) throws BbmRemoveException{
		try {			
			String where = "SELECT ID FROM " + getFieldInfo().getTableName() + " A WHERE A.FORECASTPROFILEID IN " +
				getDMO().createInClause(forecastProfileIDs);
			deleteObjects(where);
		} catch (JdmoException ex) {
			throw new BbmRemoveException(ex);
		}	
	}	
}
