package com.verint.ejb.bbm.forecast.ejb;


import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmTimeSeriesException;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.util.TraceOperator;
import com.verint.ejb.bbm.forecast.model.ForecastProfileList;

import java.util.*;

/**
 * Helper class to hold the result of data relating to a forecast recalculation for a collection of ForecastProfiles.
 *
 * Holds both forecast time series data and a collection of ForecastProfileLists for outbound queues.
 */
class ForecastCalculationResult {

	private final Map<ID, Collection<TraceCube>> calculatedForecastData;
	private final Map<ID, Collection<ForecastProfileList>> calculatedForecastProfileLists;

	/**
	 * Constructor which stores the calculated forecast data and ForecastProfileLists for each SPQueue that was being
	 * reforecasted.
	 *
	 * @param calculatedForecastData - Key - spQueueId, Value - A forecast tracecube for each forecast profile associated
	 *                              to the SPQueue
	 * @param calculatedForecastProfileLists - Key - spQueueId, Value - Collection of calculated forecast lists for
	 *                                      the associated SPQueue
	 */
	ForecastCalculationResult(Map<ID, Collection<TraceCube>> calculatedForecastData,
			Map<ID, Collection<ForecastProfileList>> calculatedForecastProfileLists) {

		this.calculatedForecastData = calculatedForecastData;
		this.calculatedForecastProfileLists = calculatedForecastProfileLists;
	}

	/**
	 * Returns a set of concatenated forecasted trace cubes of the recalculated forecast data.  There will be
	 * one concatenated trace cube per SPQueue.  Multi-week ranges are concatenated into one single cube.
	 */
	Map<ID, TraceCube> getConcatenatedForecastData() throws BbmTimeSeriesException {
		Map<ID, TraceCube> forecastDataBySpQueueId = new HashMap<>();
		for (ID spQueueId : calculatedForecastData.keySet()) {
			forecastDataBySpQueueId.put(spQueueId,
					TraceOperator.concatenateTraceCubes(calculatedForecastData.get(spQueueId)));
		}

		return forecastDataBySpQueueId;
	}

	/**
	 * Returns a collection of all ForecastProfileLists that have been recalculated/generated.  The returned collection
	 * may contain ForecastProfileLists belonging to different SPQueues.
	 */
	Collection<ForecastProfileList> getForecastProfileLists() {
		Set<ForecastProfileList> retVal = new HashSet<>();
		for (ID spQueueId : calculatedForecastProfileLists.keySet()) {
			retVal.addAll(calculatedForecastProfileLists.get(spQueueId));
		}
		return retVal;
	}
}
