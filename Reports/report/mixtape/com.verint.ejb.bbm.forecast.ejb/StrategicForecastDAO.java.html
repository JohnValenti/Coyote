package com.verint.ejb.bbm.forecast.ejb;

import com.bluepumpkin.common.jdmo.JdmoException;

/**
 * DAO for the StrategicForecast table.  Extending {@link DAOBase} is largely
 * ceremonial.  About all this class inherits is the {@link Jdmo}.
 */
public class StrategicForecastDAO extends DAOBase<StrategicForecast> {

	private static final StrategicForecastFieldInfo m_fieldInfo = new StrategicForecastFieldInfo();

	public StrategicForecastDAO() {
		super();
	}

	public StrategicForecastDAO(Jdmo jdmo) {
		super(jdmo);
	}

	/**
	 * Returns a trace cube containing any strategic forecast volume or AHT values
	 * that exist within the specified time range.  Any part of the trace cube
	 * for which values were not available will contain Trace.TRACENA.
	 *
	 * @param queueId
	 * @param startTime
	 * @param endTime
	 * @return
	 * @throws BbmTimeSeriesException
	 * @throws JdmoException
	 */
	public ForecastTraceCube getStrategicForecastTraceCube(ID queueId, Date startTime, Date endTime)
			throws BbmTimeSeriesException, BbmFinderException, JdmoException {
		ForecastTraceCube forecast = new ForecastTraceCube(queueId, startTime, endTime);
		forecast.initTraceValue(Trace.CV, Trace.TRACENA);
		forecast.initTraceValue(Trace.AHT, Trace.TRACENA);
		JdmoRowset forecastRows = getStrategicForecastRowset(queueId, startTime, endTime);
		while (forecastRows.next()) {
			Date time = TimeZoneUtil.toDate(forecastRows.getTimestamp(StrategicForecastFieldInfo.Column.TIME.name()));
			int volume = forecastRows.getInt(StrategicForecastFieldInfo.Column.CALLVOLUME.name());
			int aht = forecastRows.getInt(StrategicForecastFieldInfo.Column.AHT.name());
			forecast.setTraceValue(Trace.CV, volume, time, true);
			forecast.setTraceValue(Trace.AHT, aht, time, true);
		}
		//compute cv_vh trace type needed for aggregations. the ID parameter is unused in the calcComputedTraceValues function.
		forecast.calcComputedTraceValues(new ID());
		return forecast;
	}

	/**
	 * Queries the {@code STRATEGICFORECAST} table for the strategic forecast entries
	 * that have the specified queue ID and fall within the specified date range.
	 *
	 * @param queueId
	 * @param startTime
	 * @param endTime
	 * @return
	 * @throws JdmoException
	 */
	private JdmoRowset getStrategicForecastRowset(ID queueId, Date startTime, Date endTime)
			throws JdmoException, BbmFinderException {
		// SELECT TIME, CALLVOLUME, AHT from STRATEGICFORECAST
		//    where QUEUEID = (queueId) and TIME < (endTime) and not TIME < (startTime)  
		StringBuffer queryBuffer = new StringBuffer();
		queryBuffer.append("select ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.TIME.name());
		queryBuffer.append(", ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.CALLVOLUME.name());
		queryBuffer.append(", ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.AHT.name());
		queryBuffer.append(" from ");
		queryBuffer.append(m_fieldInfo.getTableName());
		queryBuffer.append(" where ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.QUEUEID.name());
		queryBuffer.append(" = ");
		JdmoUtil.asSqlLiteral(DAOUtil.mapIDToDEID(queueId, new QueueFieldInfo()), queryBuffer);
		queryBuffer.append(" and ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.TIME.name());
		queryBuffer.append(" < ");
		JdmoUtil.asSqlLiteral(endTime, queryBuffer);
		queryBuffer.append(" and not ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.TIME.name());
		queryBuffer.append(" < ");
		JdmoUtil.asSqlLiteral(startTime, queryBuffer);
		queryBuffer.append(" order by ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.TIME.name());
		return m_dmo.createRowset(queryBuffer.toString());
	}

	public void deleteStrategicForecasts(ID queueId, Date startTime, Date endTime)
			throws JdmoException, BbmFinderException, BbmRemoveException {
		StringBuffer queryBuffer = new StringBuffer();
		queryBuffer.append("delete ");
		queryBuffer.append(" from ");
		queryBuffer.append(m_fieldInfo.getTableName());
		queryBuffer.append(" where ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.QUEUEID.name());
		queryBuffer.append(" = ");
		JdmoUtil.asSqlLiteral(DAOUtil.mapIDToDEID(queueId, new QueueFieldInfo()), queryBuffer);
		queryBuffer.append(" and ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.TIME.name());
		queryBuffer.append(" < ");
		JdmoUtil.asSqlLiteral(endTime, queryBuffer);
		queryBuffer.append(" and not ");
		queryBuffer.append(StrategicForecastFieldInfo.Column.TIME.name());
		queryBuffer.append(" < ");
		JdmoUtil.asSqlLiteral(startTime, queryBuffer);
		m_dmo.execute(queryBuffer.toString());
	}


	@Override
	protected StrategicForecast createValueObject() {
		return new StrategicForecast();
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}
}
