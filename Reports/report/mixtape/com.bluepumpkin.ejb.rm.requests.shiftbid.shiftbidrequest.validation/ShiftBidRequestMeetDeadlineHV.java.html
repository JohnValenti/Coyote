/**
 * Verifies if another ShiftBidRequest has the same set of biddable schedules as this.
 *
 * @author tuanphan
 *
 * To change this generated comment go to
 * Window>Preferences>Java>Code Generation>Code and Comments
 */
public class ShiftBidRequestMeetDeadlineHV implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

    private static final String m_className = ShiftBidRequestMeetDeadlineHV.class.getName();

    private static Category m_cat = Log.initCategory(m_className);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

        try  {
        	
            ShiftBidRequest sbr = (ShiftBidRequest) validatable;
            Date submittedTime = sbr.getSubmittedOn();
            Date deadlineTime;
            // QC125415 If ShiftBidRequestMeetDeadlineHV_disable is true, we check auction deadline, otherwise, personal deadline.
            if("true".equalsIgnoreCase(BbmManagerFactory.getDBConfigManager().getValue("ShiftBidRequestMeetDeadlineHV_disable"))){
            	deadlineTime = sbr.getOptMethods().getShiftBidAuction().getAuctionDeadLine();
            } else {
            	deadlineTime = sbr.getOptMethods().getShiftBidder().getDeadlineDate();
            }
            
            if(submittedTime.after(deadlineTime)) {
		        ValidationResult result = ValidationUtil.setHardValidationResult(sbr, 
		        	RmEjbBundleKey.SBR_REQUEST_AFTER_DEADLINE, deadlineTime, m_className);
		        return result;
            }
            return null;
        } finally  {
            m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
        }
    }
}
