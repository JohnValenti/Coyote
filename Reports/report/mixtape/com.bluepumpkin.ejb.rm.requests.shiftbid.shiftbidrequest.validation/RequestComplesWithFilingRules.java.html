public class RequestComplesWithFilingRules implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;
    
    private static final String m_className = RequestComplesWithFilingRules.class.getName();
    
    private static final Category m_cat = Log.initCategory(m_className);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");
        
        try {
            ShiftBidRequest sbr =  (ShiftBidRequest) validatable;
            
            // get the associated auction
            ShiftBidAuction sbAuction = sbr.getOptMethods().getShiftBidAuction();
            
            // Build a filing rule checker for this request
            ValidationCache validationCache = sbr.getValidationCache();
            FilingRuleChecker ckr = validationCache.getFilingRuleChecker(null);         
            
            Date fileDate = sbr.getSubmittedOn();
            TimeRange timeRange = validationCache.getDateRange();
                
            ValidationResult result = null;
             // checks 
            if (!ckr.requestComplies(fileDate, timeRange.getStartDate(), 
                timeRange.getEndDate(), sbAuction.getAuctionDeadLine())) { 
                // Since sometimes it might be mystifying why a request fails, we log
                // the rule that failed the requst
                RequestFilingRule rule = ckr.getLastFailingRule();                
    
                m_cat.l7dDebug(RmEjbLogBundleKey.SBR_FILING_RULE_FAILED, new Object[] {sbr.getName(), sbr.getID(), rule.getID()});
                result = ValidationUtil.setSoftValidationResult(sbr,
                    RmEjbBundleKey.SBR_FILING_RULE_FAILED, sbr.getName(), m_className);
            }
            
            return result;
        } finally {
            m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
        }
    }
