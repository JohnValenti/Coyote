package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.validation;

/**
 *
 * Verify if setting "Shift Blocks Per Bid" (auction level setting) is not violated.
 */
public class MaxShiftBlocksPerBidNotExceededHV implements Validator {
    private static String m_className = MaxShiftBlocksPerBidNotExceededHV.class.getName();

    private static final Category m_cat = Log.initCategory(m_className);


    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {

    	ValidationResult valResult = null;
        try {
            ShiftBidRequest sbReq = (ShiftBidRequest) validatable;
            ShiftBidAuction shiftBidAuction = sbReq.getOptMethods().getShiftBidAuction();
            // rule only applies to multi shift auctions
            if (!shiftBidAuction.getIsMultiShift()){
            	return valResult;
            }

            // get "Max Shift Blocks Per Bid" setting.
            int maxShiftBlocksPerBid = shiftBidAuction.getMaxShiftBlocksPerBid();
            if(maxShiftBlocksPerBid == 0) {
            	return valResult;
            }

            List biddableScheduleInstances = sbReq.getOptMethods().getShiftBidRequestBiddableSchedules();

            // rule only applies to multi shift auctions and the value for the Max Shift Blocks should be set
            if(maxShiftBlocksPerBid > 0 && shiftBidAuction.getIsMultiShift()) {
	            	// get # of shift blocks submitted in this request for this auction
	                int numOfShiftBlocksPerBid = biddableScheduleInstances == null? 0: biddableScheduleInstances.size();

	                // check if # of shift blocks in Bid <= "maxShiftBlocksPerBid" setting.
	                if (numOfShiftBlocksPerBid > maxShiftBlocksPerBid) {
	                    valResult = ValidationUtil.setHardValidationResult(sbReq, RmEjbBundleKey.SBR_MAXSHIFTBLOCKSPERBID_EXCEEDED,
	                        NumberFactory.newInteger(maxShiftBlocksPerBid), m_className);
	                }

	                return valResult;
            	}
        } finally {
            m_cat.debug("Validation Rule: " + m_className + " done");
         }
		return valResult;
    }
}
