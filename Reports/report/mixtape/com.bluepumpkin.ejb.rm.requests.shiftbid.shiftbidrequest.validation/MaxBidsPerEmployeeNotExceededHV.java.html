 * Verify if setting "Max Bids Per Employee" (org level setting) is not violated.
 */
public class MaxBidsPerEmployeeNotExceededHV implements Validator {
    private static String m_className = MaxBidsPerEmployeeNotExceededHV.class.getName();

    private String[] m_statusArr = new String[] {
            RequestAuditTrail.STATUS_APPROVED,
            RequestAuditTrail.STATUS_DENIED,
            RequestAuditTrail.STATUS_ESCALATED,
            RequestAuditTrail.STATUS_PENDING,
            RequestAuditTrail.STATUS_TENTATIVE
        };

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        
        ShiftBidRequestDAO sbReqDAO = null;
        try {
            // get "Max Bids Per Employee" setting.
            ShiftBidRequest sbReq = (ShiftBidRequest) validatable;
            ShiftBidRequestValidationCache sbReqValCache = sbReq.getValidationCacheSBR();
            OrganizationSetting empOrgSetting = sbReqValCache.getOrgSettingForOrgID(sbReqValCache.getOrgID());
            int maxBidsPerEmpSetting = empOrgSetting.getShiftBidMaxBidsPerEmployee();
            
            // get # of bids in system submitted by employee for this auction 
            // in states other than (withdrawn or invalid)
            sbReqDAO = new ShiftBidRequestDAO(ShiftBidRequest.DL_BASIC);
            int numOfBidsSubmitted = sbReqValCache.getNumOfBidsSubmittedForBidderID(sbReq.getShiftBidderID(), 
                sbReq.getShiftBidAuctionID(), m_statusArr, sbReqDAO);
            
            // check if # of bids in system <= "MaxBidsPerEmployee" setting.
            ValidationResult valResult = null;            
            if (numOfBidsSubmitted >= maxBidsPerEmpSetting) {
                valResult = ValidationUtil.setHardValidationResult(sbReq, RmEjbBundleKey.SBR_MAXBIDSPEREMPLOYEESETTING_EXCEEDED, 
                    NumberFactory.newInteger(maxBidsPerEmpSetting), m_className);
            } 
            
            return valResult;
        } finally {
            if (sbReqDAO != null) sbReqDAO.cleanUp();
        }
    }
