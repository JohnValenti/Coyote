package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.validation;

import java.io.Serializable;

import org.apache.log4j.Priority;

import com.bluepumpkin.common.logging.Category;

import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidRequest;

/**
 * @author rrajendran
 *
 * Verify that the shift assignments associated with the given shift bid request
 * do not overlap a non working holiday.
 */
public class BiddableSchedOrgHolidayNoOverlapHV implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

    private static final String m_className = BiddableSchedOrgHolidayNoOverlapHV.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");
        try  {
            ShiftBidRequest sbr = (ShiftBidRequest) validatable;
            ShiftBidRequestValidationCache sbrCache = sbr.getValidationCacheSBR();

            // obtain Time Map for non working holidays.
            TimeMap holsNonWorkingTimeMap = sbrCache.getHolidaysNonWorkingTimeMap();

            // obtain the shift assignments for the biddable schedule
            List shiftAssns = sbr.getOptMethods().getShiftAssignments();
            Event event = RequestUtil.verifyOnBitsContainEvents(holsNonWorkingTimeMap, shiftAssns);
            if (event != null)  {
                ValidationResult result = ValidationUtil.setAndLogHardValidationResult(sbr,
										RmEjbBundleKey.SBR_SA_OVERLAPS_NONWORKING_HOL,
										RmEjbLogBundleKey.SBR_SA_OVERLAPS_NONWORKING_HOL,
										new Serializable[] {event.getID()}, m_className);

                return result;
            }

            return null;
        } finally {
            m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
        }
    }
}
