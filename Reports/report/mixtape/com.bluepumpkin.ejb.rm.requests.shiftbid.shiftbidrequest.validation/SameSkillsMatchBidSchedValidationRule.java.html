 *  ### Employee receiving the shift can have more skills

/**
 * @author Sameet Joshi
 *         <p/>
 *         To change this generated comment go to
 *         Window>Preferences>Java>Code Generation>Code and Comments
 */
public class SameSkillsMatchBidSchedValidationRule implements Validator {
	private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;
	private static final String m_className = SameSkillsMatchBidSchedValidationRule.class.getName();

	private static final Category m_cat = Log.initCategory(m_className);

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	public ValidationResult validate(Validatable validatable) throws Exception {
		m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

		try {
			ShiftBidRequest sbReq = (ShiftBidRequest) validatable;
			ID empID = sbReq.getEmployeeID();
			Collection skillAssnsForBidder = ShiftBidAuctionUtil.getSkillAssnsForBidder(sbReq);

			// get 'not-deleted' skillIDs for bidder.  See javadoc for CacheSkill.getSkillIDsNotDeleted() method.
			Collection skillIDsForBidder = RequestUtil.getListOfFieldValuesFromVOBases(skillAssnsForBidder, SkillAssignmentFieldInfo.SKILLASSIGNMENT_SKILLID);
			Set skillIDsForBidderNotDeleted = new HashSet(WfmCacheUtil.getSkillIDsNotDeleted(skillIDsForBidder));

			WorkResourceManager wrm = null;
			ValidationResult valResult = null;
			// iterate thru' the biddable schedules for the request.
			List bidScheds = sbReq.getOptMethods().getBiddableSchedules();
			for (Iterator bidSchedsIter = bidScheds.iterator(); bidSchedsIter.hasNext();) {
				BiddableSchedule bidSched = (BiddableSchedule) bidSchedsIter.next();

				// bidSched associated with this request can be null if the associated bidSched has
				// bidSched.numOfAvailInstances == 0.  Note that execution will never reach this point with
				// this condition bidSched == null since
				// the hard validator RequestBiddableSchedulesAreValidHV.java, which is run prior to this
				// validator, takes care of this.  Still checking, just in case.
				if (bidSched == null) {
					return ValidationUtil.setAndLogHardValidationResult(sbReq,
					        RmEjbBundleKey.SBR_BIDSCHED_HAS_NOMORE_BIDSCHEDINSTS,
					        RmEjbLogBundleKey.SBR_BIDSCHED_HAS_NOMORE_BIDSCHEDINSTS,
					        new Serializable[]{sbReq.getName(), sbReq.getID()},
					        m_className);
				}
				// obtain a biddable schedule instance for the biddbale schedule
				BiddableScheduleInstance bidSchedInst = bidSched.getOptMethods().getFirstAvailBiddableSchedInst();
				ID phantomID = bidSchedInst.getPhantomEmployeeID();

				// get phantom
				wrm = (wrm == null) ? BbmManagerFactory.getWorkResourceManager() : wrm;
				ScheduleAccessManager sam = WfmManagerFactory.getScheduleAccessManager();
				List skillAssnsForPhantom = ShiftBidAuctionUtil.getSkillAssignsForPhanID(phantomID, sam);
				Collection skillIDsForPhantom = RequestUtil.getListOfFieldValuesFromVOBases(skillAssnsForPhantom, SkillAssignmentFieldInfo.SKILLASSIGNMENT_SKILLID);
				Set skillIDsForPhantomNotDeleted = new HashSet(WfmCacheUtil.getSkillIDsNotDeleted(skillIDsForPhantom));
				String phantomNameToDisplay = sam.getEmployeeTemplateByID(sam.getPhantomByID(phantomID).getEmployeeTemplateID()).getName() + ", " + sam.getPhantomByID(phantomID).getName();
				// four possibilities:
				//   no phantom skills, no emp skills.
				//   no phan skills, emp skills.
				//   phan skills, no emp skills.
				//   phan skills, emp skills.
				// check if the phantom skill assignments to see if they match the employee's skill assignments.
				// if no skill assignments were found.
				if (skillIDsForPhantomNotDeleted == null && skillIDsForBidderNotDeleted == null) return null;
				// if no skill assignments were found.
				if (skillIDsForPhantomNotDeleted.isEmpty() && skillIDsForBidderNotDeleted.isEmpty()) return null;
				if (!skillIDsForPhantomNotDeleted.containsAll(skillIDsForBidderNotDeleted) ||
				        !skillIDsForBidderNotDeleted.containsAll(skillIDsForPhantomNotDeleted)) {
					m_cat.l7dDebug(RmEjbLogBundleKey.SBR_PHAN_BIDDER_SKILL_NOT_EXACT_MATCH, new Object[]{phantomNameToDisplay, sbReq.getValidationCacheSBR().getEmployeeNameByID(empID)});
					valResult = ValidationUtil.setSoftValidationResult(sbReq,
					        RmEjbBundleKey.SBR_PHAN_BIDDER_SKILL_NOT_EXACT_MATCH,
					        new Serializable[]{phantomNameToDisplay, sbReq.getValidationCacheSBR().getEmployeeNameByID(empID)}, m_className);
					//return valResult; we defer until all skill mismatches are found.
				}
			}
			return valResult;
		} finally {
			m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
		}
	}


}
