/**
 * Verfies the following:
 * <li> ShiftAssignments associated with the shift bid request do not overlap any shifts in the
 * unpublished schedule.
 * <li> No two shift assignments fall on the same org day.
 *
 * @author rrajendran
 *
 * To change this generated comment go to
 * Window>Preferences>Java>Code Generation>Code and Comments
 */
public class BiddableScheduleDoesNotOverlapShiftAssignmentsHV implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

    private static final String m_className = BiddableScheduleDoesNotOverlapShiftAssignmentsHV.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

        try {
            ShiftBidRequest sbr = (ShiftBidRequest) validatable;
            ShiftBidRequestValidationCache sbrValidationCache = sbr.getValidationCacheSBR();
            ShiftBidAuction shiftBidAuction = sbr.getOptMethods().getShiftBidAuction();
            boolean isMultiShiftBlocksBidding = shiftBidAuction.getIsMultiShift();

            // shift assignments associated with the shift bid request
            List sbrShiftAssns = sbr.getOptMethods().getShiftAssignments();

            // shift assignments in the unpublished schedule.  do not have to check the published
            // schedule since the unpublished schedule will replace the published schedule when the request is approved.
            TimeMap schedShiftAssnsTimeMap = sbrValidationCache.getShiftsUnpubTimeMap();

            // verify if the request's shifts overlap the schedule's shift.
            Event event = RequestUtil.verifyOnBitsOverlapEvents(schedShiftAssnsTimeMap, sbrShiftAssns);
            if (event != null)  {
                ValidationResult result = ValidationUtil.setHardValidationResult(sbr,
                	RmEjbBundleKey.SBR_OVERLAPS_SCHED_SA, event.getStartTime(), event.getEndTime(), m_className);
                return result;
            }

            // get the schedule shift assignments for the request period
            Collection schedShiftAssns = sbrValidationCache.getShiftsUnpubDuringPeriod();

            // get the organization for the request's employee
            Organization empOrg = sbrValidationCache.getOrg();

            //build a Map that maps the 'org day start for the shift's start time' to the shift.
            Map orgDayStartForShiftStartTimeToShiftMap = new HashMap();
            for (Iterator shiftIter = schedShiftAssns.iterator(); shiftIter.hasNext();) {
                ShiftAssignment shift = (ShiftAssignment) shiftIter.next();

                // get the org day start time on which the shift falls.
                Date orgDayStartForShift = TOCalcUtil.getDateForOrgDayStart(empOrg, shift.getStartTime());

                // verify that another shift does not exist on the same org day (quite unlikely but just in case).
                if (orgDayStartForShiftStartTimeToShiftMap.containsKey(orgDayStartForShift))
                    throw new IllegalArgumentException("Two shifts start on the same day in the schedule: " +
                        shift.getID() + " and " + ((ShiftAssignment)orgDayStartForShiftStartTimeToShiftMap.get(orgDayStartForShift)).getID());

                // add the mapping from org day start time for shift's start time to the shift.
                orgDayStartForShiftStartTimeToShiftMap.put(orgDayStartForShift, shift);
            }

            //If not a multi shift block bidding, check overlapping in an org day
            if(!isMultiShiftBlocksBidding) {
                // now ensure that none of the shift assignments associated with this shift bid request falls on the same
                // org day as a shift in the unpublished schedule.
                for (Iterator sbrShiftIter = sbrShiftAssns.iterator(); sbrShiftIter.hasNext();) {
                    ShiftAssignment sbrShift = (ShiftAssignment) sbrShiftIter.next();
                    Date orgDayStartForSbrShift = TOCalcUtil.getDateForOrgDayStart(empOrg, sbrShift.getStartTime());

                    // Error: a shift exists for the same org day in the schedule.
                    if (orgDayStartForShiftStartTimeToShiftMap.containsKey(orgDayStartForSbrShift))  {
    					ShiftAssignment schedSA = (ShiftAssignment) orgDayStartForShiftStartTimeToShiftMap.get(orgDayStartForSbrShift);

                        ValidationResult result = ValidationUtil.setHardValidationResult(sbr,
                        	RmEjbBundleKey.SBR_OVERLAPS_SA_ORGDAY, sbrShift.getStartTime(), sbrShift.getEndTime(),
                        	schedSA.getStartTime(), schedSA.getEndTime(), m_className);

                        return result;
                    }
                }
            }

            return null;
        } finally {
            m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
        }
    }
}
