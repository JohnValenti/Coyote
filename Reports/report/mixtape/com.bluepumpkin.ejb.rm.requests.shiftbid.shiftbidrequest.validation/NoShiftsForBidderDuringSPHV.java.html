/**
 * Ensures that the bidder does not have any shift assignments during the SP associated with
 * the auction.
 *
 * Title:        Blue Pumpkin Software Request Management Module
 * Copyright:    Copyright (c) 2001-2004
 * Company:      Blue Pumpkin Software, inc
 * @author       Raja Rajendran
 */
public class NoShiftsForBidderDuringSPHV implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

    private static final String m_className = NoShiftsForBidderDuringSPHV.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);


    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

        ShiftBidderDAO bidderDAO = null;
        Map empIDToNumOfShiftsMap = null;
        try {
            ShiftBidRequest sbReq = (ShiftBidRequest) validatable;

            ShiftBidAuction sbAuction = sbReq.getOptMethods().getShiftBidAuction();
            ID bidderEmpID = sbReq.getEmployeeID();

            // obtain the # of shifts for the bidder during the SP.
            bidderDAO = new ShiftBidderDAO(ShiftBidder.DL_BASIC);
            if(!sbAuction.getIncPrevSchedEmp()) {
            	empIDToNumOfShiftsMap = bidderDAO.getNumOfShiftsDuringAuctionForEmps(
                Collections.singleton(bidderEmpID), sbAuction.getStartTime(), sbAuction.getEndTime() );
            }
            // if no shifts exist, then request passes this validation rule
            if (empIDToNumOfShiftsMap == null || empIDToNumOfShiftsMap.isEmpty())
                return null;

            // shifts exist during the SP for this request.  So request fails validation.
            return ValidationUtil.setAndLogHardValidationResult(sbReq,
								RmEjbBundleKey.SBR_BIDDER_HAS_SHIFTS_DURING_AUC_SP,
								RmEjbLogBundleKey.SBR_BIDDER_HAS_SHIFTS_DURING_AUC_SP,
                new Serializable[] {bidderEmpID, (Serializable)empIDToNumOfShiftsMap.get(bidderEmpID)},
								m_className);
//        } c_atch (Exception e) {
//            t_hrow RequestUtil.createValidationExceptionWrapper(e, m_cat);
        } finally {
            if (bidderDAO != null) bidderDAO.cleanUp();
            m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
        }
    }

}
