package com.bluepumpkin.ejb.rm.setup.filingrules;

import java.rmi.RemoteException;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.witness.ejb.bbm.copy.CopyableObjectPlugin;
import com.witness.ejb.bbm.copy.ejb.CopyDAO;
import com.witness.ejb.bbm.copy.model.CopyResult;

public abstract class RequestFilingRuleCopyPlugin extends CopyableObjectPlugin {

	public CopyResult copy(ID idObj, ID idOrg, boolean bForce, CopyDAO pDAO,
			String sCopyPrefix) throws BbmException {
		try {
			RmManagerFactory.getInstance().getFilingRuleManager()
					.copyRequestFilingRule(idObj, idOrg);
		} catch (RemoteException e) {
			throw new BbmException(e);
		}
		return new CopyResult();
	}

	public boolean isMultiTargetCopyAllowed() {
		return false;
	}

	public String convertType(Localizer localizer, boolean isPlural) {
		String rbKey = isPlural ? RmEjbBundleKey.OBJECT_FILING_RULE_PLURAL : RmEjbBundleKey.OBJECT_FILING_RULE; 
		return localizer.i18n(RmEjbBundleKey.BUNDLE_NAME, rbKey);
	}

	public abstract ID getModificationPrivilegeID();
}
