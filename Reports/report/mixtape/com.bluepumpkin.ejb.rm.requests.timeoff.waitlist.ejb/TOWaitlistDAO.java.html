package com.bluepumpkin.ejb.rm.requests.timeoff.waitlist.ejb;

/**
 * DAO class for TOWaitlist
 */
public class TOWaitlistDAO extends RmDAOBase<TOWaitlist> {
    // TOChoiceFieldInfo for the DAO class.
    private static FieldInfo m_fieldInfo = new TOWaitlistFieldInfo();
    private static Category m_cat = Log.initCategory(TOWaitlistDAO.class.getName());

    /**
     * Constructor
     */
    public TOWaitlistDAO() {
        super();
    }

    /**
     * Constructor
     *
     * @param - Jdmo - Jdmo object instance
     * @see
     */
    public TOWaitlistDAO(Jdmo dmo) {
        super(dmo);
    }

    @Override
	protected FieldInfo getFieldInfo() {
        return TOWaitlistDAO.m_fieldInfo;
    }

    protected Category getCategory() {
        return TOWaitlistDAO.m_cat;
    }

    /**
     * Creates the value object.
     *
     * @return - returns new Value object
     * @see com.bluepumpkin.ejb.bbm.vo.ValueObjectBase
     * @see com.bluepumpkin.ejb.rm.requests.timeoff.waitlist.model.TOWaitlist
     */
	@Override
	protected TOWaitlist createValueObject() {
		return new TOWaitlist();
    }
    /**
     * Finds TimeOff Waitlist for the TimeOff Waitlist id object passed as argument.
     *
     * @param - ID -  ID object for Time off Choice.
     * @return - TOWaitlist Object.
     * @throw BbmFinderException
     */
    public TOWaitlist getTOWaitlistRequestById(ID pTOWaitlistId) throws BbmFinderException {
        return getObjectByID(pTOWaitlistId);
    }

    public TOWaitlist getWaitlistByTimeOffRequestId(ID toRequestID) throws BbmFinderException {
        TOWaitlist toWaitlist = null;
        Collection waitlistCol = getObjectsByParentID(toRequestID);
        if (waitlistCol != null) {
            for (Iterator iterator = waitlistCol.iterator(); iterator.hasNext();) {
                toWaitlist = (TOWaitlist) iterator.next();
            }
        }
        return toWaitlist;
    }
    /**
     * Create new TimeOff Waitlist for the TimeOff Waitlist object passed as argument.
     *
     * @param - pTOWaitlist -  TOWaitlist object for Time off Choice.
     * @return - ID object for created TOWaitlist.
     * @throw BbmCreateException
     */
    public ID createTOWaitlistRequest(TOWaitlist pTOWaitlist) throws RmException, BbmCreateException {
        if (pTOWaitlist.getTORequestID() == null) {
            throw RequestUtil.createRmException(RmEjbLogBundleKey.REQUEST_ID_NULL, TOWaitlistDAO.m_cat);
        }
        if (pTOWaitlist.getTOWaitlistCreationDate() == null) {
            throw RequestUtil.createRmException(RmEjbLogBundleKey.TOWAITLIST_CREATION_DATE_NULL, TOWaitlistDAO.m_cat);
        }
        if (pTOWaitlist.getTOWaitlistExpiryDate() == null) {
            throw RequestUtil.createRmException(RmEjbLogBundleKey.TOWAITLIST_EXPIRY_DATE_NULL, TOWaitlistDAO.m_cat);
        }
        ID l_TOWaitlistId = createObject(pTOWaitlist);
        return l_TOWaitlistId;

    }

    /**
     * Update TimeOff Waitlist database for the TimeOff Waitlist object passed as argument.
     *
     * @param - pTOWaitlist -  TOWaitlist object for Time off Choice.
     * @throw BbmUpdateException
     */
    public void updateTOWaitlistRequest(TOWaitlist pTOWaitlist) throws BbmUpdateException, MultiUserException {
        updateObject(pTOWaitlist);
    }

    /**
     * Delete TimeOff Waitlist database for the TimeOff Waitlist id object passed as argument.
     *
     * @param - pTOWaitlistID -  ID object for Time off Choice.
     * @throw BbmRemoveException
     * @see
     */
    public void deleteTOWaitlistRequest(ID pTOWaitlistID) throws  BbmRemoveException {
        deleteObject(pTOWaitlistID);
    }

	public Collection getWaitlistedTORequestIdsThatArePastExpiryDate() throws Exception {
		StringBuffer query = new StringBuffer(256);
		query.append("SELECT A.ID FROM REQUEST A, ");
		query.append(getFieldInfo().getTableName()).append(" B ");
		query.append("WHERE A.ID= B.TIMEOFFREQUESTID and B.EXPIRYDATE<=? AND REQUESTSTATUS='");
		query.append(RequestAuditTrail.STATUS_WAITLIST).append("'");
        List toReqIDs = DAOUtil.getIDsUsingSQLQuery(getDMO(), query.toString(),
				new Object[]{new Date()},
				new DAOUtil.ColumnMetaData[]{DAOUtil.ColumnMetaData.getColMetaDataForDate()
				});
		return toReqIDs;
	}
    public Collection getAllWaitlistedTORequestIds() throws Exception {
        StringBuffer query = new StringBuffer(256);
        query.append("SELECT A.ID FROM REQUEST A, ");
        query.append(getFieldInfo().getTableName()).append(" B ");
        query.append("WHERE A.ID= B.TIMEOFFREQUESTID AND REQUESTSTATUS=?");
        List toReqIDs = DAOUtil.getIDsUsingSQLQuery(getDMO(), query.toString(),
                new Object[]{RequestAuditTrail.STATUS_WAITLIST},
                new DAOUtil.ColumnMetaData[]{DAOUtil.ColumnMetaData.getColMetaDataForString()
                });
        return toReqIDs;
    }
}
