package com.bluepumpkin.web.bbm.campaign.sp;

import com.witness.web.uif.pagecomponent.selection.SelectorPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;

/*
 * This is an UI that shows a disabled edit and a button to popup a modal dialog to pick schedule periods
 * This extends SchedulingPeriodSelectorPC to add the ability to show only scheduling periods for which 
 * all the selected employees can work. "Can work" meaning that the employee has been either added to the scheduling period
 * or they were/are in an organization which is linked to the scheduling period and which have "Use All Employees in Linked Organizations" checked  
 */
public class EmployeeSchedulingPeriodSelectorPC extends SchedulingPeriodSelectorPC {

	private static final long serialVersionUID = 2441658048013968213L;

	String selectedEmployeeIDs;

	public static final String EMPLOYEE_SP_SELECTOR_HANDLER = "employee_sp_selector_pu";

	public static final String FIELD_NAME_SELECTED_EMPLOYEES = "selectedEmployees";

	public EmployeeSchedulingPeriodSelectorPC(RequestContext context, ToolbarPC toolbar) {
		super(context, toolbar);
	}

	@Override
	protected void addSelectorPopupArg() {
		//If PopupManger.js sees comma separated values then it assumes the arguments are comma separated field names and the values 
		//must be populated by looking up the field values. In this case are not passing comma separated field names so change 
		//the selected employee ids to be '|' separated 
		//This | separated list will be passed to EmployeeSchedulingPeriodSelectorPopupPM
		String adjustedEmployeeIDs = selectedEmployeeIDs == null ? "" : selectedEmployeeIDs.replace(",", "|");

		String queryString = String.format("isDateEnabled=false&sPMode=0&%s=%s", FIELD_NAME_SELECTED_EMPLOYEES, adjustedEmployeeIDs);
		SelectorPC.addSelectorPopupArg(this, m_name, EMPLOYEE_SP_SELECTOR_HANDLER, queryString, SELECTOR_WIDTH_HEIGHT, false, false);
	}

	public void setSelectedEmployees(String selectedIDs) {
		selectedEmployeeIDs = selectedIDs;
	}

}
