import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.pulse.model.HistoryTrackingView;
import com.bluepumpkin.ejb.bbm.pulse.model.HistoryTrackingViewFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**
 * HistoryTrackingViewDAO is not to be accessed by other objects directly
 * Use default package level restriction
 */
class HistoryTrackingViewDAO extends DAOBase {

	private static FieldInfo m_fieldInfo = new HistoryTrackingViewFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public HistoryTrackingViewDAO() {
		 super();
	}

	public HistoryTrackingViewDAO(Jdmo dmo) {
		 super(dmo);
	}

	protected ValueObjectBase createValueObject()
	{
		return new HistoryTrackingView();
	}

	/** create a HistoryTrackingView */
	public ID createTrackingView(HistoryTrackingView objValue) throws BbmCreateException
	{
		// Convert the definition into String
		objValue.saveChartDef();
		return createObject(objValue);
	}

	/**
	 *   get an employee's HistoryTrackingView definitions
	 *
	 *   @parameter ID, employee id
	 *
	 *   @return a HistoryTrackingView object
	 */
	public HistoryTrackingView getHistoryTrackingView(ID userID) throws BbmFinderException {
		if (userID == null) {
			return null;
		}

		StringBuffer strWhere = new StringBuffer(30);
		strWhere.append("A.BPUSERID = ").append(userID);
		Collection<HistoryTrackingView> col = getObjects(new String(strWhere));
		if (col.isEmpty()) {
			return null;
		}
		HistoryTrackingView tv = col.iterator().next();
		// Next line is to populate the view definition
		tv.retrieveChart();
		return tv;
	}

	/**
	 * get one TrackView object based on its ID
	 *
	 * @param ID, TrackView ID
	 *
	 * @return HistoryTrackingView
	 */
	public HistoryTrackingView getHistoryTrackingViewByID(ID id) throws BbmFinderException
	{
		if (id == null) return null;
		HistoryTrackingView tv = (HistoryTrackingView)getObjectByID(id);
		// Next line is to populate the view definition
		tv.retrieveChart();
		return tv;
	}
}
