import java.util.Collection;
import java.util.Collections;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.pulse.model.PulseNoteQueue;
import com.bluepumpkin.ejb.bbm.pulse.model.PulseNoteQueueFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

/**
 * PulseNoteQueueDAO is not to be accessed by other objects directly
 * Use default package level restriction
 */
class PulseNoteQueueDAO extends DAOBase<PulseNoteQueue> {

	private static FieldInfo m_fieldInfo = new PulseNoteQueueFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public PulseNoteQueueDAO() {
		 super();
	}

	public PulseNoteQueueDAO(Jdmo dmo) {
		 super(dmo);
	}
	
	protected PulseNoteQueue createValueObject() {
		return new PulseNoteQueue();
	}

	
	
	/** create a createPulseNoteQueue */
	public ID createPulseNoteQueue(PulseNoteQueue objValue) throws BbmCreateException	{		
		return createObject(objValue);
	}	
	
	/**
	 *   get note asignments to queues
	 *
	 *   @param noteIds, Collection
	 *   @param starttime, Date
	 *   @param endtime, Date	 
	 *   @return a Collection of PulseNote objects
	 */
	public Collection<PulseNoteQueue> getPulseNoteQueues(Collection<ID> noteIds) throws Exception	{
		if (noteIds == null) {
			return Collections.emptyList();
		}

		// get the notes within the given time period
		StringBuffer strWhere = new StringBuffer(200);
		strWhere.append(" PULSENOTEID IN ");
		strWhere.append(m_dmo.createInClause(noteIds));
		
		return getObjects(new String(strWhere));
	}	
	
	/**
	 * 
	 * @param noteId
	 * @throws Exception
	 */
	public void deleteByPulseNoteId (ID noteId) throws Exception {
		if (noteId == null) return;
		StringBuffer sql = new StringBuffer();
		sql.append(" SELECT ID FROM PULSENOTEQUEUE WHERE PULSENOTEID = "+noteId);
		deleteObjects(sql.toString());
	}
	
	/**
	 * 
	 * @param noteIds
	 * @throws Exception
	 */
	public void deleteByPulseNoteIds (Collection<ID> noteIds) throws Exception{
		if (noteIds == null) return;
		StringBuffer sql = new StringBuffer();
		sql.append(" SELECT ID FROM PULSENOTEQUEUE WHERE PULSENOTEID IN ");
		sql.append(m_dmo.createInClause(noteIds));
		deleteObjects(sql.toString());
	}
}
