import java.sql.Timestamp;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.pulse.model.PulseNote;
import com.bluepumpkin.ejb.bbm.pulse.model.PulseNoteFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.core.base.MultiUserException;

/**
 * TrackingViewDAO is not to be accessed by other objects directly
 * Use default package level restriction
 */
class PulseNoteDAO extends DAOBase<PulseNote> {

	private static FieldInfo m_fieldInfo = new PulseNoteFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public PulseNoteDAO() {
		 super();
	}

	public PulseNoteDAO(Jdmo dmo) {
		 super(dmo);
	}

	protected PulseNote createValueObject() {
		return (new PulseNote());
	}

	protected boolean appendFromWhereStatement(StringBuffer strSQL) {
		strSQL.append(" FROM PULSENOTE A, PULSENOTEQUEUE B, FORECASTUSAGE C ");
		strSQL.append(" WHERE A.ID = B.PULSENOTEID ");
		strSQL.append(" AND A.FORECASTUSAGEID = C.ID ");
		return true;
	}
	
	/**
	 * create a Pulse note
	 * @param objValue
	 * @return
	 * @throws BbmCreateException
	 */
	public ID createPulseNote(PulseNote objValue) throws BbmCreateException	{		
		return createObject(objValue);
	}
	
	/**
	 * update a Pulse note
	 * @param objValue
	 * @throws BbmUpdateException
	 * @throws MultiUserException
	 */
	public void updatePulseNote(PulseNote objValue) throws BbmUpdateException, MultiUserException	{		
		updateObject(objValue);
	}

	/**
	 *   get an employee's PulseNotes
	 *
	 *   @param queueIds, Collection
	 *   @param starttime, Date
	 *   @param endtime, Date	 
	 *   @return a Collection of PulseNote objects
	 */
	public Collection<PulseNote> getPulseNotes(Collection<ID> queueIds, Date starttime, Date endtime) throws Exception	{
		if (queueIds == null || starttime == null || endtime==null) {
			return Collections.emptyList();
		}

		// get the notes within the given time period
		StringBuffer strWhere = new StringBuffer(200);
		strWhere.append(" A.STARTTIME <=");
		strWhere.append("'" + JdmoUtil.formatDBString(new Timestamp(endtime.getTime())) + "'");
		strWhere.append(" AND A.ENDTIME >=");
		strWhere.append("'" + JdmoUtil.formatDBString(new Timestamp(starttime.getTime())) + "'");		
		strWhere.append(" AND B.QUEUEID IN ");
		strWhere.append(m_dmo.createInClause(queueIds));
		strWhere.append(" GROUP BY A.ID, A.CREATEDDATE, A.CREATEDBY, A.MODIFIEDDATE, ");
		strWhere.append(" A.MODIFIEDBY, A.STARTTIME, A.ENDTIME, A.FORECASTUSAGEID, A.NOTE ");
		
		Collection<PulseNote> notes = getObjects(new String(strWhere));
		return notes;
	}	
}
