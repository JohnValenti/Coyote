package com.witness.web.applet;

import java.net.ConnectException;

/**
 * Title:       AppletConnectionException
 * Description: Exception thrown when Applet Connection occurs
 * Copyright:   Copyright (c) 2005
 * Company:     Witness Systems, Inc
 *
 * @author dsu
 *         Created on Aug 11, 2005
 */
public class AppletConnectionException extends ConnectException {
	private int code=-1;

	/**
	 * Constructor
	 */
	public AppletConnectionException() {
	}

	/**
	 * Constructor
	 *
	 * @param code The code stating the error condition
	 */
	public AppletConnectionException(int code) {
		this.code = code;
	}

	/**
	 * Return The code stating the error condition
	 */
	public int getCode() {
		return this.code;
	}
}
