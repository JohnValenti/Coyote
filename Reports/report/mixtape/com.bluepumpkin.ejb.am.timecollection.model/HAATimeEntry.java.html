package com.bluepumpkin.ejb.am.timecollection.model;

import java.io.Serializable;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

/*
 * Title:        Blue Pumpkin Software
 * Description:  Historical Entry
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * Created:      Jul 24, 2003
 * @author       SSong
 * @version		 1.0
 */
public class HAATimeEntry implements Serializable, Comparable {
	static final long serialVersionUID = -8992049173432266554L;

	private ID m_ID;
	private ID m_dsFailureID;
	private ID m_EmployeeID;
	private ID m_ActivityID;
	private int m_PunchTypeCode;
	// client provide Date object
	private Date m_PunchTime;
	// backend use milliseconds representation
	private long m_PunchTimestamp;

	private volatile int m_Hash = 0;

	public HAATimeEntry() {
		super();
	}

	/**
	 * Used by BPX to construct an HAA Entry
	 * @param ID, employeeID
	 * @param ID, activityID
	 * @param Date, punchTime
	 * @param int, punchType
	 */
	public HAATimeEntry(ID employeeID, ID activityID, Date punchTime, int punchType) {
		m_EmployeeID = employeeID;
		m_ActivityID = activityID;
		m_PunchTime = punchTime;
		m_PunchTimestamp = punchTime.getTime();
		m_PunchTypeCode = punchType;
	}

	/**
	 * Used by backend to retrieve HAA Entry
	 * @param ID, id
	 * @param ID, dataSourceFailureID
	 * @param ID, employeeID
	 * @param ID, activityID
	 * @param long, punchTimestamp
	 * @param int, punchType
	 */
	public HAATimeEntry(ID id, ID dsFailureID, ID employeeID, ID activityID, long punchTimestamp, int punchType) {
		m_ID = id;
		m_dsFailureID = dsFailureID;
		m_EmployeeID = employeeID;
		m_ActivityID = activityID;
		m_PunchTime = new Date(punchTimestamp);
		m_PunchTimestamp = punchTimestamp;
		m_PunchTypeCode = punchType;
	}

	/* (non-Javadoc)
	 * @see java.lang.Comparable#compareTo(java.lang.Object)
	 */
	public int compareTo(Object o) {
		if (o instanceof HAATimeEntry) {
			HAATimeEntry oa = (HAATimeEntry)o;
			if (oa.getPunchTimestamp() < getPunchTimestamp())
				return 1;
			if (oa.getPunchTimestamp() > getPunchTimestamp())
				return -1;
			return 0;
		}
		return 1;
	}

	/* (non-Javadoc)
	 * @see java.lang.Object#equals(java.lang.Object)
	 */
	public boolean equals(Object o) {
		if (o instanceof HAATimeEntry) {
			HAATimeEntry oa = (HAATimeEntry)o;
			if (oa.getPunchTimestamp() == getPunchTimestamp())
				if (oa.getPunchTypeCode() == getPunchTypeCode())
					if (oa.getActivityID().equals(getActivityID()))
						return oa.getEmployeeID().equals(getEmployeeID());
		}
		return false;
	}

	/* (non-Javadoc)
	 * @see java.lang.Object#hashCode()
	 */
	public int hashCode() {
		if (m_Hash == 0) {
			StringBuffer hash = new StringBuffer();
			if (m_ID != null) {
				hash.append(m_ID);
			}
			hash.append(getPunchTimestamp());
			hash.append(getPunchTypeCode());
			hash.append(getActivityID());
			hash.append(getEmployeeID());
			m_Hash = hash.toString().hashCode();
		}
		return m_Hash;
	}

	/* (non-Javadoc)
	 * @see java.lang.Object#toString()
	 */
	public String toString() {
		StringBuffer sb = new StringBuffer(80);
		if (getID()!= null) {
			sb.append("ID is ").append(getID());
		}
		if (getEmployeeID() != null) {
			sb.append(" For Employee ID ").append(getEmployeeID());
		}
		if (getPunchTime() != null) {
			sb.append(" Punch at ").append(getPunchTime()).append(" , milliseconds format ").append(getPunchTimestamp());
		}

		sb.append(" PuncType is ").append(getPunchTypeCode());
		return sb.toString();
	}

	/**
	 * @return
	 */
	public ID getEmployeeID() {
		return m_EmployeeID;
	}

	/**
	 * @param employeeID
	 */
	public void setEmployeeID(ID employeeID) {
		m_EmployeeID = employeeID;
	}

	/**
	 * @return
	 */
	public ID getActivityID() {
		return m_ActivityID;
	}

	/**
	 * @param m_id
	 */
	public void setActivityID(ID id) {
		m_ActivityID = id;
	}

	/**
	 * @return
	 */
	public ID getID() {
		return m_ID;
	}

	/**
	 * @param m_id
	 */
	public void setID(ID id) {
		m_ID = id;
	}

	/**
	 * @return
	 */
	private long getPunchTimestamp() {
		return m_PunchTimestamp;
	}

	/**
	 * @param pucnchTimestamp
	 */
	public void setPunchTimestamp(long pucnchTimestamp) {
		m_PunchTimestamp = pucnchTimestamp;
	}

	/**
	 * @return
	 */
	public Date getPunchTime() {
		return m_PunchTime;
	}

	/**
	 * @param punchTime
	 */
	public void setPunchTime(Date punchTime) {
		m_PunchTime = punchTime;
	}

	/**
	 * @return
	 */
	public int getPunchTypeCode() {
		return m_PunchTypeCode;
	}

	/**
	 * @param punchTypeCode
	 */
	public void setPunchTypeCode(int punchTypeCode) {
		m_PunchTypeCode = punchTypeCode;
	}

	public void setDSFailureID(ID id) {
		m_dsFailureID = id;
	}

	public ID getDSFailureID() {
		return m_dsFailureID;
	}
}
