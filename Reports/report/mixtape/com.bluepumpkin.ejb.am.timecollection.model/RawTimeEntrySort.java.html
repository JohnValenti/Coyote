package com.bluepumpkin.ejb.am.timecollection.model;

import com.bluepumpkin.ejb.bbm.activity.model.EmployeeEvent;

/**
* <B>RawTimeEntrySort</B>
* <P>
* sorts collection of RawTimeEntrys
* <P>
	* <B>EMPLOYEEID_ORDER</B>
	* <P>
	* returns a collection of RawTimeEntrySort by 'employeeid'
	* 2nd order sort is by 'punchtime'
	* <P>
	* usage:<br>
	* Collections.sort( (List)rawTimeEntries, RawTimeEntrySort.EMPLOYEEID_ORDER);
	* <P>
	*
	* <B>PUNCHTIME_ORDER</B>
	* <P>
	* returns a collection of RawTimeEntrySort by 'punchtime'
	* <P>
	* usage:<br>
	* Collections.sort( (List)rawTimeEntries, RawTimeEntrySort.PUNCHTIME_ORDER);
	* <P>
	*
*/

public class RawTimeEntrySort {

	public RawTimeEntrySort() {
	}
	
	static final public Comparator EMPLOYEEID_ORDER = new Comparator() {
		public int compare(Object object1, Object object2) {
			EmployeeEvent event1 = (EmployeeEvent)object1;
			EmployeeEvent event2 = (EmployeeEvent)object2;
			int employeeIdCompare = event1.getEmployeeID().compareTo(event2.getEmployeeID());
			if( employeeIdCompare != 0 )
				return employeeIdCompare;
			return (event1.getSortTime()).compareTo(event2.getSortTime());
		}
	};

	static final public Comparator PUNCHTIME_ORDER = new Comparator() {
		public int compare(Object object1, Object object2) {
			EmployeeEvent event1 = (EmployeeEvent)object1;
			EmployeeEvent event2 = (EmployeeEvent)object2;
			return (event1.getSortTime().compareTo(event2.getSortTime()));
		}
	};	
}
