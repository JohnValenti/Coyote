package com.bluepumpkin.ejb.am.timecollection.model;

import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;
import java.io.Externalizable;
import java.sql.Timestamp;
/**
 * Title:        Blue Pumpkin Software Activity Management
 * Description: Information for an activity
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author Vincent Yu
 * @author Sheng Song, enhance Externalization
 * @version 1.0
 */

public class LogEntry implements Externalizable {
	static final long serialVersionUID = -120991761248312752L;

	/**
	* Name for activity type
	*/
	private String activityTypeName;
	/**
	* Name for activity
	*/
	private String activityName;
	/**
	* start time of this activity
	*/
	private Timestamp start;
	/**
	* duration of this activity
	*/
	private long duration;
	/**
	* remark by the employee for this activity
	*/
	private String remark;

	public LogEntry() {}

	public LogEntry(String activityTypeName, String activityName, Timestamp start, String remark) {
		this.activityTypeName = activityTypeName;
		this.activityName = activityName;
		this.start = start;
		this.remark = remark;
	}
	/**
	* set duration for this activity
	*/
	public void setDuration(long duration) { this.duration = duration; }
	/**
	* accessor for activity type name
	*/
	public String getActivityTypeName() { return activityTypeName; }
	/**
	* accessor for activity name
	*/
	public String getActivityName() { return activityName; }
	/**
	* accessor for start time
	*/
	public Timestamp getStart() { return start; }
	/**
	* accessor for duration
	*/
	public long getDuration() { return duration; }
	/**
	* accessor for employee remark
	*/
	public String getRemark() { return remark; }
	public String toString() {
		return "Activity Type: " + activityTypeName +
			", Activity: " + activityName +
			", Start: " + start +
			", Duration: " + duration +
			", Remark: '" + remark + "'";
	}
	/* (non-Javadoc)
	 * @see java.io.Externalizable#readExternal(java.io.ObjectInput)
	 */
	public void readExternal(ObjectInput in)
		throws IOException, ClassNotFoundException {
		activityTypeName = (String)in.readObject();
		activityName = (String)in.readObject();
		start = (Timestamp)in.readObject();
		remark = (String)in.readObject();
		duration = in.readLong();
	}

	/* (non-Javadoc)
	 * @see java.io.Externalizable#writeExternal(java.io.ObjectOutput)
	 */
	public void writeExternal(ObjectOutput out) throws IOException {
		out.writeObject(activityTypeName);
		out.writeObject(activityName);
		out.writeObject(start);
		out.writeObject(remark);
		out.writeLong(duration);
	}
}
