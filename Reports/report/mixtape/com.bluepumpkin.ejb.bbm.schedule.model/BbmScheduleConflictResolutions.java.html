package com.bluepumpkin.ejb.bbm.schedule.model;

/**
 * Title:        Blue Pumpkin Software Basic Business Model
 * Description:  This object describes how to resolve conflicts (schedule
 *               constraint violations) when they are encountered.
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       Greg Fihctenholtz
 * @version      1.0
 */
import java.util.HashMap;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.util.NumberFactory;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbBundleKey;

public class BbmScheduleConflictResolutions implements java.io.Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;

	/**
	 * With this resolution, the shift overlapped with the newly created or
	 * updated shift, template or event will be deleted.
	 */
	public static final int RESOLUTION_REMOVE_SHIFT = 1;

	/**
	 * With this resolution, the calendar event overlapped with the newly
	 * created or updated shift, template or event will be deleted
	 */
	public static final int RESOLUTION_REMOVE_CALENDAREVENT = 2;

	/**
	 * With this resolution, the recurring instance overlapped with the
	 * conflicting shift, template or event will be deleted. Here means creating
	 * "holes", i.e. exceptions to the recurring event template rule.
	 */
	public static final int RESOLUTION_REMOVE_RECURRINGINSTANCE = 3;

	private HashMap<Integer, Integer> m_mapResolutions = new HashMap<Integer, Integer>();

	public void addResolution(int conflictType, int resolution) {
		m_mapResolutions.put(NumberFactory.newInteger(conflictType), NumberFactory.newInteger(resolution));
	}

	public int getResolution(int conflictType) {
		Integer i = m_mapResolutions.get(NumberFactory.newInteger(conflictType));
		return i == null ? -1 : i.intValue();
	}

	/**
	 * Small helper method.
	 */
	private static String getMessageId(int resolution) {
		switch (resolution) {
		case RESOLUTION_REMOVE_CALENDAREVENT:
			return BbmEjbBundleKey.SCHEDULE_RESOLUTION_REMOVE_CALENDAREVENT;
		case RESOLUTION_REMOVE_RECURRINGINSTANCE:
			return BbmEjbBundleKey.SCHEDULE_RESOLUTION_REMOVE_RECURRINGINSTANCE;
		case RESOLUTION_REMOVE_SHIFT:
			return BbmEjbBundleKey.SCHEDULE_RESOLUTION_REMOVE_SHIFT;
		}
		return null;
	}

	/**
	 * Returns the text for the given schedule conflict resolution message in
	 * the specified locale.
	 * 
	 * @deprecated by com.bluepumpkin.common.l10n.Localizer public static String
	 *             getLocalizedMessage(int resolution, Locale locale) { String
	 *             msgID = getMessageId(resolution); return
	 *             StringUtil.formatMessage(BbmEjbBundleKey.BUNDLE, locale,
	 *             msgID, null); }
	 */

	/**
	 * Returns the text for the given schedule conflict resolution message in
	 * the specified locale.
	 */
	public static String getLocalizedMessage(int resolution, Localizer localizer) {
		String msgID = getMessageId(resolution);
		return localizer.i18n(BbmEjbBundleKey.BUNDLE, msgID, (Object[]) null);
	}
}
