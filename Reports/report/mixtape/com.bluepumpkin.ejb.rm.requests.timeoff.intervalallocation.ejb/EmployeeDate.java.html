package com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

public class EmployeeDate {

	final int employeeID;
	final Date date;

	public EmployeeDate(ID id, Date dt) {
		this(id.toInt(), dt);
	}
	public EmployeeDate(int id, Date dt) {
		this.employeeID = id;
		this.date = dt;
	}

	@Override
	public boolean equals(Object o) {
		if (o == this) {
			return true;
		}

		if (!(o instanceof EmployeeDate)) {
			return false;
		}

		EmployeeDate other = (EmployeeDate) o;
		return other.employeeID == employeeID && other.date.getTime() == date.getTime();

	}

	@Override
	public int hashCode() {
		int result = 17;
		result = 31 * result + employeeID;
		result = 31 * result + date.hashCode();
		return result;
	}

}
