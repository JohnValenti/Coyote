package com.bluepumpkin.web.am.user.preferences;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Set;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.util.StringUtil;
import com.witness.ejb.core.licensing.LicenseKeys;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;

import com.bluepumpkin.ejb.core.CoreManagerFactory;
import com.bluepumpkin.web.am.adherence.keys.AdherenceKeys;
import com.bluepumpkin.web.am.adherence.util.AdherenceUtils;
import com.bluepumpkin.web.am.l10n.AmWebBundleKey;

public class UserPrefAMPC extends UserPrefPC {
	private LicenseManager m_licManager;
	private boolean m_isShowWhoisInEnabled=false;
	private boolean m_isMyTimeSortOrderEnabled=false;
	private boolean m_isAdherenceAutoRefreshEnabled=false;
	private boolean m_isAdherenceCollapsedViewEnabled=false;
	private boolean m_isPulseAutoRefreshEnabled=false;

	private boolean m_isWhoIsInShown;
	private String m_selectedMyTimeSortOrder;
	private String m_selectedAdherenceAutoRefresh;
	private String m_selectedAdherenceCollapsedView;
	private String m_selectedPulseAutoRefresh;

	public UserPrefAMPC(RequestContext context) {
		super(context);
		setTitle(i18n(m_common_bundle, UIFWebBundleKey.USER_PREF_AM));

		m_isShowWhoisInEnabled = isAuthorized(PrivilegeKeys.AM_VIEWWHOISINSTATUS);
		m_isMyTimeSortOrderEnabled = isAuthorized(PrivilegeKeys.AM_MYTIMETIMERECORDS);

		boolean isAuthToViewAdherence = isAuthorized(PrivilegeKeys.AM_VIEWADHERENCE);
		m_isAdherenceAutoRefreshEnabled = isAuthToViewAdherence &&
            (isLicensed(LicenseKeys.APP_ADHERENCE_EXPRESS) || isLicensed(LicenseKeys.APP_ADHERENCE));
		m_isAdherenceCollapsedViewEnabled = isAuthToViewAdherence;

		boolean isAuthToViewPulse = isAuthorized(PrivilegeKeys.FS_VIEWPULSE);
		m_isPulseAutoRefreshEnabled = isAuthToViewPulse && isLicensed(LicenseKeys.APP_FS);
	}

	public Collection getUserPrefKey() {
		ArrayList keys = new ArrayList(5);
		keys.add(USER_SHOW_WHOISIN);
		if (m_isMyTimeSortOrderEnabled) keys.add(USER_MYTIME_SORT_ORDER);
		if (m_isAdherenceAutoRefreshEnabled) keys.add(USER_ADHERENCE_AUTO_REFRESH);
		if (m_isAdherenceCollapsedViewEnabled) keys.add(USER_ADHERENCE_COLLAPSED_VIEW);
		if (m_isPulseAutoRefreshEnabled) keys.add(USER_PULSE_AUTO_REFRESH);
		return keys;
	}

	protected void initSupportData() throws Exception {
		m_isWhoIsInShown = getBooleanPreference(USER_SHOW_WHOISIN, false);
		m_selectedMyTimeSortOrder = getPreference(USER_MYTIME_SORT_ORDER);
		m_selectedAdherenceAutoRefresh = getPreference(USER_ADHERENCE_AUTO_REFRESH);
		m_selectedAdherenceCollapsedView = getPreference(USER_ADHERENCE_COLLAPSED_VIEW);
		m_selectedPulseAutoRefresh = getPreference(USER_PULSE_AUTO_REFRESH);
	}

	protected void initContent() throws Exception {
		// Show Who Is In
		if (m_isShowWhoisInEnabled) {
			HtmlBooleanCheckbox input = HtmlBooleanCheckbox.createInput(
					USER_SHOW_WHOISIN, i18n(m_bundle, UIFWebBundleKey.SHOW_WHOISIN),
					onChangeJS, m_isWhoIsInShown, false);
			addRow(input.getLabelDisplay(), input.getInputDisplay());
		}

		// Mytime Sort Order
		if (m_isMyTimeSortOrderEnabled) {
			addRowWithLabelKey(UIFWebBundleKey.MYTIME_SORT_ORDER, getMytimeSortOrderSelect());
		}

		// Adherence Auto-Refresh
		if (m_isAdherenceAutoRefreshEnabled) {
			addRowWithLabelKey(UIFWebBundleKey.ADHERENCE_AUTO_REFRESH,	getAdherenceAutoRefreshSelect());
		}

		// Adherence Collapsed View
		if (m_isAdherenceCollapsedViewEnabled) {
			addRowWithLabelKey(UIFWebBundleKey.ADHERENCE_COLLAPSED_VIEW,	getAdherenceCollapsedViewSelect());
		}

		// Pulse Auto-Refresh
		if (m_isPulseAutoRefreshEnabled) {
			addRowWithLabelKey(UIFWebBundleKey.PULSE_AUTO_REFRESH,	getPulseAutoRefreshSelect());
		}

	}

	/**
	 * Returns the My Time sort order selection in an Html Select string
	 */
	private String getMytimeSortOrderSelect() {
		ArrayList sortOptions = new ArrayList(2);
		sortOptions.add(new StringsPair("1", i18n(m_bundle, UIFWebBundleKey.ASCENDING)));
		sortOptions.add(new StringsPair("0", i18n(m_bundle, UIFWebBundleKey.DESCENDING)));
		return HtmlUtil.makeSelectInput(USER_MYTIME_SORT_ORDER, m_selectedMyTimeSortOrder, sortOptions);
	}

	/**
	 * Returns the Adherence Auto-Refresh selection in an Html Select string
	 */
	private String getAdherenceAutoRefreshSelect() {
		ArrayList options = getAutoRefreshOptions(UserPreferenceKeys.ADHERENCE_AUTO_REFRESH_OPTIONS);
		return HtmlUtil.makeSelectInput(USER_ADHERENCE_AUTO_REFRESH, m_selectedAdherenceAutoRefresh, options);
	}

	/**
	 * Returns the Adherence Collapsed view selection in an Html Select string
	 */
	private String getAdherenceCollapsedViewSelect() {
		ArrayList options = new ArrayList(4);
		//check if settings should be retrieved from the profile
		boolean hasSchedule = false;
		if (isLicensed(LicenseKeys.APP_FS))
			hasSchedule = true;

		String actualStr = i18n(m_bundle,UIFWebBundleKey.ACTUAL);

		if (isLicensed(LicenseKeys.APP_DESKTOPMONITORING)
				|| isLicensed(LicenseKeys.APP_ACDMONITORING)){
			try {
				actualStr = BbmManagerFactory.getDBConfigManager().getValue(ConfigKey.DESKTOP_MONITORING_ACTUALS_PRIMARY);
			} catch(Exception ex) {}
		}
		if (StringUtil.isEmpty(actualStr))
			actualStr = m_context.getLocalizer().i18n(AmWebBundleKey.BUNDLE_NAME,AmWebBundleKey.ADHERENCE_TIMELINE_PRIMARY);

		if (hasSchedule)
			options.add(new StringsPair(UserPreferenceKeys.ADHERENCE_COLLAPSED_VIEW_SCHEDULED,
					i18n(m_bundle,UIFWebBundleKey.SCHEDULED)));
		if (isLicensed(LicenseKeys.APP_ADHERENCE)
				|| isLicensed(LicenseKeys.APP_ADHERENCE_EXPRESS)
				|| isLicensed(LicenseKeys.APP_DESKTOPMONITORING)
				|| isLicensed(LicenseKeys.APP_ACDMONITORING))
			options.add(new StringsPair(UserPreferenceKeys.ADHERENCE_COLLAPSED_VIEW_ACTUAL,
					actualStr));
		options.add(new StringsPair(UserPreferenceKeys.ADHERENCE_COLLAPSED_VIEW_EXCEPTION,
				i18n(m_bundle,UIFWebBundleKey.EXCEPTION)));

		//add secondary view type if DM license is available
		if (isLicensed(LicenseKeys.APP_DESKTOPMONITORING)){
			String secondary = null;
			try {
				secondary = BbmManagerFactory.getDBConfigManager().getValue(ConfigKey.DESKTOP_MONITORING_ACTUALS_SECONDARY);
			} catch(Exception ex) {}
			if (StringUtil.isEmpty(secondary))
				secondary = m_context.getLocalizer().i18n(AmWebBundleKey.BUNDLE_NAME,AmWebBundleKey.ADHERENCE_TIMELINE_SECONDARY);
			options.add(new StringsPair(AdherenceKeys.TRACKING,
					secondary));
		}

		return HtmlUtil.makeSelectInput(USER_ADHERENCE_COLLAPSED_VIEW, m_selectedAdherenceCollapsedView,options);
	}

	/**
	 * Returns the Pulse Auto-Refresh selection in an Html Select string
	 */
	private String getPulseAutoRefreshSelect() {
		ArrayList options = getAutoRefreshOptions(UserPreferenceKeys.PULSE_AUTO_REFRESH_OPTIONS);
		return HtmlUtil.makeSelectInput(USER_PULSE_AUTO_REFRESH, m_selectedPulseAutoRefresh,options);
	}


	private ArrayList getAutoRefreshOptions(int[] optionsInSeconds) {
		ArrayList options = new ArrayList(optionsInSeconds.length+1);

		// adds no-refresh first
		options.add(new StringsPair(UserPreferenceKeys.NO_AUTO_REFRESH,	i18n(m_bundle,UIFWebBundleKey.NO_AUTO_REFRESH)));

		for (int i = 0; i < optionsInSeconds.length; i++) {
			int minutes = optionsInSeconds[i] / 60;
			String suffix = minutes > 1 ?
					i18n(m_bundle,UIFWebBundleKey.MINUTES) :
					i18n(m_bundle,UIFWebBundleKey.MINUTE);
			options.add(new StringsPair(String.valueOf(optionsInSeconds[i]), String.valueOf(minutes) + " " + suffix));
		}
		return options;
	}

	protected boolean isLicensed(String app) {
		boolean result=false;
		try {
			if (m_licManager==null) m_licManager=CoreManagerFactory.getLicenseManager();
			result=(m_licManager!=null && m_licManager.isLicensed(app));
		} catch(Exception ex) {}

		return result;
	}
}
