package com.bluepumpkin.ejb.rm.l10n;

import java.io.Serializable;

/**
 * @author rrajendran
 *
 * Wraps a resource bundle key.  Used to mark the msgKey (which is just a string) as
 * a resource bundle msgKey.  This tagging allows the msgKey to be used for localization.
 * See {@link com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil#getLocalizedMessage(String, String, Object[], Localizer, TimeZone) getLocalizedMessage()}
 * for further details.
 */
public class ResourceBundleKeyWrapper implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private String m_resBundleMsgKey = null;
	private final static String m_resBundleName = RmEjbBundleKey.BUNDLE_NAME;
	
	public ResourceBundleKeyWrapper(String msgKey) {
		m_resBundleMsgKey = msgKey;
	}
	
	
	/**
	 * @return
	 */
	public static String getResBundleName() {
		return m_resBundleName;
	}

	/**
	 * @return
	 */
	public String getResBundleMsgKey() {
		return m_resBundleMsgKey;
	}

}
