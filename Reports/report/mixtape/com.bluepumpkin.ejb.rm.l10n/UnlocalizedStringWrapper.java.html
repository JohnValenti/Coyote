package com.bluepumpkin.ejb.rm.l10n;

import com.bluepumpkin.common.localization.Localizer;

/**
 * @author RRajendran
 *
 * Used to wrap unlocalized strings which can later be localized by
 * common logic (residing in RequestUtil.getLocalizedString() currently).
 * 
 */
public class UnlocalizedStringWrapper {
	private String m_unlocString;
	
	public UnlocalizedStringWrapper(String unlocString) {
		m_unlocString = unlocString;
	}
	
	/**
	 * return the embedded unlocalized string.
	 * 
	 * @return
	 */
	public String getUnlocalizedString() {
		return m_unlocString;
	}
	
	/**
	 * Localize the encapsulated unlocalized string by looking it up
	 * in the resource bundle, using the unlocalized bundle as the msgKey.
	 *  
	 * @param localizer
	 * @return
	 */
	public String getLocalizedString(Localizer localizer) {
		return localizer.i18n(RmEjbBundleKey.BUNDLE_NAME, m_unlocString);
	}
}
