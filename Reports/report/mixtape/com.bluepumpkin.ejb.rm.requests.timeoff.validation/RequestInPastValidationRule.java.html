package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.Date;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoiceGroup;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.RmUtil;
import com.bluepumpkin.ejb.rm.util.TORequestUtil;

/**
 * Rule in UI: Requested Time Off is for the past date
 * <p/>
 * Check if request is for the past date. this rule will raise alert for only
 * those requests that are created for start time that falls before submitted
 * date of the request. I believe this prevents users from submitting requests
 * for timeoff taken in the past.
 * <p/>
 */
public class RequestInPastValidationRule extends TimeoffValidationRule {
	private static final String CLASSNAME = RequestInPastValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASSNAME);

	@Override
	protected void validateTimeOffChoiceGroup(TORequest request, TOChoiceGroup choiceGroup) throws Exception {
		String methodName = "validateTimeOffChoiceGroup";
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod(methodName, request));
		}

		ValidationResult result = null;
		for (TOChoice toChoice : choiceGroup.getChoices()) {

			if (!validateTOChoice(request, toChoice)) {
				Date startDate = toChoice.getStartDate();
				Date endDate = toChoice.getEndDate();
				// Build validation result, store in choice.
				result = ValidationUtil.setSoftValidationResult(toChoice, RmEjbBundleKey.TO_REQUEST_FOR_PAST_DATE,
						getEmployeeNameByID(request), startDate, endDate, CLASSNAME);
			}
		}

		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpExitMethod(methodName, result));
		}
	}

	// Returns false if it fails validation
	boolean validateTOChoice(TORequest request, TOChoice choice) {
		return !TORequestUtil.isTimeOffChoiceForPastDate(request, choice);
	}

	protected String getEmployeeNameByID(TORequest request) throws Exception {
		return request.getCache().getEmployeeNameByID();
	}
}
