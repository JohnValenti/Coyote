/**
 * <p> Rule from UI: Minimum duration of request with a net staffing surplus: X (percent).
 * Will flag any time off requests that do not satisfy the minimum required net-staffing shortage in percent
 * Formula:
 * (#periods where net staffing is improved / (#periods where net staffing is improved + #periods where net staffing is worsened) )*100
 * Periods will be calculated based on 15-minute intervals, regardless of LTI. However, the 15-minute values take LTI into account.
 * We do not care about intervals where the net staffing is not changed. But we do take threshold into account. So if net staffing
 * for an interval before the request is 2, and after approval, net staffing is 1, and the threshold is 2, then this interval is
 * considered to be unchanged rather than improved, since both values fall under the threshold and are considered equivalent to 0.
 * Default value = 100%
 * Sample Error Message Shown:
 * Request violates the net staffing requirements for the requested time frame:
 * Queue q1:
 * From 10/10/2011 2:00 am PDT to 10/10/2011 4:00 am PDT
 * From 10/10/2011 8:00 am PDT to 10/10/2011 10:00 am PDT
 * Queue 123 phone QUEUE1:
 * From 10/10/2011 12:15 PM to 07/23/2012 1:00 PM
 * From 10/10/2011 4:45 PM to 07/23/2012 5:00 PM
 */
public class NetStaffingValidationRule implements Validator {
	private static final String m_className = NetStaffingValidationRule.class.getName();

	private static final Category m_cat = Log.initCategory(m_className);
	public static final String MIN_SURPLUS_DURATION_VALUE = "VTO_MIN_SURPLUS_DURATION_VALUE";
	public static final int MINUTES = 1;
	public static final int PERCENT = 0;
	public static final ID RULE_ID = new ID(-194045);
	public static Integer duration = null;
	

	/**
	 * Validate a time off request.
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		TORequest tor = (TORequest) validatable;
		String methodName = "validate";
		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpEnterMethod(methodName, tor));
		}
		ValidationResult result = null;
		Iterator<TOChoice> it = tor.getRequestChoiceList().iterator();
		TOValidationCache cache = tor.getCache();
		Organization org = getOrganization(tor);
		TimeZone tz = org.getTimeZone();
		int minReqdNetStaffSurplusPercent = getRuleParameter(org.getID());


		while (it.hasNext()) {
			TOChoice choice = it.next();
			Date startDate = choice.getStartDate();
			Date endDate = choice.getEndDate();
			TimeRange choiceRange = new TimeRange(startDate, endDate);
			duration = getDuration();
			boolean durViolationFound = checkDurationFound(tz, startDate, endDate, duration);
			if (durViolationFound) {
				result = ValidationUtil.setSoftValidationResult(choice, RmEjbBundleKey.TO_CHOICE_MAX_DURATION_NETSTAFFING, startDate, endDate, m_className);
				continue;
			}

			
			boolean stafingExists = doesStaffingExistForPeriod(tor, choice);

			if (!stafingExists) {
				result = ValidationUtil.setSoftValidationResult(choice, RmEjbBundleKey.CS_NETSTAFF_NOT_AVAILABLE, m_className);
				continue;
			}

			NetStaffingCube preNsCube = getPreTOApprovalNetStaffingCube(tor, choice); //the Tracecubes as they would look before approval of the request.
			NetStaffingCube postNsCube = getPostTOApprovalNetStaffingCube(tor, choice); //the Tracecubes as they would look after approval of the request.
			//Only continue to validate when Queues Found that match the skill set of EMP (preNsCube!=null && postNsCube !=null ) )
			if(preNsCube!=null && postNsCube!=null){ 
				HashMap<ID, SPQueue> spQueueMap = postNsCube.getSPQueueMap();
	
				HashMap<ID, TreeSet<TimeRange>> spQTimeRangeWorsened = new HashMap<ID, TreeSet<TimeRange>>();
				Collection<ShiftAssignment> pubShifts = getPublishedShiftAssignments(tor, choice);
				ArrayList<TimeRange> trCol = new ArrayList<TimeRange>();
				for (Iterator<ShiftAssignment> i = pubShifts.iterator(); i.hasNext();) {
					ShiftAssignment sa = i.next();
					if (choiceRange.getEndDate().before(sa.getStartTime()) || choiceRange.getStartDate().after(sa.getEndTime())) {
						continue;
					}
					TimeRange shiftTr = new TimeRange(choiceRange.includes(sa.getStartTime()) ? sa.getStartTime() : choiceRange.getStartDate(),
					        choiceRange.includes(sa.getEndTime()) ? sa.getEndTime() : choiceRange.getEndDate());
					trCol.add(shiftTr);
				}
	
				Pair<Integer, Integer> numIntervalsImprovedAndWorsened = getNumIntervalsImprovedAndWorsened(spQTimeRangeWorsened, trCol, preNsCube, postNsCube, cache);
				int numIntervalsImproved = numIntervalsImprovedAndWorsened.getFirst();
				int numIntervalsWorsened = numIntervalsImprovedAndWorsened.getSecond();
				int numIntervalsChanged = numIntervalsImproved + numIntervalsWorsened;
				double percentOfIntervalsImproved = (numIntervalsChanged==0) ? 100.0d :
						(((double)(numIntervalsImproved)) / ((double)numIntervalsChanged) * 100.0d); //we don't care about the unchanged intervals
	
	
				if (percentOfIntervalsImproved < minReqdNetStaffSurplusPercent)
				{
					StringBuffer value = new StringBuffer();
					Set qIDSet = TOHoursPerDayUtil.extractFieldSet(SPQueueFieldInfo.SPQUEUE_QUEUEID, spQueueMap.values());
					Map queIDNameMap = WfmManagerFactory.getWorkloadManager().getQueueNamesByIDs(qIDSet);
					for (ID spQID : spQTimeRangeWorsened.keySet()) {
						TreeSet<TimeRange> timeRangeSet = spQTimeRangeWorsened.get(spQID);
						SPQueue spQueue = spQueueMap.get(spQID);
						String qName = spQueue.getQueueID() == null ? "" : (String) queIDNameMap.get(spQueue.getQueueID());
						generateString(value, timeRangeSet, qName, tz);
					}
					result = ValidationUtil.setSoftValidationResult(choice, RmEjbBundleKey.TO_NETSTAFF_SURPLUS, value.toString(), m_className);
				}
			}
		}
		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
		}
		return result;
	}

	/**
	 * Determines the number of 15-minute intervals in the requested date range in which the net staffing
	 * would be improved and worsened if the request is approved. We also keep track of the SPQueue->intervals that
	 * would see net staffing worsen. We need this information in order to compare the percentage of improvement
	 * versus the goal percentage for the validation rule.
	 * @param spQTimeRangeWorsened - In this map, we will store the SPQueues that contain worsened (15-minute) intervals.
	 *                               mapped to the worsened intervals. Worsened intervals are the 15-minute intervals in
	 *                               which net staffing would get worse if the request is approved.
	 * @param trCol - List of timeRanges for the request, in chronological order
	 * @param preNsCube - the NetStaffingCube prior to approving the request.
	 * @param postNsCube - the NetStaffingCube that would exist after approving the request.
	 * @param cache - the cache to use when getting the necessary aggregations, or any other cacheable items.
	 * @return - A Pair containing the number of 15-minute intervals in the requested date range that would see a net staffing improvement
	 *           were the request to be approved, and the number of 15-minute intervals in the requested date range that would see a net
	 *           staffing worsening were the request to be approved.
	 */
	Pair<Integer, Integer> getNumIntervalsImprovedAndWorsened(HashMap<ID, TreeSet<TimeRange>> spQTimeRangeWorsened, List<TimeRange> trCol,
															  NetStaffingCube preNsCube, NetStaffingCube postNsCube, TOValidationCache cache)
	{
		Pair<Integer, Integer> numIntervalsImprovedAndWorsened = null;
		int numIntervalsImproved = 0;
		int numIntervalsWorsened = 0;

		if (trCol != null &&  trCol.size()>0)
		{
			TimeRange requestDateRange = ValidationUtil.getRequestDateRange(trCol);
			Date curIntervalStart = requestDateRange.getStartDate();
			Date curIntervalEnd = new Date(curIntervalStart.getTime() + Trace.INTERVAL_IN_MILLIS);
			while (!curIntervalEnd.after(requestDateRange.getEndDate())) //(for each interval of the request range)
			{
				if (ValidationUtil.isValidInterval(curIntervalStart, trCol))
				{
					boolean anyQueueImprovedInCurInterval = false;
					boolean anyQueueUnchangedInCurInterval = false;
					HashSet<SPQueue> spQueuesWorsenedThisInterval = new HashSet();

					//For Time Off Requests:
					//If ALL queues in this interval has a net staffing improvement, then this interval is considered an improvement.
					//Otherwise, if ANY queue in this interval has a net staffing worsening, then this interval is considered a worsening.
					//Otherwise, this interval is considered unchanged.
					HashMap<SPQueue, Double> spqIDToNSBeforeMap = ValidationUtil.getFTEDPastThreshold(preNsCube, curIntervalStart, curIntervalEnd, "PRE", cache);
					HashMap<SPQueue, Double> spqIDToNSAfterMap = ValidationUtil.getFTEDPastThreshold(postNsCube, curIntervalStart, curIntervalEnd, "POST", cache);

					for (Iterator it = spqIDToNSBeforeMap.keySet().iterator(); it.hasNext();)
					{
						SPQueue spQueue = (SPQueue)it.next();

						double netStaffBefore = Math.abs(spqIDToNSBeforeMap.get(spQueue));
						double netStaffAfter = Math.abs(spqIDToNSAfterMap.get(spQueue));

						if ((netStaffAfter < netStaffBefore))
						{
							anyQueueImprovedInCurInterval = true;
						}
						else if (netStaffAfter > netStaffBefore)
						{
							spQueuesWorsenedThisInterval.add(spQueue);
						}
						else
						{
							anyQueueUnchangedInCurInterval = true;
						}
					}

					if (spQueuesWorsenedThisInterval.size()>0)
					{
						numIntervalsWorsened++;
						for (SPQueue curSPQueue : spQueuesWorsenedThisInterval)
						{
							ID spqID = curSPQueue.getID();
							TreeSet curSPQWorserSet = spQTimeRangeWorsened.get(spqID);
							if (curSPQWorserSet == null) {
								curSPQWorserSet = new TreeSet<TimeRange>();
							}
							curSPQWorserSet.add(new TimeRange(curIntervalStart, curIntervalEnd));

							spQTimeRangeWorsened.put(spqID, curSPQWorserSet);
						}
					}
					else if (anyQueueImprovedInCurInterval && !anyQueueUnchangedInCurInterval)
					{
						numIntervalsImproved++;
					}
				}
				curIntervalStart = curIntervalEnd;
				curIntervalEnd = new Date(curIntervalEnd.getTime() + Trace.INTERVAL_IN_MILLIS);
			}
		}

		return new Pair(numIntervalsImproved, numIntervalsWorsened);
	}


	public void generateString(StringBuffer value, Set defaultingSet, String queueName, TimeZone tz) {
		Localizer localizer= getLocalizer();
		ResourceBundle rmBundle = localizer.getBundle(RmEjbBundleKey.BUNDLE_NAME);
		value.append(localizer.i18n(rmBundle, RmEjbBundleKey.QUEUE_NAME, new Object[] { queueName }));
		value.append("<BR>");
		TreeSet violationTreeSet = TimeRange.mergeAdjacentTimeRanges(defaultingSet, 60000);
		for (Iterator iterator = violationTreeSet.iterator(); iterator.hasNext();) {
			TimeRange tr = (TimeRange) iterator.next();
			String formatStDateTime = localizer.formatDateTime(tr.getStartDate(), tz);
			String formatEnDateTime = localizer.formatDateTime(tr.getEndDate(), tz);
			value.append(localizer.i18n(rmBundle, RmEjbBundleKey.TimePeriodFromToData, new Object[] { formatStDateTime, formatEnDateTime }));
			value.append("<BR>");
		}
	}
	@SuppressWarnings("rawtypes")
	protected int getRuleParameter(ID orgId) throws BbmCreateException, BbmFinderException, RemoteException{
		ValidationRuleManager valMan = RmManagerFactory.getInstance().getValidationRuleManager();
		Map ruleParams = valMan.getValidationRuleParams(orgId, RULE_ID);
		return  ((Integer) ruleParams.get(MIN_SURPLUS_DURATION_VALUE)).intValue();
	}
	protected int getDuration() throws BbmEJBCreateException, RemoteException{
		int result = BbmManagerFactory.getDBConfigManager(false).getIntValue(ConfigKey.RM_MAX_LENGTH_ALLOWED_NETSTAFFING);
			if (result == ConfigManager.NOVALUE) {
				result = 7;
			}
		return result;
	}
	protected boolean checkDurationFound(TimeZone tz,Date startDate,Date endDate,int duration ){
		return DateUtil.compareDayDiffWithAllowedCount(tz, startDate, endDate, duration) > 0;
	}
	protected boolean doesStaffingExistForPeriod(TORequest tor,TOChoice choice) throws Exception{
		return getTOValidationCache(tor).doesStaffingExistForPeriod(tor, choice);
	}
	protected NetStaffingCube getPreTOApprovalNetStaffingCube(TORequest tor,TOChoice choice) throws Exception{
		return getTOValidationCache(tor).getPreTOApprovalNetStaffingCube(tor, choice);
	}
	protected NetStaffingCube getPostTOApprovalNetStaffingCube(TORequest tor,TOChoice choice) throws Exception{
		return getTOValidationCache(tor).getPostTOApprovalNetStaffingCube(tor, choice);
	}
	@SuppressWarnings("unchecked")
	private Collection<ShiftAssignment> getPublishedShiftAssignments(TORequest tor, TOChoice choice) throws Exception {
		return getTOValidationCache(tor).getPublishedShiftAssignments(tor,choice);
	}
	protected TOValidationCache getTOValidationCache(TORequest toRequest) {
		return toRequest.getCache();
	}
	private Organization getOrganization(TORequest tor) throws Exception {
		return getTOValidationCache(tor).getOrg();
	}
	protected Localizer getLocalizer() {
		return DefaultLocalizationManager.getDefaultInstance().getLocalizer();
	}
