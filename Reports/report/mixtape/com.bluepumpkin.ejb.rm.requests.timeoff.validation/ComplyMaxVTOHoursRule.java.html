/**
 * Rule in UI: Requested Time Off is for the past date
 * <p/>
 * Check if request is for the past date.
 * this rule will raise alert for only those requests that are created for start time
 * that falls before creation date of the request.
 * <p/>
 */
public class ComplyMaxVTOHoursRule implements Validator {
	private static final String CLASSNAME = ComplyMaxVTOHoursRule.class.getName();

	private static final Category LOG = Log.initCategory(CLASSNAME);

	/**
	 * Validate a time off request.
	 */
	@SuppressWarnings("unchecked")
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		TORequest tor = (TORequest) validatable;
		String methodName = "validate";
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod(methodName, tor));
		}

		// If Validation By Type, then all activities may use for this validator
		if (!LicenseUtil.isAdvancedRMLicense() && !tor.getTimeOffType().equals(Activity.ACTIVITY_VOLUNTARY_TIMEOFF)) {
			return null;
		}

		ValidationResult result = null;
		Iterator<TOChoice> choiceListIter = tor.getRequestChoiceList().iterator();
		Employee emp = tor.getCache().getEmployee(Employee.DETAIL_LEVEL_EMPLOYEE_TIMEBASEDPROPERTY);
		ID empID = tor.getEmployeeID();
		Organization org = tor.getCache().getOrg();

		while (choiceListIter.hasNext()) {
			List<ShiftsConflict> llNewConflicts = new LinkedList<ShiftsConflict>();
			TOChoice choice = choiceListIter.next();
			Date startDate = choice.getStartDate();
			Date endDate = choice.getEndDate();
			TimeRange weekRange = new TimeRange(TOCalcUtil.getDateForOrgWeekStart(org, startDate), TOCalcUtil.getDateForOrgWeekEnd(org, endDate));

			//Get the max VTO per day and week for the employee's orgweek that includes the start of the current choice.
			WorkResourceMinMaxHour wrMinMax = tor.getCache().getMinMaxHourAssignment(empID, weekRange.getStartDate()); //this returns minutes. We should refactor method name to say minutes.
			int maxVTOPerDay = wrMinMax.getMaxVTOHoursPerDay();
			int maxVTOPerWeek = wrMinMax.getMaxVTOHoursPerWeek();

			//Accumulator for the scheduled+requested VTO minutes for the entire weekRange
			float weeklyVTOMinutes = 0.0f;

			//Accumulator for the scheduled+requested VTO minutes for each day in the weekRange
			HashMap<Date, Float> dailyVTOMap = new HashMap<Date, Float>();

			//Get the VTO hours per day which are already scheduled for the employee
			Collection hrsPerDayCol = TOCalendarUtil.getHoursPerDayForRange(Collections.singletonList(empID),
			        weekRange, Collections.singletonList(Activity.ACTIVITY_VOLUNTARY_TIMEOFF), true);

			//Iterate though the already-scheduled VTO per day and store it in the daily/weekly accumulators
			if (hrsPerDayCol != null) {
				for (Iterator<TOHoursPerDay> hrsPerDayIter = hrsPerDayCol.iterator(); hrsPerDayIter.hasNext();) {
					TOHoursPerDay hoursPerDay = hrsPerDayIter.next();
					weeklyVTOMinutes += getAccountedMinutes(hoursPerDay, weekRange); //GQ: Isn't this just accumulating all weeks if the choice spans more than one week?

					TimeRange dayRangeStart = new TimeRange(TOCalcUtil.getDateForOrgDayStart(org, hoursPerDay.getStartTime()), TOCalcUtil.getDateForOrgDayEnd(org, hoursPerDay.getStartTime()));
					float durationForDayStart = getAccountedMinutes(hoursPerDay, dayRangeStart);
					dailyVTOMap.put(dayRangeStart.getStartDate(), durationForDayStart);
				}
			}

			//Get the VTO hours per day for the request
			TOHoursPerDay reqHrsPerDay = choice.getHoursPerDay();

			//Iterate though the requested VTO per day and add it to the daily/weekly accumulators, and validate against the max
			if (reqHrsPerDay != null) {
				Calendar weekIterator = Calendar.getInstance(org.getTimeZone());
				weekIterator.setTime(weekRange.getStartDate());

				while (weekIterator.getTime().before(weekRange.getEndDate())) {
					Date curOrgDayStart = weekIterator.getTime();
					weekIterator.add(Calendar.DATE, 1);
					weekIterator.getTime();

					TimeRange dayRangeStart = new TimeRange(TOCalcUtil.getDateForOrgDayStart(org, curOrgDayStart), TOCalcUtil.getDateForOrgDayEnd(org, curOrgDayStart));
					float scheduledDurationForDayStart = dailyVTOMap.get(dayRangeStart.getStartDate())==null?0.0f : dailyVTOMap.get(dayRangeStart.getStartDate());
					float totalDurationForDayStart = getAccountedMinutes(reqHrsPerDay, dayRangeStart) + scheduledDurationForDayStart;

					//check daily vto
					if (maxVTOPerDay < totalDurationForDayStart)
					{
						ShiftsConflict cConflict = new ShiftsConflict(ShiftsConflict.SHIFT_VTO_DAILY, maxVTOPerDay, dayRangeStart.getStartDate());
						llNewConflicts.add(cConflict);
					}
				}
			}

			weeklyVTOMinutes += getAccountedMinutes(reqHrsPerDay, weekRange);

			//check weekly VTO
			if (weeklyVTOMinutes > maxVTOPerWeek) {
				ShiftsConflict cConflict = new ShiftsConflict(ShiftsConflict.SHIFT_VTO_WEEKLY, maxVTOPerWeek, weekRange.getStartDate());
				llNewConflicts.add(cConflict);
			}

			if (llNewConflicts != null && !llNewConflicts.isEmpty()) {
				result= RequestUtil.setValidationResult(empID, tor.getCache().getEmployeeNameByID(empID),choice , llNewConflicts, CLASSNAME);
			}
		}

		return result;
	}

	public float getAccountedMinutes(TOHoursPerDay hoursPerDay, TimeRange range) {
		float duration = 0.0f;
		Date hoursPerDayStart = hoursPerDay.getStartTime();
		Date hoursPerDayEnd = hoursPerDay.getEndTime();
		Date hoursPerDayEndNonInc = new Date(hoursPerDayEnd.getTime()-1); //subtract 1 millisecond since the end time may include the next instant.

		if (hoursPerDayStart.after(range.getEndDate()) || hoursPerDayEndNonInc.before(range.getStartDate())) {
			return duration; // this event is not relevant
		}
		boolean isToEvtStartBeforeStart = hoursPerDayStart.before(range.getStartDate());
		boolean isToEvtEndAfterEnd = hoursPerDayEndNonInc.after(range.getEndDate());
		if (isToEvtStartBeforeStart || isToEvtEndAfterEnd) {
			DailyHoursBuckets dailyBuckets = null;
			try {
				dailyBuckets = TOHoursPerDayUtil.getDailyHoursBucketsForDateRange(null, hoursPerDay,
				        isToEvtStartBeforeStart ? range.getStartDate() : hoursPerDayStart,
				        isToEvtEndAfterEnd ? range.getEndDate() : hoursPerDayEnd);
			} catch (Exception e) {
				LOG.error(e);
			}
			duration = dailyBuckets.getTotalHours() * 60;
		} else {
			duration = hoursPerDay.getTotalMinutes();
		}
		return duration;
	}
}
