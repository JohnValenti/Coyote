/**
 * "TimeoffIntervalAllocationReferencesScheduleRule" validation rule for time off requests.
 * For activity types ONLY marked as "interval allocation" (i.e., and not also "daily") verify
 * that the employee has a reference schedule.
 * If no reference schedule exists for the employee, a hard validation exception is raised.
 **/

public class TimeoffIntervalAllocationReferenceScheduleRule implements Validator {

    private static final Category LOG = Log.initCategory( TimeoffIntervalAllocationReferenceScheduleRule.class.getName());


    @Override
    public ValidationResult validate(Validatable validatable) throws Exception {

        String methodName = "validate";
        if (LOG.isDebugEnabled()){
        	LOG.debug(RmUtil.dumpEnterMethod(methodName, validatable));
        }

        ValidationResult result = null;

        TORequest toRequest = (TORequest)validatable;

        // If we are not using the interval pool, no need to validate
        boolean useIntervalPool = isUseIntervalTOPool(toRequest);
        if(!useIntervalPool){
        	return result;
        }


		ID empId = toRequest.getEmployeeID();

		if(!validateReferenceScheduleGood(toRequest)){
        	result = ValidationUtil
    			.setHardValidationResult(toRequest,
    				RmEjbBundleKey.TIMEOFFINTERVALALLOCATIONSEGMENT_NO_REFERENCE_SCHEDULE,
    				getEmployeeByID(empId), this.getClass().getName());

		}


        if (LOG.isDebugEnabled()) {
        	LOG.debug(RmUtil.dumpExitMethod(methodName, result));
        }
        return result;
    }

	protected Employee getEmployeeByID(ID empId) throws Exception {
		return ValidationUtil.getEmployeeByID(empId, null, Employee.DETAIL_LEVEL_EMPLOYEE_BASIC);
	}

	protected boolean validateReferenceScheduleGood(TORequest toRequest) throws Exception {
		return TimeOffIntervalAllocationUtil.validateReferenceScheduleGood(toRequest);
	}

	protected boolean isUseIntervalTOPool(TORequest toRequest) throws Exception {
		return TORequestUtil.isUseIntervalTOPool(toRequest);
	}
}
