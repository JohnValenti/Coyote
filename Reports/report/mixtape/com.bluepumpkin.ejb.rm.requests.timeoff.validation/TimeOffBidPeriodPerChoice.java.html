package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

/**
 * Title:        TimeOffBidPeriodPerChoiceHV
 * Description:  Hard validation alert when a Time Off Request of activity type specified for a bid is made during an open bid period by an employee in the bid,
 * 				 for a period in the request period of the bid, the hard validation rule for "Minimum of X Weeks/Days per Choice" and "Maximum of X Weeks/Days per Choice"
 * 				 is applied upon submission where appropriate. Besides that, if the minimum/maximum uni is week, the bid duration should be increments of weeks
 * 				 This validation is invoked only if there is Advanced RM license
 * Copyright:    Copyright (c) 2016
 * Company:      Verint System, Inc
 * @author       Truc Pham
 * @version      1.0
**/
import java.util.Date;
import java.util.Iterator;
import java.util.TimeZone;
import java.util.concurrent.TimeUnit;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.localization.DefaultLocalizationManager;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.ejb.TimeOffBidUtil;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBid;

/**
 *         <p/>
 *        When a Time Off Request of activity type specified for employee added to the bid while the bid is open,
 * 		  for a period in the request period of the bid, the hard validation rule for "Minimum of X Weeks/Days per Choice" and "Maximum of X Weeks/Days per Choice"
 * 			is applied upon submission where appropriate...
 * 		  Besides that, if the minimum/maximum uni is week, the bid duration should be increments of weeks
 *         <p/>
 *         <p> Note: This validation is invoked only if there is Advanced RM license.
 */
public class TimeOffBidPeriodPerChoice implements Validator {

	private static final String CLASS_NAME = TimeOffBidPeriodPerChoice.class.getName();

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	@SuppressWarnings("rawtypes")
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		TORequest toReq = (TORequest) validatable;
		
		ID timeOffType = toReq.getTimeOffType();
		ID employeeId = toReq.getEmployeeID();
		TimeOffBid bid = getOpenTimeOffBidForActivityIdAndEmployeId(timeOffType, employeeId, toReq.getSubmittedOn());
		if (bid != null) {
			String bidName = bid.getName();
			long bidRequestStart = bid.getRequestStartTime().getTime();
			long bidRequestEnd = bid.getRequestEndTime().getTime();
			int minDaysRequired = TimeOffBidUtil.getMinimumDaysPerChoice(bid);
			int maxDaysRequired = TimeOffBidUtil.getMaximumsPerChoice(bid);
			long dayInMiliseconds = TimeUnit.DAYS.toMillis(1);
			long minimumDurationRequired = minDaysRequired * dayInMiliseconds;
			long maximumDurationRequired = maxDaysRequired * dayInMiliseconds;
			Localizer localizer = getLocalizer();
			String minPeriodText = TimeOffBidUtil.getPeriodText(localizer, bid.getMinValuePerChoice(), bid.getMinUnitPerChoice());
			String maxPeriodText = TimeOffBidUtil.getPeriodText(localizer, bid.getMaxValuePerChoice(), bid.getMaxUnitPerChoice());
			TimeZone empTimeZone = getTimeZoneOfEmployee(employeeId);
			for (Iterator it = toReq.getRequestChoiceList().iterator(); it.hasNext();) {
				TOChoice choice = (TOChoice) it.next();
				long choiceStart = choice.getStartDate().getTime();
				long choiceEnd = choice.getEndDate().getTime();
				long choiceDuration = TimeOffBidUtil.getRealChoiceDuration(choice.getStartDate(), choice.getEndDate(), empTimeZone);
				boolean choicePeriodInBidRequestPeriod = bidRequestStart <= choiceStart && choiceEnd <= bidRequestEnd;
				if (choicePeriodInBidRequestPeriod) {
					//validate Min/Max for Time off choice
					result = validateMinMaxPeriod(choice, choiceDuration, minimumDurationRequired, minPeriodText, maximumDurationRequired,
							maxPeriodText, bidName);
				}
			}
		}
		return result;
	}

	protected ValidationResult validateMinMaxPeriod(TOChoice choice, long choiceDuration, long minimumDurationRequired, String minPeriodText,
			long maximumDurationRequired, String maxPeriodText, String bidName) {
		ValidationResult result = null;
		if (choiceDuration < minimumDurationRequired) {
			result = ValidationUtil.setSoftValidationResult(choice, RmEjbBundleKey.TIMEOFF_BID_TOCHOICE_VIOLATE_MIN_PERIOD_VALUE,
					choice.getStartDate(), choice.getEndDate(), minPeriodText, bidName, CLASS_NAME);
			return result; //no point in going ahead from here as the validation is already failed.
		}
		if (choiceDuration > maximumDurationRequired) {
			result = ValidationUtil.setSoftValidationResult(choice, RmEjbBundleKey.TIMEOFF_BID_TOCHOICE_VIOLATE_MAX_PERIOD_VALUE,
					choice.getStartDate(), choice.getEndDate(), maxPeriodText, bidName, CLASS_NAME);
			return result; //no point in going ahead from here as the validation is already failed.
		}
		return result;
	}

	protected TimeOffBid getOpenTimeOffBidForActivityIdAndEmployeId(ID timeOffType, ID employeeId, Date submittedDate) throws Exception {
		return TimeOffBidUtil.getOpenTimeOffBidForActivityIdAndEmployeId(timeOffType, employeeId, submittedDate);
	}

	protected Localizer getLocalizer() {
		return DefaultLocalizationManager.getDefaultInstance().getLocalizer();
	}
	protected TimeZone getTimeZoneOfEmployee(ID employeeId) throws BbmFinderException {
		return TimeOffBidUtil.getTimeZoneOfEmployee(employeeId);
	}
}
