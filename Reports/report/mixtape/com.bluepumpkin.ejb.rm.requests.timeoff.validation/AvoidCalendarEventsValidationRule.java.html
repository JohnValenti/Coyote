package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.Collection;

import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEvent;

/**
 * Rule in UI: Request avoids scheduled calendar events.
 * <p/>
 * Check each TOChoice does not overlap a scheduled calendar event (An event
 * in the CALENDAREVENTASSIGNMENT table with
 * event type == {@link com.bluepumpkin.ejb.bbm.activity.model.Event#EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT})
 * <p/>
 * <p/>
 * Applies only for "debit" and "debit-if" requests.  Does not apply for "don't debit" requests
 * as unavailability events cannot overlap shifts (rule not skipped as no shift will be found
 * (enforced by hard validation)).
 * <p/>
 * <p/>
 * Applies equally for tentative and non tentative requests (no unpublished calendar event assignments used).
 */
public class AvoidCalendarEventsValidationRule implements Validator {

	private static final String m_className = AvoidCalendarEventsValidationRule.class.getName();

	private static final Category m_cat = Log.initCategory(m_className);
	public static final String CALENDAR_EVENT_ACTIVITY = "CALENDAR_EVENT_ACTIVITY";
	public static final ID RULE_ID = new ID(-194006);

	/**
	 * Validate a time off request.
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		TORequest tor = (TORequest) validatable;

		String methodName = "validate";
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		// Check for calendar events
		Collection calEvents = tor.getCache().getCalEventsDuringPeriod(true, false, false);
		// Loop through the choices and check that each not overlap a calendar event
		TOChoice toChoice = tor.getTOChoiceForValidation();

		if (toChoice == null) {
			Iterator it = tor.getRequestChoiceList().iterator();
			while (it.hasNext()) {
				toChoice = (TOChoice) it.next();

				result = setValidationResultForAvoidCalendarEvents(calEvents, tor, toChoice,
				        getActivityCollFromMap(tor, toChoice));
				// if the TOChoice  overlaps with a calendar event.
			} // end while (it.hasNext())
		} else {    // validate one TO CHOICE ONLY
			result = setValidationResultForAvoidCalendarEvents(calEvents, tor, toChoice,
			        getActivityCollFromMap(tor, toChoice));
		}
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
		return result;
	}

	private Collection getActivityCollFromMap(TORequest tor, TOChoice toChoice) throws Exception {
		//get activities to validate against from rule
		//get the rule params
		ID activityID = tor.getRequestSubType();
		ValidationRuleManager valMan = RmManagerFactory.getInstance().getValidationRuleManager();
		Map<String, Integer> ruleParams = valMan.getValidationRuleParams(
				tor.getValidationCache()
 		        .getOrgIDForEmployeeDuringPeriod(tor.getEmployeeID(), toChoice.getStartDate(), toChoice.getEndDate()), RULE_ID, activityID);

		ArrayList activityColl = new ArrayList(ruleParams.size());
		if (ruleParams != null && ruleParams.isEmpty())
			return null;
		for (Iterator it = ruleParams.values().iterator(); it.hasNext();) {
			activityColl.add(new ID(((Integer) it.next()).intValue()));
		}
		return activityColl;
	}

	public ValidationResult setValidationResultForAvoidCalendarEvents(Collection calEvents,
	                                                                  TORequest tor, TOChoice toChoice, Collection activityColl) throws Exception {
		ValidationResult result = null;

		for (Iterator it = calEvents.iterator(); it.hasNext();) {
			CalendarEvent calEvent = (CalendarEvent) it.next();
			//set validation result for the TOCHOICE only if the calendar event falls in the choice date range
			if (intersect(toChoice, calEvent)) {
				if (activityColl == null || activityColl.contains(calEvent.getActivityID()) ||
				        activityColl.contains(new ID(-2))) {
					result = ValidationUtil.setSoftValidationResult(toChoice,
					        RmEjbBundleKey.TO_CONTAINED_IN_CALEVT, tor.getCache().getEmployeeNameByID(),
					        toChoice.getStartDate(), toChoice.getEndDate(), m_className);
				}
			}
		}
		return result;
	}

	protected static boolean intersect(TOChoice toChoice, CalendarEvent calEvent) {
		return calEvent.getEndTime().after(toChoice.getStartDate()) && calEvent.getStartTime().before(toChoice.getEndDate());
	}
}
