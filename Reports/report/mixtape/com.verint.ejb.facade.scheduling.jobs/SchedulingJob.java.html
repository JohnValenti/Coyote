package com.verint.ejb.facade.scheduling.jobs;

import org.quartz.StatefulJob;
import org.quartz.JobExecutionContext;
import org.quartz.JobExecutionException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.Lock;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.facade.FacadeManagerFactory;
import com.verint.ejb.facade.scheduling.ScheduleMessageScheduler;

import java.util.*;

/**
 * Class to Schedule the requests in the WSAutoScheduleRequest table
 */
public class SchedulingJob implements StatefulJob {
	private static Category m_cat = Log.initCategory(SchedulingJob.class.getName());
	/**
	 * lock to ensure that this job is not reexecuted while an earlier instance
	 * is still executing.
	 */
	static Lock m_jobExecLock = new Lock();
	/*
	 * @see org.quartz.Job#execute(org.quartz.JobExecutionContext)
	 */
	static long lastRunTime = 0;
	static int lastKnownRunHour = 0;

	public void execute(JobExecutionContext execCtxt) throws JobExecutionException {
		String methodName = "execute";
		long start = System.currentTimeMillis();
		try {
			// try to obtain the lock.
			if (!m_jobExecLock.lock()) {
				m_cat.debug("Quartz job not executed as an earlier job is still executing");
				return;
			}

			ScheduleMessageScheduler scheduler = FacadeManagerFactory.getScheduleMessageScheduler();
			updateStatus(scheduler);
			scheduleRequests(scheduler);
			clearOrphanRecords(scheduler);
		} catch (Exception e) {			
			m_cat.error(methodName, e);
			throw new JobExecutionException(e, false);
		}finally{
			// release lock if it is being held.
			if (m_jobExecLock.isLocked()) {
				m_jobExecLock.releaseLock();
			}
			m_cat.debug("Time for Execution = " + (System.currentTimeMillis() - start) / 1000 + " seconds");
		}
	}

	private void updateStatus(ScheduleMessageScheduler scheduler) {
		try {
			scheduler.updateStatusAndClearCompletedRequests();
		} catch (Exception any) {
			m_cat.error("Could not update/clear scheduling states.", any);
		}
	}

	private void scheduleRequests(ScheduleMessageScheduler scheduler) {
		try {
			scheduler.scheduleReadyRequests();
		} catch (Exception any) {
			m_cat.error("Could not schedule requests.", any);
		}
	}

	private void clearOrphanRecords(ScheduleMessageScheduler scheduler) {
		try {
			scheduler.clearOrphanRequests();
		} catch (Exception any) {
			m_cat.error("Could not clear new / in-progress scheduling requests not mapped to message", any);
		}
	}
}
