package com.verint.web.rm.rest.entity.summary;

import com.verint.web.rest.entity.AbstractWFOEntity;

public abstract class RequestSummaryEntity extends AbstractWFOEntity {

	private int requestsTotal;
	private int requestsExpired;
	private int requestsPending;
	private int requestsApproved;
	private int requestsDenied;
	private int requestsInvalid;

	public RequestSummaryEntity() { // NOSONAR
	}

	public int getRequestsTotal() {
		return requestsTotal;
	}

	public void setRequestsTotal(int requestsTotal) {
		this.requestsTotal = requestsTotal;
	}

	public int getRequestsExpired() {
		return requestsExpired;
	}

	public void setRequestsExpired(int requestsExpired) {
		this.requestsExpired = requestsExpired;
	}

	public int getRequestsPending() {
		return requestsPending;
	}

	public void setRequestsPending(int requestsPending) {
		this.requestsPending = requestsPending;
	}

	public int getRequestsApproved() {
		return requestsApproved;
	}

	public void setRequestsApproved(int requestsApproved) {
		this.requestsApproved = requestsApproved;
	}

	public int getRequestsDenied() {
		return requestsDenied;
	}

	public void setRequestsDenied(int requestsDenied) {
		this.requestsDenied = requestsDenied;
	}

	public int getRequestsInvalid() {
		return requestsInvalid;
	}

	public void setRequestsInvalid(int requestsInvalid) {
		this.requestsInvalid = requestsInvalid;
	}

}
