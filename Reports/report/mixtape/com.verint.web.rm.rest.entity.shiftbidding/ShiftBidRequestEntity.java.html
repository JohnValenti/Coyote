package com.verint.web.rm.rest.entity.shiftbidding;

import java.util.ArrayList;
import java.util.List;

import com.bluepumpkin.ejb.rm.requests.common.model.Request;
import com.verint.web.rm.rest.entity.RequestEntity;

public class ShiftBidRequestEntity extends RequestEntity {

	public static final String TYPE = Request.REQUESTTYPE_SHIFTBID;

	private int auctionID;
	private boolean isBonusUsed;
	private String name;
	private int preference;
	private final List<BiddableScheduleEntity> schedules = new ArrayList<BiddableScheduleEntity>();

	public ShiftBidRequestEntity() {
		super(TYPE);
	}

	public int getAuctionID() {
		return auctionID;
	}

	public void setAuctionID(int auctionID) {
		this.auctionID = auctionID;
	}

	public boolean isBonusUsed() {
		return isBonusUsed;
	}

	public void setBonusUsed(boolean isBonusUsed) {
		this.isBonusUsed = isBonusUsed;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public int getPreference() {
		return preference;
	}

	public void setPreference(int preference) {
		this.preference = preference;
	}

	public List<BiddableScheduleEntity> getSchedules() {
		return schedules;
	}

	public static String getType() {
		return TYPE;
	}
}
