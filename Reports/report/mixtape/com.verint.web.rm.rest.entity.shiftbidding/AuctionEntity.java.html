package com.verint.web.rm.rest.entity.shiftbidding;

import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidauction.model.ShiftBidAuction;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidder;
import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rest.util.FormatUtils;

/**
 * Represents an Auction.
 */
public class AuctionEntity extends AbstractWFOEntity {

	private final String auctionDeadline;
	private final int bonusForNoBidsAccepted;
	private String deadline;
	private final String endTime;
	private final boolean fullPeriodSchedule;
	private final boolean isMultiShift;
	private final int maxShiftBlocksPerBid;
	private final int maxHoursPerDay;
	private final int maxConsecDaysPerBid;
	private final int minHoursPerDay;
	private final String name;
	private int shiftBidderID;
	private final String startTime;
	private final boolean usesScoring;
	private final int status;

	private int employeeRequestCount;
	private int maxPreference;

	public AuctionEntity(ShiftBidAuction auction) {
		setId(auction.getID().toString());
		auctionDeadline = FormatUtils.dateToISO8601SecPrecisionString(auction.getAuctionDeadLine());
		bonusForNoBidsAccepted = auction.getBonusForNoBidsAccepted();
		endTime = FormatUtils.dateToISO8601SecPrecisionString(auction.getEndTime());
		fullPeriodSchedule = auction.getIsFullPeriodSchedule();
		isMultiShift = auction.getIsMultiShift();
		maxConsecDaysPerBid = auction.getMaxConsecDaysPerBid();
		maxHoursPerDay = auction.getMaxHoursPerDay();
		maxShiftBlocksPerBid = auction.getMaxShiftBlocksPerBid();
		minHoursPerDay = auction.getMinHoursPerDay();
		name = auction.getName();
		startTime = FormatUtils.dateToISO8601SecPrecisionString(auction.getStartTime());
		usesScoring = auction.getUsesScoring();
		status = auction.getStatus();
	}

	public String getAuctionDeadline() {
		return auctionDeadline;
	}

	public int getBonusForNoBidsAccepted() {
		return bonusForNoBidsAccepted;
	}

	public String getDeadline() {
		return deadline;
	}

	public int getEmployeeRequestCount() {
		return employeeRequestCount;
	}

	public void setEmployeeRequestCount(int employeeRequestCount) {
		this.employeeRequestCount = employeeRequestCount;
	}

	public String getEndTime() {
		return endTime;
	}

	public boolean isFullPeriodSchedule() {
		return fullPeriodSchedule;
	}

	public boolean isMultiShift() {
		return isMultiShift;
	}

	public int getMaxShiftBlocksPerBid() {
		return maxShiftBlocksPerBid;
	}

	public int getMaxHoursPerDay() {
		return maxHoursPerDay;
	}

	public int getMaxConsecDaysPerBid() {
		return maxConsecDaysPerBid;
	}

	public int getMinHoursPerDay() {
		return minHoursPerDay;
	}

	public int getMaxPreference() {
		return maxPreference;
	}

	public void setMaxPreference(int maxPreference) {
		this.maxPreference = maxPreference;
	}

	public String getName() {
		return name;
	}

	public int getShiftBidderID() {
		return shiftBidderID;
	}

	public void setShiftBidderID(int shiftBidderID) {
		this.shiftBidderID = shiftBidderID;
	}

	public String getStartTime() {
		return startTime;
	}

	public boolean isUsesScoring() {
		return usesScoring;
	}

	public int getStatus() {
		return status;
	}

	/**
	 * Sets the various ShiftBidder properties.
	 */
	public void setShiftBidder(ShiftBidder bidder) {
		shiftBidderID = bidder.getID().toInt();
		this.deadline = FormatUtils.dateToISO8601SecPrecisionString(bidder.getDeadlineDate());
	}
}
