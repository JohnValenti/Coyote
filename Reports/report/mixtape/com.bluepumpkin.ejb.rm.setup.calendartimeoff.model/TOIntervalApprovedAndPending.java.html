package com.bluepumpkin.ejb.rm.setup.calendartimeoff.model;

import java.util.Date;

public class TOIntervalApprovedAndPending extends TOIntervalAllocation { // NOSONAR

	private static final long serialVersionUID = 1L;

	int approvedMinutesForEmployee;
	int approvedMinutes;
	int pendingMinutesForEmployee;
	int pendingMinutes;

	public TOIntervalApprovedAndPending() {
		super();
	}

	public TOIntervalApprovedAndPending(Date startTime, Date endTime) {
		super(startTime, endTime);
	}

	public int getApprovedMinutes() {
		return approvedMinutes;
	}

	public void setApprovedMinutes(int approvedMinutes) {
		this.approvedMinutes = approvedMinutes;
	}

	public int getApprovedMinutesForEmployee() {
		return approvedMinutesForEmployee;
	}

	public void setApprovedMinutesForEmployee(int approvedMinutesForEmployee) {
		this.approvedMinutesForEmployee = approvedMinutesForEmployee;
	}

	public int getPendingMinutes() {
		return pendingMinutes;
	}

	public float getPendingHours() {
		return pendingMinutes / 60.0f;
	}

	public float getApprovedHours() {
		return approvedMinutes / 60.0f;
	}

	public float getEmployeePendingHours() {
		return pendingMinutesForEmployee / 60.0f;
	}

	public float getEmployeeApprovedHours() {
		return approvedMinutesForEmployee / 60.0f;
	}

	public void setPendingMinutes(int pendingMinutes) {
		this.pendingMinutes = pendingMinutes;
	}

	public int getPendingMinutesForEmployee() {
		return pendingMinutesForEmployee;
	}

	public void setPendingMinutesForEmployee(int pendingMinutesForEmployee) {
		this.pendingMinutesForEmployee = pendingMinutesForEmployee;
	}
}
