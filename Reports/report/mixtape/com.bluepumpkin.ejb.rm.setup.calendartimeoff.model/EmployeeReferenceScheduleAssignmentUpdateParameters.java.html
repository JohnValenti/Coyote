package com.bluepumpkin.ejb.rm.setup.calendartimeoff.model;

import java.util.Date;



import com.bluepumpkin.common.datatypes.ID;

/**
 * This is used as a parameter object for the database update.
 * see EmployeeReferenceScheduleAssignmentDAO.updateEmployeeReferenceScheduleEffectivity
 * EmployeeReferenceScheduleAssignment is used to read data, this class is used to update multiple employees. 
 */
public class EmployeeReferenceScheduleAssignmentUpdateParameters implements java.io.Serializable {
	
	private static final long serialVersionUID = 4749105671242990537L;
	
	ID schedulingPeriodDEID;
	Date startTime;
	Date endTime;
	boolean updateSpID;
	boolean updateStartTime;
	boolean updateEndTime;
	
	public ID getSchedulingPeriodDEID() {
		return schedulingPeriodDEID;
	}
	public void setSchedulingPeriodDEID(ID spID) {
		this.schedulingPeriodDEID = spID;
	}
	public Date getStartTime() {
		return startTime;
	}
	public void setStartTime(Date startTime) {
		this.startTime = startTime;
	}
	public Date getEndTime() {
		return endTime;
	}
	public void setEndTime(Date endTime) {
		this.endTime = endTime;
	}
	public boolean isUpdateSpID() {
		return updateSpID;
	}
	public void setUpdateSpID(boolean updateSpID) {
		this.updateSpID = updateSpID;
	}
	public boolean isUpdateStartTime() {
		return updateStartTime;
	}
	public void setUpdateStartTime(boolean updateStartTime) {
		this.updateStartTime = updateStartTime;
	}
	public boolean isUpdateEndTime() {
		return updateEndTime;
	}
	public void setUpdateEndTime(boolean updateEndTime) {
		this.updateEndTime = updateEndTime;
	}
	

}
