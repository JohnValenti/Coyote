package com.bluepumpkin.ejb.rm.setup.calendartimeoff.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class TOIntervalAllocation extends ValueObjectBase {

	private static final long serialVersionUID = 1L;
	private static final FieldInfo FIELD_INFO = new TOIntervalAllocationFieldInfo();

	public TOIntervalAllocation() {
		super();
	}

	public TOIntervalAllocation(Date startTime, Date endTime) {
		this();
		this.setStartTime(startTime);
		this.setEndTime(endTime);
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return TOIntervalAllocation.FIELD_INFO;
	}

	public ID getTOPoolID() {
		return getFieldValueID(TOIntervalAllocationFieldInfo.TOPOOLID);
	}

	public void setTOPoolID(ID id) {
		setFieldValue(TOIntervalAllocationFieldInfo.TOPOOLID, id);
	}

	public Date getStartTime() {
		return getFieldValueDate(TOIntervalAllocationFieldInfo.STARTTIME);
	}

	public void setStartTime(Date dt) {
		setFieldValue(TOIntervalAllocationFieldInfo.STARTTIME, dt);
	}

	public Date getEndTime() {
		return getFieldValueDate(TOIntervalAllocationFieldInfo.ENDTIME);
	}

	public void setEndTime(Date dt) {
		setFieldValue(TOIntervalAllocationFieldInfo.ENDTIME, dt);
	}

	public float getAllocatedHours() {
		return getFieldValueFloat(TOIntervalAllocationFieldInfo.ALLOCATEDHOURS);
	}

	public void setAllocatedHours(float hours) {
		setFieldValue(TOIntervalAllocationFieldInfo.ALLOCATEDHOURS, hours);
	}
}
