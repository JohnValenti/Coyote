/**
 * Wrapper around a an array of floats. Each array element represents an org
 * day's allocation such as available TO Hours, computed TO hours etc. Can
 * retrieve and set elements in the array using a 'date' as an index (instead of
 * using a numeric index).
 * 
 * <p>
 * Allocates a float array representing hours per organization day. These can be
 * available hours, request hours, taken hours.... This class also provides
 * operations to compare HoursBucket objects.
 */
public class DailyHoursBuckets implements IDailyHoursBuckets {
	private static final String CLASSNAME = DailyHoursBuckets.class.getName();
	private static Category LOG = Log.initCategory(CLASSNAME);

	 * 2 = overlaps scheduled and unscheduled period

	 * 0.

	protected TimeRange dailyHrsBucketsRangeAsOrgDayStart = null; // m_DHBRangeInOrgDays

	public DailyHoursBuckets(int dayBoundaryOffset, TimeZone tz, TimeRange range, boolean inclHrsPerDayDetails) {
		this(dayBoundaryOffset, tz, range);
		this.inclTOHrsPerDayDetails = inclHrsPerDayDetails;
		if (this.inclTOHrsPerDayDetails) {
			TOEventDetailArray = new ArrayList[buckets.length];
		}

	/**
	 * Creates a new instance of HoursBucket
	 * 
	 * @param dayBoundaryOffset
	 *            the integer number of minutes from midnight, local time that
	 *            markes the start of a new organization day.
	 * @param tz
	 *            the organization time zone
	 * @param range
	 *            the time range for the DailyHoursBuckets object
	 */
	public DailyHoursBuckets(int aDayBoundaryOffset, TimeZone tz, TimeRange range) {

		if (aDayBoundaryOffset < 0 || aDayBoundaryOffset >= 60 * 24) {
			throw new IllegalArgumentException("aDayBoundaryOffset " + aDayBoundaryOffset);
		}
		dayBoundaryOffsetInMins = aDayBoundaryOffset;

		if (tz == null) {
			throw new IllegalArgumentException("TimeZone is null");
		}
		timeZone = tz;

		if (range == null) {
			throw new IllegalArgumentException("TimeRange is null");
		}
		// bad hack sameet Apr 2010
		// we should not create dailyhourbuckets with 1 extra day if enddate is
		// same as start of org day
		// because it is redundant; no hours will ever be accounted for that
		// day.
		if (TOCalcUtil.getDateForOrgDayStart(timeZone, dayBoundaryOffsetInMins, range.getEndDate()).equals(range.getEndDate())) {
			LOG.debug(" deducting 1 msec from end date :" + range.getEndDate());
			range.setEndDate(new Date(range.getEndDate().getTime() - 1));
		}

		dailyHrsBucketsRangeAsOrgDayStart = new TimeRange(TOCalcUtil.getDateForOrgDayStart(timeZone, dayBoundaryOffsetInMins, range.getStartDate()),
				TOCalcUtil.getDateForOrgDayStart(timeZone, dayBoundaryOffsetInMins, range.getEndDate()));

		dailyHrsBucketsRangeEndAsOrgDayEnd = TOCalcUtil.getDateForOrgDayEnd(timeZone, dayBoundaryOffsetInMins, dailyHrsBucketsRangeAsOrgDayStart.getEndDate());

		dailyHrsBucketsRangeStartAsOrgDayStartCal = Calendar.getInstance(timeZone);
		dailyHrsBucketsRangeStartAsOrgDayStartCal.setTime(dailyHrsBucketsRangeAsOrgDayStart.getStartDate());

		// get the difference, in days, between the range's end date and
		// m_rangeStartOrgDayCal.
		int endIndex = getIndex(range.getEndDate());

		// Allocate the bucket array
		// The size of the array is one more than the index of the last day
		buckets = new float[endIndex + 1];
	}

	public DailyHoursBuckets(DailyHoursBuckets copy) {
		dayBoundaryOffsetInMins = copy.dayBoundaryOffsetInMins;
		timeZone = copy.timeZone;
		dailyHrsBucketsRangeAsOrgDayStart = copy.dailyHrsBucketsRangeAsOrgDayStart;
		dailyHrsBucketsRangeEndAsOrgDayEnd = copy.dailyHrsBucketsRangeEndAsOrgDayEnd;
		dailyHrsBucketsRangeStartAsOrgDayStartCal = copy.dailyHrsBucketsRangeStartAsOrgDayStartCal;

		buckets = new float[copy.buckets.length];

		for (int i = 0; i < copy.buckets.length; i++) {
			buckets[i] = copy.buckets[i];
		}

		inclTOHrsPerDayDetails = copy.inclTOHrsPerDayDetails;
		if (inclTOHrsPerDayDetails) {
			TOEventDetailArray = new ArrayList[copy.TOEventDetailArray.length];
			for (int i = 0; i < copy.TOEventDetailArray.length; i++) {
				TOEventDetailArray[i] = new ArrayList();
				TOEventDetailArray[i].addAll(copy.TOEventDetailArray[i]);
			}
		}

		totalHours = copy.totalHours;
		periodFlag = copy.periodFlag;
		totalHoursScheduled = copy.totalHoursScheduled;
		totalHoursDefaulted = copy.totalHoursDefaulted;
	}

	/**
	 * Add the selected amount to the bucket that starts on the day indicated by
	 * the local date.
	 * <p>
	 * Note: only the year/month/day fields of the given local date are used.
	 * The hour/minute/second fields are initialized using this bucket object's
	 * organization dayBoundaryOffset.
	 */
	public void addToBucket(LocalDate ld, float hours, int sourceFlag) {
		// Add hours to the bucket
		buckets[getIndex(ld)] += hours;

		if (sourceFlag == DailyHoursBuckets.HOURS_SOURCE_SCHEDULE) {
			totalHoursScheduled += hours;
		} else if (sourceFlag == DailyHoursBuckets.HOURS_SOURCE_DEFAULT) {
			totalHoursDefaulted += hours;
		}

		// invalidate cache
		totalHours = -1;
	}

	public void addToBucket(Date d, float hours, ValueObjectBase hrsPerDay) {
		addToBucket(d, hours, hrsPerDay, DailyHoursBuckets.HOURS_SOURCE_UNINITIALIZED);
	}

	/**
	 * Add the selected amount to the bucket that starts on the day that
	 * contains the given date.
	 */
	public void addToBucket(Date d, float hours, ValueObjectBase hrsPerDay, int sourceFlag) {
		// Add hours to the bucket
		int index = getIndex(d);
		buckets[index] += hours;
		if (sourceFlag == DailyHoursBuckets.HOURS_SOURCE_SCHEDULE) {
			totalHoursScheduled += hours;
		} else if (sourceFlag == DailyHoursBuckets.HOURS_SOURCE_DEFAULT) {
			totalHoursDefaulted += hours;
		}

		if (TOEventDetailArray[index] == null) {
			TOEventDetailArray[index] = new ArrayList();
		}
		TOEventDetailArray[index].add(hrsPerDay);
		// invalidate cache
		totalHours = -1;
	}

	public void addToBucket(Date d, float hours) {

	public boolean isDateInRange(Date dt) {
		return (dt != null && dt.before(getDHBRangeEndAsOrgDayEnd()) && !dt.before(getDHBRangeStartAsOrgDayStart()));
	}

	// returns Hours for a given date without considering Time component.
	// the API will internally use the start of Day for the relevant org for the
	// given date.
	// Day start will be computed based on Timezone and day boundary of the Org
	// to which the buckets belong to;
	public Date getOrgDayStart(Calendar calInp) {
		Calendar cal = Calendar.getInstance(getTimeZone());
		cal.set(Calendar.YEAR, calInp.get(Calendar.YEAR));
		cal.set(Calendar.MONTH, calInp.get(Calendar.MONTH));
		cal.set(Calendar.DAY_OF_MONTH, calInp.get(Calendar.DAY_OF_MONTH));
		TOCalcUtil.setCalMinsOffsetFromMidnight(cal, dayBoundaryOffsetInMins);
		return cal.getTime();
	}

	private void subFromBucket(LocalDate ld, float hours, int sourceFlag) {
		// Add hours to the bucket
		buckets[getIndex(ld)] -= hours;

		if (sourceFlag == DailyHoursBuckets.HOURS_SOURCE_SCHEDULE) {
			totalHoursScheduled -= hours;
		} else if (sourceFlag == DailyHoursBuckets.HOURS_SOURCE_DEFAULT) {
			totalHoursDefaulted -= hours;
		}
		// invalidate cache
		totalHours = -1;
	}

	/**
	 * subtracts the hours from this bucket using the given bucket
	 * 
	 * @param srcBuckets
	 * @param start
	 * @param end
	 */
	public void subFromBucket(DailyHoursBuckets srcBuckets, Date start, Date end) {
		subFromBucket(srcBuckets, start, end, DailyHoursBuckets.HOURS_SOURCE_UNINITIALIZED);
	}

	public void subFromBucket(DailyHoursBuckets srcBuckets, Date start, Date end, int sourceFlag) {
		Date currDate = start;
		while (!end.before(currDate)) {
			float hrs = srcBuckets.getBucketHours(currDate);
			subFromBucket(currDate, hrs, sourceFlag);
			currDate = TOCalcUtil.getDateForwardOneDay(currDate, getTimeZone());
		}
	}

	public void resetBuckets() {
		Arrays.fill(buckets, 0.0f);
		totalHoursScheduled = 0;
		totalHoursDefaulted = 0;

		if (inclTOHrsPerDayDetails) {
			TOEventDetailArray = new ArrayList[buckets.length];
		}
	}

	public int getTotalMinutes() {
		return Math.round(getTotalHours() * 60f);
	}

	public float getTotalHours() {
		// return cached value if cache is still valid.
		if (totalHours >= 0)
			return totalHours;

		// recompute as cache is invalid.
		totalHours = 0;
		for (int i = 0; i < buckets.length; i++) {
			totalHours += buckets[i];
		}

		return totalHours;
	}

	/**
	 * Check that the allocation of hours for each day in this set of buckets is
	 * at least as large as the amount of time in the requested buckets. Return
	 * the first date (beginning of org day) that has the requeted time exceed
	 * the available.
	 * 
	 * @param requestedBuckets
	 *            buckets representing a time off request
	 * @return the first date that doesn't have enough available hours or null
	 *         if there are enough available hours for all the days in the
	 *         request.
	 */
	public Date checkAllocation(DailyHoursBuckets requestedBuckets) {
		Date result = null;

		Calendar vCal = (Calendar) requestedBuckets.dailyHrsBucketsRangeStartAsOrgDayStartCal.clone();
		Date d = vCal.getTime();
		Date end = requestedBuckets.dailyHrsBucketsRangeAsOrgDayStart.getEndDate();
		do {
			// QA 81106/93213 - Time off requests are triggering the
			// "Organization has available time off hours" validation rule when
			// the day after the request does not have enough hours available
			// if the request is one day request but the bucket have two item,
			// the second one
			// is 0.0 and for the 'this' buckes the second item is -4.0 then the
			// if will fail
			// so we should check the request bucket hours > 0 and > 'this'
			// bucket hours
			if (requestedBuckets.getBucketHours(d) > 0 && requestedBuckets.getBucketHours(d) > this.getBucketHours(d)) {
				// found a day that doesn't have enough hours
				result = d;
				break;
			}
			vCal.add(Calendar.DATE, 1);
			d = vCal.getTime();
			
			// QA 77903 - Time-zone differences allow auto-approval of vacation
			// requests when request adjusts to cross day boundry
			// if the bucket has 2 items which has end date = start date + 1
			// then above while will be fail
		} while (!d.after(end));
		return result;
	}

	/**
	 * Find the index for a day given a LocalDate that specifies the
	 * month/day/year
	 * 
	 * <p>
	 * Note: the hours/minute/second fields of the given local date are not
	 * used. Instead they are set using the org's dayBoundaryOffset.
	 */
	public int getIndex(LocalDate ld) {
		// Compute a date object for the beginning of the organization day
		// on the given local date
		int m_dayBoundaryHour = 0;
		int m_dayBoundaryMinute = 0;
		Calendar vCal = Calendar.getInstance(timeZone);
		vCal.setTime(ld.getTime(timeZone));
		vCal.set(Calendar.HOUR_OF_DAY, m_dayBoundaryHour);
		vCal.set(Calendar.MINUTE, m_dayBoundaryMinute);
		vCal.set(Calendar.SECOND, 0);
		vCal.set(Calendar.MILLISECOND, 0);
		// Get the index for the day
		return getIndex(vCal.getTime());
	}

	/**
	 * Get the day index for the given date. The org day start, for the given
	 * date, is used to compute the index. Index computation takes into account
	 * that the interval between the given date 'dt' and the buckets's start
	 * date (as org day) could span year boundaries.
	 * 
	 * @param dt
	 *            the date of interest
	 * @return the index of the bucket in the bucket array corresponding to the
	 *         given date
	 */
	public int getIndex(Date givenDate) {
		Date givenDateAsOrgDayStart = TOCalcUtil.getDateForOrgDayStart(timeZone, dayBoundaryOffsetInMins, givenDate);

		Calendar givenDateAsOrgDayStartCal = Calendar.getInstance(timeZone);
		givenDateAsOrgDayStartCal.setTime(givenDateAsOrgDayStart);

		// Make sure that 'givenDate' is after this DailyBucketHours range's
		// start date.
		if (givenDateAsOrgDayStartCal.before(dailyHrsBucketsRangeStartAsOrgDayStartCal)) {
			throw new IllegalArgumentException("Cannot compute index for date " + givenDateAsOrgDayStart + " since it is before "
					+ dailyHrsBucketsRangeStartAsOrgDayStartCal.getTime());
		}

		// Ensure that the givenDate is before this DailyBucketHours range's end
		// date.
		if (givenDateAsOrgDayStart.after(dailyHrsBucketsRangeEndAsOrgDayEnd)) {
			throw new IllegalArgumentException("Cannot compute index for date " + givenDateAsOrgDayStart + " since it is after "
					+ dailyHrsBucketsRangeEndAsOrgDayEnd);
		}

		// Calculate the day difference between the computed orgDayStart
		// containing the given dt
		// and the org day for the start of this bucket's time range.
		int diff = 0;
		// DailyBucketHour range's start date (adjusted to the beginning
		// of the org day) in the same year as the givenDate.
		// DBH == DailyBucketHours;
		Calendar DBHRangeStartOrgDayCal = (Calendar) dailyHrsBucketsRangeStartAsOrgDayStartCal.clone();
		// get year for the orgDayStart that contains dt.
		int yearForGivenDateOrgDayStart = givenDateAsOrgDayStartCal.get(Calendar.YEAR);
		// if firstOrgDay points to a prior calendar year
		while (DBHRangeStartOrgDayCal.get(Calendar.YEAR) < yearForGivenDateOrgDayStart) {
			int dayInYear = DBHRangeStartOrgDayCal.get(Calendar.DAY_OF_YEAR);
			int lastDayOfYear = DBHRangeStartOrgDayCal.getActualMaximum(Calendar.DAY_OF_YEAR);
			// number of days to get the
													// rest of year.
			diff += lastDayOfYear - dayInYear + 1; 
			// set firstOrgDay to point the beginning of the next year
			DBHRangeStartOrgDayCal.roll(Calendar.YEAR, true);
			DBHRangeStartOrgDayCal.set(Calendar.MONTH, Calendar.JANUARY);
			DBHRangeStartOrgDayCal.set(Calendar.DAY_OF_MONTH, 1);
		}
		// Now the calendars point to the same year
		diff += (givenDateAsOrgDayStartCal.get(Calendar.DAY_OF_YEAR) - DBHRangeStartOrgDayCal.get(Calendar.DAY_OF_YEAR));

		// The difference is the index
		return diff;
	}
