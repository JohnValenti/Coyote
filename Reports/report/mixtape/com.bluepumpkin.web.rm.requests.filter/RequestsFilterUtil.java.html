package com.bluepumpkin.web.rm.requests.filter;

import java.util.ArrayList;
import java.util.Iterator;
import java.util.List;

import com.bluepumpkin.common.datatypes.IDStringPair;
import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.localization.Localizer;
import com.witness.web.uif.pagecomponent.list.OptionData;
import com.bluepumpkin.web.rm.requests.RequestUtil;

/**

public class RequestsFilterUtil {
  
  /** Creates a new instance of RequestsFilterUtil */
  private RequestsFilterUtil() {}
  
  /**
   * Create Filter Options from specified Filter Names
   */
  public static ArrayList createFilterOptions(IDStringPair[] fNames, String selectedItem) {
		ArrayList result = new ArrayList();
   
    if (fNames!=null && fNames.length>0) {
      for(int i=0; i<fNames.length; i++) {
        IDStringPair fName = fNames[i];
        String name = fName.getString();
        String id = fName.getID().toString();
        OptionData data = new OptionData(name, id);
        if (id.equalsIgnoreCase(selectedItem)) {
          data.setSelected(true);
        }
        result.add(data);
      }
    }
    return result;
  }
	
	/**
	 * Create Requests Status Options
	 */
	public static List createStatusOptions(List reqStatus, Localizer localizer) {
		ArrayList result = new ArrayList();
   
		if (reqStatus!=null && !reqStatus.isEmpty()) {
			for(Iterator it=reqStatus.iterator(); it.hasNext(); ) {
				String status = (String)it.next();
				String[] imgAndAltText = RequestUtil.getImgAndL10nDisplayForStatus(status, localizer);
				String display = imgAndAltText==null ? status : imgAndAltText[1];
				result.add( new StringsPair(status, display) );
			}
		}
		
    return result;
	}
}
