package com.verint.web.fs.skills;

import java.rmi.RemoteException;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Comparator;
import java.util.LinkedHashMap;
import java.util.List;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.skill.model.Skill;
import com.bluepumpkin.ejb.bbm.vo.DBSortField;
import com.bluepumpkin.web.bbm.workload.WorkloadModelHandler;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.MultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.InsertableMultiColListPC;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.SortUtil;

public class SkillListPC extends InsertableMultiColListPC {
	private static final long serialVersionUID = 1L;
	
	private Collection<StringsPair> m_mediaStringPairs;
	
	public SkillListPC(RequestContext context, Collection<Skill> skills) {
		super(context);
		setName("availableSkillsTable");
		m_mediaStringPairs = getMediaStringsPair();
		initRows(skills);
		initHeader();
	}

	/**
	 * Return Collection of Media StringsPair (String mediaId, String mediaName)
	 */
	private Collection<StringsPair> getMediaStringsPair() {
		try {
			return WorkloadModelHandler.getMediaWithIDStringsPairList(m_context, m_localizer);
		} catch (RemoteException e) {
			return new ArrayList<StringsPair>();
		} catch (BbmException e) {
			return new ArrayList<StringsPair>();
		}
	}
	
	
	protected void initHeader() {
		TableHeaderPC header = getHeader();
		header.setDefaultSortColumn("skillname");
		header.setPartialSortable(true);
		header.addColumnHeaderData("skillname", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SKILLNAME), false);
		header.addColumnHeaderData("skillmedia", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SKILLMEDIA), false);
		header.addColumnHeaderData("skilldescription", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SKILLDESCRIPTION), false);
		header.addColumnHeaderData("skillorg", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SKILLORGANIZATION), false);
	}

	private void initRows(Collection<Skill> skills) {
		List<MultiColumnNodeData> rows = super.getListModel();

		if (skills != null) {
			for (Skill skill : skills) {
				String detailRowID = getName() + "__detail" + getJSSuffix(skill);
				DefaultMultiColumnNodeData mcnd = new DefaultMultiColumnNodeData(skill.getID());

				// Name
				mcnd.add(str(skill.getName()));
				
				//skill media
				String medianame = "";
				for (StringsPair pair : m_mediaStringPairs) {
					if (pair.getKey().trim().equalsIgnoreCase(skill.getMediaID().toString().trim()))
						medianame = pair.getValue();
				}
				mcnd.add(str(medianame));
				
				//skill description
				mcnd.add(str(skill.getDescription()));
				
				//skill org
				mcnd.add(str(skill.getOrganizationName()));
				
				setRowAttributes(mcnd, skill);
				
				rows.add(mcnd);
				
			}
		}

		SortUtil.sort(rows, 0, getHeader().isAscending(), m_localizer);
	}

	private String getJSSuffix(Skill skill) {
		return skill.getID().toString().replace('-', 'n');
	}
	
	/**
	 * Sub-classes can override this method to set attributes on the MultiColumnNodeData.
	 * @param mcnd MultiColumnNodeData that is populated with the Shift's values. Set the node attributes on this object.
	 * @param shift that is represented by this row. Use this object to determine if attributes should be set on the MultiColumnNodeData
	 */
	protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, Skill skill) {}
	
	/**
	 * Sub-classes can override this method to set attributes on the MultiColumnNodeData.
	 * @param mcnd MultiColumnNodeData that is populated with the Shift's values. Set the node attributes on this object.
	 * @param shift that is represented by this row. Use this object to determine if attributes should be set on the MultiColumnNodeData
	 */
	protected void setDetailRowAttributes(DefaultMultiColumnNodeData mcnd, DefaultMultiColumnNodeData mcndDetail, 
			Skill skill) {}
	
	private String str(Object v) {
		return v == null ? "" : v.toString();
	}
	
	public LinkedHashMap<DBSortField, Comparator> getSortComparators() {
		LinkedHashMap<DBSortField, Comparator> comparators = new LinkedHashMap<DBSortField, Comparator>();
		for (SkillSortField sortField : getDBSortFieldIndices()) {
			switch (sortField) {
				case ID:
					comparators.put(SkillSortField.ID, null);
					break;
				case NAME:
					comparators.put(SkillSortField.NAME, null);
					break;
				case MEDIA:
					comparators.put(SkillSortField.MEDIA, null);
					break;
				case DESCRIPTION:
					comparators.put(SkillSortField.DESCRIPTION, null);
					break;
				case ORGANIZATION:
					comparators.put(SkillSortField.ORGANIZATION, null);
					break;
			}
		}
		
		return comparators;
	}
	
	public ArrayList<SkillSortField> getDBSortFieldIndices() {
		ArrayList<SkillSortField> retVal = new ArrayList<SkillSortField>();
		if (getHeader().getSortColumnIndexes() == null) {
			//Default sort index is by ID (chronological order in which shift event was created)
			retVal.add(SkillSortField.ID);
		} else {
			for (int sortField : getHeader().getSortColumnIndexes()) {
				switch (sortField) {
					case 1: 
						retVal.add(SkillSortField.NAME);
						break;
					case 2: 
						retVal.add(SkillSortField.MEDIA);
						break;
					case 3: 
						retVal.add(SkillSortField.DESCRIPTION);
						break;
					case 4: 
						retVal.add(SkillSortField.ORGANIZATION);
						break;
				}
			}
		}		
		return retVal;
	}
}//
