package com.verint.web.fs.skills;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.bbm.workload.queue.selection.QueueFilterUtil;
import com.bluepumpkin.web.core.pagecomponent.time.TimePeriodSelectorController;
import com.bluepumpkin.web.core.pagecomponent.time.TimePeriodSelectorSPController;
import com.verint.web.fs.workrules.assignment.WorkRulesEmployeeAssignmentUtil;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;

public class SkillsEmployeeAssignmentCampaignPM extends
		SkillsEmployeeAssignmentPM {
	private static final long serialVersionUID = 1L;
	
	public SkillsEmployeeAssignmentCampaignPM(RequestContext context) {
		super(context);
		isByOrg = false;
		try {
			ID spID = QueueFilterUtil.retrieveSchedulingPeriodID(context);
			if (spID != null) {
				selectedSP = CampaignModelHandler.getSchedulingPeriodByID(
					m_context, spID);
			}
			if (selectedSP != null) {
				selectedCampaign = CampaignModelHandler.getCampaign(context, selectedSP.getCampaignID());
			}
		} catch(Exception e) {
			handleUnableToLoadDataError(log, e);
		}
	}
	public static SkillsEmployeeAssignmentCampaignPM getInstance(RequestContext context) {
		return (SkillsEmployeeAssignmentCampaignPM) PageModel.getInstance(context, SkillsEmployeeAssignmentCampaignPM.class);
	}
	@Override
	protected TimePeriodSelectorController getTimePeriodSelectorController() {
		if (selectedCampaign != null && selectedSP != null) {
			return new TimePeriodSelectorSPController(selectedCampaign, selectedSP);
		}
		return null;
	}

	
	protected void loadData() throws Exception {
		if (WorkRulesEmployeeAssignmentUtil.checkIfEmployeeSelectionIsValid(getSelectedID(), getRequestContext()) == false) {
			setSelectedID(null);
			clearCachedSelectedID();
		}
		super.loadData();
	}
	
}
