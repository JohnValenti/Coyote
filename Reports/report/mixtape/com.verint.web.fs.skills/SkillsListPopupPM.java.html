package com.verint.web.fs.skills;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.HashSet;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.skill.model.Skill;
import com.bluepumpkin.ejb.bbm.workresource.model.Employee;
import com.bluepumpkin.web.bbm.employee.EmployeeModelHandler;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.WorkRulesSelectionPopupPM;
import com.verint.web.fs.workrules.assignment.EmployeeAssignmentInlineEditComponentRH;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarTextButton;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.CacheUtil;

public class SkillsListPopupPM extends WorkRulesSelectionPopupPM {
	private static final long serialVersionUID = 1L;

	public static final String FORM_ACTION = "skill_selection_list_popup";
	public static final String POPUP_ID = "POPUP_SKILL";
	public static final String POPUP_ARGS = "schedulingPeriodID,viewStartDate,viewEndDate,selectedIDs";
	public static final String POPUP_ARG_SELECTED_IDS = "selectedIDs";
	public static final String POPUP_ARG_SKILL_ID = "skillID";
	public static final String POPUP_ARG_IS_INFO_POPUP = "isInfoPopup";
	
	private Collection<Skill> skills = new ArrayList<Skill>();
	private HashSet<ID> assignedSkillIDs = new HashSet<ID>();
	private Date viewStartDate;
	private Date viewEndDate;
	private ID schedulingPeriodID;
	private Collection<Employee> selectedEmployees;
	private Collection<ID> employeeIDs;
	
	public SkillsListPopupPM(RequestContext context) {
		super(context);
	}
	
	public static SkillsListPopupPM getInstance(RequestContext context) {
		return (SkillsListPopupPM)getInstance(context, SkillsListPopupPM.class);
	}
	
	protected void initToolbar() {
		ToolbarTextButton addButton = m_toolbar.createCloseWindowButton(
				"addSkills", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
				FsWebBundleKey.FS_ADD_SKILLS), true);	
		
		StringBuffer addButtonJS = new StringBuffer();
		addButtonJS.append(m_toolbar.getName())
		.append(".disableButton('addSkills'); ")
		.append("RowInsertionUtil.addNewDynamicRow('skill', eval('window.opener.skillTable'), ")
		.append("'" + EmployeeAssignmentInlineEditComponentRH.URL + "', ")
		.append("'" + EmployeeAssignmentInlineEditComponentRH.PARAM_TYPE + "=" +
				EmployeeAssignmentInlineEditComponentRH.SKILL_TYPE + "&")
		.append(EmployeeAssignmentInlineEditComponentRH.PARAM_EMPLOYEE_IDS + "=" +
				StringUtil.join(employeeIDs, StringUtil.DELIM) + "&");
		if (schedulingPeriodID != null) {		//if in campaign mode, we use the viewStartDate/endDates always
			addButtonJS.append(EmployeeAssignmentInlineEditComponentRH.PARAM_START_DATE + "=" + viewStartDate.getTime() + "&")
			.append(EmployeeAssignmentInlineEditComponentRH.PARAM_END_DATE + "=" + viewEndDate.getTime())
			.append("&" + EmployeeAssignmentInlineEditComponentRH.PARAM_SP_ID + "=" + schedulingPeriodID.toString());
		} else if (selectedEmployees.size() == 1) {	//start date = employee start date. end date = blank	
			addButtonJS.append(EmployeeAssignmentInlineEditComponentRH.PARAM_START_DATE + "=" +
					selectedEmployees.iterator().next().getStartTime().getTime());		
		} else { //multiple employees selected.  start date = employee start date if all selected employees share the same start date, * otherwise.  end date = blank
			Date startDate = selectedEmployees.iterator().next().getStartTime();
			boolean areStartDatesCommon = true;
			for (Employee employee : selectedEmployees) {
				if (employee.getStartTime().equals(startDate) == false) {
					areStartDatesCommon = false;
					break;
				}
			}
			if (areStartDatesCommon) {
				addButtonJS.append(EmployeeAssignmentInlineEditComponentRH.PARAM_START_DATE + "=" + startDate.getTime());
			} else {
				addButtonJS.append(EmployeeAssignmentInlineEditComponentRH.PARAM_START_DATE + "=" + MULTI_VALUE_DATA);
			}
		}
		
		//add the view start/end dates
		addButtonJS.append("&" + EmployeeAssignmentInlineEditComponentRH.PARAM_VIEW_START_DATE + "=" + viewStartDate.getTime());
		addButtonJS.append("&" + EmployeeAssignmentInlineEditComponentRH.PARAM_VIEW_END_DATE + "=" + viewEndDate.getTime());
		addButtonJS.append("');");
		
		addButton.setCustomJS(addButtonJS.toString());
		m_toolbar.addToolbarElement(addButton);
		
		m_toolbar.addCloseWindowTextButton("cancel", m_localizer.i18n(
				CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_CANCEL), true);
	}
	
	protected void initUtilityPane() {
		getUtilityPane().setIsHelpEnabled(false);
	}
	
	protected void initContentTitle() {
		m_contentTitlePC.setPageTitle(getLocalizer().i18n(
			FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.PEOPLE_SKILL));
	}
	
	protected void loadData() throws Exception {
		String employeeIDStr = CacheUtil.getCachedValue(m_context, CacheUtil.CACHE_MODE_SESSION,
				SkillsEmployeeAssignmentPM.EMPLOYEE_ID_LIST_CONTEXT_KEY);
		if (StringUtil.isEmpty(employeeIDStr) == false) employeeIDs = StringUtil.parseIDStringToCollection(employeeIDStr);
		
		if (employeeIDs != null && employeeIDs.size() > 0) {
			selectedEmployees = EmployeeModelHandler.getEmployeesByIDs(getRequestContext(), employeeIDs);
			skills = SkillsAssignmentMH.getEmployeeSkills(m_context, employeeIDs, viewStartDate, viewEndDate);
			//Remove skills that have already been assigned to all the selected employees
			if (skills != null && assignedSkillIDs != null) {
				for (Iterator<Skill> iter = skills.iterator(); iter.hasNext();) {
					if (assignedSkillIDs.contains(iter.next().getID())) {
						iter.remove();
					}
				}
			}
		} else {
			throw new Exception(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_ASSIGNMENT_RULE_POPUP_NO_EMPLOYEES_ERROR));
		}
	}

	protected SkillListPC initList() {
		return new SkillListPC(m_context, skills);
	}
	
	public String getHtmlDisplay() {
		if (m_list != null) {
			m_list.setFirstColumnWidth("30%");
			return m_list.getHtmlDisplay();
		}
		return "";
	}
	
	public void setSelectedIDs(String selectedIDs) {
		//This is a hack in order to prevent a rare bug in which the user selects
		//enough skills so that the sum of the selected skill IDs causes the URL
		//to get truncated (IE supports a max of 2048 characters in the URL).  We will
		//remove the last (possibly truncated) ID in this case, in order to prevent the
		//system from interpreting that as a different ID.  Missing the truncated IDs isn't
		//horrible, since the only drawback is the user will be able to reselect the same skill
		//again which will overwrite the existing selection.
		if (selectedIDs != null && selectedIDs.length() > 1900) {
			selectedIDs = selectedIDs.substring(0, selectedIDs.lastIndexOf(","));
		}
		this.assignedSkillIDs = new HashSet<ID>(StringUtil.parseIDStringToCollection(selectedIDs));
	}
	
	public void setViewStartDate(String startDate) {
		viewStartDate = new Date(Long.parseLong(startDate));
	}
	
	public void setViewEndDate(String endDate) {
		viewEndDate = new Date(Long.parseLong(endDate));
	}
	
	public void setSchedulingPeriodID(ID schedulingPeriodID) {
		this.schedulingPeriodID = schedulingPeriodID;
	}
}
