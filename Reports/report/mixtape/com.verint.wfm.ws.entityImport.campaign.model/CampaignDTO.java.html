package com.verint.wfm.ws.entityImport.campaign.model;

import javax.xml.bind.annotation.XmlType;

@XmlType(name = "CampaignDTO", propOrder = { "externalSourceId", "name", "description", "timeZoneId", "weekStart",
		"dayBoundary", "distributed" })
public class CampaignDTO {
	private static final long serialVersionUID = 1L;

	protected String m_externalSourceId;
	protected String m_name;
	protected String m_description;
	protected String m_timeZoneId;
	protected short m_weekStart;
	protected int m_dayBoundary;
	protected boolean m_isDistributed;

	public CampaignDTO() {
	}

	public String getExternalSourceId() {
		return m_externalSourceId;
	}

	public void setExternalSourceId(String externalSourceId) {
		m_externalSourceId = externalSourceId;
	}

	public String getName() {
		return m_name;
	}

	public void setName(String name) {
		m_name = name;
	}

	public String getDescription() {
		return m_description;
	}

	public void setDescription(String description) {
		m_description = description;
	}

	public String getTimeZoneId() {
		return m_timeZoneId;
	}

	public void setTimeZoneId(String timeZoneId) {
		m_timeZoneId = timeZoneId;
	}

	public short getWeekStart() {
		return m_weekStart;
	}

	public void setWeekStart(short weekStart) {
		m_weekStart = weekStart;
	}

	public int getDayBoundary() {
		return m_dayBoundary;
	}

	public void setDayBoundary(int dayBoundary) {
		m_dayBoundary = dayBoundary;
	}

	public boolean isDistributed() {
		return m_isDistributed;
	}

	public void setDistributed(boolean isDistributed) {
		m_isDistributed = isDistributed;
	}
}
