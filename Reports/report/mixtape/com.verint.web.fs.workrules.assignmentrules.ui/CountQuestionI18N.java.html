package com.verint.web.fs.workrules.assignmentrules.ui;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;
import com.verint.web.fs.workrules.assignmentrules.question.CountQuestion.CountResponseType;
import com.verint.web.fs.workrules.assignmentrules.question.Question;

class CountQuestionI18N extends AbstractQuestionI18N {
	@Override
	public String getResponseKey(Response response) {
		ResponseType responseType = response.getResponseType();
		if (responseType == null)
			return null;
		if (!(responseType instanceof CountResponseType))
			throw new IllegalArgumentException("Expected responseType to be of type "
					+ CountResponseType.class.getCanonicalName() + " instead of "
					+ responseType.getClass().getCanonicalName());
		
		CountResponseType crt = (CountResponseType) responseType;
		switch (crt) {
			case AtLeast:
				return FsWebBundleKey.ASSIGNMENT_RULE_NUM_AT_LEAST;
			case AtMost:
				return FsWebBundleKey.ASSIGNMENT_RULE_NUM_NO_MORE_THAN;
			case Between:
				return FsWebBundleKey.ASSIGNMENT_RULE_NUM_BETWEEN;
			case Exactly:
				return FsWebBundleKey.ASSIGNMENT_RULE_NUM_EXACTLY;
			case FairNumber:
				return FsWebBundleKey.ASSIGNMENT_RULE_NUM_FAIR;
			default:
				throw new IllegalArgumentException("Unknown value of: " + crt);
		}
	}

	@Override
	public String getQuestionText(Localizer localizer, Question question) {
		return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.ASSIGNMENT_RULE_TEXT_COUNT);
	}
}
