package com.verint.web.fs.workrules.assignmentrules.ui;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.question.Question;
import com.verint.web.fs.workrules.assignmentrules.question.WorkUnitQuestion.WorkUnitResponse;

class WorkUnitQuestionI18N extends AbstractQuestionI18N {

	@Override
	protected String getResponseKey(Response response) {
		if (!(response instanceof WorkUnitResponse)) {
			throw new IllegalArgumentException("Expected response type: " + WorkUnitResponse.class.getName()
					+ " Actual Type: " + response.getClass().getName());
		}

		WorkUnitResponse wur = (WorkUnitResponse) response;
		switch (wur.getResponseType()) {
			case Activities:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_ACTIVITY;
			case AnyDay:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_DAY;
			case ConsecutiveDay:
			case ConsecutiveDayOff:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_CONSECUTIVE_DAY;
			case ConsecutiveDayOfWeek:
			case ConsecutiveDayOfWeekOff:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_CONSECUTIVE_DAY_OF_WEEK;
			case ConsecutiveFullWeekend:
			case ConsecutiveFullWeekendOff:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_CONSECUTIVE_FULL_WEEKEND;
			case ConsecutivePartialWeekend:
			case ConsecutivePartialWeekendOff:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_CONSECUTIVE_PARTIAL_WEEKEND;
			case DayOfWeek:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_DAY_OF_WEEK;
	  /* Dollars is no longer used, so the string was removed.
			case Dollars:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_DOLLARS;
			*/
			case FullWeekend:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_FULL_WEEKEND;
			case Hours:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_HOURS;
			case MinShiftEventSpacing:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_MIN_SHIFT_EVENT_SPACING;
			case MinShiftSpacing:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_MIN_SHIFT_SPACING;
			case FullWeekendLabeledAsPartial:
			case PartialWeekend:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_PARTIAL_WEEKEND;
				/* Rotation is covered in a different screen, so it is never an option in the Assignment Rules screen.
			case Rotation:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_ROTATION;
				*/
			case Shift:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_SHIFT;
			case Weekend:
				return FsWebBundleKey.ASSIGNMENT_RULE_WORK_UNIT_WEEKEND;
			default:
				throw new IllegalArgumentException("Unexpected response type: " + wur.getResponseType());
		}
	}

	@Override
	public String getQuestionText(Localizer localizer, Question question) {
		return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.ASSIGNMENT_RULE_TEXT_WORK_UNIT);
	}
}
