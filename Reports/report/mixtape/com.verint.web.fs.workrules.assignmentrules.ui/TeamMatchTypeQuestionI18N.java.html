package com.verint.web.fs.workrules.assignmentrules.ui;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;
import com.verint.web.fs.workrules.assignmentrules.question.Question;
import com.verint.web.fs.workrules.assignmentrules.question.TeamMatchQuestion.TeamMatchResponseType;

class TeamMatchTypeQuestionI18N extends AbstractQuestionI18N {
	@Override
	public String getResponseKey(Response response) {
		ResponseType responseType = response.getResponseType();
		if (!(responseType instanceof TeamMatchResponseType))
			throw new IllegalArgumentException("Expect a response type of type: "
					+ TeamMatchResponseType.class.getName());
		TeamMatchResponseType matchType = (TeamMatchResponseType) responseType;
		switch(matchType) {
			case AllOther:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_SUBJECT_ALL_OTHER;
			case Supervisor:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_SUBJECT_SUPERVISOR;
			case TeamLead:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_SUBJECT_LEAD;
			default: 
				throw new IllegalArgumentException("Unknown response type: " + matchType.name());
		}
	}

	@Override
	public String getQuestionText(Localizer localizer, Question question) {
		return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.ASSIGNMENT_RULE_TEXT_TEAM_MATCH);
	}

}
