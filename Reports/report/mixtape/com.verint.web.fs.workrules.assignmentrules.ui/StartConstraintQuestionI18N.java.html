package com.verint.web.fs.workrules.assignmentrules.ui;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.question.Question;
import com.verint.web.fs.workrules.assignmentrules.question.StartConstraintQuestion.StartConstraintResponseType;

class StartConstraintQuestionI18N extends AbstractQuestionI18N implements QuestionI18N {
	@Override
	protected String getResponseKey(Response response) {
		switch ((StartConstraintResponseType)response.getResponseType()) {
		case SameTime:
			return FsWebBundleKey.ASSIGNMENT_RULE_SAME_TIME;
			default:
				throw new IllegalArgumentException("Unexpected response type: " + response.getResponseType());
		}
	}

	@Override
	public String getQuestionText(Localizer localizer, Question question) {
		return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.ASSIGNMENT_RULE_TEXT_START_CONSTRAINT);
	}

}
