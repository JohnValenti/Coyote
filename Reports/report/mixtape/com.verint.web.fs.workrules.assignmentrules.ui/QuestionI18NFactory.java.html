package com.verint.web.fs.workrules.assignmentrules.ui;

import com.verint.web.fs.workrules.assignmentrules.question.QuestionType;

public class QuestionI18NFactory {
	public static QuestionI18N getI18NInfo(QuestionType type) {
		switch(type) {
			case RuleType:
				return new RuleTypeQuestionI18N();
			case Count:
				return new CountQuestionI18N();
			case CalendarPeriod:
				return new PeriodQuestionI18N();
			case TeamConstraintType:
				return new TeamConstraintTypeQuestionI18N();
			case TeamMatchType:
				return new TeamMatchTypeQuestionI18N();
			case TimeOfDay:
			case TimeOfDayAndBetween:
				return new TimePeriodOfDayQuestionI18N();
			case WorkUnit:
				return new WorkUnitQuestionI18N();
			case AlignmentDate:
				return new AlignmentDateQuestionI18N();
			case EndDate:
				return new EndDateQuestionI18N();
			case StartConstraint:
				return new StartConstraintQuestionI18N();
			case StartUnit:
				return new StartUnitQuestionI18N();
			case Tolerance:
			case TeamTolerance:
				return new ToleranceQuestionI18N();
			default:
				throw new IllegalArgumentException("Unknown question type: " + type.name());
		}
	}
}
