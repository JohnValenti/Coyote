package com.bluepumpkin.web.rm.requests.shiftswap;

import java.util.ArrayList;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.requests.swap.request.model.ShiftSwapRequest;
import com.bluepumpkin.web.rm.IHavePageMessage;
import com.witness.web.uif.base.Message;

public class ShiftSwapRequestParameters implements IHavePageMessage {

	private final List<Message> messages = new ArrayList<Message>();

	private ID agentID;
	private Date expirationDate;
	private boolean negotiable;
	private boolean isSwapPartial;
	private boolean isPickupPartial;
	private ID postingID;
	private Date swapStartDate;
	private Date swapEndDate;
	private Date requestDate1;

	private Date pickupStartDate;
	private Date pickupEndDate;
	private String swapType;

	private Date requestDate2;

	private boolean submitToMgr;
	private String negotiationComment;

	private String shiftType;

	private ID savedRequestID;

	private ShiftSwapRequestFormPopupPM model;

	public ShiftSwapRequestParameters() { // NOSONAR

	}

	public ShiftSwapRequestParameters(ShiftSwapRequestFormPopupPM model) {
		this.model = model;
	}

	public ID getAgentID() {
		return agentID;
	}

	public void setAgentID(ID agentID) {
		this.agentID = agentID;
	}

	public Date getExpirationDate() {
		return expirationDate;
	}

	public void setExpirationDate(Date expirationDate) {
		this.expirationDate = expirationDate;
	}

	public boolean isNegotiable() {
		return negotiable;
	}

	public void setNegotiable(boolean isNegotiable) {
		this.negotiable = isNegotiable;
	}

	public boolean getIsSwapPartial() {
		return isSwapPartial;
	}

	public void setIsSwapPartial(boolean isSwapPartial) {
		this.isSwapPartial = isSwapPartial;
	}

	public boolean getIsPickupPartial() {
		return isPickupPartial;
	}

	public void setIsPickupPartial(boolean isPickupPartial) {
		this.isPickupPartial = isPickupPartial;
	}

	public ID getPostingID() {
		return postingID;
	}

	public void setPostingID(ID postingID) {
		this.postingID = postingID;
	}

	public Date getSwapStartDate() {
		return swapStartDate;
	}

	public void setSwapStartDate(Date swapStartDate) {
		this.swapStartDate = swapStartDate;
	}

	public Date getSwapEndDate() {
		return swapEndDate;
	}

	public void setSwapEndDate(Date swapEndDate) {
		this.swapEndDate = swapEndDate;
	}

	public Date getRequestDate1() {
		return requestDate1;
	}

	/**
	 * My Entire Day
	 */
	public void setRequestDate1(Date requestDate1) {
		this.requestDate1 = requestDate1;
	}

	public Date getPickupStartDate() {
		return pickupStartDate;
	}

	public void setPickupStartDate(Date pickupStartDate) {
		this.pickupStartDate = pickupStartDate;
	}

	public Date getPickupEndDate() {
		return pickupEndDate;
	}

	public void setPickupEndDate(Date pickupEndDate) {
		this.pickupEndDate = pickupEndDate;
	}

	public String getSwapType() {
		return swapType;
	}

	public void setSwapType(String swapType) {
		this.swapType = swapType;
	}

	/**
	 * Agent2's Entire Day
	 */
	public Date getRequestDate2() {
		return requestDate2;
	}

	public void setRequestDate2(Date requestDate2) {
		this.requestDate2 = requestDate2;
	}

	public boolean isSubmitToMgr() {
		return submitToMgr;
	}

	public void setSubmitToMgr(boolean isSubmitToMgr) {
		this.submitToMgr = isSubmitToMgr;
	}

	public String getNegotiationComment() {
		return negotiationComment;
	}

	public void setNegotiationComment(String negotiationComment) {
		this.negotiationComment = negotiationComment;
	}

	public String getShiftType() {
		return shiftType;
	}

	public void setShiftType(String shiftType) {
		this.shiftType = shiftType;
	}

	public List<Message> getMessages() {
		return messages;
	}

	/**
	 * Return true if it is a Two Way Swap
	 */
	public boolean isTwoWaySwap() {
		return ShiftSwapRequest.SHIFTSWAP_TYPE_TWOWAY.equals(getSwapType());
	}

	public ID getSavedRequestID() {
		return savedRequestID;
	}

	public void setSavedRequestID(ID requestID) {
		savedRequestID = requestID;
	}

	// IHavePageMessage implementation

	@Override
	public Message addPageMessage(int messageType, String messageID, String msgIDrBundleBaseName) {
		return addPageMessage(messageType, messageID, msgIDrBundleBaseName, null);
	}

	@Override
	public Message addPageMessage(int messageType, String messageID, String msgIDrBundleBaseName, String[] messageArgs) {
		// add directly to the form because we want to preserve message order. There was a case where the logic added an error message
		// and then there was an exception. The UI showed it in the right order where the service only returned the exception.
		// this preserves the form's original behavior in case other cases exist.

		if (model != null) {
			return model.addPageMessage(messageType, messageID, msgIDrBundleBaseName, messageArgs);
		}
		Message message = new Message(messageType, messageID, msgIDrBundleBaseName, messageArgs);
		messages.add(message);
		return message;
	}
}
