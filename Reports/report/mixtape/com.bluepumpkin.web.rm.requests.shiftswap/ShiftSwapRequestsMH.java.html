    RmManagerFactory rmf = RmManagerFactory.getInstance();

	/**
	 * Return Employee Names One can Swap Shift with.
	 */
	public static Collection<EmployeeName> getEmpNamesOneCanSwapShiftWith(RequestContext context, ID empID)
			throws RemoteException, BbmException {
		ShiftSwapRequestManager ssrm = getSSManager();

		@SuppressWarnings("unchecked")
		Collection<Employee> empList = ssrm.getEmployeesOneCanSwapWith(empID);
		List<ID> empIDs = new ArrayList<ID>(empList.size());
		for (Iterator<Employee> it = empList.iterator(); it.hasNext();) {
			Employee emp = it.next();
			empIDs.add(emp.getID());
		}

		Map<ID, EmployeeName> eNameMap = EmployeeModelHandler.getEmployeeNamesByIDs(context, empIDs);
		return eNameMap.values();
	}
	
	/**
	 * Return Map of EmployeeID and EmployeeName for ShiftSwapRequest
	 */
	public static Map getEmpNameMapForRequest(RequestContext context, ShiftSwapRequest request)
	throws RemoteException, BbmException {
    List empIDList = request.getEmployees();		
		return EmployeeModelHandler.getEmployeeNamesByIDs(context, empIDList);
	}	
	
	/**
	 * Return Collection of Unique Employee IDs for specified collection of
	 * Shift Swap Requests
	 */
	public static Collection getUniqueEmpIDs(Collection requests) {
		HashSet empIDSet = new HashSet(requests.size());
				
		for (Iterator it = requests.iterator(); it.hasNext(); ) {
			ShiftSwapRequest ssr = (ShiftSwapRequest)it.next();
			empIDSet.addAll(ssr.getEmployees());
		}
		
		return empIDSet;
	}

	public static Pair<Boolean, String> requestWithdrawOfApprovedRequest(ID requestID, String requestType,
			String requestVersion, String requestStatus, String comment) throws Exception {
		Pair<Boolean, String> pair = new Pair<Boolean, String>(new Boolean("true"), ""); //return true by default. false is explicitly set in one scenario only
		ShiftSwapRequest request = getSSRequest(requestID);
		//if request is approved then invoke the approved TO Withdraw process
		if (request.getRequestStatus().equals(RequestAuditTrail.STATUS_APPROVED)) {
			pair = getSSManager().isEligibleForRequestWithdrawOfApprovedRequest(request);
			if (pair.getFirst().booleanValue()) {
				if (RequestAuditTrail.STATUS_WITHDRAWN.equals(requestStatus)){
					getSSManager().requestWithdrawOfApprovedRequest(request, comment);
				} else if (RequestAuditTrail.STATUS_WITHDRAW_NEGOTIATION.equals(requestStatus)) {
					getSSManager().requestWithdrawOfApprovedRequestNegotiation(request, comment);
				}
			}
		} else {
			RequestsMH.changeRequestStatus(requestID, requestType, requestVersion, requestStatus, comment);
		}
		return pair;
	}

	public static Pair<Boolean, String> acceptWithdrawOfApprovedRequest(RequestContext context, ID requestID, String comment) throws Exception {
		ShiftSwapRequestManager ssrm = getSSManager();
		ShiftSwapRequest request = getSSRequest(requestID);
		Pair<Boolean, String> pair = ssrm.isEligibleForRequestWithdrawOfApprovedRequest(request);
		if (pair.getFirst().booleanValue()) {
			ssrm.acceptWithdrawOfApprovedRequest(request, comment);
		}
		return pair;
	}
	
	public static void rejectWithdrawOfApprovedRequest(RequestContext context, ID requestID, String comment) throws Exception {
		ShiftSwapRequestManager ssrm = getSSManager();
		ShiftSwapRequest request = getSSRequest(requestID);
		getSSManager().rejectWithdrawOfApprovedRequest(request, comment);
	}

	}
