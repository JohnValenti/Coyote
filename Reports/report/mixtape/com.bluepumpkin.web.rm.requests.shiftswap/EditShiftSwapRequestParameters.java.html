package com.bluepumpkin.web.rm.requests.shiftswap;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

public class EditShiftSwapRequestParameters {

	// fields for editing
	private ID requestID;
	private Date expirationDate;
	private boolean isNegotiable;
	private String negotiationComment;
	private boolean isSubmitToMgr;
	private String comment;

	// fields for processing
	private boolean withdrawConfirmed;
	private boolean hasError; // means unauthorized and message will be on the context

	public EditShiftSwapRequestParameters() { // NOSONAR
	}

	public ID getRequestID() {
		return requestID;
	}

	public void setRequestID(ID requestID) {
		this.requestID = requestID;
	}

	public Date getExpirationDate() {
		return expirationDate;
	}

	public void setExpirationDate(Date expirationDate) {
		this.expirationDate = expirationDate;
	}

	public boolean isNegotiable() {
		return isNegotiable;
	}

	public void setNegotiable(boolean isNegotiable) {
		this.isNegotiable = isNegotiable;
	}

	public String getNegotiationComment() {
		return negotiationComment;
	}

	public void setNegotiationComment(String negotiationComment) {
		this.negotiationComment = negotiationComment;
	}

	public boolean isSubmitToMgr() {
		return isSubmitToMgr;
	}

	public void setSubmitToMgr(boolean isSubmitToMgr) {
		this.isSubmitToMgr = isSubmitToMgr;
	}

	public String getComment() {
		return comment;
	}

	public void setComment(String comment) {
		this.comment = comment;
	}

	public boolean isWithdrawConfirmed() {
		return withdrawConfirmed;
	}

	public void setWithdrawConfirmed(boolean withdrawConfirmed) {
		this.withdrawConfirmed = withdrawConfirmed;
	}

	public boolean isHasError() {
		return hasError;
	}

	public void setHasError(boolean hasError) {
		this.hasError = hasError;
	}

}
