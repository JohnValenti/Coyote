package com.bluepumpkin.web.rm.requests.shiftswap;

import java.util.*;

import com.bluepumpkin.ejb.rm.requests.swap.request.model.*;
import com.bluepumpkin.ejb.rm.requests.common.model.*;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;

import com.witness.web.uif.pagecomponent.*;
import com.witness.web.uif.base.*;
import com.witness.web.uif.l10n.*;
import com.witness.web.uif.pagemodel.*;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;

import com.bluepumpkin.web.rm.keys.*;
import com.bluepumpkin.web.rm.l10n.*;

/**
 * Title:        MyShiftSwapRequestsPM
 * Description:  Page Model for My Shift Swap Requests
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, inc
 * @author       Zhibo, David Su
 * @version      1.0
 */
public class MyShiftSwapRequestsPM extends MyRequestsPM {
	private Map m_empNameMap = Collections.EMPTY_MAP;
	private Map m_empOrgMap = Collections.EMPTY_MAP;

	/**
	 * Constructs a schedule view page model shell.
	 */
	public MyShiftSwapRequestsPM(RequestContext context)	{
		super(context, Request.REQUESTTYPE_SHIFTSWAP);		
	}

	/**
	 * Add Page Not Enabled Message
	 */
	protected void addPageNotEnabledMsg() {
		SwapUtil.addSSNotEnabledMsg(this);
	}

	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, MyShiftSwapRequestsPM.class);
	}

	/**
	 * Populate Model
	 */
	protected void loadData() throws Exception {
		if (isAuthToViewSSRqts()) {
			try {
				m_orgSetting = ShiftSwapRequestsMH.getOrgSettingForEmployee(m_context, m_empID);
				m_isPageEnabled = m_orgSetting.getEnableShiftSwap();

				if (m_isPageEnabled) {
					m_requestList = ShiftSwapRequestsMH.getRequestsByEmployee(m_empID, m_isIncludeExpired);

					Collection empIDList = ShiftSwapRequestsMH.getUniqueEmpIDs(m_requestList);
					m_empNameMap = ShiftSwapRequestsMH.getEmpNameMap(m_context, empIDList);
					m_empOrgMap = ShiftSwapRequestsMH.getEmpOrgMap(m_context, empIDList);
				}
			} catch (RmHardValidationException ex) {
				Message msg = handleUnableToLoadDataError(log, ex);
				msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
			}
		}
	}

	/**
	 * Initialize List Header
	 */
	protected void initHeader() {
		TableHeaderPC header = m_list.getHeader();
		header.setPartialSortable(true);

		addHeaderData(header, m_common_bundle, UIFWebBundleKey.NAME, true);
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.STATUS, true);
		addHeaderData(header, m_bundle, RmWebBundleKey.SHIFT_DATE, false);
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.START, false);
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.END, false);
		addHeaderData(header, m_bundle, RmWebBundleKey.POST_EXPIRES, false);
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.CONFIRM, false);
		addHeaderData(header, m_bundle, RmWebBundleKey.SWAPTYPE, true);
		addCommonHeaderData();
	}

	/**
	 * Create Row Data From Request
	 */
	protected MultiColumnNodeData createRowData(RequestAggregate request) {
		ShiftSwapRequest ssRequest = (ShiftSwapRequest)request;

		String rowID = request.getID().toString();
		DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData(rowID);

		String agentNames = SwapUtil.createAgentNames(m_user, ssRequest,
				m_empNameMap, m_empOrgMap, m_localizer);
		row_data.add(agentNames);

		String status = createStatusIcon(request);
		row_data.add(status);

		SwapUtil.addShiftDates(ssRequest, m_common_bundle, m_localizer, m_view_tz, row_data);
		SwapUtil.addPostExpirationDate(ssRequest, m_localizer, m_view_tz, row_data);
		SwapUtil.addConfirmData(ssRequest, m_localizer, row_data);
		SwapUtil.addSwapTypeImage(ssRequest.getSwapType(), m_bundle, m_localizer, row_data);

		addCommonListData(request, row_data);
		return row_data;
	}

	/**
	 * Render Request Comments
	 */
	protected String renderComments(RequestAggregate request) {
		String status = request.getRequestStatus();
		if (RequestAuditTrail.STATUS_NEGOTIATION.equals(status)) {
			m_messagePC.clear();
			String nComment = ((ShiftSwapRequest)request).getNegotiationComment();
			m_messagePC.addMessage(new Message(Message.INFO_TYPE, nComment));
			return m_messagePC.getHtmlDisplay();
		} else {
			return super.renderComments(request);
		}
	}

	/**
	 * Add Create Request Button
	 */
	protected void addCreateRequestButton() {
		boolean isBtnEnabled = m_isPageEnabled && isAuthToEditSSRqts();
		String newRequestLabel = i18n(m_bundle, RmWebBundleKey.CREATE_NEW_REQUEST);
		m_toolbar.addPopupWindowTextButton(RmPageAction.POPUP_SHIFTSWAP_REQUEST_FORM_ACTION,
				newRequestLabel, isBtnEnabled);
	}
}
