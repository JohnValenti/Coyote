package com.verint.wfm.ws.entityImport.activity.model;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;

import javax.xml.bind.annotation.XmlTransient;
import javax.xml.bind.annotation.XmlType;

import org.apache.commons.collections.CollectionUtils;

@XmlType(name = "ActivityDTO", propOrder = { "externalSourceId", "name", "description", "organizationName",
		"activityType", "paid", "adherenceTolerance", "guiColor", "timeoff", "usedInShiftEvent", "usedInCalendarEvent",
		"usedInShift", "mediaIds", "additionalQueueExternalSourceIds" })
public class ActivityDTO {
	private static final long serialVersionUID = 1L;

	protected String m_externalSourceId;
	protected String m_name;
	protected String m_description;
	protected String m_organizationName;
	protected String m_activityType;
	protected boolean m_paid;
	protected int m_adherenceTolerance;
	protected String m_guiColor;
	protected boolean m_isTimeoff;
	protected boolean m_isUsedInShiftEvent;
	protected boolean m_isUsedInCalendarEvent;
	protected boolean m_isUsedInShift;
	protected List<String> m_mediaIds;
	protected Collection<String> m_additionalQueueExternalSourceIds = new ArrayList<String>();

	public ActivityDTO() {
		super();
	}

	public String getExternalSourceId() {
		return m_externalSourceId;
	}

	public void setExternalSourceId(String externalSourceId) {
		m_externalSourceId = externalSourceId;
	}
	
	public String getName() {
		return m_name;
	}

	public void setName(String name) {
		m_name = name;
	}

	public String getDescription() {
		return m_description;
	}

	public void setDescription(String description) {
		m_description = description;
	}

	public String getOrganizationName() {
		return m_organizationName;
	}

	public void setOrganizationName(String organizationName) {
		m_organizationName = organizationName;
	}

	public String getActivityType() {
		return m_activityType;
	}

	public void setActivityType(String activityType) {
		m_activityType = activityType;
	}

	public boolean isPaid() {
		return m_paid;
	}

	public void setPaid(boolean paid) {
		m_paid = paid;
	}

	public int getAdherenceTolerance() {
		return m_adherenceTolerance;
	}

	public void setAdherenceTolerance(int adherenceTolerance) {
		m_adherenceTolerance = adherenceTolerance;
	}

	public boolean isTimeoff() {
		return m_isTimeoff;
	}

	public void setTimeoff(boolean isTimeoff) {
		m_isTimeoff = isTimeoff;
	}

	public String getGuiColor() {
		return m_guiColor;
	}

	public void setGuiColor(String guiColor) {
		m_guiColor = guiColor;
	}

	public boolean isUsedInShiftEvent() {
		return m_isUsedInShiftEvent;
	}

	public void setUsedInShiftEvent(boolean isUsedInShiftEvent) {
		m_isUsedInShiftEvent = isUsedInShiftEvent;
	}

	public boolean isUsedInCalendarEvent() {
		return m_isUsedInCalendarEvent;
	}

	public void setUsedInCalendarEvent(boolean isUsedInCalendarEvent) {
		m_isUsedInCalendarEvent = isUsedInCalendarEvent;
	}

	public boolean isUsedInShift() {
		return m_isUsedInShift;
	}

	public void setUsedInShift(boolean isUsedInShift) {
		m_isUsedInShift = isUsedInShift;
	}

	public List<String> getMediaIds() {
		return m_mediaIds;
	}

	public void setMediaIds(List<String> mediaIds) {
		m_mediaIds = mediaIds;
	}

	public Collection<String> getAdditionalQueueExternalSourceIds() {
		return m_additionalQueueExternalSourceIds;
	}

	public void setAdditionalQueueExternalSourceIds(Collection<String> additionalQueueExternalSourceIds) {
		m_additionalQueueExternalSourceIds = additionalQueueExternalSourceIds;
	}

	@XmlTransient
	public boolean isQueueHopping() {
		// There is queue hopping whenever a non-empty list of queues is provided
		return !CollectionUtils.isEmpty(m_additionalQueueExternalSourceIds);
	}
}
