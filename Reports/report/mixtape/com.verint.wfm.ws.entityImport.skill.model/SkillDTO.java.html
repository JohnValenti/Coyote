package com.verint.wfm.ws.entityImport.skill.model;

import javax.xml.bind.annotation.XmlType;

@XmlType(name = "SkillDTO", propOrder = { "externalSourceId", "name", "description", "mediaTypeId",
		"organizationName" })
public class SkillDTO {
	private static final long serialVersionUID = 1L;

	protected String m_externalSourceId;
	protected String m_name;
	protected String m_description;
	protected String m_mediaTypeId;
	protected String m_organizationName;

	public SkillDTO() {
		super();
	}

	public String getExternalSourceId() {
		return m_externalSourceId;
	}

	public void setExternalSourceId(String externalSourceId) {
		m_externalSourceId = externalSourceId;
	}

	public String getName() {
		return m_name;
	}

	public void setName(String name) {
		m_name = name;
	}

	public String getDescription() {
		return m_description;
	}

	public void setDescription(String description) {
		m_description = description;
	}

	public String getOrganizationName() {
		return m_organizationName;
	}

	public void setOrganizationName(String organizationName) {
		m_organizationName = organizationName;
	}

	public String getMediaTypeId() {
		return m_mediaTypeId;
	}

	public void setMediaTypeId(String mediaTypeId) {
		m_mediaTypeId = mediaTypeId;
	}

	public String toString() {
		StringBuffer buf = new StringBuffer();
		buf.append("[");
		buf.append(" Name=");
		buf.append(m_name);
		buf.append(" Description=");
		buf.append(m_description);
		buf.append(" MediaID=");
		buf.append(m_mediaTypeId);
		buf.append("]");

		return buf.toString();
	}
}
