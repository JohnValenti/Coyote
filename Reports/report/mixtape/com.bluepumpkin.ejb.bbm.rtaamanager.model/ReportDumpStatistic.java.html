package com.bluepumpkin.ejb.bbm.rtaamanager.model;

import java.util.HashMap;
import java.io.Serializable;

public class ReportDumpStatistic implements Serializable {

	private static final long serialVersionUID = 1L;

	private static class ThreadLocalMap extends ThreadLocal {
		public Object initialValue() {
			return new HashMap();
		}

		public HashMap getMap() {
			return (HashMap) super.get();
		}
	}

	private ThreadLocalMap map = new ThreadLocalMap();

	public ReportDumpStatistic() {
		clear();
	}

	public void clear() {
		Long zero = new Long(0);
		HashMap stat = map.getMap();
		stat.clear();

		stat.put("m_lTimeSpentPerCatInitialize", zero);
		stat.put("m_lTimeSpentPerCatChangedIntervals", zero);
		stat.put("m_lTimeSpentPerCatGetRTAAData", zero);
		stat.put("m_lTimeSpentPerCatTimeoffCalculation", zero);
		stat.put("m_lTimeSpentPerCatUpdateRTAAData", zero);
		stat.put("m_lTimeSpentPerCatUpdateRTAADataStitching", zero);
		stat.put("m_lTimeSpentPerCatUpdateEmployeeFilters", zero);
		stat.put("m_lTimeSpentPerCatUpdateOrganizationDays", zero);
		stat.put("m_lTotalPublishedPlannedEventsDumped", zero);
		stat.put("m_lTotalUnpublishedPlannedEventsDumped", zero);
		stat.put("m_lTotalActualEventsDumped", zero);
		stat.put("m_lTotalAdherenceExceptionsDumped", zero);
	}

	public void add(String key, long value) {
		HashMap stat = map.getMap();
		if (stat == null) {
			stat = new HashMap();
		}

		long cur = 0;
		if (stat.get(key) != null) {
			cur = ((Long) (stat.get(key))).longValue();
		}
		stat.put(key, new Long(cur + value));
	}

	public long get(String key) {
		HashMap stat = map.getMap();
		if (stat.get(key) == null) {
			return new Long(0);
		}

		return ((Long) stat.get(key)).longValue();
	}

}
