package com.bluepumpkin.ejb.bbm.rtaamanager.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class AdherenceException extends ValueObjectBase implements Comparable {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new AdherenceExceptionFieldInfo();
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}
	
	public AdherenceException() {
		super();
	}
	
	public AdherenceException(ID id, ID employeeID, ID activityID, Date startTime, Date endTime, boolean isApproved, String approvedBy, ID plannedEventTimeLineID) {
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_ID, id);
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_EMPLOYEEID, employeeID);
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_PLANNEDACTIVITYID, activityID);
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_STARTTIME, startTime);
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_ENDTIME, endTime);
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_ISAPPROVED, isApproved);
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_APPROVEDBY, approvedBy);
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_PLANNEDEVENTTIMELINEID, plannedEventTimeLineID);
	}
	
	public Date getStartTime() {
		return getFieldValueDate(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_STARTTIME);
	}
	
	public Date getEndTime() {
		return getFieldValueDate(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_ENDTIME);
	}
	
	public ID getPlannedActivityID() {
		return getFieldValueID(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_PLANNEDACTIVITYID);
	}
	
	public String getApprovedBy() {
		return getFormattedFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_APPROVEDBY);
	}
	
	public boolean isApproved() {
		return getFieldValueBoolean(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_ISAPPROVED);
	}
	
	
	public void setStartTime(Date val) {
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_STARTTIME, val);
	}

	public void setEndTime(Date val) {
		setFieldValue(AdherenceExceptionFieldInfo.ADHERENCEEXCEPTION_ENDTIME, val);
	}
	
	
	public boolean isBasicallyEqual (AdherenceException event) {
		if (getPlannedActivityID().equals(event.getPlannedActivityID()) &&
			getApprovedBy().equals(event.getApprovedBy()) &&
			isApproved() == event.isApproved())
			return true;
		return false;
	}
	
	public int compareTo(Object o){
		AdherenceException item2 = (AdherenceException) o;
        
        long lDateDiff = getStartTime().getTime() - item2.getStartTime().getTime();
        
        if (lDateDiff < 0 ) return -1;
        if (lDateDiff > 0) return 1;
        return 0;        
    }

}
