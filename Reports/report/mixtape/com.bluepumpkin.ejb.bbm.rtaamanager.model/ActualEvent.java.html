package com.bluepumpkin.ejb.bbm.rtaamanager.model;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

import java.util.Date;
import com.bluepumpkin.common.datatypes.ID;

public class ActualEvent extends ValueObjectBase implements Comparable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new ActualEventFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	private UniqueKey uniqueKey;

	public ActualEvent() {
		super();
	}

	public ActualEvent(ID employeeID, ID activityID, Date startTime, Date endTime, boolean isPaid,
			Date timeRecordStart, Date timeRecordEnd, ID exception, ID orgDay, ID spDay) {
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_EMPLOYEEID, employeeID);
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_ACTIVITYID, activityID);
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_STARTTIME, startTime);
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_ENDTIME, endTime);
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_ISPAID, isPaid);
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_TIMERECORDSTARTTIME, timeRecordStart);
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_TIMERECORDENDTIME, timeRecordEnd);
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_ADHERENCEEXCEPTIONID, exception);
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_ORGANIZATIONDAYID, orgDay);
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_SPDAYID, spDay);

		uniqueKey = new UniqueKey(getEmployeeID(), getStartTime(), getEndTime());
	}

	public ID getEmployeeID() {
		return getFieldValueID(ActualEventFieldInfo.ACTUALEVENT_EMPLOYEEID);
	}

	public Date getStartTime() {
		return getFieldValueDate(ActualEventFieldInfo.ACTUALEVENT_STARTTIME);
	}

	public Date getEndTime() {
		return getFieldValueDate(ActualEventFieldInfo.ACTUALEVENT_ENDTIME);
	}

	public ID getActivityID() {
		return getFieldValueID(ActualEventFieldInfo.ACTUALEVENT_ACTIVITYID);
	}

	public Date getTimeRecordStartTime() {
		return getFieldValueDate(ActualEventFieldInfo.ACTUALEVENT_TIMERECORDSTARTTIME);
	}

	public Date getTimeRecordEndTime() {
		return getFieldValueDate(ActualEventFieldInfo.ACTUALEVENT_TIMERECORDENDTIME);
	}

	public boolean isPaid() {
		return getFieldValueBoolean(ActualEventFieldInfo.ACTUALEVENT_ISPAID);
	}

	public void setStartTime(Date val) {
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_STARTTIME, val);
	}

	public void setEndTime(Date val) {
		setFieldValue(ActualEventFieldInfo.ACTUALEVENT_ENDTIME, val);
	}

	public boolean isBasicallyEqual(ActualEvent event) {
		if (((getTimeRecordStartTime() == null && event.getTimeRecordStartTime() == null) || (getTimeRecordStartTime() != null
				&& event.getTimeRecordStartTime() != null && getTimeRecordStartTime().equals(
				event.getTimeRecordStartTime()))
				&& (getTimeRecordEndTime() != null && event.getTimeRecordEndTime() != null && getTimeRecordEndTime()
						.equals(event.getTimeRecordEndTime())))
				&& getActivityID().equals(event.getActivityID()) && isPaid() == event.isPaid())
			return true;
		return false;
	}

	public int compareTo(Object o) {
		ActualEvent item2 = (ActualEvent) o;

		long lDateDiff = getStartTime().getTime() - item2.getStartTime().getTime();

		if (lDateDiff < 0)
			return -1;
		if (lDateDiff > 0)
			return 1;
		return 0;
	}

	public boolean equals(Object obj) {
		if (obj == null || !(obj instanceof ActualEvent))
			return false;

		ActualEvent evt = (ActualEvent) obj;
		return (evt.getEmployeeID() != null && evt.getEmployeeID().equals(this.getEmployeeID())
				&& evt.getStartTime() != null && evt.getStartTime().equals(this.getStartTime())
				&& evt.getEndTime() != null && evt.getEndTime().equals(this.getEndTime()));
	}

	public int hashCode() {
		ID empId = getEmployeeID();
		Date dtStart = getStartTime();
		Date dtEnd = getEndTime();
		long hash = (empId == null ? 0 : empId.hashCode()) + (dtEnd == null ? 0 : dtEnd.hashCode())
				+ (dtStart == null ? 0 : dtStart.hashCode());
		return (int) hash % Integer.MAX_VALUE;
	}

	public String toString() {
		StringBuffer sb = new StringBuffer(1024);
		sb.append("ActualEvent@");
		sb.append(Integer.toHexString(hashCode()));
		sb.append('[');
		FieldInfoEntry fieldEntries[] = m_fieldInfo.getFieldInfo();
		for (int j = 0; j < fieldEntries.length; j++) {
			sb.append(fieldEntries[j].getFieldName());
			sb.append(" = ");
			sb.append(getFieldValue(j));
			sb.append('\n');
		}
		sb.append(']');
		return sb.toString();
	}

	public UniqueKey getUniqueKey() {
		if (uniqueKey == null)
			uniqueKey = new UniqueKey(getEmployeeID(), getStartTime(), getEndTime());

		return uniqueKey;
	}

	public class UniqueKey {
		private ID employeeID;
		private Date startTime;
		private Date endTime;

		public UniqueKey(ID employeeID, Date startTime, Date endTime) {
			this.employeeID = employeeID;
			this.startTime = startTime;
			this.endTime = endTime;
		}

		public ID getEmployeeID() {
			return employeeID;
		}

		public Date getStartTime() {
			return startTime;
		}

		public Date getEndTime() {
			return endTime;
		}
	}
}
