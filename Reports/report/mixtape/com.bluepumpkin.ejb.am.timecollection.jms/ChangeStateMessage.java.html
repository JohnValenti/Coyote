package com.bluepumpkin.ejb.am.timecollection.jms;

import java.io.IOException;
import java.io.ObjectInputStream;
import java.io.ObjectOutputStream;
import java.sql.Timestamp;
import java.util.Date;

import javax.jms.JMSException;
import javax.jms.Message;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.JMSMessageBase;
import com.bluepumpkin.ejb.am.timecollection.model.RawTimeEntry;
//import com.bluepumpkin.ejb.bbm.config.ConfigKey;

/**
 * Title:        Blue Pumpkin Software Activity Management
 * Description:
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */


public class ChangeStateMessage extends JMSMessageBase {
	/** serialization/deserialization compatible with previous versions. */
	static final long serialVersionUID = -7033829920852279574L;

	private String m_id;
	private ID  m_oldActivityId;
	private ID  m_newActivityId;
	private ID  m_employeeId;
	private String  m_deviceId;
	private Date m_date;

	/**
	 * default constructor
	 */
	public ChangeStateMessage() {
	}

	/**
	 *  constructor - ChangeStateMessage
	 *
	 *  @param id             message id, is id of RawTimeEntry also
	 *  @param oldActivityId   ID of  previous activity
	 *  @param newActivityId   ID of  new activity
	 *  @param employeeId     ID of employee
	 *  @param deviceeId      ID of user device
	 *  @param remark         user added remark
	 *  @param punchTime      time of punch (start time)
	 */
	public ChangeStateMessage(  String id,
															ID oldActivityId,
															ID newActivityId,
															ID employeeId,
															String deviceId,
															Date date ) {
		this.m_oldActivityId = oldActivityId;
		this.m_newActivityId = newActivityId;
		this.m_employeeId = employeeId;
		this.m_deviceId = deviceId;
		this.m_date = date;
		this.setId(id);

	}

	/**
	 *  constructor - ChangeStateMessage
	 *
	 *  @param rawTimeEntry   RawTimeEntry to init data
	 *  @param deviceeId      ID of user device
	 */
	public ChangeStateMessage(  RawTimeEntry rawTimeEntry,
															String deviceId ) {
		this.m_oldActivityId = null;
		this.m_newActivityId = rawTimeEntry.getActivityId();
		this.m_employeeId = rawTimeEntry.getEmployeeID();
		this.m_deviceId = deviceId;
		this.m_date = rawTimeEntry.getPunchDate();

	}

	/**
	 *  constructor - ChangeStateMessage
	 *
	 *  @param changeStateMessage   ChangeStateMessage to copy
	 */
	public ChangeStateMessage(  ChangeStateMessage changeStateMessage ) {

		this.m_id = changeStateMessage.m_id;
		this.m_oldActivityId = changeStateMessage.m_oldActivityId;
		this.m_newActivityId = changeStateMessage.m_newActivityId;
		this.m_employeeId = changeStateMessage.m_employeeId;
		this.m_deviceId = changeStateMessage.m_deviceId;
		this.m_date = changeStateMessage.m_date;

	}

	/**
	 * setMessageProperties
	 *
	 * sets the header in the message given
	 * adds id, deviceId employeeId in header
	 * this allows for filtering/selector of messages
	 */
	public void setMessageProperties ( Message message ) throws JMSException {
		if (message != null) {
			message.setStringProperty("id", this.m_id);
			message.setStringProperty("employeeId", this.m_employeeId.toString());
			message.setStringProperty("deviceId", this.m_deviceId);
		}
	}

	/*
	* getters/setters
	*/
	public String getId() { return m_id; }
	public void setId( String id ) {
		this.m_id = id;
	}

	public ID getOldActivityId() { return m_oldActivityId; }
	public void setOldActivityId( ID oldActivityId ) { this.m_oldActivityId = oldActivityId; }

	public ID getNewActivityId() { return m_newActivityId; }
	public void setNewActivitId( ID newActivityId ) { this.m_newActivityId = newActivityId; }

	public ID getEmployeeId() { return m_employeeId; }
	public void setEmployeeId( ID employeeId ) { this.m_employeeId = employeeId; }

	public String getDeviceId() { return m_deviceId; }
	public void setDeviceId( String deviceId ) { this.m_deviceId = deviceId; }

	public Date getDate() { return m_date; }
	public void setDate( Date date ) { this.m_date = date; }

	public String getTimeoutDbKey() {
//		return ConfigKey.MSG_TIMEOUT_AGENTCONTROL_AM;
		return "";
	}
	/*
	*  getPunchTime
	*  @deprecated, see getPunchDate
	*/
	public Timestamp getTimestamp() { return new Timestamp(m_date.getTime()); }

	/**
	* print object variables
	*/
	public String toString () {
		StringBuffer buffer = new StringBuffer();
		buffer.append("\nChangeStateMessage.m_id: "+m_id);
		buffer.append("\nChangeStateMessage.m_oldActivityId: "+m_oldActivityId);
		buffer.append("\nChangeStateMessage.m_newActivityId: "+m_newActivityId);
		buffer.append("\nChangeStateMessage.m_employeeId: "+m_employeeId);
		buffer.append("\nChangeStateMessage.m_deviceId: "+m_deviceId);
		buffer.append("\nChangeStateMessage.m_date: "+m_date);
		return buffer.toString();
	}

	private void writeObject(ObjectOutputStream out) throws IOException {
		out.defaultWriteObject();
		out.writeObject(m_id);
		out.writeObject(m_oldActivityId);
		out.writeObject(m_newActivityId);
		out.writeObject(m_employeeId);
		out.writeObject(m_deviceId);
		out.writeObject(m_date);
	}

	private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
		in.defaultReadObject();
		m_id = (String)in.readObject();
		m_oldActivityId = (ID)in.readObject();
		m_newActivityId = (ID)in.readObject();
		m_employeeId = (ID)in.readObject();
		m_deviceId = (String)in.readObject();
		m_date = (Date)in.readObject();
	}
}
