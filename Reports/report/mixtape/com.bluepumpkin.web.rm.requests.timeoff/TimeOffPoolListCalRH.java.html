package com.bluepumpkin.web.rm.requests.timeoff;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.workresource.OrganizationScopeManagerProxy;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;
import com.bluepumpkin.web.core.base.RequestHandler;
import com.bluepumpkin.web.rm.keys.RmPageAction;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        TimeOffPoolListCalRH
 * Description:  Handle requests from the Time Off Pools List page for the
 *               "Requests \ Time Off Calendar" module, as well as requests for
 *               the "Requests \ Time Off Calendar" page itself.
 * Copyright:    Copyright (c) 2010
 * Company:      Verint, Inc
 * @author       Gonzalo Quintanar
 */
public class TimeOffPoolListCalRH extends RequestHandler
{
	private static final Category log = Log.initCategory(TimeOffPoolListCalRH.class.getName());

    private static final String VIEW_PRIVILEGE = PrivilegeKeys.TOM_VIEWTIMEOFFCALENDAR;
	public static final String LIST_PAGE = "listPage";
	public static final String EDIT_PAGE = "editPage";
	public static final String CONFIRM_PAGE = "confirmPage";


	/**
	 * Process Request for the User page
	 */
	@Override
	public void processRequest(RequestContext context) throws Exception
    {
		if (isAction(context, PageAction.VIEW) || isAction(context, PageAction.VIEW_DETAIL)) //these actions comes from the list page
        {
			goToFormPage(context);
        }
		else if (isAction(context, PageAction.REFRESH)) //this action comes from the form page
        {
			goToFormPage(context);
        }
		else if (isAction(context,RmPageAction.TIMEOFF_ALLOCATION_TYPE_CHANGE_ACTION)) //this action comes from the form page
        {
			AgentsTimeOffCalendarPM model = new AgentsTimeOffCalendarPM(context);
			model.extractParameters(context.getRequest());
			String allocationType = context.getRequest().getParameter(model.ALLOCATION_TYPE_FN);
			if(!StringUtil.isEmpty(allocationType)) {
				model.setAllocationType(allocationType);
			} else{
				model.setAllocationType(TimeOffCalendarPC.DAILY_ALLOCATION_DROPDOWN_VALUE);
			}
			context.setPageModel(model);
			setNextPage(context, EDIT_PAGE);
        }
		else
        {
			setNextPage(context, LIST_PAGE);
		}
	}

	/**
	 * To go Form Page
	 */
	private void goToFormPage(RequestContext context) {
		if (isAuthorized(context, VIEW_PRIVILEGE) || isAuthorized(context, PrivilegeKeys.TOM_VIEWAVAILABLETIMEOFF)) {
			setNextPage(context, EDIT_PAGE);
		} else {
			setNextPage(context, LIST_PAGE);
		}
	}

	protected boolean isAuthorizedToView(RequestContext context, ID orgID) {
		return isValidSelectedOrg(context, orgID) &&
		        (context.getUser().isAuthorized(VIEW_PRIVILEGE, OrganizationScopeManagerProxy.getScopeID(orgID))
		        || context.getUser().isAuthorized(PrivilegeKeys.TOM_VIEWAVAILABLETIMEOFF, OrganizationScopeManagerProxy.getScopeID(orgID)));
	}

    /**
     * Return true if selected Org is valid
     */
    protected boolean isValidSelectedOrg(RequestContext context, ID orgID) {
        return orgID!=null &&
                OrganizationModelHandler.existsOrganization(context, orgID) &&
                !OrganizationModelHandler.ROOT_ORG_ID.equals(orgID);
    }

}
