	protected String getTableName() {

	/**
	 * Returns the contents of the SQL WHERE clause that is used to fetch the
	 * events for a given workresource in a given time period
	 *
	 * The implementation of this method relies on the fact that there are
	 * columns STARTTIME, DURATION and WORKRESOURCEID in the table which this
	 * DAO class is modeling, i.e. it will work with mostl DAO classes that
	 * model single-attendee events. Certain derived classes will override this
	 * method to provide the appropriate SQL where clause contents for their
	 * case. For example CalendarEventAssignmentDAO class will override this
	 * method to provide the appropriate SQL for multi-attendee events.
	 */
	protected StringBuffer getWhereClauseForGetEventsQueryByTimeRange(Date dtStart, Date dtEnd) throws JdmoException {
		StringBuffer strSQL = new StringBuffer(100);
		strSQL.append(m_dmo.dateAddMinutes("A.DURATION", "A.STARTTIME"));
		strSQL.append(" > '" + JdmoUtil.formatDBString(dtStart));
		strSQL.append("' AND ");

		// to enhance the sql statement performance.
		// add a.starttime > dtstart - 24 hours for querying shiftassignment
		// and shiftassignmentpub table to constraint the duration check range.
		String tableName = getTableName().toUpperCase();
		if ((tableName.equals("SHIFTASSIGNMENTPUB") || tableName.equals("SHIFTASSIGNMENT"))) {
			Calendar aCalendar = new GregorianCalendar(TimeZone.getTimeZone("GMT"));
			aCalendar.setTime(dtStart);
			aCalendar.add(Calendar.DATE, -1);
			Date oneDayBack = aCalendar.getTime();

			strSQL.append(" A.STARTTIME > '" + JdmoUtil.formatDBString(oneDayBack));
			strSQL.append("' AND ");
		}
		if (dtEnd != null) {
			strSQL.append(" A.STARTTIME < '" + JdmoUtil.formatDBString(dtEnd));
			strSQL.append("' AND ");
		}
		return strSQL;
	}

	/**
	 * Returns the contents of the SQL WHERE clause that is used to fetch the
	 * events for a given set of work resources in a given time period
	 */
	protected StringBuffer getWhereClauseForGetEventsQuery(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd)
			throws JdmoException {
		StringBuffer strSQL = getWhereClauseForGetEventsQueryByTimeRange(dtStart, dtEnd);
		strSQL.append(" A.WORKRESOURCEID IN ");
		strSQL.append(m_dmo.createInClause(workResourceIDs));
		return strSQL;
	}

	protected StringBuffer getWhereClauseForGetEventsQueryByStartTime(Collection<ID> workResourceIDs, Date dtStart,
			Date dtEnd) throws JdmoException {
		StringBuffer strSQL = new StringBuffer(200);
		strSQL.append("A.STARTTIME");
		strSQL.append(" >= '" + JdmoUtil.formatDBString(dtStart));
		strSQL.append("' AND ");
		if (dtEnd != null) {
			strSQL.append(" A.STARTTIME <= '" + JdmoUtil.formatDBString(dtEnd));
			strSQL.append("' AND ");
		}
		strSQL.append(" A.WORKRESOURCEID IN ");
		strSQL.append(m_dmo.createInClause(workResourceIDs));
		return strSQL;
	}

	/**
	 * Returns the contents of the SQL WHERE clause that is used to fetch the
	 * events that fall within the already published periods for a given set of
	 * work resources in a given time period
	 */
	protected StringBuffer getWhereClauseForGetEventsInPublishedPeriodsQuery(Collection<ID> workResourceIDs, Date dtStart,
			Date dtEnd) throws JdmoException {
		StringBuffer strSQL = getWhereClauseForGetEventsQuery(workResourceIDs, dtStart, dtEnd);
		strSQL.append(" AND EXISTS ");
		strSQL.append(" (SELECT 1 ");
		strSQL.append(" FROM PUBLISHINGPERIOD B ");
		strSQL.append(" WHERE A.WORKRESOURCEID = B.WORKRESOURCEID AND ");
		strSQL.append(" A.STARTTIME < B.ENDTIME AND ");
		strSQL.append(m_dmo.dateAddMinutes("A.DURATION", "A.STARTTIME"));
		strSQL.append(" > B.STARTTIME) ");
		return strSQL;
	}

	/**
	 * Returns a collection of schedule events for multiple work resources
	 */
	public Collection<Collection<T>> getEventsForWorkResources(Collection<ID> workResourceIDs,
			Date dtStart, Date dtEnd) throws BbmFinderException {
		try {
			StringBuffer strSQL = getWhereClauseForGetEventsQuery(workResourceIDs, dtStart, dtEnd);

			// QA 38793:Silk 80826 To filter the shifts after termination date
			return groupNFilterEvents(workResourceIDs, dtStart, dtEnd, strSQL.toString());

		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}

	// QA 38793 To filter the shifts after termination date
	/*
	 * Invokes ScheduleDAOUtil to group the events by work resources and finds
	 * the terminated employees and filters the events after termination date
	 */
	protected Collection<Collection<T>> groupNFilterEvents(Collection<ID> workResourceIDs, Date dtStart,
			Date dtEnd, String strSQL) throws BbmFinderException {
		Collection<Collection<T>> events = ScheduleDAOUtil.groupEventsByWorkResources(getObjects(strSQL), workResourceIDs);

		// Query for the terminated Employees
		Map<ID, Date> terminatedEmpList = getTerminatedEmployees(workResourceIDs, dtStart, dtEnd);
		// If any terminated employees exist, then filter the events list by end
		// date
		if (terminatedEmpList != null && !terminatedEmpList.isEmpty()) {
			return ScheduleDAOUtil.filterShiftsByEmpEndDate(events, workResourceIDs, terminatedEmpList);
		}

		return events;
	}

	protected void deleteEventsGivenWhereClause(String whereClause) throws BbmRemoveException {
		try {
			StringBuffer strSQL = new StringBuffer(200);
			strSQL.append("SELECT ID FROM ");
			strSQL.append(getTableName());
			strSQL.append(" A ");
			strSQL.append("WHERE ");
			strSQL.append(whereClause);
			DAOUtil.deleteObjects(m_dmo, getTableName(), strSQL.toString());
		} catch (JdmoException e) {
			throw new BbmRemoveException(e);
		}
	}

	/**
	 * Deletes schedule events for a set of work resources in a given time
	 * window
	 */
	public void deleteEventsForWorkResources(Collection<ID> workResourceIDs, Date startTime, Date endTime)
			throws BbmRemoveException {
		try {
			StringBuffer where = getWhereClauseForGetEventsQuery(workResourceIDs, startTime, endTime);
			deleteEventsGivenWhereClause(where.toString());
		} catch (JdmoException e) {
			throw new BbmRemoveException(e);
		}
	}

	/**
	 * Provides a Map of employee Id and termination date
	 */
	protected Map<ID, Date> getTerminatedEmployees(Collection<ID> workResourceIDs, Date startTime, Date endTime)
			throws BbmFinderException {
		try {
			// create the sql statement
			StringBuffer sqlQuery = new StringBuffer(128);
			sqlQuery.append("SELECT ID, ENDTIME FROM EMPLOYEEAM ");
			// Add the where clause
			sqlQuery.append(" WHERE  ");
			if (endTime != null) {
				sqlQuery.append(" EMPLOYEEAM.ENDTIME ");
				sqlQuery.append(" <= '" + JdmoUtil.formatDBString(endTime));
				sqlQuery.append("' AND ");
			}
			sqlQuery.append(" EMPLOYEEAM.ID IN ");
			sqlQuery.append(m_dmo.createInClause(workResourceIDs));

			// Obtain the workresource list
			Map<ID, Date> workResources = new java.util.HashMap<ID, Date>();
			ID id = null;
			Date empEndTime = null;
			JdmoRowset rs = m_dmo.createRowset(sqlQuery.toString());
			while (rs.next()) {
				id = rs.getID(1);
				empEndTime = rs.getTimestamp(2);
				workResources.put(id, empEndTime);
			}
			return workResources;
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
	}
}
