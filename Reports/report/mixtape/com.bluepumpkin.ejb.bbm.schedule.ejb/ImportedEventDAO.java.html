import java.util.Arrays;
import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.schedule.model.ImportedEvent;
import com.bluepumpkin.ejb.bbm.schedule.model.ImportedEventFields;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

class ImportedEventDAO extends EventAssignmentDAO<ImportedEvent> {
	// meta data info.
	private static FieldInfo m_fieldInfo = new ImportedEventFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ImportedEventDAO() {
		super();
	}

	public ImportedEventDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ImportedEvent createValueObject() {
		return new ImportedEvent();
	}

	/** override from base class */
	protected DAOBase createChildDAO(int iType) {
		// only has one child object
		if (iType == ImportedEventFields.CHILD_EVENT) {
			return this;
		} else
			return null; // should never happen
	}

	/**
	 * This is an override of the base class
	 */
	protected StringBuffer getWhereClauseForGetEventsQuery(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd)
			throws JdmoException {
		StringBuffer strWhere = super.getWhereClauseForGetEventsQuery(workResourceIDs, dtStart, dtEnd);
		// I asked Ed H. whether or not the resulting collection should include
		// children events and parent events (e.g. shifts assignments and shift
		// event assignments), and he said that the collection should only
		// include
		// the parent events e.g. shift assignments, which means we should
		// filter
		// out any records with non-null PARENTEVENTID field
		strWhere.append(" AND A.PARENTEVENTID IS NULL ");
		return strWhere;
	}

	/**
	 * Returns the collection of imported events for a single work resource
	 * (e.g. an employee) with the given ID. An employee's event will be
	 * returned as part of the collection of the event's duration interval
	 * overlaps with the requested time interval (start date to end date)
	 */
	public Collection<ImportedEvent> getEventsForWorkResource(ID workResourceID, Collection<ID> activityCol, Date dtStart, Date dtEnd)
			throws BbmFinderException {
		try {
			// Filtering based on Activity IDs
			StringBuffer strSQL = getWhereClauseForGetEventsQuery(Arrays.asList(workResourceID), dtStart, dtEnd);
			strSQL.append(" AND A.ACTIVITYID IN ");
			strSQL.append(m_dmo.createInClause(activityCol));
			return getObjects(strSQL.toString());
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}
}
