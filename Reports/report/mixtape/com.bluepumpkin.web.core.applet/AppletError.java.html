package com.bluepumpkin.web.core.applet;

import java.io.Serializable;

public class AppletError implements Serializable {
    protected String m_className;
    String [] m_messages;
    public AppletError(String className, String [] messages) {
        m_className = className;
        m_messages = messages;
    }

    public String getClassName() {
        return m_className;
    }

    public String[] getMessages() {
        return m_messages;
    }

    public String toString() {
        String s = "AppletError[class=" + m_className + "messages={";
        if (m_messages != null && m_messages.length > 0) {
            for (int i = 0; i < m_messages.length; i++) {
                s += "\"" + m_messages[i] + "\"";
                if (i < m_messages.length - 1) {
                    s += ",";
                }
            }
        }
        s += "}";
        return s;
    }
}
