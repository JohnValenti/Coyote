package com.bluepumpkin.ejb.facade.export.ejb;

import java.util.Collection;
import java.rmi.RemoteException;

import com.bluepumpkin.ejb.facade.Log;
import com.bluepumpkin.ejb.facade.base.FacadeException;

import com.bluepumpkin.common.logging.Category;

import com.bluepumpkin.ejb.core.base.SessionEJBBase;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;


/**
 * Title:        Blue Pumpkin Software Facade Module
 * Description:  PeopleFacade implementation
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author       lixin zhang
 * @version      1.0

public class ExportManagerEJB extends SessionEJBBase
{
  /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
private static Category m_cat = Log.initCategory(ExportManagerEJB.class.getName());

  /** override the base class to provide the appropriate logging category */
  protected Category getCategory()
  {
          return m_cat;
  }

  public Collection getValueObjects( FieldInfo info,
		  								String selectStmt,
										String fromWhereClause )
  					throws FacadeException, RemoteException
  {
	  ExportDAO dao = new ExportDAO(info,
			  							selectStmt,
			  							fromWhereClause);
	  try
	  {
		  return dao.getObjects("");
	  }
	  catch( Exception ex )
	  {
		  this.handleException(ex);
		  throw new FacadeException(ex);
	  }
	  
	  finally
	  {
		  dao.cleanUp();
	  }
	}  
}
