package com.verint.web.rest.wfm.resource.shiftassignment;

import com.verint.web.rest.wfm.repository.WfoRelationshipRepository;
import com.verint.web.rest.wfm.repository.WfoResourceRepository;
import com.verint.web.rest.wfm.resource.shifteventassignment.ShiftEventAssignmentEntity;
import com.verint.web.rest.wfm.resource.shifteventassignment.ShiftEventAssignmentRepository;
import io.katharsis.repository.annotations.JsonApiRelationshipRepository;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.inject.Inject;

@JsonApiRelationshipRepository(source = ShiftAssignmentEntity.class, target = ShiftEventAssignmentEntity.class)
public class ShiftToShiftEventRepository
		extends WfoRelationshipRepository<Integer, Integer, ShiftAssignmentEntity, ShiftEventAssignmentEntity> {

	private final static Logger log = LoggerFactory.getLogger(ShiftToShiftEventRepository.class.getName());

	@Inject
	private ShiftAssignmentRepository saRepository;

	@Inject
	private ShiftEventAssignmentRepository seaRepository;

	@Override
	public WfoResourceRepository<Integer, ShiftAssignmentEntity> getParentRepository() {
		return saRepository;
	}

	@Override
	public WfoResourceRepository<Integer, ShiftEventAssignmentEntity> getChildRepository() {
		return seaRepository;
	}

	@Override
	protected Logger logger() {
		return log;
	}
}
