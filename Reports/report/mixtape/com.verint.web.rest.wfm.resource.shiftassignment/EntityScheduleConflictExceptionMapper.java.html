package com.verint.web.rest.wfm.resource.shiftassignment;

import java.util.Collection;
import java.util.stream.Collectors;

import io.katharsis.errorhandling.ErrorData;
import io.katharsis.errorhandling.ErrorResponse;
import io.katharsis.errorhandling.mapper.ExceptionMapper;
import io.katharsis.legacy.queryParams.errorhandling.ExceptionMapperProvider;

@ExceptionMapperProvider
public class EntityScheduleConflictExceptionMapper implements ExceptionMapper<EntityScheduleConflictException> {

	@Override
	public ErrorResponse toErrorResponse(EntityScheduleConflictException exception) {
		Collection<ErrorData> errorData = exception.getDetails().stream()
				.map(detail -> ErrorData.builder().setTitle(exception.getTitle()).setDetail(detail).build())
				.collect(Collectors.toList());
		return ErrorResponse.builder()
				.setStatus(exception.getHttpStatusCode())
				.setErrorData(errorData)
				.build();
	}

	@Override
	public EntityScheduleConflictException fromErrorResponse(ErrorResponse errorResponse) {
		return null;
	}

	@Override
	public boolean accepts(ErrorResponse errorResponse) {
		return false;
	}
}
