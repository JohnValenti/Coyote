package com.verint.web.rest.wfm.resource.shiftassignment;

import java.util.Collection;
import java.util.TimeZone;
import java.util.stream.Collectors;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.schedule.model.BbmScheduleConflict;
import com.verint.ejb.wfm.l10n.WfmCommonBundleKey;

import io.katharsis.repository.response.HttpStatus;

public class EntityScheduleConflictException extends RuntimeException {

	private final Collection<BbmScheduleConflict> conflicts;
	private final TimeZone tz;
	private final Localizer localizer;

	public EntityScheduleConflictException(Localizer localizer, TimeZone tz, Collection<BbmScheduleConflict> conflicts) {
		this.localizer = localizer;
		this.conflicts = conflicts;
		this.tz = tz;
	}

	String getTitle() {
		return localizer.i18n(WfmCommonBundleKey.BUNDLE_NAME, WfmCommonBundleKey.SCHEDULE_CONFLICT_EXCEPTION);
	}

	int getHttpStatusCode() {
		return HttpStatus.CONFLICT_409;
	}

	Collection<String> getDetails() {
		return conflicts.stream()
				.map(conflict -> conflict.getLocalizedMessage(localizer, tz))
				.collect(Collectors.toList());
	}
}
