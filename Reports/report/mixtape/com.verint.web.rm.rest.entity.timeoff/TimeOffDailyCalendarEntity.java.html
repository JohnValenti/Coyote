package com.verint.web.rm.rest.entity.timeoff;

import java.util.ArrayList;
import java.util.Date;
import java.util.List;

import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOCalendarDayData;

@XmlRootElement(name = "timeOffCalendar")
public class TimeOffDailyCalendarEntity extends TimeOffCalendarEntity {

	private final List<TimeOffCalendarDayEntity> days = new ArrayList<TimeOffCalendarDayEntity>();

	public TimeOffDailyCalendarEntity(Date startDate, Date endDate, boolean isTOPoolView,
			List<TOCalendarDayData> daysData) {
		super(startDate, endDate, isTOPoolView);

		for (TOCalendarDayData day : daysData) {
			days.add(new TimeOffCalendarDayEntity(day));
		}
	}

	public List<TimeOffCalendarDayEntity> getDays() {
		return days;
	}
}
