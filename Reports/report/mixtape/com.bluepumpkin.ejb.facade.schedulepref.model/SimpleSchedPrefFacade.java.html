package com.bluepumpkin.ejb.facade.schedulepref.model;

import java.util.Collection;

import com.bluepumpkin.common.datatypes.*;

import com.bluepumpkin.ejb.bbm.schedulepref.model.*;
import com.bluepumpkin.ejb.bbm.effectivity.*;
 
public class SimpleSchedPrefFacade extends SimpleSchedulePreference {

    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;

	private static final int NPARENTFIELDS = SimpleSchedulePreferenceFieldInfo.SIMPLESCHEDPREF_NUMOFFIELDS;
    
    public static final int STARTTIME_LOCAL = NPARENTFIELDS + 0;
    public static final int ENDTIME_LOCAL = NPARENTFIELDS + 1;
    
    public static final int NUMOFFIELDS_EXT = 2;
    
    private Object m_fields_ext[] = new Object[NUMOFFIELDS_EXT];
    
    public SimpleSchedPrefFacade() {
    }
    
    public Object getFieldValue(int iField)
    {
        if (iField < NPARENTFIELDS)
           return super.getFieldValue(iField);
           
        return m_fields_ext[iField - NPARENTFIELDS];
    }
    
    public void setFieldValue(int iField, Object objValue)
    {
        if (iField < NPARENTFIELDS)
           super.setFieldValue(iField, objValue);
        else
           m_fields_ext[iField - NPARENTFIELDS] = objValue;
    }
    
    public void setFieldNull(int iField)
	{
        if (iField < NPARENTFIELDS)
           super.setFieldNull(iField);
        else
           m_fields_ext[iField - NPARENTFIELDS] = m_nullObject;
	}

	public boolean isFieldNull(int iField)
	{
         if (iField < NPARENTFIELDS)
           return super.isFieldNull(iField);
        else
           return (m_fields_ext[iField - NPARENTFIELDS] instanceof NullObject);
	}
    
    public void setFieldMultiValue(int iField)
    {
        if (iField < NPARENTFIELDS)
           super.setFieldMultiValue(iField);
        else
           m_fields_ext[iField - NPARENTFIELDS] = m_multiValueObject;
	}
    
    public boolean isFieldMultiValue(int iField)
    {
        if (iField < NPARENTFIELDS)
           return super.isFieldMultiValue(iField);
        else
           return (m_fields_ext[iField - NPARENTFIELDS] instanceof MultiValueObject);
	}
    
    public LocalDate getStartTimeLocal() 
    {
        return (LocalDate) m_fields_ext[STARTTIME_LOCAL - NPARENTFIELDS]; 
    }
    
    public LocalDate getEndTimeLocal() 
    {
        return (LocalDate) m_fields_ext[ENDTIME_LOCAL - NPARENTFIELDS]; 
    }
    
    public void setStartTimeLocal(LocalDate dtStart) 
    { 
        m_fields_ext[STARTTIME_LOCAL - NPARENTFIELDS] = dtStart; 
    }
    
    public void setEndTimeLocal(LocalDate dtEnd) 
    { 
        m_fields_ext[ENDTIME_LOCAL - NPARENTFIELDS] = dtEnd;  
    } 
    
    public void convertEffectivityToLocalDate(Collection colTZPeriods)
    {
        setStartTimeLocal(TimePeriodUtil.convertGMTToLocalDate(colTZPeriods, getStartTime()));
        setEndTimeLocal(TimePeriodUtil.convertGMTToLocalDate(colTZPeriods, getEndTime()));
    }
    
    public void convertEffectivityToGMTDate(Collection colTZPeriods)
    {
        Object dtStartLocal = m_fields_ext[STARTTIME_LOCAL - NPARENTFIELDS];
        if (dtStartLocal != null && dtStartLocal instanceof MultiValueObject)
           this.setFieldMultiValue(SimpleSchedulePreferenceFieldInfo.SIMPLESCHEDPREF_STARTTIME);
        else
           setStartTime(TimePeriodUtil.convertLocalDateToGMT(colTZPeriods, getStartTimeLocal()));
        
        Object dtEndLocal = m_fields_ext[ENDTIME_LOCAL - NPARENTFIELDS];
        if (dtEndLocal != null && dtEndLocal instanceof MultiValueObject)
           this.setFieldMultiValue(SimpleSchedulePreferenceFieldInfo.SIMPLESCHEDPREF_ENDTIME);
        else
           setEndTime(TimePeriodUtil.convertLocalDateToGMT(colTZPeriods, getEndTimeLocal()));
    }
}
