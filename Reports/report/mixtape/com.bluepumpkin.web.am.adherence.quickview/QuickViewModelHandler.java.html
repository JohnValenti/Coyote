package com.bluepumpkin.web.am.adherence.quickview;

import java.rmi.RemoteException;
import java.util.Collection;

import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.rtaamanager.model.AdheranceStatusData;
import com.bluepumpkin.web.core.base.ModelHandler;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        Model Handler for QuickView
 * Description:  Model Handler for QuickView
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       Shimon Kakon
 * @version 1.0
 */

public class QuickViewModelHandler extends ModelHandler {

	public static AdheranceStatusData getAdherenceStatusData(RequestContext context, Collection colEmployeeIds,
		Collection colPushedEmployeeIds, int sortField, boolean isAscending, int offset,
		int employeePerPage)
		throws BbmEJBCreateException, BbmFinderException, RemoteException {

		return BbmManagerFactory.getRTAAManager(context.isInWhatIfMode()).getAdheranceStatusData(colEmployeeIds,
			colPushedEmployeeIds, sortField, isAscending, offset, employeePerPage, context.getViewingTimeZone());
	}
}
