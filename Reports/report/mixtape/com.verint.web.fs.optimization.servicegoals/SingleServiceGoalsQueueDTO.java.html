package com.verint.web.fs.optimization.servicegoals;

import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.timeseries.model.ClientServiceGoalTraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastTraceCube;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.verint.web.fs.optimization.SingleQueueDTO;

/**
 * The Class SingleServiceGoalsQueueDTO. Instances of this class represent data transfer objects for single queue selections
 * in the context of service goals.
 */
public class SingleServiceGoalsQueueDTO extends SingleQueueDTO {

	private static final long serialVersionUID = 8287845643011662873L;
	
	private ClientServiceGoalTraceCube m_serviceGoals;
	private ForecastTraceCube m_forecast;

	/**
	 * Creates a new instance of SingleServiceGoalsQueueDTO.
	 * 
	 * @param queue the queue
	 * @param spQueue the sp queue
	 */
	public SingleServiceGoalsQueueDTO(Queue queue, SPQueue spQueue) {
		super(queue, spQueue);
	}
	
	/**
	 * Instantiates a new single service goals queue dto.
	 * 
	 * @param queue the queue
	 * @param spQueue the sp queue
	 * @param serviceGoals the service goals trace cube
	 * * @param forecast the forecast trace cube
	 */
	public SingleServiceGoalsQueueDTO(Queue queue, SPQueue spQueue, 
			ClientServiceGoalTraceCube serviceGoals, ForecastTraceCube forecast) {
		super(queue, spQueue);
		m_serviceGoals = serviceGoals;
		m_forecast = forecast;
	}
	
	/**
	 * Gets the service goals.
	 * 
	 * @return the service goals trace cube
	 */
	public ClientServiceGoalTraceCube getServiceGoals() {
		return m_serviceGoals;
	}

	/**
	 * Gets the forecast.
	 *
	 * @return the forecast trace cube
	 */
	public ForecastTraceCube getForecast() {
		return m_forecast;
	}
}
