public class QueueAttributePairValueDropDownDAO extends DAOBase {

		private static FieldInfo m_fieldInfo = new QueueAttributePairValueDropDownFieldInfo();
		protected FieldInfo getFieldInfo() { return m_fieldInfo; }

		public QueueAttributePairValueDropDownDAO() {
            super();
		}

		public QueueAttributePairValueDropDownDAO(Jdmo dmo) {
            super(dmo);
		}

		protected ValueObjectBase createValueObject()
		{
            return (new QueueAttributePairDropDownValue());
		}
		
		/**
		 *  delete objects with parent ids
		 *  NOTE: if the parent table has both ID and SID columns,
		 *  you must pass in the ID value 
		 */
		public void deleteObjectsWithParentIDsAndNonSpecifiedAttributes(Collection<ID> parentIDs, Collection<ID> attributeIDsToBeRetained)
			throws BbmRemoveException
		{
			try {
				StringBuffer buf = new StringBuffer(100);
				buf.append("SELECT " + m_fieldInfo.getIDFieldName() + " FROM ");
				buf.append(m_fieldInfo.getTableName());
				buf.append(" WHERE ");
				buf.append(m_fieldInfo.getParentIDFieldName());
				buf.append(" IN ");
				buf.append(getDMO().createInClause(parentIDs));
				if(attributeIDsToBeRetained != null && attributeIDsToBeRetained.size() > 0) {
					buf.append(" AND QUEUEATTRIBUTEID NOT IN ");
					buf.append(getDMO().createInClause(attributeIDsToBeRetained));
				}
				
				deleteObjects(buf.toString());
			} catch (JdmoException ex) {
				throw new BbmRemoveException(ex);
			}
		}
}
