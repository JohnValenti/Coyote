import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.workload.model.Media;
import com.bluepumpkin.ejb.bbm.workload.model.MediaFieldInfo;

public class MediaDAO extends DAOBase {

		private static FieldInfo m_fieldInfo = new MediaFieldInfo();
		protected FieldInfo getFieldInfo() { return m_fieldInfo; }

		public MediaDAO() {
            super();
		}

		public MediaDAO(Jdmo dmo) {
            super(dmo);
		}

		protected ValueObjectBase createValueObject()
		{
            return (new Media());
		}
}
