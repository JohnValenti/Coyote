package com.verint.ejb.bbm.forecast;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Iterator;
import java.util.Date;
import java.util.GregorianCalendar;
import java.util.Calendar;
import java.util.TimeZone;


import com.verint.ejb.bbm.forecast.model.ForecastProfileProject;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.ActualTraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;


public class ForecastProfileProjectUtil {
	
//	generate forecasted projects given historytimeseries and weights
	private static TimeZone gmtTimeZone = TimeZone.getTimeZone("GMT");
	
	public static Collection createForecastedProjectsFromHistory(Collection <ActualTraceCube> historyTimeSeries, 
			ArrayList weights, boolean bIsWeekly, Date weekStartDt, Date weekEndDt)
	{
		ArrayList<ForecastProfileProject> forecastedProject = new ArrayList<ForecastProfileProject>();
		
		double dTotalSecondsWeighted = 0.0;
		double[] dTotalSecondsWeightedPerDay = new double[7]; //7 - days a week
		double dTotalWeights = 0.0;
		double[] dTotalWeightsPerDay = new double[7];
		boolean noTotalData = true;
		boolean[] noTotalDataDays = new boolean[7];
		for(int n = 0; n<7; n++)
			noTotalDataDays[n] = true;
		
		int numWeeks = weights.size();
		double[] weightsPerWeek = new double[numWeeks];
		for(int n =0; n< numWeeks; n++)
		{
			weightsPerWeek[n] = (Double)weights.get(n);
		}
		int weekCount = -1;
		Iterator iter = historyTimeSeries.iterator();
		while(iter.hasNext())
		{			
			weekCount++;
			ActualTraceCube series = (ActualTraceCube)iter.next();
			double[] cvArray =  series.getTraceValueD(Trace.CV);// for CV
			double[] ahtArray = series.getTraceValueD(Trace.AHT); // for AHT
			int  arraySize = cvArray.length;
			
			double dWeekTotalSeconds = 0.0;
			boolean noWeekData = true;
			double[] dWeekTotalSecondsPerDay = new double[7]; // 7 days in a week
			boolean[] noDayData = new boolean[7];
			for(int n =0; n< 7; n++)
				noDayData[n] = true;
			
			int numPointsInDay = 96;
			int day = 0;
			
			
			for (int span = 0; span < arraySize; span++)
			{ // Calculate minutes spent on queue for the history week
				if (span != 0 && day != 6 && span % numPointsInDay == 0)
					++day;
				double dCV = cvArray[span];
				double dAHT = ahtArray[span];
				if (dCV != -1.0 && dAHT != -1.0) 
				{
					if (bIsWeekly)
					{
						dWeekTotalSeconds += dCV * dAHT;
						noWeekData = false;
					}
					else
					{
						dWeekTotalSecondsPerDay[day] += dCV * dAHT;
						noDayData[day] = false;
					}
				}
			}
//			Multiply in the weights for this week.
			double dWeight = weightsPerWeek[weekCount];
			
			if (!noWeekData)
			{
				noTotalData = false;
				dTotalSecondsWeighted += dWeekTotalSeconds * dWeight;
				dTotalWeights += dWeight;
			}
			// Multiply in the weights for each day in this week.
			for (int j=0; j< 7; j++)
			{
				if (!noDayData[j])
				{
					noTotalDataDays[j] = false;
					dTotalSecondsWeightedPerDay[j] += dWeekTotalSecondsPerDay[j] * dWeight;
					dTotalWeightsPerDay[j] += dWeight;
				}
			}
		}
		
		// Create the Weekly Projects
		if (!noTotalData)
		{
			double length = (dTotalSecondsWeighted / dTotalWeights) / 60.0;
			ForecastProfileProject project = new ForecastProfileProject();
			project.setStartTime(weekStartDt);
			project.setEndTime(weekEndDt);
			project.setTotalDuration((int)length);
			forecastedProject.add(project);
		}
		
		// Create the daily projects
		for (int j=0; j<7; j++)
		{
			if (!noTotalDataDays[j])
			{
				double length = (dTotalSecondsWeightedPerDay[j] / dTotalWeightsPerDay[j]) / 60.0;
				GregorianCalendar startcal = new GregorianCalendar(gmtTimeZone);
				GregorianCalendar endCal = new GregorianCalendar(gmtTimeZone);
				startcal.setTime(weekStartDt);
				startcal.add(Calendar.DATE,j);
				endCal = (GregorianCalendar)startcal.clone();
				endCal.add(Calendar.DATE,1);
				
				ForecastProfileProject project = new ForecastProfileProject();
				project.setStartTime(startcal.getTime());
				project.setEndTime(endCal.getTime());
				project.setTotalDuration((int)length);
				forecastedProject.add(project);
			}
		}
		return forecastedProject;
	}
	
	
	private static boolean getIsDateFallsBetweenThisRange(GregorianCalendar date, GregorianCalendar startDate, GregorianCalendar endDate )
	{
		return (date.after(startDate) && date.before(endDate)) || date.equals(startDate) || date.equals(endDate);
	}
}
