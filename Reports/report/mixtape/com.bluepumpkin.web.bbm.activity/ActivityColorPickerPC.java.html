	ActivityColorPickerPC(RequestContext context, String colorFieldName,
		String initialColor, ResourceBundle bbmBundle) {
		super(context);
		m_colorFieldName = colorFieldName;

		m_initialColor = initialColor;
		setName("ActivityColorPickerPC");
		initLocalizedStrings();
		constructColorTable();
	}

	private void constructColorTable() {
		Object[] firstColParams = new Object[2];
		Object[] secondColParams = new Object[1];
		Object[] colorTableParams = new Object[6];

		// Start Div
		htmlDisplay.append("<div id=\"")
			.append(getName())
			.append("Wrapper\">");

		// construct external table
		htmlDisplay.append(m_externalTableStart);

		// First row - the suggested color table
		htmlDisplay.append(m_externalTableRowStart);
		firstColParams[0] = i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_COLOR_SUGGESTED_LABEL);
		firstColParams[1] = i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_COLOR_SUGGESTED_HINT);
		htmlDisplay.append(MessageFormat.format( m_externalTableFirstColTemplate, firstColParams));
		// Add internalTable
		StringBuffer internalColorTable = new StringBuffer(4000);
		internalColorTable.append(m_internalTableStart);
		internalColorTable.append(m_internalTableShim);
		// add color rows
		for(int i = 0; i < m_suggestedColorsNumber; i += 2) {
			colorTableParams[0] = "'" + m_activityFamiliesColors[i] + "'";
			colorTableParams[1] = m_activityFamiliesColors[i];
			colorTableParams[2] = m_activityFamilies[i];
			if(!StringUtil.isEmpty(m_activityFamiliesColors[i+1])) {
				colorTableParams[3] = "'" + m_activityFamiliesColors[i+1] + "'";
				colorTableParams[4] = m_activityFamiliesColors[i+1];
				colorTableParams[5] = m_activityFamilies[i+1];
				internalColorTable.append(MessageFormat.format(
					m_internalTableRowTemplate, colorTableParams));
			}
			else {
				internalColorTable.append(MessageFormat.format(
					m_internalTableRowOneColorTemplate, colorTableParams));
			}
		}
		internalColorTable.append(m_tableEnd);
		secondColParams[0] = internalColorTable.toString();
		htmlDisplay.append(MessageFormat.format( m_externalTableSecondColTemplate, secondColParams));
		htmlDisplay.append(m_externalTableRowEnd);

		// add a spacer row
		htmlDisplay.append(m_externalTableRowStart);
		htmlDisplay.append(m_externalTableSpacer);
		htmlDisplay.append(m_externalTableRowEnd);

		// Third row - the color picker
		htmlDisplay.append(m_externalTableRowStart);
		firstColParams[0] = i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_COLOR_OTHER_LABEL);
		firstColParams[1] = i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_COLOR_OTHER_HINT);
		htmlDisplay.append(MessageFormat.format( m_externalTableFirstColTemplate, firstColParams));
		secondColParams[0] = HtmlUtil.makeColorPickerCaller(m_colorFieldName,
			m_initialColor, COLOR_ON_CHNAGE_EVENT, m_localizer);
		htmlDisplay.append(MessageFormat.format( m_externalTableSecondColTemplate, secondColParams));
		htmlDisplay.append(m_externalTableRowEnd);
		htmlDisplay.append(m_externalTableSpacer);

		// End table
		htmlDisplay.append(m_tableEnd);

		// End Div
		 htmlDisplay.append("</div>");


	} // end of method constructColorTable()
