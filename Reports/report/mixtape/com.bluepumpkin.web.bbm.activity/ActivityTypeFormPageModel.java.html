public class ActivityTypeFormPageModel extends OrgWorkpaneFormPageModel {
	private static final Category log = Log.initCategory(ActivityTypeFormPageModel.class.getName());
	private ActivityCategory m_activityType;
	private boolean m_hasMyTimeLicense = false;
	private boolean m_hasAccrualLicense = false;

	// Fields name
	public static final String NAME_FN = "activityTypeName";
	public static final String DESCRIPTION_FN = "activityTypeDescription";
	public static final String VISIBLE_FOR_MY_TIME_FN = "visibleForMyTime";
	public static final String IS_TO_ALLOTMENT_FN = "isToAllotment";
	public static final String TO_ACCRUAL_SCHEDULE_FN = "toAccrualSchedule";
	public static final String TO_ACCRUAL_POLICY_FN = "toAccrualPolicy";

	// Constants
	public static final boolean IS_VISIBLE_FOR_MYTIME_DEFAULT = true;

	private static boolean IS_EDITABLE = true;

	/**
	 * Constructor
	 *
	 * @param context the Request Context
	 */
	public ActivityTypeFormPageModel(RequestContext context) {
		super(context);
		m_activityType = new ActivityCategory();
	}


	public boolean extractParameters(HttpServletRequest request) {
		if(super.extractParameters(request)){
			m_activityType.setOrganizationId( getSelectedIDObject() );
			// get the value for check boxes since checkboxes values are not
			// submitted with a form if they are not checked
			String visibleForMyTime = request.getParameter(ActivityTypeFormPageModel.VISIBLE_FOR_MY_TIME_FN);
			setVisibleForMyTime((visibleForMyTime != null));
			return true;
		}

		return false;
	}



	/**
	 * Return instance of Model which is ready to be rendered
	 *
	 * @param context the Request Context
	 */
	public static ActivityTypeFormPageModel getInstance(RequestContext context) {
		return (ActivityTypeFormPageModel)getInstance(context, ActivityTypeFormPageModel.class);
	}

	/**
	 * Return the form title
	 *
	 * @return form title
	 */
	public String getFormTitle() {
		return i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_TYPE_FORM_TITLE);
	}


	public ID getActivityTypeID() {
		if (m_performActionOnIDs == null) {
			return null;
		}
		else {
			return m_performActionOnIDs[getEditIndex()];
		}
	}

	/**
	 * Returns the organization dispaly label
	 */
	public String getOrganizationLabel() {
		return i18n(m_bbmBundle, BbmWebBundleKey.LIST_HEADER_OWNER_ORG);
	}

		public String getActivityTypeOrganizationName() {
		try{
			return OrganizationModelHandler.getOrganizationName(m_context, m_activityType.getOrganizationId());
		} catch (Exception ex){
			return this.getOrganizationName();
		}
	}

	public String getActivityTypeName() {
		return m_activityType.getName() == null ? null : m_activityType.getName().trim();
	}

	public void setActivityTypeName(String name) {
		m_activityType.setName((name == null ? name : name.trim()));
	}

	public String getActivityTypeDescription() {
		return m_activityType.getDescription();
	}

	public void setActivityTypeDescription(String desc) {
		m_activityType.setDescription(StringUtil.convertEmptyStringToNull(desc));
	}

	public boolean getVisibleForMyTime() {
		if(this.isNew()) {
			return IS_VISIBLE_FOR_MYTIME_DEFAULT;
		}
		else {
			return m_activityType.isVisibleToTimeCollection();
		}
	}


	public void setVisibleForMyTime(boolean visible) {
		m_activityType.setVisibleToTimeCollection(visible);
	}

	public boolean isToAllotment() {
		return m_activityType.isTOAllotment();
	}

	public void setIsToAllotment(boolean bValue) {
		m_activityType.setTOAllotment(bValue);
	}

	public int getToAccrualSchedule() {
		return m_activityType.getTOAccrualSchedule();
	}

	public void setToAccrualSchedule(int value) {
		m_activityType.setTOAccrualSchedule(value);
	}

	public int getToAccrualPolicy() {
		return m_activityType.getTOAccrualPolicy();
	}

	public void setToAccrualPolicy(int value) {
		m_activityType.setTOAccrualPolicy(value);
	}


	/**
	 * Gets the activity type value object
	 *
	 * @return activity type value object
	 */
	public ActivityCategory getActivityType() {
		m_activityType.setID(getActivityTypeID());
		return m_activityType;
	}



	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return "activity_type_form";
	}


	/**
	 * Set the validators to the form fields
	 *
	 */
	protected void initFieldValidators() {
		// Add Field Validators
		addValidator(new IsEnteredFieldValidator(this, NAME_FN,
				UIFWebBundleKey.LABEL_NAME, UIFWebBundleKey.BUNDLE_NAME));
	}


	/**
	 * Returns a (single) entity type.
	 */
	protected String getTitleEntityType() {
		return this.getFormTitle();
	}


	/**
	 * Returns an entity name. Used for title.
	 */
	protected String getEntityName() {
		return this.getActivityTypeName();
	}



	/**
	 * createFormModel
	 *
	 */
	protected void createFormModel() {

		try {
			if( !this.isNew() ) {
				// Edit mode - get the info from backend
				m_activityType = ActivityTypeModelHandler.getActivityTypeByID(m_context, getCurrentID());
			}
			else {
				// Create new mode - set the org name and ID
				if (hasSelectedID()) {
					ID orgID = getSelectedIDObject();
					m_activityType.setOrganizationId(orgID);
				}
			}
			m_initFailure = false;
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
			m_initFailure = true;
		}
	}


	/**
	 * Initialize Form Display
	 */
	protected void initForm() {
		checkForLicenses();
		FormTwoColumnPC formPC = new FormTwoColumnPC(m_context);
		addChildComponent(formPC);
		setForm(formPC);
		formPC.setTitle(getFormTitle());

		formPC.addRow(getOrganizationLabel(), getActivityTypeOrganizationName());

		if (IS_EDITABLE){//show editable fields
			formPC.addRow(getNameLabel(),
				HtmlUtil.makeTextInput(NAME_FN, 50, 50, getActivityTypeName(), JSUtil.ON_CHANGE));

			formPC.addRow(getDescriptionLabel(),
				HtmlUtil.makeTextInput(DESCRIPTION_FN, 50, 50, getActivityTypeDescription(), JSUtil.ON_CHANGE));

			if (m_hasMyTimeLicense){
				formPC.addRow(getVisibleForMyTimeLabel(),
					HtmlChoiceInput.makeChkBoxInput(VISIBLE_FOR_MY_TIME_FN, "true", "", getVisibleForMyTime(), false, JSUtil.ON_CHANGE));
			}
		} else {//show read only fields
			formPC.addRow(getNameLabel(), getActivityTypeName());

			formPC.addRow(getDescriptionLabel(), getActivityTypeDescription());

			if (m_hasMyTimeLicense){
				formPC.addRow(getVisibleForMyTimeLabel(),	m_localizer.formatYesNo(getVisibleForMyTime()));
			}
		}
		
		HtmlBooleanCheckbox cb = new HtmlBooleanCheckbox();
		cb.setInputName(IS_TO_ALLOTMENT_FN);
		cb.setIsChecked(isToAllotment());
		cb.setIsDisabled(!IS_EDITABLE);
		formPC.addRow(i18n(m_bbmBundle, m_hasAccrualLicense ? 
				BbmWebBundleKey.ACTIVITY_LABEL_TIMEOFF_WITH_ACCRUAL :
				BbmWebBundleKey.ACTIVITY_LABEL_TIMEOFF_WITH_ALLOTMENT), cb);
		
		if (m_hasAccrualLicense) {
			ArrayList options = new ArrayList();
			options.add(new StringsPair(String.valueOf(Activity.TO_ACCRUAL_SCHEDULE_DAILY),
					ActivityModelHandler.getAccrualScheduleStr(m_localizer, Activity.TO_ACCRUAL_SCHEDULE_DAILY)));
			options.add(new StringsPair(String.valueOf(Activity.TO_ACCRUAL_SCHEDULE_WEEKLY),
					ActivityModelHandler.getAccrualScheduleStr(m_localizer, Activity.TO_ACCRUAL_SCHEDULE_WEEKLY)));
			options.add(new StringsPair(String.valueOf(Activity.TO_ACCRUAL_SCHEDULE_SEMI_MONTHLY),
					ActivityModelHandler.getAccrualScheduleStr(m_localizer, Activity.TO_ACCRUAL_SCHEDULE_SEMI_MONTHLY)));
			options.add(new StringsPair(String.valueOf(Activity.TO_ACCRUAL_SCHEDULE_MONTHLY),
					ActivityModelHandler.getAccrualScheduleStr(m_localizer, Activity.TO_ACCRUAL_SCHEDULE_MONTHLY)));
			options.add(new StringsPair(String.valueOf(Activity.TO_ACCRUAL_SCHEDULE_YEARLY),
					ActivityModelHandler.getAccrualScheduleStr(m_localizer, Activity.TO_ACCRUAL_SCHEDULE_YEARLY)));
			formPC.addRow(i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_LABEL_TIMEOFF_ACCRUAL_SCHEDULE),
					HtmlUtil.makeSelectInput(TO_ACCRUAL_SCHEDULE_FN,
							new String[]{String.valueOf(getToAccrualSchedule())},
							JSUtil.ON_CHANGE, options, false, IS_EDITABLE));
			
			options = new ArrayList();
			options.add(new StringsPair(String.valueOf(Activity.ACCRUAL_POLICY_ALL_HOURS_ON_START),
					ActivityModelHandler.getAccrualPolicyStr(m_localizer, Activity.ACCRUAL_POLICY_ALL_HOURS_ON_START)));
			options.add(new StringsPair(String.valueOf(Activity.ACCRUAL_POLICY_PRORATED_ON_START),
					ActivityModelHandler.getAccrualPolicyStr(m_localizer, Activity.ACCRUAL_POLICY_PRORATED_ON_START)));
			options.add(new StringsPair(String.valueOf(Activity.ACCRUAL_POLICY_ALL_HOURS_ON_NEXT),
					ActivityModelHandler.getAccrualPolicyStr(m_localizer, Activity.ACCRUAL_POLICY_ALL_HOURS_ON_NEXT)));
			options.add(new StringsPair(String.valueOf(Activity.ACCRUAL_POLICY_PRORATED_ON_NEXT),
					ActivityModelHandler.getAccrualPolicyStr(m_localizer, Activity.ACCRUAL_POLICY_PRORATED_ON_NEXT)));
			formPC.addRow(i18n(m_bbmBundle, BbmWebBundleKey.ACCRUAL_POLICY),
					HtmlUtil.makeSelectInput(TO_ACCRUAL_POLICY_FN,
							new String[]{String.valueOf(getToAccrualPolicy())},
							JSUtil.ON_CHANGE, options, false, IS_EDITABLE));
		}
	}


	private String getNameLabel() {
		return i18n(m_common_bundle, UIFWebBundleKey.LABEL_NAME);
	}

	private String getDescriptionLabel() {
		return i18n(m_common_bundle, UIFWebBundleKey.LABEL_DESCRIPTION);
	}

	private String getVisibleForMyTimeLabel() {
		return i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_TYPE_FORM_LABEL_MY_TIME);
	}

	private void checkForLicenses(){
		try {
			m_hasMyTimeLicense = CoreManagerFactory.getLicenseManager().isLicensed(LicenseKeys.APP_MY_TIME);
		} catch (Exception ex){
			log.l7dError(BbmWebBundleKey.UNABLE_TO_LOAD_DATA, ex);
		}
		
		m_hasAccrualLicense = TOAccrualCalculator.hasLicenseForAccrual();
	}
