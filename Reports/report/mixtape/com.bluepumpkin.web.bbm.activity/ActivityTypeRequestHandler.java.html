package com.bluepumpkin.web.bbm.activity;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;
import com.bluepumpkin.ejb.bbm.base.BbmCreateDuplicateKeyException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateDuplicateKeyException;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.bbm.l10n.BbmWebLogBundleKey;
import com.witness.web.uif.base.Message;
import com.bluepumpkin.web.core.base.RequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

import java.util.ResourceBundle;

/**
 * Title: ActivityTypeRequestHandler Description: Handle requests from Activity
 * Type list and form pages Copyright: Copyright (c) 2001 Company: Blue Pumpkin
 * Software, Inc
 * 
 * @author Shimon Kakon
 * @version 1.0
 */
public class ActivityTypeRequestHandler extends RequestHandler {
	private static final Category log = Log.initCategory(ActivityTypeRequestHandler.class.getName());
	private ResourceBundle m_bbmBundle;
	private static final String ACTIVITY_TYPE_LIST_PAGE = "activityTypeListPage";
	private static final String ACTIVITY_TYPE_FORM_PAGE = "activityTypeFormPage";
	private static final String CONFIGURE_PRIVILEGE = PrivilegeKeys.CORE_VIEWACTIVITIES;
	private Localizer m_localizer;

	/**
	 * Process Requests for the Activity Type area
	 */
	public void processRequest(RequestContext context) {
		m_localizer = context.getLocalizer();
		m_bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);

		// Process requested action
		if (isAction(context, PageAction.EDIT)) {
			processListEdit(context);
		} else if (isAction(context, PageAction.ADD)) {
			processListCreateNew(context);
		} else if (isAction(context, PageAction.DELETE)) {
			processListDelete(context);
		} else if (isAction(context, PageAction.CREATE_NEW)) {
			processFormCreateNew(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processFormSave(context);
		} else if (isAction(context, PageAction.CANCEL)) {
			processFormCancel(context);
		} else if (isAction(context, PageAction.NEW_ID)) {
			gotoList(context);
		} else {
			gotoList(context);
		}
	}

	/**
	 * Process Create new Action from list
	 */
	private void processListCreateNew(RequestContext context) {
		// validate request - check if organization was selected
		String selectedOrg = context.getRequest().getParameter(ActivityTypeListPageModel.SELECTED_ID_FN);

		if (!StringUtil.isEmpty(selectedOrg)) {
			gotoForm(context);
		} else {
			// Oops... this should never happened. must have selected
			// organization
			// before creating new activity type. Log and put a msg in page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_TYPE_NO_ORG_CREATE_FAILURE);

			ActivityTypeListPageModel model = ActivityTypeListPageModel.getInstance(context);
			model.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_TYPE_SELECT_ORG_BEFORE_CREATE));
			gotoList(context);
		}
	}

	/**
	 * Process Edit Action from list
	 */
	private void processListEdit(RequestContext context) {
		String selectedRow = context.getRequest().getParameter(ActivityTypeListPageModel.PERFORM_ACTION_ON_ID_FN);

		if (!StringUtil.isEmpty(selectedRow)) {
			gotoForm(context);
		} else {
			// Oops... this should never happened. must have selected a row
			// before editing. Log and put a msg in page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_TYPE_NO_SELECTION_EDIT_FAILURE);

			ActivityTypeListPageModel model = ActivityTypeListPageModel.getInstance(context);
			model.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_TYPE_SELECT_BEFORE_EDIT));
			gotoList(context);
		}
	}

	/**
	 * Process delete Action from list
	 */
	private void processListDelete(RequestContext context) {
		if (!isAuthorized(context, CONFIGURE_PRIVILEGE)) {
			gotoList(context);
			return;
		}

		ActivityTypeListPageModel listModel = null;
		String selectedRow = context.getRequest().getParameter(ActivityTypeListPageModel.PERFORM_ACTION_ON_ID_FN);

		if (!StringUtil.isEmpty(selectedRow)) {
			ID[] performActionOnID = StringUtil.parseIDString(selectedRow);
			listModel = new ActivityTypeListPageModel(context);
			context.setPageModel(listModel);

			// Process Delete
			processDelete(context, performActionOnID[0], ActivityCategory.OBJECT_TYPE_ACTIVITYCATEGORY, listModel,
					BbmWebBundleKey.ACTIVITY_TYPE_DELETE_FAILURE, BbmWebBundleKey.BUNDLE_NAME,
					BbmWebBundleKey.ACTIVITY_TYPE_DELETE_FAILURE, log, true);

		} else {
			// Oops... this should never happened. must have selected a row
			// before deleting. Log and put a msg in page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_TYPE_NO_SELECTION_DELETE_FAILURE);

			listModel = ActivityTypeListPageModel.getInstance(context);
			listModel.addPageMessage(Message.ERROR_TYPE, BbmWebBundleKey.ACTIVITY_TYPE_SELECT_BEFORE_DELETE,
					BbmWebBundleKey.BUNDLE_NAME);
		}

		// Go back to the list
		gotoList(context);
	}

	/**
	 * Process create new Action from form
	 */
	private void processFormCreateNew(RequestContext context) {

		try {
			// Create new Activity Type
			ActivityCategory activityType = extractForm(context);
			ID newID = ActivityTypeModelHandler.createNewActivityType(context, activityType);

			// No exceptions - so prepare to go back to the list
			// Load the List model, Set the edited row to be pre-selected in the
			// list
			ActivityTypeListPageModel listModel = new ActivityTypeListPageModel(context);
			listModel.setPerformActionOnIDs(new ID[] { newID });
			// Cache the model and go back to list
			context.setPageModel(listModel);
			gotoList(context);

		} catch (BbmCreateDuplicateKeyException e) {
			// add message to the page
			ActivityTypeFormPageModel formModel = ActivityTypeFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_TYPE_CREATE_NEW_FAILURE_DUPLICATE));
			// Go back to the form
			gotoForm(context);

		} catch (Exception e) {
			// Log the message and add it to the page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_TYPE_CREATE_NEW_FAILURE);

			ActivityTypeFormPageModel formModel = ActivityTypeFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_TYPE_CREATE_NEW_FAILURE));
			// Go back to the form
			gotoForm(context);
		}
	}

	/**
	 * Process Save Action from form
	 */
	private void processFormSave(RequestContext context) {
		try {
			// edit Activity Type after extracting from the form
			ActivityCategory activityType = extractForm(context);
			ActivityTypeModelHandler.updateActivityType(context, activityType);

			// No exceptions - so prepare to go back to the list
			// Load the List model, Set the edited row to be pre-selected in the
			// list
			ActivityTypeListPageModel listModel = new ActivityTypeListPageModel(context);
			listModel.setPerformActionOnIDs(new ID[] { activityType.getID() });
			// Cache the model and go back to list
			context.setPageModel(listModel);
			gotoList(context);

		} catch (BbmUpdateDuplicateKeyException e) {
			// add message to the page
			ActivityTypeFormPageModel formModel = ActivityTypeFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_TYPE_UPDATE_FAILURE_DUPLICATE));
			// Go back to the form
			gotoForm(context);

		} catch (Exception e) {
			// Log the message and add it to the page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_TYPE_UPDATE_FAILURE);

			ActivityTypeFormPageModel formModel = ActivityTypeFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_TYPE_UPDATE_FAILURE));
			// Go back to the form
			gotoForm(context);
		}

	}

	/**
	 * Process Cancel Action from form
	 */
	private void processFormCancel(RequestContext context) {
		ActivityTypeListPageModel listModel = new ActivityTypeListPageModel(context);
		String strID = context.getRequest().getParameter(ActivityTypeFormPageModel.PERFORM_ACTION_ON_ID_FN);

		// Set the edited row to be pre-selected in the list
		if (!StringUtil.isEmpty(strID)) {
			listModel.setPerformActionOnID(strID);
		}

		// Cache the model and go back to list
		context.setPageModel(listModel);
		gotoList(context);
	}

	/**
	 * Go to List
	 */
	private void gotoList(RequestContext context) {
		setNextPage(context, ACTIVITY_TYPE_LIST_PAGE);
	}

	/**
	 * Go to Form
	 */
	private void gotoForm(RequestContext context) {
		if (isAuthorized(context, CONFIGURE_PRIVILEGE)) {
			setNextPage(context, ACTIVITY_TYPE_FORM_PAGE);
		} else {
			gotoList(context);
		}
	}

	/**
	 * extract parameters from submitted form
	 * 
	 * @return the activity category value object created from the extracted
	 *         form
	 */
	private ActivityCategory extractForm(RequestContext context) {
		ActivityTypeFormPageModel formModel = new ActivityTypeFormPageModel(context);
		formModel.extractParameters(context.getRequest());
		return formModel.getActivityType();
	}

}
