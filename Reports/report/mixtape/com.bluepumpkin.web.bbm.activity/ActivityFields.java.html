	// Prevent instantiation
	private ActivityFields(){}

	public static short[] getDisplayMap(ID id) {
		if( (id == null) || (!systemActivitiesDisplayMap.containsKey(id)) ) {
			return editableDisplayMap;
		}
		else {
			return (short[])systemActivitiesDisplayMap.get(id);
		}
	}

	public static boolean isSystemActivity(ID id){
		return systemActivitiesDisplayMap.containsKey(id);
	}

	public static boolean hasDisplayableField(short[] displayMap, short project) {
		short begin = 0;
		short end = TOTAL_FIELDS_COUNT;

		switch(project) {
			case FS:
				begin = FS_FIELDS_START;
				end = FS_FIELDS_END;
				break;
			case RM:
				begin = RM_FIELDS_START;
				end = RM_FIELDS_END;
				break;
			case AM:
				begin = AM_FIELDS_START;
				end = AM_FIELDS_END;
				break;
            case ADVISOR:
				begin = ADVISOR_FIELDS_START;
				end = ADVISOR_FIELDS_END;
				break;
		}

		for(int i=begin; i<=end;i++) {
			if(displayMap[i] == DISPLAYABLE || displayMap[i] == EDITABLE) return true;
		}
		return false;
	}
