package com.bluepumpkin.web.bbm.activity;

import java.rmi.RemoteException;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Iterator;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.ejb.bbm.Filter.Filter;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategoryFilter;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.bluepumpkin.ejb.core.base.MultiUserException;
import com.witness.ejb.core.security.model.User;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        ActivityTypeModelHandler
 * Description:  Handle interactions with Activity Manager EJB
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author       Shimon Kakon
 * @version 1.0
 */
public class ActivityTypeModelHandler extends ActivityModelHandler {
	/**
	 * Return all Activities Types given organization ID
	 *
	 * @param orgId - The Organization ID
	 * @param filterSystemDefined - should the activity type defined in the system level
	 * be included
	 * @throws RemoteException
	 * @throws BbmException
	 * @return Colection of activity type value objects
	 */
	public static Collection<ActivityCategory> getActivityTypes(RequestContext context, ID orgId, boolean filterSystemDefined)
		throws RemoteException, BbmException {

		// Create the filter
		ActivityCategoryFilter filter = new ActivityCategoryFilter();
		if(filterSystemDefined) {
			filter.setActivityCategoryId(ActivityCategory.SYSTEM_DEFINED, Filter.OP_NOT_EQUAL);
		}

		return getActivityManager(context).findOrganizationActivityCategories(orgId, filter);
	}


	/**
	 * Return Activity Type by ID
	 *
	 * @param id - The activity type ID
	 * @throws RemoteException
	 * @throws BbmException
	 * @return activity type value object
	 */
	 public static ActivityCategory getActivityTypeByID(RequestContext context, ID id)
		throws RemoteException, BbmException {
		return getActivityManager(context).findActivityCategoryById(id);
	}


	/**
	 * Return Activity Types by ID
	 *
	 * @param ids - Collection of activity type IDs
	 * @throws RemoteException
	 * @throws BbmException
	 * @return activity type value object
	 */
	 public static Collection<ActivityCategory> getActivityTypesByID(RequestContext context, Collection ids)
		throws RemoteException, BbmException {
		return getActivityManager(context).findActivityCategories(ids);
	}


	/**
	 * Create New Activity Type
	 *
	 * @param activityType - activity type value object to create
	 * @throws RemoteException
	 * @throws BbmException
	 * @return new activity type ID
	 */
	public static ID createNewActivityType(RequestContext context, ActivityCategory activityType)
			throws RemoteException, BbmException {
		return getActivityManager(context).createActivityCategory(activityType);
	}


	/**
	 * Update Activity Type
	 *
	 * @param activityType - activity type value object to update
	 * @throws RemoteException
	 * @throws BbmException
	 */
	public static void updateActivityType(RequestContext context, ActivityCategory activityType)
			throws RemoteException, BbmException, MultiUserException {
		getActivityManager(context).updateActivityCategory(activityType);
	}


	/**
	 * Create a hashmap activityTypeID, activityTypeName (key, value) combination
	 * @param activityTypesIDs activityTypeIDs
	 * return HashMap of activityTypeID, activityTypeName
	 */
	public static Map<ID, ActivityCategory> buildActivityTypeMap(RequestContext context, Collection activityTypesIDs)
		throws RemoteException, BbmException {

		if (activityTypesIDs == null || activityTypesIDs.isEmpty()){
			return Collections.EMPTY_MAP;
		}
		Collection activityTypes = getActivityTypesByID(context, activityTypesIDs);
		HashMap activityTypeMap = new HashMap(activityTypes.size());
		if (activityTypes != null && activityTypes.size() > 0){
			for (Iterator it = activityTypes.iterator(); it.hasNext();) {
				ActivityCategory activityType = (ActivityCategory)it.next();
				activityTypeMap.put(activityType.getID(), activityType);
			}
		}
		return activityTypeMap;
	}



	/**
	 * Return an array list of activity type StringsPair
	 * Activity type id is the key and activity type name is the value
	 *
	 * @param orgId - The Organization ID
	 * @throws RemoteException
	 * @throws BbmException
	 * @return Colection of activity type StringsPair
	 */
	public static Collection<StringsPair> getActivityTypesStringsPair(RequestContext context, ID orgId, boolean filterSystemDefined)
		throws RemoteException, BbmException {

		Collection<StringsPair> activityTypeStringsPair = null;
		Collection<ActivityCategory> activityTypesList = getActivityTypes(context, orgId, filterSystemDefined);

		if(activityTypesList != null && activityTypesList.size() > 0) {
			ActivityCategory activityType = null;
			activityTypeStringsPair = new ArrayList<StringsPair>(activityTypesList.size());
			for(Iterator<ActivityCategory> it = activityTypesList.iterator(); it.hasNext();) {
				activityType = it.next();
				activityTypeStringsPair.add(new StringsPair(
					activityType.getID().toString(), activityType.getName()) );
			}
		}
		else {
			activityTypeStringsPair = Collections.emptyList();
		}

		return activityTypeStringsPair;
	}

	/**
	 * Return all activity types that are visible to the given user.
	 *	
	 * @param user  The user object.
	 *
	 * @return A collection of ActivityCategory objects.
	 * @throws RemoteException
	 * @throws BbmException
	 */
	 public static Collection getAllActivityTypesVisibleToUser(RequestContext context, User user) throws Exception {
		HashSet set = new HashSet(64);

		Collection userOrgIDs = user.getAuthorizedScopes(PrivilegeKeys.CORE_VIEWACTIVITIES_ID, Organization.SCOPE_TYPE);

		if (userOrgIDs != null) {
			for (Iterator orgIter = userOrgIDs.iterator(); orgIter.hasNext();) {
				ID orgID = (ID)orgIter.next();

				Collection actTypes = getActivityTypes(context, orgID, true);
				if (actTypes != null) {
					for (Iterator actIter = actTypes.iterator(); actIter.hasNext();) {
						set.add(actIter.next());
					}
				}
			}
		}
		return set;
	}
}
