package com.bluepumpkin.web.bbm.activity;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityProperties;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;
import com.witness.web.uif.util.js.JSUtil;
import com.witness.web.uif.l10n.UIFWebBundleKey;

import java.util.*;

/**
 * Title:        ActivityAdvisorUsage
 * Description:  A collapsable container for the Activity's Advisor usage attributes
 * Copyright:    Copyright (c) 2004
 * Company:      Blue Pumpkin Software, Inc
 * @author       Swati Tewari
 * @version 1.0
 */

public class ActivityAdvisorUsage extends ActivityUsage {

	public ActivityAdvisorUsage(RequestContext context, ResourceBundle bundle, Activity activity, ActivityProperties activityProperties) {
		super(context, bundle, activity, activityProperties);
	}

	public void initForDisplay() {
		if (m_isInitializedForDisplay) return;
		else {
			super.initForDisplay();
			if(m_visible) {
				setTitle(i18n(m_bundle, BbmWebBundleKey.ACTIVITY_TITLE_ADVISOR_USAGE));
				setName("advisorUsage");
			}
		}
	}

	protected void initFields() {
		if(m_visible) {
			m_fields = new StringBuffer(512);
			int i = 0;
			// Source Measure
			Map coreSourceMeasureMap;
			switch (m_displayMap[ActivityFields.SOURCE_MEASURE]) {
				case ActivityFields.EDITABLE:
					ArrayList options = new ArrayList();
					options.add(new StringsPair("", i18n(m_common_bundle, UIFWebBundleKey.NONE)));
					coreSourceMeasureMap = ActivityModelHandler.getCoreSourceMeasureMap(m_context);
					if (coreSourceMeasureMap != null) {
						for (Iterator it = coreSourceMeasureMap.entrySet().iterator(); it.hasNext();) {
							Map.Entry entry = (Map.Entry) it.next();
							options.add(new StringsPair(entry.getKey().toString(), entry.getValue().toString()));
						}
					}
					String selected = "";
					if (m_activity.getSourceMeasureId() != null)
						selected = m_activity.getSourceMeasureId().toString();
					m_fields.append(addToTemplate(BbmWebBundleKey.ACTIVITY_LABEL_ADV_SOURCE_MEASURE, i++));
					addComponent(HtmlUtil.makeSelectInput(ActivityFields.SOURCE_MEASURE_FN,
							selected, JSUtil.ON_CHANGE, options));
					break;
				case ActivityFields.DISPLAYABLE:
					String sourceName = i18n(m_common_bundle, UIFWebBundleKey.NONE);
					coreSourceMeasureMap = ActivityModelHandler.getCoreSourceMeasureMap(m_context);
					if (m_activity.getSourceMeasureId() != null && coreSourceMeasureMap != null &&
							coreSourceMeasureMap.keySet().contains(m_activity.getSourceMeasureId())){
						try{
							sourceName = (String)coreSourceMeasureMap.get(m_activity.getSourceMeasureId());
						} catch (Exception ex) {}
					}
					m_fields.append(addToTemplate(BbmWebBundleKey.ACTIVITY_LABEL_ADV_SOURCE_MEASURE, i++));
					addComponent(sourceName);
				case ActivityFields.NON_DISPLAYABLE:
					addHiddenField(ActivityFields.SOURCE_MEASURE);
					break;
				default:
					throw new IllegalStateException();
			}
		} else { // container is invisible
			addHiddenField(ActivityFields.SOURCE_MEASURE);
		}
	}

	public ID getSourceMeasureId() {
		return m_activity.getSourceMeasureId();
	}

	public void setSourceMeasureId(ID sourceMeasureId) {
		m_activity.setSourceMeasureId(sourceMeasureId);
	}

	public void setDefaultValues() {};

	private void addHiddenField(short field) {
		switch(field) {
			case ActivityFields.SOURCE_MEASURE:
				m_hiddenFields.append(HtmlUtil.makeHiddenInput(ActivityFields.SOURCE_MEASURE_FN,
						m_activity.getSourceMeasureId() ));
				break;
		}
		m_hiddenFields.append("\n");
	}
}
