package com.bluepumpkin.ejb.rm.setup.calendartimeoff.ejb;

import java.sql.Timestamp;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.EmployeeReferenceScheduleAssignment;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.EmployeeReferenceScheduleAssignmentUpdateParameters;
import com.verint.ejb.wfm.util.DAOUtil;

/*
 * This class contains the database access code for the setup of employee reference schedule. 
 * The back end time off calculation database access code is in com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb.ReferenceScheduleAssignmentsDAO
 */
public class EmployeeReferenceScheduleAssignmentDAO {

	private Jdmo dmo = null;


	public EmployeeReferenceScheduleAssignmentDAO() {
		dmo = new Jdmo(false);
	}

	public void cleanUp() {
		if (dmo != null) {
			dmo.cleanUp();
			dmo = null;
		}
	}


	/**
	 * @return
	 *	The valid reference schedule assignments for the specified employees in the date range [dtStart,dtEnd) 
	 */
	public List<EmployeeReferenceScheduleAssignment> getValidAssignments(Collection<ID> empIDs, Date dtStart, Date dtEnd) throws JdmoException {

		if (empIDs == null || empIDs.isEmpty()) {
			return Collections.emptyList();
		}

		String sqlStart = JdmoUtil.formatDBString(dtStart);

		//@formatter:off
		String sql =
				"SELECT A.ID , A.SPID,  A.WORKRESOURCEID, A.STARTTIME, " 
				+ "case when EAM.ENDTIME < A.EndTIme then EAM.ENDTIME else A.ENDTIME "
				+ "END as EndTime "
				+ "FROM WORKRESOURCEREFSCHEDULE A "
				+ "inner join SP on SP.ID = A.SPID "
				+ "inner join EMPLOYEEAM EAM on EAM.ID = A.WORKRESOURCEID "
				+ "where 	(EAM.EndTime is null or EAM.EndTime >= '%s') and " 
				+ "A.StartTime < '%s' and (A.EndTime is null or A.EndTime > '%s' ) "
				+ "AND A.WORKRESOURCEID IN %s "
				+ "order by A.STARTTIME ";
		//@formatter:on
		sql = String.format(sql, sqlStart, JdmoUtil.formatDBString(dtEnd), sqlStart, dmo.createInClause(empIDs));

		return rowSetToList(sql);
	}

	/**
	 * @return
	 *	The valid reference schedule assignments for the specified employees at the given date 
	 */
	public List<EmployeeReferenceScheduleAssignment> getValidAssignments(Collection<ID> empIDs, Date dt) throws JdmoException {

		if (empIDs == null || empIDs.isEmpty()) {
			return Collections.emptyList();
		}

		String sqlDate = JdmoUtil.formatDBString(dt);

		//@formatter:off
		String sql =
				"SELECT A.ID , A.SPID,  A.WORKRESOURCEID, A.STARTTIME, " 
				+ "case when EAM.ENDTIME < A.EndTIme then EAM.ENDTIME else A.ENDTIME "
				+ "END as EndTime "
				+ "FROM WORKRESOURCEREFSCHEDULE A "
				+ "inner join SP on SP.ID = A.SPID "
				+ "inner join EMPLOYEEAM EAM on EAM.ID = A.WORKRESOURCEID "
				+ "where 	(EAM.EndTime is null or EAM.EndTime >= '%s') and " 
				+ "A.StartTime <= '%s' and (A.EndTime is null or A.EndTime > '%s' ) "
				+ "AND A.WORKRESOURCEID IN %s "
				+ "order by A.STARTTIME ";
		//@formatter:on
		sql = String.format(sql, sqlDate, sqlDate, sqlDate, dmo.createInClause(empIDs));

		return rowSetToList(sql);
	}

	
	private List<EmployeeReferenceScheduleAssignment> rowSetToList(String sql) throws JdmoException {
		JdmoRowset res = dmo.createRowset(sql);

		List<EmployeeReferenceScheduleAssignment> assignments = new ArrayList<EmployeeReferenceScheduleAssignment>();

		while (res.next()) {
			EmployeeReferenceScheduleAssignment assignment = new EmployeeReferenceScheduleAssignment();

			assignment.setID(res.getID("ID"));
			assignment.setSchedulingPeriodDEID(res.getID("SPID"));
			assignment.setWorkResourceID(res.getID("WORKRESOURCEID"));
			assignment.setStartTime(TimeZoneUtil.toDate(res.getTimestamp("STARTTIME")));
			assignment.setEndTime(TimeZoneUtil.toDate(res.getTimestamp("EndTime")));
			assignments.add(assignment);
		}

		return assignments;
	}
	
	/**
	 * @return
	 * The scheduling period DEIDs which are common to all the specified employees. The number of scheduling period DEIDs that are returned 
	 * are limited by limitListSize parameter.  
	 */
	public List<ID> getCommonSchedulingPeriodDEIDs(Collection<ID> employeeIds, int limitListSize) throws JdmoException{

		if (employeeIds == null || employeeIds.isEmpty()) {
			return Collections.emptyList();
		}
		
		JdmoQuery jdmoQuery = dmo.createQuery("RM_GET_COMMON_SPDEID_FOR_EMPLOYEES", Jdmo.STORPROC_QUERY);
		String xml = DAOUtil.converIdsToXml(employeeIds);
		jdmoQuery.setParSqlXml(1, xml);
		jdmoQuery.setParInt(2, limitListSize);
		JdmoRowset res = dmo.createRowset(jdmoQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

		List<ID> result = new ArrayList<ID>(limitListSize);

		while (res.next()) {
			result.add(res.getID("SPID"));
		}
		return result;
	}
	
	/**
	 * When there are no valid reference schedule assignments for an employee at a given date we need to present the user with "filler assignments"
	 * These assignments would fill the gap present at the given date for each employee. 
	 * The start time  of each filler assignment is the end of the previous assignment for the employee. If there is no previous assignment then we use the 
	 * employee's start time (the hire date)
	 * The end time of each filler assignment is the start of the next assignment for the employee. If there is no next assignment then we use the 
	 * employee's end time (the fire date or null if they are active)
	 * 
	 * The scheduling period is a random scheduling period that all the given employees can work in. 
	 *  
	 * @return
	 *  The filler assignments for the employee at the specified date. If the employee has a valid assignment then no assignment is returned for that employee. 
	 * @throws JdmoException 
	 */
	public List<EmployeeReferenceScheduleAssignment> getFillerAssignments(Collection<ID> empIDs, Date dt) throws JdmoException{
		
		if (empIDs == null || empIDs.isEmpty()) {
			return Collections.emptyList();
		}

		List<EmployeeReferenceScheduleAssignment> fillers = new ArrayList<EmployeeReferenceScheduleAssignment>();
		List<ID> validFillerSpDEIDs = getCommonSchedulingPeriodDEIDs(empIDs, 1);
		if(validFillerSpDEIDs.isEmpty()) {
			//there is no scheduling period that is valid for all of these employees. 
			return fillers;
		}
		 
		
		JdmoQuery  jdmoQuery = dmo.createQuery("RM_GET_FILLER_REF_SCHEDULE_ASSIGNMENTS", Jdmo.STORPROC_QUERY);
		String xml = DAOUtil.converIdsToXml(empIDs);
        jdmoQuery.setParSqlXml(1, xml);
        jdmoQuery.setParTimestamp(2, new Timestamp(dt.getTime()));
        JdmoRowset res = dmo.createRowset(jdmoQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
        
        
        //any scheduling period DEID that all of the specified employees can work will do as the filler 
        ID fillerSpDEID = validFillerSpDEIDs.get(0);
        
        while (res.next()) {
			EmployeeReferenceScheduleAssignment assignment = new EmployeeReferenceScheduleAssignment();

			assignment.setSchedulingPeriodDEID(fillerSpDEID);
			assignment.setWorkResourceID(res.getID("EmployeeID"));
			assignment.setStartTime(TimeZoneUtil.toDate(res.getTimestamp("StartTime")));
			assignment.setEndTime(TimeZoneUtil.toDate(res.getTimestamp("EndTime")));
			fillers.add(assignment);
		}
        
        return fillers;
	}

	public void updateEmployeeReferenceScheduleEffectivity(Collection<ID> empIDs, Date asOfDate, EmployeeReferenceScheduleAssignmentUpdateParameters assignmentsToUpdate) throws JdmoException {

		if (empIDs == null || empIDs.isEmpty()) {
			return;
		}

		int option = (assignmentsToUpdate.isUpdateStartTime()? 1 : 0) | (assignmentsToUpdate.isUpdateEndTime()? 2 : 0) | (assignmentsToUpdate.isUpdateSpID()? 4 : 0);
		String xml = DAOUtil.converIdsToXml(empIDs);
		
		JdmoQuery  jdmoQuery = dmo.createQuery("RM_UPDATE_REF_SCHEDULE_ASSIGNMENTS", Jdmo.STORPROC_QUERY_NORS);
        jdmoQuery.setParSqlXml(1, xml);
        jdmoQuery.setParTimestamp(2, new Timestamp(asOfDate.getTime()));
        jdmoQuery.setParID(3, assignmentsToUpdate.getSchedulingPeriodDEID());
        jdmoQuery.setParTimestamp(4, TimeZoneUtil.toTimestamp(assignmentsToUpdate.getStartTime()));
        jdmoQuery.setParTimestamp(5, TimeZoneUtil.toTimestamp(assignmentsToUpdate.getEndTime()));
        jdmoQuery.setParInt(6, option);
        dmo.execute(jdmoQuery);

	}

}
