package com.bluepumpkin.ejb.rm.setup.calendartimeoff.ejb;

/**

public final class TOPoolDAO extends DAOBase {

	private static FieldInfo m_fieldInfo = new TOPoolFieldInfo(); //OrganizationFieldInfo();

	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public TOPoolDAO() {
		super();
	}

	public TOPoolDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ValueObjectBase createValueObject() {
		return (new TOPool());
	}

	public String getTOPoolName(ID idTOPool) throws BbmObjectNotFoundException, BbmFinderException {
		String strSQL = "SELECT NAME from TOPOOL where ID = " + idTOPool;
		try {
			JdmoRowset rs = m_dmo.createRowset(strSQL);
			if (rs.next())
				return rs.getString(1);
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
		throw new BbmObjectNotFoundException();
	}

	public Collection getTOPoolsByIDs(Collection colIDs) throws BbmFinderException {
		StringBuffer strWhere = new StringBuffer("A.ID IN ");
		try {
			strWhere.append(m_dmo.createInClause(colIDs));
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
		return getObjects(new String(strWhere));
	}

	/**
	 * get TOPool ids given their names
	 *
	 * @param setTOPoolNames a hashset of organization names
	 * @return a HashMap of (TOPoolName, TOPoolID) pairs
	 */
	public HashMap getTOPoolIDsByNames(HashSet setTOPoolNames) throws BbmFinderException {
		return DAOUtil.getObjectIDsByNames(setTOPoolNames, "NAME", "TOPOOL", m_dmo);
	}

	/**
	 * Returns collection of TOPools which can be assignhed to an employee of given orgID.
	 *
	 * @param orgID          ID of Organization for which TOPools are requested.
	 * @param inclParentOrgs return TOPools created at parent orgs as well.
	 * @return Returns collection of TOPools
	 * @throws com.bluepumpkin.ejb.bbm.base.BbmFinderException
	 *          if error occurs while reading from DB
	 */
	public Collection getTOPoolsForOrg(ID orgID, boolean inclParentOrgs) throws Exception {
			ArrayList orgIDs = new ArrayList();
			if (inclParentOrgs) {
				orgIDs.addAll(BbmManagerFactory.getWorkResourceManager().getOrganizationsParentsByIDs(Collections.singletonList(orgID)));
			}
			if (orgIDs != null && !orgIDs.contains(orgID)) {
				orgIDs.add(orgID);
			}
			StringBuffer strWhere = new StringBuffer("A.ORGANIZATIONID IN ").append(m_dmo.createInClause(orgIDs));
			return getObjects(new String(strWhere));
	}


	public Collection getOrgIDsUnderTOPool(ID pTOPoolID) throws Exception {
		TOPool toPool = (TOPool) getObjectByID(pTOPoolID);
		HashSet orgIDs = new HashSet();
		orgIDs.add(toPool.getOrganizationId());
		orgIDs.addAll(BbmManagerFactory.getWorkResourceManager().getOrganizationsChildrenByIDs(orgIDs));
		return orgIDs;
	}

	public Collection getTOPoolsForOrgIDs(Collection orgIDs) throws Exception {
		HashSet orgIDSet = new HashSet();
		orgIDSet.addAll(orgIDs);
		orgIDSet.addAll(BbmManagerFactory.getWorkResourceManager().getOrganizationsParentsByIDs(orgIDSet));
		StringBuffer strWhere = new StringBuffer("A.ORGANIZATIONID IN ").append(m_dmo.createInClause(orgIDSet));
		return getObjects(strWhere.toString());
	}
	public Collection getTOPoolIDsWithWaitListScanEnabled() throws Exception {
		String strSql = "Select ID from TOPOOL where ISWAITLIST=1 and ISWAITLISTSCAN=1";
		ArrayList arrItems = new ArrayList();
		try {
			JdmoRowset rs = m_dmo.createRowset(strSql);
			long lCount = m_lRecords;
			while (rs.next() && (lCount != 0)) {
				arrItems.add(rs.getID(1));
				lCount--;
			}
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
		return arrItems;
	}
	/**
	 * Returns collection of all TOPools in the system. Couldn't be simpler.
	 * @return Returns collection of TOPools
	 */
	public Collection getAllTOPools() throws Exception {
		return getObjects("");
	}
	
}
