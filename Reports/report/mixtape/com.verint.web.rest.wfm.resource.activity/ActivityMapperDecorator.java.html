package com.verint.web.rest.wfm.resource.activity;

import com.bluepumpkin.ejb.bbm.activity.model.Activity;

import java.util.ArrayList;
import java.util.Collection;

public abstract class ActivityMapperDecorator implements ActivityMapper {
	private final ActivityMapper delegate;

	public ActivityMapperDecorator(ActivityMapper delegate) {
		this.delegate = delegate;
	}

	@Override
	public ActivityEntity activityToEntity(Activity source) {
		ActivityEntity activityEntity = delegate.activityToEntity(source);
		activityEntity.setTimeOff(source.isTimeoff());
		return activityEntity;
	}

	@Override
	public Activity entityToActivity(ActivityEntity source) {
		Activity activity = delegate.entityToActivity(source);
		activity.setTimeoff(source.getTimeOff());
		return activity;
	}

	@Override
	public Collection<ActivityEntity> activitiesToEntities(Collection<Activity> source) {
		if (source == null) {
			return null;
		}

		Collection<ActivityEntity> collection = new ArrayList<ActivityEntity>();
		for (Activity activity : source) {
			collection.add(activityToEntity(activity));
		}
		return collection;
	}
}
