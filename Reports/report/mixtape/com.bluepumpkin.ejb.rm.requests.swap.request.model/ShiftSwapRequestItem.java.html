import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectNode;
import com.bluepumpkin.ejb.rm.requests.swap.shiftitem.model.ShiftSwapItem;

public class ShiftSwapRequestItem extends ValueObjectNode{
	private static final FieldInfo m_fieldInfo = new ShiftSwapRequestItemFieldInfo();

	public ShiftSwapRequestItem() {
		super();
	}
    
    public ShiftSwapRequestItem(ShiftSwapItem item, int order) {
        this.setShiftSwapItem(item);
        this.setItemOrder(order);
    }
       
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

    //getters
	public ID getShiftSwapRequestId() {
		return getFieldValueID(ShiftSwapRequestItemFieldInfo.SHIFTSWAPREQUESTITEM_I_SHIFTSWAPREQUESTID);
	}

	public ID getShiftSwapItemId() {
		return this.getFieldValueID(ShiftSwapRequestItemFieldInfo.SHIFTSWAPREQUESTITEM_I_SHIFTSWAPITEMID);
	}

    public int getShiftSwapItemOrder() {
		return this.getFieldValueInt(ShiftSwapRequestItemFieldInfo.SHIFTSWAPREQUESTITEM_N_ITEMORDER);
	}

    //Get ShiftSwapItem ValueObject.
    public ShiftSwapItem getShiftSwapItem() {
        Collection items = this.getChildObjects(ShiftSwapRequestItemFieldInfo.ITEM_CHILD_TYPE);

        // todo: could return a null
        Iterator itr = items.iterator();
        if (itr.hasNext()) {
           return (ShiftSwapItem)itr.next();
        }

        // todo: are we allowed to return a null?
		return new ShiftSwapItem();
	}

	// setters

    //Set ShiftSwapItem ValueObject
    public void setShiftSwapItem(ShiftSwapItem item) {
        createChildObject(ShiftSwapRequestItemFieldInfo.ITEM_CHILD_TYPE, item);
	}

    public void setItemOrder(int order) {
		setFieldValue(ShiftSwapRequestItemFieldInfo.SHIFTSWAPREQUESTITEM_N_ITEMORDER, order);
	}

    public void updateShiftSwapItem(ShiftSwapItem item) {
        //this.clearCreatedChildObjectMap(ShiftSwapRequestItemFieldInfo.ITEM_CHILD_TYPE);
        //this.clearUpdatedChildObjectMap(ShiftSwapRequestItemFieldInfo.ITEM_CHILD_TYPE);
        updateChildObject(ShiftSwapRequestItemFieldInfo.ITEM_CHILD_TYPE, item);
	}
    
    public void moveChildrenFromPersistedToUpdated() {
        // move from persisted to updated list
        ShiftSwapItem ssItem = getShiftSwapItem();
        updateShiftSwapItem(ssItem);
        
        // clear the persisted list.
        clearChildObjectMap(ShiftSwapRequestItemFieldInfo.ITEM_CHILD_TYPE);
    }
}
