package com.bluepumpkin.ejb.bbm.timeseries.util;

import java.util.Date;

/**
 * The Class OpenHoursMissingException.
 */
public class OpenHoursMissingException extends RuntimeException {

	private Date m_date;
	
	public OpenHoursMissingException(String message, Date date) {
		super(message);
		m_date = date;
	}
	
	public Date getDateWhoseOpenHoursIsMissing() {
		return m_date;
	}

	private static final long serialVersionUID = 1L;

}
