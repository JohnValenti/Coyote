package com.bluepumpkin.ejb.bbm.timerecord.ejb;

import java.util.Collection;
import java.util.Date;
import java.util.HashMap;

import com.bluepumpkin.common.cache.Cache;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.config.ConfigCacheUtil;
import com.bluepumpkin.ejb.bbm.config.ConfigKey;
import com.bluepumpkin.ejb.bbm.config.ejb.DBConfigManager;
import com.bluepumpkin.ejb.bbm.timerecord.cache.TimeTrackingCache;
import com.bluepumpkin.ejb.core.base.SessionEJBBase;

/**

 * Description:  TimeRecordManager EJB implementation

 * Company:      Blue Pumpkin Software, inc

 * @version 1.0

public class TimeTrackingManagerEJB extends SessionEJBBase
{
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;

	private final static String clsName = TimeTrackingManagerEJB.class.getName();

	private static Category m_cat = Log.initCategory(clsName);
	// override the base class to provide the appropriate logging category
	protected Category getCategory() { return m_cat; }

	// cache reference to other EJBs
	private Cache m_EventCache;
	private DBConfigManager m_DBConfigManager;

	public void onEjbCreate() 
	{
		try 
		{
			m_DBConfigManager = BbmManagerFactory.getDBConfigManager();

			if( this.isCacheEnabled() )
				m_EventCache = TimeTrackingCache.getCache();
		} 
		catch (Exception e) 
		{
			handleException("onEjbCreate", e, false);
		}
	}

	private boolean isCacheEnabled() throws Exception
	{
		boolean retVal = ConfigCacheUtil.isCacheEnabled(m_DBConfigManager, ConfigKey.TIMETRACKING_CACHE_USAGE);
		if( !retVal && this.m_EventCache != null )
		{
			this.m_EventCache.clear();
			this.m_EventCache = null;
		}
		
		return retVal;
	}
	
	private Cache getCache() throws Exception
	{
		if( this.m_EventCache == null )
			m_EventCache = TimeTrackingCache.getCache();
		
		if( this.m_EventCache != null &&  this.m_EventCache.keySet().size() <= 1 )
		{
			TimeTrackingCache.reloadCache(m_EventCache);
		}

		return m_EventCache;
	}
	
	public HashMap getEventsForWorkResource(Collection workResourceID,
												Date dtStart,
												Date dtEnd ) throws BbmFinderException
	{
		try
		{
			this.methodStart( "getEventsForWorkResource", workResourceID, dtStart, dtEnd);
			if( this.isCacheEnabled() &&  TimeTrackingCache.fitsInCache(dtStart, dtEnd) )
			{
				return TimeTrackingCache.getObjects( this.getCache(), 
														workResourceID,
														dtStart,
														dtEnd);
			}
			else
			{
				TimeTrackingDAO dao = new TimeTrackingDAO();
				HashMap hm = dao.getEventsForWorkResource(workResourceID, dtStart, dtEnd);
				TimeTrackingCache.filterNoActivity(hm);
				
				return hm;
			}
		}
		catch( Exception ex )
		{
			this.handleException(ex);
			throw new BbmFinderException(ex);
		}
		finally
		{
			this.methodFinish();
		}
	}
		
	public void createEvents(Collection events ) throws BbmCreateException
	{
		if( events == null || events.size() <= 0 )
			return;
		
		try
		{
			this.methodStart( "createEvents", events);
			TimeTrackingDAO dao = new TimeTrackingDAO();
			dao.insertEvents(events);
			
			//update the cache
			if( this.isCacheEnabled() )
				TimeTrackingCache.insertInCache( this.getCache(), events);
		}
		catch( Exception ex )
		{
			this.handleException(ex);
			throw new BbmCreateException(ex);
		}
		finally
		{
			this.methodFinish();
		}
	}
	
}
