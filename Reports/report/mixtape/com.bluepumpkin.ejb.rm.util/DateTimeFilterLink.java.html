package com.bluepumpkin.ejb.rm.util;

import java.util.Date;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.BiddableSchedule;
import java.util.TimeZone;

public class DateTimeFilterLink extends RmFilterChain
{
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	final private static int ANY_DATE = 0;
	final private static int BEFORE_DATE = 1;
	final private static int AFTER_DATE = 2;
	final private static int BETWEEN_DATE = 3;
	final private static int ANY_TIME = 4;
	final private static int BEFORE_TIME = 5;
	final private static int AFTER_TIME = 6;
	final private static int BETWEEN_TIME = 7;
	
	private int timeType = UNDEFINED;
	private String startDate;
	private String endDate;
	private String m_DateCol;

	private int m_startTime = 0;
	private int m_endTime = 0;
	private int m_dateColPos = -1;
	private int m_durColPos = -1;
	
	private TimeZone m_tz = TimeZone.getTimeZone("PST");
	
	public void setTimeZone( TimeZone tz )
	{
		this.m_tz = tz;
	}
	
	public void setBefore( Date dt)
	{
		this.timeType = BEFORE_DATE;
		this.startDate = RmFilterUtil.dateToString(dt);
	}

	public void setBefore( int sec )
	{
		this.timeType = BEFORE_TIME;
		this.m_startTime = sec;
	}

	public void setAfter( Date dt)
	{
		this.timeType = AFTER_DATE;
		this.startDate = RmFilterUtil.dateToString(dt);
	}

	public void setAfter( int sec)
	{
		this.timeType = AFTER_TIME;
		this.m_endTime = sec;
	}

	public void setEqual( Date dt)
	{
		this.timeType = ANY_DATE;
		this.startDate = RmFilterUtil.dateToString(dt);
	}

	public void setEqual( int sec)
	{
		this.timeType = ANY_TIME;
		this.m_startTime = sec;
	}

	public void setBetween( Date before, Date after )
	{
		this.timeType = BETWEEN_DATE;
		this.startDate = RmFilterUtil.dateToString(before);
		this.endDate = RmFilterUtil.dateToString(after);
	}

	public void setBetween( int before, int after )
	{
		this.timeType = BETWEEN_TIME;
		this.m_startTime = before;
		this.m_endTime = after;
	}

	public void setDateColumnName( String name )
	{
		this.m_DateCol = name;
	}

	public void setDateColumnPostion( int pos )
	{
		this.m_dateColPos = pos;
	}
	
	public void setDurationColumnPostion( int pos )
	{
		this.m_durColPos = pos;
	}

	public boolean isValidDate( Date date )
	{
		 int sec = RmFilterUtil.dateToTime(date, 0, this.m_tz);
		 boolean found = false;
		 
		  switch( this.timeType )
		  {
		  	case BETWEEN_TIME:
		  		if( sec >= this.m_startTime && sec <= this.m_endTime )
		  			found = true;
		  		break;
		  		
		  	case AFTER_TIME:
		  		if( sec >= this.m_endTime )
		  			found = true;
		  		break;

		  	case BEFORE_TIME:
		  		if( sec <= this.m_startTime )
		  			found = true;
		  		break;

		  	case ANY_TIME:
		  		if( sec == this.m_startTime )
		  			found = true;
		  		break;
		  }
		  
		  return found;
	}

	protected boolean isScheduleValid(BiddableSchedule sched)
	{
		return true;
	}

}
