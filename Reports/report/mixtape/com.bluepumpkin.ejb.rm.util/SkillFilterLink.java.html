package com.bluepumpkin.ejb.rm.util;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.skill.ejb.SkillManager;
import com.bluepumpkin.ejb.bbm.skill.model.Skill;
import com.bluepumpkin.ejb.bbm.skill.model.SkillAssignment;
import com.bluepumpkin.ejb.bbm.workresource.model.Phantom;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.BiddableSchedule;
import com.verint.ejb.wfm.WfmManagerFactory;

public class SkillFilterLink  extends RmFilterChain
{
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private ArrayList skills = new ArrayList();
	private int skillsMatchType = UNDEFINED;
	
	public SkillFilterLink ()
	{
	}

	public SkillFilterLink ( int type, Collection idColl )
	{
		this.skills.addAll(idColl);
		this.skillsMatchType = type;
	}
	
	public void setMatchType( int type )
	{
		this.skillsMatchType = type;
	}

	public void addSkillID( ID id )
	{
		this.skills.add( id.toString() );
	}
	
	protected boolean isScheduleValid(BiddableSchedule vo)
	{
			 Phantom ph = vo.getOptMethods().getEmployee();
			 Date start = new Date();
		  
			  try
			  {
				  SkillManager mgr = WfmManagerFactory.getSkillManager();
				  Collection skColl = mgr.getSkills();
				  
				  java.util.HashMap  hmSkill = new java.util.HashMap();
				  
				  Iterator skIter = skColl.iterator();
				  while( skIter.hasNext() )
				  {
					  Skill sk = (Skill)skIter.next();
					  hmSkill.put( sk.getID(), sk.getID() );
				  }
				  
				 Collection sColl = mgr.getSkillAssignmentsForTemplate(ph.getEmployeeTemplateID(), 
						 													start, 
						 													null);
				 if( sColl == null )
					 return false;
				 
				 int matchCount = 0;
				 boolean templateHasSkillsNotInFilter = false;
				 
				 //Iterate through the template's skill assignments
				 Iterator sIter = sColl.iterator();
				 for( int i=0; sIter.hasNext(); ++i )
				 {
					 SkillAssignment sa = (SkillAssignment)sIter.next();
					 String idStr = ((ID)hmSkill.get( sa.getSkillID() )).toString();
					 
					 if( this.skills.contains(idStr) )
					 {
						 ++matchCount;
					 }
					 else
					 {
					 	templateHasSkillsNotInFilter = true;
					 }
				 }

				 boolean found = false;
				 
				 switch( this.skillsMatchType )
				 {
				 	case MATCH_EXACT:
				 		if( matchCount == this.skills.size() && 
				 				sColl.size() == this.skills.size() )
				 			found = true;
				 		break;

				 	case MATCH_AT_MOST:
				 		if( !templateHasSkillsNotInFilter )
				 			found = true;
				 		break;
				 		
				 	case MATCH_AT_LEAST:
				 		if( matchCount >= this.skills.size() )
				 			found = true;
				 		break;
				 }

				 return found;
			 }
			  catch( Exception ex )
			  {
				  ex.printStackTrace();
				  return false;
			  }
	  }

}
