package com.bluepumpkin.ejb.rm.util;

import java.text.SimpleDateFormat;
import java.util.Calendar;
import java.util.Collection;
import java.util.Iterator;
import java.util.Date;
import java.util.List;
import java.util.TimeZone;

import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.common.datatypes.StringsPair; 
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.rm.requests.commonrequestmanager.model.RequestFilter;

public class RmFilterUtil
{
	  static public String getIDList( Collection coll , boolean stringValue )
	  {
		  StringBuffer buf = new StringBuffer();
		  if( coll != null  && coll.size() > 0 )
		  {
			  Iterator iter = coll.iterator();
			  buf.append("(");
			  while( iter.hasNext() )
			  {
				  ID id = (ID)iter.next();
				  if( stringValue )
					  buf.append("'");
				  buf.append(id);
	
				  if( stringValue )
					  buf.append("'");
	
				  if( iter.hasNext() )
					  buf.append(",");
			  }
			  buf.append(")");
		  }
		  
		  return buf.toString();
	  }

	static public String dateToString( Date dt )
	{
			if( dt == null )
				return "";
			else
				return JdmoUtil.formatDBString(TimeZoneUtil.toTimestamp(dt));
	}

	public static String timeToString( int sec, String pattern )
	{
		Calendar cal = Calendar.getInstance( TimeZone.getTimeZone("GMT"));
		cal.set( Calendar.HOUR_OF_DAY, 0);
		cal.set( Calendar.MINUTE, 0);
		cal.set( Calendar.SECOND, sec);
		cal.set( Calendar.MILLISECOND, 0);
		
		SimpleDateFormat fmt = new SimpleDateFormat();
		fmt.setTimeZone(cal.getTimeZone());
		fmt.applyPattern(pattern);
		return fmt.format(cal.getTime());
	}
	
	public static int dateToTime( Date date, int durInSecs, TimeZone tz )
	{
		Calendar cal = Calendar.getInstance( TimeZone.getTimeZone("GMT"));
		cal.setTime(date);
		cal.setTimeZone(tz);
		
		cal.add( Calendar.SECOND, durInSecs);
		
		int retVal = cal.get(Calendar.HOUR_OF_DAY) * 3600;
		retVal += cal.get(Calendar.MINUTE) * 60;
		retVal += cal.get(Calendar.SECOND);
		
		return retVal;
	}
	
	public static StringsPair getStartAndEndDates(RequestFilter filter, String key) {
		StringsPair pair = null;
		
        List dateRange = (List) filter.getValueForKey(key);
        Date start = null;
        Date end = null;
	    if (dateRange != null){
	    	start = (Date) dateRange.get(0);
		    end  = (Date) dateRange.get(1);
	    }
	    if (start == null)
	    	start = new Date();
	    if (end == null)
	    	end = new Date();
	    String rangeStart = JdmoUtil.formatDBString(start);
	    String rangeEnd = JdmoUtil.formatDBString(end);	

	    pair = new StringsPair(rangeStart, rangeEnd);		
		return pair;
	}

	static String getQueryBetween( Object min, Object max, String colName)
	{
		StringBuffer sql = new StringBuffer();
		sql.append("('");
  		sql.append( min );
  		sql.append("' <= ");
  		sql.append( colName );
  		sql.append(" and '");
  		sql.append( max );
  		sql.append("' >= ");
  		sql.append( colName );
  		sql.append(")");

  		return sql.toString();
	}

	static String getQueryGreaterThan( Object max, String colName)
	{
		StringBuffer sql = new StringBuffer();
  		sql.append( colName );
  		sql.append(" > ");
  		sql.append("'");
  		sql.append( max );
  		sql.append("'");
		return sql.toString();
	}

	static String getQueryLessThan( Object min, String colName)
	{
		StringBuffer sql = new StringBuffer();
  		sql.append( colName );
  		sql.append(" < ");
  		sql.append("'");
  		sql.append( min );
  		sql.append("'");
		return sql.toString();
	}
  		
	static String getQueryEqual( Object obj, String colName)
	{
		StringBuffer sql = new StringBuffer();
  		sql.append("'");
  		sql.append( obj );
  		sql.append("' = ");
  		sql.append( colName );
		return sql.toString();
	}
}
