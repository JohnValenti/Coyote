package com.bluepumpkin.ejb.rm.util;

import java.math.BigInteger;

public class Base58 {

	private static final String ALPHABET = "123456789ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnopqrstuvwxyz";
	private static final BigInteger BASE = BigInteger.valueOf(58);
	private static final BigInteger TWO_COMPL_REF = BigInteger.ONE.shiftLeft(64);

	private Base58() {

	}

	public static String encode(long input) {

		//This is just repeated subtract of the reminder and division by 58
		//However java does not have unsigned long so this is not straightforward 
		//An alternative is to use Guava and use the unsigned long library in it, but that requires yet another library.

		BigInteger value = BigInteger.valueOf(input);
		if (value.signum() < 0) {
			value = value.add(TWO_COMPL_REF);
		}

		StringBuilder s = new StringBuilder();
		while (value.compareTo(BASE) >= 0) {
			BigInteger reminder = value.mod(BASE);
			s.insert(0, ALPHABET.charAt(reminder.intValue()));
			value = value.subtract(reminder).divide(BASE);
		}

		s.insert(0, ALPHABET.charAt(value.intValue()));
		return s.toString();
	}

}
