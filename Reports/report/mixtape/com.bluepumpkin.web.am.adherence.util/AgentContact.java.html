package com.bluepumpkin.web.am.adherence.util;

import java.util.Calendar;
import java.util.Date;
import java.util.ArrayList;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bpx.balance.model.Interaction;

public class AgentContact implements java.io.Serializable {
	private Date startTime;
	private Date endTime;
	private boolean isRecorded;
	private String callID;
	private String balanceAgentID;
	private String media;
	private boolean recordedByWFM;
	private boolean primaryRecording;
	private ID dataSourceID;
	private String contactName;
	private ID employeeID;
	private ArrayList m_overlappingContactIDs;
	private ArrayList m_overlappingRecordedContactIDs;

	/**
	 * @param startTime
	 * @param endTime
	 * @param isRecorded
	 * @param callID
	 * @param balanceServerID
	 * @param balanceAgentID
	 * @param media
	 * @param recordedByWFM
	 * @param primaryRecording
	 */
	public AgentContact(Date startTime, Date endTime, boolean isRecorded, String callID, String contactName,
			String balanceAgentID, String media, boolean recordedByWFM, boolean primaryRecording, ID dsID) {
		updateTimeRange(startTime, endTime);
		this.isRecorded = isRecorded;
		this.callID = callID;
		this.balanceAgentID = balanceAgentID;
		this.media = media;
		this.recordedByWFM = recordedByWFM;
		this.primaryRecording = primaryRecording;
		this.dataSourceID = dsID;
		this.contactName = contactName;
	}

	public AgentContact(Interaction interaction) {
		updateTimeRange(interaction.getStartTime(), interaction.getEndTime());
		this.isRecorded = interaction.isRecorded();
		this.callID = interaction.getCallID();
		this.balanceAgentID = interaction.getBalanceAgentID();
		this.media = interaction.getMedia();
		this.recordedByWFM = interaction.isRecordedByWFM();
		this.primaryRecording = interaction.isPrimaryRecording();
		this.dataSourceID = interaction.getDataSourceID();
		this.contactName = interaction.getContactName();
	}

	public AgentContact() {

	}

	private void updateTimeRange(Date start, Date end) {
		if (start.equals(end)) {
			// get current time in GMT
			Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
			if ((cal.getTime().getTime() - start.getTime()) < 2 * 60 * 60 * 1000) {
				if (cal.getTime().after(start)) {
					this.endTime = cal.getTime();
				} else {
					// add 30 seconds to make valid record
					cal.add(Calendar.SECOND, 30);
					this.endTime = cal.getTime();
				}
			}
		} else {
			this.startTime = start;
			this.endTime = end;
		}
	}

	public String getBalanceAgentID() {
		return balanceAgentID;
	}

	/**

	public void setCallID(String callID) {
		this.callID = callID;

	 */
	public Date getEndTime() {
		return endTime;

	 * @param endTime

	 */
	public void setEndTime(Date endTime) {
		if (startTime != null) {
			updateTimeRange(this.startTime, endTime);
		} else {
			this.endTime = endTime;
		}
	}

		this.isRecorded = isRecorded;

	}

	/**

	 */
	public boolean isPrimaryRecording() {
		return primaryRecording;
	}

	/**

		if (this.endTime != null) {
			updateTimeRange(startTime, this.endTime);
		} else {
			this.startTime = startTime;
		}
	}

		return dataSourceID;

	public ArrayList getOverlappingContactIDs() {
		return m_overlappingContactIDs == null ? new ArrayList() : m_overlappingContactIDs;
	}

	public void setOverlappingContactIDs(ArrayList overlappingContactIDs) {
		m_overlappingContactIDs = overlappingContactIDs;
	}

	public ArrayList getOverlappingRecordedContactIDs() {
		return m_overlappingRecordedContactIDs == null ? new ArrayList() : m_overlappingRecordedContactIDs;
	}

		m_overlappingRecordedContactIDs = overlappingRecordedContactIDs;
	}

	public String toString() {
		StringBuffer sb = new StringBuffer(200);
		sb.append("\n").append("Start: " + getStartTime());
		sb.append("\n").append("End: " + getEndTime());
		sb.append("\n").append("OverlappingCallIDs: " + getOverlappingContactIDs());
		sb.append("\n").append("OverlappingRecordedCallIDs: " + getOverlappingRecordedContactIDs());
		sb.append("\n").append("isRecorded: " + this.isRecorded);
		sb.append("\n").append("isRecordedByWFM: " + this.isRecordedByWFM());
		return sb.toString();
	}
