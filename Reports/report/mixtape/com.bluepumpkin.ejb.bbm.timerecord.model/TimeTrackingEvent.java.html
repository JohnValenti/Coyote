package com.bluepumpkin.ejb.bbm.timerecord.model;

import java.util.Date;
import java.io.Serializable;

import com.bluepumpkin.common.datatypes.ID;

public class TimeTrackingEvent implements Comparable, Serializable
{
	static final long serialVersionUID = 3116005448760481130L;

	Date m_startTime;
	Date m_endTime;
	ID m_activityID;
	ID m_empID;
	ID m_dsID;
	ID m_ID;
	
	public TimeTrackingEvent() 
	{
		this.setStartTime( new Date() );
	}

	public TimeTrackingEvent( TimeTrackingEvent cur ) 
	{
		this.setActivityID( cur.getActivityID() );
		this.setDataSourceID( cur.getDataSourceID() );
		this.setEmployeeID( cur.getEmployeeID() );
		this.setStartTime( cur.getStartTime() );
		this.setEndTime( cur.getEndTime() );
		this.setID( cur.getID() );
	}
	
	public Date getStartTime() 
	{
        return this.m_startTime;
	}
	
	public void setStartTime(Date date) 
	{
		this.m_startTime = date;
	}

	public Date getEndTime() 
	{
		return this.m_endTime;
	}

	public void setEndTime(Date date) 
	{
		this.m_endTime = date;
	}

	public ID getDataSourceID() 
	{
		return this.m_dsID;
	}
	
	public void setDataSourceID(ID id) 
	{
		this.m_dsID = id;
	}

	public ID getActivityID() 
	{
		return this.m_activityID;
	}

	public void setActivityID(ID id) 
	{
		this.m_activityID = id;
	}

	public ID getEmployeeID() 
	{
		return this.m_empID;
	}

	public void setEmployeeID(ID id) 
	{
		this.m_empID = id;
	}
	
	public ID getID() 
	{
		return this.m_ID;
	}

	public void setID(ID id) 
	{
		this.m_ID = id;
	}

	public int compareTo(Object obj) 
	{
		TimeTrackingEvent tr = (TimeTrackingEvent)obj;
		return this.getStartTime().compareTo( tr.getStartTime() );
	}
	
	public int hashCode()
	{
		return this.getStartTime().hashCode();
	}
	
	public boolean equals(Object obj)
	{
		return this.compareTo(obj) == 0;
	}
	
	public String toString() 
	{
		StringBuffer sb = new StringBuffer(200);
		sb.append("[ID=");
		sb.append( this.getID() );
		sb.append(" empID=");
		sb.append( this.getEmployeeID() );
		sb.append(" activityID=");
		sb.append( this.getActivityID() );
		sb.append(" start=");
		sb.append( this.getStartTime() );
		sb.append(" end=");
		sb.append( this.getEndTime() );
		sb.append(" dsID=");
		sb.append( this.getDataSourceID() );
		sb.append("]");
		
		return sb.toString();
	}
}
