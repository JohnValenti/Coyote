package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.Collection;
import java.util.Date;
import java.util.List;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.workresource.model.OrganizationHOO;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.RmUtil;

/**
 * This is a soft validation rule to validate all makeups of a Flex Time Request
 * must be inside the Hours Of Operation
 */
public class MakeupTimeComplyWithHOOValidationRule implements Validator {
	public static final String MAKEUP_TIME_COMPLY_HOO = "MAKEUP_TIME_COMPLY_HOO";

	public static final ID RULE_ID = new ID(-194069);
	private static final String CLASS_NAME = MakeupTimeComplyWithHOOValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		LOG.debug(RmUtil.dumpEnterMethod("validate", validatable));
		TORequest tor = (TORequest) validatable;
		ValidationResult result = null;

		// Find all makeup time ranges in this flex time
		List<TimeRange> makeupTimeRanges = getMakeupDurations(tor);
		Date firstDate = makeupTimeRanges.get(0).getStartDate();
		Date lastDate = makeupTimeRanges.get(makeupTimeRanges.size() - 1).getEndDate();
		TimeRange makeupRange = new TimeRange(firstDate, lastDate);

		Collection<OrganizationHOO> orgHOOs = getHOOAssignmentsDuringPeriod(tor, makeupRange);
		TimeZone orgTZ = getOrgTimeZone(tor);

		//check each makeup time range against the HOOs
		if (!isMakeupsWithinHOO(orgTZ, makeupTimeRanges, orgHOOs)) {
			//violate the rule
			result = ValidationUtil.setSoftValidationResult(tor, RmEjbBundleKey.FLEX_REQ_MAKE_UP_MUST_BE_WITHIN_HOO,
					CLASS_NAME);
		}

		LOG.debug(RmUtil.dumpExitMethod("validate", result));
		return result;
	}

	private boolean isMakeupsWithinHOO(TimeZone orgTZ, List<TimeRange> makeupDurations,
			Collection<OrganizationHOO> orgHOOs) {
		for (TimeRange makeup : makeupDurations) {
			TimeRange hooTimeRange = getHOOOfGivenDay(orgTZ, makeup.getStartDate(), orgHOOs);

			if (hooTimeRange == null || (!makeup.equals(hooTimeRange) && !makeup.isInside(hooTimeRange))) {
				return false;
			}

		}

		return true;
	}

	private TimeRange getHOOOfGivenDay(TimeZone tz, Date dateToGetHOO, Collection<OrganizationHOO> organizationHOOs) {
		for (OrganizationHOO hoo : organizationHOOs) {
			if (hoo.isActive() && RequestUtil.isDateBetween(dateToGetHOO, hoo.getStartTime(), hoo.getEndTime())) {
				//Opening hour
				Date open = hoo.getDayOpen(dateToGetHOO, tz);
				Date close = hoo.getDayClose(dateToGetHOO, tz);
				//closing hour
				if (open != null && close != null) {
					return new TimeRange(open, close);
				}
			}
		}

		return null;
	}

	protected TimeZone getOrgTimeZone(TORequest tor) throws Exception {
		return getFTValidationCache(tor).getOrg().getTimeZone();
	}

	protected ID getOrgID(TORequest tor) {
		return getFTValidationCache(tor).getEmployeeID();
	}

	protected Collection<OrganizationHOO> getHOOAssignmentsDuringPeriod(TORequest tor, TimeRange timeORange)
			throws Exception {
		return getFTValidationCache(tor).getHOOAssignmentsDuringPeriod(timeORange);
	}

	protected List<TimeRange> getMakeupDurations(TORequest tor) throws Exception {
		return getFTValidationCache(tor).getMakeupDurations();
	}

	private FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}
}
