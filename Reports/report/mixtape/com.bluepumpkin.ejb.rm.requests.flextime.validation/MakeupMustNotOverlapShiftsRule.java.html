package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.Collection;
import java.util.Collections;
import java.util.Date;

import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;

/**
 * Validate that the makeup time for a flex time request doesn't overlap
 * itself or employee's shift assignments.
 *
 */
public class MakeupMustNotOverlapShiftsRule implements Validator {

	private static final String CLASS_NAME = MakeupMustNotOverlapShiftsRule.class.getName();

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {

		ValidationResult result = null;

		TORequest toRequest = (TORequest) validatable;

		// Find makeup durations
		Collection<TimeRange> makeupDurations = getMakeupAndGapDurations(toRequest);

		// Check for conflicts between makeup blocks
		for (TimeRange makeupDuration : makeupDurations) {
			for (TimeRange makeupDuration2 : makeupDurations) {
				if (makeupDuration != makeupDuration2 && intersect(makeupDuration.getStartDate(), makeupDuration.getEndDate(), makeupDuration2.getStartDate(), makeupDuration2.getEndDate())) {
					result = ValidationUtil.setHardValidationResult(toRequest, RmEjbBundleKey.FLEX_REQ_MAKEUP_TIME_OVERLAP_MAKEUP_TIME,
							getEmployeeNameByID(toRequest), makeupDuration.getStartDate(), makeupDuration.getEndDate(),  CLASS_NAME);
					return result;
				}
			}
		}


		// Check for conflicts between makeup blocks and existing shifts
		@SuppressWarnings("unchecked")
		Date firstDate = Collections.min(makeupDurations).getStartDate();
		@SuppressWarnings("unchecked")
		Date lastDate = Collections.max(makeupDurations).getEndDate();
		Collection<ShiftAssignment> shifts = getPublishedShiftAssignments(toRequest, firstDate, lastDate);

		for (TimeRange makeupDuration : makeupDurations) {
			for (ShiftAssignment calEvent : shifts) {
				if (intersect(makeupDuration.getStartDate(), makeupDuration.getEndDate(), calEvent.getStartTime(), calEvent.getEndTime())) {
					result = ValidationUtil.setHardValidationResult(toRequest, RmEjbBundleKey.FLEX_REQ_MAKEUP_TIME_OVERLAP_SHIFT_ASSIGNMENT,
							getEmployeeNameByID(toRequest), makeupDuration.getStartDate(), makeupDuration.getEndDate(),  CLASS_NAME);
					return result;
				}
			}
		}

		return result;
	}

	private boolean intersect(Date start1, Date end1, Date start2, Date end2) {
		return end2.after(start1) && start2.before(end1);
	}

	private String getEmployeeNameByID(TORequest toRequest) throws Exception {
		return getFTValidationCache(toRequest).getEmployeeNameByID();
	}

	private Collection<TimeRange> getMakeupAndGapDurations(TORequest toRequest) throws Exception {
		return getFTValidationCache(toRequest).getMakeupAndGapDurations();
	}

	private Collection<ShiftAssignment> getPublishedShiftAssignments(TORequest toRequest, Date start, Date end) throws Exception {
		Collection<ShiftAssignment> events = getFTValidationCache(toRequest).getPublishedShiftAssignments(new TimeRange(start, end));
		return events;
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}
}
