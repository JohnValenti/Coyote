package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;

import com.bluepumpkin.ejb.rm.RmManagerFactory;

/**
 * Modified from AvoidCalendarEventsValidationRule.java for Flex Time.
 *
 * Rule in UI: Scheduled calendar event activities that time off must avoid ___
 * <p/>
 * Check each TOChoice does not overlap a scheduled calendar event (An event
 * in the CALENDAREVENTASSIGNMENT table with
 * event type == {@link com.bluepumpkin.ejb.bbm.activity.model.Event#EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT})
 * <p/>
 */
public class FlexTimeMustAvoidSchedCalEventsValidationRule implements Validator {
	public static final String FLEX_TIME_MUST_AVOID_SCHED_CALENDAR_EVENTS = "FLEX_TIME_MUST_AVOID_SCHED_CALENDAR_EVENTS";

	public static final ID RULE_ID = new ID(-194064);

	private static final ID ALL_ACTIVITIES_OPTION_ID = new ID(-2);
	private static final String CLASS_NAME = FlexTimeMustAvoidSchedCalEventsValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	/**
	 * Validate a time off request.
	 */
	@SuppressWarnings("unchecked")
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result;

		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod("validate", validatable));
		}

		TORequest tor = (TORequest) validatable;

		// Get calendar events
		Collection<CalendarEvent> calEvents = getFTValidationCache(tor).getCalEventsDuringPeriod(true, false, false);

		// Get choices to be validated
		List<TOChoice> toChoices = getTOChoicesToBeValidated(tor);

		// Loop through all choices and check that each not overlap a calendar event
		result = validateTOChoicesNotOverlapCalendarEvents(tor, calEvents, toChoices);

		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpExitMethod("validate", result));
		}

		return result;
	}

	private ValidationResult validateTOChoicesNotOverlapCalendarEvents(TORequest tor, Collection<CalendarEvent> calEvents, List<TOChoice> toChoices) throws Exception {
		ValidationResult result;
		result = null;
		for (TOChoice toChoice : toChoices) {
			ID orgId = getFTValidationCache(tor).getOrgIDForEmployeeDuringPeriod(tor.getEmployeeID(), toChoice.getStartDate(), toChoice.getEndDate());
			ValidationResult validation = setValidationResultForAvoidCalendarEvents(calEvents, tor, toChoice, getActivityIDsToAvoidForRule(orgId, RULE_ID));
			if (result == null && validation != null) {
				result = validation;
			}
		}
		return result;
	}

	private List<TOChoice> getTOChoicesToBeValidated(TORequest tor) {
		List<TOChoice> toChoices = new ArrayList<TOChoice>();
		if (tor.getTOChoiceForValidation() != null) {
			toChoices.add(tor.getTOChoiceForValidation());
		} else {
			toChoices.addAll(tor.getRequestChoiceList());
		}
		return toChoices;
	}

	private Collection<ID> getActivityIDsToAvoidForRule(ID orgId, ID ruleId) throws Exception {
		Collection<ID> activityColl;

		// Get activities to validate against from rule params
		Map<String, Integer> ruleParams = getValidationRuleManager().getValidationRuleParams(orgId, ruleId);
		if (ruleParams != null && ruleParams.isEmpty()) {
			activityColl = null;
		} else {
			activityColl = new ArrayList<ID>();
			for (Integer id : ruleParams.values()) {
				activityColl.add(new ID(id.intValue()));
			}
		}

		return activityColl;
	}


	private ValidationResult setValidationResultForAvoidCalendarEvents(Collection<CalendarEvent> calEvents,
			TORequest tor, TOChoice toChoice, Collection<ID> activityColl) throws Exception {
		ValidationResult result = null;

		for (CalendarEvent calEvent : calEvents) {
			//set validation result for the TOCHOICE only if the calendar event falls in the choice date range
			if (intersect(toChoice, calEvent)) {
				if (activityColl == null || activityColl.contains(calEvent.getActivityID()) || activityColl.contains(ALL_ACTIVITIES_OPTION_ID)) {
					String employeeName = getFTValidationCache(tor).getEmployeeNameByID();
					ValidationResult validation = ValidationUtil.setSoftValidationResult(toChoice, RmEjbBundleKey.FLEX_REQ_CONTAINED_IN_CALEVT, employeeName, toChoice.getStartDate(), toChoice.getEndDate(), CLASS_NAME);
					if (result == null && validation != null) {
						result = validation;
					}
				}
			}
		}
		return result;
	}

	protected boolean intersect(TOChoice toChoice, CalendarEvent calEvent) {
		return calEvent.getEndTime().after(toChoice.getStartDate()) && calEvent.getStartTime().before(toChoice.getEndDate());
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}

	protected ValidationRuleManager getValidationRuleManager() throws BbmCreateException {
		return RmManagerFactory.getInstance().getValidationRuleManager();
	}
}
