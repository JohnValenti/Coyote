/**
 * Validate the max allowed numbers of Flex Time requests per period (month and
 * quarter)
 *
 */
public class MaxNumberRequestsPerPeriodValidationRule implements Validator {

	public static final String MAX_FLEX_MAKEUP_PER_PERIOD = "MAX_FLEX_MAKEUP_PER_PERIOD";

	public static final String MAX_FLEX_MAKEUP_PER_PERIOD_UNIT = "MAX_FLEX_MAKEUP_PER_PERIOD_UNIT";

	public static final int PERIOD_MONTH_INDEX = 0;
	public static final int PERIOD_QUARTER_INDEX = 1;

	public static final ID RULE_ID = new ID(-194060);

	private static final String CLASS_NAME = MaxNumberRequestsPerPeriodValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);
	private String methodName = "validate";

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		LOG.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		TORequest toRequest = (TORequest) validatable;
		ValidationResult result = null;

		ID orgID = toRequest.getCache().getOrg().getID();
		@SuppressWarnings("rawtypes")
		Map ruleParams = getValidationRuleManager().getValidationRuleParams(orgID, RULE_ID);
		int maxPerPeriod = ((Integer) ruleParams.get(MAX_FLEX_MAKEUP_PER_PERIOD)).intValue();
		int periodUnit = ((Integer) ruleParams.get(MAX_FLEX_MAKEUP_PER_PERIOD_UNIT)).intValue();
		TimeZone orgTZ = getTimeZone(toRequest);

		TOChoice toChoice = (TOChoice) toRequest.getRequestChoiceList().get(0);
		TimeRange periodTimeRange = getTimeRangeOfConfiguredPeriodUnit(periodUnit, toChoice.getStartDate(),
				getFirstDateOfYear(orgTZ), orgTZ);

		RequestCountParam param = new RequestCountParam().addEmployeeID(toRequest.getEmployeeID())
				.addStatus(RequestAuditTrail.STATUS_APPROVED).setRequestType(Request.REQUESTTYPE_FLEXTIME)
				.setTimeRange(periodTimeRange);
		int actualCount = getNumberOfRequestPerPeriod(toRequest.getEmployeeID(), param);
		/**
		 * As the Validation rule only run on non-approved TO requests Using
		 * [actualCount >= maxPerPeriod] will trigger alert for non-approved TO
		 * requests when the user meets the limit
		 */
		if (actualCount >= maxPerPeriod) {
			result = ValidationUtil.setSoftValidationResult(toRequest, RmEjbBundleKey.FLEX_REQ_MAX_FLEX_TIME_PER_PERIOD,
					actualCount + 1, periodTimeRange.getStartDate(),
					new Date(periodTimeRange.getEndDate().getTime() - 1), maxPerPeriod, CLASS_NAME);
		}

		LOG.debug(RmUtil.dumpExitMethod(methodName, result));
		return result;
	}

	private int getNumberOfRequestPerPeriod(ID empID, RequestCountParam param) throws Exception {
		Map<ID, Integer> countOfEmployees = RequestUtil.getFlexTimeRequestMgr(null, null)
				.getNumberOfRequestPerPeriod(param);
		if (countOfEmployees == null || !countOfEmployees.containsKey(empID)) {
			return 0;
		}
		return countOfEmployees.get(empID);
	}

	private TimeRange getTimeRangeOfConfiguredPeriodUnit(int periodUnit, Date timeOffStartDate, Date firstDateOfYear,
			TimeZone orgTZ) {
		return DateUtil.getFiscalRange(firstDateOfYear, timeOffStartDate, periodUnit == PERIOD_MONTH_INDEX ? 1 : 3,
				orgTZ);
	}

	private Date getFirstDateOfYear(TimeZone orgTZ) throws BbmEJBCreateException, RemoteException {
		String strFirstDateOfYearFromEM = getDbConfigManager().getValue("EnterpriseSettings/CAL_STARTING_DATE_YEAR");
		if (strFirstDateOfYearFromEM == null || strFirstDateOfYearFromEM.isEmpty()) {
			// NO EM FISCAL YEAR, USE THE STANDARD CALENDAR
			return DateUtil.getFirstDateOfCurrentCalendarYear(orgTZ);
		}

		return DateUtil.iso8601SecPrecisionStringToDate(strFirstDateOfYearFromEM, orgTZ);
	}

	protected DBConfigManager getDbConfigManager() throws BbmEJBCreateException {
		return BbmManagerFactory.getDBConfigManager();
	}

	protected ValidationRuleManager getValidationRuleManager() throws BbmCreateException {
		return RmManagerFactory.getInstance().getValidationRuleManager();
	}

	protected TimeZone getTimeZone(TORequest tor) throws Exception {
		return getFTValidationCache(tor).getOrg().getTimeZone();
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}
}
