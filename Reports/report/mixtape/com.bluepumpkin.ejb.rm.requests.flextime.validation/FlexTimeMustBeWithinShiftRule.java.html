package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;

import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.RmUtil;

/**
 * This class is an implementation of the following Hard Validation rule: <b>The
 * Flex Time Off portion MUST be within shifts.</b> Only allow Flex Time Off
 * requests to be within a shift. Cannot request flex time off that goes outside
 * shift boundaries
 */
public class FlexTimeMustBeWithinShiftRule implements Validator {
	private static final String CLASS_NAME = FlexTimeMustBeWithinShiftRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		LOG.debug(RmUtil.dumpEnterMethod("validate", validatable));

		TORequest tor = (TORequest) validatable;

		TimeRange timeOffRange = new TimeRange(tor.getFirstTOChoice().getStartDate(),
				tor.getFirstTOChoice().getEndDate());
		Collection<ShiftAssignment> shiftEvents = getPublishedShiftAssignments(tor, timeOffRange);
		List<TimeRange> shiftTimeRanges = getShiftAssignmentTimeRanges(shiftEvents);

		// will violate the rule if the flex time off time doesn't fit in any
		// shift time range
		if (!isFlexTimeOffInsideAllShiftAssignments(timeOffRange, shiftTimeRanges)) {
			result = ValidationUtil.setHardValidationResult(tor, RmEjbBundleKey.FLEX_REQ_TIME_OFF_MUST_BE_WITHIN_SHIFTS,
					CLASS_NAME);
		}

		LOG.debug(RmUtil.dumpExitMethod("validate", result));
		return result;
	}

	private boolean isFlexTimeOffInsideAllShiftAssignments(TimeRange timeOffRange, List<TimeRange> shiftTimeRanges) {
		for (TimeRange shiftTimeRange : shiftTimeRanges) {

			// if flex time off time doesn't fit any shift assignment time
			// range, this flex violate this rule
			if (!timeOffRange.equals(shiftTimeRange) && !timeOffRange.isInside(shiftTimeRange)) {
				return false;
			}
		}
		
		return true;
	}

	private List<TimeRange> getShiftAssignmentTimeRanges(Collection<ShiftAssignment> shifts) {
		List<TimeRange> shiftTimeRanges = new ArrayList<TimeRange>();
		if (shifts == null || shifts.isEmpty()) {
			return shiftTimeRanges;
		}

		for (ShiftAssignment shift : shifts) {
			TimeRange shiftTimeRange = new TimeRange(shift.getStartTime(), shift.getEndTime());
			shiftTimeRanges.add(shiftTimeRange);
		}
		return shiftTimeRanges;
	}

	protected Collection<ShiftAssignment> getPublishedShiftAssignments(TORequest tor, TimeRange eventsDuration)
			throws Exception {
		return getFTValidationCache(tor).getPublishedShiftAssignments(eventsDuration);
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}
}
