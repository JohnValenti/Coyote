@WebService(serviceName="SchedExportSvc", targetNamespace = "http://verint.com/wfm/ws/sched/SchedExportSvc")
public class SchedExportSvc {
	private static Category m_cat = Log.initCategory(SchedExportSvc.class.getName());
	
	/**
	 * Return a flattened list of scheduled published events for one employee. Shift Assignments, 
	 * Shift Activities, and Calendar Events including TimeOff Events will be returned. All events which
	 * which start between startTime and endTime will be returned.
	 *  
	 * @param employeeID the agent's unique employee ID 
	 * @param startTime the start of the period for exporting events in GMT.
	 * @param endTime the end of the period for exporting events in GMT.
	 * @param sinceTime only return events modified since this time. Reserved for future use. 
	 * @return An ordered list of Events, or no Events if the agent has no published schedule 
	 * 			between startTime and endTime
	 */
	@WebMethod  
	public ArrayList<Event> getEmployeeSchedule(int employeeID,  Date startTime, Date endTime, Date sinceTime)
	throws SchedExportSvcException {
		ArrayList<Event> schedule = new ArrayList<Event>();
		ID idEmployee = new ID(employeeID);

		// verify date range
		String msg = "";
		if (startTime==null) 
			throw new SchedExportSvcException("startTime is null");
		else if (endTime == null )
			throw new SchedExportSvcException("endTime is null");
		else if (!startTime.before(endTime)) {
			throw new SchedExportSvcException("Starttime is after endtime");
		}
		
		try {
			ScheduleAccessManager schedAccessMgr = WfmManagerFactory.getScheduleAccessManager();
			if (!publishedAfter(schedAccessMgr, idEmployee, startTime, endTime, sinceTime))
				return schedule;
			
			Calendar startCal = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
			startCal.setTime(startTime);
			Calendar endCal = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
			endCal.setTime(endTime);
			
			// add & subtract 1 day from start and end dates
			startCal.add(Calendar.DATE, -1);
			endCal.add(Calendar.DATE, 1);
		
		
	        //-- get activities map
	        HashMap<ID, Activity> activitiesMap = getActivitiesMap(); 
	 
			Collection<SimpleEvent> flattenedSchedule = null;
			flattenedSchedule = getFlattenedSchedule(schedAccessMgr, idEmployee, startCal.getTime(), endCal.getTime());
			schedule = buildEventList(idEmployee.toInt(), flattenedSchedule, activitiesMap, startTime, endTime);
		} catch (Exception e) {
			m_cat.l7dError(null, e);
			throw new SchedExportSvcException(e);
		}

		return schedule;
	}

	// return true if any portion of the schedule has been published after sinceTime; otherwise return false
	private boolean publishedAfter(ScheduleAccessManager schedAccessMgr, ID idEmployee, Date startTime, Date endTime, Date sinceTime) 
	throws RemoteException, BbmFinderException {
		boolean wasPublishedAfter = false;
		
		if (sinceTime == null) return true;
		
		//--- load published periods for the employee, from, to
		ArrayList eIDs = new ArrayList(1);  
        eIDs.add(idEmployee);

		Collection ppTimeRanges = new ArrayList(1);
        Collection pubPeriodsForEmps = schedAccessMgr.getPublishedPeriods(eIDs, startTime, endTime);

		if (pubPeriodsForEmps!=null && !pubPeriodsForEmps.isEmpty()){
			for (Iterator it1= pubPeriodsForEmps.iterator(); it1.hasNext();) {
				Collection pubPeriods = (Collection) it1.next();
				if (pubPeriods == null) break;

				ArrayList ranges = new ArrayList();
				if (pubPeriods != null) {
					for (Iterator it2 = pubPeriods.iterator(); it2.hasNext();) 
		            {
						PublishingPeriod tp = (PublishingPeriod) it2.next();
		                if (tp.getPublishTime().after(sinceTime)) {
		                	wasPublishedAfter = true;
		                	break;
		                }
					}
				}
			}
		}
        
        return wasPublishedAfter;
	}

	private HashMap<ID, Activity> getActivitiesMap() throws BbmException, RemoteException {
		ActivityManager actMgr = WfmManagerFactory.getActivityManager(false);
		
		//Get all active and deleted activities
		ActivityFilter activityFilter = new ActivityFilter(ActivityFilter.FILTER_NONE);
		
		Collection activities = actMgr.findActivities(activityFilter);
		HashMap activityMap = new HashMap();
		for (Iterator i = activities.iterator(); i.hasNext();){
			Activity activity = (Activity)i.next();
			activityMap.put(activity.getID(), activity);
		}
		return activityMap;
	}
	
	// build the list of Events
	private ArrayList<Event> buildEventList(int employeeId, Collection<SimpleEvent> flattenedSchedule, HashMap<ID,Activity> activitiesMap,
											Date startTime, Date endTime) {
		ArrayList<Event> rtnList = new ArrayList<Event>();
		for (SimpleEvent sEvent: flattenedSchedule ) {
			// ensure that events overlap with start and end times
			if (sEvent.getStartTime().before(endTime) && sEvent.getEndTime().after(startTime)) {
				Event evt = new Event();
				
				evt.setStartTime(sEvent.getStartTime());
				evt.setEndTime(sEvent.getEndTime());
				evt.setIsPaid(sEvent.getPaid());
	
				ID idAct = sEvent.getActivityID();
				Activity activity = activitiesMap.get(idAct);
				if (activity != null) {
					evt.setActivityName(activity.getName());
					evt.setActivityColor(activity.getColor());
				}
				
				evt.setEmployeeId(employeeId);
				
				rtnList.add(evt);
			}
		}
		
		return rtnList;
	}
	
	private Collection getFlattenedSchedule(ScheduleAccessManager schedAccessMgr, ID idEmployee, Date startTime, Date endTime) 
	throws BbmEJBCreateException, BbmFinderException, RemoteException {
		int eventTypeMask = com.bluepumpkin.ejb.bbm.activity.model.Event.EVENT_TYPE_ALL_CALENDAR_EVENTS
						| com.bluepumpkin.ejb.bbm.activity.model.Event.EVENT_TYPE_SHIFT_ASSIGNMENT
						| com.bluepumpkin.ejb.bbm.activity.model.Event.EVENT_TYPE_SHIFT_EVENT_ASSIGNMENT;
        Collection scheduleEvents = schedAccessMgr.getPublishedEventsForWorkResourceByType(eventTypeMask,idEmployee,startTime,endTime);
		Collection scheduleShifts = selectOnlyShiftAssignments(scheduleEvents);
        
		Collection flatSchedule = EventUtils.convertEventsToTimelineForSingleEmployee(scheduleEvents);
		Collection mergedSchedule = mergeSameEvents(flatSchedule);
		//--- mark events covered by shift extensions as overtime		
		Collection markedSchedule = markOvertimeEvents(mergedSchedule, scheduleShifts);
		
		return markedSchedule;
	}
	
	private static Collection selectOnlyShiftAssignments(Collection scheduleEvents){
		ArrayList shifts = new ArrayList();
		if (scheduleEvents != null && scheduleEvents.size() > 0) {
			for (Iterator i = scheduleEvents.iterator(); i.hasNext();){
				com.bluepumpkin.ejb.bbm.activity.model.Event event = (com.bluepumpkin.ejb.bbm.activity.model.Event) i.next();
				if( event != null){
					if(event.getEventType() == com.bluepumpkin.ejb.bbm.activity.model.Event.EVENT_TYPE_SHIFT_ASSIGNMENT)
						shifts.add(event);
				}
			}
		}
		return shifts;
	}//selectOnlyShiftAssignments
	/**
	 * merge adjacent events on the timeline, which have the same properties
	 */
	private Collection mergeSameEvents(Collection events){
		ArrayList mergedEvents = new ArrayList();
		if (events == null) return mergedEvents;
		SimpleEvent curEvent = null;
		for (Iterator it = events.iterator(); it.hasNext();){
			SimpleEvent nextEvent = (SimpleEvent) it.next();
			if (curEvent == null) { //handle first event case
				if(nextEvent != null) {
					curEvent = nextEvent;
				}
			} else { //have cur, which is previous and got next
				if (nextEvent != null) {
					// QA54539 - Cal Events together show as one event in WFO
					//As per Jason, merging events are not required. But it is not user friendly
					// to show adjacent unavailability events as separate events. So changing it ignore
					// Any calendar event assignments and time offs
					if (curEvent.getActivityID().equals(nextEvent.getActivityID())
							&& curEvent.getEndTime().equals(nextEvent.getStartTime())
							&& curEvent.getDescription().equals(nextEvent.getDescription())
							&& (curEvent.getEventType()!= com.bluepumpkin.ejb.bbm.activity.model.Event.EVENT_TYPE_ADH_CALENDAR_EVENTS)
							)
					{ //next is the same
						curEvent.setEndTime(nextEvent.getEndTime());
					} else { //next is different
						mergedEvents.add(curEvent);
						curEvent = nextEvent;
					}
				}//else skip next as null
			}
		}
		if (curEvent != null)
			mergedEvents.add(curEvent);

		return mergedEvents;
	}//mergeSameEvents

	/**
	 * mark events within shift extensions as overtime
	 * @param scheduleData - collection of SimpleEvent objects in chronological order
	 * @param scheduleShifts - collection of SimpleEvent shift assignments in chronological order
	 * @return
	 */
	private Collection markOvertimeEvents(Collection scheduleData, Collection scheduleShifts){
		ArrayList overtimePeriods = (ArrayList)findOvertimePeriods(scheduleShifts);
		//if no overtime return original collection of flattened events
		if(overtimePeriods.size()<=0) return scheduleData;
		//else repackage scheduleData collection since may have to split some events
		ArrayList markedEvents = new ArrayList();
		//for each event in scheduleData process event
		for (Iterator it = scheduleData.iterator(); it.hasNext();){
			markEventOvertime(overtimePeriods,(SimpleEvent)it.next(),markedEvents);
		}
		//
		return markedEvents;
	}//markOvertimeEvents
	
	/**
	 * find shift assignments overitme periods
	 * @param scheduleShifts
	 * @return collection of overtime TimeRanges or empty
	 */
	private static Collection findOvertimePeriods(Collection scheduleShifts){
		ArrayList overtimeRanges = new ArrayList();
		if (scheduleShifts != null && scheduleShifts.size() > 0){
			SortEvents se = new SortEvents();
			Collections.sort((List)scheduleShifts,
					se.getEventSorter(SortEvents.SORT_BY_START,SupportNavigation.SORT_ASCENDING));
			for(Iterator it = scheduleShifts.iterator();it.hasNext();) {
				ShiftAssignment sa = (ShiftAssignment)it.next();
				if (sa != null){
					if (sa.getExtensionBefore() > 0){
						TimeRange ot = new TimeRange(sa.getStartTime(),sa.getExtensionBefore()*60000);
						overtimeRanges.add(ot);
					}
					if (sa.getExtensionAfter() > 0){
						TimeRange ot = new TimeRange(new Date(sa.getEndTime().getTime()- sa.getExtensionAfter()*60000),sa.getEndTime());
						overtimeRanges.add(ot);
					}
				}
			}
		}
		return overtimeRanges;
	}//findOvertimePeriods

	/**
	 * process one SimpleEvent marking overtime and adding to marked collection
	 */
	private static void markEventOvertime(ArrayList OTPeriods, SimpleEvent event, Collection markedEvents ){
		if (OTPeriods.size()<=0){
			markedEvents.add(event);
			return;
		}
		TimeRange overtimeRange = (TimeRange)OTPeriods.get(0);
		TimeRange eventRange = new TimeRange(event.getStartTime(),event.getEndTime());
		int overlapType = eventRange.getRelativeLocationTo(overtimeRange);
		if (overlapType == TimeRange.INTERVAL_IN || overlapType == TimeRange.INTERVAL_EQUALS || 
			event.getActivityID().compareTo(Activity.ACTIVITY_SHIFT_OVERTIME_GAP) == 0){	//consider Overtime GAP belongs to Overtime
			event.setIsOvertime(true);
			markedEvents.add(event);
		} else if (overlapType == TimeRange.INTERVAL_BEFORE){
			//keep the same overtime period in the future on the timeline
			markedEvents.add(event);
		} else if (overlapType == TimeRange.INTERVAL_AFTER) {
			//try next overtime period recursively
			markEventWithNextOvertime(OTPeriods, event, markedEvents );
		} else if (overlapType == TimeRange.INTERVAL_BEFORE_AND_IN
				||overlapType == TimeRange.INTERVAL_COVERS
				&& overtimeRange.getEndDate().getTime() == eventRange.getEndDate().getTime()){
			//split event (eventStart,overtimeStart)(overtimeStart,eventEnd) and mark the second overtime
			addSplitPartOfEvent(event, null, overtimeRange.getStartDate(), false, markedEvents);
			//second event is adjusted event;
			event.setStartTime(overtimeRange.getStartDate());
			event.setIsOvertime(true);
			markedEvents.add(event);
		} else if (overlapType == TimeRange.INTERVAL_AFTER_AND_IN
				||overlapType == TimeRange.INTERVAL_COVERS
				&& overtimeRange.getStartDate().getTime() == eventRange.getStartDate().getTime()){
			//split event (eventStart,overlapEnd) (overlapEnd,eventEnd), mark first overtime and add to marked
			addSplitPartOfEvent(event, null, overtimeRange.getEndDate(), true, markedEvents);
			//try next overtime recursively with second event which is adjusted event
			event.setStartTime(overtimeRange.getEndDate());
			event.setIsOvertime(false);
			markEventWithNextOvertime(OTPeriods, event, markedEvents );
		} else { //event covers overtime, and overtime starts after event start and ends before event end
			//	split event to three events (eventStart,overtimeStart),(overtimeStart,overtimeEnd),(overtimeEnd,eventEnd)
			//	mark second overtime, move two first to marked
			addSplitPartOfEvent(event, null, overtimeRange.getStartDate(), false, markedEvents);
			addSplitPartOfEvent(event, overtimeRange.getStartDate(), overtimeRange.getEndDate(), true, markedEvents);
			//try next overtime recursively with third event which is adjusted event
			event.setStartTime(overtimeRange.getEndDate());
			event.setIsOvertime(false);
			markEventWithNextOvertime(OTPeriods, event, markedEvents );
		}
		return;
	}//markEventOvertime

	/**
	 * process one SimpleEvent marking overtime and adding to marked collection
	 * 	trying with consecutive overtime periods
	 */
	private static void markEventWithNextOvertime(ArrayList OTPeriods, SimpleEvent event, Collection markedEvents ){
		if (OTPeriods.size()>0){
			OTPeriods.remove(0);
			markEventOvertime(OTPeriods, event, markedEvents);
		} else {//no overtime left - move second event to marked
			markedEvents.add(event);
		}
		return;
	}//markEventWithNextOvertime

	/**
	 * @param event
	 * @param startOrNull
	 * @param endOrNull
	 * @param bOvertime
	 */
	private static void addSplitPartOfEvent(SimpleEvent event, Date startOrNull, Date endOrNull, boolean bOvertime, Collection markedEvents){
		SimpleEvent newEvent = new SimpleEvent(event);
		if (startOrNull!=null) newEvent.setStartTime(startOrNull);
		if (endOrNull != null) newEvent.setEndTime(endOrNull);
		newEvent.setIsOvertime(bOvertime);
		markedEvents.add(newEvent);
	}//addSplitPartOfEvent
