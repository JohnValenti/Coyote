import java.util.Collection;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.schedulepref.model.EmployeeOTVTOPref;
import com.bluepumpkin.ejb.bbm.schedulepref.model.EmployeeOTVTOPrefFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class EmployeeOTVTOPrefDAO extends DAOBase {

    private static FieldInfo m_fieldInfo = new EmployeeOTVTOPrefFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }

    public EmployeeOTVTOPrefDAO() {
         super();
    }

    public EmployeeOTVTOPrefDAO(Jdmo dmo) {
         super(dmo);
    }

    protected ValueObjectBase createValueObject()
    {
        return (new EmployeeOTVTOPref());
    } 
    
    public EmployeeOTVTOPref getObjectByEmployeeId(String strWhere) 
    	throws BbmFinderException {
    	try {
    		Collection objects = super.getObjects(strWhere);
    		if (!objects.isEmpty()) 
    			return (EmployeeOTVTOPref)objects.iterator().next();
    		return null;    			
    	}
    	catch(BbmFinderException e)
    	{
    		throw e;
    	}
    }
    
}
