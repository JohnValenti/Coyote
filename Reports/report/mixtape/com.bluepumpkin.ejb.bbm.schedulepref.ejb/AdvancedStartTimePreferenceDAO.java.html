import java.util.Collection;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.schedulepref.model.AdvancedStartTimePreference;
import com.bluepumpkin.ejb.bbm.schedulepref.model.AdvancedStartTimePreferenceFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class AdvancedStartTimePreferenceDAO extends DAOBase {

    private static FieldInfo m_fieldInfo = new AdvancedStartTimePreferenceFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }

    public AdvancedStartTimePreferenceDAO() {
         super();
    }

    public AdvancedStartTimePreferenceDAO(Jdmo dmo) {
         super(dmo);
    }

    protected ValueObjectBase createValueObject()
    {
        return (new AdvancedStartTimePreference());
    } 
    
    public void deleteObjects(Collection colIDs) throws BbmRemoveException
    {
        try
        {
            StringBuffer strSQL = new StringBuffer(30);
            strSQL.append("delete ").append(this.getFieldInfo().getTableName());
            strSQL.append(" where SCHEDULEPREFERENCEID in ");
            strSQL.append(m_dmo.createInClause(colIDs));
            
            m_dmo.executeCommand(new String(strSQL));
        }
        catch(Exception e)
        {
            throw new BbmRemoveException(e);
        }        
    }
}
