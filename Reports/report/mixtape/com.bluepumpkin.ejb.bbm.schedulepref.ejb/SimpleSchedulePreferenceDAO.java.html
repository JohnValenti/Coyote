	/**
	 * get a given employee's schedule preference assignments
	 *
	 * @param idEmployee employee id
	 * @param dtStart the time period, null date means a open period
	 * @param dtEnd: the time period, null date means a open period
	 *
	 * @return a collection of SimpleSchedulePreference objects which hold the assignment information
	 */
		public Collection getSimpleSchedulePreferences(ID idEmployee, Date dtStart, Date dtEnd) throws BbmFinderException
		{
		try 
        {
            //First, load week boundary and day offset boundary for the employee's organization
            int nWeekBoundary = -1;
            int nDayOffset = -1;
            String query = "SELECT ORG.WEEKSTARTDAY, ORG.DAYBOUNDARYOFFSET FROM ORGANIZATION ORG,EMPLOYEEAM EMP,WORKRESOURCEORGANIZATION WRSORG WHERE WRSORG.WORKRESOURCEID = EMP.ID AND WRSORG.ORGANIZATIONID = ORG.ID AND EMP.ID = " + idEmployee;

            JdmoRowset r = m_dmo.createRowset( query );
            while(r.next()) 
            {
                nWeekBoundary = r.getInt("WEEKSTARTDAY");
                nDayOffset = r.getInt("DAYBOUNDARYOFFSET");

            }
            r.close();

            if (nWeekBoundary == -1 || nDayOffset == -1) 
            {
                return null;
            } 
            else 
            {
                nWeekBoundary = (nWeekBoundary-1)*1440;
            }


					//-------------------------------------------------
            // get data and create day off and start time prefs
					//-------------------------------------------------

            ArrayList aPrefs = new ArrayList();
			
            query = "SELECT PREFERENCETYPE,PREFERREDTIME,PREFERENCE FROM EMPTIMEPREFERENCE " +
			"WHERE EMPLOYEEID = " + idEmployee;

            r = m_dmo.createRowset( query );

			int[] anDayPreference = new int[7]; //Day off preference array
			int[][] aanTimePreferences =  new int[7][96]; //Start time preference array
			for (int i = 0; i < 7; i++) 
            {
						anDayPreference[i] = 999;
				for (int j = 0; j < 96; j++) 
                {
							aanTimePreferences[i][j] = 999;
						}
					}

			while(r.next()) 
            {
					int nPreferenceType = r.getInt("PREFERENCETYPE");
					int nPreferenceTime = r.getInt("PREFERREDTIME");
					int nPreference = r.getInt("PREFERENCE");
					
    			if (nPreferenceType == 0) 
                { 
                    //start time preference
                    nPreferenceTime += nDayOffset;
					nPreferenceTime /= 15;
                    int nDay = nPreferenceTime/96;
                    nDay %= 7; //in case adding day offset makes it wrap around to before the week starts
                    int nTime = nPreferenceTime%96;
                    aanTimePreferences[nDay][nTime] = nPreference;
					}
    			else 
                {
    			    //day off preference
					anDayPreference[nPreferenceTime] = nPreference;
					}
			}

					r.close();

            //-------------------------------------------------
            // get data and create basic prefs
            //-------------------------------------------------
            
			query = "SELECT PREFERREDSTART FROM EMPLOYEEAM WHERE ID = " + idEmployee;

			r = m_dmo.createRowset( query );

			int nGeneralPreffered = 0;

			while(r.next()) {
				nGeneralPreffered = r.getInt("PREFERREDSTART");
			}

			r.close();

			//load week boundary
			nWeekBoundary = -1;
			query = "SELECT ORG.WEEKSTARTDAY FROM ORGANIZATION ORG,EMPLOYEEAM EMP,WORKRESOURCEORGANIZATION WRSORG WHERE WRSORG.WORKRESOURCEID = EMP.ID AND WRSORG.ORGANIZATIONID = ORG.ID AND EMP.ID = " + idEmployee;

			r = m_dmo.createRowset( query );

			while(r.next()) {
				nWeekBoundary = r.getInt("WEEKSTARTDAY");
			}

			r.close();
			if (nWeekBoundary == -1) {
				return null;
			} else {
				nWeekBoundary = (nWeekBoundary-1)*1440;
			}
			
			SimpleSchedulePreference pPref = new SimpleSchedulePreference();
			pPref.setEmployeeID(idEmployee);
    		pPref.setPreferredStart((short)nGeneralPreffered); //In general, start: Early/Late/No Preference

            //Find the first preference for the day off and the first start time preference.
            //Lower values have a higher preference.
			int nMinPreferedDay = 999;
			int nMinPreferedStart = 999;
    		for (int i = 0; i < 7; i++) 
            {
				if (anDayPreference[i] < nMinPreferedDay)
					nMinPreferedDay = anDayPreference[i];
    			
                for (int j = 0; j < 96; j++) 
                {
					if (aanTimePreferences[i][j] < nMinPreferedStart)
						nMinPreferedStart = aanTimePreferences[i][j];
				}
			}

            //Set the value for radio button: "Is it more important for you to have your preferred day off or to start work at the time you prefer?"
			if (nMinPreferedDay < nMinPreferedStart)
				pPref.setPreferDayOff(SimpleSchedulePreference.PREFER_DAYOFF_OVER_STARTTIME);
			else if (nMinPreferedStart < nMinPreferedDay)
				pPref.setPreferDayOff(SimpleSchedulePreference.PREFER_STARTTIME_OVER_DAYOFF);
    		else 
                pPref.setPreferDayOff(SimpleSchedulePreference.PREFER_DAYOFF_EQUAL_STARTTIME);

            //Set start time preferences
    		for (int nLevel = 0; nLevel < 3; nLevel ++) 
            {
    			for (int nDay = 0; nDay < 7; nDay++) 
                {
						int nMinPreference = 999;
						int nStartTime = -1;
						int nEndTime = -1;
    				for (int j = 0; j < 96; j++) 
                    {
    					if (aanTimePreferences[nDay][j] < nMinPreference) 
                        {
								nMinPreference = aanTimePreferences[nDay][j];
								nStartTime = j;
								nEndTime = nStartTime;
    						while (nEndTime < 95 && aanTimePreferences[nDay][nEndTime + 1] == nMinPreference)	
                            {
									nEndTime++;
								}
							}
						}
    				if (nStartTime != -1) 
                    {
							int nOffsetDay = nDay + 1 + (nWeekBoundary/1440);
							if (nOffsetDay > 7)
								nOffsetDay -= 7;

						pPref.setStartTimePreference(nOffsetDay, nLevel,
																				 (nStartTime * 15 + (nWeekBoundary % 1440)) % 1440,
																				 (nEndTime * 15 + (nWeekBoundary % 1440)) % 1440);
							for (int i = nStartTime; i <= nEndTime; i++)
							{
								aanTimePreferences[nDay][i] = 999;
							}
						}
					}
				}

            //Set the day off preferences
			Integer[] aDays = new Integer[8];
    		for (int nLevel = 0; nLevel < 7; nLevel++) 
            {
				int nMinPreference = 999;
				int nPreferedDay = -1;
    			for (int nDay = 0; nDay < 7; nDay++) 
                {
    				if (anDayPreference[nDay] < nMinPreference) 
                    {
						nMinPreference = anDayPreference[nDay];
						nPreferedDay = nDay;
					}
				}
    			if (nPreferedDay != -1) 
                {
					aDays[nLevel] = new Integer(nPreferedDay);
					anDayPreference[nPreferedDay] = 999;
				}
			}
			ArrayList aDayPrefs = new ArrayList();
    		for (int i = 0; i < 8; i++) 
            {
				if (aDays[i] != null) {
					int nOffsetDay = aDays[i].intValue() + 1 + (nWeekBoundary/1440);
					if (nOffsetDay > 7)
						nOffsetDay -= 7;

					aDayPrefs.add(new Integer(nOffsetDay));
				}
			}
			pPref.setDayOffPreference(aDayPrefs);
			
			aPrefs.add(pPref);
			return aPrefs;
		} 
        catch(JdmoException e) 
        {
					e.printStackTrace();
					throw new BbmFinderException(e);
		} 
        finally 
        {
					m_dmo.cleanUp();
				}
		}
	
		/** create a simple schedule preference object */
		public ID createSimpleSchedulePreference(SimpleSchedulePreference objValue) throws BbmCreateException
		{
					if (objValue == null) return null;
					objValue.setOverridden(false);

					ID id = super.createNoOverlapAssignment(objValue, SimpleSchedulePreferenceFieldInfo.SIMPLESCHEDPREF_EMPLOYEEID,
							-1, "ISADVANCEDPREFERENCE = 0");

					// create an advanced schedule preference out of the simple one
					AdvancedSchedulePreferenceDAO advDAO = new AdvancedSchedulePreferenceDAO(m_dmo);
					try
					{
							AdvancedSchedulePreference advPref = AdvancedSchedulePreference.getInstance(objValue);
							advDAO.createAdvancedSchedulePreference(advPref, false);
					}
					catch(Exception e) {
							advDAO.cleanUp();
					}

					return id;
		}

		/**
		 *  check whether there is simple schedule preference in the given time
		 *  period
		 */
		public boolean hasSimpleSchedulePreference(ID idEmployee, Date dtStart, Date dtEnd)
				 throws BbmFinderException
		{
				try
				{
						StringBuffer strWhere = new StringBuffer(30);

						strWhere.append("select 1 from SCHEDULEPREFERENCE A where A.ISADVANCEDPREFERENCE = 0 ");
						strWhere.append(" and A.EMPLOYEEID = ").append(idEmployee);

						if (dtStart != null)
						{
								 String strStart = "'" + JdmoUtil.formatDBString(TimeZoneUtil.toTimestamp(dtStart)) + "'";
								 strWhere.append(" and (A.ENDTIME IS NULL OR A.ENDTIME > " + strStart + ")");
						}

						if (dtEnd != null)
						{
								String strEnd = "'" + JdmoUtil.formatDBString(TimeZoneUtil.toTimestamp(dtEnd)) + "'";
								strWhere.append(" and A.STARTTIME <= " + strEnd);
						}

						return DAOUtil.dataExist(strWhere.toString(), m_dmo);
				}
				catch(Exception e) {
						throw new BbmFinderException(e);
				}
		}
