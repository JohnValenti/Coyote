package com.verint.wfm.ws.entityImport.forecast.model;

import java.util.Calendar;
import java.util.Date;
import java.util.TimeZone;

import javax.xml.bind.annotation.XmlElement;
import javax.xml.bind.annotation.XmlType;
import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;

import com.verint.wfm.ws.ServiceApplicationException;
import com.verint.wfm.ws.util.CalendarUtil;
import com.verint.wfm.ws.util.JaxbDateSerializer;

@XmlType(name = "ProjectWorkTaskDTO", propOrder = { "startTime", "endTime","duration" })
public class ProjectWorkTaskDTO {
	private Date m_startTime;
	private Date m_endTime;
	private int m_duration;

	public ProjectWorkTaskDTO() {
		super();
	}

	public ProjectWorkTaskDTO(Date startTime, Date endTime, int duration) {
		super();
		m_startTime = startTime;
		m_endTime = endTime;
		this.m_duration = duration;
	}

	@XmlElement(type=Date.class)
	@XmlJavaTypeAdapter(value = JaxbDateSerializer.class)
	public Date getStartTime() {
		return m_startTime;
	}
	
	public void setStartTime(Date startTime) {
		m_startTime = startTime;
	}
	
	@XmlElement(type=Date.class)
	@XmlJavaTypeAdapter(value = JaxbDateSerializer.class)
	public Date getEndTime() {
		return m_endTime;
	}
	public void setEndTime(Date endTime) {
		m_endTime = endTime;
	}
	
	
	public Calendar startCalendarWithZone(TimeZone timeZone) throws ServiceApplicationException {
		return CalendarUtil.convertDateToCalendarWithZone(m_startTime, timeZone, false);
	}

	public Calendar endCalendarWithZone(TimeZone timeZone) throws ServiceApplicationException {
		return CalendarUtil.convertDateToCalendarWithZone(m_endTime, timeZone, false);
	}
	
	
	public int getDuration() {
		return m_duration;
	}
	public void setDuration(int duration) {
		this.m_duration = duration;
	}
}
