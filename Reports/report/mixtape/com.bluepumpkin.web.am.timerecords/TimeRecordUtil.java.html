package com.bluepumpkin.web.am.timerecords;

import java.util.ResourceBundle;

import com.bluepumpkin.web.am.l10n.AmWebBundleKey;
import com.witness.web.uif.system.RequestContext;
import com.bluepumpkin.ejb.bbm.timerecord.model.*;
import com.bluepumpkin.common.localization.Localizer;

/**

 * @version 1.0
 */
public class TimeRecordUtil {

	/**
	 * getCreatorStr()
	 */
	public static String getCreatorStr(RequestContext context, ResourceBundle bundle, int recordType, int timeSourceCode){
		Localizer localizer = context.getLocalizer();

		//get text for creator
		String creator = null;
		if (recordType == TimeRecord.TIMEINTERVAL){
			if (timeSourceCode == TimeInterval.SYSTEM){
				creator = localizer.i18n(bundle, AmWebBundleKey.SYSTEM);
			} else if (timeSourceCode == TimeInterval.MANUAL){
				creator = localizer.i18n(bundle, AmWebBundleKey.MANAGER);
			} else if (timeSourceCode == TimeInterval.SCHEDULE){
				creator = localizer.i18n(bundle, AmWebBundleKey.SCHEDULE);
			} else if (timeSourceCode == TimeInterval.HOLIDAY){
				creator = localizer.i18n(bundle, AmWebBundleKey.HOLIDAY);
			}
		} else {
			if (timeSourceCode == TimeEntrySourceCode.MULTI){
				creator = localizer.i18n(bundle, AmWebBundleKey.MULTIPLE);
			} else if (timeSourceCode == TimeEntrySourceCode.RAWPUNCH){
				creator = localizer.i18n(bundle, AmWebBundleKey.SYSTEM);
			} else if (timeSourceCode == TimeEntrySourceCode.MYTIME){
				creator = localizer.i18n(bundle, AmWebBundleKey.MY_TIME);
			} else if (timeSourceCode == TimeEntrySourceCode.MANUAL){
				creator = localizer.i18n(bundle, AmWebBundleKey.MANAGER);
			} else if (timeSourceCode == TimeEntrySourceCode.FROMSCHEDULE){
				creator = localizer.i18n(bundle, AmWebBundleKey.SCHEDULE);
			} else if (timeSourceCode == TimeEntrySourceCode.AUTO_CLOSE){
				creator = localizer.i18n(bundle, AmWebBundleKey.SYSTEM_AUTOCLOSE);
			} else if (timeSourceCode == TimeEntrySourceCode.AUTO_MERGE){
				creator = localizer.i18n(bundle, AmWebBundleKey.SYSTEM_AUTOMERGE);
			}
		}
		return creator==null?"":creator;
	}
}
