@Path("/topools/")
@Produces("application/json")

public class TimeOffPoolRestService extends AbstractWFOEntityService<TimeOffPoolEntity>{
    private static final Category Logger = Log.initCategory(TimeOffPoolRestService.class.getName());

    @Context
    private HttpServletRequest request;
    @Context
    private HttpServletResponse response;
    @Context
    private ServletContext sc;

    public ITimeOffPoolFacade timeOffPoolFacade;

    public RequestContext getRequestContext()
    {
        return new RequestContext(request, response, sc );
    }

    public TimeOffPoolRestService() {
    }

    public ITimeOffPoolFacade getTimeOffPoolFacade() {return timeOffPoolFacade;}
    public void setTimeOffPoolFacade(ITimeOffPoolFacade timeOffPoolFacade)
    {
        this.timeOffPoolFacade = timeOffPoolFacade;
    }

    @GET
    @Path("/get/{organizationId}")
    public TimeOffPoolList<TimeOffPoolEntity> getTimeOffPoolEntityWithParameters(@PathParam("organizationId") String organizationID) throws WFOException
    {
        try
        {
            ID orgId= getOrgId(organizationID);
            if(orgId == null)
            {
                throw new BusinessWFOException("organizationId is null",Response.Status.CONFLICT);
            }

            RequestContext context = getRequestContext();
            boolean bIncludeParentOrg = false;
            return (new TimeOffPoolList<TimeOffPoolEntity>(timeOffPoolFacade.getTOPoolByOrgId(context, orgId, bIncludeParentOrg)));
        }
        catch (WFOException ex) {
            Logger.error(ex.getLocalizedMessage(), ex);
            throw ex;
        }
        catch (Exception ex) {
            Logger.error(ex.getLocalizedMessage(), ex);
            throw new SystemWFOException(ex.getLocalizedMessage());
        }
    }
    @POST
    @Path("/topool")
    @Consumes("application/json")
    public TimeOffPoolEntity createTimeOffPoolEntity(TimeOffPoolParamEntity bean) throws WFOException
    {
        try
        {
            RequestContext context = getRequestContext();
            String toPoolName = bean.getToPoolName();
            String toPoolDesc = bean.getToPoolDescription();
            ID toPoolOrganizationId = new ID(bean.getToPoolOrgID());
            String strOrgId = toPoolOrganizationId.toString();
            ID orgId = StringUtil.isEmpty(strOrgId) ? null : new ID(strOrgId);
            if(orgId == null)
            {
                throw new BusinessWFOException("organizationId is null",Response.Status.CONFLICT);
            }

            if(toPoolName == null)
            {
                throw new BusinessWFOException("time of pool name is null",Response.Status.CONFLICT);
            }

            return timeOffPoolFacade.createTimeOffPool(context, toPoolName, toPoolDesc, toPoolOrganizationId);
        }
        catch(WFOException ex)
        {
            throw ex;
        }
        catch(Exception ex)
        {
            Logger.error(ex.getLocalizedMessage(), ex);
            throw new SystemWFOException(ex.getLocalizedMessage());
        }
    }
