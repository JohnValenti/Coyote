package com.verint.web.rm.rest.service;

import java.util.Date;

import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.Consumes;
import javax.ws.rs.GET;
import javax.ws.rs.POST;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.requests.custshift.model.CustShiftReq;
import com.bluepumpkin.web.rm.Log;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.security.WFOSecurity;
import com.verint.web.rest.service.AbstractWFOService;
import com.verint.web.rm.rest.entity.RequestProcessEntity;
import com.verint.web.rm.rest.entity.custshift.CustShiftDetailEntity;
import com.verint.web.rm.rest.entity.custshift.CustShiftRequestParametersEntity;
import com.verint.web.rm.rest.entity.custshift.CustShiftRequestResponseEntity;
import com.verint.web.rm.rest.entity.custshift.ShiftOTExtensionParametersEntity;
import com.verint.web.rm.rest.facade.custshift.ICustShiftFacade;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.ejb.core.licensing.LicenseKeys;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.ejb.core.security.model.User;
import com.witness.web.uif.system.RequestContext;

@Path("/custshift/")
@Produces("application/json")
@WFOSecurity(licenses = { LicenseKeys.APP_RM_CS })
public class CustShiftRestService extends AbstractWFOService {
	private static final Category LOG = Log.initCategory(CustShiftRestService.class.getName());

	public static final String RIBBON_END_DATE_FN = "ribbonEndDate";

	public static final String EXTENSION_AFTER_DESC = "Extension After";
	public static final String EXTENSION_AFTER_ID_DESC = EXTENSION_AFTER_DESC + " ID";
	public static final String EXTENSION_AFTER_ACTIVITY_ID_DESC = EXTENSION_AFTER_DESC + " ActivityID";
	public static final String EXTENSION_AFTER_GAP_DESC = EXTENSION_AFTER_DESC + " Gap";

	public static final String EXTENSION_BEFORE_DESC = "Extension Before";
	public static final String EXTENSION_BEFORE_ID_DESC = EXTENSION_BEFORE_DESC + " ID";
	public static final String EXTENSION_BEFORE_ACTIVITY_ID_DESC = EXTENSION_BEFORE_DESC + " ActivityID";
	public static final String EXTENSION_BEFORE_GAP_DESC = EXTENSION_BEFORE_DESC + " Gap";

	@Context
	private HttpServletRequest request;
	@Context
	private HttpServletResponse response;
	@Context
	private ServletContext sc;

	private ICustShiftFacade custShiftFacade;

	public CustShiftRestService() { // NOSONAR
	}

	public RequestContext getRequestContext() {
		return new RequestContext(request, response, sc);
	}

	public ICustShiftFacade getCustShiftFacade() {
		return custShiftFacade;
	}

	public void setCustShiftFacade(ICustShiftFacade custShiftFacade) {
		this.custShiftFacade = custShiftFacade;
	}

	// ================================================================================
	// Custom Shift Request API
	// ================================================================================

	/**
	 * Creates a new custom shift request.
	 */
	@POST
	@Path("/createRequest")
	@Consumes("application/json")
	public CustShiftRequestResponseEntity createRequest(CustShiftRequestParametersEntity requestParams) throws WFOException {
		try {
			RequestContext context = getRequestContext();
			Localizer localizer = context.getLocalizer();
			assertIsAuthorizedForRequest(context);

			if (requestParams == null) {
				throw RmUtils.createInvalidParameterException(localizer, "requestParams");
			}

			CustShiftReq custShiftRequest = getRequest(context, requestParams);

			String comment = requestParams.getComment();
			if (comment != null) {
				comment = comment.trim();
			}

			return custShiftFacade.createRequest(context, custShiftRequest, comment);
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}

	/**
	 * Creates a new custom shift request.
	 */
	@POST
	@Path("/editRequest")
	@Consumes("application/json")
	public CustShiftRequestResponseEntity editRequest(CustShiftRequestParametersEntity requestParams) throws WFOException {
		try {
			RequestContext context = getRequestContext();
			Localizer localizer = context.getLocalizer();
			assertIsAuthorizedForRequest(context);

			if (requestParams == null) {
				throw RmUtils.createInvalidParameterException(localizer, "requestParams");
			}

			CustShiftReq custShiftRequest = getRequest(context, requestParams);
			custShiftRequest.setID(RmUtils.getID(localizer, requestParams.getId(), RmUtils.ID_FN));

			String comment = requestParams.getComment();
			if (comment != null) {
				comment = comment.trim();
			}

			return custShiftFacade.editRequest(context, custShiftRequest, comment);
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}

	/**
	 * Gets the custom shift data for the current employee.
	 */
	@GET
	@Path("/getDetailData")
	public CustShiftDetailEntity getDetailData() throws WFOException {
		try {
			RequestContext context = getRequestContext();
			User user = context.getUser();
			assertIsAuthorizedForRequest(context);

			Date date = RmUtils.getDate(context, request, RmUtils.DATE_FN);
			Date ribbonEndDate = RmUtils.getDateNullable(request, RIBBON_END_DATE_FN);

			return custShiftFacade.getDetailData(context, user.getEmployeeID(), date, ribbonEndDate);
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}

	/**
	 * Get the request by the specified id.
	 */
	@GET
	@Path("/get")
	@Consumes("application/json")
	public CustShiftRequestResponseEntity getRequest() throws WFOException {
		try {
			RequestContext context = getRequestContext();
			Localizer localizer = context.getLocalizer();

			ID requestID = RmUtils.getID(context, RmUtils.ID_FN);
			RmUtils.assertNonZeroParameter(localizer, requestID.toInt(), RmUtils.ID_FN);

			return custShiftFacade.getRequest(context, requestID);
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}

	/**
	 * Performs an action on an existing request.
	 */
	@POST
	@Path("/processRequest")
	@Consumes("application/json")
	public CustShiftRequestResponseEntity processRequest(RequestProcessEntity requestParams) throws WFOException {
		try {
			RequestContext context = getRequestContext();
			Localizer localizer = context.getLocalizer();

			ID requestID = new ID(requestParams.getId());
			RmUtils.assertNonZeroParameter(localizer, requestID.toInt(), RmUtils.ID_FN);

			int actionCode = requestParams.getActionCode();
			String comment = requestParams.getComment();
			if (comment != null) {
				comment = comment.trim();
			}

			return custShiftFacade.processRequest(context, requestID, actionCode, comment);
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}

	// ================================================================================
	// IMPLEMENTATION
	// ================================================================================

	private void assertIsAuthorizedForRequest(RequestContext context)
			throws BusinessWFOException {
		User user = context.getUser();
		if (user == null || user.getEmployeeID() == null) {
			RmUtils.throwUnauthorizedRequest(context);
		}
		RmUtils.assertIsAuthorized(context, PrivilegeKeys.CS_MODIFYPERSONALREQUESTS_ID);
	}

	private CustShiftReq getRequest(RequestContext context, CustShiftRequestParametersEntity requestParams)
			throws BusinessWFOException {
		CustShiftReq req = new CustShiftReq(CustShiftReq.DL_BASIC);
		Localizer localizer = context.getLocalizer();

		req.setEmployeeID(context.getUser().getEmployeeID());

		Date startTime = RmUtils.getDate(context, requestParams.getStartTime(), RmUtils.START_TIME_FN);
		if (startTime == null) {
			throw RmUtils.createInvalidParameterException(localizer, RmUtils.START_TIME_FN);
		}
		req.setStartTime(startTime);

		if (requestParams.getShiftID() != null) {
			int shiftID = requestParams.getShiftID();
			RmUtils.assertNonZeroParameter(localizer, shiftID, RmUtils.SHIFT_ID_FN);
			req.setShiftID(new ID(shiftID));
		}

		// activityID is not required for certain request types so only enforce a non-zero value if it was specified
		// the logic later will automatically change this value based on the selected shift or validate it
		if (requestParams.getActivityID() != null) {
			int activityID = requestParams.getActivityID();
			RmUtils.assertNonZeroParameter(localizer, activityID, RmUtils.ACTIVITY_ID_FN);
			req.setActivityID(new ID(activityID));
		}

		req.setIsOT(requestParams.isOvertime());

		populateExtensionAfter(req, context, requestParams.getExtensionAfter());
		populateExtensionBefore(req, context, requestParams.getExtensionBefore());

		return req;
	}

	private void populateExtensionAfter(CustShiftReq req, RequestContext context, ShiftOTExtensionParametersEntity entity)
			throws BusinessWFOException {
		if (entity == null) {
			return;
		}
		Localizer localizer = context.getLocalizer();

		ID typeID = RmUtils.getID(localizer, entity.getId(), EXTENSION_AFTER_ID_DESC);
		RmUtils.assertNonZeroParameter(localizer, typeID.toInt(), EXTENSION_AFTER_ID_DESC);
		req.setExtAfterID(typeID);

		int activityID = entity.getActivityID();
		if (activityID != 0) {
			req.setExtAfterActivityID(new ID(activityID));
		}

		req.setExtAfterGap(entity.getGap());

		req.setExtAfterIsOT(entity.isOvertime());
	}

	private void populateExtensionBefore(CustShiftReq req, RequestContext context, ShiftOTExtensionParametersEntity entity)
			throws BusinessWFOException {
		if (entity == null) {
			return;
		}
		Localizer localizer = context.getLocalizer();

		ID typeID = RmUtils.getID(localizer, entity.getId(), EXTENSION_BEFORE_ID_DESC);
		RmUtils.assertNonZeroParameter(localizer, typeID.toInt(), EXTENSION_BEFORE_ID_DESC);
		req.setExtBeforeID(typeID);

		int activityID = entity.getActivityID();
		if (activityID != 0) {
			req.setExtBeforeActivityID(new ID(activityID));
		}

		req.setExtBeforeGap(entity.getGap());

		req.setExtBeforeIsOT(entity.isOvertime());
	}
}
