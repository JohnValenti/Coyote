package com.verint.web.rm.rest.service;

import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.rm.Log;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.security.WFOSecurity;
import com.verint.web.rest.service.AbstractWFOService;
import com.verint.web.rm.rest.entity.settings.OrganizationSettingsEntity;
import com.verint.web.rm.rest.facade.IOrganizationSettingsFacade;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.web.uif.system.RequestContext;

/**
 * REST service for RM organization settings. This is being filled in as features are necessary.
 */
@Path("/settings/")
@Produces("application/json")
@WFOSecurity(licenses = {})
public class OrganizationSettingsRestService extends AbstractWFOService {
	private static final Category LOG = Log.initCategory(OrganizationSettingsRestService.class.getName());

	@Context
	private HttpServletRequest request;
	@Context
	private HttpServletResponse response;
	@Context
	private ServletContext sc;

	private IOrganizationSettingsFacade organizationSettingsFacade;

	public OrganizationSettingsRestService() {
	}

	public RequestContext getRequestContext() {
		return new RequestContext(request, response, sc);
	}

	public IOrganizationSettingsFacade getOrganizationSettingsFacade() {
		return organizationSettingsFacade;
	}

	public void setOrganizationSettingsFacade(IOrganizationSettingsFacade organizationSettingsFacade) {
		this.organizationSettingsFacade = organizationSettingsFacade;
	}

	/**
	 * Gets the employee's RM organization settings.
	 *
	 * Example:
	 * http://localhost:7001/wfo/rest/rm-api/settings/get
	 */
	@GET
	@Path("/get")
	public OrganizationSettingsEntity getSettings() throws WFOException {
		try {
			RequestContext context = getRequestContext();

			return organizationSettingsFacade.getSettingsForEmployee(context, context.getUser().getEmployeeID());
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}
}
