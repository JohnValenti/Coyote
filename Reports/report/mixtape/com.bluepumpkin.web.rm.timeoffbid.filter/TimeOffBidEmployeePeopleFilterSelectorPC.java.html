package com.bluepumpkin.web.rm.timeoffbid.filter;

/**
 * Use to display the People Filter Selector in the Title of a Page Manager (ContainerPC)
 * 
 */
public class TimeOffBidEmployeePeopleFilterSelectorPC extends PeopleFilterSelectorPC {
	private static final Category log = Log.initCategory(TimeOffBidEmployeePeopleFilterSelectorPC.class.getName());

	private final ResourceBundle bbm_Bundle;
	private boolean displayInline = true;
	private boolean isAppShell = false;
	private final static String default_NONE = "DEFAULT_NONE";
	private final static String default_ALL = "DEFAULT_ALL";

	public TimeOffBidEmployeePeopleFilterSelectorPC(RequestContext context) {
		super(context);

		bbm_Bundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);
		setIsFullDisplayMode(false);
		setIsRefreshOptionEanbled(false);
		getListbox().setIsRefreshOnChange(true);

		isAppShell = NavigationManager.getInstance().isAppShell();
	}

	/**
	 * Mode to determine if the Selector is displayed inline without other table formating.
	 * 
	 * @param displayInline true to display without formatting
	 */
	public void setDisplayInline(boolean displayInline) {
		this.displayInline = displayInline;
	}

	/**
	 * Return the state of the inline display mode.
	 * 
	 * @return true to display without formatting
	 */
	public boolean getDisplayInline() {
		return displayInline;
	}

	/**
	 * Returns the the HMTL to display this component. It will call its superclass if the inline display is set to false.
	 * 
	 * @return HTML to display this component
	 */
	@Override
	public String getHtmlDisplay() {
		// initialize the filters in the listbox
		m_listbox.setOptions(getFilters());
		StringBuilder buff = new StringBuilder(1024);

		if (displayInline) {
			buff.append("<script>var FILTER_ENABLED_" + getName() + " = '" + (isEnabled() ? "1" : "0") + "';</script>");
			if (useWrapper()) {
				buff.append("\n<div style=\"" + m_wrapperStyle + "\">");
			}

			if (m_isFullDisplayMode && m_displayLabels) {
				HtmlTableCell cellLabel = createLabelCell(getHeader());
				cellLabel.setAttribute("width", "50");
				buff.append(cellLabel);
			}

			if (isAppShell && isNewUIFilter()) {
				setUseWrapper(false);

				HtmlTableCell cell = new HtmlTableCell();
				cell.setAttribute("filterName", getName());
				cell.setAttribute("class", "newuifilter");
				cell.setAttribute("colspan", "2");
				buff.append(cell);

				HtmlTableCell hcell = new HtmlTableCell();
				hcell.setAttribute("id", m_listbox.getName() + "Wrapper");
				hcell.setStyle("display:none;");
				m_listbox.setGenerateWrapper(false);
				hcell.setInnerHtml(m_listbox);
				buff.append(hcell);
			}

			buff.append(HtmlUtil.makeHiddenInput(getName(), getSelectedFilterName() != null ? getSelectedFilterName() : ""));
			if (useWrapper()) {
				buff.append("\n</div>");
			}
		} else {
			return super.getHtmlDisplay();
		}
		return buff.toString();
	}

	/**
	 * Return Selected Filter ID
	 */
	@Override
	public ID getSelectedFilter() {
		ID filterID = null;

		String filterName = getSelectedFilterName();
		if (!StringUtil.isEmpty(filterName) && !EmployeeModelHandler.isAllFilter(filterName)) {
			Filter filter = EmployeeModelHandler.getFilterByName(m_context, filterName);
			if (filter != null) {
				filterID = filter.getFilterID();
			} else {
				// check if filterName is a numeric ID
				try {
					int filterIDVal = Integer.valueOf(filterName).intValue();
					filterID = new ID(filterIDVal);
				} catch (Exception e) {
					log.warn("Filer Name is null " + e.getLocalizedMessage());
				}
			}
		} else if (EmployeeModelHandler.isAllFilter(filterName)) {
			if (filterName != null && filterName.equals(default_NONE)) {
				filterID = new ID(PageKeys.DEFAULT_NONE);
			} else if (filterName != null && filterName.equals(default_ALL)) {
				filterID = new ID(PageKeys.DEFAULT_ALL);
			}
		}
		return filterID;
	}
}
