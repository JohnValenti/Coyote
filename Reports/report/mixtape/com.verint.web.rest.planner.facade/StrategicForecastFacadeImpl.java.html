package com.verint.web.rest.planner.facade;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.fs.Log;
import com.verint.ejb.bbm.forecast.ejb.ForecastTimeSeriesManager;
import com.verint.ejb.bbm.forecast.model.StrategicForecast;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.planner.entity.StrategicForecastEntity;
import com.verint.web.rest.exception.checked.SystemWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.util.FormatUtils;
import com.witness.web.uif.system.RequestContext;

import javax.ws.rs.core.Response;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.List;

public class StrategicForecastFacadeImpl implements IStrategicForecastFacade {
	private static final Category logger = Log.initCategory(StrategicForecastFacadeImpl.class.getName());
	private ForecastTimeSeriesManager forecastTimeSeriesManager;


	public StrategicForecastFacadeImpl() {
	}

	public void deleteStrategicForecasts(RequestContext context, ID queueID, Date startTime, Date endTime) throws WFOException {
		try {
			boolean isWhatIf = context.isInWhatIfMode();
			forecastTimeSeriesManager = WfmManagerFactory.getForecastTimeSeriesManager(isWhatIf);
			forecastTimeSeriesManager.deleteStrategicForecast(queueID, startTime, endTime);
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	public void createStrategicForecasts(RequestContext context, List<StrategicForecastEntity> strategicForecastEntities) throws WFOException {
		try {
			if (strategicForecastEntities == null || strategicForecastEntities.isEmpty()) {
				throw new BusinessWFOException("Strategic Forecast data list is empty", Response.Status.CONFLICT);
			}
			boolean isWhatIf = context.isInWhatIfMode();
			forecastTimeSeriesManager = WfmManagerFactory.getForecastTimeSeriesManager(isWhatIf);
			Collection<StrategicForecast> strategicForecasts = getStrategicForecastCollection(strategicForecastEntities);
			forecastTimeSeriesManager.createStrategicForecast(strategicForecasts);
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	private List<StrategicForecast> getStrategicForecastCollection(Collection<StrategicForecastEntity> strategicForecastEntities) {
		ArrayList<StrategicForecast> strategicForecastList = new ArrayList<StrategicForecast>();
		for (StrategicForecastEntity sfEntity : strategicForecastEntities) {
			ID queueID = new ID(sfEntity.getQueueID());
			StrategicForecast sfData = new StrategicForecast();
			Date startTime = FormatUtils.iso8601SecPrecisionStringToDate(sfEntity.getTime());
			sfData.setQueueID(queueID);
			sfData.setCallVolume(sfEntity.getCv());
			sfData.setAHT(sfEntity.getAht());
			sfData.setTime(startTime);
			strategicForecastList.add(sfData);
		}
		return strategicForecastList;
	}
}
