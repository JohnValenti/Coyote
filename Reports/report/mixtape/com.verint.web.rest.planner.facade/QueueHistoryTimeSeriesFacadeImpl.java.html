package com.verint.web.rest.planner.facade;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;

import com.bluepumpkin.ejb.bbm.timeseries.ejb.TimeSeriesManager;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.web.fs.Log;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.verint.web.rest.planner.entity.QueueHistoryTimeSeriesEntity;
import com.verint.web.rest.exception.checked.SystemWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.witness.web.uif.system.RequestContext;

import java.util.List;
import java.util.Date;
import java.util.ArrayList;
import java.util.Calendar;
import java.util.TimeZone;

public class QueueHistoryTimeSeriesFacadeImpl implements IQueueHistoryTimeSeriesFacade{

	private static final Category logger = Log.initCategory(QueueHistoryTimeSeriesFacadeImpl.class.getName());
	private TimeSeriesManager timeSeriesManager;


	public QueueHistoryTimeSeriesFacadeImpl() {
	}

	public List<QueueHistoryTimeSeriesEntity> getHistoryData(RequestContext context, ID queueID, Date startTime, Date endTime) throws WFOException {
		try {
			boolean isWhatIf = context.isInWhatIfMode();
			timeSeriesManager = WfmManagerFactory.getTimeSeriesManager(isWhatIf);
			TraceCube historyTC = timeSeriesManager.getHistoryData(queueID, startTime, endTime);
			return buildQueueHistoryDataEntity(historyTC);

		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	public List<String> getMonthYearData(RequestContext context, ID queueID, int nOffSet) throws WFOException {
		try {
			boolean isWhatIf = context.isInWhatIfMode();
			timeSeriesManager = WfmManagerFactory.getTimeSeriesManager(isWhatIf);
			return timeSeriesManager.getMonthYearListInActualTimeSeries(queueID, nOffSet);
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}

	}

	public List<String> getDayMonthYearData(RequestContext context, ID queueID, Date startTime, Date endTime) throws WFOException {
		try {
			boolean isWhatIf = context.isInWhatIfMode();
			timeSeriesManager = WfmManagerFactory.getTimeSeriesManager(isWhatIf);
			return timeSeriesManager.getDayMonthYearListInActualTimeSeries(queueID, startTime, endTime);
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	public List<String> getWeeklyListData(RequestContext context, ID queueID, Date startTime, Date endTime) throws WFOException {
		try {
			boolean isWhatIf = context.isInWhatIfMode();
			timeSeriesManager = WfmManagerFactory.getTimeSeriesManager(isWhatIf);
			return timeSeriesManager.getWeekListInActualTimeSeries(queueID, startTime, endTime);
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}

	}

	private List<QueueHistoryTimeSeriesEntity> buildQueueHistoryDataEntity(TraceCube historyTC) {
		ArrayList<QueueHistoryTimeSeriesEntity> qhtsList = new ArrayList<QueueHistoryTimeSeriesEntity>();
		Date curDate = historyTC.getRawStartDate();
		while (curDate.before(historyTC.getRawEndDate())) {
			double cv = historyTC.getTraceValue(Trace.CV, curDate);
			double aht = historyTC.getTraceValue(Trace.AHT, curDate);

			if (cv < 0) {
				cv = -1.0;
			}
			if (aht < 0) {
				aht = -1.0;
			}

			QueueHistoryTimeSeriesEntity qhtsEntity = new QueueHistoryTimeSeriesEntity();
			qhtsEntity.setTime(curDate.toString());
			qhtsEntity.setCv(cv);
			qhtsEntity.setAht(aht);

			qhtsList.add(qhtsEntity);
			curDate = getDateIncrement(curDate, 15);
		}
		return qhtsList;
	}

	private Date getDateIncrement(Date curDate, int minute) {
		Calendar cal = Calendar.getInstance();
		cal.setTimeZone(TimeZone.getTimeZone("GMT"));
		cal.setTime(curDate);
		cal.add(Calendar.MINUTE, minute);
		return cal.getTime();
	}
}
