package com.verint.web.fs.workrules.assignmentrules;

import java.util.List;

import com.verint.web.fs.workrules.assignmentrules.question.QuestionType;

public class ResponseImpl<T extends ResponseType> implements Response {

	private QuestionType questionType;
	private T responseType;
	private List<ResponseParam> parameters;
	
	public ResponseImpl(QuestionType questionType, T responseType,
			List<ResponseParam> parameters) {
		super();
		this.questionType = questionType;
		this.responseType = responseType;
		this.parameters = parameters;
	}

	@Override
	public QuestionType getQuestionType() {
		return questionType;
	}

	@Override
	public T getResponseType() {
		return responseType;
	}

	@Override
	public List<ResponseParam> getResponseParameters() {
		return parameters;
	}

	@Override
	public int hashCode() {
		final int prime = 31;
		int result = 1;
		result = prime * result + ((parameters == null) ? 0 : parameters.hashCode());
		result = prime * result + ((questionType == null) ? 0 : questionType.hashCode());
		result = prime * result + ((responseType == null) ? 0 : responseType.hashCode());
		return result;
	}

	@Override
	public boolean equals(Object obj) {
		if (this == obj)
			return true;
		if (obj == null)
			return false;
		if (getClass() != obj.getClass())
			return false;
		ResponseImpl other = (ResponseImpl) obj;
		if (parameters == null) {
			if (other.parameters != null)
				return false;
		} else if (!parameters.equals(other.parameters))
			return false;
		if (questionType != other.questionType)
			return false;
		if (responseType == null) {
			if (other.responseType != null)
				return false;
		} else if (!responseType.equals(other.responseType))
			return false;
		return true;
	}

	@Override
	public String getIDStr() {
		return getResponseType().toString();
	}

	@Override
	public String toString() {
		return getIDStr();
	}
	
	
}
