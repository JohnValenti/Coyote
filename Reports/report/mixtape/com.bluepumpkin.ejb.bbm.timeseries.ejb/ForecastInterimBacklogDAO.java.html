import java.util.ArrayList;

import java.util.Date;
import java.util.HashMap;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;

public class ForecastInterimBacklogDAO extends DAOBase<ForecastInterimBacklog> {
	private static FieldInfo m_fieldInfo = new ForecastInterimBacklogFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	/**
	* default constructor
	*/
	public ForecastInterimBacklogDAO(){
		super();
	}
	
	@Override
	protected ForecastInterimBacklog createValueObject() {
		return new ForecastInterimBacklog();
	}
	
	public static ID createForecastInterimBacklog(ForecastInterimBacklog objValue) throws JdmoException {
		Jdmo jdmo = new Jdmo();
		ID result = null;
		try {
			HashMap insertHashMap = new HashMap(4);
			insertHashMap.put("FORECASTINSTANCEID", objValue.getForecastInstanceId());
			insertHashMap.put("SPQUEUEID", objValue.getSpQueueId());
			insertHashMap.put("INTERIMBACKLOG", new Float(objValue.getInterimBacklog()));
			insertHashMap.put("INTERIMBACKLOGDATE", objValue.getInterimBacklogDate());
			
			
			result = jdmo.executeInsert(ForecastInterimBacklogFieldInfo.TABLE_NAME, insertHashMap);
			
		} finally {
			jdmo.cleanUp();
		}	
		return result;
	}
	
	public static void deleteBacklogByForecastInstanceID(ID forecastInstance) throws JdmoException {
		Jdmo jdmo = new Jdmo();
		if (forecastInstance == null) return;
		try {
			
			StringBuffer strSQL = new StringBuffer(30);
			strSQL.append("DELETE FROM ").append(ForecastInterimBacklogFieldInfo.TABLE_NAME);
			strSQL.append(" WHERE FORECASTINSTANCEID = ");
			strSQL.append(forecastInstance);
			

			jdmo.executeCommand(strSQL.toString());
		} finally {
			jdmo.cleanUp();
		}
	}

	/**
	 * This method will update the records in the FCINSTSPQINTERIMBACKLOG table if they exist
	 * for the given spQueues/forecastInstance combinations, otherwise it will create new records for them.
	 */
	public void updateInterimBacklogForForecastInstance(Collection<SPQueue> spQueues, ID forecastInstanceID) throws BbmFinderException, JdmoException, MultiUserException, BbmUpdateException, BbmCreateException {
		Collection<ID> spQueueIDs = ValueObjectUtil.getIDFromObjects(spQueues);
		HashMap<ID, SPQueue> spQueueIDMap = ValueObjectUtil.getIDObjectMap(spQueues);
		Collection<ForecastInterimBacklog> fibs = getObjects(" " +
				getFieldInfo().getFieldName(ForecastInterimBacklogFieldInfo.FORECASTBACKLOG_QUEUEID) + " IN " +
				getDMO().createInClause(spQueueIDs) + "AND " +
				getFieldInfo().getFieldName(ForecastInterimBacklogFieldInfo.FORECASTBACKLOG_FORECASTINSTANCEID) + " = " +
				JdmoUtil.asSqlLiteral(forecastInstanceID));
		for (ForecastInterimBacklog fib : fibs) {
			spQueueIDs.remove(fib.getSpQueueId());
			setInterimBacklogFields(spQueueIDMap.get(fib.getSpQueueId()), fib);
		}
		updateObjects(fibs);
		
		if (spQueueIDs.size() > 0) {
			Collection<ForecastInterimBacklog> newFibs = new ArrayList<ForecastInterimBacklog>();
			for (ID spQueueID : spQueueIDs) {
				ForecastInterimBacklog fib = createValueObject();
				fib.setForecastInstanceId(forecastInstanceID);
				setInterimBacklogFields(spQueueIDMap.get(spQueueID), fib);
				fib.setSpQueueId(spQueueID);
				newFibs.add(fib);
			}
			createObjects(newFibs);
		}
	}
	
	
	/**
	 * This method will update the records in the FCINSTSPQINTERIMBACKLOG table if they exist
	 * for the given spQueues/forecastInstance combinations, otherwise it will create new records for them.
	 */
	public Collection<ForecastInterimBacklog> getInterimBacklogForForecastInstance(Collection<ID> spQueueIDs, ID forecastInstanceID) throws BbmFinderException, JdmoException, MultiUserException, BbmUpdateException, BbmCreateException {
		//Collection<ID> spQueueIDs = ValueObjectUtil.getIDFromObjects(spQueues);
		//HashMap<ID, SPQueue> spQueueIDMap = ValueObjectUtil.getIDObjectMap(spQueues);
		Collection<ForecastInterimBacklog> fibs = getObjects(" " +
				getFieldInfo().getFieldName(ForecastInterimBacklogFieldInfo.FORECASTBACKLOG_QUEUEID) + " IN " +
				getDMO().createInClause(spQueueIDs) + "AND " +
				getFieldInfo().getFieldName(ForecastInterimBacklogFieldInfo.FORECASTBACKLOG_FORECASTINSTANCEID) + " = " +
				JdmoUtil.asSqlLiteral(forecastInstanceID));
		
		return fibs;
	}
	
	/**
	 * Sets the interim backlog value before being persisted in the DB.  If the interim backlog date
	 * is null, this means interim backlog has not been set and therefore the DB value for interim
	 * backlog needs to be set to -9999.
	 */
	private void setInterimBacklogFields(SPQueue spQueue, ForecastInterimBacklog fib) {
		if (spQueue.getInterimBacklogDate() == null) {
			fib.setInterimBacklog(-9999.0f);
			fib.setInterimBacklogDate(new Date());				
		} else {
			fib.setInterimBacklog(spQueue.getInterimBacklog());
			fib.setInterimBacklogDate(spQueue.getInterimBacklogDate());
		}		
	}
}
