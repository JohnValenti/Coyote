package com.bluepumpkin.ejb.bbm.timeseries.ejb;

import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueueFieldInfo;
import com.bluepumpkin.ejb.bbm.dao.DAOBaseWithAutoIDMapping;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastAllocationTimeSeries;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastAllocationTimeSeriesFieldInfo;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastAllocationTimeSeriesFieldInfo.Column;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

public class ForecastAllocationTimeSeriesDAO extends
		DAOBaseWithAutoIDMapping<ForecastAllocationTimeSeries> {

	private static final ForecastAllocationTimeSeriesFieldInfo fieldInfo = new ForecastAllocationTimeSeriesFieldInfo();

	public ForecastAllocationTimeSeriesDAO() {
		super();
	}

	public ForecastAllocationTimeSeriesDAO(Jdmo dmo) {
		super(dmo);
	}

	@Override
	protected ForecastAllocationTimeSeries createValueObject() {
		return new ForecastAllocationTimeSeries();
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return fieldInfo;
	}

	@Override
	protected Map<Integer, ? extends FieldInfo> getForeignKeyFieldInfo() {
		Map<Integer, FieldInfo> retVal = new HashMap<Integer, FieldInfo>();
		retVal.put(Column.SOURCESPQUEUEID.getIndex(), new SPQueueFieldInfo());
		retVal.put(Column.DESTINATIONSPQUEUEID.getIndex(), new SPQueueFieldInfo());
		return retVal;
	}
}
