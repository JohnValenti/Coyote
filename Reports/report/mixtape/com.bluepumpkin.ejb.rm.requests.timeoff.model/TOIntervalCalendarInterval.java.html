package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOIntervalAllocation;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOIntervalApprovedAndPending;

/**
 * Represents the data within a specific Time Off Calendar's Interval. Similar to TOCalendarDayData.
 */
public class TOIntervalCalendarInterval {

	private Date startTime;
	private Date endTime;
	private final Float allocatedHours;
	private TOIntervalAllocation interval;
	private ID toPoolID;
	private Float allPendingHours;
	private Float allApprovedHours;
	private Float pendingHoursForEmployee;
	private Float approvedHoursForEmployee;

	public TOIntervalCalendarInterval(Date startTime, Date endTime, Float allocatedHours) {
		this.startTime = startTime;
		this.endTime = endTime;
		this.allocatedHours = allocatedHours;
	}

	public TOIntervalCalendarInterval(TOIntervalAllocation interval, ID toPoolID) {
		this.interval = interval;
		this.startTime = interval.getStartTime();
		this.endTime = interval.getEndTime();
		this.allocatedHours = interval.getAllocatedHours();
		this.toPoolID = toPoolID;
	}

	public TOIntervalCalendarInterval(TOIntervalApprovedAndPending interval, ID toPoolID) {
		this((TOIntervalAllocation) interval, toPoolID);

		if (interval.getTOPoolID() != null) {
			this.toPoolID = interval.getTOPoolID();
		}

		allPendingHours = interval.getPendingHours();
		allApprovedHours = interval.getApprovedHours();
		pendingHoursForEmployee = interval.getEmployeePendingHours();
		approvedHoursForEmployee = interval.getEmployeeApprovedHours();
	}

	/**
	 * Creates a copy of the source instance.
	 * 
	 * @param source
	 */
	public TOIntervalCalendarInterval(TOIntervalCalendarInterval source) {
		this(source.startTime, source.endTime, source.allocatedHours);
		this.interval = source.interval;
		this.toPoolID = source.toPoolID;
		allPendingHours = source.allPendingHours;
		allApprovedHours = source.allApprovedHours;
		pendingHoursForEmployee = source.pendingHoursForEmployee;
		approvedHoursForEmployee = source.approvedHoursForEmployee;
	}

	public TOIntervalAllocation getInterval() {
		return interval;
	}

	public Date getStartTime() {
		return this.startTime;
	}

	public void setStartTime(Date startTime) {
		this.startTime = startTime;
	}

	public Date getEndTime() {
		return this.endTime;
	}

	public void setEndTime(Date endTime) {
		this.endTime = endTime;
	}

	public Float getAllocatedHours() {
		return this.allocatedHours;
	}

	public Float getAvailableHours(boolean forEmployee) {

		if (this.allocatedHours == null) {
			return 0f;
		}

		Float approvedHours = getApprovedHours(forEmployee);

		if (approvedHours == null) {
			return this.allocatedHours;
		}
		return Math.max(0f, this.allocatedHours - approvedHours);
	}


	public Float getPendingHours(boolean forEmployee) {
		return forEmployee ? this.pendingHoursForEmployee : this.allPendingHours;
	}

	public Float getApprovedHours(boolean forEmployee) {
		return forEmployee ? this.approvedHoursForEmployee : this.allApprovedHours;
	}

	public ID getTOPoolID() {
		return toPoolID;
	}

	public void setInterval(TOIntervalAllocation interval) {
		this.interval = interval;
	}

	@Override
	public String toString() {
		return String.format("\"TOP=%s%n, StartTime=%s%n EndTime=%s%n AllocatedHours=%s%n Interval=%s\"", this.toPoolID, this.startTime,
				this.endTime, this.allocatedHours, this.interval);
	}

}
