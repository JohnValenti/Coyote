package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;
import com.bluepumpkin.ejb.rm.cache.CacheUtilRM;

import java.util.*;

public class TOAccrualWeekly extends TOAccrual {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public static final String scheduleOfAccrualStr = "Semi-Monthly";
   	Integer accrualDayOfWeek = null;
	public int getAccrualDayOfWeek() {
		if (accrualDayOfWeek == null) {
			int day = org.getWeekStartDate();
			try {
				//get accrual day of week from org setting
				day = CacheUtilRM.getOrgSetting(org.getID()).getAccrualDayOfWeek();
			} catch (Exception e) {
				e.printStackTrace();
			}
			accrualDayOfWeek = new Integer(day);
		}
		return accrualDayOfWeek.intValue();
	}
	public void incrementPeriod(Calendar calStart) {
		calStart.add(Calendar.WEEK_OF_MONTH, 1);
	}

	public void decrementPeriod(Calendar calStart) {
		calStart.add(Calendar.WEEK_OF_MONTH, -1);
	}
	/* TO Weekly will init to start of the week based on org start date of the employee's org*/

	protected void initCalendarToStartOfPeriod(Calendar calStart) {
		super.initCalendarToStartOfPeriod(calStart);
		calStart.set(Calendar.DAY_OF_WEEK, getAccrualDayOfWeek());
	}

}
