package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.io.Serializable;
import java.util.Date;

/**
 * Title:        TimeOffEventSummary
 * Description:  Helper object that packages the timeoff event's duration for the Ui
 * along with the event
 * Copyright:    Copyright (c) 2005
 * Company:      Witness Systems, inc
 * @author       Swati Tewari
 * @version      1.0
 */
public class TimeOffEventSummary  implements Serializable{

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private String m_activityName;
	private float m_hoursAccounted;
	private Date m_startTime;
	private Date m_endTime;
	
	public TimeOffEventSummary(Date start, Date end,
			String activityName, float hoursAccounted) {
		m_startTime = start;
		m_endTime = end;
		m_activityName = activityName;
		m_hoursAccounted = hoursAccounted;
	}

	public Date getStartTime(){
		return m_startTime;
	}
	
	public Date getEndTime(){
		return m_endTime;
	}
	
	public String getActivityName(){
		return m_activityName;
	}
	
	public float getHoursAccounted(){
		return m_hoursAccounted;
	}
}
