package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Calendar;
import java.util.Collection;
import java.util.Date;
import java.util.List;
import java.util.Locale;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;

/**
 * Time Off Interval Calendar contains intervals and increments broken down into 15 minutes for special statuses
 * such as Blackout day, holiday, etc.
 */
public class TOIntervalCalendar {
	private static Category LOG = Log.initCategory(TOIntervalCalendar.class.getName());
	private static final int MILLISECONDS_IN_INTERVAL = 15 * 60 * 1000;

	private final List<TOIntervalCalendarInterval> intervals;
	private List<TOIntervalCalendarIncrement> increments;
	private final Date startTime;
	private final Date endTime;
	private final TimeZone timeZone;
	private final List<TOIntervalCalendarDay> days = new ArrayList<TOIntervalCalendarDay>();
	private TOIntervalCalendarDay longestDay = null;
	private int dstTransitionIndex = -1;
	private DSTTransitionMode dstTransitionMode = DSTTransitionMode.NEUTRAL;
	private final SimpleDateFormat dateFormat = new SimpleDateFormat("MM/dd/yyyy HH:mm:ss zZ", Locale.US);

	// ================================================================================
	// CONSTRUCTION
	// ================================================================================

	/**
	 * Creates a new instance based on the specified start and end times, rounding to the nearest 15 minutes if necessary.
	 */
	public TOIntervalCalendar(Date startTime, Date endTime, TimeZone timeZone) {
		this.startTime = DateUtil.roundToNearestInterval(startTime);
		this.endTime = DateUtil.roundToNearestInterval(endTime);
		this.timeZone = timeZone;
		this.intervals = new ArrayList<TOIntervalCalendarInterval>();
		createIncrements();
	}

	/**
	 * Creates a new instance based on the specified start and end times, rounding to the nearest 15 minutes if necessary.
	 * This is shorthand for creating and then adding intervals.
	 */
	public TOIntervalCalendar(Date startTime, Date endTime, TimeZone timeZone, List<TOIntervalCalendarInterval> intervals) {
		this(startTime, endTime, timeZone);

		addIntervals(intervals);
	}

		return days;

		return dstTransitionIndex;

	/**
	 * Add blackout days to the calendar. This will mark organization days if they align with the blackout day.
	 */
	public void addBlackoutDays(Collection<BlackoutDay> blackoutDays, TimeZone timeZone, int orgDayBoundaryOffset) {
		Calendar cal = Calendar.getInstance(timeZone);
		if (blackoutDays.isEmpty() || days.isEmpty()) {
			return;
		}

		int daysIndex = 0;
		TOIntervalCalendarDay day = days.get(0);
		for (BlackoutDay blackoutDay : blackoutDays) {
			LocalDate localDate = blackoutDay.getBlackoutLocalDate();
			Date orgDate = localDate.getTime(timeZone);
			cal.setTime(orgDate);
			TOCalcUtil.setCalMinsOffsetFromMidnight(cal, orgDayBoundaryOffset);

			while (day.getStartTime().before(cal.getTime())) {
				daysIndex++;
				if (daysIndex >= days.size()) {
					return;
				}
				day = days.get(daysIndex);
			}
			if (cal.getTime().equals(day.getStartTime())) {
				day.setBlackoutDay(blackoutDay);
			}
		}
	}

	/**
	 * Adds the Intervals to each increment in which they overlap. This will replace any existing assignment.
	 * Intervals are expected to be in ascending order with no overlap.
	 *
	 * @param intervals
	 */
	public void addIntervals(List<TOIntervalCalendarInterval> intervals) {
		if (intervals.isEmpty()) {
			return;
		}
		this.intervals.addAll(intervals);
		List<TOIntervalCalendarInterval> intervalQueue = new ArrayList<TOIntervalCalendarInterval>(intervals);
		TOIntervalCalendarInterval interval = intervalQueue.get(0);
		// skip to the index of this interval
		int index = getIndex(interval.getStartTime());
		index = (0 <= index && index < this.increments.size()) ? index : 0;
		for (; index < this.increments.size(); index++) {
			TOIntervalCalendarIncrement increment = this.increments.get(index);
			interval = findInterval(intervalQueue, increment.getDate());
			if (interval != null) {
				increment.setInterval(interval);
			}
		}
	}

	public TOIntervalCalendarDay getDSTTransitionDay() {
		if (dstTransitionIndex < 0) {
			return null;
		}
		for (TOIntervalCalendarDay day : days) {
			int startIndex = getIndex(day.getStartTime());
			int endIndex = getIndex(day.getEndTime());
			if (startIndex <= dstTransitionIndex && dstTransitionIndex < endIndex) {
				return day;
			}
		}
		return null;
	}

	/**
	 * Gets the DST transition mode for the specified interval.
	 */
	public DSTTransitionMode getDstTransitionMode(TOIntervalCalendarInterval interval) {
		int startIndex = getIndex(interval.getStartTime());
		int endIndex = getIndex(interval.getEndTime());
		if (startIndex <= dstTransitionIndex && dstTransitionIndex < endIndex) {
			return dstTransitionMode;
		}
		return DSTTransitionMode.NEUTRAL;
	}

	/**
	 * Gets whether a gap is before the increment.
	 */
	public boolean hasDstTransitionGapBefore(TOIntervalCalendarIncrement increment) {
		int index = getIndex(increment.getDate());
		// fallback has an extra hour, only add a gap for other days
		if (dstTransitionMode == DSTTransitionMode.FALLBACK) {
			return increment.isNonDstDayTransition();
		}
		return index == dstTransitionIndex;
	}

	/**
	 * Gets whether a gap is before the interval.
	 */
	public boolean hasDstTransitionGapBefore(TOIntervalCalendarInterval interval, TOIntervalCalendarDay day) {
		int startIndex = getIndex(interval.getStartTime());
		if (dstTransitionMode == DSTTransitionMode.SPRINGFORWARD) {
			if (startIndex == this.dstTransitionIndex) {
				if (day.getLength() == TimeZoneUtil.DAY_IN_MILLISECONDS_LONG) {
					// this day is 24 hours so the DST hour is on the previous day
					return false;
				}
				return true;
			}
			int difference = startIndex - this.dstTransitionIndex;
			if (interval.getStartTime().equals(day.getStartTime()) && 0 < difference && difference < DateUtil.INTERVALS_IN_HOUR) {
				return true;
			}
		}
		return this.increments.get(startIndex).isNonDstDayTransition();
	}

	/**
	 * Gets whether the interval contains the DST transition gap for the day. Intervals
	 * that start on the transition will return false (See hasDstTransitionGapBefore).
	 */
	public boolean hasDstTransitionGapWithin(TOIntervalCalendarInterval interval) {
		int startIndex = getIndex(interval.getStartTime());
		int endIndexExclusive = getIndex(interval.getEndTime());
		if (startIndex < dstTransitionIndex && dstTransitionIndex < endIndexExclusive) {
			if (dstTransitionMode == DSTTransitionMode.SPRINGFORWARD) {
				return true;
			}
			// FallBack already contains extra increments for the DST hour
			return false;
		}
		for (int index = startIndex + 1; index < endIndexExclusive; index++) {
			TOIntervalCalendarIncrement increment = increments.get(index);
			if (increment.isNonDstDayTransition()) {
				return true;
			}
		}
		return false;
	}

	/**
	 * Marks the increments with the specified type for each date range pair.
	 */
	public void markIncrements(Date startTime, Date endTime, int type) {
		List<TimeRange> dates = new ArrayList<TimeRange>(1);
		dates.add(new TimeRange(startTime, endTime));
		markIncrements(dates, type);
	}

	/**
	 * Marks the increments with the specified type for each date range pair.
	 */
	public void markIncrements(List<TimeRange> dates, int type) {
		for (TimeRange range : dates) {
			// adjust the range
			int startIndex = getIndex(range.getStartDate());
			int endIndexExclusive = getIndex(range.getEndDate());
			if (startIndex < 0) {
				startIndex = 0;
			}
			if (increments.size() <= endIndexExclusive) {
				endIndexExclusive = increments.size();
			}
			if (startIndex >= increments.size() || endIndexExclusive < 0) {
				continue;
			}

			// mark increments
			for (int index = startIndex; index < endIndexExclusive; index++) {
				increments.get(index).setTypeIfHigherPriority(type);
			}
		}
	}

	/**
	 * Splits the collection of intervals up into a series of days. If an interval spans days, it will be split in two.
	 * This will also find the longest day so it can be used for DST logic.
	 */
	public void splitIntoDays() {
		splitIntoDays(timeZone, startTime, endTime);
	}

	/**
	 * Splits the collection of intervals up into a series of days. If an interval spans days, it will be split in two.
	 * Days will be ignored and intervals will be ignored or trimmed to fit the specified range.
	 * This will also find the longest day so it can be used for DST logic.
	 *
	 * @param timeZone
	 *            The time zone to use for "days".
	 * @param startTime
	 *            The start time of the range. This must be within the existing range.
	 * @param endTime
	 *            The end time of the range. This must be within the existing range.
	 */
	public void splitIntoDays(TimeZone timeZone, Date startTime, Date endTime) {
		assertTimeIsWithinRange(startTime);
		assertTimeIsWithinRange(endTime);
		Calendar cal = Calendar.getInstance(timeZone);
		int offsetMinutes = DateUtil.getTimeInMinutesFromStartOfDay(startTime, cal);
		cal.setTime(startTime);
		Calendar tomorrow = (Calendar) cal.clone();
		DateUtil.addDays(tomorrow, 1, offsetMinutes);

		int intervalIndex = getFirstIntervalIndexFor(startTime);
		while (cal.getTime().before(endTime)) {
			// do not go past the endTime
			if (tomorrow.getTime().after(endTime)) {
				tomorrow.setTime(endTime);
			}
			TOIntervalCalendarDay day = new TOIntervalCalendarDay(cal.getTime(), tomorrow.getTime());
			days.add(day);
			updateLongestDay(day);
			while (intervalIndex < intervals.size()) {
				TOIntervalCalendarInterval interval = intervals.get(intervalIndex);
				if (DateUtil.isAfterOrEqual(interval.getStartTime(), tomorrow.getTime())) {
					// done with the day
					break;
				}
				if (tomorrow.getTime().before(interval.getEndTime())) {
					// split up the interval & update increments.
					TOIntervalCalendarInterval split = new TOIntervalCalendarInterval(interval);
					interval.setEndTime(tomorrow.getTime());
					split.setStartTime(tomorrow.getTime());
					intervals.add(intervalIndex + 1, split);
					updateIncrements(split);
				}
				// trim interval if it starts before the current time
				if (interval.getStartTime().before(cal.getTime())) {
					interval.setStartTime(cal.getTime());
				}
				// only add the interval if there is time (This might never be the case b/c getFirstIntervalIndexFor)
				if (interval.getStartTime().before(interval.getEndTime())) {
					day.add(interval);
				}
				intervalIndex++;
			}
			cal.setTime(tomorrow.getTime());
			DateUtil.addDays(tomorrow, 1, offsetMinutes);
		}

		updateDSTTransition(timeZone, startTime, endTime);
	}

	public int getIndex(Date date) {
		return (int) ((date.getTime() - this.startTime.getTime()) / (MILLISECONDS_IN_INTERVAL));
	}

	@Override
	public String toString() {
		return String.format("(%s - %s, Increments=%s, Intervals=%s)", this.startTime, this.endTime,
				this.increments.size(), this.intervals.size());
	}

	// ================================================================================
	// IMPLEMENTATION
	// ================================================================================

	private void assertTimeIsWithinRange(Date time) {
		TimeRange range = new TimeRange(startTime, endTime);
		if (!range.includes(time)) {
			throw new IllegalArgumentException(
					String.format("Date '%s' must be within range '%s' - '%s'.", time, startTime, endTime));
		}
	}

	/**
	 * Create the 15 minute increments between the start time and exclusive end time.
	 */
	private void createIncrements() {
		int numberOfIncrements = (int) ((this.endTime.getTime() - this.startTime.getTime()) / (MILLISECONDS_IN_INTERVAL));
		this.increments = new ArrayList<TOIntervalCalendarIncrement>(numberOfIncrements > 0 ? numberOfIncrements : 0);
		Calendar cal = Calendar.getInstance(timeZone);
		cal.setTime(this.startTime);
		while (cal.getTime().compareTo(endTime) < 0) {
			this.increments.add(new TOIntervalCalendarIncrement(cal.getTime()));
			cal.add(Calendar.MINUTE, DateUtil.TIME_INTERVAL);
		}
	}

	private int getFirstIntervalIndexFor(Date date) {
		for (int index = 0; index < intervals.size(); index++) {
			TOIntervalCalendarInterval interval = intervals.get(index);
			TimeRange range = new TimeRange(interval.getStartTime(), interval.getEndTime());
			if (range.includes(date) && !interval.getEndTime().equals(date)) {
				return index;
			}
		}
		// return size so comparisons using this index will automatically fail
		return intervals.size();
	}

	/**
	 * Gets whether the DST Transition point is within the last hour of the longest day.
	 * 1:00 am - 1:45 am EST with an org offset of 1 hour has the first 1am on the previous day.
	 * Since it is at the end, none of the other days require gaps before them.
	 */
	private boolean isDstTransitionWithinLastHourOfLongestDay(TOIntervalCalendarIncrement increment, TimeZone timeZone) {
		if (longestDay == null) {
			return false;
		}
		Calendar cal = Calendar.getInstance(timeZone);
		cal.setTime(longestDay.getEndTime());
		cal.add(Calendar.HOUR, -1);
		TimeRange range = new TimeRange(cal.getTime(), longestDay.getEndTime());
		return range.includes(increment.getDate());
	}

	/**
	 * Marks increments on non-DST days that match the DST transition point when the mode is FallBack.
	 */
	private void markNonDstDayTransitions(TOIntervalCalendarIncrement increment, TimeZone timeZone, Date startTime, Date endTime) {
		if (dstTransitionMode != DSTTransitionMode.FALLBACK) {
			return;
		}
		Calendar cal = Calendar.getInstance(timeZone);
		cal.setTime(increment.getDate());
		/*
		 * Note to self: We can't just add a day here because org offsets that align with DST cause it to pick the
		 * second 1 am during fallback. Adding days will cause the extra time to appear in the wrong place. So we
		 * add an hour to push the transition to 2 am and then mark the increments. Please note that these increments
		 * might be the start of a day, which means the gap is actually on the day before. The call to
		 * isDstTransitionWithinLastHourOfLongestDay should protect against adding gaps in that case.
		 */
		cal.add(Calendar.DATE, -1);
		cal.add(Calendar.HOUR, 1);
		debugMarkNonDstDayTransitions(increment, timeZone, startTime, endTime);
		if (!isDstTransitionWithinLastHourOfLongestDay(increment, timeZone)) {
			while (DateUtil.isBeforeOrEqual(startTime, cal.getTime())) {
				int index = getIndex(cal.getTime());
				TOIntervalCalendarIncrement inc = increments.get(index);
				inc.setIsNonDstDayTransition(true);
				cal.add(Calendar.DATE, -1);
			}
			cal.setTime(increment.getDate());
			cal.add(Calendar.DATE, 1);
			cal.add(Calendar.HOUR, 1);
			while (cal.getTime().before(endTime)) {
				int index = getIndex(cal.getTime());
				TOIntervalCalendarIncrement inc = increments.get(index);
				inc.setIsNonDstDayTransition(true);
				cal.add(Calendar.DATE, 1);
			}
		}
	}

	private void updateDSTTransition(TimeZone timeZone, Date startTime, Date endTime) {
		if (increments.isEmpty()) {
			return;
		}
		boolean prevIsInDst = timeZone.inDaylightTime(startTime);
		TOIntervalCalendarIncrement increment = null;
		for (int index = getIndex(startTime); index < increments.size(); index++) { // NOSONAR
			increment = increments.get(index);
			if (increment.getDate().after(endTime)) {
				break;
			}
			boolean isInDst = timeZone.inDaylightTime(increment.getDate());
			if (prevIsInDst && !isInDst) {
				// Fall Back / DST Ends (25 hour day)
				dstTransitionMode = DSTTransitionMode.FALLBACK;
				dstTransitionIndex = index;
				break;
			} else if (!prevIsInDst && isInDst) {
				// Spring Forward / DST Starts (23 hour day)
				dstTransitionMode = DSTTransitionMode.SPRINGFORWARD;
				dstTransitionIndex = index;
				break;
			}
		}
		markNonDstDayTransitions(increment, timeZone, startTime, endTime);
	}

	/**
	 * Updates the increments's interval value.
	 */
	private void updateIncrements(TOIntervalCalendarInterval interval) {
		int index = getIndex(interval.getStartTime());
		int endIndex = getIndex(interval.getEndTime());
		if (index < 0 || this.increments.size() < index || endIndex < 0 || this.increments.size() < endIndex) {
			return;
		}
		for (; index < endIndex; index++) {
			TOIntervalCalendarIncrement increment = this.increments.get(index);
			increment.setInterval(interval);
		}
	}

	private void updateLongestDay(TOIntervalCalendarDay day) {
		if (longestDay == null || longestDay.getLength() < day.getLength()) {
			longestDay = day;
		}
	}

	/**
	 * Finds the interval that contains the date within the queue. Unmatched items are dequeued.
	 */
	private TOIntervalCalendarInterval findInterval(List<TOIntervalCalendarInterval> intervalQueue, Date date) {
		if (intervalQueue.isEmpty()) {
			return null;
		}
		TOIntervalCalendarInterval interval = intervalQueue.get(0);
		if (date.compareTo(interval.getStartTime()) < 0) {
			// they are in order so the rest of the intervals start in the future.
			// this happens when we "append" intervals to a single TOIC.
			return null;
		}
		while (interval.getEndTime().compareTo(date) <= 0) {
			intervalQueue.remove(0);
			if (intervalQueue.isEmpty()) {
				return null;
			}
			interval = intervalQueue.get(0);
		}
		return interval;
	}

	private void debugMarkNonDstDayTransitions(TOIntervalCalendarIncrement increment, TimeZone timeZone, Date startTime, Date endTime) {
		if (!LOG.isDebugEnabled()) {
			return;
		}
		dateFormat.setTimeZone(timeZone);
		LOG.debug(dateFormat.format(startTime) + " - " + dateFormat.format(endTime));
		LOG.debug("DST Transition: " + dateFormat.format(increment.getDate()));
		LOG.debug("LongestDay: " + dateFormat.format(longestDay.getStartTime()) + " - " +
				dateFormat.format(longestDay.getEndTime()));
	}
