package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model;

import com.bluepumpkin.ejb.bbm.vo.ChildInfoEntry;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;


/**
 * Title:        BiddableShiftFieldInfo
 * Description:  meta data information for BiddableShift object
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc.
 * @author       Jagdish Seelam
 * @version 1.0
 */

public class BiddableShiftFieldInfo extends FieldInfo {

	// field id constants
	public static final int BIDDABLESHIFT_I_ID = 0;
	public static final int BIDDABLESHIFT_I_BIDDABLESCHEDULEINSTANCEID = 1;
	public static final int BIDDABLESHIFT_I_SHIFTASSIGNMENTID = 2;
    public static final int BIDDABLESHIFT_N_SHIFTORDER = 3;

	// don't start with m_fieldPrefix in base class
	public static final int BIDDABLESHIFTNUM_OF_FIELDS = 4;

	private static FieldInfoEntry m_fieldInfoEntries[] = new FieldInfoEntry[BIDDABLESHIFTNUM_OF_FIELDS];
	private static String m_strTableName = "BIDDABLESHIFT";
	
    public static final int SHIFTASSIGNMENT_CHILD_TYPE = 0;
    protected static final ChildInfoEntry[] m_childInfoEntries = 
        { ChildInfoEntry.oneToOneChild(BIDDABLESHIFT_I_SHIFTASSIGNMENTID) };
        
   // static initialization
	static {
        RequestUtil.initFieldInfoEntries(m_fieldInfoEntries, BiddableShiftFieldInfo.class,
            m_strTableName.toUpperCase() + '_', BIDDABLESHIFTNUM_OF_FIELDS);
	}

	public BiddableShiftFieldInfo() {
		super();
	}

	// implement abstract methods in parent class
	public String getTableName() { 
        return m_strTableName; 
    }
	
    public int getIDFieldIndex() { 
        return BIDDABLESHIFT_I_ID; 
    }
    
    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.bbm.vo.FieldInfo#getParentIDFieldIndex()
     */
    public int getParentIDFieldIndex() {
        return BIDDABLESHIFT_I_BIDDABLESCHEDULEINSTANCEID;
    }

	public FieldInfoEntry [] getFieldInfo() {
		return m_fieldInfoEntries;
	}
    
    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.bbm.vo.FieldInfo#getChildInfo()
     */
    public ChildInfoEntry[] getChildInfo() {
        return m_childInfoEntries;
    }
}
