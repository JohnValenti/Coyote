package com.verint.web.fs.workrules.assignment;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignWorkResource;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.bbm.workload.queue.selection.QueueFilterUtil;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.verint.web.fs.workrules.WorkRulesModelHandler;
import com.witness.web.uif.Log;
import com.witness.web.uif.system.RequestContext;

import java.util.ArrayList;
import java.util.Collection;

public class WorkRulesEmployeeAssignmentUtil {

	private static Category log = Log.initCategory(WorkRulesEmployeeAssignmentUtil.class.getName());

	/**
	 * If the selected employees are no longer part of the valid selection (as often happens when
	 * the campaign or scheduling period dropdowns are changed), then indicate that to the page model.
	 * This method assumes that the currently selected scheduling period is stored in the cache (via
	 * QueueFilterUtil.storeSchedulingPeriodID).  If an invalid or null SP ID is cached, this method
	 * will return false.
	 *
	 * @param selectedIDs - A comma delimited string of the currently selected employee IDs that we will verify
	 *                    belong to the cached scheduling period.
	 */
	public static boolean checkIfEmployeeSelectionIsValid(String selectedIDs, RequestContext context) throws Exception {
		if (selectedIDs != null && selectedIDs.length() > 0) {
			ID spID = QueueFilterUtil.retrieveSchedulingPeriodID(context);
			if (spID == null) {
				return false;
			} else {
				SchedulingPeriod sp = CampaignModelHandler.getSchedulingPeriodByID(context, spID);
				if (sp == null) {
					return false;
				}
				ID campaignID = QueueFilterUtil.retrieveCampaignID(context);
				if (campaignID == null) {
					//If the campaign ID is null but SP ID is non-null (shouldn't happen), log a warning about the inconsistent state.
					log.warn("Warning: inconsistent state detected in WorkRulesEmployeeAssignmentUtil.checkIfEmployeeSelectionIsValid." +
									"Campaign ID = null when SP ID = " + sp.getID(),
							new IllegalArgumentException());
					return false;
				}
				ArrayList<ID> campaignSPEmployees = new ArrayList<>();
				Collection<ID> selectedEmployees = StringUtil.parseIDStringToCollection(selectedIDs);
				Collection<CampaignWorkResource> campaignEmployees = WorkRulesModelHandler.getCampaignWorkResourceAssignments(context,
						campaignID, sp.getStartTime(), sp.getEndTime());

				for (CampaignWorkResource cwr : campaignEmployees) {
					campaignSPEmployees.add(cwr.getWorkResourceID());
				}

				//remove phantoms and then compare, as we are not hiding phantoms
				ScheduleAccessManager sam = WfmManagerFactory.getScheduleAccessManager(context.isInWhatIfMode());
				selectedEmployees = sam.filterPhantomIDsFromWorkResourceIDList(selectedEmployees);

				return campaignSPEmployees.containsAll(selectedEmployees);
			}
		}
		return false;
	}
}
