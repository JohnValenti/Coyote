package com.verint.web.fs.workrules.assignment;

import com.verint.web.fs.workrules.WorkRulesSelectionPopupRH;
import com.witness.web.uif.system.RequestContext;

public class RotationListPopupRH extends WorkRulesSelectionPopupRH<RotationListPopupPM> {
	
	@Override
	protected RotationListPopupPM getPageModel(RequestContext context) {
		RotationListPopupPM pm = new RotationListPopupPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
	}
	@Override
	protected void processTableSort(RequestContext context,RotationListPopupPM pageModel) {
		String sortColumn = context.getRequest().getParameter("sortColumn");
		String sortOrder = context.getRequest().getParameter("sortOrder");
	
		context.setUserProperty(RotationListPopupPM.ROTATION_POPUPLIST_SORTCOLUMN, sortColumn, true);
		context.setUserProperty(RotationListPopupPM.ROTATION_POPUPLIST_SORTORDER, sortOrder, true);

 
	}
}
