package com.verint.web.fs.workrules.assignment;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Comparator;
import java.util.HashSet;
import java.util.LinkedHashMap;
import java.util.Map;
import java.util.Set;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit.WorkUnitType;
import com.bluepumpkin.ejb.bbm.vo.DBSortField;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRuleSortField;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.OrganizationCampaignNameComparator;
import com.verint.web.fs.workrules.PriorityComparator;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.MultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.InsertableMultiColListPC;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;

public class AssignmentRuleListPC extends InsertableMultiColListPC {
	private static final long serialVersionUID = 1L;
	public static final String ASSIGNMENTRULE_LIST_SORTCOLUMN = "ASSIGNMENTRULE_LIST_SORTCOLUMN";
	public static final String ASSIGNMENTRULE_LIST_SORTORDER = "ASSIGNMENTRULE_LIST_SORTORDER";
	private Collection<ComplexWorkRule> assignmentRules;
	private Map<ID, Organization> orgMap = null;
	private Map<ID, String> assignmentRuleDescriptionMap = null;

	private boolean m_isUsedInList =false;
	public AssignmentRuleListPC(RequestContext context, Collection<ComplexWorkRule> assignmentRules,
			Map<ID, Organization> orgMap, Map<ID, String> assignmentRuleDescriptionMap) {
		this(context,assignmentRules,orgMap, assignmentRuleDescriptionMap, false);
	}
	public AssignmentRuleListPC(RequestContext context, Collection<ComplexWorkRule> assignmentRules,
			Map<ID, Organization> orgMap, Map<ID, String> assignmentRuleDescriptionMap, boolean isUsedInList) {
		super(context);

		this.assignmentRules = assignmentRules;
		this.orgMap = orgMap;
		this.assignmentRuleDescriptionMap = assignmentRuleDescriptionMap;
		m_isUsedInList = isUsedInList; ///Must be assign before initHeader
		extractParameters(context.getRequest());
		initHeader();
	}

	@Override
	public String getHtmlDisplay() {
		initRows();
		return super.getHtmlDisplay();
	}

	private void initHeader() {
		TableHeaderPC header = getHeader();
		Localizer localizer = m_context.getLocalizer();
		header.setPartialSortable(true);
		header.addColumnHeaderData(FsWebBundleKey.FS_ASSIGNMENT_RULE,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ASSIGNMENT_RULE), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_ORGANIZATION,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ORGANIZATION), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_ASSIGNMENT_RULE_PRIORITY,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ASSIGNMENT_RULE_PRIORITY), true);
		header.addColumnHeaderData(FsWebBundleKey.BUNDLE_NAME,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_DESCRIPTION), false);
		setPersitedSortHeader(m_isUsedInList,header);
	}
	
	
	private void initRows() {
		Collection<MultiColumnNodeData> rows = super.getListModel();
		
		if (assignmentRules != null) {
			for (ComplexWorkRule cwr : assignmentRules) {
				DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(
						cwr.getID());
				
				//Name
				nodeData.add(cwr.getName());
				
				//Organization
				if (cwr.getOrganizationID() != null) {
					Organization o = orgMap.get(cwr.getOrganizationID());
					if (o != null) {
						nodeData.add(o.getName());	
					} else {
						nodeData.add("");
					}
				} else {
					nodeData.add("");			
				}
				
				//Priority
				nodeData.add(new Integer(cwr.getPriority()).toString());
				
				//Description
				nodeData.add(assignmentRuleDescriptionMap.get(cwr.getID()));		
	
				setRowAttributes(nodeData, cwr);

				rows.add(nodeData);
			}
		}
	}
	
	protected void setRowAttributes(DefaultMultiColumnNodeData rowData, ComplexWorkRule cwr) {}
	
	
	@Override
	public LinkedHashMap<DBSortField, Comparator> getSortComparators() {
		return new LinkedHashMap<DBSortField, Comparator>(getSortFields());
	}

	public LinkedHashMap<ComplexWorkRuleSortField, Comparator<?>> getSortFields() {
		LinkedHashMap<ComplexWorkRuleSortField, Comparator<?>> comparators =
			new LinkedHashMap<ComplexWorkRuleSortField, Comparator<?>>();
		for (ComplexWorkRuleSortField sortField : getDBSortFieldIndices()) {
			switch (sortField) {
				case ID:
					comparators.put(ComplexWorkRuleSortField.ID, null);
					break;
				case NAME:
					comparators.put(ComplexWorkRuleSortField.NAME, null);
					break;
				case ORGANIZATION:
					comparators.put(ComplexWorkRuleSortField.ORGANIZATION, new OrganizationCampaignNameComparator(getLocalizer()));
					break;
				case PRIORITY:
					comparators.put(ComplexWorkRuleSortField.PRIORITY, null);
					break;
				default: break;
			}
		}
		
		return comparators;
	}
	
	public ArrayList<ComplexWorkRuleSortField> getDBSortFieldIndices() {
		ArrayList<ComplexWorkRuleSortField> retVal = new ArrayList<ComplexWorkRuleSortField>();
		TableHeaderPC header = getHeader();
		setPersitedSortHeader(m_isUsedInList, header);
		if (header.getSortColumnIndexes() == null) {
			//Default sort index is by ID (chronological order in which object was created)
			retVal.add(ComplexWorkRuleSortField.ID);
		} else {
			for (int sortField : header.getSortColumnIndexes()) {
				switch (sortField) {
					case 0: 
						retVal.add(ComplexWorkRuleSortField.NAME);
						break;
					case 1: 
						retVal.add(ComplexWorkRuleSortField.ORGANIZATION);
						break;
					case 2: 
						retVal.add(ComplexWorkRuleSortField.PRIORITY);
						break;
					default: break;
				}
			}
		}
		return retVal;
	}
	private void setPersitedSortHeader(boolean isUsedList,TableHeaderPC header) {
		String sortColumn = null;
		String sortOrder = null;
		if(isUsedList) {
			sortColumn = m_context.getUserProperty(ASSIGNMENTRULE_LIST_SORTCOLUMN);
			sortOrder = m_context.getUserProperty(ASSIGNMENTRULE_LIST_SORTORDER);
		}
		
		
		if (sortColumn != null) {
			header.setSortColumn(sortColumn);
		}
		if (sortOrder != null) {
			header.setSortOrder(sortOrder);
		}
	}
}
