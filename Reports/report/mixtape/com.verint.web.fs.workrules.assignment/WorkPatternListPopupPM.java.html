	protected void initToolbar() {
		if (isInfoPopup == false) {
			FindToolbarPageComponent findToolBar = new FindToolbarPageComponent(m_context, getToolbar(),
					FIND_ITEM_FN, WorkRulesSelectionPopupRH.FIND_ACTION,
					getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_FIND_WORKPATTERNS));
			getToolbar().addToolbarComponent(findToolBar);

			getToolbar().addDivider();
			ToolbarTextButton addButton = m_toolbar.createCloseWindowButton(
					"addWorkPatterns", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
							FsWebBundleKey.FS_ADD_SELECTED_WORK_PATTERN), true);

			StringBuffer addButtonJS = new StringBuffer();
			addButtonJS.append(m_toolbar.getName())
					.append(".disableButton('addWorkPatterns'); ")
					.append("RowInsertionUtil.addNewDynamicRow('workPattern', eval('window.opener.workPatternTable'), ")
					.append("'" + EmployeeAssignmentInlineEditComponentRH.URL + "', ")
					.append("'" + EmployeeAssignmentInlineEditComponentRH.PARAM_TYPE + "=" +
							EmployeeAssignmentInlineEditComponentRH.WORK_PATTERN_TYPE + "&")
					.append(EmployeeAssignmentInlineEditComponentRH.PARAM_EMPLOYEE_IDS + "=" +
							CacheUtil.getCachedValue(m_context, CacheUtil.CACHE_MODE_SESSION,
									WorkRulesEmployeeAssignmentPM.EMPLOYEE_ID_LIST_CONTEXT_KEY) + "&");
			if (schedulingPeriodID != null) {    //if in campaign mode, we use the viewStartDate/endDates always
				addButtonJS.append(EmployeeAssignmentInlineEditComponentRH.PARAM_START_DATE + "=" + viewStartDate.getTime() + "&")
						.append(EmployeeAssignmentInlineEditComponentRH.PARAM_END_DATE + "=" + viewEndDate.getTime())
						.append("&" + EmployeeAssignmentInlineEditComponentRH.PARAM_SP_ID + "=" + schedulingPeriodID.toString());
			} else if (selectedEmployees.size() == 1) {  //start date = employee start date. end date = blank
				addButtonJS.append(EmployeeAssignmentInlineEditComponentRH.PARAM_START_DATE + "=" +
						selectedEmployees.iterator().next().getStartTime().getTime());
			} else { //multiple employees selected.  start date = employee start date if all selected employees share the same start date, * otherwise.  end date = blank
				Date startDate = selectedEmployees.iterator().next().getStartTime();
				boolean areStartDatesCommon = true;
				for (Employee employee : selectedEmployees) {
					if (employee.getStartTime().equals(startDate) == false) {
						areStartDatesCommon = false;
						break;
					}
				}
				if (areStartDatesCommon) {
					addButtonJS.append(EmployeeAssignmentInlineEditComponentRH.PARAM_START_DATE + "=" + startDate.getTime());
				} else {
					addButtonJS.append(EmployeeAssignmentInlineEditComponentRH.PARAM_START_DATE + "=" + MULTI_VALUE_DATA);
				}
			}

			//add the view start/end dates
			addButtonJS.append("&" + EmployeeAssignmentInlineEditComponentRH.PARAM_VIEW_START_DATE + "=" + viewStartDate.getTime());
			addButtonJS.append("&" + EmployeeAssignmentInlineEditComponentRH.PARAM_VIEW_END_DATE + "=" + viewEndDate.getTime());
			addButtonJS.append("');");

			addButton.setCustomJS(addButtonJS.toString());
			m_toolbar.addToolbarElement(addButton);

			m_toolbar.addCloseWindowTextButton("cancel", m_localizer.i18n(
					CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_CANCEL), true);
		}
		super.initToolbar();
	}

	protected void loadData() throws Exception {
		if (schedulingPeriodID != null) {
			selectedSchedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
					m_context, schedulingPeriodID);
		}

		String employeeIDStr = CacheUtil.getCachedValue(m_context, CacheUtil.CACHE_MODE_SESSION,
				WorkRulesEmployeeAssignmentPM.EMPLOYEE_ID_LIST_CONTEXT_KEY);
		if (isInfoPopup) {
			workPatterns = new ArrayList<ShiftPattern>(WorkRulesModelHandler.getShiftPatterns(m_context, assignedWorkPatternIDs));
		} else if (employeeIDStr != null) {
			Collection<ID> employeeIDs = StringUtil.parseIDStringToCollection(employeeIDStr);
			selectedEmployees = EmployeeModelHandler.getEmployeesByIDs(
					getRequestContext(), employeeIDs);
			if (areEmployeesOfSameType(selectedEmployees)) {
				if (isPaginationNavPressed()) {
					workPatterns = new ArrayList<ShiftPattern>(WorkRulesModelHandler.getShiftPatterns(m_context, getPageIDs()));
				} else {
					PaginationPair<ShiftPattern> shiftPatternData =
							WorkRulesModelHandler.getAvailableShiftPatternsForEmployees(m_context,
									employeeIDs, viewStartDate, viewEndDate, selectedEmployees.iterator().next().getEmployeeType(),
									selectedSchedulingPeriod, m_list.getHeader().isAscending(), getPageIDSize(),
									getPagePaginationIndex(), m_list.getSortComparators(), assignedWorkPatternIDs, getTextToFind(itemToFind));
					workPatterns = new ArrayList<ShiftPattern>(shiftPatternData.getValues());
					initPaginationIDs(shiftPatternData.getIDs());
				}
			}
		} else {
			throw new Exception(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_ASSIGNMENT_RULE_POPUP_NO_EMPLOYEES_ERROR));
		}

		Collections.sort(workPatterns, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));

		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		Collection<ID> campaignIDs = new HashSet<ID>();
		Collection<ID> activityIDs = new HashSet<ID>();
		Collection<ID> empTypeIDs = new HashSet<ID>();

		for (ShiftPattern sp : workPatterns) {
			if (sp.getOrganizationID() != null) {
				orgIDs.add(sp.getOrganizationID());
			}
			if (sp.getCampaignID() != null) {
				campaignIDs.add(sp.getCampaignID());
			}
			if (sp.getEmployeeTypeID() != null) {
				empTypeIDs.add(sp.getEmployeeTypeID());
			}
			for (Shift shift : sp.getShiftPatternShiftAttributes().keySet()) {
				if (shift.getOrganizationID() != null) {
					orgIDs.add(shift.getOrganizationID());
				}
				if (shift.getCampaignID() != null) {
					campaignIDs.add(shift.getCampaignID());
				}
				if (shift.getActivityID() != null) {
					activityIDs.add(shift.getActivityID());
				}
				for (ShiftEvent se : shift.getShiftEvents()) {
					if (se.getOrganizationID() != null) {
						orgIDs.add(se.getOrganizationID());
					}
					if (se.getCampaignID() != null) {
						campaignIDs.add(se.getCampaignID());
					}
					if (se.getActivityID() != null) {
						activityIDs.add(se.getActivityID());
					}
				}
			}
			for (VTOEvent vtoe : sp.getVTOEventStartTimes().keySet()) {
				if (vtoe.getOrganizationID() != null) {
					orgIDs.add(vtoe.getOrganizationID());
				}
				if (vtoe.getCampaignID() != null) {
					campaignIDs.add(vtoe.getCampaignID());
				}
				if (vtoe.getActivityID() != null) {
					activityIDs.add(vtoe.getActivityID());
				}
			}
			for (ShiftOTExtension sote : sp.getOTExtensionStartTimes().keySet()) {
				if (sote.getOrganizationID() != null) {
					orgIDs.add(sote.getOrganizationID());
				}
				if (sote.getCampaignID() != null) {
					campaignIDs.add(sote.getCampaignID());
				}
				if (sote.getActivityID() != null) {
					activityIDs.add(sote.getActivityID());
				}
			}
		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);
		empTypeMap = EmployeeTypeModelHandler.getEmployeeTypesByIDs(m_context, empTypeIDs);
	}

	public void setSelectedIDs(String selectedIDs) {
		//This is a hack in order to prevent a rare bug in which the user selects
		//enough work patterns so that the sum of the selected work pattern IDs causes the URL
		//to get truncated (IE supports a max of 2048 characters in the URL).  We will
		//remove the last (possibly truncated) ID in this case, in order to prevent the
		//system from interpreting that as a different ID.  Missing the truncated IDs isn't
		//horrible, since the only drawback is the user will be able to reselect the same work pattern
		//again which will overwrite the existing selection.
		if (selectedIDs != null && selectedIDs.length() > 1900) {
			selectedIDs = selectedIDs.substring(0, selectedIDs.lastIndexOf(","));
		}
		this.assignedWorkPatternIDs = new HashSet<ID>(StringUtil.parseIDStringToCollection(selectedIDs));
	}

	/**
	 * Checks to see if the collection of employees are all the same type.  If they are not,
	 * then we can't load any work patterns as the ones we add need to be valid for all selected
	 * employees.
	 *
	 * @param employees
	 * @return
	 */
	private boolean areEmployeesOfSameType(Collection<Employee> employees) {
		if (employees == null || employees.size() == 0) {
			return false;
		}
		ID employeeType = employees.iterator().next().getEmployeeType();
		for (Employee employee : employees) {
			if (employee.getEmployeeType().equals(employeeType) == false) {
				return false;
			}
		}
		return true;
	}
