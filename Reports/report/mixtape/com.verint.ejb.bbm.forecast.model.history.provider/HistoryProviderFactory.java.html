package com.verint.ejb.bbm.forecast.model.history.provider;

import com.bluepumpkin.common.datatypes.TimeContext;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.workload.model.Media;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.verint.ejb.bbm.forecast.model.history.service.NotesDataService;
import com.verint.ejb.bbm.forecast.model.history.service.OutboundHistoryDataService;
import com.verint.ejb.bbm.forecast.model.history.service.VolumeAndAhtHistoryDataService;

public class HistoryProviderFactory {

	public static IHistoryProvider createHistoryProvider(Campaign campaign, Media media, Queue queue,
			TimeContext timeContext, boolean isWhatIf) throws BbmEJBCreateException {
		if (Media.isMediaImmediate(media.getID()) || Media.isMediaDeferred(media.getID())) {
			VolumeAndAhtHistoryDataService historyDataService =
					new VolumeAndAhtHistoryDataService(campaign, isWhatIf);
			NotesDataService notesDataService = new NotesDataService();
			return new QueueHistoryTimeSeriesProvider(queue, timeContext, historyDataService, notesDataService);
		}
		if (Media.isMediaOutbound(media.getID())) {
			OutboundHistoryDataService historyDataService = new OutboundHistoryDataService(campaign, isWhatIf);
			NotesDataService notesDataService = new NotesDataService();
			return new OutboundHistoryProvider(queue, timeContext, historyDataService, notesDataService);
		}
		throw new IllegalArgumentException("HistoryProviderFactory - Media ID: " + media.getID() + " is not supported");
	}
}
