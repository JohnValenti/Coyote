import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class SimpleStartTimePreference extends ValueObjectBase
{
    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new SimpleStartTimePreferenceFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }
    
    public SimpleStartTimePreference() {
        super();
    }

    public short getWeekDay()
    {
        return getFieldValueShort(SimpleStartTimePreferenceFieldInfo.SIMPLESTARTTIME_WEEKDAY);
    }

    public short getPreferenceLevel()
    {
        return getFieldValueShort(SimpleStartTimePreferenceFieldInfo.SIMPLESTARTTIME_PREFERENCELEVEL);
    }
        
    public int getStartOffset()
    {
        return getFieldValueInt(SimpleStartTimePreferenceFieldInfo.SIMPLESTARTTIME_STARTOFFSET);
    }
    
    public int getEndOffset()
    {
        return getFieldValueInt(SimpleStartTimePreferenceFieldInfo.SIMPLESTARTTIME_ENDOFFSET);
    }
    
    public void setWeekDay(short sValue)
    {
        setFieldValue(SimpleStartTimePreferenceFieldInfo.SIMPLESTARTTIME_WEEKDAY, sValue);
    }

    public void setPreferenceLevel(short sValue)
    {
        setFieldValue(SimpleStartTimePreferenceFieldInfo.SIMPLESTARTTIME_PREFERENCELEVEL, sValue);
    }
        
    public void setStartOffset(int iValue)
    {
        setFieldValue(SimpleStartTimePreferenceFieldInfo.SIMPLESTARTTIME_STARTOFFSET, iValue);
    }
    
    public void setEndOffset(int iValue)
    {
        setFieldValue(SimpleStartTimePreferenceFieldInfo.SIMPLESTARTTIME_ENDOFFSET, iValue);
    }
 
}
