import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class SchedulePrefPeriod extends ValueObjectBase
{
    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new SchedulePrefPeriodFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }
        
    public SchedulePrefPeriod() {
        super();
    }

    public ID getEmployeeID()
    {
        return getFieldValueID(SchedulePrefPeriodFieldInfo.SCHEDPREFPERIOD_EMPLOYEEID);
    }
    
    public Date getStartTime()
    {
        return getFieldValueDate(SchedulePrefPeriodFieldInfo.SCHEDPREFPERIOD_STARTTIME);
    }

    public Date getEndTime()
    {
        return getFieldValueDate(SchedulePrefPeriodFieldInfo.SCHEDPREFPERIOD_ENDTIME);
    }
    
    public boolean isOverridden() 
    {
        return getFieldValueBoolean(SchedulePrefPeriodFieldInfo.SCHEDPREFPERIOD_ISOVERRIDDEN);
    }
    
    
}
