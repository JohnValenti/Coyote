package com.verint.web.fs.optimization.forecasting;

import java.util.Collection;

import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.workload.model.Media;
import com.verint.web.fs.optimization.Response;

public class ForecastResponse<T> extends Response<T> {

	private final boolean useWeightedSmoothingInForecast;

	public ForecastResponse(Media media, Campaign campaign, SchedulingPeriod sp, T queueDto,
			Collection<CampaignHOO> hoo, boolean useWeightedSmoothingInForecast) {
		super(media, campaign, sp, queueDto, hoo);

		this.useWeightedSmoothingInForecast = useWeightedSmoothingInForecast;
	}

	public boolean isUseWeightedSmoothingInForecast() {
		return useWeightedSmoothingInForecast;
	}
}
