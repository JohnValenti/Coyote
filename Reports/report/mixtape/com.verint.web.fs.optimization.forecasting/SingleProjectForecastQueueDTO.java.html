package com.verint.web.fs.optimization.forecasting;

import java.io.Serializable;
import java.util.Collections;
import java.util.List;

import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastTraceCube;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.verint.ejb.bbm.forecast.model.ForecastProfile;
import com.verint.ejb.bbm.forecast.model.ForecastProfileProject;
import com.verint.web.fs.optimization.SingleQueueDTO;

public class SingleProjectForecastQueueDTO extends SingleForecastQueueDTO implements Serializable {

	private static final long serialVersionUID = -6260243725486658336L;
	private final List<ForecastProfile> m_forecastProfiles;
	
	private List<ForecastProfileProject> m_projectForecasts; 
	
	public SingleProjectForecastQueueDTO(Queue queue, SPQueue spQueue, List<ForecastProfile> forecastProfiles,ForecastTraceCube timeSeriesTraceCube) {
		super(queue, spQueue,forecastProfiles,timeSeriesTraceCube);
		m_forecastProfiles =forecastProfiles;
	}

	
	public SingleProjectForecastQueueDTO(Queue queue, SPQueue spQueue, List<ForecastProfile> forecastProfiles, 
			List<ForecastProfileProject> forecast,ForecastTraceCube timeSeriesTraceCube) {
		this(queue, spQueue, forecastProfiles,timeSeriesTraceCube);
		m_projectForecasts = forecast;
	}
	
	public List<ForecastProfileProject> getProjectForecast() {
		return m_projectForecasts;
	}
	
}
