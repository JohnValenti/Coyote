package com.bluepumpkin.ejb.am.mytime.model;

import java.io.Externalizable;
import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;

/**
 * Title:        Blue Pumpkin Software Activity Management
 * Description:

 * @version 1.0

/**
 * This is a specialized value object
 * It contains a list of Activities for a given ActivityCategory
 * and a list of ActivityCategories.
 */
public class MyTimeEvent implements Externalizable {

	private ID        m_activityId = null;
	private String    m_activityName = null;
	private ID        m_activityCategoryId = null;
	private String    m_activityCategoryName = null;
	private String    m_activityColor = null;
	private String    m_status;
	private String    m_notes;
	private int       m_duration;
	private Date      m_startTime;
	private int       m_punchTimeCode;
	private int       m_eventType;//Added for QA99911

	public void setActivity(Activity activity) {
		this.m_activityId = activity.getID();
		this.m_activityName = activity.getName();
	}

	public String getActivityColor() { return m_activityColor; }
	public void setActivityColor(String activityColor) {
		this.m_activityColor = activityColor;
	}

	public void setActivityCategory(ActivityCategory activityCategory) {
		this.m_activityCategoryId = activityCategory.getID();
		this.m_activityCategoryName = activityCategory.getName();
	}

	public ID getActivityId() { return m_activityId; }
	public void setActivityId(ID activityId) {
		this.m_activityId = activityId;
	}

	public ID getActivityCategoryId() { return m_activityCategoryId; }
	public void setActivityCategoryId(ID activityCategoryId) {
		this.m_activityCategoryId = activityCategoryId;
	}

	public String getActivityName() { return m_activityName; }
	public void setActivityName(String activityName) {
		this.m_activityName = activityName;
	}

	public String getActivityCategoryName() {
		return m_activityCategoryName;
	}
	public void setActivityCategoryName(String activityCategoryName) {
		this.m_activityCategoryName = activityCategoryName;
	}

	public String getNotes() { return m_notes; }
	public void setNotes(String notes) {
		this.m_notes = notes;
	}

	public String getStatus() { return m_status; }
	public void setStatus(String status) {
		this.m_status = status;
	}

	public long getDuration() { return m_duration; }
	public void setDuration(int duration) {
		this.m_duration = duration;
	}

	public Date getStartTime() { return m_startTime; }
	public void setStartTime(Date startTime) {
		this.m_startTime = startTime;
	}

	public int getPunchTimeCode() { return m_punchTimeCode; }
	public void setPunchTimeCode(int punchTimeCode) {
		this.m_punchTimeCode = punchTimeCode;
	}
	
	//Changes start for QA99911
	public int getEventType() { return m_eventType; }
	public void setEventType(int m_eventType) {
		this.m_eventType = m_eventType;
	}
	//Changes end for QA99911

	/**
	*  constructors
	*/
	public MyTimeEvent() {
	}
	public MyTimeEvent(Activity activity, ActivityCategory activityCategory) {
		this.setActivity(activity);
		this.setActivityCategory(activityCategory);
	}
	public MyTimeEvent(Activity activity) {
		this.setActivity(activity);
	}
	public MyTimeEvent(ActivityCategory activityCategory) {
		this.setActivityCategory(activityCategory);
	}

	/**
	* print object
	*/
	public String toString () {
		StringBuffer buffer = new StringBuffer(256);
		buffer.append("\n\nMyTimeEvent.m_activityId: "+m_activityId);
		buffer.append("\nMyTimeEvent.m_activityName: "+m_activityName);
		buffer.append("\nMyTimeEvent.m_activityCategoryId: "+m_activityCategoryId);
		buffer.append("\nMyTimeEvent.m_activityCategoryName: "+m_activityCategoryName);
		buffer.append("\nMyTimeEvent.m_activityColor: "+m_activityColor);
		buffer.append("\nMyTimeEvent.m_status: "+m_status);
		buffer.append("\nMyTimeEvent.m_notes: "+m_notes);
		buffer.append("\nMyTimeEvent.m_duration: "+m_duration);
		buffer.append("\nMyTimeEvent.m_startTime: "+m_startTime);
		buffer.append("\nMyTimeEvent.m_punchTimeCode: "+m_punchTimeCode);
		return buffer.toString();
	}
	/* (non-Javadoc)
	 * @see java.io.Externalizable#readExternal(java.io.ObjectInput)
	 */
	public void readExternal(ObjectInput in)
		throws IOException, ClassNotFoundException {		
		m_activityId = (ID)in.readObject();
		m_activityName = (String)in.readObject();
		m_activityCategoryId = (ID)in.readObject();
		m_activityCategoryName = (String)in.readObject();
		m_activityColor = (String)in.readObject();
		m_status = (String)in.readObject();
		m_notes = (String)in.readObject();
		m_duration = in.readInt();
		m_startTime = (Date)in.readObject();
		m_punchTimeCode = in.readInt();
		m_eventType = in.readInt();//Added for QA99911
	}

	/* (non-Javadoc)
	 * @see java.io.Externalizable#writeExternal(java.io.ObjectOutput)
	 */
	public void writeExternal(ObjectOutput out) throws IOException {
		out.writeObject(m_activityId);
		out.writeObject(m_activityName);
		out.writeObject(m_activityCategoryId);
		out.writeObject(m_activityCategoryName);
		out.writeObject(m_activityColor);
		out.writeObject(m_status);
		out.writeObject(m_notes);
		out.writeInt(m_duration);
		out.writeObject(m_startTime);
		out.writeInt(m_punchTimeCode);	
		out.writeInt(m_eventType);//Added for QA99911
	}

}
