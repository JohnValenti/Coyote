package com.bluepumpkin.ejb.am.mytime.model;

import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;

/**
 * Title:        Blue Pumpkin Software Activity Management

/**
 * This is a specialized value object
 * It contains a list of Activities for a given ActivityCategory
 * and a list of ActivityCategories.
 */
public class ActivityGroup implements java.io.Serializable {

	public final static int ALL_ACTIVITYCATEGORIES = 0;
	public final static int MYTIME_ACTIVITYCATEGORIES = 1;

	public final static int START_SHIFT = 1;
	public final static int END_SHIFT = 2;

	private Collection activities = null;
	private Collection activityCategories = null;
	private ID  activityCategoryId = null;

	public Collection getActivities() {
		return activities;
	}
	public void setActivities(Collection activities) {
		this.activities = activities;
	}

	public Collection getActivityCategories() {
		return activityCategories;
	}
	public void setActivityCategories(Collection activityCategories) {
		this.activityCategories = activityCategories;
	}

	public ActivityGroup( Collection activities, Collection activityCategories) {
		this.activities = activities;
		this.activityCategories = activityCategories;
	}

	public ActivityGroup(ID  activityCategoryId, Collection activities, Collection activityCategories) {
		this.activities = activities;
		this.activityCategories = activityCategories;
		this.activityCategoryId = activityCategoryId;
	}

	/**
	* print object
	*/
	public String toString () {
		StringBuffer buffer = new StringBuffer();
		buffer.append("\nActivityGroup.activities: "+activities);
		buffer.append("\nActivityGroup.activityCategories: "+activityCategories);
		buffer.append("\nActivityGroup.activityCategoryId: "+activityCategoryId);

		return buffer.toString();
	}
}
