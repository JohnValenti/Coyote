	* <P>Params  Collection of IDs
	* returns Collection Of ShiftSwapItem Objects
	* <P>
	*  @return  Collection Of ShiftSwapItem Objects
	*/
    public Collection getShiftSwapItemsByIDs(Collection ids) throws Exception {
          Collection items = new ArrayList();
          Iterator it = ids.iterator();
		  while (it.hasNext()){
               items.add(getShiftSwapItemByID((ID) it.next()));
          }
          return items;
    }

    /**
	* <B>getShiftSwapItemByID</B>

	* <P>
	*  @return   ShiftSwapItem Object
	*
	*/
    public ShiftSwapItem getShiftSwapItemByID(ID id) throws Exception {
        ShiftSwapItem item = new ShiftSwapItem();
        //Find shiftswapID & time order from SHIFTSWAPREQUESTITEM table
		String strQuery = "select EMPLOYEEID, STARTTIME, ENDTIME, EXPIRATIONDATE, ISNEGOTIATIONREQUIRED, SHIFTTYPE, SHIFTID " +
			   "from SHIFTSWAPITEM " +
			   "where ID = " + id;

	    m_cat.debug("strQuery: " + strQuery);

        //Set query results into shiftswap item object
//		try {
			JdmoRowset rs = m_dmo.createRowset(strQuery );
			while (rs.next())
			{
				item.setEmployeeID((ID)rs.getObject(0));
                item.setStartDate((Date)rs.getObject(1));
                item.setEndDate((Date)rs.getObject(2));
                item.setExpirationDate((Date)rs.getObject(3));
                item.setNegotiation(rs.getBoolean(4) );
                item.setShiftType(rs.getString(5));
                item.setShiftID((ID)rs.getObject(6));
			}
//		} c_atch(Exception e) {
//			m_cat.error("ShiftSwapItemDAO:getShiftSwapItemByID(ID id): " + e.getMessage() );
//		}
          return item;
    }

    public Collection createShiftSwapItems(Collection ssItems)
           throws BbmCreateException{
            Iterator it = ssItems.iterator();
            Collection ids = new ArrayList();
            ID id = null;
		    while (it.hasNext()){
