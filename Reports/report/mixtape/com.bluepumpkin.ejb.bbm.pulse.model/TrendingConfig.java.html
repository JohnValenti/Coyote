package com.bluepumpkin.ejb.bbm.pulse.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**
 *
 * This value object is to store TrendingLogic definition, which is associated with SP 
 * @author ssong
 *
 */
public class TrendingConfig extends ValueObjectBase {
	
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public static final short APPWIN_ENDOFDAY = 0;
	public static final short APPWIN_ENDOFWEEK = 1;
	public static final short APPWIN_OFFSET = 2;
    public static final short APPWIN_ENDOFPERIOD = 3;
	
	public static final short RATE_NO_DROPOFF = 0;
	public static final short RATE_LINEAR_DROPOFF = 1;
	public static final short RATE_DIMINISH_QUICK = 2;
	public static final short RATE_CARRIED_THRU = 3;
	
	private static FieldInfo m_fieldInfo = new TrendingConfigFieldInfo();
	private int MAX_SUPPORT_WINDOW = 2;
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public TrendingConfig() { super(); }

	public ID getID() {
		return getFieldValueID(TrendingConfigFieldInfo.TRENDINGCONFIG_ID);
	}
	
	public void setID(ID id) {
		setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_ID, id);
	}

	public ID getUserID() {
		return getFieldValueID(TrendingConfigFieldInfo.TRENDINGCONFIG_USERID);
	}
	
	public void setUserID(ID id) {
		setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_USERID, id);
	}
	
	public ID getSPID() {
		return getFieldValueID(TrendingConfigFieldInfo.TRENDINGCONFIG_SPID);
	}
	
	public void setSPID(ID id) {
		setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_SPID, id);
	}
	
	public void setApplyWindowType(short type) {
		setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_APPLYWINDOWTYPE, type);
	}
	
	public short getApplyWindowType() {
		return getFieldValueShort(TrendingConfigFieldInfo.TRENDINGCONFIG_APPLYWINDOWTYPE);
	}
	/**
	 * Set offset interval from now in 15 mins interval, 4 means one hour from now
	 * @param interval
	 */
	public void setApplyWindowOffset(int interval) {
		setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_APPLYWINDOWOFFSET, interval);
	}
	
	public int getApplyWindowOffset() {
		return getFieldValueInt(TrendingConfigFieldInfo.TRENDINGCONFIG_APPLYWINDOWOFFSET);
	}
	
	public int getMaxExclusionWindow() {
		return MAX_SUPPORT_WINDOW = 2;
	}
	
	public Date[] getExclusionToStart() {
		Date[] range = new Date[2];
		range[0] = getFieldValueDate(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINSTART1);
		range[1] = getFieldValueDate(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINEND1);
		return range;
	}
	
	public void setExclusionToStart(Date s1, Date s2) {
		setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINSTART1, s1);
		setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINEND1, s2);
	}
	
	public Date[][] getExclusionWindow() {		
		Date s2 = getFieldValueDate(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINSTART2);
		Date s3 = getFieldValueDate(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINSTART3);
		
		Date e2 = getFieldValueDate(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINEND2);
		Date e3 = getFieldValueDate(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINEND3);
		int lines = 0;
		if (s2 != null && e2 != null)
			lines++;
		if (s3 != null && e3 != null)
			lines++;
		Date[][] ranges = new Date[lines][2];
		lines = 0;
		if (s2 != null && e2 != null) {
			ranges[lines][0] = s2;
			ranges[lines++][1] = e2;
		}
		if (s3 != null && e3 != null) {
			ranges[lines][0] = s3;
			ranges[lines++][1] = e3;
		}
		return ranges;
	}
	
	/**
	 * only support two exclusion windows now
	 * @param excWin
	 */
	public void setExclusionWindow(Date[][] excWin) {
		for (int i=0; i<excWin.length && i<2; i++) {
			if (excWin[i] != null) {
				setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINSTART2+i, excWin[i][0]);
				setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_HSTEXCWINEND2+i, excWin[i][1]);			
			}
		}
	}
	
	public void setTrendingRate(short rate) {
		setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_TRENDINGRATE, rate);
	}
	
	public short getTrendingRate() {
		return getFieldValueShort(TrendingConfigFieldInfo.TRENDINGCONFIG_TRENDINGRATE);
	}
	
	public void saveProcess() {
		setFieldValue(TrendingConfigFieldInfo.TRENDINGCONFIG_LASTUPDATE, new Date());
	}
}
