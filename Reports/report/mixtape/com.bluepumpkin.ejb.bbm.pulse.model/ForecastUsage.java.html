	private static FieldInfo m_fieldInfo = new ForecastUsageFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public ForecastUsage() {

	// id
	public void setID(ID id) {
		setFieldValue(ForecastUsageFieldInfo.FORECASTUSAGE_ID, id);
	}	
	public ID getID() {
		return getFieldValueID(ForecastUsageFieldInfo.FORECASTUSAGE_ID);
	}
	
	// name
	public void setName(String name) {
		setFieldValue(ForecastUsageFieldInfo.FORECASTUSAGE_NAME, name);
	}
	public String getName() {
		return (String)getFieldValue(ForecastUsageFieldInfo.FORECASTUSAGE_NAME);
	}
	
	// description
	public void setDescription (String desc) {
		setFieldValue(ForecastUsageFieldInfo.FORECASTUSAGE_DESCRIPTION, desc);
	}
	public String getDescription() {
		return (String)getFieldValue(ForecastUsageFieldInfo.FORECASTUSAGE_DESCRIPTION);
	}
	
} // end of class
