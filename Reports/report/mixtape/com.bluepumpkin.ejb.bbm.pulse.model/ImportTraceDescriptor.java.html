package com.bluepumpkin.ejb.bbm.pulse.model;

import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.util.NumberFactory;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.timeseries.model.ActualTraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.util.TraceUtil;

/**
 * Title:        Blue Pumpkin Software Forecast&Schedule
 * Description:  ImportTraceDescriptor object, a descriptor object for Trace import functionality
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       Sheng Song
 * @version 2.0
 */

public class ImportTraceDescriptor extends TraceDescriptor {
//	static final long serialVersionUID = -6605727215494873032L;

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// Additional parameter with default values
	private int linesToIgnore = 1;
	protected LocalDate date;

	public ImportTraceDescriptor(String targetFile, TimeZone tz, ID id, String name, LocalDate startDate) {
		super(targetFile, tz, id, name);
		date = startDate;
		pickInitColumns();
	}

	protected void pickInitColumns() {
		// At least we need import based on Time column, default pos is 3
		// Date 2, Time 3, CV 5, SL 6, Abn 7, ASA 8, AHT 9, STF 10, OCU 11, BKG 12, FTE 13
		traceMap.put(DATE, NumberFactory.newShort((short)2));
		traceMap.put(TIME, NumberFactory.newShort((short)3));
		traceMap.put(CV, NumberFactory.newShort((short)5));
		traceMap.put(PCA, NumberFactory.newShort((short)6));
		traceMap.put(ABANDONMENT, NumberFactory.newShort((short)7));
		traceMap.put(ASA, NumberFactory.newShort((short)8));
		traceMap.put(AHT, NumberFactory.newShort((short)9));
		traceMap.put(STAFFING, NumberFactory.newShort((short)10));
		traceMap.put(OCCUPANCY, NumberFactory.newShort((short)11));
		traceMap.put(BACKLOG, NumberFactory.newShort((short)12));
		traceMap.put(FTE, NumberFactory.newShort((short)13));
		traceMap.put(DIALS, NumberFactory.newShort((short)14));
		traceMap.put(CONNECTS, NumberFactory.newShort((short)15));
		traceMap.put(RPC, NumberFactory.newShort((short)16));
		traceMap.put(RPCAHT, NumberFactory.newShort((short)17));
	}

	public void setLinesToIgnore(int lines) { linesToIgnore = lines; }
	public int getLinesToIgnore() { return linesToIgnore; }

	// If ImportDate is specified, it will be used as based date to import Trace
	public LocalDate getImportDate() { return date; }

	// If Date and StartDate are not specified, that's wrong
	// If StartDate is specified, it will overrides Date in the File
	// If StartDate is not specified, it will depends on Date column in File
	public boolean isValid() {
		if (getImportDate() == null) {
			if (getColumnPos(DATE) == null)
				return false;
		}
		return true;
	}

	public TraceCube getTraceCube() {
		try {
			return new ActualTraceCube(getQueueID(), null, null, getTraceTypes(), getTimeZone());
		} catch(Exception e) {
			return TraceUtil.META_ACTUAL;
		}
	}

	public boolean pickColumn(Short colType, short colPos) {
		// When import Abandonments with compatible mode, CV is required for conversion
		return super.pickColumn(colType, colPos);
	}

	public boolean isSupportedTraceType(Short colType) {
		short type = colType.shortValue();
		if (getColumnPos(colType) == null)
			return false;
		switch (type) {
			case Trace.ABANDONMENT:
				return true;
			case Trace.ASA:
				return true;
			case Trace.AHT:
				return true;
			case Trace.BACKLOG:
				return true;
			case Trace.CV:
				return true;
			case Trace.OCCUPANCY:
				return true;
			case Trace.PCA:
				return true;
			case Trace.STAFFING:
				return true;
			case Trace.FTE:
				return true;
			case Trace.DIALS:
				return true;
			case Trace.CONNECTS:
				return true;
			case Trace.RPC:
				return true;
			case Trace.RPCAHT:
				return true;
			case Trace.VH:
				return (Trace.isVolumeHandledEnabled());//returns true only if Volume handled is enabled
			case Trace.ASTAFFING:
				return true;
			case Trace.AFTE:
				return true;
			case Trace.NETSTAFFING:
				return true;
			default:
				return false;
		}
	}

	public TraceCube getTraceCube(Date gmtDay) {
		Date gmtNextDay = new Date(gmtDay.getTime() + TimeZoneUtil.DAY_IN_MILLISECONDS - 1000);
		try {
			return new ActualTraceCube(getQueueID(), gmtDay, gmtNextDay, getTraceTypes(), getTimeZone());
		} catch(Exception e) {
			return TraceUtil.META_ACTUAL;
		}
	}

	public TraceCube getTraceCube(Date start, Date end) {
		try {
			return new ActualTraceCube(getQueueID(), start, end, getTraceTypes(), getTimeZone());
		} catch(Exception e) {
			return TraceUtil.META_ACTUAL;
		}
	}
}
