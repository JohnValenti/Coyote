package com.bluepumpkin.ejb.bbm.pulse.model;

import java.util.Date;
import java.util.Locale;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.util.NumberFactory;
import com.bluepumpkin.ejb.bbm.timeseries.model.ActualTraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.util.TraceUtil;

/**
 * Title:        Blue Pumpkin Software Forecast&Schedule
 * Description:  ExportTraceDescriptor object, a descriptor object for Trace export functionality
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       Sheng Song
 * @version 2.0
 */

public class ExportTraceDescriptor extends TraceDescriptor {
//    static final long serialVersionUID = 256009729234716130L;

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	protected LocalDate sDate;
	protected LocalDate eDate;

	private ID acdID;
	private String acdName;

	// For export column name can be locale based
	protected Locale locale;

	public ExportTraceDescriptor(String targetFile, TimeZone tz, ID id, String name, LocalDate startDate, LocalDate endDate) {
		super(targetFile, tz, id, name);
		sDate = startDate;
		eDate = endDate;
		pickInitColumns();
	}

	protected void pickInitColumns() {
		// We will export those columns:
		// Date 2, Time 3, CV 5, SL 6, Abn 7, ASA 8, AHT 9, STF 10, OCU 11, BKG 12, FTE 13
		// With Queue Name 1, Interval 4
		traceMap.put(DATE, NumberFactory.newShort((short)2));
		traceMap.put(TIME, NumberFactory.newShort((short)3));
		traceMap.put(CV, NumberFactory.newShort((short)5));
		traceMap.put(PCA, NumberFactory.newShort((short)6));
		traceMap.put(ABANDONMENT, NumberFactory.newShort((short)7));
		traceMap.put(ASA, NumberFactory.newShort((short)8));
		traceMap.put(AHT, NumberFactory.newShort((short)9));
		traceMap.put(STAFFING, NumberFactory.newShort((short)10));
		traceMap.put(OCCUPANCY, NumberFactory.newShort((short)11));
		traceMap.put(BACKLOG, NumberFactory.newShort((short)12));
		traceMap.put(FTE, NumberFactory.newShort((short)13));
		traceMap.put(DIALS, NumberFactory.newShort((short)14));
		traceMap.put(CONNECTS, NumberFactory.newShort((short)15));
		traceMap.put(CRATE, NumberFactory.newShort((short)16));
		traceMap.put(RPC, NumberFactory.newShort((short)17));
		traceMap.put(RPCRATE, NumberFactory.newShort((short)18));
		traceMap.put(RPCAHT, NumberFactory.newShort((short)19));

	}

	public LocalDate getStartLocalDate() { return sDate; }
	public LocalDate getEndLocalDate() { return eDate; }
	public Date getStartDate() { return sDate.getTime(getTimeZone()); }
	public Date getEndDate() { return eDate.getTime(getTimeZone()); }

	public void setACDID(ID id) { acdID = id; }
	public ID getACDID() { return acdID; }
	public void setACDName(String name) { acdName = name; }
	public String getACDName() { return acdName; }

	public void setLocale(Locale locale) { this.locale = locale; }
	public Locale getLocale() {
		if (locale == null)
			return Locale.getDefault();
		return locale;
	}

	public TraceCube getTraceCube() {
		try {
			return new ActualTraceCube(getQueueID(), null, null, getTraceTypes(), getTimeZone());
		} catch(Exception e) {
			return TraceUtil.META_ACTUAL;
		}
	}

	public boolean isSupportedTraceType(Short colType) {
		short type = colType.shortValue();
		if (getColumnPos(colType) == null)
			return false;
		switch (type) {
			case Trace.ABANDONMENT:
				return true;
			case Trace.ASA:
				return true;
			case Trace.AHT:
				return true;
			case Trace.BACKLOG:
				return true;
			case Trace.CV:
				return true;
			case Trace.OCCUPANCY:
				return true;
			case Trace.PCA:
				return true;
			case Trace.STAFFING:
				return true;
			case Trace.FTE:
				return true;
			case Trace.DIALS:
				return true;
			case Trace.CONNECTS:
				return true;
			case Trace.CRATE:
				return true;
			case Trace.RPC:
				return true;
			case Trace.RPCRATE:
				return true;
			case Trace.RPCAHT:
				return true;
			case Trace.VH:
				return (Trace.isVolumeHandledEnabled());//returns true only if Volume handled is enabled
			case Trace.ASTAFFING:
				return true;
			case Trace.AFTE:
				return true;
			case Trace.NETSTAFFING:
				return true;
			default:
				return false;
		}
	}

}
