	public int compareTo(Object o) {
		if (o != null && o instanceof PulseNoteBO) {
			PulseNoteBO note = (PulseNoteBO) o;
			ID id = getID();
			ID noteId = note.getID();
			if (id == null && noteId == null) return -1;
			if (id == null && noteId != null) return -1;
			if (id != null && noteId == null) return 1;
			return id.compareTo(noteId);
		}
		return -1;
	}
	
	public Object clone() throws CloneNotSupportedException {
		PulseNoteBO clone = new PulseNoteBO();
		clone.setID(this.getID());
		clone.setActionType(this.getActionType());
		clone.setCreatedBy(this.getCreatedBy());
		clone.setCreatedDate((Date)this.getCreatedDate().clone());
		clone.setModifiedBy(this.getModifiedBy());
		clone.setModifiedDate((Date)this.getModifiedDate().clone());
		clone.setNote(this.getNote());
		clone.setStartTime((Date) this.getStartTime().clone());
		clone.setEndTime((Date) this.getEndTime().clone());
		clone.setExcludeFromForeCast(this.getExcludeFromForeCast() != null ? this.getExcludeFromForeCast() : ForecastUsage.NONE);
		
		Map cloneQs = new HashMap();
		Map<ID, Queue> thisQs = this.getQueues();
		Set<ID> keys = thisQs.keySet();
		Iterator<ID> keyIt = keys.iterator();
		while (keyIt.hasNext()) {
			ID queueId = keyIt.next(); 
			Queue queue = thisQs.get(queueId);
			cloneQs.put(new ID(queueId), queue.clone());
		}
		clone.setQueues(cloneQs);
		return clone;
	}

	/**
	 * Returns a string consisting of an alphabetized list of
	 * queues.  Each queue name is separated by a carriage return
	 * (this method is used for displaying the queues in list form
	 * in the notes dialogs).
	 * @param scopedQueues - only queues belonging to this note which are also in 
	 *                       this collection are included in the output string. If
	 *                       scopedQueues is null, then all queues belonging to this
	 *                       note will be included.
	 */
	public String getFormattedQueueString(Collection scopedQueues) {
		if (m_queues == null) return "";
		Collection queues =  m_queues.values();
		if (queues == null) return "";
		Iterator qIt = queues.iterator();
		ArrayList qNames = new ArrayList();
		while (qIt.hasNext()) {
			Queue q = (Queue) qIt.next();
			if (scopedQueues == null ||
					(scopedQueues != null && scopedQueues.contains(q))) {
				qNames.add(q.getName());
			}
		}					
		StringBuffer sb = new StringBuffer();					
		Collections.sort(qNames);
		Iterator namesIt = qNames.iterator();
		while (namesIt.hasNext()) {
			sb.append((String) namesIt.next());
			if (namesIt.hasNext())
				sb.append("\r\n");
		}
		return sb.toString();
	}
