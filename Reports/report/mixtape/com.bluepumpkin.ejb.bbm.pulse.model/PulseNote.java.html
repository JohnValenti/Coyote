	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new PulseNoteFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public PulseNote() { super(); }

	// id
	public void setID(ID id) {
		setFieldValue(PulseNoteFieldInfo.PULSENOTE_ID, id);
	}	
	public ID getID() {
		return getFieldValueID(PulseNoteFieldInfo.PULSENOTE_ID);
	}
	
	//created date
	public void setCreatedDate(Date createdDate) {
		setFieldValue(PulseNoteFieldInfo.PULSENOTE_CREATED_DATE, createdDate);
	}
	public Date getCreatedDate() {
		return getFieldValueDate(PulseNoteFieldInfo.PULSENOTE_CREATED_DATE);
	}
	
	// created by
	public void setCreatedBy (String name) {
		setFieldValue(PulseNoteFieldInfo.PULSENOTE_CREATED_BY, name);
	}
	public String getCreatedBy() {
		return getFormattedFieldValue(PulseNoteFieldInfo.PULSENOTE_CREATED_BY);
	}
	
	// modified date
	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(PulseNoteFieldInfo.PULSENOTE_MODIFIED_DATE, modifiedDate);
	}
	public Date getModifiedDate() {
		return getFieldValueDate(PulseNoteFieldInfo.PULSENOTE_MODIFIED_DATE);
	}
	
	// modified by
	public void setModifiedBy (String name) {
		setFieldValue(PulseNoteFieldInfo.PULSENOTE_MODIFIED_BY, name);
	}
	public String getModifiedBy() {
		return getFormattedFieldValue(PulseNoteFieldInfo.PULSENOTE_MODIFIED_BY);
	}

	// starttime	
	public void setStartTime(Date time) {
		setFieldValue(PulseNoteFieldInfo.PULSENOTE_STARTTIME, time);
	}
	public Date getStartTime() {
		return getFieldValueDate(PulseNoteFieldInfo.PULSENOTE_STARTTIME);
	}
	
	// endtime	
	public void setEndTime(Date time) {
		setFieldValue(PulseNoteFieldInfo.PULSENOTE_ENDTIME, time);
	}
	public Date getEndTime() {
		return getFieldValueDate(PulseNoteFieldInfo.PULSENOTE_ENDTIME);
	}
	
	// forecastUsageId
	public ID getForecastUsageId () {
		return getFieldValueID(PulseNoteFieldInfo.PULSENOTE_FORCEASTUSAGE_ID);
	}
	
	public void setForecastUsageId (ID fu) {
		setFieldValue(PulseNoteFieldInfo.PULSENOTE_FORCEASTUSAGE_ID, fu);
	}
	
	// note
	public void setNote(String note) {
		setFieldValue(PulseNoteFieldInfo.PULSENOTE_NOTE, note);
	}
	public String getNote() {
		return getFormattedFieldValue(PulseNoteFieldInfo.PULSENOTE_NOTE);
