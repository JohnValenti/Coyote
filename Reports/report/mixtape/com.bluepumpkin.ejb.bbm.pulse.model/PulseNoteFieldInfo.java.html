import com.bluepumpkin.common.datatypes.PrimitiveTypes;

public final class PulseNoteFieldInfo extends FieldInfo {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// field id constants
	public static final int PULSENOTE_ID 				= 0;
	public static final int PULSENOTE_CREATED_DATE 		= 1;
	public static final int PULSENOTE_CREATED_BY 		= 2;
	public static final int PULSENOTE_MODIFIED_DATE 	= 3;

	public static final int PULSENOTE_FORCEASTUSAGE_ID	= 7;
	public static final int PULSENOTE_NOTE 				= 8;
	
	private static final int PULSENOTE_NUMOFFIELDS 		= 9;

	private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[PULSENOTE_NUMOFFIELDS];
	private static String m_strTableName = "PULSENOTE";

	// static initialization
	static {
		m_arrFieldInfo[PULSENOTE_ID] 				= new FieldInfoEntry("ID", PrimitiveTypes.ID);
		m_arrFieldInfo[PULSENOTE_CREATED_DATE] 		= new FieldInfoEntry("CREATEDDATE", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[PULSENOTE_CREATED_BY] 		= new FieldInfoEntry("CREATEDBY", PrimitiveTypes.STRING);
		m_arrFieldInfo[PULSENOTE_MODIFIED_DATE] 	= new FieldInfoEntry("MODIFIEDDATE", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[PULSENOTE_MODIFIED_BY] 		= new FieldInfoEntry("MODIFIEDBY", PrimitiveTypes.STRING);
		m_arrFieldInfo[PULSENOTE_STARTTIME] 		= new FieldInfoEntry("STARTTIME", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[PULSENOTE_ENDTIME] 			= new FieldInfoEntry("ENDTIME", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[PULSENOTE_FORCEASTUSAGE_ID]	= new FieldInfoEntry("FORECASTUSAGEID", PrimitiveTypes.ID);
		m_arrFieldInfo[PULSENOTE_NOTE] 				= new FieldInfoEntry("NOTE", PrimitiveTypes.STRING);				
	}

	public PulseNoteFieldInfo() {
		super();
	}

	// implement abstract methods in parent class
	public String getTableName() { return m_strTableName; }
	public int getIDFieldIndex() { return PULSENOTE_ID; }
	public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }

} // end of class
