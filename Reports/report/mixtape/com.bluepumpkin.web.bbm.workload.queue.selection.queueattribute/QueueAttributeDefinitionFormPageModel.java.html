    private DynamicMultiColListWithToolbarPC initializeCustomStatusMappingTable(
    	    RequestContext context) {

    	DynamicMultiColListWithToolbarPC table = new DynamicMultiColListWithToolbarPC(
    		context);
    	table.setName(QUEUEATTRIB_DROPDOWN_TABLENAME);

    	// buttons
    	table.setIsMoveUpButtonShown(false);
    	table.setIsMoveDownButtonShown(false);
    	table.initForDisplay();

    	DynamicMultiColListPC innerTable = table.getDynamicMultiColList();
    	innerTable.setTemplateNodeData(createTemplateRowForStatusMapping(QUEUEATTRIB_DROPDOWN_TABLE_ID));

    	TableHeaderPC header = innerTable.getHeader();
    	header.addColumnHeaderData("col1", i18n(m_bbmBundle, //$NON-NLS-1$
    		BbmWebBundleKey.QUEUE_ATTRIBUTE_DEFINITION_DISTINCT_DD));
    	header.addColumnHeaderData("", ""); // for hidden id //$NON-NLS-1$ //$NON-NLS-2$

    	return table;
        }
    
    private int nextCustomStatusRow = 0;
    private MultiColumnNodeData createTemplateRowForStatusMapping(
    	    String tabAttrID) {
    	DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData("_"); //$NON-NLS-1$

    	m_inputPCOne.reset(QUEUEATTRIB_DROPDOWN_VALUES, "", FormInputPC.TYPE_STRING, false); //$NON-NLS-1$ //$NON-NLS-2$

    	node.add(HtmlUtil.makeHiddenInput(tabAttrID + "_id", "_")); //$NON-NLS-1$ //$NON-NLS-2$

    	return node;
        }

	protected void loadData() throws Exception {
		try {

			ID queueAttribDefID = getQueueAttributeDefinitionID();
			if(queueAttribDefID != null) {
				queueAttribute = modelHandler.getQueueAttributeByID(m_context, queueAttribDefID);
			}
			Collection<QueueAttributeType> attribTypes = modelHandler.getQueueAttributeTypes(m_context);
			Iterator<QueueAttributeType> iter = attribTypes.iterator();

			while(iter.hasNext()) {
				QueueAttributeType type = iter.next();
				queueAttributeTypes.add(new StringsPair(type.getID().toString(), type.getName()));
				queueAttributeTypesMap.put(type.getId(), type.getName());
			}
			addExistingDropDownData(queueAttribute.getQueueAttributeDropDown());
			
			if(userEnteredDropDownElements != null) { 
				ArrayList<DefaultMultiColumnNodeData> nodes = new ArrayList<DefaultMultiColumnNodeData>();
				for(String s : userEnteredDropDownElements) {
					if(StringUtil.isEmptyOrWhiteSpace(s)) 
						continue;
					if(existingElements.contains(s))
						continue;
					DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData("_"); //$NON-NLS-1$

			    	m_inputPCOne.reset(QUEUEATTRIB_DROPDOWN_VALUES, "", FormInputPC.TYPE_STRING, false); //$NON-NLS-1$ //$NON-NLS-2$
			    	m_inputPCOne.setInputValue(s); //$NON-NLS-1$
			    	m_inputPCOne.setAllowAllChars(false);
			    	m_inputPCOne.setOnChangeJS(JSUtil.ON_CHANGE +" validateForDuplicateDDValues(this);");
			    	
			    	node.add(m_inputPCOne.getHtmlDisplay());
			    	node.add(HtmlUtil.makeHiddenInput(QUEUEATTRIB_DROPDOWN_TABLE_ID + "_id", "_")); //$NON-NLS-1$ //$NON-NLS-2$
			    	nodes.add(node);
				}
				customSchemaTable.getDynamicMultiColList().getListModel().addAll(nodes);
			}
			
			
			if(queueAttribute.getType() == null || (!QueueAttributeType.TYPE_DROPDOWN.equals(queueAttribute.getType()))) {
				Collection elements = customSchemaTable.getToolbar().getElementList();
				if(elements.size() > 0) {
					for (Iterator it = elements.iterator(); it.hasNext();) {
						Object obj = it.next();
						if (obj instanceof ToolbarElement) {
							((ToolbarElement)obj).setEnabled(false);
						}
					}
				}
			}
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
			throw ex;
		}
	}
	
	private void addExistingDropDownData(Collection<QueueAttributeDropDown> dd) {
		Iterator<QueueAttributeDropDown> it = dd.iterator();
		ArrayList tableRows = new ArrayList();
		
		while (it.hasNext()) {
			QueueAttributeDropDown row = it.next();

		    DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData(
			    row.getID());
		    existingElements.add(row.getValue().trim());
		    m_inputPCOne.reset("", "", FormInputPC.TYPE_STRING, false); //$NON-NLS-1$ //$NON-NLS-2$
		    m_inputPCOne.setInputFN(QUEUEATTRIB_DROPDOWN_VALUES); //$NON-NLS-1$
		    m_inputPCOne.setAllowAllChars(false);
		    m_inputPCOne.setOnChangeJS("validateForDuplicateDDValues(this);");
		    
		    
		    m_inputPCOne.setInputValue(row.getValue().trim());
		    node.add(m_inputPCOne.getHtmlDisplay());

		    node.add(HtmlUtil.makeHiddenInput(QUEUEATTRIB_DROPDOWN_TABLE_ID + "_id", row.getID())); //$NON-NLS-1$
		    tableRows.add(node);
		}

		customSchemaTable.getDynamicMultiColList().setListModel(tableRows);
	}

	public QueueAttributeDefintion getQueueAttribute(HttpServletRequest request) throws Exception {
		String[] dropDownValues = request.getParameterValues(QUEUEATTRIB_DROPDOWN_VALUES);
		// Keep track of duplicate dd values and display message as needed.
		// TODO: Doing it on client side will be better. Need to add JS validation for duplicat DD values.
		HashSet ddValuesSet = new HashSet();
		
		//System.out.println("Drop down values " + dropDownValues);
		
		QueueAttributeDefintion attrib = new QueueAttributeDefintion();
		if(!StringUtil.isEmptyOrWhiteSpace(request.getParameter(QUEUE_ATTRIB_ID)))
			attrib.setID(new ID(request.getParameter(QUEUE_ATTRIB_ID)));
		
		if(!StringUtil.isEmptyOrWhiteSpace(request.getParameter(QUEUEATTRIB_DEF_NAME_FN))) {
			attrib.setName(request.getParameter(QUEUEATTRIB_DEF_NAME_FN).trim());	
		}
		
		attrib.setDescription(request.getParameter(QUEUEATTRIB_DEF_DESC_FN).trim());

		attrib.setType(new ID(request.getParameter(QUEUEATTRIB_DEF_TYPE_FN)));
		attrib.setOrganizationId(new ID(getSelectedID()));
		
		if(attrib.getType().equals(new ID(3)) && dropDownValues != null) {
			String[] dropDownID = request.getParameterValues(QUEUEATTRIB_DROPDOWN_TABLE_ID + "_id");
			ArrayList<QueueAttributeDropDown> values = new ArrayList<QueueAttributeDropDown>();
			for(int x = 1; x < dropDownValues.length; x++) {
                if(StringUtil.isEmptyOrWhiteSpace(dropDownValues[x]))
                    continue;
				if(ddValuesSet.contains(dropDownValues[x].toLowerCase().trim())){
					// throw exception. And since exception is generic add page message.
					addPageMessage(Message.ERROR_TYPE, i18n(m_bbmBundle, BbmWebBundleKey.QUEUE_ATTRIBUTE_DUPLICATE_DD_VALUES));
					return null;
				}
				QueueAttributeDropDown dd = new QueueAttributeDropDown();
				if(StringUtil.isEmptyOrWhiteSpace(dropDownID[x]) || dropDownID[x].equalsIgnoreCase("_"))
					dd.setID(null);
				else
					dd.setID(new ID(dropDownID[x]));
				dd.setValue(dropDownValues[x].trim());
				
				values.add(dd);
				ddValuesSet.add(dropDownValues[x].toLowerCase().trim());
			}
			attrib.setQueueAttributeDropDown(values);
		}
		ID modID = m_bIsInUDFMode? QueueAttributeDefintion.EMP_USER_DEFFINED_FIELD_MODULE: QueueAttributeDefintion.DEFAULT_MODULE_ID;
		attrib.setModuleId(modID);
		return attrib;
	}

	 private void setDeletedRows(Collection deletedIDs) {
			ArrayList<ID> toBeDeleted = new ArrayList();
			if (deletedIDs != null)
			    for (Iterator it = deletedIDs.iterator(); it.hasNext();) {
				try {
				    ID deletedID = new ID((String) it.next());
				    toBeDeleted.add(deletedID);
				    //System.out.println("Deleting Row " + deletedID);
				} catch (RuntimeException e) {
				    // This is possible if a new row is deleted which is not
				    // yet saved.
				    // In this event, there wont be any ids.
				}
			    }
			if(toBeDeleted.size() <= 0) {
			    return;
			}
			
			    	try { 
			    		modelHandler.deleteDropDownValues(m_context, toBeDeleted);
			    	} catch (Exception e) {
						//addPageMessage(Message.WARNING_TYPE, i18n(m_bbmBundle, "Error deleting drop down values. Make sure it is not linked to any queues - Non localized")); //$NON-NLS-1$
						handleUnableToUpdateInfo(log, e);
					}
		    }
	 
	 private boolean isInUDFMode(){
		HttpServletRequest aRequest = m_context.getRequest();
		String pathInfo = aRequest.getPathInfo();
		Screen screen = m_context.getScreen();
		String screenName = (screen == null)? "": screen.getName();
		return ("udf_def".equals(pathInfo) ||
				"/udf_def".equals(pathInfo) ||
				"UDF_DEF_FORM".equals(screenName) );
	 }
