@SuppressWarnings("serial")
public class SPQueueAttributeFilterBoxPC extends CampaignSPFilterBoxPC {
	private static final short LIST_FILTER_MAX_WIDTH = (short)125;
	private static final String OUTER_TABLE_WIDTH = "305";
	
	private QueueAttributeFilterPickerPC attribPicker;

	private ID m_mediaID;
	
	public SPQueueAttributeFilterBoxPC(RequestContext context, String name) {
		super(context, name);

		attribPicker = new QueueAttributeFilterPickerPC(context);
		addSiblingComponent(attribPicker);
	}

	protected short getListBoxDisplayMaxWidth() {
		return LIST_FILTER_MAX_WIDTH;
	}
	
	/**
	 * Initialize Component For Display
	 */
	public void initForDisplay() {
		if (m_isInitializedForDisplay) return;
		else {
			super.initForDisplay();
			
			// Initialize Media List Properties.
			attribPicker.initFilter(true);

			//Setting width to match the other drop downs.
			attribPicker.getListbox().setMaxDisplayWidth(getListBoxDisplayMaxWidth());
			attribPicker.getListbox().getSelect().setAttribute("width", getListBoxDisplayMaxWidth() + "px");
		}
	}
	
	/**
	 * Initialize Component
	 *
	 * @param mediaList Media ID used to initialize Component.
	 *   if value is not null, it will be used as the only media Option
	 */
	public void init(Collection mediaList, boolean isDistributedCampaignEnabled, boolean isDistributedCampaignOnly) throws Exception {
		initSPFilter(isDistributedCampaignEnabled, isDistributedCampaignOnly);
		initMediaFilter(mediaList);
	}
	
	/**
	 * Initialize the Media Drop Down
	 */
	//TODO: it looks like this method doesn't do anything
	private void initMediaFilter(Collection<Media> mediaList) throws Exception {
		if (mediaList!=null && !mediaList.isEmpty()) {
			ID selectedID = getMediaID();
			ArrayList<OptionData> filters = new ArrayList<OptionData>(mediaList.size() + 1);
			
			// Add blank option.
			OptionData data = new OptionData("", BLANK_ID.toString(), false);
			filters.add(data);
			
			// Iteration through the media list and add the items.
			for (Iterator<Media> it = mediaList.iterator(); it.hasNext();) {
				Media media = it.next();
				ID mediaID = media.getID();
				data = new OptionData(media.getName(), mediaID.toString(), mediaID.equals(selectedID));
				filters.add(data);
			}
		}
	}
	
	public void setMediaID(ID id) {
		if (BLANK_ID.equals(id)) id=null;
		m_mediaID=id;
	}

	public ID getMediaID() {
		return m_mediaID;
	}
	
	public ID getQueueAttributeID() {
		return attribPicker.getQueueAttributeID();
	}
	
	protected String getOuterTableWidth() {
		return OUTER_TABLE_WIDTH;
	}
	/**
	 * Return Html Display
	 * Sankar Nair - Overriding this function to ensure that
	 * the alignment of selector pane is matching to Pulse applet 
	 * when both the Campaign picker and Org picker are shown. 
	 */
		
	/**
	 * Return Bottom Row : Media List Box
	 */
	public HtmlTableRow createBottomRow() {
		ArrayList rowcells = new ArrayList();
		
		Object[] args = new Object[1];
	    args[0] = m_context.getSystemManagerFactory().getVerticalizationManager().getVerticalizedString(m_context, DefaultVerticalizationManager.QUEUE);
	    String label = MessageFormat.format( m_localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.QUEUE_ATTRIBUTE_FILTER_PICKER_CAPTION), args);

	
		HtmlTableCell cell = new HtmlTableCell(label + ":&nbsp;");
		cell.setHeight("18");
		cell.setStyleClass("headerText");
        cell.setStyle("white-space:nowrap;");
		
        rowcells.add(cell);
        rowcells.add(attribPicker);
		
		return new HtmlTableRow(rowcells);
	}
}
