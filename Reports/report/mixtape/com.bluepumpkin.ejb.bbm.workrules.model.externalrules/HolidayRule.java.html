  public void initialize(LocalDate dtStartDate, LocalDate dtEndDate, Collection pOrganizations, Collection cWorkResourceIDs)
						throws WorkRuleException {
	m_hOrgHolidays = new HashMap();
	HolidayManager pHolidayManager = null;
	try {
		pHolidayManager = BbmManagerFactory.getHolidayManager();
	}
	catch (BbmEJBCreateException e) {
	  e.printStackTrace();
	  throw new WorkRuleException("Failed to get Holiday Manager");
	}
	//for each organization
	for(Iterator i = pOrganizations.iterator(); i.hasNext();) {
	  Organization pOrganization = (Organization) i.next();
	  LinkedList llTimeSpans = new LinkedList();
	  m_hOrgHolidays.put(pOrganization.getID(),llTimeSpans);
	  if (pHolidayManager != null) {
		Collection cHolidays =null;
		 /*  REdundant code ; this is not used anymore. Sameet Apr 2010
		try {
		  cHolidays = null;// pHolidayManager.getHolidays(pOrganization.getID(),dtStartDate,dtEndDate);
		}
		catch (BbmFinderException e) {
		  e.printStackTrace();
		  throw new WorkRuleException("Failed to get Holiday Manager");
		}
		catch ( RemoteException e) {
		  e.printStackTrace();
		  throw new WorkRuleException("Failed to get Holiday Manager");
		}  */
		for(Iterator h = cHolidays.iterator(); h.hasNext();) {
		  Holiday pHoliday = (Holiday)h.next();
		  //add the holidays timespans to the org's arraylist
		  llTimeSpans.add(new TimeSpan(
			pHoliday.getStartDate().getTime(pOrganization.getTimeZone()),
			pHoliday.getEndDate().getTime(pOrganization.getTimeZone())));
		}
	  }
	}
  }

  public Collection getWorkRuleConflicts(Collection pWorkRuleParameters,int nPriority,
											 Organization pOrganization, int eOverlap,
											 Date dtStartDate, Date dtEndDate, Collection pEvents,
											 Collection pOldConflicts) {
	  //Employees who work Holidays are paid OT1...

	LinkedList pNewConflicts = new LinkedList();

	  //iterate over the event collection
	  for( Iterator i = pEvents.iterator(); i.hasNext(); ) {
		Event pEvent = (Event)i.next();
		if (WorkRuleUtil.isInPeriodAndApproved(pEvent,dtStartDate,dtEndDate,eOverlap)) {
		  //iterate over holidays for each event
		  Collection cHolidays = (Collection)m_hOrgHolidays.get(pOrganization.getID());
		  for(Iterator h = cHolidays.iterator(); h.hasNext();) {
			TimeSpan pTimeSpan = (TimeSpan)h.next();
			if (WorkRuleUtil.isInPeriod(pEvent.getStartTime(),pEvent.getEndTime(),
				pTimeSpan.dtSpanStart,pTimeSpan.dtSpanEnd,eOverlap)) {
			  Collection pSplitConflicts =
			  WorkRuleUtil.SplitConflicts(pEvent,pOldConflicts,nPriority,pEvent.getStartTime(),pEvent.getEndTime(),false);
			  pNewConflicts.addAll(pSplitConflicts);
			}
		  }
		}
	  }
	  return pNewConflicts;
  }

  public int getLookBack(Collection pWorkRuleParameters) {
	return 0;
  }

  public int getLookForward(Collection pWorkRuleParameters) {
	return 0;
  }

  public boolean isValid(Collection pWorkRuleParameters, int ePeriodType, int nPeriodLength,
	Locale locale, StringBuffer sMessage) {
	return true;
  }
