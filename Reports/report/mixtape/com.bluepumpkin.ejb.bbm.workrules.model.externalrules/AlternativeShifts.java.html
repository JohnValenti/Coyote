  public void initialize(LocalDate dtStartDate, LocalDate dtEndDate, Collection pOrganizations, Collection cWorkResourceIDs)
						throws WorkRuleException {
	m_hEmployeeImportedEvents = new HashMap();
	ScheduleAccessManager pScheduleManager = null;

	try {
		pScheduleManager = WfmManagerFactory.getScheduleAccessManager();
	}
	catch (BbmEJBCreateException e) {
	  e.printStackTrace();
	  throw new WorkRuleException("Failed to get Schedule Manager");
	}
	Date dtEarlyStart = new Date(dtStartDate.getTime().getTime() - 24*60*60*1000);
	Date dtLateEnd = new Date(dtEndDate.getTime().getTime() + 24*60*60*1000);

	//get the imported events
	try {
		Collection cEvents = pScheduleManager.getEventsForWorkResources(cWorkResourceIDs,dtEarlyStart,dtLateEnd);
		if (cEvents != null) {
			Iterator j = cEvents.iterator();
			for (Iterator i = cWorkResourceIDs.iterator();i.hasNext();) {
				m_hEmployeeImportedEvents.put(i.next(),j.next());
			}
		}
	}
	catch (BbmFinderException e) {
	  e.printStackTrace();
	  throw new WorkRuleException("Failed to get Schedules");
	}
	catch (RemoteException e) {
	  e.printStackTrace();
	  throw new WorkRuleException("Failed to get Schedules");
	}
  }

  public Collection getWorkRuleConflicts(Collection pWorkRuleParameters,int nPriority,
											 Organization pOrganization, int eOverlap,
											 Date dtStartDate, Date dtEndDate, Collection pEvents,
											 Collection pConflicts)
											 throws WorkRuleException{
	  //Employees who are work between X and Y hours and are schedule to work between X and Y hours
	  //each 1 Workday are paid OT1

	LinkedList pNewConflicts = new LinkedList();
	Iterator w = pWorkRuleParameters.iterator();
	int nMax  = (int)(60.0 * ((WorkRuleParameter)w.next()).getDoubleValue());
	int nUpperBound  = (int)(60.0 * ((WorkRuleParameter)w.next()).getDoubleValue()) ;
	int nMinScheduled = (int)(60.0 * ((WorkRuleParameter)w.next()).getDoubleValue());
	int nMaxScheduled = (int)(60.0 * ((WorkRuleParameter)w.next()).getDoubleValue());

	//iterate over the scheduled events
	int nScheduledTime = 0;
	//find the employee id
	Iterator i = pEvents.iterator();
	ID idEmployee = null;
	if (i.hasNext()) {
		Event pEvent = (Event)i.next();
		idEmployee = (ID) pEvent.getWorkResourceIDs().iterator().next();
	}
	else
		return pNewConflicts;

	//get the imported events for this employee
	Collection cImportedEvents = (Collection)m_hEmployeeImportedEvents.get(idEmployee);
	if (cImportedEvents == null)
		return pNewConflicts;

	for(i = cImportedEvents.iterator(); i.hasNext(); ) {
	  Event pEvent = (Event)i.next();
	  if (pEvent.getPaid() && pEvent.getOverlayPrecedence() == 1 &&
	  WorkRuleUtil.isInPeriod(pEvent.getStartTime(),pEvent.getEndTime(),dtStartDate,dtEndDate,eOverlap))
		nScheduledTime += pEvent.getDuration(); //what about unpaid Events...
	}

	if (nScheduledTime < nMinScheduled || nScheduledTime > nMaxScheduled)
		return pNewConflicts; //the rule does not apply

	int nOldCount = 0;
	int nTotalCount = 0;
	//iterate over the event collection
	for( i = pEvents.iterator(); i.hasNext(); ) {
	  Event pEvent = (Event)i.next();

	  if (nOldCount > nUpperBound)
		break;
	  //if the event is in the period, increment the hours count
	  if (WorkRuleUtil.isInPeriodAndApproved(pEvent,dtStartDate,dtEndDate,eOverlap)) {
		int nDuration = pEvent.getDuration() - WorkRuleUtil.GetUnpaidDuration(pEvent,pConflicts,nPriority);

		nTotalCount = nOldCount + nDuration;
		if (nTotalCount > nMax) {
		  //there is an exception here
		  int nExceptionStart = nMax;
		  if (nOldCount > nMax)
			nExceptionStart = nOldCount;
		  int nExceptionEnd = nTotalCount;
		  int nExceptionOffset = 0;
		  if (nTotalCount > nUpperBound) {
			nExceptionOffset = nExceptionEnd - nUpperBound;
			nExceptionEnd = nUpperBound;
		  }
		  //we need to create a new exception to the end of the event or to nUpperBound
		  int nExceptionDuration = nExceptionEnd - nExceptionStart;

		  Date dtViolationStart = new Date(pEvent.getEndTime().getTime() - ((nExceptionDuration + nExceptionOffset)* 60 * 1000));
		  Date dtViolationEnd = new Date(pEvent.getEndTime().getTime() - (nExceptionOffset * 60 * 1000));

		  Collection pSplitConflicts =
		  WorkRuleUtil.SplitConflicts(pEvent,pConflicts,nPriority,dtViolationStart,dtViolationEnd,true);
		  pNewConflicts.addAll(pSplitConflicts);
		}
	  }
	nOldCount = nTotalCount;
	}
	return pNewConflicts;
  }

  public boolean isValid(Collection pWorkRuleParameters, int ePeriodType, int nPeriodLength,
	Locale locale, StringBuffer sMessage)
		throws WorkRuleException {
	if (pWorkRuleParameters.size() != 4)
	  throw new WorkRuleException("Incorrect number of parameters");

	Iterator w = pWorkRuleParameters.iterator();
	double dMax = ((WorkRuleParameter)w.next()).getDoubleValue() ;
	double dUpperBound  = ((WorkRuleParameter)w.next()).getDoubleValue() ;
	double dMinScheduled = ((WorkRuleParameter)w.next()).getDoubleValue();
	double dMaxScheduled = ((WorkRuleParameter)w.next()).getDoubleValue();

	int nPeriodDays = WorkRuleUtil.getPeriodInDays(ePeriodType,nPeriodLength);

	if (dMax > nPeriodDays * 24.0) {
		Object[] aArgs = {new Double(dMax), new Integer(nPeriodDays)};
		sMessage.append(
		  MessageFormat.format("It is not possible to work more than {0} hours in {1} day period.",aArgs));
		return false;
	}

	if (dUpperBound <= dMax || dMaxScheduled <= dMinScheduled) {
	  sMessage.append("The \'more than\' value must be greater than the \'fewer than\' value.");
	  return false;
	}

	if (dUpperBound < 0.0 || dMax < 0.0 || dMaxScheduled < 0.0 || dMinScheduled < 0.0) {
	  sMessage.append("All values must be greater than 0.");
	  return false;
	}




	return true;
  }
