package com.verint.web.rest.wfm.exception.params;


import com.verint.web.rest.wfm.exception.EntityException;

public class QueryParamDateFormatException extends EntityException {

	private final String dateStringThatFailedToParse;

	public QueryParamDateFormatException(String entityName, String dateStringThatFailedToParse) {
		super(entityName, null);

		this.dateStringThatFailedToParse = dateStringThatFailedToParse;
	}

	@Override
	public String getTitle() {
		return "QueryParamDateFormatException";
	}

	@Override
	public String getDetail() {
		return "An exception occurred parsing a date filter parameter for entity " + entityName + ": Invalid date format - "
				+ dateStringThatFailedToParse + ". Must be of the format yyyy-MM-dd'T'HH:mm:ss.SSSZ";
	}
}
