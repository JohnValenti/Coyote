package com.bluepumpkin.web.rm.requests.shiftbidding;

import java.util.*;
import com.witness.web.uif.l10n.*;
import com.bluepumpkin.web.rm.l10n.*;
import com.bluepumpkin.web.rm.requests.AgentsRequestsListConfig;
import com.bluepumpkin.ejb.rm.requests.common.model.Request;

/**
 * Title:        AgentsShiftBiddingRequestsListConfig
 * Description:  List Configuration for Agents Shift Bid Requests
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, dsu
 * @version      1.0
 *
 * Created on October 11, 2002, 10:21 AM
 */
public class AgentsShiftBiddingRequestsListConfig extends AgentsRequestsListConfig {
  private List m_allFPs;
  private Map m_fpMap;

  /** Creates a new instance of AgentsShiftSwapRequestsListConfig */
  public AgentsShiftBiddingRequestsListConfig() {
    initFPs();
  }

	/**
	 * Initialize Field Properties
	 */
  private void initFPs() {
    if (m_allFPs==null)	{
			m_allFPs = new ArrayList(10);
			m_allFPs.add( createFP(Request.SORT_LAST_NAME_FIRST, UIFWebBundleKey.NAME, true) );
			m_allFPs.add( createFP(RmWebBundleKey.BID_NAME, RmWebBundleKey.BUNDLE_NAME, false) );
			m_allFPs.add( createFP(Request.SORT_STATUS, UIFWebBundleKey.STATUS, true) );
			m_allFPs.add( createFP(Request.SORT_CREATED, UIFWebBundleKey.SUBMITTED, true) );
			m_allFPs.add( createFP(Request.SORT_MODIFIED, UIFWebBundleKey.LAST_MODIFIED, true) );
			m_allFPs.add( createFPforRM(Request.SORT_SHIFTBID_ISBONUSUSED, RmWebBundleKey.USE_POINTS, true) );
			m_allFPs.add( createFP(Request.SORT_SHIFTBID_SENIORITY, UIFWebBundleKey.SENIORITY, true) );
			m_allFPs.add( createFP(Request.SORT_EMP_RANK, UIFWebBundleKey.RANK, true) );
			m_allFPs.add( createFP(Request.SORT_SHIFTBID_POINTS, UIFWebBundleKey.POINTS, true) );
			m_allFPs.add( createFP(Request.SORT_SHIFTBID_BIDDERBONUS, UIFWebBundleKey.BONUS, true) );
			m_allFPs.add( createFP(Request.SORT_SHIFTBID_SCORE, UIFWebBundleKey.SCORE, true) );
			m_allFPs.add( createFP(Request.SORT_SHIFTBID_PREFERENCE, UIFWebBundleKey.PREFERENCE, true) );
			addCommonFPs(m_allFPs);
		}

    if (m_fpMap==null) m_fpMap = createFPMap(m_allFPs);
  }

  /**
   * Return All List Config Field Properties (ListConfigFP)
   */
  public List getAllFPs() {
    return m_allFPs;
  }

  /**
   * Return a Map of All List Config Field Properties (ListConfigFP)
   * where key is ListConfigFP.getFieldName();
   */
  public Map getFPMap() {
    return m_fpMap;
  }
}
