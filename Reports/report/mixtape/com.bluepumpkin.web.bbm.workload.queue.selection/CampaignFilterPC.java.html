public class CampaignFilterPC extends FormTwoColumnPC {
	protected static final Category m_log = Log.initCategory(CampaignFilterPC.class.getName());
	private DateRangePickerPC m_drPickerPC;
	private ResourceBundle m_bundle;
	private boolean m_isCombined = false;
	private boolean m_isDisplayQueueOnly = false;
	private ID m_campaignID;
	private Collection m_campList = Collections.emptyList();
	private QueueSelectionPC m_queuePC;
	private List filterMediaList;

	/**
	 * Creates a new instance of CampaignFilterPC.
	 * @param context
	 */
	public CampaignFilterPC(RequestContext context) {
		super(context);
		m_bundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);

		m_drPickerPC = new DateRangePickerPC(context, "tPeriod");
		addChildComponent(m_drPickerPC);

		try {
			m_campList = CampaignModelHandler.getCampaignsForUser(context, context.getUser());
			if (!m_campList.isEmpty()) {
				m_campaignID = ((ValueObjectBase)m_campList.iterator().next()).getID();
			}
		} catch (Exception ex) {
			handleUnableToLoadDataError(m_log, ex);
		}

		m_queuePC = new QueueSelectionPC(context);
		m_queuePC.setName("queueList");
		addChildComponent(m_queuePC);

		setName("campFilter");
	}

	/**
	 * Initialize Component For Display
	 */
	@Override
	public void initForDisplay() {
		if (!m_isInitializedForDisplay) {
			m_queuePC.setIsBorderEnabled(false);
			m_queuePC.setIsHeaderShown(false);
			m_queuePC.setRowSelectable(true);
			m_queuePC.setIsNodeDisplayEnabled(true);
			m_queuePC.setIsInputSelectorDisplayed(false);

			super.initForDisplay();
			initContent();
		}
	}

	/**
	 * Initialize Component.
	 * @throws java.lang.Exception
	 */
	public void init() throws Exception {
		Map mediaFilter = filterMedia(WorkloadModelHandler.getMediaMap(m_context));

		ID organizationID = QueueFilterUtil.retrieveOrganizationID(m_context);
		if(isOrgSelected()) {
			m_queuePC.setCampaignModeData(null, organizationID, true, getStartDate(), getEndDate(), mediaFilter, false, false, true);
		} else {
			m_queuePC.setCampaignModeData(m_campaignID, null, false, getStartDate(), getEndDate(), mediaFilter, false, false, true);
		}
		m_queuePC.init();
	}

	private boolean isOrgSelected() {
		String scopeTypeAsInt = m_context.getUserProperty("ALERT_RULE_SCOPE");
		if (StringUtil.isEmpty(scopeTypeAsInt)) {
			return false;
		}

		int scopeType = 0;
		try {
			scopeType = Integer.parseInt(scopeTypeAsInt);
		} catch (NumberFormatException e) {
			return false;
		}
		return scopeType == NotifyRule.ORG_SCOPE;
	}

	/**
	 * Initialize Content
	 */
	private void initContent() {
		setLastRowType(LAST_ROW_NONE);
		setIsBorderEnabled(true);

		String selectedCampaignID = m_campaignID != null ? m_campaignID.toString() : "";
		if (!m_isDisplayQueueOnly) {
			m_drPickerPC.getStartDatePickerPC().setIsDateValueOptional(true);
			m_drPickerPC.getEndDatePickerPC().setIsDateValueOptional(true);
			m_drPickerPC.setOnChangeJS("submitForm();");
			String drLabel = i18n(m_common_bundle, UIFWebBundleKey.TIME_PERIOD);
			String drInput = m_drPickerPC.getHtmlDisplay();
			addRow(drLabel, drInput);

			//String cLabel = i18n(m_bundle, BbmWebBundleKey.QUEUE_SELECTOR_CAMPAIGNS);

			// Jeff Thelen: Get custom string from BPCONFIG table - for GMT clients.
			// This handles internationalization otherwise
			String cLabel = verticalize(m_bundle,BbmWebBundleKey.QUEUE_SELECTOR_CAMPAIGNS);

			ArrayList options = new ArrayList();
			options.add( new StringsPair("", ""));
			options.addAll(
					CampaignModelHandler.createDropDownValueList(m_campList, selectedCampaignID, m_localizer, true)
			);
			String campDD = HtmlUtil.makeSelectInput("campaignID", selectedCampaignID, "submitForm();", options);
			addRow(cLabel, campDD);
		} else {
			m_context.addGenericHTML(HtmlUtil.makeHiddenInput("campaignID", selectedCampaignID));
		}

		// verticalise queue
		Object[] args = new Object[1];
		args[0] = m_context.getSystemManagerFactory().getVerticalizationManager().getVerticalizedString(m_context, DefaultVerticalizationManager.QUEUES);

		String queueDisp = null;
		if (m_queuePC.isTreeEmpty()) {
			queueDisp = i18n(m_bundle, BbmWebBundleKey.QUEUE_SELECTOR_NO_RESULTS);
			queueDisp =  MessageFormat.format( queueDisp, args);
		}
		else {
			queueDisp = m_queuePC.getHtmlDisplay();
		}

		String leftItem =  MessageFormat.format( i18n(m_bundle, BbmWebBundleKey.QUEUE_SELECTOR_QUEUES), args);
		addRow( leftItem, queueDisp);
	}

	private Map filterMedia(Map allMedia) {
		List filterList = getFilterMediaList();
		if (filterList == null || filterList.isEmpty()) {
			return allMedia;
		}
		Map map = new HashMap();
		for (Iterator itr = allMedia.keySet().iterator(); itr.hasNext();) {
			Object key = itr.next();
			Media media = (Media) allMedia.get(key);
			if (filterList.contains(media.getName())) {
				map.put(key, media);
			}
		}
		return map;
	}

	public List getFilterMediaList() {
		return filterMediaList;
	}

	public void setFilterMediaList(List filterMediaList) {
		this.filterMediaList = filterMediaList;
	}

	public void setFilterMedia(String filterMedia) {
		this.filterMediaList = new ArrayList();
		StringTokenizer tokens = new StringTokenizer(filterMedia, ",;:");
		while (tokens.hasMoreElements()) {
			String token = (String)tokens.nextElement();
			this.filterMediaList.add(token.trim());
		}
	}

	public void setCampaignID(ID id) {
		m_campaignID = id;
	}

	public ID getCampaignID() {
		return m_campaignID;
	}

	public ID[] getCheckedQueueID() {
		ID[] checkedIds = m_queuePC.getCheckedID();
		List<ID> queueIdList = new ArrayList<>();

		if(checkedIds!=null){
				queueIdList.addAll(Arrays.asList(checkedIds));
			}
		ID[] addedQueueIDs = m_queuePC.getNewlyAddedID();
		for (ID addedQueueID : addedQueueIDs) {
			if (!queueIdList.contains(addedQueueID)) {
				queueIdList.add(addedQueueID);
			}
		}
		return queueIdList.toArray(new ID[0]);
	}

	public void setCheckedQueueID(ID[] queueIDs) {
		if (queueIDs != null && queueIDs.length > 0){
			m_queuePC.setCheckedID(queueIDs);
		}
	}

	public void setIsDisplayQueueOnly(boolean bSwitch) {
		m_isDisplayQueueOnly = bSwitch;
	}

	public boolean isDisplayQueueOnly() {
		return m_isDisplayQueueOnly;
	}

	public void setIsCombined(boolean bSwitch) {
		m_isCombined = bSwitch;
	}

	public boolean isCombined() {
		return m_isCombined;
	}

	public void setStartDate(Date date) {
		m_drPickerPC.getStartDatePickerPC().setDate(date);
	}

	public Date getStartDate() {
		return m_drPickerPC.getStartDatePickerPC().getDate();
	}

	public void setEndDate(Date date) {
		m_drPickerPC.getEndDatePickerPC().setDate(date);
	}

	public Date getEndDate() {
		return m_drPickerPC.getEndDatePickerPC().getDate();
	}
}
