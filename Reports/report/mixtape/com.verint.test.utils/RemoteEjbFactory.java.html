package com.verint.test.utils;


import com.bluepumpkin.ejb.bbm.JNDINames;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManager;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManagerEJBHome;
import com.bluepumpkin.ejb.core.security.weblogic.SecurityDelegator;
import com.bluepumpkin.ejb.core.security.weblogic.SecurityDelegatorHome;
import com.witness.ejb.core.security.RoleManager;
import com.witness.ejb.core.security.UserManager;
import com.witness.ejb.core.security.ejb.RoleManagerHome;
import com.witness.ejb.core.security.ejb.UserManagerHome;

import javax.naming.Context;
import javax.naming.InitialContext;
import java.util.Hashtable;

public class RemoteEjbFactory {

	private static Context ctx;

	private static UserManager um;
	private static WorkResourceManager wrm;
	private static SecurityDelegator securityDelegator;
	private static RoleManager roleManager;

	protected static String getHostAndPortForEjbRemoteAccess() {
		String host = System.getProperty("host");
		String[] temp = host.split(":");
		return temp[0] + ":7001";
	}

	protected static Context getContext() throws Exception {
		if (ctx == null) {
			Hashtable<String, String> env = new Hashtable<String, String>();
			env.put("java.naming.factory.initial", "weblogic.jndi.WLInitialContextFactory");
			env.put("java.naming.provider.url", "t3://" + getHostAndPortForEjbRemoteAccess());
			env.put("java.naming.security.principal", "wsuperuser");
			env.put("java.naming.security.credentials", "pumpkin1");
			ctx = new InitialContext(env);
		}
		return ctx;
	}

	public static UserManager getUserManager() throws Exception {
		if (um == null) {
			UserManagerHome ejbHome = (UserManagerHome) getContext().lookup(com.bluepumpkin.ejb.core.JNDINames.USERMGR_EJBHOME);
			um = ejbHome.create();
		}
		return um;
	}

	public static SecurityDelegator getSecurityDelegator() throws Exception {
		if (securityDelegator == null) {
			SecurityDelegatorHome ejbHome =
				(SecurityDelegatorHome) getContext().lookup(com.bluepumpkin.ejb.core.JNDINames.SECURITYDELEGATOR_EJBHOME);
			securityDelegator = ejbHome.create();
		}
		return securityDelegator;
	}

	public static WorkResourceManager getWorkResourceManager() throws Exception {
		if (wrm == null) {
			WorkResourceManagerEJBHome ejbHome =
				(WorkResourceManagerEJBHome) getContext().lookup(JNDINames.BBM_WORKRESOURCEMANAGER_EJBHOME);
			wrm = ejbHome.create();
		}
		return wrm;
	}

	public static RoleManager getRoleManager() throws Exception {
		if (roleManager == null) {
			RoleManagerHome ejbHome = (RoleManagerHome) getContext().lookup(com.bluepumpkin.ejb.core.JNDINames.ROLEMANAGER_EJBHOME);
			roleManager = ejbHome.create();
		}
		return roleManager;
	}
}
