	/**
	 *  This will return the next random/pre-determined number.
	 *
	 *  @param size  If we are in random mode, a random # from 0 to 'size'
	 *               will be generated.
	 *               If we are in non-random mode, it will return the
	 *               pre-determined value of its specific type.
	 *  @param type  The type of data to return, only applies when in
	 *               non-random mode.
	 *
	 *  @return The next number.
	 */
	private int nextNumber(int size, int type) {
		if (m_isRandom) {
			return m_random.nextInt(size);
		} else {
			switch (type) {
			case 0: // employee
				return m_nextEmployeeOffset;
			case 1: // event type
				return m_nextEventType;
			case 2: // slot #
				return m_nextTimeSlot;
			case 3: // activity
			default:
				int ret = m_nextActivity;

				// needs to update some value.
				m_nextActivity = (m_nextActivity + 1) % ACTIVITY_COLORS.length;
				m_nextTimeSlot = (m_nextTimeSlot + 1) % m_slots;

				if (m_nextTimeSlot == 0) {
					m_nextEventType = (m_nextEventType+1) % EVENTTYPES;
					if (m_nextEventType == 0) {
						m_nextEmployeeOffset = (m_nextEmployeeOffset+1) % m_edArray.length;
					}
				}
				return ret;
			}
		}
	}

	/**
	 *  Generates new events.  Each event generated will be shoved in a random list.
	 *
	 *  @param count  The number of events to generate.
	 */
	synchronized private void generateEvents(int count) {
		if (m_edArray == null || m_edArray.length == 0) {
			// nothing to generate...
			return;
		}
		for (int i = 0; i < count; i++) {
			// pick an employee randomly
			EmployeeData ed = m_edArray[nextNumber(m_edArray.length,0)];

			// pick a queue randomly (scheduled, actual, or exception)
			int eventType = nextNumber(EVENTTYPES,1);
			Collection eventList;
			switch (eventType) {
			case SCHEDULE_EVENTTYPE:
				eventList = ed.getPlannedEvents();
				break;
			case ACTUAL_EVENTTYPE:
				eventList = ed.getActualEvents();
				break;
			case EXCEPTIONS_EVENTTYPE:
			default:
				eventList = ed.getAdheranceViolations();
				break;
			}

			// pick a time slot randomly
			long start_time = m_start + nextNumber(m_slots,2) * m_interval;
			long end_time = start_time + m_interval;

			// determine if time slot already occupied, if so, we only modify
			// the timestamp and the activity, otherwise, we create a new
			// event.
			AdherenceEvent event = getEvent(eventList, start_time);
			ID id = new ID(nextNumber(ACTIVITY_COLORS.length,3));
			if (event == null) {
				if (eventType == EXCEPTIONS_EVENTTYPE) {
					event = new RTAAException(ed.getEmployeeID(),
											  new Date(start_time),
											  new Date(end_time),
											  id,
											  false,
											  new Date(System.currentTimeMillis()));
				} else {
					event = new SimpleEvent(new ID(m_id++),
											0,
											"SimpleEvent",
											new Date(start_time),
											new Date(end_time),
											id,
											false,
											false,
											null,
											new Date(System.currentTimeMillis()));
				}

				// add to the front
				((ArrayList)eventList).add(0,event);
			} else {
				event.setActivityID(id);
				event.setTimestamp(new Date(System.currentTimeMillis()));
				if (event instanceof RTAAException) {
					((RTAAException)event).setApproved(!((RTAAException) event).isApproved());
				}
				eventList.remove(event);

				// put back to the front
				((ArrayList)eventList).add(0,event);
			}
		}
	}

	/**
	 *  Retrieves an event from the list.
	 *
	 *  @param list  The list to retrieve from.
	 *  @param startTime  The start time used as the key retrieval
	 *
	 *  @return  An AdherenceEvent object, or <code>null</code> if not found.
	 */
	private AdherenceEvent getEvent(Collection list, long startTime) {
		for (Iterator eventIter = list.iterator(); eventIter.hasNext();) {
			AdherenceEvent event = (AdherenceEvent)eventIter.next();

			if (event.getStartTime().getTime() == startTime)
				return event;
		}
		return null;
	}

	/**
	 *  Gets the recently changed records.
	 *
	 *  @param  employeeIDs  The set of employess of interest.
	 *  @param  startDate    The start range of activities of interest.
	 *  @param  endDate      The end range of activities of interest.
	 *  @param  lastTimestamp  The time to determine "recentless"
	 *                         If <code>null</code>  returns all, or no
	 *                         filter.
	 * @param pageNum For pagination, the current page number being viewed.
	 *                When negative, there is no pagination involved.
	 * @param pageSize For pagination, the current page size
	 *                 not used, when 'pageNum' is negative.
	 * @param sortKey  The key for sorting.  If negative, use the default sort key.
	 */
	synchronized public RTAAData getAdherenceChanges(Collection employeeIDs,
													 Date startDate,
													 Date endDate,
													 Date lastTimestamp,
													 int pageNum,
													 int pageSize,
													 int sortKey)
	{
		if (startDate.getTime() != m_start || endDate.getTime() != m_end) {
			// if request dates have changes, then we clean up.
			cleanEventLists();
			m_start = startDate.getTime();
			m_end = endDate.getTime();
		}

		if (lastTimestamp == null && !AdherenceUtils.areCollectionsEqual(m_employeeIDs, employeeIDs)) {
			System.out.println("Setting employee IDs to new list " + employeeIDs + " with " + employeeIDs.size() + " employees");
			setEmployeeIDs(employeeIDs);
		}

		HashMap activities = new HashMap();
		Collection selectedEmployeeIDs = employeeIDs;

		if (pageNum >= 0) {
			// with pagination, new list of employees return
			selectedEmployeeIDs = sort(pageNum, pageSize, sortKey);
		}

		Collection employeeData = new ArrayList(selectedEmployeeIDs.size());
		m_validatedTimeStamp = new Date(System.currentTimeMillis());

		fetchChangedData(employeeData, activities, selectedEmployeeIDs, startDate, endDate, lastTimestamp);

		RTAAData ret = new RTAAData(employeeData,
									activities,
									startDate,
									endDate,
									null);
		ret.setTimestamp(m_validatedTimeStamp);
		ret.setEmployeeIDs(selectedEmployeeIDs);
		return ret;
	}

	/**
	 *  Returns a sorted list of employee IDs based on the
	 *  sorting/pagination parameters.
	 */
	private Collection sort(int pageNum, int pageSize, int sortKey) {
		EmployeeData[] tmpED = new EmployeeData[m_edArray.length];
		System.arraycopy(m_edArray, 0, tmpED, 0, tmpED.length);

		switch (sortKey) {
		case AdherenceKeys.SORT_NONE:
			//Arrays.sort(tmpED, new NameComparator(0));
			break;
		case AdherenceKeys.SORT_NAME:
			//Arrays.sort(tmpED, new NameComparator(1));
			break;
		case AdherenceKeys.SORT_START_TIME:
		case AdherenceKeys.SORT_END_TIME:
		case AdherenceKeys.SORT_SCHEDULED_START_TIME:
		case AdherenceKeys.SORT_SCHEDULED_END_TIME:
		case AdherenceKeys.SORT_ADHERENCE_STATUS:
			break;
		}

		int start = pageNum * pageSize;
		int length = pageSize;
		if (start + length > m_edArray.length) {
			length = m_edArray.length - start;
		}

		EmployeeData[] sortedEmployeeData = new EmployeeData[length];
		System.arraycopy(tmpED, start, sortedEmployeeData, 0, length);

		ArrayList returnList = new ArrayList(length);
		for (int i = 0; i < length; i++) {
			returnList.add(sortedEmployeeData[i].getEmployeeID());
		}

		return returnList;
	}

	/**
	 *  Returns a collection of items that are in 'newItems' but not in original
	 */
	private Collection difference(Collection original, Collection newItems) {
		HashSet diffSet = new HashSet(original.size());

		for (Iterator iter = newItems.iterator();iter.hasNext();) {
			ID newID = (ID)iter.next();
			if (!original.contains(newID)) {
				diffSet.add(newID);
			}
		}
		return diffSet;
	}

	synchronized public void massApproveExceptions(ID employeeID, Collection timeRangesToAuthorize, Collection timeRangesToUnauthorize) {
		Collection sortedTimeRangesToAuthorize = AdherenceUtils.sortTimeRangeList(timeRangesToAuthorize);
		Collection sortedTimeRangesToUnauthorize = AdherenceUtils.sortTimeRangeList(timeRangesToUnauthorize);
		EmployeeData empData = getEmployeeData(employeeID);
		Collection unsortedAdheranceViolations = empData.getAdheranceViolations();
		massAuthorizeExceptions(unsortedAdheranceViolations, sortedTimeRangesToAuthorize, true);
		massAuthorizeExceptions(unsortedAdheranceViolations, sortedTimeRangesToUnauthorize, false);
	}

	synchronized public void groupAuthorizeExceptions(Collection empIDs, Date start, Date end, boolean authorized)
	{
		for (Iterator empIter = empIDs.iterator(); empIter.hasNext();) {
			ID empID = (ID)empIter.next();

			EmployeeData empData = getEmployeeData(empID);
			Collection exceptions = empData.getAdheranceViolations();

			ArrayList sortedExceptions = new ArrayList(AdherenceUtils.sortExceptionList(exceptions));
			int size = sortedExceptions.size();
			
			for (int i = 0; i < size; i++) {
				RTAAException exc = (RTAAException)sortedExceptions.get(i);

				if (exc.getStartTime().after(end))
					break;

				if (AdherenceUtils.dateRangesOverlap(exc.getStartTime(), exc.getEndTime(),
													 start, end)) {

					exc.setTimestamp(new Date(System.currentTimeMillis()));
					exc.setApproved(authorized);
					exceptions.remove(exc);

					// put back to the front.
					((ArrayList) exceptions).add(0, exc);
				}
			}
		}
	}

	private void massAuthorizeExceptions(Collection listOriginalExceptions, Collection listSortedTimeRanges, boolean authorize) {
		if (listOriginalExceptions == null) {
			return;
		}
		ArrayList listSortedExceptions = new ArrayList(AdherenceUtils.sortExceptionList(listOriginalExceptions));
		int numExceptions = listSortedExceptions.size();
		ArrayList listEvents = new ArrayList(numExceptions);
		int nextTimeRangeIndex = 0;
		TimeRange [] timeRanges = (TimeRange []) listSortedTimeRanges.toArray(new TimeRange[0]);
		TimeRange nextTimeRange = timeRanges.length == 0 ? null : timeRanges[nextTimeRangeIndex];
		for (int i = 0; i < numExceptions; i++) {
			RTAAException nextException = (RTAAException) listSortedExceptions.get(i);
			// if the current exception occurs after the current scheduled event, let's iterate through
			// the list of scheduled events until we come to the next scheduled event that overlaps or is
			// after this exception
			while (nextTimeRange != null && !nextTimeRange.getEndDate().after(nextException.getStartTime())) {
				nextTimeRangeIndex++;
				nextTimeRange = nextTimeRangeIndex < timeRanges.length ? timeRanges[nextTimeRangeIndex] : null;
			}
			if (nextTimeRange == null) {
				// no more time ranges
				break;
			} else {
				if (AdherenceUtils.dateRangesOverlap(nextException.getStartTime(), nextException.getEndTime(),
						nextTimeRange.getStartDate(), nextTimeRange.getEndDate())) {
					nextException.setTimestamp(new Date(System.currentTimeMillis()));
					nextException.setApproved(authorize);
					listOriginalExceptions.remove(nextException);

					// put back to the front
					((ArrayList) listOriginalExceptions).add(0, nextException);
				}
			}
		}

	}

	// expects to be called by a synchronized method
	private void exceptionApprovalHelper(Collection exceptions, boolean shouldApprove) {
		Iterator itExceptions = exceptions.iterator();
		while (itExceptions.hasNext()) {
			RTAAException nextException = (RTAAException) itExceptions.next();
			ID nextEmpID = nextException.getEmployeeID();
			EmployeeData ed = getEmployeeData(nextEmpID);
			Collection adheranceViolations = ed.getAdheranceViolations();
			AdherenceEvent event = getEvent(adheranceViolations, nextException.getStartTime().getTime());
			if (event != null && event instanceof RTAAException) {
				System.out.println("setting event for ID=" + nextEmpID + " starting at " + (nextException.getStartTime().getTime() - m_start) + " to be " + (shouldApprove ? "approved" : "unapproved"));
				((RTAAException) event).setApproved(shouldApprove);
				event.setTimestamp(new Date(System.currentTimeMillis()));
				adheranceViolations.remove(event);
				// put back to the front
				((ArrayList) adheranceViolations).add(0, event);
			} else {
				System.out.println("FAILED TO UN/APPROVE EXCEPTION:");
				System.out.println("looking for event for employee ID = " + nextEmpID + " starting at " + (nextException.getStartTime().getTime() - m_start));
				System.out.println("exceptions in memory:");
				dumpEventList(adheranceViolations, m_start);
			}
		}
	}

	/**
	 *  Removes all of the event lists previously created in the system.
	 */
	private void cleanEventLists() {
		Collection edCollection = m_edMap.values();
		for (Iterator edIter = edCollection.iterator(); edIter.hasNext();) {
			EmployeeData ed = (EmployeeData)edIter.next();
			ed.getPlannedEvents().clear();
			ed.getActualEvents().clear();
			ed.getAdheranceViolations().clear();
		}
	}

	/**
	 *  Fetches the recently changed records, by filling in the data in
	 *  the 'employeeData' collection and the 'activities' HashMap.
	 *
	 *  @param  employeeData The collection where data is filled.
	 *  @param  activities   The set of activities.
	 *  @param  employeeIDs  The set of employess of interest.
	 *  @param  startDate    The start range of activities of interest.
	 *  @param  endDate      The end range of activities of interest.
	 *  @param  lastTimestamp  The time to determine "recentless"
	 *                         If <code>null</code>  returns all, or no
	 *                         filter.
	 */
	private void fetchChangedData(Collection employeeData,
								  HashMap activities,
								  Collection employeeIDs,
								  Date startDate,
								  Date endDate,
								  Date lastTimestamp)
	{
		for (Iterator empIter = employeeIDs.iterator();empIter.hasNext();) {
			ID empID = (ID)empIter.next();

			EmployeeData srcED = (EmployeeData)m_edMap.get(empID);

			// done, if the emp ID is unknown
			if (srcED == null)
				continue;

			Date timestamp = lastTimestamp;

			// if we actually want all the data, then overwrite validated time to
			// null.
			//if (fetchAllDataEmpIDs.contains(empID))
			//timestamp = null;

			Collection scheduled =
				filterEvents(activities, startDate, endDate, timestamp, srcED.getPlannedEvents());
			Collection actual =
				filterEvents(activities, startDate, endDate, timestamp, srcED.getActualEvents());
			Collection exceptions =
				filterEvents(activities, startDate, endDate, timestamp, srcED.getAdheranceViolations());

			EmployeeData destED =
				new EmployeeData(empID,
								 srcED.getEmployeeName(),
								 scheduled,
								 actual,
								 exceptions,
								 null,
								 null,
								 null,
								 null,
								 null,
								 false,
								 0);
			employeeData.add(destED);
		}
	}

	/**
	 *  Filter events to be returned based on the relevant date range
	 *  and the validated time stamp.  Returns a list of qualified events.
	 *
	 *  @return An ArrayList of qualified "events", or
	 *  <code>Collections.EMPTY_LIST</code> if nothing matches
	 */
	private Collection filterEvents(HashMap activities,
									Date startDate,
									Date endDate,
									Date lastTimestamp,
									Collection events)
	{
		Collection eventList = new ArrayList();
		for (Iterator eventIter = events.iterator(); eventIter.hasNext();) {
			AdherenceEvent event = (AdherenceEvent)eventIter.next();

			// the events are sorted based on the last modified timestamp
			// so we can quit once we find one that's not as recent
			// as the last validated timestamp
			if (!isMoreRecent(event.getTimestamp(), lastTimestamp))
				break;

			// make sure the events are within our range of interests
			if (AdherenceUtils.dateRangesOverlap(event.getStartTime(),
												 event.getEndTime(),
												 startDate,
												 endDate)) {

				// need to pass back a cloned item so that the client
				// can modify.
				AdherenceEvent retEvent;
				if (event instanceof SimpleEvent) {
					retEvent = new SimpleEvent(((SimpleEvent)event).getID(),
											   0,
											   "SimpleEvent",
											   event.getStartTime(),
											   event.getEndTime(),
											   event.getActivityID(),
											   false,
											   false,
											   null,
											   event.getTimestamp());
				} else {
					retEvent = new RTAAException(((RTAAException)event).getEmployeeID(),
												 event.getStartTime(),
												 event.getEndTime(),
												 event.getActivityID(),
												 ((RTAAException) event).isApproved(),
												 event.getTimestamp());
				}
				eventList.add(retEvent);

				// add to activities hashmap
				ID activityID = event.getActivityID();
				activities.put(activityID, m_activities.get(activityID));
			}
		}

		if (eventList.size() == 0)
			return Collections.EMPTY_LIST;
		return eventList;
	}

	/**
	 *  Determines if the time stamp in an event record more
	 *  recent than the last validated time stamp.
	 *
	 *  @return <code>true</code>  If event record is more recent, or if there
	 *                             is no validated time stamp.
	 *          <code>false</code> otherwise.
	 */
	private boolean isMoreRecent(Date eventTimeStamp, Date validatedTimeStamp) {
		if (validatedTimeStamp == null)
			return true;
		if (eventTimeStamp.compareTo(validatedTimeStamp) >= 0) {
			return true;
		} else {
			return false;
		}
	}

	/**
	 *  Sleeps for a while, make sure we won't be
	 *  interrupted.
	 */
	private void unInterruptedSleep() {
		long now = System.currentTimeMillis();
		long endOfSleepTime = now + m_cycleTime;
		while (now < endOfSleepTime) {
			try {
				sleep(endOfSleepTime - now);
			} catch (InterruptedException ex) {}
			now = System.currentTimeMillis();
		}
	}

	/**
	 *  Terminates the thread.
	 */
	public void terminate() {
		m_running = false;
	}

	/**
	 *
	 */
	public void memoryDump() {
		for (int i = 0; i < m_edArray.length; i++) {
			EmployeeData ed = m_edArray[i];

            System.out.println(">>>>>>> Employee ID= " + ed.getEmployeeID() + " <<<<<<<");

			System.out.println("  >>>>> ScheduledEvents<<<<<<<");
			dumpEventList(ed.getPlannedEvents());

			System.out.println("  >>>>> ActualEvents<<<<<<<");
			dumpEventList(ed.getActualEvents());

			System.out.println("  >>>>> ScheduledEvents<<<<<<<");
			dumpEventList(ed.getAdheranceViolations());
		}
	}

	static public void dumpEventList(Collection list, long offset) {
		for (Iterator eventIter = list.iterator(); eventIter.hasNext();) {
			AdherenceEvent event = (AdherenceEvent)eventIter.next();
			System.out.println("    ST=" + (event.getStartTime().getTime()-offset) +
							   " ET=" + (event.getEndTime().getTime()-offset) +
							   " Act=" + event.getActivityID() +
							   " TS=" + event.getTimestamp().getTime());
		}
	}

	static public void dumpEventList(Collection list) {
		dumpEventList(list, 0);
	}
