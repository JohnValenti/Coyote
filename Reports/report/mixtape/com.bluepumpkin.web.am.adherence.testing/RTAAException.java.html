package com.bluepumpkin.web.am.adherence.testing;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

/**
 *  Defines the records used for exception adherence events.
 */
public class RTAAException 
	extends com.bluepumpkin.ejb.bbm.rtaaservice.model.RTAAException 
	implements AdherenceEvent 
{
	private Date m_timestamp;

	public RTAAException(ID idEmployee, Date dateStart, Date dateEnd, 
						 ID idPlannedActivity, boolean bIsApproved,
						 Date timestamp)
	{
		super(idEmployee, dateStart, dateEnd, idPlannedActivity, bIsApproved, false);
		setTimestamp(timestamp);
	}

	public RTAAException(Date dateStart, Date dateEnd, 
						 ID idPlannedActivity, boolean bIsApproved,
						 Date timestamp)
	{
		super(dateStart, dateEnd, idPlannedActivity, bIsApproved);
		setTimestamp(timestamp);
	}

	public ID getActivityID() {
		return getPlannedActivityID();
	}

	public void setActivityID(ID activityID) {
		m_idPlannedActivity = activityID;
	}

	public Date getTimestamp() {
		return m_timestamp;
	}

	public void setTimestamp(Date timestamp) {
		m_timestamp = timestamp;
	}

	public void setApproved(boolean isApproved) {
		m_bIsApproved = isApproved;
	}
}
