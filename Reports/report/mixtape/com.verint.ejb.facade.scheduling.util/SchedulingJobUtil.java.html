package com.verint.ejb.facade.scheduling.util;

import java.rmi.RemoteException;
import java.util.Date;

import javax.naming.NamingException;

import org.quartz.JobDetail;
import org.quartz.Scheduler;
import org.quartz.SchedulerException;
import org.quartz.SimpleTrigger;
import com.bluepumpkin.common.base.Log;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.config.ConfigKey;
import com.bluepumpkin.ejb.bbm.quartz.QuartzUtil;
import com.verint.ejb.facade.scheduling.jobs.SchedulingJob;


public class SchedulingJobUtil {

	private static Category m_cat = Log.initCategory(SchedulingJobUtil.class.getName());

	private static String quartzWFMGroupName = "WFM";

	private static String quartzSchedulerJobName = "SchedulerQueueJob";
	
	private static String triggerName = "triggerForJob" + quartzSchedulerJobName;

	private static final int RETRYINTERVAL_IN_MILLIS = 90000;

	private SchedulingJobUtil() {
		// private constructor to make sure that this is used as static class
	}

	public static void addOrUpdateSchedulingJobIfNecessary()
			throws BbmEJBCreateException, RemoteException, NamingException, SchedulerException {
		// get the configuration value for background task execution interval.
		String intervalStr = BbmManagerFactory.getDBConfigManager()
				.getValue(ConfigKey.FACADE_SCHEDULING_RETRY_INTERVAL);
		int cfgInterval = StringUtil.isEmpty(intervalStr) ? RETRYINTERVAL_IN_MILLIS
				: Integer.parseInt(intervalStr);

		boolean jobExists = quartzJobExists(quartzSchedulerJobName, quartzWFMGroupName);

		if (jobExists) {
			QuartzUtil.removeJob(quartzSchedulerJobName, quartzWFMGroupName);
		}

		// add job
		addQuartzJob(cfgInterval);

	}

	/**
	 * @param jobGroupName
	 * @param jobName
	 */
	private static boolean quartzJobExists(String jobName, String jobGroupName)
			throws NamingException, SchedulerException {

		// if running within container.
		m_cat.debug("retrieving quartz job from JNDI jobName=" + jobName + " : jobGroupName=" + jobGroupName);

		Scheduler scheduler = com.bluepumpkin.ejb.bbm.quartz.QuartzUtil.getScheduler();
		JobDetail jobDetail = scheduler.getJobDetail(jobName, jobGroupName);

		return jobDetail != null;
	}

	/**
	 * Adds the quartz job for the SchedulingQueueJob. The trigger is set to
	 * ignore missed executions. If an execution of the job takes longer than
	 * the interval, it does not make sense to start another job immediately for
	 * the scheduling as already scheduled SPs would not be complete.
	 */
	private static void addQuartzJob(int repeatInterval) throws NamingException, SchedulerException {
		m_cat.debug("Adding quartz job from JNDI jobName=" + quartzSchedulerJobName + " : jobGroupName="
				+ quartzWFMGroupName);
		JobDetail job = new JobDetail(quartzSchedulerJobName, quartzWFMGroupName, SchedulingJob.class);

		SimpleTrigger st = new SimpleTrigger(triggerName, quartzWFMGroupName, new Date(), null,
				SimpleTrigger.REPEAT_INDEFINITELY, repeatInterval);
		st.setMisfireInstruction(SimpleTrigger.MISFIRE_INSTRUCTION_RESCHEDULE_NEXT_WITH_REMAINING_COUNT);
		QuartzUtil.getScheduler().scheduleJob(job, st);
		m_cat.debug("Added quartz job from JNDI jobName=" + quartzSchedulerJobName + " : jobGroupName="
				+ quartzWFMGroupName);
	}
}
