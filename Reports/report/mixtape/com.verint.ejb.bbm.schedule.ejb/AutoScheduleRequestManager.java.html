package com.verint.ejb.bbm.schedule.ejb;

import java.rmi.RemoteException;
import java.util.List;
import java.util.Map;

import javax.ejb.EJBObject;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.verint.ejb.bbm.schedule.model.WSAutoScheduleRequest;

/**
 * Remote Interface for creating/updating WSAutoScheduleRequests
 */
public interface AutoScheduleRequestManager extends EJBObject {

	public enum SchedulingMode {
		RECALC("R"), SCHEDULE("S"), ANALYZE("A");

		String mode;

		SchedulingMode(String mode) {
			this.mode = mode;
		}

		public String mode() {
			return mode;
		}
	}

	/**
	 * Creates scheduling requests for the given SPIds.
	 * 
	 * @return Map of spId vs Created request Id
	 */
	public Map<ID, ID> createRequests(List<ID> spIds, SchedulingMode mode, String createdBy)
			throws BbmCreateException, RemoteException;

	/**
	 * Persists the given WSAutoScheduleRequest in the database.
	 */
	public ID createSchedulingRequest(WSAutoScheduleRequest request, String modifiedBy)
			throws BbmCreateException, RemoteException;

	/**
	 * Updates the requests with the given IDs with the given status and the
	 * status messages
	 */
	public void updateSchedulingRequests(List<ID> requests, String status, String statusMessage)
			throws BbmUpdateException, RemoteException;

	/**
	 * Returns the request by the given ID
	 */
	public WSAutoScheduleRequest getSchedulingRequest(ID requestID)
			throws BbmFinderException, BbmObjectNotFoundException, RemoteException;
}
