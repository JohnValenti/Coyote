package com.bluepumpkin.ejb.bbm.queuegroup.ejb;

import java.util.Collections;
import java.util.Collection;
import java.util.ArrayList;
import java.util.Iterator;
import java.util.HashMap;

import com.bluepumpkin.common.jdmo.*;
import com.bluepumpkin.common.datatypes.ID;

import com.bluepumpkin.ejb.bbm.vo.*;
import com.bluepumpkin.ejb.bbm.dao.*;
import com.bluepumpkin.ejb.bbm.base.*;

import com.bluepumpkin.ejb.bbm.queuegroup.model.*;


public class QueueGroupDAO extends DAOBase {

        private static FieldInfo m_fieldInfo = new QueueGroupFieldInfo();
        protected FieldInfo getFieldInfo() { return m_fieldInfo; }

        public QueueGroupDAO() {
            super();
        }

        public QueueGroupDAO(Jdmo dmo) {
            super(dmo);
        }

        protected ValueObjectBase createValueObject()
        {
            return (new QueueGroup());
        }

        /**
         *  get ids of groups linked to a given queue
         *
         *  @return a collection of group ids
         *
         */
        public Collection getLinkedGroupIDsForQueue(ID idQueue) throws BbmFinderException
        {
            if (idQueue == null) return Collections.EMPTY_LIST;

            Collection colQueueGroups = getObjects(" QUEUEID = " + idQueue);
            ArrayList arrGroupIDs = new ArrayList();

            Iterator it = colQueueGroups.iterator();
            while (it.hasNext())
                arrGroupIDs.add(((QueueGroup) it.next()).getGroupID());

            return arrGroupIDs;
        }
        /**
         *  get ids of groups linked to a given queue
         *
         *  @return a collection of group ids
         *
         */
        public Collection getLinkedQueueIDsForGroup(ID idGroup) throws BbmFinderException
        {
            if (idGroup == null) return Collections.EMPTY_LIST;

            Collection colQueueGroups = getObjects(" DATASOURCEGROUPID = " + idGroup);
            ArrayList arrQueueIDs = new ArrayList();

            Iterator it = colQueueGroups.iterator();
            while (it.hasNext())
                arrQueueIDs.add(((QueueGroup) it.next()).getQueueID());

            return arrQueueIDs;
        }
        /**
         *  get ids of groups linked to a queue
         *
         *  @return a collection of group ids
         *
         */
        public Collection getAllLinkedGroupIDs() throws BbmFinderException
        {
            Collection colQueueGroups = getObjects(" QUEUEID is not NULL ");
            ArrayList arrGroupIDs = new ArrayList();

            Iterator it = colQueueGroups.iterator();
            while (it.hasNext())
                arrGroupIDs.add(((QueueGroup) it.next()).getGroupID());

            return arrGroupIDs;
            
        }
        /**
         *  get ids of groups linked to a list of queues
         *
         *  @return a HashMap contains (queueID, array list of groups)
         *
         */
        public HashMap getLinkedGroupIDsForQueues(Collection colQueueIDs) throws BbmFinderException
        {
            if (colQueueIDs == null || colQueueIDs.size() < 1) return new HashMap();

            try
            {
                Collection colQueueGroups = getObjects(" QUEUEID in " + m_dmo.createInClause(colQueueIDs));
                HashMap map = new HashMap();

                Iterator it = colQueueGroups.iterator();
                while (it.hasNext())
                {
                    QueueGroup item = (QueueGroup) it.next();

                    ArrayList arr = (ArrayList) map.get(item.getQueueID());
                    if (arr == null)
                    {
                        arr = new ArrayList();
                        map.put(item.getQueueID(), arr);
                    }

                    arr.add(item.getGroupID());
                }

                return map;
            }
            catch(JdmoException e)
            {
                throw new BbmFinderException(e);
            }
        }

        /**
         *  link groups to queue
         */
        public void linkGroupsToQueue(ID idQueue, Collection colGroupIDs) throws BbmCreateException
        {
            if (idQueue == null || colGroupIDs == null || colGroupIDs.size() < 1) return;

            try
            {
                // in case of link group to queue twice or by another user
                unLinkGroupsFromQueue(idQueue, colGroupIDs);

                Iterator it = colGroupIDs.iterator();
                while (it.hasNext())
                {
                    QueueGroup item = new QueueGroup();
                    item.setQueueID(idQueue);
                    item.setGroupID((ID) it.next());
                    createObject(item);
                }
            }
            catch(Exception e)
            {
                throw new BbmCreateException(e);
            }
        }

        /**
         *  unlink groups from queue
         */
        public void unLinkGroupsFromQueue(ID idQueue, Collection colGroupIDs) throws BbmRemoveException
        {
            if (idQueue == null || colGroupIDs == null || colGroupIDs.size() < 1) return;

            try
            {
                StringBuffer strSQL = new StringBuffer(30);
                strSQL.append("delete ").append(getFieldInfo().getTableName());
                strSQL.append(" where QUEUEID = ").append(idQueue);
                strSQL.append(" and DATASOURCEGROUPID in ").append(m_dmo.createInClause(colGroupIDs));

                m_dmo.executeCommand(new String(strSQL));
            }
            catch(JdmoException e)
            {
                throw new BbmRemoveException(e);
            }
        }

}
