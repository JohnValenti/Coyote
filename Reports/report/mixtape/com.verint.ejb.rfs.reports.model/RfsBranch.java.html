package com.verint.ejb.rfs.reports.model;

import java.util.Collection;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Iterator;
import java.util.Map;

import javax.xml.bind.annotation.XmlElement;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.ScopeID;
import com.bluepumpkin.common.datatypes.ScopeableVO;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectSimpleRoot;

public class RfsBranch extends ValueObjectSimpleRoot implements ScopeableVO {

	private static final long serialVersionUID = -2133879139894758915L;

	private static FieldInfo m_fieldInfo = new RfsBranchFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	private RfsBranch m_branchParent = null;
	private HashSet<RfsBranch> m_setChildBranches = null;
	private ScopeID m_ScopeID;
	private ScopeID m_ParentScopeID;

	public static final short SCOPE_TYPE = (short) 2;

	public RfsBranch() {
		super();
	}

	/**
	 * If this were being used in jaxb conext, this is the only field which
	 * would be unmarshaled, because the default would be none.
	 */
	@XmlElement
	public String getBranchId() {
		return getFormattedFieldValue(RfsBranchFieldInfo.BRANCH_ID_NAME);
	}

	public String getBranchLabel() {
		return getFormattedFieldValue(RfsBranchFieldInfo.BRANCH_LABEL);
	}

	//FIXME: Equals returns true if an ID object is passed in
	public boolean equals(Object obj) {
		if (obj instanceof RfsBranch) {
			ID id = ((RfsBranch) obj).getID();
			if (getID() == null || id == null) {
				return false;
			}
			return id.equals(getID());
		} else if (obj instanceof ID) {
			ID id = (ID) obj;
			if (getID() == null || id == null) {
				return false;
			}
			return id.equals(getID());
		} else {
			return false;
		}
	}

	public RfsBranch getParent() {
		return m_branchParent;
	}

	public void setParent(RfsBranch branch) {
		m_branchParent = branch;
	}

	public void addChildBranch(RfsBranch branch) {
		if (m_setChildBranches == null) {
			m_setChildBranches = new HashSet<RfsBranch>(1);
		}
		m_setChildBranches.add(branch);
	}

	public HashSet<RfsBranch> getChildBranches() {
		return m_setChildBranches;
	}

	public static RfsBranch buildBranchTree(HashMap<ID, RfsBranch> mapOrgs) {
		if (mapOrgs == null || mapOrgs.size() < 1)
			return null;

		Iterator<Map.Entry<ID, RfsBranch>> it = mapOrgs.entrySet().iterator();

		while (it.hasNext()) {
			Map.Entry<ID, RfsBranch> mapEntry = it.next();
			RfsBranch org = (RfsBranch) mapEntry.getValue();
			if (org == null)
				continue;

			RfsBranch orgParent = (RfsBranch) mapOrgs.get(org.getParentID());
			if (orgParent == null) {
				continue; // no parent
			}

			org.setParent(orgParent);
			orgParent.addChildBranch(org);
		}

		// return the organization which don't have parents
		Iterator<RfsBranch> it2 = mapOrgs.values().iterator();
		while (it2.hasNext()) {
			RfsBranch org = it2.next();
			if (org.getParent() == null) {
				return org;
			}
		}

		return null;
	}

	// assume there is an root organization
	public static RfsBranch buildBranchTree(Collection<RfsBranch> colOrgs) {
		if (colOrgs == null || colOrgs.size() < 1) {
			return null;
		}

		HashMap<ID, RfsBranch> mapOrgs = new HashMap<ID, RfsBranch>(colOrgs.size());

		Iterator<RfsBranch> it = colOrgs.iterator();
		while (it.hasNext()) {
			RfsBranch item = it.next();
			mapOrgs.put(item.getID(), item);
		}

		return buildBranchTree(mapOrgs);
	}

	@Override
	public String getIcon() {
		// TODO Auto-generated method stub
		return "";
	}

	@Override
	public String getName() {
		// TODO Auto-generated method stub
		return getFormattedFieldValue(RfsBranchFieldInfo.BRANCH_LABEL);
	}

	public ScopeID getScopeID() {
		if (m_ScopeID == null && getID() != null)
			m_ScopeID = new ScopeID(getID(), SCOPE_TYPE);
		return m_ScopeID;
	}

	/*
	 * (non-Javadoc)
	 * 
	 * @see com.bluepumpkin.common.datatypes.ScopeableVO#getParentScopeID()
	 */
	public ScopeID getParentScopeID() {
		if (m_ParentScopeID == null && getParentID() != null)
			m_ParentScopeID = new ScopeID(getParentID(), SCOPE_TYPE);
		return m_ParentScopeID;
	}

}
