package com.verint.web.rest.wfm.resource.calendareventassignment;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.*;
import com.bluepumpkin.ejb.bbm.employeefilter.ejb.EmployeeFilter;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.schedule.model.BbmScheduleConflictException;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventAssignment;
import com.bluepumpkin.ejb.core.base.MultiUserException;
import com.google.common.collect.Sets;
import com.verint.web.rest.wfm.exception.EntityCreateException;
import com.verint.web.rest.wfm.resource.shiftassignment.EntityScheduleConflictException;
import com.verint.web.rest.wfm.repository.RepositoryLogger;
import com.verint.web.rest.wfm.repository.ValidateJsonApiFilterParams;
import com.verint.web.rest.wfm.repository.WfoResourceRepository;
import com.verint.web.rest.wfm.resource.DateAndEmployeeIdComboFilterParamHandler;
import io.katharsis.queryParams.QueryParams;
import io.katharsis.repository.annotations.JsonApiResourceRepository;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.inject.Inject;
import javax.ws.rs.container.ContainerRequestContext;
import java.rmi.RemoteException;
import java.util.*;

import static com.verint.web.rest.wfm.filter.DateFilterParamHandler.END_DATE_DEFAULT_FILTER_KEY;
import static com.verint.web.rest.wfm.filter.DateFilterParamHandler.START_DATE_DEFAULT_FILTER_KEY;
import static com.verint.web.rest.wfm.filter.EmployeeFilterNameFilterParamHandler.EMPLOYEE_FILTER_NAME_FILTER_KEY;
import static com.verint.web.rest.wfm.filter.EmployeeIdsFilterParamHandler.EMPLOYEE_IDS_FILTER_KEY;
import static com.verint.web.rest.wfm.resource.calendareventassignment.CalendarEventTypeFilterParamHandler.EVENT_TYPE_FILTER_PARAM_KEY;

@JsonApiResourceRepository(CalendarEventAssignmentEntity.class)
public class CalendarEventAssignmentRepository extends WfoResourceRepository<Integer, CalendarEventAssignmentEntity> {

	private static final Logger log = LoggerFactory.getLogger(CalendarEventAssignmentRepository.class.getName());

	private static final String ENTITY_NAME = CalendarEventAssignmentEntity.class.getSimpleName();
	static final String RESOURCE_NAME = "calendareventassignments";

	@Inject
	private ScheduleAccessManager sam;

	@Inject
	private EmployeeFilter ef;

	@Override
	protected Logger logger() { return log; }

	@Override
	@RepositoryLogger
	public CalendarEventAssignmentEntity findOne(Integer id, QueryParams requestParams, ContainerRequestContext requestContext) {
		try {
			ID idObj = ID.fromInt(id);
			CalendarEventAssignment cea = sam.getCalendarEventAssignmentByID(idObj);
			return CalendarEventAssignmentMapper.INSTANCE.ceaToCeaEntity(cea);
		} catch (BbmObjectNotFoundException e) {
			throw handleBbmObjectNotFoundException(ENTITY_NAME, Integer.toString(id), e);
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, Integer.toString(id), e);
		} catch (BbmFinderException e) {
			throw handleBbmFinderException(ENTITY_NAME, Integer.toString(id), e);
		}
	}

	@Override
	@RepositoryLogger
	@ValidateJsonApiFilterParams
	public Iterable<CalendarEventAssignmentEntity> findAll(QueryParams requestParams, ContainerRequestContext requestContext) {
		try {
			CalendarEventType eventType =
					CalendarEventTypeFilterParamHandler.getFilterValue(getFilterParam(EVENT_TYPE_FILTER_PARAM_KEY));
			DateAndEmployeeIdComboFilterParamHandler filterParamHandler =
					new DateAndEmployeeIdComboFilterParamHandler(ef, getCurrentUser(requestContext), getFilterParams(),
							getValidFilterParamKeys(), RESOURCE_NAME);

			if (filterParamHandler.getEmployeeIds().isEmpty()) {
				return Collections.emptyList();
			}
			List<Collection<CalendarEventAssignment>> assignments = sam.getEventsForWorkResourcesByType(
					eventType.getMask(), filterParamHandler.getEmployeeIds(),
					filterParamHandler.getStartDate(), filterParamHandler.getEndDate());
			Set<CalendarEventAssignment> consolidatedAssignments =
					CalendarEventAssignmentEntityUtil.consolidateAndFilterAssignments(sam, assignments);
			return CalendarEventAssignmentMapper.INSTANCE.ceasToCeaEntities(consolidatedAssignments);
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, null, e);
		} catch (BbmFinderException e) {
			throw handleBbmFinderException(ENTITY_NAME, null, e);
		}
	}

	@Override
	@RepositoryLogger
	@ValidateJsonApiFilterParams
	public Iterable<CalendarEventAssignmentEntity> findAll(Iterable<Integer> ids, QueryParams queryParams, ContainerRequestContext requestContext) {
		Collection<CalendarEventAssignmentEntity> assignments = new ArrayList<>();
		for (Integer id : ids) {
			assignments.add(findOne(id, queryParams, requestContext));
		}
		return assignments;
	}

	@Override
	@RepositoryLogger
	public void delete(Integer id, ContainerRequestContext requestContext) {
		try {
			ID idObj = ID.fromInt(id);
			sam.deleteCalendarEventAssignments(Collections.singleton(idObj));
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, Integer.toString(id), e);
		} catch (MultiUserException e) {
			throw handleMultiUserException(ENTITY_NAME, Integer.toString(id), e);
		} catch (BbmRemoveException e) {
			throw handleBbmRemoveException(ENTITY_NAME, Integer.toString(id), e);
		}
	}

	@Override
	@RepositoryLogger
	public <T extends CalendarEventAssignmentEntity> T save(T entity, ContainerRequestContext requestContext) {
		String eventType = entity.getEventType();
		if (eventType == null) {
			throw new EntityCreateException(entity);
		}
		String entityID = entity.getID() == null ? null : Integer.toString(entity.getID());
		try {
			if (entityID == null) {
				CalendarEventAssignment cea = CalendarEventAssignmentEntityUtil.createCEA(entity);
				ID ceID = sam.createCalendarEventAssignment(cea);
				entity.setID(ceID.toInt());
				entity.setEventType(eventType);
				return entity;
			} else {
				if (entity.getCalendarEventAttendees().isEmpty()) {
					delete(entity.getID(), requestContext);
					return null;
				}
				return CalendarEventAssignmentRepositoryUtil.updateCEAByType(entity, sam);
			}
		} catch (BbmCreateException e) {
			throw handleBbmCreateException(entity, e);
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, entityID, e);
		} catch (BbmScheduleConflictException e) {
			log.warn("CalendarEventAssignmentRepository.save: failed to save CalendarEventAssignment", e);
			throw new EntityScheduleConflictException(getLocalizer(requestContext), getUserTimeZone(requestContext), e.getConflicts());
		} catch (MultiUserException e) {
			throw handleMultiUserException(ENTITY_NAME, entityID, e);
		} catch (BbmObjectNotFoundException e) {
			throw handleBbmObjectNotFoundException(ENTITY_NAME, entityID, e);
		} catch (BbmUpdateException e) {
			throw handleBbmUpdateException(entity, e);
		} catch (BbmFinderException e) {
			throw handleBbmFinderException(ENTITY_NAME, Integer.toString(entity.getID()), e);
		}
	}

	@Override
	protected String getResourceName() {
		return RESOURCE_NAME;
	}

	@Override
	protected Set<String> getValidFilterParamKeys() {
		return Sets.newHashSet(EMPLOYEE_IDS_FILTER_KEY,
				START_DATE_DEFAULT_FILTER_KEY, END_DATE_DEFAULT_FILTER_KEY, EVENT_TYPE_FILTER_PARAM_KEY,
				EMPLOYEE_FILTER_NAME_FILTER_KEY);
	}
}
