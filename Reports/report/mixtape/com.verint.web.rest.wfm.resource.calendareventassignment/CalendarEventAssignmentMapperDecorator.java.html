package com.verint.web.rest.wfm.resource.calendareventassignment;

import com.bluepumpkin.ejb.bbm.activity.model.Event;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEvent;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventAssignment;
import com.bluepumpkin.ejb.bbm.schedule.model.TimeOffEvent;

import java.util.ArrayList;
import java.util.Collection;

abstract class CalendarEventAssignmentMapperDecorator implements CalendarEventAssignmentMapper {

	private final CalendarEventAssignmentMapper delegate;

	CalendarEventAssignmentMapperDecorator(CalendarEventAssignmentMapper delegate) { this.delegate = delegate; }

	@Override
	public CalendarEventAssignmentEntity ceaToCeaEntity(CalendarEventAssignment source) {
		CalendarEventAssignmentEntity entity = delegate.ceaToCeaEntity(source);
		// Converting Integer representation of events to string for better readability on client side.
		switch (source.getEventType()) {
			case Event.EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT:
				entity.setEventType(CalendarEventType.calendar.name());
				int overlapType = (int) ((CalendarEvent)source).getOverlapType();
				entity.setOverlapType(overlapType);
				break;
			case Event.EVENT_TYPE_TIME_OFF:
				entity.setEventType(CalendarEventType.timeoff.name());
				entity.setCountsMinsTowardsRules(((TimeOffEvent)source).getCountsMinutesTowardsRules());
				int timeOffRule = (int) ((TimeOffEvent)source).getTimeOffRule();
				entity.setTimeOffRule(timeOffRule);
				break;
			case Event.EVENT_TYPE_UNAVAILABILITY:
				entity.setEventType(CalendarEventType.unavailability.name());
				break;
			default:
				entity.setEventType("unspecified");
				break;
		}
		entity.setIsLocked(source.isLocked());
		return entity;
	}

	@Override
	public Collection<CalendarEventAssignmentEntity> ceasToCeaEntities(Collection<CalendarEventAssignment> source) {
		Collection<CalendarEventAssignmentEntity> entities = new ArrayList<>();
		for (CalendarEventAssignment cea : source) {
			entities.add(ceaToCeaEntity(cea));
		}
		return entities;
	}

}
