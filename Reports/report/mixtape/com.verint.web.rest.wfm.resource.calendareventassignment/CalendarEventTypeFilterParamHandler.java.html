package com.verint.web.rest.wfm.resource.calendareventassignment;


import com.verint.web.rest.wfm.filter.FilterParamHandler;

import java.util.Optional;
import java.util.Set;

class CalendarEventTypeFilterParamHandler extends FilterParamHandler {

	static final String EVENT_TYPE_FILTER_PARAM_KEY = "eventType";

	/**
	 * Returns a CalendarEventType value from the given set of parameters.  If none was sent in the request,
	 * this method returns CalendarEventType.all.
	 */
	static CalendarEventType getFilterValue(Set<String> params) throws InvalidCalendarEventTypeException {
		Optional<String> stringValue = getSingleFilterValue(params);
		if (stringValue.isPresent()) {
			try {
				return CalendarEventType.valueOf(stringValue.get());
			} catch (IllegalArgumentException e) {
				throw new InvalidCalendarEventTypeException(null, stringValue.get());
			}
		}
		return CalendarEventType.all;
	}
}
