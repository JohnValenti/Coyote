package com.verint.web.rest.wfm.resource.calendareventassignment;

import com.verint.web.rest.wfm.repository.WfoRelationshipRepository;
import com.verint.web.rest.wfm.repository.WfoResourceRepository;
import com.verint.web.rest.wfm.resource.activity.ActivityEntity;
import com.verint.web.rest.wfm.resource.activity.ActivityRepository;
import io.katharsis.repository.annotations.JsonApiRelationshipRepository;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import javax.inject.Inject;

@JsonApiRelationshipRepository(source = CalendarEventAssignmentEntity.class, target = ActivityEntity.class)
public class CalendarEventAssignmentToActivityRepository
		extends WfoRelationshipRepository<Integer, Integer, CalendarEventAssignmentEntity, ActivityEntity> {

	private static final Logger log = LoggerFactory.getLogger(CalendarEventAssignmentToActivityRepository.class.getName());

	@Inject
	private CalendarEventAssignmentRepository ceaRepository;
	@Inject
	private ActivityRepository activityRepository;


	@Override
	public WfoResourceRepository<Integer, CalendarEventAssignmentEntity> getParentRepository() {
		return ceaRepository;
	}

	@Override
	public WfoResourceRepository<Integer, ActivityEntity> getChildRepository() {
		return activityRepository;
	}

	@Override
	protected Logger logger() { return log; }

}
