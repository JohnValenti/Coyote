package com.verint.web.rest.wfm.resource.calendareventassignment;

import com.bluepumpkin.ejb.bbm.activity.model.Event;

enum CalendarEventType {

	all(Event.EVENT_TYPE_ALL_CALENDAR_EVENTS),
	calendar(Event.EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT),
	unavailability(Event.EVENT_TYPE_UNAVAILABILITY),
	timeoff(Event.EVENT_TYPE_TIME_OFF);

	private final int mask;

	CalendarEventType(int mask) {
		this.mask = mask;
	}

	public int getMask() {
		return mask;
	}
}
