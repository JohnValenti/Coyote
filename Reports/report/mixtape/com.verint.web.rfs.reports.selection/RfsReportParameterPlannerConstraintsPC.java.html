		m_HCvsActualPC = new FormTwoColumnPC(context);

		m_HCvsBudgetPC.setTitle("Head Count Compared to Budget");
	//	addChildComponent(m_HCvsBudgetPC);

		m_headCountInputPC = new RfsReportParameterPlannerMultiInputPC(context, toolbar, controlNode);

	public String getHtmlDisplay(RffRequest rff, Node controlNode,boolean readOnly) {
		List<StringsPair> empCategories = rfsReportUtil.getOptions(m_context, controlNode, "", RfsReportParameterUtil.EMPLOYEE_CATEGORIES, rff.getParentId());
		if(empCategories != null) {
			
			for(Iterator<StringsPair> iter = empCategories.iterator(); iter.hasNext();) {
				StringsPair empCat = (StringsPair) iter.next();
				
				m_headCountPC.addRow(empCat.getValue(),
							m_headCountInputPC.getHtmlDisplay(rff, controlNode, readOnly, RfsReportParameterUtil.PARAM_HEADCOUNT, empCat.getKey()));
				m_percentHeadCountPC.addRow(empCat.getValue(),
						m_percentHCInputPC.getHtmlDisplay(rff, controlNode, readOnly, RfsReportParameterUtil.PARAM_PERCENT_HEADCOUNT, empCat.getKey()));
				m_percentFTEPC.addRow(empCat.getValue(),
						m_percentFTEInputPC.getHtmlDisplay(rff, controlNode, readOnly, RfsReportParameterUtil.PARAM_PERCENT_FTE, empCat.getKey()));
				m_HCvsActualPC.addRow(empCat.getValue(),
						m_HCvsActualInputPC.getHtmlDisplay(rff, controlNode, readOnly, RfsReportParameterUtil.PARAM_HC_VS_ACTUAL, empCat.getKey()));
				m_HCvsBudgetPC.addRow(empCat.getValue(),
						m_HCvsBudgetInputPC.getHtmlDisplay(rff, controlNode, readOnly, RfsReportParameterUtil.PARAM_HC_VS_BUDGET, empCat.getKey()));
				
			}
		}
		
		m_forecastPC.addRow("Max % Overstaffed", m_forecastInputPC.getHtmlDisplay());

		StringBuffer content = new StringBuffer();
		//content.append(m_headCountPC.getHtmlDisplay())
			//	.append(HtmlUtil.NEW_LINE);
				//.append(m_percentHeadCountPC.getHtmlDisplay())
				//.append(HtmlUtil.NEW_LINE)
				//.append(m_percentFTEPC.getHtmlDisplay())
				//.append(HtmlUtil.NEW_LINE);
				//.append(m_HCvsActualPC.getHtmlDisplay())
				//.append(HtmlUtil.NEW_LINE)
				//.append(m_HCvsBudgetPC.getHtmlDisplay())
			//	.append(HtmlUtil.NEW_LINE)
				//.append(m_forecastPC.getHtmlDisplay());
		return content.toString();
	}
