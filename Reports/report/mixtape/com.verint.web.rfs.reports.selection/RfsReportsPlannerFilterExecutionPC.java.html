package com.verint.web.rfs.reports.selection;

import java.util.Iterator;
import java.util.List;

import javax.servlet.http.HttpServletRequest;

import org.w3c.dom.Node;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.web.core.reports.ReportControlsXml;
import com.bluepumpkin.web.core.reports.ReportParametersPCInterface;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.bluepumpkin.web.ireport.framework.RffParameter;
import com.bluepumpkin.web.ireport.framework.RffRequest;
import com.verint.web.rfs.reports.util.RfsReportControlsXml;
import com.verint.web.rfs.reports.util.RfsReportParameterUtil;
import com.witness.web.uif.pagecomponent.CompositePC;
import com.witness.web.uif.pagecomponent.form.FormInputPC;
import com.witness.web.uif.pagecomponent.form.FormTwoColumnPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;
import com.witness.web.uif.util.html.HtmlChoiceInput;
import com.witness.web.uif.util.js.JSUtil;

public class RfsReportsPlannerFilterExecutionPC extends CompositePC implements ReportParametersPCInterface {

	private static final long serialVersionUID = 1L;
	
	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();
	private static RfsReportControlsXml rfsReportControls = RfsReportControlsXml.getInstance();
	private static RfsReportParameterUtil rfsReportUtil = RfsReportParameterUtil.getInstance();

	private FormTwoColumnPC m_executionPC = null;	
	private FormInputPC m_empCatPC = null;
	
	public RfsReportsPlannerFilterExecutionPC(RequestContext context, ToolbarPC toolbar, Node controlNode) {
		super(context);
		setName(reportControls.getControlSingleParameterName(controlNode));		
		m_empCatPC = new FormInputPC(context);
		m_executionPC = new FormTwoColumnPC(context);
		m_executionPC.setTitle(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"AUTOMATICALLY_TARGET_STAFFMIX"));
		addChildComponent(m_executionPC);
	}
	
	public void initForDisplay() {
		if (m_isInitializedForDisplay)
			return;

		super.initForDisplay();
	}
	
	public String getLabel(RffRequest rff, Node controlNode) {
		String label = reportControls.getControlLabel(controlNode);
		if (label == null || label.length() == 0) {
			label = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"CHOOSE_A_TARGET");
		}
		return label;
	}

	public String getHtmlDisplay() {
		// TODO Auto-generated method stub
		return null;
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode) {
		return getHtmlDisplay(rff, controlNode, false);
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode,boolean readOnly) {
				
		StringBuffer leftObj = new StringBuffer();
		StringBuffer rightObj = new StringBuffer();

		String currentValue = m_context.getRequest().getParameter("ClosestTo");
		
		leftObj.append(HtmlUtil.NEW_LINE)
			.append(HtmlChoiceInput.makeRadioInput("ClosestTo", 
													"ACTUAL", 
													m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"CLOSEST_TO_ACTUAL"), 
													RfsReportParameterUtil.STAFFMIX_CLOSESTTO_ACTUAL.equals(currentValue),
													JSUtil.ON_CHANGE + JSUtil.REFRESH_PAGE_NO_CONFIRM))
	  	   .append(HtmlUtil.NEW_LINE).append(HtmlUtil.NEW_LINE)
		   .append(HtmlChoiceInput.makeRadioInput("ClosestTo", 
				   									"BUDGET", 
				   									m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"CLOSEST_TO_BUDGETED"), 
				   									RfsReportParameterUtil.STAFFMIX_CLOSESTTO_BUDGET.equals(currentValue),
				   									JSUtil.ON_CHANGE + JSUtil.REFRESH_PAGE_NO_CONFIRM));	
	
		rightObj.append(HtmlUtil.NEW_LINE)
			.append(HtmlChoiceInput.makeRadioInput("ClosestTo", 
													"TARGET",
													m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"CLOSEST_TO_RATIOS"), 
													RfsReportParameterUtil.STAFFMIX_CLOSESTTO_TARGET.equals(currentValue),
													JSUtil.ON_CHANGE + JSUtil.REFRESH_PAGE_NO_CONFIRM))
			.append(HtmlUtil.NEW_LINE).append(HtmlUtil.NEW_LINE);
		
		List<StringsPair> empCategories = rfsReportUtil.getOptions(m_context, controlNode, "", RfsReportParameterUtil.EMPLOYEE_CATEGORIES, rff.getParentId());
		if(empCategories != null) {
			for(Iterator<StringsPair> iter = empCategories.iterator(); iter.hasNext();) {
				StringsPair empCat = (StringsPair) iter.next();
				
				m_empCatPC.setInputType(FormInputPC.TYPE_INTEGER);
				m_empCatPC.setInputValue("0");
				m_empCatPC.setInputFN("TargetPercentFTEs" + empCat.getKey());				
				if (RfsReportParameterUtil.STAFFMIX_CLOSESTTO_TARGET.equals(currentValue))
					m_empCatPC.setEnabled(true);
				else
					m_empCatPC.setEnabled(false); 
				
				rightObj.append(empCat.getValue())
						.append(" : ")
						.append(m_empCatPC.getHtmlDisplay())
						.append(HtmlUtil.NBSP)
						.append(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"PERCENT_FTE"))
						.append(HtmlUtil.NEW_LINE);
			}
		}		
		m_executionPC.addRow(leftObj.toString(), rightObj.toString());
		
		return m_executionPC.getHtmlDisplay();
	}

	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode) {
		Node closestToActualParamNode = rfsReportControls.getControlParamClosestToActual(controlNode);
		RffParameter closestToActualParam = rff.getRffParameter(reportControls.getParameterName(closestToActualParamNode));
		closestToActualParam.setCurrentValue(RfsReportParameterUtil.STAFFMIX_CLOSESTTO_ACTUAL.equals(request.getParameter("ClosestTo")));

		Node closestToBudgetParamNode = rfsReportControls.getControlParamClosestToBudget(controlNode);
		RffParameter closestToBudgetParam = rff.getRffParameter(reportControls.getParameterName(closestToBudgetParamNode));
		closestToBudgetParam.setCurrentValue(RfsReportParameterUtil.STAFFMIX_CLOSESTTO_BUDGET.equals(request.getParameter("ClosestTo")));

		Node closestToRatiosParamNode = rfsReportControls.getControlParamClosestToRatios(controlNode);
		RffParameter closestToRatiosParam = rff.getRffParameter(reportControls.getParameterName(closestToRatiosParamNode));
		closestToRatiosParam.setCurrentValue(RfsReportParameterUtil.STAFFMIX_CLOSESTTO_TARGET.equals(request.getParameter("ClosestTo")));

		Node targetPercentFTEsNode = rfsReportControls.getControlParamTargetPercentFTEs(controlNode);
		RffParameter targetPercentFTEsParam = rff.getRffParameter(reportControls.getParameterName(targetPercentFTEsNode));
		
		String percentFTEsValue = "";
		List<StringsPair> empCategories = rfsReportUtil.getOptions(m_context, controlNode, "", RfsReportParameterUtil.EMPLOYEE_CATEGORIES, rff.getParentId());
		if(empCategories != null) {
			for(Iterator<StringsPair> iter = empCategories.iterator(); iter.hasNext();) {
				StringsPair empCat = (StringsPair) iter.next();
				percentFTEsValue += percentFTEsValue != "" ? "," + request.getParameter("TargetPercentFTEs"+empCat.getKey()) : request.getParameter("TargetPercentFTEs"+empCat.getKey());
			}
		}
		targetPercentFTEsParam.setCurrentValue(percentFTEsValue);
	}
	
	public String getCustomJavaScript(Node controlNode) {
		return null;
	}
	
	public boolean extractParameters(HttpServletRequest request) {
		super.extractParameters(request);
		return true;
	}
}
