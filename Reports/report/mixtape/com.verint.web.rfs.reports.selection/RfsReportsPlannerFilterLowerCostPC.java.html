package com.verint.web.rfs.reports.selection;

import javax.servlet.http.HttpServletRequest;

import org.w3c.dom.Node;

import com.bluepumpkin.web.core.reports.ReportControlsXml;
import com.bluepumpkin.web.core.reports.ReportParametersPCInterface;
import com.bluepumpkin.web.ireport.framework.RffParameter;
import com.bluepumpkin.web.ireport.framework.RffRequest;
import com.witness.web.uif.pagecomponent.CompositePC;
import com.witness.web.uif.pagecomponent.form.FormInputPC;
import com.witness.web.uif.pagecomponent.form.FormTwoColumnPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;

public class RfsReportsPlannerFilterLowerCostPC extends CompositePC implements ReportParametersPCInterface {

	private static final long serialVersionUID = 1L;
	
	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();
	private FormTwoColumnPC m_lowerCostPC = null;	
	private FormInputPC m_lowerCostInputPC = null;		
	
	public RfsReportsPlannerFilterLowerCostPC(RequestContext context, ToolbarPC toolbar, Node controlNode) {
		super(context);
		setName(reportControls.getControlSingleParameterName(controlNode));
		
		m_lowerCostPC = new FormTwoColumnPC(context);
		m_lowerCostPC.setTitle("Cost");
		m_lowerCostInputPC = new FormInputPC(context);
	}
	
	
	
	public void initForDisplay() {
		if (m_isInitializedForDisplay)
			return;

		super.initForDisplay();
	}
	
	public String getLabel(RffRequest rff, Node controlNode) {
		String label = reportControls.getControlLabel(controlNode);
		if (label == null || label.length() == 0) {
			label = "Lower Cost";
		}
		return label;
	}

	public String getHtmlDisplay() {
		// TODO Auto-generated method stub
		return null;
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode) {
		return getHtmlDisplay(rff, controlNode, false);
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode,boolean readOnly) {
				
		m_lowerCostInputPC.setInputFN("UpperCountStaffMixCost");
		m_lowerCostInputPC.setInputType(FormInputPC.TYPE_INTEGER);
		m_lowerCostInputPC.setInputValue("50");
		m_lowerCostPC.addRow("Lowest N options based on cost", m_lowerCostInputPC.getHtmlDisplay());

		return m_lowerCostPC.getHtmlDisplay();
	}

	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode) {
		Node paramNode = reportControls.getParameter(controlNode);
		if (paramNode == null) {
			throw new IllegalArgumentException("A parameter was not specified for this control.");
		}
		String paramName = reportControls.getParameterName(paramNode);
		RffParameter param = rff.getRffParameter(paramName);
		if (param == null) {
			throw new IllegalArgumentException("Parameter in controls but not in report: " + paramName);
		}

//		reportUtil.setRffParameterValue(request, param, "STRING");

	}
	
	public String getCustomJavaScript(Node controlNode) {
		return null;
	}
	
	public boolean extractParameters(HttpServletRequest request) {
		super.extractParameters(request);
		return true;
	}
}
