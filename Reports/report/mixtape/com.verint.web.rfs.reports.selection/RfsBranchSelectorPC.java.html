package com.verint.web.rfs.reports.selection;

import java.util.Collection;

import javax.servlet.http.HttpServletRequest;

import com.verint.ejb.rfs.reports.model.RfsBranch;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.rfs.reports.RfsReportParametersModelHandler;
import com.witness.web.uif.pagecomponent.PageComponent;
import com.witness.web.uif.pagecomponent.selection.IDsSelectorPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.js.JSUtil;

public class RfsBranchSelectorPC extends IDsSelectorPC {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private boolean isEdited = false;
	
	public static final String REQUEST_HANDLER = "branch_selector_popup";

	private String m_currentScenario = null;
	
	public RfsBranchSelectorPC(RequestContext context, ToolbarPC toolbar) {
		super(context, toolbar, FsWebBundleKey.BUNDLE_NAME);
		setName("RfsReportParameterBranchSelectorPC");
		//m_isRefreshOnClose = true;
		setOnChange(JSUtil.REFRESH_PAGE_NO_CONFIRM);
		setTitle(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"BRANCH_PICKER"));
	}

	protected void appendValues(Object obj, StringBuffer displaySB, StringBuffer hiddenSB) {
		RfsBranch org = (RfsBranch)obj;
		if(org.getBranchId() != null) 
		{
			displaySB.append( org.getBranchLabel() );
			hiddenSB.append( org.getID());
		}
	}

	@SuppressWarnings("unchecked")
	protected Collection fetchDataFromIDs(Collection ids) throws Exception {
		return RfsReportParametersModelHandler.getBranchObjectsByIDs(m_context, ids);
	}

	public static void addSelectorPopupArg(PageComponent pc, String name, boolean isMultiSelectEnabled, boolean isRefreshOnClose, String currentScenario) {
		addSelectorPopupArg(pc, name, REQUEST_HANDLER, "Scenario="+currentScenario, SELECTOR_WIDTH_HEIGHT, isMultiSelectEnabled, isRefreshOnClose);
	}

	/**
	 * add Selector Popup Argument
	 */
	protected void addSelectorPopupArg() {
		try {
			m_currentScenario = RfsReportParametersModelHandler.getCurrentScenario(m_context);
		} catch(Exception e) {
			
		}		
		addSelectorPopupArg(this, m_name, m_isMultiSelectEnabled, m_isRefreshOnClose, m_currentScenario);
	}


	@Override
	public String getLabel() {
		// TODO Auto-generated method stub
		return m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,"MULTIPLE_BRANCHES");
	}

	public boolean isEdited() {
		return isEdited;
	}
	
	public void setEdited(boolean edited) {
		this.isEdited = edited;
	}
	
	
	public boolean extractParameters(HttpServletRequest request) {
		boolean success=super.extractParameters(request);
		String editedVal = getParameter(request, getEditedInputName());
		if (editedVal != null) {
			isEdited = Boolean.parseBoolean(editedVal);
		}
		return success;
	}
	
	public String getHtmlDisplay() {
		StringBuffer sb = new StringBuffer();
		appendEditedFlag(sb);
		return super.getHtmlDisplay() + sb.toString();
	}
	
	protected void appendEditedFlag(StringBuffer sb) {
		sb.append("<input type='hidden' id='" + getEditedInputID() + "' name='").append(getEditedInputName()).append("' value='").append(isEdited).append("'></input>");
	}
	
	private String getEditedInputName() {
		return getName() + "__edited";
	}
	
	public String getEditedInputID() {
		return getName() + "__edited_id";
	}
	
	public String getCustomJavaScript() {
	//	return JSUtil.ON_CHANGE ;
		return null;
	}


}
