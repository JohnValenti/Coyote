package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.ArrayList;
import java.util.Calendar;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit.WorkUnitType;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectRich;
import com.bluepumpkin.ejb.bbm.workrules.model.CalendarPeriod.CalendarUnit;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRuleFieldInfo.Column;

public class Rotation extends ValueObjectRich {
	private static final long serialVersionUID = 1L;
	
	// meta data information
	private static FieldInfo m_fieldInfo = new ComplexWorkRuleFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
		
	public static final String CALENDAR_PERIOD_RELATIONSHIP_KEY = "CALENDARPERIOD|ID";
	public static final String ROTATION_WORKUNIT_RELATIONSHIP_KEY = "WORKUNIT";
	
	public Rotation() {
		super();
		setFieldValue(ComplexWorkRuleFieldInfo.Column.WorkUnitCountConstraint.getIndex(), ComplexWorkRule.WorkUnitCountConstraint.Rotation.getDBValue());
	}

	public ID getID() {
		return (ID) this.getFieldValue(Column.SID.getIndex());
	}
	
	public ID getOrganizationID() {
		return (ID) this.getFieldValue(Column.OrganizationID.getIndex());
	}
	
	public void setOrganizationID(ID orgID) {
		this.setFieldValue(Column.OrganizationID.getIndex(), orgID);
	}
	
	public String getName() {
		return this.getFormattedFieldValue(Column.Name.getIndex());
	}

	public void setName(String name) {
		this.setFieldValue(Column.Name.getIndex(), name);
	}
	
	/**
	 * Returns the Alignment Date. The Alignment Date is a LocalDate specifying midnight
	 * of the date that the rotation starts. This date should fall on the owner organization's 
	 * week start.
	 * @return
	 */
	public LocalDate getAlignmentDate() {
		return getCalendarPeriod().getAlignmentDate();
	}
	
	public void setAlignmentDate(LocalDate alignmentDate) {
		alignmentDate.set(Calendar.HOUR_OF_DAY, 0);
		alignmentDate.set(Calendar.MINUTE, 0);
		alignmentDate.set(Calendar.SECOND, 0);
		alignmentDate.set(Calendar.MILLISECOND, 0);
		
		CalendarPeriod cp = getCalendarPeriod();
		// Null out ID to create a new record
		cp.setID(null);
		cp.setAlignmentDate(alignmentDate);
	}
	
	private void setCalendarPeriod(CalendarPeriod period) {
		setRelatedObjects(CALENDAR_PERIOD_RELATIONSHIP_KEY,
				new ArrayList<CalendarPeriod>(Collections.singleton(period)));
	}
	
	private CalendarPeriod getCalendarPeriod() {
		if (getRelatedObjects(CALENDAR_PERIOD_RELATIONSHIP_KEY).size() > 0) {
			return (CalendarPeriod)getRelatedObjects(CALENDAR_PERIOD_RELATIONSHIP_KEY).iterator().next();
		}
		CalendarPeriod period = new CalendarPeriod();
		period.setCalendarUnit(CalendarUnit.Week);
		setCalendarPeriod(period);
		return getCalendarPeriod();
	}
	
	public List<ShiftPattern> getShiftPatterns() {
		@SuppressWarnings("unchecked")
		Collection<WorkUnit> workUnits = getRelatedObjects(ROTATION_WORKUNIT_RELATIONSHIP_KEY);
		if (workUnits != null && !workUnits.isEmpty() ) {
			return workUnits.iterator().next().getShiftPatterns();
		}
		return Collections.emptyList();
	}
	
	public void setShiftPatterns(List<ShiftPattern> shiftPatterns) {
		@SuppressWarnings("unchecked")
		Collection<WorkUnit> workUnits = getRelatedObjects(ROTATION_WORKUNIT_RELATIONSHIP_KEY);
		if (workUnits == null || workUnits.isEmpty() ) {
			WorkUnit wu = new WorkUnit();
			wu.setWorkUnitType(WorkUnitType.Rotation);
			workUnits = Collections.singleton(wu);
			setRelatedObjects(ROTATION_WORKUNIT_RELATIONSHIP_KEY, workUnits);
		}
		workUnits.iterator().next().setShiftPatterns(shiftPatterns);
		setNumWeeks(shiftPatterns.size());
	}
	
	public String getCreatedBy() {
		return getFormattedFieldValue(Column.CreatedBy.getIndex());
	}

	public String getModifiedBy() {
		return getFormattedFieldValue(Column.ModifiedBy.getIndex());
	}

	public Date getCreationDate() {
		return getFieldValueDate(Column.CreatedOn.getIndex());
	}

	public Date getModificationDate() {
		return getFieldValueDate(Column.ModifiedOn.getIndex());
	}

	public int getNumWeeks() {
		return getCalendarPeriod().getUnitCount();
	}
	
	private void setNumWeeks(int numWeeks) {
		CalendarPeriod cp = getCalendarPeriod();
		// Null out ID to create a new record
		cp.setID(null);
		cp.setUnitCount(numWeeks);
	}
}
