package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.Calendar;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

@SuppressWarnings("serial")
public class CalendarPeriod extends ValueObjectBase {

	// meta data information
	private static FieldInfo m_fieldInfo = new CalendarPeriodFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	public enum CalendarUnit {
		Day(0, 1),
		Week(1, 7),
		Month(2, 30.5f),
		Year(3, 365.25f),
		RollingWeek(4, 7),
		Month28Day(5, 28),
		Month29Day(6, 29),
		Month30Day(7, 30),
		Month31Day(8, 31),
		;
		
		private int dbValue;
		private float numDays;
		private CalendarUnit(int dbValue, float numDays) {
			this.dbValue = dbValue;
			this.numDays = numDays;
		}
		public int getDBValue() {
			return dbValue;
		}

		/**
		 * Returns the approximate number of days in this Calendar Unit. Used for validation.
		 * @return
		 */
		public float getNumDays() {
			return numDays;
		}
		public static CalendarUnit fromDBValue(int dbValue) {
			for (CalendarUnit ut : CalendarUnit.values()) {
				if (dbValue == ut.dbValue) return ut;
			}
			throw new IllegalArgumentException("Unexpected CalendarUnit value of " + dbValue);
		}
	}

	public ID getID() {
		return new ID(this.getFormattedFieldValue(CalendarPeriodFieldInfo.CALENDARPERIOD_ID));
	}
	
	public int getUnitCount() {
		return this.getFieldValueInt(CalendarPeriodFieldInfo.CALENDARPERIOD_UNITCOUNT);
	}
	
	public void setUnitCount(int count) {
		this.setFieldValue(CalendarPeriodFieldInfo.CALENDARPERIOD_UNITCOUNT, count);
	}
	
	public LocalDate getAlignmentDate() {
		Date dbDate = this.getFieldValueDate(CalendarPeriodFieldInfo.CALENDARPERIOD_ALIGNMENTDATE);
		if (dbDate == null) return null;
		return new LocalDate(dbDate, TimeZoneUtil.GMT_TIMEZONE);
	}
	
	public void setAlignmentDate(LocalDate alignmentDate) {
		if (alignmentDate != null) {
			// Ensure that the data represents midnight, since this field is not used to store the time, only the calendar day.
			LocalDate newDate = null;
			newDate = new LocalDate(alignmentDate);
			newDate.set(Calendar.HOUR_OF_DAY, 0);
			newDate.set(Calendar.MINUTE, 0);
			newDate.set(Calendar.SECOND, 0);
			newDate.set(Calendar.MILLISECOND, 0);
			
			this.setFieldValue(CalendarPeriodFieldInfo.CALENDARPERIOD_ALIGNMENTDATE, newDate.getTime(TimeZoneUtil.GMT_TIMEZONE));
		} else {
			this.setFieldValue(CalendarPeriodFieldInfo.CALENDARPERIOD_ALIGNMENTDATE, (Date)null);
		}
	}
	
	public CalendarUnit getCalendarUnit() {
		return CalendarUnit.fromDBValue(this.getFieldValueInt(CalendarPeriodFieldInfo.CALENDARPERIOD_CALENDARUNIT));
	}
	
	public void setCalendarUnit(CalendarUnit unit) {
		setFieldValue(CalendarPeriodFieldInfo.CALENDARPERIOD_CALENDARUNIT, unit.getDBValue());
	}

}
