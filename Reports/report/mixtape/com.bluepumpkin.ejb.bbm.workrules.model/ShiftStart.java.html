package com.bluepumpkin.ejb.bbm.workrules.model;

public class ShiftStart extends ValueObjectBase {

	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	}

		return getFieldValueID(ShiftStartFieldInfo.SHIFTSTART_SHIFTID);
	}

		return getFieldValueInt(ShiftStartFieldInfo.SHIFTSTART_POSSIBLESTSTARTTIME);

	public void setPossibleStartTime(int startTime) { 
		setFieldValue(ShiftStartFieldInfo.SHIFTSTART_POSSIBLESTSTARTTIME, startTime);
	}

		return getFieldValueDate(ShiftStartFieldInfo.SHIFTSTART_CREATED);
	}

		setFieldValue(ShiftStartFieldInfo.SHIFTSTART_CREATED, createdDate);
	}

		return getFormattedFieldValue(ShiftStartFieldInfo.SHIFTSTART_CREATEDBY);
	}
	public void setCreatedBy(String createdBy) {
		setFieldValue(ShiftStartFieldInfo.SHIFTSTART_CREATEDBY, createdBy);
	}

		return getFieldValueDate(ShiftStartFieldInfo.SHIFTSTART_MODIFIED);

	public void setModifiedBy(String modifiedBy) {
