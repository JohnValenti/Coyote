import com.bluepumpkin.common.datatypes.ID;

public class SelectorElement implements java.io.Serializable {

  /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
private ID m_id;
  private String m_sName;
  private boolean m_bSelected;
  private int m_eType;


  public SelectorElement(ID id,String sName,boolean bSelected, int eType) {
    m_id = id;
    m_sName = sName;
    m_bSelected = bSelected;
    m_eType = eType;
  }


  public int getType() {return m_eType;}
  public ID getID() {return m_id;}
  public String getName() {return m_sName;}
  public boolean getSelected() {return m_bSelected;}

}
