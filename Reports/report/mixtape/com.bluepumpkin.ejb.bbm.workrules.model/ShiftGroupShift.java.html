package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class ShiftGroupShift extends ValueObjectBase {
	private static final long serialVersionUID = 1L;

	private static FieldInfo m_fieldInfo = new ShiftGroupShiftFieldInfo();
	/**
	 * Gets the day of the week starting with Monday as 1, to Sunday as 7, for each day checked for each shift rule. For consecutive days, Min is represented with 1000 and Max with 1001. For consistent starts, shifts on days with the same VALUE must start at the same time
	*/
	public int getDaySequence() { return getFieldValueInt(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_DAYSEQ); }
	public void setDaySequence(int daySequence) { setFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_DAYSEQ, daySequence); }
	
	/**
	 * Value is 1 for a day record, and represents number of days for min/max consecutive days with a ceiling of 7. Values for possible days off are indicated by a special shift id SHIFT_OFF. Values for consistent starts have SHIFTGROUPID linked to SHIFTGROUP named SAME-STARTS - value in this case is the number chosen in the drop-down for consistent starts
	 * @return
	 */
	public int getValue() { return getFieldValueInt(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_VALUE); }
	public void setValue(int value) { setFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_VALUE, value); }
	
	public Date getCreatedDate() { return getFieldValueDate(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_CREATED); }
	public void setCreatedDate(Date createdDate) { setFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_CREATED, createdDate); }
	
	public String getCreatedBy() { return getFormattedFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_CREATEDBY); }
	public void setCreatedBy(String createdBy) { setFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_CREATEDBY, createdBy); }

	public Date getModifiedDate() { return getFieldValueDate(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_MODIFIED); }
	public void setModifiedDate(Date modifiedDate) { setFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_MODIFIED, modifiedDate); }
	
	public String getModifiedBy() { return getFormattedFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_MODIFIEDBY); }
	public void setModifiedBy(String modifiedBy) { setFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_MODIFIEDBY, modifiedBy); }
	
	public ID getShiftGroupID() { return getFieldValueID(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_SHIFTGROUPID); }
	public void setShiftGroupID(ID shiftGroupID) { setFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_SHIFTGROUPID, shiftGroupID); }
	
	public ID getShiftID() { return getFieldValueID(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_SHIFTID); }
	public void setShiftID(ID shiftID) { setFieldValue(ShiftGroupShiftFieldInfo.SHIFTGROUPSHIFT_SHIFTID, shiftID); }

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}
}
