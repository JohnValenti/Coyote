	protected void buildDBRelationships() {
		seDAO = new ShiftEventDAO(m_dmo);
		sesDAO = new ShiftEventShiftDAO(m_dmo);
		ssDAO = new ShiftStartDAO(m_dmo);
		
		addRelationship(new OneToManyRelationship(this, ssDAO, getFieldInfo().getFieldIndexByName("ID"),
				ssDAO.getFieldInfo().getFieldIndexByName("SHIFTID"), Shift.SHIFT_START_RELATIONSHIP_KEY));
		addRelationship(new JunctionTableRelationship(this, seDAO, sesDAO,
				getFieldInfo().getFieldIndexByName("ID"),
				sesDAO.getFieldInfo().getFieldIndexByName("SHIFTID"),
				sesDAO.getFieldInfo().getFieldIndexByName("SHIFTEVENTID"), 
				ssDAO.getFieldInfo().getFieldIndexByName("ID"), Shift.SHIFT_EVENT_RELATIONSHIP_KEY));
	}

	public Collection<Shift> getObjects(String strWhere) throws BbmFinderException
	{
		StringBuffer sb = new StringBuffer();
			sb.append("(A.DELETEONDATE IS NULL OR A.DELETEONDATE > ");
			sb.append(JdmoUtil.asSqlLiteral(new Date(System.currentTimeMillis())));
			sb.append(")");
			sb.append(strWhere == null ? "" : " AND " + strWhere);
		StringBuffer strSQL = getQuerySQL(sb);
		return getObjectsGivenFullSqlQueryStr(strSQL.toString());
	}
	
	public void copyShifts(Collection<ID> shiftIDs, boolean isOrgMode, ID ownerID, LocaleContext localeContext, String currentUser)
	throws BbmFinderException, BbmCreateException {
		CopyDAO copyDAO = new CopyDAO();
		
		try {
			//Before we initiate the copy operation we want to clear out the IDs on the input objects and their
			// children and set the create date/user
			ArrayList<Shift> shiftsToCopy = new ArrayList<Shift>(getObjectsByIDs(shiftIDs));

			for (Shift shift : shiftsToCopy) {
				shift.setName(copyDAO.getNewNameForCopiedValueObjectWithLengthLimitation(shift.getName(), "SHIFT", localeContext, ShiftFieldInfo.SHIFT_NAME_MAX_LENGTH));
				
				if (isOrgMode) {
					shift.setCampaignID(null);
					shift.setOrganizationID(ownerID);
				} else {
					shift.setOrganizationID(null);
					shift.setCampaignID(ownerID);
				}
				
				shift.setID(null);
				shift.setDEID(null);
				shift.setCreatedBy(currentUser);
				shift.setModifiedBy(null);
				shift.setCreatedDate(new Date(System.currentTimeMillis()));
				shift.setModifiedDate(null);
				
				//Clear the IDs on associated start times
				Collection<ShiftStart> shiftStarts = new ArrayList<ShiftStart>(shift.getShiftStarts());
				for (ShiftStart shiftStart : shiftStarts) {
					shiftStart.setID(null);
					shiftStart.setCreatedBy(currentUser);
					shiftStart.setModifiedBy(null);
					shiftStart.setCreatedDate(new Date(System.currentTimeMillis()));
					shiftStart.setModifiedDate(null);
				}
				shift.setShiftStarts(shiftStarts);
			}

			createObjects(shiftsToCopy);

		} catch (BbmFinderException ex) {
			throw new BbmFinderException(ex);
		} catch (BbmCreateException ex) {
			throw new BbmCreateException(ex);
		} finally {
		}
	}

	public PaginationPair<Shift> getCompletePaginationDataSetForOrg(ID orgID, boolean isAscendingSort, int pageIDSize,
			int pageIndex, LinkedHashMap<DBSortField, Comparator> sortFieldMap,
			Collection<ID> filteredIDs, String filterByName) throws BbmFinderException {
		StringBuffer whereClause = new StringBuffer();
		whereClause.append(DAOUtil.getCurrentObjectsByOrgHierarchySQLString(orgID));
		if (filterByName != null && filterByName.length() > 0) {
			whereClause.append(" AND A." + getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_NAME) + " LIKE '%").

		}
		return getCompletePaginationDataSet(whereClause.toString(),
				isAscendingSort, pageIDSize, pageIndex, sortFieldMap, filteredIDs);
	}

	public HashMap<ID, String> getShiftNamesByOrg(ID orgID) throws BbmFinderException {
		HashMap<ID, String> retVal = new HashMap<ID, String>();
		try {
			JdmoRowset r = getDMO().createRowset("SELECT " + getFieldInfo().getIDFieldName() + "," + 
					getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_NAME) + " FROM " + getFieldInfo().getTableName() +
					" A WHERE " + DAOUtil.getCurrentObjectsByOrgHierarchySQLString(orgID));
			while (r.next()) {
				retVal.put(r.getID(getFieldInfo().getIDFieldName()),
						r.getString(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_NAME)));
			}
		} catch (JdmoException ex) {
			throw new BbmFinderException(ex);
		}
		return retVal;
	}

	@Override
	protected void buildSortQuery(List<DBSortField> sortFields, StringBuffer selectQuery, StringBuffer fromQuery,
			StringBuffer whereQuery, StringBuffer groupByQuery, StringBuffer orderByQuery) {
		for (DBSortField sortField : sortFields) {
			switch((ShiftSortField)sortField) {
				case ID:
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getIDFieldName());
					break;
				case NAME:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_NAME));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_NAME));
					break;
				case ACTIVITY:
					selectQuery.append(", (SELECT B.").append(afi.getFieldName(ActivityFieldInfo.ACTIVITY_NAME)).append(
						" FROM ").append(afi.getTableName()).append(" B WHERE B.").append(afi.getIDFieldName()).append(
						" = A.").append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_ACTIVITYID)).append(
						") AS ").append(afi.getTableName()).append("_").append(
								afi.getFieldName(ActivityFieldInfo.ACTIVITY_NAME));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(activitySortFieldKey);
					break;
				case DURATION:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_SHIFTLENGTH));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_SHIFTLENGTH));						
					break;
				case MIN_SPACING:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_MINSPACING));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_MINSPACING));						
					break;
				case MAX_SPACING:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_MAXSPACING));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_MAXSPACING));						
					break;
				case START_TIME:
					selectQuery.append(", A." + getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_ID) + ", min(C.").
						append(ssfi.getFieldName(ShiftStartFieldInfo.SHIFTSTART_POSSIBLESTSTARTTIME) + ") as STARTTIME");
					fromQuery.append(", " + ssfi.getTableName() + " C");
					fromQuery.append(", " + ofi.getTableName() + " D");
					whereQuery.append(" AND C." + ssfi.getFieldName(ShiftStartFieldInfo.SHIFTSTART_SHIFTID) +
							" = A." + getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_ID));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(" STARTTIME");
					groupByQuery.append(" GROUP BY A." + getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_ID) +
							", A." + getFieldInfo().getIDFieldName());
					break;
				case ORGANIZATION:
					selectQuery.append(", (SELECT E.").append(ofi.getFieldName(OrganizationFieldInfo.ORGANIZATION_NAME)).append(
						" FROM ").append(ofi.getTableName()).append(" E WHERE E.").append(ofi.getIDFieldName()).append(
						" = A.").append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_ORGANIZATIONID)).append(
						") AS ").append(organizationSortFieldKey);
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(organizationSortFieldKey);				
					break;
				case DESCRIPTION:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_DESCRIPTION));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_DESCRIPTION));						
					break;
				default: break;
			}
		}
	}
	
	protected Object getPaginationDataFieldValue(JdmoRowset rowData, DBSortField sortField) throws JdmoException {
		ShiftSortField ssf = (ShiftSortField)sortField;
		switch(ssf) {
			case NAME:
				return rowData.getString(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_NAME));
			case ACTIVITY:
				return rowData.getString(activitySortFieldKey);
			case DURATION:
				return rowData.getString(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_SHIFTLENGTH));
			case START_TIME:
				return rowData.getInt("STARTTIME");
			case MIN_SPACING:
				return rowData.getString(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_MINSPACING));
			case MAX_SPACING:
				return rowData.getString(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_MAXSPACING));
			case ORGANIZATION:
				return rowData.getString(organizationSortFieldKey);
			case DESCRIPTION:
				return rowData.getString(getFieldInfo().getFieldName(ShiftFieldInfo.SHIFT_DESCRIPTION));
			default: return null;
		}
	}
	
	@Override
	protected Map<Integer, ? extends FieldInfo> getForeignKeyFieldInfo() {
		Map<Integer, FieldInfo> retVal = new HashMap<Integer, FieldInfo>();
		retVal.put(ShiftFieldInfo.SHIFT_CAMPAIGNID, new CampaignFieldInfo());
		return retVal;
	}
