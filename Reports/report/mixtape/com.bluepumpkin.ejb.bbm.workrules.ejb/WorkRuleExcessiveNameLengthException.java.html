package com.bluepumpkin.ejb.bbm.workrules.ejb;

import com.bluepumpkin.ejb.bbm.base.BbmFinderException;

/**
 * Exception intended to be thrown by the Work Rule DAOs when a copy operation
 * is attempted and the new name generated for the copied work rule would be too
 * large for the name column in the database.
 */
public class WorkRuleExcessiveNameLengthException extends BbmFinderException {

	private static final long serialVersionUID = 6869917915352318430L;

	private final String nameOfViolatingWorkRule;

	public WorkRuleExcessiveNameLengthException(String nameOfViolatingWorkRule) {
		super();
		this.nameOfViolatingWorkRule = nameOfViolatingWorkRule;
	}

	public String getNameOfViolatingWorkRule() {
		return nameOfViolatingWorkRule;
	}
}
