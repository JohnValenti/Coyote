	public ShiftOTExtensionDAO() {

		super(dmo);
	}

	@Override
	protected void buildDBRelationships() {
		seDAO = new ShiftEventDAO(m_dmo);
		seoteDAO = new ShiftEventOTExtensionDAO(m_dmo);
		
		addRelationship(new JunctionTableRelationship(this, seDAO, seoteDAO,
				getFieldInfo().getFieldIndexByName("ID"),
				seoteDAO.getFieldInfo().getFieldIndexByName("SHIFTOTEXTENSIONID"),
				seoteDAO.getFieldInfo().getFieldIndexByName("SHIFTEVENTID"), 
				seDAO.getFieldInfo().getFieldIndexByName("ID"), ShiftOTExtension.SHIFT_EVENT_RELATIONSHIP_KEY ));	
		
	}

	public Collection<ShiftOTExtension> getOTExtensions(Collection<ID> otextensionIDs) throws BbmFinderException {
		try {
			return getObjects("(DELETEONDATE IS NULL OR DELETEONDATE > " +
					JdmoUtil.asSqlLiteral(new Date(System.currentTimeMillis())) +") AND SID IN (" +
					JdmoUtil.createIDList(otextensionIDs) + ")");
		} catch (Exception ex) {
			throw new BbmFinderException(ex);
		}
	}
	public void copyOTExtensions(Collection<ID> otExtensionIDs, boolean isOrgMode, ID ownerID, LocaleContext localeContext, String currentUser)
	throws BbmFinderException, BbmCreateException {
		CopyDAO copyDAO = new CopyDAO();
			
		try {
			//Before we initiate the copy operation we want to clear out the IDs on the input objects and their
			// children and set the create date/user
			ArrayList<ShiftOTExtension> otExtensionsToCopy = new ArrayList<ShiftOTExtension>(getOTExtensions(otExtensionIDs));
			
			for (ShiftOTExtension otExtension : otExtensionsToCopy) {
				otExtension.setName(copyDAO.getNewNameForCopiedValueObjectWithLengthLimitation(otExtension.getName(),
						"SHIFTOTEXTENSION", localeContext, ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_NAME_MAX_LENGTH));
				
				if (isOrgMode) {
					otExtension.setCampaignID(null);
					otExtension.setOrganizationID(ownerID);
				} else {
					otExtension.setOrganizationID(null);
					otExtension.setCampaignID(ownerID);
				}
				
				otExtension.setID(null);
				otExtension.setDEID(null);
				otExtension.setCreatedBy(currentUser);
				otExtension.setModifiedBy(null);
				otExtension.setCreatedDate(new Date(System.currentTimeMillis()));
				otExtension.setModifiedDate(null);
			}

			createObjects(otExtensionsToCopy);

		} catch (BbmFinderException ex) {
			throw new BbmFinderException(ex);
		} catch (BbmCreateException ex) {
			throw new BbmCreateException(ex);
		} finally {
			
		}
	}

	public HashMap<ID, String> getOTExtensionNamesByOrg(ID orgID) throws BbmFinderException {
		HashMap<ID, String> retVal = new HashMap<ID, String>();
		try {
			JdmoRowset r = getDMO().createRowset("SELECT " + getFieldInfo().getIDFieldName() + "," + 
					getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_NAME) + " FROM " + getFieldInfo().getTableName() +
					" A WHERE " + DAOUtil.getCurrentObjectsByOrgHierarchySQLString(orgID));
			while (r.next()) {
				retVal.put(r.getID(getFieldInfo().getIDFieldName()),
						r.getString(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_NAME)));
			}
		} catch (JdmoException ex) {
			throw new BbmFinderException(ex);
		}
		return retVal;
	}

	@Override
	protected void buildSortQuery(List<DBSortField> sortFields, StringBuffer selectQuery, StringBuffer fromQuery,
			StringBuffer whereQuery, StringBuffer groupByQuery, StringBuffer orderByQuery) {
		for (DBSortField sortField : sortFields) {
			switch((ShiftOTExtensionSortField)sortField) {
				case ID:
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getIDFieldName());
					break;
				case NAME:
					selectQuery.append(", A.").append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_NAME));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_NAME));
					break;
				case ACTIVITY:
					selectQuery.append(", (SELECT B.").append(afi.getFieldName(ActivityFieldInfo.ACTIVITY_NAME)).append(
							" FROM ").append(afi.getTableName()).append(" B WHERE B.").append(afi.getIDFieldName()).append(
							" = A.").append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_ACTIVITYID)).append(
							") AS ").append(afi.getTableName()).append("_").append(
							afi.getFieldName(ActivityFieldInfo.ACTIVITY_NAME));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(activitySortFieldKey);
					break;
				case LENGTH:
					selectQuery.append(", A.").append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_OTSHIFTLENGTH));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_OTSHIFTLENGTH));						
					break;
				case MIN_GAP:
					selectQuery.append(", A.").append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_MINGAP));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_MINGAP));						
					break;
				case MAX_GAP:
					selectQuery.append(", A.").append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_MAXGAP));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_MAXGAP));						
					break;
				case ORGANIZATION:
					selectQuery.append(", (SELECT C.").append(ofi.getFieldName(OrganizationFieldInfo.ORGANIZATION_NAME)).append(
						" FROM ").append(ofi.getTableName()).append(" C WHERE C.").append(ofi.getIDFieldName()).append(
						" = A.").append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_ORGANIZATIONID)).append(
						") AS ").append(organizationSortFieldKey);
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(organizationSortFieldKey);			
					break;
				case DESCRIPTION:
					selectQuery.append(", A.").append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_DESCRIPTION));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_DESCRIPTION));						
					break;
				default: break;
			}
		}
	}
	
	protected Object getPaginationDataFieldValue(JdmoRowset rowData, DBSortField sortField) throws JdmoException {
		ShiftOTExtensionSortField soteSortField = (ShiftOTExtensionSortField)sortField;
		switch(soteSortField) {
		case NAME:
			return rowData.getString(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_NAME));
		case ACTIVITY:
			return rowData.getString(activitySortFieldKey);
		case LENGTH:
			return rowData.getString(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_OTSHIFTLENGTH));
		case MIN_GAP:
			return rowData.getString(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_MINGAP));
		case MAX_GAP:
			return rowData.getString(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_MAXGAP));
		case ORGANIZATION:
			return rowData.getString(organizationSortFieldKey);				
		case DESCRIPTION:
			return rowData.getString(getFieldInfo().getFieldName(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_DESCRIPTION));
		default: return null;				
		}
	}
	
	@Override
	protected Map<Integer, ? extends FieldInfo> getForeignKeyFieldInfo() {
		Map<Integer, FieldInfo> retVal = new HashMap<Integer, FieldInfo>();
		retVal.put(ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_CAMPAIGNID, new CampaignFieldInfo());
		return retVal;
	}
