		sb.append("(A.DELETEONDATE IS NULL OR A.DELETEONDATE > ");
		sb.append(JdmoUtil.asSqlLiteral(new Date(System.currentTimeMillis())));
		sb.append(")");
		sb.append(strWhere == null ? "" : " AND " + strWhere);
		StringBuffer strSQL = getQuerySQL(sb);
		return getObjectsGivenFullSqlQueryStr(strSQL.toString());
	}

	public void copyProjects(
		Collection<ID> projIDs,
		boolean isOrgMode,
		ID ownerID,
		LocaleContext localeContext,
		String currentUser)
		throws BbmFinderException, BbmCreateException
	{
		CopyDAO copyDAO = new CopyDAO();

		try {
			//Before we initiate the copy operation we want to clear out the IDs on the input objects and their
			// children and set the create date/user
			ArrayList<ProjectRules> projectToCopy = new ArrayList<ProjectRules>(getObjectsByIDs(projIDs));

			for (ProjectRules projRule : projectToCopy) {
				projRule.setName(copyDAO.getNewNameForCopiedValueObjectWithLengthLimitation(projRule.getName(),
						"PROJECT", localeContext, ProjectRulesFieldInfo.PROJECT_NAME_MAX_LENGTH));

				if (isOrgMode) {
					projRule.setCampaignID(null);
					projRule.setOrganizationID(ownerID);
				}
				else {
					projRule.setOrganizationID(null);
					projRule.setCampaignID(ownerID);
				}

				projRule.setID(null);
				projRule.setDEID(null);
				projRule.setCreatedBy(currentUser);
				projRule.setModifiedBy(null);
				projRule.setCreatedDate(new Date(System.currentTimeMillis()));
				projRule.setModifiedDate(null);

				//Clear the IDs on associated start times
				Collection<ProjectStart> collProjStart = new ArrayList<ProjectStart>(projRule.getProjectStartTimes());
				for (ProjectStart projStart : collProjStart) {
					projStart.setID(null);
					projStart.setCreatedBy(currentUser);
					projStart.setModifiedBy(null);
					projStart.setCreatedDate(new Date(System.currentTimeMillis()));
					projStart.setModifiedDate(null);
				}
				projRule.setProjectStartTimes(collProjStart);
			}
			createObjects(projectToCopy);
		}
		catch (BbmFinderException ex) {
			throw new BbmFinderException(ex);
		}
		catch (BbmCreateException ex) {
			throw new BbmCreateException(ex);
		}
	}

	/**
	 * Returns a complete list of IDs and the first chunk of data to be displayed

	public PaginationPair<ProjectRules> getCompletePaginationDataSetForOrg(
		ID orgID, boolean isAscendingSort, int pageIDSize,
		int pageIndex, LinkedHashMap<DBSortField, Comparator> sortFieldMap,
		Collection<ID> filteredIDs, String filterByName)
		throws BbmFinderException
	{
		StringBuffer whereClause = new StringBuffer();
		whereClause.append(DAOUtil.getCurrentObjectsByOrgHierarchySQLString(orgID));
		if (filterByName != null && filterByName.length() > 0) {
			whereClause.append(" AND A.");
			whereClause.append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_NAME));
			whereClause.append(" LIKE '%");
			whereClause.append(filterByName).append("%'");
		}
		return super.getCompletePaginationDataSet(whereClause.toString(),
			isAscendingSort, pageIDSize, pageIndex, sortFieldMap, filteredIDs);
	}

	public HashMap<ID, String> getShiftNamesByOrg(ID orgID)
		throws BbmFinderException
	{
		HashMap<ID, String> retVal = new HashMap<ID, String>();
		try {
			JdmoRowset r = getDMO().createRowset("SELECT " + getFieldInfo().getIDFieldName() + ","
				+ getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_NAME) + " FROM " + getFieldInfo().getTableName()
				+ " A WHERE " + DAOUtil.getCurrentObjectsByOrgHierarchySQLString(orgID));
			while (r.next()) {
				retVal.put(r.getID(getFieldInfo().getIDFieldName()),
					r.getString(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_NAME)));
			}
		}
		catch (JdmoException ex) {
			throw new BbmFinderException(ex);
		}
		return retVal;
	}

	@Override
	protected void buildSortQuery(
		List<DBSortField> sortFields,
		StringBuffer selectQuery,
		StringBuffer fromQuery,
		StringBuffer whereQuery,
		StringBuffer groupByQuery,
		StringBuffer orderByQuery)
	{
		for (DBSortField sortField : sortFields) {
			switch ((ProjectRulesSortField)sortField) {
				case ID:
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getIDFieldName());
					break;
				case NAME:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_NAME));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_NAME));
					break;
				case DESCRIPTION:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_DESCRIPTION));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_DESCRIPTION));
					break;
				case ACTIVITY:
					selectQuery.append(", (SELECT B.").append(activityFieldInfo.getFieldName(ActivityFieldInfo.ACTIVITY_NAME)).append(
						" FROM ").append(activityFieldInfo.getTableName()).append(" B WHERE B.").append(activityFieldInfo.getIDFieldName()).append(
							" = A.").append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_ACTIVITYID)).append(
							") AS ").append(activityFieldInfo.getTableName()).append("_").append(
							activityFieldInfo.getFieldName(ActivityFieldInfo.ACTIVITY_NAME));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(activitySortFieldKey);
					break;
				case LENGTH:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_LENGTH));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_LENGTH));
					break;
				case ORGANIZATION:
					selectQuery.append(", (SELECT E.").append(organizationFieldInfo.getFieldName(OrganizationFieldInfo.ORGANIZATION_NAME)).append(
						" FROM ").append(organizationFieldInfo.getTableName()).append(" E WHERE E.").append(organizationFieldInfo.getIDFieldName()).append(
							" = A.").append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_ORGANIZATIONID)).append(
							") AS ").append(organizationSortFieldKey);
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(organizationSortFieldKey);
					break;
				case DAILYMAXPEREMPLOYEE:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_DAILYMAXPEREMPLOYEE));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_DAILYMAXPEREMPLOYEE));
					break;
				case WEEKLYMAXPEREMPLOYEE:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_WEEKLYMAXPEREMPLOYEE));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_WEEKLYMAXPEREMPLOYEE));
					break;
				case MAXSIMULTANEOUS:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_MAXSIMULTANEOUS));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_MAXSIMULTANEOUS));
					break;
				case START_TIME:
					selectQuery.append(", A." + getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_ID) + ", min(C.").
						append(projectStartFieldInfo.getFieldName(ProjectStartFieldInfo.PROJECTSTART_POSSIBLESTSTARTTIME) + ") as STARTTIME");
					fromQuery.append(", " + projectStartFieldInfo.getTableName() + " C");
					fromQuery.append(", " + organizationFieldInfo.getTableName() + " D");
					whereQuery.append(" AND C." + projectStartFieldInfo.getFieldName(ProjectStartFieldInfo.PROJECTSTART_PROJECTID)
						+ " = A." + getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_ID));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(" STARTTIME");
					groupByQuery.append(" GROUP BY A." + getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_ID)
						+ ", A." + getFieldInfo().getIDFieldName());
					String s = selectQuery.toString() + fromQuery.toString() + whereQuery.toString() + orderByQuery.toString() + groupByQuery.toString();
					break;
				default:
					break;
			}
		}
	}

	@Override
	protected Object getPaginationDataFieldValue(JdmoRowset rowData, DBSortField sortField)
		throws JdmoException
	{
		ProjectRulesSortField prsf = (ProjectRulesSortField)sortField;
		switch (prsf) {
			case NAME:
				return rowData.getString(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_NAME));
			case DESCRIPTION:
				return rowData.getString(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_DESCRIPTION));
			case ACTIVITY:
				return rowData.getString(activitySortFieldKey);
			case LENGTH:
				return rowData.getString(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_LENGTH));
			case ORGANIZATION:
				return rowData.getString(organizationSortFieldKey);
			case DAILYMAXPEREMPLOYEE:
				return rowData.getString(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_DAILYMAXPEREMPLOYEE));
			case WEEKLYMAXPEREMPLOYEE:
				return rowData.getString(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_WEEKLYMAXPEREMPLOYEE));
			case MAXSIMULTANEOUS:
				return rowData.getString(getFieldInfo().getFieldName(ProjectRulesFieldInfo.PROJECT_MAXSIMULTANEOUS));
			case START_TIME:
				return rowData.getInt("STARTTIME");
			default:
				return null;
		}
	}
