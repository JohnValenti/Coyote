package com.bluepumpkin.web.fs.schedule.group;

import com.witness.web.uif.pagemodel.*;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.base.*;
import com.witness.web.uif.keys.PageAction;
import com.bluepumpkin.web.bbm.people.PeopleRequestHandler;
import com.bluepumpkin.web.fs.schedule.ScheduleViewPM;
import com.bluepumpkin.web.fs.schedule.ScheduleViewUtil;
import com.bluepumpkin.web.fs.schedule.personal.PersonalGraphSchedulePM;

/**
 * Title:        GroupScheduleRH
 * Description:  Handle http requests for Group Schedule Views
 *               extends the base class to handle pagination
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * Created on October 7, 2002, 1:46 PM
 * @author       Pavel Bosin
 * @version 1.0
 */
public class GroupScheduleRH extends PeopleRequestHandler {

	/**
	 * Custom process Request
	 */
	public void processCustomRequest(RequestContext context) throws WebException {
		if (isAction(context, PageAction.FILTER_CHANGE)) {
			preProcessFilterChange(context, getPageModel(context));
		}
	}

	/**
	 * Return Page Model to perform generic action request on
	 */
	public WorkpanePageModel getPageModel(RequestContext context) {
		String viewType = getViewType(context);
		WorkpanePageModel model = getGroupPM(context, viewType);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		return model;
	}//getPageModel

	/** return WorkPane Page Model for group view */
	private WorkpanePageModel getGroupPM(RequestContext context, String viewType) {
		// Create Page Model from View Type
        ScheduleViewPM model = null;
        
		if (viewType!=null) {
			if (viewType.endsWith(GroupTextSchedulePM.FORM_ACTION))
				model = new GroupTextSchedulePM(context);
			else if (viewType.endsWith(GroupGraphSchedulePM.FORM_ACTION))
				model = new GroupGraphSchedulePM(context);
			else if (viewType.endsWith(GroupPrintSchedulePM.FORM_ACTION))
				model = new GroupPrintSchedulePM(context);
			else if (viewType.endsWith(GroupMultidaySchedulePM.FORM_ACTION))
				model = new GroupMultidaySchedulePM(context);
            else if (viewType.endsWith(GroupMultidayGraphSchedulePM.FORM_ACTION))
                model = new GroupMultidayGraphSchedulePM(context);         
		}

		// Default Page Model to Multi Day View
		if (model == null) model = new GroupMultidaySchedulePM(context);
		return model;
	}

	/**
	 * Get Request Type
	 */
	protected String getViewType(RequestContext context)
	{
		String viewType = ScheduleViewPM.getViewType(context);
		if (viewType==null) 
		{
			viewType = ScheduleViewUtil.getCachedFormAction(context);
			
			//String selectedIDs = (String)context.getAttribute(RequestContext.SESSION_SCOPE, "PEOPLE_SELECTION_ID_KEY"); 
			
			String selectedIDs = context.getRequest().getParameter("selectedID");
			
			if (viewType!=null)
			{ 
				if (viewType.endsWith(GroupGraphSchedulePM.FORM_ACTION))
				{
	                if ((selectedIDs != null) && (selectedIDs.length() > 0) && (selectedIDs.indexOf(",") == -1))				
					{
						//There is a single agent selected. We must show the multiday graph view.
						viewType = GroupMultidayGraphSchedulePM.FORM_ACTION;
	                    setNextScreenName(context, PeopleScheduleRH.SCREEN_MULTIDAY_GRAPH);
					}
	                else
	                {
	                	viewType = GroupGraphSchedulePM.FORM_ACTION;
	                	setNextScreenName(context, PeopleScheduleRH.SCREEN_GRAPH);
	                }
				}
				else if (viewType.endsWith(GroupMultidayGraphSchedulePM.FORM_ACTION))
				{
	                if ((selectedIDs != null) && (selectedIDs.length() > 0) && (selectedIDs.indexOf(",") != -1))				
					{
						//There are multiple agents selected. We must show the single day group graph view.
						viewType = GroupGraphSchedulePM.FORM_ACTION;
	                    setNextScreenName(context, PeopleScheduleRH.SCREEN_GRAPH);
					}
	                else
	                {
	                	viewType = GroupMultidayGraphSchedulePM.FORM_ACTION;
	                	setNextScreenName(context, PeopleScheduleRH.SCREEN_MULTIDAY_GRAPH);
	                }
				}
				
			}			
		}
		return viewType;
	}
}//class
