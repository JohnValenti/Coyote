  /**
   *  delete queue implements the following algorithms
   *  1) call checkLinkedObjects() to check if queue is linked to any other objects
   *  2) if queue is linked to campaign
   *        just return DeletionResult, since the deletion need user confirmation
   *     else, deleted it
   *
   */
  public DeletionResult deleteObject(ID idQueue,
							   Date dtDeleteOnDate, Localizer localizer, Object objCustomDefined)
							   throws DeleteException
  {
	  Jdmo dmo = new Jdmo(false);
	  try
	  {
		  DeletionResult dResult = checkLinkedObjects(idQueue, dmo);
		  switch (dResult.getResultCode())
		  {
			 case DeletionResult.DELETED:
				  // delete the object
				  DAOUtil.deleteObjectCascade(idQueue, "QUEUE", dmo);
				  break;

			 default:
				  // do nothing if prohibit delete or need substitute
				  break;
		  }

		  return dResult;
	  }
	  catch(Exception e)
	  {
		  throw new DeleteException(e);
	  }
	  finally
	  {
		  dmo.cleanUp();
	  }
  }


   /**
   *  force delete queue implements the following algorithms
   *  Cascade delete it.
   *
   */
  public void forceDeleteObject(ID idQueue) throws DeleteException
  {
	  Jdmo dmo = new Jdmo(false);
	  try
	  {
		   DAOUtil.deleteObjectCascade(idQueue, "QUEUE", dmo);
	  }
	  catch(Exception e)
	  {
		  throw new DeleteException(e);
	  }
	  finally
	  {
		  dmo.cleanUp();
	  }
  }
