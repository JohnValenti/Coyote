package com.verint.web.fs.dashboard.portlets.pulse;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.pulse.model.TraceChart;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;

/**
 * The Type of the statistic for which the
 * 
 * @author RPochana
 * 
 */
public enum StatisticType {
	CV(Trace.CV, FsWebBundleKey.PULSE_STATISTIC_CV), AHT(Trace.AHT,
			FsWebBundleKey.PULSE_STATISTIC_AHT), PCA(Trace.PCA,
			FsWebBundleKey.PULSE_STATISTIC_PCA), ASA(Trace.ASA,
			FsWebBundleKey.PULSE_STATISTIC_ASA), ABANDONMENT(Trace.ABANDONMENT,
			FsWebBundleKey.PULSE_STATISTIC_ABANDONMENT), BACKLOG(Trace.BACKLOG,
			FsWebBundleKey.PULSE_STATISTIC_BACKLOG), STAFFING(Trace.STAFFING,
			FsWebBundleKey.PULSE_STATISTIC_STAFFING), OCCUPANCY(
			Trace.OCCUPANCY, FsWebBundleKey.PULSE_STATISTIC_OCCUPANCY), FTE(
			Trace.FTE, FsWebBundleKey.PULSE_STATISTIC_FULL_TIME_EQUIVALENTS);

	private short m_id;
	private String m_name;

	private StatisticType(short id, String name) {
		this.m_id = id;
		this.m_name = name;
	}

	public int getId() {
		return m_id;
	}

	public String getName() {
		return m_name;
	}

	public static StatisticType getStatisticType(int typeId) {
		if (typeId == CV.m_id)
			return CV;
		if (typeId == AHT.m_id)
			return AHT;
		if (typeId == PCA.m_id)
			return PCA;
		if (typeId == ASA.m_id)
			return ASA;
		if (typeId == ABANDONMENT.m_id)
			return ABANDONMENT;
		if (typeId == BACKLOG.m_id)
			return BACKLOG;
		if (typeId == STAFFING.m_id)
			return STAFFING;
		if (typeId == OCCUPANCY.m_id)
			return OCCUPANCY;
		if (typeId == FTE.m_id)
			return FTE;

		return null;

	}

	public static Collection createDropDownValueList(Localizer localizer) {

		List<StringsPair> options = new ArrayList<StringsPair>();
		options.add(new StringsPair("" + CV.m_id, localizer.i18n(
				localizer.getBundle(FsWebBundleKey.BUNDLE_NAME), CV.m_name)));

		options.add(new StringsPair("" + AHT.m_id, localizer.i18n(
				localizer.getBundle(FsWebBundleKey.BUNDLE_NAME), AHT.m_name)));

		options.add(new StringsPair("" + PCA.m_id, localizer.i18n(
				localizer.getBundle(FsWebBundleKey.BUNDLE_NAME), PCA.m_name)));

		options.add(new StringsPair("" + ASA.m_id, localizer.i18n(
				localizer.getBundle(FsWebBundleKey.BUNDLE_NAME), ASA.m_name)));
		options.add(new StringsPair("" + ABANDONMENT.m_id, localizer.i18n(
				localizer.getBundle(FsWebBundleKey.BUNDLE_NAME),
				ABANDONMENT.m_name)));
		options.add(new StringsPair("" + BACKLOG.m_id, localizer.i18n(
				localizer.getBundle(FsWebBundleKey.BUNDLE_NAME),
				BACKLOG.m_name)));
		options.add(new StringsPair("" + STAFFING.m_id, localizer.i18n(
				localizer.getBundle(FsWebBundleKey.BUNDLE_NAME),
				STAFFING.m_name)));
		options.add(new StringsPair("" + OCCUPANCY.m_id, localizer.i18n(
				localizer.getBundle(FsWebBundleKey.BUNDLE_NAME),
				OCCUPANCY.m_name)));
		options.add(new StringsPair("" + FTE.m_id, localizer.i18n(
				localizer.getBundle(FsWebBundleKey.BUNDLE_NAME), FTE.m_name)));

		return options;
	}

	public boolean supportsForecast() {
		return TraceChart.isSupported(m_id, TraceChart.ABSOLUTE_CHART,
				TraceChart.FORECAST_LINE);
	}

	public boolean supportsRequired() {
		return TraceChart.isSupported(m_id, TraceChart.ABSOLUTE_CHART,
				TraceChart.REQUIRE_LINE);
	}

	public boolean supportsActual() {
		return TraceChart.isSupported(m_id, TraceChart.ABSOLUTE_CHART,
				TraceChart.ACTUAL_LINE);
	}
}
