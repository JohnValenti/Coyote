package com.verint.web.fs.dashboard.portlets.pulse;

import java.util.ArrayList;
import java.util.List;
import java.util.ResourceBundle;

import javax.servlet.http.HttpServletRequest;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.CompositePC;
import com.witness.web.uif.pagecomponent.form.FormInputPC;
import com.witness.web.uif.pagecomponent.picker.date.DateRangePickerPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;
import com.witness.web.uif.util.html.HtmlChoiceInput;
import com.witness.web.uif.util.js.JSUtil;

/**
 * A composite picker consisting of a DateRangePickerPC and a "last period"
 * picker. A radio button lets the user choose which picker they would like to
 * use.
 * 
 * This is similar to LastNPeriodPicker in Scorecards, However this uses DateRangePicker as opposed to DatePeriodPicker
 */
public class LastNPeriodPickerPC extends CompositePC {
	
	public static final int PERIOD_TYPE_DAILY = 0;
	public static final int PERIOD_TYPE_WEEKLY = 1;
	public static final int PERIOD_TYPE_MONTHLY = 2;
	public static final int PERIOD_TYPE_QUARTERLY = 3;
	public static final int PERIOD_TYPE_YEARLY = 4;

	private ResourceBundle m_core = null;

	private static final String PREFIX_RADIO_NAME = "RDPN_";
	private static final String LAST_PERIOD_PICKER_DD_INPUT_NAME = "lastPeriodPicker_input";
	private static final String DATE_RANGE_PICKER_NAME = "dateRangePicker";
	private static final String FN_LAST_NPERIODS_INPUT = "lastNPeriods";
	public static final String PICKER_TYPE_LAST_PERIOD = "lastPeriod";
	public static final String PICKER_TYPE_DATE_RANGE = "dateRange";

	private HtmlChoiceInput m_choiceInput = null;
	private DateRangePickerPC m_dateRangePickerPC;
	private FormInputPC m_lastNPeriodsInputPC;

	private int m_lastPeriodType = PERIOD_TYPE_DAILY;
	private String m_pickerType = PICKER_TYPE_LAST_PERIOD;

	public LastNPeriodPickerPC(RequestContext context) {
		super(context);

		m_core = m_localizer.getBundle(CoreWebBundleKey.BUNDLE_NAME);

		m_choiceInput = new HtmlChoiceInput();

		m_lastNPeriodsInputPC = new FormInputPC(context);
		m_lastNPeriodsInputPC.setInputFN(FN_LAST_NPERIODS_INPUT);
		m_lastNPeriodsInputPC.setInputValue("5");
		m_lastNPeriodsInputPC.setInputType(FormInputPC.TYPE_INTEGER);
		m_lastNPeriodsInputPC.setSize(2);
		m_lastNPeriodsInputPC.setOnChangeJS(JSUtil.SUBMIT_FORM);
		addChildComponent(m_lastNPeriodsInputPC);

		m_dateRangePickerPC = new DateRangePickerPC(m_context,
				DATE_RANGE_PICKER_NAME);
		m_dateRangePickerPC.setOnChangeJS(JSUtil.SUBMIT_FORM);
		m_dateRangePickerPC.getStartDatePickerPC().setIsTimeEnabled(false);
		m_dateRangePickerPC.getEndDatePickerPC().setIsTimeEnabled(false);
		addChildComponent(m_dateRangePickerPC);
	}

	public boolean extractParameters(HttpServletRequest request) {
		boolean success = super.extractParameters(request);

		String lastPeriodType = getParameter(request,
				LAST_PERIOD_PICKER_DD_INPUT_NAME);
		if (lastPeriodType != null) {
			m_lastPeriodType = Integer.parseInt(lastPeriodType);
		}
		String pickerType = getParameter(request, PREFIX_RADIO_NAME + getName());
		if (pickerType != null) {
			m_pickerType = pickerType;
		}
		return success;
	}

	public String getHtmlDisplay() {
		StringBuffer content = new StringBuffer(256);
		content.append("<table cellpadding=\"0\" cellspacing=\"0\" border=\"0\">");
		content.append(getHtmlDisplayLastPeriodPicker());
		content.append(getHtmlDisplayRangeDatePicker());
		content.append("</table>");
		return (content.length() > 0) ? content.toString() : null;
	}


	public DateRangePickerPC getDateRangePickerPC() {
		return m_dateRangePickerPC;
	}

	public int getLastPeriodType() {
		return m_lastPeriodType;
	}

	public void setLastPeriodType(int lastPeriodType) {
		m_lastPeriodType = lastPeriodType;
	}

	public String getPickerType() {
		return m_pickerType;
	}

	public void setPickerType(String pickerType) {
		m_pickerType = pickerType;
	}

	public int getLastNPeriods() {
		return Integer.parseInt(m_lastNPeriodsInputPC.getInputValue());
	}

	public void setLastNPeriods(int lastNPeriods) {
		m_lastNPeriodsInputPC.setInputValue(Integer.toString(lastNPeriods));
	}

	// Private methods
	private String getHtmlDisplayRangeDatePicker() {
		String radioName = getRadioName();
		String radioValue = PICKER_TYPE_DATE_RANGE;

		String strLine = HtmlUtil.NBSP + m_dateRangePickerPC.getHtmlDisplay();

		// Is the corresponding radio button selected?
		boolean isSelected = false;
		if (radioValue.equals(getPickerType())) {
			isSelected = true;
		}

		// Generate Content
		StringBuffer content = new StringBuffer(256);
		appendRadioRow(radioName, radioValue, isSelected, strLine, content);
		return content.toString();
	}

	private String getHtmlDisplayLastPeriodPicker() {
		String radioName = getRadioName();
		String radioValue = PICKER_TYPE_LAST_PERIOD;

		String label = i18n(m_common_bundle, UIFWebBundleKey.LAST);

		List options = getLastNPeriodOptions();

		String strPeriod = HtmlUtil.makeSelectInput(
				LAST_PERIOD_PICKER_DD_INPUT_NAME,
				Integer.toString(getLastPeriodType()), JSUtil.SUBMIT_FORM,
				options);

		String strLine = HtmlUtil.NBSP + label + HtmlUtil.NBSP
				+ m_lastNPeriodsInputPC.getHtmlDisplay() + HtmlUtil.NBSP
				+ strPeriod;

		// Is the corresponding radio button selected?
		boolean isSelected = false;
		if (radioValue.equals(getPickerType())) {
			isSelected = true;
		}

		// Generate Content
		StringBuffer content = new StringBuffer(256);
		appendRadioRow(radioName, radioValue, isSelected, strLine, content);
		return content.toString();
	}

	private List getLastNPeriodOptions() {
		List options = new ArrayList();
		options.add(new StringsPair(Integer
				.toString(PERIOD_TYPE_DAILY), i18n(m_core,
				CoreWebBundleKey.REPORTS_PARAMETERS_LAST_COMPLETED_DAYS)));
		options.add(new StringsPair(Integer
				.toString(PERIOD_TYPE_WEEKLY), i18n(m_core,
				CoreWebBundleKey.REPORTS_PARAMETERS_LAST_COMPLETED_WEEKS)));
		options.add(new StringsPair(Integer
				.toString(PERIOD_TYPE_MONTHLY), i18n(m_core,
				CoreWebBundleKey.REPORTS_PARAMETERS_LAST_COMPLETED_MONTHS)));
		return options;
	}

	private void appendRadioRow(String radioParam, String value,
			boolean isSelected, String content, StringBuffer sb) {
		String input = m_choiceInput.getRadioInput(radioParam, value, "",
				isSelected, false, "submitForm();");

		sb.append("<tr><td class=\"formInternalRowLight\" nowrap>");
		sb.append(input);
		sb.append("</td><td class=\"formInternalRowLight\" nowrap>");
		sb.append(content);
		sb.append("</td></tr>");
	}

	public String getRadioName() {
		return PREFIX_RADIO_NAME + getName();
	}

}
