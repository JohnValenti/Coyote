package com.bluepumpkin.ejb.rm.requests.common.model;


import com.bluepumpkin.ejb.bbm.vo.ChildInfoEntry;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;

/**
 * meta data information for Request object
 */
public class RequestFieldInfo extends FieldInfo {
    private static String REQUEST_TABLENAME = "REQUEST";
        
    // field id constants. 
    // the constants are coded as 'prefix_type_fieldname' for automatic initialization
    //  of FieldInfoEntry array using reflection.
    public static final int REQUEST_I_ID = 0;
    public static final int REQUEST_I_CREATORID = 1;
    public static final int REQUEST_S_REQUESTTYPE = 2;
    public static final int REQUEST_S_REQUESTSTATUS = 3;
    public static final int REQUEST_D_SUBMITTEDON = 4;
    public static final int REQUEST_D_LASTMODIFIEDAT = 5;
    public static final int REQUEST_I_EMPLOYEEID = 6;
    public static final int REQUEST_D_EXPIRATIONDATE = 7;
    
    // don't start with m_fieldPrefix in base class 
    private static final int REQUEST_NUM_OF_FIELDS = 8;
    
    // static initialization
    private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[REQUEST_NUM_OF_FIELDS];
    
    public static final int AUDIT_TRAIL_CHILD_TYPE = 0;
    public static final int CHILD_TYPE_EMPLOYEE = 1;    
    protected static final int REQUEST_NUM_OF_CHILDREN = 2;
    private static ChildInfoEntry[] m_childInfo = { 
        ChildInfoEntry.oneToManyChild(),
        ChildInfoEntry.oneToOneChild(REQUEST_I_EMPLOYEEID) 
    };

    static {
        assert m_childInfo.length == REQUEST_NUM_OF_CHILDREN:"m_childInfo.length != REQUEST_NUM_OF_CHILDREN: " + 
            m_childInfo.length + ',' + REQUEST_NUM_OF_CHILDREN; 
        RequestUtil.initFieldInfoEntries(m_arrFieldInfo, RequestFieldInfo.class, "REQUEST_", 
            REQUEST_NUM_OF_FIELDS);
    }

    public static final String COLNAME_REQUESTSTATUS = m_arrFieldInfo[REQUEST_S_REQUESTSTATUS].getFieldName();    
       
    // implement abstract methods in parent class
    public String getTableName() { 
        return REQUEST_TABLENAME; 
    }
    
    public int getIDFieldIndex() {
        return REQUEST_I_ID;
    }    

    public FieldInfoEntry [] getFieldInfo() { 
        return m_arrFieldInfo; 
    }
    
    // Return the child info array that contains necessary information about
    // the audit trail children entries
    public ChildInfoEntry[] getChildInfo()	{
        return m_childInfo;
    }    
}
