public class ProjectUtil {

	/**
	 * @param tc
	 * @param projStartTimes
	 * @return Utility method to create TimeOfDay objects grouped by weekday, which can later be used to 
	 *  	   intialize/create StartTimes object which would take care of creating continuous intervals of time if any are available.
	 *  	   or show them as discrete intervals if there are none.
	 *  NOTE: The TimeOfDay object implements the Comparable interface, StarTimes implement the SortedSet interface.  
	 */
	public static Map<WeekDay, Set<TimeOfDay>> getStartTimes(TimeContext tc,Collection<ProjectStart> projStartTimes,int iaOffSet){

			Map<WeekDay, Set<TimeOfDay>> weeklyMap = new HashMap<WeekDay, Set<TimeOfDay>>(WeekDay.WEEK_LENGTH);
			weeklyMap.put(WeekDay.SUNDAY, null);
			weeklyMap.put(WeekDay.MONDAY, null);
			weeklyMap.put(WeekDay.TUESDAY, null);
			weeklyMap.put(WeekDay.WEDNESDAY, null);
			weeklyMap.put(WeekDay.THURSDAY, null);
			weeklyMap.put(WeekDay.FRIDAY, null);
			weeklyMap.put(WeekDay.SATURDAY, null);
			
			if (projStartTimes != null) {
				for (ProjectStart lObjPst : projStartTimes) {
					
					HashSet<TimeOfDay> setObjTod = null;
					
					if(weeklyMap.containsKey(lObjPst.getWeekDay()) &&
							weeklyMap.get(lObjPst.getWeekDay())!= null ){
						
						setObjTod = (HashSet<TimeOfDay>) weeklyMap.get(lObjPst.getWeekDay());
				    	
				    	TimeOfDay tod = new TimeOfDay(lObjPst.getPossibleStartTime()+tc.getDayBoundary());
				    	tod.addMinutes(iaOffSet);
				    	setObjTod.add(tod);
				    	
				    } else {
				    	
				    	setObjTod = new HashSet<TimeOfDay>();
				    	TimeOfDay tod = new TimeOfDay(lObjPst.getPossibleStartTime()+tc.getDayBoundary());
				    	tod.addMinutes(iaOffSet);
				    	setObjTod.add(tod);
				    	weeklyMap.put(lObjPst.getWeekDay(),setObjTod);
				    }
				}
			}
			return weeklyMap;
		}
	

	
	
	public static Map<WeekDay, Set<ProjectStart>> getProjectStartByWeek(TimeContext tc,ProjectRules projRule){

		Map<WeekDay, Set<ProjectStart>> weeklyMap = new HashMap<WeekDay, Set<ProjectStart>>(WeekDay.WEEK_LENGTH);
		Collection<ProjectStart> projStartTimes  = projRule.getProjectStartTimes();
		weeklyMap.put(WeekDay.SUNDAY, null);
		weeklyMap.put(WeekDay.MONDAY, null);
		weeklyMap.put(WeekDay.TUESDAY, null);
		weeklyMap.put(WeekDay.WEDNESDAY, null);
		weeklyMap.put(WeekDay.THURSDAY, null);
		weeklyMap.put(WeekDay.FRIDAY, null);
		weeklyMap.put(WeekDay.SATURDAY, null);
		
		if (projStartTimes != null) {

			for (ProjectStart lObjPst : projStartTimes) {
				
				Set<ProjectStart> setObjProjStart = null;
				
				if (weeklyMap.containsKey(lObjPst.getWeekDay()) &&
						weeklyMap.get(lObjPst.getWeekDay())!= null ) {
					
					setObjProjStart = weeklyMap.get(lObjPst.getWeekDay());
			    	setObjProjStart.add(lObjPst);
			    	
			    } else {
			    	setObjProjStart = new HashSet<ProjectStart>();
			    	setObjProjStart.add(lObjPst);
			    	weeklyMap.put(lObjPst.getWeekDay(),setObjProjStart);
			    }
			}
		}
		return weeklyMap;
	}
	
	public static Set<ProjectStart> getShiftStarts(TimeContext timeContext, ID shiftID, Collection<TimeOfDay> timesOfDay) {
		HashSet<ProjectStart> retVal = new HashSet<ProjectStart>();
		for (TimeOfDay tod : timesOfDay) {
			ProjectStart ss = new ProjectStart();
			ss.setProjectID(shiftID);
			if (tod.getMinutesSinceMindnight() < timeContext.getDayBoundary()) {
				ss.setPossibleStartTime(1440 + tod.getMinutesSinceMindnight() - timeContext.getDayBoundary());
			} else {
				ss.setPossibleStartTime(tod.getMinutesSinceMindnight() - timeContext.getDayBoundary());
			}
			retVal.add(ss);
		}
		return retVal;
	}
  
	private static void removeStartTimes(TimeContext timeContext,Set<ProjectStart> projStart,
			Collection<TimeOfDay> startTimesToRemove) {
		
		if (projStart != null) {
			for (Iterator<ProjectStart> i = projStart.iterator(); i.hasNext(); ) {
				if (startTimesToRemove.contains(new TimeOfDay(i.next().getPossibleStartTime() + timeContext.getDayBoundary()))) {
					i.remove();
				}
			}
		}
	}

	private static Set<ProjectStart> addStartTimesToShift(TimeContext tx,Set<ProjectStart> setProjStart,
			Collection<TimeOfDay> startTimesToAdd,ProjectRules projRule,int daySeq) {
		
		HashSet<TimeOfDay> starts = null;
		if(setProjStart != null){
			starts = new HashSet<TimeOfDay>(ProjectUtil.getStartTimesOfDay(tx, setProjStart));
		}else{
			starts = new HashSet<TimeOfDay>();
		}
		starts.addAll(startTimesToAdd);
		
		return ProjectUtil.getProjectStarts(tx, projRule.getID(), starts,setProjStart,daySeq);

	}
	
	public static Set<ProjectStart> getProjectStarts(TimeContext tx, ID projID, Collection<TimeOfDay> timesOfDay,Set<ProjectStart> setProjStart,int daySeq) {
		
		if (setProjStart == null) {
			setProjStart = new HashSet<ProjectStart>();
		}
		
		for (TimeOfDay tod : timesOfDay) {
			ProjectStart ps = new ProjectStart();
			ps.setProjectID(projID);
			ps.setDaySeq(daySeq);
			
			if (tod.getMinutesSinceMindnight() < tx.getDayBoundary()) {
				ps.setPossibleStartTime(1440 + tod.getMinutesSinceMindnight() - tx.getDayBoundary());
			} else {
				ps.setPossibleStartTime(tod.getMinutesSinceMindnight() - tx.getDayBoundary());
			}
			setProjStart.add(ps);
		}
		return setProjStart;
	}
	
	public static Set<TimeOfDay> getStartTimesOfDay(TimeContext timeContext, Collection<ProjectStart> projStart) {
		HashSet<TimeOfDay> retVal = new HashSet<TimeOfDay>();
		for (ProjectStart ss : projStart) {
			retVal.add(new TimeOfDay(ss.getPossibleStartTime() + timeContext.getDayBoundary()));
		}
		return retVal;
	}
	
	/**
	 * @param modifiedStartTimes (this Parameter contains the Added and Removed StartTimes by Weekday
	 * @param mapOrgProjStartTimes (it contains the Original list of Start times for the Project by Weekday
	 * @param projRule (The Object to which we are going to set the Modified collection of ProjectStart Times
	 * @param tx The TimeContext allows us to get the DayBoundary and offset in the event of Time Zone for comparing the Times
	 */
	public static void prepareStartTimes(Map <WeekDay,ProjectRulesFormPM.ProjectRulesStartTimePC> modifiedStartTimes, 
								  Map <WeekDay,Set<ProjectStart>> mapOrgProjStartTimes,ProjectRules projRule,TimeContext tx ){
		
		ProjectRulesStartTimePC lObjStartTimePC = null;
		StartTimesPickerPC lObjStPicker = null;
		Set <ProjectStart> lSetProjSt = null;
		
		Collection<ProjectStart> finalProjectStarts = null;
		
		if (modifiedStartTimes != null) {
			for (WeekDay wd: WeekDay.values()) {
				lSetProjSt = mapOrgProjStartTimes.get(wd);
				lObjStartTimePC = modifiedStartTimes.get(wd);
				lObjStPicker = lObjStartTimePC.getStartPickerPC();
				
				if (lObjStartTimePC != null && lObjStartTimePC.isChecked()) {
					if (lSetProjSt != null 
							&& lObjStPicker.getRemovedStartTimes() != null &&
							lObjStPicker.getRemovedStartTimes().size() > 0) {
						
						ProjectUtil.removeStartTimes(tx,lSetProjSt,lObjStPicker.getRemovedStartTimes());
					}
					if( lObjStPicker != null && 
							(lObjStPicker.getAddedStartTimes()!= null && lObjStPicker.getAddedStartTimes().size() > 0)) {
	
						lSetProjSt = ProjectUtil.addStartTimesToShift(tx, lSetProjSt, lObjStPicker.getAddedStartTimes(), projRule,wd.getCount());
					}
					
					mapOrgProjStartTimes.put(wd, lSetProjSt);
				} else {
					mapOrgProjStartTimes.put(wd, new HashSet<ProjectStart>());
					continue;
				}
			}
			
			//Finally Prepare a collection of all the ProjectStartTimes to store in the db
			for (Iterator<WeekDay> i = mapOrgProjStartTimes.keySet().iterator(); i.hasNext();) {
				
				lSetProjSt = mapOrgProjStartTimes.get(i.next());
				
				if (finalProjectStarts == null) {
					finalProjectStarts = new HashSet<ProjectStart>();
				}
				if (lSetProjSt != null) {
					finalProjectStarts.addAll(lSetProjSt);
				}
			}
			
			//Add the So Prepared Collection to the ProjectRules object
			projRule.setProjectStartTimes(finalProjectStarts);
		}
		
	}
