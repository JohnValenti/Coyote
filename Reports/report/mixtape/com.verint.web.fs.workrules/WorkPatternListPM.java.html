	public WorkPatternListPM(RequestContext context) {
		super(context,JumpToPagePaginationPC.USED_IN_WORKPATTERN_LIST_PAGE);

		option_showTimeZoneToggle = true;
		
		removeChildComponent(m_list);
		m_list = new WorkPatternMasterDetailListPC(m_context, shiftPatterns, optimizationTargetTimeZone, timeZoneEffectiveDate, 
				isByOrg, orgMap, campaignMap, activityMap, empTypeMap,WorkPatternMasterDetailListPC.USED_IN_LIST);
		//Need to get the sort order which is get from WorkPatternMasterDetailListPC(...) for loadData(). 
		isAscendingSort = m_list.getHeader().isAscending();
		this.addChildComponent(m_list);
		this.setIsMultiRowSelectable(true);
		
		addRequiredJavaScriptFile(WorkRulesJSFileID.WORK_RULES_UTIL);
		
		toggleAll = new ToggleVisiblePC(m_context, "toggleAll");
		this.addChildComponent(toggleAll);
	}

	private ArrayList<ShiftPattern> shiftPatterns = new ArrayList<ShiftPattern>();
	@Override
	protected void loadData() throws Exception {
		super.loadData();
		int pageIndex = getPagePaginationIndex();
		int itemIndex = getEditIndex();
		int pageSize = getPageIDSize();	
		String pageAction = getPageAction();
		String strItemIndex = (String) m_context.getAttribute(RequestContext.SESSION_SCOPE, WORKPATTERNLIST_ITEMINDEX);
		if(pageAction!=null&& (pageAction.equals(PageAction.CANCEL)||pageAction.equals(PageAction.SAVE) 
				||pageAction.equals(PageAction.TABLE_SORT)||pageAction.equals(PageAction.CREATE_NEW))){
		
			if (!StringUtil.isEmpty(strItemIndex)){
					itemIndex = Integer.parseInt(strItemIndex);
					pageIndex = (itemIndex/pageSize)+ (itemIndex%pageSize > 0 ? 1: 0); 
			}
		} 
		if (isByOrg && selectedOrg != null) {
			if (isPaginationNavPressed()) {
				shiftPatterns = new ArrayList<ShiftPattern>(WorkRulesModelHandler.getShiftPatterns(m_context, getPageIDs()));
			} else {
				PaginationPair<ShiftPattern> shiftPatternData = WorkRulesModelHandler.getShiftPatternsByOrgHierarchy(m_context, 
						getSelectedIDObject(), isAscendingSort, pageSize,
						pageIndex, m_list.getSortComparators(), null,getTextToFind(itemToFind));
				shiftPatterns = new ArrayList<ShiftPattern>(shiftPatternData.getValues());
				initPaginationIDs(shiftPatternData.getIDs());					
			}
		} else if (!isByOrg && selectedSchedulingPeriod != null){
			if (isPaginationNavPressed()) {
				shiftPatterns = new ArrayList<ShiftPattern>(WorkRulesModelHandler.getShiftPatterns(m_context, getPageIDs()));
			} else {
				PaginationPair<ShiftPattern> shiftPatternData =
						WorkRulesModelHandler.getShiftPatternsBySchedulingPeriod(m_context, 
						selectedSchedulingPeriod, isAscendingSort, pageSize,
						pageIndex, m_list.getSortComparators(), null,getTextToFind(itemToFind));
				shiftPatterns = new ArrayList<ShiftPattern>(shiftPatternData.getValues());
				initPaginationIDs(shiftPatternData.getIDs());							
			}
		}
		
		Collections.sort(shiftPatterns, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));
		
		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		Collection<ID> campaignIDs = new HashSet<ID>();
		Collection<ID> activityIDs = new HashSet<ID>();
		Collection<ID> empTypeIDs = new HashSet<ID>();
		
		for (ShiftPattern sp : shiftPatterns) {
			if (sp.getOrganizationID() != null) orgIDs.add(sp.getOrganizationID());
			if (sp.getCampaignID() != null) campaignIDs.add(sp.getCampaignID());
			if (sp.getEmployeeTypeID() != null) empTypeIDs.add(sp.getEmployeeTypeID());
			for (Shift shift : sp.getShiftPatternShiftAttributes().keySet()) {
				if (shift.getOrganizationID() != null) orgIDs.add(shift.getOrganizationID());
				if (shift.getCampaignID() != null) campaignIDs.add(shift.getCampaignID());
				if (shift.getActivityID() != null) activityIDs.add(shift.getActivityID());
				for (ShiftEvent se : shift.getShiftEvents()) {
					if (se.getOrganizationID() != null) orgIDs.add(se.getOrganizationID());
					if (se.getCampaignID() != null) campaignIDs.add(se.getCampaignID());
					if (se.getActivityID() != null) activityIDs.add(se.getActivityID());
				}
			}
			for (VTOEvent vtoe : sp.getVTOEventStartTimes().keySet()) {
				if (vtoe.getOrganizationID() != null) orgIDs.add(vtoe.getOrganizationID());
				if (vtoe.getCampaignID() != null) campaignIDs.add(vtoe.getCampaignID());
				if (vtoe.getActivityID() != null) activityIDs.add(vtoe.getActivityID());
			}
			for (ShiftOTExtension sote : sp.getOTExtensionStartTimes().keySet()) {
				if (sote.getOrganizationID() != null) orgIDs.add(sote.getOrganizationID());
				if (sote.getCampaignID() != null) campaignIDs.add(sote.getCampaignID());
				if (sote.getActivityID() != null) activityIDs.add(sote.getActivityID());
			}
		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!isByOrg) campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);
		empTypeMap = EmployeeTypeModelHandler.getEmployeeTypesByIDs(m_context, empTypeIDs);
	}

	/**
		Limit the page size to 30 to prevent a Javascript message saying that the script is taking too long.
		Each work pattern can have shifts, vto events, and ot extensions. If a work pattern has all three,
		there will be 3 tables per row with a javascript representation, which can add up to a lot of work.
		See bug QC-74372 for more information.
	 */
	protected int getMaximumPaginationSize() {
		int configuredMaxPageSize = retrieveConfiguredMaxPageSize();
		if (configuredMaxPageSize < 0 || configuredMaxPageSize > 30) {
			return 30;
		} else {
			return configuredMaxPageSize;
		}
	}

	protected void initToolbar() {	
		
		if(selectedCampaign != null || selectedOrg != null) {
			FindToolbarPageComponent findToolBar = new FindToolbarPageComponent(m_context, getToolbar(),
					FIND_ITEM_FN, WorkRulesRH.FIND_ACTION,
					getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_FIND_WORKPATTERNS));	
			getToolbar().addToolbarComponent(findToolBar);
			getToolbar().addDivider(); 
			StringBuilder sbImportArgs = new StringBuilder();
			sbImportArgs.append(WorkRulesImportPopupPM.IS_BY_ORG_ARGUMENT).append("=").append(isByOrg);
			if(!isByOrg) {
				sbImportArgs.append("&");
				sbImportArgs.append(WorkRulesImportPopupPM.CAMPAIGN_ID_ARGUMENT).append("=").append(selectedCampaign.getID());
				sbImportArgs.append("&");
				sbImportArgs.append(WorkRulesImportPopupPM.SCHEDULING_PERIOD_ID_ARGUMENT).append("=").append(selectedSchedulingPeriod.getID());
			} else {
				sbImportArgs.append("&");
				sbImportArgs.append(WorkRulesImportPopupPM.ORGANIZATION_ID_ARGUMENT).append("=").append(selectedOrg.getID());
			}
			sbImportArgs.append("&");
			sbImportArgs.append(WorkRulesImportPopupPM.OPTIMIZATION_TARGET_TIMEZONE).append("=").append(optimizationTargetTimeZone.getID());
			sbImportArgs.append("&");
			sbImportArgs.append(WorkRulesImportPopupPM.TIMEZONE_EFFECTIVE_DATE).append("=").append(timeZoneEffectiveDate);
					
			addPopupArgs(WorkPatternListPM.WORK_PATTERN_IMPORT_POPUP, WorkPatternImportPopupPM.FORM_ACTION, sbImportArgs.toString(), null, "480,520,ctr,ctr", true, 1);
			ToolbarTextButton importButton = m_toolbar.createPopupWindowButton(WorkPatternListPM.WORK_PATTERN_IMPORT_POPUP, 
					i18n(m_bundle, FsWebBundleKey.TOOLBAR_IMPORT), m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES));
			m_toolbar.addToolbarElement(importButton);
			
			StringBuilder sb = new StringBuilder();
			sb.append(WorkPatternExportPopupPM.IS_BY_ORG_ARGUMENT).append("=").append(isByOrg);
			if(!isByOrg) {
				sb.append("&");
				sb.append(WorkPatternExportPopupPM.CAMPAIGN_ID_ARGUMENT).append("=").append(selectedCampaign.getID());
				sb.append("&");
				sb.append(WorkPatternExportPopupPM.SCHEDULING_PERIOD_ID_ARGUMENT).append("=").append(selectedSchedulingPeriod.getID());
			} else {
				sb.append("&");
				sb.append(WorkPatternExportPopupPM.ORGANIZATION_ID_ARGUMENT).append("=").append(selectedOrg.getID());
			}
			sb.append("&");
			sb.append(WorkPatternExportPopupPM.OPTIMIZATION_TARGET_TIMEZONE).append("=").append(optimizationTargetTimeZone.getID());
			sb.append("&");
			sb.append(WorkPatternExportPopupPM.TIMEZONE_EFFECTIVE_DATE).append("=").append(timeZoneEffectiveDate);
			
			addPopupArgs(WorkPatternListPM.WORK_PATTERN_EXPORT_POPUP, WorkPatternExportPopupPM.VIEW_ACTION, sb.toString(), null, "480,255,ctr,ctr", true, 1);
			ToolbarTextButton exportButton = m_toolbar.createPopupWindowButton(WorkPatternListPM.WORK_PATTERN_EXPORT_POPUP, 
					i18n(m_bundle, FsWebBundleKey.TOOLBAR_EXPORT), true);
			m_toolbar.addToolbarElement(exportButton);
			
			boolean hasEditPrivilege = hasEditPrivilege();
            importButton.setEnabled(hasEditPrivilege);
            exportButton.setEnabled(hasEditPrivilege);
			
			m_toolbar.addDivider();
		}
		super.initToolbar();
	}
