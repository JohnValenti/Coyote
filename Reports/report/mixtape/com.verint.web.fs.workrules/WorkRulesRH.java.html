package com.verint.web.fs.workrules;

import static com.witness.web.uif.util.TimeZoneOverrideUtil.TIME_ZONE_OVERRIDE_TOGGLE_ACTION;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.ejb.bbm.base.BbmDuplicateNameException;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.WebException;
import com.witness.web.uif.base.WorkpaneRequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.pagemodel.WorkpanePageModel;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.TimeZoneOverrideUtil;

public abstract class WorkRulesRH<T extends WorkRulesListPM, U extends WorkRulesFormPM> extends WorkpaneRequestHandler {
	
	protected static final String LIST_PAGE = "listPage";
	protected static final String FORM_PAGE = "formPage";
	public static final String WORK_RULES_COPY_PAGE_ACTION = "WORK_RULES_COPY";
	public static final String FIND_ACTION = "FIND_ACTION";
	protected boolean isByOrg = true;

	@Override
	public void processRequest(RequestContext context) throws Exception {
		try {
			if(isAction(context, WORK_RULES_COPY_PAGE_ACTION)) {
				T pm = getListPageModel(context);
				processCopy(context, pm);
			}
			if(isAction(context, PageAction.ADD)) {
				T pm = getListPageModel(context);
				processAdd(context, pm);
			}
			if(isAction(context, PageAction.EDIT)) {
				T pm = getListPageModel(context);
				processEdit(context, pm);
			}
			if(isAction(context, PageAction.DELETE)) {
				T pm = getListPageModel(context);
				processDelete(context, pm);
			}
			if(isAction(context, PageAction.IMPORT)) {
				T pm = getListPageModel(context);
				processImport(context, pm);
			}
			if(isAction(context, PageAction.EXPORT)) {
				T pm = getListPageModel(context);
				processExport(context, pm);
			}
			if(isAction(context, PageAction.DOWNLOAD)) {
				T pm = getListPageModel(context);
				processDownload(context, pm);
			}
			if(isAction(context, PageAction.PREVIOUS)) {
				T pm = getListPageModel(context);
				processPageLeft(context, pm);
			}
			if(isAction(context, PageAction.NEXT)) {
				T pm = getListPageModel(context);
				processPageRight(context, pm);
			}
			if(isAction(context, PageAction.SELECT_PAGE)) {
				T pm = getListPageModel(context);
				processPageSelect(context, pm);
			}
			if (isAction(context, TIME_ZONE_OVERRIDE_TOGGLE_ACTION)) {
				TimeZoneOverrideUtil.toggleTimeZoneOverride(context);
			}
			if (isAction(context, PageAction.SAVE)) {
				try {
					U pm = getFormPageModel(context);
					processSave(context, pm);
				} catch (Exception e) {
					setNextPage(context, FORM_PAGE);
					throw e;
				}
			}
			if (isAction(context, PageAction.CREATE_NEW)) {
				try {
					U pm = getFormPageModel(context);
					processSave(context, pm);
				} catch (Exception e) {
					setNextPage(context, FORM_PAGE);
					throw e;
				}
			}
			if (isAction(context, FIND_ACTION)) {
				T pm = getListPageModel(context);
					processFind(context,pm);
			}
			if (isAction(context, PageAction.TABLE_SORT)) {
				T pm = getListPageModel(context);
				processTableSort(context,pm);
			} 
		} catch (BbmDuplicateNameException ex) {
			PageModel pm = context.getPageModel();
			if (pm != null) pm.addPageMessage(Message.ERROR_TYPE, ex.getLocalizedMessage());
		}
		catch (Exception ex) {
			PageModel pm = context.getPageModel();
			if (pm != null) handleUnableToUpdate(pm, ex);
		}

	}
	
	abstract protected T getListPageModel(RequestContext context);
	abstract protected U getFormPageModel(RequestContext context);
	abstract protected void processImport(RequestContext context, T pageModel) throws BPException;
	abstract protected void processExport(RequestContext context, T pageModel) throws BPException;
	abstract protected void processDownload(RequestContext context, T pageModel) throws BPException;
	abstract protected void processCopy(RequestContext context, T pageModel) throws BPException;
	abstract protected void processDelete(RequestContext context, T pageModel) throws BPException;
	abstract protected void processSave(RequestContext context, U pageModel) throws BPException;

	protected void processAdd(RequestContext context, T pageModel) throws BPException {
		setNextPage(context, FORM_PAGE);
	}

	protected void processEdit(RequestContext context, T pageModel)
			throws BPException {
		setNextPage(context, FORM_PAGE);
	}
	
	protected void processPageLeft(RequestContext context, T pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexBackward();	
	}
	
	protected void processPageRight(RequestContext context, T pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexForward();
	}
	
	protected void processPageSelect(RequestContext context, T pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.setPagePaginationAndEditIndices(
				pageModel.getPaginationControl().getSelectedPaginationPageIndex());
	}
	
	protected void processFind(RequestContext context, T pageModel) throws BPException {
		setNextPage(context, LIST_PAGE);
	}
	protected void processTableSort(RequestContext context, T pageModel) throws BPException {
		setNextPage(context, LIST_PAGE);
	}
	/**
	 * WorkRulesRH uses getListPageModel and getFormPageModel.
	 */
	@Override
	public WorkpanePageModel getPageModel(RequestContext context) {
		// TODO Auto-generated method stub
		return null;
	}

	@Override
	public void processCustomRequest(RequestContext context)
			throws WebException {
		// TODO Auto-generated method stub
		
	}
}
