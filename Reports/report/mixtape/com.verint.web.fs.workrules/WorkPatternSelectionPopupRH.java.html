package com.verint.web.fs.workrules;

import com.witness.web.uif.system.RequestContext;

public class WorkPatternSelectionPopupRH extends WorkRulesSelectionPopupRH<WorkPatternSelectionPopupPM> {
	@Override
	protected WorkPatternSelectionPopupPM getPageModel(RequestContext context) {
		WorkPatternSelectionPopupPM pm = new WorkPatternSelectionPopupPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
	}

	@Override
	protected void processTableSort(RequestContext context, WorkPatternSelectionPopupPM pageModel) {
		String sortColumn = context.getRequest().getParameter("sortColumn");
		String sortOrder = context.getRequest().getParameter("sortOrder");
		context.setUserProperty(WorkPatternSelectionPopupPM.WORKPATTERN_SELECTIONPOPUP_SORTCOLUMN, sortColumn, true);
		context.setUserProperty(WorkPatternSelectionPopupPM.WORKPATTERN_SELECTIONPOPUP_SORTORDER, sortOrder, true);
	}
}
