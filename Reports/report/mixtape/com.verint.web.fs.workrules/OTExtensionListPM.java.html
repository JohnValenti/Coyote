package com.verint.web.fs.workrules;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.PaginationPair;
import com.bluepumpkin.common.datatypes.ScopeID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectIDListComparator;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.ejb.core.security.Scope;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftOTExtension;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.OTExtensionListPC;

public class OTExtensionListPM extends WorkRulesListPM{
	private static final long serialVersionUID = 1L;

	public OTExtensionListPM(RequestContext context)
	{
		super(context);
		removeChildComponent(m_list);
		m_list = new OTExtensionListPC(
				m_context,
				"otExtensionTable",
				optimizationTargetTimeZone, 
				selectedSchedulingPeriod == null ? new Date() : selectedSchedulingPeriod.getStartTime(), 
						otExtensions,
						orgMap,
						campaignMap, 
						activityMap);

		//This needs to be done here so that we can get the sort columns/order
		m_list.extractParameters(context.getRequest());
		this.addChildComponent(m_list);
		// TODO multi-edit for OT
		this.setIsMultiRowSelectable(true);
	}

	private ArrayList<ShiftOTExtension> otExtensions = new ArrayList<ShiftOTExtension>();
	private Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
	private Map<ID, Campaign> campaignMap = new HashMap<ID, Campaign>();
	private Map<ID, Activity> activityMap = new HashMap<ID, Activity>();

	protected void loadData() throws Exception
	{
		super.loadData();

		if(!hasSelectedID()) return;

		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		Collection<ID> campaignIDs = new HashSet<ID>();
		Collection<ID> activityIDs = new HashSet<ID>();
		// Retrieve shiftsOTExtension and add the selected org or campaign into the collection of objects to retrieve
		if(isByOrg)
		{
			if(selectedOrg != null)
			{
				if ( m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID, new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE))) 
				{
					if (isPaginationNavPressed()) {
						otExtensions = new ArrayList<ShiftOTExtension>(WorkRulesModelHandler.getShiftOTExtensionsByIDs(m_context, getPageIDs()));
					} 
					else
					{
						PaginationPair<ShiftOTExtension> shiftOTData = WorkRulesModelHandler.getShiftOTExtensionsByOrgHierarchy(m_context, getSelectedIDObject(),
								m_list.getHeader().isAscending(), getPageIDSize(), getPagePaginationIndex(),
								m_list.getSortComparators(), null); 
						otExtensions = new ArrayList<ShiftOTExtension>(shiftOTData.getValues());
						initPaginationIDs(shiftOTData.getIDs()); 
					}
				}
				else {
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE, FsWebBundleKey.BUNDLE_NAME);
				}
			}
		}
		else
		{
			if(selectedSchedulingPeriod != null)
			{
				if (m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID)) {
					if (isPaginationNavPressed()) {
						otExtensions = new ArrayList<ShiftOTExtension>(WorkRulesModelHandler.getShiftOTExtensionsByIDs(m_context, getPageIDs()));
					}
					else
					{
						PaginationPair<ShiftOTExtension> shiftOTData = WorkRulesModelHandler.getShiftOTExtensionsBySchedulingPeriod(m_context, selectedSchedulingPeriod,
								m_list.getHeader().isAscending(), getPageIDSize(), getPagePaginationIndex(),
								m_list.getSortComparators(), null); 
						otExtensions = new ArrayList<ShiftOTExtension>(shiftOTData.getValues());
						initPaginationIDs(shiftOTData.getIDs());
					}
				}
				else
				{
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE, FsWebBundleKey.BUNDLE_NAME);
				}

			}
		}

		Collections.sort(otExtensions, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));

		// Retrieve the needed org, campaign, and activity IDs from the shifts and associated shift events
		for (ShiftOTExtension shitOT : otExtensions) {
			if (shitOT.getOrganizationID() != null) orgIDs.add(shitOT.getOrganizationID());
			if (shitOT.getCampaignID() != null) campaignIDs.add(shitOT.getCampaignID());
			if (shitOT.getActivityID() != null) activityIDs.add(shitOT.getActivityID());
			for (ShiftEvent se : shitOT.getShiftEvents()) {
				if (se.getOrganizationID() != null) orgIDs.add(se.getOrganizationID());
				if (se.getCampaignID() != null) campaignIDs.add(se.getCampaignID());
				if (se.getActivityID() != null) activityIDs.add(se.getActivityID());
			}
		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!isByOrg) campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);

	}

	protected void initScreenContent()
	{
		// Init selectable items first to initialize all of the toggles
		initSelectableItems();
		initHeader();
	}

	protected void initHeader()
	{

	}

	protected void initSelectableItems()
	{
		this.removeChildComponent(m_list);
		m_list = new OTExtensionListPC(
				m_context,
				"otExtensionTable",
				optimizationTargetTimeZone, 
				selectedSchedulingPeriod == null ? new Date() : selectedSchedulingPeriod.getStartTime(), 
						otExtensions,
						orgMap,
						campaignMap, 
						activityMap){
			@Override
			protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, ShiftOTExtension shiftOTExtension)
			{
				super.setRowAttributes(mcnd, shiftOTExtension);
				OTExtensionListPM.this.setRowAttributes(mcnd, shiftOTExtension.getOrganizationID());
			}
			@Override
			protected void setDetailRowAttributes(DefaultMultiColumnNodeData mcnd,
					DefaultMultiColumnNodeData mcndDetail, ShiftOTExtension shiftOTExtension) {
				super.setDetailRowAttributes(mcnd, mcndDetail, shiftOTExtension);
				OTExtensionListPM.this.setDetailRowAttributes(mcnd, mcndDetail, shiftOTExtension.getOrganizationID());
			}
		};
		m_list.extractParameters(m_context.getRequest());
		this.addChildComponent(m_list);

	}

	protected void initContentTitle()
	{
		super.initContentTitle();
		
		ID selectedID = getSelectedIDObject();
		if (selectedID != null) {
			if (isByOrg) {
				if (selectedOrg != null) m_contentTitlePC.setItemName(str(selectedOrg.getName()));
				
			} else {
				//Campaign
				Campaign c = campaignMap.get(selectedID);
				if (c != null) m_contentTitlePC.setItemName(str(c.getName()));
			}
		}
		String pageTitle = str(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.WORKRULES_SHIFTS_OTEXTENSION_TITLE));
		m_contentTitlePC.setPageTitle(pageTitle);

	}

	private String str(Object v) {
		return v == null ? "" : v.toString();
	}

	public static OTExtensionListPM getInstance(RequestContext context)
	{
		return (OTExtensionListPM)PageModel.getInstance(context,OTExtensionListPM.class);
	}
}
