	public static final String SHIFT_EXPORT_POPUP = "SHIFT_EXPORT_POPUP";	// Screen name of export popup
	public static final String SHIFTLIST_SORTCOLUMN = ShiftListPC.SHIFT_LIST_SORTCOLUMN;
	public static final String SHIFTLIST_SORTORDER = ShiftListPC.SHIFT_LIST_SORTORDER;
	public static final String  SHIFTLIST_ITEMINDEX = JumpToPagePaginationPC.SHIFTLIST_ITEMINDEX;
	private boolean isAscendingSort = true;
	public ShiftListPM(RequestContext context) {
		super(context, JumpToPagePaginationPC.USED_IN_SHIFT_LIST_PAGE);
		removeChildComponent(m_list);
		m_list = new ShiftListPC(
				m_context,
				"shiftTable",
				shifts, 
				optimizationTargetTimeZone, 
				selectedSchedulingPeriod == null ? new Date() : selectedSchedulingPeriod.getStartTime(), 
				orgMap,
				campaignMap, 
				activityMap,true);
		//Need to get the sort order after initialize the component for loadData(). 
		isAscendingSort = m_list.getHeader().isAscending();
		this.addChildComponent(m_list);
		this.setIsMultiRowSelectable(true);
	}

	private ArrayList<Shift> shifts = new ArrayList<Shift>();
	private Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
	private Map<ID, Campaign> campaignMap = new HashMap<ID, Campaign>();
	private Map<ID, Activity> activityMap = new HashMap<ID, Activity>();
	
	@Override
	protected void loadData() throws Exception {
		super.loadData();
		
		if (!hasSelectedID()) return;
		int pageIndex = getPagePaginationIndex();
		int pageSize = getPageIDSize();	
		int itemIndex = getEditIndex();
		String pageAction = getPageAction();
		String strItemIndex = (String) m_context.getAttribute(RequestContext.SESSION_SCOPE, SHIFTLIST_ITEMINDEX);
		if(pageAction!=null&& (pageAction.equals(PageAction.CANCEL)||pageAction.equals(PageAction.SAVE) 
				||pageAction.equals(PageAction.TABLE_SORT)||pageAction.equals(PageAction.CREATE_NEW))){
		
			if (!StringUtil.isEmpty(strItemIndex)){
					itemIndex = Integer.parseInt(strItemIndex);
					pageIndex = (itemIndex/pageSize)+ (itemIndex%pageSize > 0 ? 1: 0); 
			}
		} 
			
		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		Collection<ID> campaignIDs = new HashSet<ID>();
		Collection<ID> activityIDs = new HashSet<ID>();

		// Retrieve shifts and add the selected org or campaign into the collection of objects to retrieve 
		if (isByOrg) {
			if (selectedOrg != null) {
				if ( m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID, new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE))) {
					if (isPaginationNavPressed()) {
						shifts = new ArrayList<Shift>(WorkRulesModelHandler.getShiftsByIDs(m_context, getPageIDs()));
					} else {
						PaginationPair<Shift> shiftData = WorkRulesModelHandler.getShiftListByOrg(m_context, getSelectedIDObject(),
								isAscendingSort, pageSize, pageIndex,
								m_list.getSortComparators(), null, getTextToFind(itemToFind));
						shifts = new ArrayList<Shift>(shiftData.getValues());
						initPaginationIDs(shiftData.getIDs());
					}
				} else {
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE, FsWebBundleKey.BUNDLE_NAME);
				}
			}
		}
		else {
			if (selectedSchedulingPeriod != null) {
				if (m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID)) {
					if (isPaginationNavPressed()) {
						shifts = new ArrayList<Shift>(WorkRulesModelHandler.getShiftsByIDs(m_context, getPageIDs()));
					} else {
						PaginationPair<Shift> shiftData = WorkRulesModelHandler.getShiftListBySchedulingPeriod(m_context, 
								selectedSchedulingPeriod, isAscendingSort, pageSize,
								pageIndex, m_list.getSortComparators(), null, getTextToFind(itemToFind));
						shifts = new ArrayList<Shift>(shiftData.getValues());
						initPaginationIDs(shiftData.getIDs());
					}
				} else {
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE, FsWebBundleKey.BUNDLE_NAME);
				}
			}
		}
		
		Collections.sort(shifts, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));

		// Retrieve the needed org, campaign, and activity IDs from the shifts and associated shift events
		for (Shift shift : shifts) {
			if (shift.getOrganizationID() != null) orgIDs.add(shift.getOrganizationID());
			if (shift.getCampaignID() != null) campaignIDs.add(shift.getCampaignID());
			if (shift.getActivityID() != null) activityIDs.add(shift.getActivityID());
			for (ShiftEvent se : shift.getShiftEvents()) {
				if (se.getOrganizationID() != null) orgIDs.add(se.getOrganizationID());
				if (se.getCampaignID() != null) campaignIDs.add(se.getCampaignID());
				if (se.getActivityID() != null) activityIDs.add(se.getActivityID());
			}
		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!isByOrg) campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);
	}

	@Override
	protected void initScreenContent() {
		// Init selectable items first to initialize all of the toggles
		initSelectableItems();
		initHeader();
	}

	@Override

	}

	@Override
	protected void initSelectableItems() {
		this.removeChildComponent(m_list);
		m_list = new ShiftListPC(
				m_context,
				"shiftTable",
				shifts, 
				optimizationTargetTimeZone, 
				selectedSchedulingPeriod == null ? new Date() : selectedSchedulingPeriod.getStartTime(), 
				orgMap,
				campaignMap, 
				activityMap,true){
			@Override
			protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, Shift shift) {
				super.setRowAttributes(mcnd, shift);
				ShiftListPM.this.setRowAttributes(mcnd, shift.getOrganizationID());
			}
			@Override
			protected void setDetailRowAttributes(DefaultMultiColumnNodeData mcnd,
					DefaultMultiColumnNodeData mcndDetail, Shift shift) {
				super.setDetailRowAttributes(mcnd, mcndDetail, shift);
				ShiftListPM.this.setDetailRowAttributes(mcnd, mcndDetail, shift.getOrganizationID());
			}
		};
		//m_list.extractParameters(m_context.getRequest());
		this.addChildComponent(m_list);
	}
	
	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		
		ID selectedID = getSelectedIDObject();
		if (selectedID != null) {
			if (isByOrg) {
				if (selectedOrg != null) m_contentTitlePC.setItemName(str(selectedOrg.getName()));
				
			} else {
				//Campaign
				Campaign c = campaignMap.get(selectedID);
				if (c != null) m_contentTitlePC.setItemName(str(c.getName()));
			}
		}
		String pageTitle = str(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.WORKRULES_SHIFTS_TITLE));
		m_contentTitlePC.setPageTitle(pageTitle);
	}

	private String str(Object v) {
		return v == null ? "" : v.toString();
	}
	
	public static ShiftListPM getInstance(RequestContext context) {
		return (ShiftListPM) PageModel.getInstance(context, ShiftListPM.class);
	}

		// This should be refactored to a WorkRulesPopup, there is too much duplication here, because these arguments are basically pertinent to all of those
		// pop ups.
		if((selectedCampaign != null &&selectedSchedulingPeriod!=null ) || selectedOrg != null) {
			FindToolbarPageComponent findToolBar = new FindToolbarPageComponent(m_context, getToolbar(),
					FIND_ITEM_FN, WorkRulesRH.FIND_ACTION,
					getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_FIND_SHIFTS_PROMPT));

			getToolbar().addDivider();

			if(!isByOrg) {
				sbImportArgs.append("&");

				sbImportArgs.append("&");
				sbImportArgs.append(ShiftExportPopupPM.SCHEDULING_PERIOD_ID_ARGUMENT).append("=").append(selectedSchedulingPeriod.getID());
			} else {
				sbImportArgs.append("&");
				sbImportArgs.append(ShiftExportPopupPM.ORGANIZATION_ID_ARGUMENT).append("=").append(selectedOrg.getID());
			}
			sbImportArgs.append("&");
			sbImportArgs.append(ShiftExportPopupPM.OPTIMIZATION_TARGET_TIMEZONE).append("=").append(optimizationTargetTimeZone.getID());
			sbImportArgs.append("&");
			sbImportArgs.append(ShiftExportPopupPM.TIMEZONE_EFFECTIVE_DATE).append("=").append(timeZoneEffectiveDate);
			
			
			addPopupArgs(ShiftListPM.SHIFT_IMPORT_POPUP, ShiftImportPopupPM.FORM_ACTION, sbImportArgs.toString(), null, "480,520,ctr,ctr", true, 1);
			ToolbarTextButton importButton = m_toolbar.createPopupWindowButton(ShiftListPM.SHIFT_IMPORT_POPUP, 
					i18n(m_bundle, FsWebBundleKey.TOOLBAR_IMPORT), m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES));
			m_toolbar.addToolbarElement(importButton);
			
			StringBuilder sb = new StringBuilder();
			sb.append(ShiftExportPopupPM.IS_BY_ORG_ARGUMENT).append("=").append(isByOrg);
			if(!isByOrg) {
				sb.append("&");
				sb.append(ShiftExportPopupPM.CAMPAIGN_ID_ARGUMENT).append("=").append(selectedCampaign.getID());
				sb.append("&");
				sb.append(ShiftExportPopupPM.SCHEDULING_PERIOD_ID_ARGUMENT).append("=").append(selectedSchedulingPeriod.getID());
			} else {
				sb.append("&");
				sb.append(ShiftExportPopupPM.ORGANIZATION_ID_ARGUMENT).append("=").append(selectedOrg.getID());
			}
			sb.append("&");
			sb.append(ShiftExportPopupPM.OPTIMIZATION_TARGET_TIMEZONE).append("=").append(optimizationTargetTimeZone.getID());
			sb.append("&");
			sb.append(ShiftExportPopupPM.TIMEZONE_EFFECTIVE_DATE).append("=").append(timeZoneEffectiveDate);
			
			addPopupArgs(ShiftListPM.SHIFT_EXPORT_POPUP, ShiftExportPopupPM.VIEW_ACTION, sb.toString(), null, "480,255,ctr,ctr", true, 1);
			ToolbarTextButton exportButton = m_toolbar.createPopupWindowButton(ShiftListPM.SHIFT_EXPORT_POPUP, 
					i18n(m_bundle, FsWebBundleKey.TOOLBAR_EXPORT), true);
			m_toolbar.addToolbarElement(exportButton);
			
			boolean hasEditPrivilege = hasEditPrivilege();
            importButton.setEnabled(hasEditPrivilege);
            exportButton.setEnabled(hasEditPrivilege);
			
			m_toolbar.addDivider();
			
			super.initToolbar();
		}
