	public Collection<ShiftOTExtension> getOTExtensions() throws BbmFinderException {
		Collection<ShiftOTExtension> ret = new ArrayList<ShiftOTExtension>(otextensions);
		if (ret.isEmpty()) ret.add(new ShiftOTExtension());

		for (ShiftOTExtension otextension : ret) {
			// Construct a shift from the form field
			if (!isMultiEdit() || activitySelectorPC.isEdited()) otextension.setActivityID(activitySelectorPC.getSelections().get(0).getID());
			if (isEditingNameAndDesc()) {
				otextension.setName(nameInputPC.getInputValue() != null ? nameInputPC.getInputValue().trim() : null);
				otextension.setDescription(descInputPC.getInputValue());
				if (isByOrg) otextension.setOrganizationID(selectedOrg.getID());
				else if (!isByOrg) otextension.setCampaignID(selectedCampaign.getID());
			}
			if (!isMultiEdit() || durationPC.isEdited()) otextension.setDuration(durationPC.getDuration().getDurationInMinutes());

			if (!isMultiEdit() || minGapPC.isEdited()) otextension.setMinGap(minGapPC.getDuration().getDurationInMinutes());
			if (!isMultiEdit() || maxGapPC.isEdited()) otextension.setMaxGap(maxGapPC.getDuration().getDurationInMinutes());
			if (shiftEventListPC.getRemovedRowIDs().length() > 0) {
				removeShiftEventsFromOTExtension(otextension, StringUtil.parseIDStringToCollection(
						shiftEventListPC.getRemovedRowIDs()));
			}			
			if (shiftEventListPC.getAddedRowIDs().length() > 0) {			
				Collection<ShiftEvent> addedShiftEvents = new ArrayList<ShiftEvent>();
				addedShiftEvents = WorkRulesModelHandler.getShiftEvents(m_context, StringUtil.parseIDStringToCollection(
						shiftEventListPC.getAddedRowIDs()));
				addShiftEventsToOTExtension(otextension, addedShiftEvents);
			}
		}
		return ret;
	}

	/**
	 * Returns an array list of all shift events belonging to the selected shifts.
	 * Also sets the multi edit common flag on each shift event so the shiftEventListPC
	 * can render the shift appropriately
	 * @return
	 */
	private Collection<ShiftEvent> getShiftEventsFromSelectedOTExtensions() {
		if (otextensions.size() == 0) return new ArrayList<ShiftEvent>();

		HashMap<ID, ShiftEvent> shiftEventMap = new HashMap<ID, ShiftEvent>();

		//initialize the map
		ShiftOTExtension tempOTExtension = otextensions.get(0);
		for (ShiftEvent se : tempOTExtension.getShiftEvents()) {
			se.setMultiEditCommon(true);
			shiftEventMap.put(se.getID(), se);
		}

		for (ShiftOTExtension otextension : otextensions) {
			for (ShiftEvent se : shiftEventMap.values()) {
				//if a shiftevent in the map does not belong to the shift in this iteration,
				// we want to set multi edit common to false as it is no longer common between
				// all shifts in the set
				if (se.isMultiEditCommon() && otextension.getShiftEvents().contains(se) == false) {
					se.setMultiEditCommon(false);
				}
			}
			for (ShiftEvent se : otextension.getShiftEvents()) {
				if (shiftEventMap.containsKey(se.getID()) == false) {
					se.setMultiEditCommon(false);
					shiftEventMap.put(se.getID(), se);
				}
			}
		}
		return new ArrayList<ShiftEvent>(shiftEventMap.values());
	}

	private void removeShiftEventsFromOTExtension(ShiftOTExtension otextension, Collection<ID> shiftEventIDsToRemove) {
		ArrayList<ShiftEvent> shiftEventsToRemove = new ArrayList<ShiftEvent>();
		for (ShiftEvent se : otextension.getShiftEvents()) {
			for (ID idToRemove : shiftEventIDsToRemove) {
				if (se.getID().equals(idToRemove)) {
					shiftEventsToRemove.add(se);
					break;
				}
			}
		}
		otextension.removeShiftEvents(shiftEventsToRemove);
	}

	//It's possible that shifts events that are being added might already be associated to a OTExtension.
	//  Only add the shift events that are not already associated
	private void addShiftEventsToOTExtension(ShiftOTExtension otextension, Collection<ShiftEvent> shiftEventsToAdd) {
		ArrayList<ShiftEvent> finalShiftEventsToAdd = new ArrayList<ShiftEvent>();
		for (ShiftEvent se : shiftEventsToAdd) {
			if (otextension.getShiftEvents().contains(se) == false) {
				finalShiftEventsToAdd.add(se);
			}
		}
		otextension.addShiftEvents(finalShiftEventsToAdd);
	}

	private int getMaxDuration(Collection<Integer> durations) {
		int maxDuration = 0;
		for (int dur : durations) {
			if (dur > maxDuration) maxDuration = dur;
		}
		return maxDuration;	
	}

	@Override
	public boolean validate() {
		boolean retVal = super.validate();

		HashMap<ID, Activity> activities = new HashMap<ID, Activity>(activityMap);

		Collection<ShiftOTExtension> editedOTExtensions = null;
		HashMap<ID, String> allOTExtensionNames = null;
		try {
			if (isByOrg) allOTExtensionNames = WorkRulesModelHandler.getOTExtensionNamesByOrg(m_context, selectedOrg.getID());
			else allOTExtensionNames = WorkRulesModelHandler.getOTExtensionNamesBySchedulingPeriod(m_context, selectedSchedulingPeriod);
			editedOTExtensions = getOTExtensions();
		} catch (BbmFinderException e1) {
			handleUnableToLoadDataError(log, e1);
			return false;
		}

		for (ShiftOTExtension otextension : editedOTExtensions) {
			Collection<ShiftEvent> shiftEvents = otextension.getShiftEvents();
			HashSet<ID> seActivityIDs = new HashSet<ID>();
			for (ShiftEvent se : shiftEvents) seActivityIDs.add(se.getActivityID());
			seActivityIDs.removeAll(activities.keySet());
			try {
				if (! seActivityIDs.isEmpty()) activities.putAll(WorkRulesModelHandler.getActivities(m_context, seActivityIDs));
			} catch (BbmFinderException e) {
				handleUnableToLoadDataError(log, e);
				return false;
			}

			int eventDurations = 0;
			int spacedEvents = 0;
			for (ShiftEvent se : shiftEvents) {
				int count = se.isFlexible() ? se.getMinCount() : 1;
				eventDurations += count * se.getDuration().getDurationInMinutes();

				// See if spacing applies to the shift event
				Activity a = activities.get(se.getActivityID());
				if (! a.isUsedInShift()) spacedEvents += count;
			}

			int otExtensionLength = otextension.getDuration();
			if (otExtensionLength < (eventDurations + ((spacedEvents - 1) * otextension.getMinGap()))) {
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_NOT_ALL_SHIFT_EVENTS_FIT_IN_OTEXTENSION));
				retVal = false;
			}
			
			if(otExtensionLength <= 0)
			{
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_OT_EXTENSION_DURATION_CANNOT_BE_ZERO));
				retVal = false;
			}
			
			if(	otextension.getMinGap() > (MAX_SHIFT_MINSPACING_IN_HOURS * 60) ||
				otextension.getMinGap()  > otExtensionLength)
			{
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.MIN_GAP_IS_OUT_OF_RANGE));
				retVal = false;
			}
				
		
			
			// Shift event start and end time vs OT extension length
			for (ShiftEvent se : shiftEvents) {

				double startWindow = se.getStart();
				double endWindow = 0;

				if(se.isAnyTime())
				{
					startWindow = 0;
					endWindow = otExtensionLength;
				}
				if( (startWindow + se.getDuration().getDurationInMinutes()) > otExtensionLength){
					addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_LINKING_SHIFT_EVENTS_EXCEEDS_OTEXTENSION_LENGTH));
					retVal = false;
				}

			}


			// Name must be unique
			if (isEditingNameAndDesc()) {
				String name = otextension.getName();
				for (ID existingShiftID : allOTExtensionNames.keySet()) {
					if (existingShiftID.equals(otextension.getID())) continue;
					if (name.equals(allOTExtensionNames.get(existingShiftID))) {
						addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_NAME_MUST_BE_UNIQUE));
						retVal = false;
					}
				}
			}

			// Max gap must be > min gap 
			if (otextension.getMinGap() > otextension.getMaxGap()) {
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.MAX_GAP_MUST_BE_GREATER_THAN_MIN_GAP));
				retVal = false;
			}


			// Cannot have a shift event with the same activity as the OT Extension
			ID activityID = otextension.getActivityID();
			for (ShiftEvent se : otextension.getShiftEvents()) {
				if (activityID.equals(se.getActivityID())) {
					addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_SHIFT_EVENT_ACTIVITY_MUST_BE_DIFFERENT_FROM_SHIFT));
					retVal = false;
				}
			}
			
		}
		return retVal;
	}

	class ShiftMaxSpacingPC extends CompositePC {

		DurationSpinnerPC durationPC;
		CheckBoxPC unlimitedCheckBoxPC;

		private boolean isMultiEdit = false;
		private boolean isEdited = false;

		ShiftMaxSpacingPC(RequestContext context, String name) {
			super(context);
			setName(name);

			addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
			setRenderJavaScriptIfDisabled(true);
			durationPC = new DurationSpinnerPC(context, getDurationName());
			addChildComponent(durationPC);
			unlimitedCheckBoxPC = new CheckBoxPC(m_context, getUnlimitedName(), false,
					getDurationName() + ".setEnabled(false);", getDurationName() + ".setEnabled(true);");
			addChildComponent(unlimitedCheckBoxPC);
		}

		public void setMultiEdit(boolean isMultiEdit) {
			this.isMultiEdit = isMultiEdit;
		}

		@Override
		public String getJavaScriptInitCode() {
			String handlerName = getName() + ".OnChange";
			return super.getJavaScriptInitCode()  + 
			"\n" + handlerName + " = function() { jQuery('input[name=" + getEditedName() + "]').val('true'); };" +
			"\n" + getUnlimitedName() + ".change(function() {" + handlerName + "();});" +
			"\n" + getDurationName() + ".change(function() {" + handlerName + "();});";
		}

		@Override
		public String getHtmlDisplay() {
			return BPMessageFormat.format("{0}{1}{2}&nbsp;&nbsp;{3}<input type='hidden' name=\"{4}\" value=\"{5}\">",
					new Object[] {
					(isMultiEdit ? "<span class='multiEditIndicator' style='padding-right: 2px;'>*</span>" : ""),
					unlimitedCheckBoxPC.getHtmlDisplay(),
					i(FsWebBundleKey.FS_UNLIMITED), 
					durationPC.getHtmlDisplay(),
					getEditedName(),
					isEdited});
		}

		private String getDurationName() {
			return getName() + "__duration";
		}
		private String getUnlimitedName() {
			return getName() + "__unlimited";
		}
		private String getEditedName() { 
			return getName() + "__edited";
		}

		@Override
		public boolean extractParameters(HttpServletRequest request) {
			if (!isAutoExtractParamEnabled()) return true;
			String editedStr = request.getParameter(getEditedName());
			if (editedStr != null) isEdited = new Boolean(editedStr);
			return super.extractParameters(request);
		}

		public boolean isEdited() {
			return isEdited;
		}

		public Duration getDuration() {
			if (unlimitedCheckBoxPC.isChecked()) {
				return Duration.fromMinutes(0);
			}
			return durationPC.getDuration();
		}

		public void setDuration(Duration duration) {
			if (duration.getDurationInMinutes() == 0) {
				unlimitedCheckBoxPC.setIsChecked(true);
				durationPC.setEnabled(false);
			} else {
				unlimitedCheckBoxPC.setIsChecked(false);
			}
			durationPC.setValue(duration);
		}

		public DurationSpinnerPC getDurationSpinnerPC() {
			return durationPC;
		}

		public CheckBoxPC getCheckBoxPC() {
			return unlimitedCheckBoxPC;
		}
	}
