public class AssignmentRulesListPM extends WorkRulesListPM {
	protected static Category log = Log.initCategory(AssignmentRulesListPM.class.getName());
	public static final String ASSIGNMENTRULE_LIST_SORTCOLUMN = AssignmentRuleListPC.ASSIGNMENTRULE_LIST_SORTCOLUMN;
	public static final String ASSIGNMENTRULE_LIST_SORTORDER = AssignmentRuleListPC.ASSIGNMENTRULE_LIST_SORTORDER;
	private List<ComplexWorkRule> complexWorkRules = Collections.emptyList();
	private HashMap<ID, Organization> orgsByID = new HashMap<>();
	private AssignmentRuleListPC m_assignmentRuleListPC;
	private Map<ID, String> assignmentRuleDescriptionMap = new HashMap<>();
	private HashMap<ID, String> cwrEmpNameMap;
	boolean isAscendingSort = true;

	public AssignmentRulesListPM(RequestContext context) {
		super(context);

		this.setJSMediator(FsJavaScriptFileID.WORK_PANE_LIST_ALT_DELETE_MEDIATOR);

		option_showTimeZoneToggle = false;
		removeChildComponent(m_list);
		m_list = new AssignmentRuleListPC(m_context, new LinkedList<>(), new HashMap<>(),
				new HashMap<>(), true);
		//Need to get the sort order after initialize the component for loadData().
		isAscendingSort = m_list.getHeader().isAscending();
		this.addChildComponent(m_list);
		setIsMultiRowSelectable(false);
	}

	@Override
	protected void loadData() throws Exception {
		super.loadData();
		int pageIndex = getPagePaginationIndex();
		int pageSize = getPageIDSize();


		ID selectedOrgID = getSelectedIDObject();
		if (selectedOrgID != null) {
			// Get Complex Work Rules
			if (m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID, new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE))) {
				WorkRuleManager wrm = WfmManagerFactory.getWorkRuleManager(m_context.isInWhatIfMode());
				if (isPaginationNavPressed()) {
					complexWorkRules = new ArrayList<>(wrm.getComplexWorkRulesByIDs(getPageIDs()));
					ArrayList<ID> pageIDs = (ArrayList<ID>) getPageIDs();
					Collections.sort(complexWorkRules, new ValueObjectIDListComparator(pageIDs));

				} else {
					PaginationPair<ComplexWorkRule> data =
							wrm.getComplexWorkRulesForOrgAndAncestors(
									selectedOrgID,
									pageSize,
									pageIndex, ((AssignmentRuleListPC) m_list).getSortFields(), isAscendingSort);
					complexWorkRules = new ArrayList<>(data.getValues());
					initPaginationIDs(data.getIDs());
					Collections.sort(complexWorkRules, new ValueObjectIDListComparator(new ArrayList<>(getPageIDs())));
				}

			} else {
				addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE, FsWebBundleKey.BUNDLE_NAME);
			}

			loadEmployeeAssignmentNames();

			// Get Owner Org Names
			HashSet<ID> ownerOrgIDs = new HashSet<ID>();
			for (ComplexWorkRule cwr : complexWorkRules) {
				ID ownerOrgID = cwr.getOrganizationID();
				if (ownerOrgID != null) {
					ownerOrgIDs.add(cwr.getOrganizationID());
				}
			}

			orgsByID = ValueObjectUtil.getIDObjectMap(OrganizationModelHandler.getOrganizationsByIDs(m_context, ownerOrgIDs));

			assignmentRuleDescriptionMap = ARParamPCUtil.buildAssignmentRuleDescriptions(m_context, orgsByID, complexWorkRules);
		}
	}


	/**
	 * Loads the names of the assigned employees, formats the lists as strings, and stores them in the cwrEmpNameMap field.
	 *
	 * @throws BbmFinderException
	 * @throws RemoteException
	 * @throws BbmException
	 */
	private void loadEmployeeAssignmentNames() throws BbmFinderException, RemoteException, BbmException {
		// First get the object representation of the assignment for the rotations
		Map<ID, ? extends Collection<WorkResourceComplexWorkRule>> rotationAssignmentMap =
				WorkRulesModelHandler.getComplexWorkRuleAssignmentsByCWRIDs(m_context, ValueObjectUtil.getIDFromObjects(complexWorkRules));

		// Now Collect the employee IDs so we can get the names
		Set<ID> workResourceIDs = new HashSet<>();
		for (Collection<WorkResourceComplexWorkRule> assignments : rotationAssignmentMap.values()) {
			for (WorkResourceComplexWorkRule assignment : assignments) {
				workResourceIDs.add(assignment.getWorkResourceID());
			}
		}

		// Get the Name abstraction for all the employees
		HashMap<ID, EmployeeName> assignmentNames = WorkResourceModelHandler.getWorkResourceManager(m_context).getWorkResourceNamesByIDs(workResourceIDs);

		cwrEmpNameMap = new HashMap<>();
		for (Entry<ID, ? extends Collection<WorkResourceComplexWorkRule>> assignmentEntry : rotationAssignmentMap.entrySet()) {
			ID rotationID = assignmentEntry.getKey();
			ArrayList<String> empNames = new ArrayList<>(assignmentEntry.getValue().size());
			for (WorkResourceComplexWorkRule assignment : assignmentEntry.getValue()) {
				try {
					// QC 109704 - handling NPE for empNames.add(null)
					empNames.add(assignmentNames.get(assignment.getWorkResourceID()).getDisplayName(m_localizer));
				} catch (NullPointerException npe) {
					addPageMessage(Message.WARNING_TYPE, FsWebBundleKey.ASSIGNMENT_RULE_WARN_RULES_LIST_EMPLOYEE_INVALID_DATA, FsWebBundleKey.BUNDLE_NAME);
					log.l7dWarn(FsWebBundleKey.ASSIGNMENT_RULE_WARN_EMPLOYEE_INVALID_DATA + assignment.getWorkResourceID(), npe); // log the employee ID for these occurrences
					empNames.add("Employee ID: " + assignment.getWorkResourceID()); // add something identifiable to empNames for these occurrences
				}
			}
			cwrEmpNameMap.put(rotationID, StringUtil.join(empNames, ", "));
		}
	}


	@Override
	protected boolean isSortSupported() {
		return true;
	}

	@Override
	protected void initHeader() {
	}

	@Override
	protected void initSelectableItems() {
		removeChildComponent(m_list);
		m_list = m_assignmentRuleListPC = new AssignmentRuleListPC(m_context, complexWorkRules, orgsByID, assignmentRuleDescriptionMap, true) {
			@Override
			protected void setRowAttributes(DefaultMultiColumnNodeData rowData,
					ComplexWorkRule cwr) {
				super.setRowAttributes(rowData, cwr);
				AssignmentRulesListPM.this.setRowAttributes(rowData, cwr.getOrganizationID());
				if (cwrEmpNameMap.containsKey(cwr.getID())) {
					String msg = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.DELETE_CONFIRMATION_WITH_ASSIGNMENTS, cwrEmpNameMap.get(cwr.getID()));
					rowData.setNodeAttribute("deleteConfirmation", msg);
				}
			}
		};
		m_assignmentRuleListPC.extractParameters(m_context.getRequest());
		addChildComponent(m_list);
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		m_contentTitlePC.setPageTitle(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ASSIGNMENT_RULES));
	}

	public static AssignmentRulesListPM getInstance(RequestContext context) {
		return (AssignmentRulesListPM) PageModel.getInstance(context, AssignmentRulesListPM.class);
	}

}
