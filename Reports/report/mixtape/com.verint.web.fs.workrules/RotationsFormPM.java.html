@SuppressWarnings("serial")
public class RotationsFormPM extends WorkRulesFormPM {
	protected static Category log = Log.initCategory(RotationsFormPM.class.getName());
	static {

		//log.setResourceBundle(new FsWebLogBundle());
		ResourceBundle resourceBundle = ResourceBundle.getBundle(FsWebLogBundleKey.BUNDLE_NAME);
		log.setResourceBundle(resourceBundle);
	}

	private static final String FN_NAME = "_rotationName";
	private static final String FN_DATE = "_rotationDate";
	private static final String FN_INITIAL_ACTION = "_rotationInitAction";

	private FormTwoColumnPC pc_form;
	private FormInputPC pc_name;
	private DatePickerPC pc_date;
	private WorkPatternSelectionPC pc_workPatterns;

	private String data_initialAction = null;
	private Rotation data_rotation = null;
	private Map<ID, Organization> data_orgMap = new HashMap<ID, Organization>();
	private Map<ID, EmployeeType> data_empTypeMap = new HashMap<ID, EmployeeType>();

	public RotationsFormPM(RequestContext context) {
		super(context);
		optimizationTargetTimeZone = null; // Null out time zone to hide it
		createComponents();
	}

	@Override
	public boolean extractParameters(HttpServletRequest request) {
		boolean retVal = super.extractParameters(request);
		String initialActionStr = request.getParameter(FN_INITIAL_ACTION);
		if (!StringUtil.isEmpty(initialActionStr))
			data_initialAction = initialActionStr;
		else
			data_initialAction = getPageAction();
		addGenericHTML(HtmlUtil.makeHiddenInput(FN_INITIAL_ACTION, data_initialAction));
		return retVal;
	}

	@Override
	protected void loadData() throws Exception {
		super.loadData();

		// Load Rotation
		if (getPerformActionOnIDSize() > 0) {
			data_rotation = WorkRulesModelHandler
					.getRotationsByIDs(m_context, Arrays.asList(getPerformActionOnIDs()))
					.iterator()
					.next();
		} else {
			data_rotation = new Rotation();
			data_rotation.setOrganizationID(getSelectedIDObject());
		}

		applyExtractedParamsToRotation();

		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		orgIDs.add(data_rotation.getOrganizationID());
		orgIDs.add(getSelectedIDObject());
		Collection<ID> empTypeIDs = new HashSet<ID>();

		for (ShiftPattern sp : data_rotation.getShiftPatterns()) {
			if (sp.getOrganizationID() != null)
				orgIDs.add(sp.getOrganizationID());
			if (sp.getEmployeeTypeID() != null)
				empTypeIDs.add(sp.getEmployeeTypeID());
		}

		data_orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		data_empTypeMap = EmployeeTypeModelHandler.getEmployeeTypesByIDs(m_context, empTypeIDs);
	}

	private void applyExtractedParamsToRotation() throws BbmFinderException {
		if (isPostBack()) {
			// Apply extracted Name to Rotation
			if (!StringUtil.isEmpty(pc_name.getInputValue()))
				data_rotation.setName(pc_name.getInputValue().trim());

			// Apply extracted date
			data_rotation.setAlignmentDate(pc_date.getLocalDate());

			// Apply extractedWorkPatterns to rotation
			ArrayList<ShiftPattern> shiftPatterns = new ArrayList<ShiftPattern>();
			List<ID> ids = pc_workPatterns.getExtractedWorkPatternIDs();
			if (ids != null && !ids.isEmpty()) {
				// We have to create a map because the getShiftPatterns method will not return duplicates, even if
				// duplicate ids are passed in.
				Map<ID, ShiftPattern> patternMap = ValueObjectUtil.getIDObjectMap(WorkRulesModelHandler
						.getShiftPatterns(m_context, ids));
				for (ID id : ids) {
					shiftPatterns.add(patternMap.get(id));
				}
				data_rotation.setShiftPatterns(shiftPatterns);
			}
		}
	}

	private boolean isPostBack() {
		return PageAction.SAVE.equals(getPageAction()) || PageAction.CREATE_NEW.equals(getPageAction());
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		m_contentTitlePC.setPageTitle(i18n(m_bundle, FsWebBundleKey.FS_ROTATIONS));
		if (data_rotation != null)
			m_contentTitlePC.setItemName(data_rotation.getName());
	}

	/**
	 * Set the data on the Components and also configuration that relies on data being loaded.
	 */
	@Override
	protected void initForm() {
		if (!isPostBack()) {
			pc_name.setInputValue(data_rotation.getName());
			pc_date.setTimeZone(selectedOrg.getTimeZone());
			pc_date.setDate(data_rotation.getAlignmentDate());
		}

		// WorkPattern Selection cannot extract itself completely.
		pc_workPatterns.setData(getSelectedIDObject(), data_rotation.getShiftPatterns(), data_empTypeMap, data_orgMap);

		// Date
		OrgDateUtil.restrictDatePickerToOrgStartWeek(
				pc_date,
				data_orgMap.get(getSelectedIDObject()),
				data_rotation.getAlignmentDate());
	}

	/**
	 * Initialize the components without setting any data on them. They will be ready for param extraction.
	 */
	private void createComponents() {
		pc_form = new FormTwoColumnPC(m_context);
		pc_form.setIsBorderEnabled(false);
		setForm(pc_form);

		// Name input
		pc_name = new FormInputPC(m_context);
		pc_name.reset(HtmlUtil.escapeAttributeValue(FN_NAME), "", FormInputPC.TYPE_STRING, true);
		pc_name.setAutoExtractParamEnabled(true);
		pc_name.setMaxLength(50);
		pc_name.setInputFieldDesc(UIFWebBundleKey.NAME, UIFWebBundleKey.BUNDLE_NAME);
		pc_name.setNotAllowedChars("'"); 
		this.addChildComponent(pc_name);
		pc_form.addRow(i18n(m_common_bundle, UIFWebBundleKey.NAME), pc_name);

		// Organization
		pc_form.addRow(i18n(m_bundle, FsWebBundleKey.FS_ORGANIZATION), new Object() {
			@Override
			public String toString() {
				if (getSelectedIDObject() != null) {
					Organization o = data_orgMap.get(getSelectedIDObject());
					if (o != null)
						return o.getName();
				}
				return "";
			}
		});

		/*
		 * // Duration HtmlSpan durationSpan = new HtmlSpan(ID_DURATION); int duration = 0; duration =
		 * rotation.getNumWeeks(); durationSpan.setInnerHtml(duration); pc_form.addRow(i18n(m_bundle,
		 * FsWebBundleKey.FS_DURATION), m_localizer.i18n(m_bundle, FsWebBundleKey.N_WEEKS, duration));
		 */

		pc_date = new DatePickerPC(m_context, FN_DATE);
		this.addChildComponent(pc_date);
		pc_form.addRow(i18n(m_bundle, FsWebBundleKey.FS_ALIGNMENT_DATE), pc_date);

		pc_workPatterns = new WorkPatternSelectionPC(m_context, "workPatternList");
		this.addChildComponent(pc_workPatterns);
		pc_form.addRow(i18n(m_bundle, FsWebBundleKey.WORK_PATTERNS), pc_workPatterns);
	}

	public static RotationsFormPM getInstance(RequestContext context) {
		return (RotationsFormPM) PageModel.getInstance(context, RotationsFormPM.class);
	}

	/**
	 * Returns Rotations represented by this form.
	 * 
	 * @return
	 */
	public Rotation getRotation() {
		return data_rotation;
	}

	@Override
	public boolean validate() {
		try {
			boolean retVal = super.validate();

			Rotation rot = getRotation();
			WorkRuleManager wrm = WfmManagerFactory.getWorkRuleManager(m_context.isInWhatIfMode());
			Collection<RotationValidationError> errors = new HashSet<WorkRuleManager.RotationValidationError>(
					wrm.validate(rot));

			if (errors.remove(RotationValidationError.NameMustBeUnique)) {
				retVal = false;
				addPageMessage(
						Message.ERROR_TYPE,
						FsWebBundleKey.OBJECT_NAME_MUST_BE_UNIQUE,
						FsWebBundleKey.BUNDLE_NAME);
			}

			if (errors.remove(RotationValidationError.TooManyWorkPatterns)) {
				retVal = false;
				addPageMessage(
						Message.ERROR_TYPE,
						FsWebBundleKey.ERROR_ROTATION_LENGTH_TOO_LONG,
						FsWebBundleKey.BUNDLE_NAME);
			}

			if (errors.remove(RotationValidationError.NotEnoughWorkPatterns)) {
				retVal = false;
				addPageMessage(
						Message.ERROR_TYPE,
						FsWebBundleKey.ERROR_ROTATION_LENGTH_TOO_SHORT,
						FsWebBundleKey.BUNDLE_NAME);
			}
			
			if (errors.remove(RotationValidationError.AlignmentDateNotAlignedWithWeekStart)) {
				retVal = false;
				addPageMessage(
						Message.ERROR_TYPE,
						FsWebBundleKey.ERROR_ROTATION_ALIGNMENT_DATE_NOT_ALIGNED_WITH_WEEK_START,
						FsWebBundleKey.BUNDLE_NAME);
			}
			

			// Any other errors should have been caught by the client-side validation, so just log them and give a
			// generic message.
			if (errors.size() > 0) {
				retVal = false;
				log.l7dError(FsWebLogBundleKey.ERROR_VALIDATING, new Object[] { StringUtil.join(errors, ", ") });
				addPageMessage(
						Message.ERROR_TYPE,
						CommonWebBundleKey.UNABLE_TO_UPDATE_INFO,
						CommonWebBundleKey.BUNDLE_NAME);
			}

			return retVal;
		} catch (Exception e) {
			handleUnableToLoadDataError(log, e);
			return false;
		}
	}

	@Override
	public String getJavaScriptInitCode() {
		return super.getJavaScriptInitCode() + "\nresetJSArray[resetJSArray.length] = \"" + getMediatorRef()
				+ ".refreshPageWithAction('" + data_initialAction + "', true);\"\n";
	}

}
