package com.verint.web.fs.workrules;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.TimeZone;

import javax.servlet.http.HttpServletRequest;

import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftFieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPatternPeriod;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPatternShiftAttributes;
import com.bluepumpkin.web.core.pagecomponent.CheckBoxPC;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.widgets.WorkGroupDropDownSelectionPC;
import com.witness.web.uif.jsp.HtmlDiv;
import com.witness.web.uif.jsp.HtmlTag;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.MultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;

public class WorkDaysInlineEditDetailListPC extends MultiColListPC {
	private static final long serialVersionUID = 1L;
	
	private ShiftPatternShiftAttributes shiftAttributes;
	private Shift shift;
	private ArrayList<String> ddNames = new ArrayList<String>();
	private boolean isEdited = false;
	private ShiftPatternPeriod shiftPatternPeriod;
	private int dayOfWeekStart = 1;

	public WorkDaysInlineEditDetailListPC(RequestContext context, Shift shift, ShiftPatternShiftAttributes shiftAttributes, Organization organization,
			Campaign campaign, TimeZone optimizationTargetTimeZone, ShiftPatternPeriod period) {
		super(context);

		setName("shift_" + shift.getID().toString() + "_detail");
		setIsBorderEnabled(true);
		setRowSelectable(false);
		
		this.shift = shift;
		this.shiftAttributes = shiftAttributes;
		this.shiftPatternPeriod = period;
		dayOfWeekStart = WorkRulesUtil.getWeekStartDate(organization, campaign);
		
		extractParameters(getRequestContext().getRequest());
	}
	
	private void initHeader() {
		TableHeaderPC header = getHeader();
		
		switch (shiftPatternPeriod) {
		case Week:
			String[] dow = m_localizer.getWeekdayNames(false);
			int[] dowOrder = m_localizer.getWeekdayValues(dayOfWeekStart);
			for (int i = 0; i < dowOrder.length; i++) {
				header.addColumnHeaderData(dow[dowOrder[i]], dow[dowOrder[i]]);
			}
			break;
		case TwentyEightDayMonth:
		case TwentyNineDayMonth:
		case ThirtyDayMonth:
		case ThirtyOneDayMonth:
			for (int i = 0; i < shiftPatternPeriod.getDaysInPeriod().length; i++) {
				header.addColumnHeaderData(Integer.toString(shiftPatternPeriod.getDaysInPeriod()[i]),
						m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MONTHLY_WORK_PATTERN_DAY_HEADER,
								new Object[] { m_localizer.formatNumber(shiftPatternPeriod.getDaysInPeriod()[i]) }));
			}
			break;
		default:
		}
		
		
		header.addColumnHeaderData(FsWebBundleKey.FS_CONSISTENT_SHIFT_EVENTS_HEADER,
				getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_CONSISTENT_SHIFT_EVENTS_HEADER));
	}
	
	public HtmlTag getWrapper() {
		if (m_wrapper == null) m_wrapper = new HtmlDiv();
		m_wrapper.setStyle("margin:20px");
		return m_wrapper;
	}
	
	@Override
	public String getJavaScriptInitCode() {
		StringBuffer retVal = new StringBuffer(super.getJavaScriptInitCode());
		if (shift.getID().toInt() != ShiftFieldInfo.POSSIBLE_DAYS_OFF_SHIFT_ID) {
			retVal.append(getName()).append(".syncCheckBoxEnabled = function () { " +
				"var groupNums = []; " +
				"for (var i = 1; i <= " + shiftPatternPeriod.getDaysInPeriod().length + "; i++) { " +
					"var selections = eval('").append(getName()).append("_' + i + '_workGroupSelectionPC.getSelections()');" +
					"for (var j in selections) { " +
						"if ( selections[j] != 0 && typeof(groupNums[selections[j]]) != 'undefined' ) {") 
							.append(getConsistentShiftActivitiesCheckboxName()).append(".setEnabled(true);" +
							"return;" +
						"}" +
						"groupNums[selections[j]] = true;" +
					"}" + 
				"}")
				.append(getConsistentShiftActivitiesCheckboxName()).append(".setEnabled(false);" + 
			"};\n")
			.append(getName()).append(".syncCheckBoxEnabled();\n\n");
		}
		
		return retVal.toString();
	}

	private StringBuffer jsInitCode = new StringBuffer();
	private void initRows() {	
		Collection<MultiColumnNodeData> rows = super.getListModel();
		
		DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(
				shift.getID());

		for (int i = 0; i < shiftPatternPeriod.getDaysInPeriod().length; i++) {
			int shiftGroup = 0;
			
			if (shiftAttributes != null) {
				if (shiftAttributes.getConsistentShiftGroupMap().get(shiftPatternPeriod.getDaysInPeriod()[i]) != null) {
					shiftGroup = shiftAttributes.getConsistentShiftGroupMap().get(shiftPatternPeriod.getDaysInPeriod()[i]);
				}
			}

			WorkGroupDropDownSelectionPC workGroupSelectionPC = getConsistentWorkGroupDDPC(shift, shiftPatternPeriod.getDaysInPeriod()[i], shiftGroup);
			addChildComponent(workGroupSelectionPC);
			jsInitCode.append(workGroupSelectionPC.getJavaScriptInitCode());
			nodeData.add(workGroupSelectionPC.getHtmlDisplay()); 
		}
		
		//Consistent Shift Activities
		CheckBoxPC consistentCheckBoxPC = getConsistentCheckBoxPC();
		addChildComponent(consistentCheckBoxPC);
		nodeData.add(consistentCheckBoxPC);
		
		jsInitCode.append(getJavaScriptInitCode());
		rows.add(nodeData);
	}
	
	private CheckBoxPC getConsistentCheckBoxPC() {
		boolean consistentShiftActivities = false;
		if (shiftAttributes != null) consistentShiftActivities = shiftAttributes.isConsistentShiftActivities();
		CheckBoxPC ccPC = new CheckBoxPC(getRequestContext(),
				getName() + "_" + shift.getID().toString() + "_consistentCheckBoxPC",
				consistentShiftActivities);
		if (shiftAttributes != null && shiftAttributes.isConsistentShiftActivitiesMultiEdit()) {
			ccPC.setMultiEdit(true);
		}
		ccPC.extractParameters(m_context.getRequest());
		if (shift != null && shiftAttributes != null) {
			if (ccPC.isEdited()) {
				isEdited = true;
				ccPC.setMultiEdit(false);
				shiftAttributes.setisConsistentShiftActivities(ccPC.isChecked());
				shiftAttributes.setIsConsistentShiftActivitesMultiEdit(false);
			}
		}
		return ccPC;
	}
	
	private WorkGroupDropDownSelectionPC getConsistentWorkGroupDDPC(Shift shift, int dow, int shiftGroup) {
		WorkGroupDropDownSelectionPC ddpc = new WorkGroupDropDownSelectionPC(getRequestContext(),
				getName() + "_" + Integer.toString(dow) + "_workGroupSelectionPC");
		ddpc.setOnChangeJS(getName() + ".syncCheckBoxEnabled();");
		ddNames.add(ddpc.getName());
		ddpc.setSelectedIDs(Collections.singleton(Integer.toString(shiftGroup)));
		if (shiftAttributes != null && shiftAttributes.isConsistentShiftGroupMultiEdit(dow)) {
			ddpc.setMultiEdit(true);
		}
		ddpc.extractParameters(m_context.getRequest());
		if (shift != null && shiftAttributes != null) {
			if (ddpc.isEdited()) {
				isEdited = true;
				ddpc.setMultiEdit(false);
				if (WorkGroupDropDownSelectionPC.NO_SELECTION_ID.equals(ddpc.getSelections().iterator().next())) {
					shiftAttributes.getConsistentShiftGroupMap().remove(dow);
				} else {
					shiftAttributes.getConsistentShiftGroupMap().put(dow, Integer.parseInt(
							ddpc.getSelections().iterator().next()));
				}
				shiftAttributes.setConsistentShiftGroupMultiEdit(dow, false);
			}
		}
		return ddpc;
	}
	
	private boolean isExtracted = false;
	@Override
	public boolean extractParameters(HttpServletRequest request) {
		if (isExtracted) return true;
		isExtracted = true;
		boolean retVal = super.extractParameters(request);

		initRows();
		initHeader();
		
		return retVal;
	}
	
	public String getJSInitCode() {
		return jsInitCode.toString();
	}
	
	public ArrayList<String> getDropDownNamesInList() {
		return ddNames;
	}
	
	private String getConsistentShiftActivitiesCheckboxName() {
		return getName() + "_" + shift.getID().toString() + "_consistentCheckBoxPC";
	}
	
	public boolean isEdited() {
		return isEdited;
	}
}
