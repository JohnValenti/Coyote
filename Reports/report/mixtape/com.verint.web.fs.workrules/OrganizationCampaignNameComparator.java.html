package com.verint.web.fs.workrules;

import java.io.Serializable;
import java.util.Comparator;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;

/**
 * This Comparator is intended to be used in list components with an "Organization" column that displays the
 * name of the record's organization (or "Local" if it belongs to the campaign instead of an organization)
 */
public class OrganizationCampaignNameComparator implements Comparator<String>, Serializable {
	private static final long serialVersionUID = 1L;

	private Localizer localizer;
	private String campaignName;
	
	public OrganizationCampaignNameComparator(Localizer localizer) {
		this.localizer = localizer;
		campaignName = localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.LOCAL);
	}
	
	public int compare(String o1, String o2) {
		if (o1 == null && o2 == null) return 0;
		if (o1 == null) o1 = campaignName;
		if (o2 == null) o2 = campaignName;
		return localizer.getCollator().compare(o1, o2);
	}
}
