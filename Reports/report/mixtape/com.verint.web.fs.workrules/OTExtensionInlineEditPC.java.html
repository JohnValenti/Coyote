public class OTExtensionInlineEditPC extends InsertableMultiColListPC {
	private static final long serialVersionUID = 1L;

	private HashMap<ShiftOTExtension, ShiftPattern.StartTime> modifiedOTExtensions = new HashMap<ShiftOTExtension, ShiftPattern.StartTime>();
	private HashMap<ShiftOTExtension, Set<ShiftPattern.StartTime>> shiftOTExtensions;
	private TimeZone optimizationTargetTimeZone;
	private Date timeZoneEffectiveDate;

	public OTExtensionInlineEditPC(RequestContext context, String name,
			Map<ShiftOTExtension, Set<ShiftPattern.StartTime>> otExtensions, boolean isByOrg, ID ownerID, TimeZone optimizationTargetTimeZone,
			Date timeZoneEffectiveDate) {
		super(context);
		setName(name);
		shiftOTExtensions = new HashMap<>(otExtensions);

		this.optimizationTargetTimeZone = optimizationTargetTimeZone;
		this.timeZoneEffectiveDate = timeZoneEffectiveDate;

		ToolbarPC otExtensionToolbar = new ToolbarPC(m_context);
		otExtensionToolbar.setName("otExtensionToolbar");
		ToolbarTextButton addButton = otExtensionToolbar.createPopupWindowButton(OTExtensionListPopupPM.POPUP_ID,
				m_localizer.i18n(CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_ADD), true);

		if (ownerID != null) {
			addButton.setAttribute("ownerID", ownerID.toString());
		}
		addButton.setAttribute("isByOrg", String.valueOf(isByOrg));
		addButton.setAttribute("isSelectionPopup", String.valueOf(true));
		addButton.setAttribute("onClick", getFilterJSForAddButton(getName(), otExtensionToolbar.getName(), OTExtensionListPopupPM.POPUP_ID));

		otExtensionToolbar.addToolbarElement(addButton);
		ToolbarTextButton removeButton = new ToolbarTextButton(otExtensionToolbar, "removeOTExtension",
				m_localizer.i18n(CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_REMOVE),
				InsertableMultiColListPC.REMOVE_ROW_BUTTON_ACTION, ToolbarTextButton.BLUE_BUTTON, false);
		otExtensionToolbar.addToolbarElement(removeButton);

		addRequiredJavaScriptFile(WorkRulesJSFileID.WORK_RULES_UTIL);
		addPopupArgs(OTExtensionListPopupPM.POPUP_ID, OTExtensionListPopupPM.FORM_ACTION, "",
				OTExtensionListPopupPM.POPUP_ARGS, "1000,600,ctr,ctr", true, 1);

		// Add javascript files required by the dropdown component. This is necessary because we dynamically
		// add rows with a dropdown without a postback. If there are no rows initially, we still need the dropdown's
		// required javascript files
		StartTimeDDPC ddpc = new StartTimeDDPC(m_context, "dummy", false, false);
		for (String file : ddpc.getRequireJavaScriptFiles()) {
			addRequiredJavaScriptFile(file);
		}

		if (shiftOTExtensions != null) {
			HashMap<ID, ShiftOTExtension> otExtensionSIDs = ValueObjectUtil.getFieldObjectMap(
					ShiftOTExtensionFieldInfo.SHIFTOTEXTENSION_SID, shiftOTExtensions.keySet());
			setRowIDs(StringUtil.createDelimitedString(
					otExtensionSIDs.keySet(), StringUtil.DELIM));
		}

		setIsBorderEnabled(true);
		setToolbar(otExtensionToolbar);
		extractParameters(m_context.getRequest());
	}

	@Override
	public String getJavaScriptInitCode() {
		StringBuffer js = new StringBuffer(super.getJavaScriptInitCode());
		if (getToolbar() != null) {
			js.append("\n").append(getName()).append(".setToolbar(").append(getToolbar().getName()).append(");");
			js.append("\n").append(getName()).append(".addButtonEnableAttribute('isDeleteable', 'removeOTExtension');\n");
		}
		return js.toString();
	}

	@Override
	public String getHtmlDisplay() {
		initHeader();
		initRows();
		return super.getHtmlDisplay();
	}

	private boolean isHeaderInited = false;

	private void initHeader() {
		if (isHeaderInited) {
			return;
		}
		isHeaderInited = true;

		TableHeaderPC header = getHeader();
		Localizer localizer = m_context.getLocalizer();
		DefaultHeaderData dhd = new DefaultHeaderData(FsWebBundleKey.FS_OT_EXTENSION,
				getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_OT_EXTENSION));
		dhd.addImageButton(getOTExtensionPopupImage(), getOTExtensionPopupCaption(), getOTExtensionPopupJS());
		dhd.setSortable(true);
		header.addColumnHeaderData(dhd);
		header.addColumnHeaderData(FsWebBundleKey.FS_START_TIMES,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_START_TIMES));
	}

	private String getOTExtensionPopupImage() {
		return ImageFileID.TOOLTIP_BUTTON;
	}

	private String getOTExtensionPopupCaption() {
		return m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_VIEW_OT_EXT_DETAILS);
	}

	private String getOTExtensionPopupJS() {
		String optimizationTargetTimeZoneStr = optimizationTargetTimeZone != null ? optimizationTargetTimeZone.getID() : "";
		long tzDateMillis = timeZoneEffectiveDate != null ? timeZoneEffectiveDate.getTime() : new Date().getTime();
		return "WorkRulesUtil.handleInfoPopup(this, " + getName() + "," + getHeader().getName() + ",'" +
				OTExtensionListPopupPM.POPUP_ID + "','" + OTExtensionListPopupPM.POPUP_ARG_VO_ID + "','" +
				OTExtensionListPopupPM.POPUP_ARG_VO_IDS + "','" + OTExtensionListPopupPM.ARG_TIME_ZONE + "," +
				OTExtensionListPopupPM.ARG_TIME_ZONE_DATE + "','" +
				optimizationTargetTimeZoneStr + "," + tzDateMillis + "');";
	}

	private boolean isRowsInited = false;

	private void initRows() {
		if (isRowsInited) {
			return;
		}
		isRowsInited = true;

		List<MultiColumnNodeData> rows = super.getListModel();

		for (ShiftOTExtension shiftOTExtension : shiftOTExtensions.keySet()) {
			DefaultMultiColumnNodeData nodeData = getNodeData(shiftOTExtension);
			rows.add(nodeData);
		}

		SortUtil.sort(getListModel(), getHeader().getColumnIndexByName(FsWebBundleKey.FS_OT_EXTENSION),
				true, m_context.getLocalizer());
	}

	/**
	 * Sub-classes can override this method to set attributes on the MultiColumnNodeData.
	 *
	 * @param mcnd       MultiColumnNodeData that is populated with the Shift Event's values. Set the node attributes on this object.
	 * @param shiftEvent that is represented by this row. Use this object to determine if attributes should be set on the MultiColumnNodeData
	 */
	protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, ShiftOTExtension shiftOTExtension) {
		if (!shiftOTExtension.isMultiEditCommon()) {
			mcnd.setNodeAttribute(ROW_ATTRIBUTE_MULTI_EDIT_NOT_COMMON, String.valueOf(true));
		}
		mcnd.setNodeAttribute(WorkpaneMultiColPM.IS_DELETEABLE, String.valueOf(true));
		mcnd.setNodeAttribute(OTExtensionListPopupPM.POPUP_ARG_VO_ID, shiftOTExtension.getID().toString());
	}

	public String getRowHtml(ShiftOTExtension shiftOTExtension) {
		StringBuffer html = new StringBuffer();
		super.appendHtmlRowData(html, getNodeData(shiftOTExtension), 0, 0);
		return html.toString();
	}

	protected DefaultMultiColumnNodeData getNodeData(ShiftOTExtension shiftOTExtension) {
		DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(
				shiftOTExtension.getID());
		String otExtensionNameHTML = shiftOTExtension.getName();
		//Name
		if (shiftOTExtension.isMultiEditCommon() == false) {
			otExtensionNameHTML = MULTI_VALUE_DATA + otExtensionNameHTML;
		}
		if (shiftOTExtension.getDeleteOnDate() != null) {
			otExtensionNameHTML = new HtmlDel(otExtensionNameHTML).toString();
		}
		nodeData.add(otExtensionNameHTML);

		//Start Times
		StartTimeDDPC ddpc = getStartTimeDDPC(shiftOTExtension);
		addChildComponent(ddpc);
		setRowAttributes(nodeData, shiftOTExtension);
		nodeData.add(ddpc);
		return nodeData;
	}

	private StartTimeDDPC getStartTimeDDPC(ShiftOTExtension shiftOTExtension) {
		boolean isDeleted = shiftOTExtension.getDeleteOnDate() != null;
		StartTimeDDPC ddpc;
		Set<ShiftPattern.StartTime> startTimes = shiftOTExtensions.get(shiftOTExtension);
		if (startTimes == null || startTimes.size() == 0) {
			ddpc = new StartTimeDDPC(getRequestContext(),
					getName() + "__" + JSUtil.getJSSuffix(shiftOTExtension.getID().toString()), false, isDeleted);
		} else {
			ddpc = new StartTimeDDPC(getRequestContext(),
					getName() + "__" + JSUtil.getJSSuffix(shiftOTExtension.getID().toString()), false, isDeleted,
					startTimes.iterator().next());
			if (startTimes.size() > 1) {
				ddpc.setMultiEdit(true);
			}
		}
		ddpc.extractParameters(m_context.getRequest());
		if (shiftOTExtensions.get(shiftOTExtension) != null && shiftOTExtensions.get(shiftOTExtension).size() > 0) {
			if (ddpc.isEdited()) {
				modifiedOTExtensions.put(shiftOTExtension, ddpc.getSelections().iterator().next());
			}
			if ((shiftOTExtension.isMultiEditCommon() || ddpc.isEdited()) && ddpc.getSelections().size() > 0) {
				shiftOTExtensions.put(shiftOTExtension, Collections.singleton(ddpc.getSelections().iterator().next()));
			}
		}
		return ddpc;
	}

	private boolean isExtracted = false;

	@Override
	public boolean extractParameters(HttpServletRequest request) {
		if (isExtracted) {
			return true;
		}
		isExtracted = true;
		boolean retVal = super.extractParameters(request);

		// removed
		for (ID removedID : getRemovedRowIDCollection()) {
			for (ShiftOTExtension sote : shiftOTExtensions.keySet()) {
				if (removedID.equals(sote.getID())) {
					shiftOTExtensions.remove(sote);
					break;
				}
			}
		}

		// added
		try {
			Collection<ID> addedExtensionIDs = getAddedRowIDCollection();
			if (addedExtensionIDs.size() > 0) {
				for (ShiftOTExtension sote : WorkRulesModelHandler.getShiftOTExtensionsByIDs(m_context, addedExtensionIDs)) {
					//we want to remove any existing extensions with the same ID as the newly added ones will trump these
					shiftOTExtensions.remove(sote);
					sote.setMultiEditCommon(true);
					shiftOTExtensions.put(sote, Collections.singleton(ShiftPattern.StartTime.StartOrEndOfShift)); // This will be updated in the initRows method call below
				}
			}
		} catch (BbmFinderException e) {
			retVal = false;
			handleUnableToLoadDataError(log, e);
		}

		initHeader();
		initRows();

		return retVal;
	}

	public HashMap<ShiftOTExtension, ShiftPattern.StartTime> getAddedOTExtensionAssignments() {
		HashMap<ShiftOTExtension, ShiftPattern.StartTime> retVal = new HashMap<>();
		Map<ID, ShiftOTExtension> extensionMap = ValueObjectUtil.getIDObjectMap(shiftOTExtensions.keySet());
		for (ID id : getAddedRowIDCollection()) {
			retVal.put(extensionMap.get(id), shiftOTExtensions.get(extensionMap.get(id)).iterator().next());
		}
		return retVal;
	}

	public HashMap<ShiftOTExtension, ShiftPattern.StartTime> getModifiedOTExtensionAssignments() {
		return modifiedOTExtensions;
	}
}
