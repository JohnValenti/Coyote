package com.verint.web.fs.workrules;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.PaginationPair;
import com.bluepumpkin.common.datatypes.ScopeID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectIDListComparator;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.VTOEvent;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.ejb.core.security.Scope;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.system.RequestContext;

import java.util.*;

public class VTOEventListPM extends WorkRulesListPM {
	private static final long serialVersionUID = 1L;

	ArrayList<VTOEvent> vtoEvents = new ArrayList<VTOEvent>();

	private Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
	private Map<ID, Campaign> campaignMap = new HashMap<ID, Campaign>();
	private Map<ID, Activity> activityMap = new HashMap<ID, Activity>();

	public VTOEventListPM(RequestContext context) {
		super(context);
		removeChildComponent(m_list);
		m_list = new VTOEventListPC(
				getRequestContext(),
				"vtoEventTable",
				vtoEvents,
				orgMap,
				campaignMap,
				activityMap);
		//This needs to be done here so that we can get the sort columns/order
		m_list.extractParameters(context.getRequest());
		this.addChildComponent(m_list);
		this.setIsMultiRowSelectable(true);
	}

	public static VTOEventListPM getInstance(RequestContext context) {
		return (VTOEventListPM) getInstance(context, VTOEventListPM.class);
	}

	protected void initContentTitle() {
		super.initContentTitle();
		String pageTitle = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.WORKRULES_VTO_EVENTS_TITLE);
		m_contentTitlePC.setPageTitle(pageTitle);
	}

	protected void loadData() throws Exception {
		super.loadData();
		// Retrieve shifts events and add the selected org or campaign into the collection of objects to retrieve
		if (isByOrg) {
			if (selectedOrg != null) {
				if (m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID,
						new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE))) {
					if (isPaginationNavPressed()) {
						vtoEvents = new ArrayList<>(WorkRulesModelHandler.getVTOEventsByIDs(m_context, getPageIDs()));
					} else {
						PaginationPair<VTOEvent> vtoEventData =
								WorkRulesModelHandler.getVTOEventsByOrgHierarchy(m_context,
										getSelectedIDObject(), m_list.getHeader().isAscending(), getPageIDSize(),
										getPagePaginationIndex(), m_list.getSortComparators(), null);
						vtoEvents = new ArrayList<>(vtoEventData.getValues());
						initPaginationIDs(vtoEventData.getIDs());
					}

				} else {
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE, FsWebBundleKey.BUNDLE_NAME);
				}
			}
		} else {
			if (selectedSchedulingPeriod != null) {
				if (m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID)) {
					if (isPaginationNavPressed()) {
						vtoEvents = new ArrayList<>(WorkRulesModelHandler.getVTOEventsByIDs(m_context, getPageIDs()));
					} else {
						PaginationPair<VTOEvent> vtoEventData =
								WorkRulesModelHandler.getVTOEventsBySchedulingPeriod(m_context,
										selectedSchedulingPeriod, m_list.getHeader().isAscending(), getPageIDSize(),
										getPagePaginationIndex(), m_list.getSortComparators(), null);
						vtoEvents = new ArrayList<>(vtoEventData.getValues());
						initPaginationIDs(vtoEventData.getIDs()); //pagination not required for VTO
					}
				} else {
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE, FsWebBundleKey.BUNDLE_NAME);
				}
			}
			//if (selectedCampaign != null) optimizationTargetTimeZone = selectedCampaign.getTimeZone();
		}

		Collections.sort(vtoEvents, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));

		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<>();
		Collection<ID> campaignIDs = new HashSet<>();
		Collection<ID> activityIDs = new HashSet<>();
		for (VTOEvent ve : vtoEvents) {
			if (ve.getOrganizationID() != null) {
				orgIDs.add(ve.getOrganizationID());
			}
			if (ve.getCampaignID() != null) {
				campaignIDs.add(ve.getCampaignID());
			}
			if (ve.getActivityID() != null) {
				activityIDs.add(ve.getActivityID());
			}
		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!isByOrg) {
			campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		}
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);
	}

	@Override
	protected void initHeader() {
	}

	@Override
	protected void initSelectableItems() {
		this.removeChildComponent(m_list);
		m_list = new VTOEventListPC(m_context, "vtoEventTable", vtoEvents,
				orgMap, campaignMap, activityMap) {
			@Override
			protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, VTOEvent vtoEvent) {
				super.setRowAttributes(mcnd, vtoEvent);
				VTOEventListPM.this.setRowAttributes(mcnd, vtoEvent.getOrganizationID());
			}
		};
		m_list.getHeader().setSortable(true);
		m_list.extractParameters(m_context.getRequest());
		this.addChildComponent(m_list);
	}
}
