package com.verint.web.fs.workrules;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.time.TimeContextFactory;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftStart;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.bluepumpkin.web.fs.shift.ShiftUtil;
import com.verint.web.fs.widgets.StartTimes;
import com.witness.web.uif.data.DataSet;
import com.witness.web.uif.data.DataTable;
import com.witness.web.uif.data.DataTableColumn;
import com.witness.web.uif.data.DataTableRow;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.TimeZoneOverrideUtil;

import java.util.Collection;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;

public class ShiftExportPopupPM extends WorkRulesExportPopupPM {
	private static final long serialVersionUID = 1L;

	public static final String VIEW_ACTION = "shift_export";
	public static final String FORM_ACTION = "shift_export_download";

	private Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
	private Map<ID, Campaign> campaignMap = new HashMap<ID, Campaign>();
	private Map<ID, Activity> activityMap = new HashMap<ID, Activity>();

	public ShiftExportPopupPM(RequestContext context) {
		super(context);

		// Disable comma delimiter because shift events field comma delimits multiple assignments.
		setIsCommaDelimiterEnabled(false);
	}

	public static ShiftExportPopupPM getInstance(RequestContext context) {
		return (ShiftExportPopupPM) getInstance(context, ShiftExportPopupPM.class);
	}

	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return FORM_ACTION;
	}

	@Override
	public String getDefaultFileName() {
		return m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_SHIFT_FILE_NAME);
	}

	@Override
	public DataSet getDataSet() throws Exception {
		DataSet ds = new DataSet();

		// Only have 1 table to export.
		DataTable table = new DataTable("Shifts");
		ds.add(table);

		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_SHIFT)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_ACTIVITY)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_DURATION)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_START_TIMES)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_MIN_SPACING)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_MAX_SPACING)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_ORGANIZATION)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_DESCRIPTION)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_SHIFT_EVENTS)));

		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		Collection<ID> campaignIDs = new HashSet<ID>();
		Collection<ID> activityIDs = new HashSet<ID>();

		Collection<Shift> shifts = null;
		if (m_isByOrganization) {
			shifts = WorkRulesModelHandler.getCurrentShiftsByOrg(m_context, new ID(m_organizationId));
		} else {
			ID spID = new ID(m_schedulingPeriodId); // not sure this is the way to create an ID.
			shifts = WorkRulesModelHandler.getCurrentShiftsBySchedulingPeriodCampaignOnly(m_context, spID);
		}

		// Retrieve the needed org, campaign, and activity IDs from the shifts and associated shift events
		for (Shift shift : shifts) {
			if (shift.getOrganizationID() != null) {
				orgIDs.add(shift.getOrganizationID());
			}
			if (shift.getCampaignID() != null) {
				campaignIDs.add(shift.getCampaignID());
			}
			if (shift.getActivityID() != null) {
				activityIDs.add(shift.getActivityID());
			}
			for (ShiftEvent se : shift.getShiftEvents()) {
				if (se.getOrganizationID() != null) {
					orgIDs.add(se.getOrganizationID());
				}
				if (se.getCampaignID() != null) {
					campaignIDs.add(se.getCampaignID());
				}
				if (se.getActivityID() != null) {
					activityIDs.add(se.getActivityID());
				}
			}
		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!m_isByOrganization) {
			campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		}
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);

		if (shifts != null) {

			for (Shift shift : shifts) {
				DataTableRow row = new DataTableRow(table, false);

				// Shift Name
				row.add(shift.getName());

				// Activity Name (not supposed to be able to create a shift without activity assigned but
				// it exists in legacy data for some reason so need to be defensive).
				String activityName = (activityMap.get(shift.getActivityID()) != null ? activityMap.get(shift.getActivityID()).getName() : "");
				row.add(activityName);

				// Duration
				row.add(m_localizer.formatDurationInMins(shift.getDuration()));

				// Start Times
				Collection<ShiftStart> startTimes = shift.getShiftStarts();
				row.add(new StartTimes(
								m_context.getLocalizer(),
								TimeZoneOverrideUtil.getTimeZoneOffset(m_context, m_optimizationTargetTimezone, m_timeZoneEffectiveDate),
								ShiftUtil.getStartTimesOfDay(
										TimeContextFactory.getTimeContext(shift, orgMap, campaignMap),
										startTimes
								)
						).toString()
				);

				//Min Spacing
				row.add(m_localizer.formatDurationInMins(shift.getMinSpacing()));

				// Max Spacing
				row.add(m_localizer.formatDurationInMins(shift.getMaxSpacing()));

				//Org or Campaign
				Organization o = orgMap.get(shift.getOrganizationID());
				row.add(o == null ? m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.LOCAL) : String.valueOf(o.getName()));

				//description
				row.add((shift.getDescription() != null ? shift.getDescription().trim() : ""));

				// Assigned Shift Events
				Collection<ShiftEvent> shiftEvents = shift.getShiftEvents();
				StringBuilder sb = new StringBuilder();
				for (ShiftEvent se : shiftEvents) {
					if (sb.length() > 0) {
						sb.append(", ");
					}
					sb.append(se.getName());
				}
				row.add(sb.toString());

				table.addRow(row);
			}
		}

		return ds;
	}


}
