package com.verint.web.fs.workrules;

import java.util.Comparator;

public class PriorityComparator implements Comparator<String> {
	@Override
	public int compare(String o1, String o2) {
		if (o1 == null && o2 == null) return 0;
		if (o1 == null && o2 != null) return -1;
		if (o2 == null && o1 != null) return 1;
		
		// If the values exist, try to parse them as numbers before sorting.
		
		Integer o1Int = 0;
		Integer o2Int = 0;
		
		boolean o1FailedParse = false;
		try {
			o1Int = Integer.valueOf(o1);
		} catch (NumberFormatException ex) {
			o1FailedParse = true;
		}
		
		boolean o2FailedParse = false;
		try {
			o2Int = Integer.valueOf(o2);
		} catch (NumberFormatException ex) {
			o2FailedParse = true;
		}
		
		if(o1FailedParse && o2FailedParse) {
			// We know both objects are non-null or a return value would have already happened
			// per the initial logic so these are both non-null strings but they couldn't be parsed
			// as numbers so just perform a string comparison on the original objects.
			return o1.compareTo(o2);
		}
		
		// If you manage to get this far, perform a numeric comparison on the two objects.
		return o1Int.compareTo(o2Int);
	}
}
