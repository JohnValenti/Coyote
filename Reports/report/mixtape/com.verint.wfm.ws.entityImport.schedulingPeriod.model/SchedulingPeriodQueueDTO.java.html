package com.verint.wfm.ws.entityImport.schedulingPeriod.model;

import java.util.Arrays;
import java.util.List;

import javax.xml.bind.annotation.XmlType;

import com.verint.wfm.ws.entityImport.model.HoursOfOperationDTO;
import com.verint.wfm.ws.entityImport.schedulingPeriod.SchedulingPeriodNotPersistedInfo;

@XmlType(name = "SchedulingPeriodQueueDTO", propOrder = { "queueExternalSourceId", "skillExternalSourceId",
		"serviceGoalType", "serviceLevelAnswerTime", "abandonmentTime", "overrideHoursOfOperationDTOs", "numSeats",
		"serviceGoalDTO", "serviceGoalSeriesDTO"  })
public class SchedulingPeriodQueueDTO {
	private static final long serialVersionUID = 1L;
	
	public static final String DEADLINE = "DEADLINE";
	public static final String AVERAGE_ANSWER_TIME = "AVERAGE_ANSWER_TIME";
	public static final String SERVICE_LEVEL_ANSWER_TIME = "SERVICE_LEVEL_ANSWER_TIME";
	final public static List<String> SERVICE_GOAL_TYPES = Arrays.asList(SERVICE_LEVEL_ANSWER_TIME, AVERAGE_ANSWER_TIME,
			DEADLINE);

	// protected SchedPeriodExternalNaturalKey m_schedPeriodExternalNaturalKey;
	protected String m_queueExternalSourceId;
	protected String m_skillExternalSourceId;
	// Choose SERVICE_LEVEL_ANSWER_TIME, AVERAGE_ANSWER_TIME, DEADLINE,
	// TODO SERVICE_LEVEL_ANSWER_TIME is always OK for any queue type
	// TODO validate AVERAGE_ANSWER_TIME not allowed when (1) there is skill or (2) back office
	// TODO validate DEADLINE not allowed when (1) Not back office
	protected String m_serviceGoalType;
	// Note: this is set only for type SERVICE_LEVEL_ANSWER_TIME
	protected Short m_serviceLevelAnswerTime;
	// Abandonment after how much time?
	protected short m_abandonmentTime;
	protected List<HoursOfOperationDTO> m_overrideHoursOfOperationDTOs;
	protected Integer m_numSeats;
	protected ServiceGoalDTO m_serviceGoalDTO;
	protected ServiceGoalSeriesDTO m_serviceGoalSeriesDTO;

	// ------------------------------------------- methods
	public SchedulingPeriodQueueDTO() {
	}

	public String getQueueExternalSourceId() {
		return m_queueExternalSourceId;
	}

	public void setQueueExternalSourceId(String queueExternalSourceId) {
		m_queueExternalSourceId = queueExternalSourceId;
	}

	public String getSkillExternalSourceId() {
		return m_skillExternalSourceId;
	}

	public void setSkillExternalSourceId(String skillExternalSourceId) {
		m_skillExternalSourceId = skillExternalSourceId;
	}

	public String getServiceGoalType() {
		return m_serviceGoalType;
	}

	public void setServiceGoalType(String serviceGoalType) {
		m_serviceGoalType = serviceGoalType;
	}

	public Short getServiceLevelAnswerTime() {
		return m_serviceLevelAnswerTime;
	}

	public void setServiceLevelAnswerTime(Short serviceLevelAnswerTime) {
		m_serviceLevelAnswerTime = serviceLevelAnswerTime;
	}

	public short getAbandonmentTime() {
		return m_abandonmentTime;
	}

	public void setAbandonmentTime(short abandonmentTime) {
		m_abandonmentTime = abandonmentTime;
	}

	public Integer getNumSeats() {
		return m_numSeats;
	}

	public void setNumSeats(Integer numSeats) {
		m_numSeats = numSeats;
	}

	public List<HoursOfOperationDTO> getOverrideHoursOfOperationDTOs() {
		return m_overrideHoursOfOperationDTOs;
	}

	public void setOverrideHoursOfOperationDTOs(List<HoursOfOperationDTO> overrideHoursOfOperationDTOs) {
		m_overrideHoursOfOperationDTOs = overrideHoursOfOperationDTOs;
	}

	public ServiceGoalDTO getServiceGoalDTO() {
		return m_serviceGoalDTO;
	}

	public void setServiceGoalDTO(ServiceGoalDTO serviceGoalDTO) {
		m_serviceGoalDTO = serviceGoalDTO;
	}
	
	public ServiceGoalSeriesDTO getServiceGoalSeriesDTO() {
		return m_serviceGoalSeriesDTO;
	}

	public void setServiceGoalSeriesDTO(ServiceGoalSeriesDTO serviceGoalSeriesDTO) {
		m_serviceGoalSeriesDTO = serviceGoalSeriesDTO;
	}

	public void validate(SchedulingPeriodNotPersistedInfo schedulingPeriodNotPersistedInfo) {

		if (this.getServiceGoalType().equals(SchedulingPeriodQueueDTO.DEADLINE)) {
			schedulingPeriodNotPersistedInfo.addSchedulingPeriodQueueServiceGoalTypeIsNotSupported(this
					.getServiceGoalType());
		}
		if (this.getServiceGoalDTO() == null && this.getServiceGoalSeriesDTO()==null) {
			schedulingPeriodNotPersistedInfo.addSchedulingPeriodQueueServiceGoalTypeIsMissing(this
					.getQueueExternalSourceId());
		}

		if (this.getServiceGoalType().equals(SchedulingPeriodQueueDTO.SERVICE_LEVEL_ANSWER_TIME)
				&& (this.getServiceLevelAnswerTime() == null) || (this.getServiceLevelAnswerTime().intValue() == 0)) {
			schedulingPeriodNotPersistedInfo.addSchedulingPeriodQueueServiceLevelAnswerTimeMissing(this
					.getQueueExternalSourceId());
		}

		if (this.getServiceGoalType().equals(SchedulingPeriodQueueDTO.SERVICE_LEVEL_ANSWER_TIME)
				&& this.getAbandonmentTime() == 0) {
			schedulingPeriodNotPersistedInfo
					.addSchedulingPeriodQueueServiceLevelAbandonmentTimeMissingWhenGoalTypeIsAnswerTime(this
							.getQueueExternalSourceId());
		}
	}
}
