package com.verint.wfm.ws.entityImport.schedulingPeriod.model;

import javax.xml.bind.annotation.XmlType;

@XmlType(name = "ServiceGoalSeriesDTO", propOrder = {
		"serviceLevelPercentSeriesEncoded", "averageAnswerTimeSeriesEncoded" })
public class ServiceGoalSeriesDTO {
	/**
	 * The Service Level Percent answered with in a specific time. The time is
	 * set at a SPQueue level, but the %answered is per hour per spqueue
	 */
	protected String m_serviceLevelPercentSeriesEncoded;
	/**
	 * Average speed of answer in seconds encoded
	 */
	protected String m_averageAnswerTimeSeriesEncoded;

	public String getServiceLevelPercentSeriesEncoded() {
		return m_serviceLevelPercentSeriesEncoded;
	}

	public void setServiceLevelPercentSeriesEncoded(
			String serviceLevelPercentSeriesEncoded) {
		this.m_serviceLevelPercentSeriesEncoded = serviceLevelPercentSeriesEncoded;
	}

	public String getAverageAnswerTimeSeriesEncoded() {
		return m_averageAnswerTimeSeriesEncoded;
	}

	public void setAverageAnswerTimeSeriesEncoded(
			String averageAnswerTimeSeriesEncoded) {
		this.m_averageAnswerTimeSeriesEncoded = averageAnswerTimeSeriesEncoded;
	}

}
