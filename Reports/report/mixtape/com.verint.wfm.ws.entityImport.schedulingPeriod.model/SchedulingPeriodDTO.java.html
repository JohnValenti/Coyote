package com.verint.wfm.ws.entityImport.schedulingPeriod.model;

import java.text.DateFormat;
import java.text.SimpleDateFormat;
import java.util.Calendar;
import java.util.Date;
import java.util.List;
import java.util.TimeZone;

import javax.xml.bind.annotation.XmlElement;
import javax.xml.bind.annotation.XmlType;
import javax.xml.bind.annotation.adapters.XmlJavaTypeAdapter;

import com.bluepumpkin.common.util.StringUtil;
import com.verint.wfm.ws.ServiceApplicationException;
import com.verint.wfm.ws.entityImport.model.HoursOfOperationDTO;
import com.verint.wfm.ws.entityImport.schedulingPeriod.SchedulingPeriodServiceDelegate;
import com.verint.wfm.ws.util.CalendarUtil;
import com.verint.wfm.ws.util.JaxbDateSerializer;

@XmlType(name = "SchedulingPeriodDTO", propOrder = { "schedulingPeriodExternalNaturalKey", "skillBased",
		"usingAllEmployees", "organizationNames", "hoursOfOperationDTOs", "schedulingPeriodQueueDTOs" })
public class SchedulingPeriodDTO {
	private static final long serialVersionUID = 1L;

	protected SchedulingPeriodExternalNaturalKey m_schedulingPeriodExternalNaturalKey;
	protected boolean m_isSkillBased;
	protected boolean m_isUsingAllEmployees;
	protected List<String> m_organizationNames;
	protected List<HoursOfOperationDTO> m_hoursOfOperationDTOs;
	protected List<SchedulingPeriodQueueDTO> m_schedulingPeriodQueueDTOs;

	public SchedulingPeriodDTO() {
	}

	public SchedulingPeriodExternalNaturalKey getSchedulingPeriodExternalNaturalKey() {
		return m_schedulingPeriodExternalNaturalKey;
	}

	public void setSchedulingPeriodExternalNaturalKey(
			SchedulingPeriodExternalNaturalKey schedulingPeriodExternalNaturalKey) {
		m_schedulingPeriodExternalNaturalKey = schedulingPeriodExternalNaturalKey;
	}

	public boolean isSkillBased() {
		return m_isSkillBased;
	}

	public void setSkillBased(boolean isSkillBased) {
		m_isSkillBased = isSkillBased;
	}

	public boolean isUsingAllEmployees() {
		return m_isUsingAllEmployees;
	}

	public void setUsingAllEmployees(boolean isUsingAllEmployees) {
		m_isUsingAllEmployees = isUsingAllEmployees;
	}

	public List<String> getOrganizationNames() {
		return m_organizationNames;
	}

	public void setOrganizationNames(List<String> organizationNames) {
		m_organizationNames = organizationNames;
	}

	public List<HoursOfOperationDTO> getHoursOfOperationDTOs() {
		return m_hoursOfOperationDTOs;
	}

	public void setHoursOfOperationDTOs(List<HoursOfOperationDTO> hoursOfOperationDTOs) {
		m_hoursOfOperationDTOs = hoursOfOperationDTOs;
	}

	public List<SchedulingPeriodQueueDTO> getSchedulingPeriodQueueDTOs() {
		return m_schedulingPeriodQueueDTOs;
	}

	public void setSchedulingPeriodQueueDTOs(List<SchedulingPeriodQueueDTO> schedulingPeriodQueueDTOs) {
		m_schedulingPeriodQueueDTOs = schedulingPeriodQueueDTOs;
	}

	@XmlType(name = "SchedulingPeriodExternalNaturalKey", propOrder = { "campaignExternalSourceId", "startDate" })
	public static class SchedulingPeriodExternalNaturalKey {
		private String m_campaignExternalSourceId;
		private Date m_startDate;

		public SchedulingPeriodExternalNaturalKey() {
		}

		public SchedulingPeriodExternalNaturalKey(String campaignExternalSourceId, Date startDate) {
			m_campaignExternalSourceId = campaignExternalSourceId;
			m_startDate = startDate;
		}

		public String getCampaignExternalSourceId() {
			return m_campaignExternalSourceId;
		}

		public void setCampaignExternalSourceId(String campaignExternalSourceId) {
			m_campaignExternalSourceId = campaignExternalSourceId;
		}

		public Calendar startCalendarWithZone(TimeZone timeZone) throws ServiceApplicationException {
			return CalendarUtil.convertMidnightDateToCalendarWithZone(m_startDate, timeZone);
		}

		@XmlElement(type = Date.class)
		@XmlJavaTypeAdapter(value = JaxbDateSerializer.class)
		public Date getStartDate() {
			return m_startDate;
		}

		public void setStartDate(Date startDate) {
			m_startDate = startDate;
		}

		@Override
		public int hashCode() {
			// CAT.error("hash inputs: " + m_campaignExternalSourceId + ", " + m_startDate);
			return m_campaignExternalSourceId.trim().hashCode() + m_startDate.hashCode();
		}

		@Override
		public boolean equals(Object obj) {
			SchedulingPeriodExternalNaturalKey k = (SchedulingPeriodExternalNaturalKey) obj;
			// CAT.error("equals this? " + m_campaignExternalSourceId.trim() + ", " + m_startDate + "that? "
			// + k.m_campaignExternalSourceId + ", " + k.m_startDate);
			// CAT.error("equals? " + m_campaignExternalSourceId.trim().equals(k.m_campaignExternalSourceId.trim()) +
			// "&&"
			// + m_startDate.equals(k.m_startDate));
			return m_campaignExternalSourceId.trim().equals(k.m_campaignExternalSourceId.trim())
					&& m_startDate.equals(k.m_startDate);
		}

		@Override
		public String toString() {
			StringBuilder b = new StringBuilder();
			if (!StringUtil.isEmptyOrWhiteSpace(this.getCampaignExternalSourceId())) {
				b.append(" Campaign External Id: ");
				b.append(this.getCampaignExternalSourceId());
			}
			if (m_startDate != null) {
				b.append(" StartDate relative format: ");

				Calendar zonelessCalendar = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
				zonelessCalendar.setTime(m_startDate);

				DateFormat zonelessFormat = new SimpleDateFormat(SchedulingPeriodServiceDelegate.DATE_FORMAT);
				zonelessFormat.setTimeZone(TimeZone.getTimeZone("UTC"));

				b.append(zonelessFormat.format(zonelessCalendar.getTime()));
			}
			return b.toString();
		}
	}

}
