package com.bluepumpkin.ejb.rm.requests.commonrequestmanager.model;

import java.util.Iterator;
import java.util.List;
import java.util.Map;
import java.util.HashMap;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeName;


/**
 * This class combines a list of requests with a Map of employee id to
 * employee name objects.
 */
public class RequestCollection implements java.io.Serializable {
    
    /**
     * The List of request objects requested.  This is the subset of the value 
     * objects requested, i.e. the current "chunk".  They are in sorted order.
     */
    protected List m_requests = null;
    
    /**
     * The List of IDs.  This is the whole list of request IDs in the requested
     * sort order.
     */
    protected List m_ids = null;
    
    /**
     * The map of employeeId to EmployeeName objects.  The map uses employee id
     * as the key and maps to an 
     * {@link com.bluepumpkin.ejb.bbm.workresource.model.EmployeeName Employeename}
     * object representing the employee's name.
     */
    protected Map m_nameMap = null;
    
    private static final long serialVersionUID = 7418492200566407143L;

	/** Creates a new instance of RequestCollection
     * @param requests a collection of request objects
     * @param nameMap a map of employeeID to EmployeeName objects.  The set of
     * employeeIds used is the set of all employeeIds used in the collection of
     * request objects.
	 * @see com.bluepumpkin.ejb.bbm.workresource.model.EmployeeName
     */
    public RequestCollection(List ids, List requests, Map nameMap) {
        // Test that the inputs are not null.  Use 
        // an empty list or map to create this instance if necessary.
        if (ids == null) {
            throw new IllegalArgumentException("request id list cannot be null, use empty list if necessary");
        }
        if (requests == null) {
            throw new IllegalArgumentException("request list cannot be null, use empty list if necessary");
        }
        if (nameMap == null) {
            throw new IllegalArgumentException("nameMap cannot be null, use empty map if necessary");
        }
        m_ids = ids;
        m_requests = requests;
        m_nameMap  = nameMap;
	}
           
    /**
     * Return an iterator over the collection of requests
     * @return an iterator over the collection of requests.
     */
    public Iterator iterator() {
        return m_requests.iterator();
    }
    
    /**
     * Return the underlying request collection.
     * @return the collection or requests stored in this object.
     */
    public List getRequests() {
        return m_requests;
    }
    
    /**
     * Returns <code>true</code> if there are no requests in the collection.
     * @return <code>true</code> if the collection is empty.
     */
    public boolean isEmpty() {
        return m_requests.isEmpty();
        }
    
    /**
     * Add a name to the map of (id,name) pairs.
     * @param empId the id of the employee whose name is being added
     * @param name the EmployeeName object containing information about this
     * employee's name.
     */
    public void put(ID empId, EmployeeName name) {
        m_nameMap.put(empId, name);
    }
    
    /**
     * Lookup a name in the employee name map
     * @param empId the id of the employee whose name is being requested
     * @return the EmployeeName object containing information about this
     * employee's name.
     */
    public EmployeeName get(ID empId) {
        return (EmployeeName)m_nameMap.get(empId);
    }
    
    /**
     * Get the map used for the to associate employeeIds with employeeName objects
     * by this request collection.
     */
    public Map getNameMap() {
        return m_nameMap;
    }

    /**
     * Get the list of ids stored in this Request Collection.
     */
    public List getIds() {
        return m_ids;
    }

}
