package com.bluepumpkin.ejb.bbm.empworkrule.model;


import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

import java.util.Calendar;
import java.util.Date;

public class WorkResourceMonthlyMinMaxHour extends ValueObjectBase {
	private static final long serialVersionUID = 1L;

	private static FieldInfo fieldInfo = new WorkResourceMonthlyMinMaxHourFieldInfo();

	private Calendar monthYearRepresentation;

	public WorkResourceMonthlyMinMaxHour() {
		super();
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return fieldInfo;
	}

	public ID getID() {
		return getFieldValueID(WorkResourceMonthlyMinMaxHourFieldInfo.Column.ID.getIndex());
	}

	public void setID(ID id) {
		setFieldValue(WorkResourceMonthlyMinMaxHourFieldInfo.Column.ID.getIndex(), id);
	}

	public ID getWorkResourceID() {
		return getFieldValueID(WorkResourceMonthlyMinMaxHourFieldInfo.Column.WORKRESOURCEID.getIndex());
	}

	public void setWorkResourceID(ID workResourceID) {
		setFieldValue(WorkResourceMonthlyMinMaxHourFieldInfo.Column.WORKRESOURCEID.getIndex(), workResourceID);
	}

	public int getCalendarMonth() {
		return getMonthYearRepresentation().get(Calendar.MONTH);
	}

	public void setCalendarMonth(int calendarMonth) {
		getMonthYearRepresentation().set(Calendar.MONTH, calendarMonth);
		refreshMonthYearField();
	}

	public int getYear() {
		return getMonthYearRepresentation().get(Calendar.YEAR);
	}

	public void setYear(int year) {
		getMonthYearRepresentation().set(Calendar.YEAR, year);
		refreshMonthYearField();
	}

	public int getMinMinutes() {
		return getFieldValueInt(WorkResourceMonthlyMinMaxHourFieldInfo.Column.MINMINUTES.getIndex());
	}

	public void setMinMinutes(int minMinutes) {
		setFieldValue(WorkResourceMonthlyMinMaxHourFieldInfo.Column.MINMINUTES.getIndex(), minMinutes);
	}

	public int getMaxMinutes() {
		return getFieldValueInt(WorkResourceMonthlyMinMaxHourFieldInfo.Column.MAXMINUTES.getIndex());
	}

	public void setMaxMinutes(int maxMinutes) {
		setFieldValue(WorkResourceMonthlyMinMaxHourFieldInfo.Column.MAXMINUTES.getIndex(), maxMinutes);
	}

	private Calendar getMonthYearRepresentation() {
		if (monthYearRepresentation == null) {
			monthYearRepresentation = Calendar.getInstance(TimeZoneUtil.GMT_TIMEZONE);
			Date date = getFieldValueDate(WorkResourceMonthlyMinMaxHourFieldInfo.Column.MONTHYEAR.getIndex());
			if (date == null) {
				monthYearRepresentation.set(1980, Calendar.JANUARY, 1, 0, 0, 0);
			} else {
				monthYearRepresentation.setTime(date);
			}
		}
		return monthYearRepresentation;
	}

	private void refreshMonthYearField() {
		setFieldValue(WorkResourceMonthlyMinMaxHourFieldInfo.Column.MONTHYEAR.getIndex(),
				getMonthYearRepresentation().getTime());
	}

	/**
	 * Returns true if the ID of this record is null (indicating it has not been persisted to the DB yet).
	 */
	public boolean isNewRecord() {
		return getID() == null;
	}
}
