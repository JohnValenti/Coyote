package com.bluepumpkin.ejb.bbm.empworkrule.model;

import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class EmpTimeBankAdjust extends ValueObjectBase {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// meta data information
	private static FieldInfo m_fieldInfo = new EmpTimeBankAdjustFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	public ID getID() {
		return new ID(this.getFormattedFieldValue(EmpTimeBankAdjustFieldInfo.EMPTIMEBANKADJUST_ID));
	}
	
	public ID getEmployeeID() {
		return this.getFieldValueID(EmpTimeBankAdjustFieldInfo.EMPTIMEBANKADJUST_EMPLOYEEID);
	}

	public int getAdjustMinute() {
		return this.getFieldValueInt(EmpTimeBankAdjustFieldInfo.EMPTIMEBANKADJUST_ADJUSTMINUTES);
	}
	
	public Date getAdjustDate() {
		return this.getFieldValueDate(EmpTimeBankAdjustFieldInfo.EMPTIMEBANKADJUST_ADJUSTDATE);
	}
	
	public void setEmployeeID(ID value) {
		this.setFieldValue(EmpTimeBankAdjustFieldInfo.EMPTIMEBANKADJUST_EMPLOYEEID, value);
	}
	
	public void setAdjustMinute(int value) {
		this.setFieldValue(EmpTimeBankAdjustFieldInfo.EMPTIMEBANKADJUST_ADJUSTMINUTES, value);
	}
	
	public void setAdjustDate(Date value) {
		this.setFieldValue(EmpTimeBankAdjustFieldInfo.EMPTIMEBANKADJUST_ADJUSTDATE, value);
	}

}
