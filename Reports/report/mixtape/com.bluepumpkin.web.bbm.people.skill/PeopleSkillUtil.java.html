package com.bluepumpkin.web.bbm.people.skill;

import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.ScopeID;
import com.bluepumpkin.ejb.bbm.workresource.OrganizationScopeManagerProxy;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.ejb.core.security.model.User;

/**
 * Title:        PeopleSkillUtil
 * Description:  Utility for People Skill code
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, dsu
 * @version      1.0
 *
 * Created on November 21, 2002, 4:55 PM
 */
public class PeopleSkillUtil {

	/** Creates a new instance of PeopleSkillUtil */
	private PeopleSkillUtil() {}

	/**
	 * Return true if User is authorized to View Employee Skill
	 */
	public static boolean isAuthToViewEmpSkillInfo(User user, ID empID, Map eOrgMap) {
		ID myEmpID = user.getEmployeeID();
		if (empID.equals(myEmpID)) {
			return user.isAuthorized(PrivilegeKeys.SS_VIEWPERSONALSKILLS_ID);
		} else {
			ID orgID = (ID)eOrgMap.get(empID);
			return user.isAuthorized(PrivilegeKeys.SS_VIEWEMPLOYEESKILLS_ID,
				OrganizationScopeManagerProxy.getScopeID(orgID));
		}
	}

	/**
	 * Return true if User is authorized to View Employee Template Skill
	 */
	public static boolean isAuthToViewEmpTemplateSkillInfo(User user, ID etID, Map etOrgMap) {
		ID orgID = (ID)etOrgMap.get(etID);
		return user.isAuthorized(PrivilegeKeys.SS_VIEWEMPLOYEESKILLS_ID,
				OrganizationScopeManagerProxy.getScopeID(orgID));
	}

	/**
	 * Return true if User is authorized to View Employee Skill and Proficiency
	 */
	public static boolean isAuthToViewEmpSnPInfo(User user, ID empID, Map eOrgMap) {
		ID myEmpID = user.getEmployeeID();
		if (empID.equals(myEmpID)) {
			return user.isAuthorized(PrivilegeKeys.SS_VIEWPERSONALSKILLS_ID) &&
			  user.isAuthorized(PrivilegeKeys.SS_VIEWPERSONALPROFICIENCIES_ID);
		} else {
			ID orgID = (ID)eOrgMap.get(empID);
			ScopeID orgScopeID = OrganizationScopeManagerProxy.getScopeID(orgID);
			return user.isAuthorized(PrivilegeKeys.SS_VIEWEMPLOYEESKILLS_ID, orgScopeID) &&
				user.isAuthorized(PrivilegeKeys.SS_VIEWEMPLOYEEPROFICIENCIES_ID, orgScopeID);
		}
	}

	/**
	 * Return true if User is authorized to View Employee Template Skill and Proficiency
	 */
	public static boolean isAuthToViewEmpTemplateSnPInfo(User user, ID etID, Map etOrgMap) {
		ID orgID = (ID)etOrgMap.get(etID);
		ScopeID orgScopeID = OrganizationScopeManagerProxy.getScopeID(orgID);
		return user.isAuthorized(PrivilegeKeys.SS_VIEWEMPLOYEESKILLS_ID, orgScopeID) &&
				user.isAuthorized(PrivilegeKeys.SS_VIEWEMPLOYEEPROFICIENCIES_ID, orgScopeID);
	}
}
