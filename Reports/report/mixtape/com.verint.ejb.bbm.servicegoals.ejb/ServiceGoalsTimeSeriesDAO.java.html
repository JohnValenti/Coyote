package com.verint.ejb.bbm.servicegoals.ejb;


import java.util.ArrayList;

import java.util.HashSet;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;

/**
 * This DAO handles the saving of service goals time series records.
 * This is an abstract DAO with 3 known subclasses: ASATimeSeriesDAO, MaxDialsTimeSeriesDAO, and ServiceLevelPercentDAO.
 * All of the database logic for these subclasses is located in this DAO, however the DAO base class specifies that DAOs
 * need to declare a createValueObject() method, which is left to the 3 subclasses in this instance.  This can probably
 * be refactored such that everything is in one class, and the subclasses are removed.
 */
public abstract class ServiceGoalsTimeSeriesDAO <T extends ServiceGoalsTimeSeriesDataItem> extends DAOBase<T> {
	private static FieldInfo m_fieldInfo = new ServiceGoalsTimeSeriesDataItemFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	/**
	* default constructor
	*/
	public ServiceGoalsTimeSeriesDAO() {}
	
	/**
	 * Saves a collection of service goal time series records for the given SPQueue.
	 * @param spQueue
	 * @param timeSeriesToSave
	 * @param name
	 */
	public void saveServiceGoalsTimeSeries(
			List<T> timeSeriesToSave,
			SPQueue spQueue,
			String name) throws BbmUpdateException, BbmRemoveException, BbmCreateException, JdmoException {
		Map<SPQueue, List<T>> tempMap = new HashMap<SPQueue, List<T>>();
		tempMap.put(spQueue, timeSeriesToSave);
		saveServiceGoalsTimeSeries(tempMap, name);
	}
	
	/**
	 * Saves a collection of service goal time series records for each of the given SPQueues.
	 * @param timeSeriesBySpQueue - key = spqueue, map = collection of service goals time series records associated to the spqueue
	 * @param name - user name of the current logged in user
	 */
	public void saveServiceGoalsTimeSeries(
			Map<SPQueue, List<T>> timeSeriesBySpQueue,
			String name) throws BbmUpdateException, BbmRemoveException, BbmCreateException, JdmoException {

		HashSet<ID> spQueueIDs =
			new HashSet<ID>(ValueObjectUtil.getFieldObjectCol(SPQueueFieldInfo.SPQUEUE_ID, timeSeriesBySpQueue.keySet()));
		
		//Delete the previous required time series records for the given spqueues
		StringBuffer query = new StringBuffer();
		query.append("DELETE " + getFieldInfo().getTableName() + " WHERE " +
				getFieldInfo().getFieldName(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_SPQUEUEID)
				+ " IN ");
		query.append(getDMO().createInClause(spQueueIDs));
		getDMO().executeCommand(query.toString());

		ArrayList<T> allTimeSeries = new ArrayList<T>();
		for (SPQueue spQueue : timeSeriesBySpQueue.keySet()) {
			for (T dataItem : timeSeriesBySpQueue.get(spQueue)) {
				dataItem.setFieldValue(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_SPQUEUEID, spQueue.getDEID());
				dataItem.setFieldValue(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_CREATEDBY, name);
				dataItem.setFieldValue(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_CREATED, new Date());
				allTimeSeries.add(dataItem);
			}
		}
		createObjects(allTimeSeries);
	}
}
