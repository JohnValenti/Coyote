package com.bluepumpkin.web.fs.calendar;

import javax.servlet.http.HttpServletRequest;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.campaign.ejb.CampaignManager;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.web.bbm.campaign.selection.CampaignSchedulingPeriodSelectionUtil;
import com.bluepumpkin.web.core.base.RequestHandler;
import com.bluepumpkin.web.core.cache.SelectionCacheUtil;
import com.bluepumpkin.web.fs.keys.FsKeys;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.system.RequestContext;

/**
 * Handles requests to the top-level Calendar window, which contains both the left-hand
 * navigation pane and the right-hand applet pane.
 */
public class CalendarRequestHandler extends RequestHandler {
	private CampaignManager m_campaignManager = null;

	/**
	 * If the Calendar is launched with an SPID in the URL, that SP and its campaign should
	 * be preselected in the navigation pane.  This method handles that.
	 */
	@Override
	public void processRequest(RequestContext context) throws Exception {
		HttpServletRequest request = context.getRequest();
		String action = request.getParameter(FsKeys.CALENDAR_PARAMETER_ACTION);
		if (FsKeys.CALENDAR_ACTION_DRILLDOWN.equals(action)) {
			String SPIDString = request.getParameter(FsKeys.CALENDAR_PARAMETER_SPID);
			if (SPIDString != null) {
				ID spid = new ID(SPIDString);
				SchedulingPeriod sp = getCampaignManager(context).getSchedulingPeriodByID(spid);
				// Set selected campaign and SP IDs using the s:201,c:6 format.
				CampaignSchedulingPeriodSelectionUtil.setSelectedAutoCachedCampaignSPID(context, sp.getCampaignID(), sp.getID());
				// Set selected campaign and SP IDs using separate IDs.
				SelectionCacheUtil.setCampaignID(context, sp.getParentID());
				SelectionCacheUtil.setSchedulingPeriodID(context, sp.getID());
			}
		}
	}

	/**
	 * Returns an instance of CampaignManager, creating one if necessary.
	 * 
	 * @param context
	 * @return
	 */
	private CampaignManager getCampaignManager(RequestContext context) {
		if (m_campaignManager == null) {
			try {
				m_campaignManager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
			} catch (BbmEJBCreateException e) {
				e.printStackTrace();
			}
		}
		return m_campaignManager;
	}
}
