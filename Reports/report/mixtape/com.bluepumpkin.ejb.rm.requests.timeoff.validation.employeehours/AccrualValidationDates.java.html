package com.bluepumpkin.ejb.rm.requests.timeoff.validation.employeehours;

import java.util.Calendar;
import java.util.Date;

import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;

public class AccrualValidationDates {

	//The request choice start date.
	Date choiceStart;
	//The request choice end date. This is exclusive
	Date choiceEnd;

	//The validation start based on the request choice start and the employee's start date.
	//This value will be ignored by the calculation if the employee has an accrual
	//In that case the start date for the calculation is the accrued at date.
	//See TOAccrual.getStartDate
	Date validationStart;
	//The validation end for this employee based on the request choice end. This is inclusive.
	Date validationEnd;

	//does the choice cross the employee's time off year boundary?
	boolean requestCrossesTimeOffYearBoundary;



	public boolean doesRequestCrossTimeOffYearBoundary() {
		return requestCrossesTimeOffYearBoundary;
	}

	public Date getValidationStart() {
		return validationStart;
	}
	public Date getValidationEnd() {
		return validationEnd;
	}

	public Date getChoiceStart() {
		return choiceStart;
	}

	public void initialize(AccrualValidationInputs inputs, TOChoice toChoice) throws Exception { // NOSONAR

		Organization org = inputs.employeeOrganization;
		choiceStart = toChoice.getStartDate();
		choiceEnd = toChoice.getEndDate();


		//start of the time off year for the choice start
		Calendar choiceStartTOYear = RequestUtil.getEmployeeTimeOffYearStart(toChoice.getStartDate(), org, inputs.employeeTimeOffYearStart);

		//start of the time off year for the choice end. Note that TOChoice.getEndDate() is exclusive. 
		Calendar choiceEndTOYear = RequestUtil.getEmployeeTimeOffYearStart(toChoice.getEndDate(), org, inputs.employeeTimeOffYearStart);

		if (choiceEndTOYear.after(choiceStartTOYear)) {
			requestCrossesTimeOffYearBoundary = true;
		}

		Date empStart = inputs.employee.getStartTime();

		//Start validation from the time off year start for the request choice start (sanity check the employee start date to make 
		//sure the validationStart is not before the employee's start date)
		//TOAccrual will ignore this value and use the last accrued date for the employee if it exists 
		validationStart = choiceStartTOYear.getTime().after(empStart) ? choiceStartTOYear.getTime() : empStart;

		if (toChoice.getEndDate().equals(choiceEndTOYear.getTime())) {
			//The choice ends right at the start of time off year for the choice end.
			//Adjustment since TOChoice.getEndDate() is exclusive. 
			validationEnd = new Date(toChoice.getEndDate().getTime() - 1);
		} else {
			//set to the end of the time off year for the choice's end
			Calendar empTOYearEndCalToChoiceEnd = RequestUtil.getEmployeeTimeOffYearEnd(toChoice.getEndDate(), org,
					inputs.employeeTimeOffYearStart);
			validationEnd = empTOYearEndCalToChoiceEnd.getTime();
		}

	}

}
