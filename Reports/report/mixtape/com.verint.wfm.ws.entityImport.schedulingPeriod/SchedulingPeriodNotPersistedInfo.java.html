package com.verint.wfm.ws.entityImport.schedulingPeriod;

import java.util.Calendar;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.verint.wfm.ws.entityImport.EntityNotPersistedInfo;
import com.verint.wfm.ws.entityImport.schedulingPeriod.model.SchedulingPeriodDTO.SchedulingPeriodExternalNaturalKey;

/**
 * For each scheduling period not persisted by createOrUpdateSchedulingPeriods(), one of these will be in the list returned
 * This is returned in lieu of an exception, because we always want to continue on to all the organizations
 *
 * @see EntityImportService.createOrUpdateSchedulingPeriods()
 */
public class SchedulingPeriodNotPersistedInfo extends EntityNotPersistedInfo<SchedulingPeriodExternalNaturalKey> {
	private static final long serialVersionUID = 1L;

	public SchedulingPeriodNotPersistedInfo() {
		super();
	}

	public SchedulingPeriodNotPersistedInfo(SchedulingPeriodExternalNaturalKey schedulingPeriodExternalNaturalKey) {
		super(schedulingPeriodExternalNaturalKey);
	}

	// -----------------------------------------
	public SchedulingPeriodExternalNaturalKey getSchedulingPeriodExternalNaturalKey() {
		return super.getEntityExternalSourceId();
	}

	/**
	 * Dummy here on server, but needed for SOAP stub
	 */
	public void setSchedulingPeriodExternalNaturalKey(
			SchedulingPeriodExternalNaturalKey schedulingPeriodExternalSourceId) {
		// do nothing
	}

	public void addSchedulingPeriodServerExceptionMessage(String exception) {
		getMessages().add(
				String.format("Exception is %s. The server could not persist scheduling period %s.", exception,
						getExternalIdSummary()));
	}

	public void addExternalNaturalKeyMissingMessage(String organizationName) {
		getMessages().add(
				String.format(
						"External Key was missing for scheduling period linked to organization with name %s.",
						organizationName));
	}

	public void addOrganizationNamesMissingMessage() {
		getMessages().add(
				String.format("For scheduling period %s, not even one organization names was provided.",
						getExternalIdSummary()));
	}

	public void addNoSuchOrganizatonNameMessage(String organizationName) {
		getMessages().add(
				String.format("For scheduling period %s, the organization specified to have name %s does not exist.",
						getExternalIdSummary(), organizationName));
	}

	
	
	public void addCampaignExternalSourceIdInKeyMissingMessage() {
		getMessages()
				.add(String
						.format("Scheduling Period with incomplete external key %s missing the required property campaign external id.",
								getExternalIdSummary()));
	}

	public void addNoSuchCampaignExternalIdMessage() {
		getMessages().add(
				String.format("In scheduling period %s, the campaign specified to have external id %s does not exist.",
						getExternalIdSummary(), getEntityExternalSourceId().getCampaignExternalSourceId()));
	}

	public void addStartDateInKeyMissingMessage() {
		getMessages().add("External Key is missing the required property scheduling period start date.");

	}

	public void addStartDateInKeyNotOnCampaignWeekStartMessage(Calendar schedulingPeriodStartDate, int campaignWeekStart) {
		getMessages().add(
				String.format("Scheduling period %s has start date %tc that is not on campaign week start %d.",
						getExternalIdSummary(), schedulingPeriodStartDate, campaignWeekStart));

	}

	public void addSchedulingPeriodQueueServiceGoalTypeIsNotSupported(String serviceGoalType) {
		getMessages().add(
				String.format("The scheduling period %s queue service goal type %s is not supported.",
						getExternalIdSummary(), serviceGoalType));
	}

	public void addSchedulingPeriodQueueServiceGoalTypeIsMissing(String queueExternalSourceId) {
		getMessages().add(
				String.format(
						"The scheduling period %s queue service goal type is missing for queue with external id %s.",
						getExternalIdSummary(), queueExternalSourceId));
	}

	public void addSchedulingPeriodQueueServiceLevelAnswerTimeMissing(String queueExternalSourceId) {
		getMessages()
				.add(String
						.format("The scheduling period %s queue service goal is specified as answer time yet the targeted answer time is missing for queue with external id %s.",
								getExternalIdSummary(), queueExternalSourceId));
	}

	public void addSchedulingPeriodQueueServiceLevelAbandonmentTimeMissingWhenGoalTypeIsAnswerTime(
			String queueExternalSourceId) {
		getMessages()
				.add(String
						.format("The scheduling period %s queue service goal is specified as answer time yet the abandonment time is missing for queue with external id %s.",
								getExternalIdSummary(), queueExternalSourceId));
	}

	public void addQueueExternalSourceIdMissingMessage() {
		
		getMessages().add(
				String.format("In scheduling period %s, the collection of scheduling period queues had an element for which queue external source id was missing.",
						getExternalIdSummary()));
	}

	public void addNoSuchQueueExternalIdMessage(String queueExternalSourceId) {
		getMessages().add(
				String.format("In scheduling period %s, the queue specified to have external id %s does not exist.",
						getExternalIdSummary(), queueExternalSourceId));
	}

	public void addNoSuchSkillExternalIdMessage(String skillExternalSourceId) {
		getMessages().add(
				String.format("In scheduling period %s, the skill specified to have external id %s does not exist.",
						getExternalIdSummary(), skillExternalSourceId));
	}

	public void addNoSuchServiceGoalType(String serviceGoalType) {
		getMessages().add(
				String.format("In scheduling period %s, the serviceGoalType %s is not known.", getExternalIdSummary(),
						serviceGoalType));
	}

	public void addTimePeriodOverlap(ID campaignID) {
		getMessages().add(getTimePeriodOverlap(campaignID));
	}

	public String getTimePeriodOverlap(ID campaignID) {
		return String.format(
				"In scheduling period %s, there was overlap with another scheduling period in same campaign with WFO id %s.",
				getExternalIdSummary(),
				campaignID);
	}

	public void addTimePeriodOverlap() {
		getMessages()
				.add(String
						.format("In scheduling period %s, there was overlap with another scheduling period in same campaign with unknown WFO id.",
								getExternalIdSummary()));
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public List<String> getMessages() {
		return super.getMessages();
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public void setMessages(List<String> messages) {
		// Do nothing
	}
}
