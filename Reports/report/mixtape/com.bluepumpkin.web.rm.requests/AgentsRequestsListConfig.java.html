package com.bluepumpkin.web.rm.requests;

import java.util.*;
import com.witness.web.uif.listconfig.*;
import com.witness.web.uif.l10n.*;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.model.Request;

/**
 * Title:        AgentsRequestsListConfig
 * Description:  List Configuration for Agents Requests
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, dsu
 * @version      1.0
 *
 * Created on October 8, 2002, 10:21 AM
 */
public class AgentsRequestsListConfig extends ListConfig {
  private List m_allFPs;
  private Map m_fpMap;

  /** Creates a new instance of AgentsRequestsListConfig */
  public AgentsRequestsListConfig() {
    initFPs();
  }

	/**
	 * Initialize Field Properties
	 */
  private void initFPs() {
    if (m_allFPs==null) {
			m_allFPs = new ArrayList(8);
			m_allFPs.add( createFP(Request.SORT_LAST_NAME_FIRST, UIFWebBundleKey.NAME, true) );
            m_allFPs.add( createFP(Request.SORT_STATUS, UIFWebBundleKey.STATUS, true) );
			m_allFPs.add( createFP(Request.SORT_TYPE, UIFWebBundleKey.TYPE, true) );
			m_allFPs.add( createFP(Request.SORT_CREATED, UIFWebBundleKey.SUBMITTED, true) );
			m_allFPs.add( createFP(Request.SORT_MODIFIED, UIFWebBundleKey.LAST_MODIFIED, true) );
            m_allFPs.add( createFP(Request.SORT_TO_SENIORITY, UIFWebBundleKey.SENIORITY, true) );
            m_allFPs.add( createFP(Request.SORT_EMP_RANK, UIFWebBundleKey.RANK, true) );
			addCommonFPs(m_allFPs);
		}

    if (m_fpMap==null) m_fpMap = createFPMap(m_allFPs);
  }

	/**
	 * Add Common Field Properties
	 */
	protected static void addCommonFPs(List list) {
		list.add( createFP(UIFWebBundleKey.EXPIRED, UIFWebBundleKey.BUNDLE_NAME, false) );
		list.add( createFP(UIFWebBundleKey.COMMENTS, UIFWebBundleKey.BUNDLE_NAME, false) );
		list.add( createFP(UIFWebBundleKey.ALERTS, UIFWebBundleKey.BUNDLE_NAME, false) );
		list.add( createFP(UIFWebBundleKey.ACTIONS, UIFWebBundleKey.BUNDLE_NAME, false) );
	}

  /**
   * Convenient method to create ListConfigFP which uses Core Web Bundle
   *
	 * @param fieldName The column identified
   * @param rbKey The Resource Bundle Key
   * @param isSortable specify whether Field is sortable
   */
  protected static ListConfigFP createFP(int fieldName, String rbKey, boolean isSortable) {
    return createFP(fieldName, rbKey, UIFWebBundleKey.BUNDLE_NAME, isSortable);
  }

	/**
	 * Convenient method to create ListConfigFP which uses Core Web Bundle
	 *
	 * @param fieldName The column identified
	 * @param rbKey The Resource Bundle Key
	 * @param isSortable specify whether Field is sortable
	 */
	protected static ListConfigFP createFPforRM(int fieldName, String rbKey, boolean isSortable) {
		return createFP(fieldName, rbKey, RmWebBundleKey.BUNDLE_NAME, isSortable);
	}

  /**
   * Return All List Config Field Properties (ListConfigFP)
   */
  public List getAllFPs() {
    return m_allFPs;
  }

  /**
   * Return a Map of All List Config Field Properties (ListConfigFP)
   * where key is ListConfigFP.getFieldName();
   */
  public Map getFPMap() {
    return m_fpMap;
  }
}
