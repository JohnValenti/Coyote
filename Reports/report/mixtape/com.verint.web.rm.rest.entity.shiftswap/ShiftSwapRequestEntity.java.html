package com.verint.web.rm.rest.entity.shiftswap;

import java.util.ArrayList;
import java.util.List;

import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.ejb.rm.requests.common.model.Request;
import com.verint.web.rm.rest.entity.RequestEntity;

/**
 * Represents a Shift Swap request.
 */
@XmlRootElement(name = "request")
public class ShiftSwapRequestEntity extends RequestEntity {

	public static final String TYPE = Request.REQUESTTYPE_SHIFTSWAP;

	private final List<ShiftSwapItemEntity> items = new ArrayList<ShiftSwapItemEntity>();
	private String negotiationComment;
	private String swapType;
	private final List<String> warnings = new ArrayList<String>();

	public ShiftSwapRequestEntity() {
		super(TYPE);
	}

	public List<ShiftSwapItemEntity> getItems() {
		return items;
	}

	public String getNegotiationComment() {
		return negotiationComment;
	}

	public void setNegotiationComment(String negotiationComment) {
		this.negotiationComment = negotiationComment;
	}

	public String getSwapType() {
		return swapType;
	}

	public void setSwapType(String swapType) {
		this.swapType = swapType;
	}

	public List<String> getWarnings() {
		return warnings;
	}
}
