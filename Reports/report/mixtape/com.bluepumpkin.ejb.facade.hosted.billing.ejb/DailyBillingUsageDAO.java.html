	/**
	 * Initiates the daily dump for the BPMAINDB.HOSTEDBILLINGDAILYUSERS and BPMAINDB.HOSTEDBILLINGDAILYENCRYPT tables. These tables
	 * are used in calculating the billing for tenant system users.
	 * <p>
	 * @param startDate a start date/time to use in dumping data, typically 12:00 AM for today's date
	 * @param endDate an end date/time to use in dumping data, typically 11:59 PM for today's date
	 * @param tz the timezone for the system performing the dump
	 * <p>
	 * @throws FacadeException
	 */
	public void dumpDailyUsage(Date startDate, Date endDate, TimeZone tz)
			throws FacadeException {
		// Use input dates for querying
		m_cat.info("dumpDailyUsage");
		m_cat.info("startDate: " + startDate);
		m_cat.info("endDate: " + endDate);
		m_cat.info("timezone: " + tz);
		String strStartDate = JdmoUtil.formatDBString(startDate);
		String strEndDate = JdmoUtil.formatDBString(endDate);

		firstOfTheMonthDate = getFirstOfTheMonthDate(endDate);
		String strFirstOfTheMonthDate = JdmoUtil.formatDBString(firstOfTheMonthDate);
		m_cat.info("strFirstOfTheMonthDate: " + strFirstOfTheMonthDate);

		dumpDailyBillingPrivilegeUsers(strStartDate, strEndDate, strFirstOfTheMonthDate);
		dumpDailyBillingWFMConsumptionUsers(strStartDate, strEndDate, strFirstOfTheMonthDate);
		dumpDailyBillingEncryptionSet(strFirstOfTheMonthDate);
	}

	public Date getFirstOfTheMonthDate(Date endDate)
			throws FacadeException {
		WorkResourceManager mgr = null;
		TimeZone yourCompanyTZ = null;
		try {
			mgr = BbmManagerFactory.getWorkResourceManager();

			Organization yourCompanyOrg = mgr.getOrganizationByID(Organization.YOUR_COMPANY_ID_OBJ);
			yourCompanyTZ = yourCompanyOrg.getTimeZone();
		} catch (Exception e) {
			throw new FacadeException(e);
		}

		Calendar cal = Calendar.getInstance(yourCompanyTZ);
		// using the end time since that value will hold today's date (modified for timezone) and the start time will hold a value
		// of (end time - 24 hours)
		cal.setTime(endDate);
		// setting day of month to 1, so all values entered into the hosted billing daily users table will have the same day value
		// this is done for performance and to snap the report to month ranges
		LocalDate ld = new LocalDate(cal.get(Calendar.YEAR), cal.get(Calendar.MONTH), 1, 0, 0);
		return ld.getTime(TimeZone.getTimeZone("GMT"));
	}

	/**
	 * This method will dump the required data to the db for users with the permissions to components associated with hosted site billing.
	 * <p>
	 * @param strStartDate the start time frame for which to run the dump, typically set by the integration server when the dump is
	 * scheduled to run
	 * @param strEndDate the end time frame for which to run the dump, typically set by the integration server when the dump is scheduled
	 * to run
	 * @param strFirstOfTheMonthDate the date for the first day of the current month
	 * <p>
	 * @throws FacadeException
	 */
	protected void dumpDailyBillingPrivilegeUsers(String strStartDate, String strEndDate, String strFirstOfTheMonthDate)
			throws FacadeException {
		StringBuilder buf = new StringBuilder(STRINGBUILDERINITIALCAPACITY);

		buf.append("SELECT DISTINCT org.ID orgID, billingSource.ID billingSourceID, employee.ID empID").
				append(" FROM ORGANIZATION org, HOSTEDBILLINGSOURCE billingSource, AUTHPRIVILEGE authPriv, EMPLOYEEAM employee,").
				append("AUTHROLEPRIVILEGE authRolePriv, AUTHACCESSRIGHT authAccRights, WORKRESOURCEORGANIZATION workResOrg, BPUSER bpuser").
				append(" WHERE workResOrg.ORGANIZATIONID = org.ID").
				append(" AND workResOrg.STARTTIME<").append(JdmoUtil.asSqlLiteral(strEndDate)).
				append(" AND isnull(workResOrg.ENDTIME,").append(JdmoUtil.asSqlLiteral(ISNULLENDDATE)).append(")>=").
				append(JdmoUtil.asSqlLiteral(strStartDate)).
				append(" AND workResOrg.WORKRESOURCEID = employee.ID").
				append(" AND workResOrg.WORKRESOURCEID = bpuser.EMPLOYEEID").
				append(" AND bpuser.ID = authAccRights.BPUSERID").
				append(" AND authAccRights.AUTHROLEID = authRolePriv.AUTHROLEID").
				append(" AND authRolePriv.AUTHPRIVILEGEID = authPriv.ID").
				append(" AND authPriv.ID = billingSource.SOURCEID").
				append(" ORDER BY org.ID");

		try {
			Map insertHashMap = new HashMap();
			JdmoRowset rs = m_dmo.createRowset(buf.toString());
			while (rs.next()) {
				// if the values aren't already in the table, add them
				if (!isDuplicateHOSTEDBILLINGDAILYUSERS(rs, strFirstOfTheMonthDate)) {
					insertHashMap.put("DAYDATE", firstOfTheMonthDate);
					insertHashMap.put("ORGANIZATIONID", rs.getID(1));
					insertHashMap.put("HOSTEDBILLINGSOURCEID", rs.getInt(2));
					insertHashMap.put("EMPLOYEEID", rs.getInt(3));

					m_dmo.addBatchInsert("HOSTEDBILLINGDAILYUSERS", (HashMap)insertHashMap);
				}
				insertHashMap.clear();
			}
			m_dmo.executeBatch();
		} catch (Exception e) {
			throw new FacadeException(e);
		}
	}

	/**
	 * This method will dump the required data to the db for users that have been consumed by components associated with hosted site
	 * billing.
	 * <p>
	 * @param strStartDate the start time frame for which to run the dump, typically set by the integration server when the dump is
	 * scheduled to run
	 * @param strEndDate the end time frame for which to run the dump, typically set by the integration server when the dump is scheduled
	 * to run
	 * @param strFirstOfTheMonthDate the days date for which the report is run
	 * <p>
	 * @throws FacadeException
	 */
	protected void dumpDailyBillingWFMConsumptionUsers(String strStartDate, String strEndDate, String strFirstOfTheMonthDate)
			throws FacadeException {
		StringBuilder buf = new StringBuilder(STRINGBUILDERINITIALCAPACITY);

		buf.append("SELECT DISTINCT org.ID orgID, " + WFMCONSUMPTIONSOURCEID + " billingSourceID, employee.ID empID").
				append(" FROM ORGANIZATION org, PLANNEDEVENTTIMELINE plannedEventTL, EMPLOYEEAM employee,").
				append(" WORKRESOURCEORGANIZATION workResOrg").
				append(" WHERE workResOrg.ORGANIZATIONID = org.ID").
				append(" AND workResOrg.STARTTIME<").append(JdmoUtil.asSqlLiteral(strEndDate)).
				append(" AND isnull(workResOrg.ENDTIME,").append(JdmoUtil.asSqlLiteral(ISNULLENDDATE)).
				append(")>=").append(JdmoUtil.asSqlLiteral(strStartDate)).
				append(" AND workResOrg.WORKRESOURCEID = plannedEventTL.WORKRESOURCEID").
				append(" AND plannedEventTL.STARTTIME >=").append(JdmoUtil.asSqlLiteral(strFirstOfTheMonthDate)).
				append(" AND plannedEventTL.STARTTIME <").append(JdmoUtil.asSqlLiteral(strEndDate)).
				append(" AND workResOrg.WORKRESOURCEID = employee.ID").
				append(" ORDER BY org.ID");

		try {
			Map insertHashMap = new HashMap();
			JdmoRowset rs = m_dmo.createRowset(buf.toString());
			while (rs.next()) {
				// if the values aren't already in the table, add them
				if (!isDuplicateHOSTEDBILLINGDAILYUSERS(rs, strFirstOfTheMonthDate)) {
					insertHashMap.put("DAYDATE", firstOfTheMonthDate);
					insertHashMap.put("ORGANIZATIONID", rs.getID(1));
					insertHashMap.put("HOSTEDBILLINGSOURCEID", rs.getInt(2));
					insertHashMap.put("EMPLOYEEID", rs.getInt(3));

					m_dmo.addBatchInsert("HOSTEDBILLINGDAILYUSERS", (HashMap)insertHashMap);
				}
				insertHashMap.clear();
			}
			m_dmo.executeBatch();
		} catch (Exception e) {
			throw new FacadeException(e);
		}
	}

	/**
	 * This method will dump the required data to the db associated with the system's security setting "Data-At-Rest Encryption",
	 * currently found at System Management -> Security: Data-At-Rest Encryption.
	 * <p>
	 * @param strFirstOfTheMonthDate the date for which to dump data, typically set by the integration server when the dump is scheduled to run
	 * <p>
	 * @throws FacadeException
	 */
	protected void dumpDailyBillingEncryptionSet(String strFirstOfTheMonthDate)
			throws FacadeException {
		try {
			// delete any tables entries for this date
			deleteFromHostedBillingDailyEncryptWhereDaydateIs(strFirstOfTheMonthDate);
			// create and execute a batch insert
			exectuteBatchForHostedBillingDailyEncrypt();
		} catch (Exception e) {
			throw new FacadeException(e);
		}
	}

	/**
	 * Deletes any entries from the BPMAINDB.HOSTEDBILLINGDAILYENCRYPT table for the date provided: this date would typically be for today's
	 * date.
	 * <p>
	 * @param strFirstOfTheMonthDate the date for which to delete entries in the HOSTEDBILLINGDAILYENCRYPT table
	 * <p>
	 * @throws JdmoException
	 */
	private void deleteFromHostedBillingDailyEncryptWhereDaydateIs(String strFirstOfTheMonthDate)
			throws JdmoException {
		StringBuilder strQuery = new StringBuilder();
		strQuery.append("delete from HOSTEDBILLINGDAILYENCRYPT where DAYDATE=").append(JdmoUtil.asSqlLiteral(strFirstOfTheMonthDate));
		m_dmo.execute(strQuery.toString());
	}

	/**
	 * Executes a batch insert to the BPMAINDB.HOSTEDBILLINGDAILYENCRYPT table using a globally defined date: this date would typically be
	 * for today's date.
	 * <p>
	 * @throws JdmoException
	 * @throws XPathExpressionException
	 */
	private void exectuteBatchForHostedBillingDailyEncrypt()
			throws JdmoException, XPathExpressionException {
		Map insertHashMap = new HashMap();
		insertHashMap.put("DAYDATE", firstOfTheMonthDate);
		insertHashMap.put("ENCRYPTIONSET", getAtRestEncryption() ? 1 : 0);
		m_dmo.addBatchInsert("HOSTEDBILLINGDAILYENCRYPT", (HashMap)insertHashMap);
		m_dmo.executeBatch();
	}

	/**
	 * For 11.2, the value should be retrieved via web service, after EM team creates the web service to retrieve the value.
	 * <p>
	 * @param partialXml
	 * <p>
	 * @return true if the ATREST_ENCRYPTION value is true, false otherwise
	 * <p>
	 * @throws JdmoException
	 * @throws XPathExpressionException
	 */
	private boolean getAtRestEncryption()
			throws JdmoException, XPathExpressionException {
		String strSelect = "select SECTIONSETTINGSXML from INSTALLDATAXML where INSTALLHIERARCHYID = -442001 and"
				+ " INSTALLSCREENMETADATAID = -6003";
		JdmoRowset rs = m_dmo.createRowset(strSelect);
		rs.next();
		String partialXml = rs.getString(1);

		StringBuilder buf = new StringBuilder("<root>");
		buf.append(partialXml).append("</root>");

		InputSource inputSource = new InputSource(new StringReader(buf.toString()));
		XPathFactory factory = XPathFactory.newInstance();
		XPath xpath = factory.newXPath();

		String value = xpath.evaluate("//Setting[@Name=\"ATREST_ENCRYPTION\"]/@Value", inputSource);
		return Boolean.valueOf(value);
	}

	private boolean isDuplicateHOSTEDBILLINGDAILYUSERS(JdmoRowset rs, String strFirstOfTheMonthDate)
			throws FacadeException {
		StringBuilder buf = new StringBuilder(STRINGBUILDERINITIALCAPACITY);
		try {
			buf.append("SELECT * FROM HOSTEDBILLINGDAILYUSERS").
					append(" WHERE DATEPART(yyyy, DAYDATE) = DATEPART(yyyy, ").append(JdmoUtil.asSqlLiteral(strFirstOfTheMonthDate)).append(")").
					append(" AND DATEPART(mm, DAYDATE) = DATEPART(mm, ").append(JdmoUtil.asSqlLiteral(strFirstOfTheMonthDate)).append(")").
					append(" AND ORGANIZATIONID = ").append(JdmoUtil.asSqlLiteral(rs.getID(1))).
					append(" AND HOSTEDBILLINGSOURCEID = ").append(JdmoUtil.asSqlLiteral(rs.getInt(2))).
					append(" AND EMPLOYEEID = ").append(JdmoUtil.asSqlLiteral(rs.getInt(3)));
			JdmoRowset rs2 = m_dmo.createRowset(buf.toString());
			return rs2.next();
		} catch (Exception e) {
			throw new FacadeException(e);
		}
	}
