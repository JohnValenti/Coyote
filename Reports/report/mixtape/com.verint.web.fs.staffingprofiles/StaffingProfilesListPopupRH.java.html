package com.verint.web.fs.staffingprofiles;

import java.util.Collection;

import com.bluepumpkin.ejb.bbm.campaign.ejb.CampaignManager;
import com.bluepumpkin.ejb.bbm.campaign.model.SPEmpTemplate;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

public class StaffingProfilesListPopupRH extends RequestHandler {
	@Override
	public void processRequest(RequestContext context) throws Exception {
		if (isAction(context, PageAction.ADD)) {
			StaffingProfilesListPopupPM pm = StaffingProfilesListPopupPM.getInstance(context);
			pm.setShouldCloseOnLoad(true);
			pm.extractParameters(context.getRequest());
			processSave(context, pm);
		} else {
			StaffingProfilesListPopupPM pm = new StaffingProfilesListPopupPM(context);
			pm.extractParameters(context.getRequest());
			context.setPageModel(pm);
		}
	}

	void processSave(RequestContext context, StaffingProfilesListPopupPM pm) throws Exception {
		Collection<SPEmpTemplate> colSPEmpTemplates = pm.getSPEmpTemplates(context);

		// Fixed for ESR ESR# #4339140 - Unable to add staffing profile to SP in
		// WHAT-IF mode (web F&S) issue details.
		CampaignManager campaignMgr = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
		for (SPEmpTemplate colspEmpTemplate : colSPEmpTemplates) {
			campaignMgr.createSPEmpTemplate(colspEmpTemplate);
		}
	}
}
