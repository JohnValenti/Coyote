package com.bluepumpkin.web.bbm.workload.queue.selection.queueattribute.filter;

import com.bluepumpkin.common.util.StringUtil;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.util.js.JSUtil;

/**

 * Copyright (c) 2003 Company: Blue Pumpkin Software, Inc
 * 
 * @author Sankar Nair
 * @version 1.0
 */
public class QueueAttributeFilterUtil {
	protected static final String NAMED_FILTER_NAME = "QueueAttributeFilterPicker";
	public static final String NAMED_FILTER_RH = "queue_attribute_filter_config";

	public static final String QUEUEATTRIBUTEFILTERIDFROMREPORTPOPUP = "QUEUEATTRIBUTEFILTERIDFROMREPORTPOPUP";
	public static final String ACTIVE_WIT_NAMEDF_FILTER_ID = "ACTIVE_WIT_NAMED_FILTER_ID";
	public static final String ACTIVE_NAMEDFILTER_DUMMY_ID = "-1";
	public static final String SAVED_NAMED_FILTER_INSTANCE_ID = "SAVED_NAMED_FILTER_INSTANCE_ID";

	public static final String WITS_REPORT_DATA_FETCHER_GCR_NAME = "OPERATIONS_WITS_REPORT_COMPNENTS_DATA_FETCHER";

	private QueueAttributeFilterUtil() {
	}

	public static void addPopupArgs(QueueAttributeFilterPickerPC pc, boolean isEditEnabled) {
		String queueAttributeId = "";
		if (pc.getQueueAttributeID() != null) {
			queueAttributeId = JSUtil.escapeStringLiteral(pc.getQueueAttributeID().toString());
		}

		String sExtraAttr = QueueAttributeFilterPageModel.QUEUE_ATTRIB_FILTER_ID_FN + "=" + queueAttributeId;
		if (StringUtil.isEmptyOrWhiteSpace(pc.getSelectedFilterName()) || pc.getQueueAttributeID() == null
				|| "0".equals(pc.getSelectedFilterName())) {
			sExtraAttr = "";
		}
		String sWindowArg = "750,700,ctr,ctr";

		pc.addPopupArgs("POPUP" + UIFWebBundleKey.FILTER_CREATE_NAMED_FILTER, QueueAttributeFilterUtil.NAMED_FILTER_RH, "",
				null, sWindowArg, true, 1);
		if (isEditEnabled) {
			pc.addPopupArgs("POPUP" + UIFWebBundleKey.FILTER_EDIT_NAMED_FILTER, QueueAttributeFilterUtil.NAMED_FILTER_RH,
					sExtraAttr, QUEUEATTRIBUTEFILTERIDFROMREPORTPOPUP, sWindowArg, true, 1);
		}
	}
}
