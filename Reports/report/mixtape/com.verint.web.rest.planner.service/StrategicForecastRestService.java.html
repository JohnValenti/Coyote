package com.verint.web.rest.planner.service;

import java.util.*;

import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.*;
import javax.ws.rs.core.Context;
import javax.ws.rs.core.Response;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.fs.Log;
import com.verint.web.rest.planner.entity.StrategicForecastEntity;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.SystemWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.planner.facade.IStrategicForecastFacade;
import com.verint.web.rest.security.WFOSecurity;
import com.verint.web.rest.service.ScheduleEventsRestService;
import com.verint.web.rest.util.FormatUtils;
import com.witness.web.uif.system.RequestContext;
import com.verint.web.rest.service.AbstractWFOEntityService;

@Path("/strategicforecast/")
@Produces("application/json")
@WFOSecurity
public class StrategicForecastRestService extends AbstractWFOEntityService<StrategicForecastEntity> {
	private static final Category logger = Log.initCategory(ScheduleEventsRestService.class.getName());
	private IStrategicForecastFacade strategicForecastFacade;
	@Context
	private HttpServletRequest request;
	@Context
	private HttpServletResponse response;
	@Context
	private ServletContext sc;

	public StrategicForecastRestService() {
	}

	public IStrategicForecastFacade getStrategicForecastFacade() {
		return strategicForecastFacade;
	}

	public void setStrategicForecastFacade(IStrategicForecastFacade strategicForecastFacade) {
		this.strategicForecastFacade = strategicForecastFacade;
	}

	public RequestContext getRequestContext() {
		return new RequestContext(request, response, sc);
	}

	@DELETE
	@Path("/{queueid}/{startDate}/{endDate}")
	public void deleteStrategicForecast(@PathParam("queueid") String queueid,
			@PathParam("startDate") String strStartTime,
			@PathParam("endDate") String strEndTime) throws WFOException {
		try {

			ID queueId = queueid == null ? null : new ID(queueid);

			Date startTime = null;
			Date endTime = null;
			startTime = FormatUtils.iso8601SecPrecisionStringToDate(strStartTime);
			endTime = FormatUtils.iso8601SecPrecisionStringToDate(strEndTime);

			if (startTime == null || endTime == null) {
				throw new BusinessWFOException("date is null", Response.Status.CONFLICT);
			}

			RequestContext context = getRequestContext();
			strategicForecastFacade.deleteStrategicForecasts(context, queueId, startTime, endTime);
		} catch (WFOException ex) {
			throw ex; // should already have been logged
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	@POST
	@Consumes("application/json")
	public void saveStrategicForecast(List<StrategicForecastEntity> strategicForecastEntities) throws WFOException {
		RequestContext context = getRequestContext();
		strategicForecastFacade.createStrategicForecasts(context, strategicForecastEntities);
	}

	@Override
	public StrategicForecastEntity getEntity(@PathParam("id") String id) throws WFOException {
		return new StrategicForecastEntity();
	}

	@Override
	public StrategicForecastEntity updateEntity(StrategicForecastEntity employee) throws WFOException {
		return new StrategicForecastEntity();
	}

	@Override
	public StrategicForecastEntity addEntity(StrategicForecastEntity employee) throws WFOException {
		return new StrategicForecastEntity();
	}

	@Override
	public StrategicForecastEntity deleteEntity(@PathParam("id") String id) throws WFOException {
		return new StrategicForecastEntity();
	}

	@Override
	public boolean addEntities(Collection<StrategicForecastEntity> entities) throws WFOException {
		return true;
	}

	@Override
	public boolean updateEntities(Collection<StrategicForecastEntity> entities) throws WFOException {
		return true;
	}

	@Override
	public boolean deleteEntities(Collection<StrategicForecastEntity> entities) throws WFOException {
		return true;
	}

	@Override
	public List<StrategicForecastEntity> getAllEntities() throws WFOException {
		return Collections.emptyList();
	}
}
