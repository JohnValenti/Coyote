package com.verint.web.rest.planner.service;


import java.util.*;
import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.*;
import javax.ws.rs.core.Context;
import javax.ws.rs.core.Response;

import com.verint.web.rest.planner.entity.TimebankEntity;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.planner.facade.ITimebankFacade;
import com.verint.web.rest.security.WFOSecurity;
import com.verint.web.rest.service.AbstractWFOEntityService;
import com.witness.web.uif.system.RequestContext;


@Path("/timebanks/")
@Produces("application/json")
@WFOSecurity
public class TimebankRestService extends AbstractWFOEntityService<TimebankEntity> {

	private ITimebankFacade timebankFacade;
	@Context
	private HttpServletRequest request;
	@Context
	private HttpServletResponse response;
	@Context
	private ServletContext sc;

	public TimebankRestService() {
	}

	public ITimebankFacade getTimebankFacade() {
		return timebankFacade;
	}

	public void setTimebankFacade(ITimebankFacade timebankFacade) {
		this.timebankFacade = timebankFacade;
	}

	public RequestContext getRequestContext() {
		return new RequestContext(request, response, sc);
	}

	@GET
	@Path("/exists/{name}")
	public Boolean checkIfTimebankExists(@PathParam("name") String timebankName) throws WFOException {
		if (timebankName == null || timebankName.isEmpty()) {
			throw new BusinessWFOException("timebank name is null", Response.Status.CONFLICT);
		}
		RequestContext context = getRequestContext();
		return timebankFacade.checkIfTimebankExists(context, timebankName);
	}

	@DELETE
	@Path("/{name}")
	public void deleteTimebank(@PathParam("name") String timebankName) throws WFOException {
		if (timebankName == null || timebankName.isEmpty()) {
			throw new BusinessWFOException("timebank name is null", Response.Status.CONFLICT);
		}
		RequestContext context = getRequestContext();
		timebankFacade.deleteTimebank(context, timebankName);
	}

	@POST
	@Consumes("application/json")
	public void saveTimbankData(List<TimebankEntity> timebankEntities) throws WFOException {
		if (timebankEntities == null || timebankEntities.isEmpty()) {
			throw new BusinessWFOException("timebank data is null", Response.Status.CONFLICT);
		}
		RequestContext context = getRequestContext();
		timebankFacade.saveTimebankData(context, timebankEntities);
	}
