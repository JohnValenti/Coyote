package com.bluepumpkin.ejb.am.payrollmanager.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
/**
 * Title:        Blue Pumpkin Software Foundation
 * Description:  DataSourcePayrollProvider
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */

public class DataSourcePayrollProvider implements Serializable
{
	private List m_Config;
	private Collection m_Mapping;
	private ID m_DSPPID;
	private ID m_DataSourceID;
	private String m_DataSourceName;
	private ID m_PayrollProviderID;
	private String m_PayrollProviderName;
	private String m_ExportDestination;
	// A cache to cache the PayrollProvider's FILELOCATION
	private String m_ConfigFile;
	// used to capture the PayrollExportConfig version
	private String m_Version;
	// A cache holder for DataSource Description
	private String m_Description;
	public DataSourcePayrollProvider() {
		super();
	}
	// Constructor used by the backend for loading
	public DataSourcePayrollProvider(ID id,
									 ID dataSourceID,
									 ID payrollProviderID,
									 String exportDestination,
									 String version) {
		m_DSPPID = id;
		m_DataSourceID = dataSourceID;
		m_PayrollProviderID = payrollProviderID;
		m_ExportDestination = exportDestination;
		m_Version = version;
	}

	/** Constructor used by the frontend for save
	 *  @deprecated
	 */
	public DataSourcePayrollProvider(String name,
									 ID payrollProviderID,
									 String exportDestination) {
		m_DataSourceName = name;
		m_PayrollProviderID = payrollProviderID;
		m_ExportDestination = exportDestination;
	}

	// Constructor used by the frontend for save
	public DataSourcePayrollProvider(String name,
									 ID payrollProviderID,
									 String exportDestination,
									 String description) {
		m_DataSourceName = name;
		m_PayrollProviderID = payrollProviderID;
		m_ExportDestination = exportDestination;
		m_Description = description;
	}

	public ID getID() { return m_DSPPID; }
	public void setID(ID id) { m_DSPPID = id; m_DataSourceID = id;}

	public ID getDataSourceID() { return m_DataSourceID; }
	public ID getPayrollProviderID() { return m_PayrollProviderID; }
	public void setPayrollProviderID(ID id) { m_PayrollProviderID = id; }
	public String getPayrollProviderName() { return m_PayrollProviderName; }
	public void setPayrollProviderName(String name) { m_PayrollProviderName = name; }
	public String getVersion() { return m_Version; }
	public void setVersion(String version) { m_Version = version; }
	public String getExportDestination() { return m_ExportDestination; }
	public void setExportDestination(String destination) {
		m_ExportDestination = destination;
	}

	// Internal Cache for payroll config
	public List getConfig() { return m_Config; }
	public void setConfig(List config) { m_Config = config; }

	// Internal Cache for EarningType mapping
	public Collection getMapping() { return m_Mapping; }
	public void setMapping(Collection map) { m_Mapping = map; }

	public String getDataSourceName() { return m_DataSourceName; }
	public void setDataSourceName(String name) { m_DataSourceName = name; }

	// Invoked in EJB to get the XML config
	public String getConfigLocation() { return m_ConfigFile; }
	// Invoked in UI to cache the value
	public void SetConfigLocation(String location) { m_ConfigFile = location; }

	public String getDescription() { return m_Description; }
	public void setDescription(String desc) { m_Description = desc; }
}
