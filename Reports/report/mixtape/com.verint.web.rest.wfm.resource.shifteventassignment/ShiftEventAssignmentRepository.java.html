@JsonApiResourceRepository(ShiftEventAssignmentEntity.class)
public class ShiftEventAssignmentRepository extends WfoResourceRepository<Integer, ShiftEventAssignmentEntity> {

	private final static Logger log = LoggerFactory.getLogger(ShiftEventAssignmentRepository.class.getName());
	private final static String ENTITY_NAME = ShiftEventAssignmentEntity.class.getSimpleName();
	public final static String RESOURCE_NAME = "shifteventassignments";

	@Inject
	private ScheduleAccessManager sam;

	@Inject
	private EmployeeFilter ef;

	@Override
	protected Logger logger() {
		return log;
	}

	@Override
	@RepositoryLogger
	public <T extends ShiftEventAssignmentEntity> T save(T entity, ContainerRequestContext requestContext) {
		String entityID = entity.getID() == null ? "" : Integer.toString(entity.getID());
		try {
			ShiftEventAssignment sea = ShiftEventAssignmentMapper.INSTANCE.entityToSea(entity);
			if (sea.getID() == null) {
				ID newID = sam.createShiftEventAssignment(sea);
				entity.setID(newID.toInt());
			} else {
				sam.updateShiftEventAssignment(sea);
			}
			return entity;
		} catch (BbmCreateException e) {
			throw handleBbmCreateException(entity, e);
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, Integer.toString(entity.getID()), e);
		} catch (BbmUpdateException e) {
			throw handleBbmUpdateException(entity, e);
//		}catch (BbmScheduleConflictException e) {
//			log.error("ShiftAssignmentRepository.save: failed to save ShiftAssignment", e);
//			throw new EntityScheduleConflictException(getLocalizer(requestContext), getUserTimeZone(requestContext), e.getConflicts());
		} catch (MultiUserException e) {
			throw handleMultiUserException(ENTITY_NAME, Integer.toString(entity.getID()), e);
		}
//		} catch (BbmFinderException e) {
//			throw handleBbmFinderException(ENTITY_NAME, entityID, e);
//		}
	}

	@Override
	@RepositoryLogger
	public ShiftEventAssignmentEntity findOne(Integer id, QueryParams requestParams, ContainerRequestContext requestContext) {
		try {
			ShiftEventAssignment sea = sam.getShiftEventAssignmentById(ID.fromInt(id));
			return ShiftEventAssignmentMapper.INSTANCE.seaToEntity(sea);
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, Integer.toString(id), e);
		} catch (BbmFinderException e) {
			throw handleBbmFinderException(ENTITY_NAME, Integer.toString(id), e);
		}
	}

	@Override
	@RepositoryLogger
	@ValidateJsonApiFilterParams
	public Iterable<ShiftEventAssignmentEntity> findAll(QueryParams queryParams, ContainerRequestContext requestContext) {
		try {
			DateAndEmployeeIdComboFilterParamHandler filterParamHandler =
					new DateAndEmployeeIdComboFilterParamHandler(ef, getCurrentUser(requestContext), getFilterParams(),
							getValidFilterParamKeys(), RESOURCE_NAME);

			if (filterParamHandler.getEmployeeIds().isEmpty()) {
				return Collections.emptyList();
			}

			//TODO: uncomment when porting to 15.2
			return Collections.emptyList();
//			return ShiftEventAssignmentMapper.INSTANCE.seasToEntities(
//					sam.getAllShiftEventAssignmentsForEmployeesBetweenDates(
//							filterParamHandler.getEmployeeIds(), filterParamHandler.getStartDate(),
//							filterParamHandler.getEndDate()));
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, ALL_IDS, e);
		} catch (BbmFinderException e) {
			throw handleBbmFinderException(ENTITY_NAME, ALL_IDS, e);
		}
	}

	@Override
	@RepositoryLogger
	public Iterable<ShiftEventAssignmentEntity> findAll(Iterable<Integer> ids, QueryParams queryParams, ContainerRequestContext requestContext) {
		try {
			return ShiftEventAssignmentMapper.INSTANCE.seasToEntities(
					sam.getShiftEventAssignmentsByIDs(IdUtil.fromInt(ids)));
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, IdUtil.toDelimitedString(ids), e);
		} catch (BbmFinderException e) {
			throw handleBbmFinderException(ENTITY_NAME, IdUtil.toDelimitedString(ids), e);
		}
	}

	@Override
	@RepositoryLogger
	public void delete(Integer id, ContainerRequestContext requestContext) {
		try {
			sam.deleteShiftEventAssignments(Collections.singleton(ID.fromInt(id)));
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, Integer.toString(id), e);
		} catch (BbmRemoveException e) {
			throw handleBbmRemoveException(ENTITY_NAME, Integer.toString(id), e);
		}
	}

	@Override
	protected String getResourceName() {
		return RESOURCE_NAME;
	}

	@Override
	protected Set<String> getValidFilterParamKeys() {
		return Sets.newHashSet(EMPLOYEE_FILTER_NAME_FILTER_KEY, EMPLOYEE_IDS_FILTER_KEY,
				START_DATE_DEFAULT_FILTER_KEY, END_DATE_DEFAULT_FILTER_KEY);
	}
}
