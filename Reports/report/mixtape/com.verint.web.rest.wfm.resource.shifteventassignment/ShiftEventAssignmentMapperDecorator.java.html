package com.verint.web.rest.wfm.resource.shifteventassignment;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftEventAssignment;
import com.verint.web.rest.wfm.resource.shiftassignment.ShiftAssignmentEntity;

import java.util.ArrayList;
import java.util.Collection;

public abstract class ShiftEventAssignmentMapperDecorator implements ShiftEventAssignmentMapper {
	private final ShiftEventAssignmentMapper delegate;

	public ShiftEventAssignmentMapperDecorator(ShiftEventAssignmentMapper delegate) {
		this.delegate = delegate;
	}

	@Override
	public ShiftEventAssignment entityToSea(ShiftEventAssignmentEntity entity) {
		ShiftEventAssignment sea = delegate.entityToSea(entity);
		sea.addWorkResourceID(ID.fromInt(entity.getWorkResourceID()));
		sea.setParentID(ID.fromInt(entity.getShiftAssignmentID()));
		return sea;
	}

	@Override
	public ShiftEventAssignmentEntity seaToEntity(ShiftEventAssignment sea) {
		ShiftEventAssignmentEntity entity = delegate.seaToEntity(sea);
		if (sea.getWorkResourceIDs() != null && !sea.getWorkResourceIDs().isEmpty()) {
			entity.setWorkResourceID(((ID) sea.getWorkResourceIDs().iterator().next()).toInt());
		}
		if (sea.getParentID() != null) {
			entity.setShiftAssignmentID(sea.getParentID().toInt());
		}
		return entity;
	}

	@Override
	public Collection<ShiftEventAssignmentEntity> seasToEntities(Collection<ShiftEventAssignment> valueObjects) {
		Collection<ShiftEventAssignmentEntity> retVal = new ArrayList<>();
		for (ShiftEventAssignment vo : valueObjects) {
			retVal.add(seaToEntity(vo));
		}

		return retVal;
		//TODO: use streams
		//valueObjects.stream().forEach(e -> seaToEntity(e));
	}
}
