package com.verint.web.rest.wfm.resource.shifteventassignment;

import javax.inject.Inject;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import com.verint.web.rest.wfm.repository.WfoRelationshipRepository;
import com.verint.web.rest.wfm.repository.WfoResourceRepository;
import com.verint.web.rest.wfm.resource.activity.ActivityEntity;
import com.verint.web.rest.wfm.resource.activity.ActivityRepository;

import io.katharsis.legacy.repository.annotations.JsonApiRelationshipRepository;

@JsonApiRelationshipRepository(source = ShiftEventAssignmentEntity.class, target = ActivityEntity.class)
public class ShiftEventToActivityRepository
		extends WfoRelationshipRepository<Integer, Integer, ShiftEventAssignmentEntity, ActivityEntity> {

	private final static Logger log = LoggerFactory.getLogger(ShiftEventToActivityRepository.class.getName());

	@Inject
	private ActivityRepository activityRepository;

	@Inject
	private ShiftEventAssignmentRepository seaRepository;

	@Override
	public WfoResourceRepository<Integer, ShiftEventAssignmentEntity> getParentRepository() {
		return seaRepository;
	}

	@Override
	public WfoResourceRepository<Integer, ActivityEntity> getChildRepository() {
		return activityRepository;
	}

	@Override
	protected Logger logger() {
		return log;
	}
}
