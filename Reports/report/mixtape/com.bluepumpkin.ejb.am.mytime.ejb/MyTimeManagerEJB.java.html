	/**
	* <B>getScheduleEvents</B>
	* <P>
	*
	*  Returns the collection of all events for a work resource (e.g. an
	*  employee) with the given ID. Both imported events and event overrides
	*  will be returned. An employee's event will be returned as part of the
	*  collection of the event's duration interval overlaps with the requested
	*  interval (start date to end date, inclusively). The resulting collection
	*  of events is not sorted in any order.
	*
	*  @param employeeId   id of employee to search
	*  @param startDate    date to begin search
	*  @param endDate      date to end search
	* <P>
	*  @return                collection of user MyTimeEvent
	*/
	public Collection getScheduleEvents(ID workResourceID, Date startDate, Date endDate) throws AmException
	{
		methodStart("getScheduleEvents", workResourceID, startDate, endDate);

		Collection workResourceEvents = null;
		Collection flattenedWorkResourceEvents = null;
		Collection mergedSchedule = null;
		ArrayList events = new ArrayList();

		try {
			workResourceEvents = scheduleAccessManager.getPublishedEventsForWorkResource(workResourceID,	startDate, endDate);
			flattenedWorkResourceEvents = EventUtils.convertEventsToTimelineForSingleEmployee(workResourceEvents);			
			mergedSchedule = EventUtils.mergeSameEvents(flattenedWorkResourceEvents);

			//-------------------------------------------------------------------------
			// get activities and place in HashMap for easy access
			//-------------------------------------------------------------------------
			HashMap activityMap = new HashMap();
			//-------------------------------------------------------------------------
			// get activity categories and place in HashMap for easy access
			//-------------------------------------------------------------------------			
			HashMap activityCategoryMap = new HashMap();			
			HashSet actIDSet = new HashSet();
			for (Iterator i = mergedSchedule.iterator(); i.hasNext();) {
				Event event = (Event)i.next();
				MyTimeEvent currentMyTimeEvent = new MyTimeEvent();
				ID activityId = event.getActivityID();
				if (activityId != null ) {
					actIDSet.add(activityId);
				}
			}
			// load the activitymap
			Collection actCol = m_ActivityManager.findActivities(actIDSet);
			HashSet actCatIDSet = new HashSet();
			for (Iterator i = actCol.iterator(); i.hasNext(); ) {
				Activity act = (Activity)i.next();
				activityMap.put(act.getID(), act);
				actCatIDSet.add(act.getActivityCategoryId());
			}
			Collection actCatCol = m_ActivityManager.findActivityCategories(actCatIDSet);
			for (Iterator i = actCatCol.iterator(); i.hasNext(); ) {
				ActivityCategory actCat = (ActivityCategory)i.next();
				activityCategoryMap.put(actCat.getID(), actCat);
			}
			//-------------------------------------------------------------------------
			// iterate thru workResourceEvents and create collection of MyTimeEvents
			// MyTimeEvents are a convenience value object for presentation level
			//-------------------------------------------------------------------------
			for (Iterator i = mergedSchedule.iterator(); i.hasNext();) {
				Event event = (Event)i.next();
				MyTimeEvent currentMyTimeEvent = new MyTimeEvent();

				ID activityId = event.getActivityID();
				if (activityId != null ) {
					Activity activity = (Activity)activityMap.get(activityId);
					ID activityCategoryId = null;
					if (activity != null) {
						activityCategoryId = activity.getActivityCategoryId();
						currentMyTimeEvent.setActivity(activity);
						currentMyTimeEvent.setActivityCategory((ActivityCategory)activityCategoryMap.get(activityCategoryId));
						currentMyTimeEvent.setActivityColor(activity.getColor());
					}
					currentMyTimeEvent.setDuration(event.getDuration());
					currentMyTimeEvent.setStartTime(event.getStartTime());

					events.add(currentMyTimeEvent);
				}
			}
			return events;
		} catch (BbmFinderException e) {
			handleException(e);
			throw new AmException(e);
		} catch (Exception e) {
			handleException(e);
			throw new AmException(e);
		} finally {
			methodFinish();
		}
	}

	/**
	* <B>getActualEvents</B>
	* <P>
	*
	*  Returns the collection of all events for a work resource (e.g. an
	*  employee) with the given ID. It always read from TimeRecord.
	*
	*  @param employeeId   id of employee to search
	*  @param startDate    date to begin search
	*  @param endDate      date to end search
	* <P>
	*  @return                collection of user TimeRecord
	*/
	public Collection getActualEvents(ID workResourceID, Date startDate, Date endDate) throws AmException
	{
		methodStart("getActualEvents", workResourceID, startDate, endDate);
		try {
			Collection timeRecords = null;
			try {
				timeRecords = m_TimeRecordManager.getValidEventsForWorkResource(workResourceID, startDate, endDate);
			} catch(BbmTimePeriodOverlapException e) {
				timeRecords = (Collection)((HashMap)e.getContent()).get(workResourceID);
			}
			if (timeRecords == null)
				return null;
			ArrayList fallsInTR = new ArrayList(timeRecords.size());
			for (Iterator itTR = timeRecords.iterator(); itTR.hasNext(); )
			{
				TimeRecord tr = (TimeRecord)itTR.next();
				// check TimeRecord is in peirod or not
				if (StateChange.isInPeriod(tr, startDate, endDate, Overlap))
				{
					fallsInTR.add(tr);
				}
			}
			return fallsInTR;
		} catch (BbmFinderException e) {
			handleException(e);
			throw new AmException(e);
		} catch (Exception e) {
			handleException(e);
			throw new AmException(e);
		} finally {
			methodFinish();
		}
	}
