package com.bluepumpkin.web.am.adherence.mapping;

import java.text.MessageFormat;
import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.HashSet;
import java.util.ResourceBundle;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.workresource.OrganizationScopeManagerProxy;
import com.bluepumpkin.web.am.Log;
import com.bluepumpkin.web.am.l10n.AmWebBundleKey;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.bbm.organization.OrgWorkpaneListPageModel;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;

/**
 * Title:        AdherenceMappingListPageModel
 * Description:  Display the list of Activity per given organization
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, inc
 * @author       Shimon Kakon
 * @version 1.0
 */
@SuppressWarnings("serial")
public class AdherenceMappingListPageModel extends OrgWorkpaneListPageModel {
	private static final Category log = Log.initCategory(AdherenceMappingListPageModel.class.getName());
	private ResourceBundle m_amBundle;
	private AdherenceMapping[] m_adherenceMappings;
	private HashMap<ID, String> m_orgMap = null;

	// sort actions
	public static final String LIST_NAME_SORT = "LIST_NAME_SORT";
	public static final String LIST_ORG_SORT = "LIST_ORG_SORT";

	// Constants
	public static final String EMPTY_MAPPED_ACTIVITIES_STRING = "--";

	/**
	 * Constructor
	 */
	public AdherenceMappingListPageModel(RequestContext context) {
		super(context);
		m_amBundle = m_localizer.getBundle(AmWebBundleKey.BUNDLE_NAME);

		// Add JavaScript Variables
		addJSVariable(DELETE_CONFIRMATION, m_amBundle,	AmWebBundleKey.ADHERENCE_MAPPING_DELETE_CONFIRMATION);
	}


	/**
	 * Return instance of Model which is ready to be rendered
	 *
	 * @param context - The request context
	 * @return Adherence Mapping List PageModel
	 */
	public static AdherenceMappingListPageModel getInstance(RequestContext context) {
		return (AdherenceMappingListPageModel)getInstance(context, AdherenceMappingListPageModel.class);
	}


	/**
	 * get URL for HTML Form Action
	 *
	 * @return the form action
	 */
	public String getFormAction() {
		return "adherence_mapping_list";
	}


	/**
	 * Initialize List
	 */
	protected void initList() {
		super.initList();
		m_list.setIsZebra(true);
		// Set up Headers
		TableHeaderPC header = m_list.getHeader();
		header.addColumnHeaderData(LIST_NAME_SORT,
				i18n(m_amBundle, AmWebBundleKey.ADHERENCE_MAPPING_LABEL_SCHEDULED_ACTIVITY), true);
		header.addColumnHeaderData("noAction",
				i18n(m_amBundle, AmWebBundleKey.ADHERENCE_MAPPING_LABEL_MAPPING), false);
		header.addColumnHeaderData(LIST_ORG_SORT,
				i18n(m_bbmBundle, BbmWebBundleKey.LIST_HEADER_OWNER_ORG), true);

		// List Sorting
		m_list.setIsAutoSortEnabled(true);
		header.setDefaultSortColumn(LIST_NAME_SORT);
	}


	/**
	 * Initialize toolbar.
	 * Override default toolbar defined in WorkpaneListPageModel
	 */
	protected void initToolbar() {
		String editLabel = i18n(m_amBundle,
			AmWebBundleKey.ADHERENCE_MAPPING_EDIT_ACTION);

		ToolbarPC toolbar = m_toolbar;
		toolbar.setAlignment(ToolbarPC.ALIGN_RIGHT);

		toolbar.addSubmitTextButton(PageAction.EDIT, editLabel, isEditActionEnabled());
	}


	/**
	 * Get the default sort order.
	 * Override default toolbar defined in WorkpaneListPageModel
	 */
	protected String getDefaultSortColumn() {
		return LIST_NAME_SORT;
	}


	/**
	 * Returns true if the list support sorting.
	 * Override default toolbar defined in WorkpaneListPageModel
	 */
	protected boolean isSortSupported() {
		return true;
	}


	/**
	 * Create the list model
	 */
	protected void createListModel()
		{
			if(!this.loadAdherenceMapping()) 
			{
				return;
			}

			try
			{
				Collection<ID> orgIDs = null; 
				if (m_adherenceMappings.length > 0)
					orgIDs = OrganizationModelHandler.getAllOrgIDsInUserScopePlusAncestors(m_context, m_context.getUser(), true);
				ArrayList listModel = new ArrayList();

			Activity activity = null;
			for(int i = 0; i < m_adherenceMappings.length; i++) 
			{
				activity = m_adherenceMappings[i].getActivity();

				// Create the row data
				ArrayList<String> rowData = new ArrayList<String>(3);
				rowData.add(activity.getName());
				rowData.add(convertEmptyStringToDashes( m_adherenceMappings[i].getMappedActivitiesString(orgIDs)));
				rowData.add((String)m_orgMap.get(activity.getOrganizationId()));

				// Add row data to list model
				ID activityID = activity.getID();
				String rowID = activityID.toString();
				DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(rowID, rowData);
				nodeData.getNodeAttributes().put(IS_EDITABLE, isRowEditable(getSelectedIDObject()));
				listModel.add(nodeData);

				// Check if row should be selected
				if (isRowSelected(activityID)) 
					m_list.addSelectedRowID(rowID);
			}
			
			m_list.setListModel(listModel);
		} 
		catch (Exception ex) 
		{
			handleUnableToLoadDataError(log, ex);
		}
	}

	/**
	 * Get entity name. Used for title.
	 */
	protected String getTitleEntityType() {
		return i18n(m_amBundle, AmWebBundleKey.ADHERENCE_MAPPING_LIST_TITLE);
	}



	/**
	 * Loads all the adherence mapping value objects
	 *
	 * @return true if load was successful
	 */
	private boolean loadAdherenceMapping() {
		try {
			m_adherenceMappings = AdherenceMappingModelHandler.getAdherenceMappings(m_context, getSelectedIDObject());
			if(m_adherenceMappings != null && m_adherenceMappings.length > 0) {
				HashSet<ID> orgIds = new HashSet<ID>(m_adherenceMappings.length);
				Activity activity = null;
				ID id = null;
				for(int i = 0; i < m_adherenceMappings.length ; i++) {
					activity = m_adherenceMappings[i].getActivity();
					id = activity.getOrganizationId();
					if(!orgIds.contains(id)) {
						orgIds.add(id);
					}
				}
				m_orgMap = OrganizationModelHandler.buildOrganizationMap(m_context, orgIds);
			}
			return true;
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
			return false;
		}
	}



	public static String convertEmptyStringToDashes(String str) {
		if(StringUtil.isEmpty(str)) {
			return AdherenceMappingListPageModel.EMPTY_MAPPED_ACTIVITIES_STRING;
		}
		else {
			return str;
		}
	}

	/**
	 * Return true if Current User is authorized to configure specified Organization
	 *
	 * Overrides the superclass implementation, which defaults to 'true'
	 */
	protected boolean isAuthorizedToConfigure(ID orgID) {
		return m_context.getUser().isAuthorized(PrivilegeKeys.AM_CONFIGUREADHERENCEMAPPING,
				OrganizationScopeManagerProxy.getScopeID(orgID));
	}


	/**
	 * Return true if Current User is authorized to configure specified Organization
	 *
	 * Overrides the superclass implementation, which defaults to 'true'
	 */
	protected boolean isAuthorizedToView(ID orgID) {
		return m_context.getUser().isAuthorized(PrivilegeKeys.AM_VIEWADHERENCEMAPPING,
				OrganizationScopeManagerProxy.getScopeID(orgID));
	}

	/**
	 * Return localized message indicating that current user is not authorized
	 * to view activities in the selected Organization.
	 */
	protected String getNotAuthorizedToViewMsg() {
		return MessageFormat.format(
			i18n(m_bbmBundle, BbmWebBundleKey.NOT_AUTHORIZED_TO_VIEW_ENTITY_IN_SELECTED_ORG),
			new Object[] { i18n(m_bbmBundle, BbmWebBundleKey.BBM_ORG_ACT_ADHERENCE) } );
	}



	/**
	 * is the edit/delete actions should be enabled?
	 * Override the OrgWorkpaneListPageModel method, to allow mapping of activities
	 * from different organization
	 * @return true if edit/delete operation shoule be enabled
	 */
	protected boolean isEditDeleteActionsEnabled(boolean isSuperEnabled) {
		return isSuperEnabled;
	}
}
