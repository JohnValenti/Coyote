package com.verint.wfm.ws.wrappers;

import java.util.Collection;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.dao.MapUtil;
import com.bluepumpkin.ejb.bbm.dao.MapUtil.ObjectType;
import com.bluepumpkin.ejb.bbm.dao.MapUtil.SystemType;

/**
 * The purpose of this class is for testing, to inject a dependency in between this package and the static methods of MapUtil.
 */
public class MapUtilWrapper {

	private static MapUtilWrapper mapUtilWrapper;

	private MapUtilWrapper() {}

	public static MapUtilWrapper getInstance() {
		if (mapUtilWrapper == null) {
			mapUtilWrapper = new MapUtilWrapper();
		}
		return mapUtilWrapper;
	}

	public void flushMap(SystemType systemType) {
		MapUtil.flushMap(systemType);
	}

	public Map<ID, ID> getExternalIDMap(SystemType sysType, ObjectType objType, Collection<ID> idColl, boolean reverseLookup)
			throws JdmoException {
		return MapUtil.getExternalIDMap(sysType, objType, idColl, reverseLookup);
	}

	public ID getExternalToWfoID(SystemType systemType, ObjectType type, ID id) throws JdmoException {
		return MapUtil.getExternalToWfoID(systemType, type, id);
	}
}
