package com.bluepumpkin.web.bbm.campaign.selection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.witness.web.uif.system.RequestContext;

public class CampaignSchedulingPeriodSelectionUtil {
	
	//This key is auto cached ID used by CampaignSchedulingPeriodSelectionPM (tree nav selection of campaign/sp)
	private static final String SELECTED_CAMPAIGN_SP_ID = "SELECTED_CAMPAIGN_SP_ID";

	private static final String PREFIX_DELIM = ":";
	public static final String CAMPAIGN_ID_PREFIX = "c" + PREFIX_DELIM;
	public static final String SP_ID_PREFIX = "s" + PREFIX_DELIM;
	
	/**
	 * Returns a comma delimited list of IDs extracted from compositeIDString
	 * @param compositeIDString - in the format "c:<campaignID1>,c:<campaignID2>,...,s:<spID1>,s:<spID2>,..."
	 * @param prefix - either c or s, to indicate which ids to extract from the composite ID string
	 * @return
	 */
	public static String extractIDs(String compositeIDString, String prefix) {
		if (compositeIDString == null) return null;
		StringBuffer ret = new StringBuffer();
		String[] ids = compositeIDString.split(StringUtil.DELIM);
		
		for (int i = 0; i < ids.length; i++) {
			if (ids[i].indexOf(prefix) > -1) {
				if (ret.length() > 0) ret.append(",");
				ret.append(ids[i].split(PREFIX_DELIM)[1]);
			}
		}
		
		return ret.toString();
	}
	
	/**
	 * Creates a composite ID string of the format: "s:<spID>,c:<campaignID>"
	 * If either spID or campaignID is null, it will not include those pieces in
	 * the composite ID string (e.g.: if spID is null, the string returned will be 'c:<campaignID>'"
	 */
	public static String buildCompositeIDString(ID spID, ID campaignID) {
		String retVal = "";
		if (spID != null) retVal += SP_ID_PREFIX + spID.toString();
		if (campaignID != null) {
			if (retVal.length() > 0) retVal += ",";
			retVal += CAMPAIGN_ID_PREFIX + campaignID.toString();
		}
		
		return retVal;
	}
	
	/**
	 * This sets the composite ID string used in the auto ID caching of CampaignSchedulingPeriodSelectionPM.
	 * This setter is here so that other selection screens that select Campaigns/SPs can sync their selection
	 * persistence with this other method of persistence.
	 */
	public static void setSelectedAutoCachedCampaignSPID(RequestContext context, ID campaignID, ID spID) {
		context.setUserProperty(SELECTED_CAMPAIGN_SP_ID,
				CampaignSchedulingPeriodSelectionUtil.buildCompositeIDString(spID, campaignID));
	}
}
