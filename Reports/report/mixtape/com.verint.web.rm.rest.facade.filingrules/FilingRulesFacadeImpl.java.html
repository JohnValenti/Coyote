package com.verint.web.rm.rest.facade.filingrules;

import java.util.Collection;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.setup.filingrules.model.RequestFilingRule;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.setup.filingrules.FilingRuleModelHandler;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.util.FormatUtils;
import com.verint.web.rm.rest.entity.filingrules.FilingRuleEntity;
import com.verint.web.rm.rest.entity.filingrules.FilingRulesEntity;
import com.verint.web.rm.rest.facade.OrganizationSettingsFacadeImpl;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.system.RequestContext;

public class FilingRulesFacadeImpl implements IFilingRulesFacade {
	private static final Category LOG = Log.initCategory(OrganizationSettingsFacadeImpl.class.getName());

	/**
	 * Gets the RM filing rules for the employee's organization.
	 */
	@Override
	public FilingRulesEntity getFilingRulesForEmployee(RequestContext context, ID empID)
			throws WFOException {
		try {
			FilingRulesEntity result = new FilingRulesEntity();
			List<FilingRuleEntity> results = result.getRules();

			RmUtils.assertIsAuthorized(context, PrivilegeKeys.TOM_VIEWPERSONALREQUESTS_ID,
					PrivilegeKeys.SS_VIEWPERSONALREQUESTS_ID, PrivilegeKeys.SB_VIEWPERSONALBIDS_ID,
					PrivilegeKeys.SS_READWRITETOSWAPBOARD_ID, PrivilegeKeys.FT_VIEWPERSONALREQUESTS_ID);

			ID orgID = OrganizationModelHandler.getEmployeeOrgID(context, empID);
			Collection<RequestFilingRule> rules = FilingRuleModelHandler.getFilingRuleList(orgID);
			if (rules == null) {
				return result;
			}
			result.setId(orgID.toString());

			for (RequestFilingRule rule : rules) {
				// Policies web page does not show agent disabled rules
				if (!rule.isApplied()) {
					continue;
				}
				FilingRuleEntity entity = new FilingRuleEntity();
				entity.setApplied(rule.isApplied());
				entity.setActivityID(RmUtils.getInteger(rule.getActivityId()));
				entity.setDateComparison(rule.getDateComparision());
				entity.setDayRange(rule.getDayRange());
				entity.setDateType(rule.getDayType());
				entity.setEndTime(FormatUtils.dateToISO8601SecPrecisionString(rule.getEndDate()));
				entity.setFileByDate(FormatUtils.dateToISO8601SecPrecisionString(rule.getFileByDate()));
				entity.setId(rule.getID().toString());
				entity.setIntervalDays(rule.getIntervalDays());
				entity.setIntervalType(rule.getIntervalType());
				entity.setRequestSubType(rule.getRequestSubType());
				entity.setRequestType(rule.getRequestType());
				entity.setStartTime(FormatUtils.dateToISO8601SecPrecisionString(rule.getStartDate()));
				entity.setTimeLine(rule.getTimeLine());
				entity.setTimeType(rule.getTimeType());

				results.add(entity);
			}
			return result;
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(ex, LOG);
		}
	}
}
