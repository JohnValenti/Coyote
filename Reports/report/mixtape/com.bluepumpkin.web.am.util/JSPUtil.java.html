package com.bluepumpkin.web.am.util;

import java.text.*;

public class JSPUtil {

	/**
	 * getDisplayDuration()
	 */
	public static int[] getDisplayDuration(int seconds){
		int[] returnValue = new int[]{0, 0, 0};
		//convert to seconds
		int hours = seconds/(60*60);
		int minutes = seconds/(60) - hours*60;
		seconds = seconds - (hours*60*60 + minutes*60);
		returnValue[0] = hours;
		returnValue[1] = minutes;
		returnValue[2] = seconds;
		return returnValue;
	}

	/**
	 * return a date object corresponding to a string and given format
	 */
	public static java.util.Date toDateObj(String date, String format){
		return toDateObj(date, format, null);
	}

	public static java.util.Date toDateObj(String date, String format, TimeZone tz){
		java.util.Date dateToReturn = null;
		try{
			SimpleDateFormat dateFormat = new SimpleDateFormat( format);
			if (tz != null)
				dateFormat.setTimeZone(tz);
			dateToReturn = dateFormat.parse(date);
		}
		catch(ParseException pe){
		}
		return dateToReturn;
	}

	/**
	 * return a string representing a date for a given format
	 */
	public static String formatDate(java.util.Date date, String format)
		throws ParseException{
		return formatDate(date, format, null);
	}

	public static String formatDate(java.util.Date date, String format, TimeZone tz)
		throws ParseException
	{
		SimpleDateFormat dateFormat = new SimpleDateFormat( format);
		if (tz != null)
			dateFormat.setTimeZone(tz);
		return	dateFormat.format(date);
	}

	/**
	 * escape the specified character
	 * @param String
	 * @param String
	 * return excaped string
	 */
	public static StringBuffer escapeString(String display, String charToEscape) {
		StringBuffer buffer = new StringBuffer(display.length());
		int end = display.length();
		int start = 0;
		int idx;
		while ( start < end ) {
			idx = display.indexOf(charToEscape,start);
			if ( idx >= 0 ) {
				buffer.append(display.substring(start,idx))
					.append("\\'");
				start = idx + 1;
			} else {
				buffer.append(display.substring(start));
				start = end;
			}
		}
		return buffer;
	}
}
