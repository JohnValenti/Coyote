package com.verint.web.rest.wfm;

import io.katharsis.dispatcher.controller.HttpMethod;
import io.katharsis.errorhandling.ErrorData;
import io.katharsis.errorhandling.exception.KatharsisMappableException;
import io.katharsis.errorhandling.mapper.KatharsisExceptionMapper;
import io.katharsis.resource.exception.RequestBodyNotFoundException;
import io.katharsis.resource.exception.ResourceException;
import io.katharsis.response.BaseResponseContext;
import io.katharsis.response.HttpStatus;
import io.katharsis.rs.type.JsonApiMediaType;

import javax.annotation.Priority;
import javax.ws.rs.Priorities;
import javax.ws.rs.WebApplicationException;
import javax.ws.rs.container.ContainerRequestContext;
import javax.ws.rs.container.ContainerResponseContext;
import javax.ws.rs.container.ContainerResponseFilter;
import java.io.IOException;

/**
 * Error response handler for HTTP Status codes that are not handled by {@link io.katharsis.rs.KatharsisFilter}
 */
@Priority(Priorities.HEADER_DECORATOR)
public class APIResponseFilter extends JsonApiErrorResponse implements ContainerResponseFilter {

	@Override
	public void filter(ContainerRequestContext requestContext, ContainerResponseContext responseContext) throws WebApplicationException {
		BaseResponseContext baseResponseContext;
		try {
			// Maps a 400 HTTP Status to one of -
			// 415 if Content-Type is missing/incorrect in request header
			// 400 for Bad Requests
			if (responseContext.getStatus() == HttpStatus.NOT_FOUND_404) {
				if (requestContext.hasEntity() &&
						!requestContext.getMediaType().isCompatible(JsonApiMediaType.APPLICATION_JSON_API_TYPE)) {
					throw new UnsupportedMediaTypeException("Media Type not compatible with JSON API specification!");
				}
				throw new RequestBodyNotFoundException(HttpMethod.valueOf(requestContext.getMethod()), requestContext.getUriInfo().getPath());
			}

			// Handles 500 HTTP Status codes, only if response did not already have that information
			if (responseContext.getStatus() == HttpStatus.INTERNAL_SERVER_ERROR_500 &&
					!responseContext.hasEntity()) {
				throw new ResourceException("Something went wrong!");
			}
		} catch (KatharsisMappableException e) {
			baseResponseContext = new KatharsisExceptionMapper().toErrorResponse(e);
			try {
				prepareResponse(requestContext, baseResponseContext, responseContext);
			} catch (IOException io) {
				throw new WebApplicationException(io);
			}
		}
	}

	private class UnsupportedMediaTypeException extends KatharsisMappableException {
		private static final String TITLE = "UnsupportedMediaType";

		public UnsupportedMediaTypeException(String message) {
			super(APIHttpStatus.UNSUPPORTED_MEDIA_TYPE_415, ErrorData.builder()
					.setTitle(TITLE)
					.setDetail(message)
					.setStatus(String.valueOf(APIHttpStatus.UNSUPPORTED_MEDIA_TYPE_415))
					.build());
		}
	}
}
