package com.bluepumpkin.web.bbm.workload.queue.mapping;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.HashMap;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.config.ConfigKey;
import com.bluepumpkin.web.core.base.ModelHandler;
import com.witness.web.uif.system.RequestContext;

/**
 * Title: QueueGroupMH Description: Handle interaction with Group EJB Copyright:
 * Copyright (c) 2002 Company: Blue Pumpkin Software, Inc
 * 
 * @author Swati Tewari
 * @version 1.0
 */

public class QueueGroupMH extends ModelHandler {
    /**
     * Get linked group IDs for given queue
     * 
     * @param ID
     *            queueID return Collection of groupIDs
     */
    public static Collection getLinkedGroupIDsForQueue(ID queueID) throws RemoteException, BbmFinderException {
        return BbmManagerFactory.getQueueGroupManager().getLinkedGroupIDsForQueue(queueID);
    }

    public static Collection getLinkedGroupIDs(RequestContext context) throws RemoteException, BbmException {
        return BbmManagerFactory.getQueueGroupManager().getAllLinkedGroupIDs();
    }

    /**
     * Link the given queue to the groups
     * 
     * @param ID
     *            queueID
     * @param Collection
     *            colgroupIDs return Collection of groupIDs
     */
    public static void linkGroupsToQueue(ID idQueue, Collection colgroupIDs) throws RemoteException, BbmCreateException {
        BbmManagerFactory.getQueueGroupManager().linkGroupsToQueue(idQueue, colgroupIDs);
    }

    /**
     * Unlink the given queue to the groups
     * 
     * @param ID
     *            queueID
     * @param Collection
     *            colgroupIDs return Collection of groupIDs
     */
    public static void unLinkGroupsFromQueue(ID idQueue, Collection colgroupIDs) throws RemoteException, BbmRemoveException {
        BbmManagerFactory.getQueueGroupManager().unLinkGroupsFromQueue(idQueue, colgroupIDs);
    }

    /**
     * Get all groups in the system return Collection of group
     */
    public static Collection getAllGroups() throws RemoteException, BbmFinderException {
        return BbmManagerFactory.getGroupManager().getGroups();
    }

    /**
     * Get all groups in the system return Collection of group
     */
    public static HashMap getLinkedGroupIDsForQueues(Collection queueIDs) throws RemoteException, BbmFinderException {
        return BbmManagerFactory.getQueueGroupManager().getLinkedGroupIDsForQueues(queueIDs);
    }

    public static boolean isShowMappedGroups() throws Exception {
        final String sValue = BbmManagerFactory.getDBConfigManager().getValue(ConfigKey.BPX_GLOBAL_DISPLAY_MAPPED_GROUPS);
        if (sValue != null) {
            return Boolean.valueOf(sValue).booleanValue();
        }
        return true;
    }
}
