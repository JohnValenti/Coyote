/**
 * <p> <b> Hard Validation Rule: </b>
 */
public class PublishedPeriodValidationRule implements Validator {

	private static Category m_cat = Log.initCategory(PublishedPeriodValidationRule.class.getName());

	public PublishedPeriodValidationRule() {};

	/* (non-Javadoc)
		* @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
		*/
	public ValidationResult validate(Validatable validatable) throws Exception {
		CustShiftReq csr = (CustShiftReq) validatable;
		String methodName = "validate";
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, csr));
		ValidationResult result = null;

		boolean pubPeriod = false;
		CustShiftValidationCache cache = (CustShiftValidationCache) csr.getCache();
		ID empID = csr.getEmployeeID();
		//Just check if the requeted range is published 
		TimeRange range = new TimeRange(csr.getStartTime(),csr.getEndTime());
		Collection pubPeriodsByEmp = cache.getScheduleAccessManager().getPublishedPeriods
		        (Collections.singletonList(empID), range.getStartDate(), range.getEndDate());

		Collection pubPeriods = null;
		if (pubPeriodsByEmp != null && !pubPeriodsByEmp.isEmpty()) {
			pubPeriods = (Collection) pubPeriodsByEmp.iterator().next();
			if (pubPeriods != null) {
				for (Iterator i = pubPeriods.iterator(); i.hasNext();) {
					PublishingPeriod tp = (PublishingPeriod) i.next();
					TimeRange pubRange = new TimeRange(tp.getStartTime(), tp.getEndTime());
					if (!tp.isPartialPublish() && range.isInside(pubRange) ) {
						//no error Return here
						pubPeriod = true;
						break;
					}
				}
			}
		}
		if (!pubPeriod) {
			result = ValidationUtil.setSoftValidationResult(csr, RmEjbBundleKey.CS_PUBLISHED_PERIOD_CHECK, null, this.getClass().getName());
		}
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName));
		return result;
	}
}
