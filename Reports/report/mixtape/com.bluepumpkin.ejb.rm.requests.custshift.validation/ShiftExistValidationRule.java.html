import java.util.Collection;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.activity.model.Event;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.custshift.model.CustShiftReq;
import com.bluepumpkin.ejb.rm.util.CustomShiftEJBUtil;
import com.bluepumpkin.ejb.rm.util.RmUtil;

/**
 * <p> <b> Hard Validation Rule: </b>
 */
public class ShiftExistValidationRule implements Validator {

	private static Category m_cat = Log.initCategory(ShiftExistValidationRule.class.getName());

	public ShiftExistValidationRule() {};

	/* (non-Javadoc)
		* @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
		*/
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		CustShiftReq csr = (CustShiftReq) validatable;
		String methodName = "validate";
		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpEnterMethod(methodName, csr));
		}

		CustShiftValidationCache cache = (CustShiftValidationCache) csr.getValidationCache();
		ScheduleAccessManager sam = cache.getScheduleAccessManager();
		//The unpublished shift assignment
		ShiftAssignment sa = cache.getExistingShiftAssignToBeDeleted(csr, sam);

		//Check if user is just requesting an extension. in that case , there needs to be an existing shift
		if (csr.getShiftID() == null && (csr.getExtAfterID() != null || csr.getExtBeforeID() != null)) {
			//raise an error if no existing shift assignment is found. Most likely the shift was deleted
			if (csr.getShiftAssignmentID() == null || sa == null) {
				return ValidationUtil.setHardValidationResult(csr, RmEjbBundleKey.CS_SCHEDULE_CHANGED, null, this.getClass().getName());
			}
		}

		ValidationResult result = handleNoShiftAssignment(csr, sam);
		if (result != null) {
			return result;
		}

		//if csr.getShiftAssignmentID return null how can sa be anything but null? 
		boolean noShift = csr.getShiftAssignmentID() == null && sa == null;
		boolean isValidShift = csr.getShiftAssignmentID() != null && sa != null && csr.getShiftAssignmentID().equals(sa.getID());
		
		
		//check if the shift has changed in unpublished world.
		if (noShift || isValidShift) {
						
			//check if the unpublished is in sync with published world.
			if (sa != null && !CustomShiftEJBUtil.isPublishedShiftInSync(sa, sam)) {	
				//the unpublished shift is not in sync with published world
				result = ValidationUtil.setHardValidationResult(csr, RmEjbBundleKey.CS_PUB_SCHEDULE_CHANGED, null, this.getClass().getName());
			}
			//no shift found in unpublished world but shift does exists in published world then raise error.
			if (sa == null && cache.getPublishedShiftAssn(csr, sam) != null) {
				result = ValidationUtil.setHardValidationResult(csr, RmEjbBundleKey.CS_PUB_SCHEDULE_CHANGED, null, this.getClass().getName());
			}
		} else {
			//the shift has changed in unpublished world
			result = ValidationUtil.setHardValidationResult(csr, RmEjbBundleKey.CS_SCHEDULE_CHANGED, null, this.getClass().getName());
		}

		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpExitMethod(methodName));
		}

		return result;
	}

	ValidationResult handleNoShiftAssignment(CustShiftReq csr, ScheduleAccessManager sam)
			throws Exception {

		if (csr.getShiftAssignmentID() != null) {
			return null;
		}

		Collection colEventsUnPub = sam.getEventsForWorkResourceByType(Event.EVENT_TYPE_SHIFT_ASSIGNMENT, csr.getEmployeeID(),
				csr.getStartTime(), csr.getEndTime());

		if (colEventsUnPub == null || colEventsUnPub.isEmpty()) {
			//the shift is not in unpublished schedules. 
			return null;
		}

		CustShiftValidationCache cache = (CustShiftValidationCache) csr.getValidationCache();
		if (cache.getPublishedShiftAssn(csr, sam) != null) {
			//In both published & unpublished there are shifts that overlap the requested range, however the request's shift assignment 
			//is null. Output an validation error that says creating this request will overlap with an existing shift. 
			//This situation happens when an existing shift crosses midnight (or a requested shift crosses midnight) and cause
			//an overlap between an existing shift and a requested shift. 

			return ValidationUtil.setHardValidationResult(csr, RmEjbBundleKey.CS_OVERLAP_EXISTING_SHIFT, null, this.getClass().getName());
		}

		//found shift in unpublished, but it is not in published? assume that the published schedule has changed
		return ValidationUtil.setHardValidationResult(csr, RmEjbBundleKey.CS_PUB_SCHEDULE_CHANGED, null, this.getClass().getName());
	}
}
