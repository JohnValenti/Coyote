package com.bluepumpkin.ejb.bbm.rtaaservice.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

public class AdheranceOverride implements java.io.Serializable
{
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	protected Date m_dateStart, m_dateEnd, m_dateLastModified;
	protected ID m_idEmployee;
	protected boolean m_bIsDeleted;
	protected String m_approveBy;
	protected String m_strComment;

	public AdheranceOverride(ID idEmployee, Date dateStart, Date dateEnd, Date dateLastModified, boolean bIsDeleted, String approveBy, String strComment)
	{
		super();
		m_idEmployee = idEmployee;
		m_dateStart = dateStart;
		m_dateEnd = dateEnd;
		m_dateLastModified = dateLastModified;
		m_bIsDeleted = bIsDeleted;
		m_approveBy = approveBy;
		m_strComment = strComment;
	}

	public AdheranceOverride(AdheranceOverride adheranceOverride)
	{
		super();
		m_idEmployee = adheranceOverride.getEmployeeID();
		m_dateStart = adheranceOverride.getStartTime();
		m_dateEnd = adheranceOverride.getEndTime();
		m_dateLastModified = adheranceOverride.getLastModifiedTime();
		m_bIsDeleted = adheranceOverride.isDeleted();
		m_approveBy = adheranceOverride.getApproveBy();
		m_strComment = adheranceOverride.getComment();
	}

	public boolean isDeleted() { return m_bIsDeleted; }
	public ID getEmployeeID()
	{
		return m_idEmployee;
	}
	public Date getStartTime()
	{
		return m_dateStart;
	}
	public void setStartTime(Date dateStart)
	{
		m_dateStart = dateStart;
	}
	public Date getEndTime()
	{
		return m_dateEnd;
	}
	public void setEndTime(Date dateEnd)
	{
		m_dateEnd = dateEnd;
	}
	public Date getLastModifiedTime()
	{
		return m_dateLastModified;
	}
	public void setLastModifiedTime(Date dateLastModified)
	{
		m_dateLastModified = dateLastModified;
	}
	public String getApproveBy()
	{
		return m_approveBy;
	}
	public void setApproveBy(String approveBy) {
		m_approveBy = approveBy;
	}

	public String getComment()
	{
		return m_strComment;
	}
	public void setComment(String strComment) {
		m_strComment = strComment;
	}


	public String toString()
	{
		StringBuffer sb = new StringBuffer(200);
		sb.append("EmployeeID: " + getEmployeeID());
		if (isDeleted())
			sb.append("\n").append("IsDeleted");
		sb.append("\n").append("Start: " + getStartTime());
		sb.append("\t").append("End: " + getEndTime());
		sb.append("\t").append("LastModified: " + getLastModifiedTime());
		sb.append("\t").append("ApproveBy: " + getApproveBy());
		sb.append("\t").append("Comment: " + getComment());
		return sb.toString();
	}
}
