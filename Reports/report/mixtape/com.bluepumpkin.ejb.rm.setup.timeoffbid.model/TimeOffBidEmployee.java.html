package com.bluepumpkin.ejb.rm.setup.timeoffbid.model;

public class TimeOffBidEmployee {
	public static final int SUCCESS = 0;
	public static final int NOT_IN_SYSTEM = 1;
	public static final int NOT_IN_BID = 2;
	public static final int WEEKS = 1;
	public static final int DAYS = 0;

	String firstName = "";
	String lastName = "";
	int maxApprovePeriod = 0;
	int periodType = 0;
	int timeOffBidID = 0;
	int status = 0;

	public TimeOffBidEmployee() {
	}

	public TimeOffBidEmployee(String firstName, String lastName, int maxApprovePeriod) {
		this.firstName = firstName;
		this.lastName = firstName;
		this.maxApprovePeriod = maxApprovePeriod;
	}

	public void setFirstName(String firstName) {
		this.firstName = firstName;
	}

	public String getFirstName() {
		return firstName;
	}

	public void setLastName(String lastName) {
		this.lastName = lastName;
	}

	public String getLastName() {
		return lastName;
	}

	public void setMaxApprovePeriod(int maxApprovePeriod) {
		this.maxApprovePeriod = maxApprovePeriod;
	}

	public int getMaxApprovePeriod() {
		return maxApprovePeriod;
	}

	public void setPeriodType(int periodType) {
		this.periodType = periodType;
	}

	public int getPeriodType() {
		return periodType;
	}

	public void setTimeOffBidID(int timeOffBidID) {
		this.timeOffBidID = timeOffBidID;
	}

	public int getTimeOffBidID() {
		return timeOffBidID;
	}

	public void setStatus(int status) {
		this.status = status;
	}

	public int getStatus() {
		return status;
	}
}
