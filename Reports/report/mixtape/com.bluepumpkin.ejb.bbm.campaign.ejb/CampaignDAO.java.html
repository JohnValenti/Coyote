	public ID createCampaign(Campaign objValue) throws JdmoException {
		// Updated for v11.
		Jdmo jdmo = new Jdmo();
		ID createdID = new ID();
		try {
			// get the time zone str id
			ID timeZoneDBID = null;
			ID tzID = objValue.getTimeZoneID();
			String getTZIDStr =
					new String("select ID  from TIMEZONE where STANDARDNAME = "
							+ "(select WINTZSTDNAME from TIMEZONEMAP where JAVATZIDENT = ?)");
			JdmoQuery getTZIDQuery = jdmo.createQuery(getTZIDStr, Jdmo.PARAM_QUERY);
			getTZIDQuery.setParString(1, tzID.toString());
			JdmoRowset rsGetTZID = jdmo.createRowset(getTZIDQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			if (rsGetTZID.next()) {
				timeZoneDBID = rsGetTZID.getID(1);
			} else {
				// include new entry.
				String strWinStdName = null;
				String getWindowsNameTZStr = new String("select WINTZSTDNAME from TIMEZONEMAP where JAVATZIDENT = ?");
				JdmoQuery getWindowsNameTZQuery = jdmo.createQuery(getWindowsNameTZStr, Jdmo.PARAM_QUERY);
				getWindowsNameTZQuery.setParString(1, tzID.toString());
				JdmoRowset rsGetWindowsNameTZ = jdmo.createRowset(getWindowsNameTZQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
				if (rsGetWindowsNameTZ.next()) {
					strWinStdName = rsGetWindowsNameTZ.getString(1);
				} else {
					// QC-114718 otherwise it will create a invalid timezone
					// with 'null' as the standard name
					throw new JdmoException("The time zone id " + tzID.toString() + " does not exist.");
				}
				timeZoneDBID = new ID(DAOUtil.getDEID(jdmo));
				String strCreateTZ =
						"INSERT INTO TIMEZONE (ID, STANDARDNAME, BIAS) values ('" + timeZoneDBID.toString() + "','"
								+ strWinStdName + "', -1)";
				jdmo.executeCommand(strCreateTZ);
			}

			ID viewtimeZoneDBID = null;
			tzID = objValue.getDefaultViewTimeZoneID();
			if (tzID != null) {
				getTZIDStr =
						new String("select ID  from TIMEZONE where STANDARDNAME = "
								+ "(select WINTZSTDNAME from TIMEZONEMAP where JAVATZIDENT = ?)");
				getTZIDQuery = jdmo.createQuery(getTZIDStr, Jdmo.PARAM_QUERY);
				getTZIDQuery.setParString(1, tzID.toString());
				rsGetTZID = jdmo.createRowset(getTZIDQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
				if (rsGetTZID.next()) {
					viewtimeZoneDBID = rsGetTZID.getID(1);
				} else // include new entry.
				{
					String strWinStdName = null;
					String getWindowsNameTZStr = new String("select WINTZSTDNAME from TIMEZONEMAP where JAVATZIDENT = ?");
					JdmoQuery getWindowsNameTZQuery = jdmo.createQuery(getWindowsNameTZStr, Jdmo.PARAM_QUERY);
					getWindowsNameTZQuery.setParString(1, tzID.toString());
					JdmoRowset rsGetWindowsNameTZ =
							jdmo.createRowset(getWindowsNameTZQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
					if (rsGetWindowsNameTZ.next()) {
						strWinStdName = rsGetWindowsNameTZ.getString(1);
						viewtimeZoneDBID = new ID(DAOUtil.getDEID(jdmo));
						String strCreateTZ =
								"INSERT INTO TIMEZONE (ID, STANDARDNAME, BIAS) values ('" + viewtimeZoneDBID.toString()
										+ "','" + strWinStdName + "', -1)";
						jdmo.executeCommand(strCreateTZ);
					} else {
						// QC-114718 otherwise it will create a invalid timezone
						// with 'null' as the standard name
						throw new JdmoException("The view time zone id " + tzID.toString() + " does not exist.");
					}
				}
			}

			// get ID from DE
			String idStr = DAOUtil.getDEID(jdmo);
			ID DEIDparentID = getDEIDFromSID("Campaign", objValue.getParentID());
			String strCreate =
					"INSERT INTO CAMPAIGN (ID, ISDISTRIBUTION, PARENTCAMPAIGNID, NAME, TIMEZONEID,"
							+ " WEEKSTART, DESCRIPTION, VIEWTIMEZONEID, PERIODICITYID, MONTHSTARTDAY) values ('"
							+ idStr
							+ "','"
							+ JdmoUtil.formatBool(objValue.getIsDistributed())
							+ "',"
							+ ((DEIDparentID != null) ? "'" : "")
							+ DEIDparentID
							+ ((DEIDparentID != null) ? "'" : "")
							+ ",'"
							+ JdmoUtil.formatDBString(objValue.getName())
							+ "','"
							+ timeZoneDBID
							+ "',"
							+ objValue.getWeekStart()
							+ ",'"
							+ JdmoUtil.formatDBString(objValue.getDescription())
							+ "',"
							+ ((viewtimeZoneDBID != null) ? "'" + viewtimeZoneDBID + "'" : "NULL")
							+ ","
							+ objValue.getPeriodicityID() + "," + objValue.getMonthStartDay() + ")";

			jdmo.executeCommand(strCreate);

			// find SID and return
			String stmt = "SELECT SID FROM CAMPAIGN WHERE ID ='" + idStr + "'";
			JdmoRowset rs = m_dmo.createRowset(stmt);
			if (rs.next()) {
				createdID = rs.getID(1);
			}
			rs.close();
		} finally {
			jdmo.cleanUp();
		}
		return createdID;

	}

	protected ID getDEIDFromSID(String TableName, ID sidValue) throws JdmoException {
		ID DEID = null;
		if (sidValue == null) {
			return DEID;
		}

		String stmt = "SELECT ID FROM " + TableName + " WHERE SID='" + sidValue + "'";
		JdmoRowset rs = m_dmo.createRowset(stmt);
		if (rs.next()) {
			DEID = rs.getID(1);
		}
		return DEID;
	}

	protected ID getSIDFromDEID(String TableName, ID idValue) throws JdmoException {
		ID SID = null;
		String stmt = "SELECT SID FROM " + TableName + " WHERE ID='" + idValue + "'";
		JdmoRowset rs = m_dmo.createRowset(stmt);
		if (rs.next()) {
			SID = rs.getID(1);
		}
		return SID;
	}

	public static TimeZone getTimeZoneByCampaignID(ID campaignSID) throws BbmFinderException {
		Jdmo dmo = new Jdmo();
		try {
			StringBuffer pStmt = new StringBuffer();
			pStmt.append("select TIMEZONEMAP.JAVATZIDENT ");
			pStmt.append(" from TIMEZONE,TIMEZONEMAP,CAMPAIGN ");
			pStmt.append(" where CAMPAIGN.TIMEZONEID=TIMEZONE.ID ");
			pStmt.append(" and TIMEZONEMAP.WINTZSTDNAME=TIMEZONE.STANDARDNAME");
			pStmt.append(" and CAMPAIGN.SID=?");

			JdmoQuery jq = dmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			jq.setParID(1, campaignSID);
			JdmoRowset rs = dmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			String timeZoneDisplayName = null;
			if (rs.next()) {
				timeZoneDisplayName = rs.getString(1);
			}
			return TimeZone.getTimeZone(timeZoneDisplayName);
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		} finally {
			dmo.cleanUp();
		}
	}

	/**
	 * Returns a map of campaignID -> campaign's TimeZone for the given set of campaign IDs.
	 *
	 * @param campaignSIDs
	 * @return Map of campaign SID -> TimeZone
	 */
	public Map<ID, TimeZone> getTimeZonesByCampaignIDs(Collection<ID> campaignSIDs) throws BbmFinderException {
		try {
			StringBuffer query = new StringBuffer();
			query.append("select CAMPAIGN.SID, TIMEZONEMAP.JAVATZIDENT ");
			query.append(" from TIMEZONE,TIMEZONEMAP,CAMPAIGN ");
			query.append(" where CAMPAIGN.TIMEZONEID=TIMEZONE.ID ");
			query.append(" and TIMEZONEMAP.WINTZSTDNAME=TIMEZONE.STANDARDNAME");
			query.append(" and CAMPAIGN.SID IN ");
			query.append(getDMO().createInClause(campaignSIDs));

			JdmoRowset rs = getDMO().createRowset(query.toString());

			Map<ID, TimeZone> retVal = new HashMap<ID, TimeZone>();
			while (rs.next()) {
				ID campaignID = rs.getID("SID");
				String timeZoneDisplayName = rs.getString("JAVATZIDENT");
				retVal.put(campaignID, TimeZone.getTimeZone(timeZoneDisplayName));
			}
			return retVal;
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}

	public Collection<Campaign> getCampaign(Collection<? extends ID> cidCampaigns, boolean isByDEID)
			throws BbmFinderException {
		try {
			// load all campaigns
			ArrayList<Campaign> aCampaigns = new ArrayList<Campaign>();

			String query =
					"SELECT C.ID, C.SID,TZMA.JAVATZIDENT TZ,TZMB.JAVATZIDENT VIEWTZ, C.NAME,C.WEEKSTART,"
							+ "C.DESCRIPTION,C.ISDISTRIBUTION,C.PERIODICITYID,C.MONTHSTARTDAY, "
							+ "(SELECT SID FROM CAMPAIGN WHERE ID = C.PARENTCAMPAIGNID) PARENTSID "
							+ "FROM CAMPAIGN C,TIMEZONE TZA , TIMEZONE TZB, TIMEZONEMAP TZMA, TIMEZONEMAP TZMB "
							+ "WHERE TIMEZONEID = TZA.ID AND TZMA.WINTZSTDNAME = TZA.STANDARDNAME AND "
							+ m_dmo.getNullFunctionName()
							+ "(VIEWTIMEZONEID,TIMEZONEID) = TZB.ID AND TZMB.WINTZSTDNAME = TZB.STANDARDNAME ";
			if (cidCampaigns != null) {
				if (isByDEID) {
					query += " AND C.ID IN " + m_dmo.createInClause(cidCampaigns);
				} else {
					query += " AND C.SID IN " + m_dmo.createInClause(cidCampaigns);
				}
			}

			JdmoRowset r = m_dmo.createRowset(query);

			// -------------------------------------------------
			// get data and create campaigns
			// -------------------------------------------------

			while (r.next()) {

				ID idCampaign1 = r.getID("SID");
				ID deID = r.getID("ID");
				TimeZone pTZ = TimeZone.getTimeZone(r.getString("TZ"));
				String sName = r.getString("NAME");
				int nWeekStart = r.getInt("WEEKSTART");
				String sDescription = r.getString("DESCRIPTION");
				ID idParentCampaign = r.getID("PARENTSID");

				int nPeriodicityID = r.getInt("PERIODICITYID");
				short nMonthStartDay = r.getShort("MONTHSTARTDAY");

				TimeZone pViewTZ = TimeZone.getTimeZone(r.getString("VIEWTZ"));
				boolean isDistributed = r.getBoolean("ISDISTRIBUTION");

				Campaign pCampaign = new Campaign();
				pCampaign.setID(idCampaign1);
				pCampaign.setDEID(deID);
				pCampaign.setName(sName);
				pCampaign.setTimeZone(pTZ);
				pCampaign.setTimeZoneID(new ID(pTZ.getID()));
				pCampaign.setSchedulingPeriodType(Campaign.CAMPAIGN_PERIOD_TYPE_WEEK);
				pCampaign.setSchedulingPeriodLength(1);
				pCampaign.setDescription(sDescription);
				pCampaign.setParentID(idParentCampaign);
				pCampaign.setWeekStart(nWeekStart);
				pCampaign.setDefaultViewTimeZone(pViewTZ);
				pCampaign.setIsDistributed(isDistributed);

				pCampaign.setPeriodicityID(nPeriodicityID);
				pCampaign.setMonthStartDay(nMonthStartDay);

				aCampaigns.add(pCampaign);
			}

			r.close();
			return aCampaigns;

		} catch (JdmoException e) {
			e.printStackTrace();
			throw new BbmFinderException(e);
		} finally {
			m_dmo.cleanUp();
		}
	}

	/**
	 * Routine to return the ID of a campaign given the name. Throws an error if name is not unique
	 *
	 * @param name
	 * @return ID
	 * @throws JdmoException
	 */
	static ID getCampaignIDByName(String name) throws JdmoException {
		String select = "SELECT sid FROM campaign WHERE name = ?";
		ID campaignID = null;
		Jdmo dmo = new Jdmo();
		JdmoRowset rs = null;
		try {
			JdmoQuery query = dmo.createQuery(select, Jdmo.PARAM_QUERY);
			query.setParString(1, name);
			rs = dmo.createRowset(query, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			if (rs.next()) {
				campaignID = rs.getID(1);
				// check unique
				if (rs.next()) {
					throw new JdmoException("Non unique queue name; " + name);
				}
			}
			return campaignID;
		} finally {
			try {
				rs.close();
			} catch (JdmoException e) {
			}
			dmo.cleanUp();
		}
	}

	 */
	public Map<ID, Campaign> getCampaignsLinkedToSchedulingPeriods(Collection<SchedulingPeriod> sps)
			throws BbmObjectNotFoundException, BbmFinderException {
		Map<ID, Campaign> campaignsBySPID = new HashMap<ID, Campaign>();
		Collection<Campaign> campaigns =
				getObjectsByIDs(ValueObjectUtil.getFieldObjectCol(SchedulingPeriodFieldInfo.SP_CAMPAIGNID, sps));
		for (SchedulingPeriod sp : sps) {
			for (Campaign campaign : campaigns) {
				if (sp.getCampaignID().equals(campaign.getID())) {
					campaignsBySPID.put(sp.getID(), campaign);
					break;
				}
			}
		}
		return campaignsBySPID;
	}

	public static void deleteCampaigns(Collection<ID> colCampaignIDs) throws BbmRemoveException {
		Jdmo dmo = new Jdmo();
		try {
			String where = "select id from CAMPAIGN where sid in (" + dmo.createInClause(colCampaignIDs) + ")";
			// delete children campaign too.
			where +=
					" OR PARENTCAMPAIGNID in (select ID from CAMPAIGN where SID in " + dmo.createInClause(colCampaignIDs)
							+ ")";
			DAOUtil.deleteObjects(dmo, "CAMPAIGN", where);
		} catch (JdmoException ex) {
			throw new BbmRemoveException(ex);
		} finally {
			dmo.cleanUp();
		}
	}

	/**
	 * Returns a map of Scheduling Period IDs to their associated campaign's name.
	 *
	 * @param spIDs - The scheduling period IDs for which we'd like to look up their associated campaign's name.
	 * @throws BbmFinderException
	 */
	public Map<ID, String> getCampaignNamesLinkedToSPs(Collection<ID> spIDs) throws BbmFinderException {
		HashMap<ID, String> retVal = new HashMap<ID, String>();
		try {
			SchedulingPeriodFieldInfo spFieldInfo = new SchedulingPeriodFieldInfo();
			/**
			 * select SP.SID, CAMPAIGN.NAME FROM CAMPAIGN, SP WHERE SP.SID IN (<spIDs>) AND CAMPAIGN.ID = SP.CAMPAIGNID
			 */
			JdmoRowset r =
					getDMO().createRowset(
							"SELECT " + spFieldInfo.getTableName() + "."
									+ spFieldInfo.getFieldName(SchedulingPeriodFieldInfo.SP_ID) + ","
									+ getFieldInfo().getTableName() + "."
									+ getFieldInfo().getFieldName(CampaignFieldInfo.CAMPAIGN_NAME) + " FROM "
									+ getFieldInfo().getTableName() + "," + spFieldInfo.getTableName() + " WHERE "
									+ spFieldInfo.getTableName() + "."
									+ spFieldInfo.getFieldName(SchedulingPeriodFieldInfo.SP_ID) + " IN "
									+ getDMO().createInClause(spIDs) + " AND " + getFieldInfo().getTableName() + "."
									+ getFieldInfo().getFieldName(CampaignFieldInfo.CAMPAIGN_ID) + " = "
									+ spFieldInfo.getTableName() + "."
									+ spFieldInfo.getFieldName(SchedulingPeriodFieldInfo.SP_CAMPAIGNID));
			while (r.next()) {
				retVal.put(r.getID(spFieldInfo.getFieldName(SchedulingPeriodFieldInfo.SP_ID)),
						r.getString(getFieldInfo().getFieldName(CampaignFieldInfo.CAMPAIGN_NAME)));
			}
		} catch (JdmoException ex) {
			throw new BbmFinderException(ex);
		}
		return retVal;
	}
