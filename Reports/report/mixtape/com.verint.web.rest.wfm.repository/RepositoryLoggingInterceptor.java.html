package com.verint.web.rest.wfm.repository;


import org.aopalliance.intercept.MethodInterceptor;
import org.aopalliance.intercept.MethodInvocation;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class RepositoryLoggingInterceptor implements MethodInterceptor {

	private final static Logger log = LoggerFactory.getLogger(RepositoryLoggingInterceptor.class);

	public Object invoke(MethodInvocation invocation) throws Throwable {
		if (log.isDebugEnabled()) {
			debugLogMethodInvocation(invocation);
		} else {
			log.info("RepositoryLoggingInterceptor: entering method " + invocation.getMethod().getName());
		}
		return invocation.proceed();
	}

	private void debugLogMethodInvocation(MethodInvocation invocation) {
		StringBuilder sb = new StringBuilder(200);
		sb.append("Repository: ").append(invocation.getThis().getClass().getSimpleName());
		sb.append(" Method: ").append(invocation.getMethod().getName());
		if (invocation.getArguments() != null && invocation.getArguments().length > 0) {
			for (int i = 0; i < invocation.getArguments().length; i++) {
				sb.append(" Parameter: ").append(invocation.getMethod().getParameters()[i].toString());
				sb.append(" ParamValue: ").append(invocation.getArguments()[i].toString());
			}
		}
		log.info(sb.toString());
	}
}
