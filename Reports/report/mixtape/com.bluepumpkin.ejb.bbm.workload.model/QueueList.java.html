package com.bluepumpkin.ejb.bbm.workload.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class QueueList extends ValueObjectBase {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new QueueListFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public QueueList() {
		super();
	}

	public QueueList(FieldInfo info) {
		super(info);
		// TODO Auto-generated constructor stub
	}

	public Date getEndTime() {
		return getFieldValueDate(QueueListFieldInfo.QUEUE_LIST_ENDTIME);
	}

	public void setEndTime(Date endTime) {
		setFieldValue(QueueListFieldInfo.QUEUE_LIST_ENDTIME,endTime);
	}

	public ID getId() {
		return getFieldValueID(QueueListFieldInfo.QUEUE_LIST_ID);
	}

	public void setId(ID id) {
		setFieldValue(QueueListFieldInfo.QUEUE_LIST_ID,id);
	}

	public ID getQueueId() {
		return getFieldValueID(QueueListFieldInfo.QUEUE_LIST_QUEUE_SID);
	}

	public void setQeueuId(ID id) {
		setFieldValue(QueueListFieldInfo.QUEUE_LIST_QUEUE_SID,id);
	}

	public int getLength() {
		return getFieldValueInt(QueueListFieldInfo.QUEUE_LIST_LENGTH);
	}

	public void setLength(int length) {
		setFieldValue(QueueListFieldInfo.QUEUE_LIST_LENGTH,length);
	}

	public int getRetries() {
		return getFieldValueInt(QueueListFieldInfo.QUEUE_LIST_RETRIES);
	}

	public void setRetries(int retries) {
		setFieldValue(QueueListFieldInfo.QUEUE_LIST_RETRIES,retries);
	}

	public Date getStartTime() {
		return getFieldValueDate(QueueListFieldInfo.QUEUE_LIST_STARTTIME);
	}

	public void setStartTime(Date startTime) {
		setFieldValue(QueueListFieldInfo.QUEUE_LIST_STARTTIME,startTime);
	}
	
	public boolean equals(Object obj)
	{
		if(!(obj instanceof QueueList)) return false;
		QueueList ql = (QueueList) obj;
		return ql.getQueueId().equals(this.getQueueId())&&
				ql.getStartTime().equals(this.getStartTime())&&
				ql.getEndTime().equals(this.getEndTime());
	}
	public int hashCode()
	{
		String hashCode = getQueueId().toString()+"|"+getEndTime().toGMTString()+"|"+getStartTime();
		return hashCode.hashCode();
	}
	public void add (QueueList ql)
	{
		setLength(ql.getLength()+getLength());
		setRetries(ql.getRetries()+getRetries());
	}
}
