package com.bluepumpkin.ejb.bbm.workload.model;

/**
 * An enumeration listing the types of Queues that are available.
 */
public enum QueueType {
	Normal,
	Distributed,
	Virtual,
	Process,
	Unknown;

	/**
	 * Converts an integer value (what is stored in the DB) to a QueueType.
	 */
	public static QueueType fromInt(int queueType) {
		switch(queueType) {
		case Queue.QUEUE_TYPE_NORMAL:
			return Normal;
		case Queue.QUEUE_TYPE_DISTRIBUTED:
			return Distributed;
		case Queue.QUEUE_TYPE_VIRTUAL:
			return Virtual;
		case Queue.QUEUE_TYPE_PROCESS:
			return Process;
		default:
			return Unknown;
		}
	}
	
	/**
	 * Converts a QueueType into an integer (what is stored in the DB).
	 */
	public static int toInt(QueueType queueType) {
		switch(queueType) {
		case Normal:
			return Queue.QUEUE_TYPE_NORMAL;
		case Distributed:
			return Queue.QUEUE_TYPE_DISTRIBUTED;
		case Virtual:
			return Queue.QUEUE_TYPE_VIRTUAL;
		case Process:
			return Queue.QUEUE_TYPE_PROCESS;
		default:
			return -1;
		}		
	}
}
