package com.bluepumpkin.ejb.bbm.activity.ejb;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityQueue;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityQueueFieldInfo;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class ActivityQueueDAO extends DAOBase {

	// meta data information
	private static FieldInfo m_fieldInfo = new ActivityQueueFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public ActivityQueueDAO(){
		super();
	}

	public ActivityQueueDAO(Jdmo dmo){
		super(dmo);
	}

	protected ValueObjectBase createValueObject() {
		return (new ActivityQueue());
	}
	
	protected Collection findQueueIdsForActivity(ID activityId)
		throws BbmFinderException {
		ArrayList<ID> list = new ArrayList<ID>(1);
		list.add(activityId);
		Object obj = findQueueForActivities(list).get(activityId);
		if (obj == null)
			return Collections.EMPTY_LIST;
		return (Collection)obj;
	}
	
	/**
	 * <B>findQueueForActivities</B>
	 * <P>
	 * find Queue Ids per activity for a given collection of activities
	 * <P>
	 *
	 * @param activityIds: collection of activity Ids
	 * @return Map of activity ID (key) and collection of Queue Ids (value)
	 */
	public Map<ID, ArrayList<ID>> findQueueForActivities(Collection<ID> activityIds)
		throws BbmFinderException {
	
		try {
			if(activityIds == null || activityIds.isEmpty()) {
				return Collections.EMPTY_MAP;
			}
			StringBuffer strWhere = new StringBuffer(30);
			strWhere.append("A.ACTIVITYID in ").append(m_dmo.createInClause(activityIds));
			strWhere.append(" ORDER BY A.ACTIVITYID ");
	
			Collection activityQueueVOCollection = getObjects(strWhere.toString());
			HashMap<ID, ArrayList<ID>> activityQueueIdsMap = new HashMap<ID, ArrayList<ID>>(activityIds.size());
			ActivityQueue activityQueue = null;
			ArrayList<ID> queueIdsCollection = null;
			for(Iterator it = activityQueueVOCollection.iterator(); it.hasNext();) {
				activityQueue = (ActivityQueue)it.next();
				if(activityQueueIdsMap.containsKey(activityQueue.getActivityId())) {
					queueIdsCollection = activityQueueIdsMap.get(activityQueue.getActivityId());
					queueIdsCollection.add(activityQueue.getQueueId());
				}
				else {
					queueIdsCollection = new ArrayList<ID>(1);
					queueIdsCollection.add(activityQueue.getQueueId());
					activityQueueIdsMap.put(activityQueue.getActivityId(), queueIdsCollection);
				}
			}
			return activityQueueIdsMap;
		} catch(Exception e) {
			 throw new BbmFinderException(e);
		}
	}
	
	/**
	 * <B>findActivityForQueues</B>
	 * <P>
	 * find Activity Ids per Queue for a given collection of queue IDs
	 * <P>
	 *
	 * @param queueIds: collection of Queue Ids
	 * @return Map of Queue ID (key) and collection of Activity Ids (value)
	 */
	public Map<ID, ArrayList<ID>> findActivityForQueues(Collection<ID> queueIds)
	throws BbmFinderException {

	try {
		if(queueIds == null || queueIds.isEmpty()) {
			return Collections.emptyMap();
		}
		StringBuffer strWhere = new StringBuffer(30);
		strWhere.append("A.QUEUEID in ").append(m_dmo.createInClause(queueIds));
		//strWhere.append(" ORDER BY A.ACTIVITYID ");

		Collection activityQueueVOCollection = getObjects(strWhere.toString());
		HashMap<ID, ArrayList<ID>> activityQueueIdsMap = new HashMap<ID, ArrayList<ID>>(queueIds.size());
		ActivityQueue activityQueue = null;
		ArrayList<ID> activityIdsCollection = null;
		for(Iterator it = activityQueueVOCollection.iterator(); it.hasNext();) {
			activityQueue = (ActivityQueue)it.next();
			if(activityQueueIdsMap.containsKey(activityQueue.getQueueId())) {
				activityIdsCollection = activityQueueIdsMap.get(activityQueue.getQueueId());
				activityIdsCollection.add(activityQueue.getActivityId());
			}
			else {
				activityIdsCollection = new ArrayList<ID>(1);
				activityIdsCollection.add(activityQueue.getActivityId());
				activityQueueIdsMap.put(activityQueue.getQueueId(), activityIdsCollection);
			}
		}
		return activityQueueIdsMap;
	} catch(Exception e) {
		 throw new BbmFinderException(e);
	}
}	
	
	protected void unlinkAllActivityQueues(ID activityId) throws BbmRemoveException {
		if (activityId == null) {
			return;
		}
		try {
			StringBuffer strSQL = new StringBuffer(30);
			strSQL.append("delete ").append(getFieldInfo().getTableName());
			strSQL.append(" where ").append("ACTIVITYID = ").append(activityId.toString());
			m_dmo.executeCommand(strSQL.toString());
		} catch(JdmoException e) {
			throw new BbmRemoveException(e);
		}
	}

	public void unlinkQueueActivities(Collection<ID> queueIds) throws BbmRemoveException {
		if (!queueIds.isEmpty()) {
			try {
				m_dmo.executeCommand("delete " + getFieldInfo().getTableName() + " where " + "QUEUEID IN"
					+ m_dmo.createInClause(queueIds));
			} catch (JdmoException e) {
				throw new BbmRemoveException(e);
			}
		}
	}

}
