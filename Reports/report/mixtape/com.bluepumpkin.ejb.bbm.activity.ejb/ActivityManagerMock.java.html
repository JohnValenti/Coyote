	@Override
	public ID createActivity(Activity activity)
			throws BbmCreateException, BbmCreateDuplicateKeyException, ActivityValidationException, RemoteException {
		ID newActivityId = ID.fromInt(activityCount);
		activityCount++;
		activity.setID(newActivityId);
		activities.put(newActivityId, activity);

		return newActivityId;
	}
