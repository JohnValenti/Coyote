package com.bluepumpkin.ejb.bbm.activity.ejb;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityTimeOffBalanceFieldInfo;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityTimeOffBalance;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;


import java.util.*;

/**
 * Title:        Blue Pumpkin Software Basic Business Model
 * Description:  DAO object to serialize a ActivityTimeOffBalance object

public class ActivityTimeOffBalanceDAO extends DAOBase {

	public static final int BALANCE_NA = -1;
	private static FieldInfo m_fieldInfo = new ActivityTimeOffBalanceFieldInfo();

	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	private static final String m_className = ActivityTimeOffBalanceDAO.class.getName();
	private static Category m_cat = Log.initCategory(m_className);

	public ActivityTimeOffBalanceDAO() {
		super();
	}

	public ActivityTimeOffBalanceDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ValueObjectBase createValueObject() {
		return (new ActivityTimeOffBalance());
	}
	/**
	 * Get ActivityID and ActivityTimeOffBalance pairs as HashMap for the OrganizationId & activityIDs
	 * passed as arguments.
	 *
	 * @param orgId   - Organization ID object.
	 * @param shallow - if true then only the ActivityTimeOffBalance defined at the org
	 *                is returned. If false then ActivityTimeOffBalance that are not defined
	 *                at the org but defined at parent orgs are added in.
	 *                The inherited values used based on the closeness to
	 *                given org. If A is parent of B is parent of C then
	 *                C will return ActivityTimeOffBalance defined a B if they are not
	 *                defined as C and will ignore those same ActivityTimeOffBalance
	 *                if they are also defined at A.
	 * @return We return the returnMap with the mapped values.
	 * @throws BbmFinderException if there is an error while executing database query
	 */
	public Collection getTOBalanceForActivities(Collection empIDs, Collection activityIDCol, Collection actCatIDCol) throws Exception {
		StringBuffer buf = new StringBuffer();
		buf.append(" A.WORKRESOURCEID IN ").append(m_dmo.createInClause(empIDs));
		if ((activityIDCol != null && !activityIDCol.isEmpty()) || (actCatIDCol != null && !actCatIDCol.isEmpty())) {
			buf.append(" AND (");
		if (activityIDCol != null && !activityIDCol.isEmpty()) {
			buf.append(" ACTIVITYID in ").append(m_dmo.createInClause(activityIDCol));
		}
		if ((activityIDCol != null && !activityIDCol.isEmpty()) && (actCatIDCol != null && !actCatIDCol.isEmpty())) {
			buf.append(" OR ");
		}
		if (actCatIDCol != null && !actCatIDCol.isEmpty()) {
			buf.append(" ACTIVITYCATEGORYID in ").append(m_dmo.createInClause(actCatIDCol));
		}
		buf.append(") ");
		}
		return getObjects(buf.toString());
	}
	public void updateOrInsertActivityTOBalance(Collection activityTOBalCol) throws Exception {
		Collection actTOBalToInsert = new ArrayList();
		Collection actTOBalToUpdate = new ArrayList();
		for (Iterator iterator = activityTOBalCol.iterator(); iterator.hasNext();) {
			ActivityTimeOffBalance bal = (ActivityTimeOffBalance) iterator.next();
			if (bal.getID() == null) {
				actTOBalToInsert.add(bal);
			} else {
				actTOBalToUpdate.add(bal);
			}
		}
		updateObjects(actTOBalToUpdate);
		createObjects(actTOBalToInsert);
	}
	public ActivityTimeOffBalance getDefaultActivityTimeOffBalance(ID empID, ID activityID) {
		ActivityTimeOffBalance balDao = new ActivityTimeOffBalance();
		balDao.setActivityID(activityID);
		balDao.setWorkResourceID(empID);
		balDao.setMaxBalance(BALANCE_NA);
		balDao.setMaxCarryOver(BALANCE_NA);
		return balDao;
	}


	public HashMap getActivityTOBalanceMap(Collection empIDs, Collection actIDs, Collection actCatIDs) throws Exception {
		HashMap empActBalMap = new HashMap();
		Collection toBalances = getTOBalanceForActivities(empIDs, actIDs, actCatIDs);
		for (Iterator iterator = toBalances.iterator(); iterator.hasNext();) {
			ActivityTimeOffBalance bal = (ActivityTimeOffBalance) iterator.next();
			HashMap actBalMap = (HashMap) empActBalMap.get(bal.getWorkResourceID());
			if (actBalMap == null) {
				actBalMap = new HashMap();
				empActBalMap.put(bal.getWorkResourceID(), actBalMap);
			}
			actBalMap.put(bal.getActivity_ActCategoryComb(), bal);
		}
		return empActBalMap;
	}

}
