package com.bluepumpkin.ejb.bbm.activity.ejb;

import java.util.Collection;
import java.util.Collections;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityProperties;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityPropertiesFieldInfo;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;
import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;

public class ActivityPropertiesDAO extends DAOBase<ActivityProperties> {
	private static Category log = Log.initCategory(ActivityPropertiesDAO.class.getName());
	private static final String eMessage = "Activity Property not found.  Returning Dummy object. ";
	final static int DUMMY_ID = -10000;
	String sqlStr = null;
	
	// meta data information
	private static FieldInfo m_fieldInfo = new ActivityPropertiesFieldInfo();
	
	//Making this public for the relationship to ShiftEventDAO
	public FieldInfo getFieldInfo() { 
		return m_fieldInfo;
	}		

	public ActivityPropertiesDAO(){
		super();
	}

	public ActivityPropertiesDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ActivityProperties createValueObject() {
		return new ActivityProperties();
	}
	
	public ID getDontExistID() {
		return new ID(DUMMY_ID);
	}
	
	public String getInsertSQLString() {
		return sqlStr;
	}
	
	protected ActivityProperties findActivityProperties(ID id) throws BbmFinderException {
		ActivityProperties activityProperties;
		try {
			activityProperties = getObjectByID(id);
		} catch (BbmObjectNotFoundException e) {
			activityProperties = new ActivityProperties();
			activityProperties.setActivityId(new ID(DUMMY_ID));
			activityProperties.setUseDailyTOPool(false);
			activityProperties.setUseIntervalTOPool(false);
			activityProperties.setTimeoffWithFlex(ActivityProperties.TIME_OFF_TIME);
			activityProperties.setUseAdvancedVTOOptions(false);
			activityProperties.setUseRequiredPayPeriodHours(false);
			log.warn(eMessage + e.getMessage());
		} catch(BbmFinderException e) {
			throw new BbmFinderException(e);
		} 
		return activityProperties;
	}
	
	@SuppressWarnings("unchecked")
	protected Collection<ActivityProperties> findActivitiesProperties(Collection<ID> ids) throws BbmFinderException {
		try {
			if(ids == null || ids.isEmpty()) {
				return Collections.emptyList();
			}
			StringBuilder strWhere = new StringBuilder(150);
			strWhere.append("A.ID in ").append(m_dmo.createInClause(ids));
			return  getObjects(strWhere.toString());
		} catch(JdmoException e) {
			throw new BbmFinderException(e);
		}
	}
	
	protected void insertActivityProperties(ActivityProperties activityProperties) throws BbmCreateException {
		try {
			sqlStr = createInsertQuery(activityProperties);
			m_dmo.executeCommand(sqlStr);
		} catch (JdmoException e) {
			throw new BbmCreateException(e);
		}
	}
	
	private String createInsertQuery(ActivityProperties activityProperties) {
		StringBuilder sqlInsertStr = new StringBuilder();
		
		sqlInsertStr.append(String.format("insert into %s values(", m_fieldInfo.getTableName()));
		FieldInfoEntry[] arrFieldInfo = m_fieldInfo.getFieldInfo();
		
		for (int idx = 0; idx < arrFieldInfo.length; idx++) {
			FieldInfoEntry fieldInfoEntry = arrFieldInfo[idx];
			if (idx > 0) {
				sqlInsertStr.append(",");
			}
			
			switch (fieldInfoEntry.getFieldType()) {
				case PrimitiveTypes.ID: 
					sqlInsertStr.append(activityProperties.getFieldValueID(idx));
					break;
				case PrimitiveTypes.BOOLEAN: 
					sqlInsertStr.append(activityProperties.getFieldValueBoolean(idx) ? 1 : 0);
					break;
				case PrimitiveTypes.SHORT:
					sqlInsertStr.append(activityProperties.getFieldValueShort(idx));
					break;
				case PrimitiveTypes.LONG: 
					sqlInsertStr.append(activityProperties.getFieldValueLong(idx));
					break;
				case PrimitiveTypes.INT: 
					sqlInsertStr.append(activityProperties.getFieldValueInt(idx));
					break;
				case PrimitiveTypes.FLOAT: 
					sqlInsertStr.append(activityProperties.getFieldValueFloat(idx));
					break;
				case PrimitiveTypes.DOUBLE: 
					sqlInsertStr.append(activityProperties.getFieldValueDouble(idx));
					break;
				default : // Default is String
					sqlInsertStr.append("'");
					sqlInsertStr.append((String)activityProperties.getFieldValue(idx));
					sqlInsertStr.append("'");
					break;
			}
		}
		sqlInsertStr.append(")");
		return sqlInsertStr.toString();
	}
}
