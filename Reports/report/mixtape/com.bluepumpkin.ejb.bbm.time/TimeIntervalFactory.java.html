package com.bluepumpkin.ejb.bbm.time;

import java.util.Date;

import com.bluepumpkin.common.datatypes.TimeContext;

/**
 * An instance of this class can be used to obtain TimeInterval instances that
 * share a particular time context. It also provides shared instances of certain
 * frequently-used time intervals.
 */
public class TimeIntervalFactory {
	private final TimeContext m_timeContext;

	private final TimeInterval m_fifteenMinutes;
	private final TimeInterval m_oneHour;
	private final TimeInterval m_oneDay;
	private final TimeInterval m_oneWeek;
	private final TimeInterval m_oneMonth;

	/**
	 * Creates an instance of TimeIntervalFactory that will produce time
	 * intervals with the specified time context.
	 * 
	 * @param timeContext
	 */
	public TimeIntervalFactory(TimeContext timeContext) {
		m_timeContext = timeContext;
		m_fifteenMinutes = createTimeInterval(TimeUnits.Minute, 15);
		m_oneHour = createTimeInterval(TimeUnits.Hour, 1);
		m_oneDay = createTimeInterval(TimeUnits.Day, 1);
		m_oneWeek = createTimeInterval(TimeUnits.Week, 1);
		m_oneMonth = createTimeInterval(TimeUnits.Month, 1);
	}

	/**
	 * Creates a time interval having the specified number of the specified
	 * units (e.g., 15 minutes, 2 weeks).
	 * 
	 * @param units
	 * @param count
	 */
	public TimeInterval createTimeInterval(TimeUnits units, int count) {
		return new TimeInterval(units, count, m_timeContext);
	}

	/**
	 * Creates a time interval spanning from the specified start time to
	 * immediately before the specified end time.
	 * 
	 * @param units
	 * @param startTime
	 * @param endTime
	 */
	public TimeInterval createTimeInterval(TimeUnits units, Date startTime, Date endTime) {
		return new TimeInterval(units, startTime, endTime, m_timeContext);
	}

	/**
	 * Returns a fifteen-minute interval with this factory's time context.
	 */
	public TimeInterval getFifteenMinutes() {
		return m_fifteenMinutes;
	}

	/**
	 * Returns a one-hour interval with this factory's time context.
	 */
	public TimeInterval getOneHour() {
		return m_oneHour;
	}

	/**
	 * Returns a one-day interval with this factory's time context.
	 */
	public TimeInterval getOneDay() {
		return m_oneDay;
	}

	/**
	 * Returns a one-week interval with this factory's time context.
	 */
	public TimeInterval getOneWeek() {
		return m_oneWeek;
	}

	/**
	 * Returns a one-month interval with this factory's time context.
	 */
	public TimeInterval getOneMonth() {
		return m_oneMonth;
	}
}
