package com.bluepumpkin.ejb.facade.hosted.billing;

import com.bluepumpkin.ejb.facade.FacadeManagerFactory;
import com.bluepumpkin.ejb.facade.hosted.billing.ejb.DailyBillingUsageManager;

import java.util.Date;

public class DumpHostedBillingUsage {
	public DumpHostedBillingUsage() {};
	
	public static void dump(Date startTime, Date endTime) throws Exception {
		DailyBillingUsageManager mgr = FacadeManagerFactory.getDailyBillingUsageManager();
		mgr.dumpDailyUsage(startTime, endTime);
	}
}
