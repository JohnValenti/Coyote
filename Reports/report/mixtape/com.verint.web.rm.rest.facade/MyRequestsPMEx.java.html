package com.verint.web.rm.rest.facade;

import java.util.HashSet;
import java.util.Map;
import java.util.Set;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;
import com.bluepumpkin.web.rm.requests.MyRequestsPM;
import com.witness.web.uif.system.RequestContext;

/**
 * Extend MyRequestsPM to get access to which actions are enabled.
 * Later this code needs to be refactored so the logic is shared between UI, REST, and backend.
 */
@SuppressWarnings("serial")
public class MyRequestsPMEx extends MyRequestsPM {

	public MyRequestsPMEx(RequestContext context) {
		super(context);
	}

	Set<String> actions = new HashSet<String>();

	/**
	 * Append Item Action if enabled
	 */
	@Override
	public boolean isItemActionEnabled(RequestAggregate request, String action) {
		boolean enabled = super.isItemActionEnabled(request, action);
		if (enabled) {
			actions.add(action);
		}
		return enabled;
	}

	/**
	 * Sets up the internal state so this can be used by REST services.
	 */
	void setupForRest(OrganizationSetting orgSettings, Map<ID, Boolean> auctionsOpenByID) {
		m_orgSetting = orgSettings;
		if (auctionsOpenByID != null) {
			m_auctionMap = auctionsOpenByID;
		}
	}
}
