public class ForecastInstanceFieldInfo extends FieldInfo {
/**

	 */
	private static final long serialVersionUID = 1L;
	//****@Do not edit till next remark****
	public static final int FORECASTINSTANCE_ID = 0;// Unique identifier for each forecast instance
	public static final int FORECASTINSTANCE_SPID = 1;// Identifier for SP
	public static final int FORECASTINSTANCE_ISBASE = 2;// Specifies if this is a base forecast
	public static final int FORECASTINSTANCE_DESCRIPTION = 3;// Description of the forecast instance
	public static final int FORECASTINSTANCE_CREATED = 4;// Date and time of instance creation
	public static final int FORECASTINSTANCE_CREATEDBY = 5;// User who created the instance
	public static final int FORECASTINSTANCE_MODIFIED = 6;// Date and time of instance modification
	public static final int FORECASTINSTANCE_MODIFIEDBY = 7;// User who modified the instance
	public static final int FORECASTINSTANCE_SID = 8;

	public static final int NUMOFFIELDS = 9;
	private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[NUMOFFIELDS];
	private static String m_strTableName = "FORECASTINSTANCE";
	static {
		m_arrFieldInfo[FORECASTINSTANCE_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
		m_arrFieldInfo[FORECASTINSTANCE_SID] = new FieldInfoEntry("SID", PrimitiveTypes.ID);
		m_arrFieldInfo[FORECASTINSTANCE_SPID] = new FieldInfoEntry("SPID", PrimitiveTypes.ID);
		m_arrFieldInfo[FORECASTINSTANCE_ISBASE] = new FieldInfoEntry("ISBASE", PrimitiveTypes.BOOLEAN);
		m_arrFieldInfo[FORECASTINSTANCE_DESCRIPTION] = new FieldInfoEntry("DESCRIPTION", PrimitiveTypes.STRING);
		m_arrFieldInfo[FORECASTINSTANCE_CREATED] = new FieldInfoEntry("CREATED", PrimitiveTypes.DATE);
		m_arrFieldInfo[FORECASTINSTANCE_CREATEDBY] = new FieldInfoEntry("CREATEDBY", PrimitiveTypes.STRING);
		m_arrFieldInfo[FORECASTINSTANCE_MODIFIED] = new FieldInfoEntry("MODIFIED", PrimitiveTypes.DATE);
		m_arrFieldInfo[FORECASTINSTANCE_MODIFIEDBY] = new FieldInfoEntry("MODIFIEDBY", PrimitiveTypes.STRING);
	}

//****@Do not edit - close****
	public ForecastInstanceFieldInfo() {
		super();
	}

	@Override
	public int getParentIDFieldIndex() {
		return FORECASTINSTANCE_SPID;
	}

	public String getTableName() { return m_strTableName; }
	public int getIDFieldIndex() { return FORECASTINSTANCE_SID; }
	public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }

}
