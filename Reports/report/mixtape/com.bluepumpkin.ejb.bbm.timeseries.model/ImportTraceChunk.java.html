package com.bluepumpkin.ejb.bbm.timeseries.model;

import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;
import java.util.Arrays;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

/**
 * Import TraceChunk used mainly for Import logic to fill in 
 * holes in the timeline.
 * It mainly to support nullify operation, user can specify -1 
 * for a given statistic. Backend will just be able to persist it.
 * @author ssong
 *
 */
public class ImportTraceChunk extends TraceChunk {
	private boolean[] m_TypesSet = new boolean[Trace.MAX_TYPE];
	
	//	 Default constructor
	public ImportTraceChunk() {
		super();
		Arrays.fill(m_TypesSet, false);
	}
	
	public ImportTraceChunk(ID queueID, Date startDate) {
		super(queueID, startDate);
		Arrays.fill(m_TypesSet, false);
	}

	public short[] getTraceTypes() {
		int size = 0;
		for (int i=0; i<m_TypesSet.length; i++) {
			if (m_TypesSet[i])
				size++;
		}
		if (size == 0)
			return null;
		short[] types = new short[size];
		int j=0;
		for (short i=0; i<m_TypesSet.length; i++) {
			if (m_TypesSet[i])
				types[j++] = i;
		}
		return types;
	}

	public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
		m_TypesSet = (boolean[])in.readObject();
		super.readExternal(in);
	}

	public void setTraceValue(short type, float value) {
		if (type < Trace.MAX_TYPE)
			m_TypesSet[type] = true;
		super.setTraceValue(type, value);
	}

	public void setTraceValues(short[] types, float[] values) {
		for (int i=0; i<types.length; i++) {
			if (types[i] < Trace.MAX_TYPE)
				m_TypesSet[types[i]] = true;
		}
		super.setTraceValues(types, values);
	}

	public void setTraceValues(short[] types, int[] values) {
		for (int i=0; i<types.length; i++) {
			if (types[i] < Trace.MAX_TYPE)
				m_TypesSet[types[i]] = true;
		}
		super.setTraceValues(types, values);
	}

	public String toString() {
		StringBuffer sb = new StringBuffer(120);
		sb.append(super.toString());
		for (int i = 0; i < m_TypesSet.length; i++) {
			if (m_TypesSet[i])
				sb.append("\t").append(getTraceName((short)i)).append(" is set");
		}
		return sb.toString();
	}

	public void writeExternal(ObjectOutput out) throws IOException {
		out.writeObject(m_TypesSet);
		super.writeExternal(out);
	}
}
