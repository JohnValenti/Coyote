package com.bluepumpkin.ejb.bbm.timeseries.model;

import java.io.Externalizable;
import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;

import com.bluepumpkin.common.datatypes.ID;

public class CompoundQueueKey implements Externalizable {

	private ID m_cpgID;
	private ID m_mediaID;
	private ID m_queueID;
	private short m_type;

	private String hashStr;

	public CompoundQueueKey() {
	}

	public CompoundQueueKey(ID campaignID, ID mediaID, ID queueID) {
		m_cpgID = campaignID;
		m_mediaID = mediaID;
		m_queueID = queueID;
	}

	public CompoundQueueKey(ID campaignID, ID mediaID, ID queueID, short type) {
		m_cpgID = campaignID;
		m_mediaID = mediaID;
		m_queueID = queueID;
		m_type = type;
	}

	public ID getCampaignID() {
		return m_cpgID;
	}

	public ID getMediaID() {
		return m_mediaID;
	}

	public ID getQueueID() {
		return m_queueID;
	}

	public short getType() {
		return m_type;
	}

	public boolean equals(Object obj) {
		if (obj instanceof CompoundQueueKey) {
			CompoundQueueKey key = (CompoundQueueKey) obj;
			if (IDEql(m_cpgID, key.getCampaignID()) && IDEql(m_mediaID, key.getMediaID())
					&& IDEql(m_queueID, key.getQueueID())) {
				return m_type == key.getType();
			}
		}
		return false;
	}

	public static boolean IDEql(ID id1, ID id2) {
		if (id1 == null && id2 == null) {
			return true;
		}
		if (id1 != null && id2 != null) {
			return id1.equals(id2);
		}
		return false;
	}

	public int hashCode() {
		if (hashStr == null) {
			StringBuffer sb = new StringBuffer(64);
			if (m_cpgID != null) {
				sb.append(m_cpgID);
			}
			sb.append("|");
			if (m_mediaID != null) {
				sb.append(m_mediaID);
			}
			sb.append("|");
			if (m_queueID != null) {
				sb.append(m_queueID);
			}
			sb.append("|").append(m_type);
			hashStr = sb.toString();
		}
		return hashStr.hashCode();
	}

	/*
	 * (non-Javadoc)
	 * 
	 * @see java.io.Externalizable#readExternal(java.io.ObjectInput)
	 */
	public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
		m_cpgID = (ID) in.readObject();
		m_mediaID = (ID) in.readObject();
		m_queueID = (ID) in.readObject();
		m_type = in.readShort();
	}

	/*
	 * (non-Javadoc)
	 * 
	 * @see java.io.Externalizable#writeExternal(java.io.ObjectOutput)
	 */
	public void writeExternal(ObjectOutput out) throws IOException {
		out.writeObject(m_cpgID);
		out.writeObject(m_mediaID);
		out.writeObject(m_queueID);
		out.writeShort(m_type);
	}
}
