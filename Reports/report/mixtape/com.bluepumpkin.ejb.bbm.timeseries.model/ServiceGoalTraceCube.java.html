package com.bluepumpkin.ejb.bbm.timeseries.model;

import java.util.Arrays;
import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.base.BbmTimeSeriesException;
import com.bluepumpkin.ejb.bbm.timeseries.util.TraceUtil;

/**
 * Title: Blue Pumpkin Software Basic Business Model Description:
 * ServiceGoalTraceCube object, mainly used by UI to load bunch of Trace Data
 * for a given period Copyright: Copyright (c) 2002 Company: Blue Pumpkin
 * Software, inc
 * 
 * @author Sheng Song
 * @version 2.0
 */
@SuppressWarnings("serial")
public class ServiceGoalTraceCube extends TraceCube {
	// static final long serialVersionUID = -3835436339421180351L;

	public static final short[] TYPES = { Trace.PCA, Trace.ASA, Trace.ABANDONMENT, Trace.DEADLINE_TIME, Trace.MAX_DIALS };
	public static final short[] PCA_ABN = { Trace.PCA, Trace.ABANDONMENT };
	public static final short[] ASA_ABN = { Trace.ASA, Trace.ABANDONMENT };

	// Default constructor
	public ServiceGoalTraceCube() {
		super();
	}

	public ServiceGoalTraceCube(short[] types) {
		super(types);
	}

	/**
	 * TraceCube is a container for types of Trace, given period for backend
	 * 
	 * @param ID
	 *            , queueID
	 * @param Date
	 *            , startDate,
	 * @param Date
	 *            , endDate,
	 */
	public ServiceGoalTraceCube(ID queueID, Date startDate, Date endDate) throws BbmTimeSeriesException {
		this(queueID, startDate, endDate, TimeZoneUtil.GMT_TIMEZONE);
	}

	/**
	 * TraceCube is a container for types of Trace, given period for backend
	 * 
	 * @param ID
	 *            , queueID
	 * @param Date
	 *            , startDate,
	 * @param Date
	 *            , endDate,
	 * @param TimeZone
	 */
	public ServiceGoalTraceCube(ID queueID, Date startDate, Date endDate, TimeZone timeZone)
			throws BbmTimeSeriesException {
		super(queueID, startDate, endDate, null);
	}

	/**
	 * TraceCube is a container for types of Trace,
	 * 
	 * @param ID
	 *            , queueID
	 * @param Date
	 *            , startDate,
	 * @param Date
	 *            , endDate,
	 * @param short[], the Trace Types to be included
	 */
	public ServiceGoalTraceCube(ID queueID, Date startDate, Date endDate, short[] types) throws BbmTimeSeriesException {
		this(queueID, startDate, endDate, types, TimeZoneUtil.GMT_TIMEZONE);
	}

	/**
	 * TraceCube is a container for types of Trace, given period
	 * 
	 * @param ID
	 *            , queueID
	 * @param Date
	 *            , startDate,
	 * @param Date
	 *            , endDate,
	 * @param short[], the Trace Types to be included
	 * @param TimeZone
	 */
	public ServiceGoalTraceCube(ID queueID, Date startDate, Date endDate, short[] types, TimeZone timeZone)
			throws BbmTimeSeriesException {
		super(queueID, startDate, endDate, types);
		checkTraceTypeSupport(types);
	}

	// Base TraceCube supports all types of Trace
	protected short[] getSupportedTraceType() {
		return TYPES;
	}

	public void setValidation(boolean flag) {
		m_EnforceValidation = flag;
	}

	/**
	 * Initialize the TraveValue Array based on Trace Type, with a constant
	 * value
	 * 
	 * @param short, type
	 * @param int, value
	 */
	public void initTraceValue(short type, int value) {
		if (m_Types == null || getTraceValuesD() == null) {
			return;
		}
		super.initTraceValue(type, value);
		switch (type) {
		case Trace.PCA:
			super.initTraceValue(Trace.ASA, Trace.TRACENA);
			break;
		case Trace.ASA:
			super.initTraceValue(Trace.PCA, Trace.TRACENA);
			break;
		}
	}

	/**
	 * Set TraceValue, starting from a given date
	 * 
	 * @param short, type
	 * @param int, value
	 * @param int, offset
	 */
	public void setTraceValue(short type, int value, int offSet) {
		int[] values = { value };
		setTraceValue(type, values, offSet);
	}

	/**
	 * Initialize the TraveValue Array based on Trace Type, from a given
	 * absoluate OffSet Will mutually clean PCA for ASA, vice versa
	 * 
	 * @param short, type
	 * @param int[], value
	 * @param int, offset
	 */
	public void setTraceValue(short type, int[] value, int offSet) {
		if (m_Types == null || getTraceValuesD() == null || value == null) {
			return;
		}
		int[] cleanUp = null;
		super.setTraceValue(type, value, offSet);
		if (m_EnforceValidation) {
			switch (type) {
			case Trace.PCA:
				if (getTraceValueD(Trace.ASA) != null) {
					cleanUp = new int[value.length];
					Arrays.fill(cleanUp, Trace.TRACENA);
					super.setTraceValue(Trace.ASA, cleanUp, offSet);
				}
				break;
			case Trace.ASA:
				if (getTraceValueD(Trace.PCA) != null) {
					cleanUp = new int[value.length];
					Arrays.fill(cleanUp, Trace.TRACENA);
					super.setTraceValue(Trace.PCA, cleanUp, offSet);
				}
				break;
			}
		}
	}

	/**
	 * Set value without validation check, for internal routine
	 */
	public void initTraceValue(short type, int[] value, Date start) {
		int offSet = TraceUtil.gapOffSet(m_StartDate, m_StartOffSet, start);
		super.setTraceValue(type, value, offSet);
	}

	/**
	 * Set TraceValue, starting from a given date
	 * 
	 * @param short, type
	 * @param int, value
	 * @param Date
	 *            , start
	 */
	public void setTraceValue(short type, int value, Date start) {
		int[] values = { value };
		setTraceValue(type, values, start);
	}

	/**
	 * Set TraceValue, starting from a given date
	 * 
	 * @param short, type
	 * @param int[], value
	 * @param Date
	 *            , start
	 */
	public void setTraceValue(short type, int[] value, Date start) {
		if (m_Types == null || getTraceValuesD() == null || value == null || start == null) {
			return;
		}
		// If try to insert Trace after the start date, it should fail
		if (m_RawStartDate == null || m_RawStartDate.after(start)) {
			return;
		}
		int offSet = TraceUtil.gapOffSet(m_StartDate, m_StartOffSet, start);
		setTraceValue(type, value, offSet);
	}

	protected String getCubeName() {
		return "ServiceGoalTraceCube";
	}

	// make a clone from given parameters
	public TraceCube newInstance(ID queueID, Date startDate, Date endDate, short[] types) throws BbmTimeSeriesException {
		return new ServiceGoalTraceCube(queueID, startDate, endDate, types);
	}
}
