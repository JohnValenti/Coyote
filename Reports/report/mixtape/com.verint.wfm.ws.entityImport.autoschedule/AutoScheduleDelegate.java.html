package com.verint.wfm.ws.entityImport.autoschedule;

public class AutoScheduleDelegate {

	private CampaignManager m_campaignManager;
	private SchedulingPeriodServiceDelegate m_schePeriodServiceDelegate;
	private static final Category CAT = Log
			.initCategory(AutoScheduleDelegate.class.getName());

	public AutoScheduleDelegate() {
		init();
	}

	/**
	 * Auto Schedules the Scheduling Periods. This method auto schedules only
	 * those scheduling periods that have not already being scheduled.
	 * 
	 * @param schedulingPeriodIDs
	 * @param systemTypeName
	 */
	public List<SchedulingNotTriggeredInfo> autoSchedule(
			Collection<SchedulingPeriodExternalNaturalKey> schedulingPeriodIDs,
			boolean overrideSchedules, String systemTypeName)
			throws ServiceApplicationException {
		List<SchedulingNotTriggeredInfo> messages = new ArrayList<SchedulingNotTriggeredInfo>();
		if (schedulingPeriodIDs == null || schedulingPeriodIDs.isEmpty()) {
			CAT.debug("No scheduling periods to schedule");
			return messages;
		}

		if (getCampaignManager() != null) {
			SystemType systemType = SystemType.valueOf(systemTypeName);
			Map<SchedulingPeriodExternalNaturalKey, ID> externalVsLocalId = m_schePeriodServiceDelegate
					.convertToID(systemType, schedulingPeriodIDs,
							new HashMap<ID, Campaign>());
			Map<ID, SchedulingPeriodExternalNaturalKey> localVsExternalID = new HashMap<ID, SchedulingPeriodExternalNaturalKey>();

			// handle if entity does not exist.
			List<ID> localIds = validateID(schedulingPeriodIDs,
					externalVsLocalId, localVsExternalID, messages);
			if (localIds.size() > 0) {
				Map<ID, SchedulingPeriodExternalNaturalKey> scheduleRequestsVsSPID = ScheduleUtil
						.getInstance().createSchedulingRequests(localIds,
								localVsExternalID, messages);
				ScheduleUtil.getInstance().scheduleRequests(
						new ArrayList<ID>(scheduleRequestsVsSPID.keySet()),
						new ArrayList<SchedulingPeriodExternalNaturalKey>(
								scheduleRequestsVsSPID.values()),overrideSchedules, messages);
			}
		} else {
			CAT.error("The Campaign Manager is not available. Could not trigger autoschedule");
			throw new ServiceApplicationException(
					"Could not trigger the schedule creation due to system error");
		}

		return messages;
	}
	
	
	public SchedulingPeriodExternalNaturalKeyMapJAXBType doesScheduleExist(
			Collection<SchedulingPeriodExternalNaturalKey> spIDs,
			String systemTypeName) throws ServiceApplicationException {
		Map<SchedulingPeriodExternalNaturalKey, Boolean> spKeyVsResult = new HashMap<SchedulingPeriodExternalNaturalKey, Boolean>();
		
		if (spIDs == null || spIDs.isEmpty()) {
			CAT.debug("No scheduling periods to check the if schedule exists");
			return new SchedulingPeriodExternalNaturalKeyMapJAXBType();
		}
	
		if (getCampaignManager() != null) {
			SystemType systemType = SystemType.valueOf(systemTypeName);
			Map<SchedulingPeriodExternalNaturalKey, ID> externalVsLocalId = m_schePeriodServiceDelegate
					.convertToID(systemType, spIDs, new HashMap<ID, Campaign>());
			
			// iterate through IDs and check
			for (Iterator<SchedulingPeriodExternalNaturalKey> it = externalVsLocalId
					.keySet().iterator(); it.hasNext();) {
				SchedulingPeriodExternalNaturalKey externalID = it.next();
				ID localID = externalVsLocalId.get(externalID);
				if (localID != null) {
					spKeyVsResult.put(externalID, SchedulingRequestUtil.getInstance()
							.doesScheduleExist(localID.toInt()));
				} else {
					spKeyVsResult.put(externalID, false);
				}
			}

		} else {
			CAT.error("The Campaign Manager is not available. Could not check schedules");
			throw new ServiceApplicationException(
					"Could not check the schedules due to system error");
		}

		// Map needs to be wrapped by a custom JAXB type
		return new SchedulingPeriodExternalNaturalKeyMapJAXBType(spKeyVsResult);
	}
	
	

	/**
	 * Checks if there is a local ID for every Scheduling period External Id and
	 * returns the list of all non-null local Ids. This method adds exception
	 * messages for non-existent ids. This method populated the Map of
	 * out_localVsExternalID.
	 * 
	 * @param schedulingPeriodIDs
	 * @param externalVsLocalId
	 * @param out_localVsExternalID
	 * @param messages
	 * @return
	 */
	private List<ID> validateID(
			Collection<SchedulingPeriodExternalNaturalKey> schedulingPeriodIDs,
			Map<SchedulingPeriodExternalNaturalKey, ID> externalVsLocalId,
			Map<ID, SchedulingPeriodExternalNaturalKey> out_localVsExternalID,
			List<SchedulingNotTriggeredInfo> messages) {
		List<ID> localIds = new ArrayList<ID>();
		for (Iterator<SchedulingPeriodExternalNaturalKey> it = schedulingPeriodIDs
				.iterator(); it.hasNext();) {
			SchedulingPeriodExternalNaturalKey key = it.next();
			ID localId = externalVsLocalId.get(key);
			if (localId == null) {
				SchedulingNotTriggeredInfo message = new SchedulingNotTriggeredInfo(
						key);
				message.addNoSuchSPExternalIdMessage();
				messages.add(message);
			} else {
				localIds.add(localId);
				out_localVsExternalID.put(localId, key);
			}

		}

		return localIds;

	}

	public List<SchedulingStatus> getSchedulingStatus(
			Collection<SchedulingPeriodExternalNaturalKey> schedulingPeriodIDs,
			String systemTypeName) throws ServiceApplicationException {
		List<SchedulingStatus> statuses = new ArrayList<SchedulingStatus>();
		if (schedulingPeriodIDs == null || schedulingPeriodIDs.isEmpty()) {
			CAT.debug("No Scheduling Periods to check the scheduling status");
			return statuses;
		}
		if (getCampaignManager() != null) {
			SystemType systemType = SystemType.valueOf(systemTypeName);
			Map<SchedulingPeriodExternalNaturalKey, ID> externalVsLocalId = m_schePeriodServiceDelegate
				.convertToID(systemType, schedulingPeriodIDs, new HashMap<ID, Campaign>());
		
			for (Iterator<SchedulingPeriodExternalNaturalKey> it = schedulingPeriodIDs.iterator(); it.hasNext();) {
				SchedulingPeriodExternalNaturalKey key = it.next();
				SchedulingStatus status = new SchedulingStatus();
				statuses.add(status);
				status.setSchedulingPeriodExternalNaturalKey(key);
		
				ID localId = externalVsLocalId.get(key);
				if (localId != null) {
					WSAutoScheduleRequest request = SchedulingRequestUtil.getInstance()
						.getSchedulingRequestWithCurrentStatusBySP(localId);
		
					if (request != null) {
						status.setStatusCode(request.getSchedulingStatus());
						status.setProgress(request.getProgress());
						// TODO might have to split to array
						status.setMessage(request.getStatusMessage());
						status.setMessageCode((String) request
								.getFieldValue(WSAutoScheduleRequestFieldInfo.STATUSMESSAGETYPE));
					} else {
						status.setStatusCode(WSAutoScheduleRequest.SCHEDULE_STATUS_UNKNOWN);
					}
				} else {
					status.setStatusCode(WSAutoScheduleRequest.SCHEDULE_STATUS_UNKNOWN);
				}
		
			}
		}
		return statuses;
	}

	public void stopScheduling(
			Collection<SchedulingPeriodExternalNaturalKey> schedulingPeriodIDs,
			String systemTypeName) throws ServiceApplicationException {
		
		if (schedulingPeriodIDs == null || schedulingPeriodIDs.isEmpty()) {
			CAT.debug("No scheudling periods to stop scheduling");
			return;
		}
		if (getCampaignManager() != null) {
			SystemType systemType = SystemType.valueOf(systemTypeName);
			Map<SchedulingPeriodExternalNaturalKey, ID> externalVsLocalId = m_schePeriodServiceDelegate
					.convertToID(systemType, schedulingPeriodIDs,
							new HashMap<ID, Campaign>());
			List<ID> localIds = new ArrayList<ID>();
			// handle if entity does not exist.
			for (Iterator<SchedulingPeriodExternalNaturalKey> it = schedulingPeriodIDs
					.iterator(); it.hasNext();) {
				SchedulingPeriodExternalNaturalKey key = it.next();
				ID localId = externalVsLocalId.get(key);
				if (localId != null) {
					localIds.add(localId);
				} else {
					CAT.error("Could not stop scheduling for Scheduling Period of external Id "
							+ key.toString());
				}

			}
			SchedulingRequestUtil.getInstance().stopScheduling(localIds, SchedulingMode.SCHEDULE, false, null);
		}

	}

	private CampaignManager getCampaignManager() {
		if (m_campaignManager == null) {
			createCampaignManager();
		}

		return m_campaignManager;
	}

	private void createCampaignManager() {
		try {
			m_campaignManager = WfmManagerFactory.getCampaignManager();
		} catch (BbmEJBCreateException e) {
			CAT.error("Error getting the CampaignManager", e);
		}
	}

	private void init() {
		createCampaignManager();
		m_schePeriodServiceDelegate = new SchedulingPeriodServiceDelegate();
		m_schePeriodServiceDelegate.postConstruct();
	}

	@PreDestroy
	public void preDestroy() {
		m_campaignManager = null;
		m_schePeriodServiceDelegate = null;
	}

}
