package com.verint.wfm.ws.entityImport.queue;

import java.util.List;

import com.verint.wfm.ws.entityImport.EntityNotPersistedInfo;

/**
 * For each queue not persisted by createOrUpdateActivities(), one of these will be in the list returned This is
 * returned in lieu of an exception, because we always want to continue on to all the activities
 * 
 * 
 * @see EntityImportService.createOrUpdateActivities()
 * 
 */
public class QueueNotPersistedInfo extends EntityNotPersistedInfo<String> {
	private static final long serialVersionUID = 1L;

	public QueueNotPersistedInfo() {
		super();
	}

	public QueueNotPersistedInfo(String queueExternalSourceId) {
		super(queueExternalSourceId);
	}

	// -----------------------------------------
	public String getQueueExternalSourceId() {
		return super.getEntityExternalSourceId();
	}

	/**
	 * Dummy here on server, but needed for SOAP stub
	 */
	public void setQueueExternalSourceId(String queueExternalSourceId) {
		// do nothing
	}

	public void addQueueServerExceptionMessage(String exception) {
		getMessages().add(
				String.format("Exception is %s. The server could not persist queue with external id %s.", exception,
						getExternalIdSummary()));
	}

	public void addExternalIdMissingMessage(String skillName) {
		getMessages().add(String.format("External id was missing for queue with name %s.", skillName));
	}

	public void addExternalIdLengthExceededMessage(int actualLength, int lengthLimit) {
		getMessages().add(
				String.format("The external id %s length %d exceeded the limit of %d.", getExternalIdSummary(),
						actualLength, lengthLimit));
	}

	public void addNameMissingMessage() {
		getMessages().add(String.format("Name was missing for queue with external id %s.", getExternalIdSummary()));
	}
	
	public void addNameHasInvalidCharactersMessage(String invalidChars) {
		getMessages()
				.add(String
						.format("The characters %s are not allowed in the queue name. The queue %s has invalid characters in the name.",
								invalidChars, getExternalIdSummary()));
	}

	public void addNameLengthExceededMessage(int actualLength, int lengthLimit) {
		getMessages().add(
				String.format("For queue with external id %s the name length %d exceeded the limit of %d.",
						getExternalIdSummary(), actualLength, lengthLimit));
	}

	public void addNameNotUniqueMessage(String ambiguousQueueWFOId, String name) {
		getMessages().add(
				String.format("For queue with external id %s, name %s conflicted with another queue with wfo sid %d.",
						getExternalIdSummary(), name, ambiguousQueueWFOId));
	}

	public void addMediaIdMissingMessage() {
		getMessages().add(
				String.format("For queue with external id %s, the media id was missing.", getExternalIdSummary()));
	}

	public void addOrganizationNameMissingMessage() {
		getMessages().add(
				String.format("For queue with external id %s, the organization name was missing.",
						getExternalIdSummary()));
	}

	public void addNoSuchOrganizatonNameMessage(String organizationName) {
		getMessages()
				.add(String
						.format("For queue with external id %s, the organization specified to have name %s does not exist.",
								getExternalIdSummary(), organizationName));
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public List<String> getMessages() {
		return super.getMessages();
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public void setMessages(List<String> messages) {
		// Do nothing
	}
}
