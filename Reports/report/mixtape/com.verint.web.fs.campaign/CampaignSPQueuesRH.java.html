package com.verint.web.fs.campaign;

import java.util.Collection;
import java.util.HashMap;

import com.bluepumpkin.common.base.BPException;

import com.bluepumpkin.ejb.bbm.campaign.ejb.CampaignManager;

import com.witness.web.uif.system.RequestContext;

public class CampaignSPQueuesRH extends WorkpaneRequestHandler {
	private static final Category log = Log.initCategory(CampaignSPQueuesRH.class.getName());
	protected static final String FORM_PAGE = "formPage";
	
	@Override
	public final void processRequest(RequestContext context) throws Exception {
		// Process requested action
		if (isAction(context, PageAction.ADD)) {
			processAdd(context);
		} else if (isAction(context, PageAction.SAVE)) {
			CampaignSPQueueslListPM pm = CampaignSPQueueslListPM.getInstance(context);
			pm.extractParameters(context.getRequest());
			processSave(context, pm);
		} else if (isAction(context, PageAction.DELETE)) {
			CampaignSPQueueslListPM pm = new CampaignSPQueueslListPM(context);
			pm.extractParameters(context.getRequest());
			processDelete(context, pm);
		} else if (isAction(context, PageAction.CANCEL)) {
			processLoadID(context);
		} else if (isAction(context, PageAction.EDIT)) {
			prepareForNextPage(context, null, FORM_PAGE);
		} else if(isAction(context, PageAction.PREVIOUS)) {
			CampaignSPQueueslListPM pm = new CampaignSPQueueslListPM(context);
			pm.extractParameters(context.getRequest());
			processPageLeft(context, pm);
		} else if(isAction(context, PageAction.NEXT)) {
			CampaignSPQueueslListPM pm = new CampaignSPQueueslListPM(context);
			pm.extractParameters(context.getRequest());
			processPageRight(context, pm);
		} else if(isAction(context, PageAction.SELECT_PAGE)) {
			CampaignSPQueueslListPM pm = new CampaignSPQueueslListPM(context);
			pm.extractParameters(context.getRequest());
			processPageSelect(context, pm);
		} else {
			String selectId = context.getParameter(WorkpaneFieldNames.SELECTED_ID_FN);
			if(selectId != null && selectId.startsWith(CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX)) {
				prepareForNextPage(context, null, FORM_PAGE);
			} else {
				prepareForNextPage(context, null, FORM_PAGE);
			}
		}
	}
	
	protected void processSave(RequestContext context, CampaignSPQueueslListPM pageModel) {
		try {
			CampaignManager camManager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
			HashMap<SPQueue, Collection<ID>> skillSPQueuemap = pageModel.getSkillForSPQueues();
			for(SPQueue spqueue: skillSPQueuemap.keySet()) {
				Collection<ID> skillIDs = skillSPQueuemap.get(spqueue);
				spqueue.setSkills(skillIDs);
				camManager.updateSPQueue(spqueue);
			}
			
			if (pageModel.getCombinedSPQueue() != null) {
				camManager.updateSPQueue(pageModel.getCombinedSPQueue());
			}
			
			pageModel.addPageMessage(new Message(Message.INFO_TYPE,
					context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
							FsWebBundleKey.FS_QUEUE_SETTINGS_SAVED_SUCCESSFULLY)));

			//save spqueue, using the cached pagemode, need to reload spqueues. 
			pageModel.resetListPC();
		}
		catch (Exception e) {
			log.l7dError(BbmWebLogBundleKey.UNABLE_TO_SAVE_DATA, e);
		}	
		prepareForNextPage(context, pageModel, FORM_PAGE);
	}
	
	protected void processAdd(RequestContext context) {
		// No implementation?
	}
	
	protected void processDelete(RequestContext context, CampaignSPQueueslListPM pageModel) {
		try {
			CampaignManager camManager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
			Collection<ID> idsToDelete = pageModel.getPerfromActionOnIDsAsCollection();
			for (ID id: idsToDelete){
				camManager.deleteSPQueue(id);
			}
			pageModel.addPageMessage(new Message(Message.INFO_TYPE,
					context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
							FsWebBundleKey.FS_QUEUES_REMOVED_SUCCESSFULLY)));
		} catch (Exception e){
			log.l7dError(FsWebLogBundleKey.UNABLE_TO_DELETE_SP_QUEUE, e);
		}
		
		prepareForNextPage(context, pageModel, FORM_PAGE);
	}
	
	protected void processLoadID(RequestContext context) {
		// No implementation?	
	}
	
	/**
	 * Prepare Model and determine which page to go next
	 */
	protected void prepareForNextPage(RequestContext context, WorkpanePageModel model, String nextPage) {
		if (model!=null) {
			context.setPageModel(model);
		}

		if (nextPage!=null) {
			setNextPage(context, nextPage);
		} else {
			setNextPage(context, FORM_PAGE);
		}
	}
	
	protected void processPageLeft(RequestContext context, CampaignSPQueueslListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexBackward();
		prepareForNextPage(context, null, FORM_PAGE);
	}
	
	protected void processPageRight(RequestContext context, CampaignSPQueueslListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexForward();
		prepareForNextPage(context, null, FORM_PAGE);
	}
	
	protected void processPageSelect(RequestContext context, CampaignSPQueueslListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.setPagePaginationAndEditIndices(
				pageModel.getPaginationControl().getSelectedPaginationPageIndex());
		prepareForNextPage(context, null, FORM_PAGE);
	}

	@Override
	public WorkpanePageModel getPageModel(RequestContext context) {
		// No implementation?
		return null;
	}

	@Override
	public void processCustomRequest(RequestContext context)
			throws WebException {
		// No implementation?
	}
}
