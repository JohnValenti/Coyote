package com.verint.web.fs.campaign;

import java.io.Serializable;
import java.util.Comparator;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.workload.model.QueueType;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;

public class QueueTypeComparator implements Comparator<QueueType>, Serializable {
	private static final long serialVersionUID = 1L;

	private Localizer localizer;
	private String queueTypeNormal;
	private String queueTypeDistributed;
	private String queueTypeVirtual;
	private String queueTypeProcess;
	
	public QueueTypeComparator(Localizer localizer) {
		this.localizer = localizer;
		queueTypeNormal = localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.QUEUE_TYPE_NORMAL);
		queueTypeDistributed = localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.QUEUE_TYPE_DISTRIBUTED);
		queueTypeVirtual = localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.QUEUE_TYPE_VIRTUAL);
		queueTypeProcess = localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.QUEUE_TYPE_PROCESS);
	}
	
	public int compare(QueueType o1, QueueType o2) {
		if (o1 == null && o2 == null) return 0;
		if (o1 == null) return -1;
		if (o2 == null) return 1;
		if (o1.equals(o2)) return 0;
		return localizer.getCollator().compare(getLocalizedStringForQueueType(o1),
				getLocalizedStringForQueueType(o2));
	}
	
	private String getLocalizedStringForQueueType(QueueType queueType) {
		switch (queueType) {
		case Normal:
			return queueTypeNormal;
		case Distributed:
			return queueTypeDistributed;
		case Virtual:
			return queueTypeVirtual;
		case Process:
			return queueTypeProcess;
		default:
			return "";
		}
	}
}
