package com.verint.web.fs.campaign;

import com.witness.web.uif.jsp.HtmlDiv;
import com.witness.web.uif.pagecomponent.PageComponent;
import com.witness.web.uif.pagemodel.WorkpaneListPM;
import com.witness.web.uif.system.RequestContext;

/**
 * A workpane with a summary section, a dividing section below that,
 * and a main section below that.  Extend from WorkpaneListPM.
 */
public abstract class WorkpaneListWithSummaryPM extends WorkpaneListPM {

	protected PageComponent summaryComponent;
	protected PageComponent listComponent;
	
	public WorkpaneListWithSummaryPM(RequestContext context, int modelType) {
		super(context, modelType);
	}
	
	public void setSummaryComponent(PageComponent summaryComponent) {
		this.summaryComponent = summaryComponent;
		removeChildComponent(summaryComponent);
		addChildComponent(summaryComponent);
	}
	
	public String getSummaryComponent() {
		if (summaryComponent == null) return "";
		return summaryComponent.getHtmlDisplay();
	}
	
	public String getDividerComponent() {
		if (summaryComponent == null) return "";
		HtmlDiv dividerDiv = new HtmlDiv();
		dividerDiv.setStyle("height:15px; width:100%");
		dividerDiv.setId(summaryComponent.getName() + "_dividerDiv");
		return dividerDiv.toString();
	}
	
	public void setListComponent(PageComponent listComponent) {
		this.listComponent = listComponent;
		removeChildComponent(listComponent);
		addChildComponent(listComponent);
	}
	
	public String getListComponent() {
		if (listComponent == null) return "";
		return listComponent.getHtmlDisplay();		
	}
}
