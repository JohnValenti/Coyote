	Collection<Queue> m_Queues = null;
	
	public CampaignSPQueuesListPopupPC(RequestContext context, Collection<Queue> queues) {
		super(context);
		// TODO Auto-generated constructor stub
		m_Queues = queues;
		initHeader();
		
		initRows();
	}
	
	private void initHeader() {
		TableHeaderPC header = getHeader();
		Localizer localizer = m_context.getLocalizer();
		header.setPartialSortable(true);
		
		header.addColumnHeaderData(FsWebBundleKey.SCHEDULINGPERIOD_NAME, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_NAME), true);

		// Enable Auto Sorting
		List<String> sortColumns = new ArrayList<String>();
		sortColumns.add(FsWebBundleKey.SCHEDULINGPERIOD_NAME);
		header.setDefaultSortColumns(sortColumns);
	}

	private void initRows() {
		Collection<MultiColumnNodeData> rows = super.getListModel();
	
		try {
			WorkloadManager wm = WfmManagerFactory.getWorkloadManager(m_context.isInWhatIfMode());
			Collection parentQueues = null;
			Set<ID> parents = new HashSet<ID>();
			for (Queue queue : m_Queues) {			
				if (queue != null && queue.getParentID() != null){
					parents.add(queue.getParentID());
				}
			}
			
			if (parents.size() > 0) {
				parentQueues = wm.getQueuesByIDs(parents);
			}
			
			for (Queue queue : m_Queues) {
				if (queue != null) {
					DefaultMultiColumnNodeData rowData = new DefaultMultiColumnNodeData(queue.getID());
					ID parentID = queue.getParentID();
					if (parentID != null && parentQueues != null) {
						for (Object parentQueueObject: parentQueues) {
							Queue parentQueue = (Queue)parentQueueObject;
							if (parentQueue.getID().equals(parentID)) {
								rowData.add(parentQueue.getName() + " / " + queue.getName());
							}							
						}
					} else { 
						rowData.add(queue.getName());
					}
					
					rows.add(rowData);
				}	
			}
		} catch (Exception e) {
			System.out.println(e.getMessage());
		}
	}
}
