  public Integer getSLWaitingTime( ID queID, Date startDate )
      throws FacadeException, RemoteException
  {
    Integer retVal = null;
    Jdmo jdmo = new Jdmo();

    try
    {
        StringBuffer sql = new StringBuffer();
        // Change starts for QA97322
        sql.append("select sp.SLANSWERWAITINGTIME from sp s,SPQUEUE sp where (sp.queueid=(select q1.id from queue q1 where q1.id =(select isnull(q.parentqueueid, q.id) from Queue q where q.sid =");
        sql.append(queID);
        sql.append(" ) and q1.queuetype=2 and s.isskillbased=1) or sp.queueid=(select q.ID from Queue q where q.sid =");
        sql.append(queID);
        sql.append(" and s.ISSKILLBASED =1) or (sp.queueid is NULL and MEDIAID='ID_PHONE') and s.ISSKILLBASED =0) and SPID = s.id  and s.FROMDATE <= ? AND TODATE >= ? order by s.FROMDATE ");
        // Change ends for QA97322
        

      JdmoQuery jq1 = jdmo.createQuery(sql.toString(), Jdmo.PARAM_QUERY);

      java.sql.Timestamp newStartDate = new java.sql.Timestamp(startDate.getTime());
      java.sql.Timestamp newEndDate = new java.sql.Timestamp(startDate.getTime() + 900000L);

      jq1.setParTimestamp( 1, newStartDate );
     jq1.setParTimestamp( 2, newEndDate );

      JdmoRowset rs = jdmo.createRowset(jq1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

      while (rs.next())
      {
        retVal =  new Integer( rs.getInt(1) );
        break;
      }

      rs.close();

      return retVal;

    }
    catch( Exception ex )
    {
        throw new FacadeException(ex);
    }

    finally
    {
      jdmo.cleanUp();
    }

  }

  public void calculateStaffing(Collection dsColl,
                                       Date startDate,
                                       Date endDate,
                                       Date lastQueryDate,
                                       long lookBackDays,
                                       long empChunkSize,
                                      boolean enableProfiling ,boolean isIncremental)
      throws FacadeException, RemoteException
  {
	  try {
    StaffingCalculator calc = new StaffingCalculator( this.getCategory() );
    calc.calculateStaffing(dsColl,
                           startDate,
                           endDate,
                           lastQueryDate,
                           lookBackDays,
                           empChunkSize,
                           enableProfiling,
                           isIncremental);
	}
	catch (com.bluepumpkin.ejb.bbm.base.FacadeException fexp) {
		throw new FacadeException(fexp);
	}
  }
