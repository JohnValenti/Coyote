package com.verint.web.rest.wfm.exception;


public class EntityCreateDuplicateKeyException extends EntityException {

	public EntityCreateDuplicateKeyException(String entityName, String id) {
		super(entityName, id);
	}

	@Override
	public String getTitle() {
		return "EntityCreateDuplicateKeyException";
	}

	@Override
	public String getDetail() {
		return "A duplicate key exception occurred when saving entity " + entityName + " with ID " + id;
	}
}
