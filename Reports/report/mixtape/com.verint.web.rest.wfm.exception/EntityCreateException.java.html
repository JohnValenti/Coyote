package com.verint.web.rest.wfm.exception;

import com.verint.web.rest.wfm.entity.Entity;
import com.verint.web.rest.wfm.util.IdUtil;

public class EntityCreateException extends EntityException {

	private final Entity entity;

	public EntityCreateException(Entity entity) {
		super(entity.getClass().getSimpleName(), IdUtil.toString(entity.getID()));

		this.entity = entity;
	}

	@Override
	public String getTitle() {
		return "EntityCreateException";
	}

	@Override
	public String getDetail() {
		return "There was a problem creating entity: " + entity.toString();
	}
}
