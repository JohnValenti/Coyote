package com.bluepumpkin.web.fs.schedule.preferences;

/**
 * Title:        SimpleSchedulePreferenceMH
 * Description:  Handle EJB interaction for Simple Schedule Preference
 *  Schedule Summary model handler handles EJB interactions for the
 *  user's schedule summary page model.  Since there are no update functions,
 *  the only interaction is retrieval of the value objects needed for the
 *  page model.
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * @author       Chris King; Pavel Bosin
 * @version 1.0
 */
public class SchedulePreferenceMH extends ModelHandler {
	//keys for the data HashMap
	public static final String SIMPLE_PREFS_KEY = "SIMPLE_PREFS_KEY";
	public static final String ORG_TZ_KEY = "ORG_TZ_KEY";
	public static final String PREFS_HISTORY_KEY = "PREFS_HISTORY_KEY";
	public static final String OTVTO_PREFS_KEY = "OTVTO_PREFS_KEY";
	/**
	 * get a HashMap with data for Simple Schedule Preferences page
	 * @param employeeID
	 */
	public static HashMap getSimplePrefs(RequestContext context, ID employeeID)	throws RemoteException, BbmException, BbmEJBCreateException {
		HashMap theData = new HashMap();
		if (employeeID != null) {
			SchedulePrefManager mgr = WfmManagerFactory.getSchedulePrefManager(context.isInWhatIfMode());

			//get simple prefs collection - it should have only one element in London and no dates selection
			Collection simpleSchedPrefs = mgr.getSimpleSchedulePreferences(employeeID, null,null);
			if (!simpleSchedPrefs.isEmpty()) {
				theData.put(SIMPLE_PREFS_KEY, simpleSchedPrefs.iterator().next());
			}

			TimeZone orgTz = OrganizationModelHandler.getEmployeeOrgTZ(context, employeeID);
			theData.put(ORG_TZ_KEY, orgTz);

			// get Over-time / Voluntary Time-off object by employee ID
			EmployeeOTVTOPref otvtoSchedPref = mgr.getEmployeeOTVTOPref(employeeID);
			if (otvtoSchedPref != null) {
				theData.put(OTVTO_PREFS_KEY, otvtoSchedPref);
			}

		}
		return theData;
	}
	
	public static boolean isDayOffSame(RequestContext context, Collection colEmpIDS)	throws RemoteException, BbmException, BbmEJBCreateException {
		if (colEmpIDS != null ) {
			
			if(colEmpIDS.isEmpty()){
				return false;
			}
			SchedulePrefManager mgr = WfmManagerFactory.getSchedulePrefManager(context.isInWhatIfMode());
			Collection baseSimpleSchedPrefs = null;
			for (Iterator iterator = colEmpIDS.iterator(); iterator.hasNext();) {
				ID employeeID = (ID) iterator.next();
				
				Collection simpleSchedPrefs = mgr.getSimpleSchedulePreferences(employeeID, null,null);
				if (!simpleSchedPrefs.isEmpty()) {
					if(baseSimpleSchedPrefs != null){
						//if(!simpleSchedPrefs.equals(baseSimpleSchedPrefs)){
						SimpleSchedulePreference basepref = (SimpleSchedulePreference)baseSimpleSchedPrefs.iterator().next();
						SimpleSchedulePreference pref = (SimpleSchedulePreference)simpleSchedPrefs.iterator().next();
						if(basepref == null || pref == null){
							return false;
						}
						Collection basedayoffpref = basepref.getDayOffPreference();
						Collection dayoffpref = pref.getDayOffPreference();
						if(basedayoffpref.equals(dayoffpref)){
							//do nothing.
						}
						else{
							return false;
						}
					}
					else{
						baseSimpleSchedPrefs = simpleSchedPrefs;
					}
				}
				else
					return false;
			}
			return true;
		}
		return false;
	}
	
	public static boolean isOTVTOOffSame(RequestContext context, Collection colEmpIDS)	throws RemoteException, BbmException, BbmEJBCreateException {
		if (colEmpIDS != null ) {
			
			if(colEmpIDS.isEmpty()){
				return false;
			}
			SchedulePrefManager mgr = WfmManagerFactory.getSchedulePrefManager(context.isInWhatIfMode());
			EmployeeOTVTOPref baseotvtopref = null;
			for (Iterator iterator = colEmpIDS.iterator(); iterator.hasNext();) {
				ID employeeID = (ID) iterator.next();
				
				EmployeeOTVTOPref otvtopref = mgr.getEmployeeOTVTOPref(employeeID);
				if (otvtopref!=null) {
					if(baseotvtopref != null){
						//if(!otvtopref.equals(baseotvtopref)){
						if (otvtopref.getOtAfter() != baseotvtopref
								.getOtAfter()
								|| otvtopref.getOtBefore() != baseotvtopref
										.getOtBefore()
								|| otvtopref.getVtoAtBeginning() != baseotvtopref
										.getVtoAtBeginning()
								|| otvtopref.getVtoAtEnd() != baseotvtopref
										.getVtoAtEnd()) {
							return false;
						}
					}
					else{
						baseotvtopref = otvtopref;
					}
				}
				else
					return false;
			}
			return true;
		}
		return false;
	}

	public static void saveSimplePrefs(RequestContext context, SimpleSchedulePreference simpleSchedPref, EmployeeOTVTOPref otvtoPref)
			throws RemoteException, BbmEJBCreateException, MultiUserException, BbmException { 
		SchedulePrefManager mgr = WfmManagerFactory.getSchedulePrefManager(context.isInWhatIfMode());
		mgr.updateSimpleSchedulePreference(simpleSchedPref);
		mgr.updateEmployeeOTVTOPref(otvtoPref);
	}

	public static void saveSimplePrefsForMultipleEmps(RequestContext context,
			Collection colEmpIDs,SimpleSchedulePreference simpleSchedPref, EmployeeOTVTOPref otvtoPref)
			throws RemoteException, BbmEJBCreateException, MultiUserException, BbmException { 
		SchedulePrefManager mgr = WfmManagerFactory.getSchedulePrefManager(context.isInWhatIfMode());
		mgr.updateSimpleSchedulePreferenceMultiEmps(colEmpIDs, simpleSchedPref, otvtoPref);
	}
}
