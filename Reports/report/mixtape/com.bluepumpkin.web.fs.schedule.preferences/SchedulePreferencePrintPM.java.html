import com.witness.web.uif.system.RequestContext;


/**
 * This class is used as a shell for the mockup pages.
 */
public class SchedulePreferencePrintPM extends SchedulePreferencePM {

	/**
	 * Constructs a schedule view page model shell.
	 */
	public SchedulePreferencePrintPM(RequestContext context)	{
		super(context, SchedulePreferencePM.VIEW_TYPE_PRINT);
	}//constructor


	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static SchedulePreferencePrintPM getInstance(RequestContext context)	{
		SchedulePreferencePrintPM model = null;

		if (context.getPageModel() instanceof SchedulePreferencePrintPM) {
			model = (SchedulePreferencePrintPM)context.getPageModel();
		} else {
			model = new SchedulePreferencePrintPM(context);
			model.extractParameters(context.getRequest());
			context.setPageModel(model);
		}
		model.populateModel(context);
		return model;
	}//getInstance
