package com.bluepumpkin.ejb.bbm.corbafacade.ejb;

import com.bluepumpkin.common.base.BPException;

public class CorbaFacadeException extends BPException
{
	/**
	 * 
	 */

	// define types
	public static final int TYPE_SYSTEM_EXCEPTION = 0;
	public static final int TYPE_BBM_FINDER_EXCEPTION = 1;
	public static final int TYPE_BBM_CREATE_EXCEPTION = 2;
	public static final int TYPE_BBM_REMOVE_EXCEPTION = 3;

	public static final int TYPE_BBM_UPDATE_DUPLICATE_AGENT_ID_EXCEPTION = 12;

	public int errType = TYPE_SYSTEM_EXCEPTION;

  public CorbaFacadeException ()
  {
			this.errType = TYPE_SYSTEM_EXCEPTION;

  }

	public int getExceptionType()

	}
}
