package com.verint.web.fs.optimization;

import java.io.Serializable;
import java.util.Collections;
import java.util.List;

public class MultiQueueDTO<TSingleQueueDTO extends SingleQueueDTO> implements Serializable {
	
	private static final long serialVersionUID = -7303643749329488136L;
	
	private final List<TSingleQueueDTO> m_queueDtos;
	
	public MultiQueueDTO(List<TSingleQueueDTO> queueDtos) {
		m_queueDtos = queueDtos;
	}
	
	public List<TSingleQueueDTO> getSingleQueueDtos() {
		return Collections.unmodifiableList(m_queueDtos);
	}
}
