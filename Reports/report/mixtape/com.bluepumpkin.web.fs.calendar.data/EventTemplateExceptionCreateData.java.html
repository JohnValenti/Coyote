public class EventTemplateExceptionCreateData implements Serializable {
    protected CalendarEventTemplate m_template;
    protected Collection m_newWorkResourceIDs;
    protected Date m_newStartDate, m_origStartDate;
    protected int m_newDuration;
	protected boolean m_newLocked;
	protected boolean m_isCreateExceptionByLock = false;
    protected CalendarEventAssignment m_exceptionEvent = null;
    protected CalendarEventTemplate m_exceptionTemplate = null;
	public EventTemplateExceptionCreateData(CalendarEventTemplate template, Collection newWorkResourceIDs, Date origStartDate, Date newStartDate, int newDuration) {
		this(template, newWorkResourceIDs, origStartDate, newStartDate, newDuration, false);
	}
    public EventTemplateExceptionCreateData(CalendarEventTemplate template, Collection newWorkResourceIDs, Date origStartDate, Date newStartDate, int newDuration, boolean newIsLocked) {
        m_template = template;
        m_newWorkResourceIDs = newWorkResourceIDs;
        m_origStartDate = origStartDate;
        m_newStartDate = newStartDate;
        m_newDuration = newDuration;
		m_newLocked = newIsLocked;
    }
    
    public EventTemplateExceptionCreateData(CalendarEventTemplate template, Collection newWorkResourceIDs, Date origStartDate, Date newStartDate, int newDuration, CalendarEventTemplate excepTemplate, boolean newIsLocked) {
        m_template = template;
        m_newWorkResourceIDs = newWorkResourceIDs;
        m_origStartDate = origStartDate;
        m_newStartDate = newStartDate;
        m_newDuration = newDuration;
		m_newLocked = newIsLocked;
		m_exceptionTemplate = excepTemplate;
    }

    public EventTemplateExceptionCreateData(CalendarEventTemplate template, Date origStartDate, CalendarEventAssignment exceptionEvent) {
        m_template = template;
        m_origStartDate = origStartDate;
        m_exceptionEvent = exceptionEvent;
    }

    public CalendarEventTemplate getTemplate() {
        return m_template;
    }

    public ID getTemplateID() {
        return m_template.getID();
    }

    public Collection getNewWorkResourceIDs() {
        if (m_exceptionEvent != null) {
            return m_exceptionEvent.getWorkResourceIDs();
        }
        return m_newWorkResourceIDs;
    }

    public Date getNewStartDate() {
        if (m_exceptionEvent != null) {
            return m_exceptionEvent.getStartTime();
        }
        return m_newStartDate;
    }

    public int getNewDuration() {
        if (m_exceptionEvent != null) {
            return m_exceptionEvent.getDuration();
        }
        return m_newDuration;
    }

	public boolean getNewLocked() {
		if (m_exceptionEvent != null) {
			return m_exceptionEvent.isLocked();
		}
		return m_newLocked;
	}

    public Date getOrigStartDate() {
        return m_origStartDate;
    }
    
    public CalendarEventTemplate getExceptionTemplate(){
    	return m_exceptionTemplate;
    }

    public CalendarEventAssignment getExceptionEvent() {
        if (m_exceptionEvent != null) {
            return m_exceptionEvent;
        }
        CalendarEventAssignment calendarEventAssignment = null;
        switch (m_template.getEventType()) {
            case CalendarEvent.EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT:
                calendarEventAssignment = new CalendarEvent();
                ((CalendarEvent) calendarEventAssignment).setOverlapType(m_template.getOverlapType());
                break;
            case CalendarEvent.EVENT_TYPE_TIME_OFF:
                calendarEventAssignment = new TimeOffEvent();
                if (m_template instanceof RecurringEventTemplate) {
                    ((TimeOffEvent) calendarEventAssignment).setCountsMinutesTowardsRules(((RecurringEventTemplate)m_template).getCountsMinutesTowardsRules());
                    ((TimeOffEvent) calendarEventAssignment).setTimeOffRule(((RecurringEventTemplate) m_template).getTimeOffRule());
                } else  {
                    ((TimeOffEvent) calendarEventAssignment).setCountsMinutesTowardsRules(0);
                    ((TimeOffEvent) calendarEventAssignment).setTimeOffRule(TimeOffEvent.TIME_OFF_RULE_USE_MINSTOWARDSRULES);
                }
                break;
            case CalendarEvent.EVENT_TYPE_UNAVAILABILITY:
                calendarEventAssignment = new UnavailabilityEvent();
                break;
        }
        if (calendarEventAssignment != null) {
            calendarEventAssignment.setStartTime(m_newStartDate);
			calendarEventAssignment.setLocked(m_newLocked);
            calendarEventAssignment.setDuration(m_newDuration);
            calendarEventAssignment.setActivityID(m_template.getActivityID());
            Iterator itEmpIDs = m_newWorkResourceIDs.iterator();
            while (itEmpIDs.hasNext()) {
                ID nextEmpID = (ID) itEmpIDs.next();
                calendarEventAssignment.addWorkResourceID(nextEmpID);
            }
            calendarEventAssignment.setDescription(m_template.getDescription());
        }
        return calendarEventAssignment;
    }
    
    public boolean isCreateExceptionByLock() {
    	return this.m_isCreateExceptionByLock;
    }
    
    public void setCreateExceptionByLock(boolean value) {
    	this.m_isCreateExceptionByLock = value;
    }
    
    public void setLock(boolean value) {
    	this.m_newLocked = value;
    }
}
