import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class QueueAttributeFilterFieldInfo extends FieldInfo {

    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// field id constants
    public static final int QAF_ID = 0;
    public static final int QAF_NAME = 1;
    public static final int QAF_ISPUBLIC = 2;
    public static final int QAF_USERID = 3;
    public static final int QAF_SHOWROOTQUEUESONLY = 4;
    
    
    public static final int QAF_NUMOFFIELDS = 5;


    private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[QAF_NUMOFFIELDS];
    private static String m_strTableName = "QUEUEATTRIBUTEFILTER";

    // static initialization
    static {
      m_arrFieldInfo[QAF_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
      m_arrFieldInfo[QAF_NAME] = new FieldInfoEntry("NAME", PrimitiveTypes.STRING);
      m_arrFieldInfo[QAF_ISPUBLIC] = new FieldInfoEntry("ISPUBLIC", PrimitiveTypes.BOOLEAN);
      m_arrFieldInfo[QAF_USERID] = new FieldInfoEntry("USERID", PrimitiveTypes.ID);
      m_arrFieldInfo[QAF_SHOWROOTQUEUESONLY] = new FieldInfoEntry("SHOWROOTQUEUESONLY", PrimitiveTypes.BOOLEAN);
    }

    public QueueAttributeFilterFieldInfo() {
        super();
    }

    public int getParentIDFieldIndex() {
    	return QAF_USERID;
    }
    
    // implement abstract methods in parent class
    public String getTableName() { return m_strTableName; }
    public int getIDFieldIndex() { return QAF_ID; }
    public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }

}
