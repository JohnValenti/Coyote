package com.bluepumpkin.ejb.bbm.activity;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.Pair;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityValidationException;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbBundleKey;
import com.witness.ejb.bbm.copy.CopyableObjectPlugin;
import com.witness.ejb.bbm.copy.ejb.CopyDAO;
import com.witness.ejb.bbm.copy.model.CopyResult;
import com.witness.ejb.core.security.PrivilegeKeys;

/**
 * @author dsu
 * @version 1.0
 * @Title: ActiviyCopyPlugin
 * @Description:
 * @Copyright: Copyright (c) Aug 15, 2006
 * @Company: Witness Systems
 */
public class ActivityCopyPlugin extends CopyableObjectPlugin {
  /**
   *
   */
  private static final long serialVersionUID = 1L;
  private static final Pair TEMP_ID_PAIR = new Pair();

  /*
   * (non-Javadoc)
   *
   * @see
   * com.witness.ejb.bbm.copy.CopyableObjectPlugin#copy(com.bluepumpkin.common
   * .datatypes.ID, com.bluepumpkin.common.datatypes.ID, boolean,
   * com.witness.ejb.bbm.copy.ejb.CopyDAO, java.lang.String)
   */
  public CopyResult copy(ID idActivity, ID idOrg, boolean bForce, CopyDAO pDAO, String sCopyPrefix)
      throws BbmCreateException, BbmFinderException, ActivityValidationException {
    return ActivityCopyUtil.copyActivity(idActivity, idOrg, bForce, pDAO, TEMP_ID_PAIR, sCopyPrefix);
  }

  /*
   * (non-Javadoc)
   *
   * @see
   * com.witness.ejb.bbm.copy.CopyableObjectPlugin#isMultiTargetCopyAllowed()
   */
  public boolean isMultiTargetCopyAllowed() {
    return false;
  }

  /*
   * (non-Javadoc)
   *
   * @see
   * com.witness.ejb.bbm.GenericObjectPlugin#convertType(com.bluepumpkin.common
   * .localization.Localizer, boolean)
   */
  public String convertType(Localizer localizer, boolean isPlural) {
    String rbKey = isPlural ? BbmEjbBundleKey.OBJECT_ACTIVITY_PLURAL : BbmEjbBundleKey.OBJECT_ACTIVITY;
    return localizer.i18n(BbmEjbBundleKey.BUNDLE_NAME, rbKey);
  }

  /*
   * (non-Javadoc)
   *
   * @see com.witness.ejb.bbm.GenericObjectPlugin#getModificationPrivilegeID()
   */
  public ID getModificationPrivilegeID() {
    return PrivilegeKeys.CORE_VIEWACTIVITIES_ID;
  }
}
