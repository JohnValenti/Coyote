package com.bluepumpkin.web.rm.requests.flextime;

import java.util.ArrayList;
import java.util.List;
import java.util.Map;
import com.bluepumpkin.ejb.rm.requests.common.model.Request;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.requests.AgentsRequestsListConfig;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.listconfig.ListConfigFP;

public class AgentsFlexTimeRequestsListConfig extends AgentsRequestsListConfig {
	private List<ListConfigFP> allFPs;
	private Map<String, ListConfigFP> fpMap;

	public AgentsFlexTimeRequestsListConfig() {
		initFPs();
	}

	/**
	 * Initialize Field Properties
	 */
	@SuppressWarnings("unchecked")
	private void initFPs() {
		if (allFPs == null) {
			allFPs = new ArrayList<ListConfigFP>();
			allFPs.add(createFP(Request.SORT_LAST_NAME_FIRST, UIFWebBundleKey.NAME, true));
			allFPs.add(createFP(Request.SORT_STATUS, UIFWebBundleKey.STATUS, true));
			allFPs.add(createFP(Request.SORT_CREATED, UIFWebBundleKey.SUBMITTED, true));
			allFPs.add(createFP(Request.SORT_MODIFIED, UIFWebBundleKey.LAST_MODIFIED, true));
			allFPs.add(createFP(Request.SORT_TO_SENIORITY, UIFWebBundleKey.SENIORITY, true));
			allFPs.add(createFP(Request.SORT_EMP_RANK, UIFWebBundleKey.RANK, true));

			allFPs.add(createFP(Request.SORT_TO_ACTIVITY, UIFWebBundleKey.TYPE, true));

			allFPs.add(createFPforRM(Request.SORT_TO_FIRST_START, RmWebBundleKey.FLEX_TIMEOFF_START_DATE_LABEL, true));
			allFPs.add(createFPforRM(Request.SORT_TO_FIRST_END, RmWebBundleKey.FLEX_TIMEOFF_END_DATE_LABEL, true));

			allFPs.add(createFPforRM(Request.SORT_TO_WAITLIST_TOCHOICE_REQ_LENGTH, RmWebBundleKey.FLEX_TIMEOFF_LENGTH_LABEL, false));
			allFPs.add(createFPforRM(Request.SORT_TO_WAITLIST_TOCHOICE_ALLOCATION_HOURS, RmWebBundleKey.TIMEOFF_HOURS_ACCOUNTED_LABEL, false));

			allFPs.add(createFP(Request.SORT_TO_POOL_NAME, RmWebBundleKey.TIME_OFF_POOL, RmWebBundleKey.BUNDLE_NAME, false));

			allFPs.add(createFP(Request.SORT_CS_STARTTIME, RmWebBundleKey.MAKEUP_START_DATE_LABEL, RmWebBundleKey.BUNDLE_NAME, true) );
			allFPs.add(createFP(Request.SORT_CS_ENDTIME, RmWebBundleKey.MAKEUP_END_DATE_LABEL, RmWebBundleKey.BUNDLE_NAME, true) );

			allFPs.add(createFP(Request.SORT_CS_ACTIVITYID, RmWebBundleKey.RM_SA_FILTER_MAKEUP_ACTIVITY, RmWebBundleKey.BUNDLE_NAME, true));


			addCommonFPs(allFPs);
		}

		if (fpMap == null) {
			fpMap = createFPMap(allFPs);
		}
	}

	/**
	 * Return All List Config Field Properties (ListConfigFP)
	 */
	@SuppressWarnings("rawtypes")
	public List getAllFPs() {
		return allFPs;
	}

	/**
	 * Return a Map of All List Config Field Properties (ListConfigFP) where key is ListConfigFP.getFieldName();
	 */
	@SuppressWarnings("rawtypes")
	public Map getFPMap() {
		return fpMap;
	}
}
