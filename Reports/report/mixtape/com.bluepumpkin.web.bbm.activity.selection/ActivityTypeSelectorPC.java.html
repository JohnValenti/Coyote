package com.bluepumpkin.web.bbm.activity.selection;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;
import com.bluepumpkin.web.bbm.activity.ActivityTypeModelHandler;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.witness.web.uif.jsp.HtmlImage;
import com.witness.web.uif.keys.ImageFileID;
import com.witness.web.uif.pagecomponent.PageComponent;
import com.witness.web.uif.pagecomponent.selection.IDsSelectorPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;


/**
 * Title:        ActivityTypeSelectorPC
 * Description:  Activity Type Selector
 * Copyright:    Copyright (c) 2004
 * Company:      Blue Pumpkin Software, inc
 * @author       Alvin Cham
 * Created on September 17, 2004, 11:10 AM
 * @version      1.0
 */
public class ActivityTypeSelectorPC extends IDsSelectorPC {
	public static final String REQUEST_HANDLER = "activity_type_selector_pu";

	/** Creates a new instance of Selector */
	public ActivityTypeSelectorPC(RequestContext context, ToolbarPC toolbar) {
		super(context, toolbar, BbmWebBundleKey.BUNDLE_NAME);
		setName("ActivityTypeSelectorPC");
	}

	/**
	 * Add Popup Argument for Selector
	 *
	 * @param pc - The Page Component where the Popup Argument will be added
	 * @param name - The unique name associated with the Popup Argument
	 * @param isMultiSelectEnabled Specifies whether the Selection is Multiple
	 *  or multiple.
	 * @param isRefreshOnClose Specifies whether Opener is refresh on Close
	 */
	public static void addSelectorPopupArg(PageComponent pc, String name,
	boolean isMultiSelectEnabled, boolean isRefreshOnClose) {
		addSelectorPopupArg(pc, name, REQUEST_HANDLER, SELECTOR_WIDTH_HEIGHT,
				isMultiSelectEnabled, isRefreshOnClose);
	}

	/**
	 * add Selector Popup Argument
	 */
	protected void addSelectorPopupArg() {
		addSelectorPopupArg(this, m_name, m_isMultiSelectEnabled, m_isRefreshOnClose);
	}

	/**
	 * Return Label for Selector
	 */
	public String getLabel() {
		return i18n(m_bundle, BbmWebBundleKey.BBM_ORG_ACT_CATEGORIES);
	}

	/**
	 * Fetch Data From Collection of IDs
	 * @param ids - Collection of IDs
	 * @return Collection of Data
	 */
	protected Collection fetchDataFromIDs(Collection ids) throws Exception {
		Collection actTypes = new ArrayList();

		Collection visibleActTypes = ActivityTypeModelHandler.getAllActivityTypesVisibleToUser(m_context, m_context.getUser());
		for (Iterator iter = visibleActTypes.iterator(); iter.hasNext();) {
			ActivityCategory actType = (ActivityCategory)iter.next();

			if (ids.contains(actType.getID()))
				actTypes.add(actType);
		}

		return actTypes;
	}

	/**
	 * Append Display and Hidden Value from Specified Object
	 * @param obj - The Object containing the Values
	 * @param displaySB - The Display Value StringBuffer
	 * @param hiddenSB - The Hidden Value StringBuffer
	 */
	protected void appendValues(Object obj, StringBuffer displaySB, StringBuffer hiddenSB) {
		ActivityCategory actType = (ActivityCategory)obj;
		displaySB.append(actType.getName());
		hiddenSB.append(actType.getID());
	}

	/**
	 * Return Button Image
	 */
	protected HtmlImage getButtonImage() {
		HtmlImage btnImage = new HtmlImage(ImageFileID.GENERIC_SELECTOR_BUTTON,
										   ImageFileID.GENERIC_SELECTOR_BUTTON_WIDTH,
										   ImageFileID.GENERIC_SELECTOR_BUTTON_HEIGHT,
										   i18n(m_bundle, BbmWebBundleKey.ACTIVITY_TYPE_SELECTOR_PAGE_TITLE));
		return btnImage;
	}

}
