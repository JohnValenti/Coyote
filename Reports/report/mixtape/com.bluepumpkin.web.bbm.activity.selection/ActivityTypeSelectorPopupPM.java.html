package com.bluepumpkin.web.bbm.activity.selection;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.activity.tree.ActivityTypeTreePC;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.pagemodel.SelectorPopupWindowPM;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        ActivityTypeSelectorPopupPM
 * Description:  Activity Type Selector Popup Page
 * Copyright:    Copyright (c) 2004
 * Company:      Blue Pumpkin Software, Inc
 * @author       Alvin Cham
 * @version      1.0
 *
 * Created on September 17, 2004, 12:27 PM
 */
public class ActivityTypeSelectorPopupPM extends SelectorPopupWindowPM {
	private static final Category m_log = Log.initCategory(ActivityTypeSelectorPopupPM.class.getName());
	private ActivityTypeTreePC m_treePC;

	/** Creates a new instance of SelectorPopupPM */
	public ActivityTypeSelectorPopupPM(RequestContext context) {
		super(context);

		m_treePC = new ActivityTypeTreePC(m_context);
		addChildComponent(m_treePC);

		init(ActivityTypeSelectorPC.REQUEST_HANDLER,
		     JavaScriptFileID.MEDIATOR_POPUP_TREE_SELECTOR,
		     BbmWebBundleKey.BUNDLE_NAME,
		     BbmWebBundleKey.ACTIVITY_TYPE_SELECTOR_PAGE_TITLE);
	}

	/**
	 * Return an instance of Page Model
	 */
	public static PageModel getInstance(RequestContext context) {
		ActivityTypeSelectorPopupPM model = null;

		PageModel cachedModel = context.getPageModel();
		if (cachedModel instanceof ActivityTypeSelectorPopupPM) {
			model = (ActivityTypeSelectorPopupPM)cachedModel;
		} else {
			// Need to create new Model
			model = new ActivityTypeSelectorPopupPM(context);
			model.extractParameters(context.getRequest());
			context.setPageModel(model);
		}

		model.populateModel();
		return model;
	}

	/**
	 * Make sure that the model is populated
	 */
	private void populateModel() {
		if (m_isInitialized) return; else m_isInitialized=true;

		try {
			m_treePC.init();
		} catch (Exception ex) {
			handleUnableToLoadDataError(m_log, ex);
		} finally {
			m_treePC.setMultiRowSelectable(isMultiSelectEnabled());
			setDisplayPC(m_treePC);
		}
	}
}
