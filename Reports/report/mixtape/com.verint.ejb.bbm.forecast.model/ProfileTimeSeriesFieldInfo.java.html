package com.verint.ejb.bbm.forecast.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class ProfileTimeSeriesFieldInfo extends FieldInfo {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;

	public static enum Column {
		ID(new FieldInfoEntry("ID", PrimitiveTypes.ID)),
		ForecastProfileID(new FieldInfoEntry("FORECASTPROFILEID", PrimitiveTypes.ID)),
		SequenceNumber(new FieldInfoEntry("SEQNUM", PrimitiveTypes.INT)),
		QueueID(new FieldInfoEntry("QUEUEID", PrimitiveTypes.ID)),
		AverageHandlingTime(new FieldInfoEntry("AHT", PrimitiveTypes.FLOAT)),
		Volume(new FieldInfoEntry("CALLVOLUME", PrimitiveTypes.FLOAT)),
		ConnectRate(new FieldInfoEntry("CRATE", PrimitiveTypes.FLOAT)),
		RightPartyConnectRate(new FieldInfoEntry("RPCRATE", PrimitiveTypes.FLOAT)),
		RightPartyConnectAverageHandlingTime(new FieldInfoEntry("RPCAHT", PrimitiveTypes.FLOAT)),
		Created(new FieldInfoEntry("CREATED", PrimitiveTypes.DATETIME)),
		CreatedBy(new FieldInfoEntry("CREATEDBY", PrimitiveTypes.STRING)),
		Modified(new FieldInfoEntry("MODIFIED", PrimitiveTypes.DATETIME)),
		ModifiedBy(new FieldInfoEntry("MODIFIEDBY", PrimitiveTypes.STRING))
		;
		static {
			int ctr = 0;
			for (Column fie : Column.values()) fie.index = ctr++;
		}
		
		private int index;
		private FieldInfoEntry fie;
		private Column(FieldInfoEntry fie) {
			this.fie = fie;
		}
		
		public FieldInfoEntry getFieldInfoEntry() {
			return this.fie;
		}
		public int getIndex() {
			return index;
		}
	}

	private static final FieldInfoEntry[] fieldInfoEntries;
	static {
		int ctr = 0;
		fieldInfoEntries = new FieldInfoEntry[Column.values().length];
		for (Column c : Column.values()) {
			fieldInfoEntries[ctr++] = c.getFieldInfoEntry();
		}
	}
	
	public FieldInfoEntry[] getFieldInfo() {
		return fieldInfoEntries;
	}

	public int getIDFieldIndex() {
		return Column.ID.getIndex();
	}

	public String getTableName() {
		return "PROFILETIMESERIES";
	}

	@Override
	public int getParentIDFieldIndex() {
		return Column.ForecastProfileID.getIndex();
	}

}
