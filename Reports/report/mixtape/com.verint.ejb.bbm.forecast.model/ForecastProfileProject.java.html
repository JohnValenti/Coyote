package com.verint.ejb.bbm.forecast.model;

import java.util.Date;
import com.bluepumpkin.common.datatypes.*;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class ForecastProfileProject extends ValueObjectBase {
	
	/**
	 * 
	 */
	private static final long serialVersionUID = 9144189204654845101L;
	private static FieldInfo m_fieldInfo = new ForecastProfileProjectFieldInfo();
	protected FieldInfo getFieldInfo() {return m_fieldInfo;}
	
	public ForecastProfileProject()
	{
		super();
	}
	
	public ID getProjectID(){
		return getFieldValueID(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_ID);
	}
	public void setProjectID(ID id)
	{
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_ID,id);
	}
	public ID getQueueID(){
		return getFieldValueID(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_QUEUEID);
	}
	public void setQueueID(ID id)
	{
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_QUEUEID,id);
	}
	public ID getForecastProfileID(){
		return getFieldValueID(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_FORECASTPROFILEID);
	}
	public void setForecastProfileID(ID id)
	{
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_FORECASTPROFILEID,id);
	}
	/**
	 * Return the start date/time of the forecast for this project queue.
	 */
	public Date getStartTime(){
		return getFieldValueDate(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_STARTTIME);
	}
	public void setStartTime(Date date)
	{
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_STARTTIME,date);
	}
	/**
	 * Return the end date/time of the forecast for this project queue.
	 */
	public Date getEndTime(){
		return getFieldValueDate(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_ENDTIME);
	}
	public void setEndTime(Date date)
	{
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_ENDTIME,date);
	}
	/**
	 * Return the forecasted number of hundredths of minutes required for this project queue between
	 * the defined start and end time; that is, 100 times the number of minutes.
	 */
	public int getTotalDuration(){
		return getFieldValueInt(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_TOTALDURATION);
	}
	public void setTotalDuration(int duration)
	{
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_TOTALDURATION,duration);
	}
	public String getCreatedBy() {
		return getFormattedFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_CREATEDBY);
	}
	
	public void setCreatedBy(String createdBy) {
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_CREATEDBY, createdBy);
	}
	
	public Date getModifiedDate() {
		return getFieldValueDate(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_MODIFIED);
	}
	
	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_MODIFIED, modifiedDate);
	}
	
	public String getModifiedBy() {
		return getFormattedFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_MODIFIEDBY);
	}
	
	public void setModifiedBy(String modifiedBy) {
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_MODIFIEDBY, modifiedBy);
	}
	public Date getCreatedDate() {
		return getFieldValueDate(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_CREATED);
	}
	public void setCreatedDate(Date createdDate) {
		setFieldValue(ForecastProfileProjectFieldInfo.FORECASTPROFILEPROJECT_CREATED, createdDate);
	}
}
