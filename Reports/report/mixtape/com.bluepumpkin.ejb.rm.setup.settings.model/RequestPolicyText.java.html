import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.rm.Log;



public class RequestPolicyText extends ValueObjectBase {
    
    private static FieldInfo m_fieldInfo = new RequestPolicyTextFieldInfo();
    
    private static Category m_cat = Log.initCategory(RequestPolicyText.class.getName());
    
    protected Category getCategory() {
        return m_cat;
    }
    
    /**
     * Default constructor for the class. 
     */
    public RequestPolicyText() {
        super();
    }

    // Database attribute index for Organization ID
    public static final int REQUESTPOLICYTEXT_I_ORGANIZATIONID = 1;

    // Database attribute index for RequestType
    public static final int REQUESTPOLICYTEXT_S_REQUESTTYPE = 2;

    // Database attribute index for Fragment Number
    public static final int REQUESTPOLICYTEXT_N_FRAGMENTNUM = 3;

    // Database attribute index for PolicyText
    public static final int REQUESTPOLICYTEXT_S_POLICYTEXT = 4;
    
    // Database attribute index for PolicyText
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }
    
    // getters
    
    /**
     * Get the OrganizationId attribute for the as ID object. 
     * @return ID - OrganizationId  for object instance.
     */
    public ID getOrganizationId() {
        return getFieldValueID(
        RequestPolicyTextFieldInfo.REQUESTPOLICYTEXT_I_ORGANIZATIONID);
    }
    
    /**
     * Get the RequestType attribute for the as string object. 
     * @return String - RequestType for object instance.
     */
    public String getRequestType() {
        return getFormattedFieldValue(
        RequestPolicyTextFieldInfo.REQUESTPOLICYTEXT_S_REQUESTTYPE);
    }
    
    /**
     * Get the Fragment Number attribute for the as int. 
     * @return int - Fragment Number for request policy text as int.
     */
    public int getFragmentNum() {
        return getFieldValueInt(RequestPolicyTextFieldInfo.REQUESTPOLICYTEXT_N_FRAGMENTNUM);
    }
    
    /**
     * Get the Rquest policy text value for fragment number attribute. 
     * @return String - string value as PolicyText for fragment number.
     */
    public String getPolicyText() {
        return getFormattedFieldValue(
        RequestPolicyTextFieldInfo.REQUESTPOLICYTEXT_S_POLICYTEXT);
    }
    
    /**
     * Set the Organization id for the object. 
     * @param ID - ID object as Organization id.
     */
    public void setOrganizationId(ID pID) {
        setFieldValue(RequestPolicyTextFieldInfo.REQUESTPOLICYTEXT_I_ORGANIZATIONID, pID);
    }
    
    /**
     * Set the Request Type for policy text. 
     * @param pRequestType - String object as Request Type.
     */
    public void setRequestType(String pRequestType) {
        setFieldValue(RequestPolicyTextFieldInfo.REQUESTPOLICYTEXT_S_REQUESTTYPE, pRequestType);
    }
    
    /**
     * Set the FragmentNumber for Request policy text. 
     * @param int - FragmentNumber for the request polict text.
     */
    public void setFragmentNum(int pFragmentNum) {
        setFieldValue(RequestPolicyTextFieldInfo.REQUESTPOLICYTEXT_N_FRAGMENTNUM, pFragmentNum);
    }
    
    /**
     * Set the Policy text. 
     * @param String - string value as policy text.
     */
    public void setPolicyText(String  pPolicyText) {
        setFieldValue(RequestPolicyTextFieldInfo.REQUESTPOLICYTEXT_S_POLICYTEXT, pPolicyText);
    }
    
}
