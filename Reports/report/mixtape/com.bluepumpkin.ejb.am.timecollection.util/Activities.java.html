public final class Activities {
	
	//Activities
	public static final int NOACTIVITY = 7;
	public static final int STARTSHIFT = -1001;
  
	public static String getActivityName(int id) {
		if (id == NOACTIVITY) return "End Shift";
		else if (id == STARTSHIFT) return "Start Shift";
		else return null;
	}
}
