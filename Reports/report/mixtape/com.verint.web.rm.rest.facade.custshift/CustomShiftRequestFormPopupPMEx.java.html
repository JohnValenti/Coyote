	/**
	 * Prepares the model for saving requests (create/edit).
	 * Mimics CustomShiftRequestsRH.processSave logic as well.
	 */
	public CustShiftReq prepareForSavingFromREST(RequestContext context, CustShiftReq request)
			throws Exception {
		// setting reload action to save skips the error condition when nothing is editable so leave it null

		setDateTimePickers(request.getStartTime());

		setCustomShiftRequest(request);

		restCustShiftReq = request;
		initializeModelForSaving();

		// Use the shared state analysis to handle many of the standard cases: no SP, new req but a shift exists, change req but no shift,
		// etc. Warning messages will be turned into errors here.
		initContentAnalyzeState();
		RmUtils.throwExceptionIfAtleastWarningMessage(context, getPageMessageList());

		initContentAnalyzeStateForREST(context, request);

		if (validate()) {
			request = getCustShiftRequestForSaving(); // NOSONAR
			if (request == null) {
				RmUtils.throwExceptionFromPageMessage(context, this);
			}
		} else {
			RmUtils.throwExceptionFromPageMessage(context, this);
		}
		return request;
	}

	private void setDateTimePickers(Date date) {
		Calendar cal = Calendar.getInstance(m_view_tz);
		cal.setTime(date);
		int hours = cal.get(Calendar.HOUR_OF_DAY);
		int minutes = cal.get(Calendar.MINUTE);
		int seconds = cal.get(Calendar.SECOND);
		m_datePickerPC.setDate(cal.getTime());
		m_timePickerPC.setTime(hours, minutes, seconds);
	}

	private void initContentAnalyzeStateForREST(RequestContext context, CustShiftReq request)
			throws BusinessWFOException, CloneNotSupportedException {
		Localizer localizer = context.getLocalizer();
		if (!isShiftEditable()) {
			// validate the shift date and time against the shift assignment, throw if anything is different
			TimeRange shiftRange = CustomShiftRequestsMH.getMainShiftTimeRange(m_shiftAssignment, m_view_tz);
			if (shiftRange != null && !shiftRange.getStartDate().equals(request.getStartTime())) {
				throw RmUtils.createInvalidParameterException(localizer, RmUtils.START_TIME_FN);
			}
		}

		if (!isActivityEditable()) {
			// change request does not validate that the activity actually matches the one in the shift
			if (m_shift != null) { // NOSONAR
				setActivityID(m_shift.getActivityID().toString());
			}
		}

		if (isShiftChangeRequest()) {
			setOvertime(false);
		}

		validateShiftID(context);
		validateActivityID(context);
		validateExtensions(context);
	}

	@Override
	protected void afterLoadCSDataIninitializeModelForSaving() throws Exception { // NOSONAR
		// if an extension already exists on the shift and it is locked, it will not be included on this request
		CustShiftReq request2 = (CustShiftReq) restCustShiftReq.clone();
		if (isShiftChangeRequest()) {
			CustomShiftUtil.convertShiftAssignmentToRequest(m_shiftAssignment, request2,
					m_saMainShiftStartTime, m_saMainShiftEndTime, m_saGapBeforeDuration, m_saGapAfterDuration, false);
			// request2 now contains the extensions based on the existing shift assignment
			if (!isExtAfterEditable()) {
				restCustShiftReq.setExtAfterActivityID(request2.getExtAfterActivityID());
				restCustShiftReq.setExtAfterDuration(request2.getExtAfterDuration());
				restCustShiftReq.setExtAfterGap(request2.getExtAfterGap());
				restCustShiftReq.setExtAfterID(request2.getExtAfterID());
				restCustShiftReq.setExtAfterIsOT(request2.getExtAfterIsOT());
			}
			if (!isExtBeforeEditable()) {
				restCustShiftReq.setExtBeforeActivityID(request2.getExtBeforeActivityID());
				restCustShiftReq.setExtBeforeDuration(request2.getExtBeforeDuration());
				restCustShiftReq.setExtBeforeGap(request2.getExtBeforeGap());
				restCustShiftReq.setExtBeforeID(request2.getExtBeforeID());
				restCustShiftReq.setExtBeforeIsOT(request2.getExtBeforeIsOT());
			}
			handleChangeRequestForExtension();
		}

		boolean hasExtension = restCustShiftReq.getExtAfterID() != null || restCustShiftReq.getExtBeforeID() != null;

		// check for OT here because the page model will force it to false, we want error for REST
		if (restCustShiftReq.isOT()) {
			if (!m_allowOTShifts) {
				String msg = m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.REQUESTS_FOR_OVERTIME_ARE_NOT_ALLOWED);
				RmUtils.throwExceptionBadRequest(msg);
			}
			if (hasExtension && isNewShiftRequest()) {
				String msg = m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.REQUESTS_FOR_OVERTIME_WITH_EXT_ARE_NOT_ALLOWED);
				RmUtils.throwExceptionBadRequest(msg);
			}
		}
		if (restCustShiftReq.getExtAfterIsOT() && !m_allowOTExtAfterShifts) {
			String msg = m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.REQUESTS_FOR_OVERTIME_EXT_AFTER_ARE_NOT_ALLOWED);
			RmUtils.throwExceptionBadRequest(msg);
		}
		if (restCustShiftReq.getExtBeforeIsOT() && !m_allowOTExtBeforeShifts) {
			String msg = m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.REQUESTS_FOR_OVERTIME_EXT_BEFORE_ARE_NOT_ALLOWED);
			RmUtils.throwExceptionBadRequest(msg);
		}
		if (!hasExtension && !m_allowRegularShifts) {
			String msg = m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.REQUESTS_FOR_REGULAR_ARE_NOT_ALLOWED);
			RmUtils.throwExceptionBadRequest(msg);
		}
		if (restCustShiftReq.getExtAfterID() != null && !m_allowRegularExtAfterShifts) {
			String msg = m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.REQUESTS_FOR_REGULAR_EXT_AFTER_ARE_NOT_ALLOWED);
			RmUtils.throwExceptionBadRequest(msg);
		}
		if (restCustShiftReq.getExtBeforeID() != null && !m_allowRegularExtBeforeShifts) {
			String msg = m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.REQUESTS_FOR_REGULAR_EXT_BEFORE_ARE_NOT_ALLOWED);
			RmUtils.throwExceptionBadRequest(msg);
		}

		// shift change requests do not allow overtime
		if (isShiftChangeRequest() && restCustShiftReq.isOT()) {
			String msg = m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.CHANGE_REQUESTS_FOR_OVERTIME_ARE_NOT_ALLOWED);
			RmUtils.throwExceptionBadRequest(msg);
		}
	}

	/**
	 * If they haven't specified a shift and activity id, they only want an extension.
	 * However, setCurrentObjects will force the shift to the first one in the collection,
	 * so lets just set these based on the shift assignment and let keepOnlyRequestDeltas remove it later.
	 */
	private void handleChangeRequestForExtension() {
		if (m_shiftAssignment == null) {
			return;
		}
		if (restCustShiftReq.getShiftID() == null) {
			if (m_shiftAssignment.getShiftID() == null) {
				// SA is custom shift, assign it the special -1 shift because we need a shift for
				// setRequestStartEndBasedOnShiftAndDatePicker. Otherwise, it will skip that logic and change the request dates to the
				// shift's start and end dates which are currently null. that cause will fail when altering just the start time of
				// a custom shift - all of this crap is due to the inability of extracting the model's logic so it can be re-used
				Shift shift = findCustomShift();
				if (shift != null) {
					restCustShiftReq.setShiftID(shift.getID());
				}
			} else {
				restCustShiftReq.setShiftID(m_shiftAssignment.getShiftID());
			}
		}
		if (restCustShiftReq.getActivityID() == null) {
			restCustShiftReq.setActivityID(m_shiftAssignment.getActivityID());
		}
	}

	private Shift findCustomShift() {
		if (m_shifts != null) {
			for (Shift shift : m_shifts) {
				if (shift.getID().equals(CUSTOM_ID)) {
					return shift;
				}
			}
		}
		return null;
	}

	private static ShiftOTExtension getShiftOTExtension(ID id, List<ShiftOTExtension> types, RequestContext context) {
		for (ShiftOTExtension type : types) {
			if (type.getID().equals(id)) {
				return type;
			}
		}
		return null;
	}

	private void validateShiftID(RequestContext context) throws BusinessWFOException {
		ID shiftID = getShiftID();
		if (shiftID == null) {
			return;
		}
		for (Shift shift : m_shifts) {
			if (shift.getID().equals(shiftID)) {
				return;
			}
		}
		throw RmUtils.createInvalidParameterException(context.getLocalizer(), RmUtils.SHIFT_ID_FN);
	}

	private void validateActivityID(RequestContext context) throws BusinessWFOException { // NOSONAR
		ID shiftID = getShiftID();
		ID activityID = getActivityID();

		if (isNewShiftRequest()) {
			if (activityID == null) {
				if (shiftID != null && m_shift != null) {
					// just force it to the shift's activity id
					m_customShiftRequest.setActivityID(m_shift.getActivityID());
					return;
				}
				// new shift without a matching/valid shiftID - error
				throw RmUtils.createInvalidParameterException(context.getLocalizer(), RmUtils.ACTIVITY_ID_FN);
			}
			// must be in collection
			boolean found = false;
			for (Activity a : m_activities) {
				if (a.getID().equals(activityID)) {
					found = true;
					break;
				}
			}
			// check the shift, but this should already be in the main collection if not null
			if (m_shift != null && m_shift.getActivityID() != null && m_shift.getActivityID().equals(activityID)) {
				found = true;
			}
			if (!found) {
				throw RmUtils.createInvalidParameterException(context.getLocalizer(), RmUtils.ACTIVITY_ID_FN);
			}
		}
	}

	private void validateExtensions(RequestContext context) throws BusinessWFOException {
		ShiftOTExtension type = validateExtID(context, getExtAfterID(), m_extAfterTypes, CustShiftRestService.EXTENSION_AFTER_ID_DESC);
		ID activityID = validateExtActivityID(getExtAfterActivityID(), m_extAfterActivities, getExtAfterID(), type, context,
				CustShiftRestService.EXTENSION_AFTER_ACTIVITY_ID_DESC);
		setExtAfterActivityID(activityID);
		validateExtGap(type, getExtAfterGap(), context, CustShiftRestService.EXTENSION_AFTER_GAP_DESC);

		type = validateExtID(context, getExtBeforeID(), m_extBeforeTypes, CustShiftRestService.EXTENSION_BEFORE_ID_DESC);
		activityID = validateExtActivityID(getExtBeforeActivityID(), m_extBeforeActivities, getExtBeforeID(), type, context,
				CustShiftRestService.EXTENSION_BEFORE_ACTIVITY_ID_DESC);
		setExtBeforeActivityID(activityID);
		validateExtGap(type, getExtBeforeGap(), context, CustShiftRestService.EXTENSION_BEFORE_GAP_DESC);
	}

	private static ID validateExtActivityID(ID id, List<Activity> activities, ID typeID, ShiftOTExtension type, RequestContext context,
			String fieldName) throws BusinessWFOException {
		if (id == null) {
			if (type != null) {
				return type.getActivityID();
			}
			if (typeID != null) {
				// activityID was specified without a shiftID - error
				throw RmUtils.createInvalidParameterException(context.getLocalizer(), fieldName);
			}
		} else {
			boolean found = false;
			for (Activity activity : activities) {
				if (activity.getID().equals(id)) {
					found = true;
				}
			}
			if (!found) {
				throw RmUtils.createInvalidParameterException(context.getLocalizer(), fieldName);
			}
		}
		return id;
	}

	/**
	 * Validates the extension's gap. 15.1.0 Sustaining does not enforce the min or max gaps so this version will only
	 * enforce positive or 0 values. In 15.1.0_135, duration is editable and gaps are optionally enforced.
	 */
	private static void validateExtGap(ShiftOTExtension type, int gap, RequestContext context, String fieldName)
			throws BusinessWFOException {
		if (type == null) {
			return;
		}
		if (gap < 0) {
			throw RmUtils.createInvalidParameterException(context.getLocalizer(), fieldName);
		}
	}

	private static ShiftOTExtension validateExtID(RequestContext context, ID id, List<ShiftOTExtension> types, String fieldName)
			throws BusinessWFOException {
		ShiftOTExtension shift = null;
		if (id != null) {
			shift = getShiftOTExtension(id, types, context);
			if (shift == null) {
				throw RmUtils.createInvalidParameterException(context.getLocalizer(), fieldName);
			}
			return shift;
		}
		return null;
	}

	public void populateDetails(RequestContext context, CustShiftDetailEntity entity, ID employeeID, Date date, Date ribbonEndDate)
			throws BusinessWFOException {
		detailEntity = entity;
		Localizer localizer = context.getLocalizer();

		setDateTimePickers(date);
		setRequestedFor(employeeID);

		if (ribbonEndDate != null) {
			setReloadAction(RELOAD_ACTION_RIBBON);
			setRibbonEndDate(ribbonEndDate);
			setRibbonStartDate(date);
		}

		initializeModel();

		entity.setId(employeeID.toString());
		entity.setInSP(m_spID != null);

		CustShiftFacadeImpl.addActivities(entity.getActivities(), m_activities);
		CustShiftFacadeImpl.addShifts(entity.getShifts(), m_shifts);

		CustShiftFacadeImpl.addActivities(entity.getExtensionAfterActivities(), m_extAfterActivities);
		CustShiftFacadeImpl.addActivities(entity.getExtensionBeforeActivities(), m_extBeforeActivities);

		CustShiftFacadeImpl.addShiftOTExtensions(entity.getExtensionAfterTypes(), m_extAfterTypes);
		CustShiftFacadeImpl.addShiftOTExtensions(entity.getExtensionBeforeTypes(), m_extBeforeTypes);

		CustShiftFacadeImpl.populateShiftAssignment(entity, m_shiftAssignment);

		entity.setGapAfterDuration(m_saGapAfterDuration);
		entity.setGapBeforeDuration(m_saGapBeforeDuration);

		entity.setMainShiftEndTime(FormatUtils.dateToISO8601SecPrecisionString(m_saMainShiftEndTime));
		entity.setMainShiftStartTime(FormatUtils.dateToISO8601SecPrecisionString(m_saMainShiftStartTime));
		entity.setMostDeficientActivity1(RmUtils.getInteger(m_mostDeficientActivity1));
		entity.setMostDeficientActivity2(RmUtils.getInteger(m_mostDeficientActivity2));

		RmUtils.addPageMessages(localizer, entity.getMessages(), getPageMessageList());

		if (ribbonEndDate != null) {
			RmUtils.throwExceptionIfAtleastWarningMessage(context, getPageMessageList());
			RequestsMetaData metaData = new RequestsMetaData(m_orgSetting);
			entity.setRequest(CustShiftFacadeImpl.createCustShiftRequestEntity(context, m_customShiftRequest, metaData));
		}
	}

	@Override
	protected void optimizeRequestForRibbonSelection() {
		// initializeModel calls optimizeRequestForRibbonSelection and then analyzeState
		// messages in optimize are OK to return to the client because they are messages concerning their selection
		// messages in initContentAnalyzeState are error messages that break the operation.

		super.optimizeRequestForRibbonSelection();
		// capture the ribbon messages here
		List<Message> messages = getPageMessageList();
		RmUtils.addPageMessages(m_localizer, detailEntity.getMessages(), messages);

		// clear them now so they will not be included when we go over initContentAnalyzeState warning messages that break the operation
		messages.clear();
	}
