package com.bluepumpkin.web.rm.auction.filter;

import java.util.*;
import com.witness.web.uif.system.*;

import com.witness.web.uif.util.html.CSSUtil;

import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarTextButton;

/**
 * Title:        BidOptionsFilterPC
 * Description:  This generates the Shift Bid Options page's Filter UI, consisting
 *               of a BidOptionsMessagePC and a popup window "Update" button.
 * Copyright:    Copyright (c) 2005
 * Company:      Witness Systems, Inc
 * @author       Gonzalo Quintanar
 * @version      1.0
 */
public class BidOptionsFilterPC extends CompositePC {
	protected ResourceBundle  m_bundle;
	
	//Constants
	private static final String FILTER_POPUP_SIZE = "800,655,ctr,ctr";
	private static final String FILTER_RH   = "bid_options_filter_config";
	private static final String FILTER_NAME = "bidOptionsFilter";

	//Components
	private ToolbarPC m_toolbar;
	protected BidOptionsMessagePC m_messageBox;

	//Data
	private BidOptionsFilter m_filter; //The data structure holding all the filter parameters
	private String m_summaryString;
	
	private ID m_auctionID;
	private ID m_bidderID;
	
	/**
	 * Constructor
	 */
	public BidOptionsFilterPC(RequestContext context, ID auctionID, ID bidderID) //, ID empID, ID campaignID) 
	{
		super(context);

		//Initialize member variables
		m_auctionID = auctionID;
		m_bidderID = bidderID;
		//m_empID = empID;
		//m_campaignID = campaignID;		

		m_bundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);

		// Create filter Message box subcomponent
		m_messageBox = createMessageBox();

		// Create Toolbar with the Go button subcomponent
		m_toolbar = new ToolbarPC(context);

		// Name this component and its sub-components
		setName(FILTER_NAME);

		// Add subcomponents to the BidOptionsFilterPC
		addChildComponent(m_toolbar);
		addChildComponent(m_messageBox);
		
		setUseWrapper(false);
	}
	
	/**
	 * Create a Bid Options Message Box Page Component
	 */
	private BidOptionsMessagePC createMessageBox() 
	{
		BidOptionsMessagePC msgPC = new BidOptionsMessagePC(m_context, "filterDescriptor");
		msgPC.setWidth("100%"); //300px
		msgPC.setHeight("22px");
		msgPC.setMessageStyle("font-size:9px;");
		msgPC.setMsgPreFmtEnabled(false);
		msgPC.setIsButtonEnabled(true);
		return msgPC;
	}
	
	/**
	 * Initialize Component For Display
	 */
	public void initForDisplay() 
	{
		if (m_isInitializedForDisplay)
		{
			return;
		}
		else 
		{   		
			super.initForDisplay();
			addPopupArgs(PageAction.UPDATE, FILTER_RH, getPopupArgs(), null, FILTER_POPUP_SIZE, true, 1);
			initMessageBox();
		}
	}

	/**
	 * Initialize the Message Box text with the filter settings strings.
	 */
	private void initMessageBox()
	{
		ArrayList msgList = new ArrayList(2);
		
		String filterHeader = i18n(m_bundle, RmWebBundleKey.RM_SA_FILTER_FILTER_SETTINGS) + ":";
		Message msg1 = new Message(Message.INFO_TYPE, filterHeader);
		msg1.setIsHtmlCodeAllowed(true);
		msgList.add(msg1);
		try
		{
			BidOptionsFilter filter = getFilter();
			if (filter != null)
			{
				String filterString = BidOptionsFilterUtil.getFilterString(filter, getAuction(), 
						m_context, m_localizer, m_localeContext);
				if (filterString.length() > 0)
				{
					Message msg2 = new Message(Message.INFO_TYPE, filterString);
					msg2.setIsHtmlCodeAllowed(true);
					msgList.add(msg2);
				}
			}
			if ((m_summaryString != null) && (m_summaryString.length() > 0))
			{
				Message msg3 = new Message(Message.INFO_TYPE, m_summaryString);
				msg3.setIsHtmlCodeAllowed(true);
				msgList.add(msg3);
			}
			
			m_messageBox.addMessage(msgList);
		}
		catch (Exception ex) 
		{
			handleUnableToLoadDataError(log, ex);
		}
		
	}
	
	/**
	 * Return true if the auction is for entire period schedules. 
	 * Return false if the auction is for individual shifts.
	 */
	private boolean getIsWeeklyAuction()
	{
		boolean isWeeklyAuction = true;
		try
		{
			ShiftBidAuction auction = AuctionMH.getAuction(m_auctionID);
			isWeeklyAuction = auction.getIsFullPeriodSchedule();
		}
		catch (RmHardValidationException ex) 
		{
			Message msg = handleUnableToLoadDataError(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));			
		} 
		catch (Exception ex) 
		{
			handleUnableToLoadDataError(log, ex);
		}
		return isWeeklyAuction;
	}
	
	/**
	 * Return the Auction. 
	 */
	private ShiftBidAuction getAuction()
	{
		ShiftBidAuction auction = null;
		try
		{
			auction = AuctionMH.getAuction(m_auctionID);
		}
		catch (RmHardValidationException ex) 
		{
			Message msg = handleUnableToLoadDataError(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));			
		} 
		catch (Exception ex) 
		{
			handleUnableToLoadDataError(log, ex);
		}
		return auction;
	}

	/**
	 * Return any Popup window arguments 
	 */
	private String getPopupArgs() 
	{
		StringBuffer sb = new StringBuffer(100);
	    //sb.append(BidOptionsFilterPopupPM.USE_SESSION_FILTER_FN).append("=true&");
	    sb.append(BidOptionsFilterPopupPM.AUCTION_ID_FN).append("=").append(m_auctionID.toInt()).append("&");
	    sb.append(BidOptionsFilterPopupPM.BIDDER_ID_FN).append("=").append(m_bidderID.toInt()).append("&");
	    //sb.append(BidOptionsFilterPopupPM.EMPLOYEE_ID_FN).append("=").append(m_empID).append("&");
	    //sb.append(BidOptionsFilterPopupPM.CAMPAIGN_ID_FN).append("=").append(m_campaignID).append("&");
	    
		return sb.toString();
	}

	public BidOptionsFilter getFilter() 
	{
		if (m_filter == null)
		{
			m_filter = (BidOptionsFilter)m_context.getAttribute(RequestContext.SESSION_SCOPE, RmKeys.BID_OPTIONS_FILTER);
		}
		return m_filter;
	}

	/**
	 * Return Html Display
	 */
	public String getHtmlDisplay() 
	{
		StringBuffer buff = new StringBuffer(1024);		
		if (useWrapper()) 
			buff.append("\n<div style=\"margin-top:12px; margin-bottom:7px\">");
		HtmlTable table = new HtmlTable("0", "0", "0");
		table.setWidth("400");
		table.add(createTableRow());
		buff.append(table);
		
		if (useWrapper()) 
			buff.append("\n</div>");

		return buff.toString();
	}
	
	/**
	 * Create Row
	 */
	protected HtmlTableRow createTableRow()
	{
		ArrayList rowcells = new ArrayList();
		HtmlTableCell cell = new HtmlTableCell(m_messageBox);
		cell.setAlign("left");
		rowcells.add(cell);

		//Add a little space before the Update button
		cell = new HtmlTableCell(HtmlUtil.NBSP);
		rowcells.add(cell);

		// Add the "Update" button, contained in a wrapper cell
		String btnLabel = i18n(m_bundle, RmWebBundleKey.RM_SA_FILTER);
		ToolbarTextButton updateButton = new ToolbarTextButton(m_toolbar, PageAction.UPDATE, btnLabel,
				ToolbarButton.POPUP_WINDOW_TYPE, ToolbarTextButton.BLUE_BUTTON, true);		
		cell = new HtmlTableCell(updateButton);
		cell.setAlign("left");
		cell.setValign("top");
		rowcells.add(cell);

		//Add a little space after the Update button
		cell = new HtmlTableCell(HtmlUtil.NBSP);
		cell.setWidth("20px");
		rowcells.add(cell);
		
		
		return new HtmlTableRow(rowcells);
	}
	
	/**
	 * Set an optional string to be appended to the end of the message box text.
	 * @param summaryString optional string to be appended to the end of the message box text
	 */
	public void setSummaryString(String summaryString)
	{
		m_summaryString = summaryString;
	}
	
}
