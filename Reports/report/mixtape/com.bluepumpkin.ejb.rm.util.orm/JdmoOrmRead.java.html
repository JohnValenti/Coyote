package com.bluepumpkin.ejb.rm.util.orm;

import java.lang.reflect.InvocationTargetException;
import java.sql.ResultSetMetaData;
import java.sql.SQLException;
import java.util.ArrayList;
import java.util.List;

import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;

/**
 * Converts native SQL to a list of objects. 
 * This is not really an ORM since this does not handle any kind of relationships. 
 * All this does is to convert the rowset from the query 
 * to a list of a single type of object.  Uses reflection so this will be slower than if you do this by hand.
 */
class JdmoOrmRead<T> {

	Class<T> entityClass;
	PropertySetter<T> propertySetter;
	String[] propertyNames;

	JdmoOrmRead(Class<T> clazz, JdmoRowset rs) throws SQLException, JdmoException {

		entityClass = clazz;
		ResultSetMetaData rsmd = rs.getMetaData();
		int count = rsmd.getColumnCount();
		propertyNames = new String[count];
		for (int i = 0; i < count; i++) {
			propertyNames[i] = rsmd.getColumnLabel(i + 1);
		}

		propertySetter = new PropertySetter<T>(clazz, propertyNames);
	}



	static <T> List<T> toObjects(Class<T> entityClass, JdmoRowset rs) {
		List<T> result = new ArrayList<T>();
		try {
			JdmoOrmRead<T> orm = new JdmoOrmRead<T>(entityClass, rs);
			while (rs.next()) {
				result.add(orm.createBeanFromResultset(rs));
			}
		} catch (Exception e) {
			throw new RmRuntimeException("Could not convert result set to objects", e);
		}
		return result;

	}

	T createBeanFromResultset(JdmoRowset rs) throws InstantiationException, IllegalAccessException, JdmoException,
			InvocationTargetException {
		T bean = entityClass.newInstance();
		for (String propertyName : propertyNames) {
			Object value = rs.getObject(propertyName);
			propertySetter.setProperty(bean, value, propertyName);
		}
		return bean;

	}

}
