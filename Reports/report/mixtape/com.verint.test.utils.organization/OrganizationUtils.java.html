package com.verint.test.utils.organization;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.verint.test.utils.DateUtils;
import com.verint.test.utils.GuidUtils;
import com.verint.test.utils.RemoteEjbFactory;

import java.util.ArrayList;
import java.util.TimeZone;

/**
 * OrganizationUtils class created to build and destory Organization objects associated with Impact360
 */
public class OrganizationUtils {

	/**
	 * Build an organization given name and description. Defaults to America/Los_Angeles time zone
	 * @param name Name of organization
	 * @param description description of the organization
	 * @return Built Organization object
	 */
	public static Organization createOrganization(String name, String description) throws Exception {

		Organization org = new Organization();
		org.setName(GuidUtils.createGuidName(name));
		org.setDescription(description);
		org.setTimeZone(TimeZone.getTimeZone("America/Los_Angeles"));
		org.setDayBoundaryOffset(0);
		org.setWeekStartDate((short)1);
		org.setParentID(Organization.YOUR_COMPANY_ID_OBJ);
		org.setID(RemoteEjbFactory.getWorkResourceManager().createOrganization(org));
		return org;
	}


	/**
	 * Build an organization.
	 * @param name Name of organization
	 * @param description description of the organization
	 * @return Built Organization object
	 */
	public static Organization createOrganization(String name, String description, String timeZone, String dayBoundary, short weekStartDay) throws Exception {

		Organization org = new Organization();
		org.setName(GuidUtils.createGuidName(name));
		org.setDescription(description);
		org.setTimeZone(TimeZone.getTimeZone(timeZone));
		org.setDayBoundaryOffset(DateUtils.timeToMinutes12HoursStr(dayBoundary));
		org.setWeekStartDate(weekStartDay);
		org.setParentID(Organization.YOUR_COMPANY_ID_OBJ);
		org.setID(RemoteEjbFactory.getWorkResourceManager().createOrganization(org));
		return org;
	}

	/**
	 *
	 * @param name
	 * @param description
	 * @param timeZone
	 * @param dayBoundary
	 * @param weekStartDay
	 * @param parentID
	 * @return
	 * @throws Exception
	 */
	public static Organization createSubOrganization(String name, String description, String timeZone, String dayBoundary, short
			weekStartDay, ID parentID) throws Exception {

		Organization org = new Organization();
		org.setName(GuidUtils.createGuidName(name));
		org.setDescription(description);
		org.setTimeZone(TimeZone.getTimeZone(timeZone));
		org.setDayBoundaryOffset(DateUtils.timeToMinutes12HoursStr(dayBoundary));
		org.setWeekStartDate(weekStartDay);
		org.setParentID(parentID);
		org.setID(RemoteEjbFactory.getWorkResourceManager().createOrganization(org));
		return org;
	}

	/**
	 * Build an organization given name and description. Defaults to America/Los_Angeles time zone
	 * which operate during 24 hours.
	 *
	 * @param name Name of organization
	 * @param description description of the organization
	 * @return Built Organization object
	 */
	public static Organization create24HoursOrganization(String name, String description) throws Exception {

		Organization org = new Organization();
		org.setName(GuidUtils.createGuidName(name));
		org.setDescription(description);
		org.setTimeZone(TimeZone.getTimeZone("America/Los_Angeles"));
		org.setDayBoundaryOffset(0);
		org.setWeekStartDate((short)1);
		org.setParentID(Organization.YOUR_COMPANY_ID_OBJ);
		org.setIs24Hour(true);
		org.setID(RemoteEjbFactory.getWorkResourceManager().createOrganization(org));
		return org;
	}

	/**
	 * Get the organization given the Org ID
	 * @param orgId Organization ID of the organization we want
	 * @return Organization with the ID
	 * @throws Exception
	 */
	public static Organization getOrgByID(ID orgId) throws Exception {
		return RemoteEjbFactory.getWorkResourceManager().getOrganizationByID(orgId);
	}

	/**
	 * Destroy an organization
	 * @param org Organization to be destroyed
	 * @throws Exception
	 */
	public static void destroyOrganization(Organization org) throws Exception {
		ArrayList<ID> ids = new ArrayList<>();
		ids.add(org.getID());
		RemoteEjbFactory.getWorkResourceManager().deleteOrganizations(ids);
	}
}
