package com.verint.test.utils.organization;


import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.workresource.ejb.OrganizationFilter;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workresource.model.OrganizationHOO;
import com.verint.test.utils.RemoteEjbFactory;

import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;
import java.util.HashSet;

public class OrganizationPersistenceUtils {

	public static ID createOrganization(Organization organization) throws Exception {
		return RemoteEjbFactory.getWorkResourceManager().createOrganization(organization);
	}

	/**
	 * deleteOrganization is replaced with deleteOrganizations to match naming conventions
	 * from other WfmRemoteEjbFactory methods.
	 **/
	@Deprecated
	public static void deleteOrganization(Collection<ID> orgIds) throws Exception {
		RemoteEjbFactory.getWorkResourceManager().deleteOrganizations(orgIds);
	}

	public static void deleteOrganization(ID orgId) throws Exception {
		deleteOrganizations(Collections.singletonList(orgId));
	}

	public static void deleteOrganizations(Collection<ID> orgIds) throws Exception {
		RemoteEjbFactory.getWorkResourceManager().deleteOrganizations(orgIds);
	}

	public static void createOrganizationHOO(OrganizationHOO hoo) throws Exception {
		RemoteEjbFactory.getWorkResourceManager().createOrganizationHOOAssignment(hoo);
	}

	public static Collection<Organization> getOrganizations() throws Exception {
		return RemoteEjbFactory.getWorkResourceManager().getAllOrganizations();
	}

	public static Organization getOrganizationById(ID id) throws Exception {
		Collection<Organization> organizations = getOrganizationsByIds(Collections.singletonList(id));
		Organization organization = null;
		if (organizations.size() == 1) {
			organization = organizations.iterator().next();
		}
		return organization;
	}

	public static Collection<Organization> getOrganizationsByIds(Collection<ID> ids) throws Exception {
		return RemoteEjbFactory.getWorkResourceManager().getOrganizationsByIDs(ids);
	}

	public static HashMap<String, ID> getOrganizationIDsByNames(HashSet<String> names) throws Exception {
		return RemoteEjbFactory.getWorkResourceManager().getOrganizationIDsByNames(names);
	}

	public static String getOrganizationName(ID orgId) throws Exception {
		return RemoteEjbFactory.getWorkResourceManager().getOrganizationName(orgId);
	}

	public static Organization getOrganizationByName(String name) throws Exception {
		OrganizationFilter organizationNameFilter = new OrganizationFilter();
		organizationNameFilter.setOrganizationName(name, OrganizationFilter.OP_EQUAL);
		Collection<Organization> organizations = RemoteEjbFactory.getWorkResourceManager().getOrganizations(organizationNameFilter);
		if (organizations.size() != 1) {
			return null;
		}
		return organizations.iterator().next();
	}
}
