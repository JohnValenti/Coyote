package com.verint.wfm.ws.util;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.activity.ejb.ActivityManager;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityFilter;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.MapUtil;
import com.bluepumpkin.ejb.bbm.dao.MapUtil.SystemType;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManager;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.verint.wfm.ws.ServiceApplicationException;

public class SyncUtil {

	private static final Map<String, ID> ORGANIZATION_NAME_ID_MAP = new HashMap<String, ID>();
	private static final Map<String, ID> ACTIVITY_NAME_ID_MAP = new HashMap<String, ID>();
	
	public static MapJAXBType areSaved(Collection<String> externalSourceIds, String systemTypeName,
			MapUtil.ObjectType objectType) throws ServiceApplicationException {

		SystemType systemType;
		try {
			systemType = SystemType.valueOf(systemTypeName);
		} catch (RuntimeException e) {
			throw new ServiceApplicationException(e);
		}

		Map<String, Boolean> preResult = new HashMap<String, Boolean>();
		if (externalSourceIds != null) {
			for (String externalSourceId : externalSourceIds) {
				ID localId;
				ID extId = ExternalIDUtil.buildExternalID(systemType,
						externalSourceId);
				try {
					localId = MapUtil.getExternalToWfoID(systemType,
							objectType, extId);
				} catch (JdmoException e) {
					throw new ServiceApplicationException(e);
				}
				// There is a saved entity precisely when we find a local id
				// that is mapped to the external id
				preResult.put(externalSourceId, (localId != null));
			}
		}

		// Map needs to be wrapped by a custom JAXB type
		return new MapJAXBType(preResult);
	}
	
	public static String doTheseExist(Collection<String> externalSourceIds, String systemTypeName, MapUtil.ObjectType objectType)
			throws ServiceApplicationException {

		SystemType systemType;
		try {
			systemType = SystemType.valueOf(systemTypeName);
		} catch (RuntimeException e) {
			throw new ServiceApplicationException(e);
		}

		StringBuilder preResult = new StringBuilder();
		if (externalSourceIds != null) {
			for (String externalSourceId : externalSourceIds) {
				ID extId = ExternalIDUtil.buildExternalID(systemType, externalSourceId);
				try {
					preResult.append(MapUtil.getExternalToWfoID(systemType, objectType, extId) != null);
				} catch (JdmoException e) {
					throw new ServiceApplicationException(e);
				}
			}
		}
		return preResult.toString();
	}

	public static ID getOrganizationIDByNameCaseInsensitive(String organizationName, WorkResourceManager manager,
			boolean alwaysResetCache)
			throws ServiceApplicationException {

		// By construction this map is of size one :)
		Map<String, ID> singletonMap = getOrganizationIDsByNamesCaseInsensitive(
				Collections.singletonList(organizationName),
				manager, alwaysResetCache);

		// Sneaky, no? Savoir-faire is everywhere!
		ID wfoId = singletonMap.values().iterator().next();
		return wfoId;
	}

	// TODO use generics or callbacks to unify these two methods
	public static ID getActivityIDByNameCaseInsensitive(String activityName, ActivityManager manager,
			boolean alwaysResetCache)
			throws ServiceApplicationException {

		// By construction this map is of size one :)
		Map<String, ID> singletonMap = getActivityIDsByNamesCaseInsensitive(
				Collections.singletonList(activityName),
				manager, alwaysResetCache);

		// Sneaky, no? Savoir-faire is everywhere!
		ID wfoId = singletonMap.values().iterator().next();
		return wfoId;
	}
	
	
	
	/**
	 * 
	 * @param alwaysResetCache
	 *            this should be true if the cache could be stale
	 */
	public static Map<String, ID> getOrganizationIDsByNamesCaseInsensitive(List<String> organizationNames,
			WorkResourceManager manager, boolean alwaysResetCache)
			throws ServiceApplicationException {

		// only populate cache if its empty, or if calling method has a reason to believe its stale
		populateOrganizationNameUpperCaseIDCachedMap(manager, alwaysResetCache);

		// we only need a subset of the cached map
		Map<String, ID> result = new HashMap<String, ID>();
		for (String organizationName : organizationNames) {
			ID wfoOrgId = ORGANIZATION_NAME_ID_MAP.get(organizationName.toUpperCase());
			// Note this can be null id when org name doesn't exist
			result.put(organizationName, wfoOrgId);
		}
		return result;
	}

	/**
	 * 
	 * @param alwaysResetCache
	 *            this should be true if the cache could be stale
	 */
	public static Map<String, ID> getActivityIDsByNamesCaseInsensitive(List<String> activityNames,
			ActivityManager manager, boolean alwaysResetCache)
			throws ServiceApplicationException {

		// only populate cache if its empty, or if calling method has a reason to believe its stale
		populateActivityNameUpperCaseIDCachedMap(manager, alwaysResetCache);

		// we only need a subset of the cached map
		Map<String, ID> result = new HashMap<String, ID>();
		for (String activityName : activityNames) {
			ID wfoactivityId = ACTIVITY_NAME_ID_MAP.get(activityName.toUpperCase());
			// Note this can be null id when org name doesn't exist
			result.put(activityName, wfoactivityId);
		}
		return result;
	}
	
	
	private static void populateOrganizationNameUpperCaseIDCachedMap(WorkResourceManager manager,
			boolean alwaysResetCache)
			throws ServiceApplicationException {
		if (ORGANIZATION_NAME_ID_MAP.isEmpty() || alwaysResetCache) {
			synchronized (ORGANIZATION_NAME_ID_MAP) {
				ORGANIZATION_NAME_ID_MAP.clear();
				try {
					Collection<Organization> organizations = manager.getFullScopeTree();
					for (Organization organization : organizations) {
						ORGANIZATION_NAME_ID_MAP.put(organization.getName().toUpperCase(), organization.getID());
					}

				} catch (BbmFinderException e) {
					throw new ServiceApplicationException(e);
				} catch (RemoteException e) {
					throw new ServiceApplicationException(e);
				} catch (Exception e) {
					throw new ServiceApplicationException(e);
				}
			}
		}
	}

	private static void populateActivityNameUpperCaseIDCachedMap(ActivityManager manager,
			boolean alwaysResetCache)
			throws ServiceApplicationException {
		if (ACTIVITY_NAME_ID_MAP.isEmpty() || alwaysResetCache) {
			synchronized (ACTIVITY_NAME_ID_MAP) {
				ACTIVITY_NAME_ID_MAP.clear();
				try {
					ActivityFilter activityFilter = new ActivityFilter(ActivityFilter.FILTER_NONE);
					Collection<Activity> activities = manager.findActivities(activityFilter);
					for (Activity activity : activities) {
						ACTIVITY_NAME_ID_MAP.put(activity.getName().toUpperCase(), activity.getID());
					}
				} catch (BbmFinderException e) {
					throw new ServiceApplicationException(e);
				} catch (RemoteException e) {
					throw new ServiceApplicationException(e);
				}
			}
		}
	}
}
