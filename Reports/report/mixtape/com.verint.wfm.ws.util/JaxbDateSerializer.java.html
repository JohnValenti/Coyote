package com.verint.wfm.ws.util;

import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.TimeZone;

import javax.xml.bind.annotation.adapters.XmlAdapter;

public class JaxbDateSerializer extends XmlAdapter<Date, Date> {

	protected SimpleDateFormat m_simpleDateFormat;

	public JaxbDateSerializer() {
		// We are expecting date as 2000-01-03T00:00:00, where time zone is unknown
		// We are using UTC as a way to essentially make java forget about the real zone!
		m_simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");      
		m_simpleDateFormat.setTimeZone(TimeZone.getTimeZone("UTC"));

	}

	/**
	 * we clone SimpleDateFormat each time because its not thread safe
	 */
	private SimpleDateFormat cloneFormat() {
		return (SimpleDateFormat) this.m_simpleDateFormat.clone();
	}

	/**
	 * Create XML
	 */
	@Override
	public Date marshal(Date date) throws Exception {
		Date reverseDummy = date;
//		System.out.println("Date after marshal: "+ cloneFormat().format(reverseDummy));		
//		System.err.println("Date after marshal: "+ cloneFormat().format(reverseDummy));		
		return reverseDummy;
	}

	/**
	 * Create java from XML
	 */
	@Override
	public Date unmarshal(Date date) throws Exception {
		Date dummy = date;
//		System.out.println("Date after unmarshal: "+ cloneFormat().format(dummy));
//		System.err.println("Date after unmarshal: "+ cloneFormat().format(dummy));

		return dummy;
	}
