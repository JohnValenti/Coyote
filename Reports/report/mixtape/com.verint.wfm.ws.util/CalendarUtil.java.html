package com.verint.wfm.ws.util;

import java.util.Calendar;
import java.util.Date;
import java.util.TimeZone;

import com.verint.wfm.ws.ServiceApplicationException;

public class CalendarUtil {

	public static final int QUARTER_HOUR_PER_DAY = 4 * 24;
	public static final int DAYS_IN_WEEK = 7;

	/**
	 * Convert a date at midnight to a Calendar with zone at midnight
	 */
	public static Calendar convertDateToCalendarWithZone(Date midnightDate, TimeZone timeZone, boolean midnightOnly) throws ServiceApplicationException {
		Calendar zonelessCalendar = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
		zonelessCalendar.setTime(midnightDate);
		if (midnightOnly && zonelessCalendar.get(Calendar.HOUR_OF_DAY)  != 0){
			throw new ServiceApplicationException("Date not midnight.. zones screwed up: " + zonelessCalendar.get(Calendar.HOUR_OF_DAY));
		}
		
		Calendar result = Calendar.getInstance(timeZone);
		result.set(zonelessCalendar.get(Calendar.YEAR), zonelessCalendar.get(Calendar.MONTH), zonelessCalendar.get(Calendar.DATE),
				zonelessCalendar.get(Calendar.HOUR_OF_DAY), zonelessCalendar.get(Calendar.MINUTE), zonelessCalendar.get(Calendar.SECOND));
		
		Calendar zonedCalendar = result;
		return zonedCalendar;
	}

	public static Calendar convertMidnightDateToCalendarWithZone(Date midnightDate, TimeZone timeZone) throws ServiceApplicationException {
		return convertDateToCalendarWithZone(midnightDate, timeZone, true);
	}
}
