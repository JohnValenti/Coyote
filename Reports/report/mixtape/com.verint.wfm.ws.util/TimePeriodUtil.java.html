package com.verint.wfm.ws.util;

import java.util.Calendar;
import java.util.Iterator;
import java.util.LinkedHashMap;
import java.util.List;

import com.bluepumpkin.common.util.DateTimeUtil;
import com.verint.wfm.ws.entityImport.model.HoursOfOperationDTO;

/**
 * TODO: merge with the other TimePeriodUtil
 * 
 * @author EClingman
 * 
 */
public class TimePeriodUtil {

	private static final int HOUR_IN_DAY = 24;

	/**
	 * We assume that the startDate is the first day of the week as far as
	 * HoursOfOperationDTO is concerned
	 * 
	 * @param startDate
	 * @param endDate
	 * @param hoursOfOperationDTO
	 * @return
	 */
	public static LinkedHashMap<Calendar, Boolean> buildOnOffHoursMap(Calendar startDate, Calendar endDate,
			List<HoursOfOperationDTO> hoursOfOperationDTOs) {

		if (startDate.after(endDate)) {
			throw new IllegalArgumentException("startDate must be before endDate");
		}

		// We will get next() below as we need them
		Iterator<HoursOfOperationDTO> iterator = hoursOfOperationDTOs.iterator();
		HoursOfOperationDTO hoursOfOperationDTO = iterator.next();

		// We can't rely on the currentTime's opinion of the hour, because it
		// might be GMT rather than local time say
		// We rely that start date is always midnight local time
		int hoursOffSet = startDate.get(Calendar.HOUR_OF_DAY);

		// We will loop from startDate to endDate in one hour increments,
		// creating a date map
		Calendar currentTime = (Calendar) startDate.clone();
		LinkedHashMap<Calendar, Boolean> onOffHoursMap = new LinkedHashMap<Calendar, Boolean>();
		int dayOfWeekCount = 1; // Must start at one not zero!
		while (currentTime.before(endDate)) {

			if (dayOfWeekCount == 8) {
				hoursOfOperationDTO = iterator.next();
				dayOfWeekCount = 1;
			}

			Boolean openPartOfHour;
			// is it closed all day
			if (hoursOfOperationDTO.getDayOpen()[dayOfWeekCount] == -1) {
				openPartOfHour = Boolean.FALSE;
			} else {
				int dayOpenHour = (int) Math.floor((hoursOfOperationDTO.getDayOpen()[dayOfWeekCount] / 60.d));
				int dayCloseHour = (int) Math
						.ceil((hoursOfOperationDTO.getDayClose()[dayOfWeekCount] / 60.d));
				// TODO Warning this will be slightly off for the cross-over to
				// daylight savings time.... Ask Tyler for advice! And perhaps
				// verify that (start - end date) equals 7 time size of
				// collection?
				int currentHourLocalTime = (currentTime.get(Calendar.HOUR_OF_DAY) - hoursOffSet + HOUR_IN_DAY)
						% HOUR_IN_DAY;

				if (currentHourLocalTime >= dayOpenHour && currentHourLocalTime < dayCloseHour) {
					openPartOfHour = Boolean.TRUE;
				} else {
					openPartOfHour = Boolean.FALSE;
				}
			}

			// get ready for next iteration
			Calendar lastTime = (Calendar) currentTime.clone();
			onOffHoursMap.put(lastTime, openPartOfHour);
			currentTime.add(Calendar.HOUR, 1);

			// If the original hour:minutes have come back we are at a new day
			// local time
			if (DateTimeUtil.equalModuloDay(startDate, currentTime)) {
				dayOfWeekCount++;
			}
		}
		return onOffHoursMap;
	}
	
	/**
	 * Converts the minute offset from 12:00am Sunday to the 'format Day hh:mm' and returns it.
	 * @param offset
	 * @return
	 */
	public static String formatToDayAndTime(int offset) {
		String[] days = { "Sunday", "Monday", "Tuesday", "Wednesday",
				"Thursday", "Friday", "Saturday", "Sunday" };
		StringBuffer timeBuffer = new StringBuffer();
		// offset is minute offset from sunday
		int hourOffset = offset / 60;
		int mins = offset % 60;
		if (hourOffset >= 24) {
			int dayOffset = hourOffset / 24;
			timeBuffer.append(days[dayOffset % 7]);
			hourOffset = hourOffset % 24;
		} else {
			timeBuffer.append(days[0]);
		}
		if(hourOffset<10)
		{
			timeBuffer.append(" 0" + hourOffset);
		} else {
			timeBuffer.append(" "+hourOffset);
		}
		// append mins offset
		if(mins<10){			
			timeBuffer.append(" 0" + mins);
		} else {
			timeBuffer.append(" "+mins);
		}
		return timeBuffer.toString();
	}	
}
