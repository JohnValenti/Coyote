package com.bluepumpkin.web.fs.schedule.summary;

import java.util.Date;

/**
 * Title: TimeBankSummaryData
 * Description: Store the data for the agent's current time bank for schedule summary page.
 * Copyright:   Copyright (c) 2009
 * Company:     Verint Systems, Inc
 * @author      Gonzalo Quintanar
 */
public class TimeBankSummaryData 
{
    //private int m_auctions;

    private String m_timeBankName;
    private Date m_startDate;
    private Date m_endDate;
    private Date m_proRateDate;
    private int m_targetHours;
    private int m_prTargetHours;
    private int m_scheduledHours;
    private int m_adjustedHours;
    private int m_balance;

    /**
     * loaded constructor
     */
    public TimeBankSummaryData(
            String timeBankName,
            Date startDate, 
            Date endDate,
            Date proRateDate,
            int targetHours,
            int prTargetHours,
            int scheduledHours,
            int adjustedHours,
            int balance) 
    {
        m_timeBankName = timeBankName;
        m_startDate = startDate;
        m_endDate = endDate;
        m_proRateDate = proRateDate;
        m_targetHours = targetHours;
        m_scheduledHours = scheduledHours;
        m_prTargetHours = prTargetHours;
        m_adjustedHours = adjustedHours;
        m_balance = balance;
    }

    public String getTimeBankName(){
        return m_timeBankName;
    }
    
    public Date getStartDate(){
        return m_startDate;
    }
    
    public Date getEndDate(){
        return m_endDate;
    }
    
    public Date getProRateDate(){
        return m_proRateDate;
    }
    
    public int getTargetHours(){
        return m_targetHours;
    }

    public int getPrTargetHours() {
        return m_prTargetHours;
    }

    public int getScheduledHours() {
        return m_scheduledHours;
    }

    public int getAdjustedHours() {
        return m_adjustedHours;
    }

    public int getBalance() {
        return m_balance;
    }
}
