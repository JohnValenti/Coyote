package com.bluepumpkin.web.fs.schedule.summary;

/**
 * Title:        ShiftSwapSummaryData
 * Description:  Value Object for Shift Swap summary data for one employee for UI display
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, inc
 * @author       Pavel Bosin, David Su
 * @version      1.0
 */
public class ShiftSwapSummaryData extends RequestSummaryData {
	private int m_postings;
	private int m_requestsNegotiation;

	/**
	 * loaded constructor
	 */
	public ShiftSwapSummaryData(
			int postings,
			int requestsTotal,
			int requestsExpired,
			int requestsNegotiation,
			int requestsPending,
			int requestsApproved,
			int requestsDenied,
			int requestsInvalid) {
		super(requestsTotal, requestsExpired, requestsPending, requestsApproved,
				requestsDenied,	requestsInvalid);
		m_postings = postings;
		m_requestsNegotiation = requestsNegotiation;
	}

	public int getPostings(){
		return m_postings;
	}

	public int getRequestsNegotiation() {
		return m_requestsNegotiation;
	}
}
