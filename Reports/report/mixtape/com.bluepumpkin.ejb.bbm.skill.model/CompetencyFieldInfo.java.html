import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class CompetencyFieldInfo extends FieldInfo {

    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// field id constants
    public static final int COMPETENCY_ID = 0;
    public static final int COMPETENCY_EXTERNALIDENT = 1;
    public static final int COMPETENCY_NAME = 2;
    public static final int COMPETENCY_DESCRIPTION = 3;
    public static final int COMPETENCY_CREATIONTIME = 4;
    public static final int COMPETENCY_MODIFICATIONTIME = 5;
	public static final int COMPETENCY_ISDELETED = 6;

    private static final int COMPETENCY_NUMOFFIELDS = 7;

    private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[COMPETENCY_NUMOFFIELDS];
    private static String m_strTableName = "COMPETENCY";

    // static initialization
    static {
      m_arrFieldInfo[COMPETENCY_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
      m_arrFieldInfo[COMPETENCY_EXTERNALIDENT] = new FieldInfoEntry("EXTERNALIDENT", PrimitiveTypes.STRING);
      m_arrFieldInfo[COMPETENCY_NAME] = new FieldInfoEntry("NAME", PrimitiveTypes.STRING);
      m_arrFieldInfo[COMPETENCY_DESCRIPTION] = new FieldInfoEntry("DESCRIPTION", PrimitiveTypes.STRING);
      m_arrFieldInfo[COMPETENCY_CREATIONTIME] = new FieldInfoEntry("CREATIONTIME", PrimitiveTypes.DATETIME);
      m_arrFieldInfo[COMPETENCY_MODIFICATIONTIME] = new FieldInfoEntry("MODIFICATIONTIME", PrimitiveTypes.DATETIME);
      m_arrFieldInfo[COMPETENCY_ISDELETED] = new FieldInfoEntry("ISDELETED", PrimitiveTypes.BOOLEAN);
    }

    public CompetencyFieldInfo() {
        super();
    }

    // implement abstract methods in parent class
    public String getTableName() { return m_strTableName; }
    public int getIDFieldIndex() { return COMPETENCY_ID; }
    public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }
    
    public static String getColumnName( int columnNo )
    {
    	if( columnNo >= m_arrFieldInfo.length )
    		return "";
    	
    	return m_arrFieldInfo[columnNo].getFieldName();
    }

}
