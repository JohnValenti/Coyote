package com.verint.ejb.rfs;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.EJBHomeFactory;
import com.bluepumpkin.ejb.bbm.Log;
import com.verint.ejb.rfs.reports.ejb.RfsReportsManager;
import com.verint.ejb.rfs.reports.ejb.RfsReportsManagerHome;


public class RfsManagerFactory {
	private static Category LOG = 
			Log.initCategory( RfsManagerFactory.class.getName() );
	public static RfsReportsManager getRfsReportsManager() throws  RfsEJBCreateException{
		try {
			EJBHomeFactory factory = EJBHomeFactory.getFactory();
			RfsReportsManagerHome mgrHome =
				(RfsReportsManagerHome) factory.lookUpHome(
					JNDINames.RFS_REPORTS_MGR_EJBHOME,
					RfsReportsManagerHome.class);

			return mgrHome.create();
		} catch (Exception e) {
			log(e);
			throw new RfsEJBCreateException("RfsReportsManager", e);
		}
	}

	private static void log(String message) {
		LOG.debug(message);
	}

	private static void log(Throwable ex) {
		LOG.error(ex);
	}

}
