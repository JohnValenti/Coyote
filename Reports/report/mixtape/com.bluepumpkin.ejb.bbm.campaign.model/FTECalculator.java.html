	private Map<ID, Double> m_hEmployeeIDsToWages;

	private TraceCube m_tRequirements;

	// Collection of ForecastProfileProject's. If not null, we also calculate the Project Statistics.
	private Collection<ForecastProfileProject> m_projects; 

	private Map<ID, String> m_projectQueueIdNameMap;
	private Map<ID, ID> m_projectQueueIdSIDMap;
	private Map<ID, Collection<ID>> m_hActivityQueues;

	private double m_defaultWage;

	/* 
	*	This will be true if at least one employee/phantom has a wage of $0.00. This indicates that we need to use the default 
	*	wage when calculating. 
	*/
	private boolean m_bUseDefaultWage; 

	// the ID of the scheduling period. We only count events for this SP.
	private ID m_SPID;

	// a map of the sp's employee/phantom ID's -> the filtered events for each
	private Map<ID, Collection<Event>> m_wrIdsToEventsMap;


	// A collection of the employee/phantom ID's to be calculated.
	private Collection<ID> wrIdsToCalculate;
	
	/**
	 * Constructor.
	 *
	 * @param hEmployeeIDsToWages         - A an ordered Map of employee ID to employee wage (Double)
	 *                                    for all of the employees in the SP
	 * @param hPhantomIDsToEmpTemplateIDs - An ordered map of PhantomIDs to EmployeeTemplateIDs
	 * @param cEmpTemplates               - The EmployeeTemplates corresponding to the Phantoms in
	 *                                    cPhantoms, followed by any remaining Staffing Profiles
	 *                                    (EmployeeTemplates) for this scheduling period that have not
	 *                                    been scheduled.
	 * @param tRequirements               - The Required FTE TraceCube for the SP
	 * @param hActivityMedias             - Map of Activity (key) and collection of Media Ids
	 *                                    (value), for all activities in the schedules for the SP
	 * @param dtStart                     - Start date for the SP
	 * @param dtEnd                       - End date for the SP
	 * @param projects                    - Collection of ForecastProfileProject's. If not null, we also
	 *                                    calculate the Project Statistics.
	 * @param projectQueueIdNameMap       - a map of project queue DEID's to names
	 * @param projectQueueIdSIDMap        - a map of project queue DEID to SID
	 * @param hActivityQueues             - a map of activityIDs (SID) to queueIDs (SID)
	 * @param spID                        - the ID of the scheduling period. We only count events for
	 *                                    this SP.
	 * @param wrIdsToEventsMap            - a map of the selected employee/phantom ID's -> the filtered
	 *                                    events for each.
	 * @param defaultWage                 - The default wage to use when the employee's wage is $0.0,
	 *                                    for calculating the "Total Scheduled Cost".
	 */
	public FTECalculator(
			Map<ID, Double> hEmployeeIDsToWages,
			Map<ID, ID> hPhantomIDsToEmpTemplateIDs,
			Collection<EmployeeTemplate> cEmpTemplates,
			TraceCube tRequirements,
			Map<ID, Collection<ID>> hActivityMedias,
			Date dtStart,
			Date dtEnd,
			Collection<ForecastProfileProject> projects,
			Map<ID, String> projectQueueIdNameMap,
			Map<ID, ID> projectQueueIdSIDMap,
			Map<ID, Collection<ID>> hActivityQueues,
			ID spID,
			Map<ID, Collection<Event>> wrIdsToEventsMap,
			double defaultWage) {
		m_hEmployeeIDsToWages = hEmployeeIDsToWages;
		m_hPhantomIDsToEmpTemplateIDs = hPhantomIDsToEmpTemplateIDs;
		m_cEmpTemplates = cEmpTemplates;
		m_tRequirements = tRequirements;
		m_hActivityMedias = hActivityMedias;
		m_dtStart = dtStart;
		m_dtEnd = dtEnd;
		m_projects = projects;
		m_projectQueueIdNameMap = projectQueueIdNameMap;
		m_projectQueueIdSIDMap = projectQueueIdSIDMap;
		m_hActivityQueues = hActivityQueues;
		m_bUseDefaultWage = false;
		m_SPID = spID;
		m_wrIdsToEventsMap = wrIdsToEventsMap;
		wrIdsToCalculate = m_wrIdsToEventsMap == null ? null : new ArrayList<ID>(m_wrIdsToEventsMap.keySet());
		m_defaultWage = defaultWage;
	}

	/**
	 * @param wrID              - The workResource ID (Employee or Phantom)
	 * @param cShiftAssignments
	 * @param dWage
	 * @param hActivityMedias
	 * @param dtStart
	 * @param anStaffing
	 * @param fteCalc
	 * @param pStat
	 */
	private void countHours(ID wrID, Collection<Event> cShiftAssignments, double dWage,
			Map<ID, Collection<ID>> hActivityMedias, Date dtStart, int[] anStaffing, FTECalculation fteCalc, 
			FTECalculation.TemplateStat pStat) {
		int nPaid = 0;
		int nContact = 0;

		// flatten the Event collection first.
		Collection<SimpleEvent> colFlattenedEvents = EventUtils.convertEventsToTimelineForSingleEmployee(cShiftAssignments,
			null, null, null, true);

		for (SimpleEvent pEvent : colFlattenedEvents) {
			// if the event has an SPID, then we must verify it. Otherwise, allow it (ex: a calendar event)
			if (pEvent.getSPID() == null || pEvent.getSPID().equals(m_SPID)) {
				if (pEvent.getPaid()) {
					nPaid += pEvent.getDuration();
				}

				Collection<ID> cMedias = hActivityMedias.get(pEvent.getActivityID());
				if (cMedias != null && !cMedias.isEmpty()) {
					nContact += pEvent.getDuration();
					incStaffing(anStaffing, dtStart, pEvent.getStartTime(), pEvent.getDuration());

					if (cMedias.contains(Media.MEDIA_ID_PROJECT)) {
						updateProjectStats(fteCalc, wrID, dWage, pEvent);
					}
				}
			}
		}

		fteCalc.m_nScheduledContact += nContact;
		fteCalc.m_nScheduledPaid += nPaid;
		fteCalc.m_dScheduledCost += (nPaid * dWage / 60.0);

		if (pStat != null) {
			// this is a non-hidden Phantom
			pStat.nContactHours += nContact;
			pStat.nPaidHours += nPaid;
			pStat.dHourlyCost += (nPaid * dWage / 60.0);
		}
	}

	/**
	 * Update project statistics for an event.
	 *
	 * @param wrID    - The workResource ID (Employee or Phantom)
	 */
	private void updateProjectStats(FTECalculation fteCalc, ID wrID, double dWage, Event pEvent) {
		Collection<ForecastProfileProject> projects = fteCalc.m_projects;

		if (fteCalc.m_activityQueues != null) {
			Collection<ID> activityQueueIDs = fteCalc.m_activityQueues.get(pEvent.getActivityID());

			if (activityQueueIDs != null && !activityQueueIDs.isEmpty()) {
				for (ForecastProfileProject project : projects) {

					ID projectQueueID = project.getQueueID();
					// get the queue SID for this DEID
					projectQueueID = fteCalc.m_projectQueueIdSIDMap.get(projectQueueID);

					if (projectQueueID != null && activityQueueIDs.contains(projectQueueID)) {
						TimeRange projectRange = new TimeRange(project.getStartTime(), project.getEndTime());
						TimeRange eventRange = new TimeRange(pEvent.getStartTime(), new Date(pEvent.getEndTime().getTime()));
						Double overlapSeconds = projectRange.getOverlapDuration(eventRange) / 1000.0;

						if (overlapSeconds > 0.0) {
							ID projectID = project.getID();
							fteCalc.addProjectEmployee(projectID, wrID);
							fteCalc.addProjectScheduled(projectID, overlapSeconds.intValue());
							fteCalc.addProjectCost(projectID, (overlapSeconds / 3600.0d) * dWage);
						}
					}
				}
			}
		}
	}

	private void incStaffing(int[] anStaffing, Date dtStart, Date dtIncStart, int nDuration) {
		int nOffset = (int) (dtIncStart.getTime() - dtStart.getTime()) / (1000 * 60);
		for (int n = 0; n < nDuration; n++) {
			if (nOffset + n >= 0 && nOffset + n < anStaffing.length) {
				anStaffing[nOffset + n]++;
			}
		}
	}

	/**
	 * @param defaultWage - The default wage to use when the emplyee's wage is $0.0, for
	 *                    calculating the "Total Scheduled Cost".
	 */
	public void setWage(double defaultWage) {
		m_defaultWage = defaultWage;
	}

	/**
	 * Calculate and return the FTECalculation data for the Web suite's FTE
	 * Calculator. You must call setEventsAndWage() prior to this method.
	 *
	 * @return The FTECalculation data for the SP
	 */
	public FTECalculation calculateFTE() {
		FTECalculation fteCalc = new FTECalculation();
		int nMinutes = (int) ((m_dtEnd.getTime() - m_dtStart.getTime()) / 60000);
		int[] anStaffing = new int[nMinutes];

		Arrays.fill(anStaffing, 0);

		if (m_projects != null) {
			fteCalc.initProjectStatistics(m_projects, m_projectQueueIdNameMap, m_projectQueueIdSIDMap, m_hActivityQueues);
		}

		for (ID employeeID : m_hEmployeeIDsToWages.keySet()) {
			if (wrIdsToCalculate.contains(employeeID)) {
				Collection<Event> cShiftAssignments = m_wrIdsToEventsMap.get(employeeID);
				if (cShiftAssignments != null) {
					double dWage = m_hEmployeeIDsToWages.get(employeeID);
					if (dWage == 0.0d) {
						dWage = m_defaultWage;
						m_bUseDefaultWage = true;
					}
					countHours(employeeID, cShiftAssignments, dWage, m_hActivityMedias, m_dtStart, anStaffing, fteCalc, null);
				}
			}
		}

		Iterator<EmployeeTemplate> etIt = m_cEmpTemplates.iterator();
		for (ID phantomID : m_hPhantomIDsToEmpTemplateIDs.keySet()) {
			if (wrIdsToCalculate.contains(phantomID)) {
				ID empTemplateID = m_hPhantomIDsToEmpTemplateIDs.get(phantomID);
				EmployeeTemplate empTemplate = etIt.next();

				FTECalculation.TemplateStat pStat = null;
				if (!empTemplate.getIsHidden()) {
					/*
					 *	This template corresponds to a Phantom that was created from a true staffing profile (rather than
					 *  having been converted from a real agent's schedule). It will appear in the "Staffing Profile Statistics".
					 */
					pStat = fteCalc.m_hTemplateStats.get(empTemplateID);
					if (pStat == null) {
						// get a new TemplateStat. Does not affect fteCalc.
						pStat = fteCalc.addTemplateStat(empTemplate.getName());

						// add the TemplateStat to fteCalc
						fteCalc.m_hTemplateStats.put(empTemplateID, pStat);
					}
					pStat.nNumberScheduled++;
				}

				Collection<Event> cShiftAssignments = m_wrIdsToEventsMap.get(phantomID);
				if (cShiftAssignments != null) {
					double dWage = empTemplate.getWage();
					if (dWage == 0.0d) {
						dWage = m_defaultWage;
						m_bUseDefaultWage = true;
					}
					countHours(phantomID, cShiftAssignments, dWage, m_hActivityMedias, m_dtStart, anStaffing, fteCalc, pStat);
				}
			}
		}

		while (etIt.hasNext()) {
			// Add any remaining Staffing Profiles linked to this SP that have not been scheduled.
			EmployeeTemplate empTemplate = etIt.next();
			if (!empTemplate.getIsHidden()) {
				FTECalculation.TemplateStat pStat = fteCalc.m_hTemplateStats.get(empTemplate.getID());
				if (pStat == null) {
					// This template is a staffing profile that hasn't been scheduled. We will set all calculations to 0.
					
					// get a new TemplateStat. Does not affect fteCalc.
					pStat = fteCalc.addTemplateStat(empTemplate.getName()); 
					
					// add the TemplateStat to fteCalc
					fteCalc.m_hTemplateStats.put(empTemplate.getID(), pStat); 
				}
			}
		}

		if (m_tRequirements != null) {
			int nLength = m_tRequirements.getMaximumLength(new short[]{Trace.FTE}) - 1;
			fteCalc.m_nRequired = (int) (60 * TraceOperator.aggregatePeriod(m_tRequirements, Trace.FTE, 0, nLength));
		}

		for (int n = 0; n < anStaffing.length; n++) {
			if (anStaffing[n] != -1) {
				if (anStaffing[n] < fteCalc.m_nMinConcurrant) {
					fteCalc.m_nMinConcurrant = anStaffing[n];
				}
				if (anStaffing[n] > fteCalc.m_nMaxConcurrant) {
					fteCalc.m_nMaxConcurrant = anStaffing[n];
				}
			}
		}
		return fteCalc;
	}

	public void refreshSchedule(Map<ID, Collection<Event>> newSchedule, Map<ID, Collection<ID>> activityMedias,
			Map<ID, Collection<ID>> activityQueues) {
		m_wrIdsToEventsMap = newSchedule;
		wrIdsToCalculate = new ArrayList<ID>(m_wrIdsToEventsMap.keySet());
		this.m_hActivityMedias = activityMedias;
		this.m_hActivityQueues = activityQueues;
	}

	public void setActivityQueues(Map<ID, Collection<ID>> activityQueues) {
		this.m_hActivityQueues = activityQueues;
	}

	public void setWrIdsToCalculate(Collection<ID> wrIdsToCalculate) {
		this.wrIdsToCalculate = wrIdsToCalculate;
	}

	public boolean useDefaultWage() {
		return m_bUseDefaultWage;
	}

	public void setDefaultWage(double defaultWage) {
		this.m_defaultWage = defaultWage;
	}

	public Map<ID, Collection<Event>> getWrIdsToEventsMap() {
		return m_wrIdsToEventsMap;
	}
}
