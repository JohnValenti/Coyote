package com.bluepumpkin.ejb.bbm.campaign.model;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class SchedulingWarning extends ValueObjectBase implements Comparable<SchedulingWarning> {

	private static final long serialVersionUID = 7309911177527415907L;

	private static FieldInfo m_fieldInfo = new SchedulingWarningFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public SchedulingWarning() {
		super();
	}

	public int getTypeIndex() {
		return this.getFieldValueInt(SchedulingWarningFieldInfo.TYPEINDEX);
	}

	public String getTypeString() {
		return this.getFormattedFieldValue(SchedulingWarningFieldInfo.TYPESTRING);
	}

	public String getWarning() {
		return this.getFormattedFieldValue(SchedulingWarningFieldInfo.WARNINGSTRING);
	}

	/**
	 * compare two warnings based on id, so we can display them the same order
	 * as F&S
	 */
	@Override
	public int compareTo(SchedulingWarning o) {
		SchedulingWarning item2 = o;

		int lDiff = getID().toInt() - item2.getID().toInt();

		if (lDiff < 0) {
			return -1;
		}
		if (lDiff > 0) {
			return 1;
		}
		return 0;
	}
}
