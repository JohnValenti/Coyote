package com.bluepumpkin.ejb.bbm.campaign.model;

/**
 * Class to store flags indicating which optional parts of an SP will be copied/cloned.
 */

public class CloneSPOptions {

	/**
	 * Stores the "Copy employee Min/Max hours, skill, and work pattern assignments" setting.
	 */
	private final boolean bCopyAssignments;

	/**
	 * Stores the "Copy Project Queue Requirements and Outbound Call Lists to partial weeks" setting.
	 */
	private final boolean bCopyProjectData;

	/**
	 * Constructor.
	 *
	 * @param copyAssignments - whether to "Copy employee Min/Max hours, skill, and work pattern assignments".
	 * @param copyProjectData - whether to "Copy Project Queue Requirements and Outbound Call Lists to partial weeks".
	 */
	public CloneSPOptions(boolean copyAssignments, boolean copyProjectData) {
		bCopyAssignments = copyAssignments;
		bCopyProjectData = copyProjectData;
	}

	/**
	 * Constructor. bCopyProjectData is set to true for compatibility with weekly SP's,
	 * since you cannot disable bCopyProjectData for them.
	 *
	 * @param copyAssignments - whether to "Copy employee Min/Max hours, skill, and work pattern assignments".
	 */
	public CloneSPOptions(boolean copyAssignments) {
		bCopyAssignments = copyAssignments;
		bCopyProjectData = true;
	}

	/**
	 * Default constructor. bCopyProjectData is set to true for compatibility with weekly SP's,
	 * since you cannot disable bCopyProjectData for them.
	 */
	public CloneSPOptions() {
		bCopyAssignments = false;
		bCopyProjectData = true;
	}

	public boolean isCopyAssignments() {
		return bCopyAssignments;
	}

	public boolean isCopyProjectData() {
		return bCopyProjectData;
	}
}
