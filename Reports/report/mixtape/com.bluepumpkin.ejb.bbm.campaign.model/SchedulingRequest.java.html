package com.bluepumpkin.ejb.bbm.campaign.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class SchedulingRequest extends ValueObjectBase {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new SchedulingRequestFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	// Values for the REQUEST field.
	public static final int CANCEL_SCHEDULE = 1;
	public static final int STOP_AND_SAVE = 2;
	public static final int CONTINUE_SCHEDULE = 3;

	public SchedulingRequest() {
		super();
	}

	public ID getSPSID() {
		return this.getFieldValueID(SchedulingRequestFieldInfo.SPID);
	}

	public void setSPSID(ID spSID) {
		this.setFieldValue(SchedulingRequestFieldInfo.SPID, spSID);
	}

	public int getRequest() {
		return this.getFieldValueInt(SchedulingRequestFieldInfo.REQUEST);
	}

	public void setRequest(int request) {
		this.setFieldValue(SchedulingRequestFieldInfo.REQUEST, request);
	}

	public String getMode() {
		return this.getFormattedFieldValue(SchedulingRequestFieldInfo.MODE);
	}

	public void setMode(String mode) {
		this.setFieldValue(SchedulingRequestFieldInfo.MODE, mode);
	}
}
