package com.bluepumpkin.ejb.bbm.campaign.model;

import java.util.Date;
import java.util.TimeZone;
import java.util.Calendar;

import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.ejb.bbm.vo.*;
import com.bluepumpkin.ejb.bbm.effectivity.*;

/**
 * This value object represents hours of operation for a single week in a
 * Scheduling Period. Each day of the week is represented with an integer value
 * from 1-7 (1 - Sunday, 7 - Saturday), matching Java's values for
 * Calendar.DAY_OF_WEEK.
 * <p>
 * Each day in the week has a pair of integer values representing the start time
 * and end time in minutes offset from midnight.
 * <p>
 * 24 hour operation (no closing time) is represented as a start time of 0 and
 * an end time of 1440 for every day in the week.
 */
public class CampaignHOO extends ValueObjectEffectivity implements HOOAssignment {
	private static final long serialVersionUID = 7994398874790153561L;

	// 24 hours * 60 minutes
	static final short MINUTES_IN_DAY = 1440;

	private static FieldInfo m_fieldInfo = new CampaignHOOFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	private LocalDate m_dtStartLocal, m_dtEndLocal;

	public CampaignHOO() {
		super();

		// set default day offset: 1440 == 24*60 is the minutes in a day
		for (short i = Calendar.SUNDAY; i <= Calendar.SATURDAY; i++) {
			setDayOpen(i, (short) 0);
			setDayClose(i, MINUTES_IN_DAY);
		}
	}

	@Override
	public ID getParentID() {
		return getCampaignID();
	}

	@Override
	public void setParentID(ID id) {
		setCampaignID(id);
	}

	public ID getCampaignID() {
		return getFieldValueID(CampaignHOOFieldInfo.CAMPAIGNHOO_CAMPAIGNID);
	}

	public ID getSPID() {
		return getFieldValueID(CampaignHOOFieldInfo.CAMPAIGNHOO_SPID);
	}

	@Override
	public Date getStartTime() {
		return getFieldValueDate(CampaignHOOFieldInfo.CAMPAIGNHOO_STARTTIME);
	}

	@Override
	public Date getEndTime() {
		return getFieldValueDate(CampaignHOOFieldInfo.CAMPAIGNHOO_ENDTIME);
	}

	public void setCampaignID(ID id) {
		setFieldValue(CampaignHOOFieldInfo.CAMPAIGNHOO_CAMPAIGNID, id);
	}

	public void setSPID(ID id) {
		setFieldValue(CampaignHOOFieldInfo.CAMPAIGNHOO_SPID, id);
	}

	@Override
	public void setStartTime(Date dt) {
		setFieldValue(CampaignHOOFieldInfo.CAMPAIGNHOO_STARTTIME, dt);
	}

	@Override
	public void setEndTime(Date dt) {
		setFieldValue(CampaignHOOFieldInfo.CAMPAIGNHOO_ENDTIME, dt);
	}

	public LocalDate getStartTimeLocal() {
		return m_dtStartLocal;
	}

	public void setStartTimeLocal(LocalDate dt) {
		m_dtStartLocal = dt;
	}

	public LocalDate getEndTimeLocal() {
		return m_dtEndLocal;
	}

	public void setEndTimeLocal(LocalDate dt) {
		m_dtEndLocal = dt;
	}

	/**
	 * return a day's open time as the minutes offset to the day's midnight
	 *
	 * @iDay from 1 to 7 indicate SUN, MON, TUE, ..., SAT
	 */
	@Override
	public short getDayOpen(short iDay) {
		if (iDay < Calendar.SUNDAY) {
			iDay = Calendar.SUNDAY;
		}
		if (iDay > Calendar.SATURDAY) {
			iDay = Calendar.SATURDAY;
		}

		return getFieldValueShort((iDay - 1) * 2 + CampaignHOOFieldInfo.CAMPAIGNHOO_SUNSTARTOFFSET);
	}

	/**
	 * return a day's close time as the minutes offset to the day's midnight
	 *
	 * @iDay from 1 to 7 indicate SUN, MON, TUE, ..., SAT
	 */
	@Override
	public short getDayClose(short iDay) {
		if (iDay < Calendar.SUNDAY) {
			iDay = Calendar.SUNDAY;
		}
		if (iDay > Calendar.SATURDAY) {
			iDay = Calendar.SATURDAY;
		}

		return getFieldValueShort((iDay - 1) * 2 + CampaignHOOFieldInfo.CAMPAIGNHOO_SUNENDOFFSET);
	}

	/**
	 * set a day's open time as the minutes offset to the day's midnight
	 *
	 * @iDay from 1 to 7 indicate SUN, MON, TUE, ..., SAT
	 */
	@Override
	public void setDayOpen(short iDay, short iOffset) {
		if (iDay < Calendar.SUNDAY) {
			iDay = Calendar.SUNDAY;
		}
		if (iDay > Calendar.SATURDAY) {
			iDay = Calendar.SATURDAY;
		}

		setFieldValue((iDay - 1) * 2 + CampaignHOOFieldInfo.CAMPAIGNHOO_SUNSTARTOFFSET, iOffset);
	}

	/**
	 * set a day's close time as the minutes offset to the day's midnight
	 *
	 * @iDay from 1 to 7 indicate SUN, MON, TUE, ..., SAT
	 */
	@Override
	public void setDayClose(short iDay, short iOffset) {
		if (iDay < Calendar.SUNDAY) {
			iDay = Calendar.SUNDAY;
		}
		if (iDay > Calendar.SATURDAY) {
			iDay = Calendar.SATURDAY;
		}

		setFieldValue((iDay - 1) * 2 + CampaignHOOFieldInfo.CAMPAIGNHOO_SUNENDOFFSET, iOffset);
	}

	/**
	 * check if it's 24 * 7, i.e. time range is 0 to 1440 for each day
	 */
	@Override
	public boolean isOpen24Hours() {
		for (short i = Calendar.SUNDAY; i <= Calendar.SATURDAY; i++) {
			if (getDayOpen(i) != 0 || getDayClose(i) != 1440) {
				return false;
			}
		}

		return true;
	}

	/**
	 * set hour's of operation as 24 * 7, i.e. time ragne is 0 to 1440 for each
	 * day
	 */
	@Override
	public void setOpen24Hours() {
		for (short i = Calendar.SUNDAY; i <= Calendar.SATURDAY; i++) {
			setDayOpen(i, (short) 0);
			setDayClose(i, MINUTES_IN_DAY);
		}
	}

	/**
	 * check if a day is active, (-1, -1) for (dayopen, dayclose) is inactive
	 * otherwise it's active
	 */
	@Override
	public boolean isActive(short iDay) {
		return !(getDayOpen(iDay) == -1 && getDayClose(iDay) == -1);
	}

	/**
	 * Returns true if at least one day has hours of operation (is active)
	 * during the week.
	 */
	public boolean isActiveDuringWeek() {
		for (short i = Calendar.SUNDAY; i <= Calendar.SATURDAY; i++) {
			if (isActive(i)) {
				return true;
			}
		}
		return false;
	}

	/**
	 * if bActive = true, set (dayopen, dayclose) to be (0, 1440) otherwise, set
	 * them to be (-1, -1)
	 */
	@Override
	public void setActive(short iDay, boolean bActive) {
		if (bActive) {
			setDayOpen(iDay, (short) 0);
			setDayClose(iDay, MINUTES_IN_DAY);
		} else {
			setDayOpen(iDay, (short) -1);
			setDayClose(iDay, (short) -1);
		}
	}

	// clone the assignment object without the ID field
	@Override
	public TimePeriod cloneTimePeriod() {
		CampaignHOO item = new CampaignHOO();

		item.setCampaignID(getCampaignID());
		item.setStartTime(getStartTime());
		item.setEndTime(getEndTime());

		for (short i = Calendar.SUNDAY; i <= Calendar.SATURDAY; i++) {
			item.setDayOpen(i, getDayOpen(i));
			item.setDayClose(i, getDayClose(i));
		}

		return item;
	}

	/**
	 * Returns the opening or closing time, applied to the specified day in the
	 * specified time zone.
	 *
	 * @param dayBoundary a reference time which is taken to be the start of the
	 *                    business day. Note that if the opening time falls before this
	 *                    time and opening time has been requested, the returned
	 *                    {@code Date} will fall before {@code dayBoundary}.
	 * @param tz          the time zone within which the opening and closing times are
	 *                    to be interpreted.
	 * @param bOpenTime   if {@code true} this method returns the opening time;
	 *                    otherwise this method returns the closing time.
	 * @return
	 */
	// TODO: Seems to me that there shouldn't be a time zone here at all.
	// Aren't HOOs specified in minutes from the start of the SP, which would be
	// minutes from getStartDate? We should be able to calculate iDay simply
	// by getting the difference in days from getStartDate() to dayBoundary.
	private Date getOffSetDay(Date dayBoundary, TimeZone tz, boolean bOpenTime) {
		// set midnight day to calendar
		Calendar cal = Calendar.getInstance(tz);
		cal.setTime(dayBoundary);

		short iDay = (short) cal.get(Calendar.DAY_OF_WEEK);
		if (!isActive(iDay)) {
			return null;
		}

		int iOffset = (bOpenTime ? getDayOpen(iDay) : getDayClose(iDay));

		cal.set(Calendar.HOUR_OF_DAY, iOffset / 60);
		cal.set(Calendar.MINUTE, iOffset % 60);
		cal.set(Calendar.SECOND, 0);
		cal.set(Calendar.MILLISECOND, 0);

		return cal.getTime();
	}

	/**
	 * return the day open time for a given day, return null if the day is not
	 * open
	 *
	 * @param dayBoundary the campaign day boundary expressed as a <code>Date</code>.
	 */
	@Override
	public Date getDayOpen(Date dayBoundary, TimeZone tz) {
		return this.getOffSetDay(dayBoundary, tz, true);
	}

	/**
	 * return the day close time for a given day, return null if the day is not
	 * open
	 *
	 * @param dayBoundary the campaign day boundary expressed as a <code>Date</code>.
	 */
	@Override
	public Date getDayClose(Date dayBoundary, TimeZone tz) {
		return this.getOffSetDay(dayBoundary, tz, false);
	}

}
