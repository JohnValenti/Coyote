package com.bluepumpkin.ejb.bbm.campaign.model;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;

/**
 * This class holds the ID of the SPQueues by QueueID and SpID. This is needed
 * to lookup the SPQueueID based on the QueueID and the SpID
 *
 */
public class SPQueueIDMap implements Serializable {

	private static final long serialVersionUID = 2729387389820923250L;

	private final Map<ID, Map<ID, ID>> spQueueIDBySpIDByQueueID = new HashMap<ID, Map<ID, ID>>();

	/**
	 * Loads the data from the JdmoRowset. The following columns are expected
	 * MEDIAID, QUEUEID, SPID, SPQUEUEID. All the parameters in each row are
	 * expected to be non-null.
	 */
	public SPQueueIDMap(JdmoRowset rs) throws JdmoException {
		if (rs == null) {
			throw new IllegalArgumentException("The result set can not be null!");
		}
		while (rs.next()) {
			String mediaId = rs.getString("MEDIAID");
			if (mediaId != null) {
				setSPQueueID(rs.getID("QUEUEID"), rs.getID("SPID"), rs.getID("SPQUEUEID"));
			}
		}
	}

	/**
	 * Returns the SPQueue ID for the given queue id and spid.Returns null if a
	 * match cannot be found
	 */
	public ID getSPQueueID(ID queueID, ID spID) {
		Map<ID, ID> spQueueIdBySpId = spQueueIDBySpIDByQueueID.get(queueID);
		if (spQueueIdBySpId != null) {
			return spQueueIdBySpId.get(spID);
		}
		return null;
	}

	/**
	 * Sets the ID for the SPQueue for a given SP and the Queue.
	 */
	private void setSPQueueID(ID queueID, ID spID, ID spqueueID) {
		Map<ID, ID> spqIdBySPId = spQueueIDBySpIDByQueueID.get(queueID);
		if (spqIdBySPId == null) {
			spqIdBySPId = new HashMap<ID, ID>();
			spQueueIDBySpIDByQueueID.put(queueID, spqIdBySPId);
		}
		spqIdBySPId.put(spID, spqueueID);
	}

}
