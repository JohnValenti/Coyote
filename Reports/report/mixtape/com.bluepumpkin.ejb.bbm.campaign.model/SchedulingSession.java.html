package com.bluepumpkin.ejb.bbm.campaign.model;

public class SchedulingSession implements java.io.Serializable {

	/**
	 * 
	 */
	private static final long serialVersionUID = -2798465089559134428L;

	private String m_userName;
	private SchedulingPeriod m_sp;
	
	public SchedulingSession () {}
	
	public String getUserName() {
		return m_userName;
	}
	
	public void setUserName(String userName) {
		this.m_userName = userName;
	}
	
	public SchedulingPeriod getSchedulingPeriod() {
		return m_sp;
	}
	
	public void setSchedulingPeriod(SchedulingPeriod sp) {
		this.m_sp = sp;
	}
	
	/**
	* print object
	*/
	public String toString () {
		StringBuffer buffer = new StringBuffer();
		buffer.append("\nusername: "+m_userName==null?"":m_userName);
		buffer.append("\nSP: "+m_sp==null?"":m_sp);
		return buffer.toString();
	}
}
