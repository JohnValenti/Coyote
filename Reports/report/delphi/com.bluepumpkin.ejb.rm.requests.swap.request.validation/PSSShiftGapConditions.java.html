package com.bluepumpkin.ejb.rm.requests.swap.request.validation;

/**
 *
 * <b>Soft Validation: applies only for partial shift swap requests</b>
 * <p>
 *    Validates the gap activity conditions
 *    - gap between RANGE_MIN_DURATION_VALUE and RANGE_MAX_DURATION_VALUE
 *    or
 *    - gap less than MIN_DURATION_VALUE
 * <p>
 *    Checks for both the shifts for a two way swap
 */
public class PSSShiftGapConditions  implements Validator
{
    private static final String m_className = PSSShiftGapConditions.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);
    
    public static final String RANGE_MIN_DURATION_VALUE = "RANGE_MIN_DURATION_VALUE";
    public static final String RANGE_MAX_DURATION_VALUE = "RANGE_MAX_DURATION_VALUE";
    public static final String MIN_GAP_DURATION_VALUE = "MIN_GAP_DURATION_VALUE";
    public static final ID RULE_ID = new ID(-194036);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    @Override
	public ValidationResult validate(Validatable validatable) throws Exception
    {
        String methodName = "validate";
        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));

        ShiftSwapRequest ssr = (ShiftSwapRequest) validatable;
        ShiftSwapValidationCache valCache = ssr.getCache();
        
        ShiftSwapItem ssItem1 = (ShiftSwapItem) ssr.getShiftSwapItems().get(0);
        ShiftSwapItem ssItem2 = (ShiftSwapItem) ssr.getShiftSwapItems().get(1);
        if (ssItem1.getIsPartial() || ssItem2.getIsPartial()) {
//        	get the resulting shifts after the swap
            ShiftAssignment[] saArray = null;
//        	get the sa array after the swap
        	saArray = ShiftSwapRequestUtil.performPartialSwap((ShiftSwapRequest)validatable, false, 
        			WfmManagerFactory.getScheduleAccessManager(), false, false);
        	
        	//for each org, find out the validator- if beeded to run for both or not
        	for (Iterator it = ssr.getValidationCache().getOrgIDs().iterator(); it.hasNext();){
	        	ID orgID = (ID)it.next();
	        	if (valCache.isValidationRequiredForOrg(orgID, "shift-swap", m_className)){
		        	//get the rule params
		        	ValidationRuleManager valMan = RmManagerFactory.getInstance().getValidationRuleManager();
					Map<String, Integer> ruleParams = valMan.getValidationRuleParams(orgID, RULE_ID);
		        	int minDuration = ruleParams.get(RANGE_MIN_DURATION_VALUE).intValue();
		        	int maxDuration = ruleParams.get(RANGE_MAX_DURATION_VALUE).intValue();
		        	int minGap = ruleParams.get(MIN_GAP_DURATION_VALUE).intValue();
		        	
		        	ScheduleAccessManager sam = RequestUtil.getScheduleAccessManager();
		        	
		        	return verifyGapCondition(ssr, saArray, minDuration, maxDuration, minGap, sam);
	        	}
        	}
        }
        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName));
        return null;
    }
    
    private ValidationResult verifyGapCondition(ShiftSwapRequest ssr, ShiftAssignment[] saArray,
    		int minDuration, int maxDuration, int minGap, ScheduleAccessManager sam) throws Exception{
    	
    	ValidationResult result = null;
    	//validate for each shift
    	for (int i = 0; i< saArray.length; i++){
    		ShiftAssignment sa = saArray[i];
    		//iterate thru shift activities to get gap activities
    		for (Iterator itr = sa.getChildren().iterator(); itr.hasNext(); ) {
        		ShiftEventAssignment shiftEventAssn = (ShiftEventAssignment) itr.next();
        		if (shiftEventAssn.getActivityID().equals(Activity.ACTIVITY_SHIFT_OVERTIME_GAP)){
        			evaluateConditions(minDuration, maxDuration, shiftEventAssn.getDuration(), minGap, result, ssr, sa);
        		}
    		}
    		
    	}
    	
    	return result;
    }
    
    private void evaluateConditions(int minDuration, int maxDuration, int gap, int minGap, 
    		ValidationResult result, ShiftSwapRequest ssr, ShiftAssignment sa) throws Exception{
    	boolean failGapCondition = false;
    	boolean failMinCondition = false;
    	
    	boolean addGapResult = false;
    	boolean addMinResult = false;
    	
    	if (gap > 0 ){
    		if (minDuration != -1 && maxDuration != -1){
    			if (gap < minDuration || gap > maxDuration){
    				failGapCondition = true;
    			}
    			if (minGap == -1 && failGapCondition){
    				addGapResult = true;
    			}
    		}
    		if (minGap != -1){
    			if (gap < minGap){
    				failMinCondition = true;
    				if (minDuration == -1 && maxDuration == -1 && failMinCondition){
    					addMinResult = true;
    				}
    			}
    		}
    		
    		if (minDuration != -1 && maxDuration != -1 &&
    				minGap != -1 &&
    				failGapCondition && failMinCondition){
    			//show both results
    			addGapResult = true;
    			addMinResult = true;
    		}
    	}
    	if (addGapResult)
    		result = ValidationUtil.setSoftValidationResult(ssr, 
                    RmEjbBundleKey.PSS_SHIFT_GAP_RANGE, 
                    String.valueOf(gap), 
                    String.valueOf(minDuration), 
                    String.valueOf(maxDuration), 
                    ssr.getValidationCache().getEmployeeNameByID(sa.getWorkResourceIDs().iterator().next()),
                    m_className);
    	if (addMinResult)
    		result = ValidationUtil.setSoftValidationResult(ssr, 
                    RmEjbBundleKey.PSS_SHIFT_GAP_MINIMUM, 
                    String.valueOf(gap), 
                    String.valueOf(minGap), 
                    ssr.getValidationCache().getEmployeeNameByID(sa.getWorkResourceIDs().iterator().next()),
                    m_className);
    }
}
