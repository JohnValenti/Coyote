package com.bluepumpkin.ejb.rm.requests.swap.request.validation;

import java.util.ArrayList;
import java.util.Calendar;
import java.util.Date;
import java.util.Iterator;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.swap.request.model.ShiftSwapRequest;
import com.bluepumpkin.ejb.rm.requests.swap.shiftitem.model.ShiftSwapItem;
import com.bluepumpkin.ejb.rm.setup.validation.ejb.ValidationRuleManager;
import com.bluepumpkin.ejb.rm.util.LicenseUtil;
import com.bluepumpkin.ejb.rm.util.ShiftSwapRequestUtil;

public class MaxNumberApprovedPartialSwapsPerShiftValidationRule implements Validator {
	public static final String MAX_APPROVED_PARTIAL_SWAPS_PER_SHIFT = "MAX_APPROVED_PARTIAL_SWAPS_PER_SHIFT";
	private static final String CLASSNAME = MaxNumberApprovedPartialSwapsPerShiftValidationRule.class.getName();
	public static final ID RULE_ID = new ID(-194072);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		ShiftSwapRequest ssReq = (ShiftSwapRequest) validatable;
		ShiftSwapValidationCache valCache = ssReq.getCache();
		ShiftSwapItem ssItem1 = ssReq.getShiftSwapItems().get(0);
		ShiftSwapItem ssItem2 = ssReq.getShiftSwapItems().get(1);
		boolean isAdvancedRMLicense = isAdvancedRMLicense();

		// If not RM Advance License, return (i.e. don't check for partial swap limit)
		if (!isAdvancedRMLicense) {
			return result;
		}

		// Check for Partial Shifts
		if (ssItem1.getIsPartial() || ssItem2.getIsPartial()) {

			//make sure the orgIDs are not duplicate
			List<ID> orgList = new ArrayList<>();
			for (Iterator<ID> it = ssReq.getValidationCache().getOrgIDs().iterator(); it.hasNext();) {
				ID orgID = it.next();
				if (!orgList.contains(orgID)) {
					orgList.add(orgID);
				}
			}

			// for each org, find out the validator- if beeded to run for both or not
			for (int i = 0; i < orgList.size(); i++) {
				ID orgID = orgList.get(i);
				if (valCache.isValidationRequiredForOrg(orgID, "shift-swap", CLASSNAME)) {
					//get the rule params
					ValidationRuleManager valMan = RmManagerFactory.getInstance().getValidationRuleManager();
					Map<String, Integer> ruleParams = valMan.getValidationRuleParams(orgID, RULE_ID);
					Integer numberOfPartialSwapsPerShift = ruleParams.get(MAX_APPROVED_PARTIAL_SWAPS_PER_SHIFT);
					int numberOfPartialSwaps = numberOfPartialSwapsPerShift.intValue();

					//validate based on shift partial swap count
					result = verifyOrgDayShift(ssReq, numberOfPartialSwaps);
				}
			}
		}

		return result;
	}

	protected ValidationResult verifyOrgDayShift(ShiftSwapRequest ssReq, int numberOfPartialSwaps) throws Exception {
		ValidationResult result = null;
		ShiftSwapItem ssItem1 = ssReq.getShiftSwapItems().get(0);
		ShiftSwapItem ssItem2 = ssReq.getShiftSwapItems().get(1);

		// Put the Shift Swap Items in order
		if (ssItem1.getID().toInt() > ssItem2.getID().toInt()) {
			ShiftSwapItem tmp = ssItem1;
			ssItem1 = ssItem2;
			ssItem2 = tmp;
		}

		// get the resulting shifts from the swap
		ShiftAssignment[] saArray = getShiftsAfterSwap(ssReq);
		ShiftAssignment sa1 = !saArray[0].getWorkResourceIDs().iterator().next().equals(ssItem2.getEmployeeID()) ? saArray[1] : saArray[0];
		ShiftAssignment sa2 = !saArray[1].getWorkResourceIDs().iterator().next().equals(ssItem1.getEmployeeID()) ? saArray[0] : saArray[1];

		//  Check the first Shift Swap Item
		if (!ssItem1.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF) && !ssItem2.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF)) {
			result = verifyOrgDayShiftCountForItem(ssItem2, numberOfPartialSwaps, ssReq, ssItem1, sa1, sa2);
		}

		//  Check the second Shift Swap Item
		if (!ssItem1.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF) && ssItem2.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF)) {
			result = verifyOrgDayShiftCountForItem(ssItem1, numberOfPartialSwaps, ssReq, ssItem2, sa1, sa2);
		}
		return result;
	}

	protected ValidationResult verifyOrgDayShiftCountForItem(ShiftSwapItem ssItem, int numberOfPartialSwaps, ShiftSwapRequest ssReq,
			                                                 ShiftSwapItem otherSSItem, ShiftAssignment sa1, ShiftAssignment sa2) throws Exception {
		ValidationResult result = null;
		int numOfPartialSwaps = 0;

		// get employee's org
		Organization org = getOrganizationForEmployeeDuringPeriod(ssReq, otherSSItem.getEmployeeID(),
			ssItem.getStartDate(), ssItem.getEndDate());
	    Calendar startCal = Calendar.getInstance(org.getTimeZone());
	    Calendar endCal = Calendar.getInstance(org.getTimeZone());

	    startCal.setTime(ssItem.getStartDate());
	    startCal.set(Calendar.HOUR_OF_DAY, 0);
	    startCal.set(Calendar.MINUTE, 0);
	    startCal.set(Calendar.SECOND, 0);
	    startCal.set(Calendar.MILLISECOND, 0);

	    endCal.setTime(ssItem.getEndDate());
	    endCal.set(Calendar.HOUR_OF_DAY, 23);
	    endCal.set(Calendar.MINUTE, 59);
	    endCal.set(Calendar.SECOND, 59);
	    endCal.set(Calendar.MILLISECOND, 999);

	    // Check the SA's the the latest End Time
	    Calendar sa1EndCal = Calendar.getInstance(org.getTimeZone());
	    Calendar sa2EndCal = Calendar.getInstance(org.getTimeZone());

	    sa1EndCal.setTime(sa1.getStartTime());
	    sa1EndCal.add(Calendar.MINUTE, sa1.getDuration());
	    sa2EndCal.setTime(sa2.getStartTime());
	    sa2EndCal.add(Calendar.MINUTE, sa2.getDuration());

	    // Expand the date range to cover the dates outside the Org Date
	    if (!endCal.getTime().after(sa1EndCal.getTime()) || !endCal.getTime().after(sa2EndCal.getTime())) {
	    	if (sa1EndCal.getTime().after(sa2EndCal.getTime())) {
	    		endCal.setTime(sa1EndCal.getTime());
	    	} else {
	    		endCal.setTime(sa2EndCal.getTime());
	    	}
	    }

	    // Get the number of Partial Swaps
	    numOfPartialSwaps = ShiftSwapRequestUtil.getNumberOfPartialSwapsForShift(ssItem, ssItem.getEmployeeID(),
			otherSSItem.getEmployeeID(), startCal.getTime(), endCal.getTime());

		// Compare and raise a validation error for each Org.
		if (numOfPartialSwaps >= numberOfPartialSwaps) {
			result = ValidationUtil.setSoftValidationResult(ssReq, RmEjbBundleKey.PSS_MAX_PARTIAL_SWAPS_FOR_SHIFT,
				numberOfPartialSwaps, CLASSNAME);
		}

		return result;
	}

	protected ValidationRuleManager getValidationRuleManager() throws BbmCreateException {
		return RmManagerFactory.getInstance().getValidationRuleManager();
	}

	protected boolean isAdvancedRMLicense() {
		return LicenseUtil.isAdvancedRMLicense();
	}

	protected Organization getOrganizationForEmployeeDuringPeriod(ShiftSwapRequest ssReq, ID employeeID, Date startDate, 
																  Date endDate) throws Exception {
		// get calendar for employee's org
		ShiftSwapValidationCache cache = (ShiftSwapValidationCache)ssReq.getValidationCache();
		return cache.getOrganizationForEmployeeDuringPeriod(employeeID, startDate, endDate);
	}

	protected ShiftAssignment [] getShiftsAfterSwap(ShiftSwapRequest ssReq) throws Exception {
		//get the shifts corresponding to the items
		return ShiftSwapRequestUtil.getShiftsAfterSwap(ssReq, ssReq.isTentative());
	}
}
