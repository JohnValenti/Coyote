public class MaxNumberApprovedPartialSwapsPerShiftValidationRule implements Validator {
	public static final String MAX_APPROVED_PARTIAL_SWAPS_PER_SHIFT = "MAX_APPROVED_PARTIAL_SWAPS_PER_SHIFT";
	private static final String className = MaxNumberApprovedPartialSwapsPerShiftValidationRule.class.getName();
	private static final Category logger = Log.initCategory(className);
	public static final ID RULE_ID = new ID(-194072);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		ShiftSwapRequest ssReq = (ShiftSwapRequest) validatable;
		ShiftSwapValidationCache valCache = ssReq.getCache();
		ShiftSwapItem ssItem1 = (ShiftSwapItem) ssReq.getShiftSwapItems().get(0);
		ShiftSwapItem ssItem2 = (ShiftSwapItem) ssReq.getShiftSwapItems().get(1);
		boolean isAdvancedRMLicense = isAdvancedRMLicense();

		// If not RM Advance License, return (i.e. don't check for partial swap limit)
		if (!isAdvancedRMLicense) {
			return result;
		}

		if (ssItem1.getIsPartial() || ssItem2.getIsPartial()) {

			//make sure the orgIDs are not duplicate
			List orgList = new ArrayList();
			for (Iterator it = ssReq.getValidationCache().getOrgIDs().iterator(); it.hasNext();) {
				ID orgID = (ID) it.next();
				if (!orgList.contains(orgID)) {
					orgList.add(orgID);
				}
			}

			// for each org, find out the validator- if beeded to run for both or not
			for (int i = 0; i < orgList.size(); i++) {
				ID orgID = (ID) orgList.get(i);
				if (valCache.isValidationRequiredForOrg(orgID, "shift-swap", className)) {
					//get the rule params
					ValidationRuleManager valMan = RmManagerFactory.getInstance().getValidationRuleManager();
					Map<String, Integer> ruleParams = valMan.getValidationRuleParams(orgID, RULE_ID);
					Integer numberOfPartialSwapsPerShift = ruleParams.get(MAX_APPROVED_PARTIAL_SWAPS_PER_SHIFT);
					int numberOfPartialSwaps = numberOfPartialSwapsPerShift.intValue();

					//validate based on shift partial swap count
					result = verifyOrgDayShift(ssReq, numberOfPartialSwaps);
				}
			}
		}

		return result;
	}

	protected ValidationResult verifyOrgDayShift(ShiftSwapRequest ssReq, int numberOfPartialSwaps) throws Exception {
		ValidationResult result = null;
		ShiftSwapItem ssItem1 = (ShiftSwapItem) ssReq.getShiftSwapItems().get(0);
		ShiftSwapItem ssItem2 = (ShiftSwapItem) ssReq.getShiftSwapItems().get(1);

		// get the resulting shifts from the swap
		ShiftAssignment[] saArray = (ShiftAssignment[])getShiftsAfterSwap(ssReq);
		ShiftAssignment sa1 = !saArray[0].getWorkResourceIDs().iterator().next().equals(ssItem2.getEmployeeID()) ? saArray[1] : saArray[0];
		ShiftAssignment sa2 = !saArray[1].getWorkResourceIDs().iterator().next().equals(ssItem1.getEmployeeID()) ? saArray[0] : saArray[1];

		//  Check the first Shift Swap Item 
		if (!ssItem1.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF) && !ssItem2.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF)) {
			result = verifyOrgDayShiftCountForItem(ssItem2, numberOfPartialSwaps, ssReq, ssItem1, sa1);
		}

		//  Check the second Shift Swap Item 
		if (!ssItem1.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF) && ssItem2.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF)) {
			result = verifyOrgDayShiftCountForItem(ssItem1, numberOfPartialSwaps, ssReq, ssItem2, sa2);
		}
		return result;
	}

	protected ValidationResult verifyOrgDayShiftCountForItem(ShiftSwapItem ssItem, int numberOfPartialSwaps, ShiftSwapRequest ssReq,
			                                                 ShiftSwapItem otherSSItem, ShiftAssignment sa) throws Exception {
		ValidationResult result = null;
		int numOfPartialSwaps = 0;
		Date [] shiftDates = null;

		// get employee's org
		Organization org = (Organization)getOrganizationForEmployeeDuringPeriod(ssReq, otherSSItem.getEmployeeID(), ssItem.getStartDate(), ssItem.getEndDate());
	    Calendar startCal = Calendar.getInstance(org.getTimeZone());
	    Calendar endCal = Calendar.getInstance(org.getTimeZone());
	    
	    startCal.setTime(ssItem.getStartDate());
	    startCal.set(Calendar.HOUR_OF_DAY, 0);
	    startCal.set(Calendar.MINUTE, 0);
	    startCal.set(Calendar.SECOND, 0);
	    startCal.set(Calendar.MILLISECOND, 0);
	    
	    endCal.setTime(ssItem.getEndDate());
	    endCal.set(Calendar.HOUR_OF_DAY, 0);
	    endCal.set(Calendar.MINUTE, 0);
	    endCal.set(Calendar.SECOND, 0);
	    endCal.set(Calendar.MILLISECOND, 0);

	    // Get Shift Start and End Dates and get the number of Partial Swaps
	    shiftDates =  getShiftStartEndDate(sa, startCal, endCal);
		numOfPartialSwaps = ShiftSwapRequestUtil.getNumberOfPartialSwapsForShift(ssItem, sa.getShiftID(), otherSSItem.getEmployeeID(), shiftDates[0], shiftDates[1]);

		// Compare and raise a validation error for each Org.
		if (numOfPartialSwaps >= numberOfPartialSwaps) {
			result = ValidationUtil.setSoftValidationResult(ssReq, RmEjbBundleKey.PSS_MAX_PARTIAL_SWAPS_FOR_SHIFT, numberOfPartialSwaps, className);
		}

		return result;
	}
	
	protected ValidationRuleManager getValidationRuleManager() throws BbmCreateException {

	}

	protected boolean isAdvancedRMLicense() {
		return LicenseUtil.isAdvancedRMLicense();
	}

	protected Organization getOrganizationForEmployeeDuringPeriod(ShiftSwapRequest ssReq, ID employeeID, Date startDate, Date endDate) throws Exception {
		// get calendar for employee's org
		ShiftSwapValidationCache cache = (ShiftSwapValidationCache)ssReq.getValidationCache();
		Organization org = cache.getOrganizationForEmployeeDuringPeriod(employeeID, startDate, endDate);
		return org;
	}
	
	protected ShiftAssignment [] getShiftsAfterSwap(ShiftSwapRequest ssReq) throws Exception {
		//get the shifts corresponding to the items
		ScheduleAccessManager sam = RequestUtil.getScheduleAccessManager();
		return ShiftSwapRequestUtil.getShiftsAfterSwap(ssReq, sam);
	}

	protected Date [] getShiftStartEndDate(ShiftAssignment sa, Calendar startCal, Calendar endCal) {
		StringBuilder shiftWhereStr = new StringBuilder();
		ShiftDAO shiftDAO = null;
		Shift shift = null;
		Date[] shiftDates = new Date[2];
		
		try {
			shiftWhereStr.append(" sid = ").append(sa.getShiftID());
			shiftDAO = new ShiftDAO();
			Collection<Shift> shifts = shiftDAO.getObjects(shiftWhereStr.toString());
			Iterator it = shifts.iterator();
			if (it != null && it.hasNext()) {
				shift = (Shift)it.next();
			}
			
			if (shift != null ) {
				Collection<ShiftStart> shiftStarts = shift.getShiftStarts();
				int startTime = getShiftStartTime(shiftStarts)/60;
				int shiftDuration = getShiftDuration(shiftStarts, shift.getDuration())/60;
				
				startCal.add(Calendar.HOUR, startTime);
				endCal.add(Calendar.HOUR, startTime + shiftDuration);
				
				shiftDates[0] = startCal.getTime();
				shiftDates[1] = endCal.getTime();
			} else {
				shiftDates[0] = startCal.getTime();
				shiftDates[1] = endCal.getTime();
			}
		} catch (Exception e) {
			// Got an exception so return the dates passed in
			shiftDates[0] = startCal.getTime();
			shiftDates[1] = endCal.getTime();
			logger.warn("Exception when retrieveing Shift" + e.getLocalizedMessage());

		}
		
		return shiftDates;
	}
	
	protected int getShiftStartTime(Collection<ShiftStart> shiftStarts) {
		int highMinutes = 99999999;
		int startOffSet = highMinutes;
		
		for(ShiftStart shiftStart : shiftStarts) {
			int possStartTime = shiftStart.getPossibleStartTime();
			if (startOffSet > possStartTime) {
				startOffSet = possStartTime;
			}
		}
		
		if (startOffSet == highMinutes) {
			startOffSet = 0;
		}
		
		return startOffSet;
	}
	
	protected int getShiftDuration(Collection<ShiftStart> shiftStarts, int duration) {
		int finalDuration = 0;
		int highDuraton = 0;
		int lowDuration = 99999999;
		
		for(ShiftStart shiftStart : shiftStarts) {
			int shiftDuration = shiftStart.getPossibleStartTime();
			if (shiftDuration > highDuraton ) {
				highDuraton = shiftDuration;
			}
			
			if (shiftDuration < lowDuration)  {
				lowDuration = shiftDuration;
			}
		}
		finalDuration = (highDuraton - lowDuration) + duration;
		return finalDuration;
	}
}
