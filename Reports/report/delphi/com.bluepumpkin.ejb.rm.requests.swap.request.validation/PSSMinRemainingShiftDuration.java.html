/**
 *
 * <b>Soft Validation: applies only for partial shift swap requests</b>
 * <p>
 *    Validates that the remaining shift after the swap is greater than the specified duration
 * <p>
 *    Checks for both the shifts for a two way swap
 */
public class PSSMinRemainingShiftDuration  implements Validator
{
    private static final String CLASS_NAME = PSSMinRemainingShiftDuration.class.getName();
    private static final Category LOGGER = Log.initCategory(CLASS_NAME);

    public static final String MIN_SHIFT_DURATION_VALUE = "MIN_SHIFT_DURATION_VALUE";
    public static final String MIN_SHIFT_DURATION_UNIT = "MIN_SHIFT_DURATION_UNIT";
    public static final int DURATION_UNIT_PERCENT = 0;
    public static final int DURATION_UNIT_MINUTES = 1;
    public static final ID RULE_ID = new ID(-194033);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception
    {
        String methodName = "validate";
        if (LOGGER.isDebugEnabled()) {
			LOGGER.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		}

        ShiftSwapRequest ssr = (ShiftSwapRequest) validatable;
        ShiftSwapValidationCache valCache = getShiftSwapValidationCache(ssr);
        ShiftSwapItem ssItem1 = ssr.getShiftSwapItems().get(0);
        ShiftSwapItem ssItem2 = ssr.getShiftSwapItems().get(1);
        ValidationResult result = null;
        if (ssItem1.getIsPartial() || ssItem2.getIsPartial()) {
        	// need to get the original shift assignments to compare/validate for percentage
        	ShiftAssignment[] saBeforeSwap = getOriginalShiftAssignments(valCache);
        	// get the resulting shifts after the swap
            ShiftAssignment[] saArray = null;
            // get the sa array after the swap
        	saArray = performPartialSwap(ssr);
            //get the rule params
        	for (int j=0; j<ssr.getShiftSwapItems().size(); j++){
	        	result = verifyMinRemainingShiftDuration(saBeforeSwap,saArray, validatable, ssr.getShiftSwapItems().get(j), valCache);
        	}

        }
        if (LOGGER.isDebugEnabled()) {
			LOGGER.debug(RmUtil.dumpExitMethod(methodName));
		}
        return result;
    }

	private ValidationResult verifyMinRemainingShiftDuration(ShiftAssignment[] saBeforeSwap, ShiftAssignment[] saArrayAfterSwap,
			Validatable validatable, ShiftSwapItem ssItemGiveOut, ShiftSwapValidationCache valCache) throws Exception {
		if (ssItemGiveOut.isSwapTypeShift()) {
			ID empId = ssItemGiveOut.getEmployeeID();
			ID orgId = valCache.getOrgIDForEmployeeDuringPeriod(empId, ssItemGiveOut.getStartDate(), ssItemGiveOut.getEndDate());
			boolean isValidationRequiredForOrg = valCache.isValidationRequiredForOrg(orgId, Request.REQUESTTYPE_SHIFTSWAP, CLASS_NAME);
			if (!isValidationRequiredForOrg) {
				return null;
			}
			Pair<Integer, Integer> paramValues = getUnitAndValueInOrg(orgId);
			int unit = paramValues.getFirst().intValue();
			int duration = paramValues.getSecond().intValue();
			int durationAfter = 0;

			/*Just handle here to avoid duplicated ShiftAssignments resulted from ShiftSwapRequestUtil.getShiftsAfterSwap(ssReq, false) */
			HashSet<ShiftAssignment> shiftAssignmentsAfter = new HashSet<ShiftAssignment>();
			for (int i = 0; i < saArrayAfterSwap.length; i++) {
				ShiftAssignment saAfter = saArrayAfterSwap[i];
				shiftAssignmentsAfter.add(saAfter);
			}
			for (ShiftAssignment saAfter : shiftAssignmentsAfter) {
				if (saAfter != null && saAfter.getWorkResourceIDs().iterator().next().equals(empId)) {
					durationAfter = saAfter.getDuration();
					break;
				}
			}
			if (unit == DURATION_UNIT_PERCENT) {
				int durationBefore = 0;
				for (int i = 0; i < saBeforeSwap.length; i++) {
					ShiftAssignment saBefore = saBeforeSwap[i];
					if (saBefore != null && saBefore.getWorkResourceIDs().iterator().next().equals(empId)) {
						durationBefore = saBefore.getDuration();
						break;
					}
				}
				//get the actual duration from the shift
				duration = (durationBefore * duration) / 100;
			}
			if (durationAfter < duration) {
				return ValidationUtil.setSoftValidationResult(validatable, RmEjbBundleKey.PSS_REM_SHIFT_DURATION_LESS_THAN_MIN, 
						ssItemGiveOut.getStartDate(), ssItemGiveOut.getEndDate(), 
						String.valueOf(duration), valCache.getEmployeeNameByID(ssItemGiveOut.getEmployeeID()), CLASS_NAME);
			}
		}
		return null;

	}
    protected ShiftSwapValidationCache getShiftSwapValidationCache(ShiftSwapRequest ssRequest) {
		return ssRequest.getCache();
	}

    protected ShiftAssignment[] getOriginalShiftAssignments( ShiftSwapValidationCache valCache) throws Exception{
    	ScheduleAccessManager sam = RequestUtil.getScheduleAccessManager();
    	return valCache.getShiftAssignsForReqSameOrgDay(true, sam);
    }
    protected ShiftAssignment[] performPartialSwap(ShiftSwapRequest ssReq) throws Exception{
    	return ShiftSwapRequestUtil.getShiftsAfterSwap(ssReq, ssReq.isTentative());

    }
    protected int getDuration(Map<String, Integer> ruleParams ){
    	return ruleParams.get(MIN_SHIFT_DURATION_VALUE).intValue();
    }
    protected int getUnit(Map<String, Integer> ruleParams ){
    	return ruleParams.get(MIN_SHIFT_DURATION_UNIT).intValue();
    }
    protected List<TimeRange> getGapsOfShiftAssignment(ShiftSwapValidationCache valCache, ShiftAssignment sa) throws Exception{
    	return valCache.getAllGaps(sa);
    }
    protected Pair<Integer, Integer> getUnitAndValueInOrg(ID orgID) throws BbmCreateException, BbmFinderException, RemoteException {
		Pair<Integer, Integer> result = new Pair<Integer, Integer>(Integer.valueOf(-1), Integer.valueOf(-1));
		ValidationRuleManager valMan = RmManagerFactory.getInstance().getValidationRuleManager();
		Map<String, Integer> ruleParams = valMan.getValidationRuleParams(orgID, RULE_ID);
		LOGGER.debug("orgID =" + orgID + "have ruleParams="+ ruleParams);
		if(ruleParams!=null&& !ruleParams.isEmpty()) {
			Integer unit = ruleParams.get(MIN_SHIFT_DURATION_UNIT);
			Integer value = ruleParams.get(MIN_SHIFT_DURATION_VALUE);
			result.setFirst(unit);
			result.setSecond(value);
			}
		return result;
	}
}
