/**
 * @author rrajendran
 *
 * Ensures that any existing shifts will not overlap with the swapped shift.
 *
 * <p> For two way swaps, both shiftswap items are checked.  For one way swaps, only the shiftswap
 * item of type 'shiftswap' is checked and the shiftswap item of type 'timeoff' is not checked.
 *
 * <p> For tentatively approved check only the published schedule.
 */
public class NoShiftsOnReceivedShiftOrgDayHV implements Validator {
	public static final String CLASSNAME = NoShiftsOnReceivedShiftOrgDayHV.class.getName();

	private ShiftSwapValidationCache m_vc = null;
	private Localizer m_localizer= null;
	private ShiftSwapRequest m_ssReq = null;

	//used for unit testing.
	void setShiftSwapValidationCache(ShiftSwapValidationCache cache) {
		m_vc = cache;
	}

	public Localizer getLocalizer(){
		if (m_localizer==null){
			m_localizer = DefaultLocalizationManager.getDefaultInstance().getLocalizer();
		}
		return m_localizer;
	}

	public void setLocalizer(Localizer l){
		m_localizer=l;
	}

	public void setRequest(ShiftSwapRequest r){
		m_ssReq=r;
	}

	protected boolean isAdvancedRMLicense() {
		return LicenseUtil.isAdvancedRMLicense();
	}


	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {

		m_ssReq = (ShiftSwapRequest) validatable;

		if (RequestAuditTrail.STATUS_WITHDRAW_REQUEST.equals(m_ssReq.getRequestStatus())) {
			// do not checks overlap for withdrawal requests
			return null;
		} else {

			ScheduleAccessManager sam = RequestUtil.getScheduleAccessManager();

			m_vc = (ShiftSwapValidationCache) m_ssReq.getValidationCache();

			List<ShiftSwapItem> ssItems = m_ssReq.getShiftSwapItems();
			ShiftSwapItem ssItem1 = ssItems.get(0);
			ShiftSwapItem ssItem2 = ssItems.get(1);
			// check the published schedule for any shift during the organization day of the received shift.
			ValidationResult valResult = verifyNoShiftAssnOnReceivedShiftOrgDay(m_ssReq, ssItem1, ssItem2, true, sam);
			if (valResult != null || m_ssReq.isTentative()) {
				//If the request is tentatively approved we do not need to check the un-published schedules since the shifts have 
				//already been swapped for the un-published schedules  
				return valResult;
			}

			//check the unpublished schedules
			return verifyNoShiftAssnOnReceivedShiftOrgDay(m_ssReq, ssItem1, ssItem2, false, sam);
		}
	}

	/**
	 * check if there is no existing shift on the received shift's org day for both 'ssItem1' and
	 * 'ssItem2'.
	 *
	 * @param ssReq
	 * @param ssItem1
	 * @param ssItem2
	 * @param published
	 * @param sam
	 * @return
	 * @throws Exception
	 */
	public ValidationResult verifyNoShiftAssnOnReceivedShiftOrgDay(ShiftSwapRequest ssReq,
		ShiftSwapItem ssItem1, ShiftSwapItem ssItem2, boolean published, ScheduleAccessManager sam) throws Exception {
	      //  AS == After swap;  BS == Before Swap.
			//
			//  BS: ssItem1, empID1                    AS: ssItem2, empID1
			//  ===================                   - - - - - - - - - - -
			//         |                                        ^
			//         |                                        |
			//         |                                        |
			//         V                                        |
			//  - - - - - - - - - -                    ====================
			//  AS: ssItem1, empID2                       BS: ssItem2, empID2
				
			// shiftAssignmentsOnReceivedShiftOrgDay.
	        //
	        // [0] == shiftAssn same org day for ssItem1 with empID2
	        // [1] == shiftAssn same org day for ssItem2 with empID1
		ShiftAssignment[] SAsForExportedShift =
			m_vc.getShiftAssignsForReqSameOrgDay(published, sam);

		ShiftAssignment SAForExportedShiftForEmp2 =null;
		ShiftAssignment SAForExportedShiftForEmp1 = null;

		// check ssItem1 received by emp2.
		if (ssItem1.isSwapTypeShift()) {
			SAForExportedShiftForEmp2 = SAsForExportedShift[1];
		}

		// check ssItem2 received by emp1.
		if (ssItem2.isSwapTypeShift()) {
			SAForExportedShiftForEmp1 = SAsForExportedShift[0];
		}
		//2 way full swap
		if (isTwoWayFullSwap(ssReq, ssItem1, ssItem2, SAForExportedShiftForEmp2, SAForExportedShiftForEmp1)) {
			//Get all shift assignments of receiver during the time range  giver's item
			Collection<ShiftAssignment> getShiftAssignmentFullSwap_Item1 = getShiftAssignmentsInSwapTimeRange(
					SAForExportedShiftForEmp2.getWorkResourceID(), ssItem1.getStartDate(), ssItem1.getEndDate(), sam, published);
			//Get all shift assignment of giver during time range receiver 's item to check the opposite site
			Collection<ShiftAssignment> getShiftAssignmentFullSwap_Item2 = getShiftAssignmentsInSwapTimeRange(
					SAForExportedShiftForEmp1.getWorkResourceID(), ssItem2.getStartDate(), ssItem2.getEndDate(), sam, published);

			//Case1: If found no shift assignments on the time range of giver's item  => it is good to swap
			//                          |---------------------|
			// |------------------| 
			//Case 2: If found 1 shift assignment and the shift assignment is the receiver and vice versa=>it is good to full swap
			// |-----------------|    
			//             |---------------------|

			//Case 3 : Explain in getOverlappedShift

			ShiftAssignment overlappedShiftForGiverItem1 = getOverlapedShift(getShiftAssignmentFullSwap_Item1, SAForExportedShiftForEmp2);
			if (overlappedShiftForGiverItem1 != null) {
				//Find the overlap range between the range of item1 and the overlap shift
				Timeline timeLine = (new Timeline(ssItem1.getStartDate(), ssItem1.getEndDate())).intersect(new Timeline(
						overlappedShiftForGiverItem1.getStartTime(), overlappedShiftForGiverItem1.getEndTime()));
				TimeRange newRange = timeLine.isEmpty() ? new TimeRange(ssItem1.getStartDate(), ssItem1.getEndDate()) : timeLine
						.getTimeRanges().get(0);
				return ValidationUtil.setHardValidationResult(m_ssReq,
						RmEjbBundleKey.SS_SHIFT_OVERLAPS_ON_RECEIVED_SHIFT_ORG_DAY_FULL_2WAY,
						m_vc.getEmployeeNameByID(SAForExportedShiftForEmp2.getWorkResourceID()), newRange.getStartDate(),
						newRange.getEndDate(), CLASSNAME);
			}
			ShiftAssignment overlappedShiftForGiverItem2 = getOverlapedShift(getShiftAssignmentFullSwap_Item2, SAForExportedShiftForEmp1);
			if (overlappedShiftForGiverItem2 != null) {
				//Find the overlap range between the range of item1 and the overlap shift
				Timeline timeLine = (new Timeline(ssItem2.getStartDate(), ssItem2.getEndDate())).intersect(new Timeline(
						overlappedShiftForGiverItem2.getStartTime(), overlappedShiftForGiverItem2.getEndTime()));
				TimeRange newRange = timeLine.isEmpty() ? new TimeRange(ssItem2.getStartDate(), ssItem2.getEndDate()) : timeLine
						.getTimeRanges().get(0);
				return ValidationUtil.setHardValidationResult(m_ssReq,
						RmEjbBundleKey.SS_SHIFT_OVERLAPS_ON_RECEIVED_SHIFT_ORG_DAY_FULL_2WAY,
						m_vc.getEmployeeNameByID(SAForExportedShiftForEmp1.getWorkResourceID()), newRange.getStartDate(),
						newRange.getEndDate(), CLASSNAME);
			}

		} else {//Other cases

			ShiftAssignment[] SAsOnReceivedShiftOrgDay = m_vc.getSwappedSAForReqWithStartDateOnSameOrgDay(published, sam);
			if (m_vc.isFullSwapOnDifferentDay()) {
				//full swap not on same day. check that shifts don't intersect
				//Verify swap item will not intersect with existing shift
				if (SAForExportedShiftForEmp2 != null) {
					ValidationResult res = verifyCollision(ssReq, ssItem2, ssItem1.getEmployeeID(), SAForExportedShiftForEmp2,
							SAsOnReceivedShiftOrgDay[1], published);
					if (res != null) {
						return res;
					}
				}

				//Verify second swap item will not intersect with existing shift
				if (SAForExportedShiftForEmp1 != null) {
					ValidationResult res = verifyCollision(ssReq, ssItem1, ssItem2.getEmployeeID(), SAForExportedShiftForEmp1,
							SAsOnReceivedShiftOrgDay[0], published);
					if (res != null) {
						return res;
					}
				}
			}
		}

		return null;

	}

	private boolean isTwoWayFullSwap(ShiftSwapRequest ssReq, ShiftSwapItem ssItem1, ShiftSwapItem ssItem2,
			ShiftAssignment SAForExportedShiftForEmp2, ShiftAssignment SAForExportedShiftForEmp1) {

		if (ssReq.isOneWay() || ssItem1.getIsPartial() || ssItem2.getIsPartial()) {
			return false;
		}
		return SAForExportedShiftForEmp2 != null && SAForExportedShiftForEmp1 != null;
	}
	
	private ValidationResult verifyCollision(ShiftSwapRequest ssReq, ShiftSwapItem ssItem, ID swapedEmployeeID, ShiftAssignment giver,
			ShiftAssignment receiver, boolean published) throws Exception {
		//Partial - we need to create a shift with the partial portion. Assumption is that there are no gaps in the portion being given.
		if (ssItem.getIsPartial()) {
			giver = ShiftSwapRequestApprovalManager.createSAFromParams(ssItem.getStartDate(), giver.getActivityID(), null,
					(int) ssItem.getDuration(), swapedEmployeeID, giver.getShiftID());
		}
		boolean isOverlap = RequestUtil.isOverlap(giver, receiver);
		if (isOverlap) {
			return ValidationUtil.setHardValidationResult(ssReq, RmEjbBundleKey.SS_SHIFT_OVERLAPS_ON_RECEIVED_SHIFT_ORG_DAY, m_vc
					.getEmployeeNameByID(swapedEmployeeID), receiver.getStartTime(), receiver.getEndTime(),
					published ? new ResourceBundleKeyWrapper(RmEjbBundleKey.GEN_PUBLISHED) : new ResourceBundleKeyWrapper(
							RmEjbBundleKey.GEN_UNPUBLISHED), CLASSNAME);
		}
		return null;
	}

	//Get all shift assignments of receiver during the time range of shift item of giver
	@SuppressWarnings("unchecked")
	protected Collection<ShiftAssignment> getShiftAssignmentsInSwapTimeRange(ID receiverID, Date startDate, Date endDate,
			ScheduleAccessManager sam, boolean published) throws Exception {
		//Find all shift assignments of the receiver in the date range of the giver

		if (published) {
			return sam.getPublishedEventsForWorkResourceByType(Event.EVENT_TYPE_SHIFT_ASSIGNMENT,
					receiverID, startDate, endDate);
		}
		return sam.getAllShiftAssignmentsForEmployeesBetweenDates(Collections.singletonList(receiverID), startDate, endDate);

	}
			
	/* Find the overlap shift when comparing 1 giver 's shift assignment with list of receiver 's shifts assignment
	 * */
	private ShiftAssignment getOverlapedShift(Collection<ShiftAssignment> listShiftAssignments, ShiftAssignment exportedShift) {
		ShiftAssignment overlappedShift = null;
		if (listShiftAssignments != null && listShiftAssignments.size() == 1 && !listShiftAssignments.contains(exportedShift)) {
			//Time Range of Giver shift item   |_____________________________|
			//                  ES   |--------------|                        SwapedShift   |---------------|
			overlappedShift = listShiftAssignments.iterator().next();
		} else if (listShiftAssignments != null && listShiftAssignments.size() > 1) {
			//Time Range of Giver shift item   |_____________________________|
			//   SwapedShift         |--------------|         ES   |---------------|
			Collection<ShiftAssignment> tempList = new ArrayList<ShiftAssignment>();
			for (ShiftAssignment shiftAssignment : listShiftAssignments) {
				if (shiftAssignment.equals(exportedShift)) {
					tempList.add(shiftAssignment);
					break;
				}
			}
			listShiftAssignments.removeAll(tempList);
			overlappedShift = listShiftAssignments.iterator().next();
		}
		return overlappedShift;

	}
