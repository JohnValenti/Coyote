	/** create a child DAO object given the type */
	@Override
	@SuppressWarnings("rawtypes")
	protected DAOBase createChildDAO(int childType) {
		if (childType == TORequestFieldInfo.TO_CHOICE_CHILD_TYPE) {
			return new TOChoiceDAO(m_dmo);
		}
		if (childType == TORequestFieldInfo.TO_WAITLIST_CHILD_TYPE) {
			return new TOWaitlistDAO(m_dmo);
		}
		if (childType == TORequestFieldInfo.TO_WITHDRAW_CHILD_TYPE) {
			return new TOWithdrawDAO(m_dmo);
		}
		if (childType == TORequestFieldInfo.TO_FLEXMAKEUP_CHILD_TYPE) {
			return new FlexRequestMakeupDAO(m_dmo);
		}
		throw new IllegalArgumentException("childType = " + childType);
	}

	@Override
	protected boolean needToLoadChildObjects(int childType) {
		boolean result = false;
		if (childType == TORequestFieldInfo.TO_CHOICE_CHILD_TYPE) {
			result = true;
		} else if (childType == TORequestFieldInfo.TO_WAITLIST_CHILD_TYPE
				&& ((getDetailLevel() & RequestDetailLevel.DL_TIMEOFF_WAITLIST) != 0)) {
			result = true;
		} else if (childType == TORequestFieldInfo.TO_WITHDRAW_CHILD_TYPE
				&& ((getDetailLevel() & RequestDetailLevel.DL_TIMEOFF_WITHDRAW) != 0)) {
			result = true;
		} else if (childType == TORequestFieldInfo.TO_FLEXMAKEUP_CHILD_TYPE) {
			result = true;
		}
		return result;
	}

        Collection requests = Collections.emptyList();

        saveAndSetDetailLevel(detailLevel);

              if ((detailLevel & TORequest.DL_TIMEOFF_FLEXMAKEUP) > 0) {
                      requests = getObjects("A.EMPLOYEEID=" + employeeID + " AND B.FLEXTYPE=" + TORequest.FLEXTYPE_FLEXWITHMAKEUP);

            	   requests = getObjects("A.EMPLOYEEID=" + employeeID + " AND B.FLEXTYPE=" + TORequest.FLEXTYPE_NONE);
               }
        } finally  {
            resetDetailLevel();
        }

        return requests;
    }

	/**
	 * Find a collection of time off requests for the given employees that are <li>part of the pending hour calculation <li>
	 * whose first TOChoice begins in the specified range <li>not of debittype = dont-debit
	 *
	 * @param empIDs the employee ids
	 * @param start beginnging of range (inclusive)
	 * @param end ending of range (exclusive)
	 * @param onlyUseAllotmentAct flag to filter non allotment activities
	 * @return a collection of TORequest objects
	 */
	public Collection getPendingHourStatusesAndDebitReqsForEmpsInDateRange(Collection empIDs, Date start, Date end,
			long detailLevel, boolean onlyUseAllotmentAct) throws Exception {
		saveAndSetDetailLevel(detailLevel);

		try {
			// the statuses we are concerned about.
			String[] statusArr = TORequest.PENDING_HOUR_STATUSES;

			// Note: query optimized during perf tuning.
			//
			// Get the collection of IDs we're after
			StringBuffer query = new StringBuffer(256);
			query.append("SELECT A.ID FROM REQUEST A, TIMEOFFREQUEST B, TIMEOFFREQUESTCHOICE C ");
			if (onlyUseAllotmentAct)
				query.append(", ACTIVITY D ");

			query.append("WHERE A.ID=B.ID ").append(" AND B.ID=C.TIMEOFFREQUESTID ").append(" AND C.RANK=1 ")
					.append(" AND NOT B.DEBITTYPE = '").append(TORequest.DEBITTYPE_DONT_DEBIT)
					.append("' AND A.REQUESTSTATUS IN ").append(DAOUtil.createInClause(statusArr))
					.append(" AND A.EMPLOYEEID IN ").append(m_dmo.createInClause(empIDs))
					.append("  AND ((C.STARTTIME BETWEEN '").append(JdmoUtil.formatDBString(start)).append("' AND '")
					.append(JdmoUtil.formatDBString(end)).append("') OR (C.ENDTIME BETWEEN '")
					.append(JdmoUtil.formatDBString(start)).append("' AND '").append(JdmoUtil.formatDBString(end))
					.append("') OR ").append("(C.STARTTIME < '").append(JdmoUtil.formatDBString(start)).append("' AND ")
					.append("C.ENDTIME > '").append(JdmoUtil.formatDBString(end)).append("')) ");

			if (onlyUseAllotmentAct)
				query.append(" and B.ACTIVITYID = D.ID AND D.ISTIMEOFFWITHALLOTMENT=1 ");

			List toReqIDs = DAOUtil.getIDsUsingSQLQuery(getDMO(), query.toString());

			return getObjectsByIDs(toReqIDs);
			// } c_atch (Exception e) {
			// t_hrow RequestUtil.createBbmFinderExceptionWrapper(e, m_cat);
			// }
		} finally {
			resetDetailLevel();
		}
	}

	public Collection getPendingTORequestsForDateAndEmpIDs(Date date, Collection empIDs, long detailLevel)
			throws BbmFinderException, JdmoException {
		saveAndSetDetailLevel(detailLevel);
		try {
			// get collection of TORequestIDs from TOChoices that fall on the given day.
			TOChoiceDAO toChoiceDAO = (TOChoiceDAO) createChildDAO(TORequestFieldInfo.TO_CHOICE_CHILD_TYPE);
			Collection TOReqIDs = toChoiceDAO.getTORequestIDsByDate(date);

			// Use collection of TORequestIDs to obtain a list of TORequests and filter out pending requests.
			// WHERE id IN (....) AND STATUS = 'PENDING'
			// TODO: do we include 'tentatively approved' requests?
			StringBuffer whereClause = new StringBuffer();
			whereClause.append(" A.ID IN ").append(m_dmo.createInClause(TOReqIDs)).append(" AND STATUS = '")
					.append(RequestAuditTrail.STATUS_PENDING).append("'").append(" AND A.EMPLOYEEID IN ")
					.append(m_dmo.createInClause(empIDs));

			return getObjects(whereClause.toString());
		} finally {
			resetDetailLevel();
		}
	}

	/**
	 * Delete all TimeOff requests for employees in the given organization created in the given range.
	 */
	public void deleteRequestsByStartDateRange(ID organizationID, boolean isForBranch, TimeRange range, String requestType, ID subType)
			throws Exception {
		StringBuffer selectClause = new StringBuffer(128);
		StringBuffer fromClause = new StringBuffer(128);
		StringBuffer whereClause = new StringBuffer(128);

		getSQLFilterForRequestDeletionForStartDateRange(organizationID, isForBranch, requestType, subType, range.getStartDate(),
				range.getEndDate(), selectClause, fromClause, whereClause);
		String sql = RequestUtil.concatStrBuffers(selectClause, fromClause, whereClause, " ");

		if (!StringUtil.isEmpty(sql)) {
			deleteObjects(sql);
		}
	}

	/**
	 * Provides the SQL query used to filter requests to be deleted by request start date range
	 * <p>
	 * UI options for request purging:<br>
	 *
	 * <pre>
	 * Requests Type  "All | Shift Bidding | Shift Swap | Time Off".<br>
	 *    For requests created from  "Current Organization | Organization Branch"<br>
	 *    For requests starting from -  "fromDate" to "toDate"  (startDate != null; endDate != null)<br>
	 * </pre>
	 *
	 * Note: This was an abstract method earlier. concrete implementations in the subclasses were commented and a common
	 * implementation was implemented here.
	 *
	 * @param organizationID
	 * @param isForBranch
	 * @param requestType
	 * @param subType
	 * @param startDate
	 * @param endDate
	 * @param selectClause
	 * @param fromClause
	 * @param whereClause
	 * @return
	 */
	protected void getSQLFilterForRequestDeletionForStartDateRange(ID orgID, boolean isForBranch, String requestType, ID subType,
			Date startDate, Date endDate, StringBuffer selectClause, StringBuffer fromClause, StringBuffer whereClause)
			throws Exception {

		Collection<ID> orgEmpIDs = getEmployeeIDsInOrganization(orgID, isForBranch, startDate, endDate);

		selectClause.append("SELECT REQ.ID ");
		fromClause.append("FROM REQUEST REQ, TIMEOFFREQUESTCHOICE TOCHOICE");

		// append where clause conditions
		whereClause.append(" WHERE REQ.EMPLOYEEID IN ").append(createInClause(orgEmpIDs));
		whereClause.append(" AND REQ.REQUESTTYPE = '").append(getRequestType()).append('\'');
		whereClause.append(" AND TOCHOICE.TIMEOFFREQUESTID = REQ.ID");

		appendRequestTypeAndSubType(requestType, subType, fromClause, whereClause);

		// and tochoice.startdate between startDate and endDate
		whereClause.append(" AND TOCHOICE.STARTTIME BETWEEN '").append(JdmoUtil.formatDBString(startDate)).append("' AND '")
				.append(JdmoUtil.formatDBString(endDate)).append('\'');
	}

	}

	private void appendRequestTypeAndSubType(String requestType, ID subType, StringBuffer fromClause, StringBuffer whereClause) {
		fromClause.append(", TIMEOFFREQUEST TOREQ ");
		whereClause.append(" AND REQ.ID = TOREQ.ID AND TOREQ.FLEXTYPE = " + (requestType.equals(Request.REQUESTTYPE_FLEXTIME) ? TORequest.FLEXTYPE_FLEXWITHMAKEUP : TORequest.FLEXTYPE_NONE) + ' ');
		if (subType != null) {
			whereClause.append(" AND TOREQ.ACTIVITYID = " + subType + ' ');
		}
	}

		finally {

		}

	}

	public Collection<ID> getApprovedFlexRequestIDs(ID employeeID, TimeRange timeOffRange, boolean includeLimit) throws BbmFinderException {
		Collection<ID> requestIDs = new ArrayList<ID>();

		try {
			String startDate = JdmoUtil.formatDBString(timeOffRange.getStartDate());
			String endDate = JdmoUtil.formatDBString(timeOffRange.getEndDate());


			StringBuilder query = new StringBuilder();
			query.append("SELECT DISTINCT(REQ.ID) REQUESTID FROM REQUEST REQ, TIMEOFFREQUEST TOREQ, TIMEOFFREQUESTCHOICE TOCHOICE ");
			query.append(" WHERE REQ.ID = TOREQ.ID AND TOREQ.ID = TOCHOICE.TIMEOFFREQUESTID");
			query.append(" AND REQ.REQUESTSTATUS = 'approved'");
			query.append(" AND REQ.REQUESTTYPE = 'time-off'");
			query.append(" AND TOREQ.FLEXTYPE = " + TORequest.FLEXTYPE_FLEXWITHMAKEUP);
			query.append(" AND REQ.EMPLOYEEID IN ").append(createInClause(Arrays.asList(employeeID)));
			// includeLimit- represent if we want to pull the records with the timeOffRange included or excluded
			if (includeLimit) {
				query.append(" AND (TOCHOICE.ENDTIME >= '" + startDate + "') AND (TOCHOICE.STARTTIME <= '" + endDate + "')");
			} else {
				query.append(" AND (TOCHOICE.ENDTIME > '" + startDate + "') AND (TOCHOICE.STARTTIME < '" + endDate + "')");
			}
			JdmoRowset rs = m_dmo.createRowset(query.toString());
			while (rs.next()) {
				ID requestID = rs.getID("REQUESTID");
				requestIDs.add(requestID);
			}

		} catch(JdmoException e) {
			throw new BbmFinderException(e);
		}
		return requestIDs;
	}




	/*
	 * An advanced vto request is created with a single TOChoice that spans the whole day for which the employee is requesting time off
	 * When such a request is approved we calculate the TOChoices that would pass the configured validation rules.  When changing 
	 * the status to approved for such a request we need to remove the existing TOChoice that spans the whole day
	 * and then put in the calculated TOChoices.  
	 */
	private void fixupTORequestForAdvancedVto(TORequest toRequest) throws MultiUserException, BbmUpdateException {

		if (!toRequest.isAdvancedVTORequest() || !toRequest.isTimeoffChoicesForAdvancedVtoInitialized()) {
			return;
		}

		List<TOChoice> approvedAdvanceVtoChoice = toRequest.getApprovedAdvancedVtoChoices();
		if (approvedAdvanceVtoChoice.isEmpty()) {
			return;
		}

		// approvedAdvanceVtoChoice is the calculated TOChoice which does not exists in the database at this point
		int rank = 1;
		for (TOChoice choice : approvedAdvanceVtoChoice) {
			choice.setID(null);
			choice.setRank(rank);
			choice.setUserRank(rank);
			rank++;
		}

		toRequest.setRequestChoiceList(approvedAdvanceVtoChoice);
		//set the approved choice to null
		//create the calculated TOChoice and update the TimeOffRequest and set the approved choice to null
		super.updateObject(toRequest);
		//need to call this otherwise the next call to updateObject will try to create time off choices again 
		toRequest.moveChildrenFromCreatedToPersisted();
		//remove all the calculated time off choices
		toRequest.resetCalculatedTimeoffChoicesForAdvancedVto();

	}

	/**
	 * For the map of quarterly timeranges, get the number of approved timeoff choices for the employee
	 * @param employeeID
	 * @param orgTimeZone 
	 * @param ArrayList<TimeRange> list of quarterly time ranges
	 * @return Map<TimeRange, Integer> quarterly time ranges x approved timeoff choices
	 * @throws JdmoException
	 */
	public Map<TimeRange, Integer> getApprovedChoiceCountForQuarter(final ID employeeID, final Date start, 
			final Date end, TimeZone orgTimeZone, ID activityID)
			throws JdmoException {

		Map<TimeRange, Integer> returnMap = new HashMap<TimeRange, Integer>();

		String sql = "SELECT * FROM RM_GetApprovedTimeOffbyQuarter(?, ?, ?, ?, ?)";

		JdmoRowset res = JdmoPreparedStatement.createRowset(m_dmo, sql, start, end, employeeID, orgTimeZone.getID(), activityID);

		while (res.next()) {
			Date s = res.getTimestamp("StartTime");
			Date e = res.getTimestamp("EndTime");
			int value = res.getInt("Nbr");
			TimeRange r = new TimeRange(s, e);

			returnMap.put(r, value);
		}


		return returnMap;
	}
