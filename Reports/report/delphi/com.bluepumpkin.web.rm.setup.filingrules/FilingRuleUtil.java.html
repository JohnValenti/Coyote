package com.bluepumpkin.web.rm.setup.filingrules;

public class FilingRuleUtil extends RmSetupUtil {
	private static final String WHEN_TO_FILE_LABEL = "WHEN_TO_FILE_LABEL";

	/** Creates a new instance of FilingRuleUtil */
	private FilingRuleUtil() {
	}

	/**
	 * Initialize Filing Rule List Header
	 */
	public static void initListHeader(TableHeaderPC header, ResourceBundle rmBundle, ResourceBundle bbmBundle, boolean isFullList, Localizer localizer) {
		// Setup Headers
		if (!isFullList) {
			header.setStyleClass(CSSUtil.CLASS_INNER_HEADER_BORDER);
		}

		if (isFullList) {
			header.addColumnHeaderData(APPLY_LABEL, localizer.i18n(rmBundle, RmWebBundleKey.ORG_RM_FILING_RULE_APPLY_LABEL));
		}
		header.addColumnHeaderData(REQ_TYPE_LABEL, localizer.i18n(rmBundle, RmWebBundleKey.ORG_RM_FILING_RULE_REQ_TYPE_LABEL));
		header.addColumnHeaderData(WHICH_REQUEST_LABEL, localizer.i18n(rmBundle, RmWebBundleKey.ORG_RM_FILING_RULE_WHICH_LABEL));
		header.addColumnHeaderData(WHEN_TO_FILE_LABEL, localizer.i18n(rmBundle, RmWebBundleKey.ORG_RM_FILING_RULE_WHEN_LABEL));
		if (isFullList) {
			header.addColumnHeaderData(ORGANIZATION_LABEL, localizer.i18n(bbmBundle, BbmWebBundleKey.LIST_HEADER_OWNER_ORG));
		}
	}

	/**
	 * Add Data Range Data to a row on the Filing Rules list page.
	 */
	private static void addDateRangeData(RequestContext context, RequestFilingRule rule, ResourceBundle rmBundle, TimeZone tz, DefaultMultiColumnNodeData row_data) {
		String dataToAdd = RequestUtil.getDateRangeData(context.getLocalizer(), rule, rmBundle, tz);

		if (dataToAdd != null) {
			row_data.add(dataToAdd);
		}
	}

	/**
	 * Add File by Date Data to a row on the Filing Rules list page.
	 */
	private static void addFileByDateData(RequestContext context, RequestFilingRule rule, ResourceBundle rmBundle, Localizer localizer, TimeZone tz, DefaultMultiColumnNodeData row_data) {
		String dataToAdd = RequestUtil.getFileByData(rule, rmBundle, localizer, tz);

		if (dataToAdd != null) {
			row_data.add(dataToAdd);
		}
	}

	/**
	 * Create Rule Data Row for the Filing Rules list page.
	 */
	public static DefaultMultiColumnNodeData createRuleDataRow(RequestContext context, RequestFilingRule rule, ResourceBundle rmBundle, String chkbox, String orgName, boolean isFullList, Localizer localizer, TimeZone tz) {

		// Create the row data
		String rowID = rule.getID().toString();
		DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData(rowID);
		String requestType = rule.getRequestType();
		boolean isTimeOff = (Request.REQUESTTYPE_TIMEOFF.equals(requestType) || requestType == null);
		boolean isShiftBid = Request.REQUESTTYPE_SHIFTBID.equals(requestType);
		boolean isTimeOffWithdraw = Request.REQUESTTYPE_TIMEOFF_WITHDRAW.equals(requestType);
		boolean isCustomShift = Request.REQUESTTYPE_CUSTSHIFT.equals(requestType);
		boolean isFlexTime = Request.REQUESTTYPE_FLEXTIME.equals(requestType);
		boolean isShiftSwapWithdraw = Request.REQUESTTYPE_SHIFTSWAP_WITHDRAW.equals(requestType);

		// Checkbox cell
		if (isFullList) {
			row_data.add(chkbox);
		}
		
		// Request Type cell
		String rTypeRBKEY = getRequestTypeKey(isTimeOff, isShiftBid, isTimeOffWithdraw, isCustomShift, isFlexTime,isShiftSwapWithdraw);
		String rType = localizer.i18n(rmBundle, rTypeRBKEY);
		ResourceBundle rmEjbBundle = localizer.getBundle(RmEjbBundleKey.BUNDLE_NAME);
		
		String subTypes = null;
		try {
			subTypes = getSubtypesString(context, rule, rmEjbBundle, localizer, isTimeOff, isCustomShift, isFlexTime);
		} catch (Exception ex) {
			//do nothing, since an empty subtypes string is not critical
		}
		if (subTypes != null) {
			rType += " (" + subTypes + ")";
		}
		row_data.add(rType);

		addDateRangeData(context, rule, rmEjbBundle, tz, row_data);
		addFileByDateData(context, rule, rmEjbBundle, localizer, tz, row_data);

		if (isFullList) {
			row_data.add(orgName);
		}

		return row_data;
	}

	/**
	 * Get the request type bundle key.
	 */
	private static String getRequestTypeKey(boolean isTimeOff, boolean isShiftBid, 
			boolean isTimeOffWithdraw, boolean isCustomShift, boolean isFlexTime, boolean isShiftSwapWithdraw) {
		String rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_SHIFT_SWAP;
		if (isTimeOff) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_TIME_OFF;
		} else if (isShiftSwapWithdraw) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_SHIFT_SWAP_WITHDRAW;
		} else if (isTimeOffWithdraw) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_TIME_OFF_WITHDRAW;
		} else if (isShiftBid) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_SHIFT_BIDDING;
		} else if (isCustomShift) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_CUSTOM_SHIFT;
		} else if (isFlexTime) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_FLEX_TIME;
		}
		return rTypeRBKEY;
	}

	/**
	 * Get the subtypes string for a filing rule, which is displayed in the "Request Type" column
	 * of the filing rules list.
	 */
	private static String getSubtypesString(RequestContext context, RequestFilingRule rule, ResourceBundle rmBundle, 
			Localizer localizer, boolean isTimeOff, boolean isCustomShift, boolean isFlexTime) 
			throws BbmObjectNotFoundException, BbmFinderException, RemoteException, BbmException {
		String subTypes = null;

		if (isTimeOff || isFlexTime) {
			subTypes = RequestUtil.getTimeOffOrFlexTimeSubtypesString(context.isInWhatIfMode(), rule, rmBundle, localizer, isTimeOff);
		} else if (isCustomShift) {
			subTypes = RequestUtil.getCustomShiftSubtypesString(rule, rmBundle, localizer);
		}
		return subTypes;
	}

	/**
	 * Create No Data Message Row
	 */
	public static DefaultMultiColumnNodeData createNoDataRow(ResourceBundle rmBundle, boolean isFullList, Localizer localizer) {
		DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData("-1");
		int colspan = isFullList ? 5 : 3;
		String noDataMsg = localizer.i18n(rmBundle, RmWebBundleKey.ORG_RM_FILING_RULE_LIST_NO_DATA);
		row_data.add(noDataMsg, "", colspan);
		row_data.setSelectable(false);
		return row_data;
	}
}
