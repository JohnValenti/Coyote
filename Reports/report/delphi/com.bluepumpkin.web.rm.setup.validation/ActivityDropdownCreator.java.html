package com.bluepumpkin.web.rm.setup.validation;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;
import java.util.ResourceBundle;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityFilter;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;
import com.bluepumpkin.web.bbm.activity.ActivityModelHandler;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;
import com.witness.web.uif.util.js.JSUtil;

public class ActivityDropdownCreator {

	Collection<Activity> activities;
	Localizer localizer;
	ResourceBundle rmBundle;

	private static final String NONE = "-1";
	private static final String ALL = "-2";
	//see PageComponent.MULTI_VALUE_DATA
	private static final String MULTI_VALUE_DATA = "*";

	ActivityDropdownCreator(Localizer lz, ResourceBundle rmBundle) {
		localizer = lz;
		this.rmBundle = rmBundle;
	}

	@SuppressWarnings("unchecked")
	void intialize(String selectedOrganizationID, RequestContext context) {
		try {
			ActivityFilter activityFilter = new ActivityFilter();
			activityFilter.setDeleted(false);
			activities = ActivityModelHandler.getActivities(context, new ID(selectedOrganizationID), activityFilter);
		} catch (Exception e) {
			throw new RmRuntimeException(e);
		}
	}

	String getHtml(String key, Collection<Object> selectedValues, boolean usedInShift, boolean includeNone, boolean includeAll) {
		return getHtml(key, selectedValues, usedInShift, includeNone, includeAll, false);
	}

	String getHtml(String key, Collection<Object> selectedValues, boolean usedInShift, boolean includeNone, boolean includeAll,
			boolean selctionIsMultiValued) {

		List<StringsPair> options = new ArrayList<StringsPair>(2);
		
		if(selctionIsMultiValued) {
			options.add(new StringsPair(MULTI_VALUE_DATA, MULTI_VALUE_DATA));
		}

		if (includeNone) {
			options.add(new StringsPair(NONE, ""));
		}

		if (includeAll) {
			options.add(new StringsPair(ALL, localizer.i18n(rmBundle, RmWebBundleKey.REQUEST_TYPE_ALL)));
		}

		for (Activity act : activities) {
			addtoOptions(options, usedInShift, act);
		}

		String[] selected = getSelected(selectedValues, selctionIsMultiValued);

		return HtmlUtil.makeSelectInput(key, selected, JSUtil.ON_CHANGE, options, true, true);
	}

	private static void addtoOptions(List<StringsPair> options, boolean usedInShift, Activity act) {
		if (usedInShift) {
			if (act.isUsedInShiftEvent()) {
				options.add(new StringsPair(act.getID().toString(), act.getName()));
			}
		} else if (act.isUsedInCalendarEvent()) {//calendar event
			options.add(new StringsPair(act.getID().toString(), act.getName()));
		}
	}

	private static String[] getSelected(Collection<Object> selectedValues, boolean selctionIsMultiValued) {
		if (selctionIsMultiValued) {
			return new String[] {};
		}

		String[] selected = new String[selectedValues.size()];
		int i = 0;
		for (Object selVal : selectedValues) {
			selected[i] = getActivityID(selVal);
			i++;
		}
		return selected;
	}

	private static String getActivityID(Object selectedValue) {
		//when the selection is loaded from the database the map contains Integers, 
		//but in extractParamMap the code puts ID into this map
		if (selectedValue instanceof ID) {
			return Integer.toString(((ID) selectedValue).toInt());
		}
		return ((Integer) selectedValue).toString();

	}


}
