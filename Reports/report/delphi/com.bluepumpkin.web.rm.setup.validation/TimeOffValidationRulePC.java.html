package com.bluepumpkin.web.rm.setup.validation;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collection;
import java.util.HashSet;
import java.util.List;
import java.util.ResourceBundle;
import java.util.Set;

import javax.servlet.http.HttpServletRequest;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityProperties;
import com.bluepumpkin.ejb.rm.util.RmUtil;
import com.bluepumpkin.web.rm.keys.RmJavaScriptFileID;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.setup.RmSetupUtil;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

public class TimeOffValidationRulePC extends MultiColListPC {

	private static final long serialVersionUID = 1L;
	private static final String VALIDATIONBYTYPE_FN = "VALIDATIONBYTYPE_FN";
	private static final String MULTI_VALUED_KEYS_FN = "multiValuedRuleKeys";
	private static final String ALL = "all";

	private final transient ResourceBundle rmBundle;
	private final ValidationPageModel pageModel;


	private Set<String> multiValuedKeys;

	public TimeOffValidationRulePC(RequestContext context,
			ValidationPageModel pageModel, ResourceBundle rmBundle) {
		super(context);
		this.rmBundle = rmBundle;
		this.pageModel = pageModel;
		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
		addRequiredJavaScriptFile(RmJavaScriptFileID.TIME_OFF_VALIDATION_RULES);
	}

	public DefaultMultiColumnNodeData createValidateByTypeDropDown(boolean enabled) {
		String dropDownString = getValidateByTypeDropDownHtml(enabled);
		String label = m_localizer.i18n(rmBundle, RmWebBundleKey.VALIDATIONBYTYPE_ACTIVITIES);
		DefaultMultiColumnNodeData rowData = new DefaultMultiColumnNodeData("r0");
		rowData.add(label + ": " + dropDownString);
		return rowData;
	}

	public List<String> getApplicableTimeOffValidatorsOnFlexRequests() {
		return Arrays.asList(RmSetupUtil.getTimeOffKeysForFlexTimeAutoProcessing());
	}

	public boolean isSelectedActivitiesHasFlexActivity() {
		if (isAllActivitiesSelected()) {
			return false;
		}

		for (ActivityProperties pro : this.pageModel.getSelectedActivities()) {
			if (pro.getTimeoffWithFlex() == ActivityProperties.FLEX_TIME) {
				return true;
				
			}
		}

		return false;
	}
	
	public List<String> getApplicableTimeOffValidatorsAdvancedVTO() {
		return Arrays.asList(RmSetupUtil.getApplicableTimeOffValidatorsAdvancedVTO());
	}
	
	public boolean isSelectedActivitiesHasAdvancedVTOOptions() {
		if (isAllActivitiesSelected()) {
			return false;
		}

		for (ActivityProperties pro : this.pageModel.getSelectedActivities()) {
			if (pro.isUseAdvancedVTOOptions()) {
				return true;
			}
		}

		return false;
	}
	
	public boolean isOnlyActivitiesFlexTypeSelected() {
		if (isAllActivitiesSelected()) {
			return false;
		}

		for (ActivityProperties pro : this.pageModel.getSelectedActivities()) {
			if (pro.getTimeoffWithFlex() != ActivityProperties.FLEX_TIME) {
				return false;
			}
		}

		return true;
	}
	
	public boolean isOnlyActivitiesHasAdvancedVTOOptionsSelected() {
		
		if (isAllActivitiesSelected()) {
			return false;
		}

		for (ActivityProperties pro : this.pageModel.getSelectedActivities()) {
			if (!pro.isUseAdvancedVTOOptions()) {
				return false;
			}
		}

		return true;
	}
	
	public boolean isAllActivitiesSelected() {
		// This means "all selected"
		if (this.pageModel.getSelectedActivities() == null || this.pageModel.getSelectedActivities().isEmpty()) {
			return true;
		}
		
		return false;
	}	
	

	private String getValidateByTypeDropDownHtml(boolean enabled) {

		List<StringsPair> options = new ArrayList<StringsPair>(1);
		options.add(new StringsPair(ALL, i18n(rmBundle, RmWebBundleKey.REQUEST_TYPE_ALL)));

		for (Activity activity : pageModel.timeoffRequestableActivities) {
			options.add(new StringsPair(activity.getID().toString(), activity.getName()));
		}

		String[] selected = null;
		if (hasSelectedActivities()) {
			List<ID> selectedIDs = pageModel.getSelectedTimeoffTypeActivityIDs();
			selected = new String[selectedIDs.size()];
			int i = 0;
			for (ID id : selectedIDs) {
				selected[i++] = id.toString();
			}
		}

		if (selected == null) {
			selected = new String[1];
			selected[0] = ALL;
		}

		return HtmlUtil.makeSelectInput(VALIDATIONBYTYPE_FN, selected, "", options, true, enabled);
	}

	private boolean hasSelectedActivities() {
		return !pageModel.getSelectedTimeoffTypeActivityIDs().isEmpty();
	}

	protected void extractActivityIDs(HttpServletRequest request) {
		String[] extractedValues = request.getParameterValues(VALIDATIONBYTYPE_FN);
		saveSelectedActivitiesToPageModel(extractedValues);
	}




	private void saveSelectedActivitiesToPageModel(String[] selectedItems) {

		if (selectedItems == null || selectedItems.length < 1) {
			return;
		}

		if (ALL.equalsIgnoreCase(selectedItems[0])) {
			pageModel.setAllRequestableActivitesSelected(true);
			return;
		}

		pageModel.setAllRequestableActivitesSelected(false);

		List<ID> ids = new ArrayList<ID>(selectedItems.length);
		for (String val : selectedItems) {
			if (StringUtil.isEmpty(val)) {
				continue;
			}
			ids.add(RmUtil.getIntIDFromString(val));
		}
		pageModel.setSelectedTimeoffTypeActivityIDs(ids);
	}

	public void addMultiValuedRuleKey(String key) {
		if (multiValuedKeys == null) {
			multiValuedKeys = new HashSet<String>();
		}
		multiValuedKeys.add(key);
	}

	public void setMultiValuedRuleKeys(String multiValuedKeys) {
		for (String key : StringUtil.parseStringToCollection(multiValuedKeys, ",")) {
			if (key != null && !key.isEmpty()) {
				addMultiValuedRuleKey(key);
			}
		}
	}

	public boolean isKeyMulitValued(String key) {
		if (multiValuedKeys == null) {
			return false;
		}
		return multiValuedKeys.contains(key);
	}

	@Override
	public String getRequiredHTML() {
		if (multiValuedKeys == null) {
			return super.getRequiredHTML();
		}

		String keys = RmUtils.join(multiValuedKeys, ',');
		return super.getRequiredHTML() + HtmlUtil.makeHiddenInput(MULTI_VALUED_KEYS_FN, keys);
	}

}
