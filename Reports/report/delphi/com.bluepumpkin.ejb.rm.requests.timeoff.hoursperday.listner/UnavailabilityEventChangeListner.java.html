/**
 * Sends UnavailabilityEvent updates to RM/IFs through remote cache manager. This class hooks
 * the UnavailabilityEvent table changes (CRUD) and sends the appropriate notifications through
 * PeopleCacheProcessor. PeopleCacheProcessor will be used by the RemoteCacheManager to send
 * UnavailabilityEvent updates to RM/IFs.
 */
public class UnavailabilityEventChangeListner implements DAOChangeHook {

	private static Category m_cat = Log.initCategory(UnavailabilityEventChangeListner.class.getName());
	/**
	 * Notifies RemoteCacheManager to send UnavailabilityEvent creation.
	 * Will be called when UnavailabilityEvent table has a new entry created as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.UnavailabilityEvent.model.UnavailabilityEvent objects that were created.
	 */
	public void OnCreateObject(ValueObjectBase[] valueObjects) throws JdmoException {
		if (m_cat.isDebugEnabled()) m_cat.debug("UnavailabilityEvent Hook OnCreateObject valueObjects = " + valueObjects);
		if (valueObjects != null) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			m_cat.info("UnavailabilityEventChangeListner; create called with null or no objects created; passed valueObjects="+valueObjects );
		}
	}

	/**
	 * Notifies RemoteCacheManager to send UnavailabilityEvent deletion.
	 * Will be called when UnavailabilityEvent table has an entry deleted as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.UnavailabilityEvent.model.UnavailabilityEvent objects that were deleted.
	 */
	public void OnDeleteObject(ValueObjectBase[] valueObjects) throws JdmoException {
		// When an UnavailabilityEvent is deleted the send notification
		if (m_cat.isDebugEnabled()) m_cat.debug("UnavailabilityEvent Hook OnDeleteObject valueObjects = " + valueObjects);
		if (valueObjects != null && valueObjects.length > 0) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			m_cat.info("UnavailabilityEventChangeListner; delete called with null or no objects deleted; passed valueObjects="+valueObjects );
		}
	}

	/**
	 * Notifies RemoteCacheManager to send UnavailabilityEvent updates.
	 * Will be called when UnavailabilityEvent table has been updated as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param old and current value objects com.bluepumpkin.ejb.bbm.UnavailabilityEvent.model.UnavailabilityEvent
	 *            will be in oldAndNewObjects.
	 */
	public void OnUpdateObject(ValueObjectPairList oldAndNewObjects) throws JdmoException {
		Collection unavailabilityCol = new ArrayList();
		if (oldAndNewObjects != null && oldAndNewObjects.getCurrentValues() != null) {
			if (m_cat.isDebugEnabled()) m_cat.debug("UnavailabilityEvent Hook OnUpdateObject sending update notification " + oldAndNewObjects.getCurrentValues());
			ValueObjectBase[] newValueObjects = oldAndNewObjects.getCurrentValues();
			ValueObjectBase[] oldValueObjects = oldAndNewObjects.getOldValues();
			for (int i = 0; i < newValueObjects.length; i++) {
				CalendarEventAssignment newCalEvent = (CalendarEventAssignment) newValueObjects[i];
				CalendarEventAssignment oldCalEvent = (CalendarEventAssignment) oldValueObjects[i];
				try {
					if (newCalEvent.getEventType() == Event.EVENT_TYPE_UNAVAILABILITY) {
						notifyTOHoursPerDay(newCalEvent);
						if (!newCalEvent.getStartTime().equals(oldCalEvent.getStartTime())
						        || !newCalEvent.getEndTime().equals(oldCalEvent.getEndTime())) {
							notifyTOHoursPerDay(oldCalEvent);
						}
					}
				} catch (BbmUpdateException e) {
					e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
				}
			}
		} else {
			m_cat.info("UnavailabilityEventChangeListner; update called with null or no objects updated; passed oldAndNewObjects="+oldAndNewObjects );

		}
	}

	public void updateTOHoursPerDayForValueObjects(ValueObjectBase[] valueObjects) {
		try {
			for (int i = 0; i < valueObjects.length; i++) {
				CalendarEventAssignment calEvent = (CalendarEventAssignment) valueObjects[i];
				if (calEvent.getEventType() == Event.EVENT_TYPE_UNAVAILABILITY) {
					notifyTOHoursPerDay(calEvent);

				}
			}
			//getTOHoursPerDayManager().updateTOHoursPerDayForEmp(Collections.singleton(calEvent.gete));
		} catch (Exception e) {
			e.printStackTrace();
		}
	}

	public void notifyTOHoursPerDay(CalendarEventAssignment calEvent) throws BbmUpdateException {
		if(RequestUtil.isRMDebugEnabled())m_cat.debug("UnavailabilityEventChange EMPID=" + calEvent.getWorkResourceIDs() + " START:" + calEvent.getStartTime() + " : ENDDATE=" + calEvent.getEndTime());
		TONotifyMessage changeStateMessage = new TONotifyMessage(calEvent.getWorkResourceIDs(),
		        calEvent.getStartTime(), calEvent.getEndTime(), TONotifyMessage.TO_NOTIFY_EMP_UNAVAIL_CHANGE);
		changeStateMessage.sendMessage();
	}

	public TOHoursPerDayManager getTOHoursPerDayManager() throws BbmCreateException {
		return RmManagerFactory.getInstance(true).getTOHoursPerDayManager(null, null);
	}

}
