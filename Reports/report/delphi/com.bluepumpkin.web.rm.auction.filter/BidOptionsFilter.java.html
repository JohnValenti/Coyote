public class BidOptionsFilter
{
	//Constant values
	public static final int UNDEFINED_INT = -1;
	public static final float UNDEFINED_FLOAT = -1.0f;
	public static final String UNDEFINED_STR = "UNDEFINED";
	
	public static final String TEMPLATE_NAME = "templateName";

	public static final int EQUAL = 0;
	public static final int MIN = 1;
	public static final int MAX = 2;
	public static final int BETWEEN = 3;
	
	public static final int LENGTH_USE_ACTUAL = 0;
	public static final int LENGTH_USE_PAID_TIME = 1;
	
	public static final int SKILLS_MATCH_EXACT = 0;
	public static final int SKILLS_MATCH_AT_LEAST = 1;
	public static final int SKILLS_MATCH_AT_MOST = 2;
	
	//The data structure holding all the user's filter settings
	Map<Object, Object> map;
	
    /**
     * Constructor. Initialize this filter with all the defaults.
     */
	public BidOptionsFilter() {
		map = new HashMap<Object, Object>(40);
    	
    	//Set all required default values
    	
    	map.put("shiftIdMatchAll", Boolean.FALSE);
    	map.put("shiftLengthUse", new Integer(LENGTH_USE_ACTUAL));
    	map.put("shiftLengthMatchAll", Boolean.FALSE);
    	map.put("shiftStartMatchAll", Boolean.FALSE);
    	map.put("shiftEndMatchAll", Boolean.FALSE);    	

    	map.put("scheduleLengthUse", new Integer(LENGTH_USE_ACTUAL));

    	map.put("activity1IdMatchAll", Boolean.FALSE);    	
    	map.put("activity1LengthMatchAll", Boolean.FALSE);    	
    	map.put("activity1StartMatchAll", Boolean.FALSE);    	
    	
    	map.put("activity2IdMatchAll", Boolean.FALSE);    	
    	map.put("activity2LengthMatchAll", Boolean.FALSE);
    	map.put("activity2StartMatchAll", Boolean.FALSE);    	
    	
    	map.put("skillsMatchType", new Integer(SKILLS_MATCH_AT_LEAST));
    
    }
	
	public Object get(Object key) {
		return map.get(key);
	}
	
	public Object put(Object key, Object val) {
		return map.put(key, val);
	}

	public Object remove(Object key) {
		return map.remove(key);
	}

	public int getInt(Object key) {
		Object obj = map.get(key);
		if (obj instanceof Integer) {
				return ((Integer)obj).intValue();
		}
		return UNDEFINED_INT;
	}
	
	public float getFloat(Object key) {
		Object obj = map.get(key);
		if (obj instanceof Float) {
				return ((Float)obj).floatValue();
		}
		return UNDEFINED_FLOAT;
	}

	public ID getID(Object key) {
		Object obj = map.get(key);
		if (obj instanceof ID) {
				return (ID)obj;
		}
		return null;
	}

	public String getIDString(Object key) {
		Object obj = map.get(key);
		if (obj instanceof ID) {
			return Integer.toString(((ID) obj).toInt());
		}
		return "UNDEFINED";
	}

	public boolean getBool(Object key) {
		Object obj = map.get(key);
		if (obj instanceof Boolean) {
			return ((Boolean) obj).booleanValue();
		}

		return false;
	}

	public String getString(Object key) {
		Object obj = map.get(key);

		if (obj instanceof String) {
				return (String)obj;
		}
		return "UNDEFINED";
	}
	
	public boolean containsKey(Object key) {
		return map.containsKey(key);
	}

	public Collection getList(Object key) {
		Object obj = map.get(key);

		if (obj instanceof Collection) {
			return (Collection) obj;
		}

		return null;
	}
	
}
