package com.bluepumpkin.web.rm.auction.filter;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.keys.RmKeys;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.keys.PageAction;
import com.bluepumpkin.web.core.base.RequestHandler;

/**
 * Title:        BidOptionsFilterRH
 * Description:  Handle requests from the Bid Options Filter configuration popup.
 * Copyright:    Copyright (c) 2005
 * Company:      Witness Systems, Inc.
 * @author       Gonzalo Quintanar
 * @version      1.0
 */
public class BidOptionsFilterRH extends RequestHandler 
{
	private static final Category log = 
		Log.initCategory(BidOptionsFilterRH.class.getName());

   	/**
	 * Method invoked by Framework to Process the Request
	 */
    public void processRequest(RequestContext context) throws Exception 
	{
		if (isAction(context, PageAction.SAVE)) 
		{
			processSave(context);
		} 
		else if (isAction(context, "CLEAR_ACTION"))
		{
			processClearForm(context);
		}
	}

    /** 
     * Handle the Save request
     */       
	public void processSave(RequestContext context) 
	{
		// get data from the request
 		BidOptionsFilterPopupPM model = new BidOptionsFilterPopupPM(context);
		model.extractParameters(context.getRequest());
		model.validate();

		if (!model.hasError()) 
		{
			try 
			{
				//Get the filter which is already initialized with all the form fields (thanks to extractParameters)
				BidOptionsFilter filter = model.getFilter();
				
				//Store filter in session for use by Page Model
				if (filter != null)
					context.setAttribute( RequestContext.SESSION_SCOPE, RmKeys.BID_OPTIONS_FILTER, filter);
            } 
			catch(Exception ex) 
			{
				model.handleUnableToUpdateInfo(m_log, ex);
			}
		}
		model.setPageDirty(false);
		model.setIsRefreshOpenerOnClose(true);
		model.setIsCloseAfterLoad( !model.hasError() );
		
		//cache model to show error messages to user
		context.setPageModel(model);
	}

    /** 
     * Handle the Clear Form request
     */       
	public void processClearForm(RequestContext context) 
	{
		// get data from the request
 		BidOptionsFilterPopupPM model = new BidOptionsFilterPopupPM(context);
		model.extractParameters(context.getRequest());

		int auctionID = model.getAuctionID();
		int bidderID = model.getBidderID();
		
		//Clear everything in the model except the required popup params
		model = new BidOptionsFilterPopupPM(context);
		model.setAuctionID(auctionID);
		model.setBidderID(bidderID);

		//Reload the popup
		model.setPageDirty(false);
		model.setIsRefreshOpenerOnClose(true);
		model.setIsCloseAfterLoad(false);
		
		//cache model to show the cleared form to the user
		context.setPageModel(model);
	}

}
