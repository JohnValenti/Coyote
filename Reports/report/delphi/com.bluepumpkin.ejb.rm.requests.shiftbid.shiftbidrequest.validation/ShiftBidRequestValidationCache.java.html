public class ShiftBidRequestValidationCache extends ValidationCache {
    private static final Category m_cat = Log.initCategory(ShiftBidRequestValidationCache.class.getName());

    private ShiftBidder m_shiftBidder;

    private Collection m_bidScheds;

    private ShiftBidAuction m_sbAuction;

    private ID m_orgID;

    private TimeRange m_timeRange;

    private ID m_empID;

    private Pair m_empIDTimeRangePair;

    private final String CACHEKEY_NUMOFBIDS_SUBMITTED_FOR_EMP = "CACHEKEY_NUMOFBIDS_SUBMITTED_FOR_EMP";
    private static final String PHANTOM_NAME = "PHANTOM_NAME:";

    private ShiftBidRequest m_sbr;

    public ShiftBidRequestValidationCache(ShiftBidRequest sbr)  {
        super(sbr);
        m_sbr = sbr;
    }

    public Pair getEmpIDTimeRangePair() {
        if (m_empIDTimeRangePair != null) return m_empIDTimeRangePair;

        Collection empIDTimeRangePairs = m_sbr.getEmpIDTimeRangePairs();
        assert empIDTimeRangePairs.size() == 1:empIDTimeRangePairs.size();

        return (m_empIDTimeRangePair = (Pair) empIDTimeRangePairs.iterator().next());
    }

    @Override
	public ID getEmployeeID()  {
        if (m_empID != null) return m_empID;

        return (m_empID = m_sbr.getEmployeeID());
    }

    @Override
	public TimeRange getDateRange()  {
        if (m_timeRange != null) return m_timeRange;

        return (m_timeRange = (TimeRange) getEmpIDTimeRangePair().getSecond());
    }

    public ID getOrgID() throws Exception {
    	if (m_orgID != null) return m_orgID;

    	TimeRange range = getDateRange();
    	return (m_orgID = getOrgIDForEmployeeDuringPeriod(getEmployeeID(), range.getStartDate(), range.getEndDate()));
    }

    public ShiftBidAuction getShiftBidAuction() throws Exception {
        if (m_sbAuction != null) return m_sbAuction;

        ShiftBidAuctionManager sbAuctionMgr = RequestUtil.getShiftBidAuctionMgr(null, null);
        m_sbAuction = sbAuctionMgr.getAuctionByID(m_sbr.getShiftBidAuctionID(), ShiftBidAuction.DL_BASIC);
        return m_sbAuction;
    }

    public Collection getBiddableSchedules() throws Exception {
        BiddableScheduleDAO bidSchedDAO = null;
        try {
            if (m_bidScheds != null) return m_bidScheds;

            bidSchedDAO = new BiddableScheduleDAO(BiddableSchedule.DL_SHIFT_ASSIGNMENTS_COLL);
            m_bidScheds = bidSchedDAO.getBiddableSchedulesByIDs(
                m_sbr.getOptMethods().getBiddableScheduleIDs(), false, BiddableSchedule.DL_SHIFT_ASSIGNMENTS_COLL);

            return m_bidScheds;
        } finally {
            if (bidSchedDAO != null) bidSchedDAO.cleanUp();
        }
    }

    /**
     * @return
     */
    public ShiftBidder getShiftBidder() throws Exception {
        ShiftBidderDAO shiftBidderDAO = null;
        try {
            if (m_shiftBidder != null) return m_shiftBidder;

            shiftBidderDAO = new ShiftBidderDAO(ShiftBidder.DL_EMPLOYEE);

            if (m_sbr.getShiftBidderID() != null)
                m_shiftBidder = shiftBidderDAO.getShiftBidderByID(m_sbr.getShiftBidderID(), ShiftBidder.DL_EMPLOYEE);
            else if (m_sbr.getEmployeeID() != null)
                m_shiftBidder = shiftBidderDAO.getShiftBidderForEmpAndAuctionID(m_sbr.getEmployeeID(),
                    m_sbr.getShiftBidAuctionID(), ShiftBidder.DL_EMPLOYEE);

            return m_shiftBidder;
        } finally {
            if (shiftBidderDAO != null) shiftBidderDAO.cleanUp();
        }
    }

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.ValidationCache#getValidators()
     */
    @Override
	public Collection<String> getValidators() throws Exception {
        return super.getValidators(getOrgID(), m_sbr.getRequestType());
    }

    /**
     * @param id
     * @param id2
     * @param sbReqDAO
     * @return
     */
    public int getNumOfBidsSubmittedForBidderID(ID shiftBidderID, ID auctionID, String[] statuses,
        ShiftBidRequestDAO sbReqDAO) throws Exception {

        String cacheKey = RequestUtil.makeCacheKey(CACHEKEY_NUMOFBIDS_SUBMITTED_FOR_EMP, shiftBidderID, auctionID,
            StringUtil.createDelimitedString(statuses, ","));
        Integer numOfBidsSubmittedInt = (Integer) get(cacheKey);

        if (numOfBidsSubmittedInt != null) {
            if (m_cat.isDebugEnabled()) m_cat.debug(getCacheHitMsg(cacheKey, numOfBidsSubmittedInt));
            return numOfBidsSubmittedInt.intValue();
        }

        int numOfBidsSubmitted = sbReqDAO.getNumOfBidsSubmittedForBidderID(shiftBidderID, auctionID, statuses);

        put(cacheKey, new Integer(numOfBidsSubmitted));

        return numOfBidsSubmitted;
    }

    /**
     * Used in error messages. Employee name is formatted simplistically.
     */
    public String getPhantomNameByID(ID id) throws Exception {
        // key for cache
        String key = PHANTOM_NAME + id;
        //retrieve from cache.
        String name = (String) get(key);

        // if cache hit, return name.
        if (name != null) return name;

        Phantom phantom = null;
        ScheduleAccessManager sam = WfmManagerFactory.getScheduleAccessManager();
        phantom = sam.getPhantomByID(id);

        // if no locale info, format name ourselves, simplistically
        // todo: why isn't locale used?
        if (phantom != null) {
            name = phantom.getName();
        }

        // store in cache
        put(key, (name == null)?"":name);
        return name;
    }
