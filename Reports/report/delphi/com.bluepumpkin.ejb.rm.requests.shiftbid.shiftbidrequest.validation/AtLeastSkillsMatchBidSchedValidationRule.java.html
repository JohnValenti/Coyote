package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.validation;

import java.io.Serializable;
import java.util.Collection;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

import org.apache.log4j.Priority;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.BiddableSchedule;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidRequest;
import com.bluepumpkin.ejb.rm.util.ShiftBidAuctionUtil;
import com.verint.ejb.wfm.cache.WfmCacheUtil;

/**
 * Validates that the bidder has all the skills needed for each biddable schedule 
 */
public class AtLeastSkillsMatchBidSchedValidationRule implements Validator {
	private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;
	private static final String m_className = AtLeastSkillsMatchBidSchedValidationRule.class.getName();
	private static final Category m_cat = Log.initCategory(m_className);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

		try {

			ShiftBidRequest sbReq = (ShiftBidRequest) validatable;

			Set<ID> skillIDsNotMatched = new HashSet<ID>();
			Set<ID> bidderSkillIds = ShiftBidAuctionUtil.getActiveSkillsForBidder(sbReq);

			ValidationResult valResult = null;
			// iterate thru' the biddable schedules for the request.
			List<BiddableSchedule> bidScheds = sbReq.getOptMethods().getBiddableSchedules();
			for (BiddableSchedule bidSched : bidScheds) {

				Collection<ID> skillsNeededForSchedule = bidSched.getSkillIds();

				// Check that the bidder has all the skills needed for this biddable schedule
				for (ID skillNeeded : skillsNeededForSchedule) {
					//skillIDsNotMatched is used to avoid duplicate validation errors for the same skill
					if (!bidderSkillIds.contains(skillNeeded) && !skillIDsNotMatched.contains(skillNeeded)) {
						skillIDsNotMatched.add(skillNeeded);
						valResult = ValidationUtil.setSoftValidationResult(sbReq,
								RmEjbBundleKey.SBR_PHAN_BIDDER_SKILL_NON_MATCH,
								new Serializable[] { WfmCacheUtil.getSkillByID(skillNeeded).getName() }, m_className);
					}
				}

			}
			return valResult;
		} finally {
			m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
		}
	}
}
