public class NoShiftBidRequestDuplicateNamesHV implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

	public static final String CLASS_NAME = NoShiftBidRequestDuplicateNamesHV.class.getName();
	private static final Category m_cat = Log.initCategory(CLASS_NAME);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    @Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		m_cat.log(m_tracePriority, "Validation Rule: " + CLASS_NAME + " started");

        ShiftBidRequestDAO sbReqDAO = null;
        try {
            ShiftBidRequest sbReq = (ShiftBidRequest) validatable;

            sbReqDAO = new ShiftBidRequestDAO(ShiftBidRequest.DL_BASIC);
            List reqIDsWithGivenName = sbReqDAO.checkIfRequestWithNameExists(sbReq.getShiftBidderID(), sbReq.getName());

			// A request being created has its ID set to null.
			ID sbReqID = sbReq.getID();
			boolean reqBeingCreated = sbReqID == null;
            
            // for a request being created, verify if the name does not already exist.
            boolean duplicateFound = true;
            if ( (reqBeingCreated && reqIDsWithGivenName.isEmpty()) ) {
                duplicateFound = false;
			// for a request being updated, verify if the name already exists for another request.
            } else if ( !reqBeingCreated ) {
				// remove 'self' from list.
				reqIDsWithGivenName.remove(sbReqID);
				if ( reqIDsWithGivenName.isEmpty() ) {
					duplicateFound = false;
				}
            }

            return (duplicateFound)?
            	ValidationUtil.setAndLogHardValidationResult(sbReq,
									RmEjbBundleKey.SBR_REQ_WITH_SAME_NAME_ALREADY_EXISTS,
									RmEjbLogBundleKey.SBR_REQ_WITH_SAME_NAME_ALREADY_EXISTS,
            			new Serializable[] {sbReq.getName(), sbReq.getShiftBidderID()},
							CLASS_NAME) : null;
            	
        } finally {
			if (sbReqDAO != null) {
				sbReqDAO.cleanUp();
			}
			m_cat.log(m_tracePriority, "Validation Rule: " + CLASS_NAME + " done");
        }
    }
}
