package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.validation;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.List;

import org.apache.log4j.Priority;

import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.activity.model.Event;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidauction.model.ShiftBidAuction;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidRequest;
import com.bluepumpkin.ejb.rm.util.ShiftBidAuctionUtil;

/**
 * This rule is to check all published schedules in an whole SP wasn't changed.
 * The rule compare between published schedules and unpublished schedule
 *
 */
public class AgentSchedulesInSPNotChangeHV implements Validator {

	private static final Priority TRACEPRIORITY = RequestUtil.m_defaultTracePriority;
	private static final String CLASS_NAME = AgentSchedulesInSPNotChangeHV.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	@SuppressWarnings("unchecked")
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		LOG.log(TRACEPRIORITY, "Validation Rule: " + CLASS_NAME + " started");

		try {
			ShiftBidRequest sbr = (ShiftBidRequest) validatable;

			boolean isMultiShiftBlocksBidding = getShiftBidAuction(sbr).getIsMultiShift();

			if (isMultiShiftBlocksBidding) {
				List<ShiftAssignment> pubShiftAssignments = new ArrayList<ShiftAssignment>(getPublishedSchedules(sbr));
				List<ShiftAssignment> unPubShiftAssignments = new ArrayList<ShiftAssignment>(
						getUnPublishedSchedules(sbr));
				ShiftBidAuctionUtil.ShiftAssignmentComparator comparator = new ShiftBidAuctionUtil.ShiftAssignmentComparator();
				Collections.sort(pubShiftAssignments, comparator);
				Collections.sort(unPubShiftAssignments, comparator);

				if (!pubShiftAssignments.equals(unPubShiftAssignments)) {
					return ValidationUtil.setHardValidationResult(sbr,
							RmEjbBundleKey.SBR_MULTI_SHIFTS_BIDDING_SCHEDULES_CHANGED, CLASS_NAME);
				}
			}
		} finally {
			LOG.log(TRACEPRIORITY, "Validation Rule: " + CLASS_NAME + " ended");
		}

		return null;
	}

	protected ShiftBidAuction getShiftBidAuction(ShiftBidRequest sbr) {
		return sbr.getOptMethods().getShiftBidAuction();
	}

	@SuppressWarnings("unchecked")
	protected Collection<ShiftAssignment> getUnPublishedSchedules(ShiftBidRequest sbReq) throws Exception {
		ShiftBidAuction sba = getShiftBidAuction(sbReq);
		TimeRange spTimeRange = new TimeRange(sba.getStartTime(), sba.getEndTime());
		return ValidationUtil.getEventsForWorkResourceByType(Event.EVENT_TYPE_SHIFT_ASSIGNMENT, sbReq.getEmployeeID(),
				spTimeRange.getStartDate(), spTimeRange.getEndDate());
	}

	@SuppressWarnings("unchecked")
	protected Collection<ShiftAssignment> getPublishedSchedules(ShiftBidRequest sbReq) throws Exception {
		ShiftBidAuction sba = getShiftBidAuction(sbReq);
		TimeRange spTimeRange = new TimeRange(sba.getStartTime(), sba.getEndTime());
		return ValidationUtil.getPublishedEventsForWorkResourceByType(Event.EVENT_TYPE_SHIFT_ASSIGNMENT,
				sbReq.getEmployeeID(), spTimeRange.getStartDate(), spTimeRange.getEndDate());
	}
}
