package com.verint.web.rm.rest.facade.timeoff;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;

import javax.ws.rs.core.Response;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOPool;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.setup.allocations.AvailableTimeOffModelHandler;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.SystemWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.util.SecurityUtils;
import com.verint.web.rm.rest.entity.timeoff.TimeOffPoolEntity;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.ejb.core.security.model.User;
import com.witness.web.uif.system.RequestContext;

public class TimeOffPoolFacadeImpl implements ITimeOffPoolFacade {
	private static final Category logger = Log.initCategory(TimeOffPoolFacadeImpl.class.getName());

	public TimeOffPoolFacadeImpl() { // NOSONAR
	}

	@Override
	public List<TimeOffPoolEntity> getTOPoolByOrgId(RequestContext context, ID organizationID, boolean bIncludeParentOrg) throws WFOException {
		try {

			List<TimeOffPoolEntity> timeoffPoolEntityList = new ArrayList<TimeOffPoolEntity>();
			User user = context.getUser();
			boolean isAuthorized = false;
			if (organizationID != null) {
				isAuthorized = SecurityUtils.isPrivWithOrgScope(user, PrivilegeKeys.TOM_VIEWAVAILABLETIMEOFF, false, organizationID);
			}

			if (!isAuthorized) {

				throw new BusinessWFOException("Not authorized to get time off pools", Response.Status.CONFLICT);
			}
			Collection<TOPool> timeOffPoolCollection = AvailableTimeOffModelHandler.getToPoolsForOrg(organizationID, bIncludeParentOrg);

			if (timeOffPoolCollection != null && !timeOffPoolCollection.isEmpty()) {
				timeoffPoolEntityList = buildTimeOfPoolEntityList(timeOffPoolCollection);
			}
			return timeoffPoolEntityList;

		} catch (BusinessWFOException ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw ex;
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	@Override
	public TimeOffPoolEntity createTimeOffPool(RequestContext context, String toPoolName, String toPoolDesc, ID organizationID) throws WFOException {
		try {
			context = context;
			User user = context.getUser();
			boolean isAuthorized = false;
			if (organizationID != null) {
				isAuthorized = SecurityUtils.isPrivWithOrgScope(user, PrivilegeKeys.TOM_EDITAVAILABLETIMEOFF, false, organizationID);
			}

			if (!isAuthorized) {

				throw new BusinessWFOException("Not authorized to create time off pools", Response.Status.CONFLICT);
			}
			boolean isToPoolNameValid = isNameValid(toPoolName);
			if (!isToPoolNameValid) {
				throw new BusinessWFOException(RmWebBundleKey.TIME_OFF_POOL_MISSING_NAME, Response.Status.CONFLICT);
			}

			boolean isTOPoolNameExist = isNameAlreadyExist(toPoolName);
			if (!isTOPoolNameExist) {
				TOPool toPool = new TOPool();
				toPool.setName(toPoolName);
				toPool.setDescription(toPoolDesc);
				toPool.setOrganizationId(organizationID);
				ID toPoolId = AvailableTimeOffModelHandler.createTOPool(toPool);
				toPool.setId(toPoolId);
				return createAndGetTimeOffPoolEntity(toPool);

			} else {
				throw new BusinessWFOException(RmWebBundleKey.TIME_OFF_POOL_DUPLICATE_NAME, Response.Status.OK);
			}
		} catch (BusinessWFOException ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw ex;
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	private boolean isNameAlreadyExist(String name) throws Exception {
		try {
			ID toPoolID = AvailableTimeOffModelHandler.getPoolIDForName(name);
			if (toPoolID == null) {
				return false;
			} else {
				return true;
			}
		} catch (Exception e) {
			throw e;
		}
	}

	private TimeOffPoolEntity createAndGetTimeOffPoolEntity(TOPool toPool) {
		TimeOffPoolEntity toPoolEntity = new TimeOffPoolEntity();
		ID toPoolId = toPool.getId();
		toPoolEntity.setToPoolID(toPoolId.toInt());
		toPoolEntity.setToPoolName(toPool.getName());
		toPoolEntity.setToPoolDescription(toPool.getDescription());

		ID toPoolOrganizationId = toPool.getOrganizationId();
		toPoolEntity.setToPoolOrgID(toPoolOrganizationId.toInt());
		return toPoolEntity;
	}

	private boolean isNameValid(String name) {
		if (name == null || name.length() == 0) {
			return false;
		}
		return true;
	}


	private List<TimeOffPoolEntity> buildTimeOfPoolEntityList(Collection<TOPool> timeOffPools) {
		List<TimeOffPoolEntity> entityList = new ArrayList<TimeOffPoolEntity>();
		for (TOPool toPool : timeOffPools) {
			TimeOffPoolEntity toPoolEntity = createTOPoolEntity(toPool);
			entityList.add(toPoolEntity);
		}
		return entityList;
	}

	private TimeOffPoolEntity createTOPoolEntity(TOPool toPool) {
		TimeOffPoolEntity toPoolEntity = new TimeOffPoolEntity();
		ID toPoolId = toPool.getId();
		ID toPoolOrgId = toPool.getOrganizationId();
		String toPoolName = toPool.getName();
		String toPoolDesc = toPool.getDescription();

		toPoolEntity.setToPoolID(toPoolId.toInt());
		toPoolEntity.setToPoolOrgID(toPoolOrgId.toInt());
		toPoolEntity.setToPoolName(toPoolName);
		toPoolEntity.setToPoolDescription(toPoolDesc);

		return toPoolEntity;
	}

}
