package com.bluepumpkin.ejb.rm.requests.timeoff.waitlist.model;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;

/**
 * meta data information for Time off Request object, bet requests
 */
public class TOWaitlistFieldInfo extends FieldInfo {
	public final static int TIMEOFFWAITLIST_I_ID = 0;
	public final static int TIMEOFFWAITLIST_D_CREATEDON = 1;
	public final static int TIMEOFFWAITLIST_D_EXPIRYDATE = 2;
	public final static int TIMEOFFWAITLIST_I_CREATORID = 3;
	public final static int TIMEOFFWAITLIST_D_LASTMODIFIEDAT = 4;
	public final static int TIMEOFFWAITLIST_I_MODIFIEDBY = 5;
	public final static int TIMEOFFWAITLIST_I_TIMEOFFREQUESTID = 6;
	public final static int TIMEOFFWAITLIST_B_ISMANUALLYADDED = 7;
	private final static int TIMEOFFWAITLIST_NUM_OF_FIELDS = 8;

	private static String m_strTableName = "TIMEOFFWAITLIST";
	private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[TIMEOFFWAITLIST_NUM_OF_FIELDS];

	static {
		RequestUtil.initFieldInfoEntries(m_arrFieldInfo, TOWaitlistFieldInfo.class, "TIMEOFFWAITLIST_", TIMEOFFWAITLIST_NUM_OF_FIELDS);
	}

	// implement abstract methods from parent class
	public String getTableName() {
		return m_strTableName;
	}

	public int getIDFieldIndex() {
		return TIMEOFFWAITLIST_I_ID;
	}

	public FieldInfoEntry[] getFieldInfo() {
		return m_arrFieldInfo;
	}

	protected int getNumFields() {
		return TIMEOFFWAITLIST_NUM_OF_FIELDS;
	}

	public int getParentIDFieldIndex() {
		return TIMEOFFWAITLIST_I_TIMEOFFREQUESTID;
	}
}
