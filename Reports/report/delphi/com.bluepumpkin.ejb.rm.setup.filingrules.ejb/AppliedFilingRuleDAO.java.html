    /**
     * Update the applied status for the for the RequestFilingRules
     * whose ids and status pair are passed as Map argument.
     * @param pOrganizationId -  ID object for organization.
     * @param idToAppliedMap - RequestFilingRuleID -> isApplied (true or false)
     * @throws BbmUpdateException if there is an error.
     */
	public void saveAppliedStatus(ID pOrganizationId, Map idToAppliedMap) throws Exception {
		//get the RequestFilingRules that need isApplied status to be updated
		RequestFilingRuleDAO requestFilingRuleDAO = new RequestFilingRuleDAO();
		Set filingRuleIDs = idToAppliedMap.keySet();
		Collection filingRules = requestFilingRuleDAO.getObjectsByIDs(filingRuleIDs); //rule.isApplied will not be set here

		//create a HashMap of ID -> RequestFilingRule
		HashMap idToRuleMap = new HashMap(filingRules.size());
		for (Iterator itr = filingRules.iterator(); itr.hasNext();) {
			RequestFilingRule rule = (RequestFilingRule) itr.next();
			idToRuleMap.put(rule.getID(), rule);
		}

		HashMap ruleToAppliedMap = new HashMap();
		for (Iterator itr = idToAppliedMap.keySet().iterator(); itr.hasNext();) {
			ID ruleID = (ID) itr.next();
			boolean newIsApplied = idToAppliedMap.get(ruleID).toString().equals("true");

			RequestFilingRule filingRule = (RequestFilingRule) idToRuleMap.get(ruleID);

			ruleToAppliedMap.put(filingRule, new Boolean(newIsApplied));

			//only subOrgAllowOverride scope rules allow update of isApplied
			if (filingRule.getOrgScope().equals(RequestFilingRule.SCOPE_SUB_ORGS_ALLOW_OVERRIDE)) {
				AppliedFilingRule appliedFilingRule = getAppliedRuleOfInterest(pOrganizationId, ruleID);
				if (appliedFilingRule != null) {
					//this was already an override, and it will remain an override
					if (appliedFilingRule.getApplied() != newIsApplied) {
						appliedFilingRule.setApplied(newIsApplied);
						updateObject(appliedFilingRule);
					}
				} else if (needToOverrideForApplied(pOrganizationId, ruleID, newIsApplied)) //if the inherited isApplied value != newIsApplied
				{
					//this is a new override
					appliedFilingRule = new AppliedFilingRule();
					appliedFilingRule.setOrganizationId(pOrganizationId);
					appliedFilingRule.setApplied(newIsApplied);
					appliedFilingRule.setRequestFilingRuleId(ruleID);
					createObject(appliedFilingRule);
				}
			}
		}
	}//saveAppliedStatus

    /**
     * Determine whether we need to override an inherited AppliedFilingRule (if one exists at a parent org).
     * @param orgID
     * @param ruleID
     * @param newIsApplied
     * @return true if we need to override an inherited AppliedFilingRule. If none exists at a parent org,
     *         we still return true because we need to create one. Return false if an AppliedFilingRule
     *         exists at a parent org that has the same applied value that we want (newIsApplied).
     */
	private boolean needToOverrideForApplied(ID orgID, ID ruleID, boolean newIsApplied) throws Exception {
		//get the list of parent orgs in order of immediate parent first, to root org last
		List parentOrgs = CacheUtilBBM.getParentOrgIDs(orgID, getDMO()); //list will not include orgID, since we don't need it
		for (Iterator orgIt = parentOrgs.iterator(); orgIt.hasNext();) {
			ID curOrgId = (ID) orgIt.next();
			AppliedFilingRule appliedFilingRule = getAppliedRuleOfInterest(curOrgId, ruleID);
			if (appliedFilingRule != null) {
				//found the rule. This first one is the only one we care about, since it's value is flowing down to us.
				return (appliedFilingRule.getApplied() != newIsApplied);
			}
		}

		//ruleID was not linked to any of our parent orgs. We will need to create a link of our own.
		return true;
	}

    /**
     * Get all AppliedFilingRules for any combination of the orgIDs and filingRuleIDs parameters.
     * @param orgIDs - Organization IDs.
     * @param filingRuleIDs - RequestFilingRule IDs.
     * @return all AppliedFilingRules for any combination of the orgIDs and filingRuleIDs parameters.
     */
	private AppliedFilingRule getAppliedRuleOfInterest(ID orgID, ID filingRuleID) throws Exception {
		AppliedFilingRule appliedFilingRule = null;

		StringBuffer strQuery = null;
		JdmoQuery jQuery1 = null;
		strQuery = getSelectStatement();
		strQuery.append(" FROM APPLIEDFILINGRULE A ");
		strQuery.append(" WHERE ORGANIZATIONID = ");
		//Fix QC196696: SQL inject-1st error
		strQuery.append(JdmoUtil.asSqlLiteral(orgID.toString()));
		strQuery.append(" AND REQUESTFILINGRULEID = ");
		strQuery.append(filingRuleID.toString());

		jQuery1 = m_dmo.createQuery(strQuery.toString(), Jdmo.STATIC_QUERY);
		JdmoRowset rs = m_dmo.createRowset(jQuery1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

		if (rs.next()) {
			appliedFilingRule = (AppliedFilingRule) readObjectFromDB(rs);
		}
		rs.close();
		return appliedFilingRule;
	}

	public void deleteByFilingRuleID(ID pFilingRuleID) throws Exception {
		StringBuilder lStmt;
		lStmt = new StringBuilder(" DELETE APPLIEDFILINGRULE ");
		lStmt.append(" WHERE REQUESTFILINGRULEID = ?");
		Object[] parm = new Object[1];
		parm[0] = pFilingRuleID;

		m_dmo.executePCommand(lStmt.toString(), parm);
	}

    /**
     * Create one AppliedFilingRule for each RequestFilingRule in a Collection.
     * @param organizationId - the org ID to use for each AppliedFilingRule
     * @param rules - a collection of RequestFilingRule's to create AppliedFilingRule's for.
     * @throws BbmCreateException
     */
	public void createObjects(ID organizationId, Collection rules) throws BbmCreateException {
		Iterator itr = rules.iterator();
		while (itr.hasNext()) {
			RequestFilingRule rule = (RequestFilingRule) itr.next();

			AppliedFilingRule appliedFilingRule = new AppliedFilingRule();
			appliedFilingRule.setOrganizationId(organizationId);

			boolean bIsApplied = true;
			if (rule.getOrgScope().equals(RequestFilingRule.SCOPE_THIS_ORG_ONLY) && !rule.getOrganizationId().equals(organizationId)) {
				bIsApplied = false;
			}

			appliedFilingRule.setApplied(bIsApplied);
			appliedFilingRule.setRequestFilingRuleId(rule.getID());
			createObject(appliedFilingRule);
		}
	}

	/**
	 * Create one AppliedFilingRule for the newly created RequestFilingRule.
	 * @param rule
	 * @throws BbmCreateException
	 */
	public void createObjects(RequestFilingRule rule) throws BbmCreateException {
		AppliedFilingRule appliedFilingRule = new AppliedFilingRule();
		appliedFilingRule.setOrganizationId(rule.getOrganizationId());
		appliedFilingRule.setApplied(true); //isApplied will always be true for this parent org
		appliedFilingRule.setRequestFilingRuleId(rule.getID());
		createObject(appliedFilingRule);
	}

	/**
	 * If a RequestFilingRule's ORGSCOPE changed, then we remove all child overrides for that rule.
	 */
	public void updateObjects(RequestFilingRule rule) throws Exception {
		// check if the user changed the ORGSCOPE value
		StringBuilder strQuery = new StringBuilder(64);
		strQuery.append("SELECT * ");
		strQuery.append(" FROM REQUESTFILINGRULE ");
		strQuery.append(" WHERE ID = ?");
		strQuery.append(" AND ORGSCOPE != ?");
		JdmoQuery jQuery1 = m_dmo.createQuery(strQuery.toString(), Jdmo.PARAM_QUERY);
		jQuery1.setParString(1, rule.getID().toString());
		jQuery1.setParString(2, rule.getOrgScope());
		JdmoRowset rs = m_dmo.createRowset(jQuery1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
		boolean doUpdate = rs.next();
		rs.close();

		if (doUpdate) {
			//First delete any child org isApplied overrides for this rule
			StringBuilder lStmt;
			lStmt = new StringBuilder(" DELETE FROM APPLIEDFILINGRULE ");
			lStmt.append(" WHERE REQUESTFILINGRULEID = ?");
			lStmt.append(" AND ORGANIZATIONID <> ?");
			Object[] parm = new Object[2];
			parm[0] = rule.getID().toString();
			parm[1] = rule.getOrganizationId().toString();
			m_dmo.executePCommand(lStmt.toString(), parm);

			//Now we need to update the APPLIEDFILINGRULE for the rule's org to use isApplied=true, which is the default value for all org scopes
			lStmt = new StringBuilder(" UPDATE APPLIEDFILINGRULE ");
			lStmt.append(" SET ISAPPLIED = 1");
			lStmt.append(" WHERE REQUESTFILINGRULEID = ?");
			lStmt.append(" AND ORGANIZATIONID = ?");
			parm = new Object[2];
			parm[0] = rule.getID().toString();
			parm[1] = rule.getOrganizationId().toString();
			m_dmo.executePCommand(lStmt.toString(), parm);
		}
	}
