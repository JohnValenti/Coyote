package com.bluepumpkin.ejb.rm.util;

import java.util.Collection;
import java.util.HashSet;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.timeoffcalculator.TONotifyMessage;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;

/**
 * @author Sameet Joshi
 *         <p/>
 */
public class TONotifyMessageClient {
	private static Category m_cat = Log.initCategory(TONotifyMessageClient.class.getName());

	public static void scanTOWaitlist(TORequest reqAgg, Collection<TOChoice> apprChoices) throws Exception {
		for (TOChoice choice : apprChoices) {
			scanTOWaitlist(reqAgg, choice);
		}
	}

	public static void scanTOWaitlist(TORequest reqAgg, TOChoice apprChoice) throws Exception {
		//This message is for waitlist purposes only .
		// Hours per day will be refreshed by unpublish event from schedule access manager API
		if (reqAgg.getCache().getTOPool() != null) {
			TONotifyMessage msg = new TONotifyMessage(TONotifyMessage.TO_NOTIFY_APPROVED_TOREQUEST_WITHDRAW);
			msg.setTOPoolId(reqAgg.getCache().getTOPool().getID());
			msg.setStartDate(apprChoice.getStartDate());
			msg.setEndDate(apprChoice.getEndDate());
			msg.sendMessage();
		}
	}

	public static void scanTOWaitlist(ID orgID, boolean includeChildOrgs, int toNotifyType) throws BbmUpdateException {
		HashSet orgList = new HashSet();
		orgList.add(orgID);
		if (includeChildOrgs) {
			try {
				Collection childOrgList = BbmManagerFactory.getWorkResourceManager().getOrganizationsChildrenByIDs(orgList);
				if (childOrgList != null && !childOrgList.isEmpty()) {
					orgList.addAll(childOrgList);
				}
			} catch (Exception e) {
				throw new BbmUpdateException(e);
			}
		}
		TONotifyMessage changeStateMessage = new TONotifyMessage(toNotifyType);
		changeStateMessage.setObject(orgList);
		changeStateMessage.sendMessage();
	}
}
