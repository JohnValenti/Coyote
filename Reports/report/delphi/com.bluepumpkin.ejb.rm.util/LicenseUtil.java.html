package com.bluepumpkin.ejb.rm.util;


import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.core.CoreManagerFactory;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.witness.ejb.core.licensing.LicenseKeys;
import com.witness.ejb.core.licensing.ejb.LicenseManager;

public class LicenseUtil {
	private static final Category LOGGER = Log.initCategory(LicenseUtil.class.getName());

	private static volatile Boolean hasAdvancedRMLicense; // NOSONAR
	private static volatile Boolean hasShiftBiddingLicense; // NOSONAR
	private static volatile Boolean hasShiftSwapLicense; // NOSONAR
	private static volatile Boolean hasTimeOffManagerLicense; // NOSONAR

	private LicenseUtil() {
		super();
	}

	public static boolean isAdvancedRMLicense() {
		return true;
	}

	public static boolean hasShiftBiddingLicense() {
		
		if (hasShiftBiddingLicense == null) {
			synchronized (LicenseUtil.class) {
				if (hasShiftBiddingLicense == null) {
					hasShiftBiddingLicense = hasLicense(LicenseKeys.APP_RM_SB);
				}
			}
		}
		return hasShiftBiddingLicense;
	}

	public static boolean hasShiftSwapLicense() {
		if (hasShiftSwapLicense == null) {
			synchronized (LicenseUtil.class) {
				if (hasShiftSwapLicense == null) {
					hasShiftSwapLicense = hasLicense(LicenseKeys.APP_RM_SS);
				}
			}
		}
		return hasShiftSwapLicense;
	}

	public static boolean hasTimeOffManagerLicense() {
		if (hasTimeOffManagerLicense == null) {
			synchronized (LicenseUtil.class) {
				if (hasTimeOffManagerLicense == null) {
					hasTimeOffManagerLicense = hasLicense(LicenseKeys.APP_RM_TOM);
				}
			}
		}
		return hasTimeOffManagerLicense;
	}

	private static boolean hasLicense(String licenseKey) {
		try {
			LicenseManager licManager = CoreManagerFactory.getLicenseManager();
			if (licManager.isLicensed(licenseKey)) {
				return true;
			}
		} catch (Exception e) {
			LOGGER.l7dError(RmEjbBundleKey.UNABLE_TO_GET_LICENSE, e);
		}
		return false;
	}

	// A temporary method to check for the RM license for Validation By Type
	// This may not be needed in the end if Validation By Type is not tied to the RM License
	// This is pending currently (May 2016)
	public static boolean isAdvancedRMLicenseForValidationByType(){
		return isAdvancedRMLicense();
	}

	// ShiftSwapCalendarInfo is implemented in 15.1.0 130, but not 15.1.1!
	public static boolean isAdvancedRMShiftSwapCalendarInfo() {
		return isAdvancedRMLicense();
	}
}
