package com.bluepumpkin.ejb.rm.util;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public abstract class RmDAOBase<T extends ValueObjectBase> extends DAOBase<T> {

	public RmDAOBase() {

	}

	public RmDAOBase(Jdmo dmo) {
		super(dmo);
	}

	public RmDAOBase(Jdmo dmo, boolean isDmoCreated) {
		super(dmo, isDmoCreated);
	}

	@Override
	public Collection<T> getObjectsByIDs(Collection<? extends ID> arrIDs) throws BbmFinderException {
		if (arrIDs == null || arrIDs.isEmpty()) {
			return Collections.emptyList();
		}
		StringBuffer strWhere = new StringBuffer("A." + m_fieldInfo.getIDFieldName() + " IN  ?");
		return getObjectsWhere(strWhere, arrIDs);
	}

	public Collection<T> getObjectsWhere(StringBuffer where, Object... parameters) throws BbmFinderException {
		String sql = getQuerySQL(where).toString();
		return getObjectsWithFullSqlQueryStr(sql, parameters);
	}

	public Collection<T> getObjectsWhere(JdmoPreparedStatementBuilder where) throws BbmFinderException {
		StringBuffer whereSql = new StringBuffer(where.getSql());
		String sql = getQuerySQL(whereSql).toString();
		return getObjectsWithFullSqlQueryStr(sql, where.getParameters());
	}

	public Collection<T> getObjectsWithFullSqlQueryStr(JdmoPreparedStatementBuilder builder) throws BbmFinderException {
		return getObjectsWithFullSqlQueryStr(builder.getSql(), builder.getParameters());
	}


	public Collection<T> getObjectsWithFullSqlQueryStr(String sql, Object... parameters) throws BbmFinderException {
		try {
			JdmoRowset res = JdmoPreparedStatement.createRowset(m_dmo, sql, parameters);
			return getObjectsFromRowset(res);
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}

	
	public int executeSql(Jdmo dmo, String sql, Object... parameters) throws BbmFinderException {
		try {
			return JdmoPreparedStatement.executeSql(dmo, sql, parameters);
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}

	public Collection<T> getObjectsFromRowset(JdmoRowset rs) throws BbmFinderException {
		List<T> arrItems = new ArrayList<T>();
		// call jDMO to fetch the data
		try {
			long lCount = m_lRecords;
			while (rs.next() && (lCount != 0)) {
				T item = readObjectFromDB(rs);
				arrItems.add(item);
				lCount--;
			}
			updateExternalIDInBatch(arrItems);
			onRetrieve(arrItems);
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}

		return this.filterObjectsFromFullSqlQuery(arrItems);
	}

}
