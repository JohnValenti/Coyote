package com.bluepumpkin.ejb.rm.util;

import com.bluepumpkin.common.jdmo.Jdmo;

public class BasicJdmoDao {

	protected Jdmo dmo = null;

	public BasicJdmoDao() {
		dmo = new Jdmo(false);
	}

	public BasicJdmoDao(Jdmo dmo) {
		this.dmo = dmo;
	}

	public void cleanUp() {
		if (dmo != null) {
			dmo.cleanUp();
			dmo = null;
		}
	}

	public static void cleanUp(BasicJdmoDao dmo) {
		if (dmo != null) {
			dmo.cleanUp();
		}
	}
}
