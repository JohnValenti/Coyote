package com.bluepumpkin.web.rm.requests;

import com.bluepumpkin.ejb.rm.requests.common.model.Request;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAuditTrail;
import com.witness.web.uif.system.RequestContext;
import com.bluepumpkin.web.rm.requests.shiftbidding.AgentsShiftBiddingRequestsPM;
import com.bluepumpkin.web.rm.requests.shiftswap.AgentsShiftSwapRequestsPM;
import com.bluepumpkin.web.rm.requests.timeoff.AgentsTimeOffRequestsPM;
import com.bluepumpkin.web.rm.requests.flextime.AgentsFlexTimeRequestsPM;
import com.bluepumpkin.web.rm.requests.custshift.*;
/**
 * Title:		     AgentsRequestsRH
 * Description:  Manager View Request List page Request Handler
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, inc
 * @author       David Su
 * @version      1.0
 */
public class AgentsRequestsRH extends RequestsRH {
	private static final String SCREEN_ALL_REQUESTS = "AGENTS_REQUESTS";
	private static final String SCREEN_TIMEOFF_REQUESTS = "AGENTS_TIMEOFF_REQUESTS";
	private static final String SCREEN_FLEXTIME_REQUESTS = "AGENTS_FLEXTIME_REQUESTS";
	private static final String SCREEN_SHIFTSWAP_REQUESTS = "AGENTS_SHIFTSWAP_REQUESTS";
	private static final String SCREEN_SHIFTBID_REQUESTS = "AGENTS_SHIFTBID_REQUESTS";
	private static final String SCREEN_CUSTSHIFT_REQUESTS = "AGENTS_CUSTSHIFT_REQUESTS";

	/**
	 * Process Request for the User page
	 *
	 * @param context The Request Context
	 */  
	public void processRequest(RequestContext context) throws Exception {
		super.processRequest(context);

		String screenName = findNextScreen(context);
		setNextScreenName(context, screenName);
	}

	/**
	 * Determine which Screen to go based on Request Filter
	 */
	protected String findNextScreen(RequestContext context) {
		String result = SCREEN_ALL_REQUESTS;

		String reqType = RequestUtil.getRequestType(context);
		if (Request.REQUESTTYPE_TIMEOFF.equals(reqType) ||
				RequestAuditTrail.STATUS_WAITLIST.equals(reqType)) {
			result = SCREEN_TIMEOFF_REQUESTS;
		} else if (Request.REQUESTTYPE_FLEXTIME.equals(reqType)) {
			result = SCREEN_FLEXTIME_REQUESTS;
		} else if (Request.REQUESTTYPE_SHIFTSWAP.equals(reqType)) {
			result = SCREEN_SHIFTSWAP_REQUESTS;
		} else if (Request.REQUESTTYPE_SHIFTBID.equals(reqType)) {
			result = SCREEN_SHIFTBID_REQUESTS;
		} else if (Request.REQUESTTYPE_CUSTSHIFT.equals(reqType)) {
			result = SCREEN_CUSTSHIFT_REQUESTS;
		}

		return result;
	}

	/**
	 * Create AgentsRequestsPM or subclass
	 *
	 * NOTE: Meant to be overwritten by subclass
	 */
	protected AgentsRequestsPM createAgentsRequestsPM(RequestContext context) {
		AgentsRequestsPM model = null;

		String reqType = RequestUtil.getRequestType(context);
		if (Request.REQUESTTYPE_TIMEOFF.equals(reqType) ||
				RequestAuditTrail.STATUS_WAITLIST.equals(reqType)) {
			model = new AgentsTimeOffRequestsPM(context);
		} else if (Request.REQUESTTYPE_FLEXTIME.equals(reqType)) {
			model = new AgentsFlexTimeRequestsPM(context);
		} else if (Request.REQUESTTYPE_SHIFTSWAP.equals(reqType)) {
			model = new AgentsShiftSwapRequestsPM(context);
		} else if (Request.REQUESTTYPE_SHIFTBID.equals(reqType)) {
			model = new AgentsShiftBiddingRequestsPM(context);
		} else if (Request.REQUESTTYPE_CUSTSHIFT.equals(reqType)) {
			model = new AgentsCustomShiftRequestsPM(context);
		} else {
			model = super.createAgentsRequestsPM(context);
		}

		return model;
	}
}
