package com.verint.web.rm.rest.entity.timeoff;

import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOCalendarDayData;
import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rest.util.FormatUtils;

public class TimeOffCalendarDayEntity extends AbstractWFOEntity {

	private final float allocatedHours;
	private final float approvedHours;
	private final float availableHours;
	private final String date;
	private final int dayType;
	private final float pendingHours;

	public TimeOffCalendarDayEntity(TOCalendarDayData day) {
		date = FormatUtils.dateToISO8601SecPrecisionString(day.getDate());
		dayType = day.getDayType();
		allocatedHours = day.getHourAvail();
		availableHours = day.getHourRemaining();
		pendingHours = day.getHourPending();
		approvedHours = day.getScheduledTO();
	}

	public float getAllocatedHours() {
		return allocatedHours;
	}

	public float getApprovedHours() {
		return approvedHours;
	}

	public float getAvailableHours() {
		return availableHours;
	}

	public String getDate() {
		return date;
	}

	public int getDayType() {
		return dayType;
	}

	public float getPendingHours() {
		return pendingHours;
	}
}
