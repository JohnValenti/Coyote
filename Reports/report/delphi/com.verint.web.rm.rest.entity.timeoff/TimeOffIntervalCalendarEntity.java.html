package com.verint.web.rm.rest.entity.timeoff;

import java.util.ArrayList;
import java.util.List;

import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendar;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendarInterval;

@XmlRootElement(name = "timeOffIntervalCalendar")
public class TimeOffIntervalCalendarEntity extends TimeOffCalendarEntity {

	private final List<TimeOffIntervalCalendarIntervalEntity> intervals;

	public TimeOffIntervalCalendarEntity(TOIntervalCalendar toic, boolean isTOPoolView, ID activityID) {
		super(toic.getStartTime(), toic.getEndTime(), isTOPoolView, activityID);

		List<TOIntervalCalendarInterval> toicIntervals = toic.getIntervals();
		intervals = new ArrayList<TimeOffIntervalCalendarIntervalEntity>(toicIntervals.size());
		for (TOIntervalCalendarInterval toicInterval : toicIntervals) {
			intervals.add(new TimeOffIntervalCalendarIntervalEntity(toic, toicInterval, isTOPoolView));
		}
	}

	public List<TimeOffIntervalCalendarIntervalEntity> getIntervals() {
		return intervals;
	}
}
