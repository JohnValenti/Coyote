package com.bluepumpkin.web.rm.auction;

import com.witness.web.uif.base.*;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.pagemodel.SettingPM;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.l10n.RmWebLogBundleKey;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.keys.RmPageAction;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.bluepumpkin.ejb.bbm.base.BbmCreateDuplicateKeyException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateDuplicateKeyException;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidauction.model.ShiftBidAuction;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;

/**
 * File:         AuctionSettingsRH
 * Description:  Request Handler for for the shift bidding auctions create/edit
 *               it is used either in the work pane or in the ext. list selection pane
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       pbosin, David Su
 * @version      1.0
 * Created on July 29, 2002, 1:22 PM
 */
public class AuctionSettingsRH extends SettingRH {
	private static final Category LOG = Log.initCategory(AuctionSettingsRH.class.getName());

	/**
	 * Process Request actions for the Auctions Employees List page
	 *
	 * @param context The Request Context
	 * @throws java.rmi.RemoteException
	 */
	@Override
	public void processRequest(RequestContext context) throws Exception, WebException {
		if (isAction(context, RmPageAction.OPEN_AUCTION)) {
			processAuctionStatus(context, true, false);
		} else if (isAction(context, RmPageAction.CLOSE_AUCTION)) {
			processAuctionStatus(context, false, false);
		} else if (isAction(context, RmPageAction.CLOSE_AUCTION_AND_ASSIGN_BONUS)) {
			processAuctionStatus(context, false, true);
		} else {
			super.processRequest(context);
		}
	}

	/**
	 * Return New Instance of SettingPM
	 */
	@Override
	protected SettingPM newSettingPM(RequestContext context) {
		return new AuctionSettingsPM(context);
	}

	/**
	 * Process Auction Status
	 */
	protected void processAuctionStatus(RequestContext context, boolean isOpenAuction, boolean isAddNoneApprBonus) {
		AuctionSettingsPM model = new AuctionSettingsPM(context);
		AuctionUtil.processAuctionStatus(context, model, isOpenAuction, isAddNoneApprBonus);
		prepareForNextPage(context, model, null);
	}

	/**
	 * Process Ceate New Action
	 */
	@Override
	protected void processAdd(RequestContext context) {
		AuctionSettingsPM model = new AuctionSettingsPM(context);

		if (isAuthorized(context, PrivilegeKeys.SB_MODIFYAUCTIONS)) {
			// Use same information
			model.extractParameters(context.getRequest());

			// Blank out name
			model.setSelectedID(null);
			model.getAuction().setName("");

			// Indicate it is in add mode
			model.setRefreshSelection(true);
			model.setIsAddMode(true);
		}

		prepareForNextPage(context, model, FORM_PAGE);
	}

	/**
	 * Process Save Action
	 */
	@Override
	protected void processSave(RequestContext context){
		AuctionSettingsPM model = new AuctionSettingsPM(context);

		if (isAuthorized(context, PrivilegeKeys.SB_MODIFYAUCTIONS)) {
			try {
				// Use same information
				model.extractParameters(context.getRequest());
				ShiftBidAuction existingAuction=null;
				if (model.getSelectedIDObject()!=null){
					ID auctionID = model.getSelectedIDObject();
					existingAuction = AuctionMH.getAuction(auctionID);
				}
				model.validate(existingAuction);

				if (!model.hasError()) {
						ShiftBidAuction auction = model.getAuction();
						if (model.isAddMode()) {
							//Fix default status for new auction is always Open
							//Since there is an issue : if the selected bid before is closed and when creating a new auction. the status will be Closed by default.
							auction.setStatus(ShiftBidAuction.STATUS_OPEN_FOR_BIDDING);
							ID newID = AuctionMH.createAuction(auction);
							model.setSelectedID(newID.toString());
						} else {
							auction.setOrganizationId(existingAuction!=null?existingAuction.getOrganizationId():auction.getOrganizationId());
							auction.setEmployeeTypeSid(existingAuction!=null?existingAuction.getEmployeeTypeSid():auction.getEmployeeTypeSid());
							AuctionMH.updateAuction(auction);
						}
						model.setRefreshSelection(true);
						model.setIsAddMode(false);
				}
			} catch (BbmCreateDuplicateKeyException ex) {
				model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_AUCTION_NAME_EXISTS,
						RmWebBundleKey.BUNDLE_NAME);
			} catch (BbmUpdateDuplicateKeyException ex) {
				model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_AUCTION_NAME_EXISTS,
						RmWebBundleKey.BUNDLE_NAME);
			} catch (RmHardValidationException ex) {
				Message msg = model.handleUnableToUpdateInfo(LOG, ex);
				msg.setAssociatedMessage(ex.getLocalizedMessage(context.getLocalizer(), context.getViewingTimeZone()));
			} catch (Exception ex) {
				Message msg = new Message(Message.ERROR_TYPE,
						RmWebBundleKey.UNABLE_TO_SAVE_AUCTION, RmWebBundleKey.BUNDLE_NAME);
				model.addPageMessage(msg);
				LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_AUCTION, ex);
			}

		}

		prepareForNextPage(context, model, null);
	}

	/**
	 * Process Delete Action
	 */
	@Override
	protected void processDelete(RequestContext context) {
		AuctionSettingsPM model = new AuctionSettingsPM(context);

		if (isAuthorized(context, PrivilegeKeys.SB_MODIFYAUCTIONS)) {
			ID selectedID = getSelectedID(context);
			if (selectedID!=null) {
				try {
					AuctionMH.removeAuction(selectedID);
					model.setRefreshSelection(true);
				} catch(RmHardValidationException ex) {
					Message msg = model.handleUnableToUpdateInfo(LOG, ex);
					msg.setAssociatedMessage(ex.getLocalizedMessage(context.getLocalizer(), context.getViewingTimeZone()));
				} catch(Exception ex) {
					Message msg = new Message(Message.ERROR_TYPE,
							RmWebBundleKey.UNABLE_TO_REMOVE_AUCTION, RmWebBundleKey.BUNDLE_NAME);
					model.addPageMessage(msg);
					LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_REMOVE_AUCTION, ex);
				}
			}
		}

		prepareForNextPage(context, model, null);
	}
}
