public class AuctionSchedulesPM extends AuctionBonusDescriptionListPM {
	private static String FORM_ACTION = "rm_sa_schedules";

	private final ResourceBundle m_bbm_bundle;
	private SchedulingPeriod m_schedPeriod;
	private HashMap<ID, String> m_empTemplateNameMap;
	private Map<ID, Boolean> orgDescSettings;
	private TimeZone m_campaignTz;
	private Collection<ID> spOrgList = Collections.EMPTY_LIST;
	private Collection<ID> viewUserScopeOrgIDs = Collections.EMPTY_LIST;
	private boolean displayDescColumn = false;
	private User viewingUser = null;

	/**
	 * Constructor
	 */
	public AuctionSchedulesPM(RequestContext context) {
		super(context, FORM_ACTION, RmWebBundleKey.RM_SA_SCHEDULES);

		m_bbm_bundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);
		setJSMediator(RmJavaScriptFileID.MEDIATOR_SCHEDULES);
		
		viewingUser = context.getUser();

		setDefaultPageIDSize(10);
		setIsPaginatingSelectedIDs(false);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, AuctionSchedulesPM.class);
	}

	/**
	 * Load Data
	 * @throws Exception
	 */
	@Override
	protected void loadData() throws Exception {
		try {
			if (hasSelectedID()) {
				ID auctionID = getSelectedIDObject();
				auction = AuctionUtil.getAuctionAndValidate(auctionID, m_context, this);
				m_schedPeriod = auction.getOptMethods().getSP();
				m_campaignTz = CampaignModelHandler.getCampaign(m_context, m_schedPeriod.getCampaignID()).getTimeZone();
				orgList = AuctionBonusDescriptionListPM.getOrgList(m_context, auction);
				spOrgList = CampaignModelHandler.getLinkedOrgsforSP(m_context, m_schedPeriod.getID());
				orgDescSettings = (Map<ID, Boolean>) AuctionUtil.setOrgDisplayDecription(spOrgList);

				// Get User's Scoped OrgIDs
				if (viewUserScopeOrgIDs.isEmpty()) {
					viewUserScopeOrgIDs = OrganizationModelHandler.getAllOrgIdsInUserScope(m_context, viewingUser);
				}
				
				if (isPaginationNavPressed()) {
					//get next chunk
					listData = AuctionMH.getBiddableSchedules(auction.getID(), orgID, getPageIDs());
				} else {
                    Pair biddableSchedPair = AuctionMH.getBiddableSchedulesForAuction(auctionID, orgID, null, null, false,
                                getSortColumn(),
                                isSortAscending(),
                                getPageIDSize(),
                                AuctionUtil.getWeekRanges(m_schedPeriod, auction.getOptMethods().getCampaign().getTimeZone()));
					listData = (Collection) biddableSchedPair.getSecond();

					initPaginationIDs((Collection) biddableSchedPair.getFirst());
				}
				addPopupArgs(RmPageAction.POPUP_AGENT_SKILLS_ACTION, RmKeys.RH_AGENT_SKILLS,
						null, "employeeID,empTemplateID", "450,350,ctr,ctr", true, 1);
				ShiftDetailsPopupPM.addPopupArgs(this, RmPageAction.POPUP_SHIFT_DETAILS_ACTION);
			} else {
				addSelectAuctionMsg();
			}
		} catch (RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataErrorAndLogAsDebug(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
		} catch (RmException ex) {
			if (!StringUtil.isEmpty(ex.getLocalizedMessage())) {
				addPageMessage(Message.ERROR_TYPE, ex.getLocalizedMessage());			
			} else {
				Message msg = handleUnableToLoadDataErrorAndLogAsDebug(log, ex);
				msg.setAssociatedMessage(ex.getLocalizedMessage(m_localeContext.getLanguageLocale()));
			}
		}
	}

	/**
	 * Initialize Header
	 */
	@Override
	protected void initHeader() {
		TableHeaderPC header = m_list.getHeader();

		displayDescColumn = AuctionUtil.getDisplayDescriptionColumn(spOrgList, viewUserScopeOrgIDs, orgDescSettings);

		header.setDefaultSortColumn(String.valueOf(BiddableSchedule.SORTBY_NONE));

		header.addColumnHeaderData(String.valueOf(BiddableSchedule.SORTBY_PHANTOMNAME),
				i18n(m_bbm_bundle, BbmWebBundleKey.TEMPLATE_NAME));

		if (hasRMAdvanceLicense && displayDescColumn) {
			header.addColumnHeaderData(String.valueOf(BiddableSchedule.SORTBY_DESCRIPTION),
					i18n(m_common_bundle, UIFWebBundleKey.DESCRIPTION));
		}

		header.addColumnHeaderData(String.valueOf(BiddableSchedule.SORTBY_NUMBER_AVAIALBLE),
				i18n(m_common_bundle, UIFWebBundleKey.AVAILABLE)); //.setIsForNumeric(true);

		header.addColumnHeaderData(String.valueOf(BiddableSchedule.SORTBY_BONUSASSIGNED),
				i18n(m_common_bundle, UIFWebBundleKey.BONUS));

		header.addColumnHeaderData(String.valueOf(BiddableSchedule.SORTBY_HOURS),
				i18n(m_common_bundle, UIFWebBundleKey.HOURS)); //.setIsForNumeric(true);

		AuctionUtil.addShiftDatesToHeader(m_schedPeriod, m_localizer, m_campaignTz, header, false);
	}

	/**
	 * Initialize List
	 */
	@Override
	protected void initSelectableItems() {
		if (listData.isEmpty() || !isPageEnabled) {
			return;
		}

		ArrayList listModel = new ArrayList(listData.size());

		m_empTemplateNameMap = AuctionUtil.getEmpTemplateNameMap(listData, m_context);
		boolean isEditable = isEditActionEnabled();
		for (Iterator it = listData.iterator(); it.hasNext();) {
			BiddableSchedule schedule = (BiddableSchedule) it.next();
			String rowID = schedule.getID().toString();
			int avail = schedule.getNumOfAvailableInstances();
			int bonus = schedule.getBonusPoints();
			String description = schedule.getDescription();

			Collection shifts = schedule.getOptMethods().getShiftAssignments();

			DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData(rowID);
			node.add(getTemplateUI(schedule), CSSUtil.STYLE_NORMAL_LEFT); //508

			if (hasRMAdvanceLicense && displayDescColumn) {
				node.add(getDescriptionUI(rowID, description, isEditable));
			}

			node.add(new NumberWrapper(avail, m_localizer), CSSUtil.STYLE_NUMERIC);
			node.add(getBonusUI(rowID, bonus, isEditable));
			node.add(AuctionUtil.getScheduleHourWrapper(shifts, m_localizer), CSSUtil.STYLE_NUMERIC);

			AuctionUtil.addShiftsData(m_schedPeriod, shifts, m_localizer, tz, m_campaignTz, node);

			node.setSelectable(false);
			listModel.add(node);
		}

		m_list.setListModel(listModel);
	}

	@Override
	protected boolean hasEditableField() {
		return true;
	}

	//////////////////////////////////////////////////////////////////////////////
	// Helper Code
	//////////////////////////////////////////////////////////////////////////////
	private BasicWrapper getTemplateUI(BiddableSchedule schedule) {
		Phantom phantomEmp = schedule.getOptMethods().getEmployee();
		ID etID = phantomEmp.getEmployeeTemplateID();

		EmployeeName name = new EmployeeName();
		name.setFirstName(phantomEmp.getName());
		name.setLastName(m_empTemplateNameMap.get(etID));

		String eNameDisplay = m_localizer.formatName(name);

		String alt = i18n(m_bbm_bundle, BbmWebBundleKey.BBM_PEOPLE_SKILLS);
		String onClickJS = "pageMediator.popupAgentsSkills('','" + etID + "');";
		String display = HtmlLinkUtil.createActiveLink(eNameDisplay, alt, "", onClickJS);
		return new BasicWrapper(eNameDisplay, display);
	}

	/**
	 * Determines if a biddeable schedule description text box and whether
	 * it is editable based on the Schedule Period Organization associations.
	 * 
	 * @param schedule Schedule that can be bided on.
	 * @return True if the Organization Setting Display Description is checked, false otherwise.
	 */
	private boolean displayDescriptionEditBox(BiddableSchedule schedule) {
		Boolean displayDecription;

		displayDecription = orgDescSettings.get(schedule.getOrgID());
		return displayDecription.booleanValue();
	}
