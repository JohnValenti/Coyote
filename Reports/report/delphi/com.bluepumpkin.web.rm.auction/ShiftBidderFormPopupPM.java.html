public class ShiftBidderFormPopupPM extends PopupWindowPM {
	private static final String RANK_BY_SCORE = String.valueOf(ShiftBidder.RANKBY_SCORE);
	private static final String RANK_BY_SENIORITY = String.valueOf(ShiftBidder.RANKBY_SENIORITY);
	private static final String RANK_BY_EMPRANK = String.valueOf(ShiftBidder.RANKBY_EMPLOYEERANK);
	private static final String RANK_BY_POINTS = String.valueOf(ShiftBidder.RANKBY_ACCUMULATEDBONUS);
	private static final String RANK_BY_BONUS = String.valueOf(ShiftBidder.RANKBY_SHIFTBIDDER_BONUS);

	private static final String REQUEST_HANDLER = "shiftbidder_form_pu";

	private static final String AUCTION_ID_FN = "auctionID";
	private static final String WAITING_IDS_FN = "waitingIDs";
	private static final String SELECTED_IDS_FN = "selectedIDs";
	private static final String ORG_ID_FN = "orgID";

	private static final String IS_ADD_SERIALLY = "isAddSerially";
	private static final String RANK_BY_FN = "rankBy";
	private static final String THINK_TIME_FN = "thinkTime";

	private static final String IS_ADD_ALL_WAITING_FN = "isAddAllWaiting";
	private static final String DEADLINE_FN = "deadline";


	private static final String POPUP_ARGS = AUCTION_ID_FN;
	private static final String POPUP_VALPARAM_FOR_OBO = IS_ADD_SERIALLY + "=true";

	private static final String AFTER_MAIN_JS = "\n" +
			"function afterMain() { \n" +
			"  var openerMediator = window.opener.pageMediator; \n" +
			"  var wIDs = openerMediator.getExpandoAttribute('" + WAITING_IDS_FN + "'); \n" +
			"  var sIDs = openerMediator.getExpandoAttribute('" + SELECTED_IDS_FN + "'); \n" +
			"  pageMediator.setValue('" + WAITING_IDS_FN + "', wIDs); \n" +
			"  pageMediator.setValue('" + SELECTED_IDS_FN + "', sIDs); \n" +
			"}";

	private FormInputPC m_inputPC;
	private FormTwoColumnPC m_formPC;
	private DatePickerPC m_datePickerPC;
	private ResourceBundle m_bundle;

	private ID m_auctionID;
	private String m_waitingIDs;
	private String m_selectedIDs;
	private int m_rankBy;
	private int m_thinkTime;
	private ShiftBidAuction m_auction;
	private Date m_deadline;
	private boolean m_isAddSerially = false;
	private boolean m_isAddAllWaiting = false;
	private ID m_orgID;

	public ShiftBidderFormPopupPM(RequestContext context) {
		super(context);
		m_bundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);

		m_inputPC = new FormInputPC(context);
		addChildComponent(m_inputPC);

		m_datePickerPC = new DatePickerPC(context, DEADLINE_FN);
		m_datePickerPC.setIsTimeEnabled(true);
		addChildComponent(m_datePickerPC);

		m_formPC = new FormTwoColumnPC(context);
		addChildComponent(m_formPC);
	}
	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, ShiftBidderFormPopupPM.class);
	}

	/**
	 * Add Popup Args for ShiftDetails Popup to specified component with associated
	 * page Action
	 */
	public static void addPopupArgs(PageComponent pc, String pageAction) {
		pc.addPopupArgs(pageAction, REQUEST_HANDLER, null, POPUP_ARGS,
				"400,300,ctr,ctr", true, 1);
	}

	/**
	 * Add Popup Args for ShiftDetails Popup for Adding One by One
	 * to specified component with associated page Action
	 */
	public static void addPopupArgsForOneByOne(PageComponent pc, String pageAction) {
		pc.addPopupArgs(pageAction, REQUEST_HANDLER, POPUP_VALPARAM_FOR_OBO,
				POPUP_ARGS,	"400,300,ctr,ctr", true, 1);
	}

	/**
	 * Initialize Model with Data
	 */
	protected void initializeModel() {
		if (m_isInitialized) return; else m_isInitialized = true;
		try {
			if (m_auctionID!=null) {
				m_auction = AuctionMH.getAuction(m_auctionID);
				Date deadline = m_auction.getAuctionDeadLine();

				Date lowerLimit = new Date();
				if (deadline.before(lowerLimit)) {
					lowerLimit = deadline;
					addPageMessage(Message.WARNING_TYPE,
							RmWebBundleKey.AUCTION_HAS_EXPIRED,	RmWebBundleKey.BUNDLE_NAME);
				}

				m_datePickerPC.setDate(deadline);
				m_datePickerPC.setLowerLimit(lowerLimit);
				m_datePickerPC.setUpperLimit(deadline);
			}
		} catch(RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataErrorAndLogAsDebug(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		} finally {
			initContentTitle();
			initContent();
			initToolbar();
		}
	}

	/**
	 * Initialize Content Title
	 */
	protected void initContentTitle() {
		String rbKey = m_isAddSerially
				? RmWebBundleKey.ADD_SERIALLY_TO_AUCTION
				: RmWebBundleKey.ADD_TO_AUCTION;
		String pageTitle = i18n(m_bundle, rbKey);
		m_contentTitlePC.setPageTitle(pageTitle);

		String itemName = m_auction!=null?m_auction.getName():"";
		m_contentTitlePC.setItemName(itemName);
	}

	/**
	 * Initialize Content
	 */
	protected void initContent() {
		m_formPC.addRow(
				i18n(m_common_bundle, UIFWebBundleKey.ADD), getAddTypeSelector()
		);

		if (m_isAddSerially) {
			m_formPC.addRow(
					i18n(m_common_bundle, UIFWebBundleKey.ORDER), getRankSelector()
			);

			m_formPC.addRow(
					i18n(m_common_bundle, UIFWebBundleKey.TIME_INTERVAL), getThinkTimeUI()
			);
		} else {
			m_datePickerPC.setInputFieldDesc(UIFWebBundleKey.DEADLINE, UIFWebBundleKey.BUNDLE_NAME);
			m_formPC.addRow(
					i18n(m_common_bundle, UIFWebBundleKey.DEADLINE),
					m_datePickerPC.getHtmlDisplay()
			);
		}
	}

	/**
	 * Prepare toolbar for rendering
	 */
	protected void initToolbar() {
		String saveLbl = i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_SAVE);
		m_toolbar.addValidateSubmitTextButton(RmPageAction.ADD_TO_AUCTION, saveLbl, true);
		ToolbarUtil.addCloseWindowCancelBtn(m_toolbar, true);
	}

	/**
	 * Generate Custom JavaScript Code
	 *
	 * Default behavior is to retrieve sub-component JavaScript Init code
	 */
	public String getCustomJavaScript() {
		return super.getCustomJavaScript() + AFTER_MAIN_JS;
	}

	/**
	 * Return the Html code to render the display of the component
	 */
	public String getHtmlDisplay() {
		return m_formPC.getHtmlDisplay();
	}

	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return REQUEST_HANDLER;
	}

	/**
	 * Return Required HTML for this component
	 */
	public String getRequiredHTML() {
		return super.getRequiredHTML() +
				HtmlUtil.makeHiddenInput(AUCTION_ID_FN, m_auctionID) +
				HtmlUtil.makeHiddenInput(IS_ADD_SERIALLY, m_isAddSerially) +
				HtmlUtil.makeHiddenInput(WAITING_IDS_FN, m_waitingIDs) +
				HtmlUtil.makeHiddenInput(SELECTED_IDS_FN, m_selectedIDs) +
				HtmlUtil.makeHiddenInput(ORG_ID_FN, m_orgID);
	}

	/**
	 * Retrieve request parameters from request object to the page model object.
	 */
	public boolean extractParameters(HttpServletRequest request) {
		boolean success=super.extractParameters(request);

		m_deadline = m_datePickerPC.extractDate(DEADLINE_FN);

		return success;
	}

	//////////////////////////////////////////////////////////////////////////////
	// Helper Methods
	//////////////////////////////////////////////////////////////////////////////
	private void addRankOption(String rankBy, String rbKey, List options) {
		options.add( new StringsPair(rankBy, i18n(m_common_bundle, rbKey)) );
	}

	private String getRankSelector() {
		ArrayList options = new ArrayList(4);

		if (m_auction!=null && m_auction.getUsesScoring()) {
			addRankOption(RANK_BY_SCORE, UIFWebBundleKey.SCORE, options);
		}
		addRankOption(RANK_BY_SENIORITY, UIFWebBundleKey.SENIORITY, options);
		addRankOption(RANK_BY_EMPRANK, UIFWebBundleKey.RANK, options);
		addRankOption(RANK_BY_POINTS, UIFWebBundleKey.POINTS, options);
		addRankOption(RANK_BY_BONUS, UIFWebBundleKey.BONUS, options);

		return HtmlUtil.makeSelectInput(RANK_BY_FN, "", JSUtil.ON_CHANGE, options);
	}

	private String getThinkTimeUI() {
		m_inputPC.reset();
		m_inputPC.setInputValue("60");
		m_inputPC.setIsRequired(true);
		m_inputPC.setInputFN(THINK_TIME_FN);
		m_inputPC.setInputType(FormInputPC.TYPE_INTEGER);
		m_inputPC.setInputFieldDesc(UIFWebBundleKey.TIME_INTERVAL, UIFWebBundleKey.BUNDLE_NAME);
		m_inputPC.initForDisplay();
		return m_inputPC + " (" + i18n(m_common_bundle, UIFWebBundleKey.MINUTES) + ")";
	}

	private String getAddTypeSelector() {
		ArrayList options = new ArrayList(2);
		options.add(
				new StringsPair("false",
						i18n(m_bundle, RmWebBundleKey.ONLY_SELECTED_N_WAITING))
		);
		options.add(
				new StringsPair("true",
						i18n(m_bundle, RmWebBundleKey.ALL_WAITING))
		);

		String selectedVal = String.valueOf(m_isAddAllWaiting);
		return HtmlUtil.makeSelectInput(IS_ADD_ALL_WAITING_FN, selectedVal,
				JSUtil.ON_CHANGE, options);
	}
}
