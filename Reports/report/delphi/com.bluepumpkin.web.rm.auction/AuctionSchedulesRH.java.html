package com.bluepumpkin.web.rm.auction;

import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.keys.RmPageAction;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.l10n.RmWebLogBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.WebException;
import com.witness.web.uif.base.WorkpaneRequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.pagemodel.WorkpanePageModel;
import com.witness.web.uif.system.RequestContext;

/**
 * File:         AuctionSchedulesRH
 * Description:  Request Handler for for the list of schedules in shift bidding auctions 
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       pbosin, David Su
 * @version      1.0
 * Created on July 29, 2002, 1:22 PM
 */
public class AuctionSchedulesRH extends WorkpaneRequestHandler {
	private static final Category m_log = Log.initCategory(AuctionSchedulesRH.class.getName());

	/**
	 * Process Request actions for the Auctions Schedules List page
	 *
	 * @param context The Request Context
	 */
	@Override
	public void processCustomRequest(RequestContext context) throws WebException {
		if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		} else if (isAction(context, RmPageAction.RESYNC_SHIFT_BID_SCHEDULES)) {
			processResyncSchedules(context);
		}
	}//processRequest

	private void processResyncSchedules(RequestContext context) {
		AuctionSchedulesPM model = new AuctionSchedulesPM(context);
		model.extractParameters(context.getRequest());
		ID auctionID = model.getSelectedIDObject();
		if (auctionID != null) {
			String message = AuctionMH.resynchronizeAuctionSchedules(context, auctionID, true);
			model.addPageMessage(Message.INFO_TYPE, message);
			context.setPageModel(model);
		}
	}

	/**
	 * Process Save Action
	 */
	private void processSave(RequestContext context) {
		AuctionSchedulesPM model = new AuctionSchedulesPM(context);
		model.extractParameters(context.getRequest());
		Map<ID, String> bonusMap = model.getBonusMap();
		Map<ID, String> descriptionMap = model.getDescriptionMap();
		boolean hasAdvancedRMLicense = model.isAdvancedRMLicense();
		ID auctionID = model.getSelectedIDObject();
		
		if (!bonusMap.isEmpty() && !descriptionMap.isEmpty()) {
			try {
				AuctionMH.updateSchedulesBonusAndDescription(auctionID, bonusMap, descriptionMap);
			} catch(RmHardValidationException ex) {
				Message msg = model.addPageMessage(Message.ERROR_TYPE,
						RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
				msg.setAssociatedMessage(ex.getLocalizedMessage(context.getLocalizer(), context.getViewingTimeZone()));
				m_log.l7dDebug(RmWebLogBundleKey.UNABLE_TO_UPDATE_INFO, ex);
			} catch (Exception ex) {
				model.addPageMessage(Message.ERROR_TYPE,
						RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
				m_log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPDATE_INFO, ex);
			}
		} else {
			if (!bonusMap.isEmpty()) {
				try {
					AuctionMH.updateSchedulesBonus(auctionID, bonusMap);
				} catch(RmHardValidationException ex) {
					Message msg = model.addPageMessage(Message.ERROR_TYPE,
							RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
					msg.setAssociatedMessage(ex.getLocalizedMessage(context.getLocalizer(), context.getViewingTimeZone()));
					m_log.l7dDebug(RmWebLogBundleKey.UNABLE_TO_UPDATE_INFO, ex);
				} catch (Exception ex) {
					model.addPageMessage(Message.ERROR_TYPE,
							RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
					m_log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPDATE_INFO, ex);
				}
			}
		}
		context.setPageModel(model);
	}

    /**
	 * Return Page Model to perform generic action request on
	 */
	@Override
	public WorkpanePageModel getPageModel(RequestContext context) {
		AuctionSchedulesPM model = new AuctionSchedulesPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		return model;
	}
}
