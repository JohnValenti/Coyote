public class AuctionEmployeesRH extends WorkpaneRequestHandler {
	private static final Category LOG = Log.initCategory(AuctionEmployeesRH.class.getName());
	private static final String BID_OPTIONS_SCREEN_NAME= "RM_SA_BID_OPTIONS";
	private static final String GET_METHOD= "GET";
	private static final String PAGE_ACTION_STR= "pageAction";
	private static final String APPSCAN_ERROR_LOG = "CSRF security constraint violated. Cannot GET with a pageAction parameter";
	/**
	 * Process Request actions for the Auctions Employees List page
	 *
	 * @param context The Request Context
	 */
	@Override
	public void processCustomRequest(RequestContext context) throws Exception {

		if (isAction(context, RmPageAction.OPEN_AUCTION)) {
			processAuctionStatus(context, true, false);
		} else if (isAction(context, RmPageAction.CLOSE_AUCTION)) {
			processAuctionStatus(context, false, false);
		} else if (isAction(context, RmPageAction.CLOSE_AUCTION_AND_ASSIGN_BONUS)) {
			processAuctionStatus(context, false, true);
		} else if (isAction(context, RmPageAction.BID_FOR_EMP)) {
			processBidForEmp(context);
		} else if (isAction(context, RmPageAction.ADD_TO_AUCTION)) {
			processAdd(context);
		} else if (isAction(context, RmPageAction.STOP_ONE_BY_ONE_ADD_TO_AUCTION)) {
			processStopSerializedAdd(context);
		} else if (isAction(context, PageAction.REMOVE)) {
			processRemove(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		} else if (isAction(context, RmPageAction.WITHDRAW_APPROVED_BID)) {
			processWithdrawApprovedBid(context);
		} else {
			String method = context.getRequest().getMethod();
			String queryString = context.getRequest().getQueryString();
			//Cannot GET with a pageAction parameter.
			boolean inValid =  GET_METHOD.equals(method) && !StringUtil.isEmpty(queryString) && queryString.indexOf(PAGE_ACTION_STR)>-1;
			//Fix Appscan issue "Body Parameter Accepted in Query"
			if (inValid) {
				LOG.error(APPSCAN_ERROR_LOG);
				throw new Exception(APPSCAN_ERROR_LOG);
			}
		}
	}

	private static void processAuctionStatus(RequestContext context, boolean isOpenAuction, boolean isAddNoneApprBonus) {
		AuctionEmployeesPM model = new AuctionEmployeesPM(context);
		AuctionUtil.processAuctionStatus(context, model, isOpenAuction, isAddNoneApprBonus);
	}

	private static void processBidForEmp(RequestContext context) {
		String sBidderID = context.getRequest().getParameter(WorkpaneFieldNames.PERFORM_ACTION_ON_ID_FN);
		BidOptionsPM model = new BidOptionsPM(context);
		model.extractParameters(context.getRequest(), WorkpaneFieldNames.SELECTED_ID_FN);
		model.setBidderID(new ID(sBidderID));
		context.setPageModel(model);
		context.setNextScreenName(BID_OPTIONS_SCREEN_NAME);
	}

	private static void processAdd(RequestContext context) {
		ShiftBidderFormPopupPM model = new ShiftBidderFormPopupPM(context);
		model.extractParameters(context.getRequest());
		Collection waitingIDs = new ArrayList();

		try {
			//get waiting ids for the auction
			Collection bidderColl = AuctionMH.getBidders(model.getAuctionID(), model.getOrgID());

			for (Iterator it = bidderColl.iterator(); it.hasNext();) {
				ShiftBidder shiftBidder = (ShiftBidder) it.next();
				if (shiftBidder.getStatus() == ShiftBidder.STATUS_NOT_ADDED) {
					waitingIDs.add(shiftBidder.getID());
				}
			}

			if (!waitingIDs.isEmpty()) {

				ID auctionID = model.getAuctionID();
				Collection bidderIDs = waitingIDs;
				if (!model.isAddAllWaiting()) {
					// Filter out bidders that are not waiting
					Collection selectedIDs = model.getSelectedIDList();
					bidderIDs = CollectionUtil.createIntersection(waitingIDs, selectedIDs);
				}
				if (model.isAddSerially()) {
					int rankBy = model.getRankBy();
					int thinkTime = model.getThinkTime();
					AuctionMH.startAuctionSerialization(bidderIDs, rankBy, thinkTime, auctionID);
				} else {
					Date deadline = model.getDeadline();
					AuctionMH.addShiftBidderToAuction(bidderIDs, deadline, auctionID);
				}
			}
		} catch (RmHardValidationException ex) {
			Message msg = model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
			msg.setAssociatedMessage(ex.getLocalizedMessage(context.getLocalizer(), context.getViewingTimeZone()));
			LOG.l7dDebug(RmWebLogBundleKey.UNABLE_TO_UPDATE_INFO, ex);
		} catch (Exception ex) {
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
			LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}

		model.setIsRefreshOpenerOnClose(true);
		model.setIsCloseAfterLoad(!model.hasError());
		context.setPageModel(model);
	}

	private static void processStopSerializedAdd(RequestContext context) {
		AuctionEmployeesPM model = new AuctionEmployeesPM(context);
		model.extractParameters(context.getRequest());

		ID auctionID = model.getSelectedIDObject();
		try {

			LOG.l7dDebug(RmWebLogBundleKey.UNABLE_TO_UPDATE_INFO, ex);
		} catch (Exception ex) {
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
			LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}

		context.setPageModel(model);
	}

	private static void processRemove(RequestContext context) {
		AuctionEmployeesPM model = new AuctionEmployeesPM(context);
		model.extractParameters(context.getRequest());

		ID auctionID = model.getSelectedIDObject();
		Collection waitingIDs = model.getWaitingIDs();
		Collection bidderIDs = model.getPerfromActionOnIDsAsCollection();
		// Filter out bidders that are waiting
		bidderIDs = CollectionUtil.createDifference(bidderIDs, waitingIDs);

		try {
			AuctionMH.removeShiftBidderFromAuction(bidderIDs, auctionID);
		} catch (RmHardValidationException ex) {
			Message msg = model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
			msg.setAssociatedMessage(ex.getLocalizedMessage(context.getLocalizer(), context.getViewingTimeZone()));
			LOG.l7dDebug(RmWebLogBundleKey.UNABLE_TO_UPDATE_INFO, ex);
		} catch (Exception ex) {
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
			LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}

		context.setPageModel(model);
	}

	/**
	 * Process Save Action
	 */
	private static void processSave(RequestContext context) {
		AuctionEmployeesPM model = new AuctionEmployeesPM(context);
		model.extractParameters(context.getRequest());
		ID auctionID = model.getSelectedIDObject();
		Map bonusMap = model.getBonusMap();
		Map deadlineMap = model.getDeadlineMap();

		try {
			AuctionMH.updateShiftBidders(auctionID, bonusMap, deadlineMap);
		} catch(RmHardValidationException ex) {
			Message msg = model.addPageMessage(Message.ERROR_TYPE,
					RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
			msg.setAssociatedMessage(ex.getLocalizedMessage(context.getLocalizer(), context.getViewingTimeZone()));
			LOG.l7dDebug(RmWebLogBundleKey.UNABLE_TO_UPDATE_INFO, ex);
		} catch(Exception ex) {
			model.addPageMessage(Message.ERROR_TYPE,
					RmWebBundleKey.UNABLE_TO_UPDATE_INFO,	RmWebBundleKey.BUNDLE_NAME);
			LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}

		context.setPageModel(model);
	}

	private static void processWithdrawApprovedBid(RequestContext context) {
		AuctionEmployeesPM model = new AuctionEmployeesPM(context);
		model.extractParameters(context.getRequest());
		try{
			ID auctionID = model.getSelectedIDObject();
			String sBidderID = context.getRequest().getParameter(WorkpaneFieldNames.PERFORM_ACTION_ON_ID_FN);

			ID approvedReqID = AuctionMH.getApprovedRequestID(new ID(sBidderID), auctionID);

			ShiftBidRequest request = null;

			if (approvedReqID != null) {
				request = (ShiftBidRequest)RmManagerFactory.getInstance().getShiftBidRequestManager().getRequestByID(approvedReqID,
						false, false, ShiftBidRequest.DL_BASIC | ShiftBidRequest.DL_BIDDABLESCHEDULE_IDS
		        		| ShiftBidRequest.DL_SHIFT_ASSIGNMENTS_COLL
		        		| RequestDetailLevel.DL_INCLUDE_ZERO_NUM_OF_AVAIL_INSTANCES
						);
			}
			if (request != null && request.getRequestStatus().equals(RequestAuditTrail.STATUS_APPROVED)){
				AuctionMH.withdrawApprovedBidRequest(request, null);
			} else {
				model.addPageMessage(Message.ERROR_TYPE,
						RmWebBundleKey.SHIFT_SWAP_CANNOT_WITHDRAW_APPROVED,	RmWebBundleKey.BUNDLE_NAME);
			}
		}catch(RmHardValidationException x) {
			model.addPageMessage(Message.ERROR_TYPE, x.getMessage());
		}catch (Exception ex){

			model.addPageMessage(Message.ERROR_TYPE,
					RmWebBundleKey.UNABLE_TO_UPDATE_INFO, RmWebBundleKey.BUNDLE_NAME);
			LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}

		context.setPageModel(model);
	}

    /**
	 * Return Page Model to perform generic action request on
	 */
	@Override
	public WorkpanePageModel getPageModel(RequestContext context) {
		AuctionEmployeesPM model = createAuctionEmployeesPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		return model;
	}

    /**
	 * Create AgentsRequestsPM or subclass
	 *
	 * NOTE: Meant to be overwritten by subclass
	 */
	protected AuctionEmployeesPM createAuctionEmployeesPM(RequestContext context) {
		return new AuctionEmployeesPM(context);
	}
