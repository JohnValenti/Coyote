package com.bluepumpkin.web.rm.auction;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.rm.Log;
import com.witness.web.uif.base.WebException;
import com.witness.web.uif.pagemodel.WorkpanePageModel;
import com.witness.web.uif.system.RequestContext;

/**
 * File:         MyBidOptionsRH
 * Description:  Request Handler for for the list of bid options for an employee
 * Copyright:    Copyright (c) 2004
 * Company:      Blue Pumpkin Software, inc
 * @author       Swati Tewari
 * @version      1.0
 * Created on June 23, 2004, 10:25 AM
 */
public class MyBidOptionsRH extends BaseBidOptionsRH {
	private static final Category LOG = Log.initCategory(MyBidOptionsRH.class.getName());

	/**
	 * Process Request actoins
	 */
	@Override
	public void processCustomRequest(RequestContext context) throws WebException {

		if (isAction(context, BaseBidOptionsPM.SORT_BY_START_ACTION)) {
			processSortBy(context, BaseBidOptionsPM.SORT_BY_START_ACTION);
		} else if (isAction(context, BaseBidOptionsPM.SORT_BY_END_ACTION)) {
			processSortBy(context, BaseBidOptionsPM.SORT_BY_END_ACTION);
		} else if (isAction(context, BaseBidOptionsPM.SORT_BY_LENGTH_ACTION)) {
			processSortBy(context, BaseBidOptionsPM.SORT_BY_LENGTH_ACTION);
		} else {
			super.processCustomRequest(context);
		}
	}
	
	private static void processSortBy(RequestContext context, String sortType) 
	{
		MyBidOptionsPM model = new MyBidOptionsPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);		
		model.setSortType(sortType);
	}
	

    /**
	 * Return Page Model to perform generic action request on
	 */
	@Override
	public WorkpanePageModel getPageModel(RequestContext context) {
        MyBidOptionsPM model =  new MyBidOptionsPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		return model;
	}
}
