/**
 * Rule in UI: Request complies with filing rules.
 *
 * Check that the request is submitted enough time in advance according to the
 * filing rules configured for the employee's organization.
 */
public class FilingRuleComplianceValidationRule implements Validator {

	private static final String CLASS_NAME = FilingRuleComplianceValidationRule.class.getName();

	private static final Category LOG = Log.initCategory(CLASS_NAME);

	/**
	 * Validate a shift swap request.
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		ShiftSwapRequest ssr = (ShiftSwapRequest) validatable;

		String methodName = "validate";
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		}

		// Build a filing rule validation checker for this request
		ID orgID = ssr.getCache().getOrgIDs().iterator().next();
		ShiftSwapWithdraw ssWithdraw = ssr.getWithdrawInfo();
		FilingRuleChecker ckr =FilingRuleCheckerHelper.getChecker(orgID,
				Request.REQUESTTYPE_SHIFTSWAP_WITHDRAW,
				null,
				ssWithdraw.getValidationTimeRange());

		result = setValidationResultForFilingRuleCompliance(ssr, ssWithdraw, ckr);
		if (LOG.isDebugEnabled()){
			LOG.debug(RmUtil.dumpExitMethod(methodName, result));
		}
		return result;
	}

	public ValidationResult setValidationResultForFilingRuleCompliance(ShiftSwapRequest ssr, ShiftSwapWithdraw ssWithdraw, FilingRuleChecker ckr) throws Exception {
		String methodName = "setValidationResultForFilingRuleCompliance";
		ValidationResult result = null;
		Date fileDate = ssWithdraw.getCreationDate();
		TimeRange timeRange =ssWithdraw.getValidationTimeRange();
		if (!ckr.requestComplies(fileDate, timeRange.getStartDate(), timeRange.getEndDate(),
				timeRange.getStartDate())) {
			ShiftSwapItem ssItem = ssWithdraw.getValidationItem();
			RequestFilingRule rule = ckr.getLastFailingRule();
			if (LOG!=null){
				LOG.debug("shift swap Withdraw request id:" + ssr.getID() + " failed because " +
						"ssWithdraw id:" + ssWithdraw.getID() + " did not pass filing rule with id:" + rule.getID());
			}

			result = ValidationUtil.setSoftValidationResult(ssr,
					//Time off message is good for swap message
					RmEjbBundleKey.APPROVED_TO_WITHDRAW_FAILED_FILING_RULE, ssr.getCache().getEmployeeNameByID(ssItem.getEmployeeID()),
					timeRange.getStartDate(), timeRange.getEndDate(), ssr.getCache().getOrgNameForMsg(),
					CLASS_NAME);
		}
		return result;
	}
}
