package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.lang.reflect.Method;
import java.util.HashMap;
import java.util.Map;

import org.springframework.core.annotation.AnnotationUtils;

import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;

import net.sf.cglib.proxy.Enhancer;
import net.sf.cglib.proxy.MethodInterceptor;
import net.sf.cglib.proxy.MethodProxy;

public class FTValidationCacheFactory {

	private static final int NO_PARAM_KEY = 0;

	private static final int NULL_PARAM_KEY = 53;

	private static final int MAGIC_NUM_31 = 31;

	private static final int MAGIC_NUM_17 = 17;

	private FTValidationCacheFactory() {
		// to hide public constructor
	}

	/**
	 * Create an instance of FTValidationCache with @Cacheable support.
	 *
	 * @param tor
	 * @return FTValidationCache instance
	 */
	public static FTValidationCache newFTValidationCache(TORequest tor) {
		return createProxyObject(FTValidationCache.class, tor);
	}

	@SuppressWarnings("unchecked")
	private static <T> T createProxyObject(Class<T> clazz, TORequest tor) {
		MethodInterceptor methodInterceptor = new ProxyObjectMethodImpl();
		Enhancer enhancer = new Enhancer();
		enhancer.setSuperclass(clazz);
		enhancer.setCallback(methodInterceptor);
		return (T) enhancer.create(new Class[]{TORequest.class}, new Object[]{tor});
	}

	private static class ProxyObjectMethodImpl implements MethodInterceptor {
		private Map<String, Map<Object, Object>> caches = new HashMap<String, Map<Object, Object>>();

		public Object intercept(Object proxy, Method method, Object[] args, MethodProxy methodProxy) throws Throwable {
			Cacheable methodData = AnnotationUtils.findAnnotation(method, Cacheable.class);
			if (methodData == null) {
				return methodProxy.invokeSuper(proxy, args);
			} else {
				Object key = generateKey(args);

				Map<Object, Object> cacheForMethod = caches.get(method.getName());
				if (cacheForMethod == null) {
					cacheForMethod = new HashMap<Object, Object>();
					caches.put(method.getName(), cacheForMethod);
				}

				if (!cacheForMethod.containsKey(key)) {
					Object value = methodProxy.invokeSuper(proxy, args);
					cacheForMethod.put(key, value);
				}
				return cacheForMethod.get(key);
			}
		}

		/**
		 * From org.springframework.cache.interceptor.DefaultKeyGenerator.
		 *
		 * @param params
		 * @return
		 */
		private Object generateKey(Object... params) {
			if (params.length == 1) {
				return params[0] == null ? NULL_PARAM_KEY : params[0];
			}
			if (params.length == 0) {
				return NO_PARAM_KEY;
			}
			int hashCode = MAGIC_NUM_17;
			for (Object object : params) {
				hashCode = MAGIC_NUM_31 * hashCode + (object == null ? NULL_PARAM_KEY : object.hashCode());
			}
			return Integer.valueOf(hashCode);
		}
	}
}
