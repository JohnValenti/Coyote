package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.Date;
import java.util.List;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexRequestMakeup;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;
import com.bluepumpkin.ejb.rm.setup.validation.ejb.ValidationRuleManager;

/**
 * Validate the makeup extensions must not violate org settings (Allow make up Requests before/after and in gaps)
 *
 */
public class MakeupExtensionsMustAdhereToOrgSettingsRule implements Validator{

	private static final String CLASS_NAME = MakeupExtensionsMustAdhereToOrgSettingsRule.class.getName();
	private String violationType;

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		Date violationDate = getFirstViolatedDate((TORequest)validatable);
		if(violationDate != null) {
			//Does violate the rule, use the setHardValidationResult to return the error message
			return ValidationUtil.setHardValidationResult((TORequest)validatable,
											violationType,
											violationDate,
											CLASS_NAME);
		}
		//The TORequest passes the rule, no violation
		return null;
	}

	private Date getFirstViolatedDate(TORequest toRequest) throws Exception {
		List<FlexRequestMakeup> flexRequestMakeupCache = toRequest.getFlexRequestMakeupList();

		if (flexRequestMakeupCache == null) {
			return null;
		}

		FTValidationCache cache = getFTValidationCache(toRequest);
		ID orgID = cache.getOrg().getID();
		OrganizationSetting orgSettings = cache.getOrgSettingForOrgID(orgID);

		for (FlexRequestMakeup flex: flexRequestMakeupCache) {
			if (!orgSettings.getAllowMakeupBeforeShift() &&
					flex.getExtBeforeDuration() > 0 && flex.getExtBeforeActivityID() != null) {
				violationType = RmEjbBundleKey.FLEX_REQ_MAKEUP_BEFORE_SHIFT_VIOLATES_ORG_SETTINGS;
				return flex.getShiftStartTime();
			}
			if (!orgSettings.getAllowMakeupAfterShift() &&
					flex.getExtAfterDuration() > 0 && flex.getExtAfterActivityID() != null) {
				violationType = RmEjbBundleKey.FLEX_REQ_MAKEUP_AFTER_SHIFT_VIOLATES_ORG_SETTINGS;
				return flex.getShiftStartTime();
			}
			if (!orgSettings.getAllowMakeupInGap() && flex.getFlexReqMakeupGaps() != null && !flex.getFlexReqMakeupGaps().isEmpty()) {
				violationType = RmEjbBundleKey.FLEX_REQ_MAKEUP_IN_GAP_VIOLATES_ORG_SETTINGS;
				return flex.getShiftStartTime();
			}
		}

		return null;
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}

	protected ValidationRuleManager getValidationRuleManager() throws BbmCreateException {
		return RmManagerFactory.getInstance().getValidationRuleManager();
	}
}
