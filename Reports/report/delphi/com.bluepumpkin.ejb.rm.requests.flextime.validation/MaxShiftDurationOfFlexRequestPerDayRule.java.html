/**
 * This class implement the validation rule: Maximum shift duration allowed ___
 * minutes per "Organization Day" or "24 Hour Period"
 *
 */
public class MaxShiftDurationOfFlexRequestPerDayRule implements Validator {
	private static final String CLASS_NAME = MaxShiftDurationOfFlexRequestPerDayRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	public static final String FT_MAX_SHIFT_DURATION_VALUE = "FT_MAX_SHIFT_DURATION_VALUE";
	public static final String FT_MAX_SHIFT_DURATION_UNIT = "FT_MAX_SHIFT_DURATION_UNIT";
	public static final int MAX_SHIFT_DURATION_UNIT_ORGHOO = 0;
	public static final int MAX_SHIFT_DURATION_UNIT_24HOURS = 1;
	public static final ID RULE_ID = new ID(-194074);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod("validate", validatable));
		}

		TORequest toRequest = (TORequest) validatable;
		ID orgID = toRequest.getCache().getOrg().getID();

		@SuppressWarnings("rawtypes")
		Map ruleParams = getValidationRuleManager().getValidationRuleParams(orgID, RULE_ID);
		int maxDuration = ((Integer) ruleParams.get(FT_MAX_SHIFT_DURATION_VALUE)).intValue();
		int unit = ((Integer) ruleParams.get(FT_MAX_SHIFT_DURATION_UNIT)).intValue();

		ValidationResult result;
		if (MAX_SHIFT_DURATION_UNIT_ORGHOO == unit) {
			result = verifyOrgDayShiftDuration(toRequest, maxDuration);
		} else {
			result = verify24HourShiftDuration(toRequest, maxDuration);
		}

		return result;
	}

	protected ValidationResult verifyOrgDayShiftDuration(TORequest toRequest, int maxDuration) throws Exception {
		ValidationResult result = null;
		TimeZone orgTZ = getTimeZone(toRequest);
		int orgDayBoundaryOffset = getOrg(toRequest).getDayBoundaryOffset();

		for (FlexRequestMakeup makeup : toRequest.getFlexRequestMakeupList()) {

			TimeRange orgDayBoundaryTimeRange = getOrgDayBoundaryWithOffsetTimeRange(makeup.getShiftStartTime(), orgTZ, orgDayBoundaryOffset);
			int totalNewPaidDuration = getPaidDurationOfShiftsInTimeRange(toRequest, orgDayBoundaryTimeRange, orgDayBoundaryTimeRange, true);

			if (totalNewPaidDuration > maxDuration) {
				int totalCurrentPaidDuration = getPaidDurationOfShiftsInTimeRange(toRequest, orgDayBoundaryTimeRange, orgDayBoundaryTimeRange, false);

				if (totalNewPaidDuration > totalCurrentPaidDuration) {
					result = ValidationUtil.setSoftValidationResult(toRequest, RmEjbBundleKey.FLEX_REQ_MAX_SHIFT_DURATION_EXCEED_ORG_DAY, totalNewPaidDuration, String.valueOf(maxDuration),
							CLASS_NAME);
					break;
				}
			}
		}

		return result;
	}

	protected ValidationResult verify24HourShiftDuration(TORequest toRequest, int maxDuration) throws Exception {
		ValidationResult result = null;
		TimeZone orgTZ = getTimeZone(toRequest);
		Calendar startCal = Calendar.getInstance(orgTZ);
		Calendar iterCal = Calendar.getInstance(orgTZ);

		for (FlexRequestMakeup makeup : toRequest.getFlexRequestMakeupList()) {
			TimeRange timeRange24Hrs = get24HoursTimeRange(makeup.getShiftStartTime(), makeup.getShiftEndTime(), orgTZ);

			Collection<ShiftAssignment> shiftAssignments = getPublishedShiftAssignments(toRequest, timeRange24Hrs);

			TimeRange impactedRange = getPeriodRange(shiftAssignments, makeup);
			startCal.setTime(impactedRange.getStartDate());
			startCal.add(Calendar.DAY_OF_MONTH, -1);
			iterCal.setTime(startCal.getTime());
			iterCal.add(Calendar.DAY_OF_MONTH, 1);
			Date endPeriod = impactedRange.getEndDate();

			//Build largest time ranges to query all shift assignments in that range, so that, the result will be cached
			TimeRange shiftsTimeRange = new TimeRange(startCal.getTime(), addHours(orgTZ, endPeriod, 24));

			while (startCal.getTime().before(endPeriod)) {

				TimeRange iterTimeRange24Hrs = new TimeRange(startCal.getTime(), iterCal.getTime());
				int totalNewPaidDuration = getPaidDurationOfShiftsInTimeRange(toRequest, shiftsTimeRange, iterTimeRange24Hrs, true);

				if (totalNewPaidDuration > maxDuration) {
					int totalCurrentPaidDuration = getPaidDurationOfShiftsInTimeRange(toRequest, shiftsTimeRange, iterTimeRange24Hrs, false);

					if (totalNewPaidDuration > totalCurrentPaidDuration) {
						result = ValidationUtil.setSoftValidationResult(toRequest, RmEjbBundleKey.FLEX_REQ_MAX_SHIFT_DURATION_EXCEED_24HRS, totalNewPaidDuration, String.valueOf(maxDuration),
								CLASS_NAME);
						break;
					}
				}

				startCal.add(Calendar.MINUTE, 15);
				iterCal.add(Calendar.MINUTE, 15);
			}

			if (result != null) {
				break;
			}
		}

		return result;
	}

	private int getPaidDurationOfShiftsInTimeRange(TORequest toRequest, TimeRange shiftsTimeRange, TimeRange boundaryTimeRange, boolean countCurrentRequest) throws Exception {
		int totalPaidDuration = 0;
		Collection<ShiftAssignment> shiftAssignments = getPublishedShiftAssignments(toRequest, shiftsTimeRange);
		Collection<TimeRange> nonTimeOffRanges = getNonTimeOffRanges(toRequest, boundaryTimeRange, shiftsTimeRange, countCurrentRequest);

		for (TimeRange duration : nonTimeOffRanges) {
			// paid duration of existing shifts + makeup duration
			totalPaidDuration += getPaidDurationOfShiftsInTimeRange(toRequest, shiftAssignments, duration) + (countCurrentRequest ? getDurationOfMakeupsWithinTimeRange(toRequest, duration) : 0);
		}

		return totalPaidDuration;
	}

	@SuppressWarnings("unchecked")
	private Collection<TimeRange> getNonTimeOffRanges(TORequest toRequest, TimeRange boundaryTimeRange, TimeRange shiftsTimeRange, boolean countCurrentRequest) throws Exception {
		Collection<TimeRange> nonTimeOffRanges = new ArrayList<TimeRange>();
		if (countCurrentRequest) {
			nonTimeOffRanges.addAll(boundaryTimeRange.splitOrTrim(new TimeRange(toRequest.getFirstTOChoice().getStartDate(), toRequest.getFirstTOChoice().getEndDate())));
		}
		else {
			nonTimeOffRanges.add(boundaryTimeRange);
		}

		Collection<CalendarEventAssignment> calEvents = getPublishedCalendarEvents(toRequest, shiftsTimeRange);

		for (CalendarEventAssignment calEvent : calEvents) {
			if (isUnpaidCalendarEvent(toRequest, calEvent)) {
				Collection<TimeRange> newNonTimeOffRanges = new ArrayList<TimeRange>();
				for (TimeRange duration : nonTimeOffRanges) {
					newNonTimeOffRanges.addAll(duration.splitOrTrim(new TimeRange(calEvent.getStartTime(), calEvent.getEndTime())));
				}
				nonTimeOffRanges = newNonTimeOffRanges;
			}
		}

		return nonTimeOffRanges;
	}

	private Date addHours(TimeZone orgTZ, Date date, int hours) {
		Calendar cal = Calendar.getInstance(orgTZ);
		cal.setTime(date);
		cal.add(Calendar.HOUR, hours);
		return cal.getTime();
	}

	/**
	 * Get PAID duration in MINUTES of shifts within the org time range
	 *
	 * @param cusShiftCache
	 * @param shiftAssignments
	 * @param boundaryTimeRange
	 * @return PAID duration in MINUTES of shifts within the org time range
	 * @throws BbmException
	 * @throws RemoteException
	 * @throws BbmFinderException
	 * @throws BbmObjectNotFoundException
	 */
	private int getPaidDurationOfShiftsInTimeRange(TORequest toRequest, Collection<ShiftAssignment> shiftAssignments, TimeRange boundaryTimeRange) throws Exception {
		// Create custom shift Cache from a dummy custom shift
		CustShiftValidationCache cusShiftCache = new CustShiftValidationCache(new CustShiftReq(RequestDetailLevel.DL_BASIC));
		int paidDuration = 0;
		for (ShiftAssignment shiftAssignment : shiftAssignments) {
			if (TimePeriodUtil.overlap(shiftAssignment.getStartTime(), shiftAssignment.getEndTime(), boundaryTimeRange.getStartDate(), boundaryTimeRange.getEndDate())) {
				paidDuration += cusShiftCache.getPaidDurationForSegment(shiftAssignment, boundaryTimeRange.getStartDate(), boundaryTimeRange.getEndDate());
			}
		}

		return paidDuration;
	}

	private boolean isUnpaidCalendarEvent(TORequest tor, CalendarEventAssignment event) throws Exception {
		if (event.getEventType() == Event.EVENT_TYPE_TIME_OFF
				|| event.getEventType() == Event.EVENT_TYPE_UNAVAILABILITY) {
			return true;
		} else if (event.getEventType() == Event.EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT) {
			Activity activity = getActivityByID(tor, event.getActivityID());
			if (activity != null && !activity.isPaid()) {
				return true;
			}
		}

		return false;
	}

	/**
	 * Get makeup duration in MINUTES within a time range
	 *
	 * @param toRequest
	 * @param timeRange
	 * @return duration in MINUTES
	 */
	private int getDurationOfMakeupsWithinTimeRange(TORequest toRequest, TimeRange timeRange) {
		long makeupDurationInTimeRange = 0;

		for(FlexRequestMakeup makeup: toRequest.getFlexRequestMakeupList()) {
			// The Before Extension
			TimeRange extBeforeTimeRange = makeup.getExtBeforeTimeRange();
			if (extBeforeTimeRange != null) {
				makeupDurationInTimeRange = timeRange.getOverlapDuration(extBeforeTimeRange);
			}

			// The After Extension
			TimeRange extAfterTimeRange = makeup.getExtAfterTimeRange();
			if (extAfterTimeRange != null) {
				makeupDurationInTimeRange += timeRange.getOverlapDuration(extAfterTimeRange);
			}

			for (FlexReqMakeupGap gap : makeup.getFlexReqMakeupGaps()) {
				makeupDurationInTimeRange += timeRange.getOverlapDuration(gap.getMakeupTimeRange());
			}
		}

		return (int) makeupDurationInTimeRange / (1000 * 60);
	}

	private TimeRange getOrgDayBoundaryWithOffsetTimeRange(Date shiftStartTime, TimeZone orgTZ,
			int orgDayBoundaryOffsetInMins) {
		Calendar startCal = Calendar.getInstance(orgTZ);
		startCal.setTime(shiftStartTime);
		startCal.set(Calendar.HOUR_OF_DAY, orgDayBoundaryOffsetInMins / 60);
		startCal.set(Calendar.MINUTE, orgDayBoundaryOffsetInMins % 60);
		startCal.set(Calendar.SECOND, 0);
		startCal.add(Calendar.MILLISECOND, 0);

		if (startCal.getTime().after(shiftStartTime)) {
			startCal.add(Calendar.DAY_OF_MONTH, -1);
		}

		Calendar endCal = Calendar.getInstance(orgTZ);
		endCal.setTime(startCal.getTime());
		endCal.add(Calendar.DAY_OF_MONTH, 1);
		return new TimeRange(startCal.getTime(), endCal.getTime());
	}

	private TimeRange get24HoursTimeRange(Date shiftStartTime, Date endStartTime, TimeZone orgTZ) {
		Calendar startCal = Calendar.getInstance(orgTZ);
		startCal.setTime(shiftStartTime);
		startCal.set(Calendar.HOUR_OF_DAY, 0);
		startCal.set(Calendar.MINUTE, 0);
		startCal.set(Calendar.SECOND, 0);
		startCal.add(Calendar.DAY_OF_MONTH, -1);

		Calendar endCal = Calendar.getInstance(orgTZ);
		endCal.setTime(endStartTime);
		endCal.add(Calendar.DAY_OF_MONTH, 1);
		return new TimeRange(startCal.getTime(), endCal.getTime());
	}

	private TimeRange getPeriodRange(Collection<ShiftAssignment> saColl, FlexRequestMakeup makeup) {

		TimeRange makeupRange = new TimeRange(makeup.getStartTime(), makeup.getEndTime());

		for (ShiftAssignment sAssgn : saColl) {
			TimeRange sAssgnRange = new TimeRange(sAssgn.getStartTime(), sAssgn.getEndTime());
			int adjacency = sAssgnRange.getAdjacency(makeupRange, 0);
			long overlapDuration = sAssgnRange.getOverlapDuration(makeupRange);
			if (adjacency == TimeRange.ADJACENT_BEFORE || adjacency == TimeRange.ADJACENT_AFTER || overlapDuration > 0) {
				return new TimeRange(DateUtil.minOfTwoDates(makeupRange.getStartDate(), sAssgnRange.getStartDate()), DateUtil.maxOfTwoDates(makeupRange.getEndDate(), sAssgnRange.getEndDate()));
			}
		}
		return makeupRange;
	}

	protected ShiftAssignment getNewShiftAssignToBeCreated(CustShiftValidationCache cusShiftCache,
			CustShiftReq custShiftReq) throws Exception {
		return cusShiftCache.getNewShiftAssignToBeCreated(custShiftReq, cusShiftCache.getScheduleAccessManager());
	}

	protected Collection<ShiftAssignment> getPublishedShiftAssignments(TORequest toRequest, TimeRange duration)
			throws Exception {
		return getFTValidationCache(toRequest).getPublishedShiftAssignments(duration);
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}

	protected ValidationRuleManager getValidationRuleManager() throws BbmCreateException {
		return RmManagerFactory.getInstance().getValidationRuleManager();
	}

	protected TimeZone getTimeZone(TORequest tor) throws Exception {
		return getFTValidationCache(tor).getOrg().getTimeZone();
	}

	protected Organization getOrg(TORequest tor) throws Exception {
		return getFTValidationCache(tor).getOrg();
	}

	protected Collection<CalendarEventAssignment> getPublishedCalendarEvents(TORequest tor, TimeRange eventsDuration)
			throws Exception {
		return getFTValidationCache(tor).getPublishedCalendarEvents(eventsDuration);
	}

	public Activity getActivityByID(TORequest tor, ID activityID) throws Exception {
		return getFTValidationCache(tor).getActivityByID(activityID);
	}
}
