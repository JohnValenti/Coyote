package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

/**
 * Title:        TimeOffBidRequestNotOverlapValidationRule
 * Description:  Validate a time off bid request to make sure it does not overlap with any approved time of bid request choice.
 * 				 The overlapping here is when the new request choice covers or overlapped partially with approved TO request choice.
 * 				 The case of new request choice within approved choice does not count here because in this case, it violates the 
 * 				  hard validation rule "WorkableTimeValidationRule" 
 * 				 This validation is invoked only if there is Advanced RM license
 * Copyright:    Copyright (c) 2016
 * Company:      Verint System, Inc
 * @author       Truc Pham
 * @version      1.0
**/
import java.util.Collection;
import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.localization.DefaultLocalizationManager;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.ejb.TimeOffBidUtil;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBid;

public class TimeOffBidRequestNotOverlapValidationRule implements Validator {
	private static final String clsName = TimeOffBidRequestNotOverlapValidationRule.class.getName();

	@SuppressWarnings("rawtypes")
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		TORequest toReq = (TORequest) validatable;
		ID timeOffType = toReq.getTimeOffType();
		ID employeeId = toReq.getEmployeeID();
		TimeOffBid bid = getOpenTimeOffBidForActivityIdAndEmployeId(timeOffType, employeeId, toReq.getSubmittedOn());
		if (bid != null) {
			long bidRequestStart = bid.getRequestStartTime().getTime();
			long bidRequestEnd = bid.getRequestEndTime().getTime();
			Collection<ID> activitiesIdInBid = getActivityIDsByBidId(bid.getId());
			Collection<TOChoice> approvedChoices = getApprovedTimeOffRequestChoiceDuringOpenBid(bid.getId(), employeeId, activitiesIdInBid);
			for (Iterator it = toReq.getRequestChoiceList().iterator(); it.hasNext();) {
				TOChoice choice = (TOChoice) it.next();
				long choiceStart = choice.getStartDate().getTime();
				long choiceEnd = choice.getEndDate().getTime();
				boolean choicePeriodInBidRequestPeriod = bidRequestStart <= choiceStart && choiceEnd <= bidRequestEnd;
				TimeRange toChoiceRange = new TimeRange(choice.getStartDate(), choice.getEndDate());
				TimeRange approveChoiceRange = getApprovedChoiceRangeOverlapWithTOChoice(toChoiceRange, approvedChoices);
				if (choicePeriodInBidRequestPeriod && approveChoiceRange != null) {
					result = ValidationUtil.setSoftValidationResult(choice,
							RmEjbBundleKey.TIMEOFF_BID_VALIDATION_TOCHOICE_OVERLAP_APPROVED_REQUESTS, getEmployeeName(toReq),
							choice.getStartDate(), choice.getEndDate(), approveChoiceRange.getStartDate(),
							approveChoiceRange.getEndDate(), clsName);
				}
			}
		}
		return result;
	}

	private TimeRange getApprovedChoiceRangeOverlapWithTOChoice(TimeRange toChoiceRange, Collection<TOChoice> approvedChoices) {
		TimeRange resultRange = null;
		boolean isOverlapped = false;
		for (Iterator<TOChoice> it = approvedChoices.iterator(); it.hasNext();) {
			TOChoice apprChoice = it.next();
			TimeRange approveChoiceRange = new TimeRange(apprChoice.getStartDate(), apprChoice.getEndDate());
			boolean isCover = approveChoiceRange.isInside(toChoiceRange);
			//We dont count toChoiceRange.isInside(approveChoiceRange) since in this case, 
			//The given range is completely contained within a block of scheduled time off 
			boolean toChoiceOverlapStart= toChoiceRange.getStartDate().before(apprChoice.getStartDate())
					&& apprChoice.getStartDate().before(toChoiceRange.getEndDate());
			boolean toChoiceOverlapEnd= toChoiceRange.getStartDate().before(apprChoice.getEndDate())
					&& apprChoice.getEndDate().before(toChoiceRange.getEndDate());
			isOverlapped = isCover || toChoiceOverlapStart || toChoiceOverlapEnd;
			if (isOverlapped) {
				resultRange = approveChoiceRange;
				break;
			}
		}
		return resultRange;
	}

	protected TimeOffBid getOpenTimeOffBidForActivityIdAndEmployeId(ID timeOffType, ID employeeId, Date submittedDate) throws Exception {
		return TimeOffBidUtil.getOpenTimeOffBidForActivityIdAndEmployeId(timeOffType, employeeId, submittedDate);
	}

	protected Collection<TOChoice> getApprovedTimeOffRequestChoiceDuringOpenBid(ID bidId, ID employeeId, Collection<ID> activityIds) throws Exception {
		return TimeOffBidUtil.getApprovedTimeOffRequestChoiceDuringOpenBid(bidId, employeeId, activityIds);
	}
	public Collection<ID> getActivityIDsByBidId(ID bidId) throws Exception{
		return TimeOffBidUtil.getActivityIDsByBidId(bidId);
	}
	protected String getEmployeeName(TORequest toReq) throws Exception {
		return toReq.getCache().getEmployeeNameByID();
	}

	protected Localizer getLocalizer() {
		return DefaultLocalizationManager.getDefaultInstance().getLocalizer();
	}
}
