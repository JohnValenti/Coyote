package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.Date;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.FilingRuleChecker;

import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResultAggregator;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;

/**
 * Rule in UI: Request complies with filing rules.
 * <p/>
 * Check that the rquest is submitted enough time in advance according to the
 * filing rules configured for the employee's organization.
 * <p/>
 * <p/>
 * Applies equally for debit, debit-if and dont-debit requests
 * <p/>
 * <p/>
 * Applies equally for tentative and non tentative requests.
 */
public class FilingRuleComplianceValidationRule implements Validator {

    private static final String m_className = FilingRuleComplianceValidationRule.class.getName();

    private static final Category m_cat = Log.initCategory(m_className);

    /**
     * Validate a time off request.
     */
    @Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResultAggregator result = new ValidationResultAggregator();
        TORequest tor = (TORequest) validatable;

        String methodName = "validate";
        if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		}

        // Build a filing rule validation checker for this request
        FilingRuleChecker ckr = tor.getCache().getFilingRuleChecker(null);

        // Loop through the time off choices and check each one against the
        // filing rules
        // Check against blackout days
        // Check to see if the blackout days timemap is already in the cache
		for (TOChoice toChoice : tor.getTOChoicesAdjustedForValidation()) {
			//For TOB groupApproval,if the choice violates any validation rule ran before , do not run the rule and continue with next choice
			if(tor.isRunForGroupTOBApproval()&& !toChoice.getValidationResults(true).isEmpty()){
				continue;
			}
			ckr.resetLastFailingRule();
			result.add(setValidationResultForFilingRuleCompliance(tor, toChoice, ckr));
        }

        if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
		}
		return result.get();
    }

    public ValidationResult setValidationResultForFilingRuleCompliance(TORequest tor, TOChoice choice, FilingRuleChecker ckr) throws Exception {
        String methodName = "setValidationResultForFilingRuleCompliance";
        ValidationResult result = null;
        Date fileDate = tor.getSubmittedOn();
        if (!ckr.requestComplies(fileDate, choice.getStartDate(), choice.getEndDate(),
                choice.getStartDate())) {
            // Since sometimes it might be mystifying why a request fails, we log
            // the rule that failed the requst
            RequestFilingRule rule = ckr.getLastFailingRule();
            m_cat.debug("time off request id:" + tor.getID() + " failed because " +
                    "choice id:" + choice.getID() + " did not pass filing rule with id:" + rule.getID());

            result = ValidationUtil.setSoftValidationResult(choice,
                    RmEjbBundleKey.TO_FAILED_FILING_RULE, tor.getCache().getEmployeeNameByID(),
                    choice.getStartDate(), choice.getEndDate(), tor.getCache().getOrgNameForMsg(),
                    m_className);
        }
        return result;
    }
}
