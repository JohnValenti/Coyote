package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.Iterator;
import java.util.List;

import com.bluepumpkin.ejb.bbm.timeoffcalculator.DailyHoursBuckets;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;

/**
 * @author rrajendran
 *         <p/>
 *         Raises an alert if any of the TOChoices have zero length.
 *         <p/>
 *         <p> Note: This validation is invoked only if the detailLevel is {@link TORequest#DL_TIMEOFF_CHOICES_LENGTH DL_TIMEOFF_CHOICES_LENGTH}.
 *         See {@link com.bluepumpkin.ejb.rm.requests.timeoff.ejb.TORequestManagerEJB#m_softValidatorDescMandArr m_softValidatorDescMandArr}
 *         for further details.
 */
public class TimeOffChoiceHasZeroLength implements Validator {

	private static String m_className = TimeOffChoiceHasZeroLength.class.getName();

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		TORequest toReq = (TORequest) validatable;
		// run this validation only if the detail level is right.
		boolean choicesLenDetLevel = toReq.isDetailLevelSet(TORequest.DL_TIMEOFF_CHOICES_LENGTH);
		if (!choicesLenDetLevel) {
			return null;
		}
		//QC 45226
		//Do not validate if TORequest not of type "debit" since it will map to an
		// unavailability event which can never have underlying shift ;
		// hence length is always zero
		if (toReq.isUnavailability()) {
			return null;
		}
		// compute the length for TOChoices
		TORequestUtil.fillTOChoicesLengthIfNecess(toReq, false);
		// iterate over each choice
		ValidationResult result = null;
		for (TOChoice toChoice : toReq.getRequestChoiceList()) {
			//For TOB groupApproval,if the choice violates any validation rule ran before , do not run the rule and continue with next choice
			if(toReq.isRunForGroupTOBApproval()&& !toChoice.getValidationResults(true).isEmpty()){
				continue;
			}
			result = setValidationResultAsNeeded(toChoice);
			if (result != null) {
				break;//no point in going ahead from here as the validation is already failed.
			}
		}
		return result;
	}

	public ValidationResult setValidationResultAsNeeded(TOChoice toChoice) {
		ValidationResult result = null;
		if (toChoice.getLength() == 0.0) {
			int periodFlag = toChoice.getPeriodFlag();
			switch (periodFlag) {
				case DailyHoursBuckets.DAILYHOURSBUCKETS_PERIOD_SCHED:
					//JT
					if(toChoice.getIsInUnpublishedGap()){
						// Same as Unpublished. The choice falls into the unpublished gap between published and last published
						result = ValidationUtil.setSoftValidationResult(toChoice,
						        RmEjbBundleKey.TO_CHOICE_HAS_ZERO_LENGTH_UNSCHED,
						        toChoice.getStartDate(), toChoice.getEndDate(), m_className);
					} else{
					result = ValidationUtil.setSoftValidationResult(toChoice,
					        RmEjbBundleKey.TO_CHOICE_HAS_ZERO_LENGTH_SCHED,
					        toChoice.getStartDate(), toChoice.getEndDate(), m_className);
					}
					break;

				case DailyHoursBuckets.DAILYHOURSBUCKETS_PERIOD_OVERLAP:
					result = ValidationUtil.setSoftValidationResult(toChoice,
					        RmEjbBundleKey.TO_CHOICE_HAS_ZERO_LENGTH_OVERLAP,
					        toChoice.getStartDate(), toChoice.getEndDate(), m_className);

					break;

				case DailyHoursBuckets.DAILYHOURSBUCKETS_PERIOD_UNSCHED:
					result = ValidationUtil.setSoftValidationResult(toChoice,
					        RmEjbBundleKey.TO_CHOICE_HAS_ZERO_LENGTH_UNSCHED,
					        toChoice.getStartDate(), toChoice.getEndDate(), m_className);

					break;

				default: // shouldn't get here, but if we do, use most generic message
					result = ValidationUtil.setSoftValidationResult(toChoice,
					        RmEjbBundleKey.TO_CHOICE_HAS_ZERO_LENGTH_OVERLAP,
					        toChoice.getStartDate(), toChoice.getEndDate(), m_className);
			}
		}
		return result;
	}


}
