package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.validation.ejb.ValidationRuleManager;
import com.bluepumpkin.ejb.rm.util.DateUtil;
import com.bluepumpkin.ejb.rm.util.RmUtil;

public class MinimumRequestDurationValidationRule implements Validator {
	public static final String MIN_REQUEST_DURATION_VALUE = "MIN_REQUEST_DURATION_VALUE";

	public static final String MIN_REQUEST_DURATION_UNIT = "MIN_REQUEST_DURATION_UNIT";

	public static final ID RULE_ID = new ID(-194008);

	private static final String m_className = MinimumRequestDurationValidationRule.class.getName();
	String methodName = "validate";
	private static final Category m_cat = Log.initCategory(m_className);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		TORequest tor = (TORequest) validatable;
		//The rule is not applicable for Anytime VTO requests
		if(tor.isAdvancedVTORequest()) {
			return null;
		}
		ValidationResult result = null;
		ID orgID = null;
		TOValidationCache valCache = tor.getCache();
		ID activityID = tor.getRequestSubType();
		for (TOChoice choice : tor.getRequestChoiceList()) {
			//For TOB groupApproval,if the choice violates any validation rule ran before , do not run the rule and continue with next choice
			if(tor.isRunForGroupTOBApproval()&& !choice.getValidationResults(true).isEmpty()){
				continue;
			}
			orgID = ValidationUtil.getOrgIDForEmployeeDuringPeriod(
					tor.getEmployeeID(), choice.getStartDate(),
					choice.getEndDate());
			if (!valCache.isValidationRequiredForOrgAndActivity(orgID,
					"time-off", m_className, activityID)) {
				continue;
			}

			ValidationRuleManager valMan = RmManagerFactory.getInstance()
					.getValidationRuleManager();
			Map<String, Integer> ruleParams = valMan.getValidationRuleParams(
					orgID, RULE_ID, activityID);
			int duration = ruleParams.get(MIN_REQUEST_DURATION_VALUE);
			int unit = ruleParams.get(MIN_REQUEST_DURATION_UNIT);
			boolean violationFound = false;
			if (unit == 2) {
				violationFound = DateUtil.compareDayDiffWithAllowedCount(
						valCache.getOrg().getTimeZone(), choice.getStartDate(),
						choice.getEndDate(), duration) < 0;
			} else {
				long valDuration = 0;
				if (unit == 1) {
					valDuration = duration * DateUtil.HOUR_IN_MSECS;
				} else {
					valDuration = duration * DateUtil.MINUTE_IN_MSECS;
				}
				long choiceDuration = choice.getEndDate().getTime()
						- choice.getStartDate().getTime();
				violationFound = choiceDuration < valDuration;
			}
			if (violationFound) {
				result = ValidationUtil
						.setSoftValidationResult(choice,
								RmEjbBundleKey.TO_CHOICE_MIN_DURATION,
								choice.getStartDate(), choice.getEndDate(),
								m_className);
			}

		}
		m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
		return result;
	}
}
