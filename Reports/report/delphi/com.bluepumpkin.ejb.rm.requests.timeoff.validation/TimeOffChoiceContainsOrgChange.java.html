package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.Log;

/**
 * @author rrajendran
 * 
 * Raises an alert if any of the TOChoices contains an org change.
 * 
 * <p> Note: This validation is invoked only if the detailLevel is {@link TORequest#DL_TIMEOFF_CHOICES_LENGTH}.
 * See {@link com.bluepumpkin.ejb.rm.requests.timeoff.ejb.TORequestManagerEJB#m_softValidatorDescMandArr m_softValidatorDescMandArr}
 * for further details.
 */
public class TimeOffChoiceContainsOrgChange implements Validator {
    
    private static String m_className = TimeOffChoiceContainsOrgChange.class.getName();
    private static Category m_cat = Log.initCategory(m_className);
    
    /**
     * Validate a time off request.  Ensure that the request is for time off
     * when the employee might be working.
     */
    public ValidationResult validate(Validatable validatable) throws Exception
    {
         String methodName = "validate";
         if (m_cat.isDebugEnabled()){
        	 m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));
         }

         TORequest tor = (TORequest)validatable;

         // Check against the hours of operation for the employee's organization
         // in the date range
         ValidationResult result = checkForOrgChange(tor);

         if (m_cat.isDebugEnabled()){
        	 m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
         }

         return result;
    }

    /**
     * Check the choices in the time off request against the non-working
     * holidays specified for the organization.
     */
    protected ValidationResult checkForOrgChange(TORequest tor) throws Exception {
       ValidationResult result = null;

       // Loop through the choices and check whether any of them has multiple work resource assignments
       for (Iterator it = tor.getRequestChoiceList().iterator(); it.hasNext(); ) {
         TOChoice c = (TOChoice)it.next();

         Collection wraColl = ValidationUtil.getWorkResAssnWithTZForWRID(tor.getEmployeeID(),
        		 c.getStartDate(), c.getEndDate());
         if (wraColl != null && wraColl.size() > 1) {
             // The given range contains org assignment change.  
             result = ValidationUtil.setSoftValidationResult(c,
                 RmEjbBundleKey.TO_CHOICE_CONTAINS_ORG_CHANGE, c.getStartDate(), 
				 		c.getEndDate(), m_className);
         }
       } // end while (it.hasNext())

       return result;
    }
}
