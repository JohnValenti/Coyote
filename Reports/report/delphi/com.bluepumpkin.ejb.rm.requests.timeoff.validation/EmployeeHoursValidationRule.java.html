package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.localization.DefaultLocalizationManager;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidatorForMultiChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.requests.timeoff.validation.employeehours.EmployeeHoursValidationRuleMultiChoice;

/**
 * Rule in UI: Employee has enough time off hours.
 * <p/>
 * For each time off year during the requested period,
 * <ul>
 * <li>Get the employee total time off allotment for the given time off type.
 * </li>
 * <li>Get time off events of the given type for the employee.</li>
 * <li>Subtract scheduled time off from the total allotment.</li>
 * <li>Check if there is enough left for the requested time off.</li>
 * </ul>
 * <p/>
 * <p/>
 * Applies only for debit and "debit-if" requests. Doesn't apply for
 * "don't debit" requests as they do not have the "time off length" property.
 * <p/>
 * <p/>
 */

/*
 * The  logic has been moved to EmployeeHoursValidationRuleMultiChoice. 
 * This class is kept as a shell so that it is easy to verify the old logic in git
 * When the logic is stable EmployeeHoursValidationRuleMultiChoice can be merged back into this class 
 * and  EmployeeHoursValidationRuleMultiChoice can be removed. 
 */

public class EmployeeHoursValidationRule implements ValidatorForMultiChoice   {

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		return validateForMultiChoice(validatable);
	}

	@Override
	public ValidationResult validateForMultiChoice(Validatable validatable) throws Exception {
		TORequest tor = (TORequest) validatable;
		EmployeeHoursValidationRuleMultiChoice multiValidator = new EmployeeHoursValidationRuleMultiChoice(getLocalizer());
		return multiValidator.validateForMultiChoice(tor);
	}

	protected Localizer getLocalizer() {
		return DefaultLocalizationManager.getDefaultInstance().getLocalizer();
	}

}
