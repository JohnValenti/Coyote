package com.verint.web.rm.rest.entity.shiftbidding;

import java.util.ArrayList;
import java.util.Date;
import java.util.List;

import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rest.util.FormatUtils;

/**
 * Represents a Shift.
 */
public class ShiftEntity extends AbstractWFOEntity {

	private String endDate;
	private String startDate;
	private final List<ShiftEventEntity> shiftEvents = new ArrayList<ShiftEventEntity>();

	public void setStartDate(Date startDate) {
		this.startDate = FormatUtils.dateToISO8601SecPrecisionString(startDate);
	}

	public void setEndDate(Date endDate) {
		this.endDate = FormatUtils.dateToISO8601SecPrecisionString(endDate);
	}

	public String getEndDate() {
		return endDate;
	}

	public String getStartDate() {
		return startDate;
	}

	public List<ShiftEventEntity> getShiftEvents() {
		return shiftEvents;
	}
}
