package com.verint.web.rm.rest.service;

import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.rm.Log;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.security.WFOSecurity;
import com.verint.web.rest.service.AbstractWFOService;
import com.verint.web.rm.rest.entity.validation.ValidationRulesResponseEntity;
import com.verint.web.rm.rest.facade.validation.IValidationFacade;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.web.uif.system.RequestContext;

/**
 * REST service for RM filing rules. This is being filled in as features are necessary.
 */
@Path("/validation/")
@Produces("application/json")
@WFOSecurity(licenses = {})
public class ValidationRestService extends AbstractWFOService {
	private static final Category LOG = Log.initCategory(ValidationRestService.class.getName());

	@Context
	private HttpServletRequest request;
	@Context
	private HttpServletResponse response;
	@Context
	private ServletContext sc;

	private IValidationFacade validationFacade;

	public ValidationRestService() {
	}

	public RequestContext getRequestContext() {
		return new RequestContext(request, response, sc);
	}

	public IValidationFacade getValidationFacade() {
		return validationFacade;
	}

	public void setValidationFacade(IValidationFacade validationFacade) {
		this.validationFacade = validationFacade;
	}

	/**
	 * Gets the RM validation settings for the employee's organization.
	 *
	 * Example:
	 * http://localhost:7001/wfo/rest/rm-api/filing-rules/get
	 */
	@GET
	@Path("/get")
	public ValidationRulesResponseEntity getFilingRulesForOrg() throws WFOException {
		try {
			RequestContext context = getRequestContext();

			ID empID = context.getUser().getEmployeeID();

			return validationFacade.getValidationSettingsForEmployee(context, empID);
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}
}
