package com.verint.web.rm.rest.service;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;

import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.PathParam;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;
import javax.ws.rs.core.Response;
import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.web.rm.Log;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.SystemWFOException;

import com.verint.web.rest.service.AbstractWFOEntityService;
import com.verint.web.rm.rest.entity.timeoff.TimeOffDataEntity;
import com.verint.web.rm.rest.facade.timeoff.ITimeOffDataFacade;
import com.witness.ejb.core.security.model.User;
import com.witness.web.uif.system.RequestContext;

@Path("/TOData/")
@Produces("application/json")
@WFOSecurity
public class TimeOffDataRestService extends AbstractWFOEntityService<TimeOffDataEntity> {
	private static final Category logger = Log.initCategory(TimeOffDataRestService.class.getName());

	@Context
	private HttpServletRequest request;
	@Context
	private HttpServletResponse response;
	@Context
	private ServletContext sc;

	public ITimeOffDataFacade timeOffDataFacade;

	public RequestContext getRequestContext() {
		return new RequestContext(request, response, sc );
	}

	public TimeOffDataRestService() { // NOSONAR
	}

	public ITimeOffDataFacade getTimeOffDataFacade() {
		return timeOffDataFacade;
	}

	public void setTimeOffDataFacade(ITimeOffDataFacade timeOffDataFacade) {
		this.timeOffDataFacade = timeOffDataFacade;
	}

	@GET
	@Path("/get")
	public TimeOffDataEntity getTODataEntity() throws WFOException {
		try {
			ID empID = null;

			String strEmpId = request.getParameter("employeeId");
			if (StringUtil.isEmpty(strEmpId)) {
				throw new BusinessWFOException("no employeeId found", Response.Status.CONFLICT);
			}

			empID = new ID(strEmpId);
			RequestContext context = getRequestContext();
			User user = getUser(request);

			return timeOffDataFacade.getTimeOffDataById(context, empID, user);
		} catch (WFOException ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw ex; // should have already been logged
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	@Override
	public TimeOffDataEntity getEntity(@PathParam("id") String employeeId) throws WFOException {
		return null;
	}


	@Override
	public TimeOffDataList<TimeOffDataEntity> getAllEntities() throws WFOException {
		return null;
	}


	@Override
	public TimeOffDataEntity updateEntity(TimeOffDataEntity employee) throws WFOException {
		return new TimeOffDataEntity();
	}

	@Override
	public TimeOffDataEntity addEntity(TimeOffDataEntity employee) throws WFOException {
		return new TimeOffDataEntity();
	}

	@Override
	public TimeOffDataEntity deleteEntity(@PathParam("id") String id) throws WFOException {
		return new TimeOffDataEntity();
	}

	@Override
	public boolean addEntities(Collection<TimeOffDataEntity> entities) throws WFOException {
		return true;
	}

	@Override
	public boolean updateEntities(Collection<TimeOffDataEntity> entities) throws WFOException {
		return true;
	}

	@Override
	public boolean deleteEntities(Collection<TimeOffDataEntity> entities) throws WFOException {
		return true;
	}

	@XmlRootElement(name = "timeoffData")
	private class TimeOffDataList<TimeOffDataEntity> extends ArrayList<TimeOffDataEntity>{
		public TimeOffDataList(List<TimeOffDataEntity> wrapped) {
			super(wrapped);
		}
	}
}
