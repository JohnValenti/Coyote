    /**
     * Update the applied status for the for the AutoProcessingRules
     * whose ids and status pair are passed as Map argument.
     * @param pOrganizationId -  ID object for organization.
     * @param idToAppliedMap - AutoProcessingRuleID -> isApplied (true or false)
     * @throws BbmUpdateException if there is an error.
     */
    public void saveAppliedStatus(ID pOrganizationId,  Map idToAppliedMap)
    throws Exception 
    {
        //get the AutoProcessingRules that need isApplied status to be updated
        AutoProcessingRuleDAO autoProcessingRuleDAO = new AutoProcessingRuleDAO();
        Set autoProcessingRuleIDs = idToAppliedMap.keySet();
        Collection autoProcessingRules = autoProcessingRuleDAO.getObjectsByIDs(autoProcessingRuleIDs); //rule.isApplied will not be set here
        
        //create a HashMap of ID -> AutoProcessingRule
        HashMap idToRuleMap = new HashMap(autoProcessingRules.size());
        for (Iterator itr = autoProcessingRules.iterator(); itr.hasNext(); ) 
        {
            AutoProcessingRule rule = (AutoProcessingRule) itr.next();
            idToRuleMap.put(rule.getID(), rule);
        }
                
        HashMap ruleToAppliedMap = new HashMap();        
        for (Iterator itr = idToAppliedMap.keySet().iterator(); itr.hasNext(); ) 
        {
            ID ruleID = (ID) itr.next();            
            boolean newIsApplied = idToAppliedMap.get(ruleID).toString().equals("true");
            
            AutoProcessingRule autoProcessingRule = (AutoProcessingRule)idToRuleMap.get(ruleID);
            
            ruleToAppliedMap.put(autoProcessingRule, new Boolean(newIsApplied));
            
            //only subOrgAllowOverride scope rules allow update of isApplied
            if (autoProcessingRule.getOrgScope().equals(AutoProcessingRule.SCOPE_SUB_ORGS_ALLOW_OVERRIDE))
            {
                AppliedAutoProcessingRule appliedAutoProcessingRule = getAppliedRuleOfInterest(pOrganizationId, ruleID);
                if (appliedAutoProcessingRule != null)
                {
                    //this was already an override, and it will remain an override
                    if (appliedAutoProcessingRule.getApplied() != newIsApplied)   
                    {
                        appliedAutoProcessingRule.setApplied(newIsApplied);
                        updateObject(appliedAutoProcessingRule);
                    }
                }
                else if (needToOverrideForApplied(pOrganizationId, ruleID, newIsApplied)) //if the inherited isApplied value != newIsApplied
                {
                    //this is a new override
                    appliedAutoProcessingRule = new AppliedAutoProcessingRule();
                    appliedAutoProcessingRule.setOrganizationID(pOrganizationId);
                    appliedAutoProcessingRule.setApplied(newIsApplied);
                    appliedAutoProcessingRule.setAutoProcessingRuleID(ruleID);
                    createObject(appliedAutoProcessingRule);
                }
            }
        }
    }//saveAppliedStatus
    
    /**
     * Determine whether we need to override an inherited AppliedAutoProcessingRule (if one exists at a parent org).
     * @param orgID
     * @param ruleID
     * @param newIsApplied
     * @return true if we need to override an inherited AppliedAutoProcessingRule. If none exists at a parent org,
     *         we still return true because we need to create one. Return false if an AppliedAutoProcessingRule
     *         exists at a parent org that has the same applied value that we want (newIsApplied).
     */
    private boolean needToOverrideForApplied(ID orgID, ID ruleID, boolean newIsApplied)
    throws Exception
    {
        //get the list of parent orgs in order of immediate parent first, to root org last
        List parentOrgs = CacheUtilBBM.getParentOrgIDs(orgID, getDMO()); //list will not include orgID, since we don't need it
        for (Iterator orgIt=parentOrgs.iterator(); orgIt.hasNext();)    
        {
            ID curOrgId = (ID)orgIt.next();
            AppliedAutoProcessingRule appliedAutoProcessingRule = getAppliedRuleOfInterest(curOrgId, ruleID);
            if (appliedAutoProcessingRule != null)
            {
                //found the rule. This first one is the only one we care about, since it's value is flowing down to us.
                return (appliedAutoProcessingRule.getApplied() != newIsApplied);
            }
        }
        
        //ruleID was not linked to any of our parent orgs. We will need to create a link of our own.
        return true;
    }
    
    /**
     * Get all AppliedAutoProcessingRules for any combination of the orgIDs and autoProcessingRuleIDs parameters.
     * @param orgIDs - Organization IDs.
     * @param autoProcessingRuleIDs - RequestAutoProcessingRule IDs.
     * @return all AppliedAutoProcessingRules for any combination of the orgIDs and autoProcessingRuleIDs parameters.
     */
    private AppliedAutoProcessingRule getAppliedRuleOfInterest(ID orgID, ID autoProcessingRuleID)
    throws Exception
    {
        AppliedAutoProcessingRule appliedAutoProcessingRule = null;
        
        StringBuffer strQuery = null;
        JdmoQuery jQuery1 = null;        
        
        strQuery = getSelectStatement();
        strQuery.append(" FROM APPLIEDAUTOPROCESSINGRULE A ");
        strQuery.append(" WHERE ORGANIZATIONID = ");
        strQuery.append(orgID.toString());
        strQuery.append(" AND AUTOPROCESSINGRULEID = ");
        strQuery.append(autoProcessingRuleID.toString());
        
        jQuery1 = m_dmo.createQuery(strQuery.toString(), Jdmo.STATIC_QUERY);
        JdmoRowset rs = m_dmo.createRowset(jQuery1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

        if (rs.next()) 
        {
            appliedAutoProcessingRule = (AppliedAutoProcessingRule) readObjectFromDB(rs);
        }
        rs.close();        
        return appliedAutoProcessingRule;
    }

    public void deleteByAutoProcessingRuleID(ID pAutoProcessingRuleID)
    throws Exception 
    {
        StringBuffer lStmt;
        lStmt = new StringBuffer( " DELETE APPLIEDAUTOPROCESSINGRULE ");
        lStmt.append(" WHERE AUTOPROCESSINGRULEID = ?");
        Object[] parm = new Object[1];
        parm[0] = pAutoProcessingRuleID;

        m_dmo.executePCommand(lStmt.toString(), parm);
    }
    
    /**
     * Create one AppliedAutoProcessingRule for each AutoProcessingRule in a Collection.
     * @param organizationId - the org ID to use for each AppliedAutoProcessingRule
     * @param rules - a collection of AutoProcessingRule's to create AppliedAutoProcessingRule's for.
     * @throws BbmCreateException
     */
    public void createObjects(ID organizationId, Collection rules)
    throws BbmCreateException
    {
        Iterator itr = rules.iterator();
        while (itr.hasNext()) 
        {
            AutoProcessingRule rule = (AutoProcessingRule)itr.next();

            AppliedAutoProcessingRule appliedAutoProcessingRule = new AppliedAutoProcessingRule();
            appliedAutoProcessingRule.setOrganizationID(organizationId);
                        
            boolean bIsApplied = true;
            if (rule.getOrgScope().equals(AutoProcessingRule.SCOPE_THIS_ORG_ONLY) && !rule.getOrganizationId().equals(organizationId))
                bIsApplied = false;
            
            appliedAutoProcessingRule.setApplied(bIsApplied);                        
            appliedAutoProcessingRule.setAutoProcessingRuleID(rule.getID());
            createObject(appliedAutoProcessingRule);
        }
    }
    
    /**
     * Create one AppliedAutoProcessingRule for the newly created AutoProcessingRule.
     * @param rule
     * @throws BbmCreateException
     */
    public void createObjects(AutoProcessingRule rule)
    throws BbmCreateException
    {
        AppliedAutoProcessingRule appliedAutoProcessingRule = new AppliedAutoProcessingRule();
        appliedAutoProcessingRule.setOrganizationID(rule.getOrganizationId());
        appliedAutoProcessingRule.setApplied(true); //isApplied will always be true for this parent org
        appliedAutoProcessingRule.setAutoProcessingRuleID(rule.getID());
        createObject(appliedAutoProcessingRule);
    }
    
    /**
     * If an AutoProcessingRule's ORGSCOPE changed, then we remove all child overrides for that rule,
     * and update that rule's APPLIEDAUTOPROCESSINGRULE.isApplied to true. 
     */
    public void updateObjects(AutoProcessingRule rule)
    throws Exception
    {
        // check if the user changed the ORGSCOPE value
        StringBuffer strQuery = new StringBuffer(64);
        strQuery.append("SELECT * ");
        strQuery.append(" FROM AUTOPROCESSINGRULE ");
        strQuery.append(" WHERE ID = ?");
        strQuery.append(" AND ORGSCOPE != ?");
        JdmoQuery jQuery1 = m_dmo.createQuery(strQuery.toString(), Jdmo.PARAM_QUERY);        
        jQuery1.setParString(1, rule.getID().toString());
        jQuery1.setParString(2, rule.getOrgScope());        
        JdmoRowset rs = m_dmo.createRowset(jQuery1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);       
        boolean doUpdate = rs.next();
        rs.close();
        
        if (doUpdate)
        {
            //First delete any child org isApplied overrides for this rule
            StringBuffer lStmt;
            lStmt = new StringBuffer( " DELETE FROM APPLIEDAUTOPROCESSINGRULE ");            
            lStmt.append(" WHERE AUTOPROCESSINGRULEID = ?");            
            lStmt.append(" AND ORGANIZATIONID <> ?");            
            Object[] parm = new Object[2];
            parm[0] = rule.getID().toString();
            parm[1] = rule.getOrganizationId().toString();
            m_dmo.executePCommand(lStmt.toString(), parm);
            
            //Now we need to update the APPLIEDAUTOPROCESSINGRULE for the rule's org to use isApplied=true, 
            //which is the default value for all org scopes
            lStmt = new StringBuffer( " UPDATE APPLIEDAUTOPROCESSINGRULE ");   
            lStmt.append(" SET ISAPPLIED = 1");  
            lStmt.append(" WHERE AUTOPROCESSINGRULEID = ?");            
            lStmt.append(" AND ORGANIZATIONID = ?");            
            parm = new Object[2];
            parm[0] = rule.getID().toString();
            parm[1] = rule.getOrganizationId().toString();
            m_dmo.executePCommand(lStmt.toString(), parm);
        }
    }
