package com.bluepumpkin.ejb.rm.requests.swap.request.ejb;

import java.util.Collections;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.rm.ShiftSwapCalendarInfo;
import com.bluepumpkin.ejb.bbm.rm.ShiftSwapManagerBridge;
import com.bluepumpkin.ejb.rm.requests.swap.request.model.ShiftSwapRequest;
import com.bluepumpkin.ejb.rm.util.LicenseUtil;


public class ShiftSwapManagerBridgeImpl implements ShiftSwapManagerBridge {

	
	/* getListApprovedShiftSwapCalendarInfo
	 * 
	 * Return a list of approved ShiftSwapItems (shift types only) for the date range and employee IDs
	 * Dates must be in GMT
	 * 
	 * @param Date startDateTimeGMT start of range to search
	 * @param Date endDateTimeGMT end of range to search
	 * @param List<ID> employeeIds  employee IDs to search
	 * @returns List<ShiftSwapCalendarInfo>
	 * 
	 */
	public List<ShiftSwapCalendarInfo> getListApprovedShiftSwapCalendarInfo(Date startDateTimeGMT, Date endDateTimeGMT, List<ID> employeeIds)
			throws BbmFinderException {
		
		
		// If there is no advanced license, return empty. 
		// This is to be implemented in 15.1.0 130, but not 15.1.1
		if (!LicenseUtil.isAdvancedRMShiftSwapCalendarInfo()) {
			return Collections.emptyList();
		}

		long detailLevel = ShiftSwapRequest.DL_BASIC | ShiftSwapRequest.DL_SHIFTSWAP_ITEMS;
		ShiftSwapRequestDAO dao = new ShiftSwapRequestDAO(detailLevel);
		return dao.getListApprovedShiftSwapCalendarInfo(startDateTimeGMT, endDateTimeGMT, employeeIds);

	}
}
