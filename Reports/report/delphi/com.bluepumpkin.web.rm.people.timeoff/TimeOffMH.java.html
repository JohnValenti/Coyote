package com.bluepumpkin.web.rm.people.timeoff;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.HashSet;
import java.util.List;
import java.util.Map;
import java.util.Set;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.workresource.model.Employee;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.ejb.CalendarTimeOffDayFacade;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.EmployeeReferenceScheduleAssignment;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.EmployeeReferenceScheduleAssignmentUpdateParameters;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOPoolAssignmentsByActivityUpdateParameters;

public class TimeOffMH {
	
	/**
	 * @return A map of employee Id to a reference schedule assignment for the passed in as of date
	 */
	public static Map<ID, EmployeeReferenceScheduleAssignment> getReferenceScheduleForWorkResources(Collection<ID> empIDs, Date dt) throws BbmCreateException, BbmFinderException, RemoteException    {
        CalendarTimeOffDayFacade ctodf = RmManagerFactory.getInstance().getTimeOffDayFacade();
        return ctodf.getReferenceScheduleForWorkResources(empIDs, dt);        
    }
	
	/**
	 * @return A list of the reference schedule assignments for the specified employees for the time range [start,end)
	 */
	public static List<EmployeeReferenceScheduleAssignment> getValidReferenceScheduleAssignments(Collection<ID> employeeIds, Date start,
			Date end) throws Exception { // NOSONAR
        CalendarTimeOffDayFacade ctodf = RmManagerFactory.getInstance().getTimeOffDayFacade();
        return ctodf.getValidReferenceScheduleAssignments(employeeIds, start, end); 
    }
	
	/**
	 * @return A list of the reference schedule assignments for the specified employee for the time range [start,end)
	 */
	public static List<EmployeeReferenceScheduleAssignment> getValidReferenceScheduleAssignments(ID employeeId, Date start, Date end)
			throws Exception { // NOSONAR
		return TimeOffMH.getValidReferenceScheduleAssignments( Collections.singleton(employeeId), start, end );
	}
	
	/**
	 * @return True if there is at least one scheduling period that is common for all the specified employees
	 */
	public static boolean hasAtLeastOneCommonSchedulingPeriod(Collection<ID> employeeIds) throws Exception { // NOSONAR
		CalendarTimeOffDayFacade ctodf = RmManagerFactory.getInstance().getTimeOffDayFacade();
		return ctodf.hasAtLeastOneCommonSchedulingPeriod(employeeIds);
	}
	

	public static void updateEmployeeReferenceScheduleEffectivity(Employee emp, Date asOfDate,
			EmployeeReferenceScheduleAssignmentUpdateParameters update) throws Exception { // NOSONAR
		CalendarTimeOffDayFacade ctodf = RmManagerFactory.getInstance().getTimeOffDayFacade();
		Set<ID> empIDSet = getEmployeeIdsFromEmployee(emp);

		ctodf.updateEmployeeReferenceScheduleEffectivity(empIDSet, asOfDate, update);
	}

	@SuppressWarnings("unchecked")
	private static Set<ID> getEmployeeIdsFromEmployee(Employee emp) {
		Set<ID> empIDSet = new HashSet<ID>();
		empIDSet.add(emp.getID());
		if (emp.getMultiEditIDs() != null && !emp.getMultiEditIDs().isEmpty()) {
			empIDSet.addAll(emp.getMultiEditIDs());
		}
		return empIDSet;
	}

	public static List<EmployeeReferenceScheduleAssignment> getFillerAssignments(Collection<ID> employeeIds, Date viewDate)
			throws Exception { // NOSONAR
		CalendarTimeOffDayFacade ctodf = RmManagerFactory.getInstance().getTimeOffDayFacade();
        return ctodf.getFillerAssignments(employeeIds, viewDate); 
	}

	public static void updateTimeOffPoolsByActivity(Employee emp, Date asOfDate,
			List<TOPoolAssignmentsByActivityUpdateParameters> updateParameters) throws Exception {
		Set<ID> empIDSet = getEmployeeIdsFromEmployee(emp);
		CalendarTimeOffDayFacade ctodf = RmManagerFactory.getInstance().getTimeOffDayFacade();
		ctodf.updateTOPoolAssignmentsByActivity(empIDSet, asOfDate, updateParameters);
	}

}
