package com.bluepumpkin.ejb.rm.requests.common.validation;

import java.util.Collection;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.bluepumpkin.ejb.rm.setup.autoprocessing.ejb.AutoProcessingRuleManager;

/**
 * Helper class to create instances of AutoProcessingRuleChecker
 */
public class AutoProcessingRuleCheckerHelper {

    /**
     * Create a AutoProcessingRuleChecker.
     * @param orgId the ID of the organization
     * @param request the request to be auto processed
     * @param activityId, an optional input for requests that have associated activities
     * (like timeoff).  This can be null.
     * @param range the time range of interest
     * @return a  AutoProcessingRuleChecker ready to check requests matching the input specification
     * @exception com.bluepumpkin.ejb.bbm.base.BbmFinderException when information necessary
     * to build the checker cannot be obtained.
     */
    public static AutoProcessingRuleChecker getChecker(ID orgId, RequestAggregate request,
                           ID activityId, TimeRange range, Collection validators,String requestType)
        throws Exception {
        // get org from ID
        Organization org = ValidationUtil.getOrganizationByID(orgId);
        return getChecker(org,request,activityId,range, validators,requestType);
    }

    /**
     * Create a AutoProcessingRuleChecker.
     *
     * @param org the organization.  Usually the org for the request's employee.
     *
     * @param request the request to be auto processed.
     *
     * @param activityId, an optional input for requests that have associated activities
     * (like timeoff).  This can be null.
     * @param range the time range of interest.  Usually the time range of the request.
     *
     * @param enabledValidators a Collection of validation rule class names.  These are the soft validators
     * enabled for the org the request's employee belongs to.
     *
     * @return a  AutoProcessingRuleChecker ready to check requests matching the input specification
     * @exception com.bluepumpkin.ejb.bbm.base.BbmFinderException when information necessary
     * to build the checker cannot be obtained.
     */
    public static AutoProcessingRuleChecker getChecker(Organization org, RequestAggregate request,
                           ID activityId, TimeRange range, Collection enabledValidators,String requestType)
        throws Exception {

        // Get the request autoprocessing rules specified

        Collection rules = null;
//        try {
            // find all enabled autoProcessing rules for the given org ID and for the given request type.
            AutoProcessingRuleManager arm = RmManagerFactory.getInstance().getAutoProcessingRuleManager();
            rules = arm.findEnabledAutoProcessingRules(org.getID(), requestType, activityId);
//        }  c_atch (Exception e) {
//            //log.error(e);
//          t_hrow RequestUtil.createValidationExceptionWrapper(e, log);
//        }

        // Get the holidays for the organization during the period
        TimeZone tz = org.getTimeZone();
        Collection holidays = ValidationUtil.getHolidays(org, tz, range);

        // Build a checker with the information we've gathered.
        return new AutoProcessingRuleChecker(rules, holidays, request, tz, org.getDayBoundaryOffset(), enabledValidators);
    }


    /** Dont create instances of this class */
    private AutoProcessingRuleCheckerHelper() {
    }
    private static Category log = Log.initCategory( AutoProcessingRuleCheckerHelper.class.getName());
}
