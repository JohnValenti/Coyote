    /**
     * Create the IN clause argument for the valid final states.
     * Assumes that valid final states not empty.
     */
    public static final String inClauseForFinalStates() {
        if ( m_inClauseForFinalStates == null ) {
            StringBuffer buf = new StringBuffer(256);
            char sep = ' ';

//            buf.append("(");
            for (Iterator it = VALID_FINAL_STATES.iterator(); it.hasNext();) {
                buf.append(sep).append('\'').append(it.next()).append('\'');
                sep = ',';
            }
//            buf.append(")");

            m_inClauseForFinalStates = buf.toString();
        }

        return m_inClauseForFinalStates;
    }
    /**
     * Create the IN clause argument for the valid final states.
     * Assumes that valid expired states not empty.
     */
    public static final String inClauseForExpiredStates() {
        if ( m_inClauseForExpiredStates == null ) {
            StringBuffer buf = new StringBuffer(256);
            char sep = ' ';

            buf.append("(");
            for (Iterator it = VALID_EXPIRED_STATES.iterator(); it.hasNext();) {
                buf.append(sep).append('\'').append(it.next()).append('\'');
                sep = ',';
            }
            buf.append(")");

            m_inClauseForExpiredStates = buf.toString();
        }

        return m_inClauseForExpiredStates;
    }

    /**
     * Create the IN clause argument for the valid final states.
     * Assumes that valid expired states not empty.
     */
	public static String getInClauseTOPendingStatuses() {
		if (m_inClauseForPendingStates == null) {
			StringBuffer sbPendingClause = new StringBuffer();
			for (int i = 0; i < TORequest.PENDING_HOUR_STATUSES.length; i++) {
				sbPendingClause.append((i==0?"(":",")).append("'").append(TORequest.PENDING_HOUR_STATUSES[i]).append("'");
			}
			sbPendingClause.append(")");
			m_inClauseForPendingStates = sbPendingClause.toString();
		}
		return m_inClauseForPendingStates;
	}


    private static HashMap initTransitionValidationMap() {
/*
*
*
*
*
*                            |--> Approved <---------
*                 Waitlist-->|       ^               |
*                   ^  |--<--+-<-----|               |
*                   |  v     |       |               |
*  negotiation --> Pending ->|--> Tentative<---------+
*                            |       |               |
*                            |       v               |
*                            |--> Denied ---------> Escalated.
*
*
* Note: All non terminal states are editable which means state transitions from a non-terminal
* state to the same must be allowed.
*
*               Pending   Approved   Denied   Escalated   Negotiation   Tentative   withdrawn   invalid   Waitlist
* Pending          x         x          x                                  x            x          x       x
* Approved                                                                              x
* Denied                                x          x                                               x
* Escalated                  x          x          x                       x            x          x
* Negotiation      x                                            x                       x          x
* Tentative        x         x          x                                  x            x          x
* Withdrawn
* Invalid
* Waitlist         x         x          x                                  *            x          x   	   x
*/

        HashMap map = new HashMap(101);

        map.put(RequestAuditTrail.STATUS_PENDING,
            new String[] {
                STATUS_PENDING, STATUS_APPROVED,
				STATUS_DENIED, STATUS_TENTATIVE,
				STATUS_WITHDRAWN, STATUS_INVALID,STATUS_WAITLIST
            });

		map.put(STATUS_APPROVED, new String[] { STATUS_APPROVED,STATUS_WITHDRAWN ,STATUS_INVALID});

		// out of synch: 'invalid'
		map.put(STATUS_DENIED,
			new String[] { STATUS_ESCALATED, STATUS_INVALID, STATUS_DENIED });

		map.put(STATUS_ESCALATED,
			new String[] {
				STATUS_APPROVED, STATUS_DENIED,
				STATUS_ESCALATED, STATUS_TENTATIVE,
                STATUS_WITHDRAWN, STATUS_INVALID});

        map.put(STATUS_NEGOTIATION,
            new String[] {
				STATUS_PENDING, STATUS_NEGOTIATION,
				STATUS_WITHDRAWN, STATUS_INVALID
            });

		// out of synch: 'tentative'.
		map.put(STATUS_TENTATIVE,
			new String[] {
				STATUS_PENDING, STATUS_APPROVED,
				STATUS_DENIED, STATUS_TENTATIVE,
                STATUS_WITHDRAWN, STATUS_INVALID
			});

        map.put(STATUS_WITHDRAWN, new String[] {  });

        map.put(STATUS_INVALID, new String[] {  });

        map.put(RequestAuditTrail.STATUS_WAITLIST,
            new String[] {
                STATUS_PENDING, STATUS_APPROVED,
				STATUS_DENIED, STATUS_TENTATIVE,
				STATUS_WITHDRAWN, STATUS_INVALID ,STATUS_WAITLIST
            });
        return map;
    }
