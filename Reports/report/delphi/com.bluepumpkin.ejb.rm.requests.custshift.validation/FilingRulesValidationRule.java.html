/**
 * <p> Rule from UI: Request complies with filing rules.
 * <p/>
 * <p> Ensures that the request to be validated passes the filing rules for the organization
 * <p/>
 * <p> Applies equally for 1 and 2 way swaps.
 * <p/>
 * <p> Applies equally for tentative and non-tentative requests (unpublished schedule not used).
 */
public class FilingRulesValidationRule implements Validator {
	private static final String m_className = FilingRulesValidationRule.class.getName();

	private static Category m_cat = Log.initCategory(m_className);

	public ValidationResult validate(Validatable validatable) throws Exception {
		CustShiftReq custShiftReq = (CustShiftReq) validatable;
		String methodName = "validate";
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, custShiftReq));

		CustShiftValidationCache vc = (CustShiftValidationCache) custShiftReq.getValidationCache();
		ValidationResult result = null;
		FilingRuleChecker ckr = vc.getFilingRuleChecker(custShiftReq);    //obtain the rule checker
		ckr.resetLastFailingRule();

		/* <p> see {@link RequestFilingRule RequestFilingRule} for the difference in
		 * filing rule type "interval (days in advance)" between SS/TO reqs and SB reqs.
		 */
		// if request fails filing rule checker
		if (!ckr.requestComplies(custShiftReq.getSubmittedOn(), custShiftReq.getStartTime(), custShiftReq.getEndTime(),
		        custShiftReq.getStartTime())) {
			// Since sometimes it might be mystifying why a request fails, we log
			// the rule that failed the request
			RequestFilingRule rule = ckr.getLastFailingRule();  //get the failing rule
			m_cat.debug("cust-shift req id:" + custShiftReq.getID() + " failed because it did not pass filing rule with id:" + rule.getID());
			// obtain the organization name
			Organization org = vc.getOrganizationForEmployeeDuringPeriod(custShiftReq.getEmployeeID(),
			        custShiftReq.getStartTime(), custShiftReq.getEndTime());
			String orgName = "";
			if (org != null) orgName = org.getName();

			// obtain employee name and day range.
			String empName = vc.getEmployeeNameByID(custShiftReq.getEmployeeID());
			String dayRange = rule.getDayRange();

			if (dayRange.equals(RequestFilingRule.DAYRANGE_ALL_DAYS)) {
				result = ValidationUtil.setSoftValidationResult(custShiftReq, RmEjbBundleKey.SS_FILING_RULES_ALLDAYS, orgName,
				        empName, custShiftReq.getStartTime(), custShiftReq.getEndTime(), m_className);
			} else if (dayRange.equals(RequestFilingRule.DAYRANGE_DATES)) {
				result = ValidationUtil.setSoftValidationResult(custShiftReq,
				        RmEjbBundleKey.SS_FILING_RULES_DAYRANGE,
				        new Serializable[]{orgName, rule.getStartDate(), rule.getEndDate(),
				                           empName, custShiftReq.getStartTime(), custShiftReq.getEndTime()}, m_className);
			} else {
				throw RequestUtil.createIllegalArgumentException(dayRange, m_cat);
			}
		}
		if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName, result));

		return result;
	}
}
