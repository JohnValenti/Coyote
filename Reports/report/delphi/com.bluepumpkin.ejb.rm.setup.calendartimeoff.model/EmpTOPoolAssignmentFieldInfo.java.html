package com.bluepumpkin.ejb.rm.setup.calendartimeoff.model;

import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

/**

public final class EmpTOPoolAssignmentFieldInfo extends FieldInfo {

    // field id constants
    public static final int EMPTOPOOL_ID = 0;
    public static final int EMPTOPOOL_TOPOOLID = 1;
    public static final int EMPTOPOOL_WORKRESOURCEID = 2;
    public static final int EMPTOPOOL_STARTTIME = 3;
    public static final int EMPTOPOOL_ENDTIME = 4;
    public static final int EMPTOPOOL_ACTIVITYID = 5;

    private static final int EMP_TOPOOL_NUMOFFIELDS = 6;

    private static FieldInfoEntry arrFieldInfo[] = new FieldInfoEntry[EMP_TOPOOL_NUMOFFIELDS];
    private static String strTableName = "WORKRESOURCETOPOOL";

    // static initialization
    static {
      arrFieldInfo[EMPTOPOOL_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
      arrFieldInfo[EMPTOPOOL_TOPOOLID] = new FieldInfoEntry("TOPOOLID", PrimitiveTypes.ID);
      arrFieldInfo[EMPTOPOOL_WORKRESOURCEID] = new FieldInfoEntry("WORKRESOURCEID", PrimitiveTypes.ID);
      arrFieldInfo[EMPTOPOOL_STARTTIME] = new FieldInfoEntry("STARTTIME", PrimitiveTypes.DATETIME);
      arrFieldInfo[EMPTOPOOL_ENDTIME] = new FieldInfoEntry("ENDTIME", PrimitiveTypes.DATETIME);
      arrFieldInfo[EMPTOPOOL_ACTIVITYID] = new FieldInfoEntry("ACTIVITYID", PrimitiveTypes.ID);
    }

    public EmpTOPoolAssignmentFieldInfo() {
        super();
        setNotifiable(true);
    }

	// implement abstract methods in parent class
	public String getTableName() {
		return strTableName;
	}

	@Override
	public int getIDFieldIndex() {
		return EMPTOPOOL_ID;
	}

	@Override
	public int getParentIDFieldIndex() {
		return EMPTOPOOL_WORKRESOURCEID;
	}

	@Override
	public FieldInfoEntry[] getFieldInfo() {
		return arrFieldInfo;
	}
}
