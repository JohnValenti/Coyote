	public static final int TOPOOL_WLPRIORITYORDER1 = 6;

	public static final int TOPOOL_WLPRIORITYORDER3 = 8;
	public static final int TOPOOL_LASTMODIFIEDAT =   9;

	private static final int TOPOOL_NUMOFFIELDS = 11;

	private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[TOPOOL_NUMOFFIELDS];
	private static String m_strTableName = "TOPOOL";

	// static initialization
	static {
		m_arrFieldInfo[TOPOOL_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
		m_arrFieldInfo[TOPOOL_NAME] = new FieldInfoEntry("NAME", PrimitiveTypes.STRING);
		m_arrFieldInfo[TOPOOL_DESCRIPTION] = new FieldInfoEntry("DESCRIPTION", PrimitiveTypes.STRING);

		m_arrFieldInfo[TOPOOL_WLPRIORITYORDER1] = new FieldInfoEntry("WLPRIORITYORDER1", PrimitiveTypes.STRING);
		m_arrFieldInfo[TOPOOL_WLPRIORITYORDER2] = new FieldInfoEntry("WLPRIORITYORDER2", PrimitiveTypes.STRING);
		m_arrFieldInfo[TOPOOL_WLPRIORITYORDER3] = new FieldInfoEntry("WLPRIORITYORDER3", PrimitiveTypes.STRING);

		m_arrFieldInfo[TOPOOL_MODIFIEDBY] = new FieldInfoEntry("MODIFIEDBY", PrimitiveTypes.STRING);
	}

	public TOPoolFieldInfo() {
		super();
		setNotifiable(true);
	}

	// implement abstract methods in parent class
	public String getTableName() { return m_strTableName; }

	public int getIDFieldIndex() { return TOPOOL_ID; }

	public FieldInfoEntry[] getFieldInfo() { return m_arrFieldInfo; }
