public class AutoProcessingListPageModel extends RmSettingsListPM { 
	private ValidationRulesMap m_vrMap;

	/**
	 * Constructor
	 */
	public AutoProcessingListPageModel(RequestContext context) {
		super(context,
				PrivilegeKeys.TOM_SETAUTOPROCESSING_ID,
				PrivilegeKeys.SS_SETAUTOPROCESSING_ID,
				PrivilegeKeys.SB_SETAUTOPROCESSING_ID,
				PrivilegeKeys.CS_SETAUTOPROCESSING_ID, 
				PrivilegeKeys.FT_SETAUTOPROCESSING_ID);

		m_vrMap = ValidationRulesMap.getInstance();

		// Enable Object Copy
		initCopyObjType(AutoProcessingRule.OBJECT_TYPE_AUTOPROCESSINGRULE_CS);
		setName("autoProcessingRuleList");
	}

	/**
	 * Return an instance of AutoProcessingListPageModel ready to be rendered in
	 * JSP
	 * @param   context RequestContext
	 * @return  AutoProcessingListPageModel  instance of AutoProcessingListPageModel
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, AutoProcessingListPageModel.class);
	}

	/**
	 * Loads all the RuleList value objects for the selected Organization ID
	 */
	@Override
	protected Collection loadSpecificRules() throws Exception {
		return AutoProcessingModelHandler.getAutoProcessingRuleList(getSelectedIDObject(),
				Request.REQUESTTYPE_ALL);
	}

	/**
	 * Initialize List, sets the list header
	 */
	@Override
	protected void initList() {
		super.initList();

		// Setup Headers
		AutoProcessingUtil.initListHeader(m_list.getHeader(), m_rmBundle, m_bbmBundle, true, m_localizer);

		// Setup basic table property
		m_list.setMultiRowSelectable(false);
	}

	/**
	 * Initialize toolbar
	 */
	@Override
	protected void initToolbar() {
		String applyRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_AUTO_PROCESSING_SAVE_ACTION);

		String addRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_AUTO_PROCESSING_CREATE_NEW_ACTION);

		String copyRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_AUTO_PROCESSING_COPY_ACTION);

		String editRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_AUTO_PROCESSING_EDIT_ACTION);

		String delRuleLabel = i18n(m_rmBundle,
				RmWebBundleKey.ORG_RM_AUTO_PROCESSING_DELETE_ACTION);

		useDefaultToolbar(applyRuleLabel, addRuleLabel, copyRuleLabel,
				editRuleLabel, delRuleLabel);
	}

	/**
	 * Populates the list data in the model
	 */
	@Override
	protected void createListModel() {
		if (!loadRules()) {
			return;
		}

		ArrayList listModel = new ArrayList();

		if (!m_isEmpty) {
			ID selectedOrgID = getSelectedIDObject();
			boolean isTOConfigEnabled = isAuthorizedToConfigureTOM(selectedOrgID);
			boolean isSSConfigEnabled = isAuthorizedToConfigureSS(selectedOrgID);
			boolean isSBConfigEnabled = isAuthorizedToConfigureSB(selectedOrgID);
			boolean isSCConfigEnabled = isAuthorizedToConfigureSC(selectedOrgID);
			boolean isFTConfigEnabled = isAuthorizedToConfigureFT(selectedOrgID);
			
			for (Iterator it = m_ruleList.iterator(); it.hasNext(); ) {
				AutoProcessingRule rule = (AutoProcessingRule)it.next();

				// Create the row data
				String rowID = rule.getID().toString();
				String requestType = rule.getRequestType();

				boolean isTimeOff = (Request.REQUESTTYPE_TIMEOFF.equals(requestType) || requestType==null);
				boolean isFlexTime = Request.REQUESTTYPE_FLEXTIME.equals(requestType);
				boolean isShiftSwap = Request.REQUESTTYPE_SHIFTSWAP.equals(requestType);
				boolean isShiftBid = Request.REQUESTTYPE_SHIFTBID.equals(requestType);
				boolean isShiftSwapWithdraw =Request.REQUESTTYPE_SHIFTSWAP_WITHDRAW.equals(requestType);
				boolean isCustomShift = Request.REQUESTTYPE_CUSTSHIFT.equals(requestType);

				boolean isConfigEnabled = isTimeOff ? isTOConfigEnabled :
						(isShiftBid ? isSBConfigEnabled : 
						(isShiftSwap || isShiftSwapWithdraw ? isSSConfigEnabled : isSCConfigEnabled));
				isConfigEnabled = isFlexTime ? isFTConfigEnabled : isConfigEnabled;

				String chkbox = AutoProcessingUtil.createApplyChkBox(rowID,
						rule.getApplied(), (rule.getCanChangeApply() && isConfigEnabled));

				String orgName = getOrganizationName(rule.getOrganizationId());
				TimeZone tz = RmSetupUtil.getOrgTimeZone(m_context, rule.getOrganizationId());
				DefaultMultiColumnNodeData row_data = AutoProcessingUtil.createRuleDataRow(
						m_context, rule, m_rmBundle, m_vrMap, chkbox, orgName, true, tz, m_localizer);
				row_data.setSelectable(isConfigEnabled);

				// Add node Attributes, sets the
				// Also need to check if this rule belongs to this organization
				// or is inherited. Combination
				// would decide if the row is editable.
				String sIsEditable = isRowEditable(rule.getOrganizationId());
				String copyObjType = isShiftSwap ? AutoProcessingRule.OBJECT_TYPE_AUTOPROCESSINGRULE_SS :
					(isShiftBid ? AutoProcessingRule.OBJECT_TYPE_AUTOPROCESSINGRULE_SB : 
						(isCustomShift ? AutoProcessingRule.OBJECT_TYPE_AUTOPROCESSINGRULE_CS : AutoProcessingRule.OBJECT_TYPE_AUTOPROCESSINGRULE_TOM));

				copyObjType = isFlexTime ? AutoProcessingRule.OBJECT_TYPE_AUTOPROCESSINGRULE_FT : copyObjType;
				copyObjType = isShiftSwapWithdraw ? AutoProcessingRule.OBJECT_TYPE_AUTOPROCESSINGRULE_SSW : copyObjType;

				AutoProcessingUtil.addRowAttr(sIsEditable, isConfigEnabled, copyObjType, row_data);

				// Add row data to list model
				listModel.add(row_data);

				// Check if row should be selected, assuming only one should be
				// selected
				if (OrganizationModelHandler.UNREASONABLE_ORG_ID.equals(m_selectedRowOrgID)) {
					if (isRowSelected(rule.getID())) {
						m_list.addSelectedRowID(rowID);
						m_selectedRowOrgID = rule.getOrganizationId();
					}
				}
			} //end for
		} else {
			// Create/Add no data msg Row
			MultiColumnNodeData row_data = AutoProcessingUtil.createNoDataRow(m_rmBundle, true, m_localizer);
			listModel.add(row_data);
		}

		m_list.setListModel(listModel);
	}
	@Override
	protected void initDialog(RequestContext context) {
		
		if(LicenseUtil.isAdvancedRMLicense()){
			ID selectedOrgID = getSelectedIDObject();
			initDialogMenuAndAddTimeOffRules(selectedOrgID,context);
			addFlexTimeToDialogMenu(selectedOrgID,context);
			addShiftRulesToDialogMenu(selectedOrgID,context);
		} else {
			super.initDialog(context);
		}
	}

	/**
		 * Return the Selected AutoProcessingRule IDs
	 */
	public ID[] getSelectedAutoProcessingRuleIDs() {
		return m_performActionOnIDs;
	}

		return i18n(m_rmBundle, RmWebBundleKey.ORG_RM_AUTO_PROCESSING_LABEL);
	}

	/**
	 * Return URL for HTML Form Action
	 * @return 	String	Identifies the URL
	 */

	public String getFormAction() {
		return "request_processing_autoprocessinglist";
	}

	@Override
	protected String getDialogHeader() {
		return i18n(m_rmBundle, RmWebBundleKey.ORG_RM_CREATE_NEW_AUTO_PROCESS_RULE);
	}
}
