package com.bluepumpkin.web.rm.setup.autoprocessing;

/**
 * Title          AutoProcessingEditPageModel
 * Description    Used to display the Create/Edit Auto Processing Rule page
 * Copyright      Copyright (c) 2002
 * Company        Blue Pumpkin Software, Inc
 * @author        Howard Mullings
 * @version       1.0
 */
public class AutoProcessingEditPageModel extends OrgWorkpaneFormPageModel {

	private static final String NEW_TIMEOFF = "NEW_TIME_OFF";
    private static final String NEW_TIMEOFF_WITHDRAW = "NEW_TIMEOFF_WITHDRAW";
	private static final String NEW_SHIFTSWAP = "NEW_SHIFT_SWAP";
	private static final String NEW_SHIFTBID = "NEW_SHIFT_BID";
	private static final String NEW_CUSTOMSHIFT  = "NEW_CUSTOM_SHIFT";
	private static final String NEW_FLEXTIME  = "NEW_FLEX_TIME";
	private static final String NEW_SHIFTSWAP_WITHDRAW  = "NEW_SHIFTSWAP_WITHDRAW";

	private static final Category log =
		Log.initCategory(AutoProcessingEditPageModel.class.getName());

	private final ResourceBundle       m_rmBundle;

	private AutoProcessingRule   m_autoProcessingRule;
	private final AutoProcessingRulePC m_autoProcessingRulePC;
	private Map                  m_validationRules;
	private Collection<StringsPair> m_timeOffTypeList;
	private final Collection           m_dayTypeList;

	/**
	 * Constructor
	 */
	public AutoProcessingEditPageModel(RequestContext context) {
		super(context);
		m_rmBundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);
		extractParameters(context.getRequest(), WorkpanePageModel.SELECTED_ID_FN);
		// Component specific required JavaScript Files
		addRequiredJavaScriptFile(JavaScriptFileID.MEDIATOR_WORKPANE);
		addRequiredJavaScriptFile(JavaScriptFileID.DIALOG_PICKERS);
		// define javascript methods that enable/disable items on the page
		addRequiredJavaScriptFile(RmJavaScriptFileID.ENABLE_RULES);
		setJSMediator(JavaScriptFileID.MEDIATOR_WORKPANE_GENERIC);

		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);

		ResourceBundle rmEjbBundle = m_localizer.getBundle(RmEjbBundleKey.BUNDLE_NAME);
		m_dayTypeList = AutoProcessingModelHandler.getDayTypeList(rmEjbBundle, m_localizer);
		String createRuleType = (String)context.getAttribute(RequestContext.REQUEST_SCOPE, AutoProcessingRequestHandler.CREATE_RULE_TYPE);

		m_autoProcessingRule = new AutoProcessingRule();
		m_autoProcessingRule.setOrganizationId(this.getSelectedIDObject());

		if (createRuleType != null){
			m_autoProcessingRule.setRequestType(createRuleType);
		}
		else {
			m_autoProcessingRule.setRequestType(
				createRuleType = Request.REQUESTTYPE_TIMEOFF);
			extractParameters(context.getRequest(), WorkpanePageModel.SELECTED_ID_FN);
		}

		try {
			if (getSelectedIDObject() != null) {
				m_validationRules = ValidationModelHandler.getValidationRules(Request.REQUESTTYPE_ALL, getSelectedIDObject());
			}
		} catch (Exception e) {
			log.l7dError(RmWebLogBundleKey.UNABLE_TO_LOAD_DATA, e);
		}

		m_autoProcessingRulePC = new AutoProcessingRulePC(context,
														  m_autoProcessingRule,
														  m_validationRules, RmSetupUtil.getOrgTimeZone(context, this.getSelectedIDObject()));

		m_autoProcessingRulePC.setOrganizationID(this.getSelectedIDObject());
		initForm();

		addChildComponent(m_autoProcessingRulePC);

		// Have to add the date picker PC to the page model otherwise the
		// system doesn't generate javascript initialization code correctly
		// because the dataRangePC is not a child of the m_autoProcessingRulePC.
		addChildComponent(m_autoProcessingRulePC.m_dateRangePC);
	}

	/**
	 * Load all of the time off types into our member variable. The time off types depend
	 * on the request type (Time Off or Flex Time).
	 */
	protected void loadTimeOffTypeList() {
		try {
			if (getSelectedIDObject() != null) {
				boolean isFlex = Request.REQUESTTYPE_FLEXTIME.equals(getCreateRuleType());
				m_timeOffTypeList = AutoProcessingModelHandler.getTimeOffTypesExcludingVto(getSelectedIDObject(), isFlex);
			}
		} catch (Exception e) {
			log.l7dError(RmWebLogBundleKey.UNABLE_TO_LOAD_DATA, e);
		}

		if (m_timeOffTypeList == null) {
			m_timeOffTypeList = new ArrayList<StringsPair>();
		}
	}

	/**
	 * Return an instance of AutoProcessingEditPageModel ready to be rendered in JSP
	 * @param   context RequestContext
	 * @return  AutoProcessingEditPageModel  instance of AutoProcessingEditPageModel
	 */
	public static AutoProcessingEditPageModel getInstance(RequestContext context) {
		return (AutoProcessingEditPageModel)getInstance(context, AutoProcessingEditPageModel.class);
	}

	/**
	 * Initialize Form Display
	 */
	@Override
	protected void initForm() {
		//invoke the methods that create the PC
		m_autoProcessingRulePC.setPageType(getCreateRuleType());
		m_autoProcessingRulePC.setAutoProcessingRule(m_autoProcessingRule);
		m_autoProcessingRulePC.setDayTypeList(m_dayTypeList);
		m_autoProcessingRulePC.setTimeOffTypeList(m_timeOffTypeList);
	}

	/**
	 * Create Form Model
	 */
	@Override
	protected void createFormModel() {

		try {
			// BLOCK CALLED IN CASE OF EDIT.....
			if (getRuleID() != null && !hasError()) {
				m_autoProcessingRule = AutoProcessingModelHandler.getAutoProcessingRule(getRuleID());
			}

			loadTimeOffTypeList();

		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		}
	}

	/**
	 * Initialize the Toolbar
	 * Adds the Save, Cancel and Revert button
	 */
	@Override
	protected void initToolbar() {
		m_toolbar.setName(getName() + "_toolbar");
		boolean isBtnEnabled = isAuthorizedToConfigure();
		ToolbarUtil.addValidateSaveBtn(m_toolbar, isBtnEnabled);
		ToolbarUtil.addConfirmDoneBtn(m_toolbar, isBtnEnabled);
		ToolbarUtil.addResetBtn(m_toolbar, isBtnEnabled);
	}

	/**
	 * Return the form title
	 * @return String Identifeis the form title
	 */
	@Override
	public String getFormTitle() {

		MessageFormat formTitle = null;
		Object[] requestType = null;
		if (isNew()) {
			formTitle = new MessageFormat(i18n(m_rmBundle, RmWebBundleKey.ORG_RM_CREATE_RULE_TEMPLATE));

		} else {
			formTitle = new MessageFormat(i18n(m_rmBundle, RmWebBundleKey.ORG_RM_EDIT_RULE_TEMPLATE));

		}

		if (getCreateRuleType().equals(Request.REQUESTTYPE_TIMEOFF)) {
			requestType = new Object[] {
				i18n(m_rmBundle,RmWebBundleKey.REQUEST_TYPE_TIME_OFF)};
		}else if (getCreateRuleType().equals(Request.REQUESTTYPE_TIMEOFF_WITHDRAW)) {
			requestType = new Object[] {
				i18n(m_rmBundle,RmWebBundleKey.REQUEST_TYPE_TIME_OFF_WITHDRAW)};
		} else if (getCreateRuleType().equals(Request.REQUESTTYPE_SHIFTSWAP)) {
			requestType = new Object[] {
				i18n(m_rmBundle,RmWebBundleKey.REQUEST_TYPE_SHIFT_SWAP)};
		} else if(getCreateRuleType().equals(Request.REQUESTTYPE_CUSTSHIFT)) {
			requestType = new Object[] {
				i18n(m_rmBundle,RmWebBundleKey.REQUEST_TYPE_CUSTOM_SHIFT)};
		} else if (getCreateRuleType().equals(Request.REQUESTTYPE_FLEXTIME)) {
			requestType = new Object[] {
				i18n(m_rmBundle,RmWebBundleKey.REQUEST_TYPE_FLEX_TIME)};
		} else if (getCreateRuleType().equals(Request.REQUESTTYPE_SHIFTSWAP_WITHDRAW)) {
			requestType = new Object[] {
				i18n(m_rmBundle,RmWebBundleKey.REQUEST_TYPE_SHIFT_SWAP_WITHDRAW)};
		} else {
			requestType = new Object[] {
				i18n(m_rmBundle,RmWebBundleKey.REQUEST_TYPE_SHIFT_BIDDING)};
		}

		return formTitle.format(requestType);
	}

	/**
	 * Gets entity name. Used for title.
	 * @return String Identifies the title entity
	 */
	@Override
	protected String getTitleEntityType() {
		return i18n(m_rmBundle, RmWebBundleKey.ORG_RM_AUTO_PROCESSING_LABEL);
	}

	/**
	 * Returns an entity name. Used for title.
	 * @return String Identifies the entity name
	 */
	@Override
	protected String getEntityName() {
		return "";
	}

	/**
	 * Return URL for HTML Form Action
	 * @return 	String	Identifies the URL
	 */
	@Override
	public String getFormAction() {
		return "request_processing_autoprocessinglist";
	}

	/**
	 * @return ID	Identifies the AutoProcessingRuleID
	 */
	public ID getRuleID() {
		if (isAction(m_context, NEW_TIMEOFF) ||
                isAction(m_context, NEW_SHIFTBID) ||
                isAction(m_context, NEW_SHIFTSWAP) ||
                isAction(m_context, NEW_TIMEOFF_WITHDRAW) ||
                isAction(m_context, NEW_CUSTOMSHIFT) ||
                isAction(m_context, NEW_FLEXTIME) ||
                isAction(m_context, NEW_SHIFTSWAP_WITHDRAW)) {
            setPerformActionOnID(null);
            return null;
        }
		if (m_performActionOnIDs==null) {
			return null;
		}
		return m_performActionOnIDs[getEditIndex()];
	}

	/**
	 * Return Page Action passed to Request Handler
	 */
	public String getPageAction(RequestContext context) {
	  return context.getRequest().getParameter(PageAction.PAGE_ACTION);
	}

    /**
     * Return true if request action is the same the specified action
     */
	public boolean isAction(RequestContext context, String action){
		String pageAction = getPageAction(context);
		if (pageAction==null || action==null) {
			return false;
		}

		return pageAction.equals(action);
	}

	/** required abstract method that is not needed! */
	@Override
	protected void initFieldValidators() { }

	/**
	 * Displays the page Header
	 * @return Identifies the page Header
	 */
	public String getHeaderText() {
		return getFormTitle();
	}

	/**
	 * set and get Methods
	 */
	public void setCreateRuleType(String requestType) {
		m_autoProcessingRule.setRequestType(requestType);
	}

	public String getCreateRuleType() {
		return m_autoProcessingRule.getRequestType();
	}

	public AutoProcessingRulePC getAutoProcessingRuleForm() {
		return m_autoProcessingRulePC;
	}

	public AutoProcessingRule getAutoProcessingRule() {
		return m_autoProcessingRule;
	}

	public void setDateRangeStart(Date startDate) {
		m_autoProcessingRule.setStartDate(startDate);
	}

	public void setDateRangeEnd(Date endDate) {
		m_autoProcessingRule.setEndDate(endDate);
	}

	@Override
	public boolean extractParameters(HttpServletRequest request) {
		boolean success=super.extractParameters(request);

		// have to extract the date picker values directly.
		// I'm not sure why this doesn't happen automatically.
		if (m_autoProcessingRulePC != null) {
			m_autoProcessingRulePC.m_dateRangePC.extractParameters(request);
			setDateRangeStart(m_autoProcessingRulePC.m_dateRangePC.getStartDatePickerPC().getDate());
			setDateRangeEnd(m_autoProcessingRulePC.m_dateRangePC.getEndDatePickerPC().getDate());
		}

		return success;
	}

	public ResourceBundle getBundle() {
		return m_rmBundle;
	}

}
