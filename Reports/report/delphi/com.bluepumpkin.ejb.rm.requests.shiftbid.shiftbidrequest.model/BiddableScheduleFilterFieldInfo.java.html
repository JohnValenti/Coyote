package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model;

public class BiddableScheduleFilterFieldInfo extends BiddableScheduleFieldInfo 
{
	public static final int SHIFTASSIGN_SHIFTID = BIDDABLESCHEDULENUM_OF_FIELDS;
	public static final int SHIFTASSIGN_STARTTIME = SHIFTASSIGN_SHIFTID + 1;
	public static final int SHIFTASSIGN_DURATION = SHIFTASSIGN_STARTTIME + 1;
	public static final int SHIFTEVENTASSIGN_STARTTIME = SHIFTASSIGN_DURATION + 1;
	public static final int SHIFTEVENTASSIGN_DURATION = SHIFTEVENTASSIGN_STARTTIME + 1;
	public static final int PHANTOM_EMPTEMPLATEID = SHIFTEVENTASSIGN_DURATION + 1;
	
	private ArrayList m_List = new ArrayList();
	
	public BiddableScheduleFilterFieldInfo()
	{
		FieldInfoEntry []arr =  super.getFieldInfo();
		for( int i=0; i < arr.length; ++i)
		{
				this.m_List.add( arr[i] );
		}

	      this.m_List.add( new FieldInfoEntry("sa_shiftid", PrimitiveTypes.ID) );
	      this.m_List.add(  new FieldInfoEntry("sa_starttime", PrimitiveTypes.DATETIME) );
	      this.m_List.add(  new FieldInfoEntry("sa_duration", PrimitiveTypes.INT) );
	      this.m_List.add(  new FieldInfoEntry("sea_starttime", PrimitiveTypes.DATETIME) );
	      this.m_List.add(  new FieldInfoEntry("sea_duration", PrimitiveTypes.INT) );
	      this.m_List.add(  new FieldInfoEntry("ph_emptemplateid", PrimitiveTypes.ID) );
	}

    public FieldInfoEntry [] getFieldInfo() 
    {
    	return (FieldInfoEntry[])this.m_List.toArray( new FieldInfoEntry[this.m_List.size()] );
    }

}
