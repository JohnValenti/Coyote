package com.bluepumpkin.ejb.rm.requests.swap.request.model;

import com.bluepumpkin.ejb.bbm.vo.ChildInfoEntry;

import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;


/**
 * meta data information for ShiftSwapRequest object
 */
public class ShiftSwapRequestFieldInfo extends FieldInfo {
    public final static int SHIFTSWAPREQUEST_I_ID = 0;
    public final static int SHIFTSWAPREQUEST_S_SWAPTYPE = 1;
    public final static int SHIFTSWAPREQUEST_S_NEGOTIATIONCOMMENT = 2;
    
    protected final static int SHIFTSWAPREQUEST_NUM_OF_FIELDS = 3;
    
    private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[SHIFTSWAPREQUEST_NUM_OF_FIELDS];
    
    public final static int REQUESTITEM_CHILD_TYPE = 0;
    public final static int SS_WITHDRAW_CHILD_TYPE = 1;
    protected final static int SS_REQUEST_NUM_OF_CHILDREN = 2;
    
    private static ChildInfoEntry[] m_childInfo = { ChildInfoEntry.oneToManyChild(),ChildInfoEntry.oneToManyChild() };
    
    private final static String SHIFTSWAP_TABLENAME = "SHIFTSWAPREQUEST";

    static {
        RequestUtil.initFieldInfoEntries(m_arrFieldInfo, ShiftSwapRequestFieldInfo.class,
                "SHIFTSWAPREQUEST_", SHIFTSWAPREQUEST_NUM_OF_FIELDS);
    }

    // private constructor to prevent instance creation
    public ShiftSwapRequestFieldInfo() {
    }

    /** get meta data information of all fields  */
    public FieldInfoEntry[] getFieldInfo() {
        return m_arrFieldInfo;
    }

    // implement abstract methods from parent class
    public String getTableName() {
        return SHIFTSWAP_TABLENAME;
    }

    // Return the child info array that contains necessary information about
    // the time off choice children entries and any entries from our superclass.
    public ChildInfoEntry[] getChildInfo() {
        return m_childInfo;
    }

    public int getIDFieldIndex() {
        return SHIFTSWAPREQUEST_I_ID;
    }
}
