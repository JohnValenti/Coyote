/**
 * Rule to validate that the shifts, which are part of the posting, are still
 * valid.
 * For partial postings, validate that the entire posting does not comprise of gap activity
 *
 * todo: for a time off posting, confirm that agent has a timeoff.
 * todo: check if posting has expired.
 * todo: auto purge for postings. requests are purged thru' the UI.
 */
public class ShiftExistValidationRule implements Validator{
   private static Category m_cat = Log.initCategory( ShiftExistValidationRule.class.getName());

   private String m_className = ShiftExistValidationRule.class.getName();

   protected Category getCategory() {
      return m_cat;
   }

   public ShiftExistValidationRule(){};

   public ValidationResult validate(Validatable validatable) throws Exception {
        ValidationResult result = null;
        m_cat.debug("Enter ShiftSwapPosting Validate: ShiftExistValidationRule");

        ShiftSwapPosting ssp = (ShiftSwapPosting)validatable;
        ID empId= ssp.getEmployeeID();
        ShiftSwapItem item = ssp.getShiftSwapItem();
        
        // this validation does not apply to swap posting of type "time-off" 
        if (ShiftSwapItem.SWAPITEMTYPE_TIMEOFF.equals(item.getShiftType())) {
            return null;
        }
        
        Date startDate = item.getStartDate();
        Date endDate = item.getEndDate();

	    ScheduleAccessManager sam = WfmManagerFactory.getScheduleAccessManager();

        // get all shift published assns	
	    Collection pubShiftAssns = RequestUtil.getEventsDuringPeriod(empId, 
													 Event.EVENT_TYPE_SHIFT_ASSIGNMENT,
	                                                 true,
	                                                 startDate,
	                                                 endDate,
													 sam);
	
        boolean bPubShiftExists = false;

		//get the exact event
		for (Iterator i = pubShiftAssns.iterator(); i.hasNext();){
			Event event = (Event)i.next();
			
			if (!item.getIsPartial() && event.getStartTime().equals(startDate) &&
					event.getEndTime().equals(endDate)){
				bPubShiftExists = true;
				break;
			} else if (TimePeriodUtil.inside(startDate, endDate, 
					event.getStartTime(),event.getEndTime())){
//				check if the entire posting is a gap activity
//				iterate through shift activities to look for gap
	    		for (Iterator itr = ((ShiftAssignment)event).getChildObjects(0).iterator(); itr.hasNext(); ) {
	    			ShiftEventAssignment shiftEventAssn = (ShiftEventAssignment) itr.next();
	    			if (shiftEventAssn.getActivityID().equals(Activity.ACTIVITY_SHIFT_OVERTIME_GAP) &&
	    					TimePeriodUtil.inside(item.getStartDate(), item.getEndDate(),
	    							shiftEventAssn.getStartTime(), shiftEventAssn.getEndTime())){
	    				//throw hard validation
	    				return ValidationUtil.setHardValidationResult( ssp, RmEjbBundleKey.SSP_POSTING_IS_GAP, m_className);		
	    			}
	    		}
				bPubShiftExists = true;
				break;
			}
		}

		 // if no shift found, then raise a hard validation error.
        if( !bPubShiftExists ){
            result = ValidationUtil.setHardValidationResult(ssp, RmEjbBundleKey.SSP_SHIFT_EXIST,
             empId, startDate, endDate, m_className);
        }

		// get all shift unpublished assns	
	    Collection unPubShiftAssns = RequestUtil.getEventsDuringPeriod(empId, 
													 Event.EVENT_TYPE_SHIFT_ASSIGNMENT,
	                                                 false,
	                                                 startDate,
	                                                 endDate,
													 sam);
	
        boolean bUnPubShiftExists = false;

		//get the exact event
		for (Iterator i = unPubShiftAssns.iterator(); i.hasNext();){
			Event event = (Event)i.next();
			if (!item.getIsPartial() && event.getStartTime().equals(startDate) &&
					event.getEndTime().equals(endDate)){
				bUnPubShiftExists = true;
				break;
			} else if (TimePeriodUtil.inside(startDate, endDate, 
					event.getStartTime(),event.getEndTime())){
//				check if the entire posting is a gap activity
//				iterate through shift activities to look for gap
	    		for (Iterator itr = ((ShiftAssignment)event).getChildObjects(0).iterator(); itr.hasNext(); ) {
	    			ShiftEventAssignment shiftEventAssn = (ShiftEventAssignment) itr.next();
	    			if (shiftEventAssn.getActivityID().equals(Activity.ACTIVITY_SHIFT_OVERTIME_GAP) &&
	    					TimePeriodUtil.inside(item.getStartDate(), item.getEndDate(),
	    							shiftEventAssn.getStartTime(), shiftEventAssn.getEndTime())){
	    				//throw hard validation
	    				return ValidationUtil.setHardValidationResult( ssp, RmEjbBundleKey.SSP_POSTING_IS_GAP, m_className);		
	    			}
	    		}
				bUnPubShiftExists = true;
				break;
			}
		}

        // if no shift found, then raise a hard validation error.
        if( !bUnPubShiftExists ){
			String employeeName = null;
			Employee employee = ValidationUtil.getEmployeeByID(empId, null, Employee.DETAIL_LEVEL_EMPLOYEE_BASIC);
			if (employee != null) {
				employeeName = employee.getEmployeeName();
			}
            result = ValidationUtil.setHardValidationResult(ssp, RmEjbBundleKey.SS_CANNOT_FIND_UNPUBSHIFT_FOR_SSITEM,
				startDate, endDate, employeeName, m_className);
        }

        return result;
   }

}
