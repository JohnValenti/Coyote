import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManager;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.swap.posting.model.ShiftSwapPosting;

/**
 * Rule to validate that the agent is still part of the organization.
 */
public class AgentGoodStandingValidationRule implements Validator{
   private final static String m_className = AgentGoodStandingValidationRule.class.getName();

   private final static Category m_cat = Log.initCategory( AgentGoodStandingValidationRule.class.getName());

   protected Category getCategory() {
      return m_cat;
   }

   public AgentGoodStandingValidationRule(){};

   public ValidationResult validate(Validatable validatable) throws Exception {
        ValidationResult result = null;
        m_cat.debug("Enter ShiftSwapPosting Validate: AgentGoodStandingValidationRule");

        ShiftSwapPosting ssp = (ShiftSwapPosting)validatable;
        ID empId= ssp.getEmployeeID();
        Date dtNow = new Date();

        Collection wrAssnsForEmp = null;
        WorkResourceManager wrMgr = BbmManagerFactory.getWorkResourceManager();
        wrAssnsForEmp = wrMgr.getWorkResourceAssignments(
              empId, dtNow, dtNow , false);

        // if no work resource assignments found, then employee is not associated with any organization currently. 
        if( wrAssnsForEmp.isEmpty() ) {
            result = ValidationUtil.setHardValidationResult(ssp,
              RmEjbBundleKey.SSP_NO_GOOD_STANDING, empId, m_className);
        }

        return result;
   }

}
