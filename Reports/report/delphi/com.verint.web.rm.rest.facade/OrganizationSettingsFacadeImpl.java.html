package com.verint.web.rm.rest.facade;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.requests.RequestsMH;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rm.rest.entity.settings.CustShiftSettingsEntity;
import com.verint.web.rm.rest.entity.settings.OrganizationSettingsEntity;
import com.verint.web.rm.rest.entity.settings.ShiftBiddingSettingsEntity;
import com.verint.web.rm.rest.entity.settings.ShiftSwapSettingsEntity;
import com.verint.web.rm.rest.entity.settings.TimeOffSettingsEntity;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.ejb.core.security.model.User;
import com.witness.web.uif.system.RequestContext;

public class OrganizationSettingsFacadeImpl implements IOrganizationSettingsFacade {
	private static final Category LOG = Log.initCategory(OrganizationSettingsFacadeImpl.class.getName());

	public OrganizationSettingsFacadeImpl() { // NOSONAR
	}

	/**
	 * Gets the employee's RM organization settings.
	 */
	@Override
	public OrganizationSettingsEntity getSettingsForEmployee(RequestContext context, ID employeeID)
			throws WFOException {
		try {
			OrganizationSettingsEntity result = new OrganizationSettingsEntity();

			OrganizationSetting orgSettings = RequestsMH.getOrgSettingForEmployee(context, employeeID);
			if (orgSettings == null) {
				return result;
			}
			result.setId(orgSettings.getCurrentOrgId().toString());
			result.setAllowEscalation(orgSettings.getAllowAgentEscalations());

			User user = context.getUser();
			if (user.isAuthorized(PrivilegeKeys.CS_VIEWPERSONALREQUESTS_ID)) {
				populateCustShiftSettings(result.getCustShift(), orgSettings);
			}

			if (user.isAuthorized(PrivilegeKeys.SB_VIEWPERSONALBIDS_ID)) {
				ShiftBiddingSettingsEntity sb = result.getShiftBidding();
				sb.setDisplayBidTemplateDes(orgSettings.getDisplayBidTemplateDes());
				sb.setMaxBids(orgSettings.getShiftBidMaxBidsPerEmployee());
				sb.setMaxPreference(orgSettings.getShiftBidMaxPref());
				sb.setPolicy(orgSettings.getShiftBidPolicy());
			}

			if (user.isAuthorized(PrivilegeKeys.SS_VIEWPERSONALREQUESTS_ID)) {
				ShiftSwapSettingsEntity shiftSwap = result.getShiftSwap();
				shiftSwap.setAllowPartialShiftPickup(orgSettings.getAllowPartialShiftPickup());
				shiftSwap.setAllowPartialShiftSwap(orgSettings.getAllowPartialShiftSwap());
				shiftSwap.setDisableOneWayShiftSwap(orgSettings.getDisableOneWayShiftSwap());
				shiftSwap.setEnableShiftSwap(orgSettings.getEnableShiftSwap());
				shiftSwap.setShiftSwapMaxNegotiationUpdates(orgSettings.getShiftSwapMaxNegotiationUpdates());
				shiftSwap.setPolicy(orgSettings.getShiftSwapPolicy());
			}

			if (user.isAuthorized(PrivilegeKeys.TOM_VIEWPERSONALREQUESTS_ID)) {
				TimeOffSettingsEntity to = result.getTimeOff();
				to.setPolicy(orgSettings.getTimeOffPolicy());
			}

			return result;
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(ex, LOG);
		}
	}

	private void populateCustShiftSettings(CustShiftSettingsEntity entity, OrganizationSetting orgSettings) {
		entity.setAllowChangeShiftRequests(orgSettings.getAllowShiftChangeRequests());
		entity.setAllowNewShiftRequests(orgSettings.getEnableCustomShiftRequests());
		entity.setAllowRequestsForOvertimeExtensionsAfterShift(orgSettings.getAllowOTAfterShift());
		entity.setAllowRequestsForOvertimeExtensionsBeforeShift(orgSettings.getAllowOTBeforeShift());
		entity.setAllowRequestsForOvertimeShifts(orgSettings.getAllowEntireOTShift());
		entity.setAllowRequestsForRegularExtensionsAfterShift(orgSettings.getAllowRegularExtensionAfterShift());
		entity.setAllowRequestsForRegularExtensionsBeforeShift(orgSettings.getAllowRegularExtensionBeforeShift());
		entity.setAllowRequestsForRegularShifts(orgSettings.getAllowEntireRegularShift());
		entity.setPolicy(orgSettings.getCustomShiftPolicy());
	}
}
