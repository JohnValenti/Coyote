package com.bluepumpkin.web.rm.setup.purging;

import java.util.Arrays;
import java.util.Date;
import com.bluepumpkin.ejb.rm.util.DateUtil;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.rm.util.LicenseUtil;
import com.bluepumpkin.web.bbm.organization.OrganizationRequestHandler;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.l10n.RmWebLogBundleKey;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

/**
 * Title          PurgingRequestHandler
 * Description    Handle request for the Purging page
 * Copyright      Copyright (c) 2002
 * Company        Blue Pumpkin Software, Inc
 * @author        Shubhangi Chavan
 * @version       1.0
 */
public class PurgingRequestHandler extends OrganizationRequestHandler {

	private static final String[] PURGE_PRIVS = new String[] {
		PrivilegeKeys.TOM_PURGEREQUESTS,
		PrivilegeKeys.FT_PURGEREQUESTS,
		PrivilegeKeys.SS_PURGEREQUESTS,
		PrivilegeKeys.SB_PURGEREQUESTS,
		PrivilegeKeys.CS_PURGEREQUESTS
	};



	private static final Category log =
			Log.initCategory(PurgingRequestHandler.class.getName());

	/**
	 * Process Request for the User page
	 * @param   context   RequestContext
	 * @throws Exception
	 */
	@Override
	public void processRequest(RequestContext context) throws Exception {
		if (isAction(context,PageAction.DELETE)) {
			processDelete(context);
		}
	}

	/**
	 * Processes the Delete action
	 * @param context RequestContext The Request context object
	 */
	private void processDelete(RequestContext context) {

		if (!isAuthorizedForOrg(context, getEffectivePrivilageKeys(), true)) {
			return;
		}

		PurgingPageModel model = new PurgingPageModel(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);

		if (model.validate()) {
			// invoke the delete method purgeRequest(...) here.
			String whatRequest = model.getWhatRequest();
			if (whatRequest == null) {
				return;
			}

			String requestType = model.getRequestType();
			ID orgId = model.getOrgID();
			boolean isForBranch = model.isForBranch();
			try {
				if (whatRequest.equals(PurgingPageModel.WHAT_REQ_ADVANCE_DAYS_VAL)) {
					int advancedCount = Integer.parseInt(model.getAdvanceCount());
					PurgingModelHandler.purgeRequest(context, requestType, orgId, advancedCount, isForBranch);
				} else if (whatRequest.equals(PurgingPageModel.WHAT_REQ_DATE_VAL)) {
					Date startDate = model.getStartDate().getTime(TimeZoneUtil.GMT_TIMEZONE);
					Date endDate = model.getEndDate().getTime(TimeZoneUtil.GMT_TIMEZONE);
					// Convert from local to GMT
					startDate = DateUtil.convertLocalDateToGMT(startDate, context.getViewingTimeZone());
					endDate = DateUtil.convertLocalDateToGMT(endDate, context.getViewingTimeZone());

					PurgingModelHandler.purgeRequest(context, requestType, orgId, startDate, endDate, isForBranch);
				} else if (whatRequest.equals(PurgingPageModel.WHAT_REQ_START_DATE_VAL)) {
					Date startDate = model.getRequestStartDate().getTime(TimeZoneUtil.GMT_TIMEZONE);
					Date endDate = model.getRequestEndDate().getTime(TimeZoneUtil.GMT_TIMEZONE);
					// Convert from local to GMT
					startDate = DateUtil.convertLocalDateToGMT(startDate, context.getViewingTimeZone());
					endDate = DateUtil.convertLocalDateToGMT(endDate, context.getViewingTimeZone());

					PurgingModelHandler.purgeRequestByRequestStart(context, requestType, orgId, startDate, endDate, isForBranch);
				} else {
					PurgingModelHandler.purgeRequest(context, requestType, orgId, isForBranch);
				}
				model.addPageMessage(Message.INFO_TYPE, RmWebBundleKey.ORG_RM_PURGING_DELETE_SUCCESS, RmWebBundleKey.BUNDLE_NAME);
			} catch (RmHardValidationException ex) {
				log.l7dError(RmWebLogBundleKey.ORG_RM_PURGING_DELETE_FAILURE, ex);
				Message msg = model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.ORG_RM_PURGING_DELETE_FAILURE,
						RmWebBundleKey.BUNDLE_NAME);
				msg.setAssociatedMessage(ex.getLocalizedMessage(context.getLocalizer(), context.getViewingTimeZone()));
			} catch (Exception ex) {
				log.l7dError(RmWebLogBundleKey.ORG_RM_PURGING_DELETE_FAILURE, ex);
				String msgID = RmWebBundleKey.ORG_RM_PURGING_DELETE_FAILURE;
				model.addPageMessage(Message.ERROR_TYPE, msgID, RmWebBundleKey.BUNDLE_NAME);
			}
		}

	}

	String[] getEffectivePrivilageKeys() {
		if (LicenseUtil.isAdvancedRMLicense()) {
			return PURGE_PRIVS;
		}

		//No flex-time if the advanced license is off
		return Arrays.stream(PURGE_PRIVS)
				.filter(x -> !PrivilegeKeys.FT_PURGEREQUESTS.equalsIgnoreCase(x))
				.toArray(String[]::new);
	}

}
