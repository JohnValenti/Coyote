package com.bluepumpkin.web.rm.requests.flextime;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.keys.RmKeys;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.requests.RequestDetailsPopupPM;
import com.bluepumpkin.web.rm.requests.timeoff.TimeOffRequestUtil;
import com.bluepumpkin.web.rm.requests.timeoff.TimeOffRequestsMH;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;

public class FlexTimeRequestDetailsPopupPM extends RequestDetailsPopupPM {

	/**
	 *
	 */
	private static final long serialVersionUID = 1L;
	private static final Category LOG = Log.initCategory(FlexTimeRequestDetailsPopupPM.class.getName());
	private static final String SHIFT_REQUEST_PC = "shiftRequestPC";

	private FlexTimeRequestsMH modelHandler;
	private String m_toTypeName;
	private MultiColListPC m_choiceListPC;
	private FlexTimeShiftRequestPC shiftRequestPC;

	public FlexTimeRequestDetailsPopupPM(RequestContext context) {
		super(context, RmKeys.RH_FLEXTIME_DETAILS);
		modelHandler = new FlexTimeRequestsMH();
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, FlexTimeRequestDetailsPopupPM.class);
	}

	@Override
	protected RequestAggregate fetchRequest(ID requestID) throws RmHardValidationException, Exception {
		return modelHandler.getFTRequest(getRequestID());
	}

	@Override
	protected void loadSpecifiedData() throws Exception {
		m_valResults = m_request.getValidationResults(false);
		m_toTypeName = TimeOffRequestsMH.getTimeOffTypeName(m_context, (TORequest) m_request);
	}

	public void initChoiceList() {
		TORequest toRequest = (TORequest) m_request;
		m_choiceListPC = createNewList();

		m_choiceListPC.setTableAttribute("summary",i18n(m_bundle, RmWebBundleKey.TO_REQUEST_POPUP_CHOICES_TABLE_SUMMARY));

		String reqStatus = toRequest.getRequestStatus();
		ID selectedChoiceID = toRequest.getApprovedChoiceID();
		TimeOffRequestUtil.TimeOffChoiceListPCConfig config = new TimeOffRequestUtil.TimeOffChoiceListPCConfig()
																	.setIsSelectable(false)
																	.setIsShowSelectedInfo(false)
																	.setIsShowAlert(false)
																	.setIsEnforceCustomTimeOffHoursFormat(true)
																	.setIsRowNumberEnabled(false)
																	.setTimeOffChoiceLabelBundleKey(RmWebBundleKey.FLEXTIME_CHOICES);
		TimeOffRequestUtil.initTimeOffChoiceListPC(m_context, m_choiceListPC, toRequest, selectedChoiceID,
				m_common_bundle, m_bundle, config);
	}

	private void initMakeupList() {
		TORequest toRequest = (TORequest) m_request;

		shiftRequestPC = new FlexTimeShiftRequestPC(getRequestContext(), SHIFT_REQUEST_PC, toRequest, true);
		shiftRequestPC.initializeModel(toRequest, toRequest.getEmployeeID(), null);
		try {
			shiftRequestPC.initializeContent();
		} catch (Exception e) {
			LOG.l7dError("FlexTimeRequestDetailsPopupPM.initMakeupList", e);
		}

		m_containerList.add(shiftRequestPC);
	}

	/**
	 * Initialize Main List
	 */
	@Override
	protected void initMainList() {
		super.initMainList();

		String title = i18n(m_bundle, RmWebBundleKey.REQUEST_SUMMARY);
		m_mainContainerPC.setTitle(title);

		m_mainContainerPC.setSummary(i18n(m_bundle, RmWebBundleKey.REQUEST_SUMMARY_TABLE));

		TORequest toRequest = (TORequest) m_request;
		m_mainContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.EMPLOYEE_NAME_LABEL),
				getEmployeeName(toRequest.getEmployeeID()));

		m_mainContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.TIMEOFF_TYPE_LABEL), m_toTypeName);
	}

	@Override
	protected void initContent() {
		initMainList();
		initChoiceList();
		initMakeupList();
		initAlertList();
		initAuditTrailList();
	}

	@Override
	protected String getContentTitleItemName() {
		return i18n(m_bundle, RmWebBundleKey.REQUEST_TYPE_FLEX_TIME);
	}
}
