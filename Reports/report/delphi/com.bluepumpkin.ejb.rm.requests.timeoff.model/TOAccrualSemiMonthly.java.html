package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.util.*;

public class TOAccrualSemiMonthly extends TOAccrual {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public static final String scheduleOfAccrualStr = "Semi-Monthly";

	public void incrementPeriod(Calendar calStart) {
		if (calStart.get(Calendar.DAY_OF_MONTH) >= 15) {
			calStart.set(Calendar.DAY_OF_MONTH, 1);
			calStart.add(Calendar.MONTH, 1);                                                   
		} else {
			calStart.set(Calendar.DAY_OF_MONTH, 15);
		}
	}

	public void decrementPeriod(Calendar calStart) {
		if (calStart.get(Calendar.DAY_OF_MONTH) > 15) { // if more than 15 then set to 15
			calStart.set(Calendar.DAY_OF_MONTH, 15);
		} else if (calStart.get(Calendar.DAY_OF_MONTH) == 1) {
			calStart.add(Calendar.MONTH, -1);     //if 1st then set to 15th of prev month
			calStart.set(Calendar.DAY_OF_MONTH, 15);
		} else {                                   //if from from 2to 15 then set to 1
			calStart.set(Calendar.DAY_OF_MONTH, 1);
		}
	}

	protected void initCalendarToStartOfPeriod(Calendar calStart) {
		super.initCalendarToStartOfPeriod(calStart);
		if (calStart.get(Calendar.DAY_OF_MONTH) >= 15) {
			calStart.set(Calendar.DAY_OF_MONTH, 15);
		}else{
			calStart.set(Calendar.DAY_OF_MONTH, 1);
		}
	}

}
