package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeTimeOffAccrued;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeTimeOffYearly;

import java.io.Serializable;

/**
 * Value Object for One Time Off Activity summary data in hours for UI display
 */
public class TimeOffActivitySummary implements Serializable {

	/**
	 * Color with which the activity is displayed on the screen - represented
	 * as the standard 6-digit hexadecimal RGP value.  Same value as stored in
	 * COLOR column of ACTIVITY table.
	 */
	private String m_activityColor;

	/**
	 * Short, unique name for the activity.  Same value as stored in NAME
	 * column of ACTIVITY table.
	 */
	private String m_activityName;

	public TimeOffActivitySummary(ID m_activityID, ActivityCategory m_activityCategory, String m_activityName, String m_activityColor) {
		this.m_activityID = m_activityID;
		this.m_activityCategory = m_activityCategory;
		this.m_activityName = m_activityName;
		this.m_activityColor = m_activityColor;
	}

	public ID getActivityID() {
		return m_activityID;
	}

	public void setActivityID(ID m_activityID) {
		this.m_activityID = m_activityID;
	}

	public ActivityCategory getActivityCategory() {
		return m_activityCategory;
	}

	public void setActivityCategory(ActivityCategory m_activityCategory) {
		this.m_activityCategory = m_activityCategory;
	}

	private ActivityCategory m_activityCategory;

	private ID m_activityID;

	/**
	 * Total hours allowed for a type for a given employee
	 */
	private float m_total = 0.0f;

	/**
	 * Hours used to date by given employee
	 */
	private float m_used = 0.0f;

	/**
	 * Hours scheduled for employee
	 */
	private float m_scheduled = 0.0f;

	/**
	 * Hours remaining for the year for this activity for a given employee
	 */
	private float m_remaining = 0.0f;


	public float getCarryOver() {
		return m_CarryOver;
	}

	public void setCarryOver(float carryover) {
		this.m_CarryOver = carryover;
	}

	/**
	 * Hours carried forward from previous year for this activity for a given employee
	 */
	private float m_CarryOver = 0.0f;


	/**
	 * Number of hours pending for this activity for employee.  This is computed
	 * by totalling the first TOChoice for all requests currently pending for
	 * the given employee.
	 */
	private float m_pending = 0.0f;

	private static final long serialVersionUID = 3277209927817594850L;

   	/**
	 * Story 5861 - Time Summary/My Schedule Summary Time-Off Changes
	 * The last updated EmployeeTimeOffAccrued info.
	 */
	private EmployeeTimeOffAccrued m_accruedTO;

	/**
	 * Default constructor
	 */
	public TimeOffActivitySummary() {
	}//constructor

	/**
	 * Loaded constructor
	 */
	public TimeOffActivitySummary(ID activityID, String activityColor, String activityName,
	                              float total, float used, float scheduled, float remaining, float pending) {
		m_activityID = activityID;
		m_activityColor = activityColor;
		m_activityName = activityName;
		m_total = total;
		m_used = used;
		m_scheduled = scheduled;
		m_remaining = remaining;
		m_pending = pending;
	}//constructor


	//===== getters and setters
	public void setActivityColor(String activityColor) {
		m_activityColor = activityColor;
	}

	public String getActivityColor() {
		return m_activityColor;
	}

	public void setActivityName(String activityName) {
		m_activityName = activityName;
	}

	public String getActivityName() {
		return m_activityName;
	}

	public void setTotal(float value) {
		m_total = value;
	}

	public void addToTotal(float value) {
		m_total += value;
	}

	public float getTotal() {
		return m_total;
	}

	public void setUsed(float value) {
		m_used = value;
	}

	public void addToUsed(float value) {
		m_used += value;
	}

	public float getUsed() {
		return m_used;
	}

	public void setScheduled(float value) {
		m_scheduled = value;
	}

	public void addToScheduled(float value) {
		m_scheduled += value;
	}

	public float getScheduled() {
		return m_scheduled;
	}

	public void setRemaining(float value) {
		m_remaining = value;
	}

	public void addToRemaining(float value) {
		m_remaining += value;
	}

	public float getRemaining() {
		return m_remaining;
	}

	public void setPending(float value) {
		m_pending = value;
	}

	public void addToPending(float value) {
		m_pending += value;
	}

	public float getPending() {
		return m_pending;
	}

	/*This is required as a unique key to be used instead of getActivityID() used earlier in all APIS for uniquness
	for multiedits or hashmaps */
	public String getActivity_ActCategoryComb() {
		return Activity.getActivity_ActCategoryComb(getActivityID(), getActivityCategory() != null ? getActivityCategory().getID() : null);
	}

	public String toString() {
		return "ACT-ACTCATID=" + getActivity_ActCategoryComb() +
		        "\ttotal = " + m_total +
		        "\tused = " + m_used +
		        "\tscheduled = " + m_scheduled +
		        "\tremaining = " + m_remaining +
		        "\tCarryOver = " + m_CarryOver +
		        "\tpending = " + m_pending;
	}

	public boolean compareWithTimeOffYearly(EmployeeTimeOffYearly yrly) {
		return (yrly != null
		        && compareDouble(yrly.getCarryover(), getCarryOver())
		        && compareDouble(yrly.getRemaining(), getRemaining())
		        && compareDouble(yrly.getUsed(), getUsed())
		        && compareDouble(yrly.getScheduled(), getScheduled())
		        && compareDouble(yrly.getPending(), getPending()));
	}

	public boolean compareDouble(double num1, double num2) {
		return (Math.abs(num1 - num2) < 0.0001);
	}

	/**
	 * Story 5861 - Time Summary/My Schedule Summary Time-Off Changes
	 * @return The last updated EmployeeTimeOffAccrued info.
	 */
	public EmployeeTimeOffAccrued getEmployeeTimeOffAccrued() {
		return m_accruedTO;
	}

	/**
	 * Story 5861 - Time Summary/My Schedule Summary Time-Off Changes
	 * @return Set the last updated EmployeeTimeOffAccrued info.
	 */
	public void setEmployeeTimeOffAccrued(EmployeeTimeOffAccrued m_accruedTO) {
		this.m_accruedTO = m_accruedTO;
	}

}//class TimeOffActivitySummary
