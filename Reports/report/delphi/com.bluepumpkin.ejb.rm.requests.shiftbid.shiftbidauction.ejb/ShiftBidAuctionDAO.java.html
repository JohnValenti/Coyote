    /**
     * The From and Where clause are customized by:
     * <li> if detailLevel == DL_CAMPAIGN.
     * <li> filter by EMPID
     * <li> filter by campaignIDs
     *
     * @see com.bluepumpkin.ejb.bbm.dao.DAOBase#appendFromWhereStatement(java.lang.StringBuffer)
     */
    @Override
	protected boolean appendFromWhereStatement(StringBuffer strSQL)   {
        try {
            boolean whereAppended = super.appendFromWhereStatement(strSQL);

            // Append FROM clauses
            boolean fetchCampaign = (getDetailLevel() & ShiftBidAuction.DL_CAMPAIGN) != 0;

            if (fetchCampaign) {
                strSQL.append(", CAMPAIGN CAMP, TIMEZONE TZ ");
            }

            Map filterExprsMap = getFilterExprsMap();
            boolean filterByEmpAMID = filterExprsMap.containsKey(FilterExpr.LHS_EMPID);
            if (filterByEmpAMID) {
                strSQL.append(", SHIFTBIDDER SB ");
            }

            // Append where clauses.

            // One way the DL_INCLUDE_EXPIRED flag can be set is when expired shift bid requests
            // are loaded.  In this case, ShiftBidREquestDAO has DL_INCLUDE_EXPIRED set as its
            // detail level.  SBReqDAO propogates this detailLevel to the ShiftBidAuctionDAO it
            // creates as a child DAO.
            boolean excExpired = (getDetailLevel() & ShiftBidAuction.DL_EXCLUDE_EXPIRED_AUCTION) != 0;
            // if expired requests are to be excluded, choose only auctions that start in the future.
            if (excExpired) {
                whereAppended = DAOUtil.appendToWhereClause(strSQL,
                    " A.STARTTIME > '" + JdmoUtil.formatDBString(new Date()) + '\'', whereAppended);
            }

       if (fetchCampaign) {
                whereAppended = DAOUtil.appendToWhereClause(strSQL,
                    COLNAME_CAMPAIGNID + " = CAMP.SID AND CAMP.TIMEZONEID = TZ.ID ", whereAppended);
            }

            boolean filterByCampIDs = filterExprsMap.containsKey(FilterExpr.LHS_CAMPAIGNIDS);
            if (filterByCampIDs) {
                FilterExpr empIDFilterExpr = (FilterExpr) filterExprsMap.get(FilterExpr.LHS_CAMPAIGNIDS);
                Collection campIDs = (Collection) empIDFilterExpr.getRHS();

                whereAppended = DAOUtil.appendToWhereClause(strSQL,
                    COLNAME_CAMPAIGNID + " IN " + m_dmo.createInClause(campIDs), whereAppended);
            }

            if (filterByEmpAMID) {
                // get employee ID
                FilterExpr empIDFilterExpr = (FilterExpr) filterExprsMap.get(FilterExpr.LHS_EMPID);
                ID empID = (ID) empIDFilterExpr.getRHS();

                // clear the filterExpr for employeeID
                clearFilterExpr(FilterExpr.LHS_EMPID);

                // given EMPID -> ShiftBidder.EmployeeID --> ShiftBidder.ShiftBidAuctionID.
				whereAppended = DAOUtil.appendToWhereClause(strSQL, " A.ID = SB.SHIFTBIDAUCTIONID AND " +
                    " SB.EMPLOYEEID = " + JdmoUtil.asSqlLiteral(empID) + " AND SB.STATUS = " + ShiftBidder.STATUS_ADDED, whereAppended);
            }

            return whereAppended;
        } catch (Exception e) {
            throw new RuntimeException(e.getMessage(), e);
        }
    }

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.bbm.dao.DAOBase#readObjectFromDB(com.bluepumpkin.common.jdmo.JdmoRowset)
     */
    @Override
	public ValueObjectBase readObjectFromDB(JdmoRowset rs) throws Exception {
        ShiftBidAuction sbAuction = (ShiftBidAuction) super.readObjectFromDB(rs);

        boolean fetchCampaign = (getDetailLevel() & ShiftBidAuction.DL_CAMPAIGN) != 0;
        if (fetchCampaign)  {
            SchedulingPeriod sp = new SchedulingPeriod();
            sp.setID(rs.getID("SPID")); //from shiftbidauction table.
            sp.setCampaignID(rs.getID("CAMPSID"));
            sp.setStartTime(TimeZoneUtil.toDate(rs.getTimestamp("STARTTIME"))); //equivalent to SP.FROMDATE
            sp.setEndTime(TimeZoneUtil.toDate(rs.getTimestamp("ENDTIME"))); //equivalent to SP.TODATE
            sbAuction.getSetters().setSP(sp);

            Campaign camp = new Campaign();
            camp.setID(rs.getID("CAMPSID"));
            camp.setName(rs.getString("CAMPNAME"));
            String tzName = rs.getString("STANDARDNAME");
            camp.setTimeZone(TimeZone.getTimeZone(tzName));
            sbAuction.getSetters().setCampaign(camp);
        }

        return sbAuction;
    }

    /**
     * Get Auctions for given campaign effectivities.  'null' for empID implies all auctions
     * defined in the system. It is wrong since QC203581:Multi-Tenant environment supervisor
     * without campaign scope assignment can see all auctions.
     * It should not return all auctions in the systems. It should return empty.
     * @param userID
     * @param detailLevel
     * @return
     */
    public Collection getAuctionsForCampaignIDs(Collection campIDs, long detailLevel) throws Exception {
        saveAndSetDetailLevel(detailLevel);
        try  {
            if (campIDs != null && !campIDs.isEmpty()) {
				setFilterExpr(new FilterExpr(FilterExpr.LHS_CAMPAIGNIDS, FilterExpr.OP_IN, campIDs), true);
				return getObjects(null);
			} else {
				return Collections.emptyList();
			}

        } finally  {
            resetDetailLevel();
            clearFilterExprsMap();
        }
    }

        	if (employeeTypeSID == null) {
            	whereClause.append(" AND A.EMPLOYEETYPESID IS NULL").append(employeeTypeSID);
        	} else {
            	whereClause.append(" AND A.EMPLOYEETYPESID = ").append(employeeTypeSID);
        	}

        } finally  {
            resetDetailLevel();
        }
    }

    /**
     * @param spID
     */
    public Collection getAuctionsForSPIDORGIDEMTYSID(ID spID, Collection<ID> organizationIDs, ID employeeTypeSID, long detailLevel) throws BbmFinderException {
        saveAndSetDetailLevel(detailLevel);

        try  {
        	StringBuilder whereClause = new StringBuilder();
        	whereClause.append(" A.SPID = ").append(spID);
        	if (organizationIDs == null || organizationIDs.isEmpty()) {
        		whereClause.append(" AND A.ORGANIZATIONID IS NULL");
        	} else {
        		whereClause.append(" AND A.ORGANIZATIONID in ").append(m_dmo.createInClause(organizationIDs));
        	}

        	if (employeeTypeSID == null) {
            	whereClause.append(" AND (A.EMPLOYEETYPESID IS NULL OR A.EMPLOYEETYPESID IS NOT NULL)");
        	} else {
            	whereClause.append(" AND (A.EMPLOYEETYPESID = ").append(employeeTypeSID).append(" OR A.EMPLOYEETYPESID IS NULL) ");
        	}
            return getObjects(whereClause.toString());
        } catch (JdmoException e) {
        	throw new BbmFinderException(e);
        } finally  {
            resetDetailLevel();
        }
    }
    /**
	* Return the pair of Id and Names of Shift Bid Auction Names in given organization Ids-employees of those organizations added to those shift bid auctions
	* @param organizationIDs
	*/
    public Collection<StringsPair> getAuctionListForFilter(Collection<ID> orgIds) throws BbmFinderException{
    	try{
    		Collection<StringsPair> shiftBidAuctions = new ArrayList<StringsPair>();
    		StringBuilder strSQL = new StringBuilder(200);
    		strSQL.append("SELECT distinct A.ID,A.NAME  from SHIFTBIDAUCTION A, SHIFTBIDDER B,WORKRESOURCEORGANIZATION WRO ");
    		strSQL.append("Where A.ID = b.SHIFTBIDAUCTIONID and  B.EMPLOYEEID = WRO.WORKRESOURCEID and  WRO.ORGANIZATIONID in ");
    		strSQL.append(m_dmo.createInClause(orgIds));
    		JdmoRowset rs = m_dmo.createRowset(strSQL.toString());
    		while (rs.next()) {
				String bidId = rs.getString("ID");
				String bidName = rs.getString("NAME");
				shiftBidAuctions.add(new StringsPair(bidId,bidName));
			}
    		return shiftBidAuctions;
    	}catch(JdmoException e){
    		throw new BbmFinderException(e);
    	}
    }
