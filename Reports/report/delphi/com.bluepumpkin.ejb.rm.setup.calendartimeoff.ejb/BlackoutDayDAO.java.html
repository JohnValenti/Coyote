public class BlackoutDayDAO extends DAOBase {

	private static FieldInfo m_fieldInfo = new BlackoutDayFieldInfo();

	private static TimeZone s_gmt = TimeZone.getTimeZone("GMT");

	@Override
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public BlackoutDayDAO() {
		super();
	}

	public BlackoutDayDAO(Jdmo dmo) {
		super(dmo);
	}

	@Override
	protected ValueObjectBase createValueObject() {
		return (new BlackoutDay());
	}

	public BlackoutDay[] findWithinMonth(ID pTOPoolID, MonthRange monthRange) throws Exception {
		BlackoutDay blackoutDays[] = new BlackoutDay[31];
		Calendar c = new GregorianCalendar();
		String strQuery = " SELECT A.ID, A.TOPOOLID, A.DAYDATE FROM TIMEOFFBLACKOUTDAY A " +
		        " WHERE A.TOPOOLID = " + JdmoUtil.asSqlLiteral(pTOPoolID) +
		        " AND A.DAYDATE >= '" + JdmoUtil.formatDBString(monthRange.startDate) +
		        "' AND A.DAYDATE < '" + JdmoUtil.formatDBString(monthRange.endDate) +
		        "'";
		JdmoRowset rs = m_dmo.createRowset(strQuery);
		while (rs.next()) {
			BlackoutDay item = (BlackoutDay) readObjectFromDB(rs);
			c.setTime(item.getBlackoutDate());
			int dayOfMonth = c.get(Calendar.DAY_OF_MONTH) - c.getMinimum(Calendar.DAY_OF_MONTH);
			blackoutDays[dayOfMonth] = item;
		}
		return blackoutDays;
	}

	/**
	 * Gets the employee's effective blackout days for the time range.
	 *
	 * @param employeeID
	 * @param startTime
	 * @param endTime
	 *            Exclusive end time.
	 *  @param ID activityID for multiple pools --pools are tied to activities
	 * @return
	 * @throws Exception
	 */
	public List<BlackoutDay> getEffectiveBlackoutDaysForEmployee(ID employeeID, Date startTime, Date endTime, ID activityID) throws Exception {
		List<BlackoutDay> blackoutDays = new ArrayList<BlackoutDay>();
		JdmoQuery jdmoQuery = m_dmo.createQuery("RM_GetEffectiveBlackoutDaysForEmployee", Jdmo.STORPROC_QUERY);
//JT
		jdmoQuery.setParID(1, employeeID);
		jdmoQuery.setParTimestamp(2, new Timestamp(startTime.getTime()));
		jdmoQuery.setParTimestamp(3, new Timestamp(endTime.getTime()));
		if(activityID != null){
			jdmoQuery.setParID(4, activityID);//NOSONAR
		} else{
			jdmoQuery.setParNull(4, Types.INTEGER);//NOSONAR
		}

		
		JdmoRowset rs = m_dmo.createRowset(jdmoQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
		while (rs.next()) {
			blackoutDays.add((BlackoutDay) readObjectFromDB(rs));
		}
		return blackoutDays;
	}

	public void deleteInRange(ID pTOPoolID, LocalDate start, LocalDate end) throws Exception {
		// Make sure that both start and end or not null and that end is after start
		if (start == null || end == null || !end.after(start)) {
			throw new IllegalArgumentException("invalid start or end date: " + start + " - " + end);
		}
		StringBuffer strQuery = new StringBuffer(512);
		//Fix the 3rd of QC 196696
	    strQuery.append(" SELECT ID FROM  TIMEOFFBLACKOUTDAY A where A.TOPOOLID = ").append(JdmoUtil.asSqlLiteral(pTOPoolID));
		strQuery.append(" AND A.DAYDATE >= '").append(JdmoUtil.formatDBString(start.getTime(s_gmt)));
		strQuery.append("' AND A.DAYDATE <= '").append(JdmoUtil.formatDBString(end.getTime(s_gmt))).append("'");
		deleteObjects(strQuery.toString());
	}

	public Collection<BlackoutDay> findInRange(ID pTOPoolID, LocalDate start, LocalDate end) throws BbmFinderException {
		// Make sure that both start and end or not null and that end is after start
		if (start == null || end == null || !end.after(start)) {
			throw new IllegalArgumentException("invalid start or end date: " + start + " - " + end);
		}
		StringBuffer strQuery = new StringBuffer(512);
		strQuery.append("A.TOPOOLID = ").append(JdmoUtil.asSqlLiteral(pTOPoolID));
		strQuery.append(" AND A.DAYDATE >= '").append(JdmoUtil.formatDBString(start.getTime(s_gmt)));
		strQuery.append("' AND A.DAYDATE <= '").append(JdmoUtil.formatDBString(end.getTime(s_gmt)));
		strQuery.append("' ORDER BY A.DAYDATE ASC");
		return getObjects(strQuery.toString());
	}


	/**
	 * Save a month's worth of blackout days to a group of organizations.
	 * The array must reference a complete months worth of data.
	 *
	 * @param pTOPoolID       - TO Pool ID.
	 * @param monthRange      - the start and end time for the month
	 * @param newBlackoutDays - the blackout days to save.
	 * if the array is empty then all the values for blackoutday for the given month will be erased from DB.
	 */
	public void saveBlackoutDays(ID pTOPoolID, MonthRange monthRange, BlackoutDay newBlackoutDays[]) throws Exception {
		ArrayList createList = new ArrayList();
		HashMap oldBlackoutDaysMap = new HashMap(31);
		BlackoutDay oldBlackoutDays[] = findWithinMonth(pTOPoolID, monthRange);
		int l = oldBlackoutDays != null ? oldBlackoutDays.length : 0;
		for (int i = 0; i < l; i++) {
			BlackoutDay oldItem = oldBlackoutDays[i];
			if (oldItem != null&&oldItem.getBlackoutLocalDate()!=null) {
				oldBlackoutDaysMap.put(oldItem.getBlackoutLocalDate(), oldItem);
			}
		}
		l = newBlackoutDays != null ? newBlackoutDays.length : 0;
		for (int i = 0; i < l; i++) {
			BlackoutDay newBlackoutDay = newBlackoutDays[i];
			if (newBlackoutDay != null && newBlackoutDay.getBlackoutLocalDate()!=null) {
				BlackoutDay oldBlackoutDay = (BlackoutDay) oldBlackoutDaysMap.get(newBlackoutDay.getBlackoutLocalDate());
				if (oldBlackoutDay == null) {                 //item exists only in new so create it
					newBlackoutDay.setTOPoolID(pTOPoolID);
					newBlackoutDay.setID(null);
					createList.add(newBlackoutDay);
				} else {
					//Item exists in new and old so no new entries to be updated or inserted for this day
					//********** Be sure to remove it from the OLD List because we will use the old list that remains
					// after the newBlackoutDays iteration for deletion purposes.
					oldBlackoutDaysMap.remove(oldBlackoutDay.getBlackoutLocalDate());
				}
			}
		}
		if (!createList.isEmpty()) {
			createObjects(createList);
		}
		if (oldBlackoutDaysMap != null && !oldBlackoutDaysMap.isEmpty()) {
			deleteObjects(ValueObjectUtil.getIDObjectMap(oldBlackoutDaysMap.values()).keySet());
		}
	}

	/**
	 * Save blackout days for a given date range to a group of organizations.
	 *
	 * @param pTOPoolID - TOPool Id.
	 * @param start-    Localdate start of range
	 * @param end-      Localdate end of range
	 * @param delete    - whether to delet blackout days.
	 */
	public void saveBlackoutDaysForRange(ID pTOPoolID, LocalDate start, LocalDate end, boolean delete) throws Exception {
		ArrayList createList = new ArrayList();
		deleteInRange(pTOPoolID, start, end);
		LocalDate sDay = new LocalDate(start);
		LocalDate eDay = new LocalDate(end);
		while (eDay.after(sDay) && !delete) {
			BlackoutDay newItem = new BlackoutDay();
			newItem.setTOPoolID(pTOPoolID);
			newItem.setBlackoutDate(sDay);
			newItem.setID(null);
			createList.add(newItem);
			sDay.add(Calendar.DATE, 1);
		}
		if (!createList.isEmpty()) {
			createObjects(createList);
		}
	}
}
