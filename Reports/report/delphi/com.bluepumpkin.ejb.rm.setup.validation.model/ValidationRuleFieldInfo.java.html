package com.bluepumpkin.ejb.rm.setup.validation.model;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;

/**
 * Title:        ValidationRuleFieldInfo
 * Description:  meta data class for ValidationRule
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc.
 * @author       Shailendra Sharma
 * @version 	 1.0
 */

public class ValidationRuleFieldInfo extends FieldInfo {    

	/* ID for the Validation Rule */
	public static final int VALIDATIONRULE_I_ID = 0;

	/* Request Type for this Validation Rule. It's values are 
	 * the specified in Request object.
	 * @see com.bluepumpkin.ejb.rm.requests.common.model.Request
	 */
	public static final int VALIDATIONRULE_S_REQUESTTYPE = 1;

	/* Name of the validator class for the validation*/
	public static final int VALIDATIONRULE_S_VALIDATOR = 2;

	/* Resource bundle name for internationalization. */
	public static final int VALIDATIONRULE_S_BUNDLENAME = 3;

	/* Resource bundle name key for the resource bundle. */
	public static final int VALIDATIONRULE_S_BUNDLEKEY = 4;

	/* boolean value for external type*/
	public static final int VALIDATIONRULE_B_ISEXTERNAL = 5;

	protected static final int VALIDATIONRULE_NUMOFFIELDS = 6;

	private static FieldInfoEntry m_arrFieldInfo[] =
		new FieldInfoEntry[VALIDATIONRULE_NUMOFFIELDS];

	private static String m_strTableName = "VALIDATIONRULE";

    static 
    {
        RequestUtil.initFieldInfoEntries(m_arrFieldInfo, ValidationRuleFieldInfo.class, 
            "VALIDATIONRULE_", VALIDATIONRULE_NUMOFFIELDS);
    }
    
	public ValidationRuleFieldInfo() {
		super();
	}

	// implement abstract methods in parent class
	public String getTableName() {
		return m_strTableName;
	}
	public int getIDFieldIndex() {
		return VALIDATIONRULE_I_ID;
	}
	public FieldInfoEntry [] getFieldInfo() {
		return m_arrFieldInfo;
	}
}
