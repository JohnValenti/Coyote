	public static final int MAX_HEAT_MAP_TOLERANCE = 100;

	/**
	 * Contact queue above required tolerance setting, for use in the web calendar FTE ribbon.
	 * @param value : range 0 - 100.
	 */
	public void setContactAboveReqTolerance(int value) {
		if (value < 0) {
			value = 0;
