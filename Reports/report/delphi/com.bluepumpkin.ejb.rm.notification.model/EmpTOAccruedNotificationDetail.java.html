package com.bluepumpkin.ejb.rm.notification.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;

/**
 * This class encapsulates all the necessary details to send a Emp TO Balance
 * change notification (notification when balance is imported using agent sync).
 * <p/>
 * This class lives in BBM, rather than RM, because of deserialization problems
 * in the Notification Handler message driven bean. When an instance of this
 * class, if defined in RM, is passed by RM to BBM, BBM's notification handler
 * MDB genearates an "Unable to find class for RequestNotifiationDetail"
 * exception. Adding RmEjbTierUtils.jar to notificationEjb.jar's manifest class
 * path fixes this.
 * <p/>
 * See class {@link com.bluepumpkin.ejb.bbm.notifyrules } for more details.
 * <p/>
 * Title: Blue Pumpkin Software Request Management Module Copyright: Copyright
 * (c) 2001-2004 Company: Blue Pumpkin Software, inc
 *
 * @author Sameet Joshi
 */
public class EmpTOAccruedNotificationDetail implements Serializable {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private Map<ID, List<EmployeeTimeOffAccrued>> empTOAccrdMap;
	private Map<ID, List<EmployeeTimeOffAccrued>> orgTOAccrdMap;

	public Map<ID, List<EmployeeTimeOffAccrued>> getOrgTOAccrdMap() {
		return orgTOAccrdMap;
	}

	public Map<ID, List<EmployeeTimeOffAccrued>> getEmpTOAccrdMap() {
		return empTOAccrdMap;
	}

	public EmpTOAccruedNotificationDetail(Collection<EmployeeTimeOffAccrued> changedTOAccrued) throws Exception {
		WorkResourceManager wrm = BbmManagerFactory.getWorkResourceManager();
		setEmpTOAccrdMap(changedTOAccrued, wrm);
		setOrgTOAccruedMap(empTOAccrdMap, wrm);
	}

	private void setEmpTOAccrdMap(Collection<EmployeeTimeOffAccrued> accruedCol, WorkResourceManager wrm) throws Exception {
		if (wrm == null)
			wrm = BbmManagerFactory.getWorkResourceManager();
		this.empTOAccrdMap = DAOUtil.groupByField(EmployeeTimeOffAccruedFieldInfo.TMOFFACCRUED_EMPLOYEEID, accruedCol);
	}

	private void setOrgTOAccruedMap(Map<ID, List<EmployeeTimeOffAccrued>> empMap, WorkResourceManager wrm) throws Exception {
		orgTOAccrdMap = new HashMap();
		Map workResIDToworkResAssnsMap = wrm.getWRAssignmentsWithTimeZone(empMap.keySet());
		// select the organization where Accrual date falls between org start
		// date and end date
		Set<ID> empSet = workResIDToworkResAssnsMap.keySet();
		for (ID empID : empSet) {

			List<EmployeeTimeOffAccrued> empTOAccrCol = empMap.get(empID);
			if (empTOAccrCol == null || empTOAccrCol.isEmpty()) {
				continue;
			}
			WorkResourceAssignment wra = null;
			for (EmployeeTimeOffAccrued accrd : empTOAccrCol) {
				if (wra == null || !fallsIn(accrd.getAccruedAtDate(), wra.getStartTime(), wra.getEndTime())) {
					wra = getWRAForStartDate((Collection) workResIDToworkResAssnsMap.get(empID), empID, accrd.getAccruedAtDate());
				}
				List<EmployeeTimeOffAccrued> accruedList = orgTOAccrdMap.get(wra.getOrganizationID());
				if (accruedList == null) {
					accruedList = new ArrayList<EmployeeTimeOffAccrued>();
					orgTOAccrdMap.put(wra.getOrganizationID(), accruedList);
				}
				accruedList.add(accrd);
			}
		}
	}

	private static WorkResourceAssignment getWRAForStartDate(Collection wrAssns, ID empID, Date date) throws Exception {
		WorkResourceAssignment wra = null;
		for (Iterator iterator = wrAssns.iterator(); iterator.hasNext();) {
			wra = (WorkResourceAssignment) iterator.next();
			if (fallsIn(date, wra.getStartTime(), wra.getEndTime())) {
				break;
			}
			wra = null;
		}
		if (wra == null) {
			throw new RmRuntimeException(empID + ": Employee is not assigned to any organization; date=" + date);
		}
		return wra;
	}


}
