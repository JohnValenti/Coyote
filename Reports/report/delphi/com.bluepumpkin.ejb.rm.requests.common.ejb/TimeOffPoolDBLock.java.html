package com.bluepumpkin.ejb.rm.requests.common.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.config.ConfigKey;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOPool;
import com.bluepumpkin.ejb.rm.util.JdmoPreparedStatement;

/** 
 * Serialize auto processing of time off requests by putting an update lock on an employee's time off pool record if the option is enabled 
 * in BPConfig
 */

public class TimeOffPoolDBLock {

	private static final boolean SERIALIZE_AUTO_PROCESSING;

	static {
		SERIALIZE_AUTO_PROCESSING = isSerialzeAutoProcessingEnabled();
	}

	private TimeOffPoolDBLock() {

	}

	private static boolean isSerialzeAutoProcessingEnabled() {
		try {
			return BbmManagerFactory.getDBConfigManager().getBooleanValue(ConfigKey.SERIALIZE_AUTO_PROCESSING);
		} catch (Exception ex) { // NOSONAR
			return false;
		}
	}

	public static void getTimeOffPoolLock(RequestAggregate request) throws Exception { // NOSONAR

		if (!SERIALIZE_AUTO_PROCESSING || request == null || !request.isTimeOffRequest()) {
			return;
		}

		TORequest timeOffRequest = (TORequest) request;
		TOPool pool = timeOffRequest.getCache().getTOPool();
		if (pool == null) {
			return;
		}

		getUpdateLock(pool.getId());
	}

	private static void getUpdateLock(ID poolID) {
		Jdmo dmo = null;
		try {
			dmo = new Jdmo();
			//This puts an update lock on the TOPOOL row with the specified ID till the end of the current transaction. 
			//Any updates to this row, or a select with(updlock) for this same ID will block till the transaction that created the lock is done
			JdmoPreparedStatement.createRowset(dmo, "select ID from TOPOOL with(updlock) where ID=?", poolID);
		} catch (JdmoException e) {
			throw new RmRuntimeException("Unable to acquire a lock on the employee's time off pool", e);
		} finally {
			if (dmo != null) {
				dmo.cleanUp();
			}
		}

	}

}
