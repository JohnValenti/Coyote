        assert (m_sortDirection == SupportNavigation.SORT_ASCENDING || m_sortDirection == SupportNavigation.SORT_DESCENDING);

    protected void addDetailLevel(long detailLevel)  {
        m_detailLevel |= detailLevel;
    }
    
    void saveAndSetDetailLevel(long detailLevel)  {
        m_savedDetailLevel = m_detailLevel;
        m_detailLevel = detailLevel | RequestDetailLevel.DL_BASIC; //DL_BASIC must always be present.
    }
    
    void resetDetailLevel()  {
        m_detailLevel = m_savedDetailLevel;
    }
    
    long getDetailLevel()  {
        return m_detailLevel;
    }
    
    void setFilterExpr(FilterExpr filterExpr)  {
        setFilterExpr(filterExpr, false);
    }
    
    void setFilterExpr(FilterExpr filterExpr, boolean clearMap)  {
        m_filterExprsMap = (m_filterExprsMap == null)?new HashMap():m_filterExprsMap;

        if (clearMap) m_filterExprsMap.clear();
        
        m_filterExprsMap.put(filterExpr.getLHS(), filterExpr);    
    }
    
    void clearFilterExpr(Object LHSKey)  {
        if (m_filterExprsMap != null) m_filterExprsMap.remove(LHSKey);    
    }
    
    void clearFilterExprsMap()  {
        if (m_filterExprsMap != null) m_filterExprsMap.clear();
    }
    
    Map getFilterExprsMap()  {
        return (m_filterExprsMap == null)? Collections.emptyMap() :m_filterExprsMap;
    }
