    /** @return true if need to load the child objects */
    @Override
	protected boolean needToLoadChildObjects()
    {
        long detailLevelToCheck = (Request.DL_AUDIT_TRAIL | Request.DL_EMPLOYEE);
        return (getDetailLevel() & detailLevelToCheck) != 0;
    }

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.bbm.dao.DAONode#needToLoadChildObjects(int)
     */
    @Override
	protected boolean needToLoadChildObjects(int childType) {
        boolean needToLoad = false;

        switch (childType) {
            case RequestFieldInfo.AUDIT_TRAIL_CHILD_TYPE:
                needToLoad = (getDetailLevel() & Request.DL_AUDIT_TRAIL) != 0; 
                break;
            case RequestFieldInfo.CHILD_TYPE_EMPLOYEE:
                needToLoad = (getDetailLevel() & Request.DL_EMPLOYEE) != 0;
                break;
        }
        
        return needToLoad;
    }
    
    /** create a child DAO object given the type */
    @Override
	protected DAOBase createChildDAO(int iType) {
        if (iType == RequestFieldInfo.AUDIT_TRAIL_CHILD_TYPE) {
            return new RequestAuditTrailDAO(getDMO());
        } else if (iType == RequestFieldInfo.CHILD_TYPE_EMPLOYEE) {
            return new EmployeeDAO(getDMO());
        }

        return null;
    }

    /**
     * Creates a row in the Request table, and an associated row in the
     * RequestAuditTrail table with <tt>comment</tt>.
     * @param request An instance of a Request object with data to be persisted.
     * @param comment A comment to be put in the audit trail.
     * @throws BbmCreateException
     * @throws BbmFinderException
     */
    public ID createRequest(Request request, String pAuditTrailNote, User createUser)
        throws Exception
    {
        try {
            // Set the last modified time and save the time value to use in the
            // audit trail record below
            Date dtNow = new Date();
            request.setLastModifiedAt(dtNow);

            // Set the modified by field with the login name
            request.setUpdateUser(createUser.getUserName());

            RequestAuditTrail raTrail = createAuditTrailObject(request.getRequestStatus(), 
                    dtNow, pAuditTrailNote, RmEjbBundleKey.CREATED, createUser.getID());
            //set status change to true
            raTrail.setIsStatusChange(true);
            request.setAuditTrail(raTrail);

            // create the request
            ID requestID = super.createObject(request);

            moveCreatedChildrenToPersisted(request);

            return requestID;
//        } c_atch (BbmCreateException e) {
//            m_cat.l7dError(RmEjbBundleKey.CREATE_FAILURE, new String[] { "REQUEST" });
//            t_hrow e;
        } finally {
            request.clearCreatedChildObjectMap(RequestFieldInfo.AUDIT_TRAIL_CHILD_TYPE);
        }
    }

    private void moveCreatedChildrenToPersisted(Request request) {
        // Get list of created children
        Collection createdChildren = request.getCreatedChildObjects(
                                            RequestFieldInfo.AUDIT_TRAIL_CHILD_TYPE);
        // Copy created children to the fetched children set.
        for (Iterator itr = createdChildren.iterator(); itr.hasNext(); ) {
            request.fillChildObject(RequestFieldInfo.AUDIT_TRAIL_CHILD_TYPE, (ValueObjectBase) itr.next());
        }

        // clear the created list after copy.
        request.clearCreatedChildObjectMap(RequestFieldInfo.AUDIT_TRAIL_CHILD_TYPE);
    }

    private long getDetailLevel() {
        return m_detailLevel;
    }

    private void saveAndSetDetailLevel(long detailLevel) {
        m_savedDetailLevel = m_detailLevel;
        m_detailLevel = detailLevel;
    }

    private void resetDetailLevel() {
        m_detailLevel = m_savedDetailLevel;
    }

    public Request getRequestByID(ID id, long detailLevel)
        throws BbmObjectNotFoundException, BbmFinderException

        saveAndSetDetailLevel(detailLevel);
        try  {
            return (Request) getObjectByID(id);
        } finally  {

        }
    }

    /**
     * Used by CommonRequestManagerEJB during group processing of requests.
     *
     * @return a Collection of Request objects whose ID is in ids.
     */
    public Collection getRequestsByIDs(Collection ids, long detailLevel)
        throws BbmFinderException
    {
        saveAndSetDetailLevel(detailLevel);
        try  {
            Collection coll = this.getObjectsByIDs(ids);
            return (coll != null)?coll:Collections.EMPTY_LIST;
        } finally  {
            resetDetailLevel();
        }
    }

    /**
     * Updates a Request object, also adding a new entry in the REQUESTAUDITTRAIL.
     *
     * @param request The Request to be updated.
     * @param comment A comment to be entered in the audit trail. If the comment
     * is null, the default comment of "updated" is entered in the audit trail.
     * @param checkStateTransition if <tt>true</tt> check if there is a state transition
     * and if so, if it is valid. When called from changeRequestState, this flag
     * should be set to <tt>false</tt>.
     * @throws BbmUpdateException
     * @throws MultiUserException
     */
    protected void updateRequest(Request request, String comment, User updateUser)
        throws Exception {
        if (request == null) {
            throw new BbmUpdateException(new NullPointerException());
        }

        try {
            Request oldRequest = (Request) getObjectByID(request.getID());

            // Copy the employee id if it is not set in the updating request
            if (request.getEmployeeID() == null) {
                request.setEmployeeID(oldRequest.getEmployeeID());
            }

			if(updateUser==null){ // to avoid null pointer exception thrown when update is invoked by system process like TO Waitlist Scan 
				 updateUser  = new SuperUserFacade();
			}
            // Set the last modified date and save the value for use in the audit trail
            Date dtNow = new Date();
            request.setLastModifiedAt(dtNow);

            // Set the modified by field with the login name
            request.setUpdateUser(updateUser.getUserName());

//            setExpirationDate(request);

            if (!request.isAuditTrailSet()) {
                RequestAuditTrail raTrail = createAuditTrailObject(request.getRequestStatus(),
                        dtNow, comment, RmEjbBundleKey.UPDATED, updateUser.getID());

                //set status change
                if (!oldRequest.getRequestStatus().equals(request.getRequestStatus()))
                    raTrail.setIsStatusChange(true);

                //add the created audit trail to the Request object.
                request.setAuditTrail(raTrail);
            }

            // update the request
            updateObject(request);

            moveCreatedChildrenToPersisted(request);

//        } c_atch (BbmCreateException e) {
//            t_hrow new BbmUpdateException(e);
//        } c_atch (BbmFinderException e) {
//            t_hrow new BbmUpdateException(e);
        } finally {
            request.clearCreatedChildObjectMap(RequestFieldInfo.AUDIT_TRAIL_CHILD_TYPE);
        }
    }

    public static RequestAuditTrail createAuditTrailObject(String status, Date dtNow, String auditTrailNote,

        if (StringUtil.isEmpty(auditTrailNote)) {
        	Localizer localizer = DefaultLocalizationManager.getLocalizer(userID);
            auditTrailNote = localizer.i18n(m_rmEjbResBundleInAppDefLocale, defaultCommentKey);
            //ESR  Bad translation (French)- ESR#4825726, translate to UTF-8 since this string will be stored in db.
            auditTrailNote = new String(auditTrailNote.getBytes("ISO-8859-1"), "UTF-8");
        }

        return new RequestAuditTrail(userID, status, dtNow, auditTrailNote);
    }

	/**
	 * Count number of request type, and status per employeeID within a time
	 * range of time off request dates
	 * 
	 * @param countParam
	 * @return
	 * @throws JdmoException
	 */
	public Map<ID, Integer> getNumberOfRequestPerPeriod(RequestCountParam countParam) throws JdmoException {
		Map<ID, Integer> map = new HashMap<ID, Integer>();
		StringBuilder sql = new StringBuilder();

		sql.append("select R.EMPLOYEEID, count(*) NUM from REQUEST R")
				.append(" join TIMEOFFREQUEST TOR on TOR.ID = R.ID")
				.append(" join TIMEOFFREQUESTCHOICE TOC on TOC.TIMEOFFREQUESTID = TOR.ID")
				.append(" where R.REQUESTSTATUS in ('").append(StringUtil.join(countParam.getStatuses(), "','"))
				.append("')").append(" and R.EMPLOYEEID in ").append(m_dmo.createInClause(countParam.getEmployeeIDs()));

		if (Request.REQUESTTYPE_FLEXTIME.equalsIgnoreCase(countParam.getRequestType())) {
			sql.append(" and R.REQUESTTYPE = '").append(Request.REQUESTTYPE_TIMEOFF).append("' and TOR.FLEXTYPE = 1");
		} else {
			sql.append(" and R.REQUESTTYPE = ").append(countParam.getRequestType()).append(" and R.FLEXTYPE = 0");
		}

		sql.append(" and TOC.STARTTIME >= ?").append(" and TOC.STARTTIME < ?").append(" group by R.EMPLOYEEID");

		JdmoQuery jQuery = m_dmo.createQuery(sql.toString(), Jdmo.PARAM_QUERY);
		jQuery.setParTimestamp(1, new Timestamp(countParam.getTimeRange().getStartDate().getTime()));
		jQuery.setParTimestamp(2, new Timestamp(countParam.getTimeRange().getEndDate().getTime()));
		JdmoRowset rs = m_dmo.createRowset(jQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

		while (rs.next()) {
			ID empID = rs.getID("EMPLOYEEID");
			Integer count = rs.getInt("NUM");
			map.put(empID, count);
		}

		return map;
	}
