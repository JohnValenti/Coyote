package com.bluepumpkin.ejb.rm.notification.rulecheckers;

import java.util.HashMap;
import java.util.Iterator;

import javax.xml.transform.Transformer;

import org.w3c.dom.Element;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.notifydelivery.ejb.NotifyMessageMap;
import com.bluepumpkin.ejb.bbm.notifydelivery.model.NotifyDeliveryMedia;
import com.bluepumpkin.ejb.bbm.notifydelivery.model.NotifyMessageInternal;
import com.bluepumpkin.ejb.bbm.notifydelivery.model.NotifyRuleDelivery;
import com.bluepumpkin.ejb.bbm.notifyrules.model.ActionDetail;

public class IntervalBasedRequestStatusChangeActionDetail extends ActionDetail {
	HashMap m_mapCustomizedeEmailMsg = null;
	HashMap m_mapCustomizedePopupMsg = null;
	
	public IntervalBasedRequestStatusChangeActionDetail() {	}
	
	public IntervalBasedRequestStatusChangeActionDetail (HashMap mapCustomizedeEmailMsg, HashMap mapCustomizedePopupMsg) {
		m_mapCustomizedeEmailMsg = mapCustomizedeEmailMsg;
		m_mapCustomizedePopupMsg = mapCustomizedePopupMsg;
	}
	
	public Object clone() throws CloneNotSupportedException {
		IntervalBasedRequestStatusChangeActionDetail cloned = (IntervalBasedRequestStatusChangeActionDetail)super.clone();
		
		cloned.m_mapCustomizedeEmailMsg = this.m_mapCustomizedeEmailMsg;
		cloned.m_mapCustomizedePopupMsg = this.m_mapCustomizedePopupMsg;
		
		return cloned;
	}
	
	public ActionDetail getNewInstance() {
		return new IntervalBasedRequestStatusChangeActionDetail();
	}

	public NotifyMessageMap getAllCustomizedMessages(NotifyMessageMap messageMap, Transformer transformer, NotifyRuleDelivery delivery)
			throws Exception{
//      set actionDetail as per delivery targets of the rule
		HashMap mapForDelivery = new HashMap();
		if (delivery.getDeliveryMediaID().equals(NotifyDeliveryMedia.NOTIFY_DELIVERY_MEDIA_EMAIL_ID)) {
			mapForDelivery = m_mapCustomizedeEmailMsg;
		} else {
			mapForDelivery = m_mapCustomizedePopupMsg;
		}
        if (mapForDelivery != null) {
        	for (Iterator it = mapForDelivery.keySet().iterator(); it.hasNext();) {
        		Object obj = it.next();	  
        		if (obj instanceof String){
        			NotifyMessageInternal msgInternal = new NotifyMessageInternal(obj, 
        					xmlToText((Element)mapForDelivery.get(obj)), transformer);
	    			messageMap.put((String)obj, msgInternal);
        		}
        		else {
        			ID workResourceID = (ID)obj;
        			NotifyMessageInternal msgInternal = new NotifyMessageInternal(workResourceID, 
        					xmlToText((Element)mapForDelivery.get(workResourceID)), transformer);
	    			messageMap.put(workResourceID, msgInternal);
        		}
        	}
        }
		return messageMap;
	}
	
	public boolean isAllMessageCustomized() {
		return true;
	}
}
