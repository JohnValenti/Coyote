package com.bluepumpkin.web.rm.setup.allocations;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.bbm.organization.OrganizationRequestHandler;
import com.bluepumpkin.web.rm.Log;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        SetBlackoutDayRangeRequestHandler
 * Description:  Handle requests for setting blackout day range
 * Copyright:    Copyright (c) 2005
 * Company:      Blue Pumpkin Software, Inc
 * @author       Gonzalo Quintanar
 * @version      1.0
 */
public class SetBlackoutDayRangeRequestHandler extends OrganizationRequestHandler {
	private static final Category log = 
		Log.initCategory(SetBlackoutDayRangeRequestHandler.class.getName());

   	/**
	 * Method invoked by Framework to Process the Request
	 */
    @Override
	public void processRequest(RequestContext context) throws Exception {
		if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		} 
	}

    /** 
     * Handle the Save request
     */       
	public void processSave(RequestContext context) {
 		if (!isAuthorizedForOrg(context, 
								PrivilegeKeys.TOM_EDITAVAILABLETIMEOFF,
								true)) {
			return;
		}
		// get data from the request
		SetBlackoutDayRangePopupPM model = new SetBlackoutDayRangePopupPM(context);
		model.extractParameters(context.getRequest());
		model.validate();

		if (!model.hasError()) {
			try {
				AvailableTimeOffModelHandler.saveBlackoutDaysForRange(
                    model.getStartDate(), model.getEndDate(), model.getPoolID(), model.isCreateBlackoutDays());
			
            } catch(Exception ex) {
				model.handleUnableToUpdateInfo(m_log, ex);
			}
		}

		model.setPageDirty(false); //Is this needed?
		model.setIsRefreshOpenerOnClose(true);
		//Use the new variable isCloseAfterLoadCustomize instead of using parent's variable isCloseAfterLoad
		//to handle for Chrome 
		model.setCloseAfterLoadCustomize( !model.hasError() );
		context.setPageModel(model);
	}

}
