	private static final Category log = Log.initCategory(AvailableTimeOffImportRequestHandler.class.getName());

	/**
	 * Processes Request submitted from the AvailableTimeOff pages
	 *
	 * @param context The Request Context
	 */
	@Override
	public void processRequest(RequestContext context) throws Exception {
		//make page model
		AvailableTimeOffImportPageModel model;
		boolean isRMAdvancedLicense = LicenseUtil.isAdvancedRMLicense();
		model = new AvailableTimeOffImportPageModel(context, isRMAdvancedLicense);

		if (!isAuthorizedForOrg(context, PrivilegeKeys.TOM_EDITAVAILABLETIMEOFF, true)) {
			return;
		}

		// can't import available time off hours if those ours are defined
		// by another org.
		//ID orgId = OrganizationUtil.getCachedOrgIDIfValid(context);
		//ID ownerOrgId = AvailableTimeOffModelHandler.getCalendarTrickleOrgId(orgId);
		//if (ownerOrgId != null && !ownerOrgId.equals(orgId)) {
		// there is an time off hour owner org and this org is not it!
		//	return;
		//}

		// get data from the request
		model.extractParameters(context.getRequest());
		int[] result = { 0, 0, 0 };
		if (isAction(context, PageAction.IMPORT)) {
			int allocationType = model.getAllocationType();
			//New interval import for Advanced License
			if (isRMAdvancedLicense && allocationType == model.INTERVAL_ALLOCATION) {
				try {
					result = processIntervalUpload(model, context);
				} catch (ParseException ex) {
					log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
					model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.TIMEOFF_POOL_IMPORT_INVALID_DATE, RmWebBundleKey.BUNDLE_NAME);
				} catch (NumberFormatException ex) {
					log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
					model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.TIMEOFF_POOL_IMPORT_INVALID_RANGE, RmWebBundleKey.BUNDLE_NAME);
				} catch (BbmCreateDuplicateKeyException ex) {
					log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
					model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.TIMEOFF_POOL_IMPORT_SAME_START_DATE, RmWebBundleKey.BUNDLE_NAME);

				} catch (Exception ex) {
					log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
					model.addPageMessage(Message.ERROR_TYPE, BbmWebBundleKey.UploadDataFailure, BbmWebBundleKey.BUNDLE_NAME);
				}
			}
			//Import pool
			else {
				try {
					model.processUpload(context);
				} catch (Exception ex) {
					log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
					model.addPageMessage(Message.ERROR_TYPE, BbmWebBundleKey.UploadDataFailure, BbmWebBundleKey.BUNDLE_NAME);
				}
			}
		}
		//Cache the AvailableTimeOffPageModel in the Request
		//For Normal license close the popup when import is successfully
		if (isAction(context, PageAction.IMPORT) && !isRMAdvancedLicense) {
			model.setPageDirty(false);
			model.setIsRefreshOpenerOnClose(true);
			model.setIsCloseAfterLoad(!model.hasError());
		}
		//For AdvancedRM license, show messages just in case users want to import again for other allocations
		if (isAction(context, PageAction.IMPORT) && model.getPageAction() != null && model.getPageAction().equals(AvailableTimeOffImportPageModel.SUCCESS) && isRMAdvancedLicense) {
			if (model.getAllocationType() == model.DAILY_ALLOCATION) {
				model.addPageMessage(Message.INFO_TYPE, RmWebBundleKey.TIMEOFF_POOL_IMPORT_SUCCESS, RmWebBundleKey.BUNDLE_NAME,
						new String[] { model.i18n(RmWebBundleKey.TIMEOFF_POOL_DAILY_ALLOCATIONS) });
			} else {
				if (result[2] > 0) {
					model.addPageMessage(Message.INFO_TYPE, RmWebBundleKey.TIMEOFF_INTERVAL_IMPORT_SUCCESS_WITH_DELETE_INFO,
							RmWebBundleKey.BUNDLE_NAME, new String[] { String.valueOf(result[0]), String.valueOf(result[1]), String.valueOf(result[2])});
				} else {
					model.addPageMessage(Message.INFO_TYPE, RmWebBundleKey.TIMEOFF_INTERVAL_IMPORT_SUCCESS_WITHOUT_DELETE_INFO,
							RmWebBundleKey.BUNDLE_NAME, new String[] { String.valueOf(result[0]), String.valueOf(result[1]) });
				}
			}
			model.setIsRefreshOpenerOnClose(true);
		}
		context.setPageModel(model);
	}

	public int[] processIntervalUpload(AvailableTimeOffImportPageModel model, RequestContext context) throws ParseException, NumberFormatException, BbmCreateDuplicateKeyException, Exception {
		int[] result = { 0, 0, 0 };
		try {
			HttpServletRequest request = context.getRequest();
			@SuppressWarnings("rawtypes")
			Map fileMap = ParameterUtil.getExtractedFileMap(request);

			FileItem file = (FileItem) fileMap.get("file");
			String fileContent = IOUtil.readStringFromTextFile(file, 10 * 1024 * 1024, context);

			// now process the info to save data
			String delimiter = model.getDelimiter();
			int numLines = 0;
			try {
				numLines = StringUtil.isEmpty(model.getNumLines()) ? 0 : Integer.parseInt(model.getNumLines());
			} catch (Exception e) {
				log.debug(e);
			}

			if (delimiter == null || ("tab").equals(delimiter)) {
				delimiter = "\t";
			}

			int count = 1;
			ID poolID = model.getPoolID();
			ID orgPoolID = AvailableTimeOffModelHandler.getTOPool(poolID).getOrganizationId();
			Organization org = OrganizationModelHandler.getOrganization(context, orgPoolID);
			TimeZone orgTz = org.getTimeZone();
			HashMap<LocalDate, List<TOIntervalAllocation>> hashTimeOffIntervals = new HashMap<LocalDate, List<TOIntervalAllocation>>();
			List<TOIntervalAllocation> timeoffIntervals = new ArrayList<TOIntervalAllocation>();
			//This collection to contains string built from start(hour+minute) and end(start+minute) to check duplicate entry
			Collection<String> startEndPairs = new ArrayList<String>();
			StringTokenizer aToken = new StringTokenizer(fileContent, "\r\n");

			while (aToken.hasMoreTokens()) {
				String data = aToken.nextToken();
				if (count > numLines) {
					List<String> dataList = ImportExportUtil.parseData(data, delimiter);
					// create a new employee and set the fields as got from the file
					int[] colIndexArray = new int[4];
					colIndexArray[0] = Integer.parseInt(model.getDate());
					colIndexArray[1] = Integer.parseInt(model.getHours());
					colIndexArray[2] = Integer.parseInt(model.getStartTime());
					colIndexArray[3] = Integer.parseInt(model.getEndTime());
					Pair<LocalDate, TOIntervalAllocation> pairObject = model.buildTimeOffIntervalObjectByRawData(colIndexArray, dataList, poolID, context.getLocalizer(), orgTz);
					LocalDate shortDate = pairObject.getFirst();
					TOIntervalAllocation timeOffInterval = pairObject.getSecond();

					if (shortDate != null && timeOffInterval != null) {
						if (hashTimeOffIntervals.get(shortDate) == null) { //gets the value for an id)
							timeoffIntervals = new ArrayList<TOIntervalAllocation>();
							startEndPairs = new ArrayList<String>();
						}
						String startEndPair = Integer.toString(timeOffInterval.getStartTime().getHours()) + Integer.toString(timeOffInterval.getStartTime().getMinutes()) + Integer.toString(timeOffInterval.getEndTime().getHours()) + Integer.toString(timeOffInterval.getEndTime().getMinutes());

						//Only add unique entries (not same start date) to the list.
						//The duplicated issue happens for Daylight saving time when the clock moves forward 1 hour
						if (!startEndPairs.contains(startEndPair)) {
							timeoffIntervals.add(timeOffInterval);
							startEndPairs.add(startEndPair);
						}
						if (!timeoffIntervals.isEmpty()) {
							hashTimeOffIntervals.put(shortDate, timeoffIntervals);
						}
					}
				} else {
					count++;
				}
			}
			// now try to save everything
			result = AvailableTimeOffModelHandler.saveCalendarTimeOffIntervals(hashTimeOffIntervals, poolID, orgTz);
			model.setPageAction(model.SUCCESS);
		} catch (ParseException ex) {
			log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
			throw ex;

		} catch (NumberFormatException ex) {
			log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
			throw ex;
		} catch (BbmCreateDuplicateKeyException ex) {
			log.l7dError(RmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
			throw ex;
		}

		return result;
	}
