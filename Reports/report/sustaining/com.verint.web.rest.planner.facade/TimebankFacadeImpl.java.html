package com.verint.web.rest.planner.facade;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.workrules.ejb.WorkRuleManager;
import com.bluepumpkin.ejb.bbm.workrules.timebanking.model.TimeBankPlannerData;
import com.bluepumpkin.web.fs.Log;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.SystemWFOException;
import com.verint.web.rest.planner.facade.ITimebankFacade;
import com.verint.web.rest.planner.entity.TimebankEntity;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.util.FormatUtils;
import com.witness.web.uif.system.RequestContext;

import javax.ws.rs.core.Response;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.List;

public class TimebankFacadeImpl implements ITimebankFacade {
	private static final Category logger = Log.initCategory(TimebankFacadeImpl.class.getName());
	private static final int MAXNUM = 54;

	public TimebankFacadeImpl() {
	}

	public boolean checkIfTimebankExists(RequestContext context, String timebankName) throws WFOException {
		try {
			WorkRuleManager wrm = WfmManagerFactory.getWorkRuleManager(context.isInWhatIfMode());
			return wrm.isTimebankPlannerDataNameExists(timebankName);
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	public void deleteTimebank(RequestContext context, String timebankName) throws WFOException {
		try {
			WorkRuleManager wrm = WfmManagerFactory.getWorkRuleManager(context.isInWhatIfMode());
			wrm.deleteTimebankPlannerData(timebankName);
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	public void saveTimebankData(RequestContext context, List<TimebankEntity> timebankEntities) throws WFOException {
		try {
			if (timebankEntities == null || timebankEntities.isEmpty()) {
				throw new BusinessWFOException("Timebank Planner data list is empty", Response.Status.CONFLICT);
			}
			WorkRuleManager wrm = WfmManagerFactory.getWorkRuleManager(context.isInWhatIfMode());
			Collection<TimeBankPlannerData> timeBankPlannerData = getTimebankCollection(timebankEntities);
			wrm.saveTimebankPlannerData(timeBankPlannerData);
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	private List<TimeBankPlannerData> getTimebankCollection(Collection<TimebankEntity> timebankEntities) throws WFOException {
		try {
			ArrayList<TimeBankPlannerData> timebankEntityList = new ArrayList<TimeBankPlannerData>();
			for (TimebankEntity tbEntity : timebankEntities) {
				TimeBankPlannerData tbPlannerData = new TimeBankPlannerData();

				String name = tbEntity.getName();
				int orgID = tbEntity.getOrgId();
				Date startTime = FormatUtils.iso8601SecPrecisionStringToDate(tbEntity.getStartTime());
				Date endTime = FormatUtils.iso8601SecPrecisionStringToDate(tbEntity.getEndTime());
				int dayOfWeek = tbEntity.getDayOfWeek();

				tbPlannerData.setName(name);
				tbPlannerData.setStartDate(startTime);
				tbPlannerData.setEndDate(endTime);
				tbPlannerData.setOrgId(orgID);
				tbPlannerData.setStartDayOfWeek(dayOfWeek);

				int[] timebankHours = getTimebankHours(tbEntity.getWeekNum(), tbEntity.getWeekHour());
				tbPlannerData.setTimebankPlannerData(timebankHours);
				timebankEntityList.add(tbPlannerData);
			}
			return timebankEntityList;
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	private int[] getTimebankHours(String weekNum, String strWeekHour) throws WFOException {
		String[] weekHourStr = strWeekHour.split(",");
		int[] weekHour = new int[MAXNUM];
		for (int i = 0; i < weekHourStr.length; i++) {
			try {
				weekHour[i] = Integer.parseInt(weekHourStr[i]);
			} catch (NumberFormatException nfex) {
				logger.error(nfex.getLocalizedMessage(), nfex);
				throw new SystemWFOException(nfex.getLocalizedMessage());
			}
		}
		return weekHour;
	}
}
