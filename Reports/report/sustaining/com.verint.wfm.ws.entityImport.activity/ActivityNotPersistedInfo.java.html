package com.verint.wfm.ws.entityImport.activity;

import java.util.ArrayList;
import java.util.Collection;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.verint.wfm.ws.entityImport.EntityNotPersistedInfo;

/**
 * For each activity not persisted by createOrUpdateActivities(), one of these will be in the list returned This is
 * returned in lieu of an exception, because we always want to continue on to all the activities
 * 
 * 
 * @see EntityImportService.createOrUpdateActivities()
 * 
 */
public class ActivityNotPersistedInfo extends EntityNotPersistedInfo<String> {
	private static final long serialVersionUID = 1L;

	public ActivityNotPersistedInfo() {
		super();
	}

	public ActivityNotPersistedInfo(String activityName) {
		super(activityName);
	}

	// -----------------------------------------
	public String getActivityIdentifier() {
		return super.getEntityExternalSourceId();
	}

	/**
	 * Dummy here on server, but needed for SOAP stub
	 */
	public void setActivityIdentifier(String activityIdentifier) {
		// do nothing
	}

	public void addActivityServerExceptionMessage(String exception) {
		getMessages().add(
				String.format("Exception is %s. The server could not persist activity with name %s.", exception,
						getExternalIdSummary()));
	}

	public void addNameMissingMessage() {
		getMessages().add(String.format("Name was missing for activity with name %s.", getExternalIdSummary()));
	}
	
	public void addNameHasInvalidCharactersMessage(String invalidChars) {
		getMessages()
				.add(String
						.format("The characters %s are not allowed in the activity name. The activity %s has invalid characters in the name.",
								invalidChars, getExternalIdSummary()));
	}

	public void addNameLengthExceededMessage(int actualLength, int lengthLimit) {
		getMessages().add(
				String.format("For activity with name %s the name length %d exceeded the limit of %d.",
						getExternalIdSummary(), actualLength, lengthLimit));
	}

	
	public void addActivityTypeMissingMessage() {
		getMessages().add(
				String.format("For activity with name %s, the activity type was missing.",
						getExternalIdSummary()));
	}

	public void addNoSuchActivityTypeMessage(String activityType) {
		getMessages()
				.add(String
						.format("For activity with name %s, the activity type %s does not exist.",
								getExternalIdSummary(), activityType));
	}
	
	public void addOrganizationNameMissingMessage() {
		getMessages().add(
				String.format("For activity with name %s, the organization name was missing.",
						getExternalIdSummary()));
	}

	public void addNoSuchOrganizatonNameMessage(String organizationName) {
		getMessages()
				.add(String
						.format("For activity with name %s, the organization specified to have name %s does not exist.",
								getExternalIdSummary(), organizationName));
	}

	public void addNoSuchQueueExternalIdsMessage(Collection<ID> noSuchQueueEternalIds) {
		
		Collection<String> noSuchQueueExternalIdStrings = new ArrayList<String>();
		for (ID  externalId : noSuchQueueEternalIds) {
			noSuchQueueExternalIdStrings.add(externalId.toString());
		}
		
		getMessages()
				.add(String
						.format("For activity with name %s, the linked queues specified to have external ids %s do not exist.",
								getExternalIdSummary(), noSuchQueueExternalIdStrings.toString()));
	}
	
	public void addUnknownValidationMessage() {
		getMessages().add(
				String.format("The system rejected activity with name %s as invalud for an unknown reason.",
						getExternalIdSummary()));
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public List<String> getMessages() {
		return super.getMessages();
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public void setMessages(List<String> messages) {
		// Do nothing
	}
}
