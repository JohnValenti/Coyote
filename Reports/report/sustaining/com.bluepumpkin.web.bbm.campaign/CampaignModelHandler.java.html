	/**
	 * Create Campaing Map
	 *
	 * @param camList The List of Campaigns
	 * @param isNameOnly if true the value is only Campaign Name
	 */
	public static Map<ID, Object> createCampaignMap(Collection<Campaign> camList, boolean isNameOnly) {
		if (camList==null || camList.isEmpty()) return Collections.emptyMap();

		Map<ID, Object> map = new HashMap<>(camList.size());
		for (Campaign cam : camList) {
			if (isNameOnly) {
				map.put(cam.getID(), cam.getName());
			} else {
				map.put(cam.getID(), cam);
			}
		}
		return map;
	}

	/**
	 * Return Queue ID Objects assigned to specified Campaign for the time range
	 *
	 * @param isSubQueuesIncluded - Specify whether Sub Queues are included
	 */
	public static Collection getQueueIDsAssignedToCampaign(RequestContext context, ID campaignID, Date startDate,
	                                                       Date endDate, boolean isSubQueuesIncluded) throws RemoteException, BbmException {
		Collection result = Collections.emptyList();
		Collection<CampaignQueue> campQueueAssgnList = getCampaignManager(context).getCampaignQueueAssignments(campaignID, startDate, 
			endDate);
		if (!campQueueAssgnList.isEmpty()) {
			result = new ArrayList(campQueueAssgnList.size());
			for (CampaignQueue cq : campQueueAssgnList) {
				result.add(cq.getQueueID());
			}
			// Fetch Sub Queue ID as well if needed
			if (isSubQueuesIncluded) {
				Collection subQueueIDs = WorkloadModelHandler.getSubQueues(context, result);
				if (!subQueueIDs.isEmpty()) {
					TreeSet tmpQIDs = new TreeSet();
					tmpQIDs.addAll(result);
					tmpQIDs.addAll(subQueueIDs);
					result = tmpQIDs;
				}
			}
		}
		return result;
	}

	/**
	 * Return Queue Objects assigned to specified Campaign for the time range
	 *
	 * @param isSubQueuesIncluded - Specify whether Sub Queues are included
	 */
	public static Collection getQueuesAssignedToCampaign(RequestContext context, ID campaignID, Date startDate,
			Date endDate, boolean isSubQueuesIncluded) throws RemoteException, BbmException {
		Collection queues = Collections.emptyList();

		Collection<ID> queueIDs = getQueueIDsAssignedToCampaign(context, campaignID, startDate, endDate);
		if (!queueIDs.isEmpty()) {
			// Fetch Sub Queue ID as well if needed
			if (isSubQueuesIncluded) {
				Collection subQueueIDs = WorkloadModelHandler.getSubQueues(context, queueIDs);
				if (!subQueueIDs.isEmpty()) {
					ArrayList tmpQIDs = new ArrayList(queueIDs.size() + subQueueIDs.size());
					tmpQIDs.addAll(queueIDs);
					tmpQIDs.addAll(subQueueIDs);
					queueIDs = tmpQIDs;
				}
			}
			queues = WorkloadModelHandler.getQueuesByIDs(context, queueIDs);
		}
		return queues;
	}

	/**
	 * Return All Queues assigned to campaigns visible to specified User and time range
	 */
	public static Collection getQueuesAssignedToCampaignsForUser(RequestContext context, User user, Date startDate,
			Date endDate, boolean includeSubqueues) throws RemoteException, BbmException {
		Collection colIDs = user.getAuthorizedScopes(PrivilegeKeys.FS_VIEWCAMPAIGN_ID, Campaign.SCOPE_TYPE);
		if (colIDs.isEmpty()) {
			return Collections.emptyList();
		}

		//now iterate thru the collection to get queue ID
		Set queueIDs = new HashSet();
		for (Object colID : colIDs) {
			ID campaignID = (ID) colID;
			Collection tmpQueueIDs = getQueueIDsAssignedToCampaign(context, campaignID, startDate, endDate, includeSubqueues);
			queueIDs.addAll(tmpQueueIDs);
		}

		return WorkloadModelHandler.getQueuesByIDs(context, queueIDs);
	}

	public static Collection createDropDownValueList(Collection campaignList, String selectedID, Localizer localizer, boolean isCreateStringsPair) {
		if (campaignList.isEmpty()) {
			return Collections.emptyList();
		}

		campaignList = SortUtil.sortBasicVO(campaignList, localizer);
		TreeNodeHelper treeNodeHelper = new DefaultTreeNodeHelper(campaignList, ROOT_OBJ, new BasicVOTreeNodeConverter(false));
		TreeModel treeModel = TreeModelBuilder.buildTreeModel(campaignList, treeNodeHelper);
		DefaultMutableTreeNode root = (DefaultMutableTreeNode)treeModel.getRoot();
		return TreeModelBuilder.createDropDownValueList(root, selectedID, false, isCreateStringsPair);
	}

	/**
	 * Return Set containing queue ID that has partial assignments for specified
	 * campaign and time period.
	 *
	 */
	public static Set getPartialAssignQueueSet(RequestContext context, ID campaignID, Date startDate,
			Date endDate) throws RemoteException, BbmException {
		Collection<CampaignQueue> cqList = getCampaignManager(context).getCampaignQueueAssignments(campaignID, startDate, endDate);

		if (cqList.isEmpty()) {
			return Collections.emptySet();
		}

		// Create partially assigned map
		Set paSet = new HashSet(cqList.size());
		for (CampaignQueue cq : cqList) {
			boolean isPA = !TimePeriodUtil.inside(startDate, endDate, cq.getStartTime(), cq.getEndTime());
			if (isPA) {
				paSet.add(cq.getQueueID());
			}
		}

		return paSet;
	}

	/**
	 * Returns a collection of Campaigns visible to the given user.  The collection will also include the scheduling periods
	 * belonging to these campaigns (these are included in the same collection).
	 */
	public static Collection<?> getCampaignsAndAssociatedSchedulingPeriodsForUser(RequestContext context, User user,
			boolean getDistributedCampaigns) throws BbmException, RemoteException {
		Collection<Campaign> campaigns = CampaignModelHandler.getCampaignsForUser(context, user);
		if (!getDistributedCampaigns) {
			//Filter out distributed and sub campaigns
			campaigns.removeIf(c -> c.getIsDistributed() || c.getParentID() != null);
		}
		Collection<SchedulingPeriod> sps =
			CampaignModelHandler.getSchedulingPeriods(context, campaigns);
		Collection retVal = new ArrayList();
		retVal.addAll(campaigns);
		retVal.addAll(sps);
		return retVal;
	}

	 */
	public static void updateSPShrinkage(RequestContext context, Collection<SPShrinkage> spShrinkageCol) throws BbmException, MultiUserException, RemoteException {
		getCampaignManager(context).updateSPShrinkage(spShrinkageCol);
		Collection<SPQueue> spQueues = getCampaignManager(context).getSPQueuesFromSPShrinkage(spShrinkageCol);

		FteRequirementsUtil.recalculateFteRequirementsForSPQueues(ValueObjectUtil.getIDFromObjects(spQueues), context.isInWhatIfMode());
	}
