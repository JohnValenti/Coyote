package com.bluepumpkin.ejb.facade.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.witness.ejb.core.security.model.AccessRight;

public class CFAccessRight implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id;
	public ID roleId;
	public CFScopeID scopeId;
	
	public static CFAccessRight convert(AccessRight from) {
		CFAccessRight to = new CFAccessRight();
		
		to.id = from.getID();
		to.roleId = from.getRoleID();
		to.scopeId = CFScopeID.convert(from.getScopeID());
		
		return to;
	}
	
	public static CFAccessRight[] convert(Collection from) {
		if (from == null || from.isEmpty()) {
			return new CFAccessRight[0];
		}
		
		CFAccessRight[] to = new CFAccessRight[from.size()];
		AccessRight fromObj;
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++) {
			fromObj = (AccessRight)i.next();
			to[ix] = CFAccessRight.convert(fromObj);
		}
		
		return to;
	}
}
