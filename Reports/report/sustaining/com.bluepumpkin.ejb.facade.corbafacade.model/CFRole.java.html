package com.bluepumpkin.ejb.facade.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.witness.ejb.core.security.model.Role;

public class CFRole implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id;
	public String name;
	public String description;
	public boolean isDefault;	
	public CFRolePrivilege[] rolePrivileges;	
	public int[] privilegeScopeTypes;
	
	public static CFRole convert(Role from) {
		CFRole to = new CFRole();
		
		to.id = from.getID();
		to.name = from.getName();
		to.description = from.getDescription();
		to.isDefault = from.isDefault();
		to.rolePrivileges = CFRolePrivilege.convert(from.getPrivileges());
		
		return to;
	}
	
	public static CFRole[] convert(Collection from) {
		if (from == null || from.isEmpty()) {
			return new CFRole[0];
		}
		
		CFRole[] to = new CFRole[from.size()];
		Role fromObj;
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++) {
			fromObj = (Role)i.next();
			to[ix] = CFRole.convert(fromObj);
		}
		
		return to;
	}
}
