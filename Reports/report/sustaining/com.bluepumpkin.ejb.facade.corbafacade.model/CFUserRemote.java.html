package com.bluepumpkin.ejb.facade.corbafacade.model;

import java.io.Serializable;

import com.bluepumpkin.common.datatypes.ID;
import com.witness.ejb.core.security.model.UserRemote;

public class CFUserRemote implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id;
	public ID employeeId;
	public String userName;
	public int pwdChange;
	public int version = -1;
	public int globalVersion = -1;
	public boolean isSuperUser = false;

	public CFAccessRight[] accessRights;
	public CFRole[] roles;
	public CFPrivilege[] priviledges;
	
	
	public static CFUserRemote convert(UserRemote from) {
		CFUserRemote to = new CFUserRemote();
		
		to.id = from.getID();
		to.employeeId = from.getEmployeeID();
		to.userName = from.getUserName();
		to.pwdChange = from.getForcePasswordChange();
		to.version = from.getUserVersion();
		to.globalVersion = from.getVersion();
		to.isSuperUser = from.isSuperUser();

		if (!to.isSuperUser)
		{
			to.accessRights = CFAccessRight.convert(from.getAccessRights());
			to.roles = CFRole.convert(from.getAllRoles().values());
			to.priviledges = CFPrivilege.convert(from.getAllPrivileges().values());
		}
		return to;
	}
}
