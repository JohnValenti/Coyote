package com.bluepumpkin.ejb.facade.corbafacade.model;

import java.io.Serializable;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.ScopeID;

public class CFScopeID implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id;
	public int scopeType;
	
	public static CFScopeID convert(ScopeID from) {
		CFScopeID to = new CFScopeID();
		
		to.id = from.getID();
		to.scopeType = from.getType();
		
		return to;
	}
}
