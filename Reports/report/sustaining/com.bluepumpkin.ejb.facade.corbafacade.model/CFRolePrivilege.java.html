package com.bluepumpkin.ejb.facade.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.witness.ejb.core.security.model.RolePrivilege;
import com.witness.ejb.core.security.model.RoleScopedPrivilege;
import com.witness.ejb.core.security.model.RoleSecureFieldPrivilege;

public class CFRolePrivilege implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public int type = -1;
	public static final int SECUREFIELD_TYPE = 1;
	public static final int SCOPED_TYPE = 2;
	public int fieldKey = -1;
	public CFScopeID scopeId;
	public ID id;
	public ID roleId;
	public ID privilegeId;
	
	public static CFRolePrivilege convert(RolePrivilege from) {
		CFRolePrivilege to = new CFRolePrivilege();
		
		to.id = from.getID();
		to.roleId = from.getRoleID();
		to.privilegeId = from.getPrivilegeID();
		
		return to;
	}
	
	public static CFRolePrivilege[] convert(Collection from) {
		if (from == null || from.isEmpty()) {
			return new CFRolePrivilege[0];
		}
		
		CFRolePrivilege[] to = new CFRolePrivilege[from.size()];
		Object fromObj;
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++) {
			fromObj = i.next();
			if (fromObj instanceof RoleSecureFieldPrivilege) {
				RoleSecureFieldPrivilege pri = (RoleSecureFieldPrivilege)fromObj;
				to[ix] = CFRolePrivilege.convert(pri);
				to[ix].type = SECUREFIELD_TYPE;
				to[ix].fieldKey = pri.getSecureFieldKey();
			} else if (fromObj instanceof RoleScopedPrivilege) {
				RoleScopedPrivilege scopedPri = (RoleScopedPrivilege)fromObj;
				to[ix] = CFRolePrivilege.convert(scopedPri);
				to[ix].type = SCOPED_TYPE;
				to[ix].scopeId = CFScopeID.convert(scopedPri.getScopeID());
			} else
				to[ix] = CFRolePrivilege.convert((RolePrivilege)fromObj);
		}
		
		return to;
	}
}
