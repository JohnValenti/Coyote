package com.bluepumpkin.ejb.facade.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.witness.ejb.core.security.model.Privilege;

public class CFPrivilege implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id;
	public ID parentId;
	public String name;
	public String licenseCategory;
	public String applicationName;
	public String moduleName;
	public int[] scopeType;
	public boolean isScopeRequired;
	
	public static CFPrivilege convert(Privilege from) {
		CFPrivilege to = new CFPrivilege();
		
		to.id = from.getID();
		to.parentId = from.getParentID();
		to.name = from.getName();
		to.licenseCategory = from.getLicenseCategory();
		to.applicationName = from.getApplication();
		to.moduleName = from.getModule();
		short[] types = from.getScopeType();
		to.scopeType = new int[types.length];
		for (int i = 0; i < types.length; i++) {
			to.scopeType[i] = types[i];
		}
		to.isScopeRequired = from.isScopedAssignmentRequired();
		
		return to;
	}
	
	public static CFPrivilege[] convert(Collection from) {
		if (from == null || from.isEmpty()) {
			return new CFPrivilege[0];
		}
		
		CFPrivilege[] to = new CFPrivilege[from.size()];
		Privilege fromObj;
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++) {
			fromObj = (Privilege)i.next();
			to[ix] = CFPrivilege.convert(fromObj);
		}
		
		return to;
	}
}
