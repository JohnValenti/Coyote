    /**
     * Saves deadline goals time series data for a deferred media queue.  This includes deadline goal time series data (saved
     * as service level percent data in the DB) and "deadline time" time series data (saved as ASA data in the DB).
     */
    public void saveDeadlineGoalsTimeSeries(List<ServiceLevelPercentTimeSeriesDataItem> deadlineGoalTimeSeries,
			List<ASATimeSeriesDataItem> deadlineTimeTimeSeries, SPQueue spQueue) throws BbmUpdateException {
    	methodStart("saveDeadlineGoalsTimeSeries", deadlineGoalTimeSeries, deadlineTimeTimeSeries, spQueue);
    	DeadlineGoalsTimeSeriesDAO dao = new DeadlineGoalsTimeSeriesDAO();
        try {
        	List<DeadlineGoalsTimeSeriesDataItem> fullDeadlineGoalTimeSeries = new ArrayList<DeadlineGoalsTimeSeriesDataItem>();
        	Map<Date, Double> deadlineGoalValueMap = new HashMap<Date, Double>();
        	Map<Date, Double> deadlineTimeValueMap = new HashMap<Date, Double>();
        	
        	for (ServiceLevelPercentTimeSeriesDataItem item : deadlineGoalTimeSeries) {
        		if (item != null && item.getTime() != null) {
        			deadlineGoalValueMap.put(item.getTime(), item.getValue());
        		}
        	}
        	
        	for (ASATimeSeriesDataItem item : deadlineTimeTimeSeries) {
        		if (item != null && item.getTime() != null) {
        			deadlineTimeValueMap.put(item.getTime(), item.getValue());
        		}
        	}
        	
        	for (Date recordTime : deadlineGoalValueMap.keySet()) {
        		fullDeadlineGoalTimeSeries.add(new DeadlineGoalsTimeSeriesDataItem(
        				recordTime, deadlineGoalValueMap.get(recordTime), deadlineTimeValueMap.get(recordTime)));
        	}
        	
        	dao.saveServiceGoalsTimeSeries(fullDeadlineGoalTimeSeries, spQueue, m_sessionContext.getCallerPrincipal().getName());
		} catch (BbmRemoveException e) {
			handleException(e);
			throw new BbmUpdateException(e);
		} catch (BbmCreateException e) {
			handleException(e);
			throw new BbmUpdateException(e);
		} catch (JdmoException e) {
			handleException(e);
			throw new BbmUpdateException(e);
		} catch (BbmUpdateException e) {
			handleException(e);
			throw e;
		} finally {
        	dao.cleanUp();
            methodFinish();
        }    	
    }
