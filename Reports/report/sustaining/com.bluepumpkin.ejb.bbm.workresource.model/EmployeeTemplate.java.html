public final class EmployeeTemplate extends ValueObjectNode {
	/**
	 *

	private static final long serialVersionUID = 1L;

	// meta data information
	private static FieldInfo m_fieldInfo = new EmployeeTemplateFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public EmployeeTemplate() {
		super();
	}

	public ID getOrganizationID() {
		return (ID) getFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_ORGANIZATIONID);
	}

	public void setOrganizationID(ID idValue) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_ORGANIZATIONID, idValue);
	}

	public String getName() {
		return getFormattedFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_NAME);
	}

	public void setName(String strValue) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_NAME, strValue);
	}

	public double getWage() {
		return getFieldValueDouble(EmployeeTemplateFieldInfo.EMPTEMPLATE_WAGE);
	}

	public void setWage(double dblWage) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_WAGE, dblWage);
	}

	public boolean getIsHidden() {
		return getFieldValueBoolean(EmployeeTemplateFieldInfo.EMPTEMPLATE_ISHIDDEN);
	}

	public void setIsHidden(boolean isHidden) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_ISHIDDEN, isHidden);
	}

	public double getProficiency() {
		return getFieldValueDouble(EmployeeTemplateFieldInfo.EMPTEMPLATE_PROFICIENCY);
	}

	public void setProficiency(double dValue) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_PROFICIENCY, dValue);
	}

	public int getConnectionID() {
		return getFieldValueInt(EmployeeTemplateFieldInfo.EMPTEMPLATE_CONNECTIONID);
	}

	public void setConnectionID(int iValue) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_CONNECTIONID, iValue);
	}

	public void setDEID(ID deID) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_CONNECTIONID, deID);
	}

	public Date getCreatedDate() {
		return getFieldValueDate(EmployeeTemplateFieldInfo.EMPTEMPLATE_CREATED);
	}

	public void setCreatedDate(Date createdDate) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_CREATEDBY, createdDate);
	}

	public String getCreatedBy() {
		return getFormattedFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_CREATEDBY);
	}

	public void setCreatedBy(String createdBy) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_CREATEDBY, createdBy);
	}

	public Date getModifiedDate() {
		return getFieldValueDate(EmployeeTemplateFieldInfo.EMPTEMPLATE_MODIFIED);
	}

	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_MODIFIED, modifiedDate);
	}

	public String getModifiedBy() {
		return getFormattedFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_MODIFIEDBY);
	}

	public void setModifiedBy(String modifiedBy) {
		setFieldValue(EmployeeTemplateFieldInfo.EMPTEMPLATE_MODIFIEDBY, modifiedBy);
	}

		return m_aSkillAssignments;

	public void setSkillAssignments(Collection<SkillAssignment> skills) {
		m_aSkillAssignments = new ArrayList<SkillAssignment>(skills);
	}

	List<ID> m_aAssignmentRuleIDs = new ArrayList<ID>();

	public List<ID> getAssignmentRuleIDs() {
