 * Description:  Supports Effective Date popup window for employee(s) Time Off Pool assignment.

public class TOPoolEffectivityPopupPM extends BasePageModel 
{
	public static final String FORM_ACTION = "pool_eff_popup";
	public static final String CUSTOM_ON_SET_JS = "updateOrgDataToParentWindow()"; //function is defined in EffectiveDatesUtil.js

	private String m_poolID = "";   //the selected employee's time off poolID on the AsOfDate (viewDate)
	private String m_poolName = ""; //the selected employee's time off pool on the AsOfDate (viewDate)
    private LocalDate m_viewDate;   //the AsOfDate (viewDate)
	//if  not null this popup is being loaded to set the time off pool for the given activity 
	private ID activityID;

	/**
	 * Constructor

		super(context, CUSTOM_ON_SET_JS);

			addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
			addRequiredJavaScriptFile(RmJavaScriptFileID.TIME_OFF_POOL_BY_ACTIVITY_POPUP);
			this.m_onSetJS = "timeOffPoolByActivity.defaultSet()";
		}
	}
	
	@Override

	/**

		return getInstance(context, TOPoolEffectivityPopupPM.class);
	}

	@Override
	protected void initializeModel() 
	{
		if (m_isInitialized)
			return;
		else
			m_isInitialized = true;
		
		try 
		{
			//selectedID is a comma-delimitted string of empIDs
			if (selectedID!=null && selectedID.length()>0)
			{
				//We know at least one employee is selected
				ID[] ids = StringUtil.parseIDString(selectedID); 
				if (ids!=null) //(m_empID != null) //we cannot rely on m_empID because it has a weird format when multiple emps are selected
				{			
					if (ids.length==1)
					{
						//Only one employee is selected. Get his pool effectivity history.
						Calendar bigBangCal = Calendar.getInstance();
						bigBangCal.set(Calendar.YEAR, 1900);
						Calendar armageddonCal = Calendar.getInstance();
						armageddonCal.set(Calendar.YEAR, 2100);
						List<EmpTOPoolAssignment> poolHistoryList = TOPoolUtil.getEmpTOPoolAssignmentsByActivity(m_empID,
								bigBangCal.getTime(), armageddonCal.getTime(), activityID);
						populateHistory(poolHistoryList);
					}
					
					//One or more employees are selected. We do not show effectivity history for multiple employees.
					//Make the collection of pools at the employees' common orgs and parent orgs (including their AsOfDate pool assignment)
					Date asOfDate = getAsOfDate();
					Collection pools = getCommonPools(selectedID, asOfDate);
					populateContainer(pools);
				}
			}
		} 
		catch (Exception ex) 
		{
			handleUnableToLoadDataError(log, ex);
		} 
		finally 
		{
			initContentTitle();
			initToolbar();
		}
	}
	
	private Date getAsOfDate() {
		if (m_viewDate != null) {
			return m_viewDate.getTime(TimeZone.getTimeZone("GMT"));
		}

		TOPoolSelectorContainerPC poolDataPC = (TOPoolSelectorContainerPC) m_dataPC;
		if (poolDataPC == null) {
			throw new RmRuntimeException("Could not get as of date");
		}

		return poolDataPC.getShowAsDate();
	}

	/**
	 * Get the ArrayList of pools at the employees' common orgs and parent orgs (including their AsOfDate pool assignment)
	 * @param empIDs - A comma-delimitted string of employee IDs for whom we are viewing the pool assignment effectivities.
	 * @param asOfDate - The view date for the pool assignments.
	 * @return A collection of TOPool objects that are available to ALL of the employees on the asOfDate. Pools available to
	 *         only some of the employees on the asOfDate will be excluded from the list. 
	 */
	Collection getCommonPools(String empIDs, Date asOfDate) throws Exception
	{
		Collection commonPools = new ArrayList();
		ID[] ids = StringUtil.parseIDString(empIDs);
		if (ids!=null && ids.length>0) 
		{
			for (int i=0; i<ids.length; i++)
			{
				ID asOfOrgID = getEmployeeOrgID(m_context, ids[i], asOfDate);
				Collection curPools;
				if (asOfOrgID != null)
					curPools = PeopleMH.getTOPoolsForOrg(asOfOrgID, true);
				else
					curPools = new ArrayList();

				//add this employee's AsOfDate pool if not already in the list
				curPools = addAsOfDatePool(curPools);

				if (i==0)
				{
					commonPools = curPools;
				}
				else
				{
					//only keep the curPools that are already in commonPools
					commonPools.retainAll(curPools);
				}				
			}
		}			
		return commonPools;
	}
	
	/**
	 * Return the Organization ID for the specified Employee
	 * @param empID The Employee ID
	 * @param asOfDate - date for which you want to get the orgID
	 * @return ID The Organization ID for the specified Employee
	 */
	private ID getEmployeeOrgID(RequestContext context, ID empID, Date asOfDate) throws RemoteException, BbmException {
		//fetch all assignments and then look for overlap
		try {
			Collection wrList = CacheUtilBBM.getWorkResourceAssnWithTZForWRID(empID);
			//now look for overlap with current time
			for (Iterator it= wrList.iterator(); it.hasNext();){
				WorkResourceAssignment wra = (WorkResourceAssignment)it.next();
				if (TimePeriodUtil.contains(wra.getStartTime(), wra.getEndTime(),
						asOfDate)){
					return wra.getOrganizationID();
				}
			}
			//still not found: use first available org
			WorkResourceAssignment wra = (WorkResourceAssignment)wrList.iterator().next();
			return wra.getOrganizationID();
		} catch (Exception ex){
			return null;
		}
	}
	
	/**
	 * Insert the user's pool on the AsOfDate if not already in the list.
	 * @param pools - a list of TOPools.
	 */
	private Collection addAsOfDatePool(Collection pools)
	{
		//m_poolID is the asOfDatePool ID, so it cannot be null
		if (m_poolID != null && !m_poolID.equals("*") && !m_poolID.isEmpty())
		{
			ID asOfDatePoolID = new ID(m_poolID);
			TOPool asOfDatePool = new TOPool();
			asOfDatePool.setID(asOfDatePoolID);
			asOfDatePool.setName(m_poolName);		
			
			if (pools == null)
			{
				pools = new ArrayList();
				pools.add(asOfDatePool);
				return pools;
			}
			else
			{
				for (Iterator it=pools.iterator(); it.hasNext();)
				{
					TOPool pool = (TOPool)it.next();
					if (pool.getId().equals(asOfDatePoolID))
						return pools;
				}
			}
			
			pools.add(asOfDatePool);
		}
		return pools;
	}

	/**
	 * Get the ID of the selected employee from the selection pane, and set m_empID and m_employee accordingly.
	 * @param aVal
	 */
	public void setPerformActionOnID(String aVal)
	{
		//let the parent class handle it
		setSelectedID(aVal);
		
		
		if (m_empID != null)
		{
			try
			{
				//get the employee
				m_employee = ValidationUtil.getEmployeeByID(m_empID, null, Employee.DETAIL_LEVEL_EMPLOYEE_TIMEBASEDPROPERTY_HISTORY);
			}
			catch (Exception e)
			{			
			}
		}
		
	}
	
	private void populateHistory(List<EmpTOPoolAssignment> poolHistoryList) 
	{
		m_historyPC.setHeightInPixels(160);
		TableHeaderPC headerPC = m_historyPC.getHeader();
		headerPC.addColumnHeaderData("", m_localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.FROM));
		headerPC.addColumnHeaderData("", m_localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.TO));
		headerPC.addColumnHeaderData("", m_localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.PEOPLE_TIMEOFF_POOL_HEADER));
		
		ArrayList listModel = new ArrayList(poolHistoryList.size());
		for(EmpTOPoolAssignment assignment : poolHistoryList ) 
		{
			
			DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData();
			node.setSelectable(false);
			node.add( formatDate(assignment.getStartTime()) );
			node.add( formatEndDate(assignment.getEndTime()) );
			node.add( assignment.getTOPoolName());
			listModel.add(node);
		}
		
		m_historyPC.setListModel(listModel);
		
	}
	
	/**
	 * Code to populate page model components
	 * @param toPools - The collection of TOPool's to add to the dropdown selector.
	 */
	private void populateContainer(Collection toPools)
	{	
		TOPoolSelectorContainerPC poolDataPC = (TOPoolSelectorContainerPC)m_dataPC;
		poolDataPC.setPoolId(m_poolID);
		poolDataPC.setDropdownData(toPools);
		
		//If the employee is not currently assigned to any pool, then the startDate and endDate will be null.
		//We cannot allow effective dates of null,null. We will force the user to set at least the startDate, which
		//means that the current view period radio button option for the effective dates must be disabled.
		if (m_poolID == null || m_poolID.equals(""))
		{
			poolDataPC.setOnlyViewPeriodEditable(false);
			poolDataPC.setOnlyCurPeriodEditable(false);
		}
	}
	
	/**
	 * Return URL for HTML Form Action
	 */
	@Override
	public String getFormAction()

	}
	
	/**
	 * Return Required HTML for this component
	 */
	@Override
	public String getRequiredHTML() {

		String html = super.getRequiredHTML() +
				HtmlUtil.makeHiddenInput(WorkpaneFieldNames.PERFORM_ACTION_ON_ID_FN, selectedID); //store the empIDs
		if (activityID != null) {
			HtmlUtil.makeHiddenInput("ActivityID", activityID);
		}
		return html;
	}

	public void setActivityID(ID activityID) {
		this.activityID = activityID;
		if (activityID != null) {
			addRequiredJavaScriptFile(RmJavaScriptFileID.TIME_OFF_POOL_BY_ACTIVITY_POPUP);
			this.m_onSetJS = String.format("timeOffPoolByActivity.set(%d);", activityID.toInt());
		}
	}
