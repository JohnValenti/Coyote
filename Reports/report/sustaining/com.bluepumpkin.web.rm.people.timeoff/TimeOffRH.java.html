public class TimeOffRH extends PeopleRequestHandler {

	/**
	 * Processes Request
	 *
	 * @param context The Request Context
	 */
	@Override
	public void processRequest(RequestContext context) throws Exception {
		if (isAction(context, PageAction.FILTER_CHANGE)) {
			processFilterChange(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		}
	}

	/**
	 * Process Filter Change
	 */
	private void processFilterChange(RequestContext context) {
		TimeOffPM model = new TimeOffPM(context);
		preProcessFilterChange(context, model);
	}

	protected void processSave(RequestContext context) 
	{
		if (!isAuthorized(context, PrivilegeKeys.TOM_EDITAVAILABLETIMEOFF)) 
			return;

		TimeOffPM model = new TimeOffPM(context);
		model.extractParameters(context.getRequest());
		model.validate();

		if (!model.hasError()) 
		{
			try 
			{
				Employee emp = model.getEmployee();
				boolean isMultiEdit = model.isMultiEdit();
				if (emp.getID() != null) 
				{
					if (!isMultiEdit) 
					{
						PeopleMH.updateEmployeeTimeOff(emp);
					} 
					else 
					{
						Collection exceptionCol = PeopleMH.updateEmployeeTimeOffMultiEdit(emp);
						if (exceptionCol != null && exceptionCol.size() > 0) 
						{
							Iterator it = exceptionCol.iterator();
							FacadeException anException = (FacadeException) it.next();
							Locale langLocale = context.getLocaleContext().getLanguageLocale();
							model.addPageMessage(Message.ERROR_TYPE, anException.getFirstException().getLocalizedMessage(langLocale));
						}
					}
					
					//Update the Time Off Pool effective dates. This will work for multi-edit, 
					//since the Employee object will have all the empIDs in emp.getMultiEditIDs().
					if (model.getMode().equals(TimeOffPoolAssignmentPC.MODE_SAVE_POOL)) 
					{
						Object startDate = model.getTOPoolEffStartDate();
						Object endDate = model.getTOPoolEffEndDate();
						ID poolID = model.getPoolID();
						PeopleMH.updateEmployeeTimeOffPoolEffectivity(emp, model.getAsOfDate(), poolID, startDate, endDate);
					}					
					
					if( model.referenceScheduleNeedsUpdate()) {
						EmployeeReferenceScheduleAssignmentUpdateParameters update = model.getUpdateAssignment();
						if(update != null) {
							TimeOffMH.updateEmployeeReferenceScheduleEffectivity(emp, model.getAsOfDate(), update);
						}
					}
					
					List<TOPoolAssignmentsByActivityUpdateParameters> modifiedPools = model.getModifiedPoolAssignmentsByActivity();
					TimeOffMH.updateTimeOffPoolsByActivity(emp, model.getAsOfDate(), modifiedPools);
					
					model.updateData();
				}
			} 
			catch (Exception ex) 
			{
				model.handleUnableToUpdateInfo(m_log, ex);
			}
		}

		context.setPageModel(model);
	}
	

}
