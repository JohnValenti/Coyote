package com.bluepumpkin.ejb.facade.schedulepref.ejb;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

import com.bluepumpkin.ejb.core.base.SessionEJBBase;

import com.bluepumpkin.common.logging.Category;

import com.bluepumpkin.ejb.facade.base.*;
import com.bluepumpkin.ejb.facade.schedulepref.model.*;
import com.bluepumpkin.ejb.facade.Log;

/**
 * Title:        Blue Pumpkin Software Facade Module
 * Description:  PeopleFacade implementation
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author       lixin zhang
 * @version      1.0
 */

public class SchedPrefFacadeManagerEJB extends SessionEJBBase {

    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static Category m_cat = Log.initCategory(SchedPrefFacadeManagerEJB.class.getName());

    /** override the base class to provide the appropriate logging category */
    protected Category getCategory()
    {
	return m_cat;
    }

    {
	super.init(SchedPrefFacadeManagerEJB.class.getName());
    }

    /**********************************************************************
     * for simple schedule preference
     **********************************************************************/
    /**
     *   get simple schedule preference for a given employee
     *
     *   @param idEmployee employee id
     *   @param dtViewDate view date
     *
     *   @return SimpleSchedPrefFacade, return null if object not found
     *
     */
    public SimpleSchedPrefFacade getSimpleSchedulePreference(ID idEmployee,
	 Date dtViewDate) throws FacadeException
    {
	methodStart("getSimpleSchedulePreference", idEmployee, dtViewDate);
	try
	{
	    return SimpleSchedPrefDAO.getSimpleSchedulePreference(idEmployee, dtViewDate);
	}
	catch(Exception e)
	{
	    handleException(e, false);      // no rollback
	    throw new FacadeException(e);
	}
	finally
	{
	    methodFinish();
	}
    }


    /**
     *   update simple schedule preference
     *
     */
    public ID updateSimpleSchedulePreference(SimpleSchedPrefFacade prefFacade)
	    throws FacadeException
    {
	methodStart("updateSimpleSchedulePreference", prefFacade);
	try
	{
	    return SimpleSchedPrefDAO.updateSimpleSchedulePreference(prefFacade);
	}
	catch(Exception e)
	{
	    handleException(e, false);      // no rollback
	    throw new FacadeException(e);
	}
	finally
	{
	    methodFinish();
	}
    }




}
