package com.bluepumpkin.web.bbm.people.staffingprofiles;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.ScopeID;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManager;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeTemplate;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.ejb.core.security.Scope;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarUtil;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.js.JSUtil;

/**
 * This Page Model is responsible for showing the Staffing Profiles linked to organizations (not to be confused with
 * com.verint.web.fs.staffingprofiles.StaffingProfilesListPM, which shows the Staffing Profiles linked
 * to an SP).
 */
public class StaffingProfilesListPM extends WorkRulesListPM {
	private static final long serialVersionUID = 1L;
	public static final String OBJECT_TYPE_EMPLOYEE_TEMPLATE = "EMPLOYEETEMPLATE"; 

	private Map<ID, Organization> orgMap = null;
	private Map<ID, Campaign> campaignMap = null;
	
	public StaffingProfilesListPM(RequestContext context) {
		super(context);

		option_showTimeZoneToggle = false;
		
		removeChildComponent(m_list);
		m_list = new MultiColListPC(context);
		m_list.setIsAutoSortEnabled(true);
		this.addChildComponent(m_list);
		this.setIsMultiRowSelectable(true);
	}

	private Collection<EmployeeTemplate> empTemplates = new ArrayList<EmployeeTemplate>();
	@Override
	protected void loadData() throws Exception {
		super.loadData();
		ScheduleAccessManager sam = WfmManagerFactory.getScheduleAccessManager(m_context.isInWhatIfMode());
		WorkResourceManager wrMgr = BbmManagerFactory.getWorkResourceManager(m_context.isInWhatIfMode());
		if (isByOrg && selectedOrg != null) {
			empTemplates = sam.getEmployeeTemplatesInOrg(selectedOrg.getID());
		} 
		
		orgMap = new HashMap<ID, Organization>();
		for (EmployeeTemplate sf : empTemplates) {
			ID id = sf.getOrganizationID();
			if (!orgMap.containsKey(id))
			{
				Organization org = wrMgr.getOrganizationByID(id);
				orgMap.put(id, org);
			}
		}
	}

	@Override
	protected void initHeader() {
	}

	public void initForDisplay() {
		if (m_isInitializedForDisplay) return;
		else {
			super.initForDisplay();

			// Enable Object Copy
			initCopyObjType(StaffingProfilesListPM.OBJECT_TYPE_EMPLOYEE_TEMPLATE);
		}
	}
	
	@Override
	protected void initSelectableItems() {
		this.removeChildComponent(m_list);
		m_list = new StaffingProfilesListPC(m_context, empTemplates, optimizationTargetTimeZone, 
				isByOrg, orgMap, campaignMap) {
			@Override
			protected void setRowAttributes(DefaultMultiColumnNodeData rowData,
					EmployeeTemplate empTemplate) {
				super.setRowAttributes(rowData, empTemplate);
				StaffingProfilesListPM.this.setRowAttributes(rowData, empTemplate.getOrganizationID());
				
				HashMap<String, String> nodeAttributes = rowData.getNodeAttributes();
				toggleBtnOnNodeSelect(rowData, WorkRulesRH.WORK_RULES_COPY_PAGE_ACTION,
						Boolean.parseBoolean(isAuthorized()));
				nodeAttributes.put(IS_EDITABLE,	isAuthorized());
				nodeAttributes.put(IS_DELETEABLE, isAuthorized());
				nodeAttributes.put(COPY_ITEM_NAME, empTemplate.getName());
			}	
		};
		m_list.setIsAutoSortEnabled(true);
		m_list.extractParameters(m_context.getRequest());
		this.addChildComponent(m_list);
	}
	
	protected String isRowCopyable(ID activityID, ID rowOrgID) {
		if (rowOrgID.equals(Organization.ROOT_ORG_ID_OBJ)) {			
			return "false";
		}
		return "true";
	}


	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		m_contentTitlePC.setPageTitle(m_localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.FS_STAFFING_PROFILES));
	}

	public static StaffingProfilesListPM getInstance(RequestContext context) {
		return (StaffingProfilesListPM)PageModel.getInstance(context, StaffingProfilesListPM.class);
	}
	
	@Override
	protected void initToolbar() {
		//Enable pagination
		m_toolbar.setPagination("test", this, true, JSUtil.REFRESH_PAGE);

		boolean enabled = false;
		if (selectedOrg != null) {
			enabled = m_context.getUser().isAuthorized(PrivilegeKeys.CORE_CONFIGUREORGANIZATION,
					new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE)) && 
					(m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES_ID,
							new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE)));
		}
		
		ToolbarUtil.addCreateBtn(m_toolbar, enabled);
		ToolbarUtil.addEditBtn(m_toolbar, false);
		ToolbarUtil.addDeleteBtn(m_toolbar, false);
		m_toolbar.addSubmitTextButton(WorkRulesRH.WORK_RULES_COPY_PAGE_ACTION, m_toolbar.i18nCore(UIFWebBundleKey.TOOLBAR_COPY), false);
	}

	protected String isAuthorized() {
		if (selectedOrg == null) return "false";
         if(m_context.getUser().isAuthorized(PrivilegeKeys.CORE_CONFIGUREORGANIZATION,
        		 new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE)) && 
        		 (m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES_ID,
        				 new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE)))) {      	 
        	 return "true";
         } else {
        	 return "false";
         }
    }
}
