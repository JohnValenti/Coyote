package com.bluepumpkin.web.bbm.people.staffingprofiles;

import static com.witness.web.uif.util.TimeZoneOverrideUtil.TIME_ZONE_OVERRIDE_TOGGLE_ACTION;

import java.rmi.RemoteException;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.web.core.base.RequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.TimeZoneOverrideUtil;

public abstract class WorkRulesRH<T extends WorkRulesListPM, U extends StaffingProfilesFormPM> extends RequestHandler {
	
	protected static final String LIST_PAGE = "staffingprofileslistPage";
	protected static final String FORM_PAGE = "staffingprofilesformPage";
	public static final String WORK_RULES_COPY_PAGE_ACTION = "WORK_RULES_COPY";
	
	protected boolean isByOrg = true;

	@Override
	public final void processRequest(RequestContext context) throws Exception {
		try {
			if(isAction(context, WORK_RULES_COPY_PAGE_ACTION)) {
				T pm = getListPageModel(context);
				processCopy(context, pm);
			}
			else if(isAction(context, PageAction.ADD)) {
				T pm = getListPageModel(context);
				processAdd(context, pm);
			}
			else if(isAction(context, PageAction.EDIT)) {
				T pm = getListPageModel(context);
				processEdit(context, pm);
			}
			else if(isAction(context, PageAction.DELETE)) {
				T pm = getListPageModel(context);
				processDelete(context, pm);
			}
			else if(isAction(context, PageAction.IMPORT)) {
				T pm = getListPageModel(context);
				processImport(context, pm);
			}
			else if(isAction(context, PageAction.EXPORT)) {
				T pm = getListPageModel(context);
				processExport(context, pm);
			}
			else if(isAction(context, PageAction.PREVIOUS)) {
				T pm = getListPageModel(context);
				processPageLeft(context, pm);
			}
			else if(isAction(context, PageAction.NEXT)) {
				T pm = getListPageModel(context);
				processPageRight(context, pm);
			}
			else if (isAction(context, TIME_ZONE_OVERRIDE_TOGGLE_ACTION)) {
				TimeZoneOverrideUtil.toggleTimeZoneOverride(context);
			}
			else if (isAction(context, PageAction.SAVE)) {
				U pm = getFormPageModel(context);
				processSave(context, pm);
			}
			else if (isAction(context, PageAction.CREATE_NEW)) {
				U pm = getFormPageModel(context);
				processSave(context, pm);
			} else {
				gotoList(context);
			}
		} catch (BPException ex) {
			PageModel pm = context.getPageModel();
			if (pm != null) handleUnableToUpdate(pm, ex);
		}

	}
	
	private void gotoList(RequestContext context) {
		setNextPage(context, LIST_PAGE);
	}
	
	abstract protected T getListPageModel(RequestContext context);
	abstract protected U getFormPageModel(RequestContext context);
	abstract protected void processImport(RequestContext context, T pageModel) throws BPException;
	abstract protected void processExport(RequestContext context, T pageModel) throws BPException;
	abstract protected void processCopy(RequestContext context, T pageModel) throws BPException;
	abstract protected void processDelete(RequestContext context, T pageModel) throws BPException, RemoteException;
	abstract protected void processSave(RequestContext context, U pageModel) throws BPException;

	protected void processAdd(RequestContext context, T pageModel) throws BPException {
		setNextPage(context, FORM_PAGE);
	}

	protected void processEdit(RequestContext context, T pageModel)
			throws BPException {
		setNextPage(context, FORM_PAGE);
	}
	
	protected void processPageLeft(RequestContext context, T pageModel) throws BPException {
		pageModel.extractParameters(context.getRequest());
		pageModel.adjustEditIndexBackward();		
		context.setPageModel(pageModel);
	}
	
	protected void processPageRight(RequestContext context, T pageModel) throws BPException {
		pageModel.extractParameters(context.getRequest());
		pageModel.adjustEditIndexForward();			
		context.setPageModel(pageModel);
	}
}
