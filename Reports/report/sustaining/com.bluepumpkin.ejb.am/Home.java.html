import com.bluepumpkin.common.util.SystemProperties;
import com.bluepumpkin.ejb.am.timecollection.ejb.TimeCollectorManagerHome;

import javax.naming.Context;
import javax.naming.InitialContext;
import javax.naming.NamingException;
import javax.rmi.PortableRemoteObject;
import java.util.Hashtable;


/**
 * This is a utility class for obtaining EJB references and other stuff
 */
public final class Home {
  /**
   * returns weblogic-specific initial context to be used for JNDI lookups
   */
  public static Context getInitialContext() throws NamingException {
    Hashtable h = new Hashtable();
    h.put(Context.INITIAL_CONTEXT_FACTORY, SystemProperties.get("CONTEXT_FACTORY"));
    h.put(Context.PROVIDER_URL, SystemProperties.get("PROVIDER_URL"));
    return new InitialContext(h);
  }

  /**
   * returns the home interface to TimeCollectorManager bject
   */
  public static TimeCollectorManagerHome getTimeCollectorManagerHome()
      throws Exception {
    try {
      Object objref = getInitialContext().lookup("bluepumpkin/timecollectormanager");
      return (TimeCollectorManagerHome)
          PortableRemoteObject.narrow(objref, TimeCollectorManagerHome.class);
    } catch (Exception e) {
      throw e;
    }
  }

}
