package com.bluepumpkin.ejb.bbm.rm;

import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.core.CoreManagerFactory;
import com.witness.ejb.core.gcr.ejb.GCRManager;
import com.witness.ejb.core.gcr.model.GCREntry;

public class ShiftSwapCalendarInfoManager {
	private ShiftSwapManagerBridge shiftSwapManagerBridge;

	public ShiftSwapCalendarInfoManager() throws Exception {
		this.shiftSwapManagerBridge = this.getShiftSwapManagerBridge();
	}

	
	// Remove returning shift swap calendar info until defects are worked out.
	// Per Richard, for 15.2.3 (HFR3)
	// Return empty list temporarily.
	public List<ShiftSwapCalendarInfo> getListApprovedShiftSwapCalendarInfo(Date startDateTimeGMT, Date endDateTimeGMT, List<ID> employeeIds)//NOSONAR
			throws Exception { //NOSONAR
		return Collections.emptyList();
	}

	@SuppressWarnings("rawtypes")
	private ShiftSwapManagerBridge getShiftSwapManagerBridge() throws Exception {
		if (shiftSwapManagerBridge == null) {
			GCRManager gcrManager = CoreManagerFactory.getGCRManagerRemote(false);
			Collection entries = gcrManager.getGCREntryOfType(ShiftSwapManagerBridge.GCRTYPE);
			Class clazz = Class.forName(((GCREntry) (entries.iterator().next())).getHook());
			shiftSwapManagerBridge = (ShiftSwapManagerBridge) clazz.newInstance();
		}
		return shiftSwapManagerBridge;
	}

}
