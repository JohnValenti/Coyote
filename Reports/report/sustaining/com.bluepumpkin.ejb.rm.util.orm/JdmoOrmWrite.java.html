package com.bluepumpkin.ejb.rm.util.orm;

import java.lang.reflect.InvocationTargetException;
import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.Set;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoPCommand;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;
import com.bluepumpkin.ejb.rm.util.DAOUtil;
import com.bluepumpkin.ejb.rm.util.DbObjectTypeUtil;
import com.bluepumpkin.ejb.rm.util.JdmoPreparedStatement;
import com.bluepumpkin.ejb.rm.util.RmUtil;

class JdmoOrmWrite<T> {

	PropertyGetter<T> propertyGetter;
	Class<T> entityClass;
	List<String> columnNames;

	JdmoOrmWrite(Class<T> entityClass){
		this(entityClass, null);
	}

	JdmoOrmWrite(Class<T> entityClass, WriteOptions options) {
		Set<String> includedProperties = options == null ? null : options.includedProperties;
		Set<String> excludedProperties = options == null ? null : options.excludedProperties;

		propertyGetter = new PropertyGetter<T>(entityClass,
				includedProperties,
				excludedProperties);
		this.entityClass = entityClass;
		this.columnNames = getColumnNames(options);
	}

	//converts property names to column names if aliases are specified
	List<String> getColumnNames(WriteOptions options){

		List<String> propertyNames = propertyGetter.getGetterPropertyNames();
		if (options == null || options.propertyAliases == null || options.propertyAliases.size() == 0) {
			//no column aliases are defined so the column names are the same as the property names
			return propertyNames;
		}

		List<String> result = new ArrayList<String>(propertyNames.size());
		for (String propName : propertyNames) {

			String columnName = options.propertyAliases.get(propName);
			if (columnName == null) {
				result.add(propName);
			} else {
				result.add(columnName);
			}
		}

		return result;
	}

	void insertIntoTempTable(Jdmo dmo, String tempTableName, Collection<T> objects) {
		if(tempTableName == null || tempTableName.isEmpty()) {
			throw new RmRuntimeException("Temp table name was not specified");
		}

		try {
			DAOUtil.createTempTable(dmo, tempTableName, getTempTableFieldDefinitions());
		} catch (JdmoException e) {
			throw new RmRuntimeException(e);
		}

		insert(dmo, tempTableName, objects);
	}

	void insert(Jdmo dmo, String insertTableName, Collection<T> objects) {
		try {
			String tableName = getAdjustedTableName(insertTableName);

			if (columnNames.isEmpty()) {
				throw new RmRuntimeException("JdmoOrm insert fails since there are no get methods on the entity class");
			}

			String sql = String.format("insert into [%s](%s) values (%s)",
					tableName,
					getCommaSeparatedColumnNames(),
					RmUtil.repeat("?", ",", columnNames.size()));

			JdmoPCommand pc = dmo.createPCommand(sql);
			HashMap<Integer, Object> param = new HashMap<Integer, Object>();
			for (T value : objects) {

				loadParamMap(param, value);

				pc.setParams(param);
				pc.addBatch();
				param.clear();
			}

			dmo.executeBatchPCommand(pc);
		} catch (Exception e) {
			throw RmRuntimeException.toRuntimeException(e);
		}
	}

	void loadParamMap(Map<Integer, Object> param, T value) throws IllegalAccessException, InvocationTargetException {
		for (int i = 0; i < columnNames.size(); i++) {
			Object colValue = propertyGetter.getProperty(value, i);
			if (colValue == null) {
				colValue = JdmoPreparedStatement.getNullObjectForType(propertyGetter.getPropertyType(i));
			}
			param.put(i + 1, colValue);
		}
	}

	String getAdjustedTableName(String tableName) {
		return tableName == null || tableName.isEmpty() ? propertyGetter.tableName : tableName;
	}

	String getCommaSeparatedColumnNames() {
		return StringUtil.join(columnNames, ",");
	}

	void merge(Jdmo dmo, String mergeTableName, Collection<T> objects) {

		String tableName = getAdjustedTableName(mergeTableName);
		try {
			fixUpMissingIds(dmo, tableName, objects);
			insertIntoTempTableForMerge(dmo, tableName, objects);
			mergeFromTempTable(dmo, tableName);
		} catch (Exception e) {
			throw RmRuntimeException.toRuntimeException(e);
		}
	}

	void fixUpMissingIds(Jdmo dmo, String tableName, Collection<T> objects) throws IllegalAccessException,
			InvocationTargetException, JdmoException {

		List<T> nullIDObjects = getObjectsWithNullIDs(objects);
		if (nullIDObjects.isEmpty()) {
			return;
		}

		int startID = getNextKey(dmo, tableName, nullIDObjects.size());

		PropertySetter<T> idSetter = new PropertySetter<T>(this.entityClass, new String[] { "ID" });

		for (int i = 0; i < nullIDObjects.size(); i++) {
			ID id = ID.fromInt(startID + i);
			Object nullIDObject = nullIDObjects.get(i);
			if (!idSetter.setProperty(nullIDObject, id, "ID")) {
				throw new RmRuntimeException("Could not set new ID value to object: " + nullIDObject);
			}
		}
	}

	//this is a separate function just for ease of unit testing fixUpMissingIds
	int getNextKey(Jdmo dmo, String tableName, int numKeysToReserve) throws JdmoException {
		return DAOUtil.getNextKey(dmo, tableName, numKeysToReserve);
	}

	List<T> getObjectsWithNullIDs(Collection<T> objects) throws IllegalAccessException,
			InvocationTargetException {
		List<T> result = new ArrayList<T>();
		for (T obj : objects) {

			Object id = propertyGetter.getProperty(obj, "ID");
			if (id == null) {
				result.add(obj);
			}
		}
		return result;
	}

	void insertIntoTempTableForMerge(Jdmo dmo, String tableName, Collection<T> objects) throws JdmoException {
		insertIntoTempTable(dmo, getMergeTempTableName(tableName), objects);
	}

	String getMergeTempTableName(String tableName) {
		return "#merge_" +  tableName;
	}

	//returns a string of type fieldName1 fieldType1, fieldName2 fieldType2 that can be used with DAOUtil.createTempTable
	String getTempTableFieldDefinitions() {
		StringBuilder builder = new StringBuilder();
		int columnCount = columnNames.size();
		for (int i = 0; i < columnCount; i++) {
			builder.append(columnNames.get(i));
			builder.append(" ");
			String sqlType = DbObjectTypeUtil.getSqlTypeFromJavaType(propertyGetter.getPropertyType(i));
			builder.append(sqlType);

			if (i != (columnCount - 1)) {
				builder.append(",");
			}
		}
		return builder.toString();
	}

	void mergeFromTempTable(Jdmo dmo,String tableName) throws JdmoException {

		String mergeSql = ";merge %s as dest "
				+ " using ( select * from %s ) as src "
				+ " on src.ID = dest.ID "
				+ " when matched then update set %s "
				+ " when not matched by target then "
				+ " insert (%s) Values (%s); ";

		mergeSql = String.format(mergeSql,
				//destination table name
				tableName,
				//source table name 
				getMergeTempTableName(tableName),
				//update list of the form dest.field = src.field
				getUpdateList(),
				// column list
				getCommaSeparatedColumnNames(),
				//source field column list
				getPrefixedColumnNames("src.")
				);

		dmo.execute(mergeSql);

	}

	String getUpdateList() {
		List<String> updateFields = new ArrayList<String>(columnNames.size());
		for (String name : columnNames) {
			if ("ID".equalsIgnoreCase(name)) {
				continue;
			}
			updateFields.add(String.format("dest.%s = src.%s", name, name));
		}
		return StringUtil.join(updateFields, ",");
	}

	String getPrefixedColumnNames(String prefix) {
		List<String> fieldList = new ArrayList<String>(columnNames.size());
		for (String name : columnNames) {
			fieldList.add(prefix + name);
		}
		return StringUtil.join(fieldList, ",");
	}

}
