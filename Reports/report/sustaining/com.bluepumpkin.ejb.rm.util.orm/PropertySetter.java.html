package com.bluepumpkin.ejb.rm.util.orm;

import java.lang.reflect.InvocationTargetException;
import java.lang.reflect.Method;
import java.util.Collection;
import java.util.Date;
import java.util.Map;
import java.util.TreeMap;

import com.bluepumpkin.common.datatypes.ID;

class PropertySetter<T> {

	//Property name (without the "set") to BaseAccessor map
	Map<String, BaseAccessor> setterCache;

	PropertySetter(Class<T> theClass, String[] propertyNames) {
		setterCache = new TreeMap<String, BaseAccessor>(String.CASE_INSENSITIVE_ORDER);
		Collection<Method> methods = OrmUtils.getMethods(theClass);
		for (String propName : propertyNames) {
			setterCache.put(propName, getSetterMethod(methods, propName));
		}
	}

	boolean setProperty(Object target, Object value, String propertyName) throws IllegalAccessException,
			InvocationTargetException {

		BaseAccessor setter = setterCache.get(propertyName);
		if (setter != null) {
			setter.setProperty(target, value);
			return true;
		}
		return false;
	}

	private static BaseAccessor getSetterMethod(Collection<Method> methods, String propertyName) {

		for (Method method : methods) {
			final String methodName = method.getName();
			if (method.getParameterTypes().length == 1 && methodName.startsWith("set")) {
				String accessorMMethodName = methodName.substring(3);
				if (accessorMMethodName.equalsIgnoreCase(propertyName)) {
					return getSetter(method);
				}
			}
		}
		return null;
	}

	private static BaseAccessor getSetter(Method m) {
		Class<?> paramType = m.getParameterTypes()[0];
		//custom setter for ID
		if (paramType == ID.class) {
			return new IDAccessor(m);
		}
		if (paramType == Date.class) {
			return new DateAccessor(m);
		}
		return new BaseAccessor(m);
	}

}
