package com.bluepumpkin.ejb.rm.util.orm;

import java.util.Map;
import java.util.Set;
import java.util.TreeMap;

import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;
import com.bluepumpkin.ejb.rm.util.RmUtil;

public class WriteOptions {

	String tableName;
	Set<String> excludedProperties;
	Set<String> includedProperties;
	//Property name to DB column name
	Map<String, String> propertyAliases;

	public WriteOptions() { // NOSONAR

	}

	public WriteOptions(String tableName) {
		setTableName(tableName);
	}

	public void setTableName(String tableName) {
		this.tableName = tableName;
	}

	/**
	 * Comma separated list of property names to exclude from the write operations. 
	 * If nothing is specified then no properties are excluded.
	 * The names should be the property names in the entity class before any aliases are applied
	 */
	public void setExcludedProperties(String excludeProps) {
		excludedProperties = RmUtil.parseStringToCaseInsensitiveSet(excludeProps.replaceAll("\\s+", ""), ",");
	}

	/**
	 * Comma separated list of property names to include in the write operations.
	 * If this is specified only the specified properties are included in write operations. 
	 * If nothing is specified then all properties are included.   
	 * The names should be the property names in the entity class before any aliases are applied
	 */
	public void setIncludedProperties(String includedProps) {
		includedProperties = RmUtil.parseStringToCaseInsensitiveSet(includedProps.replaceAll("\\s+", ""), ",");
	}

	/**
	 * Comma separated list of  property to database column aliases
	 * Column aliases are specified using the following format:
	 * Property name in the entity class as database column name
	 * for example:
	 * prop1 as col1, prop2 as col2 
	 * Implies that the entity class has a method called getProp1() and the database has a column called col1 and that getProp1() value
	 * should be persisted to the column col1
	 * @param aliases
	 */
	public void setAliases(String aliases) {

		propertyAliases = new TreeMap<String, String>(String.CASE_INSENSITIVE_ORDER);

		for (String alias : StringUtil.parseStringToCollection(aliases, ",")) {
			String[] tokens = alias.split("(?i)\\s+as\\s+");
			if (tokens.length != 2) {
				throw new RmRuntimeException("JdmoOrm: Could not parse column aliases");
			}

			String propertyName = tokens[0].trim();
			String columnName = tokens[1].trim();

			if ("ID".equalsIgnoreCase(propertyName)) {
				throw new RmRuntimeException("JdmoOrm: Column alias for ID is not current supported");
			}

			if (propertyName.isEmpty() || columnName.isEmpty()) {
				throw new RmRuntimeException("JdmoOrm: Could not parse column aliases. Property or column name is missing");
			}

			propertyAliases.put(propertyName, columnName);
		}

	}


}
