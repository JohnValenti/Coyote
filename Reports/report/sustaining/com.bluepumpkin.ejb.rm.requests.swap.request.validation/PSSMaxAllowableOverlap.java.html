/**
 *
 * <b>Soft Validation: applies for all kinds of shift swap requests</b>
 * <p>
 *    Validates that the overlap duration of swapped durations and existing shifts is not over the maximum time specified in the rule
 * <p>
 *    Checks for both the shifts for a two way swap
 */
public class PSSMaxAllowableOverlap  implements Validator
{
    private static final String m_className = PSSMaxAllowableOverlap.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);

    public static final String MAX_OVERLAP_VALUE = "MAX_OVERLAP_VALUE";
    public static final ID RULE_ID = new ID(-194037);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    @Override
    public ValidationResult validate(Validatable validatable) throws Exception
    {
        String methodName = "validate";
        if (m_cat.isDebugEnabled()) {
            m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));
        }

        ShiftSwapRequest ssr = (ShiftSwapRequest) validatable;
        ShiftSwapValidationCache valCache = ssr.getCache();
        ValidationResult result = null;
        //get the shifts corresponding to the items
        ScheduleAccessManager sam = RequestUtil.getScheduleAccessManager();

        // ESR 4373486 Fix by Angela Yu
        // shiftsToStartInsideBoundary should be false
        ShiftAssignment[] sas = ShiftSwapRequestUtil.getSwappedShiftAssignsForReq(ssr, true, false, false, sam);

        Collection<ID> orgList = new HashSet<ID>(ssr.getValidationCache().getOrgIDs());

//        	for each org, find out the validator- if beeded to run for both or not
        for (ID orgID : orgList) {
            if (valCache.isValidationRequiredForOrg(orgID, Request.REQUESTTYPE_SHIFTSWAP, m_className)) {
                //get the rule params
                ValidationRuleManager valMan = RmManagerFactory.getInstance().getValidationRuleManager();
                Map<String, Integer> ruleParams = valMan.getValidationRuleParams(orgID, RULE_ID);
                int duration = ruleParams.get(MAX_OVERLAP_VALUE);

                for (int j = 0; j < sas.length; j++) {

                    ValidationResult valResult = verifyMaxAllowableOverlap(sas[j], duration, ssr, ssr.getShiftSwapItems().get(j), valCache);
                    if (valResult != null && result == null) {
                        // set result but do not stop the check so all violations will be found
                        result = valResult;
                    }
                }
            }
        }
        if (m_cat.isDebugEnabled()) {
            m_cat.debug(RmUtil.dumpExitMethod(methodName));
        }
        return result;
    }


    private ValidationResult verifyMaxAllowableOverlap(ShiftAssignment sa, int duration,
            ShiftSwapRequest ssr, ShiftSwapItem ssItem, ShiftSwapValidationCache valCache) throws Exception{

        // if there is no existing shift on receiving side, skip the check
        if (sa == null) {
            return null;
        }
        // if item type is "time-off", skip the check
        if (!ssItem.isSwapTypeShift()) {
            return null;
        }

        // find overlap time
        Timeline ssItemTime = new Timeline(ssItem.getStartDate(), ssItem.getEndDate());
        Timeline shiftTime = new Timeline(sa.getStartTime(), sa.getEndTime());
        Timeline overlapTimeShiftAndRequest = shiftTime.intersect(ssItemTime);
        @SuppressWarnings("unchecked")
        Timeline gapTime = Timeline.newInstance(ValidationCache.getAllGaps(sa, Collections.emptyList()));

        // if two swapped blocks overlap, exclude the common time of the blocks from calculation
        Timeline overlapTimeOfSwappedItems = getOverlapTimeOfSwappedItems(ssr);

        Timeline actualOverlapTimeOfShiftAndRequest = overlapTimeShiftAndRequest.subtract(gapTime).subtract(overlapTimeOfSwappedItems);

        long overlapDuration = actualOverlapTimeOfShiftAndRequest.getTotalTimeInMiliseconds() / (1000*60);

        if (overlapDuration > duration) {
            return ValidationUtil.setSoftValidationResult(ssr,
                    RmEjbBundleKey.PSS_SHIFT_OVERLAP_DURATION_MORE_THAN_MAX,
                    String.valueOf(duration),
                    ssItem.getStartDate(),
                    ssItem.getEndDate(),
                    valCache.getEmployeeNameByID(ssItem.getEmployeeID()),
                    m_className);
        }
        return null;
    }

    private Timeline getOverlapTimeOfSwappedItems(ShiftSwapRequest ssr) {
        Timeline result;
        ShiftSwapItem ssItem1 = ssr.getShiftSwapItems().get(0);
        ShiftSwapItem ssItem2 = ssr.getShiftSwapItems().get(1);
        if (ssItem1.isSwapTypeShift() && ssItem2.isSwapTypeShift()) {
            Timeline ssItem1Time = new Timeline(ssItem1.getStartDate(), ssItem1.getEndDate());
            Timeline ssItem2Time = new Timeline(ssItem2.getStartDate(), ssItem2.getEndDate());
            result = ssItem1Time.intersect(ssItem2Time);
        } else {
            result = new Timeline();
        }
        return result;
    }

}
