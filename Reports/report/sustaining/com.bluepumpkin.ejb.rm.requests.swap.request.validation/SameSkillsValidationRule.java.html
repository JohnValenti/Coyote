/**
 * <li> Rule from UI: Both agents have ****exactly**** same skills for the active campaign queue
 * <p/>
 * <li> Ensure that the relevant skills for an employee
 * giving up his shift matches the skills of the employee
 * receiving the shift during the time period of the shift involved.
 * <p/>
 * todo: talk to jason.  need to compare only relevant skills ie. relevant skills of
 * the employee whose shift is being given up must match the skills of the employee
 * receiving it during the shift duration.
 * ### Employee receiving the shift cannot have more skills
 * <p/>
 * For 2 way swap, validaton done for both shift swap items of the swap request <br>
 * For 1 way swap, validation done only for the swap item with the shift. <br>
 * <p/>
 * <p/>
 * Rule applies equally for tentative and non tentative requests as only
 * skills are verified ( no shift information is used)
 */
public class SameSkillsValidationRule implements Validator {
	private static final String CLASS_NAME = SameSkillsValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	private ShiftSwapValidationCache validationCache;

	public SameSkillsValidationRule() {
	};

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ShiftSwapRequest ssr = (ShiftSwapRequest) validatable;
		String methodName = "validate";
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod(methodName, ssr));
		}

		ValidationResult valResult = null;

		validationCache = (ShiftSwapValidationCache) ssr.getValidationCache();

		// ssitems is an ordered list item(0) ... item(N-1)
		// item(A) gives its shift to item((A+1)modN)
		// item(A) receives its new shift from item((A+N-1)modN)
		List ssItems = ssr.getShiftSwapItems();

		int itemsSize = ssItems.size();
		for (int i = 0; i < itemsSize; i++) {
			int inIndex = (i + itemsSize - 1) % itemsSize;
			ShiftSwapItem itemIn = (ShiftSwapItem) ssItems.get(inIndex);
			ShiftSwapItem itemOut = (ShiftSwapItem) ssItems.get(i);

			// no skills required for time off
			if (itemIn.getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF))
				continue;

			// the SkillAssignments need to be compared for two employees,
			// for the time period encompassing the shift being received.
			Date itemInStart = itemIn.getStartDate();
			Date itemInEnd = itemIn.getEndDate();
			List<ShiftSwapItem> items = new ArrayList<>();
            items.add(itemIn);
            items.add(itemOut);
			Set<ID> skillsLikedToSP = new HashSet<>(validationCache.getRelevantSkillsForSchedulingPeriod(items));
			
			@SuppressWarnings("unchecked")
			Collection<SkillAssignment> inEmpSkillAssignments = validationCache.getSkillsForEmployeeDuringPeriod(itemIn.getEmployeeID(),
					itemInStart,
					itemInEnd);
			@SuppressWarnings("unchecked")
			Collection<SkillAssignment> outEmpSkillAssignments = validationCache.getSkillsForEmployeeDuringPeriod(itemOut.getEmployeeID(),
					itemInStart,
					itemInEnd);
			ValidationResult tempValResult = compareSkills(ssr,
					skillsLikedToSP,
			        getSkillIDs(inEmpSkillAssignments),
			        getSkillIDs(outEmpSkillAssignments),
			        itemOut.getEmployeeID(),
			        itemInStart,
			        itemInEnd);

			// overwrite the actual validation result only if the last validation failed, ie result is 'non null'.
			// Otherwise, a validation result set in the first iteration might be overwritten by a 'null'
			// validation result in the 2nd iteration.
			valResult = (tempValResult != null) ? tempValResult : valResult;
		}

		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpExitMethod(methodName));
		}
		return valResult;
	}
	
	private Set<ID> getSkillIDs(Collection<SkillAssignment> skillAssignments) {
		Set<ID> set = new HashSet<>();
		if (skillAssignments != null) {
			for (SkillAssignment assignment : skillAssignments) {
				set.add(assignment.getSkillID());
			}
		}
		return set;
	}

	/**
	 * @param skillsAssnIn  the Collection of SkillAssignments from the employee
	 *                      who is giving a shift away
	 * @param skillsAssnOut the Collection of SkillAssignments for the employee
	 *                      who is receiving a shift
	 * @param idEmp         the ID of the employee who is receiving a shift
	 * @return a ValidationResult containing information about the employee
	 *         who is lacking some Skill, and the name of the Skill which is lacking.
	 * @throws Exception
	 */
	ValidationResult compareSkills(Validatable validatable, Set<ID> skillsLinkedToSP, Set<ID> skillAssnsForItemInEmp,
			Set<ID> skillAssnsForItemOutEmp, ID idEmp, Date itemInStart, Date itemInEnd) throws Exception {

		if (skillAssnsForItemOutEmp.equals(skillAssnsForItemInEmp)) {
			//The skills match
			return null;
		}
		
		//If the receiving employee is missing a skill that the employee giving away the shift has, then report it as an error
		for (ID inSkillID : skillAssnsForItemInEmp) {
			if (skillsLinkedToSP.contains(inSkillID) && !skillAssnsForItemOutEmp.contains(inSkillID)) {
				return getValidationResult(validatable, idEmp, itemInStart, itemInEnd, inSkillID);
			}
		}
		return null;
	}

	//just so that we can override for unit testing
	ValidationResult getValidationResult(Validatable validatable, ID idEmp, Date itemInStart,
			Date itemInEnd, ID skillID) throws Exception {
		return ValidationUtil.setSoftValidationResult(validatable, RmEjbBundleKey.SS_SAME_SKILLS,
				validationCache.getEmployeeNameByID(idEmp), ValidationUtil.getSkillName(skillID),
				itemInStart, itemInEnd, CLASS_NAME);
	}
}
