package com.verint.web.rest.planner.service;

import java.util.ArrayList;
import java.util.Collection;

import java.util.List;

import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

import javax.ws.rs.core.Response;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.PathParam;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;
import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.fs.Log;
import com.verint.web.rest.planner.entity.QueuesEntity;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.SystemWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.planner.facade.IQueuesFacade;
import com.verint.web.rest.security.WFOSecurity;
import com.verint.web.rest.service.AbstractWFOEntityService;
import com.verint.web.rest.util.FormatUtils;
import com.witness.web.uif.system.RequestContext;

@Path("/queues/")
@Produces("application/json")
@WFOSecurity
public class QueuesRestService extends AbstractWFOEntityService<QueuesEntity> {
	private static final Category logger = Log.initCategory(QueuesRestService.class.getName());
	private IQueuesFacade queuesFacade;
	@Context
	private HttpServletRequest request;
	@Context
	private HttpServletResponse response;
	@Context
	private ServletContext sc;

	public QueuesRestService() {
	}

	public IQueuesFacade getQueuesFacade() {
		return queuesFacade;
	}

	public void setQueuesFacade(IQueuesFacade queuesFacade) {
		this.queuesFacade = queuesFacade;
	}

	public RequestContext getRequestContext() {
		return new RequestContext(request, response, sc);
	}

	@GET
	@Path("getQueuesWithParameter/{isExport}/{isLTF}")
	public QueuesList<QueuesEntity> getQueues(@PathParam("isExport") String isExport,
			@PathParam("isLTF") String isLTF) throws WFOException {
		try {

			RequestContext context = getRequestContext();
			boolean bIsExport = "true".equals(isExport);
			boolean bIsLTF = "true".equals(isLTF);

			return new QueuesList<QueuesEntity>(
					queuesFacade.getQueues(context, bIsExport, bIsLTF));

		} catch (WFOException ex) {
			throw ex; // should already have been logged
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	@GET
	@Path("/getqueuesintheperiod/{startDate}/{endDate}")
	public QueuesList<QueuesEntity> getQueuesWithDataInThePeriod(@PathParam("startDate") String strStartTime,
			@PathParam("endDate") String strEndTime) throws WFOException {
		try {

			RequestContext context = getRequestContext();
			Date startTime = null;
			Date endTime = null;
			startTime = FormatUtils.iso8601SecPrecisionStringToDate(strStartTime);
			endTime = FormatUtils.iso8601SecPrecisionStringToDate(strEndTime);

			if (startTime == null || endTime == null) {
				throw new BusinessWFOException("Date is null", Response.Status.CONFLICT);
			}
			return new QueuesList<QueuesEntity>(
					queuesFacade.getQueuesWithHistoryTimeseriesInThePeriod(context, startTime, endTime));

		} catch (WFOException ex) {
			throw ex; // should already have been logged
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	@Override
	public QueuesList<QueuesEntity> getAllEntities() throws WFOException {
		return null;
	}

	@Override
	public QueuesEntity getEntity(@PathParam("id") String id) throws WFOException {
		return new QueuesEntity();
	}

	@Override
	public QueuesEntity updateEntity(QueuesEntity employee) throws WFOException {
		return new QueuesEntity();
	}

	@Override
	public QueuesEntity addEntity(QueuesEntity employee) throws WFOException {
		return new QueuesEntity();
	}

	@Override
	public QueuesEntity deleteEntity(@PathParam("id") String id) throws WFOException {
		return new QueuesEntity();
	}

	@Override
	public boolean addEntities(Collection<QueuesEntity> entities) throws WFOException {
		return true;
	}

	@Override
	public boolean updateEntities(Collection<QueuesEntity> entities) throws WFOException {
		return true;
	}

	@Override
	public boolean deleteEntities(Collection<QueuesEntity> entities) throws WFOException {
		return true;
	}

	@XmlRootElement(name = "queues")
	private class QueuesList<QueuesEntity> extends ArrayList<QueuesEntity> {
		public QueuesList(List<QueuesEntity> wrapped) {
			super(wrapped);
		}
	}
}
