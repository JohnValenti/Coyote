package com.bluepumpkin.ejb.am.rtaaexceptioncategory.ejb;

import java.util.Collection;

import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;

import com.bluepumpkin.ejb.am.rtaaexceptioncategory.model.RTAAExceptionCategoryFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class RTAAExceptionCategoryDAO extends DAOBase {

		private static FieldInfo m_fieldInfo = new RTAAExceptionCategoryFieldInfo();
		protected FieldInfo getFieldInfo() { return m_fieldInfo; }

		public RTAAExceptionCategoryDAO() {
			 super();
		}

		public RTAAExceptionCategoryDAO(Jdmo dmo) {
			 super(dmo);
		}

		protected ValueObjectBase createValueObject()
		{
				return (new RTAAExceptionCategory());
		}
        
        /**
         *   get exception categories for a list given orgs and its parent orgs of a specified time period 
         *   @return a hashmap of (idOrg, collection of Holidays)
         */
		public Collection getRTAAExceptionCategories(Collection idOrgColl)
															 throws BbmFinderException
		{ 
			// format the query
			StringBuffer strWhere = new StringBuffer(30);

			try {
				strWhere.append("A.ORGANIZATIONID in ").append(m_dmo.createInClause(idOrgColl));
				
				Collection col = getObjects(strWhere.toString());
	            if (col == null || col.isEmpty()) return Collections.emptyList();
	            
	            return getObjects(strWhere.toString(), null);
			} catch(JdmoException e) {
				throw new BbmFinderException(e);
			}
		}
}
