package com.bluepumpkin.ejb.rm.setup.validation.model;

import java.io.Serializable;
import java.util.Map;

/**
 * Title:        ValidationSetting
 * Description:  Represents an organization's Validation Rule configurations
 * Copyright:    Copyright (c) 2009
 * Company:      Verint Systems, Inc.
 * @author       Gonzalo Quintanar
 */
public class ValidationSetting implements Serializable
{
    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public Map<String, String > settings; //all the key/value pairs for this org (currentOrgId)
    public Map ancestors; //maps each inheritted key to the ancestor orgName from which it was inheritted
    private String currentOrgId;
    private boolean hasChildren = false; //tells us whether currentOrgId has any children
   

    public ValidationSetting(Map<String, String > settings) 
    {
    	this.settings = settings;
    }
    
    public ValidationSetting(Map<String, String > settings, String currentOrgId) 
    {
    	
    	this.settings = settings;
        this.currentOrgId = currentOrgId;
       
    }
 
    public ValidationSetting(Map<String, String > settings, Map ancestors, String currentOrgId) 
    {
    	this.settings = settings;
    	this.ancestors = ancestors;
    	this.currentOrgId = currentOrgId;
    	
    }
    
    public Map getValues() {
        return settings;
    }
    public void setValues(Map<String, String > settings) {
    	this.settings = settings;
    }

    public Map getAncestors() {
        return ancestors;
    }
    public void setAncestors(Map ancestors) {
    	this.ancestors = ancestors;
    }

    public void setValue(String key, String value) {
        settings.put(key, value);
    }
    public String getValue(String key) {
        return (String)settings.get(key);
    }

    public void setAncestor(String key, String ancestor) {
    	this.ancestors.put(key, ancestor);
    }
    public String getAncestor(String key) {
        return (String)ancestors.get(key);
    }

    public void setHasChildren(boolean hasChildren)
    {
    	this.hasChildren = hasChildren;
    }
    public boolean getHasChildren()
    {
        return hasChildren;
    }

}
