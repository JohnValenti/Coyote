package com.bluepumpkin.web.core.applet.debug;

import java.io.Serializable;

import com.bluepumpkin.web.core.keys.PageKeys;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        AppletDebugManager
 * Description:  Responsible for Keeping track of Applet Debug properties.
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author       Alvin Cham
 * @version 1.0
 */
public class AppletDebugManager implements Serializable {
	private static final AppletDebugManager m_manager = new AppletDebugManager();

	/**
	 * Private Constructor
	 */
	private AppletDebugManager() {}

	/**
	 * Return Instance of AppletDebugManager
	 */
	public static AppletDebugManager getInstance() { return m_manager; };

	/**
	 * Return TRUE if AppletDebug is enabled
	 */
	public boolean isEnabled(RequestContext context) {
		Boolean b = (Boolean)context.getAttribute(RequestContext.APPLICATION_SCOPE, PageKeys.APPLET_DEBUG);
		if (b == null)
			return false;
		return b.booleanValue();
	}

	public void setEnabled(RequestContext context, boolean enabled) {
		context.setAttribute(RequestContext.APPLICATION_SCOPE, PageKeys.APPLET_DEBUG, Boolean.valueOf(enabled));
	}
}
