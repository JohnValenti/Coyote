package com.bluepumpkin.web.core.applet.debug;

import com.witness.web.uif.l10n.*;
import com.witness.web.uif.base.*;
import com.witness.web.uif.keys.*;
import com.witness.web.uif.system.*;

/**
 * Title:        AppletDebugRequestHandler.java
 * Description:  Handles Applet Debug Page Request
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       Alvin Cham
 * @version      1.0
 *
 * Created on Feb 23, 2004, 11:02 AM
 */
public class AppletDebugRequestHandler extends RequestHandler {

	/** Process the request.  This is the base class for all RequestHandler classes.
	 * Use RequestHandler classes to process those kinds of tasks
	 * that are not page or web display specific.  For example, an UpdateEmployeeInfo
	 * request might save new information into an Employement Model, but doesn't
	 * do anything that results in data being viewed.
	 * <br><br>
	 * @param context This is the request context object that contains all of the
	 * containers used by the request.  Passed into the class by the
	 * RequestProcessor when the RequestHandler is invoked.
	 * <br><br>
	 * @exception WebException - usually subclassed to the actual exception type.
	 */
	public void processRequest(RequestContext context) throws WebException {
		// Verify privilege
		if (!isAuthorized(context)) return;

		AppletDebugManager manager = AppletDebugManager.getInstance();

		// Perform action
		if (isAction(context, PageAction.ENABLE_APPLET_DEBUG)) {
			manager.setEnabled(context, true);
		} else if (isAction(context, PageAction.DISABLE_APPLET_DEBUG)) {
			manager.setEnabled(context, false);
		}

		// Extract parameter
		AppletDebugPageModel model = new AppletDebugPageModel(context);
		model.extractParameters(context.getRequest());

		// Initialise Model with Hit Manager information
		context.setPageModel(model);
	}

	/**
	 * Verify that current user contain the specified privilege
	 *
	 * Add default error message if not authorized
	 */
	protected boolean isAuthorized(RequestContext context) {
		boolean isAuthorized = AppletDebugPageModel.isAuthorized(context.getUser());

		// Create Error Message
		if (!isAuthorized) {
			Message errorMsg = new Message(Message.ERROR_TYPE,
					UIFWebBundleKey.NO_PRIVILEGE_ACCESS_DENIED,
					UIFWebBundleKey.BUNDLE_NAME);
			context.addPageMessage(errorMsg);
		}

		return isAuthorized;
	}    
}
