package com.verint.web.fs.campaign;

import java.util.Arrays;
import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmTimePeriodOverlapException;
import com.bluepumpkin.ejb.bbm.campaign.ejb.CampaignManager;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.workload.ejb.WorkloadManager;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

public class CampaignSPQueueListPopupRH extends RequestHandler {
	@Override
	public void processRequest(RequestContext context) throws Exception {
		if(isAction(context, PageAction.ADD)) {
			CampaignSPQueueListPopupPM pm = CampaignSPQueueListPopupPM.getInstance(context);
			pm.setShouldCloseOnLoad(true);
			pm.extractParameters(context.getRequest());
			processSave(context, pm);
		} else {
			CampaignSPQueueListPopupPM pm = new CampaignSPQueueListPopupPM(context);
			pm.extractParameters(context.getRequest());
			context.setPageModel(pm);
		}
	}
	
	 void processSave(RequestContext context, CampaignSPQueueListPopupPM pm) {
		try {
			Collection<ID> col= pm.getSelectedQueueIDs(context);
			CampaignManager camMgr = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
			WorkloadManager wMgr = WfmManagerFactory.getWorkloadManager(context.isInWhatIfMode());
			ID spID = pm.getSPID();
			for (ID id: col) {
	            SPQueue spqObjValue = new SPQueue();
	            spqObjValue.setSpID(spID);
	            spqObjValue.setQueueID(id);
	            Queue queue = wMgr.getQueueByID(id);
	            spqObjValue.setMediaID(queue.getMediaID());
	            camMgr.createSPQueue(spqObjValue);
			}
		} catch(BbmTimePeriodOverlapException e){
			Object [] args = e.getParamObject();
			String[] stringArray = Arrays.copyOf(args, args.length, String[].class);
			pm.addPageMessage(Message.ERROR_TYPE ,FsWebBundleKey.INVALID_QUEUE_TO_SP_LINKING, 
					FsWebBundleKey.BUNDLE_NAME, stringArray);
			pm.setShouldCloseOnLoad(false);
			context.setPageModel(pm);
		} catch (Exception e){
			pm.addPageMessage(Message.ERROR_TYPE , e.getLocalizedMessage());
			pm.setShouldCloseOnLoad(false);
			context.setPageModel(pm);
		}
		
	}
}
