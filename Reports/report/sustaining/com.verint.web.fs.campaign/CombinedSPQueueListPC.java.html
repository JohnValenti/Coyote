package com.verint.web.fs.campaign;

import java.util.Collection;

import javax.servlet.http.HttpServletRequest;

import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.MultiColumnNodeData;
import com.witness.web.uif.pagecomponent.form.FormInputPC;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;

/**
 * A "list" which contains one element - a combined queue.  This is shown
 * in a separate list from the other queues in order to better distinguish it.
 */
public class CombinedSPQueueListPC extends MultiColListPC {

	private SPQueue spQueueCombined;
	private FormInputPC m_formInputPC;
	
	public CombinedSPQueueListPC(RequestContext context, String name, SPQueue spQueueCombined, SchedulingPeriod sp) {
		super(context);

		setName(name);
		this.spQueueCombined = spQueueCombined;
		m_formInputPC = new FormInputPC(context);
		addChildComponent(m_formInputPC);
		setRowSelectable(false);

		extractParameters(m_context.getRequest());
		
		initRows();
		initHeader();
	}
	
	protected void initHeader() {
		TableHeaderPC header = getHeader();

		header.addColumnHeaderData(FsWebBundleKey.SCHEDULINGPERIOD_LTI,
				m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_LTI), false);
		header.addColumnHeaderData(FsWebBundleKey.SCHEDULINGPERIOD_THRESHOLD_OVER,
				m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_THRESHOLD_OVER), false);
		header.addColumnHeaderData(FsWebBundleKey.SCHEDULINGPERIOD_THRESHOLD_UNDER,
				m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_THRESHOLD_UNDER), false);
	}

	private void initRows() {
		Collection<MultiColumnNodeData> rows = super.getListModel();

		if (spQueueCombined != null) {
			DefaultMultiColumnNodeData mcnd = new DefaultMultiColumnNodeData(spQueueCombined.getID());
			
			CampaignSPQueueUtil.addLeastTimeIntervalCell(spQueueCombined, m_localizer, mcnd);
			CampaignSPQueueUtil.addThresholdOverCell(spQueueCombined, m_formInputPC, m_localizer, mcnd);
			CampaignSPQueueUtil.addThresholdUnderCell(spQueueCombined, m_formInputPC, m_localizer, mcnd);
			
			rows.add(mcnd);
		}
	}
	
	public boolean extractParameters(HttpServletRequest request) {
		boolean result = super.extractParameters(request);

		if (spQueueCombined != null) {
			//read the Least Time Interval (if any) for this spqueue
			String paramName = CampaignSPQueueUtil.SELECTION_SPLTI_FN + "_" + spQueueCombined.getID().toString();
			String val =  request.getParameter(paramName);
			if (val != null) {
				int iVal = Integer.parseInt(val);
				spQueueCombined.setLeastTimeInterval(iVal);
			}
			
			//read the Threshold Over (if any) for this spqueue
			paramName = CampaignSPQueueUtil.SELECTION_SPTHRESHOLDOVER_FN + "_" + spQueueCombined.getID().toString();
			val =  request.getParameter(paramName);
			if (val != null) {
				int iVal = Integer.parseInt(val);
				spQueueCombined.setFTEThresholdOver(iVal);
			}
			
			//read the Threshold Over Type (if any) for this spqueue
			paramName = CampaignSPQueueUtil.SELECTION_SPTHRESHOLDOVERTYPE_FN + "_" + spQueueCombined.getID().toString();
			val =  request.getParameter(paramName);
			if (val != null) {
				int iVal = Integer.parseInt(val);
				spQueueCombined.setFTEThresholdOverPercentage(iVal==1);
			}
			
			//read the Threshold Under (if any) for this spqueue
			paramName = CampaignSPQueueUtil.SELECTION_SPTHRESHOLDUNDER_FN + "_" + spQueueCombined.getID().toString();
			val =  request.getParameter(paramName);
			if (val != null) {
				int iVal = Integer.parseInt(val);
				spQueueCombined.setFTEThresholdUnder(iVal);
			}
			
			//read the Threshold Under Type (if any) for this spqueue
			paramName = CampaignSPQueueUtil.SELECTION_SPTHRESHOLDUNDERTYPE_FN + "_" + spQueueCombined.getID().toString();
			val =  request.getParameter(paramName);
			if (val != null) {
				int iVal = Integer.parseInt(val);
				spQueueCombined.setFTEThresholdUnderPercentage(iVal==1);
			}
		}
		
		return result;
	}
}
