	protected void processSaveCampaign(RequestContext context) {
		CampaignSettingsPageModel model = new CampaignSettingsPageModel(context);
		model.extractParameters(context.getRequest());
		String strIsCreate = context.getParameter(CampaignSettingsPageModel.CAMP_IS_CREATE);
		boolean isCreate = strIsCreate.equals("true");
		if (isCreate) {
			try {
				Campaign campaign = model.getCampaign();
				// if this is a subcampaign, its weekstart has been set to its parent's weekstart.
				if (campaign.getParentID() == null) {
					short weekStartDay = model.getWeekStartDay();
					int dayboundary = model.getDayBoundaryOffSet();
					int weekStart = (weekStartDay > 0 ? ((weekStartDay - 1) * 1440) : 0) + dayboundary;
					campaign.setWeekStart(weekStart);
				}

				CampaignManager camManager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
				ID newID = camManager.createCampaign(campaign);

				ScopeID scopeID = new ScopeID(newID, Scope.CAMPAIGN_SCOPE);
				UserManager userManager = CoreManagerFactory.getUserManager(context.isInWhatIfMode());
				userManager.addUserScope(context.getUser().getID(), scopeID);
				userManager.updateUser(context.getUser());
				UserCacheManager.getInstance().refreshCachedUser(context);

				String newIDStr = CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX + newID.toString();
				model.setSelectedID(newIDStr);
				model.setIsAddMode(false);
				CacheUtil.setCachedValue(context, CacheUtil.CACHE_MODE_PROFILE, "SELECTED_CAMPAIGN_SP_ID", newIDStr);
				SelectionCacheUtil.setCampaignID(context, new ID(newID));
				SelectionCacheUtil.setSchedulingPeriodID(context, null);
				model.addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_SAVE_CAMPAIGN_SUCCESS, FsWebBundleKey.BUNDLE_NAME);
			} catch (BbmCreateDuplicateKeyException duplicateKeyException) {
				model.addPageMessage(Message.ERROR_TYPE, FsWebBundleKey.FS_CREATE_CAMPAIGN_DUP_NAME,
						FsWebBundleKey.BUNDLE_NAME);
			} catch (BbmCreateException createException) {
				model.addPageMessage(Message.ERROR_TYPE, createException.getLocalizedMessage());
			} catch (Exception e) {
				model.addPageMessage(Message.ERROR_TYPE, e.getLocalizedMessage());
			}
		} else {
			try {
				String campID = model.getSelectedID();
				ID campaignID =
						StringUtil.parseIDStringToCollection(
								CampaignSchedulingPeriodSelectionUtil.extractIDs(campID,
										CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX)).iterator().next();

				short weekStartDay = model.getWeekStartDay();
				int dayboundary = model.getDayBoundaryOffSet();
				int weekStart = (weekStartDay > 0 ? ((weekStartDay - 1) * 1440) : 0) + dayboundary;
				int periodicityID = model.getPeriodicityID();
				short monthStartDay = model.getMonthStartDay();

				CampaignManager camManager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
				Campaign campToSave = camManager.getCampaignByID(campaignID);
				ArrayList<Campaign> campaigns = new ArrayList<Campaign>();
				Campaign campaign = model.getCampaign();
				campToSave.setName(campaign.getName());
				campToSave.setDescription(campaign.getDescription());
				campToSave.setTimeZone(campaign.getTimeZone());
				campToSave.setTimeZoneID(campaign.getTimeZoneID());
				campToSave.setWeekStart(weekStart);
				campToSave.setPeriodicityID(periodicityID);
				campToSave.setMonthStartDay(monthStartDay);

				campaigns.add(campToSave);
				camManager.updateCampaigns(campaigns);
				model.addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_SAVE_CAMPAIGN_SUCCESS, FsWebBundleKey.BUNDLE_NAME);
			} catch (Exception e) {
				model.handleUnableToLoadDataError(m_log, e);
			}
		}
		// Need to refresh Selection Pane
		model.setRefreshSelection(true);
		prepareForNextPage(context, model, CAMPAIGN_FORM_PAGE);
	}

	@Override
	protected void processSave(RequestContext context) {
		CampaignSPSettingsPageModel model = new CampaignSPSettingsPageModel(context);
		model.extractParameters(context.getRequest());
		try {
			String strSPID = model.getSelectedID();
			ID spID =
					StringUtil
							.parseIDStringToCollection(
									CampaignSchedulingPeriodSelectionUtil.extractIDs(strSPID,
											CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX)).iterator().next();

			CampaignManager camManager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
			SchedulingPeriod spToSave = camManager.getSchedulingPeriodByID(spID);

			// ESR4314016 - make sure orgs that were out of user scope are not removed from the SP.
			Collection<ID> previousLinkedOrgIdsWithoutPriviledge = camManager.getLinkedOrgsforSP(spID);
			Collection<ID> previousLinkedOrgIdsWithPriviledge =
					OrganizationModelHandler.getAllOrgIDsInUserScopePlusAncestors(context, context.getUser(), false);
			previousLinkedOrgIdsWithoutPriviledge.removeAll(previousLinkedOrgIdsWithPriviledge);

			ID campaignID = spToSave.getCampaignID();
			Campaign selectedCampaign = camManager.getCampaignByID(campaignID);

			TimeZone campaignTimeZone = selectedCampaign.getTimeZone();
			TimeZone viewTimeZone = getViewingTimeZone(context, selectedCampaign);

			long dateLong = spToSave.getStartTime().getTime();

			int timezoneOffSet = (viewTimeZone.getOffset(dateLong) - campaignTimeZone.getOffset(dateLong)) / (60 * 1000);
			int weekStartDay = SchedulingPeriodUtil.getWeekStartDay(selectedCampaign.getWeekStart());
			int weekStart = selectedCampaign.getWeekStart() + timezoneOffSet;
			int dayBoundary = (weekStart < 0 ? weekStart + 1440 : weekStart) % 1440;

			if (model.validateHOO(timezoneOffSet, dayBoundary)) {
				model.setIsAddMode(false);
				String timeRange = context.getParameter("timePeriodSelectorPC__dropdown__input");
				StringTokenizer st = new StringTokenizer(timeRange, "-");
				String start = "";
				String end = "";
				while (st.hasMoreTokens()) {
					start = st.nextToken();
					end = st.nextToken();
				}
				long lstart = 0;
				long lend = 0;
				lstart = Long.parseLong(start);
				lend = Long.parseLong(end);
				Date startDate = new Date(lstart);
				LocalDate hooStartTimeLocal = new LocalDate(startDate, TimeZone.getTimeZone("GMT"));
				Date endDate = new Date(lend);
				long diffDays = (endDate.getTime() - startDate.getTime()) / (24 * 60 * 60 * 1000);
				boolean isMultiWeek = diffDays > 8;

				CampaignHOO newHoo = model.getHoo();
				SchedulingPeriod sp = model.getSP();
				int weekIndex = 0;
				if (!isMultiWeek) {
					weekIndex = SchedulingPeriodUtil.getWeekIndex(selectedCampaign, spToSave, hooStartTimeLocal);
				}

				spToSave.setDescription(sp.getDescription());
				spToSave.setSkillBased(sp.getSkillBased());
				spToSave.setUsingAllEmployees(sp.isUsingAllEmployees());

				Collection<ID> allSPOrgs = model.getOrgIDs();
				allSPOrgs.addAll(previousLinkedOrgIdsWithoutPriviledge);
				spToSave.setOrgs(allSPOrgs);

				ArrayList<SchedulingPeriod> coll = new ArrayList<SchedulingPeriod>();
				coll.add(spToSave);
				camManager.updateSchedulingPeriods(coll);
				List<CampaignHOO> oldHoos = camManager.getCampaignHOOAssignmentsBySP(spToSave.getID());

				Map<CampaignHOO, Pair<Integer, Integer>> partialWeekHOOExclusions =
						SchedulingPeriodUtil.getPartialWeekHOOExclusions(selectedCampaign, spToSave, oldHoos,
								weekStartDay);

				for (int hooIndex = 0; hooIndex < oldHoos.size(); hooIndex++) {
					if (!isMultiWeek && hooIndex != weekIndex) {
						continue;
					}
					CampaignHOO oldhoo = oldHoos.get(hooIndex);

					for (short i = 1; i <= 7; i++) {
						short dayOffset = convertDayOfWeekToDayOffset(i, weekStartDay);
						if (newHoo.isActive(i)
								&& SchedulingPeriodUtil.isValidHOOForDay(partialWeekHOOExclusions, oldhoo, dayOffset)) {
							// if Active and value is -1, it's multi-valued, and keep the old value
							if (newHoo.getDayOpen(i) != -1) {
								short adjustedStartTime =
										CampaignSPSettingsPageModel.getAdjustedStartTime(newHoo.getDayOpen(i),
												timezoneOffSet);
								oldhoo.setDayOpen(i, adjustedStartTime);
							}

							if (newHoo.getDayClose(i) != -1) {
								short adjustedEndTime =
										CampaignSPSettingsPageModel.getAdjustedEndTime(newHoo.getDayClose(i),
												timezoneOffSet, dayBoundary);
								oldhoo.setDayClose(i, adjustedEndTime);
							}
						} else {
							oldhoo.setActive(i, false);
						}
					}
					camManager.updateCampaignHOOAssignment(oldhoo);
				}
				model.addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_SAVE_SUCCESS, FsWebBundleKey.BUNDLE_NAME);
			}
		} catch (BbmUpdateException createException) {
			model.addPageMessage(Message.ERROR_TYPE, createException.getLocalizedMessage());
		} catch (Exception e) {
			model.handleUnableToLoadDataError(m_log, e);
		}

		prepareForNextPage(context, model, FORM_PAGE);
	}
	
	/**
	 * Inspects the override setting to determine which time zone should be used.
	 * This is a convenience method that just uses the <code>isTimeZoneOverridden</code>
	 * method to switch between the RequestContext's viewing time zone and the default view time zone.
	 * <p>
	 * If the time zone is overridden, return the context's view time zone. If it is not
	 * overridden, return the default view time zone. The default view time zone comes from
	 * getDefaultViewTimeZone() for child campaigns, and from getTimeZone() for all other campaigns.
	 * @param context
	 * @param selectedCampaign
	 * @return
	 */
	protected TimeZone getViewingTimeZone(RequestContext context, Campaign selectedCampaign) {
		if (!TimeZoneOverrideUtil.isTimeZoneOverridden(context)) {
			TimeZone viewTimeZone = selectedCampaign.getDefaultViewTimeZone();
			if (viewTimeZone == null) {
				viewTimeZone = selectedCampaign.getTimeZone();
			}
			return viewTimeZone;
		} else {
			return context.getViewingTimeZone();
		}
	}
	
	/**
	 * Given a day of the week (1=Sunday..7=Saturday), its day offset for the SP week.
	 * For example, if the week start day is Monday, then Monday has an offset of 1,
	 * Tuesday has an offset of 2, and Sunday has an offset of 7.
	 * @param dayOfWeek - The day of the week (1..7) whose day offset within the SP week you need.
	 * @return The day offset (1..7).
	 */
	short convertDayOfWeekToDayOffset(short dayOfWeek, int weekStartDay) {
		int dayOffset = dayOfWeek - weekStartDay + 1;
		if (dayOfWeek < weekStartDay) {
			dayOffset += 7;
		}
		return (short)dayOffset; 
	}

	private void processAddSP(RequestContext context) {
		CampaignCreateSPPM model = new CampaignCreateSPPM(context);
		model.extractParameters(context.getRequest());
		boolean bCopyEmployeeData = model.getIsCopyEmployeeData();
		boolean bCopyProjectData = model.getIsCopyProjectData();
		boolean isMonthly = model.isMonthlyCampaign();

		CampaignSPSettingsPageModel spModel = new CampaignSPSettingsPageModel(context);
		SchedulingPeriod sp = model.getSchedulingPeriod();
		int selection = model.getSelection();
		try {
			CampaignManager camManager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
			DBConfigManager configManager = BbmManagerFactory.getDBConfigManager(context.isInWhatIfMode());
			String selectId = context.getParameter(WorkpaneFieldNames.SELECTED_ID_FN);

			ID campaignID = null;
			ID newID = null;
			if (selectId.startsWith(CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX)) {
				ID spID =
						StringUtil
								.parseIDStringToCollection(
										CampaignSchedulingPeriodSelectionUtil.extractIDs(selectId,
												CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX)).iterator().next();
				SchedulingPeriod curSP = camManager.getSchedulingPeriodByID(spID);
				campaignID = curSP.getCampaignID();
			} else if (selectId.startsWith(CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX)) {
				campaignID =
						StringUtil
								.parseIDStringToCollection(
										CampaignSchedulingPeriodSelectionUtil.extractIDs(selectId,
												CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX)).iterator().next();
			}
			sp.setCampaignID(campaignID);
			if (sp.getMaxPhantoms() == 0) {
				// Default value.
				sp.setMaxPhantoms(9999);
			}

			LocalDate start = sp.getStartTimeLocal();
			LocalDate end = sp.getEndTimeLocal();
			long numberofWeeks =
					(end.getTime().getTime() - start.getTime().getTime() + 24 * 60 * 60 * 1000) / (7 * 24 * 60 * 60 * 1000);
			if (isMonthly) {
				List<Pair<LocalDate, LocalDate>> localDates =
						SchedulingPeriodUtil.getWeeksForMonthlySP(start, model.getWeekStartDay());
				numberofWeeks = localDates.size();
			}

			if (selection == 0) {
				newID = doCreateAsEmpty(model, isMonthly, sp, camManager, campaignID, start, numberofWeeks);
			} else if (selection == 1) {
				newID = doCopyDataFromPreviousWeeks(context, model, bCopyEmployeeData, spModel, sp, campaignID,
						numberofWeeks);
			} else if (selection == 2) {
				newID = doCopyDataFromSelectPeriod(context, model, bCopyEmployeeData, spModel, sp, camManager, configManager);
			} else if (selection == 3) {
				CloneSPOptions copyFlags = new CloneSPOptions(bCopyEmployeeData, bCopyProjectData);
				newID = doCustomizeWeekSelection(context, model, copyFlags, spModel, sp, camManager, configManager,
								campaignID, numberofWeeks);
			}

			if (newID == null) {
				// an error must have occurred
				return;
			} else {
				String newIDStr = CampaignSchedulingPeriodSelectionUtil.buildCompositeIDString(newID, campaignID);
				spModel.setSelectedID(newIDStr);
				spModel.setIsAddMode(false);

				// TODO: we probably don't need to set the auto cached ID here since it should be taken care
				// of automatically via the selection pane, but I didn't want to risk removing it since it is late
				// in the dev cycle
				CampaignSchedulingPeriodSelectionUtil.setSelectedAutoCachedCampaignSPID(context, campaignID, newID);

				// Cache the Campaign and SP IDs so that other screens (profiles, employee assignment of work rules / skills)
				// will have this selection stick.
				SelectionCacheUtil.setCampaignID(context, campaignID);
				SelectionCacheUtil.setSchedulingPeriodID(context, newID);
			}
		} catch (Exception e) {
			spModel.handleUnableToLoadDataError(m_log, e);
		}
		spModel.setRefreshSelection(true);
		prepareForNextPage(context, spModel, FORM_PAGE);
	}

	/**
	 * @param context
	 * @param model
	 * @param bCopyEmployeeData
	 * @param spModel
	 * @param sp - Sp object representing the new SP to be cloned.  Some fields are not yet set, such as ID.
	 * @param numberofWeeks - The size of the new SP in weeks
	 * @throws BbmObjectNotFoundException
	 * @throws RemoteException
	 * @throws BbmFinderException
	 * @throws Exception
	 */
	private ID doCopyDataFromPreviousWeeks(RequestContext context, CampaignCreateSPPM model, boolean bCopyEmployeeData,
			CampaignSPSettingsPageModel spModel, SchedulingPeriod sp, ID campaignID, long numberofWeeks) throws Exception {

		CampaignManager camManager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
		DBConfigManager configManager = BbmManagerFactory.getDBConfigManager(context.isInWhatIfMode());

		ID newID = null;

		LocalDate spStartLocal = new LocalDate(sp.getStartTimeLocal());
		LocalDate spEndLocal = new LocalDate(sp.getEndTimeLocal());

		spStartLocal.add(Calendar.DAY_OF_YEAR, -(int) numberofWeeks * 7);
		spEndLocal.add(Calendar.DAY_OF_YEAR, -(int) numberofWeeks * 7);

		Campaign campaign = camManager.getCampaignByID(campaignID);

		Date spStart = TimePeriodUtil.convertToGMTFromLocalDateTimeInTimeZone(spStartLocal, campaign.getTimeZone());
		Date spEnd = TimePeriodUtil.convertToGMTFromLocalDateTimeInTimeZone(spEndLocal, campaign.getTimeZone());

		Collection<CampaignHOO> hoos = camManager.getCampaignHOOAssignments(campaignID, spStart, spEnd, false);

		try {
			newID = cloneSP(camManager, configManager, sp, hoos, new CloneSPOptions(bCopyEmployeeData),
					false, context.getLocaleContext(), spModel);
		} catch (BbmCloningConflictException e) {
			Collection<BbmCloningConflict> conflicts = e.getConflicts();
			String str = "";
			BbmCloningConflict conflicttest = conflicts.iterator().next();
			if (conflicttest.getConflictType() != BbmCloningConflict.CONFLICT_SHIFT_ASSIGNMENTS_EXIST) {
				for (BbmCloningConflict conflict : conflicts) {
					if (conflict.getConflictType() != BbmCloningConflict.CONFLICT_SHIFT_ASSIGNMENTS_EXIST) {
						// GMT clients may have custom strings in BPCONFIG. We have to verticalize them here, now.
						String stringMessage = getConflictMessage(conflict, context);
						if (stringMessage != null) {
							str += stringMessage + "\n";
						}
					}
				}
				model.addPageMessage(Message.ERROR_TYPE, str);
				prepareForNextPage(context, model, CAMPAIGN_CREATE_SP_PAGE);
				return null;
			} else if (conflicttest.getConflictType() == BbmCloningConflict.CONFLICT_SHIFT_ASSIGNMENTS_EXIST) {
				newID =
						cloneSP(camManager, configManager, sp, hoos, new CloneSPOptions(bCopyEmployeeData),
								true, context.getLocaleContext(), spModel);

				model.addPageMessage(Message.ERROR_TYPE, str);
				prepareForNextPage(context, model, CAMPAIGN_CREATE_SP_PAGE);
			}
		} catch (Exception e) {
			model.handleUnableToLoadDataError(m_log, e);
			prepareForNextPage(context, model, CAMPAIGN_CREATE_SP_PAGE);
			return null;
		}
		return newID;
	}

	/**
	 * @param context
	 * @param model
	 * @param bCopyEmployeeData
	 * @param spModel
	 * @param sp
	 * @param camManager
	 * @param configManager
	 * @param newID
	 * @return
	 * @throws BbmFinderException
	 * @throws RemoteException
	 * @throws Exception
	 */
	private ID doCopyDataFromSelectPeriod(RequestContext context, CampaignCreateSPPM model, boolean bCopyEmployeeData,
			CampaignSPSettingsPageModel spModel, SchedulingPeriod sp, CampaignManager camManager,
			DBConfigManager configManager) throws Exception {
		ID newID = null;

		String strId = context.getParameter(CampaignCreateSPPM.CAMP_SP_CREATE_PREV_SECECT);
		ID sourceSpId = ID.fromInt(Integer.parseInt(strId));
		Collection<CampaignHOO> hoos = camManager.getCampaignHOOAssignmentsBySP(sourceSpId);

		try {
			newID = cloneSP(camManager, configManager, sp, hoos, new CloneSPOptions(bCopyEmployeeData), 
					false, context.getLocaleContext(), spModel);
		} catch (BbmCloningConflictException e) {
			Collection<BbmCloningConflict> conflicts = e.getConflicts();
			String str = "";
			BbmCloningConflict conflicttest = conflicts.iterator().next();
			if (conflicttest.getConflictType() != BbmCloningConflict.CONFLICT_SHIFT_ASSIGNMENTS_EXIST) {
				for (BbmCloningConflict conflict : conflicts) {
					if (conflict.getConflictType() != BbmCloningConflict.CONFLICT_SHIFT_ASSIGNMENTS_EXIST) {
						// GMT clients may have custom strings in BPCONFIG. We have to verticalize them here, now.
						String stringMessage = getConflictMessage(conflict, context);
						if (stringMessage != null) {
							str += stringMessage + "\n";
						}
					}
				}
				model.addPageMessage(Message.ERROR_TYPE, str);
				prepareForNextPage(context, model, CAMPAIGN_CREATE_SP_PAGE);

				return null;
			} else if (conflicttest.getConflictType() == BbmCloningConflict.CONFLICT_SHIFT_ASSIGNMENTS_EXIST) {
				newID = cloneSP(camManager, configManager, sp, hoos, new CloneSPOptions(bCopyEmployeeData),
						true, context.getLocaleContext(), spModel);
			}

		} catch (Exception e) {
			model.handleUnableToLoadDataError(m_log, e);
			prepareForNextPage(context, model, CAMPAIGN_CREATE_SP_PAGE);
			return null;
		}
		return newID;
	}

	 * @param numberofWeeks
	 * @return
	 * @throws BbmFinderException

	protected ID doCustomizeWeekSelection(RequestContext context, CampaignCreateSPPM model, CloneSPOptions copyFlags,
			CampaignSPSettingsPageModel spModel, SchedulingPeriod sp, CampaignManager camManager,
			DBConfigManager configManager, ID campaignID, long numberofWeeks) throws BbmFinderException, RemoteException,
			BbmException, Exception {
		ID newID = null;
		String strIDs = context.getParameter(CampaignCreateSPPM.SELECTED_HOO_ID);
		ArrayList<CampaignHOO> hoos = new ArrayList<CampaignHOO>();
		if (!strIDs.equals("")) {
			// Customize Week Dialog was launched. strIDs contains the selected HOO Id's from the form variable.
			ArrayList<ID> ids = (ArrayList<ID>) StringUtil.parseIDStringToCollection(strIDs);
			for (int i = 0; i < ids.size(); i++) {
				CampaignHOO hoo = camManager.getCampaignHOOAssignmentByID(ids.get(i));
				hoos.add(hoo);
			}
		} else {
			// Customize Week Dialog was not launched. Get the default week (the last one).
			ArrayList<CampaignHOO> totalHoos = getAllCampaignHOOAssignments(context, sp, campaignID);
			CampaignHOO theLastHOO = totalHoos.get(totalHoos.size() - 1);
			for (int i = 0; i < numberofWeeks; i++) {
				hoos.add(theLastHOO);
			}
		}

		try {
			newID = cloneSP(camManager, configManager, sp, hoos, copyFlags, false, context.getLocaleContext(), spModel);
		} catch (BbmCloningConflictException e) {
			Collection<BbmCloningConflict> conflicts = e.getConflicts();
			String str = "";
			BbmCloningConflict conflicttest = conflicts.iterator().next();
			if (conflicttest.getConflictType() != BbmCloningConflict.CONFLICT_SHIFT_ASSIGNMENTS_EXIST) {
				for (BbmCloningConflict conflict : conflicts) {
					if (conflict.getConflictType() != BbmCloningConflict.CONFLICT_SHIFT_ASSIGNMENTS_EXIST) {
						// GMT clients may have custom strings in BPCONFIG. We have to verticalize them here, now.
						String stringMessage = getConflictMessage(conflict, context);
						if (stringMessage != null) {
							str += stringMessage + "\n";
						}
					}
				}
				model.addPageMessage(Message.ERROR_TYPE, str);
				prepareForNextPage(context, model, CAMPAIGN_CREATE_SP_PAGE);
				return null;
			} else if (conflicttest.getConflictType() == BbmCloningConflict.CONFLICT_SHIFT_ASSIGNMENTS_EXIST) {
				newID = cloneSP(camManager, configManager, sp, hoos, copyFlags, true, context.getLocaleContext(), spModel);
			}
		} catch (Exception e) {
			model.handleUnableToLoadDataError(m_log, e);
			prepareForNextPage(context, model, CAMPAIGN_CREATE_SP_PAGE);
			return null;
		}
		return newID;
	}

	protected ArrayList<CampaignHOO> getAllCampaignHOOAssignments(RequestContext context, SchedulingPeriod sp, ID campaignID)

		Date startDate = cal.getTime();
		cal.add(java.util.Calendar.YEAR, 20);
		Date endDate = cal.getTime();
		// end of workaround

		ArrayList<CampaignHOO> totalHoos =
				new ArrayList<CampaignHOO>(CampaignModelHandler.getCampaignHOOAssignments(context, campaignID, startDate,
						endDate));
		return totalHoos;
	}

	/**
	 * Create an SP with all CampaignHOO entries for all week(s), with all of the days in all of the week(s) set to inactive
	 * HOO.
	 * @param model
	 * @param isMonthly
	 * @param sp
	 * @param camManager
	 * @param campaignID
	 * @param start
	 * @param numberofWeeks
	 * @return
	 * @throws BbmCreateException
	 * @throws RemoteException
	 */
	protected ID doCreateAsEmpty(CampaignCreateSPPM model, boolean isMonthly, SchedulingPeriod sp,
			CampaignManager camManager, ID campaignID, LocalDate start, long numberofWeeks) throws BbmCreateException,
			RemoteException {
		ID newID = null;
		;
		newID = camManager.createSchedulingPeriod(sp);
		if (!isMonthly) {
			createCampaignHOOAssignmentsForWeeklySP(camManager, campaignID, newID, start, numberofWeeks);
		} else {
			createCampaignHOOAssignmentsForMonthlySP(camManager, campaignID, newID, start, model);
		}
		return newID;
	}

	/**
	 * For a Weekly SP (single or multi week), create and save all CampaignHOO entries for all week(s), with all of the days
	 * in all of the week(s) set to inactive HOO.
	 * @param camManager - instance of the EJB
	 * @param campaignID - the campaign ID
	 * @param spID - the SPID
	 * @param spStartDate - the start of the SP
	 * @param numberofWeeks - the number of weeks in the SP
	 * @throws BbmCreateException
	 * @throws RemoteException
	 */
	protected void createCampaignHOOAssignmentsForWeeklySP(CampaignManager camManager, ID campaignID, ID spID,
			LocalDate spStartDate, long numberofWeeks) throws BbmCreateException, RemoteException {
		LocalDate start = new LocalDate(spStartDate);
		for (int i = 0; i < numberofWeeks; i++) {
			LocalDate spEnd = null;
			if (i != 0) {
				start.add(Calendar.DAY_OF_YEAR, 7);
			}
			spEnd = new LocalDate(start);
			spEnd.add(Calendar.DAY_OF_YEAR, 7);
			createCampaignHOOAssignment(camManager, campaignID, spID, start, spEnd);
		}
	}

	/**
	 * For a Monthly SP, create and save all CampaignHOO entries for all weeks int he month, with all of the days in all of
	 * the weeks set to inactive HOO.
	 * @param camManager - instance of the EJB
	 * @param campaignID - the campaign ID
	 * @param spID - the SPID
	 * @param start - the start of the SP
	 * @param numberofWeeks - the number of weeks in the SP
	 * @throws BbmCreateException
	 * @throws RemoteException
	 */
	protected void createCampaignHOOAssignmentsForMonthlySP(CampaignManager camManager, ID campaignID, ID spID,
			LocalDate spStartDate, CampaignCreateSPPM model) throws BbmCreateException, RemoteException {

		List<Pair<LocalDate, LocalDate>> localDates =
				SchedulingPeriodUtil.getWeeksForMonthlySP(spStartDate, model.getWeekStartDay());

		for (Pair<LocalDate, LocalDate> datePair : localDates) {
			LocalDate start = datePair.getFirst();
			LocalDate end = datePair.getSecond();
			// end must include the next day
			end.add(Calendar.DATE, 1);
			createCampaignHOOAssignment(camManager, campaignID, spID, start, end);
		}
	}

	/**
	 * Create and save a CampaignHOO entry with all of the days in the week set to inactive HOO.
	 * @param camManager - instance of the EJB
	 * @param campaignID - the campaign ID
	 * @param spID - the SPID
	 * @param start - the CampaignHOO's StartTimeLocal
	 * @param end - the CampaignHOO's EndTimeLocal
	 * @throws BbmCreateException
	 * @throws RemoteException
	 */
	protected void createCampaignHOOAssignment(CampaignManager camManager, ID campaignID, ID spID, LocalDate start,
			LocalDate end) throws BbmCreateException, RemoteException {
		CampaignHOO cpnHOO = new CampaignHOO();
		cpnHOO.setCampaignID(campaignID);
		cpnHOO.setSPID(spID);
		cpnHOO.setActive((short) 1, false);
		cpnHOO.setActive((short) 2, false);
		cpnHOO.setActive((short) 3, false);
		cpnHOO.setActive((short) 4, false);
		cpnHOO.setActive((short) 5, false);
		cpnHOO.setActive((short) 6, false);
		cpnHOO.setActive((short) 7, false);
		cpnHOO.setStartTimeLocal(start);
		cpnHOO.setEndTimeLocal(end);
		camManager.createCampaignHOOAssignment(cpnHOO);
	}

	// / Test for verticalized strings
	// / If not found, return the standard message from the conflict
	// / This is needed for GMT clients who may have custom strings in BPCONFIG
	protected String getConflictMessage(BbmCloningConflict conflict, RequestContext context) {
		String stringVerticalized = null;

		if (conflict.getConflictType() == BbmCloningConflict.CONFLICT_QUEUES_IN_OTHER_CAMPAIGN) {
			Object params = StringUtils.join(conflict.getNames().toArray(new String[conflict.getNames().size()]), ", ");
			ResourceBundle bundle = ResourceBundle.getBundle(BbmEjbBundleKey.BUNDLE);
			stringVerticalized =
					context.getSystemManagerFactory()
							.getVerticalizationManager()
							.getFormattedVerticalizedString(context.getLocalizer(), bundle,
									BbmEjbBundleKey.CLONE_CONFLICT_QUEUES_IN_OTHER_CAMPAIGN, params);
		} else if (conflict.getConflictType() == BbmCloningConflict.CONFLICT_MULTIWEEK_QUEUES_DONT_MATCH) {
			ResourceBundle bundle = ResourceBundle.getBundle(BbmEjbBundleKey.BUNDLE);
			stringVerticalized =
					context.getSystemManagerFactory()
							.getVerticalizationManager()
							.getVerticalizedString(context.getLocalizer(), bundle,
									BbmEjbBundleKey.CLONE_CONFLICT_QUEUES_DONT_MATCH);
		} else if (conflict.getConflictType() == BbmCloningConflict.CONFLICT_MULTIWEEK_SKILLBASED_DOESNT_MATCH) {
			ResourceBundle bundle = ResourceBundle.getBundle(BbmEjbBundleKey.BUNDLE);
			stringVerticalized =
					context.getSystemManagerFactory()
							.getVerticalizationManager()
							.getVerticalizedString(context.getLocalizer(), bundle,
									BbmEjbBundleKey.CLONE_CONFLICT_SKILLBASED_DOESNT_MATCH);
		}

		if (stringVerticalized != null) {
			return stringVerticalized;
		} else {
			return conflict.getMessage();
		}
	}

	public void processRequest(RequestContext context) throws Exception {
		// Process requested action
		if (isAction(context, PageAction.FORCE_REFRESH)) {
			precessRefreshCreateSP(context);
		} else if (isAction(context, PageAction.NEW_ID)) {
			String selectId = context.getParameter(WorkpaneFieldNames.SELECTED_ID_FN);
			if (selectId.startsWith(CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX)) {
				processLoadSPID(context);
			} else {
				processLoadCampaignID(context);
			}
		} else if (isAction(context, PageAction.SP_REFRESH_HOO_WITH_ORG)) {
			processRefreshHOO(context);
		} else if (isAction(context, PageAction.SP_CREATE_SUB_CAMPAIGN)) {
			processAddSubCampaign(context);
		} else if (isAction(context, PageAction.NEW)) {
			processAddCampaign(context);
		} else if (isAction(context, PageAction.SP_CREATE)) {
			processAddSP(context);
		} else if (isAction(context, PageAction.ADD)) {
			processAdd(context);
		} else if (isAction(context, PageAction.CAMPAIGN_SAVE)) {
			processSaveCampaign(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		} else if (isAction(context, TIME_ZONE_OVERRIDE_TOGGLE_ACTION)) {
			TimeZoneOverrideUtil.toggleTimeZoneOverride(context);
			prepareForNextPage(context, null, FORM_PAGE);
		} else if (isAction(context, PageAction.DELETE)) {
			processDelete(context);
		} else if (isAction(context, PageAction.CANCEL)) {
			if (isWorkpaneMode(context)) {
				SettingPM model = newSettingPM(context);
				model.extractParameters(context.getRequest(), WorkpaneFieldNames.SELECTED_ID_FN);
				model.setIsAddMode(false);

				String selectId = context.getParameter(WorkpaneFieldNames.SELECTED_ID_FN);
				if (selectId != null && selectId.startsWith(CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX)) {
					prepareForNextPage(context, model, FORM_PAGE);
				} else {
					prepareForNextPage(context, model, CAMPAIGN_FORM_PAGE);
				}
			} else {
				prepareForNextPage(context, null, SUMMARY_PAGE);
			}
		} else if (isAction(context, PageAction.EDIT)) {
			prepareForNextPage(context, null, FORM_PAGE);
		} else {
			String selectId = context.getUserProperty("SELECTED_CAMPAIGN_SP_ID");
			if (selectId != null && selectId.startsWith(CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX)) {
				prepareForNextPage(context, null, FORM_PAGE);
			} else if (selectId != null && selectId.startsWith(CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX)) {
				prepareForNextPage(context, null, CAMPAIGN_FORM_PAGE);
			} else {
				if (context.getPathInfo().equals(CAMPAIGN_FORM_PAGE)) {
					prepareForNextPage(context, null, CAMPAIGN_FORM_PAGE);
				} else {
					prepareForNextPage(context, null, DEFAULT_PAGE);
				}
			}
		}
	}

	/**
	 * The main starting point to handle cloning (copying) an existing SP's data into the new SP. Handles
	 * timeouts if the cloning takes too long. Relies on CampaignManager.cloneBaseDataForSchedulingPeriod() to do actual cloning.
	 * @param camManager - CampaignManager object which will do the actual cloning.
	 * @param configManager - Needed to get BPCONFIG setting for CLONE_SP_TRANSACTION_TIMEOUT
	 * @param sp - the new SP (the target of the cloning).
	 * @param hoos - Hours Of Operation data.
	 * @param copyFlags - A set of boolean values that indicate which features need to be copied/cloned.
	 * @param bIgnoreWarnings - whether to ignore warnings when cloning
	 * @param localeCtx - LocaleContext object.
	 * @param model - the page model which is only used for displaying warning messages to.
	 * @return - The new SP's ID.
	 * @throws Exception
	 */
	private ID cloneSP(CampaignManager camManager, DBConfigManager configManager, SchedulingPeriod sp,
			Collection<CampaignHOO> hoos, CloneSPOptions copyFlags, boolean bIgnoreWarnings, LocaleContext localeCtx,
			CampaignSPSettingsPageModel model) throws Exception {
		Context c = new InitialContext();
		UserTransaction ut = (UserTransaction) c.lookup("java:comp/UserTransaction");
		// Default timeout of 1200 seconds (20 minutes)
		int clonetimeout = 1200;
		try {
			int nUserTransactionTimeOut = configManager.getIntValue(ConfigKey.CLONE_SP_TRANSACTION_TIMEOUT);
			// not sure what is up limit, set 6000 here.
			if (nUserTransactionTimeOut >= 120 && nUserTransactionTimeOut <= 6000) {
				clonetimeout = nUserTransactionTimeOut;
			}
		} catch (Exception e) {
			m_log.debug("Error when retrieving transaction timeout for Clone SP operation", e);
		}
		ut.setTransactionTimeout(clonetimeout);
		ut.begin();
		ID newID;
		try {
			newID = camManager.cloneSchedulingPeriod(sp, hoos, copyFlags, bIgnoreWarnings, localeCtx);
			ut.commit();
		} catch (Exception ex) {
			ut.rollback();
			throw ex;
		}
		try {
			camManager.cloneLSGTemplateForNewSP(newID, hoos);
		} catch (Exception e) {
			model.addPageMessage(Message.WARNING_TYPE,
					model.getLocalizer().i18n(model.getResourceBundle(), FsWebBundleKey.LSG_CLONE_FAILED));
		}
		return newID;
	}
