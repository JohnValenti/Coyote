package com.verint.web.fs.campaign;

import java.util.ArrayList;
import java.util.List;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.form.FormInputPC;
import com.witness.web.uif.util.HtmlUtil;
import com.witness.web.uif.util.js.JSUtil;

public class CampaignSPQueueUtil {

	public final static String SELECTION_SPLTI_FN = "selectionSPLeastTimeInterval";
	public final static String SELECTION_SPTHRESHOLDOVER_FN = "selectionSPThresholdOver";
	public final static String SELECTION_SPTHRESHOLDOVERTYPE_FN = "selectionSPThresholdOverType";
	public final static String SELECTION_SPTHRESHOLDUNDER_FN = "selectionSPThresholdUnder";
	public final static String SELECTION_SPTHRESHOLDUNDERTYPE_FN = "selectionSPThresholdUnderType";
	
	private CampaignSPQueueUtil() {};

	/**
	 * Add a dropdown containing the following Least Time Interval options: 15-Minutes, Hours, Days, Weeks.
	 * The default selection should be 15-Minutes for new queues. For existing queues, the value is read
	 * from the SPQUEUE table (if non-skilled SP, we only get the value for the Combined queue for that SP).
	 */
	public static void addLeastTimeIntervalCell(SPQueue spQueue, Localizer localizer, DefaultMultiColumnNodeData rowData) {
		int lti = spQueue.getLeastTimeInterval();		
		List<StringsPair> intervalOptions = new ArrayList<StringsPair>();
		intervalOptions.add(new StringsPair("0", localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_15MIN)));
		intervalOptions.add(new StringsPair("1", localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_HOURS)));
		intervalOptions.add(new StringsPair("2", localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_DAYS)));
		intervalOptions.add(new StringsPair("3", localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_WEEKS)));
		rowData.add(HtmlUtil.makeSelectInput(SELECTION_SPLTI_FN + "_" + spQueue.getID().toString(),
					new String[]{""+lti}, JSUtil.ON_CHANGE, intervalOptions, false, true));
	}
	
	/**
	 * Add a cell containing a Threshold Over Input field and a dropdown for "Percent", "Counts".
	 * The default values should be empty/Percent for new queues. For existing queues, the values are read
	 * from the SPQUEUE table (if non-skilled SP, we only get these values for the Combined queue for that SP).
	 */
	public static void addThresholdOverCell(SPQueue spQueue,
			FormInputPC formInputPC,
			Localizer localizer,
			DefaultMultiColumnNodeData rowData) {
		//create the HTML for the input field
		int thresholdOver = spQueue.getFTEThresholdOver();
		String inputHTML = getInput(formInputPC, SELECTION_SPTHRESHOLDOVER_FN + "_" + spQueue.getID().toString(), 
				""+thresholdOver, FormInputPC.TYPE_INTEGER,	FsWebBundleKey.SCHEDULINGPERIOD_THRESHOLD_OVER, true);
		
		//create the HTML for the drop down selector
		boolean isPercentage = spQueue.isFTEThresholdOverPercentage();
		int type = isPercentage ? 1 : 0;
		List<StringsPair> typeOptions = new ArrayList<StringsPair>();
		typeOptions.add(new StringsPair("0", localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_COUNT)));
		typeOptions.add(new StringsPair("1", localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_PERCENT)));
		String dropdownHTML = HtmlUtil.makeSelectInput(SELECTION_SPTHRESHOLDOVERTYPE_FN + "_" + spQueue.getID().toString(),
				new String[]{""+type}, JSUtil.ON_CHANGE, typeOptions, false, true);
		
		rowData.add(inputHTML + "&nbsp;" + dropdownHTML);
	}
	
	/**
	 * Add a cell containing a Threshold Under Input field and a dropdown for "Percent", "Counts".
	 * The default values should be empty/Percent for new queues. For existing queues, the values are read
	 * from the SPQUEUE table (if non-skilled SP, we only get these values for the Combined queue for that SP).
	 */
	public static void addThresholdUnderCell(SPQueue spQueue,
			FormInputPC formInputPC,
			Localizer localizer,
			DefaultMultiColumnNodeData rowData) {
		//create the HTML for the input field
		int thresholdUnder = spQueue.getFTEThresholdUnder();
		String inputHTML = getInput(formInputPC, SELECTION_SPTHRESHOLDUNDER_FN + "_" + spQueue.getID().toString(), 
				""+thresholdUnder, FormInputPC.TYPE_INTEGER,	FsWebBundleKey.SCHEDULINGPERIOD_THRESHOLD_UNDER, true);
		
		//create the HTML for the drop down selector
		boolean isPercentage = spQueue.isFTEThresholdUnderPercentage();
		int type = isPercentage ? 1 : 0;
		ArrayList<StringsPair> typeOptions = new ArrayList<StringsPair>();
		typeOptions.add(new StringsPair("0", localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_COUNT)));
		typeOptions.add(new StringsPair("1", localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.SCHEDULINGPERIOD_PERCENT)));
		String dropdownHTML = HtmlUtil.makeSelectInput(SELECTION_SPTHRESHOLDUNDERTYPE_FN + "_" + spQueue.getID().toString(),
				new String[]{""+type}, JSUtil.ON_CHANGE, typeOptions, false, true);
		
		rowData.add(inputHTML + "&nbsp;" + dropdownHTML);
	}
	
	private static String getInput(FormInputPC formInputPC, String fn, String value, int type, String rbKey, boolean isEditable) {
		formInputPC.reset();
		formInputPC.setInputFN(fn);
		formInputPC.setInputValue(value);
		formInputPC.setInputType(type);
		formInputPC.setIsReadOnly(!isEditable);
		formInputPC.setInputFieldDesc(rbKey, FsWebBundleKey.BUNDLE_NAME);
		formInputPC.initForDisplay();
		formInputPC.setLowerLimit(0);
		formInputPC.setStyle("vertical-align: middle; ");
		return formInputPC.getHtmlDisplay();
	}
}
