public class CampaignSPQueueslListPM extends WorkpaneListWithSummaryPM {
	private static final long serialVersionUID = 1L;

	private ResourceBundle m_bundle;
	private boolean isSkillBased = false;
	private List<SPQueue> spQueues = new ArrayList<SPQueue>();
	private SPQueue spQueueCombined; //the Combined queue (in a non-skilled SP only)
	private String selectedSchedulingPeriodID = null;
	private SchedulingPeriod selectedSchedulingPeriod = null;
	private JumpToPagePaginationPC paginationPC;
	private CombinedSPQueueListPC combinedQueueListPC;

	public CampaignSPQueueslListPM(RequestContext context) {
		super(context, PageModel.FRAMED_MODEL);

		m_bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);
		
		paginationPC = new JumpToPagePaginationPC(getToolbar(), getLocalizer());
		addChildComponent(paginationPC);
		
		setDefaultPageIDSize(50);

		removeChildComponent(m_list);
		m_list = new CampaignSPQueuesListPC(m_context, spQueues, selectedSchedulingPeriod);
		m_list.setIsCancelSelectEventPropagation(false);
		setListComponent(m_list);
		//addChildComponent(m_list);

		//This needs to be done here so that we can get the sort columns/order
		m_list.extractParameters(context.getRequest());

		setIsPaginatingSelectedIDs(false);
		this.setIsMultiRowSelectable(true);
	}

	@Override
	protected void loadData() throws Exception {
		super.loadData();

		if (getSelectedSchedulingPeriodID() != null && getSelectedSchedulingPeriodID().length() > 0) {
			selectedSchedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
					m_context, new ID(getSelectedSchedulingPeriodID()));
		}

		if (selectedSchedulingPeriod != null) {
			if (!m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWCAMPAIGN_ID,
					new ScopeID(selectedSchedulingPeriod.getCampaignID(), Scope.CAMPAIGN_SCOPE))) {
				addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_SP_SCOPE, FsWebBundleKey.BUNDLE_NAME);
				return;
			}

			isSkillBased = selectedSchedulingPeriod.getSkillBased();
			loadQueues();
		}
	}
	
	public void setSelectedID(String ids) {
		super.setSelectedID(ids);
		setSelectedSchedulingPeriodID(CampaignSchedulingPeriodSelectionUtil.extractIDs(
			getSelectedID(), CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX));
	}
	
	public String getSelectedSchedulingPeriodID() { return selectedSchedulingPeriodID; }	
	public void setSelectedSchedulingPeriodID(String selectedSchedulingPeriodIDs) { selectedSchedulingPeriodID = selectedSchedulingPeriodIDs; }

	private void loadQueues() throws Exception {
		CampaignManager campaignMgr = WfmManagerFactory.getCampaignManager(m_context.isInWhatIfMode());
		String pageAction = getPageAction();
		if (isPaginationNavPressed() || PageAction.SAVE.equals(pageAction)) {
			spQueues = new ArrayList<SPQueue>(campaignMgr.getSPQueuesByIDs(getPageIDs()));
		} else {
			PaginationPair<SPQueue> spQueueData = 
					campaignMgr.getPaginatedSPQueuesForSchedulingPeriod(selectedSchedulingPeriod.getID(),
							m_list.getHeader().isAscending(), getPageIDSize(), getPagePaginationIndex(),
							m_list.getSortComparators(), null);
			spQueues = new ArrayList<SPQueue>(spQueueData.getValues());
			initPaginationIDs(spQueueData.getIDs());
		}
		
		//The paginated IDs are already sorted, so sort the list of this page of spqueues to match this sorted list of IDs
		Collections.sort(spQueues, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));
		
		if (!isSkillBased) {
			spQueueCombined = campaignMgr.getCombinedSPQueue(selectedSchedulingPeriod.getID(), Media.MEDIA_ID_PHONE);
		}
	}
	
	@Override
	protected void initHeader() {
	}

	@Override
	protected void initSelectableItems() {
		if (selectedSchedulingPeriod == null) {
			return;
		}
		resetListPC();
		m_list.extractParameters(m_context.getRequest()); 
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		m_contentTitlePC.setPageTitle(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.CAMPAIGNS_QUEUE_TAB_TITLE)); 
	}

	public static CampaignSPQueueslListPM getInstance(RequestContext context) {
		return (CampaignSPQueueslListPM)PageModel.getInstance(context, CampaignSPQueueslListPM.class);
	}
	
	@Override
	protected void initToolbar() {
		if (selectedSchedulingPeriod == null)
			return;
		try {
			CampaignManager camManager = WfmManagerFactory.getCampaignManager(m_context.isInWhatIfMode());	
			ID campaignID = selectedSchedulingPeriod.getCampaignID();
			Campaign campaign = camManager.getCampaignByID(campaignID);
			
			//Enable pagination
			enableToolbarPagination(getPaginationControl());
			
			if(campaign.getIsDistributed())
				return;
			
			boolean createEnabled = m_context.getUser().isAuthorized(PrivilegeKeys.FS_CREATECAMPAIGN);
			boolean configEnabled = m_context.getUser().isAuthorized(PrivilegeKeys.FS_CONFIGURECAMPAIGN);
			boolean editEnabled = createEnabled || configEnabled;
			
			ToolbarTextButton addButton = m_toolbar.createPopupWindowButton(CampaignSPQueueListPopupPM.POPUP_ID,
					i18n(m_bundle, FsWebBundleKey.CAMPAIGN_QUEUE_ADD), editEnabled);
	
			if (selectedSchedulingPeriod != null) {
				addButton.setAttribute("ownerID", selectedSchedulingPeriod.getID().toString());
			}
			addButton.setAttribute("isByOrg", String.valueOf(true));
			addButton.setAttribute("isSelectionPopup", String.valueOf(true));
			
			String strSelectedIDs = "";
			if (spQueues != null && spQueues.size()>0) {
				ArrayList<ID> idList = new ArrayList<ID> ();
				for (SPQueue spQueue: spQueues){
					idList.add(spQueue.getQueueID());
				}		
	
				strSelectedIDs = StringUtil.createDelimitedString(
						idList, StringUtil.DELIM);
			}
			
			addButton.setAttribute("selectedIDs", strSelectedIDs);
			addButton.setAttribute("onClick",
					m_toolbar.getName() + ".onSelect('" + CampaignSPQueueListPopupPM.POPUP_ID + "', this);"); 
	
			m_toolbar.addToolbarElement(addButton); 
			
			addPopupArgs(CampaignSPQueueListPopupPM.POPUP_ID, CampaignSPQueueListPopupPM.FORM_ACTION, "",
					CampaignSPQueueListPopupPM.POPUP_ARGS, "600,500,ctr,ctr", true, 1); 

			m_toolbar.addDeleteTextButton(PageAction.DELETE, i18n(m_bundle, FsWebBundleKey.CAMPAIGN_QUEUE_REMOVE), false);
			
			String saveDisplay = i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_SAVE);
			String resetDisplay = i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_RESET);
	
			m_toolbar.addValidateSubmitTextButton(PageAction.SAVE, saveDisplay, editEnabled); 
			m_toolbar.addResetTextButton(PageAction.RESET, resetDisplay, editEnabled);
		}
		catch(Exception ex) {
			handleUnableToLoadDataError(log, ex);
		}
	}

	/**

		return super.extractParameters(request);
	}
	
	protected void initScreenContent() {
		if (selectedSchedulingPeriod == null) {
			addPageMessage(Message.INSTRUCTION_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.PLEASE_SELECT_A_SP));
			return;
		}
		paginationPC = new JumpToPagePaginationPC(getToolbar(), getLocalizer());
		addChildComponent(paginationPC);
		resetListPC();
	}
	
	/**
	 * Re-factored this method because the list was being reset several places and properties
	 * weren't always set the same.
	 */
	 void resetListPC() {
		if (selectedSchedulingPeriod != null) {
			removeChildComponent(m_list);
			m_list = new CampaignSPQueuesListPC(m_context, spQueues, selectedSchedulingPeriod);
			m_list.setIsCancelSelectEventPropagation(false);
	
			if (PageAction.TABLE_SORT.equals(getPageAction()) ||
					PageAction.SAVE.equals(getPageAction()) ||
					isPaginationNavPressed()) {
				m_list.extractParameters(m_context.getRequest());
			}
			addChildComponent(m_list);
			
			//Unskilled SPs will have their combined queue rendered in its own container at the top of the page, with the
			//linked queues rendered in a separate container below that.
			if (!isSkillBased) {
				combinedQueueListPC = new CombinedSPQueueListPC(m_context, "combinedSPQueueListPC", spQueueCombined, selectedSchedulingPeriod);
				combinedQueueListPC.setStyle("border-bottom-color: #7085a8; border-bottom-width:1px;");
				CollapsibleContainerPC combinedSpQueueContainer = new CollapsibleContainerPC(m_context);
				combinedSpQueueContainer.setToggleable(false);
				combinedSpQueueContainer.setName("combinedSpQueueContainer");
				combinedSpQueueContainer.setTitle(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
						FsWebBundleKey.FS_COMBINED_QUEUE));
				combinedSpQueueContainer.setIsBorderEnabled(false);
				combinedQueueListPC.getWrapper().setAttribute("style", "width:100%; margin-bottom: 10px; border-bottom-color: #7085a8; border-bottom-width:1px;");
				combinedQueueListPC.setIsInnerTable(true);
				combinedQueueListPC.setIsBorderEnabled(false);
				combinedQueueListPC.setIsCancelSelectEventPropagation(false);
				combinedSpQueueContainer.addComponent(combinedQueueListPC);
				setSummaryComponent(combinedSpQueueContainer);
	
				CollapsibleContainerPC linkedSpQueueContainer = new CollapsibleContainerPC(m_context);
				linkedSpQueueContainer.setToggleable(false);
				linkedSpQueueContainer.setName("linkedSpQueueContainer");
				linkedSpQueueContainer.setTitle(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
						FsWebBundleKey.FS_LINKED_QUEUES));
				linkedSpQueueContainer.setIsBorderEnabled(false);
				m_list.setIsInnerTable(true);
				m_list.setIsBorderEnabled(false);
				m_list.setIsCancelSelectEventPropagation(false);
				linkedSpQueueContainer.addComponent(m_list);
				setListComponent(linkedSpQueueContainer);
			} else {
				removeChildComponent(m_list);
				setListComponent(m_list);
			}
		}
	}
	
	public HashMap<SPQueue, Collection<ID>> getSkillForSPQueues() {
		if (m_list instanceof CampaignSPQueuesListPC)
			return ((CampaignSPQueuesListPC)m_list).getSkillForSPQueues();
		else
			return null;
	}
	
	public JumpToPagePaginationPC getPaginationControl() {
		return paginationPC;
	}

	/**
	 * Returns the combined queue for the unskilled SP (returns null if the selected
	 * SP is skilled).
	 */
	public SPQueue getCombinedSPQueue() {
		return spQueueCombined;
	}
}
