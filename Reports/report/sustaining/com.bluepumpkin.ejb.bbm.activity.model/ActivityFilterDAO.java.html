package com.bluepumpkin.ejb.bbm.activity.model;

import com.bluepumpkin.ejb.bbm.Filter.Filter;
import com.bluepumpkin.ejb.bbm.Filter.FilterDAO;

/**
 * Title: Blue Pumpkin Software Basic Business Model Description: Copyright:
 * Copyright (c) 2001 Company: Blue Pumpkin Software, inc
 * 
 * @author
 * @version 1.0
 */

public class ActivityFilterDAO extends FilterDAO {

	/**
	 * Create activity filter DAO with an activity filter
	 * @param objActivityFilter
	 */
	public ActivityFilterDAO(Filter objActivityFilter) {
		super(objActivityFilter);
	}
	
	/**
	 * Create activity filter with a filter
	 * @param objFilter	Filter
	 * @return FilterDAO
	 */
	public FilterDAO createFilterDAO(Filter objFilter) {
		return new ActivityFilterDAO(objFilter);
	}

}
