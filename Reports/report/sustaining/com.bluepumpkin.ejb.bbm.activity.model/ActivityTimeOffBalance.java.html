package com.bluepumpkin.ejb.bbm.activity.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**
 * Title: Verint Software Foundation Description: ActivityTimeOffBalance holds a
 * record in ActivityTimeOffBalance table Copyright: Copyright (c) 2008 Company:
 * Verint, inc
 *
 * @author Sameet Joshi
 * @version 1.0 \

public final class ActivityTimeOffBalance extends ValueObjectBase {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// meta data info.
	private static FieldInfo m_fieldInfo = new ActivityTimeOffBalanceFieldInfo();
	// used by UI to display activity name
	// set to a very high value
	public final static int MAX_BALANCE_NA = 999999999;

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ActivityTimeOffBalance() {
		super();
	}

	@Override
	public ID getID() {
		return getFieldValueID(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_ID);
	}

	@Override
	public void setID(ID idEmployee) {
		setFieldValue(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_ID, idEmployee);
	}

	public ID getWorkResourceID() {
		return getFieldValueID(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_WORKRESOURCEID);
	}

	public ID getActivityID() {
		return getFieldValueID(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_ACTIVITYID);
	}

	public void setActivityCategoryId(ID id) {
		setFieldValue(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_ACTIVITYCATEGORYID, id);
	}

	public ID getActivityCategoryId() {
		return getFieldValueID(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_ACTIVITYCATEGORYID);
	}

	public double getMaxBalance() {
		return getFieldValueDouble(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_MAXBALANCE);
	}

	public double getMaxCarryOver() {
		return getFieldValueDouble(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_MAXCARRYOVER);
	}

	public boolean isZeroMaxCarryOver() {
		return ((int) getMaxCarryOver()) == 0;
	}

	public boolean maxCarryOverNotApplicable() {
		return ((int) getMaxCarryOver()) == ActivityTimeOffBalance.MAX_BALANCE_NA;
	}

	public void setWorkResourceID(ID empID) {
		setFieldValue(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_WORKRESOURCEID, empID);
	}

	public void setActivityID(ID idActivityID) {
		setFieldValue(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_ACTIVITYID, idActivityID);
	}

	public void setModifiedBy(String strValue) {
		setFieldValue(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_MODIFIEDBY, strValue);
	}

	public String getModifiedBy() {
		return getFormattedFieldValue(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_MODIFIEDBY);
	}

	public void setMaxBalance(double dblValue) {
		setFieldValue(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_MAXBALANCE, dblValue);
	}

	public void setMaxCarryOver(double dblValue) {
		setFieldValue(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_MAXCARRYOVER, dblValue);
	}

	public void setLastModifiedDate(Date modifiedDate) {
		setFieldValue(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_LASTMODIFIEDAT, modifiedDate);
	}

	public Date getLastModifiedDate() {
		return getFieldValueDate(ActivityTimeOffBalanceFieldInfo.ACT_TMOFF_BAL_LASTMODIFIEDAT);
	}

	public static ActivityTimeOffBalance getDefault(ID empID, ID activityID, ID actCatID) {
		ActivityTimeOffBalance bal = new ActivityTimeOffBalance();
		bal.setWorkResourceID(empID);
		bal.setActivityCategoryId(actCatID);
		bal.setActivityID(activityID);
		bal.setMaxBalance(MAX_BALANCE_NA);
		bal.setMaxCarryOver(MAX_BALANCE_NA);
		return bal;
	}

	public String getActivity_ActCategoryComb() {
		return Activity.getActivity_ActCategoryComb(getActivityID(), getActivityCategoryId());
	}
}
