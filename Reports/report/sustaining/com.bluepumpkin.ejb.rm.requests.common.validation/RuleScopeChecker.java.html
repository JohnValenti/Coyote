package com.bluepumpkin.ejb.rm.requests.common.validation;

import java.util.BitSet;
import java.util.Calendar;
import java.util.Collection;
import java.util.Date;
import java.util.Iterator;
import java.util.TimeZone;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.holiday.model.Holiday;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.setup.RuleScope;
import com.bluepumpkin.ejb.rm.util.RmUtil;

public class RuleScopeChecker {
   private static Category m_cat = Log.initCategory(RuleScopeChecker.class.getName());
   
  	/**
	 * Test to see if the given filing rule applies to the request, given the request's start date.
	 *
	 * @param rule       the rule to test with
	 * @param dtStart    the date/time that the requested period starts
	 * @param holiday    true if the requested range includes time on a working (unpaid) holiday
	 * @param daysOfWeek a bitset that indicates which organization days of the week are involved
	 *                   in the request.
	 */
	protected static boolean ruleApplies(RuleScope ruleScope, Date dtStart, boolean holiday, BitSet daysOfWeek) {
		boolean result = false;

		String FILING_RULE_APPLIES = "filing rule's scope applies: ";

		// Check what type of rule is used here
		String dayRange = ruleScope.getDayRange();
		if (dayRange.equals(RuleScope.DAYRANGE_ALL_DAYS)) {
			// simplest case, the rule applies
			m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYRANGE_ALL_DAYS);
			result = true;
		} else if (dayRange.equals(RuleScope.DAYRANGE_DAY_TYPE)) {
			// Test based on type of day
			String dayType = ruleScope.getDayType();
			if (dayType.equals(RuleScope.DAYTYPE_WORKINGHOLIDAY)) {
				if (holiday) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_WORKINGHOLIDAY);
					result = true;
				}
			} else if (dayType.equals(RuleScope.DAYTYPE_WEEKEND)) {
				int[] weekendDays = RmUtil.getWeekEndDays();
				if (daysOfWeek.get(weekendDays[0]) || daysOfWeek.get(weekendDays[1])) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_WEEKEND);
					result = true;
				}
			} else if (dayType.equals(RuleScope.DAYTYPE_WEEKDAY)) {
				int[] weekDays = RmUtil.getWeekDays();
				if (daysOfWeek.get(weekDays[0]) || 
						daysOfWeek.get(weekDays[1]) || 
						daysOfWeek.get(weekDays[2]) || 
						daysOfWeek.get(weekDays[3]) || 
						daysOfWeek.get(weekDays[4])) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_WEEKDAY);
					result = true;
				}
			} else if (dayType.equals(RuleScope.DAYTYPE_MONDAY)) {
				if (daysOfWeek.get(Calendar.MONDAY)) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_MONDAY);
					result = true;
				}
			} else if (dayType.equals(RuleScope.DAYTYPE_TUESDAY)) {
				if (daysOfWeek.get(Calendar.TUESDAY)) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_TUESDAY);
					result = true;
				}
			} else if (dayType.equals(RuleScope.DAYTYPE_WEDNESDAY)) {
				if (daysOfWeek.get(Calendar.WEDNESDAY)) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_WEDNESDAY);
					result = true;
				}
			} else if (dayType.equals(RuleScope.DAYTYPE_THURSDAY)) {
				if (daysOfWeek.get(Calendar.THURSDAY)) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_THURSDAY);
					result = true;
				}
			} else if (dayType.equals(RuleScope.DAYTYPE_FRIDAY)) {
				if (daysOfWeek.get(Calendar.FRIDAY)) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_FRIDAY);
					result = true;
				}
			} else if (dayType.equals(RuleScope.DAYTYPE_SATURDAY)) {
				if (daysOfWeek.get(Calendar.SATURDAY)) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_SATURDAY);
					result = true;
				}
			} else if (dayType.equals(RuleScope.DAYTYPE_SUNDAY)) {
				if (daysOfWeek.get(Calendar.SUNDAY)) {
					m_cat.debug(FILING_RULE_APPLIES + RuleScope.DAYTYPE_SUNDAY);
					result = true;
				}
			} // default, fall through for unknown day type
		} else if (dayRange.equals(RuleScope.DAYRANGE_DATES)) {
			// Checked with Howard.  The dates are supposed to be organization (non-local) dates
			// and specify the date and time so checking here is simple
			// Does the start date fall within the rule's date range?
			if (!dtStart.before(ruleScope.getStartDate()) &&
			        dtStart.before(ruleScope.getEndDate())) { 
				m_cat.debug(FILING_RULE_APPLIES + RmUtil.dumpCommaSeparated(RuleScope.DAYRANGE_DATES, dayRange));
				result = true;
			}
		}
		// let unknowns fall through and do not apply.

		if (!result) {
			m_cat.debug("Filing rule's scope does not apply: " +
			        RmUtil.dumpCommaSeparated(ruleScope.getDayRange(), ruleScope.getDayType()));
		}

		return result;
	}

    /**
     * Test to see if the given range overlaps time on a holiday.
     * @param dtStart the start of the range
     * @param dtEnd the end of the range
     */
    protected static boolean overlapsTimeOnAWorkingHoliday(Collection holidays, TimeZone tz,
                                                    Date dtStart, Date dtEnd) {
        boolean result = false;
        if(holidays!=null && !holidays.isEmpty()) {
	        for (Iterator it = holidays.iterator(); it.hasNext();) {
		        Holiday h = (Holiday) it.next();
		        if (h.isUnPaidHoliday()) {
			        // only consider un-paid holidays, they figure into filing rule calculations
			        // Get the absolute date range for the holdiay by converting
			        // from local date to date using the timezone
			        Date hStart = h.getStartDate().getTime(tz);
			        Date hEnd = h.getEndDate().getTime(tz);
	
			        // Test to see if the request range overlaps the holiday at all
			        if (dtStart.before(hEnd) && dtEnd.after(hStart)) {
				        result = true;
				        break;
			        }
		        }
	        }
        }
        return result;
    }
    
    /**
     * Build up a map of the days of the week that this request covers (in
     * whole or in part).
     * @param dtStart the start of the range
     * @param dtEnd the end of the range
     * @return a BitSet which uses the Calendar days of the week constants
     * as indexes.  A bit is set if the range includes that day, subject to 
     * the timezone and dayboundary offset used to construct this checker.
     */
    protected static BitSet buildDaysBitSet(Date dtStart, Date dtEnd,
                                            TimeZone tz, int dayBoundaryOffset) {
        BitSet result = new BitSet();
        
        Calendar cal = Calendar.getInstance(tz);
        cal.setTime(dtStart);
        
        // Find relative minute offset for the start time
        int startOffset = cal.get(Calendar.HOUR_OF_DAY)*60 + cal.get(Calendar.MINUTE);
        
        // Set the calendar to the beginning of the organization day
        cal.set(Calendar.HOUR_OF_DAY, dayBoundaryOffset/60);
        cal.set(Calendar.MINUTE, dayBoundaryOffset%60);
        cal.getTime();  // force calculation of fields
        
        // Find the beginning of the organization day that contains the range start
        if (startOffset < dayBoundaryOffset) {
            // The organization day began on the previous calendar day
            cal.add(Calendar.DATE, -1);
        } 
        
        // Now, increment the calendar one day at a time as long it points to the
        // beginning of an organization day that is before the end date.  For
        // each day, set the corresponding day-of-week bit on in the result
        do {
            result.set( cal.get(Calendar.DAY_OF_WEEK) );            
            cal.add(Calendar.DATE,1);
        } while (cal.getTime().before(dtEnd));
        
        return result;
    }
}
