package com.verint.web.rm.rest.service;

import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.rm.Log;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.security.WFOSecurity;
import com.verint.web.rest.service.AbstractWFOService;
import com.verint.web.rm.rest.entity.RequestListEntity;
import com.verint.web.rm.rest.facade.IRequestFacade;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.system.RequestContext;

@Path("/requests/")
@Produces("application/json")
@WFOSecurity
public class RequestRestService extends AbstractWFOService {
	private static final Category LOG = Log.initCategory(RequestRestService.class.getName());
	private static final String INCLUDE_EXPIRED_FN = "includeExpired";
	private static final String INCLUDE_WAITLIST_RANK_FN = "includeWaitlistRank";
	private static final String TYPE_FN = "type";

	@Context
	private HttpServletRequest request;
	@Context
	private HttpServletResponse response;
	@Context
	private ServletContext sc;

	private IRequestFacade requestFacade;

	public RequestRestService() { // NOSONAR
	}

	public RequestContext getRequestContext() {
		return new RequestContext(request, response, sc);
	}

	public void setRequestFacade(IRequestFacade requestFacade) {
		this.requestFacade = requestFacade;
	}

	/**
	 * Gets the current employee's requests.
	 *
	 * Parameters
	 * type: The type of requests to return. Defaults to all.
	 * includeExpired: (boolean) True to return expired requests.
	 *
	 * Example:
	 * http://localhost:7001/wfo/rest/rm-api/requests/getMyRequests?type=all&includeExpired=true
	 */
	@GET
	@Path("/getMyRequests")
	public RequestListEntity getMyRequests() throws WFOException {
		try {
			RequestContext context = getRequestContext();
			RmUtils.assertIsAuthorized(context, PrivilegeKeys.TOM_VIEWPERSONALREQUESTS_ID);

			ID employeeID = context.getUser().getEmployeeID();
			String type = request.getParameter(TYPE_FN);
			boolean includeExpired = RmUtils.getBoolean(request, INCLUDE_EXPIRED_FN, false);
			boolean includeWaitlistRank = RmUtils.getBoolean(request, INCLUDE_WAITLIST_RANK_FN, false);

			return requestFacade.getRequestsForEmp(context, employeeID, type, includeExpired, includeWaitlistRank);
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}
}
