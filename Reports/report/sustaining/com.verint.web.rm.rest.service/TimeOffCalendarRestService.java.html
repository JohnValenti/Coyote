package com.verint.web.rm.rest.service;

import java.util.Date;

import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;

import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.rm.Log;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.security.WFOSecurity;
import com.verint.web.rest.service.AbstractWFOService;
import com.verint.web.rm.rest.entity.timeoff.TimeOffDailyCalendarEntity;
import com.verint.web.rm.rest.entity.timeoff.TimeOffIntervalCalendarEntity;
import com.verint.web.rm.rest.facade.timeoff.ITimeOffCalendarFacade;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.ejb.core.licensing.LicenseKeys;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.system.RequestContext;

@Path("/timeoffcalendar/")
@Produces("application/json")
@WFOSecurity(licenses = { LicenseKeys.APP_RM_TOM })
public class TimeOffCalendarRestService extends AbstractWFOService {
	private static final Category LOG = Log.initCategory(TimeOffCalendarRestService.class.getName());
	private static final String IS_TOPOOL_VIEW_FN = "isTOPoolView";

	@Context
	private HttpServletRequest request;
	@Context
	private HttpServletResponse response;
	@Context
	private ServletContext sc;

	private ITimeOffCalendarFacade timeOffCalendarFacade;

	public TimeOffCalendarRestService() {
	}

	public RequestContext getRequestContext() {
		return new RequestContext(request, response, sc);
	}

	public ITimeOffCalendarFacade getTimeOffCalendarFacade() {
		return timeOffCalendarFacade;
	}

	public void setTimeOffCalendarFacade(ITimeOffCalendarFacade timeOffCalendarFacade) {
		this.timeOffCalendarFacade = timeOffCalendarFacade;
	}

	/**
	 * Gets the daily time-off calendar. This only gets the current employee's TOC.
	 *
	 * Parameters:
	 * startTime: (date) The start date & time of the request.
	 * endTime: (date) The end date & time of the request.
	 * isTOPoolView: (boolean) True to get the TOP TOC. False to get the employee's TOC. Defaults to false.
	 *
	 * Example:
	 * http://localhost:7001/wfo/rest/rm-api/timeoffcalendar/getDaily?startDate=2016-01-01T05:00:00Z&endDate=2017-01-01T05:00:00Z
	 */
	@GET
	@Path("/getDaily")
	public TimeOffDailyCalendarEntity getDailyTimeOffCalendar() throws WFOException {
		try {
			RequestContext context = getRequestContext();
			RmUtils.assertIsAuthorized(context, PrivilegeKeys.TOM_VIEWPERSONALCALENDAR_ID);

			boolean isTOPoolView = RmUtils.getBoolean(request, IS_TOPOOL_VIEW_FN, false);
			TimeRange timeRange = getTimeRange(context);

			return timeOffCalendarFacade.getDailyCalendarForEmployee(context, timeRange, isTOPoolView);
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}

	/**
	 * Gets the interval time-off calendar. This only gets the current employee's TOC.
	 *
	 * Parameters:
	 * startTime: (date) The start date & time of the request.
	 * endTime: (date) The end date & time of the request.
	 * isTOPoolView: (boolean) True to get the TOP TOC. False to get the employee's TOC. Defaults to false.
	 *
	 * Example:
	 * http://localhost:7001/wfo/rest/rm-api/timeoffcalendar/getInterval?startDate=2016-11-01T05:00:00Z&endDate=2016-12-01T05:00:00Z
	 */
	@GET
	@Path("/getInterval")
	public TimeOffIntervalCalendarEntity getIntervalTimeOffCalendar() throws WFOException {
		try {
			RequestContext context = getRequestContext();
			RmUtils.assertHasAdvancedRmLicense(context);
			RmUtils.assertIsAuthorized(context, PrivilegeKeys.TOM_VIEWPERSONALINTERVALCALENDAR_ID);

			boolean isTOPoolView = RmUtils.getBoolean(request, IS_TOPOOL_VIEW_FN, false);
			TimeRange timeRange = getTimeRange(context);

			return timeOffCalendarFacade.getIntervalCalendarForEmployee(context, timeRange, isTOPoolView);
		} catch (Exception ex) {
			throw RmUtils.handleServiceException(ex, LOG);
		}
	}

	private TimeRange getTimeRange(RequestContext context) throws BusinessWFOException {
		Date startDate = RmUtils.getDate(context, request, RmUtils.START_DATE_FN);
		Date endDate = RmUtils.getDate(context, request, RmUtils.END_DATE_FN);
		return new TimeRange(startDate, endDate);
	}
}
