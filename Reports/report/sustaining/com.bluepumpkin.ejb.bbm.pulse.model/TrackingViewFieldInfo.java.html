import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public final class TrackingViewFieldInfo extends FieldInfo {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	// field id constants
	public static final int TRACKINGVIEW_ID = 0;
	public static final int TRACKINGVIEW_USERID = 1;
	public static final int TRACKINGVIEW_VIEWNAME = 2;
	public static final int TRACKINGVIEW_VIEWDESC = 3;	
	public static final int TRACKINGVIEW_VIEWDEF = 4;	
	public static final int TRACKINGVIEW_ENABLETRENDING = 5;
	public static final int TRACKINGVIEW_USEHOO = 6;
	public static final int TRACKINGVIEW_GLOBAL = 7;
	public static final int TRACKINGVIEW_LASTUPDATE = 8;
	
	private static final int TRACKINGVIEW_NUMOFFIELDS = 9;

	private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[TRACKINGVIEW_NUMOFFIELDS];
	private static String m_strTableName = "USERTRACEVIEW";

	// static initialization
	static {
		m_arrFieldInfo[TRACKINGVIEW_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
		m_arrFieldInfo[TRACKINGVIEW_USERID] = new FieldInfoEntry("BPUSERID", PrimitiveTypes.ID);
		m_arrFieldInfo[TRACKINGVIEW_VIEWNAME] = new FieldInfoEntry("NAME", PrimitiveTypes.STRING);
		m_arrFieldInfo[TRACKINGVIEW_VIEWDESC] = new FieldInfoEntry("DESCRIPTION", PrimitiveTypes.STRING);
		m_arrFieldInfo[TRACKINGVIEW_VIEWDEF] = new FieldInfoEntry("VIEWDEFINITION", PrimitiveTypes.STRING);
		m_arrFieldInfo[TRACKINGVIEW_ENABLETRENDING] = new FieldInfoEntry("ISTRENDING", PrimitiveTypes.BOOLEAN);
		m_arrFieldInfo[TRACKINGVIEW_USEHOO] = new FieldInfoEntry("USEHOO", PrimitiveTypes.BOOLEAN);
		m_arrFieldInfo[TRACKINGVIEW_GLOBAL] = new FieldInfoEntry("ISGLOBALVIEW", PrimitiveTypes.BOOLEAN);
		m_arrFieldInfo[TRACKINGVIEW_LASTUPDATE] = new FieldInfoEntry("LASTMODIFIEDAT", PrimitiveTypes.DATE);		
	}

	public TrackingViewFieldInfo() {
		super();
	}

	// implement abstract methods in parent class
	public String getTableName() { return m_strTableName; }
	public int getIDFieldIndex() { return TRACKINGVIEW_ID; }
	public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }
}
