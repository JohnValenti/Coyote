	/**
	 * Add Data Range Data
	 */
	private static void addDateRangeData(RequestContext context,
			AutoProcessingRule rule, ResourceBundle rmBundle,
			TimeZone tz, DefaultMultiColumnNodeData row_data) {
		Localizer localizer = context.getLocalizer();
		ResourceBundle rmEjbBundle = localizer.getBundle(RmEjbBundleKey.BUNDLE_NAME);
		String dayRange = rule.getDayRange();
		if (dayRange == null || dayRange.equals(DAYRANGE_ALL_DAYS)) {
			row_data.add(RequestUtil.getAllDaysLabel(rmEjbBundle, localizer) );
		} else if (dayRange.equals(DAYRANGE_DAY_TYPE)) {
			String dayType = rule.getDayType();
			Collection dTypeList = AutoProcessingModelHandler.getDayTypeList(rmEjbBundle, localizer);
			Object args[] = new Object[]{ RequestUtil.getSelectLabel(dayType, dTypeList) }; 
			row_data.add( localizer.i18n(rmEjbBundle, RmEjbBundleKey.ORG_RM_FILING_RULE_LIST_FOR_DAYS_LABEL, args));
		} else if (dayRange.equals(DAYRANGE_DATES)) {
			String strFromDate = localizer.formatDateTime(rule.getStartDate(), tz);
			String strToDate = localizer.formatDateTime(rule.getEndDate(), tz);
			Object args[] = new Object[]{ strFromDate, strToDate };			  
			row_data.add( localizer.i18n(rmEjbBundle, RmEjbBundleKey.ORG_RM_FILING_RULE_FROM_TO_DATE, args));
		}
	}
	
	/**
	 * Add Auto Process Val Data
	 */
	private static void addAutoProcessValData(AutoProcessingRule rule,
			ResourceBundle rmBundle, ValidationRulesMap vrMap,
			Localizer localizer, DefaultMultiColumnNodeData row_data, boolean isTabbable, boolean useAriaTooltip) {
		StringBuffer sb = new StringBuffer(128);
		Collection validations = rule.getAutoProcessingValidations();
		if (rule.isApproval()) {
			String type = rule.getApprovalType();
			sb.append("<nobr>");
			if (StringUtil.isEmpty(type) || type.equals(APPROVAL_DENIAL_TYPE_NO)) {
				sb.append(HtmlUtil.getTabbableLabel(getApproveChoiceNoLabel(rmBundle, localizer), null));
			} else if (type.equals(APPROVAL_DENIAL_TYPE_YES)) {
				sb.append(HtmlUtil.getTabbableLabel(getApproveChoiceYesLabel(rmBundle, localizer), null));
			} else if (type.equals(APPROVAL_DENIAL_TYPE_VIOLATION)) {				
				sb.append(HtmlUtil.getTabbableLabel(getApproveChoiceViolationLabel(rmBundle, localizer), null));
				if (validations != null && validations.size() != 0) {
					Iterator itr = validations.iterator();
					while (itr.hasNext()) {
						AutoProcessingValidation apv = (AutoProcessingValidation)itr.next();
						if (apv.isApprove()) appendValInfo(localizer, vrMap, apv, sb, isTabbable, useAriaTooltip);
					}
				}
			}
			sb.append("</nobr>");
		}
		if (rule.isDenial()) {
			if (rule.isApproval()) sb.append("<br>");

			sb.append("<nobr>");
			String type = rule.getDenialType();
			if (StringUtil.isEmpty(type) || type.equals(APPROVAL_DENIAL_TYPE_NO)) {
				sb.append(HtmlUtil.getTabbableLabel(getDenyChoiceNoLabel(rmBundle, localizer), null));
			} else if (type.equals(APPROVAL_DENIAL_TYPE_YES)) {
				sb.append(HtmlUtil.getTabbableLabel(getDenyChoiceYesLabel(rmBundle, localizer), null));
			} else if (type.equals(APPROVAL_DENIAL_TYPE_VIOLATION)) {
				sb.append(HtmlUtil.getTabbableLabel(getDenyChoiceViolationLabel(rmBundle, localizer), null));
				if (validations != null && validations.size() != 0) {
					Iterator itr = validations.iterator();
					while (itr.hasNext()) {
						AutoProcessingValidation apv = (AutoProcessingValidation)itr.next();
						if (apv.isDeny())	appendValInfo(localizer, vrMap, apv, sb, isTabbable, useAriaTooltip);
					}
				}
			}
			sb.append("</nobr>");
		}
		if (rule.isWaitlist()) {
			if (rule.isApproval()) sb.append("<br>");

			sb.append("<nobr>");
			String type = rule.getWaitlistType();
			if (StringUtil.isEmpty(type) || type.equals(APPROVAL_DENIAL_TYPE_NO)) {
				sb.append(HtmlUtil.getTabbableLabel(getWaitlistChoiceNoLabel(rmBundle, localizer), null));
			} else if (type.equals(APPROVAL_DENIAL_TYPE_YES)) {
				sb.append(HtmlUtil.getTabbableLabel(getWaitlistChoiceYesLabel(rmBundle, localizer), null));
			} else if (type.equals(APPROVAL_DENIAL_TYPE_VIOLATION)) {
				sb.append(HtmlUtil.getTabbableLabel(getWaitlistChoiceViolationLabel(rmBundle, localizer), null));
				if (validations != null && validations.size() != 0) {
					Iterator itr = validations.iterator();
					while (itr.hasNext()) {
						AutoProcessingValidation apv = (AutoProcessingValidation)itr.next();
						if (apv.isWailist())	appendValInfo(localizer, vrMap, apv, sb, isTabbable, useAriaTooltip);
					}
				}
			}
			sb.append("</nobr>");
		}
		row_data.add(sb);
	}

	/**
	 * Create Rule Data Row
	 */
	public static DefaultMultiColumnNodeData createRuleDataRow(
			RequestContext context, AutoProcessingRule rule,
			ResourceBundle rmBundle, ValidationRulesMap vrMap,
			String chkbox, String orgName, boolean isFullList, TimeZone tz, Localizer localizer, boolean isTabbable, boolean useAriaTooltip) {
		// Create the row data
		String rowID = rule.getID().toString();
		DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData(rowID);
		String requestType = rule.getRequestType();

		boolean isTimeOff = (Request.REQUESTTYPE_TIMEOFF.equals(requestType) || requestType==null);
		boolean isShiftBid = Request.REQUESTTYPE_SHIFTBID.equals(requestType);
        boolean isShiftSwap = Request.REQUESTTYPE_SHIFTSWAP.equals(requestType);
        boolean isApprovedTimeOffWithdraw = Request.REQUESTTYPE_TIMEOFF_WITHDRAW.equals(requestType);
        boolean isCustomShift = Request.REQUESTTYPE_CUSTSHIFT.equals(requestType);
		boolean isFlexTime = Request.REQUESTTYPE_FLEXTIME.equals(requestType);
		boolean isShiftSwapWithdraw = Request.REQUESTTYPE_SHIFTSWAP_WITHDRAW.equals(requestType);

		// Checkbox cell
		if (isFullList)	row_data.add(chkbox);

		// Request Type cell
		String rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_TIME_OFF;
		if (isShiftBid) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_SHIFT_BIDDING;
		} else if (isShiftSwap) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_SHIFT_SWAP;
		} else if (isApprovedTimeOffWithdraw) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_TIME_OFF_WITHDRAW;
		} else if (isCustomShift) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_CUSTOM_SHIFT;
		} else if (isFlexTime) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_FLEX_TIME;
		} else if (isShiftSwapWithdraw) {
			rTypeRBKEY = RmWebBundleKey.REQUEST_TYPE_SHIFT_SWAP_WITHDRAW;
		}
        
		String rType = localizer.i18n(rmBundle, rTypeRBKEY);
		if (isTimeOff || isFlexTime) {
			//append activity name
			try {
				rType+=" ( " + ActivityModelHandler.getActivityByID(context, rule.getActivityId()).getName() + " )";
			} catch (Exception ex){}
		}
		else if (isCustomShift) { //Append Custom Shift Sub-Types
			String subTypes = "";
			
			ResourceBundle rmEjbBundle = localizer.getBundle(RmEjbBundleKey.BUNDLE_NAME);			
			boolean foundOne = false;
			if (rule.getRequestSubTypeRegNewShift()) {
				subTypes = localizer.i18n(rmEjbBundle, RmEjbBundleKey.ORG_RM_FILING_RULE_REG_NEW_SHIFT);
				foundOne = true;
			}
			if (rule.getRequestSubTypeRegShiftChange()) {
				subTypes += (foundOne?", ":"") + localizer.i18n(rmEjbBundle, RmEjbBundleKey.ORG_RM_FILING_RULE_REG_SHIFT_CHANGE);
				foundOne = true;
			}
			if (rule.getRequestSubTypeOTNewShift()) {
				subTypes += (foundOne?", ":"") + localizer.i18n(rmEjbBundle, RmEjbBundleKey.ORG_RM_FILING_RULE_OT_NEW_SHIFT);
				foundOne = true;
			}
			if (rule.getRequestSubTypeOTShiftChange()) {
				subTypes += (foundOne?", ":"") + localizer.i18n(rmEjbBundle, RmEjbBundleKey.ORG_RM_FILING_RULE_OT_SHIFT_CHANGE);
			}
			
			rType += " (" + subTypes + ")";
		}
		
		row_data.add(rType);

		// Date Range
		addDateRangeData(context, rule, rmBundle, tz, row_data);

		// Auto Processing Val Data
		addAutoProcessValData(rule, rmBundle, vrMap, localizer, row_data, isTabbable, useAriaTooltip);

		if (isFullList)	row_data.add(orgName);

		return row_data;
	}

	/**
	 * Append Validation Information
	 */
	public static void appendValInfo(Localizer localizer, ValidationRulesMap vrMap,
			AutoProcessingValidation apv, StringBuffer sb, boolean isTabbable, boolean useAriaTooltip) {
		try {
			String rbKey = apv.getBundleKey();
			ValidationResources vrs = vrMap.getValResource(rbKey);
			if (vrs != null) {
				String iconPath = vrs.getIconPath(localizer);
				sb.append(HtmlUtil.imageTag(iconPath, vrs.getJSMessage(localizer), isTabbable, useAriaTooltip));
				sb.append(" ");
			}
		} catch (Exception e) {
		} // ignore
	}
