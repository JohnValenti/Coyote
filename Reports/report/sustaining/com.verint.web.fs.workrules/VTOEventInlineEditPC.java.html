public class VTOEventInlineEditPC extends InsertableMultiColListPC {
	private static final long serialVersionUID = 1L;

	private HashMap<VTOEvent, ShiftPattern.StartTime> modifiedEvents = new HashMap<VTOEvent, ShiftPattern.StartTime>();
	private HashMap<VTOEvent, Set<ShiftPattern.StartTime>> events;
	private ToolbarTextButton addButton;
	
	public VTOEventInlineEditPC(RequestContext context, String name, Map<VTOEvent, Set<ShiftPattern.StartTime>> vtoEvents, boolean isByOrg, ID ownerID) {
		super(context);
		setName(name);
		events = new HashMap<VTOEvent, Set<ShiftPattern.StartTime>>(vtoEvents);

		addPopupArgs(POPUP_ID, VTOEventListPopupPM.FORM_ACTION, "",
				POPUP_ARGS, "1000,600,ctr,ctr", true, 1);

		// Add javascript files required by the dropdown component. This is necessary because we dynamically 
		// add rows with a dropdown without a postback. If there are no rows initially, we still need the dropdown's
		// required javascript files
		StartTimeDDPC ddpc = new StartTimeDDPC(m_context, "dummy", false, false);
		for (String file : ddpc.getRequireJavaScriptFiles()) addRequiredJavaScriptFile(file);
		
		if (ownerID != null) {
			ToolbarPC toolbar = new ToolbarPC(m_context);
			toolbar.setName(getToolbarName());
			addButton = toolbar.createPopupWindowButton(POPUP_ID,
					m_localizer.i18n(CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_ADD), true);
			addButton.setAttribute("ownerID", ownerID.toString());
			addButton.setAttribute("isByOrg", String.valueOf(isByOrg));
			addButton.setAttribute("isSelectionPopup", String.valueOf(true));
			addButton.setAttribute("onClick", getFilterJSForAddButton(getName(), toolbar.getName(), POPUP_ID));

			toolbar.addToolbarElement(addButton);
			ToolbarTextButton removeButton = new ToolbarTextButton(toolbar, "removeVTOEvent",
					m_localizer.i18n(CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_REMOVE),
					InsertableMultiColListPC.REMOVE_ROW_BUTTON_ACTION, ToolbarTextButton.BLUE_BUTTON, false);
			toolbar.addToolbarElement(removeButton);
			setToolbar(toolbar);
		}
		setIsBorderEnabled(true);
		extractParameters(m_context.getRequest());
	}
	
	@Override
	public String getJavaScriptInitCode() {
		StringBuffer js = new StringBuffer(super.getJavaScriptInitCode());
		if (getToolbar() != null) {
			js.append("\n").append(getName()).append(".setToolbar(").append(getToolbar().getName()).append(");");
			js.append("\n").append(getName()).append(".addButtonEnableAttribute('isDeleteable', 'removeVTOEvent');\n");
		}
		return js.toString();
	}

	@Override
	public void setName(String name) {
		super.setName(name);
		ToolbarPC toolbar = getToolbar();
		if (toolbar!= null) {
			toolbar.setName(getToolbarName());
			addButton.setAttribute("onClick",
				toolbar.getName() + ".onSelect('" + POPUP_ID + "', this);");
		}
	}

	private String getToolbarName() {
		return getName() + "__toolbar"; 
	}


	private boolean isHeaderInited = false;
	private void initHeader() {
		if (isHeaderInited) return;
		isHeaderInited = true;
		
		TableHeaderPC header = getHeader();
		Localizer localizer = m_context.getLocalizer();
		DefaultHeaderData dhd = new DefaultHeaderData(FsWebBundleKey.FS_VTO_EVENT,
				getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_VTO_EVENT));
		dhd.addImageButton(getVTOEventPopupImage(), getVTOEventPopupCaption(), getVTOEventPopupJS());
		dhd.setSortable(true);
		header.addColumnHeaderData(dhd);
		header.addColumnHeaderData(FsWebBundleKey.FS_START_TIMES,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_START_TIMES));
	}
	
	private String getVTOEventPopupImage() {
		return ImageFileID.TOOLTIP_BUTTON;
	}
	
	private String getVTOEventPopupCaption() {
		return m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_VIEW_VTO_EVENT_DETAILS);
	}
	
	private String getVTOEventPopupJS() {
		return "WorkRulesUtil.handleInfoPopup(this, " + getName() + "," + getHeader().getName() + ",'" +
			VTOEventListPopupPM.POPUP_ID + "','" + VTOEventListPopupPM.POPUP_ARG_VTO_EVENT_ID + "','" +
			VTOEventListPopupPM.POPUP_ARG_VTO_EVENT_IDS + "');";	
	}

	private boolean isRowsInited = false;
	private void initRows() {
		if (isRowsInited) return;
		isRowsInited = true;
		
		List<MultiColumnNodeData> rows = super.getListModel();

		ArrayList<ID> ids = new ArrayList<ID>();
		for (VTOEvent vtoEvent : events.keySet()) {
			DefaultMultiColumnNodeData nodeData = getNodeData(vtoEvent);
			rows.add(nodeData);
			ids.add(vtoEvent.getID());
		}

		SortUtil.sort(getListModel(), getHeader().getColumnIndexByName(FsWebBundleKey.FS_VTO_EVENT),
				true, m_context.getLocalizer());
		
		this.setRowIDs(StringUtil.createDelimitedString(ids.toArray(new ID[ids.size()])));
	}
	
	protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, VTOEvent vtoEvent) {
		if (!vtoEvent.isMultiEditCommon()) mcnd.setNodeAttribute(ROW_ATTRIBUTE_MULTI_EDIT_NOT_COMMON, String.valueOf(true));
		mcnd.setNodeAttribute(WorkpaneMultiColPM.IS_DELETEABLE, String.valueOf(true));
		mcnd.setNodeAttribute(VTOEventListPopupPM.POPUP_ARG_VTO_EVENT_ID, vtoEvent.getID().toString());
	}
	
	/**
	 * To be called by the request handler that dynamically returns a html row via ajax.
	 * @param VTOEvent
	 * @return
	 */
	public String getRowHtml(VTOEvent VTOEvent) {
		StringBuffer html = new StringBuffer();
		super.appendHtmlRowData(html, getNodeData(VTOEvent), 0, 0);
		return html.toString();
	}
	
	protected DefaultMultiColumnNodeData getNodeData(VTOEvent vtoEvent) {
		DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(
				vtoEvent.getID());
		String vtoEventNameHTML = vtoEvent.getName();
		//Name
		if (vtoEvent.isMultiEditCommon() == false) {
			vtoEventNameHTML = MULTI_VALUE_DATA + vtoEventNameHTML;
		} 
		
		if (vtoEvent.getDeleteOnDate() != null) {
			vtoEventNameHTML = new HtmlDel(vtoEventNameHTML).toString();
		}
		
		nodeData.add(vtoEventNameHTML);
		
		
		//Start Times
		StartTimeDDPC ddpc = getStartTimeDDPC(vtoEvent);
		addChildComponent(ddpc);
		nodeData.add(ddpc);
		
		setRowAttributes(nodeData, vtoEvent);
		return nodeData;
	}
	
	private StartTimeDDPC getStartTimeDDPC(VTOEvent vtoEvent) {
		boolean isDeleted = vtoEvent.getDeleteOnDate() != null;
		StartTimeDDPC ddpc;
		Set<ShiftPattern.StartTime> startTimes = events.get(vtoEvent);
		if (startTimes == null || startTimes.size() == 0) {
			ddpc = new StartTimeDDPC(getRequestContext(),
				getName() + "__" + JSUtil.getJSSuffix(vtoEvent.getID().toString()), true, isDeleted);
		} else {
			ddpc = new StartTimeDDPC(getRequestContext(),
					getName() + "__" + JSUtil.getJSSuffix(vtoEvent.getID().toString()), true, isDeleted, startTimes.iterator().next());
			if (startTimes.size() > 1) ddpc.setMultiEdit(true);
		}
		ddpc.extractParameters(m_context.getRequest());
		if (events.get(vtoEvent) != null && events.get(vtoEvent).size() > 0) {
			if (ddpc.isEdited()) {
				modifiedEvents.put(vtoEvent, ddpc.getSelections().iterator().next());
			}
			if ((vtoEvent.isMultiEditCommon() || ddpc.isEdited())  && ddpc.getSelections().size() > 0) { 
				events.put(vtoEvent, Collections.singleton(ddpc.getSelections().iterator().next()));
			}
		}
		return ddpc;
	}
	
	private boolean isExtracted = false;
	@Override
	public boolean extractParameters(HttpServletRequest request) {
		if (isExtracted) return true;
		isExtracted = true;
		boolean retVal = super.extractParameters(request);
		
		// removed
		for (ID removedID : getRemovedRowIDCollection()) {
			for (VTOEvent vtoE : events.keySet()) {
				if (removedID.equals(vtoE.getID())) {
					events.remove(vtoE);
					break;
				}
			}
		}
		
		// added
		try {
			Collection<ID> addedEventIDs = getAddedRowIDCollection();
			if (addedEventIDs.size() > 0) {
				for (VTOEvent vtoEvent : WorkRulesModelHandler.getVTOEventsByIDs(m_context, addedEventIDs)) {
					//we want to remove any existing events with the same ID as the newly added ones will trump these
					events.remove(vtoEvent);
					vtoEvent.setMultiEditCommon(true);
					events.put(vtoEvent, Collections.singleton(ShiftPattern.StartTime.StartOrEndOfShift)); // This will be updated in the initRows method call below
				}
			}
		} catch (BbmFinderException e) {
			retVal = false;
			handleUnableToLoadDataError(log, e);
		}

		initHeader();
		initRows();
		
		return retVal;
	}

	public HashMap<VTOEvent, ShiftPattern.StartTime> getAddedVTOEventAssignments() {
		HashMap<VTOEvent, ShiftPattern.StartTime> retVal = new HashMap<VTOEvent, ShiftPattern.StartTime>();
		Map<ID, VTOEvent> eventMap = ValueObjectUtil.getIDObjectMap(events.keySet());
		for (ID id : getAddedRowIDCollection()) {
			retVal.put(eventMap.get(id), events.get(eventMap.get(id)).iterator().next());
		}
		return retVal;
	}
	
	public HashMap<VTOEvent, ShiftPattern.StartTime> getModifiedVTOEventAssignments() {
		return modifiedEvents;
	}
}
