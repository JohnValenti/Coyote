package com.verint.web.fs.workrules;

import java.util.Collection;

import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.web.bbm.activity.ActivityListPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.SortUtil;

/**
 * Similar to ActivityListPC but with only about half of the columns rendered.
 * Currently used by the Shift Event creation screen.
 */
public class CompactActivityListPC extends ActivityListPC {
	private static final long serialVersionUID = 1L;
	
	public CompactActivityListPC(RequestContext context, String name, Collection<Activity> activities) {
		super(context, name, activities);
	}
	
	/**
	 * ActivityListPC super class calls this method to determine if the full set of columns is to be rendered.
	 */
	protected boolean isFullList() {
		return false;
	}
	
	protected void init(RequestContext context, String name, Collection<Activity> activities)
	{
		super.init(context, name, activities);
		//Sort the rows by item name.  User sorting is disabled for this list, so we will sort
		//by activity name as a default.
		SortUtil.sort(getListModel(), getHeader().getColumnIndexByName(getNameColumnIdentifier()),
				true, m_context.getLocalizer());
	}
}
