package com.verint.web.fs.workrules;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.workrules.model.Rotation;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

public class RotationsRH extends WorkRulesRH<RotationsListPM, RotationsFormPM> {
	@Override
	public void processRequest(RequestContext context) throws Exception {
		String pageAction = getPageAction(context);
		//Remove Session Item Index when pageAction is null or refresh or is created new!!!
		if (pageAction == null || pageAction.equals(PageAction.NEW_ID) || pageAction.equals(PageAction.REFRESH)) {
			String strItemIndex = (String) context.getAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX);
			if (!StringUtil.isEmpty(strItemIndex)) {
				context.removeAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX);
			}
		} else {
			super.processRequest(context);
		}

	}

	@Override
	protected RotationsListPM getListPageModel(RequestContext context) {
		// Don't getInstance yet because the pagination needs to be set before the data is loaded
		// Instead of calling getInstance, instantiate and extract params.
		RotationsListPM pm = new RotationsListPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
	}

	@Override
	protected RotationsFormPM getFormPageModel(RequestContext context) {
		return RotationsFormPM.getInstance(context);
	}

	@Override
	protected void processImport(RequestContext context,
			RotationsListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processExport(RequestContext context,
			RotationsListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processCopy(RequestContext context, RotationsListPM pageModel)
			throws BPException {
		WorkRulesModelHandler.copyRotationsToOrg(
				context,
				pageModel.getPerfromActionOnIDsAsCollection(),
				pageModel.getSelectedIDObject(),
				pageModel.getLocalizer().getLocaleContext());
		context.setPageModel(null); // reset the page model to refresh the list

	}

	@Override
	protected void processDelete(RequestContext context,
			RotationsListPM pageModel) throws BPException {
		WorkRulesModelHandler.deleteRotations(context, pageModel.getPerfromActionOnIDsAsCollection());
		context.setPageModel(null); // reset the page model to remove the deleted entries
	}

	@Override
	protected void processSave(RequestContext context, RotationsFormPM pageModel)
			throws BPException {
		if (pageModel.validate()) {
			Rotation rot = pageModel.getRotation();
			WorkRulesModelHandler.saveRotation(context, rot);
		} else {
			setNextPage(context, FORM_PAGE);
		}
	}

	@Override
	protected void processDownload(RequestContext context,
			RotationsListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processFind(RequestContext context, RotationsListPM pageModel) throws BPException {
		pageModel.extractParameters(context.getRequest());
		context.setPageModel(pageModel);
	}

	@Override
	protected void processTableSort(RequestContext context, RotationsListPM pageModel) {
		String sortColumn = context.getRequest().getParameter("sortColumn");
		String sortOrder = context.getRequest().getParameter("sortOrder");
		context.setUserProperty(RotationsListPM.ROTATION_LIST_SORTCOLUMN, sortColumn, true);
		context.setUserProperty(RotationsListPM.ROTATION_LIST_SORTORDER, sortOrder, true);
	}

	@Override
	protected void processPageLeft(RequestContext context, RotationsListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexBackward();
		String strItemIndex =
				(String) context.getAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex)) {
			context.removeAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX);
		}
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX, Integer.toString(itemIndex));
	}

	@Override
	protected void processPageRight(RequestContext context, RotationsListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexForward();
		String strItemIndex =
				(String) context.getAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex)) {
			context.removeAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX);
		}
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX, Integer.toString(itemIndex));
	}

	@Override
	protected void processPageSelect(RequestContext context, RotationsListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);

		pageModel.setPagePaginationAndEditIndices(
				pageModel.getPaginationControl().getSelectedPaginationPageIndex());
		String strItemIndex =
				(String) context.getAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex)) {
			context.removeAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX);
		}
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE, RotationsListPM.ROTATIONLIST_ITEMINDEX, Integer.toString(itemIndex));
	}
}
