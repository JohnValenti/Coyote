package com.verint.web.fs.workrules;

import com.bluepumpkin.common.datatypes.Duration;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignOrg;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectIDListComparator;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.VTOEvent;
import com.bluepumpkin.web.bbm.activity.ActivityModelHandler;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.widgets.ActivitySelectorDDPC;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.form.FormInputPC;

import com.witness.web.uif.pagecomponent.picker.time.DurationSpinnerPC;

import java.util.*;

@SuppressWarnings("serial")
public class VTOEventFormPM extends WorkRulesFormPM {

	private final ResourceBundle m_bundle;
	private static final String FN_VTO_EVENT_NAME = "vtoEventName";
	private static final String FN_VTO_EVENT_DESC = "shiftEventDescription";
	private static final String JSVARNAME_ACTIVITY_ID = "activityID";
	private static final String JSVARNAME_MINIMUM_DURATION = "minCount";
	private static final String JSVARNAME_MAXIMUM_DURATION = "maxCount";

	// pc_ prefix is for page components
	private FormTwoColumnPC pc_form;
	private FormInputPC pc_name;
	private FormInputPC pc_description;

	private ActivitySelectorDDPC pc_activity;
	private DurationSpinnerPC pc_minimumDuration;
	private DurationSpinnerPC pc_maximumDuration;

	// Data
	private List<VTOEvent> vtoEvents = new ArrayList<VTOEvent>();
	protected Map<ID, Activity> activityMap = new HashMap<ID, Activity>();
	private String m_optimizationTargetName = "";
	protected Collection<Integer> minDurationList = new HashSet<Integer>();
	protected Collection<Integer> maxDurationList = new HashSet<Integer>();

	public VTOEventFormPM(RequestContext context) {
		super(context);
		setRenderJavaScriptIfDisabled(true);
		setName("WorkRulesShiftEventsFormPM");
		m_bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);
	}

	public static VTOEventFormPM getInstance(RequestContext context) {
		return (VTOEventFormPM) getInstance(context, VTOEventFormPM.class);
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		m_contentTitlePC.setPageTitle(m_localizer
				.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.WORKRULES_VTO_EVENTS_TITLE));
		if (isMultiEdit()) {
			m_contentTitlePC.setItemName(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.N_SHIFT_EVENTS,
					new Integer(vtoEvents.size())));
		} else if (vtoEvents.size() > 0) {
			m_contentTitlePC.setItemName(vtoEvents.iterator().next().getName());
		}
	}

	@Override
	protected void loadData() throws Exception {
		super.loadData();

		if (getPerformActionOnIDSize() > 0) {
			List<ID> paoIDs = Arrays.asList(getPerformActionOnIDs());
			vtoEvents = new ArrayList<VTOEvent>(WorkRulesModelHandler.getVTOEventsByIDs(m_context, paoIDs));
			Collections.sort(vtoEvents, new ValueObjectIDListComparator(paoIDs));

			// For single edit, get optimization target name (either org name or
			// "Local")
			if (vtoEvents.size() == 1) {
				VTOEvent vto = vtoEvents.iterator().next();
				if (vto.getCampaignID() != null) {
					m_optimizationTargetName = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.LOCAL);
				}
				if (vto.getOrganizationID() != null) {
					Organization o = OrganizationModelHandler.getOrganization(m_context, vto.getOrganizationID());
					if (o != null) {
						m_optimizationTargetName = o.getName();
					}
				}
			}
			Collection<ID> activityIDs = new ArrayList<ID>();
			for (VTOEvent vto : vtoEvents) {
				ID id = vto.getActivityID();
				if (id != null) {
					activityIDs.add(id);
				}
				minDurationList.add(vto.getMinDuration());
				maxDurationList.add(vto.getMaxDuration());
			}
			Collection<Activity> activities = ActivityModelHandler.getActivitiesByIDs(m_context, activityIDs);
			for (Activity a : activities) {
				activityMap.put(a.getID(), a);
			}
		}

		// If editing a new VTO Event, or multi-editing
		if (StringUtil.isEmpty(m_optimizationTargetName)) {
			if (isByOrg && selectedOrg != null) {
				m_optimizationTargetName = selectedOrg.getName();
			} else if (!isByOrg && selectedCampaign != null) {
				m_optimizationTargetName = selectedCampaign.getName();
			}
		}
	}

	@Override
	protected void initForm() {

		pc_form = new FormTwoColumnPC(m_context);
		setForm(pc_form);

		try {
			VTOEvent vto = new VTOEvent();
			if (vtoEvents.size() > 0) {
				vto = vtoEvents.iterator().next();
			}

			// Org or Campaign
			// Net to build a list of organizations to use for scoped
			// retrieval of valid activities.
			Collection<ID> orgs = new ArrayList<ID>();
			if (isByOrg) {
				String orgName = "";
				if (selectedOrg != null) {
					orgName = selectedOrg.getName();
					orgs.add(selectedOrg.getID());
				}
				pc_form.addRow(i(FsWebBundleKey.FS_ORGANIZATION), orgName);
			} else {
				String campaignName = "";
				if (selectedCampaign != null) {
					campaignName = selectedCampaign.getName();

					Collection<CampaignOrg> orgAssigned = CampaignModelHandler.getCampaignOrgAssignments(m_context,
							selectedCampaign.getID(), selectedSchedulingPeriod.getStartTime(),
							selectedSchedulingPeriod.getEndTime());
					for (CampaignOrg co : orgAssigned) {
						orgs.add(co.getOrganizationID());
					}
				}
				pc_form.addRow(i(FsWebBundleKey.FS_CAMPAIGN), campaignName);
			}

			if (isMultiEdit() == false) {
				// Name
				pc_name = new FormInputPC(m_context);
				pc_name.reset(FN_VTO_EVENT_NAME, vto.getName(), FormInputPC.TYPE_STRING, true);
				pc_name.setAutoExtractParamEnabled(true);
				pc_name.setMaxLength(50);
				pc_name.setInputFieldDesc(UIFWebBundleKey.NAME, UIFWebBundleKey.BUNDLE_NAME);
				pc_name.setIsRequired(true);
				pc_name.setInputValue(vto.getName());
				this.addChildComponent(pc_name);
				pc_form.addRow(i18n(m_common_bundle, UIFWebBundleKey.NAME), pc_name);

				// Description
				pc_description = new FormInputPC(m_context);
				pc_description.reset(FN_VTO_EVENT_DESC, vto.getName(), FormInputPC.TYPE_STRING, false);
				pc_description.setAutoExtractParamEnabled(true);
				pc_description.setMaxLength(250);
				pc_description.setInputFieldDesc(UIFWebBundleKey.DESCRIPTION, UIFWebBundleKey.BUNDLE_NAME);
				pc_description.setInputValue(vto.getDescription());
				this.addChildComponent(pc_description);
				pc_form.addRow(i18n(m_common_bundle, UIFWebBundleKey.DESCRIPTION), pc_description);
			}

			Collection<ID> activityIDs = new ArrayList<ID>();
			ID VToActivityID = new ID("-4136");
			activityIDs.add(VToActivityID);
			Collection<Activity> activities = ActivityModelHandler.getActivitiesByIDs(getRequestContext(), activityIDs);
			if (activityMap.size() > 0) {
				pc_activity = new ActivitySelectorDDPC(m_context, JSVARNAME_ACTIVITY_ID, activities, activityMap.keySet()
						.iterator().next().toString());
			} else {
				pc_activity = new ActivitySelectorDDPC(m_context, JSVARNAME_ACTIVITY_ID, activities);
			}
			pc_activity.setMultiEdit(activityMap.size() > 1);
			this.addChildComponent(pc_activity);
			pc_form.addRow(i18n(m_bundle, FsWebBundleKey.FS_ACTIVITY), pc_activity);
			pc_activity.setEnabled(false);

			// Min gap
			pc_minimumDuration = new DurationSpinnerPC(m_context, JSVARNAME_MINIMUM_DURATION);
			pc_minimumDuration.setMinutesIncrement(15);
			pc_minimumDuration.setMinTime(0, 15, 0);
			pc_minimumDuration.setMaxTime(23, 0, 0);
			if (vto != null) {
				pc_minimumDuration.setValue(Duration.fromMinutes(vto.getMinDuration()));
			}
			if (minDurationList.size() > 1) {
				pc_minimumDuration.setMultiEdit(true);
			}
			this.addChildComponent(pc_minimumDuration);
			pc_form.addRow(i(FsWebBundleKey.FS_MIN_DURATION), pc_minimumDuration);

			// Max spacing
			pc_maximumDuration = new DurationSpinnerPC(m_context, JSVARNAME_MAXIMUM_DURATION);
			pc_maximumDuration.setMinutesIncrement(15);
			pc_maximumDuration.setMinTime(0, 15, 0);
			pc_maximumDuration.setMaxTime(15, 0, 0);
			if (vtoEvents.size() == 0) {
				// Default max duration = 1 hour
				pc_maximumDuration.setValue(Duration.fromMinutes(60));
			} else if (vto != null) {
				pc_maximumDuration.setValue(Duration.fromMinutes(vto.getMaxDuration()));
			}
			pc_maximumDuration.setMultiEdit(maxDurationList.size() > 1);
			this.addChildComponent(pc_maximumDuration);
			pc_form.addRow(i(FsWebBundleKey.FS_MAX_DURATION), pc_maximumDuration);

			// Set page component state from submitted values
			if (null != getPageAction() && !PageAction.EDIT.equals(getPageAction())) { // Is
				// postback
				this.extractParameters(m_context.getRequest());
			}
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		}
	}

	/**
	 * Gets vto events represented by this form.
	 *
	 * @return
	 */
	public Collection<VTOEvent> getVTOEvents() {
		Collection<VTOEvent> ret = new ArrayList<VTOEvent>(vtoEvents);
		if (ret.isEmpty()) {
			ret.add(new VTOEvent());
		}

		boolean isMultiEdit = isMultiEdit();

		for (VTOEvent vto : ret) {
			// Construct a shift event from the form field
			if (!isMultiEdit() || pc_activity.isEdited()) {
				vto.setActivityID(pc_activity.getSelections().get(0).getID());
			}
			if (!isMultiEdit) {
				vto.setName(pc_name.getInputValue() != null ? pc_name.getInputValue().trim() : null);
				vto.setDescription(pc_description.getInputValue());
				if (isByOrg) {
					vto.setOrganizationID(selectedOrg.getID());
				} else if (!isByOrg) {
					vto.setCampaignID(selectedCampaign.getDEID());
				}
			}

			if (!isMultiEdit || pc_maximumDuration.isEdited()) {
				vto.setMaxDuration(pc_maximumDuration.getDuration().getDurationInMinutes());
			}
			if (!isMultiEdit || pc_minimumDuration.isEdited()) {
				vto.setMinDuration(pc_minimumDuration.getDuration().getDurationInMinutes());
			}
		}
		return ret;
	}

	@Override
	public boolean validate() {
		boolean retVal = super.validate();
		HashMap<ID, Activity> activities = new HashMap<ID, Activity>(activityMap);

		Collection<VTOEvent> editedVTOEvents;
		HashMap<ID, String> allVTOEventNames;
		try {
			if (isByOrg) {
				allVTOEventNames = WorkRulesModelHandler.getVTONamesByOrg(m_context, selectedOrg.getID());
			} else {
				allVTOEventNames = WorkRulesModelHandler.getVTONamesBySchedulingPeriod(m_context, selectedSchedulingPeriod);
			}
			editedVTOEvents = getVTOEvents();
		} catch (BbmFinderException e1) {
			handleUnableToLoadDataError(log, e1);
			return false;
		}

		for (VTOEvent vtoevent : editedVTOEvents) {
			HashSet<ID> vtoActivityIDs = new HashSet<ID>();
			for (VTOEvent vto : vtoEvents) {
				vtoActivityIDs.add(vto.getActivityID());
			}
			vtoActivityIDs.removeAll(activities.keySet());
			try {
				if (!vtoActivityIDs.isEmpty()) {
					activities.putAll(WorkRulesModelHandler.getActivities(m_context, vtoActivityIDs));
				}
			} catch (BbmFinderException e) {
				handleUnableToLoadDataError(log, e);
				return false;
			}

			// Name must be unique
			if (isEditingNameAndDesc()) {
				String name = vtoevent.getName();
				for (ID existingVTOEventID : allVTOEventNames.keySet()) {
					if (existingVTOEventID.equals(vtoevent.getID())) {
						continue;
					}
					if (name.equals(allVTOEventNames.get(existingVTOEventID))) {
						addPageMessage(Message.ERROR_TYPE,
								m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_NAME_MUST_BE_UNIQUE));
						retVal = false;
					}
				}
			}

			// Max gap must be > min gap
			if (vtoevent.getMinDuration() > vtoevent.getMaxDuration()) {
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
						FsWebBundleKey.WORKRULES_MAX_DURATION_MUST_BE_GREATER_THAN_MIN_DURATION));
				retVal = false;
			}

		}
		return retVal;
	}

	private String i(String key) {
		return m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, key);
	}

	private boolean isEditingNameAndDesc() {
		return !isMultiEdit();
	}
}
