	@Override
	protected void loadData() throws Exception {
		super.loadData();
		WorkRuleManager wrm = WfmManagerFactory.getWorkRuleManager(m_context.isInWhatIfMode());

		if (getPerformActionOnIDSize() > 0) {
			shifts = new ArrayList<Shift>(wrm.getShiftsByIDs(Arrays.asList(getPerformActionOnIDs())));
		} else { // new shift, set duration to 8 hours by default
			durations.add(8 * 60);
		}
		Collection<ID> orgIDs = new HashSet<ID>();
		Collection<ID> campaignIDs = new HashSet<ID>();
		for (Shift shift : shifts) {
			shiftMap.put(shift.getID(), shift);
			if (shift.getActivityID() != null) {
				shiftActivityIDs.add(shift.getActivityID());
			}
			if (shift.getOrganizationID() != null) {
				orgIDs.add(shift.getOrganizationID());
			}
			if (!isByOrg && shift.getCampaignID() != null) {
				campaignIDs.add(shift.getCampaignID());
			}
			durations.add(shift.getDuration());
			minSpacings.add(shift.getMinSpacing());
			maxSpacings.add(shift.getMaxSpacing());
			for (ShiftEvent se : shift.getShiftEvents()) {
				if (se.getOrganizationID() != null) {
					orgIDs.add(se.getOrganizationID());
				}
				if (!isByOrg && se.getCampaignID() != null) {
					campaignIDs.add(se.getCampaignID());
				}
				if (se.getActivityID() != null) {
					shiftEventActivityIDs.add(se.getActivityID());
				}
			}
		}

		if (isByOrg) {
			if (selectedOrg != null) {
				orgIDs.add(selectedOrg.getID());
			}
		} else if (selectedCampaign != null) {
			campaignIDs.add(selectedCampaign.getID());
		}

		organizationMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!isByOrg) {
			campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		}
		Collection<ID> activityIDs = new HashSet<ID>(shiftActivityIDs);
		activityIDs.addAll(shiftEventActivityIDs);
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);

		if (isByOrg) {
			if (selectedOrg != null) {
				optimizationTargetTimeZone = selectedOrg.getTimeZone();
			}
		} else {
			if (selectedCampaign != null) {
				optimizationTargetTimeZone = selectedCampaign.getTimeZone();
			}
		}
		if (optimizationTargetTimeZone == null) {
			optimizationTargetTimeZone = m_context.getViewingTimeZone(); // Shouldn't
			// happen,
			// but
			// good
			// to
			// have
			// a
			// default
		}

		if (selectedOrg != null) {
			dayBoundaryOffset = selectedOrg.getDayBoundaryOffset();
		} else {
			dayBoundaryOffset = selectedCampaign.getDayBoundaryOffset();
		}
	}

	public Collection<Shift> getShifts() throws BbmFinderException {
		Collection<Shift> ret = new ArrayList<Shift>(shifts);
		if (ret.isEmpty()) {
			ret.add(new Shift());
		}

		for (Shift shift : ret) {
			// Construct a shift from the form field
			if (!isMultiEdit() || activitySelectorPC.isEdited()) {
				shift.setActivityID(activitySelectorPC.getSelections().get(0).getID());
			}
			if (isEditingNameAndDesc()) {
				shift.setName(nameInputPC.getInputValue() != null ? nameInputPC.getInputValue().trim() : null);
				shift.setDescription(descInputPC.getInputValue());
				if (isByOrg) {
					shift.setOrganizationID(selectedOrg.getID());
				} else if (!isByOrg) {
					shift.setCampaignID(selectedCampaign.getID());
				}
			}
			if (!isMultiEdit() || durationPC.isEdited()) {
				shift.setDuration(durationPC.getDuration().getDurationInMinutes());
			}

			removeStartTimesFromShift(shift, startTimePickerPC.getRemovedStartTimes());
			addStartTimesToShift(shift, startTimePickerPC.getAddedStartTimes());

			if (!isMultiEdit() || minSpacingPC.isEdited()) {
				shift.setMinSpacing(minSpacingPC.getDuration().getDurationInMinutes());
			}
			if (!isMultiEdit() || maxSpacingPC.isEdited()) {
				shift.setMaxSpacing(maxSpacingPC.getDuration().getDurationInMinutes());
			}
			if (shiftEventListPC.getRemovedRowIDs().length() > 0) {
				removeShiftEventsFromShift(shift, StringUtil.parseIDStringToCollection(shiftEventListPC.getRemovedRowIDs()));
			}
			if (shiftEventListPC.getAddedRowIDs().length() > 0) {
				Collection<ShiftEvent> addedShiftEvents = new ArrayList<ShiftEvent>();
				addedShiftEvents = WorkRulesModelHandler.getShiftEvents(m_context,
						StringUtil.parseIDStringToCollection(shiftEventListPC.getAddedRowIDs()));
				addShiftEventsToShift(shift, addedShiftEvents);
			}
		}
		return ret;
	}

	private void initStartTimes() {
		HashSet<TimeOfDay> startTimes = new HashSet<TimeOfDay>();
		HashSet<TimeOfDay> multiSelectStartTimes = new HashSet<TimeOfDay>();
		for (Shift loopShift : shiftMap.values()) {
			startTimes.addAll(ShiftUtil.getStartTimesOfDay(
					TimeContextFactory.getTimeContext(loopShift, organizationMap, campaignMap), loopShift.getShiftStarts()));
		}
		for (TimeOfDay startTime : startTimes) {
			for (Shift loopShift : shiftMap.values()) {
				if (ShiftUtil.getStartTimesOfDay(TimeContextFactory.getTimeContext(loopShift, organizationMap, campaignMap),
						loopShift.getShiftStarts()).contains(startTime) == false) {
					multiSelectStartTimes.add(startTime);
					break;
				}
			}
		}
		startTimes.removeAll(multiSelectStartTimes);
		startTimePickerPC.setStartTimes(startTimes);
		startTimePickerPC.setMultiSelectedStartTimes(multiSelectStartTimes);
	}

	/**
	 * Returns an array list of all shift events belonging to the selected
	 * shifts. Also sets the multi edit common flag on each shift event so the
	 * shiftEventListPC can render the shift appropriately
	 *
	 * @return
	 */
	private Collection<ShiftEvent> getShiftEventsFromSelectedShifts() {
		if (shifts.size() == 0) {
			return new ArrayList<ShiftEvent>();
		}

		HashMap<ID, ShiftEvent> shiftEventMap = new HashMap<ID, ShiftEvent>();

		// initialize the map
		Shift tempShift = shifts.get(0);
		for (ShiftEvent se : new ArrayList<ShiftEvent>(tempShift.getShiftEvents())) {
			se.setMultiEditCommon(true);
			shiftEventMap.put(se.getID(), se);
		}

		for (Shift shift : shifts) {
			for (ShiftEvent se : shiftEventMap.values()) {
				// if a shiftevent in the map does not belong to the shift in
				// this iteration,
				// we want to set multi edit common to false as it is no longer
				// common between
				// all shifts in the set
				if (se.isMultiEditCommon() && shift.getShiftEvents().contains(se) == false) {
					se.setMultiEditCommon(false);
				}
			}
			for (ShiftEvent se : new ArrayList<ShiftEvent>(shift.getShiftEvents())) {
				if (shiftEventMap.containsKey(se.getID()) == false) {
					se.setMultiEditCommon(false);
					shiftEventMap.put(se.getID(), se);
				}
			}
		}
		return new ArrayList<ShiftEvent>(shiftEventMap.values());
	}

	private void removeStartTimesFromShift(Shift shift, Collection<TimeOfDay> startTimesToRemove) {
		TimeContext timeContext = TimeContextFactory.getTimeContext(shift, organizationMap, campaignMap);
		Collection<ShiftStart> shiftStartsToSet = new HashSet<ShiftStart>(shift.getShiftStarts());
		for (Iterator<ShiftStart> i = shiftStartsToSet.iterator(); i.hasNext(); ) {
			if (startTimesToRemove.contains(new TimeOfDay(i.next().getPossibleStartTime() + timeContext.getDayBoundary()))) {
				i.remove();
			}
		}
		shift.setShiftStarts(shiftStartsToSet);
	}

	private void addStartTimesToShift(Shift shift, Collection<TimeOfDay> startTimesToAdd) {
		TimeContext timeContext = TimeContextFactory.getTimeContext(shift, organizationMap, campaignMap);
		HashSet<TimeOfDay> starts = new HashSet<TimeOfDay>(ShiftUtil.getStartTimesOfDay(timeContext, shift.getShiftStarts()));
		starts.addAll(startTimesToAdd);
		shift.setShiftStarts(ShiftUtil.getShiftStarts(timeContext, shift.getID(), starts));
	}

	private void removeShiftEventsFromShift(Shift shift, Collection<ID> shiftEventIDsToRemove) {
		ArrayList<ShiftEvent> shiftEventsToRemove = new ArrayList<ShiftEvent>();
		for (ShiftEvent se : shift.getShiftEvents()) {
			for (ID idToRemove : shiftEventIDsToRemove) {
				if (se.getID().equals(idToRemove)) {
					shiftEventsToRemove.add(se);
					break;
				}
			}
		}
		shift.removeShiftEvents(shiftEventsToRemove);
	}

	// It's possible that shifts events that are being added might already be
	// associated to a shift.
	// Only add the shift events that are not already associated
	private void addShiftEventsToShift(Shift shift, Collection<ShiftEvent> shiftEvents) {
		ArrayList<ShiftEvent> shiftEventsToAdd = new ArrayList<ShiftEvent>();
		for (ShiftEvent se : shiftEvents) {
			if (shift.getShiftEvents().contains(se) == false) {
				shiftEventsToAdd.add(se);
			}
		}
		shift.addShiftEvents(shiftEventsToAdd);
	}

	private String i(String key) {
		return m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, key);
	}

	private int getMaxDuration(Collection<Integer> durations) {
		int maxDuration = 0;
		for (int dur : durations) {
			if (dur > maxDuration) {
				maxDuration = dur;
			}
		}
		return maxDuration;
	}

	private boolean isEditingNameAndDesc() {
		return !isMultiEdit();
	}

	@Override
	public boolean validate() {
		boolean retVal = super.validate();

		Collection<Shift> editedShifts = null;
		HashMap<ID, String> allShiftNames = null;
		try {
			if (isByOrg) {
				allShiftNames = WorkRulesModelHandler.getShiftNamesByOrg(m_context, selectedOrg.getID());
			} else {
				allShiftNames = WorkRulesModelHandler.getShiftNamesBySchedulingPeriod(m_context, selectedSchedulingPeriod);
			}
			editedShifts = getShifts();
		} catch (BbmFinderException e1) {
			handleUnableToLoadDataError(log, e1);
			return false;
		}

		Map<String, TimeZone> shiftTimeZones = ShiftUtil.getTimeZoneForShifts(editedShifts, organizationMap, campaignMap);
		Map<String, TimeZone> shiftEventTimeZones = ShiftUtil.getTimeZoneForShiftEventsInShifts(editedShifts, organizationMap,
				campaignMap);

		for (Shift shift : editedShifts) {
			// Name must be unique
			if (isEditingNameAndDesc()) {
				String name = shift.getName();
				for (ID existingShiftID : allShiftNames.keySet()) {
					if (existingShiftID.equals(shift.getID())) {
						continue;
					}
					if (name.equals(allShiftNames.get(existingShiftID))) {
						addPageMessage(Message.ERROR_TYPE,
								m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_NAME_MUST_BE_UNIQUE));
						retVal = false;
					}
				}
			}

			// Max spacing must be > min spacing (0 max means unlimited)
			if (shift.getMaxSpacing() != 0) {
				if (shift.getMinSpacing() > shift.getMaxSpacing()) {
					addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
							FsWebBundleKey.MAX_SPACING_MUST_BE_GREATER_THAN_MIN_SPACING));
					retVal = false;
				}
			}

			// Cannot have a shift event with the same activity as the shift
			ID activityID = shift.getActivityID();
			for (ShiftEvent se : shift.getShiftEvents()) {
				if (activityID.equals(se.getActivityID())) {
					addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
							FsWebBundleKey.FS_SHIFT_EVENT_ACTIVITY_MUST_BE_DIFFERENT_FROM_SHIFT));
					retVal = false;
				}
			}

			// Must have a valid start time
			boolean validStartTimeFound = false;
			int earliestStart = startTimePickerPC.getEarliestStartTime();
			int latestEnd = startTimePickerPC.getLatestEndTime();
			boolean is24Hr = (latestEnd - earliestStart == 1440);
			for (ShiftStart ss : shift.getShiftStarts()) {
				// If we have a "24 hour" organization/campaign (where you have at least two adjacent
				// days where the end time of one day is the same time as the start time of the subsequent
				// day in the week and the org allows shifts to cross the day boundary, then all start times are valid.
				if (canShiftAssignmentsCrossDayBoundary && is24Hr) {
					validStartTimeFound = true;
					break;
				}
				if ((ss.getPossibleStartTime() + dayBoundaryOffset >= earliestStart)
						&& (ss.getPossibleStartTime() + dayBoundaryOffset + shift.getDuration() <= latestEnd)) {
					validStartTimeFound = true;
					break;
				}
			}
			if (!validStartTimeFound) {
				addPageMessage(Message.ERROR_TYPE,
						m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MUST_HAVE_START_TIME));
				retVal = false;
			}

			// Check for valid shift event placement.
			// Both primary and secondary shift events need to be checked separately
			// A work type shift event is one whose activity has been marked as "can be used in shift"

			//Validate work type events
			ArrayList<Message> validationResults = ShiftUtil.validateShiftEventAssignments(m_localizer, dayBoundaryOffset,
					shift, shift.getWorkTypeShiftEvents(), true, shiftTimeZones, shiftEventTimeZones);
			for (Message m : validationResults) {
				addPageMessage(m);
			}
			if (validationResults.size() > 0) {
				retVal = false;
			}

			//Validate break type events
			validationResults = ShiftUtil.validateShiftEventAssignments(m_localizer, dayBoundaryOffset, shift,
					shift.getBreakTypeShiftEvents(), false, shiftTimeZones, shiftEventTimeZones);
			for (Message m : validationResults) {
				addPageMessage(m);
			}
			if (validationResults.size() > 0) {
				retVal = false;
			}
		}
		return retVal;
	}

	class ShiftMaxSpacingPC extends CompositePC {

		DurationSpinnerPC durationPC;
		CheckBoxPC unlimitedCheckBoxPC;

		private boolean isMultiEdit = false;
		private boolean isEdited = false;

		ShiftMaxSpacingPC(RequestContext context, String name) {
			super(context);
			setName(name);

			addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
			setRenderJavaScriptIfDisabled(true);
			durationPC = new DurationSpinnerPC(context, getDurationName());
			addChildComponent(durationPC);
			unlimitedCheckBoxPC = new CheckBoxPC(m_context, getUnlimitedName(), false, getDurationName()
					+ ".setEnabled(false);", getDurationName() + ".setEnabled(true);");
			addChildComponent(unlimitedCheckBoxPC);
		}

		public void setMultiEdit(boolean isMultiEdit) {
			this.isMultiEdit = isMultiEdit;
		}

		@Override
		public String getJavaScriptInitCode() {
			String handlerName = getName() + ".OnChange";
			return super.getJavaScriptInitCode() + "\n" + handlerName + " = function() { jQuery('input[name="
					+ getEditedName() + "]').val('true'); };" + "\n" + getUnlimitedName() + ".change(function() {"
					+ handlerName + "();});" + "\n" + getDurationName() + ".change(function() {" + handlerName + "();});";
		}

		@Override
		public String getHtmlDisplay() {
			return BPMessageFormat.format(
					"{0}{1}{2}&nbsp;&nbsp;{3}<input type='hidden' name=\"{4}\" value=\"{5}\">",
					new Object[]{
							(isMultiEdit ? "<span class='multiEditIndicator' style='padding-right: 2px;'>*</span>" : ""),
							unlimitedCheckBoxPC.getHtmlDisplay(), i(FsWebBundleKey.FS_UNLIMITED),
							durationPC.getHtmlDisplay(), getEditedName(), isEdited});
		}

		private String getDurationName() {
			return getName() + "__duration";
		}

		private String getUnlimitedName() {
			return getName() + "__unlimited";
		}

		private String getEditedName() {
			return getName() + "__edited";
		}

		@Override
		public boolean extractParameters(HttpServletRequest request) {
			if (!isAutoExtractParamEnabled()) {
				return true;
			}
			String editedStr = request.getParameter(getEditedName());
			if (editedStr != null) {
				isEdited = new Boolean(editedStr);
			}
			return super.extractParameters(request);
		}

		public boolean isEdited() {
			return isEdited;
		}

		public Duration getDuration() {
			if (unlimitedCheckBoxPC.isChecked()) {
				return Duration.fromMinutes(0);
			}
			return durationPC.getDuration();
		}

		public void setDuration(Duration duration) {
			if (duration.getDurationInMinutes() == 0) {
				unlimitedCheckBoxPC.setIsChecked(true);
				durationPC.setEnabled(false);
			} else {
				unlimitedCheckBoxPC.setIsChecked(false);
			}
			durationPC.setValue(duration);
		}

		public DurationSpinnerPC getDurationSpinnerPC() {
			return durationPC;
		}

		public CheckBoxPC getCheckBoxPC() {
			return unlimitedCheckBoxPC;
		}
	}
