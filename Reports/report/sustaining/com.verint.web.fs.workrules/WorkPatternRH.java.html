package com.verint.web.fs.workrules;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.workrules.ejb.WorkRuleExcessiveNameLengthException;
import com.bluepumpkin.ejb.bbm.workrules.ejb.WorkRuleManager;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.l10n.BbmWebLogBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

import java.util.Collection;

public class WorkPatternRH extends WorkRulesRH<WorkPatternListPM, WorkPatternFormPM> {
	private static final Category log = Log.initCategory(WorkPatternRH.class.getName());

	@Override
	protected WorkPatternFormPM getFormPageModel(RequestContext context) {
		return WorkPatternFormPM.getInstance(context);
	}

	@Override
	protected WorkPatternListPM getListPageModel(RequestContext context) {
		WorkPatternListPM pm = new WorkPatternListPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
	}

	@Override
	public void processRequest(RequestContext context) throws Exception {
		String pageAction = getPageAction(context);
		//Remove Session Item Index when pageAction is null or refresh or is created new
		if (pageAction == null || pageAction.equals(PageAction.NEW_ID) || pageAction.equals(PageAction.REFRESH)) {
			String strItemIndex = (String) context.getAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX);
			if (!StringUtil.isEmpty(strItemIndex)) {
				context.removeAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX);
			}
		} else {
			super.processRequest(context);
			//If the user changed the selected shift pattern period, we will catch it here and then
			//refresh the form page.
			if (isAction(context, WorkPatternPeriodDropDownPC.WORK_PATTERN_PERIOD_CHANGED_ACTION)) {
				WorkPatternFormPM pm = getFormPageModel(context);
				context.setPageModel(pm);
				setNextPage(context, FORM_PAGE);
			}
		}

	}

	@Override
	protected void processCopy(RequestContext context, WorkPatternListPM pageModel)
			throws BPException {
		try {
			if (pageModel.isByOrg) {
				WorkRulesModelHandler.copyShiftPatterns(context, pageModel.getPerfromActionOnIDsAsCollection(),
						pageModel.isByOrg, pageModel.getSelectedIDObject(), pageModel.getLocalizer().getLocaleContext());
			} else {
				pageModel.loadData();      //we need to do this to initialize the campaign/sp value objects
				WorkRulesModelHandler.copyShiftPatterns(context, pageModel.getPerfromActionOnIDsAsCollection(),
						pageModel.isByOrg, pageModel.selectedCampaign.getDEID(),
						pageModel.getLocalizer().getLocaleContext());
			}
			context.setPageModel(null);
			pageModel = getListPageModel(context);    //reload the new shifts
			pageModel.addPageMessage(Message.RESPONSE_TYPE,
					FsWebBundleKey.FS_COPY_WORK_PATTERNS_SUCCESS, FsWebBundleKey.BUNDLE_NAME);
		} catch (WorkRuleExcessiveNameLengthException ex) {
			String localizedErrorMsg = context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_WORK_PATTERN_NAME_TOO_LONG, ex.getNameOfViolatingWorkRule());
			pageModel.addPageMessage(Message.ERROR_TYPE, localizedErrorMsg);
			throw new BPException(ex);
		} catch (Exception ex) {
			pageModel.addPageMessage(Message.ERROR_TYPE,
					FsWebBundleKey.FS_UNABLE_TO_COPY_WORK_PATTERNS, FsWebBundleKey.BUNDLE_NAME);
			throw new BPException(ex);
		} finally {
			context.setPageModel(pageModel);
		}
	}

	@Override
	protected void processDelete(RequestContext context,
			WorkPatternListPM pageModel) throws BPException {
		ScheduleAccessManager sam = WfmManagerFactory.getScheduleAccessManager(context.isInWhatIfMode());
		WorkRuleManager wrm = WfmManagerFactory.getWorkRuleManager(context.isInWhatIfMode());

		try {
			Collection profilesUsingPattern =
					sam.getEmployeeTemplatesUsingShiftPattern(pageModel.getPerfromActionOnIDsAsCollection());
			Collection<String> rotationNames = wrm.getRotationRulesNameByShiftPatternIDs(pageModel.getPerfromActionOnIDsAsCollection());

			if (profilesUsingPattern.size() == 0 && rotationNames.size() == 0) {
				WorkRulesModelHandler.deleteShiftPatterns(context, pageModel.getPerfromActionOnIDsAsCollection());
				context.setPageModel(null); // reset the page model to remove the deleted entries
			} else {
				// Create staffing profile list, just show 3 or 4 and then append " and x others"
				StringBuilder sb = new StringBuilder();
				StringBuilder rotaionSB = new StringBuilder();

				int iDisplayLimit = 3;  // we'll only show the first three names
				int iRotationNameLimit = 10; //we'll only show the first 10 names;
				int iCount = 1;

				for (Object entry : profilesUsingPattern) {
					if (iCount == iDisplayLimit) {
						break;
					}
					com.bluepumpkin.ejb.bbm.workresource.model.EmployeeTemplate template =
							(com.bluepumpkin.ejb.bbm.workresource.model.EmployeeTemplate) entry;
					if (sb.length() > 0) {
						sb.append(", ");
					}
					sb.append(template.getName());
					iCount++;
				}

				iCount = 1;
				for (Object rotationName : rotationNames) {
					if (iCount > iRotationNameLimit) {
						break;
					}
					if (rotaionSB.length() > 0) {
						rotaionSB.append(", ");
					}
					rotaionSB.append((String) rotationName);
					iCount++;
				}

				//Shift pattern is used in rotation rules
				if (rotationNames.size() > iRotationNameLimit) {
					rotaionSB.append(context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
							FsWebBundleKey.FS_WORK_PATTERN_IN_USE_BY_ROTATION_RULES_LIST,
							String.valueOf(rotationNames.size() - iRotationNameLimit)));
				}

				if (rotationNames.size() > 0) {
					pageModel.addPageMessage(Message.ERROR_TYPE,
							FsWebBundleKey.FS_WORK_PATTERN_IN_USE_BY_ROTATION_RULES, FsWebBundleKey.BUNDLE_NAME,
							new String[]{rotaionSB.toString()});
				}

				//Shift pattern is used in staffing profile
				if (profilesUsingPattern.size() > iDisplayLimit) {
					sb.append(context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
							FsWebBundleKey.FS_WORK_PATTERN_IN_USE_BY_PROFILES_LIST,
							String.valueOf(profilesUsingPattern.size() - iDisplayLimit)));
				}

				if (profilesUsingPattern.size() > 0) {
					pageModel.addPageMessage(Message.ERROR_TYPE,
							FsWebBundleKey.FS_WORK_PATTERN_IN_USE_BY_PROFILES, FsWebBundleKey.BUNDLE_NAME,
							new String[]{sb.toString()});
				}

				context.setPageModel(pageModel);
			}
		} catch (Exception ex) {
			pageModel.addPageMessage(Message.ERROR_TYPE,
					FsWebBundleKey.FS_WORK_PATTERN_DELETE_ERROR, FsWebBundleKey.BUNDLE_NAME);
			throw new BPException(ex);
		}
	}

	@Override
	protected void processExport(RequestContext context,
			WorkPatternListPM pageModel) throws BPException {
		WorkPatternExportPopupPM model = new WorkPatternExportPopupPM(context);
		model.extractParameters(context.getRequest());
		model.processExport(context);
		context.setPageModel(model);
	}

	@Override
	protected void processImport(RequestContext context,
			WorkPatternListPM pageModel) throws BPException {
		try {
			WorkPatternImportPopupPM model = new WorkPatternImportPopupPM(context);
			model.extractParameters(context.getRequest());
			model.processImport(context);

			if (model.getPageAction() != null && model.getPageAction().equals(ShiftImportPopupPM.SUCCESS)) {
				model.setShowDoneButton(true);
			}
			context.setPageModel(model);
		} catch (Exception ex) {
			log.l7dError(BbmWebLogBundleKey.UNABLE_TO_UPLOAD_DATA_FROM_FILE, ex);
		}
	}

	@Override
	protected void processSave(RequestContext context, WorkPatternFormPM pageModel)
			throws BPException {
		try {
			if (pageModel.validate()) {
				if (pageModel.isNew()) {
					WorkRulesModelHandler.createShiftPattern(context, pageModel.getShiftPatterns().iterator().next());
				} else {
					WorkRulesModelHandler.updateShiftPatterns(context, pageModel.getShiftPatterns());
				}
			} else {
				setNextPage(context, "formPage");
			}
		} catch (BbmException e) {
			setNextPage(context, "formPage");
			throw e;
		}
	}

	@Override
	protected void processDownload(RequestContext context,
			WorkPatternListPM pageModel) throws BPException {
		try {
			WorkPatternExportPopupPM model = new WorkPatternExportPopupPM(context);
			model.extractParameters(context.getRequest());
			boolean isSuccess = model.processExport(context);

			model.addPageMessage(Message.INFO_TYPE, "Export completed successfully!");
			context.setPageModel(model);

			if (!isSuccess) {
				setNextPage(context, "exportPage");
			}
		} catch (Exception ex) {
			log.l7dError(BbmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}
	}

	@Override
	protected void processFind(RequestContext context, WorkPatternListPM pageModel) throws BPException {
		pageModel.extractParameters(context.getRequest());
		context.setPageModel(pageModel);
	}

	@Override
	protected void processTableSort(RequestContext context, WorkPatternListPM pageModel) {
		String sortColumn = context.getRequest().getParameter("sortColumn");
		String sortOrder = context.getRequest().getParameter("sortOrder");
		context.setUserProperty(WorkPatternListPM.WORKPATTERN_LIST_SORTCOLUMN, sortColumn, true);
		context.setUserProperty(WorkPatternListPM.WORKPATTERN_LIST_SORTORDER, sortOrder, true);
	}

	@Override
	protected void processPageLeft(RequestContext context, WorkPatternListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexBackward();
		String strItemIndex =
				(String) context.getAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex)) {
			context.removeAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX);
		}
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX, Integer.toString(itemIndex));
	}

	@Override
	protected void processPageRight(RequestContext context, WorkPatternListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);
		pageModel.adjustEditIndexForward();
		String strItemIndex =
				(String) context.getAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex)) {
			context.removeAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX);
		}
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX, Integer.toString(itemIndex));
	}

	@Override
	protected void processPageSelect(RequestContext context, WorkPatternListPM pageModel) throws BPException {
		loadIDSelection(context, pageModel);

		pageModel.setPagePaginationAndEditIndices(
				pageModel.getPaginationControl().getSelectedPaginationPageIndex());
		String strItemIndex =
				(String) context.getAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX);
		if (!StringUtil.isEmpty(strItemIndex)) {
			context.removeAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX);
		}
		int itemIndex = pageModel.getEditIndex();
		context.setAttribute(RequestContext.SESSION_SCOPE, WorkPatternListPM.WORKPATTERNLIST_ITEMINDEX, Integer.toString(itemIndex));
	}
}
