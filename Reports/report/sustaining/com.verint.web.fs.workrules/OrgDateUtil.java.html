package com.verint.web.fs.workrules;

import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.witness.web.uif.pagecomponent.picker.date.DatePickerPC;

import java.util.Calendar;
import java.util.Date;
import java.util.HashSet;

public class OrgDateUtil {
	/**
	 * Utility method to set the disabled days on the date picker such that the user must pick the days of the week
	 * corresponding to the org's week start.
	 * <p>
	 * This method will also set the time zone on the DatePickerPC to that of the Org.
	 *
	 * @param datePicker
	 * @param o
	 * @param defaultDate
	 */
	public static void restrictDatePickerToOrgStartWeek(DatePickerPC datePicker, Organization o, LocalDate defaultDate) {
		int weekStart = o.getWeekStartDate();
		datePicker.setFirstDayOfWeek(weekStart);
		HashSet<Integer> disabledDaysCol = new HashSet<Integer>();
		disabledDaysCol.add(Calendar.SUNDAY);
		disabledDaysCol.add(Calendar.MONDAY);
		disabledDaysCol.add(Calendar.TUESDAY);
		disabledDaysCol.add(Calendar.WEDNESDAY);
		disabledDaysCol.add(Calendar.THURSDAY);
		disabledDaysCol.add(Calendar.FRIDAY);
		disabledDaysCol.add(Calendar.SATURDAY);
		disabledDaysCol.remove(weekStart);
		int[] disabledDays = new int[disabledDaysCol.size()];
		int ctr = 0;
		for (int day : disabledDaysCol) {
			disabledDays[ctr++] = day;
		}
		datePicker.setDisabledDays(disabledDays);
		if (defaultDate == null) {
			defaultDate = new LocalDate(new Date(), o.getTimeZone());
			defaultDate.set(Calendar.HOUR_OF_DAY, 0);
			defaultDate.set(Calendar.MINUTE, 0);
			defaultDate.set(Calendar.SECOND, 0);
			defaultDate.set(Calendar.MILLISECOND, 0);
			defaultDate.set(Calendar.DAY_OF_WEEK, weekStart);
		}
		datePicker.setTimeZone(o.getTimeZone());
		datePicker.setDate(defaultDate);
	}

	/**
	 * Returns a new date that is aligned to the organization's week start. This method
	 * sets the day of week, hour of day, and minute values.
	 *
	 * @param org
	 * @param date
	 * @return
	 */
	public static Date snapDateToWeekStart(Organization org, Date date) {
		Calendar c = Calendar.getInstance(org.getTimeZone());
		c.setTime(date);
		c.set(Calendar.DAY_OF_WEEK, org.getWeekStartDay().getCalendarConstant());
		c.set(Calendar.HOUR_OF_DAY, org.getDayBoundaryOffset() / 60);
		c.set(Calendar.MINUTE, org.getDayBoundaryOffset() % 60);
		c.set(Calendar.SECOND, 0);
		c.set(Calendar.MILLISECOND, 0);

		return c.getTime();
	}
}
