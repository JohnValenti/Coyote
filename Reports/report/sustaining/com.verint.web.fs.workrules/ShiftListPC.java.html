	public ShiftListPC(RequestContext context, String name, Collection<Shift> shifts,
			TimeZone optimizationTargetTimezone, Date timeZoneEffectiveDate, Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap,
			Map<ID, Activity> activityMap, boolean isUsedInList) {
		super(context);
		setName(name);
		setFirstColumnWidth("2.5%");

		if (optimizationTargetTimezone != null) {
			this.optimizationTargetTimeZone = optimizationTargetTimezone;
		} else {
			this.optimizationTargetTimeZone = getRequestContext().getViewingTimeZone();
		}
		this.timeZoneEffectiveDate = timeZoneEffectiveDate;
		this.orgMap = orgMap;
		this.campaignMap = campaignMap;
		this.activityMap = activityMap;

		toggleAll = new ToggleVisiblePC(m_context, "toggleAll");
		this.addChildComponent(toggleAll);

		initRows(shifts);
		m_isUsedInList = isUsedInList; //Must be assign before initHeader
		initHeader();

		addRequiredJavaScriptFile(WorkRulesJSFileID.WORK_RULES_UTIL);
		addPopupArgs(ActivityPopupPM.POPUP_ID, ActivityPopupPM.FORM_ACTION, "",
				ActivityPopupPM.POPUP_ARGS, "1000,600,ctr,ctr", true, 1);

	}

	protected void initHeader() {
		TableHeaderPC header = getHeader();
		header.setPartialSortable(true);
		header.addColumnHeaderData("toggleAll", toggleAll.getHtmlDisplay(), false);
		header.addColumnHeaderData(SHIFT_NAME_HEADER, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_SHIFT), true);
		DefaultHeaderData dhd = new DefaultHeaderData("shiftActivity",
				getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ACTIVITY));
		dhd.addImageButton(getActivityPopupImage(), getActivityPopupCaption(), getActivityPopupJS());
		dhd.setSortable(true);
		header.addColumnHeaderData(dhd);
		header.addColumnHeaderData(SHIFT_LENGTH_HEADER, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_DURATION), true);
		header.addColumnHeaderData(SHIFT_STARTTIME, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_START_TIMES), true);
		header.addColumnHeaderData(SHIFT_MIN_SPACING, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MIN_SPACING), true);
		header.addColumnHeaderData(SHIFT_MAX_SPACING, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MAX_SPACING), true);
		header.addColumnHeaderData(SHIFT_ORGANIZATION, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ORGANIZATION), true);
		header.addColumnHeaderData(SHIFT_DESCRIPTION, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_DESCRIPTION), true);
		setPersitedSortHeader(m_isUsedInList, header);
	}

	private void initRows(Collection<Shift> shifts) {
		Collection<MultiColumnNodeData> rows = super.getListModel();

		if (shifts != null) {
			for (Shift shift : shifts) {
				String detailRowID = getName() + "__detail" + getJSSuffix(shift);
				DefaultMultiColumnNodeData mcnd = new DefaultMultiColumnNodeData(shift.getID());
				boolean hasShiftEvents = shift.getShiftEvents().size() > 0;
				if (!hasShiftEvents) {
					mcnd.add(" ");
				} else {
					// Toggle
					ToggleVisiblePC toggle = new ToggleVisiblePC(m_context, "toggle" + getJSSuffix(shift), Collections.singleton(detailRowID));
					toggleAll.addChildComponent(toggle);
					mcnd.add(toggle.getHtmlDisplay());
				}
				// Name
				mcnd.add(WorkRulesUtil.getEscapedStringRepresentation(shift.getName()));
				// Activity
				Activity a = activityMap.get(shift.getActivityID());
				mcnd.add(a == null ? "" : a.getName());
				//length
				mcnd.add(m_localizer.formatDurationInMins(shift.getDuration()));
				//Start Times
				TimeContext shiftTimeContext = TimeContextFactory.getTimeContext(shift, orgMap, campaignMap);
				mcnd.add(new StartTimes(
						m_context.getLocalizer(),
						WorkRulesUtil.getTotalTimeZoneOffsetForWorkRuleDisplay(m_context,
								optimizationTargetTimeZone, shiftTimeContext.getTimeZone(), timeZoneEffectiveDate),
						ShiftUtil.getStartTimesOfDay(
								shiftTimeContext,
								shift.getShiftStarts())));
				//Min Spacing
				mcnd.add(m_localizer.formatDurationInMins(shift.getMinSpacing()));
				// Max Spacing
				mcnd.add(m_localizer.formatDurationInMins(shift.getMaxSpacing()));
				//Org or Campaign
				Organization o = orgMap.get(shift.getOrganizationID());
				mcnd.add(o == null ? m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.LOCAL) : WorkRulesUtil.getEscapedStringRepresentation(o.getName()));
				//description
				mcnd.add(WorkRulesUtil.getEscapedStringRepresentation(shift.getDescription()));

				setRowAttributes(mcnd, shift);

				if (!hasShiftEvents) {
					rows.add(mcnd);
				} else {
					// Shift Events
					DefaultMultiColumnNodeData mcndDetail = new DefaultMultiColumnNodeData(shift.getID());
					setDetailRowAttributes(mcnd, mcndDetail, shift);
					ShiftEventListPC mclpc = new ShiftEventListPC(m_context,
							"shift" + getJSSuffix(shift) + "_events", shift.getShiftEvents(),
							optimizationTargetTimeZone, timeZoneEffectiveDate, orgMap, campaignMap, activityMap, true);
					mclpc.setIsUserSortable(false);
					mclpc.setIsBorderEnabled(true);
					mclpc.setRowSelectable(false);
					addChildComponent(mclpc);
					HtmlTableRow row = new HtmlTableRow();
					row.add("<td colspan='9'>" + mclpc + "</td>"); // Hard code the colspan for now, since we init selectable items before the header because of the toggles
					mcndDetail.setNodeAttribute("id", detailRowID);
					mcndDetail.setNodeAttribute("masterID", shift.getID().toString());
					mcndDetail.setNodeAttribute("style", "display: none;");
					mcndDetail.setNodeAttribute("multiSelectGroup", "detail");
					mcndDetail.add(row);

					MasterDetailRow mdr = new MasterDetailRow(mcnd, mcndDetail);
					rows.add(mdr);
				}
			}
		}
	}

	private String getJSSuffix(Shift shift) {
		return shift.getID().toString().replace('-', 'n');
	}

	public LinkedHashMap<DBSortField, Comparator> getSortComparators() {
		LinkedHashMap<DBSortField, Comparator> comparators =
				new LinkedHashMap<DBSortField, Comparator>();
		for (ShiftSortField sortField : getDBSortFieldIndices()) {
			switch (sortField) {
				case ID:
					comparators.put(ShiftSortField.ID, null);
					break;
				case NAME:
					comparators.put(ShiftSortField.NAME, null);
					break;
				case ACTIVITY:
					comparators.put(ShiftSortField.ACTIVITY, null);
					break;
				case DURATION:
					comparators.put(ShiftSortField.DURATION, null);
					break;
				case START_TIME:
					comparators.put(ShiftSortField.START_TIME, null);
					break;
				case MIN_SPACING:
					comparators.put(ShiftSortField.MIN_SPACING, null);
					break;
				case MAX_SPACING:
					comparators.put(ShiftSortField.MAX_SPACING, null);
					break;
				case ORGANIZATION:
					comparators.put(ShiftSortField.ORGANIZATION, new OrganizationCampaignNameComparator(getLocalizer()));
					break;
				case DESCRIPTION:
					comparators.put(ShiftSortField.DESCRIPTION, null);
					break;
			}
		}

		return comparators;
	}

	public ArrayList<ShiftSortField> getDBSortFieldIndices() {
		ArrayList<ShiftSortField> retVal = new ArrayList<ShiftSortField>();
		TableHeaderPC header = getHeader();
		setPersitedSortHeader(m_isUsedInList, header);
		if (header.getSortColumnIndexes() == null) {
			//Default sort index is by ID (chronological order in which shift event was created)
			retVal.add(ShiftSortField.ID);
		} else {
			for (int sortField : header.getSortColumnIndexes()) {
				switch (sortField) {
					case 1:
						retVal.add(ShiftSortField.NAME);
						break;
					case 2:
						retVal.add(ShiftSortField.ACTIVITY);
						break;
					case 3:
						retVal.add(ShiftSortField.DURATION);
						break;
					case 4:
						retVal.add(ShiftSortField.START_TIME);
						break;
					case 5:
						retVal.add(ShiftSortField.MIN_SPACING);
						break;
					case 6:
						retVal.add(ShiftSortField.MAX_SPACING);
						break;
					case 7:
						retVal.add(ShiftSortField.ORGANIZATION);
						break;
					case 8:
						retVal.add(ShiftSortField.DESCRIPTION);
						break;
				}
			}
		}
		return retVal;
	}

	private void setPersitedSortHeader(boolean isUsedInList, TableHeaderPC header) {
		String sortColumn = null;
		String sortOrder = null;
		if (isUsedInList) {
			sortColumn = m_context.getUserProperty(SHIFT_LIST_SORTCOLUMN);
			sortOrder = m_context.getUserProperty(SHIFT_LIST_SORTORDER);
		}
		if (sortColumn != null) {
			header.setSortColumn(sortColumn);
		}
		if (sortOrder != null) {
			header.setSortOrder(sortOrder);
		}
	}
