package com.verint.web.fs.workrules;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.bbm.campaign.selection.CampaignSchedulingPeriodSelectionUtil;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.pagecomponent.picker.time.TimeZoneOverrideTogglePC;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.pagemodel.WorkpaneFormPM;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.TimeZoneOverrideUtil;

import java.util.Date;
import java.util.ResourceBundle;
import java.util.TimeZone;

@SuppressWarnings("serial")
public abstract class WorkRulesFormPM extends WorkpaneFormPM {
	protected final ResourceBundle m_bundle;
	protected boolean isByOrg = true;
	protected Organization selectedOrg = null;
	protected Campaign selectedCampaign = null;
	protected SchedulingPeriod selectedSchedulingPeriod = null;
	protected TimeZone optimizationTargetTimeZone;
	protected Date timeZoneEffectiveDate = new Date();
	private String m_selectedCampaignID;
	private String m_selectedSchedulingPeriodID;

	public WorkRulesFormPM(RequestContext context) {
		super(context, PageModel.FRAMED_MODEL);
		setRenderJavaScriptIfDisabled(true);
		optimizationTargetTimeZone = context.getViewingTimeZone();
		m_bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);
	}

	protected boolean isMultiEdit() {
		return getPerformActionOnIDSize() > 1;
	}

	protected boolean isCreateNew() {
		return getPerformActionOnIDSize() == 0;
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();

		// Add read-only time zone information to page header
		if (optimizationTargetTimeZone != null) {
			m_contentTitlePC.addActiveComponent(new TimeZoneOverrideTogglePC(m_context, m_toolbar, optimizationTargetTimeZone, true));
		}
	}

	@Override
	protected void loadData() throws Exception {
		super.loadData();
		ID selectedID = getSelectedIDObject();
		if (selectedID != null) {
			if (isByOrg) {
				selectedOrg = OrganizationModelHandler.getOrganization(m_context, selectedID);
				optimizationTargetTimeZone = selectedOrg.getTimeZone();
				timeZoneEffectiveDate = new Date();
			} else {
				//Campaign mode : retrieve the scheduling period and campaign objects from the selected IDs
				if (!StringUtil.isEmpty(getSelectedSchedulingPeriodID())) {
					selectedSchedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
							m_context, new ID(getSelectedSchedulingPeriodID()));
					timeZoneEffectiveDate = selectedSchedulingPeriod.getStartTime();
				}
				if (!StringUtil.isEmpty(getSelectedCampaignID())) {
					selectedCampaign = CampaignModelHandler.getCampaign(m_context, new ID(getSelectedCampaignID()));
					optimizationTargetTimeZone = selectedCampaign.getTimeZone();
				}
			}
		}
	}

	protected int getTimeZoneOffset() {
		return TimeZoneOverrideUtil.getTimeZoneOffset(m_context, optimizationTargetTimeZone, new Date());
	}

	//Overriding this so that cached IDs are set properly
	@Override
	public void setSelectedID(String ids) {
		super.setSelectedID(ids);
		if (!isByOrg) {
			setSelectedCampaignID(CampaignSchedulingPeriodSelectionUtil.extractIDs(
					getSelectedID(), CampaignSchedulingPeriodSelectionUtil.CAMPAIGN_ID_PREFIX));
			setSelectedSchedulingPeriodID(CampaignSchedulingPeriodSelectionUtil.extractIDs(
					getSelectedID(), CampaignSchedulingPeriodSelectionUtil.SP_ID_PREFIX));
		}
	}

	public String getSelectedCampaignID() {
		return m_selectedCampaignID;
	}

	public void setSelectedCampaignID(String selectedCampaignID) {
		m_selectedCampaignID = selectedCampaignID;
	}

	public String getSelectedSchedulingPeriodID() {
		return m_selectedSchedulingPeriodID;
	}

	public void setSelectedSchedulingPeriodID(String selectedSchedulingPeriodID) {
		m_selectedSchedulingPeriodID = selectedSchedulingPeriodID;
	}
}
