	@Override
	protected void loadData() throws Exception {
		super.loadData();
		WorkRuleManager wrm = WfmManagerFactory.getWorkRuleManager(m_context
				.isInWhatIfMode());

		if (getPerformActionOnIDSize() > 0) {
			projectRule = new ArrayList<>(
					wrm.getProjectRulesByID(Arrays
							.asList(getPerformActionOnIDs())));
		}

		Collection<ID> orgIDs = new HashSet<>();
		Collection<ID> campaignIDs = new HashSet<>();

		for (ProjectRules lProjRule : projectRule) {
			projectRuleMap.put(lProjRule.getID(), lProjRule);

			if (lProjRule.getActivityID() != null) {
				activityIDs.add(lProjRule.getActivityID());
			}

			if (lProjRule.getOrganizationID() != null) {
				orgIDs.add(lProjRule.getOrganizationID());
			}

			if (!isByOrg && lProjRule.getCampaignID() != null) {
				campaignIDs.add(lProjRule.getCampaignID());
			}
		}

		if (isByOrg) {
			if (selectedOrg != null) {
				orgIDs.add(selectedOrg.getID());
			}
		} else if (selectedCampaign != null) {
			campaignIDs.add(selectedCampaign.getID());
		}

		organizationMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!isByOrg) {
			campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		}

		activityIDs.addAll(shiftEventActivityIDs);
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);

		if (isByOrg) {
			if (selectedOrg != null) {
				optimizationTargetTimeZone = selectedOrg.getTimeZone();
			}
		} else {
			if (selectedCampaign != null) {
				optimizationTargetTimeZone = selectedCampaign.getTimeZone();
			}
		}

		if (optimizationTargetTimeZone == null) {
			optimizationTargetTimeZone = m_context.getViewingTimeZone(); // Shouldn't happen, but good to have a default
		}

		if (selectedOrg != null) {
			dayBoundaryOffset = selectedOrg.getDayBoundaryOffset();
		} else {

		}
	}

	private void initStartTimes(ProjectRules objProjRule,
			FormTwoColumnPC objaFormPC) throws BbmException, RemoteException {

		Map<WeekDay, Set<TimeOfDay>> objMapTod;
		int dayOffset = 0;

		int weekStartDay = 0;

		Collection<OrganizationHOO> orgHOO;
		Collection<CampaignHOO> campaignHOO;
		Collection refHOO;  //FIXME: This collection could be either of type OrganizationHOO or CampaignHOO depending on value of isByOrg

		FormTwoColumnPC stFormPC = new FormTwoColumnPC(m_context);
		stFormPC.setIsBorderEnabled(true);
		stFormPC.setTitle(localizeIt(FsWebBundleKey.FS_PROJECT_STARTTIMES_FORM_TITLE));
		// TODO
		/**
		 * Make a call to the Project util to get the Start times for the given
		 * Project Rule grouped by Week While iterating through the loop, get
		 * the Operating hours for each weekday set them to the StartTimePC
		 *
		 * NOTE: Should see how it effects as this method is being called only
		 * when the action is type edit
		 */

		// Initializing the dayOffset value
		if (isByOrg && selectedOrg != null) {
			dayOffset = selectedOrg.getDayBoundaryOffset();
		} else if (!isByOrg && selectedCampaign != null) {
			dayOffset = selectedCampaign.getWeekStart() % 1440;
		}

		// Set invalid start times based on the HOO for the given Campaign or
		// Organization
		if (isByOrg) {
			// Dates are parameters but aren't actually used by the function
			// strangely enough. Just pass in any date
			orgHOO = OrganizationModelHandler.getOrganizationHOOAssignments(
					getRequestContext(), selectedOrg.getID(), new Date(),
					new Date());

			weekStartDay = selectedOrg.getWeekStartDay().get0SundayCount();

			refHOO = orgHOO;
		} else {
			campaignHOO = CampaignModelHandler.getCampaignHOOAssignments(
					getRequestContext(), selectedCampaign.getID(),
					selectedSchedulingPeriod.getStartTime(),
					selectedSchedulingPeriod.getEndTime());
			refHOO = campaignHOO;

			weekStartDay = selectedCampaign.getWeekStart() / 1440;
		}
		//Adjusting the value for Sunday as WeekDay class treats Sunday as 7th day
		if (weekStartDay == 0) {
			weekStartDay = WeekDay.SUNDAY.getCount();
		}

		// BEGIN modifying the Project Start times for each weekday
		objMapTod = ProjectUtil.getStartTimes(TimeContextFactory.getTimeContext(objProjRule, organizationMap, campaignMap),
				objProjRule.getProjectStartTimes(),
				TimeZoneOverrideUtil.getTimeZoneOffset(m_context, optimizationTargetTimeZone, timeZoneEffectiveDate));

		if (objMapTod != null && objMapTod.size() > 0) {
			Set<TimeOfDay> lSetTod = new HashSet<TimeOfDay>();
			startTimesByWeek = new HashMap<>();
			ProjectRulesStartTimePC projStPC;
			short dayOpen = 0;
			WeekDay wd;

			for (short iDay = 0; iDay < WeekDay.WEEK_LENGTH; iDay++) {

				short iWeekday = (short) (iDay + weekStartDay);
				if (iWeekday >= WeekDay.WEEK_LENGTH) {
					iWeekday -= WeekDay.WEEK_LENGTH;
				}

				wd = WeekDay.values()[iWeekday];

				dayOpen = (short) ((wd.getCount() + 1) > 7 ? (wd.getCount() + 1) - 7
						: (wd.getCount() + 1));

				if (objMapTod.get(wd) != null) {
					lSetTod = objMapTod.get(wd);
				} else {
					lSetTod = new HashSet<>();
				}

				int earliestStart = ((HOOAssignment) refHOO.iterator().next()).getDayOpen(dayOpen);
				int latestEnd = ((HOOAssignment) refHOO.iterator().next()).getDayClose(dayOpen);
				boolean is24Hr = (latestEnd - earliestStart == 1440);
				projStPC = new ProjectRulesStartTimePC(m_context,
						wd.toString(), dayOffset, lSetTod,
						selectedOrg != null ? is24Hr && selectedOrg.canCrossDayBoundary() : false);
				projStPC.getStartPickerPC().setEarliestStartTime(earliestStart);
				projStPC.getStartPickerPC().setLatestEndTime(latestEnd);
				projStPC.getStartPickerPC().setShiftLength(
						durationPC.getDuration().getDurationInMinutes());
				projStPC.setAutoExtractParamEnabled(true);
				//Adding Start Times to a class Level variable
				startTimesByWeek.put(wd, projStPC);

				stFormPC.addRow(m_localizer.getWeekdayNames(true)[dayOpen], projStPC);
			}
		}
		addChildComponent(stFormPC);
		addForm(stFormPC);

	}

	public Collection<Activity> getProjectMediaActivities(Collection<Activity> activities) {
		Map<ID, Collection<Media>> objMediaMap;
		HashSet<ID> activityIds = new HashSet<>(activities.size());
		ID id;

		try {
			for (Activity activity : activities) {

				id = activity.getID();
				if (!activityIds.contains(id)) {
					activityIds.add(id);
				}
			}

			id = null;

			objMediaMap = ActivityModelHandler.getActivitiesMediaMap(m_context, activityIds);

			if (objMediaMap != null) {
				Collection<Media> collMed = null;
				boolean isProject = false;
				for (Iterator<ID> i = objMediaMap.keySet().iterator(); i.hasNext(); ) {
					id = i.next();
					collMed = objMediaMap.get(id);

					for (Media m : collMed) {
						if (m.getId().equals(Media.MEDIA_ID_PROJECT)) {
							isProject = true;
						} else {
							continue;
						}
					}
					if (!isProject) {
						Activity toRemove = new Activity();
						toRemove.setID(id);
						activities.remove(toRemove);
					}

					isProject = false;
				}
			}

			return activities;
		} catch (RemoteException e) {
			handleUnableToLoadDataError(log, e);
			e.printStackTrace();
		} catch (BbmException e) {
			handleUnableToLoadDataError(log, e);
			e.printStackTrace();
		}

		return activities;
	}

	public Collection<ProjectRules> getRules() throws BbmFinderException {
		Collection<ProjectRules> ret = new ArrayList<ProjectRules>(projectRule);
		if (ret.isEmpty()) {
			ret.add(new ProjectRules());
		}

		for (ProjectRules rule : ret) { // Construct a shift from the form field

			if (!activitySelectorPC.getSelections().isEmpty() && (activitySelectorPC.isEdited()) || !isMultiEdit()) {
				rule.setActivityID(activitySelectorPC.getSelections().get(0).getID());
			}

			if (isEditingNameAndDesc()) {
				rule.setName(nameInputPC.getInputValue() != null ? nameInputPC.getInputValue().trim() : null);
				rule.setDescription(descInputPC.getInputValue());

				if (isByOrg) {
					rule.setOrganizationID(selectedOrg.getID());
				} else if (!isByOrg) {
					rule.setCampaignID(selectedCampaign.getID());
				}
			}

			if (!isMultiEdit() || durationPC.isEdited()) {
				rule.setDuration(durationPC.getDuration().getDurationInMinutes());
			}
		}
		return ret;
	}

	public Collection<ProjectRules> getProjectRule()
			throws BbmFinderException {
		Collection<ProjectRules> ret = new ArrayList<ProjectRules>(projectRule);

		TimeContext tx;

		if (ret.isEmpty()) {
			ret.add(new ProjectRules());
		}

		for (ProjectRules projRule : ret) {
			// Construct a shift from the form field
			if (!activitySelectorPC.getSelections().isEmpty() && (activitySelectorPC.isEdited() || !isMultiEdit())) {
				projRule.setActivityID(activitySelectorPC.getSelections().get(0).getID());
			}
			if (isEditingNameAndDesc()) {
				projRule.setName(nameInputPC.getInputValue());
				projRule.setDescription(descInputPC.getInputValue());
				if (isByOrg) {
					projRule.setOrganizationID(selectedOrg.getID()); // check for null
				} else if (!isByOrg) {
					projRule.setCampaignID(selectedCampaign.getID()); // check for null
				}
			}
			if (durationPC != null) {
				projRule.setDuration(durationPC.getDuration().getDurationInMinutes());
			}
			// Reading changes made to the Employee Counts
			// user receives a popup if these are left empty, so not checking for null
			projRule.setDailyMaxPerEmployee(dailyMaxCountPC.getEmployeeCount());
			projRule.setWeeklyMaxPerEmployee(weeklyMaxCountPC.getEmployeeCount());
			projRule.setMaxSimultaneous(simultMaxCountPC.getEmployeeCount());

			//Reading and Storing changes made to the Start Times by week
	  /*
       * Get the Start Times set by Week from the Map by each weekday
			 * Add and Remove Start Times from each Weekday.
			 * Finally prepare a collection of the Start Times for the whole week
			 * While Adding Start Times make sure all the parameters are set
			 */
			tx = TimeContextFactory.getTimeContext(projRule, organizationMap, campaignMap);

			Map<WeekDay, Set<ProjectStart>> mapOrgProjStartTimes = ProjectUtil.getProjectStartByWeek(tx, projRule);
			ProjectUtil.prepareStartTimes(startTimesByWeek, mapOrgProjStartTimes, projRule, tx);
		}
		return ret;
	}

					FormInputPC.TYPE_INTEGER, true);
			countPC.setLowerLimit(EmployeeCountPC.MIN_EMPLOYEE_COUNT);
			countPC.setUpperLimit(EmployeeCountPC.MAX_EMPLOYEE_COUNT);
			countPC.setAutoExtractParamEnabled(true);

			if (iCnt == EmployeeCountPC.MAX_EMPLOYEE_COUNT) {
				unlimitedCheckBoxPC = new CheckBoxPC(context,
						getUnlimitedName(), true, "jQuery('input[name="
						+ getCountName() + "]').val("
						+ EmployeeCountPC.MAX_EMPLOYEE_COUNT
						+ ").attr('readOnly','readonly').hide(); ",
						"jQuery('input[name=" + getCountName() + "]').val(jQuery('input[name="
								+ getCountName() + "]').attr('lowerlimit')).removeAttr('readOnly').show().focus();");
				unlimitedCheckBoxPC.setStyleClass("vAlignMiddle");
				countPC.setEnabled(false);
				countPC.setVisible(false);

			} else {
				unlimitedCheckBoxPC = new CheckBoxPC(context,
						getUnlimitedName(), false, "jQuery('input[name="
						+ getCountName() + "]').val("
						+ EmployeeCountPC.MAX_EMPLOYEE_COUNT
						+ ").attr('readOnly','readonly').hide(); ",
						"jQuery('input[name=" + getCountName() + "]').val("
								+ iCnt + ").removeAttr('readOnly').show().focus();");
				unlimitedCheckBoxPC.setStyleClass("vAlignMiddle");
			}
			addChildComponent(unlimitedCheckBoxPC);
			addChildComponent(countPC);

		}

		@Override
		public String getJavaScriptInitCode() {
			String handlerName = getName() + ".OnChange";
			StringBuffer javascriptSb = new StringBuffer();
			javascriptSb.append(super.getJavaScriptInitCode());
			javascriptSb.append("\n" + handlerName
					+ " = function() { jQuery('input[name=" + getEditedName()
					+ "]').val('true'); };");
			javascriptSb.append("\n" + getUnlimitedName()
					+ ".change(function() {");
			javascriptSb.append(handlerName + "();");
			javascriptSb.append("});");
			javascriptSb.append("\n jQuery('input[name="
					+ this.unlimitedCheckBoxPC.getInputName()
					+ "]').click(function() { ");
			javascriptSb.append("jQuery('input[name=" + getCountName()
					+ "]').toggleClass('"
					+ FormInputPC.INPUT_TEXT_READ_ONLY_STYLE
					+ "').toggleClass('" + FormInputPC.DEFAULT_STYLE + "');");
			javascriptSb.append("});");

			return javascriptSb.toString();

		}

		@Override
		public String getHtmlDisplay() {
			return BPMessageFormat
					.format(
							"{0}{1}&nbsp;&nbsp;{2}<input type='hidden' name=\"{3}\" value=\"{4}\">",
							new Object[]{
									unlimitedCheckBoxPC.getHtmlDisplay(),
									localizeIt(FsWebBundleKey.FS_UNLIMITED),
									countPC.getHtmlDisplay(), getEditedName(),
									isEdited});
		}

			return getName() + "__count";
		}

		private String getUnlimitedName() {
			return getName() + "__unlimited";
		}

			return getName() + "__edited";
		}

			if (!isAutoExtractParamEnabled()) {
				return true;
			}
			String editedStr = request.getParameter(getEditedName());
			if (editedStr != null) {
				isEdited = new Boolean(editedStr);
			}
			return super.extractParameters(request);
		}

		}

			return countPC;
		}

		public void setEmpCount(FormInputPC empCount) {
			if (empCount != null && empCount.getInputValue().equals(MAX_EMPLOYEE_COUNT)) {
				unlimitedCheckBoxPC.setIsChecked(true);
				countPC.setEnabled(false);
			} else {
				unlimitedCheckBoxPC.setIsChecked(false);
			}
			countPC.setInputValue(empCount.getInputValue());
		}

		public CheckBoxPC getCheckBoxPC() {
			return unlimitedCheckBoxPC;
		}

		public int getEmployeeCount() {
			if (this.countPC != null && this.countPC.getInputValue() != null) {
				return Integer.parseInt(this.countPC.getInputValue());
			} else {
				return 0;
			}
		}

		private StartTimesPickerPC stPickerPC;

		private CheckBoxPC weekDayCheck;
		private boolean isEdited = false;
		private boolean canShiftAssignmentsCrossDayBoundary = false;

		ProjectRulesStartTimePC(RequestContext context, String name,
				int dayOffset, Collection<TimeOfDay> projStTimes, boolean canShiftAssignmentsCrossDayBoundary) {
			super(context);

			setName(name);
			stPickerPC = new StartTimesPickerPC(context, getStartPickerName(),
					Duration.fromMinutes(getTimeZoneOffset()), Duration
					.fromMinutes(dayOffset), canShiftAssignmentsCrossDayBoundary);
			this.canShiftAssignmentsCrossDayBoundary = canShiftAssignmentsCrossDayBoundary;

			stPickerPC.setStartTimes(projStTimes);
			stPickerPC.setAutoExtractParamEnabled(true);


			if (projStTimes != null && projStTimes.size() > 0) {
				weekDayCheck = new CheckBoxPC(context, getWeekBoxName(), true,
						"jQuery(\"#" + getStartPickerName()
								+ "_table\").css('display','');", "jQuery(\"#"
						+ getStartPickerName()
						+ "_table\").css('display','none');" + getStartPickerName() + ".reset();");

			} else {
				weekDayCheck = new CheckBoxPC(context, getWeekBoxName(), false,
						"jQuery(\"#" + getStartPickerName()
								+ "_table\").css('display','');", "jQuery(\"#"
						+ getStartPickerName()
						+ "_table\").css('display','none');" + getStartPickerName() + ".reset();");
				stPickerPC.setDisplay(false);
			}
			weekDayCheck.setAutoExtractParamEnabled(true);
			addChildComponent(weekDayCheck);
			addChildComponent(stPickerPC);
		}

		@Override
		public String getHtmlDisplay() {
			StringBuffer html = new StringBuffer();
			html.append(BPMessageFormat
					.format(

									stPickerPC.getHtmlDisplay(),
									getEditedName(), isEdited}));

					append(canShiftAssignmentsCrossDayBoundary).append("'></input>");
			return html.toString();
		}
