package com.verint.web.fs.workrules;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.PaginationPair;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectIDListComparator;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.VTOEvent;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarTextButton;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;

import java.util.*;

public class VTOEventListPopupPM extends WorkRulesSelectionPopupPM {
	private static final long serialVersionUID = 1L;

	public static final String FORM_ACTION = "vtoevent_list_popup";
	public static final String POPUP_ID = "POPUP_VTO_EVENT";
	public static final String POPUP_ARG_VTO_EVENT_ID = "vtoEventID";
	public static final String POPUP_ARG_VTO_EVENT_IDS = "vtoEventIDs";
	public static final String POPUP_ARGS = "isByOrg,isSelectionPopup,ownerID,selectedIDs,vtoEventIDs";

	private boolean isByOrg;
	private ID ownerID;
	private HashSet<ID> ids = new HashSet<ID>();
	private String vtoEventIDs = "";
	private ArrayList<VTOEvent> vtoEvents = new ArrayList<VTOEvent>();
	private Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
	private Map<ID, Campaign> campaignMap = new HashMap<ID, Campaign>();
	private Map<ID, Activity> activityMap = new HashMap<ID, Activity>();

	// Flag to determine whether it is an informational pop up or selection popup.
	private boolean isSelectionPopup = false;

	public VTOEventListPopupPM(RequestContext context) {
		super(context);
	}

	@Override
	protected void initializeModel() {
		if (m_isInitialized) {
			return;
		}
		super.initializeModel();
		if (isSelectionPopup) {
			// if it is a selection pop-up, add the buttons to support
			// selection.
			ToolbarTextButton addButton = m_toolbar.createCloseWindowButton(POPUP_ID,
					m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ADD_VTO_EVENTS), true);
			addButton.setCustomJS(m_toolbar.getName() + ".disableButton('" + POPUP_ID + "'); WorkRulesUtil.addVTOEvents();");
			m_toolbar.addToolbarElement(addButton);
			m_toolbar.addCloseWindowTextButton("cancel",
					m_localizer.i18n(CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_CANCEL), true);
		}
		m_isInitialized = true;
	}

	@Override
	protected void loadData() throws Exception {
		if (ownerID != null) {
			if (isByOrg) {
				if (isPaginationNavPressed()) {
					vtoEvents = new ArrayList<>(WorkRulesModelHandler.getVTOEventsByIDs(m_context, getPageIDs()));
				} else {
					PaginationPair<VTOEvent> vtoData = WorkRulesModelHandler.getVTOEventsByOrgHierarchy(m_context, ownerID,
							m_list.getHeader().isAscending(), getPageIDSize(), getPagePaginationIndex(),
							m_list.getSortComparators(), ids);
					vtoEvents = new ArrayList<>(vtoData.getValues());
					initPaginationIDs(vtoData.getIDs());
				}
			} else {
				SchedulingPeriod selectedSchedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(m_context, ownerID);
				if (isPaginationNavPressed()) {
					vtoEvents = new ArrayList<>(WorkRulesModelHandler.getVTOEventsByIDs(m_context, getPageIDs()));
				} else {
					PaginationPair<VTOEvent> vtoData = WorkRulesModelHandler.getVTOEventsBySchedulingPeriod(m_context,
							selectedSchedulingPeriod, m_list.getHeader().isAscending(), getPageIDSize(),
							getPagePaginationIndex(), m_list.getSortComparators(), ids);
					vtoEvents = new ArrayList<>(vtoData.getValues());
					initPaginationIDs(vtoData.getIDs());
				}
			}
		} else {
			vtoEvents = new ArrayList<>(WorkRulesModelHandler.getVTOEventsByIDs(m_context,
					StringUtil.parseIDStringToCollection(vtoEventIDs)));
		}

		Collections.sort(vtoEvents, new ValueObjectIDListComparator(new ArrayList<>(getPageIDs())));

		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<>();
		Collection<ID> campaignIDs = new HashSet<>();
		Collection<ID> activityIDs = new HashSet<>();
		for (VTOEvent vtoE : vtoEvents) {
			if (vtoE.getOrganizationID() != null) {
				orgIDs.add(vtoE.getOrganizationID());
			}
			if (vtoE.getCampaignID() != null) {
				campaignIDs.add(vtoE.getCampaignID());
			}
			if (vtoE.getActivityID() != null) {
				activityIDs.add(vtoE.getActivityID());
			}
		}
		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!isByOrg) {
			campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		}
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);
	}

	@Override
	protected void initUtilityPane() {
		super.initUtilityPane();
		getUtilityPane().setIsHelpEnabled(false);
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		if (isSelectionPopup) {
			m_contentTitlePC.setPageTitle(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_VTO_EVENTS));
		} else {
			m_contentTitlePC.setPageTitle(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_VTO_EVENT_DETAILS));
		}
	}

	public void setSelectedIDs(String ids) {
		this.ids = new HashSet<>(StringUtil.parseIDStringToCollection(ids));
	}

	public static VTOEventListPopupPM getInstance(RequestContext context) {
		return (VTOEventListPopupPM) PageModel.getInstance(context, VTOEventListPopupPM.class);
	}

	public void setIsSelectionPopup(boolean isSelectionPopup) {
		this.isSelectionPopup = isSelectionPopup;
	}

	public void setIsByOrg(boolean isByOrg) {
		this.isByOrg = isByOrg;
	}

	public void setOwnerID(String ownerID) {
		this.ownerID = ID.fromInt(Integer.parseInt(ownerID));
	}

	public void setVtoEventIDs(String vtoEventIDs) {
		this.vtoEventIDs = vtoEventIDs;
	}

	@Override
	protected VTOEventListPC initList() {
		return new VTOEventListPC(getRequestContext(), "vtoEventTable", vtoEvents, orgMap, campaignMap, activityMap);
	}
}
