public class VTOEventListPC extends InsertableMultiColListPC {
	private static final long serialVersionUID = 1L;

	private static final String POPUP_ACTIVITY = "POPUP_ACTIVITY";
	private static final String POPUP_ACTIVITY_ARGS = "activityIDs";

	//Shift pattern screen has an additional column - Start Times
	private boolean isShiftPatternScreen = false;
	private LinkedHashMap<VTOEvent, ShiftPattern.StartTime> vtoEventStartTimeMap =
			new LinkedHashMap<VTOEvent, ShiftPattern.StartTime>();
	private Map<ID, Organization> orgMap = null;
	private Map<ID, Campaign> campaignMap = null;
	private Map<ID, Activity> activityMap = null;
	private boolean isChildTable = false;

	public VTOEventListPC(RequestContext context, String name, Collection<VTOEvent> vtoEvents,
			Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap, Map<ID, Activity> activityMap) {
		super(context);
		setName(name);

		this.orgMap = orgMap;
		this.campaignMap = campaignMap;
		this.activityMap = activityMap;

		if (vtoEvents != null) {
			for (VTOEvent event : vtoEvents) {
				vtoEventStartTimeMap.put(event, null);
			}
		}

		initHeader();
		initRows(vtoEventStartTimeMap);

		addRequiredJavaScriptFile(WorkRulesJSFileID.WORK_RULES_UTIL);
		addPopupArgs(POPUP_ACTIVITY, ActivityPopupPM.FORM_ACTION, "",
				POPUP_ACTIVITY_ARGS, "1000,600,ctr,ctr", true, 1);
		setIsCancelSelectEventPropagation(false);
	}

	public VTOEventListPC(RequestContext context, String name, ShiftPattern shiftPattern, Date viewDate,
			Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap, Map<ID, Activity> activityMap,
			boolean isChildTable) {
		super(context);
		setName(name);

		this.isChildTable = isChildTable;
		isShiftPatternScreen = true;
		this.orgMap = orgMap;
		this.campaignMap = campaignMap;
		this.activityMap = activityMap;

		vtoEventStartTimeMap = new LinkedHashMap<VTOEvent, ShiftPattern.StartTime>(shiftPattern.getVTOEventStartTimes());
		// Filter out vto events deleted before view date
		for (Iterator<Map.Entry<VTOEvent, ShiftPattern.StartTime>> iter = vtoEventStartTimeMap.entrySet().iterator(); iter.hasNext(); ) {
			VTOEvent vto = iter.next().getKey();
			if (vto.getDeleteOnDate() != null && vto.getDeleteOnDate().before(viewDate)) {
				iter.remove();
			}
		}

		initHeader();
		initRows(vtoEventStartTimeMap);

		addPopupArgs(POPUP_ACTIVITY, ActivityPopupPM.FORM_ACTION, "",
				POPUP_ACTIVITY_ARGS, "1000,600,ctr,ctr", true, 1);
		setIsCancelSelectEventPropagation(false);
	}

	private boolean isHeaderInited = false;

	private void initHeader() {
		if (isHeaderInited) {
			return;
		}
		isHeaderInited = true;
		TableHeaderPC header = getHeader();
		Localizer localizer = m_context.getLocalizer();
		header.addColumnHeaderData(FsWebBundleKey.FS_VTO_EVENT, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_VTO_EVENT), true);
		DefaultHeaderData dhd = new DefaultHeaderData(FsWebBundleKey.FS_EVENT_ACTIVITY,
				localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_EVENT_ACTIVITY));
		dhd.addImageButton(getActivityPopupImage(), getActivityPopupCaption(), getActivityPopupJS());
		dhd.setSortable(true);
		header.addColumnHeaderData(dhd);
		header.addColumnHeaderData(FsWebBundleKey.FS_MIN_DURATION, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MIN_DURATION), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_MAX_DURATION, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MAX_DURATION), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_ORGANIZATION, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ORGANIZATION), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_DESCRIPTION, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_DESCRIPTION), true);

		if (isShiftPatternScreen) {
			header.addColumnHeaderData(FsWebBundleKey.FS_START_TIMES, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_START_TIMES));
		}
	}

	public HtmlTag getWrapper() {
		if (isChildTable) {
			if (m_wrapper == null) {
				m_wrapper = new HtmlDiv();
			}
			m_wrapper.setStyle("margin:20px");
			return m_wrapper;
		} else {
			return super.getWrapper();
		}
	}

	private String getActivityPopupImage() {
		return ImageFileID.TOOLTIP_BUTTON;
	}

	private String getActivityPopupCaption() {
		return m_localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.ACTIVITY_VIEW_ACTION);
	}

	private String getActivityPopupJS() {
		return "WorkRulesUtil.handleInfoPopup(this, " + getName() + "," + getHeader().getName() + ",'" +
				ActivityPopupPM.POPUP_ID + "','" + ActivityPopupPM.POPUP_ARG_VO_ID + "','" +
				ActivityPopupPM.POPUP_ARG_VO_IDS + "');";
	}

	private void initRows(HashMap<VTOEvent, ShiftPattern.StartTime> vtoEventMap) {
		Collection<MultiColumnNodeData> rows = super.getListModel();

		String optimizationTargetName = "";
		for (VTOEvent vtoEvent : vtoEventMap.keySet()) {
			DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(
					vtoEvent.getID());

			if (vtoEvent.getOrganizationID() != null) {
				Organization o = orgMap.get(vtoEvent.getOrganizationID());
				if (o != null) {
					optimizationTargetName = o.getName();
				}
			} else {
				Campaign c = campaignMap.get(vtoEvent.getCampaignID());
				if (c != null) {
					optimizationTargetName = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.LOCAL);
				}
			}

			//Name
			if (vtoEvent.getDeleteOnDate() != null) {
				HtmlDel delTag = new HtmlDel(TimeZone.getDefault(), vtoEvent.getDeleteOnDate(), vtoEvent.getName());
				nodeData.add(delTag);
			} else {
				nodeData.add(vtoEvent.getName());
			}

			//Activity
			Activity a = activityMap.get(vtoEvent.getActivityID());
			nodeData.add(a.getName());

			//Duration
			nodeData.add(getLocalizer().formatDurationInMins(vtoEvent.getMinDuration()));
			nodeData.add(getLocalizer().formatDurationInMins(vtoEvent.getMaxDuration()));

			//Organization / campaign name
			nodeData.add(optimizationTargetName);

			//Description
			nodeData.add(vtoEvent.getDescription());

			//Start Times
			if (isShiftPatternScreen) {
				nodeData.add(WorkRulesUtil.getVTOEventStartTimeText(getLocalizer(),
						vtoEventMap.get(vtoEvent)));
			}

			setRowAttributes(nodeData, vtoEvent);

			rows.add(nodeData);
		}
	}

	protected void setRowAttributes(DefaultMultiColumnNodeData rowData, VTOEvent vtoEvent) {
		if (vtoEvent.getActivityID() != null) {
			rowData.setNodeAttribute(ActivityPopupPM.POPUP_ARG_VO_ID, vtoEvent.getActivityID().toString());
		}
	}

	public LinkedHashMap<DBSortField, Comparator> getSortComparators() {
		LinkedHashMap<DBSortField, Comparator> comparators =
				new LinkedHashMap<DBSortField, Comparator>();
		for (VTOEventSortField sortField : getDBSortFieldIndices()) {
			switch (sortField) {
				case ID:
					comparators.put(VTOEventSortField.ID, null);
					break;
				case NAME:
					comparators.put(VTOEventSortField.NAME, null);
					break;
				case ACTIVITY:
					comparators.put(VTOEventSortField.ACTIVITY, null);
					break;
				case MIN_DURATION:
					comparators.put(VTOEventSortField.MIN_DURATION, null);
					break;
				case MAX_DURATION:
					comparators.put(VTOEventSortField.MAX_DURATION, null);
					break;
				case ORGANIZATION:
					comparators.put(VTOEventSortField.ORGANIZATION,
							new OrganizationCampaignNameComparator(getLocalizer()));
					break;
				case DESCRIPTION:
					comparators.put(VTOEventSortField.DESCRIPTION, null);
					break;
			}
		}

		return comparators;
	}

	public ArrayList<VTOEventSortField> getDBSortFieldIndices() {
		ArrayList<VTOEventSortField> retVal = new ArrayList<VTOEventSortField>();
		if (getHeader().getSortColumnIndexes() == null) {
			//Default sort index is by ID (chronological order in which object was created)
			retVal.add(VTOEventSortField.ID);
		} else {
			for (int sortField : getHeader().getSortColumnIndexes()) {
				switch (sortField) {
					case 0:
						retVal.add(VTOEventSortField.NAME);
						break;
					case 1:
						retVal.add(VTOEventSortField.ACTIVITY);
						break;
					case 2:
						retVal.add(VTOEventSortField.MIN_DURATION);
						break;
					case 3:
						retVal.add(VTOEventSortField.MAX_DURATION);
						break;
					case 4:
						retVal.add(VTOEventSortField.ORGANIZATION);
						break;
					case 5:
						retVal.add(VTOEventSortField.DESCRIPTION);
						break;
					default:
						break;
				}
			}
		}
		return retVal;
	}

	/**
	 * Setting this to false will prevent the user from being able to change
	 * the sort order by clicking on the column header.  This behavior may be
	 * desirable if this list is used as an assignment list for another master
	 * list.  If this value is set to false, the list will be sorted by shift event
	 * name.  Default value is true.
	 */
	public void setIsUserSortable(boolean isUserSortable) {
		if (this.getHeader() != null) {
			this.getHeader().setSortable(isUserSortable);
			this.getHeader().setPartialSortable(isUserSortable);
		}

		//If user sorting is disabled, we will have a default sort which is by item name.  We are sorting here because
		//the data model (retrieved using getListModel()) has already been initialized in the constructor.
		if (isUserSortable == false) {
			SortUtil.sort(getListModel(), getHeader().getColumnIndexByName(FsWebBundleKey.FS_VTO_EVENT),
					true, m_context.getLocalizer());
		}
	}
}
