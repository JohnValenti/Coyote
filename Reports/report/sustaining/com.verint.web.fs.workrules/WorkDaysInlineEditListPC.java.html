	@Override

	private void initHeader() {
		TableHeaderPC header = getHeader();
		Localizer localizer = m_context.getLocalizer();
		header.addColumnHeaderData("toggleAll", toggleAll.getHtmlDisplay(), false);
		DefaultHeaderData dhd = new DefaultHeaderData(FsWebBundleKey.FS_SHIFT,
				getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_SHIFT));
		dhd.addImageButton(getShiftPopupImage(), getShiftPopupCaption(), getShiftPopupJS());
		dhd.setSortable(true);
		header.addColumnHeaderData(dhd);

		switch (shiftPatternPeriod) {
			case Week:
				String[] dow = m_localizer.getWeekdayNames(false);
				int[] dowOrder = m_localizer.getWeekdayValues(dayOfWeekStart);
				for (int i = 0; i < dowOrder.length; i++) {
					header.addColumnHeaderData(dow[dowOrder[i]], dow[dowOrder[i]]);
				}
				break;
			case TwentyEightDayMonth:
			case TwentyNineDayMonth:
			case ThirtyDayMonth:
			case ThirtyOneDayMonth:
				for (int i = 0; i < shiftPatternPeriod.getDaysInPeriod().length; i++) {
					header.addColumnHeaderData(Integer.toString(shiftPatternPeriod.getDaysInPeriod()[i]),
							m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MONTHLY_WORK_PATTERN_DAY_HEADER,
									new Object[]{m_localizer.formatNumber(shiftPatternPeriod.getDaysInPeriod()[i])}));
				}
				break;
			default:
		}

		header.addColumnHeaderData(FsWebBundleKey.FS_MIN_CONSECUTIVE_DAYS_HEADER, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MIN_CONSECUTIVE_DAYS_HEADER));
		header.addColumnHeaderData(FsWebBundleKey.FS_MAX_CONSECUTIVE_DAYS_HEADER, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MAX_CONSECUTIVE_DAYS_HEADER));
	}

	private boolean isRowsInited = false;

	@SuppressWarnings("unchecked")
	private void initRows() {
		if (isRowsInited) {
			return;
		}
		isRowsInited = true;

		List<MultiColumnNodeData> rows = super.getListModel();

		ArrayList<ID> ids = new ArrayList<ID>();
		for (Shift shift : shiftPatternShiftAttributes.keySet()) {
			MultiColumnNodeData nodeData = buildNodeData(shift);
			rows.add(nodeData);
			ids.add(shift.getID());
		}

		SortUtil.sort(getListModel(), getHeader().getColumnIndexByName(FsWebBundleKey.FS_SHIFT),
				true, m_context.getLocalizer());

		this.setRowIDs(StringUtil.createDelimitedString(ids.toArray(new ID[ids.size()])));
	}

	protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, Shift shift) {
		if (!shift.isMultiEditCommon()) {
			mcnd.setNodeAttribute(ROW_ATTRIBUTE_MULTI_EDIT_NOT_COMMON, String.valueOf(true));
		}
		//possible days off shifts are not removable
		if (shift.getID().toInt() == ShiftFieldInfo.POSSIBLE_DAYS_OFF_SHIFT_ID) {
			mcnd.setNodeAttribute(WorkpaneMultiColPM.IS_DELETEABLE, String.valueOf(false));
		} else {
			mcnd.setNodeAttribute(WorkpaneMultiColPM.IS_DELETEABLE, String.valueOf(true));
		}
		mcnd.setNodeAttribute(ShiftListPopupPM.ARG_SHIFT_ID, shift.getID().toString());
	}

	/**

	 *
	 * @param Collection<Shift>
	 * @return

	public String getRowHtml(Collection<Shift> shifts) {
		StringBuffer html = new StringBuffer();
		for (Shift shift : shifts) {
			super.getListModel().add(buildNodeData(shift));
		}

		super.appendRowData(html);

		//Create the javascript to instantiate the objects for the inline controls
		html.append(HtmlUtil.jsTag(jsInitCode.toString(), false));
		jsInitCode.setLength(0);

		return html.toString();
	}

	private StringBuffer jsInitCode = new StringBuffer();

	private MultiColumnNodeData buildNodeData(Shift shift) {
		String detailRowID = getName() + "__detail" + JSUtil.getJSSuffix(shift.getID().toString());
		DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(
				shift.getID().toString());

		// Toggle
		ToggleVisiblePC toggle = new ToggleVisiblePC(m_context, "toggle" +
				JSUtil.getJSSuffix(shift.getID().toString()), Collections.singleton(detailRowID));
		if (shift.getID().toInt() != ShiftFieldInfo.POSSIBLE_DAYS_OFF_SHIFT_ID) {
			if (isAjaxRequest) {
				addChildComponent(toggle);
				jsInitCode.append(toggle.getJavaScriptInitCode());
				jsInitCode.append(toggleAll.getName() + ".addChild(" + toggle.getName() + ");");
			} else {
				toggleAll.addChildComponent(toggle);
			}
			nodeData.add(toggle.getHtmlDisplay());
		} else {
			nodeData.add("");
		}

		//Name
		if (shift.isMultiEditCommon() == false) {
			nodeData.add(MULTI_VALUE_DATA + shift.getName());
		} else {
			nodeData.add(shift.getName());
		}

		//Occurrence
		for (int i = 0; i < shiftPatternPeriod.getDaysInPeriod().length; i++) {
			boolean isActive = true;

			if (shiftPatternShiftAttributes.get(shift) != null) {
				isActive = shiftPatternShiftAttributes.get(shift).getOccurrenceDays().contains(
						shiftPatternPeriod.getDaysInPeriod()[i]);
			}

			CheckBoxPC dayBoundaryCheckBoxPC = getOccurrenceCheckBoxPC(shift,
					shiftPatternPeriod.getDaysInPeriod()[i], isActive);
			addChildComponent(dayBoundaryCheckBoxPC);
			if (isAjaxRequest) {
				jsInitCode.append(dayBoundaryCheckBoxPC.getJavaScriptInitCode());
			}
			nodeData.add(dayBoundaryCheckBoxPC.getHtmlDisplay());
		}

		initMinMaxConsecutiveDays(nodeData, shift);

		setRowAttributes(nodeData, shift);

		// DOW details
		if (shift.getID().toInt() != ShiftFieldInfo.POSSIBLE_DAYS_OFF_SHIFT_ID) {
			return new MasterDetailRow(nodeData, getDetailRow(shift, toggle, detailRowID));
		} else {
			return nodeData;
		}
	}

	private DefaultMultiColumnNodeData getDetailRow(Shift shift, ToggleVisiblePC toggle, String detailRowID) {
		DefaultMultiColumnNodeData nodeDetail = new DefaultMultiColumnNodeData(shift.getID());
		WorkDaysInlineEditDetailListPC detail_pc = new WorkDaysInlineEditDetailListPC(m_context,
				shift, shiftPatternShiftAttributes.get(shift), organization, campaign, optimizationTargetTimeZone,
				shiftPatternPeriod);

		detail_pc.extractParameters(getRequestContext().getRequest());
		if (shiftPatternShiftAttributes != null && shiftPatternShiftAttributes.get(shift) != null) {
			if (detail_pc.isEdited()) {
				if (modifiedShifts.containsKey(shift) == false) {
					modifiedShifts.put(shift, shiftPatternShiftAttributes.get(shift));
				}
			}
		}

		addChildComponent(detail_pc);
		HtmlTableRow row = new HtmlTableRow();
		int colSpan;
		switch (shiftPatternPeriod) {
			case Week:
				colSpan = 11;
				break;
			case TwentyEightDayMonth:
				colSpan = 32;
				break;
			case TwentyNineDayMonth:
				colSpan = 33;
				break;
			case ThirtyDayMonth:
				colSpan = 34;
				break;
			case ThirtyOneDayMonth:
				colSpan = 35;
				break;
			default:
				colSpan = 11;
				break;
		}
		row.add("<td colspan='" + Integer.toString(colSpan) + "'>" + detail_pc + "</td>"); // Hard code the colspan for now, since we init selectable items before the header because of the toggles
		nodeDetail.setNodeAttribute("id", detailRowID);
		nodeDetail.setNodeAttribute("masterID", shift.getID().toString());
		nodeDetail.setNodeAttribute("style", "display: none; padding: 20px;");
		nodeDetail.add(row);
		jsInitCode.append(detail_pc.getJSInitCode());

		//For some reason the sizing is screwed up for drop downs that are initially hidden on page load,
		//  so we will need to resize them when the toggle is clicked
		StringBuffer onChangeJS = new StringBuffer();
		for (String dd : detail_pc.getDropDownNamesInList()) {
			onChangeJS.append(dd + ".resize();");
		}
		toggle.setOnChangeJS(onChangeJS.toString());
		onResetDetailJS.append(toggle.getName()).append(".change(function() { ").append(onChangeJS).append(" } );\n");

		return nodeDetail;
	}

	private CheckBoxPC getOccurrenceCheckBoxPC(Shift shift, int dow, boolean isActive) {
		CheckBoxPC ccPC = new CheckBoxPC(getRequestContext(),
				getName() + "_" + Integer.toString(dow) + "_occurrenceCheckBoxPC_" +
						JSUtil.getJSSuffix(shift.getID().toString()),
				isActive);
		if (shiftPatternShiftAttributes != null && shiftPatternShiftAttributes.get(shift) != null) {
			if (shiftPatternShiftAttributes.get(shift).isOccurrenceDayMultiEdit(dow)) {
				ccPC.setMultiEdit(true);
			}
		}
		ccPC.extractParameters(getRequestContext().getRequest());
		if (shiftPatternShiftAttributes != null && shiftPatternShiftAttributes.get(shift) != null) {
			if (ccPC.isEdited() || (ccPC.isMultiEdit() == false && (isActive != ccPC.isChecked()))) {
				ccPC.setMultiEdit(false);
				if (modifiedShifts.containsKey(shift)) {
					if (ccPC.isChecked()) {
						modifiedShifts.get(shift).getOccurrenceDays().add(dow);
					} else {
						modifiedShifts.get(shift).getOccurrenceDays().remove(dow);
					}
				} else {
					if (ccPC.isChecked()) {
						shiftPatternShiftAttributes.get(shift).getOccurrenceDays().add(dow);
					} else {
						shiftPatternShiftAttributes.get(shift).getOccurrenceDays().remove(dow);
					}

					modifiedShifts.put(shift, shiftPatternShiftAttributes.get(shift));
				}
				modifiedShifts.get(shift).setOccurrenceDayMultiEdit(dow, false);
			}
		}
		return ccPC;
	}

	/**
	 * Initializes the min/max days drop downs and binds client side behavior.
	 *
	 * @param shift
	 */
	private void initMinMaxConsecutiveDays(DefaultMultiColumnNodeData nodeData, Shift shift) {
		String componentNameMin = getName() + "_minDaysSelectionPC_" + JSUtil.getJSSuffix(shift.getID().toString());
		String componentNameMax = getName() + "_maxDaysSelectionPC_" + JSUtil.getJSSuffix(shift.getID().toString());

		// Configure the min drop down
		IntegerDropDownSelection ddMinDays = new IntegerDropDownSelection(getRequestContext(), componentNameMin, 0, 7);
		if (shiftPatternShiftAttributes != null && shiftPatternShiftAttributes.get(shift) != null) {
			ddMinDays.setSelectedIDs(Collections.singleton(Integer.toString(shiftPatternShiftAttributes == null ?
					0 : shiftPatternShiftAttributes.get(shift).getMinConsecutiveDay())));
			if (shiftPatternShiftAttributes.get(shift).isMinConsecutiveDaysMultiEdit()) {
				ddMinDays.setMultiEdit(true);
			}
		}

		ddMinDays.extractParameters(m_context.getRequest());
		if (shiftPatternShiftAttributes != null && shiftPatternShiftAttributes.get(shift) != null) {
			if (ddMinDays.isEdited()) {
				ddMinDays.setMultiEdit(false);
				if (modifiedShifts.containsKey(shift)) {
					modifiedShifts.get(shift).setMinConsecutiveDay(ddMinDays.getSelections().iterator().next());
				} else {
					shiftPatternShiftAttributes.get(shift).setMinConsecutiveDay(ddMinDays.getSelections().iterator().next());
					modifiedShifts.put(shift, shiftPatternShiftAttributes.get(shift));
				}
				modifiedShifts.get(shift).setIsMinConsecutiveDaysMultiEdit(false);
			}
		}

		String minDayOnChangeJS =
				"\n var minCountSelections = " + componentNameMin + ".getSelections();"
						+ "\n var maxCountSelections = " + componentNameMax + ".getSelections();"
						+ "\n if (minCountSelections.length > 0) {"
						+ "\n if (maxCountSelections.length > 0 && parseInt(minCountSelections[0]) > parseInt(maxCountSelections[0])) { "
						+ "\n " + componentNameMax + ".setSelections([minCountSelections[0]]);"
						+ "\n }"
						+ "\n }";

		ddMinDays.setUseDefaultOnChange(false);
		ddMinDays.setOnChangeJS(minDayOnChangeJS);

		addChildComponent(ddMinDays);
		nodeData.add(ddMinDays.getHtmlDisplay());

		// Configure the max drop down
		IntegerDropDownSelection ddMaxDays = new IntegerDropDownSelection(getRequestContext(), componentNameMax, 0, 7);
		if (shiftPatternShiftAttributes != null && shiftPatternShiftAttributes.get(shift) != null) {
			ddMaxDays.setSelectedIDs(Collections.singleton(Integer.toString(shiftPatternShiftAttributes == null ?
					0 : shiftPatternShiftAttributes.get(shift).getMaxConsecutiveDay())));
			if (shiftPatternShiftAttributes.get(shift).isMaxConsecutiveDaysMultiEdit()) {
				ddMaxDays.setMultiEdit(true);
			}
		}
		ddMaxDays.extractParameters(m_context.getRequest());
		if (shiftPatternShiftAttributes != null && shiftPatternShiftAttributes.keySet().contains(shift)) {
			if (ddMaxDays.isEdited()) {
				ddMaxDays.setMultiEdit(false);
				if (modifiedShifts.containsKey(shift)) {
					modifiedShifts.get(shift).setMaxConsecutiveDay(ddMaxDays.getSelections().iterator().next());
				} else {
					shiftPatternShiftAttributes.get(shift).setMaxConsecutiveDay(
							ddMaxDays.getSelections().iterator().next());
					modifiedShifts.put(shift, shiftPatternShiftAttributes.get(shift));
				}
				modifiedShifts.get(shift).setIsMaxConsecutiveDaysMultiEdit(false);
			}
		}

		String maxDayOnChangeJS =
				"\n var minCountSelections = " + componentNameMin + ".getSelections();"
						+ "\n var maxCountSelections = " + componentNameMax + ".getSelections();"
						+ "\n if (maxCountSelections.length > 0) {"
						+ "\n if (minCountSelections.length > 0 && parseInt(minCountSelections[0]) > parseInt(maxCountSelections[0])) { "
						+ "\n " + componentNameMin + ".setSelections([maxCountSelections[0]]);"
						+ "\n }"
						+ "\n }";

		ddMaxDays.setUseDefaultOnChange(false);
		ddMaxDays.setOnChangeJS(maxDayOnChangeJS);

		addChildComponent(ddMaxDays);
		nodeData.add(ddMaxDays.getHtmlDisplay());

		if (isAjaxRequest) {
			jsInitCode.append(ddMinDays.getJavaScriptInitCode());
			jsInitCode.append(ddMaxDays.getJavaScriptInitCode());
		}
	}

	public String getOnResetJS() {
		StringBuffer js = new StringBuffer();
		js.append(getToggleAllPC().getName() + ".initialize();\n");
		for (Shift shift : shiftPatternShiftAttributes.keySet()) {
			if (shift.getID().toInt() != ShiftFieldInfo.POSSIBLE_DAYS_OFF_SHIFT_ID) {
				js.append("toggle" + JSUtil.getJSSuffix(shift.getID().toString()) + ".initialize();\n");
			}
		}
		js.append(onResetDetailJS.toString());
		return js.toString();
	}

	private boolean isExtracted = false;

	@Override
	public boolean extractParameters(HttpServletRequest request) {
		if (isExtracted) {
			return true;
		}
		isExtracted = true;
		boolean retVal = super.extractParameters(request);

		// removed
		for (ID removedID : getRemovedRowIDCollection()) {
			for (Shift shift : shiftPatternShiftAttributes.keySet()) {
				if (removedID.equals(shift.getID())) {
					shiftPatternShiftAttributes.remove(shift);
					break;
				}
			}
		}

		// added
		try {
			Collection<ID> addedShiftIDs = getAddedRowIDCollection();
			if (addedShiftIDs.size() > 0) {
				for (Shift shift : WorkRulesModelHandler.getShiftsByIDs(m_context, addedShiftIDs)) {
					shift.setMultiEditCommon(true);
					shiftPatternShiftAttributes.put(shift, new ShiftPatternShiftAttributes()); // This will be updated in the initRows method call below
				}
			}
		} catch (BbmFinderException e) {
			retVal = false;
			handleUnableToLoadDataError(log, e);
		}

		initHeader();
		initRows();

		return retVal;
	}

	public Collection<ID> getRemovedRowIDCollection() {
		return StringUtil.parseIDStringToCollection(m_removedRowIDs);
	}

	public Collection<ID> getAddedRowIDCollection() {
		return StringUtil.parseIDStringToCollection(m_addedRowIDs);
	}

	public HashMap<Shift, ShiftPatternShiftAttributes> getAddedShiftAssignments() {
		HashMap<Shift, ShiftPatternShiftAttributes> retVal = new HashMap<Shift, ShiftPatternShiftAttributes>();
		Map<ID, Shift> shiftMap = ValueObjectUtil.getIDObjectMap(shiftPatternShiftAttributes.keySet());
		for (ID id : getAddedRowIDCollection()) {
			retVal.put(shiftMap.get(id), shiftPatternShiftAttributes.get(shiftMap.get(id)));
		}
		return retVal;
	}

	public HashMap<Shift, ShiftPatternShiftAttributes> getModifiedShiftAssignments() {
		return modifiedShifts;
	}
