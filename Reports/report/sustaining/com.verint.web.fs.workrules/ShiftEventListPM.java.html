public class ShiftEventListPM extends WorkRulesListPM {
	private static final long serialVersionUID = 1L;

	public static final String SHIFT_EVENT_IMPORT_POPUP = "SHIFT_EVENT_IMPORT_POPUP";
	public static final String SHIFT_EVENT_EXPORT_POPUP = "SHIFT_EVENT_EXPORT_POPUP";
	public static final String SHIFTEVENTLIST_SORTCOLUMN = ShiftEventListPC.SHIFTEVENTLIST_SORTCOLUMN;
	public static final String SHIFTEVENTLIST_SORTORDER = ShiftEventListPC.SHIFTEVENTLIST_SORTORDER;
	public static final String SHIFTEVENTLIST_ITEMINDEX = JumpToPagePaginationPC.SHIFTEVENTLIST_ITEMINDEX;
	ArrayList<ShiftEvent> shiftEvents = new ArrayList<ShiftEvent>();
	TimeZone optimizationTargetTimeZone;

	private Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
	private Map<ID, Campaign> campaignMap = new HashMap<ID, Campaign>();
	private Map<ID, Activity> activityMap = new HashMap<ID, Activity>();
	private boolean isAscendingSort = true;

	public ShiftEventListPM(RequestContext context) {
		super(context, JumpToPagePaginationPC.USED_IN_SHIFTEVENT_LIST_PAGE);
		removeChildComponent(m_list);
		m_list = new ShiftEventListPC(getRequestContext(), "shiftEventTable", shiftEvents, optimizationTargetTimeZone,
				selectedSchedulingPeriod == null ? new Date() : selectedSchedulingPeriod.getStartTime(), orgMap,
				campaignMap, activityMap, true, false);
		// Need to get the sort order after initialize the component for
		// loadData().
		isAscendingSort = m_list.getHeader().isAscending();
		this.addChildComponent(m_list);
		this.setIsMultiRowSelectable(true);
	}

	public static ShiftEventListPM getInstance(RequestContext context) {
		return (ShiftEventListPM) getInstance(context, ShiftEventListPM.class);
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		String pageTitle = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.WORKRULES_SHIFT_EVENTS_TITLE);
		m_contentTitlePC.setPageTitle(pageTitle);
	}

	@Override
	protected void loadData() throws Exception {
		super.loadData();
		int pageIndex = getPagePaginationIndex();
		int pageSize = getPageIDSize();
		int itemIndex = getEditIndex();
		String pageAction = getPageAction();
		String strItemIndex = (String) m_context.getAttribute(RequestContext.SESSION_SCOPE, SHIFTEVENTLIST_ITEMINDEX);
		if (pageAction != null
				&& (pageAction.equals(PageAction.CANCEL) || pageAction.equals(PageAction.SAVE)
				|| pageAction.equals(PageAction.TABLE_SORT) || pageAction.equals(PageAction.CREATE_NEW))) {

			if (!StringUtil.isEmpty(strItemIndex)) {
				itemIndex = Integer.parseInt(strItemIndex);
				pageIndex = (itemIndex / pageSize) + (itemIndex % pageSize > 0 ? 1 : 0);
			}
		}
		// Retrieve shifts events and add the selected org or campaign into the
		// collection of objects to retrieve
		if (isByOrg) {
			if (selectedOrg != null) {
				if (m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID,
						new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE))) {
					if (isPaginationNavPressed()) {
						shiftEvents = new ArrayList<ShiftEvent>(
								WorkRulesModelHandler.getShiftEvents(m_context, getPageIDs()));
					} else {
						PaginationPair<ShiftEvent> shiftEventData = WorkRulesModelHandler.getShiftEventsByOrgHierarchy(
								m_context, getSelectedIDObject(), isAscendingSort, pageSize, pageIndex,
								m_list.getSortComparators(), null, getTextToFind(itemToFind));
						shiftEvents = new ArrayList<ShiftEvent>(shiftEventData.getValues());
						initPaginationIDs(shiftEventData.getIDs());
					}
				} else {
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE,
							FsWebBundleKey.BUNDLE_NAME);
				}
				optimizationTargetTimeZone = selectedOrg.getTimeZone();
			}
		} else {
			if (selectedSchedulingPeriod != null) {
				if (m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID)) {
					if (isPaginationNavPressed()) {
						shiftEvents = new ArrayList<ShiftEvent>(
								WorkRulesModelHandler.getShiftEvents(m_context, getPageIDs()));
					} else {
						PaginationPair<ShiftEvent> shiftEventData = WorkRulesModelHandler.getShiftEventsBySchedulingPeriod(
								m_context, selectedSchedulingPeriod, isAscendingSort, pageSize, pageIndex,
								m_list.getSortComparators(), null, getTextToFind(itemToFind));
						shiftEvents = new ArrayList<ShiftEvent>(shiftEventData.getValues());
						initPaginationIDs(shiftEventData.getIDs());
					}
				} else {
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE,
							FsWebBundleKey.BUNDLE_NAME);
				}
			}
			if (selectedCampaign != null) {
				optimizationTargetTimeZone = selectedCampaign.getTimeZone();
			}
		}

		Collections.sort(shiftEvents, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));

		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		Collection<ID> campaignIDs = new HashSet<ID>();
		Collection<ID> activityIDs = new HashSet<ID>();
		for (ShiftEvent se : shiftEvents) {
			if (se.getOrganizationID() != null) {
				orgIDs.add(se.getOrganizationID());
			}
			if (se.getCampaignID() != null) {
				campaignIDs.add(se.getCampaignID());
			}
			if (se.getActivityID() != null) {
				activityIDs.add(se.getActivityID());
			}
		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!isByOrg) {
			campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		}
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);
	}

	@Override
	protected void initHeader() {
	}

	@Override
	protected void initSelectableItems() {
		this.removeChildComponent(m_list);
		m_list = new ShiftEventListPC(m_context, "shiftEventTable", shiftEvents, optimizationTargetTimeZone,
				selectedSchedulingPeriod == null ? new Date() : selectedSchedulingPeriod.getStartTime(), orgMap,
				campaignMap, activityMap, true, false) {
			@Override
			protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, ShiftEvent shiftEvent) {
				super.setRowAttributes(mcnd, shiftEvent);
				ShiftEventListPM.this.setRowAttributes(mcnd, shiftEvent.getOrganizationID());
			}
		};
		this.addChildComponent(m_list);
	}

	@Override
	protected void initToolbar() {

		if ((selectedCampaign != null && selectedSchedulingPeriod != null) || selectedOrg != null) {
			FindToolbarPageComponent findToolBar = new FindToolbarPageComponent(m_context, getToolbar(), FIND_ITEM_FN,
					WorkRulesRH.FIND_ACTION, getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_FIND_SHIFT_EVENTS_PROMPT));
			getToolbar().addToolbarComponent(findToolBar);
			getToolbar().addDivider();
			StringBuilder sbImportArgs = new StringBuilder();
			sbImportArgs.append(WorkRulesExportPopupPM.IS_BY_ORG_ARGUMENT).append("=").append(isByOrg);
			if (!isByOrg) {
				sbImportArgs.append("&");
				sbImportArgs.append(WorkRulesExportPopupPM.CAMPAIGN_ID_ARGUMENT).append("=")
						.append(selectedCampaign.getID());
				sbImportArgs.append("&");
				sbImportArgs.append(WorkRulesExportPopupPM.SCHEDULING_PERIOD_ID_ARGUMENT).append("=")
						.append(selectedSchedulingPeriod.getID());
			} else {
				sbImportArgs.append("&");
				sbImportArgs.append(WorkRulesExportPopupPM.ORGANIZATION_ID_ARGUMENT).append("=").append(selectedOrg.getID());
			}
			sbImportArgs.append("&");
			sbImportArgs.append(WorkRulesExportPopupPM.OPTIMIZATION_TARGET_TIMEZONE).append("=")
					.append(optimizationTargetTimeZone.getID());
			sbImportArgs.append("&");
			sbImportArgs.append(WorkRulesExportPopupPM.TIMEZONE_EFFECTIVE_DATE).append("=").append(timeZoneEffectiveDate);

			addPopupArgs(ShiftEventListPM.SHIFT_EVENT_IMPORT_POPUP, ShiftEventImportPopupPM.FORM_ACTION,
					sbImportArgs.toString(), null, "480,520,ctr,ctr", true, 1);
			ToolbarTextButton importButton = m_toolbar.createPopupWindowButton(ShiftEventListPM.SHIFT_EVENT_IMPORT_POPUP,
					i18n(m_bundle, CommonWebBundleKey.TOOLBAR_IMPORT),
					m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES));
			m_toolbar.addToolbarElement(importButton);

			StringBuilder sb = new StringBuilder();
			sb.append(WorkRulesExportPopupPM.IS_BY_ORG_ARGUMENT).append("=").append(isByOrg);
			if (!isByOrg) {
				sb.append("&");
				sb.append(WorkRulesExportPopupPM.CAMPAIGN_ID_ARGUMENT).append("=").append(selectedCampaign.getID());
				sb.append("&");
				sb.append(WorkRulesExportPopupPM.SCHEDULING_PERIOD_ID_ARGUMENT).append("=")
						.append(selectedSchedulingPeriod.getID());
			} else {
				sb.append("&");
				sb.append(WorkRulesExportPopupPM.ORGANIZATION_ID_ARGUMENT).append("=").append(selectedOrg.getID());
			}
			sb.append("&");
			sb.append(WorkRulesExportPopupPM.OPTIMIZATION_TARGET_TIMEZONE).append("=")
					.append(optimizationTargetTimeZone.getID());
			sb.append("&");
			sb.append(WorkRulesExportPopupPM.TIMEZONE_EFFECTIVE_DATE).append("=").append(timeZoneEffectiveDate);

			addPopupArgs(ShiftEventListPM.SHIFT_EVENT_EXPORT_POPUP, ShiftEventExportPopupPM.VIEW_ACTION, sb.toString(),
					null, "480,255,ctr,ctr", true, 1);
			ToolbarTextButton exportButton = m_toolbar.createPopupWindowButton(ShiftEventListPM.SHIFT_EVENT_EXPORT_POPUP,
					i18n(m_bundle, CommonWebBundleKey.TOOLBAR_EXPORT), true);
			m_toolbar.addToolbarElement(exportButton);

			boolean hasEditPrivilege = hasEditPrivilege();
			importButton.setEnabled(hasEditPrivilege);
			exportButton.setEnabled(hasEditPrivilege);

			m_toolbar.addDivider();
		}
		super.initToolbar();
	}
}
