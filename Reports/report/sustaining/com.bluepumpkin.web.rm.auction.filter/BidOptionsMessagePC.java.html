package com.bluepumpkin.web.rm.auction.filter;

import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.keys.ImageFileID;
import com.witness.web.uif.pagecomponent.message.MessagePC;

import java.util.ResourceBundle;

/**
 * Title:        BidOptionsMessagePC
 * Description:  Component to Render Filter settings in the Bid Options page.
 * Copyright:    Copyright (c) 2005
 * Company:      Witness Systems, Inc
 * @author       Gonzalo Quintanar
 * @version 1.0
 */
public class BidOptionsMessagePC extends MessagePC {

	/** Creates a new instance of BidOptionsMessagePC */
	public BidOptionsMessagePC(RequestContext context) {
		super(context);		
	}

	/** Creates a new instance of BidOptionsMessagePC */
	public BidOptionsMessagePC(RequestContext context, String name) {
		super(context, name);
	}
	
	/** Creates a new instance of BidOptionsMessagePC */
	public BidOptionsMessagePC(RequestContext context, String name, boolean isSuppressOutputIfEmpty,
			boolean isOverflowEanbled, boolean isMsgPreFmtEnabled, boolean isButtonEnabled, 
			String width, String height, String wrapperCSS, String msgStyle) 
	{
		super(context, name);
		setSupressOutputIfEmpty(isSuppressOutputIfEmpty);
		setIsOverflowEnabled(isOverflowEanbled);
		setMsgPreFmtEnabled(isMsgPreFmtEnabled);
		setIsButtonEnabled(isButtonEnabled);
		setWidth(width);
		setHeight(height);
		setWrapperCSS(wrapperCSS);
		setMessageStyle(msgStyle);
	}

	/**
	 * Append Button
	 */
	protected void appendButton(StringBuffer sb) {
		ResourceBundle bundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);
		String alt = i18n(bundle, RmWebBundleKey.RM_SHOW_ALL_FILTERS);
		String js = "expandMsgsInDialog('" + getName() + "');";
		sb.append("<button title=\"").append(alt).append("\" class=\"imageButton\" style=\"width:34px; height:24px;\" onClick=\"")
				.append(js).append("\"><img id=\"pgMsgBtn_img_id\" class=\"pageMessageBtn\" src=\"")
				.append(ImageFileID.TOGGLE_MSG_BUTTON)
				.append("\" alt=\"").append(alt)
				.append("\" /></button>");
	}
}
