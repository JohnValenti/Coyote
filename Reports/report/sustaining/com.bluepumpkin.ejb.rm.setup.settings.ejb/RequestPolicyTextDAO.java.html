import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

import com.bluepumpkin.ejb.rm.setup.settings.model.RequestPolicyText;
import com.bluepumpkin.ejb.rm.setup.settings.model.RequestPolicyTextFieldInfo;
import com.bluepumpkin.ejb.rm.util.RmUtil;


public class RequestPolicyTextDAO extends DAOBase {
    private static FieldInfo m_fieldInfo = new RequestPolicyTextFieldInfo();

    protected FieldInfo getFieldInfo() { return m_fieldInfo; }

    private static Category m_cat = Log.initCategory(RequestPolicyTextDAO.class.getName());

    //Maximum length of a policy text fragment.
    private static final int MAX_TEXTSIZE = 1023;

    private static final String TABLE_NAME = "REQUESTPOLICYTEXT";

    protected Category getCategory() {
        return m_cat;
    }

    /**
     * Default Constructor for the Class with Jdmo instance as argument.
     */
    public RequestPolicyTextDAO() {
        super();
    }

    /**
     * Constructor for the Class with Jdmo instance as argument.
     * @param Jdmo - Jdmo object instance.
     */
    public RequestPolicyTextDAO(Jdmo dmo) {
        super(dmo);
    }


    /**
     * Get RequestPolicyText object ValueObjectBase.
     * @return ValueObjectBase - RequestPolicyText object instance as
     * ValueObjectBase.
     */
    protected ValueObjectBase createValueObject() {
        return new RequestPolicyText();
    }

    /**
     * Get Policy Text as string for Request Type specified in requestType
     * for the Organization id pOrganizationId. This function joins the
     * Policy Text from a number of string fragments each less then
     * MAX_TEXTSIZE to a single String.
     *
     * @param pOrganizationId - Organization ID.
     * @param requestType - String id of Request Types.
     *
     * @throws BbmFinderException If there is an error while executing database
     * query
     */
    public String getRequestPolicy(ID pOrganizationId, String requestType)
    throws Exception {
        StringBuffer policyTextChunks = new StringBuffer("");
        boolean isPolicyNull = true;

        StringBuffer lStmt = new StringBuffer(512);
        lStmt.append(" SELECT FRAGMENTNUM, POLICYTEXT ");
        lStmt.append(" FROM  REQUESTPOLICYTEXT ");
        lStmt.append(" WHERE ORGANIZATIONID = ? ");
        lStmt.append(" AND REQUESTTYPE = ? ");
        lStmt.append(" ORDER BY FRAGMENTNUM");

        JdmoQuery jQuery1 = m_dmo.createQuery(lStmt.toString(),
                                              Jdmo.PARAM_QUERY);
        jQuery1.setParID(1, pOrganizationId);
        jQuery1.setParString(2, requestType);

        JdmoRowset rs = m_dmo.createRowset(jQuery1,
                                           Jdmo.FORWARD_ONLY,
                                           Jdmo.READ_ONLY);

        while (rs.next()) {
            String textStr = rs.getString("POLICYTEXT");
            if (textStr != null){
                policyTextChunks.append(textStr);
                isPolicyNull = false;
            }
        }

        return isPolicyNull ? null : policyTextChunks.toString();
    }


    /**
     * Insert the the Policy Text specified in policyText for Request Type
     * specified in requestType for the pOrganizationId. This function
     * breaks the Policy Text specified in policyText in to
     * a number of fragments if string policyText length is over MAX_TEXTSIZE
     *
     * @param pOrganizationId - Organization ID object.
     * @param requestType - Collection of String values for Request Types.
     * @param policyText - Collection of String values for Request Types.
     *
     * @throws BbmCreateException if there is an error while executing database
     * query
     */
    public void createRequestPolicy(ID pOrganizationId,
                                    String requestType,
                                    String policyText)
    throws Exception {

        createRequestPolicy(pOrganizationId,
                            requestType,
                            getPolicyTextArray(policyText));
    }

    /**
     * Insert the the Policy Text specified in policyTextChunks for Request Type
     * specified in requestType for the pOrganizationId. This function
     * breaks the Policy Text specified in policyText in to
     * a number of fragments if string policyText length is over MAX_TEXTSIZE
     *
     * @param pOrganizationId - Organization ID.
     * @param requestType - String if of Request Types.
     * @param policyTextChunks - Array of String values for Request Types.
     *
     * @throws BbmCreateException if there is a database error
     */
    private void createRequestPolicy(ID pOrganizationId,
                                     String requestType,
                                     String policyText[])
    throws Exception {
        HashMap lColumnValHashMap = new HashMap();
        for(int i=0; i < policyText.length; i ++) 
        {
            lColumnValHashMap.put("ORGANIZATIONID",pOrganizationId);
            lColumnValHashMap.put("REQUESTTYPE",requestType);
            lColumnValHashMap.put("FRAGMENTNUM",new Integer(i));
            String policyTextStr = policyText[i];

            lColumnValHashMap.put("POLICYTEXT", policyTextStr);
            
            m_dmo.addBatchInsert(TABLE_NAME, lColumnValHashMap);
        }

        m_dmo.executeBatch();
    }

    /**
     * Replace the request policies of the given organizations.
     *
     * @param orgIdList - Collection of Organization IDs.
     * @param requestType - String if of Request Types.
     * @param policyText - Array of String values for Request Types.
     *
     * @throws BbmUpdateException if there is a database error
     */
    public void updateRequestPolicy(Collection orgIdList,
                                    String requestType,
                                    String policyText)
        throws Exception
    {
        ArrayList deleteRequestTypes = new ArrayList(1);
        deleteRequestTypes.add(requestType);
        deleteRequestPolicy(orgIdList, deleteRequestTypes);

        String[] policyTextChunks = getPolicyTextArray(policyText);
        Iterator itr = orgIdList.iterator();
        while (itr.hasNext()) {
            ID pOrganizationId = (ID)itr.next();
            createRequestPolicy(pOrganizationId,
                                requestType,
                                policyTextChunks);
        }
    }

    /**
     * Delete all Policy for Request Types specified in requestTypeProp
     * collection for the all OrganizationId in collection orgIdList.
     *
     * @param orgIdList - Collection of Organization ID object.
     * @param requestTypeProp - Collection of String values for Request Types. f null, we delete ALL types.
     *
     * @throws BbmRemoveException if there is an error while executing database
     *                            query
     */
    public void deleteRequestPolicy(Collection orgIdList,
                                    Collection requestTypeProp)
    throws Exception {

        StringBuffer strQuery = new StringBuffer(128);
        strQuery.append(" DELETE ");
        strQuery.append(" FROM REQUESTPOLICYTEXT ");
        strQuery.append(" WHERE ORGANIZATIONID IN ");
        strQuery.append(m_dmo.createInClause(orgIdList));
        if (requestTypeProp != null)
        {
            strQuery.append(" AND REQUESTTYPE IN ");
            strQuery.append(RmUtil.createInClause(requestTypeProp));
        }
        m_dmo.executeCommand(strQuery.toString());
    }

    /**
     * Chunck policy text into pieces small enough to fit in a
     * row fo the table.
     *
     * @param policyText - the text to chunk
     *
     * @return String[] - the text broken into pieces
     */
    private static String[] getPolicyTextArray(String policyText) {
        String[] policyTextChunks = null;
        if (policyText==null) policyText = "";
        int lStrLength = policyText.length();
        if (MAX_TEXTSIZE < lStrLength) {
            //find the no of fragments
            int lNoOfFragments = lStrLength / MAX_TEXTSIZE;

            if ((lStrLength % MAX_TEXTSIZE) > 0) {
                lNoOfFragments++;
            }

            policyTextChunks = new String[lNoOfFragments];

            for (int i = 0; i < lNoOfFragments; i++) {
                int lEndIndex = (i*MAX_TEXTSIZE ) + MAX_TEXTSIZE-1;

                if (lStrLength > lEndIndex )
                    policyTextChunks[i] = policyText.substring(i*MAX_TEXTSIZE,
                                                           lEndIndex+1);
                else
                    policyTextChunks[i] = policyText.substring(i*MAX_TEXTSIZE,
                                                           lStrLength);
            }
        }
        else {
            policyTextChunks = new String[1];
            policyTextChunks[0] = policyText;
        }
        return policyTextChunks;
    }
}
