package com.verint.web.rm.rest.entity.shiftswap;

import javax.xml.bind.annotation.XmlRootElement;

import com.verint.web.rest.entity.AbstractWFOEntity;

/**
 * Represents the fields to edit a Shift Swap request.
 */
@XmlRootElement(name = "request")
public class EditShiftSwapRequestEntity extends AbstractWFOEntity {

	// negotiation fields
	private String expirationDate;
	private boolean negotiation;
	private String negotiationComment;

	// submit fields
	private boolean submitToManager;
	private String comment;

	public EditShiftSwapRequestEntity() { // NOSONAR
	}

	public String getExpirationDate() {
		return expirationDate;
	}

	public void setExpirationDate(String expirationDate) {
		this.expirationDate = expirationDate;
	}

	public boolean isNegotiation() {
		return negotiation;
	}

	public void setNegotiation(boolean isNegotiation) {
		this.negotiation = isNegotiation;
	}

	public String getNegotiationComment() {
		return negotiationComment;
	}

	public void setNegotiationComment(String negotiationComment) {
		this.negotiationComment = negotiationComment;
	}

	public boolean isSubmitToManager() {
		return submitToManager;
	}

	public void setSubmitToManager(boolean isSubmitToManager) {
		this.submitToManager = isSubmitToManager;
	}

	public String getComment() {
		return comment;
	}

	public void setComment(String comment) {
		this.comment = comment;
	}

}
