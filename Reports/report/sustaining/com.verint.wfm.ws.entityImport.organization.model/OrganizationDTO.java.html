package com.verint.wfm.ws.entityImport.organization.model;

import javax.xml.bind.annotation.XmlType;

import com.verint.wfm.ws.entityImport.model.HoursOfOperationDTO;

@XmlType(name = "OrganizationDTO", propOrder = { "externalSourceId", "name", "description", "timeZoneId",
		"weekStartDate", "numSeats", "minTimeBetweenShift", "maxConsecutiveWorkingDay",
		"parentOrganizationName", "hoursOfOperationDTO" })
public class OrganizationDTO {
	private static final long serialVersionUID = 1L;

	protected String externalSourceId;
	protected String name;
	protected String description;
	protected String timeZoneId;
	protected Short weekStartDate;
	protected Integer numSeats;
	protected int minTimeBetweenShift;
	protected int maxConsecutiveWorkingDay;
	protected String parentOrganizationName;
	protected HoursOfOperationDTO hoursOfOperationDTO;

	public OrganizationDTO() {}

	public String getExternalSourceId() {
		return externalSourceId;
	}

	public void setExternalSourceId(String externalSourceId) {
		this.externalSourceId = externalSourceId;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public String getDescription() {
		return description;
	}

	public void setDescription(String description) {
		this.description = description;
	}

	public String getTimeZoneId() {
		return timeZoneId;
	}

	public void setTimeZoneId(String timeZoneId) {
		this.timeZoneId = timeZoneId;
	}

	public Short getWeekStartDate() {
		return weekStartDate;
	}

	public void setWeekStartDate(Short weekStartDate) {
		this.weekStartDate = weekStartDate;
	}

	public Integer getNumSeats() {
		return numSeats;
	}

	public void setNumSeats(Integer numSeats) {
		this.numSeats = numSeats;
	}

	public int getMinTimeBetweenShift() {
		return minTimeBetweenShift;
	}

	public void setMinTimeBetweenShift(int minTimeBetweenShift) {
		this.minTimeBetweenShift = minTimeBetweenShift;
	}

	public int getMaxConsecutiveWorkingDay() {
		return maxConsecutiveWorkingDay;
	}

	public void setMaxConsecutiveWorkingDay(int maxConsecutiveWorkingDay) {
		this.maxConsecutiveWorkingDay = maxConsecutiveWorkingDay;
	}

	public String getParentOrganizationName() {
		return parentOrganizationName;
	}

	public void setParentOrganizationName(String parentOrganizationName) {
		this.parentOrganizationName = parentOrganizationName;
	}

	public HoursOfOperationDTO getHoursOfOperationDTO() {
		return hoursOfOperationDTO;
	}

	public void setHoursOfOperationDTO(HoursOfOperationDTO hoursOfOperationDTO) {
		this.hoursOfOperationDTO = hoursOfOperationDTO;
	}
}
