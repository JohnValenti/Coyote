package com.bluepumpkin.web.bbm.workload.queue.selection;

import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;

public class ServiceGoalsSPQueueSelectionStrategy implements ISPQueueSelectionStrategy {

	/**
	 * Combined media root node is not visible for Service Goals.
	 */
	public boolean isCombinedMediaRootNodeVisible() {
		return false;
	}

	/**
	 * Constituent queues are visible only for skill based SPs.
	 */
	public boolean areConstituentQueuesVisible(SchedulingPeriod sp) {
		return sp.getSkillBased();
	}

	/**
	 * Combined queues are selectable only for non-skilled SPs.
	 */
	public boolean areCombinedQueuesSelectable(SchedulingPeriod sp) {
		return sp.getSkillBased() == false;
	}

}
