package com.bluepumpkin.web.bbm.workload.queue.selection;

import com.bluepumpkin.web.core.base.RequestHandler;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        QueueSelectorRH
 * Description:  Queue Selector Request Handler
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, dsu
 * @version      1.0
 *
 * Created on January 6, 2003, 2:50 PM
 */
public class QueueSelectorRH extends RequestHandler {
	private static String FORM_PAGE = "QUEUE_SELECTOR_FORM";

	/**
	 * Process Requests for the queue area
	 */
	public void processRequest(RequestContext context) {
		processNext(context);
	}

	/**
	 * Process Next
	 */
	private void processNext(RequestContext context) {
		QueueSelectorFormPopupPM model = new QueueSelectorFormPopupPM(context);
		model.extractParameters(context.getRequest());

		if (model.validate()) {
			storeQueueValues(context, model);
			setNextScreenName(context, FORM_PAGE);
		}
	}
	
	private void storeQueueValues(RequestContext context, QueueSelectorFormPopupPM model) {
		QueueFilterUtil.storeCampaignID(context, model.getCampaignID());
		QueueFilterUtil.storeIsCombined(context, model.isCombined());
		QueueFilterUtil.storeTimePeriodStart(context, model.getStartDate());
		QueueFilterUtil.storeTimePeriodEnd(context, model.getEndDate());
	}
}
