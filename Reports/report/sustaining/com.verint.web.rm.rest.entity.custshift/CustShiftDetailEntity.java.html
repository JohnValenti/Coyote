package com.verint.web.rm.rest.entity.custshift;

import java.util.ArrayList;
import java.util.List;

import javax.xml.bind.annotation.XmlRootElement;

import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rm.rest.entity.Entity;
import com.verint.web.rm.rest.entity.MessageEntity;
import com.verint.web.rm.rest.entity.TimeRangeEntity;

@XmlRootElement(name = "details")
public class CustShiftDetailEntity extends AbstractWFOEntity {

	private final List<Entity> activities = new ArrayList<Entity>();
	private final List<ShiftEntity> shifts = new ArrayList<ShiftEntity>();
	private final List<Entity> extensionAfterActivities = new ArrayList<Entity>();
	private final List<ShiftOTExtensionEntity> extensionAfterTypes = new ArrayList<ShiftOTExtensionEntity>();
	private final List<Entity> extensionBeforeActivities = new ArrayList<Entity>();
	private final List<ShiftOTExtensionEntity> extensionBeforeTypes = new ArrayList<ShiftOTExtensionEntity>();
	private final List<Entity> gapActivities = new ArrayList<Entity>();
	private final List<ShiftOTExtensionEntity> gapTypes = new ArrayList<ShiftOTExtensionEntity>();
	private final List<TimeRangeEntity> gaps = new ArrayList<TimeRangeEntity>();
	private ShiftAssignmentEntity shiftAssignment;
	private final List<MessageEntity> messages = new ArrayList<MessageEntity>();
	private CustShiftRequestEntity request;

	private int gapAfterDuration;
	private int gapBeforeDuration;
	private String mainShiftEndTime;
	private String mainShiftStartTime;
	private Integer mostDeficientActivity1;
	private Integer mostDeficientActivity2;

	/**
	 * Whether the employee is in a SP during the date requested. If there is no SP, they cannot make shift change requests.
	 */
	private boolean inSP;

	public CustShiftDetailEntity() {
		super();
	}

	public List<Entity> getActivities() {
		return activities;
	}

	public List<ShiftEntity> getShifts() {
		return shifts;
	}

	public List<Entity> getExtensionAfterActivities() {
		return extensionAfterActivities;
	}

	public List<ShiftOTExtensionEntity> getExtensionAfterTypes() {
		return extensionAfterTypes;
	}

	public List<Entity> getExtensionBeforeActivities() {
		return extensionBeforeActivities;
	}

	public List<ShiftOTExtensionEntity> getExtensionBeforeTypes() {
		return extensionBeforeTypes;
	}

	public List<Entity> getGapActivities() {
		return gapActivities;
	}

	public List<ShiftOTExtensionEntity> getGapTypes() {
		return gapTypes;
	}

	public List<TimeRangeEntity> getGaps() {
		return gaps;
	}

	public boolean isInSP() {
		return inSP;
	}

	public void setInSP(boolean inSP) {
		this.inSP = inSP;
	}

	public ShiftAssignmentEntity getShiftAssignment() {
		return shiftAssignment;
	}

	public void setShiftAssignment(ShiftAssignmentEntity shiftAssignment) {
		this.shiftAssignment = shiftAssignment;
	}

	public int getGapAfterDuration() {
		return gapAfterDuration;
	}

	public void setGapAfterDuration(int gapAfterDuration) {
		this.gapAfterDuration = gapAfterDuration;
	}

	public int getGapBeforeDuration() {
		return gapBeforeDuration;
	}

	public void setGapBeforeDuration(int gapBeforeDuration) {
		this.gapBeforeDuration = gapBeforeDuration;
	}

	public String getMainShiftEndTime() {
		return mainShiftEndTime;
	}

	public void setMainShiftEndTime(String mainShiftEndTime) {
		this.mainShiftEndTime = mainShiftEndTime;
	}

	public String getMainShiftStartTime() {
		return mainShiftStartTime;
	}

	public void setMainShiftStartTime(String mainShiftStartTime) {
		this.mainShiftStartTime = mainShiftStartTime;
	}

	public Integer getMostDeficientActivity1() {
		return mostDeficientActivity1;
	}

	public void setMostDeficientActivity1(Integer mostDeficientActivity1) {
		this.mostDeficientActivity1 = mostDeficientActivity1;
	}

	public Integer getMostDeficientActivity2() {
		return mostDeficientActivity2;
	}

	public void setMostDeficientActivity2(Integer mostDeficientActivity2) {
		this.mostDeficientActivity2 = mostDeficientActivity2;
	}

	public List<MessageEntity> getMessages() {
		return messages;
	}

	public CustShiftRequestEntity getRequest() {
		return request;
	}

	public void setRequest(CustShiftRequestEntity request) {
		this.request = request;
	}
}
