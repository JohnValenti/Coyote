    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.bbm.notifyrules.model.NotifyRuleChecker#testRule(com.bluepumpkin.ejb.bbm.notifyrules.model.NotifyRule, java.lang.Object)

    public ActionDetail testRule(NotifyRule rule, Object actionContext) throws Exception {
        // if notification detail object is not the type we need, return. 
        if (! (actionContext instanceof SerializedAuctionDetail) ) 
            return null;
        
        // get rule's orgID
        if (rule.getScopeType() != Organization.SCOPE_TYPE) {
        	//should throw exception here ?????
        	return null;
        }
        ID ruleOrgID = rule.getScopeID();
        
        SerializedAuctionDetail serAucDetail = (SerializedAuctionDetail) actionContext;
        m_cat.debug("Running rule for: " + serAucDetail);
        
        // check if the rule's org matches the employee's
        Date currDate = new Date(); //TODO: must we use auction's time range instead of the current date?
        ID nextEmpID = serAucDetail.getNextEmployeeID();
        // TODO: move to common package.
        ID empOrgID = TOCalcUtil.getCurrentOrgIDForEmployeeID(nextEmpID);
        // get emp orgID's parent org IDs.
        Collection empOrgParentIDs = CacheUtilBBM.getParentOrgIDs(empOrgID, null);
        
        // Does employee's orgID match the rule's orgID or one of its children's orgID?  
        // If not, rule does not apply.            
        if ( !ruleOrgID.equals(empOrgID) && !empOrgParentIDs.contains(ruleOrgID) ) {
            m_cat.debug("Rule failed: empOrgID != ruleOrgID and empOrgID != any ruleOrgID's children: " + empOrgID + " != " + ruleOrgID);
            return null; 
        }
            
        // rule applies.  construct the auctionDetail message.
        //
        // You have been added to the auction <auction name>.  Your bid deadline is <time>.  Please
        // complete your bidding before the deadline.
        ID nextBidderEmpID = serAucDetail.getNextEmployeeID();
        String msg = RequestUtil.getLocalizedMsgInAppDefLocale(RmEjbBundleKey.SBR_SERIALAUCTIONNOTIFICATION,
            ValidationUtil.getEmployeeByID(nextBidderEmpID, null, Employee.DETAIL_LEVEL_EMPLOYEE_BASIC),
            serAucDetail.getAuctionName(), serAucDetail.getNextEmployeeBidDeadline(), 
            RequestUtil.getTimezoneCurrentForEmployeeID(nextBidderEmpID));
                            
        ActionDetail actionDetail = new ActionDetail();
        actionDetail.setObjectType(ActionDetail.WORKRESOUCEID);
		Document doc = m_parser.newDocument();
        Element xmlRootElem = RequestUtil.createElemAndAppendAsChild(null, doc, RULE_XMLROOT_ELEMENTNAME, msg);

		Localizer localizer = RequestUtil.getLocalizerForAppDefLocale();
		TimeZone timeZone = RequestUtil.getTimezoneCurrentForEmployeeID(nextBidderEmpID);
		String empName = RequestUtil.getLocalizedEmpNameByID(nextBidderEmpID, localizer);
		
		RequestUtil.createElemAndAppendAsChild(xmlRootElem, null, RULE_XML_EMPLOYEE_NAME, empName);
		
		RequestUtil.createElemAndAppendAsChild(xmlRootElem, null, RULE_XML_AUCTION_NAME, serAucDetail.getAuctionName());
		
		RequestUtil.createElemAndAppendAsChild(xmlRootElem, null, RULE_XML_DEADLINE_TIME,
			localizer.formatDateTime(serAucDetail.getNextEmployeeBidDeadline(), timeZone) + " "
				+ localizer.formatTimeZone(timeZone.getID()));
		
        actionDetail.setDetails(nextEmpID, xmlRootElem);

        m_cat.debug("Rule fired successfully: " + 
            RequestUtil.getXMLStringForElemOrDoc(null, xmlRootElem.getOwnerDocument()));
        return actionDetail;
    }
