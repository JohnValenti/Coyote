	/**
	 * Initialize Model with Data
	 */
	@Override
	protected void initializeModel() {
		if (m_isInitialized) return; else m_isInitialized = true;
		try {
			if (m_mode == MODE_CREATE) {
				// Validate Selected Schedule
				m_isRequestValid = validateRequestForCreation();

				// Load information to create new Request
				Collection bsIDList = StringUtil.parseIDStringToCollection(m_bSchedIDs);
				m_shiftAssignList = AuctionMH.getShiftAssignments(m_auctionID, bsIDList);
				m_auction = AuctionMH.getAuction(m_auctionID);
			} else {
				// Load Shift Bid Request
				boolean isViewDetail = m_mode == MODE_VIEW;
				boolean isEdit = m_mode == MODE_EDIT;
				ID reqID = m_sbRequest.getID();
				ShiftBidRequest sbRequest = AuctionMH.getRequest(reqID, isViewDetail || isEdit);
				setShiftBidRequest( sbRequest );
				if (sbRequest != null){
					m_shiftAssignList = m_sbRequest.getOptMethods().getStitchedShiftAssignments();
					m_auction = m_sbRequest.getOptMethods().getShiftBidAuction();

					if (isViewDetail || isEdit) loadRequestDetailInfo(m_sbRequest);
				} else {
					this.addPageMessage(Message.ERROR_TYPE, i18n(m_bundle, RmWebBundleKey.REQUEST_NOT_FOUND_ERROR));
				}
			}

			if (m_sbRequest != null){
				// Retrieve Shift Bidder
				ID bidderID = m_sbRequest.getShiftBidderID();
				m_shiftBidder = AuctionMH.getShiftBidderByID(bidderID);
				m_empName = AuctionMH.getEmployeeNameByBidderID(m_context, bidderID);
				if (m_mode!=MODE_VIEW) {
					m_maxPrefValue = AuctionMH.getEmpMaxPrefLevel(m_context, m_shiftBidder.getEmployeeID());
				}

				// Make sure Scheduling Period is available
				m_sp = m_auction.getOptMethods().getSP();
				if (m_sp==null) {
					m_sp = AuctionMH.getAuction(m_auction.getID()).getOptMethods().getSP();
				}

				m_actMap = ActivityModelHandler.getAllActivitiesMap(m_context);
				m_campaignTz = CampaignModelHandler.getCampaign(m_context, m_sp.getCampaignID()).getTimeZone();
			}
		} catch(RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataErrorAndLogAsDebug(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		} finally {
			initContentTitle();
			if (m_sbRequest != null){
				initContent();
				initToolbar();
			} else {
				initToolbar(false);
			}
		}
	}

	/**
	 * Validate Request for Creation where it also computest the request rank
	 */
	private final boolean validateRequestForCreation() {
		if (isCloseAfterLoad()) return true;

		boolean result = false;
		try {
			ShiftBidRequest request = getShiftBidRequest();

			// validate request and computes rank as well
			request = AuctionMH.validateRequestAndComputeRank(request);
			setShiftBidRequest(request);

			// Make sure there is no hard validation errors
			result = true;
			Collection valResultList = request.getValidationResults(true);
			for(Iterator it=valResultList.iterator(); it.hasNext(); ) {
				ValidationResult vr = (ValidationResult)it.next();
				if (!vr.isSoft() && !NoShiftBidRequestDuplicateNamesHV.CLASS_NAME.equalsIgnoreCase(vr.getValidatorName())) {
					//a hard validation error that is not a duplicate name 
					result = false;
					break;
				}
			}
		} catch(RmHardValidationException ex) {
			Message msg = addPageMessage(Message.ERROR_TYPE,
					RmWebBundleKey.SCHEDULE_VALIDATION_FAILED, RmWebBundleKey.BUNDLE_NAME);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
			LOG.l7dDebug(RmWebLogBundleKey.SCHEDULE_VALIDATION_FAILED, ex);
		} catch (Exception ex) {
			String rbKey = RmWebBundleKey.SCHEDULE_VALIDATION_FAILED;
			Message msg = new Message(Message.ERROR_TYPE, rbKey, RmWebBundleKey.BUNDLE_NAME);
			addPageMessage(msg);
			LOG.l7dError(RmWebLogBundleKey.SCHEDULE_VALIDATION_FAILED, ex);
			result = false;
		}
		return result;
	}

	/**
	 * Initialize Content
	 */
	@Override
	protected void initContent() {

		if (m_isRequestSaved) {
			RequestUtil.addSavedRequestMsg(m_sbRequest, this);
		}

		initRequestContainer();
		initAuctionContainer();
		initScheduleContainer();

		// If Save action didn't add error page message, show validation messages (if any)
		if (m_mode == MODE_CREATE && !hasPageError()) {
			// Create Page Message from Validation Result
			Collection vrList = m_sbRequest.getValidationResults(true);
			for(Iterator it=vrList.iterator(); it.hasNext(); ) {
				ValidationResult vr = (ValidationResult)it.next();
				int msgType = vr.isSoft() ? Message.WARNING_TYPE : Message.ERROR_TYPE;
				addPageMessage(msgType, vr.getLocalizedMessage(m_localizer, m_view_tz));
			}
		} else if (m_mode == MODE_VIEW || m_mode == MODE_EDIT) {
			initAlertList();
			initAuditTrailList();
		}
	}

	private void initAuctionContainer() {
		m_auctionContainerPC = createFormPC("auctionContainer");

		String title = i18n(m_bundle, RmWebBundleKey.AUCTION) + ": " + m_auction.getName();
		m_auctionContainerPC.setTitle(title);

		m_auctionContainerPC.addRow(
				i18n(m_common_bundle, UIFWebBundleKey.SCHEDULE_PERIOD),
				SchedulingPeriodUtil.getSPDisplay(m_sp, m_localizer, m_campaignTz)
		);

		String hourDisplay = AuctionUtil.getScheduleHourDisplay(m_shiftAssignList, m_localizer);
		m_auctionContainerPC.addRow(
				i18n(m_bundle, RmWebBundleKey.TOTAL_SCHEDULE_HOURS),
				hourDisplay
		);

		boolean isUseScoring = m_auction.getUsesScoring();
		if (isUseScoring) {
			m_auctionContainerPC.addRow(
					i18n(m_bundle, RmWebBundleKey.NONE_APPR_BONUS),
					m_localizer.formatNumber(m_auction.getBonusForNoBidsAccepted())
			);

			m_auctionContainerPC.addRow(
					i18n(m_bundle, RmWebBundleKey.BID_RANK),  
					createRankForBidMsg()
			);
		}

		m_auctionContainerPC.addRow(
				i18n(m_bundle, RmWebBundleKey.AUCTION_DEADLINE),
				m_localizer.formatDateTime(m_auction.getAuctionDeadLine(), m_view_tz)
		);

		Date deadline = m_shiftBidder.getDeadlineDate();
		if (deadline!=null) {
			m_auctionContainerPC.addRow(
					i18n(m_bundle, RmWebBundleKey.PERSONAL_DEADLINE),
					m_localizer.formatDateTime(deadline, m_view_tz)
			);
		}
        
        m_auctionContainerPC.setTableSummary(i18n(m_bundle, RmWebBundleKey.SHIFT_BID_AUCTION_TABLE_SUMMARY));
	}

	private void initRequestContainer() {
		boolean isViewMode = !isEditableMode();
		m_requestContainerPC = createFormPC("requestContainer");

		String title = i18n(m_bundle, RmWebBundleKey.SHIFTBID_REQUEST);
		m_requestContainerPC.setTitle(title);

		// Only display employee name if it is not the current user
		if (m_empName!=null && !m_empName.getID().equals(m_empID)) {
			m_requestContainerPC.addRow(
					i18n(m_bundle, RmWebBundleKey.EMPLOYEE_NAME_LABEL),
					m_localizer.formatName(m_empName)
			);
		}

		m_requestContainerPC.addRow(
				i18n(m_bundle, RmWebBundleKey.BID_NAME),
				createBidNameUI(isViewMode)
		);

		String preferenceLabel = HtmlUtil.createLabel(i18n(m_common_bundle, UIFWebBundleKey.PREFERENCE), PREF_FN);
		m_requestContainerPC.addRow(
				preferenceLabel,
				createPreferenceUI(isViewMode)
		);

		if (m_auction.getUsesScoring()) {
			String label = i18n(m_bundle, RmWebBundleKey.USE_POINTS);
			HtmlChoiceInput choice = HtmlChoiceInput.newInstance(USE_POINTS_FN,
					"true",label, m_sbRequest.getIsBonusUsed(), JSUtil.ON_CHANGE, true);
			choice.setIsDisabled(isViewMode);
            String choiceHtml = choice.getHtmlDisplay();
			m_requestContainerPC.addRow(label, choiceHtml);
		}

		if (MODE_EDIT == m_mode || MODE_VIEW == m_mode) {
			m_requestContainerPC.addRow(
					i18n(m_common_bundle, UIFWebBundleKey.COMMENT_HISTORY),
					createCommentHistoryPC()
			);
		}

		if (!isViewMode) {
			m_requestContainerPC.addRow(
					i18n(m_common_bundle, UIFWebBundleKey.COMMENTS),
					RequestUtil.createCommentUI(COMMENT_FN, m_comment, 100, 3, this, i18n(m_common_bundle, UIFWebBundleKey.COMMENTS))
			);
		}
        
        m_requestContainerPC.setTableSummary(i18n(m_bundle, RmWebBundleKey.SHIFT_BID_REQUEST_TABLE_SUMMARY));
	}

	private String createRankForBidMsg() {
		String sPattern = i18n(m_bundle, RmWebBundleKey.RANK_FOR_BID_WITH_SCORE_MSG);
		Object rankWithPoints = m_sbRequest.getOptMethods().getRankWithBonus().getFirst();
		Object rankWithoutPoints = m_sbRequest.getOptMethods().getRankWithoutBonus().getFirst();
		Object totalBids = m_sbRequest.getOptMethods().getRankWithBonus().getSecond();
		Object[] args = new Object[] {
			rankWithPoints,
			rankWithoutPoints,
			totalBids
		};
		return BPMessageFormat.format(sPattern, args);
	}

	private String createBidNameUI(boolean isViewMode) {
		String value = m_sbRequest.getName();
		if (isViewMode) return value;

		m_inputPC.setInputFN(BID_NAME_FN);
		m_inputPC.setInputFieldDesc(RmWebBundleKey.BID_NAME, RmWebBundleKey.BUNDLE_NAME);
		m_inputPC.setInputValue(value);
		m_inputPC.setIsRequired(true);
		m_inputPC.setMaxLength(DBInfo.SHIFTBIDAUCTION_NAME_MAXLENGTH);
		m_inputPC.initForDisplay();
		return m_inputPC.toString();
	}

	private String createPreferenceUI(boolean isViewMode) {
		String value = String.valueOf(m_sbRequest.getPreference());
		if (isViewMode) {
			return value;
		}

		return ShiftBidUtil.createBidPrefDD(PREF_FN, value, m_maxPrefValue) +
				HtmlUtil.NBSP +	i18n(m_bundle, RmWebBundleKey.BID_PREF_DESC);
	}
