package com.bluepumpkin.web.rm.auction;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.Iterator;
import java.util.ResourceBundle;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.ejb.bbm.campaign.CampaignScopeManagerProxy;
import com.bluepumpkin.ejb.bbm.workresource.OrganizationScopeManagerProxy;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidder;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.rm.auction.filter.BidOptionsFilterPC;

/**
 * File:         BidOptionsPM
 * Description:  Page model for the Bid Options list in shift bidding auctions 
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       pbosin, David Su
 * @version      1.0
 * Created on July 29, 2002, 1:04 PM
 */
public class BidOptionsPM extends BaseBidOptionsPM {
	private static String FORM_ACTION = "rm_sa_bid_options";
	private Collection m_bidderList = Collections.emptyList();
	private ID m_campaignID;
	private ID m_empOrgID;

	/**
	 * Constructor
	 */
	public BidOptionsPM(RequestContext context) {
		super(context, FRAMED_MODEL, FORM_ACTION);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, BidOptionsPM.class);
	}

	/**
	 * Load Data
	 * @throws Exception
	 */
	@Override
	protected void loadData() throws Exception {
		try {
			if (hasSelectedID()) {
				ID auctionID = getSelectedIDObject();
				loadBiddersList(auctionID);

				// Find Bidder Employee Organization ID
				m_empID = findEmpID(m_bidderID, m_bidderList);
				m_empOrgID = AuctionMH.getOrgIDForEmp(m_context, m_empID);

				if (m_bidderList.isEmpty() && m_isPageEnabled) {
					// Add Warning that no bidders are available for bidding
					addPageMessage(Message.WARNING_TYPE, RmWebBundleKey.NO_EMP_AVAILABLE_FOR_BIDDING,
							RmWebBundleKey.BUNDLE_NAME);
				} else if (m_bidderID == null || m_empID == null) {
					// Add Message to select Employee
					addPageMessage(Message.RESPONSE_TYPE, RmWebBundleKey.SELECT_EMP_FOR_BIDDING,
							RmWebBundleKey.BUNDLE_NAME);
				} else {
					loadDataForBidder(auctionID, m_bidderID, m_empID, true);
					m_campaignID = m_auction.getOptMethods().getCampaignID();

					//Add the Bid Options filter component
					m_filterPC = new BidOptionsFilterPC(m_context, auctionID, m_bidderID);
					m_filterPC.setSummaryString(getSortString());
					addChildComponent(m_filterPC);
				}
			} else {
				addPageMessage(Message.RESPONSE_TYPE, RmWebBundleKey.SELECT_AUCTION_MSG,
						RmWebBundleKey.BUNDLE_NAME);
			}
		} catch (RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataErrorAndLogAsDebug(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
		}
	}

	private void loadBiddersList(ID auctionID) throws Exception { // NOSONAR
		if (m_bidderList.isEmpty()) {
			m_bidderList = AuctionMH.getBiddersAddedToAuction(auctionID);
		}
	}

	@Override
	public ID getEmployeeID() {

		if (m_empID != null) {
			return m_empID;
		}

		if (hasSelectedID()) {
			ID auctionID = getSelectedIDObject();

			try {
				loadBiddersList(auctionID);
			} catch (Exception e) {
				throw RmRuntimeException.toRuntimeException(e);
			}

			// Find Bidder Employee Organization ID
			m_empID = findEmpID(m_bidderID, m_bidderList);
		}

		return m_empID;
	}

	/**
	 * Initialize Content Title
	 */
	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		if (!hasSelectedID() || !m_isPageEnabled) 
			return;

		if (!m_bidderList.isEmpty()) {
			if (m_filterPC != null)
				m_contentTitlePC.addActiveComponent(m_filterPC);

			ResourceBundle bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);
			String bidderLbl = HtmlUtil.createLabel(i18n(bbmBundle, BbmWebBundleKey.Employee), BIDDER_ID_FN);
			String bidderSelector = getBidderSelector();
			m_contentTitlePC.addActiveComponent(bidderLbl, bidderSelector, true);
		}
	}

	
	/**
	 * Return Auction Name UI
	 */
	@Override
	protected String getAuctionNameUI() {
		return m_auction==null?"":m_auction.getName();
	}

	/**
	 * Return Deadline to be shown in Content Title
	 */
	@Override
	protected Date getDeadline() {
		return m_auction.getAuctionDeadLine();
	}

	/**
	 * Return true if current user is authorized to configure entities in a page
	 */
	@Override
	protected boolean isAuthorizedToConfigure() {
		User user = m_context.getUser();

		return !m_bidderList.isEmpty() && (
				user.isAuthorized(PrivilegeKeys.SB_MODIFYAUCTIONS, CampaignScopeManagerProxy.getScopeID(m_campaignID)) ||
				user.isAuthorized(PrivilegeKeys.SB_MODIFYSHIFTBIDSFOREMP, OrganizationScopeManagerProxy.getScopeID(m_empOrgID)));
	}

	//////////////////////////////////////////////////////////////////////////////
	// Helper Methods
	//////////////////////////////////////////////////////////////////////////////
	private static ID findEmpID(ID bidderID, Collection bidderList) {
		ID empID = null;

		if (bidderID != null && !bidderList.isEmpty()) {
			for (Iterator it = bidderList.iterator(); it.hasNext();) {
				ShiftBidder bidder = (ShiftBidder) it.next();
				if (bidder.getID().equals(bidderID)) {
					empID = bidder.getEmployeeID();
					break;
				}
			}
		}

		return empID;
	}

	private String getBidderSelector() {
		ArrayList options = new ArrayList(m_bidderList.size());
		options.add(new StringsPair("", ""));
		for(Iterator it=m_bidderList.iterator(); it.hasNext(); ) {
			ShiftBidder shiftBidder = (ShiftBidder) it.next();
			String key = shiftBidder.getID().toString();
			String value = m_localizer.formatName(shiftBidder.getOptMethods().getEmployeeBasic());
			options.add(new StringsPair(key, value));
		}

		String selectedID = m_bidderID == null ? null : m_bidderID.toString();
		return HtmlUtil.makeSelectInput(BIDDER_ID_FN, selectedID, JSUtil.REFRESH_PAGE,
				options);
	}

}
