package com.bluepumpkin.web.rm.auction;

public class MyBidOptionsPM extends BaseBidOptionsPM {
	private static String FORM_ACTION = "my_requests_bids";

	private Collection m_auctionOptionList = Collections.EMPTY_LIST;
	private ShiftBidder m_shiftBidder;

	/**
	 * Constructor
	 */
	public MyBidOptionsPM(RequestContext context) {
		super(context, BASIC_MODEL, FORM_ACTION);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, MyBidOptionsPM.class);
	}

	/**
	 * Load Data
	 * @throws Exception
	 */
	@Override
	protected void loadData() throws Exception {
		try {
			setPageDirty(false);
			Collection auctionList = AuctionMH.getAuctionsForBidder(m_context.getUser());
			if (auctionList.isEmpty()) {
				addPageMessage(Message.INFO_TYPE,
						RmWebBundleKey.NOT_ADDED_TO_AUCTIONS, RmWebBundleKey.BUNDLE_NAME);
				return;
			}

			initAuctionData(auctionList);

			if (hasSelectedID()) {

				ID auctionID = getSelectedIDObject();
				m_empID = m_context.getUser().getEmployeeID();

				if (m_shiftBidder == null) {
					m_shiftBidder = AuctionMH.getShiftBidder(auctionID, m_empID);
				}

				if (m_shiftBidder != null) {

					m_bidderID = m_shiftBidder.getID();
					super.loadDataForBidder(auctionID, m_bidderID, m_empID, false);

					//Add the Bid Options filter component
					m_filterPC = new BidOptionsFilterPC(m_context, auctionID, m_bidderID);
					m_filterPC.setSummaryString(getSortString());
					addChildComponent(m_filterPC);
				}
			}
		} catch (RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataErrorAndLogAsDebug(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
		}
	}

	@Override
	public ID getBidderID() {

		if (m_bidderID != null) {
			return m_bidderID;
		}

		ID auctionID = getSelectedIDObject();
		m_empID = m_context.getUser().getEmployeeID();
		try {
			m_shiftBidder = AuctionMH.getShiftBidder(auctionID, m_empID);
		} catch (Exception e) {
			throw new RmRuntimeException(e);
		}
		return m_shiftBidder.getID();
	}

	@Override
	public ID getEmployeeID() {
		if (m_empID != null) {
			return m_empID;
		}
		m_empID = m_context.getUser().getEmployeeID();
		return m_empID;
	}

	/**
	 * Initialize Auction Data
	 */
	private void initAuctionData(Collection auctionList) {
		m_auctionOptionList = new ArrayList(auctionList.size());
		String selectedID = getSelectedID();
		String matchedID = null;
		for(Iterator it=auctionList.iterator(); it.hasNext(); ) {
			ShiftBidAuction auction = (ShiftBidAuction)it.next();
			String key = auction.getID().toString();
			if (matchedID==null || selectedID.equals(key)) matchedID=key;

			String value = auction.getName();
			m_auctionOptionList.add( new StringsPair(key, value) );
		}
		setSelectedID(matchedID);
	}

	/**
	 * Return Auction Name UI
	 */
	@Override
	protected String getAuctionNameUI() {
		String selectedID = getSelectedID();
        
		String title = HtmlUtil.createHiddenLabel(i18n(m_bundle, RmWebBundleKey.AUCTION), SELECTED_ID_FN);

        return title + " " + HtmlUtil.makeSelectInput(SELECTED_ID_FN, m_auctionOptionList,
                null, selectedID, "", JSUtil.REFRESH_PAGE, true);
	}

	/**
	 * Return Deadline to be shown in Content Title
	 */
	@Override
	protected Date getDeadline() {
		if (m_shiftBidder != null)
			return m_shiftBidder.getDeadlineDate();
		else
			return null;
	}

	/**
	 * Initialize Content Title
	 */
	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		
		String name = getAuctionNameUI();
		Date deadline = getDeadline();
		//JT
		String bidApprovalOrder = getBidApprovalOrderString();
		
		AuctionUtil.initContentTitle(m_context, m_isPageEnabled, name, deadline, m_contentTitlePC, m_filterPC, bidApprovalOrder);
	}
	
	
	

	private String getBidApprovalOrderString() {
		String bidApprovalOrder = "";
		if (LicenseUtil.isAdvancedRMLicense() &&  m_auction != null && m_auction.getIsDisplayBasedOn()) {
			int bidOrderBasis = this.m_auction.getValueDisplayBasedOn();
			Pair<Integer, Integer> order = ShiftBidAuctionUtil.calculateBidApprovalOrder(m_auction, this.m_shiftBidder, bidOrderBasis);
			if(order != null){
				Object[] params = { order.getFirst(), order.getSecond() };
				bidApprovalOrder = this.m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.BID_APPROVAL_MSG, params);
			}
		}
		
		return bidApprovalOrder;

	}
	
	@Override
	protected String getMinimumBidCountStringForEmployee() {
		String minBidString = "";
		if (LicenseUtil.isAdvancedRMLicense() && m_auction.getIsRequiredBids()) {
			Integer numBidsRequired = getMinimumNumberOfBids();
			String numBidsString = numBidsRequired.toString();
			Object[] params = { numBidsString };
			minBidString = this.m_localizer.i18n(RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.MINIMUM_BID_MSG_EMPLOYEE, params);
		}

		return minBidString;

	}
	
	/**
	 * Return true if current user is authorized to configure entities in a page
	 */
	@Override
	protected boolean isAuthorizedToConfigure() {
		return m_context.getUser().isAuthorized(PrivilegeKeys.SB_MODIFYPERSONALBIDS_ID);
	}
}
