package com.bluepumpkin.web.rm.auction;

import java.io.OutputStream;
import java.nio.charset.Charset;
import java.util.Collection;
import java.util.Collections;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

import org.json.JSONObject;

import weblogic.wsee.util.StringUtil;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.core.pagecomponent.DropDownSelectionPC;
import com.witness.web.uif.Log;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.pagecomponent.PageComponent;
import com.witness.web.uif.system.RequestContext;

public class BidDisplayOptionsReloadRH extends RequestHandler {
	protected static final Category LOGGER = Log.initCategory(PageComponent.class.getName());
	private final static String JSON_CONTENT_TYPE = "text/json; charset=UTF-8";
	public static final String RELOAD_CONTROL_DATA_URL = "reload_display_options";

	@Override
	public void processRequest(RequestContext context) throws Exception {
		//get controls data in JSON format
		JSONObject response = getBackgroundResponse(context);
		//write XML data to the HttpServletResponse
		HttpServletResponse res = context.getResponse();
		OutputStream out = res.getOutputStream();
		res.getOutputStream();
		res.setStatus(HttpServletResponse.SC_OK);
		res.setContentType(JSON_CONTENT_TYPE);

		out.write(response.toString().getBytes(Charset.forName("UTF-8")));

		return;
	}

	/**
	 * get  response data to refresh controls
	 * @return JSON Response string
	 *
	 */

	private JSONObject getBackgroundResponse(RequestContext context) {
		JSONObject response = new JSONObject();
		HttpServletRequest request = context.getRequest();
		String checkedScore = request.getParameter("checkedScore");
		Collection<StringsPair> displayOptions = Collections.emptyList();
		try {
			if (!StringUtil.isEmpty(checkedScore) &&checkedScore.equalsIgnoreCase("true")) {
				displayOptions=AuctionUtil.getDisplayOptions(true, context);
			} else {
				displayOptions=AuctionUtil.getDisplayOptions(false, context);
			}
			DropDownSelectionPC<StringsPair> displaySelectorPC = new DropDownSelectionPC<StringsPair>(context,
					AuctionSettingsPM.VALUE_DISPLAY_BASED_ON_FN);
			for (StringsPair a : displayOptions) {
				displaySelectorPC.addOption(a.getKey(), a.getValue(), a,false);
			}
			String htmlDisplay = displaySelectorPC.getHtmlDisplay();
			int index = htmlDisplay.indexOf("<span BP_OBJ_TYPE=\"DROP_DOWN_PICKER\"");
			String selectorHtml = htmlDisplay.substring(index, htmlDisplay.length());
			response.put("Result", selectorHtml);
			response.put("Status", "OK");

		} catch (Exception e) {
			LOGGER.error(e.getLocalizedMessage(), e);
		}

		return response;
	}

}
