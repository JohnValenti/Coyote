/**
 * File:         AuctionSummaryPM
 * Description:  Page model for the extended list of shift bidding auctions
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc

 * @version      1.0

public class AuctionSummaryPM extends SummaryPM {
	private static String FORM_ACTION = "auction_summary";

	private MultiColListPC m_auctionListPC;
	private ResourceBundle m_bundle;
	private ResourceBundle m_bbm_bundle;
	private Collection m_auctions = Collections.emptyList();
	private User m_user;
	private TimeZone m_tz;

	private String m_fullScheduleLbl;
	private String m_shiftsLbl;
	private String m_openLbl;
	private String m_closedLbl;

	/**
	 * Constructor
	 */
	public AuctionSummaryPM(RequestContext context) {
		super(context);
		setFormAction(FORM_ACTION);
		m_bundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);
		m_bbm_bundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);

		// Init and add Multi Column List sub-component
		m_auctionListPC = new MultiColListPC(context);
		m_auctionListPC.setMultiRowSelectable(false);
		addChildComponent(m_auctionListPC);

		m_fullScheduleLbl = i18n(m_bundle, RmWebBundleKey.FULL_SCHEDULE);
		m_shiftsLbl = i18n(m_bundle, RmWebBundleKey.SHIFTS);
		m_openLbl = i18n(m_common_bundle, UIFWebBundleKey.STATUS_OPEN);
		m_closedLbl = i18n(m_common_bundle, UIFWebBundleKey.STATUS_CLOSED);

		m_user = context.getUser();
		m_tz = m_context.getViewingTimeZone();

		// Add JavaScript Variables
		addJSVariable(DELETE_CONFIRMATION, m_bundle, RmWebBundleKey.DELETE_AUCTION_CONFIRM);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, AuctionSummaryPM.class);
	}

	/**
	 * Initialize Selectable Items
	 *
	 * @throws Exception
	 */
	@Override
	protected void loadData() throws Exception {
		try {
			m_auctions = AuctionMH.getAuctionsForManagerWithCampaign(m_context.getUser());
		} catch (RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataErrorAndLogAsDebug(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
		}
	}

	/**
	 * Initialize Content Title
	 */
	@Override
	protected void initContentTitle() {
		m_contentTitlePC.setPageTitle(i18n(m_bundle, RmWebBundleKey.RM_SA_AUCTION_SUMMARY));
	}

	private GenericHeaderData addHeader(TableHeaderPC header, ResourceBundle bundle, String rbKey) {
		return header.addColumnHeaderData(rbKey, i18n(bundle, rbKey));
	}

	/**
	 * Initialize Header
	 */
	@Override
	protected void initHeader() {
		TableHeaderPC header = m_auctionListPC.getHeader();
		addHeader(header, m_bundle, RmWebBundleKey.RM_SA_AUCTION_NAME);
		addHeader(header, m_common_bundle, UIFWebBundleKey.STATUS);
		addHeader(header, m_bbm_bundle, BbmWebBundleKey.CAMPAIGN);
		addHeader(header, m_common_bundle, UIFWebBundleKey.SCHEDULE_PERIOD);
		addHeader(header, m_bundle, RmWebBundleKey.BIDDING_FOR);
		addHeader(header, m_common_bundle, UIFWebBundleKey.DEADLINE);
		addHeader(header, m_bundle, RmWebBundleKey.NONE_APPR_BONUS).setIsForNumeric(true);
		addHeader(header, m_bundle, RmWebBundleKey.USE_SCORING);
		addHeader(header, m_bbm_bundle, BbmWebBundleKey.BY_ORG);
		addHeader(header, m_common_bundle, UIFWebBundleKey.AVAILABLE).setIsForNumeric(true);
		addHeader(header, m_bundle, RmWebBundleKey.PEOPLE_BIDDING).setIsForNumeric(true);
	}

	/**
	 * Initialize Selectable Items
	 */
	@Override
	protected void initSelectableItems() {
		if (m_auctions.isEmpty()) {
			return;
		}

		ArrayList listModel = new ArrayList(m_auctions.size());
		for (Iterator it = m_auctions.iterator(); it.hasNext();) {
			ShiftBidAuction auction = (ShiftBidAuction) it.next();
			addRowData(auction, listModel);

		}
		m_auctionListPC.addSelectedRowID(getSelectedID());

		m_auctionListPC.setListModel(listModel);
	}

	/**
	 * Return the Selectable Items
	 */
	@Override
	public SelectableItems getSelectableItems() {
		return m_auctionListPC;
	}

	/**
	 * Initialize toolbar
	 */
	@Override
	protected void initToolbar() {
		useDefaultToolbar(false);
	}

	/**
	 * is the add action enabled?
	 */
	@Override
	protected boolean isAddActionEnabled() {
		return true;
	}

	//////////////////////////////////////////////////////////////////////////////
	// Helper Methods
	//////////////////////////////////////////////////////////////////////////////
	private void addRowData(ShiftBidAuction auction, Collection listModel) {
		String rowID = auction.getID().toString();
		String name = HtmlUtil.escapeAttributeValue(auction.getName());
		String bidFor = auction.getIsFullPeriodSchedule() ? m_fullScheduleLbl : m_shiftsLbl;

		boolean isOpen = auction.getStatus() == ShiftBidAuction.STATUS_OPEN_FOR_BIDDING;
		String status = isOpen ? m_openLbl : m_closedLbl;
		Campaign camp = auction.getOptMethods().getCampaign();
		SchedulingPeriod sp = auction.getOptMethods().getSP();
		DateWrapper deadlineDW = new DateWrapper(auction.getAuctionDeadLine(), m_localizer, m_tz, RegionalFormatBundleKey.DATE_FORMAT,
				RegionalFormatBundleKey.TIME_FORMAT);

		int iPoints = auction.getBonusForNoBidsAccepted();
		boolean bUseScoring = auction.getUsesScoring();
		boolean bBidOnOrg = auction.getBidsOnlyWithinOrg();
		int iAvail = auction.getNumOfBiddableScheduleInstances();
		int iBidders = auction.getNumOfBidders();

		DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData(rowID);
		node.add(name);
		node.add(status);
		node.add(camp.getName());
		node.add(SchedulingPeriodUtil.getSPDisplayWrapper(sp, m_localizer, m_tz));
		node.add(bidFor);
		node.add(deadlineDW);
		node.add(new NumberWrapper(iPoints, m_localizer), CSSUtil.STYLE_NUMERIC);
		node.add(m_localizer.formatYesNo(bUseScoring));
		node.add(m_localizer.formatYesNo(bBidOnOrg));
		node.add(new NumberWrapper(iAvail, m_localizer), CSSUtil.STYLE_NUMERIC);
		node.add(new NumberWrapper(iBidders, m_localizer), CSSUtil.STYLE_NUMERIC);
		listModel.add(node);

		ID campID = auction.getOptMethods().getCampaign().getID();
		boolean isEditable = m_user.isAuthorized(PrivilegeKeys.SB_MODIFYAUCTIONS, CampaignScopeManagerProxy.getScopeID(campID));
		node.getNodeAttributes().put(IS_EDITABLE, String.valueOf(isEditable));
	}
}
