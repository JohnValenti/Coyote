package com.bluepumpkin.ejb.bbm.workload.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Collections;

/**
 * This is a POJO for storing the Paginated Queue Data
 * 
 * @author Jinal Shah
 */
public class QueuePaginationData implements Serializable {

	private static final long serialVersionUID = -4163471992636653797L;
	private Collection<Queue> queues = Collections.emptyList();
	private int total = 0;

	/**
	 * @return Returns the paginated queues.
	 */
	public Collection<Queue> getQueues() {
		return queues;
	}

	/**
	 * @param queues
	 *            The paginated groups to set.
	 */
	public void setQueues(Collection<Queue> queues) {
		this.queues = queues;
	}

	/**
	 * @return Returns the total.
	 */
	public int getTotal() {
		return total;
	}

	/**
	 * @param total
	 *            The total number of records to be set.
	 */
	public void setTotal(int total) {
		this.total = total;
	}
}
