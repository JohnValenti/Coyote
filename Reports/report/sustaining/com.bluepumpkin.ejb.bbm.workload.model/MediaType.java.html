package com.bluepumpkin.ejb.bbm.workload.model;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;

/**
 * An enumeration containing the media types that can be associated to a Queue.
 */
public enum MediaType {
	IMMEDIATE,
	DEFERRED,
	OUTBOUND,
	PROJECT,
	UNKNOWN,
	NULL;	//NULL media type is used to reference a "combined-combined" selection, in other words the combination
			//  of several combined queues.  This selection technically does not have a single media type, so we want
			//  a  NULL media type to represent this.  In this case, the media Id actually is null in the database.
	
	public static MediaType get(Media media){
		if (media == null) return NULL;
		ID id = media.getId();
		return get(id);
	}
	
	public static MediaType get(ID mediaId){
		if (mediaId == null) return NULL;
		if (mediaId.equals(Media.MEDIA_ID_FACETOFACE) ||
			mediaId.equals(Media.MEDIA_ID_PHONE) ||
			mediaId.equals(Media.MEDIA_ID_CHAT) ||
			mediaId.equals(Media.MEDIA_ID_VOIP)){
			return MediaType.IMMEDIATE;
		}
		else if (mediaId.equals(Media.MEDIA_ID_OPERATIONS) ||
				mediaId.equals(Media.MEDIA_ID_EMAIL) ||
				mediaId.equals(Media.MEDIA_ID_FAX) ||
				mediaId.equals(Media.MEDIA_ID_CALLBACK) ||
				mediaId.equals(Media.MEDIA_ID_SOCIAL_POST) || 
				mediaId.equals(Media.MEDIA_ID_RESOLUTION)) {
			return MediaType.DEFERRED;
		}
		else if (mediaId.equals(Media.MEDIA_ID_PHONE_OUTBOUND)) {
			return MediaType.OUTBOUND;
		}
		else if (mediaId.equals(Media.MEDIA_ID_PROJECT)) {
			return MediaType.PROJECT;
		}
		return MediaType.UNKNOWN;
	}

	/**
	 * Groups the given spQueues by MediaType and returns the grouping as a map keyed on the
	 * associated MediaType.
	 * If an SPQueue does not have a media type associated to it, it will be grouped under MediaType.UNKNOWN.
	 * 
	 * @return Map: key = media type, value = Collection of SPQueues associated to that media type, a subset
	 * of the parameter spQueues.
	 */
	public static Map<MediaType, Collection<SPQueue>> getMediaTypeMapForSPQueues(Collection<SPQueue> spQueues) {
		Map<MediaType, Collection<SPQueue>> retVal = new HashMap<MediaType, Collection<SPQueue>>();
		for (SPQueue spQueue : spQueues) {
			if (spQueue.getMediaID() == null) {
				if (retVal.containsKey(MediaType.UNKNOWN) == false) {
					retVal.put(MediaType.UNKNOWN, new ArrayList<SPQueue>());
				}
				retVal.get(MediaType.UNKNOWN).add(spQueue);
			} else {
				MediaType mediaType = get(spQueue.getMediaID());
				if (retVal.containsKey(mediaType) == false) {
					retVal.put(mediaType, new ArrayList<SPQueue>());
				}
				retVal.get(mediaType).add(spQueue);
			}
		}
		return retVal;
	}
}
