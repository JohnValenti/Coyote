import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.common.datatypes.ID;

import java.util.Date;

public class CombinedQueueStaffing extends ValueObjectBase
{
    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new CombinedQueueStaffingFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }
		private ID mediaID;

    public CombinedQueueStaffing()
    {
       super();
    }

		public void setMediaID( ID id )
		{
				this.mediaID = id;
		}

	public ID getMediaID()
	{
		return this.mediaID;
	}

    public String getSPQueueID()
    {
       return getFormattedFieldValue(CombinedQueueStaffingFieldInfo.COMBINEDQUEUE_SPQUEUEID);
    }

		public Date getTimeInterval()
		{
				return getFieldValueDate(CombinedQueueStaffingFieldInfo.COMBINEDQUEUE_TIMEINTERVAL);
		}

		public float getStaffing()
		{
				Float val = (Float)getFieldValue(CombinedQueueStaffingFieldInfo.COMBINEDQUEUE_STAFFING);
				return val != null ? val.floatValue() : 0;
		}

    public void setSPQueueID(String strValue)
    {
       setFieldValue(CombinedQueueStaffingFieldInfo.COMBINEDQUEUE_SPQUEUEID, strValue);
    }

		public void setTimeInterval( Date value)
		{
				setFieldValue(CombinedQueueStaffingFieldInfo.COMBINEDQUEUE_TIMEINTERVAL, value);
		}

		public void setStaffing(float value)
		{
				setFieldValue(CombinedQueueStaffingFieldInfo.COMBINEDQUEUE_STAFFING, value);
		}

}
