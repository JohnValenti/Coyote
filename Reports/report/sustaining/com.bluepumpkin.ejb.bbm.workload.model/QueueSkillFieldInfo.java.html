public class QueueSkillFieldInfo extends FieldInfo {

	// field id constants

    public static final int QUEUESKILL_QUEUEID = 1;
    public static final int QUEUESKILL_SKILLID = 2;

    private static final int QUEUE_NUMOFFIELDS = 3;

    private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[QUEUE_NUMOFFIELDS];

      m_arrFieldInfo[QUEUESKILL_SKILLID] = new FieldInfoEntry("SKILLID", PrimitiveTypes.ID);
    }

    public QueueSkillFieldInfo() {
        super();
    }

    // implement abstract methods in parent class
    public String getTableName() { return m_strTableName; }
    public int getIDFieldIndex() { return QUEUESKILL_ID; }
    public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }

}
