package com.bluepumpkin.ejb.rm.setup.autoprocessing.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;


/**
 * Title:        AppliedAutoProcessingRule
 * Description:  Represents a AppliedAutoProcessingRule for the Autoprocessing rule.
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc.
 * @author       Shailendra Sharma
 * @version 	 1.0
 */

public class AppliedAutoProcessingRule extends ValueObjectBase {

	private static FieldInfo m_fieldInfo = new AppliedAutoProcessingRuleFieldInfo();

    /**
     * The boolean value telling us whether the user can change the isApplied value for this rule.
     */
    private boolean m_canChangeApply = false; 
    
    
	public AppliedAutoProcessingRule() {
		super();
	}

	protected FieldInfo getFieldInfo() { return m_fieldInfo; }


	// getters

	public ID getOrganizationID() {
		return getFieldValueID(AppliedAutoProcessingRuleFieldInfo.APPLIEDAUTOPROCESSINGRULE_I_ORGANIZATIONID);
	}

	public ID getAutoProcessingRuleID() {
		return getFieldValueID(AppliedAutoProcessingRuleFieldInfo.APPLIEDAUTOPROCESSINGRULE_I_AUTOPROCESSINGRULEID);
	}


	public boolean getApplied() {
		return getFieldValueBoolean(AppliedAutoProcessingRuleFieldInfo.APPLIEDAUTOPROCESSINGRULE_B_ISAPPLIED);
	}


	// setters

	public void setOrganizationID(ID pOrganizationId) {
		setFieldValue(AppliedAutoProcessingRuleFieldInfo.APPLIEDAUTOPROCESSINGRULE_I_ORGANIZATIONID, pOrganizationId);
	}

	public void setAutoProcessingRuleID(ID pAutoProcessingRuleId) {
		setFieldValue(AppliedAutoProcessingRuleFieldInfo.APPLIEDAUTOPROCESSINGRULE_I_AUTOPROCESSINGRULEID, pAutoProcessingRuleId);
	}


	public void setApplied(boolean pIsApplied) {
		setFieldValue(AppliedAutoProcessingRuleFieldInfo.APPLIEDAUTOPROCESSINGRULE_B_ISAPPLIED, pIsApplied);
	}
    
    /**
     * Returns the boolean value telling us whether the user can change the isApplied value for this rule.
     */
    public boolean getCanChangeApply() {
        return m_canChangeApply;
    }
    
    /** 
     * Sets the boolean value telling us whether the user can change the isApplied value for this rule.
     */
    public void setCanChangeApply(boolean  pCanChangeApply) {
        m_canChangeApply = pCanChangeApply;
    }


}
