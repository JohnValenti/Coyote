package com.bluepumpkin.web.fs.calendar;

import java.util.Collection;

import com.bluepumpkin.web.core.applet.AppletPC;
import com.bluepumpkin.web.fs.keys.FsKeys;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.system.RequestContext;

@SuppressWarnings("serial")
public class CalendarAppletPC extends AppletPC {

	//FIXME: This is never accessed anywhere.
    protected boolean canSchedule = false;
    
    public CalendarAppletPC(RequestContext context) {
        super(context);
        setName("CalendarAppletPC");
        
		Collection authorizedCampaignsForSchedule = context.getUser().getAuthorizedScopes(PrivilegeKeys.FS_VIEWORGSCHEDULES);
        canSchedule = authorizedCampaignsForSchedule != null && authorizedCampaignsForSchedule.size() > 0;
        // Check the query string for startup parameters and copy them to the HTML
        // tag if needed
		String actionParameter = context.getParameter(FsKeys.CALENDAR_PARAMETER_ACTION);
		if (FsKeys.CALENDAR_ACTION_DRILLDOWN.equals(actionParameter)) {
			// Drill down to a specified SP and date
			String SPID = context.getParameter(FsKeys.CALENDAR_PARAMETER_SPID);
			String date = context.getParameter(FsKeys.CALENDAR_PARAMETER_DATE);
			if (SPID != null && date != null) {
				addParameter(FsKeys.CALENDAR_PARAMETER_ACTION, FsKeys.CALENDAR_ACTION_DRILLDOWN);
				addParameter(FsKeys.CALENDAR_PARAMETER_SPID, SPID);
				addParameter(FsKeys.CALENDAR_PARAMETER_DATE, date);
			}
		}
    }

    @Override
    public String getCustomJavaScript()
    {              
    	String warning = m_localizer.i18n(m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME), FsWebBundleKey.FS_CAL_CONFIRM_NOTPUBLISH);
    	return "parent.LOSE_CHANGES_WARNING =  \"" + warning + "\";"
    		+ "var LOSE_CHANGES_WARNING =\"" + warning + "\";";
    }
}
