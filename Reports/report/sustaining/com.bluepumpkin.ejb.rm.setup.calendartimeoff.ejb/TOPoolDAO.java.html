package com.bluepumpkin.ejb.rm.setup.calendartimeoff.ejb;

/**

public final class TOPoolDAO extends DAOBase<TOPool> {

	private static FieldInfo fieldInfo = new TOPoolFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() { return fieldInfo; }

	public TOPoolDAO() {
		super();
	}

	public TOPoolDAO(Jdmo dmo) {
		super(dmo);
	}

	@Override
	protected TOPool createValueObject() {
		return (new TOPool());
	}

	public String getTOPoolName(ID idTOPool) throws BbmFinderException {
		String strSQL = "SELECT NAME from TOPOOL where ID = " + JdmoUtil.asSqlLiteral(idTOPool);
		try {
			JdmoRowset rs = m_dmo.createRowset(strSQL);
			if (rs.next()) {
				return rs.getString(1);
			}
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
		throw new BbmObjectNotFoundException();
	}

	public Collection<TOPool> getTOPoolsByIDs(Collection<ID> ids) throws BbmFinderException {
		StringBuilder strWhere = new StringBuilder("A.ID IN ");
		try {
			strWhere.append(m_dmo.createInClause(ids));
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
		return getObjects(new String(strWhere));
	}

	/**
	 * get TOPool ids given their names
	 *
	 * @param setTOPoolNames a hashset of organization names
	 * @return a HashMap of (TOPoolName, TOPoolID) pairs
	 */
	public HashMap getTOPoolIDsByNames(HashSet setTOPoolNames) throws BbmFinderException {
		return DAOUtil.getObjectIDsByNames(setTOPoolNames, "NAME", "TOPOOL", m_dmo);
	}

	/**
	 * Returns collection of TOPools which can be assigned to an employee of given orgID.
	 *
	 * @param orgID          ID of Organization for which TOPools are requested.
	 * @param inclParentOrgs return TOPools created at parent orgs as well.
	 * @return Returns collection of TOPools
	 * @throws com.bluepumpkin.ejb.bbm.base.BbmFinderException
	 *          if error occurs while reading from DB
	 */
	public Collection<TOPool> getTOPoolsForOrg(ID orgID, boolean inclParentOrgs) throws Exception {
		List<ID> orgIDs = new ArrayList<>();
		if (inclParentOrgs) {
			orgIDs.addAll(BbmManagerFactory.getWorkResourceManager().getOrganizationsParentsByIDs(Collections.singletonList(orgID)));
		}
		if (orgIDs != null && !orgIDs.contains(orgID)) {
			orgIDs.add(orgID);
		}
		StringBuilder strWhere = new StringBuilder("A.ORGANIZATIONID IN ").append(m_dmo.createInClause(orgIDs));
		return getObjects(new String(strWhere));
	}

	public Collection<ID> getOrgIDsUnderTOPool(ID pTOPoolID) throws Exception {
		TOPool toPool = (TOPool) getObjectByID(pTOPoolID);
		Set<ID> orgIDs = new HashSet<>();
		orgIDs.add(toPool.getOrganizationId());
		orgIDs.addAll(BbmManagerFactory.getWorkResourceManager().getOrganizationsChildrenByIDs(orgIDs));
		return orgIDs;
	}

	public Collection<TOPool> getTOPoolsForOrgIDs(Collection<ID> orgIDs) throws Exception {
		Set<ID> orgIDSet = new HashSet<>();
		orgIDSet.addAll(orgIDs);
		orgIDSet.addAll(BbmManagerFactory.getWorkResourceManager().getOrganizationsParentsByIDs(orgIDSet));
		StringBuilder strWhere = new StringBuilder("A.ORGANIZATIONID IN ").append(m_dmo.createInClause(orgIDSet));
		return getObjects(strWhere.toString());
	}

	public Collection<ID> getTOPoolIDsWithWaitListScanEnabled() throws Exception {
		String strSql = "Select ID from TOPOOL where ISWAITLIST=1 and ISWAITLISTSCAN=1";
		List<ID> arrItems = new ArrayList<>();
		try {
			JdmoRowset rs = m_dmo.createRowset(strSql);
			long lCount = m_lRecords;
			while (rs.next() && (lCount != 0)) {
				arrItems.add(rs.getID(1));
				lCount--;
			}
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
		return arrItems;
	}

	/**
	 * Returns collection of all TOPools in the system. Couldn't be simpler.
	 * @return Returns collection of TOPools
	 */
	public Collection<TOPool> getAllTOPools() throws Exception {
		return getObjects("");
	}

	public Map<ID, TOPoolAssignmentByActivity> getTOPoolAssignmentForDisplay(Collection<ID> employeeIds, Date asOfDate,
			List<ID> activityIDs) {

		List<TOPoolAssignmentByActivity> list = JdmoOrm.toObjectsFromStoredProcedure(TOPoolAssignmentByActivity.class, m_dmo,
				"RM_GetTOPoolByActivity", employeeIds, activityIDs, asOfDate);

		Map<ID, TOPoolAssignmentByActivity> result = new HashMap<ID, TOPoolAssignmentByActivity>();
		for (TOPoolAssignmentByActivity activity : list) {
			result.put(activity.getActivityID(), activity);
		}

		return result;
	}

	public void updateTOPoolAssignmentsByActivity(Collection<ID> employeeIDs, Date asOfDate,
			List<TOPoolAssignmentsByActivityUpdateParameters> updateParameters) {
		WriteOptions options = new WriteOptions("#ActivityList");
		JdmoOrm.insertToTempTable(m_dmo, TOPoolAssignmentsByActivityUpdateParameters.class, updateParameters, options);
		JdmoPreparedStatement.executeStoredProcedure(m_dmo, "RM_UpdateTOPoolByActivity", employeeIDs, asOfDate);
	}
}
