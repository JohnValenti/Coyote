	public List<EmpTOPoolAssignment> getValidEmpTOPoolAssignmentsByPool(ID toPoolID, Collection<ID> empIDs, Date dtStart, 
			Date dtEnd, boolean bCheckOverlap)
			throws Exception { //NOSONAR
		//@formatter: off
		String sql = "SELECT A.ID WORKRESOURCETOPOOLID, A.TOPOOLID, B.NAME, A.WORKRESOURCEID, "  
				+ "A.STARTTIME, A.ENDTIME , A.ACTIVITYID, C.ENDTIME TERMINATIONDATE "
				+ "FROM WORKRESOURCETOPOOL A "
				+ "inner join TOPOOL B on A.TOPOOLID = B.ID "
				+ "inner join EMPLOYEEAM C  on  C.ID=A.WORKRESOURCEID "
				+ " where ( C.ENDTIME IS NULL OR C.ENDTIME >= ? ) "
				+ "	 AND	 A.STARTTIME <= ?  AND  (A.ENDTIME IS NULL OR A.ENDTIME > ? ) ";

		//@formatter: on

		JdmoPreparedStatementBuilder builder = new JdmoPreparedStatementBuilder();
		builder.add(sql, dtStart, dtEnd, dtStart);

		if (empIDs != null && !empIDs.isEmpty()) {
			builder.add(" and A.WORKRESOURCEID in ? ", empIDs);
		}

		if (toPoolID != null) {
			builder.add(" AND  A.TOPOOLID= ? ", toPoolID);
		}

		builder.add(" ORDER BY A.STARTTIME ASC ");
		
		JdmoRowset res = JdmoPreparedStatement.createRowset(m_dmo, builder);
		
		ArrayList<EmpTOPoolAssignment> wraList = new ArrayList<EmpTOPoolAssignment>(2);
		while (res.next()) {
			EmpTOPoolAssignment eTOPoolAss = new EmpTOPoolAssignment();
			eTOPoolAss.setOverlap(false);
			eTOPoolAss.setID(res.getID("WORKRESOURCETOPOOLID"));
			eTOPoolAss.setTOPoolID(res.getID("TOPOOLID"));
			eTOPoolAss.setWorkResourceID(res.getID("WORKRESOURCEID"));
			eTOPoolAss.setTOPoolName(res.getString("NAME"));
			eTOPoolAss.setStartTime(TimeZoneUtil.toDate(res.getTimestamp("STARTTIME")));
			eTOPoolAss.setEndTime(TimeZoneUtil.toDate(res.getTimestamp("ENDTIME")));
			eTOPoolAss.setActivityID(res.getID("ACTIVITYID"));
			Date terminationDate = res.getTimestamp("TERMINATIONDATE");
			if (!checkValidityAndAdjustEndDate(eTOPoolAss, terminationDate)) {
				continue;
			}
			wraList.add(eTOPoolAss);
		}
		if (bCheckOverlap) {
			checkOverlap(wraList);
		}
		return wraList;
	}

	private boolean checkValidityAndAdjustEndDate(EmpTOPoolAssignment eTOPoolAss, Date terminationDate) {
		//If Employee termination date found in Employee AM then
		if (terminationDate != null) {
			//If Termination date is before start date then no point going ahead  for this TOPool
			if (eTOPoolAss.getStartTime() == null || terminationDate.before(eTOPoolAss.getStartTime())) {
				return false;
			}
			//If Termination date is before End Date for TOPool then adjust end date accordingly
			if ((eTOPoolAss.getEndTime() == null || terminationDate.before(eTOPoolAss.getEndTime()))) {
				eTOPoolAss.setEndTime(terminationDate);
			}
		}
		return true;
	}

	private void checkOverlap(ArrayList arr) throws BbmFinderException {
		boolean bHasOverlap = false;
		ResourceAssignmentAlert alert = null;
		int size = arr.size();
		for (int i = 1; i < size; i++) {
			EmpTOPoolAssignment item_i = (EmpTOPoolAssignment) arr.get(i);
			for (int j = i - 1; j >= 0; j--) {
				EmpTOPoolAssignment item_j = (EmpTOPoolAssignment) arr.get(j);
				if (item_i.overlap(item_j)) {
					item_i.setOverlap(true);
					item_j.setOverlap(true);
					if (!bHasOverlap) {
						alert = new ResourceAssignmentAlert(item_i.getParentID(),
						        item_i.getStartTime(), item_i.getEndTime(),
						        item_j.getStartTime(), item_j.getEndTime(),
						        true);
					}
					bHasOverlap = true;
				} else {
					if (!item_j.isOverlap())
						break;  // no overlap with previous ones
				}
			}
		}

		if (bHasOverlap) {
			BbmTimePeriodOverlapException e = new BbmTimePeriodOverlapException();
			e.setContent(alert);
			throw e;
		}
	}

	/**
	 * check whether there are employees in given an organization
	 */
	public boolean hasEmployeeInTOPool(ID toPoolID) throws BbmFinderException {
		if (toPoolID == null) return false;

		try {
			StringBuffer strSQL = new StringBuffer();
			strSQL.append("select count(1) NUMOFEMP from ").append(getFieldInfo().getTableName());
			strSQL.append(" where TOPOOLID = " + toPoolID);

			JdmoRowset rs = m_dmo.createRowset(new String(strSQL));

			if (rs.next()) {
				int num = rs.getInt(1);
				return (num > 0);
			}
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}

		return false;
	}

	/**
	 * Gets the time off calendar data configured for all days
	 * in the given range, inclusive
	 *
	 * @param empIDs  a collection of employee ids
	 * @param dtStart - beginning of the range
	 * @param dtEnd   - end of the range
	 * @return a hashmap of (employeeid, EmpTOPoolAssignment Collection) pairs
	 * @throws com.bluepumpkin.ejb.bbm.base.BbmFinderException
	 *
	 */
	public Map<ID, List<EmpTOPoolAssignment>> getTOPoolsForEmployees(Collection<ID> empIDs, Date dtStart, Date dtEnd) throws Exception {
		List<EmpTOPoolAssignment> list = getValidEmpTOPoolAssignments(empIDs, dtStart, dtEnd, false, false);
		HashMap<ID, List<EmpTOPoolAssignment>> mapResults = new HashMap<ID, List<EmpTOPoolAssignment>>();
		
		for(EmpTOPoolAssignment assignment : list ){
			ID employeeID = assignment.getWorkResourceID();
			if(employeeID == null ){
				continue;
			}
			
			List<EmpTOPoolAssignment> empTOPoolList = mapResults.get(employeeID);
			if (empTOPoolList == null) {
				empTOPoolList = new ArrayList<EmpTOPoolAssignment>();
				mapResults.put(employeeID, empTOPoolList);
			}
			
			empTOPoolList.add(assignment);
		}
		
		return mapResults;
	}

	public Set<ID> getTOPoolIDSetForEmployees(Collection<ID> empIDs, Date dtStart, Date dtEnd) throws Exception {
		Set<ID> set = new HashSet<>(empIDs.size());
		if (empIDs == null || empIDs.isEmpty()) {
			return set;
		}
		StringBuffer strSQL = new StringBuffer();
		strSQL.append("SELECT DISTINCT TOPOOLID  FROM WORKRESOURCETOPOOL A WHERE ");
		strSQL.append(" A.WORKRESOURCEID IN ").append(m_dmo.createInClause(empIDs));
		if (dtEnd != null)
			strSQL.append(" AND A.STARTTIME <= '").append(JdmoUtil.formatDBString(dtEnd)).append("' ");
		if (dtStart != null)
			strSQL.append(" AND  (A.ENDTIME IS NULL OR A.ENDTIME >= '").append(JdmoUtil.formatDBString(dtStart)).append("') ");
		JdmoRowset rs = m_dmo.createRowset(new String(strSQL));
		while (rs.next()) {
			set.add(rs.getID(1));
		}
		return set;
	}
	
	/**
	 * Return the end date of the last pool assignment after asOfDate for the empID.
	 * @param empID    - The employee ID that we need to find the previous pool assignment for.
	 * @param asOfDate - The date that the pool assignment must occur before.
	 * @return
	 */
	public Date getPreviousPoolAssignmentEnd(ID empID, Date asOfDate) throws Exception
	{
		Date prevPoolEnd = null;
		StringBuffer strSQL = new StringBuffer();
		strSQL.append("SELECT MAX(ENDTIME) FROM WORKRESOURCETOPOOL WHERE WORKRESOURCEID = ");
		strSQL.append(empID.toString());
		strSQL.append(" AND ENDTIME IS NOT NULL AND ENDTIME < '").append(JdmoUtil.formatDBString(asOfDate)).append("' ");

		JdmoRowset rs = m_dmo.createRowset(new String(strSQL));
		if (rs.next()) {
			prevPoolEnd = TimeZoneUtil.toDate(rs.getTimestamp(1));
		}
		return prevPoolEnd;
	}

	}
