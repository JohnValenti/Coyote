package com.bluepumpkin.ejb.rm.setup.calendartimeoff.ejb;

import java.util.Calendar;
import java.util.Date;
import java.util.TimeZone;


/**
 * Title:        MonthRange
 * Description:  Store two dates that represent the boundary of a month
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc.
 * @author       Howard Mullings
 * @version 1.0
 */

public class MonthRange {
	public Date startDate;
	public Date endDate;
	public MonthRange(int pMonth, int pYear) {
		Calendar c = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
		c.clear();
		c.set(Calendar.YEAR, pYear);
		c.set(Calendar.MONTH, pMonth);
		c.set(Calendar.DAY_OF_MONTH, c.getMinimum(Calendar.DAY_OF_MONTH));
		c.set(Calendar.HOUR_OF_DAY, c.getMinimum(Calendar.HOUR_OF_DAY));
		c.set(Calendar.MINUTE, c.getMinimum(Calendar.MINUTE));
		c.set(Calendar.SECOND, c.getMinimum(Calendar.SECOND));

		startDate = c.getTime();

		c.add(Calendar.MONTH, 1);
		endDate = c.getTime();
	}
}
