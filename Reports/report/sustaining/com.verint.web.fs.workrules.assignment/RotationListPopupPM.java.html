package com.verint.web.fs.workrules.assignment;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.PaginationPair;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.vo.DBSortField;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectIDListComparator;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.Rotation;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.widgets.FindToolbarPageComponent;
import com.verint.web.fs.workrules.WorkRulesModelHandler;
import com.verint.web.fs.workrules.WorkRulesSelectionPopupPM;
import com.verint.web.fs.workrules.WorkRulesSelectionPopupRH;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarTextButton;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.CacheUtil;

import java.util.*;

public class RotationListPopupPM extends WorkRulesSelectionPopupPM {
	private static final long serialVersionUID = 1L;

	public static final String FORM_ACTION = "rotation_list_popup";
	public static final String POPUP_ID = "POPUP_ROTATION";
	public static final String POPUP_ARGS = "selectedIDs,viewStartDate,viewEndDate,isInfoPopup";
	public static final String POPUP_ARG_SELECTED_IDS = "selectedIDs";
	public static final String POPUP_ARG_ROTATION_ID = "rotationID";
	public static final String POPUP_ARG_IS_INFO_POPUP = "isInfoPopup";
	public static final String ROTATION_POPUPLIST_SORTCOLUMN = RotationListPC.ROTATION_POPUPLIST_SORTCOLUMN;
	public static final String ROTATION_POPUPLIST_SORTORDER = RotationListPC.ROTATION_POPUPLIST_SORTORDER;

	private ArrayList<Rotation> rotations;
	private HashSet<ID> selectedIDs = new HashSet<ID>();
	private Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
	private Calendar viewStartDate;
	private Calendar viewEndDate;
	private boolean isInfoPopup = false;    //if this is an info popup, the add button will be removed and we will be displaying the rotations given in the selectedIDs argument
	private RotationListPC m_rotationListPC;

	public RotationListPopupPM(RequestContext context) {
		super(context);
		setIsMultiRowSelectable(false);
	}

	public static RotationListPopupPM getInstance(RequestContext context) {
		return (RotationListPopupPM) getInstance(context, RotationListPopupPM.class);
	}

	protected void initializeModel() {
		if (m_isInitialized) {
			return;
		}
		super.initializeModel();

		if (isInfoPopup == false) {
			FindToolbarPageComponent findToolBar = new FindToolbarPageComponent(m_context, getToolbar(),
					FIND_ITEM_FN, WorkRulesSelectionPopupRH.FIND_ACTION,
					getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_FIND_ROTATIONS));
			getToolbar().addToolbarComponent(findToolBar);
			getToolbar().addDivider();
			ToolbarTextButton addButton = m_toolbar.createCloseWindowButton(
					"addRotations", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
							FsWebBundleKey.FS_ADD_SELECTED_ROTATION), true);
			StringBuffer addButtonJS = new StringBuffer();
			addButtonJS.append(m_toolbar.getName())
					.append(".disableButton('addRotations'); ")
					.append("if(window.opener.rotationTable) {")
					.append(" RowInsertionUtil.addNewDynamicRow('rotation', eval('window.opener.rotationTable'), ")
					.append("'" + EmployeeAssignmentInlineEditComponentRH.URL + "', ")
					.append("'" + EmployeeAssignmentInlineEditComponentRH.PARAM_TYPE + "=" +
							EmployeeAssignmentInlineEditComponentRH.ROTATION_RULE_TYPE + "&")
					.append(EmployeeAssignmentInlineEditComponentRH.PARAM_START_DATE + "=" + viewStartDate.getTimeInMillis() + "&")
					.append(EmployeeAssignmentInlineEditComponentRH.PARAM_END_DATE + "=" + viewEndDate.getTimeInMillis())
					.append("'); }");

			addButton.setCustomJS(addButtonJS.toString());
			m_toolbar.addToolbarElement(addButton);
			m_toolbar.addCloseWindowTextButton("cancel", m_localizer.i18n(
					CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_CANCEL), true);
		}
	}

	protected void initUtilityPane() {
		getUtilityPane().setIsHelpEnabled(false);
	}

	protected void initContentTitle() {
		m_contentTitlePC.setPageTitle(getLocalizer().i18n(
				FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ROTATION_DETAILS));
	}

	protected void loadData() throws Exception {
		String employeeIDStr = CacheUtil.getCachedValue(m_context, CacheUtil.CACHE_MODE_SESSION,
				WorkRulesEmployeeAssignmentPM.EMPLOYEE_ID_LIST_CONTEXT_KEY);
		if (isInfoPopup) {
			rotations = new ArrayList<Rotation>(WorkRulesModelHandler.getRotationsByIDs(m_context, selectedIDs));
		} else if (employeeIDStr != null) {
			Collection<ID> employeeIDs = StringUtil.parseIDStringToCollection(employeeIDStr);
			if (isPaginationNavPressed()) {
				rotations = new ArrayList<Rotation>(WorkRulesModelHandler.getRotationsByIDs(m_context, getPageIDs()));
			} else {
				PaginationPair<Rotation> rotationData = WorkRulesModelHandler.getAvailableRotationsForEmployees(m_context,
						employeeIDs, viewStartDate.getTime(), viewEndDate.getTime(), m_list.getHeader().isAscending(), getPageIDSize(),
						getPagePaginationIndex(), new LinkedHashMap<DBSortField, Comparator>(m_rotationListPC.getSortFields()), selectedIDs, getTextToFind(itemToFind));
				rotations = new ArrayList<Rotation>(rotationData.getValues());
				initPaginationIDs(rotationData.getIDs());
			}

			Collections.sort(rotations, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));
		} else {
			throw new Exception(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_ASSIGNMENT_RULE_POPUP_NO_EMPLOYEES_ERROR));
		}
		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		for (Rotation rotation : rotations) {
			if (rotation.getOrganizationID() != null) {
				orgIDs.add(rotation.getOrganizationID());
			}
		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
	}

	protected RotationListPC initList() {
		m_rotationListPC = new RotationListPC(getRequestContext(), rotations, orgMap, true);
		m_rotationListPC.setMultiRowSelectable(false);
		return m_rotationListPC;
	}

	public String getHtmlDisplay() {
		if (m_list != null) {
			return m_list.getHtmlDisplay();
		}
		return "";
	}

	public void setSelectedIDs(String selectedIDs) {
		this.selectedIDs = new HashSet<ID>(StringUtil.parseIDStringToCollection(selectedIDs));
	}

	public void setViewStartDate(String startDate) {
		viewStartDate = Calendar.getInstance();
		viewStartDate.setTimeInMillis(Long.parseLong(startDate));
	}

	public void setViewEndDate(String endDate) {
		viewEndDate = Calendar.getInstance();
		viewEndDate.setTimeInMillis(Long.parseLong(endDate));
	}

	public void setIsInfoPopup(boolean isInfoPopup) {
		this.isInfoPopup = isInfoPopup;
	}
}
