package com.verint.web.fs.workrules.assignment;

import com.bluepumpkin.web.bbm.people.filter.PeopleFilterBoxPC;
import com.bluepumpkin.web.bbm.people.filter.PeopleFilterParams;
import com.bluepumpkin.web.bbm.workload.queue.selection.CampaignSPFilterBoxPC;
import com.witness.web.uif.jsp.HtmlTableRow;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

import java.util.ArrayList;

/**
 * This page component renders the following items:
 * A campaign selector dropdown
 * A scheduling period selector dropdown
 * A person filter dropdown
 * A "find person" text field allowing the user to search for a particular employee
 */
public class CampaignSPPeopleFilterBoxPC extends CampaignSPFilterBoxPC {
	private static final long serialVersionUID = 1L;

	private static final short SELECTION_BOX_WIDTH = 235;
	
	PeopleFilterBoxPC peopleFilterBoxPC;

	public CampaignSPPeopleFilterBoxPC(RequestContext context, String name) {
		super(context, name);

		m_hasMultipleBottomRows = true;

		//Using the default name for PeopleFilterBoxPC (rather than handing in a more unique name in the constructor) because
		//PeopleFilterPageModel unfortunately hard-codes the name "peopleFilter" and refers to it as the name for the component
		//when setting this compnent's selected value in the parent window.  If the default name is not used, the "Run" button on
		//the Create Filter popup screen will not work.
		peopleFilterBoxPC = new PeopleFilterBoxPC(context);

		peopleFilterBoxPC.setDisplayLabels(true);
		peopleFilterBoxPC.setHasUnsavedFilter(true);  //This enables the "Run" button on the Create Filter popup
		setDisplayLabels(true);
		addSiblingComponent(peopleFilterBoxPC);
		setEnableFilterSync(false);
	}

	/**
	 * Initialize Component For Display
	 */
	public void initForDisplay() {
		if (m_isInitializedForDisplay) {
			return;
		} else {
			super.initForDisplay();

			peopleFilterBoxPC.getListbox().setMaxDisplayWidth(getListBoxDisplayMaxWidth());

			//The bottom row (find employee text box) actually consists of 3 cells.  We need to set the
			// colSpan of the dropdowns so that the controls are properly aligned.
			String colSpan = createBottomRow().getRowCells().size() == 0 ? "2" : createBottomRow().getRowCells().size() + "";
			m_listbox.getWrapper().setAttribute("colspan", colSpan);

			m_listbox.getSelect().setListWidthProp(SELECTION_BOX_WIDTH);
			m_schedulingPeriodListBox.getWrapper().setAttribute("colspan", colSpan);
			if (showGenericWeekSelector()) {
				m_weekSelectorPC.setOnChangeJS("top.selectionMediator.campaignSPListBoxChanged();" + getOnDateChangeJS());
			}
			peopleFilterBoxPC.getListbox().getWrapper().setAttribute("colspan", colSpan);
		}
	}

	public ArrayList<HtmlTableRow> createBottomRows() {
		ArrayList<HtmlTableRow> rows = new ArrayList<HtmlTableRow>();
		rows.add(peopleFilterBoxPC.createTopRow());
		rows.add(peopleFilterBoxPC.createBottomRow());
		return rows;
	}

	public String getHtmlDisplay() {
		//Ensure that the person filter has its drop down populated before it is rendered.
		//  (This is normally done in getHtmlDisplay but our superclass, CampaignSPFilterBoxPC, does not
		//  contain a person filter).
		peopleFilterBoxPC.getListbox().setOptions(peopleFilterBoxPC.getFilters());

		//We need to add an additional hidden input here that is used by the people filter
		//for the quick info text popup.
		StringBuffer html = new StringBuffer();
		html.append(super.getHtmlDisplay());
		html.append(HtmlUtil.makeHiddenInput(peopleFilterBoxPC.getName(), ""));
		return html.toString();
	}

	public void initPersonFilter() {
		peopleFilterBoxPC.initFilter();
	}

	public PeopleFilterParams getFilterParams() {
		return peopleFilterBoxPC.getFilterParams();
	}

	public void initFilter() throws Exception {
		initPersonFilter();
		initSPFilter(true, false);
	}

	public PeopleFilterBoxPC getPeopleFilterBoxPC() {
		return peopleFilterBoxPC;
	}
}
