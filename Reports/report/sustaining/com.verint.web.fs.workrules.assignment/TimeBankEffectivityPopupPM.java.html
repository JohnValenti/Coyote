package com.verint.web.fs.workrules.assignment;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.l10n.RegionalFormatBundleKey;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.empworkrule.model.EmpTimeBank;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectUtil;
import com.bluepumpkin.ejb.bbm.workresource.model.Employee;
import com.bluepumpkin.ejb.bbm.workrules.model.TimeBank;
import com.bluepumpkin.web.bbm.people.profile.ProfileModelHandler;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.WorkRulesModelHandler;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.MultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.pagemodel.PopupWindowListPM;
import com.witness.web.uif.system.RequestContext;

import java.util.*;

/**
 * Displays time bank assignment history in a popup window for a specific employee.
 */
public class TimeBankEffectivityPopupPM extends PopupWindowListPM {

	public static final String FORM_ACTION = "timebank_effectivity_popup";
	public static final String POPUP_ID = "POPUP_TIMEBANK_EFFECTIVITY";
	public static final String POPUP_ARGS = "employeeID,viewDate";
	public static final String POPUP_ARG_EMPLOYEE_ID = "employeeID";
	public static final String POPUP_ARG_VIEW_DATE = "viewDate";

	private ID employeeID;
	private Date viewDate;
	private Employee employee;
	private Collection<EmpTimeBank> timeBankAssignments;
	private HashMap<ID, TimeBank> timeBanks;

	public TimeBankEffectivityPopupPM(RequestContext context) {
		super(context);

		m_list = new MultiColListPC(context);

		addChildComponent(m_list);
	}

	public static TimeBankEffectivityPopupPM getInstance(RequestContext context) {
		return (TimeBankEffectivityPopupPM) getInstance(context, TimeBankEffectivityPopupPM.class);
	}

	public void initContentTitle() {
		m_contentTitlePC.setPageTitle(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_TIME_BANK_TITLE));
		if (employee != null) {
			m_contentTitlePC.setItemName(m_localizer.formatName(employee));
		}
	}

	protected void loadData() throws Exception {
		if (employeeID != null) {
			employee = new ProfileModelHandler().getEmpWithHistoryData(employeeID, viewDate, m_context.isInWhatIfMode());
			Date historyStart = new Date(new Date().getTime() - (100L * 365 * 24 * 60 * 60 * 1000)), historyEnd = new Date(new Date().getTime() + (100L * 365 * 24 * 60 * 60 * 1000));
			HashMap<ID, Collection<EmpTimeBank>> timeBankAssignmentMap = WorkRulesModelHandler.getTimeBankAssignments(m_context, Arrays.asList(employeeID), historyStart, historyEnd);
			timeBankAssignments = timeBankAssignmentMap.get(employeeID);
			if (timeBankAssignments != null) {
				ArrayList<ID> timeBankIDs = new ArrayList<ID>();
				for (EmpTimeBank etb : timeBankAssignments) {
					timeBankIDs.add(etb.getTimeBankID());
				}
				Collection<TimeBank> timeBankCol = WorkRulesModelHandler.getTimeBanksByIDs(m_context, timeBankIDs);
				timeBanks = ValueObjectUtil.getIDObjectMap(timeBankCol);

				List<MultiColumnNodeData> historyList = m_list.getListModel();
				for (EmpTimeBank etb : timeBankAssignments) {
					DefaultMultiColumnNodeData mcnd = new DefaultMultiColumnNodeData();
					//Time Banks are stored in the DB at GMT time and with an exclusive end date.  We need to subtract a day
					//from the end date for display purposes.
					mcnd.add(m_localizer.formatDate(etb.getStartTime(), TimeZoneUtil.GMT_TIMEZONE, RegionalFormatBundleKey.DATE_FORMAT));
					Calendar endDateDisplay = Calendar.getInstance(TimeZoneUtil.GMT_TIMEZONE);
					endDateDisplay.setTime(etb.getEndTime());
					endDateDisplay.add(Calendar.DAY_OF_YEAR, -1);
					mcnd.add(m_localizer.formatDate(endDateDisplay.getTime(), TimeZoneUtil.GMT_TIMEZONE, RegionalFormatBundleKey.DATE_FORMAT));
					TimeBank tb = timeBanks.get(etb.getTimeBankID());
					if (tb != null) {
						mcnd.add(tb.getName());
					} else {
						mcnd.add("");
					}
					historyList.add(mcnd);
				}
			}
		}
	}

	public void setEmployeeID(ID employeeID) {
		this.employeeID = employeeID;
	}

	public void setViewDate(Date viewDate) {
		this.viewDate = viewDate;
	}

	@Override
	protected void initHeader() {
		TableHeaderPC header = m_list.getHeader();
		header.addColumnHeaderData("startDate", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_START_DATE));
		header.addColumnHeaderData("endDate", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_END_DATE));
		header.addColumnHeaderData("timeBankName", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_TIME_BANK));
	}

	@Override
	protected void initSelectableItems() {

	}

	public String getHtmlDisplay() {
		return m_list.getHtmlDisplay();
	}
}
