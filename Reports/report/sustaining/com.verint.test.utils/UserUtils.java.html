package com.verint.test.utils;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.core.security.weblogic.SecurityDelegator;
import com.witness.ejb.core.security.RoleManager;
import com.witness.ejb.core.security.UserManager;
import com.witness.ejb.core.security.model.AccessRight;
import com.witness.ejb.core.security.model.Role;
import com.witness.ejb.core.security.model.User;
import com.witness.ejb.core.security.model.UserFacade;

import java.util.Collection;
import java.util.List;

/**
 * UserUtils class created to build and destroy User objects in the Impact360 application.
 */
public class UserUtils {

	/**
	 * Deletes the user from the underlying user database from Impact360
	 *
	 * @param username Username of the user
	 * @throws Exception
	 */
	public static void deleteUserPassword(String username) throws Exception {
		Jdmo jdmo = null;
		try {
			String fullHostName = DBUtils.getDBHost();
			jdmo = DBUtils.createJdmo(fullHostName, DBUtils.getDBName(), DBUtils.getDBUser(), DBUtils.getDBPass());
			jdmo.execute("DELETE FROM DBREALMUSER WHERE NAME = '" + username + "'");
		} catch (Exception ex) {
			ex.printStackTrace();
			throw ex;
		} finally {
			if (jdmo != null) {
				jdmo.cleanUp();
			}
		}
	}

	public static User getUserByID(ID userID) throws Exception {
		return RemoteEjbFactory.getUserManager().getUserByID(userID);
	}

	public static ID createUser(ID employeeID, String username, String password) throws Exception {
		ID userID;
		UserFacade user = new UserFacade(employeeID, username);
		user.setStatus(User.ACTIVE);

		UserManager userManager = RemoteEjbFactory.getUserManager();
		userID = userManager.createUser(user);

		SecurityDelegator securityDelegator = RemoteEjbFactory.getSecurityDelegator();
		securityDelegator.addUser(username, password);
		return userID;
	}

	/**
	 * This calls the UserManagerEJB's updateUserAccessRight, which overwrites all current access rights for the given user with the set of
	 * new access rights.
	 * @param username
	 * @param roleName
	 * @param orgID
	 * @throws Exception 
	 */
	public static void assignAccessRights(ID userID, List<AccessRight> accessRightsList) throws Exception {
		UserManager userManager = RemoteEjbFactory.getUserManager();
		userManager.updateUserAccessRight(userID, accessRightsList);
	}

	public static ID getRoleId(String roleName) throws Exception {
		ID roleID = null;
		RoleManager roleManager = RemoteEjbFactory.getRoleManager();
		Collection roles = roleManager.getRoles();
		for (Object roleObj : roles) {
			Role role = (Role) roleObj;
			if (role.getName().equals(roleName)) {
				roleID = role.getID();
			}
		}
		return roleID;
	}
}
