package com.verint.test.utils;


public class GuidUtils {
	public static boolean useGuidsForGeneratedData() {
		return Boolean.parseBoolean(System.getProperty("useGuid"));
	}

	public static String generateGuidPrefix() {
		return java.util.UUID.randomUUID().toString().substring(0, 8) + "_";
	}

	/**
	 * Check if name should be prefixed with Guid random generated string.
	 * If yes return String similar to 05bg5855_name,
	 * otherwise just original name
	 *
	 * @param name
	 * @return
	 */
	public static String createGuidName(String name) {
		if (useGuidsForGeneratedData()) {
			return generateGuidPrefix() + name;
		}
		return name;
	}
}
