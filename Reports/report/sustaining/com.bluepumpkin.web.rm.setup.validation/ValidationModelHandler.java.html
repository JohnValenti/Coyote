	/**

		return WfmManagerFactory.getActivityManager(context.isInWhatIfMode());

    /**
     * Return ValidationSetting object, which includes the bundlekey->ValidationRule
     * map, as well as ancestor org information .
     * @param requestType - One of: 'time-off', 'shift-swap', 'shift-swap-withdraw', 'shift-bid', 'time-off-withdraw', 'all'
     * @param orgID   ID  Identifes the organizationId
     * @param shallow - If true, we will search only the specified org. If false, and
     *           the rules are not found at the specified org, then we search the ancestor
     *           orgs until we find them.
     *
     * @return  The ValidationSetting object.
     */
    public static ValidationSetting getSettings(String requestType, ID orgID, boolean shallow, List<ID> listSelectedTimeOffTypeActivityIDs)
        throws BbmFinderException, BbmCreateException, RemoteException 
    {
        RmManagerFactory factory = RmManagerFactory.getInstance();
        
        // Invoke the facade method here
        return factory.getValidationRuleManager().getSettings(requestType, orgID, shallow, listSelectedTimeOffTypeActivityIDs);
    }
    
	/**
	 * Return a collection of Validation Rules for a particular org and request type.
     * @param requestType - One of: 'time-off', 'shift-swap', 'shift-swap-withdraw', 'shift-bid', 'time-off-withdraw', 'all'
	 * @param orgID	ID	Identifes the organizationId
     * @param shallow - If true, we will search only the specified org. If false, and
     *           the rules are not found at the specified org, then we search the ancestor

     *

	public static Map findValidationRules(String requestType, ID orgID, boolean shallow, List<ID> listSelectedTimeOffTypeActivityIDs)
			throws BbmFinderException, BbmCreateException, RemoteException
    {
		// Invoke the facade method here
		return RmManagerFactory.getInstance().getValidationRuleManager().

	}
	
	public static Map findValidationRules(String requestType, ID orgID, boolean shallow)
			throws BbmFinderException, BbmCreateException, RemoteException
    {
		// Invoke the facade method here
		List<ID> empty = Collections.emptyList();
		return RmManagerFactory.getInstance().getValidationRuleManager().
				findValidationRules(requestType, orgID, shallow, empty);
	}
    
    /**
     * Return a collection of Validation Rules for a particular org and request type.
     * If the rules are not found at the specified org, then we search the ancestor
     * orgs until we find them.
     * @param requestType - One of: 'time-off', 'shift-swap', 'shift-swap-withdraw', 'shift-bid', 'time-off-withdraw', 'all'
     * @param orgID   ID  Identifes the organizationId
     *           
     * @return  Collection collection of ValidationRules
     */
    public static Map getValidationRules(String requestType,ID orgID)
            throws BbmFinderException, BbmCreateException, RemoteException
    {
        return findValidationRules(requestType, orgID, false);
    }
    

	/**
	 * Configures the validation rules for an Organization.
	 * @param  validationRules	HashMap collection of Validation Rules
	 * @param  orgID	ID	Identifes the organizationId
     * @param shallow - If true, we will update only the specified org. If false, and
     *           then we update the child orgs too, if any. In other words, ALL of the
     *           child orgs' rules or ALL types will be removed, and they will inherit
     *           their settings from this org.
	 */
	public static void updateValidationRules(Map validationRules,ID orgID, boolean shallow, List<ID> listSelectedTimeOffTypeActivityIDs)
			throws BbmCreateException, BbmUpdateException, RemoteException
    {
		// Invoke the facade method here
		RmManagerFactory.getInstance().getValidationRuleManager().
				updateValidationRules(validationRules,orgID, shallow, listSelectedTimeOffTypeActivityIDs);
	}

	/**
	 * Get all the validation resources. This is a map indexed by the
	 * validator class name. The value is a ValidationResource object.
	 * The ValidationResource object contains the icon and label for a 
	 * validation rule.
	 *
	 * @return HashMap of validationClass to ValidationResources object.
	 */
	public static Map getVRClassToVRMap() {
		if (m_vrClassToVRMap==null) {
			synchronized(ValidationModelHandler.class) {
				if (m_vrClassToVRMap==null) {
					m_vrClassToVRMap = new HashMap(31);

					// get the mapping of classname to resource key.
					Map map;
					try {
						map = RmManagerFactory.getInstance().getValidationRuleManager().
								getValidationRuleMap();
					}	catch (Exception e) {
						category.error("", e);
						return Collections.emptyMap();
					}

					// get the mapping for classname to ValidationResource.
					ValidationRulesMap vrMap = ValidationRulesMap.getInstance();
					Iterator itr = map.keySet().iterator();
					while (itr.hasNext()) {
						String className = (String)itr.next();
						String resKey = (String)map.get(className);
						ValidationResources vresource = vrMap.getValResource(resKey);
						if (vresource != null) {
							m_vrClassToVRMap.put(className, 	vresource);
						}
					}
				}
			}
		}
		return m_vrClassToVRMap;
	}

	/**
	 * Get a validation resource given a validator class name.
	 * The ValidationResource object contains the icon and label for a 
	 * validation rule.
	 *
	 * @param validatorClassName - name of the validator class. This is returned
	 *                             in the conflict objects from the backend.
	 * @return ValidationResources - contains the validation rule icon name and
	 *                               label.
	 *                               The method will return null if the validator 
	 *                               class is not associated with a validation 
	 *                               rule.
	 */
	public static ValidationResources getValidationResource(String validatorClassName) {
		Map map = getVRClassToVRMap();
		if (map == null) return null;

		ValidationResources vr = (ValidationResources)map.get(validatorClassName);
		return vr;
	}

	@SuppressWarnings("unchecked")
	public static List<Activity> getTimeOffRequestableActivities(RequestContext context, ID orgId)
			throws RemoteException, BbmException {
		ActivityFilter activityFilter = new ActivityFilter();
		activityFilter.setDeleted(false);
		activityFilter.setRequestable(true);
		return new ArrayList<Activity>(ActivityModelHandler
				.getActivities(context, orgId, activityFilter));
	}

	/**
	 * Gets a collection of the effective ValidationRules and parameter values for the org.
	 * If there are multiple rules for different activity (validation by time off type), a rule is returned for each rule and activity.

			throws BbmFinderException, RemoteException, BbmCreateException {
		return RmManagerFactory.getInstance().getValidationRuleManager().

	}
