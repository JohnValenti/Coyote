    public static class Assignment {
        public static class Compare implements Comparator<Assignment>
        {
            public int compare(Assignment a1, Assignment a2)
            {
                if (a1.binding == null)
                {
                    if (a2.binding == null)
                        return 0;
                    return -1;
                }
                if (a2.binding == null)
                {
                    if (a1.binding == null)
                        return 0;
                    return 1;
                }
                if (a1.binding == null && a2.binding != null)
                    return -1;
                if (a1.binding.start < a2.binding.start)
                    return -1;
                if (a1.binding.start > a2.binding.start)
                    return 0;
                return 0;
            }
        }

        private final void BindMaxSpacingScore(int dir)
        {
            int leftSpacing = scenario.agent.GetLeftSpacing(1, binding.start);
            int rightSpacing = scenario.agent.GetRightSpacing(1, binding.start + shiftEvent.duration);
            int maxSpacing = scenario.shift.shiftEventSpacing.getSecond();
            double leftScore = Math.pow(Math.max(0, leftSpacing - maxSpacing), 2);
            double rightScore =  Math.pow(Math.max(0, rightSpacing - maxSpacing), 2);
            double score = leftScore + rightScore;
            Logger.Log((dir < 0 ? "Unbinding " : "Binding ") + score + (dir < 0 ? " from " : " to ") + " max spacing (" + scenario.IntervalToTimeSpan(maxSpacing) + ") score: LeftSpacing = " + scenario.IntervalToTimeSpan(leftSpacing) + "(" + leftScore + "), RightSpacing = " + scenario.IntervalToTimeSpan(rightSpacing) + "(" + rightScore + ")");
            Logger.Tab();
            scenario.objectiveFunction.BindConstraint(score * dir);
            Logger.Untab();
        }
        private final void BindPrevMaxSpacingScore(int dir)
        {
            int spacing = scenario.agent.GetLeftSpacing(1, binding.start) + scenario.agent.GetRightSpacing(1, binding.start);
            int maxSpacing = scenario.shift.shiftEventSpacing.getSecond();
            double score = Math.pow(Math.max(0, spacing - maxSpacing), 2);
            Logger.Log((dir < 0 ? "Unbinding " : "Binding ") + score + (dir < 0 ? " from " : " to ") + " max spacing (" + scenario.IntervalToTimeSpan(maxSpacing) + ") score: Spacing = " + scenario.IntervalToTimeSpan(spacing));
            Logger.Tab();
            scenario.objectiveFunction.BindConstraint(score * dir);
            Logger.Untab();
        }
        public final void Bind(Binding binding)
        {
            this.binding = binding;
            if (binding != null)
            {
                Logger.Log("Binding " + binding);
                Logger.Tab();
                if (!shiftEvent.isWork && binding.activity != null)
                {
                    BindPrevMaxSpacingScore(-1);
                    BindMaxSpacingScore(1);
                }
                shiftEvent.Bind(shift, binding.start, binding.activity);
                Logger.Untab();
            }
        }
        public final void Unbind(Binding binding)
        {
            Logger.Log("Unbinding " + binding);
            Logger.Tab();
            shiftEvent.Unbind(shift, binding.start, binding.activity);
            if (!shiftEvent.isWork && binding.activity != null)
            {
                BindPrevMaxSpacingScore(1);
                BindMaxSpacingScore(-1);
            }
            this.binding = null;
            Logger.Untab();
        }

        private boolean hasRequiredSkill(Binding binding) {
        	if (binding.activity != null && binding.activity.skillRequired && !binding.activity.skilledQueues.isEmpty()) {
        		return true;
        	}
        	return false;

        public final void Schedule()
        {
            if (locked)
                return;
            Logger.Log("Scheduling: " + this + " " + scenario.objectiveFunction.toString());
            Logger.Tab();
            if (this.binding != null)
                Unbind(this.binding);
            double originalScore = scenario.objectiveFunction.GetScore();
            double bestScore = originalScore;
            double bestPotentialBindingScore = originalScore;
            int first = 0;
            for (ShiftEvent.Binding binding : domain)
            {
            	// QC160771: if the shift event min count is 0, and the agent doesn't have skill, 
            	// don't create shift event. However, if the shift event min count is 1, and 
            	// the agent doesn't have skill, create/not create both are ok. Right now, we
            	// choose to create the shift event.
            	if (shiftEvent.countConstraint.getFirst() == 0 && !hasRequiredSkill(binding)) {
            		continue;
            	}
            	
            	String reason = IsConsistent(binding);
                if (reason != null)
                {
                    Logger.Log(binding + " is inconsistent because -- " + reason);
                    continue;
                }
                Logger.Log("Trying... " + binding);
                Logger.Tab();
                Bind(binding);
                double score = scenario.objectiveFunction.GetScore();
                if(first==0){
                	bestPotentialBindingScore = score+1;
                	first++;
                }
                Logger.Log("Consider... current " + scenario.objectiveFunction.toString() + " .vs. best score = " + bestScore);
                if (score < bestScore)
                {
                    Logger.Log("New best");
                    Logger.Tab();
                    SaveAsBest();
                    Logger.Untab();
                    bestScore = score;
                } else if(score < bestPotentialBindingScore){
                	Logger.Log("New potential best");
                    Logger.Tab();
                    SaveAsBest();
                    Logger.Untab();
                    bestPotentialBindingScore = score;
                }
                Unbind(binding);
                if (scenario.objectiveFunction.GetScore() != originalScore)
                    Logger.Log("Reversability Error: ObjectiveFunction = " + scenario.objectiveFunction.toString());

                Logger.Untab();
            }
            Bind(bestBinding);
            Logger.Log("Best binding = " + this);
            Logger.Untab();
        }

    public final void Initialize(Shift shift)
    {
        Logger.Log(name);
        Logger.Tab();
        BindScore();

        domain = new ArrayList<Binding>();
        int granularity;
        if (countConstraint.getFirst() < countConstraint.getSecond())
            granularity = duration;
        else
            granularity = 15;
        Logger.Log("Building domain for " + name + ": Granularity = " + scenario.IntervalToTimeSpan(granularity));
        Logger.Tab();

        for (Activity activity : activities)
        {
            for (int time = startWindow.getFirst(); time <= startWindow.getSecond(); time += granularity)
            {
                Binding binding = new Binding(scenario, time, activity);
                Logger.Log("Adding " + binding);
                domain.add(binding);
            }
        }
        if (countConstraint.getFirst() < countConstraint.getSecond())
        {
            Logger.Log("Adding NotAssigned since optional");
            domain.add(new Binding(scenario, -1, null));
        }
        Logger.Untab();

        assignments = new ArrayList<Assignment>();
        Logger.Log("Creating " + countConstraint.getSecond() + " assignments");
        for (int n = 0; n < countConstraint.getSecond(); n++)
            assignments.add(new Assignment(this, shift));
        Logger.Untab();

    }
    public final ArrayList<Binding> GetDomain()
    {
        return domain;
    }
    private void UnbindScore()
    {
        if (countConstraint.getFirst() < countConstraint.getSecond())
            scenario.objectiveFunction.BindConstraint(-Math.max(0, countConstraint.getFirst() - count));
    }
    private void BindScore()
    {
        if (countConstraint.getFirst() < countConstraint.getSecond())
            scenario.objectiveFunction.BindConstraint(Math.max(0, countConstraint.getFirst() - count));
    }
    public String IsConsistent(int start, Activity activity)
    {
        if (activity == null)
            return null;
        return activity.IsConsistent(start, start + duration);
    }
    public final boolean Load(Shift shift, Activity activity, Date start)
    {
        for (Assignment a : assignments)
        {
            if (a.binding == null)
                a.Bind(new Binding(scenario, scenario.DateTimeToInterval(start), activity));
        }
        return true;
    }
    public final void Bind(Shift shift, int start, Activity activity)
    {
        UnbindScore();
        if (activity != null)
        {
            shift.activity.UnbindStaffingFTE(start, start + duration);
            activity.BindStaffingFTE(start, start + duration);
        }
        count++;
        BindScore();
    }
    public final void Unbind(Shift shift, int start, Activity activity)
    {
        UnbindScore();
        if (activity != null)
        {
            shift.activity.BindStaffingFTE(start, start + duration);
            activity.UnbindStaffingFTE(start, start + duration);
        }
        count--;
        BindScore();
    }

        Logger.Tab();

        Logger.Untab();

    }
