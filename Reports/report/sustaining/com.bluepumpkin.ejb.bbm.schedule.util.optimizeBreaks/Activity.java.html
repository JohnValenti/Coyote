package com.bluepumpkin.ejb.bbm.schedule.util.optimizeBreaks;

import java.util.*;

public class Activity {
	
	private int elbowRoom;
	public int layer;
	private Scenario scenario;
	public String name;
	public String id;
	// for employee within the optization window
	public ArrayList<Queue> skilledQueues; 
	public boolean skillRequired = false;
	
	public Activity(Scenario scenario, String id, String name, int layer, ArrayList<Queue> skilledQueues, boolean skillRequired) {
		this.scenario = scenario;
		this.id = id;
		this.name = "Activity-" + name;
		this.layer = layer;
		this.skilledQueues = skilledQueues;
		this.skillRequired = skillRequired;
	}
	
	public void Initialize() {
		if (layer == 1) {
			elbowRoom = scenario.shift.shiftEventSpacing.getFirst() / 2;
		}
	}
	
	public String IsConsistent(int start, int end) {
		return scenario.agent.IsAvailable(layer, start - elbowRoom, end + elbowRoom);
	}
	
	public final void BindStaffingFTE(int start, int end) {
		Logger.Log("Bind " + name + " from " + scenario.IntervalToDateTime(start) + " to " + scenario.IntervalToDateTime(end));
		Logger.Tab();
		scenario.agent.SetAvailability(layer, start - elbowRoom, end + elbowRoom, false);
		for (Queue queue : skilledQueues) {
			queue.BindStaffingFTE(start, end, queue.proficiency * scenario.agent.proficiency);
		}
		Logger.Untab();
	}
	
	public final void UnbindStaffingFTE(int start, int end) {
		Logger.Log("Unbind " + name + " from " + scenario.IntervalToDateTime(start) + " to " + scenario.IntervalToDateTime(end));
		Logger.Tab();
		scenario.agent.SetAvailability(layer, start - elbowRoom, end + elbowRoom, true);
		for (Queue queue : skilledQueues) {
			queue.BindStaffingFTE(start, end, queue.proficiency * -scenario.agent.proficiency);
		}
		Logger.Untab();
	}
	
	public final void Log() {
		Logger.Log(name);
		Logger.Tab();
		for (Queue queue : skilledQueues) {
			Logger.Log(queue.name + ": Proficiency = " + queue.proficiency);
		}
		Logger.Untab();
	}
}
