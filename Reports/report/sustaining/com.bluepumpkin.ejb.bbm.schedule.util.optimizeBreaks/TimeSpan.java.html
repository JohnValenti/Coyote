package com.bluepumpkin.ejb.bbm.schedule.util.optimizeBreaks;
import java.util.*;
public class TimeSpan {
	public TimeSpan(Date start, Date end)
	{
		Calendar calStart = new GregorianCalendar();
		calStart.setTime(start);
		Calendar calEnd = new GregorianCalendar();
		calEnd.setTime(end);
		TotalMinutes = (int) ((calEnd.getTimeInMillis() - calStart.getTimeInMillis()) / 60000); 
	}
	public TimeSpan(int hours, int minutes, int seconds)
	{
		TotalMinutes = (hours * 60) + minutes; 
	}
	public Date addTo(Date date)
	{
		Calendar cal = new GregorianCalendar();
		cal.setTime(date);
		cal.add(Calendar.MINUTE, TotalMinutes);
		return cal.getTime();
	}
	public Date subFrom(Date date)
	{
		Calendar cal = new GregorianCalendar();
		cal.setTime(date);
		cal.add(Calendar.MINUTE, -TotalMinutes);
		return cal.getTime();
	}
	@Override
	public String toString()
	{
		if (TotalMinutes < 0)
			return "-" + ((-TotalMinutes / 60) + ":" + String.format("%02d", new Integer(-TotalMinutes % 60))); 
		return ((TotalMinutes / 60) + ":" + String.format("%02d", new Integer(TotalMinutes % 60))); 
	}
	public int TotalMinutes; 
}
