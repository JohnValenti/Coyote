package com.bluepumpkin.web.rm.setup.settings;

import java.util.ArrayList;

import com.witness.web.uif.pagecomponent.container.Collapsible2ColFormPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;

/**
 * Title          ColorGenericPC
 * Description    Display Time Off Calendar Color
 * Copyright      Copyright (c) 2002
 * Company        Blue Pumpkin Software, Inc
 * @author        Shubhangi Chavan
 * @version       1.0
 */
public class ColorGenericPC extends Collapsible2ColFormPC {

	private static String PC_KEY = "color_generic_pc";

	/** holder of list of ColorField(s) */
	private ArrayList m_colorFields = new ArrayList();

	/** bundle key for the title */
	private String m_titleKey;

	/** flag to determine if this component is disabled */
	private boolean m_disabled;

	/** hold field name, field label and field value for display */
	private static class ColorField {
		public String name;  // field name
		public String label;  // field label
		public String value; // field color value (eg #000000)
		public boolean enabled;
		public ColorField(String name, String label, String value,
				boolean enabled) {
			this.name = name;
			this.label = label;
			this.value = value;
			this.enabled = enabled;
		}
	}

	/**
	 * Constructs the TimeOff calendar color page component.
	 * @param context RequestContext.
	 */
	public ColorGenericPC(RequestContext context) {
		super(context, RmWebBundleKey.BUNDLE_NAME);
		setName(PC_KEY);
	}

	/**
	 * getters and setters for data fields
	 */
	public ArrayList getColorFields() {
		return m_colorFields;
	}

	public void setColorFields(ArrayList m_colorFields) {
		m_colorFields.addAll(m_colorFields);
	}

	public void addColorFields(String name,
			String label,
			String value,
			boolean enabled) {
		m_colorFields.add(new ColorField(name, label, value, enabled));
	}

	public String getTitleKey() {
		return m_titleKey;
	}

	public void setTitleKey(String key) {
		m_titleKey = key;
	}

	public boolean getDisabled() {
		return m_disabled;
	}

	public void setDisabled(boolean on) {
		m_disabled = on;
	}

	/**
	 * Get string of the HTML representing this component.
	 */
	public String getHtmlDisplay() {
        if (m_titleKey != null)
            setTitle("&nbsp;" + i18n(m_bundle, m_titleKey));

		int l = m_colorFields.size();
		for (int iArg = 0; iArg < l; iArg++) {
			ColorField cf = (ColorField)m_colorFields.get(iArg);

			// Add the TimeOff day and the color
			addRowWithLabelKey(cf.label,
					HtmlUtil.makeColorPickerCaller(cf.name, cf.value, "", !cf.enabled, m_localizer)
			);
		}

		return super.getHtmlDisplay();
	}
}
