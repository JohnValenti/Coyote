/*
 * TimeOffIntervalPaidSegmentCalculator
 * 
 * We use the following as our template for what we need to do.
 * com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.ejb.TOHoursPerDayManagerEJB.updateTOHoursPerDayForTORequest
 */

public class TimeOffIntervalPaidSegmentCalculator {

	private static final Category LOG = Log.initCategory(TimeOffIntervalPaidSegmentCalculator.class.getName());
	private TORequest toRequest = null;
	private ID empId = null;
	private Map<Object, Object> empOrgAssignments = Collections.emptyMap();
	private Organization employeeOrg = null;
	private List<TimeOffIntervalPaidSegment> listOfSegments = new ArrayList<TimeOffIntervalPaidSegment>();
	private PublishingPeriodStatus publishingPeriodStatus = null;
	private List<ID> listTimeOffChoiceIDsToUpdate = null;
	// We keep a map of choices to reference schedules so we can update the toChoice reference schedule when storing to the db
	private final Map<ID, ID> mapTOChoiceReferenceScheduleID = new HashMap<ID, ID>();
	ReferenceScheduleLoader loader;
	
	
	
	
	
	//
	// updateTimeOffIntervalPaidSegmentsChoices
	// Called from updates to schedules. We will only calculate and recreate the choices in the listOfTimeOffChoiceIDsToUpdate
	//
	public List<TimeOffIntervalPaidSegment> updateTimeOffIntervalPaidSegmentsChoices(
			TORequest request, List<ID> listOfTimeOffChoiceIDsToUpdate) throws Exception { // NOSONAR
	
		this.listTimeOffChoiceIDsToUpdate = listOfTimeOffChoiceIDsToUpdate;
		
		if( listTimeOffChoiceIDsToUpdate == null || listTimeOffChoiceIDsToUpdate.isEmpty()){
			return Collections.emptyList();
		}
		
		return generateTimeOffIntervalPaidSegmentsForTORequest(request);
	}
	
	
	
	
	public List<TimeOffIntervalPaidSegment> generateTimeOffIntervalPaidSegmentsForTORequest(
			TORequest request) throws Exception { // NOSONAR

		this.toRequest  = request;
		this.empId = toRequest.getEmployeeID();
		this.listOfSegments = new ArrayList<TimeOffIntervalPaidSegment>();

		// Evidently we don't do anything in don't debit cases.
		if (toRequest.getTimeOffDebitType().equals(TORequest.DEBITTYPE_DONT_DEBIT)) {
			LOG.info("Ignoring TO Request since it is of TYPE DONT-DEBIT:" + toRequest);
			return this.listOfSegments;
		}
		
		validateRequest();
		setEmployeeOrg();
		publishingPeriodStatus = TimeOffIntervalAllocationUtil.initPublishedPeriodStatusForTimeoffRequest(this.toRequest);
		
		
		List<TOChoice> choiceList = toRequest.getRequestChoiceList();
		
		initializeReferenceScheduleLoader(choiceList);

		for (TOChoice toChoice : choiceList) {
			if (listTimeOffChoiceIDsToUpdate != null && !listTimeOffChoiceIDsToUpdate.contains(toChoice.getID())) {
					continue;
			}

			generateTimeOffIntervalPaidSegmentsForTOChoice(toChoice);
		}
		
		return getListOfSegments();
	
	}
	
	void initializeReferenceScheduleLoader(List<TOChoice> choiceList) throws BbmFinderException {
		loader = new ReferenceScheduleLoader();

		List<EmployeeDate> employeeDates = new ArrayList<EmployeeDate>();
		for (TOChoice choice : choiceList) {

			if (listTimeOffChoiceIDsToUpdate != null && !listTimeOffChoiceIDsToUpdate.contains(choice.getID())) {
				continue;
			}

			Status publishedStatus = TimeOffIntervalAllocationUtil.getPublishedPeriodStatusForTimeoffChoice(this.publishingPeriodStatus,
					this.empId, choice);
			if (publishedStatus == Status.FULLY_UNPUBLISHED) {
				employeeDates.add(new EmployeeDate(this.empId, choice.getStartDate()));
			}
		}

		if (employeeDates.isEmpty()) {
			return;
		}

		loader.initialize(employeeDates);
	}
	


	void generateTimeOffIntervalPaidSegmentsForTOChoice(
			TOChoice toChoice) throws Exception { // NOSONAR
		
		Status publishedStatus = TimeOffIntervalAllocationUtil.getPublishedPeriodStatusForTimeoffChoice(this.publishingPeriodStatus, this.empId,toChoice);
        
		if (publishedStatus == Status.FULLY_PUBLISHED) {
			generateSegmentsForTOChoiceFromPublished(toChoice);

		} else if (publishedStatus == Status.FULLY_UNPUBLISHED) {
			// Wholly unpublished. Use the reference schedules
			generateSegmentsForTOChoiceFromReference(toChoice);
		} else if (publishedStatus == Status.HAS_GAPS_IN_PUBLISHING && TORequestUtil.isUseIntervalTOPool(toRequest)) {
			 // If we are not using the interval pool, no need to throw an error
				throw RequestUtil
						.createAndLogRmHardValidationException(
								RmEjbBundleKey.TIMEOFFINTERVALALLOCATIONSEGMENT_SCHEDULES_PARTLY_PUBLISHED,
								RmEjbLogBundleKey.TIMEOFFINTERVALALLOCATIONSEGMENT_SCHEDULES_PARTLY_PUBLISHED,
								null, LOG);
		}
	}




	private void generateSegmentsForTOChoiceFromReference(TOChoice toChoice) throws Exception { // NOSONAR
			generateSegmentsForTOChoice(toChoice, false);
		}



	private void generateSegmentsForTOChoiceFromPublished(TOChoice toChoice) throws Exception { // NOSONAR
		generateSegmentsForTOChoice(toChoice,true);
	}



	@SuppressWarnings("unchecked")
	void setEmployeeOrg() throws Exception{ // NOSONAR
		TOChoice theFirstChoice = toRequest.getFirstTOChoice();
		this.empOrgAssignments = TimeOffIntervalAllocationUtil.getEmpOrgAssignments(Collections.singletonList(empId), theFirstChoice.getStartDate(),  theFirstChoice.getEndDate());
		this.employeeOrg = TimeOffIntervalAllocationUtil.getEmpOrgForDate(empOrgAssignments, empId, theFirstChoice.getStartDate());
	}


	void validateRequest() throws RmException {
		if (toRequest == null) {
			throw RequestUtil
					.createRmException(
							RmEjbLogBundleKey.TIMEOFFINTERVALALLOCATIONSEGMENT_TOREQUEST_INVALID,
							LOG);
		}
	
		
		if (toRequest.getFirstTOChoice() == null) {
			throw RequestUtil
					.createRmException(
							RmEjbLogBundleKey.TIMEOFFINTERVALALLOCATIONSEGMENT_TOCHOICE_INVALID,
							LOG);
		}
		
		@SuppressWarnings("unchecked")
		List<TOChoice> choiceList = toRequest.getRequestChoiceList();
		if( choiceList==null || choiceList.isEmpty()){
			throw RequestUtil
			.createRmException(
					RmEjbLogBundleKey.INVALID_CHOICE_LIST,
					LOG);
		}
		
	}


	private void generateSegmentsForTOChoice(TOChoice toChoice, boolean usePublishedSchedules) throws Exception{ // NOSONAR
		
		if (toChoice == null) {
			return;
		}

		if (usePublishedSchedules) {
			generateSegmentsForPublishedSchedules(toChoice);
		} else {
			generateSegmentsForReferenceSchedules(toChoice);
		}

	}

	void generateSegmentsForPublishedSchedules(TOChoice toChoice) throws Exception{ // NOSONAR
		
		// Get the shifts, then cap them to the
		// toChoice range.
		Date startDay = toChoice.getStartDate();
		// Aligning the end with the next day start to make it exclusive.
		Date endDay = toChoice.getEndDate();

		// Set the choiceID's Reference Schedule to null, since we're using published.
		this.mapTOChoiceReferenceScheduleID.put(toChoice.getID(), null );
		
		
		Collection<SimpleEvent> publishedShiftEvents = getPublishedShiftEvents(
				startDay, endDay);
		
		if (publishedShiftEvents == null || publishedShiftEvents.isEmpty()) {
			return;
		}

		// Cap each segment to the toChoice range
		
		// This will chop the ShiftEvents into discrete segments and also cap
		// them to the range passed in
		Collection<SimpleEvent> listSimpleEvent = EventUtils
				.convertEventsToTimelineForSingleEmployee(publishedShiftEvents,
						startDay, endDay, null);

		
		for (SimpleEvent simpleEvent : listSimpleEvent) {
			if (simpleEvent.getPaid()) {
				TimeOffIntervalPaidSegment segment = TimeOffIntervalAllocationUtil
						.createTimeOffIntervalPaidSegmentFromSimpleEvent(
								toChoice, simpleEvent);
				listOfSegments.add(segment);
			}
		}
	
	}

	




	void generateSegmentsForReferenceSchedules(TOChoice toChoice) {
		// For a full day, get the shifts from the reference schedule,
		// then cap them to the toChoice range.

		Date startDay = toChoice.getStartDate();
		Date endDay = toChoice.getEndDate();

		try {
			
			Collection<ShiftAssignment> referencedShiftEvents = getReferenceShiftEvents(startDay, endDay, toChoice);
			
			if (referencedShiftEvents == null
					|| referencedShiftEvents.isEmpty()) {
				return;
			}

			// Cap each segment to the toChoice range
			// This will chop the ShiftEvents into discrete segments and also
			// cap them to the range passed in
			Collection<SimpleEvent> listSimpleEvent = EventUtils
					.convertEventsToTimelineForSingleEmployee(
							referencedShiftEvents, startDay, endDay,
							null);

			for (SimpleEvent simpleEvent : listSimpleEvent) {
				if (simpleEvent.getPaid()) {
					TimeOffIntervalPaidSegment segment = TimeOffIntervalAllocationUtil
							.createTimeOffIntervalPaidSegmentFromSimpleEvent(
									toChoice, simpleEvent);
					listOfSegments.add(segment);
				}
			}
		} catch (Exception e) {
			LOG.error(e.getMessage(), e);
		}

	}

	

	@SuppressWarnings("unchecked")
	Collection<SimpleEvent> getPublishedShiftEvents(Date startDay,
			Date endDay) throws Exception { // NOSONAR

		ScheduleAccessManager scheduleAccessManager = RequestUtil
				.getScheduleAccessManager();

		//Return the shifts for the full day
		return RequestUtil.getShiftAssignsDuringPeriod(this.empId, startDay, endDay, true,
						scheduleAccessManager);
	}

	Collection<ShiftAssignment> getReferenceShiftEvents(Date startDay,
			Date endDay, TOChoice toChoice) {

		// Org is current only for the first choice start date...
		TimeZone tz = this.employeeOrg.getTimeZone();

		// Add the referenceScheduleID to the map of Choices so we can
		// update the choice later.
		ID referenceScheduleID = loader.getRefrenceScheduleID(this.empId,
				startDay);
		this.mapTOChoiceReferenceScheduleID.put(toChoice.getID(),
				referenceScheduleID);

		return loader.getReferenceSchedules(this.empId, startDay, endDay, tz);

	}

	public List<TimeOffIntervalPaidSegment> getListOfSegments() {
		return listOfSegments;
	}

	public Map<ID, ID> getMapTOChoiceReferenceScheduleID() {
		return mapTOChoiceReferenceScheduleID;
	}

	


	
	
	
}
