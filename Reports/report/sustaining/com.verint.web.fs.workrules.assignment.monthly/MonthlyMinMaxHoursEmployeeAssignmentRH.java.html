package com.verint.web.fs.workrules.assignment.monthly;


import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.empworkrule.ejb.EmpWorkRuleManager;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkResourceMonthlyMinMaxHour;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

import java.util.List;
import java.util.Map;

public class MonthlyMinMaxHoursEmployeeAssignmentRH extends RequestHandler {

	protected MonthlyMinMaxHoursEmployeeAssignmentPM getPageModel(RequestContext context) {
		return MonthlyMinMaxHoursEmployeeAssignmentPM.getInstance(context);
	}

	@Override
	public void processRequest(RequestContext context) throws Exception {
		if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		}
	}

	protected void processSave(RequestContext context) throws BPException {
		MonthlyMinMaxHoursEmployeeAssignmentPM pageModel = getPageModel(context);
		try {
			pageModel.extractParameters(context.getRequest());
			Map<ID, List<WorkResourceMonthlyMinMaxHour>> hoursAssignments = pageModel.getMonthlyMinMaxHourAssignments();

			EmpWorkRuleManager ewrm = WfmManagerFactory.getEmpWorkRuleManager(context.isInWhatIfMode());
			ewrm.saveMonthlyMinMaxHourAssignments(hoursAssignments);

			//Add message to screen to indicate a successful save
			pageModel.addPageMessage(Message.INSTRUCTION_TYPE,
						BbmWebBundleKey.UPDATED_EMPS_STATUS, BbmWebBundleKey.BUNDLE_NAME,
						new String[]{String.valueOf(pageModel.getSelectedIDSize())});

		} catch (Exception ex) {
			handleUnableToUpdate(pageModel, ex);
		}
	}
}
