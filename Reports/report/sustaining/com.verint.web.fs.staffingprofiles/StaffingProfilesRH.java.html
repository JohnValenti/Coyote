package com.verint.web.fs.staffingprofiles;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.campaign.ejb.CampaignManager;
import com.bluepumpkin.ejb.bbm.campaign.model.SPEmpTemplate;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.web.core.base.RequestHandler;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;

import java.rmi.RemoteException;
import java.util.ArrayList;
import java.util.Collection;

public class StaffingProfilesRH extends RequestHandler {
	public static final String STAFFING_PROFILE_ADD = "STAFFING_PROFILE_ADD";
	public static final String STAFFING_PROFILE_REMOVE = "STAFFING_PROFILE_REMOVE";
	public static final String STAFFING_PROFILE_SAVE = "STAFFING_PROFILE_SAVE";
	public static final String STAFFING_PROFILE_REVERT = "STAFFING_PROFILE_REVERT";

	public void processRequest(RequestContext context) throws Exception {
		if (isAction(context, PageAction.SAVE)) {
			try {
				StaffingProfilesListPM pm = StaffingProfilesListPM.getInstance(context);
				processSave(context, pm);
			} catch (Exception ex) {
				PageModel pm = context.getPageModel();
				if (pm != null) {
					pm.addPageMessage(Message.ERROR_TYPE,
							FsWebBundleKey.STAFFING_PROFILES_ERROR_DURING_SAVE, FsWebBundleKey.BUNDLE_NAME);
				}
			}
		} else if (isAction(context, PageAction.DELETE)) {
			try {
				StaffingProfilesListPM pm = StaffingProfilesListPM.getInstance(context);
				processDelete(context, pm);
			} catch (Exception ex) {
				PageModel pm = context.getPageModel();
				if (pm != null) {
					pm.addPageMessage(Message.ERROR_TYPE,
							FsWebBundleKey.STAFFING_PROFILES_ERROR_DURING_REMOVE, FsWebBundleKey.BUNDLE_NAME);
				}
			}
		}
	}

	protected void processDelete(RequestContext context, StaffingProfilesListPM pageModel) throws BPException {
		CampaignManager campaignMgr = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
		try {
			campaignMgr.deleteSPEmpTemplate(pageModel.getPerfromActionOnIDsAsCollection());
		} catch (Exception rEx) {
			throw new BPException(rEx);
		}
		context.setPageModel(null); // reset the page model to remove the deleted entries
	}

	protected void processSave(RequestContext context, StaffingProfilesListPM pageModel) throws BPException {
		if (pageModel.validate()) {
			CampaignManager campaignMgr = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
			Collection<SPEmpTemplate> spEmpTemplates = pageModel.getSPEmpTemplates();
			Collection<SchedulingPeriod> spCol = new ArrayList<SchedulingPeriod>();
			try {
				campaignMgr.updateSPEmpTemplate(spEmpTemplates);
				//
				ID id = pageModel.getSchedulingPeriodID();
				SchedulingPeriod sp = campaignMgr.getSchedulingPeriodByID(id);
				sp.setMinPhantoms(pageModel.getSPMinPhantoms());
				sp.setMaxPhantoms(pageModel.getSPMaxPhantoms());
				spCol.add(sp);
				campaignMgr.updateSchedulingPeriods(spCol);
			} catch (RemoteException ex) {
				throw new BPException(ex);
			}
			context.setPageModel(null); // reset the pagemodel to set the updated values
		} else {
			context.setPageModel(pageModel);
		}
	}
}
