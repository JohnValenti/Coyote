package com.verint.web.fs.staffingprofiles;

/**
 * This Page Model is responsible for showing the Staffing Profiles linked to an SP (not to be confused with
 * com.bluepumpkin.web.bbm.people.staffingprofiles.StaffingProfilesListPM, which shows the Staffing Profiles linked
 * to organizations).
 */
public class StaffingProfilesListPM extends WorkRulesListPM {
	private static final long serialVersionUID = 1L;

	protected ToggleVisiblePC toggleAll;

	protected ID spid = null;
	private Map<ID, Organization> orgMap = null;
	private Map<ID, Campaign> campaignMap = null;
	private Map<ID, Activity> activityMap = null;
	private Map<ID, EmployeeType> empTypeMap = null;
	private Collection<SPEmpTemplate> spEmpTemplates = new ArrayList<SPEmpTemplate>();
	private int minPhantoms = 0;
	private int maxPhantoms = 9999;

	public StaffingProfilesListPM(RequestContext context) {
		super(context);

		//This is only required because this class extends WorkRulesListPM.  Not sure why this class
		//needs to extend from WorkRulesListPM.
		isByOrg = false;

		option_showTimeZoneToggle = false;

		removeChildComponent(m_list);
		m_list = new MultiColMasterDetailListPC(context);
		m_list.setIsAutoSortEnabled(true);
		this.addChildComponent(m_list);
		this.setIsMultiRowSelectable(true);

		toggleAll = new ToggleVisiblePC(m_context, "toggleAll");
		this.addChildComponent(toggleAll);
	}

	@Override
	protected void loadData() throws Exception {
		super.loadData();

		if (selectedSchedulingPeriod != null) {
			CampaignManager campaignMgr = WfmManagerFactory.getCampaignManager(m_context.isInWhatIfMode());
			spid = selectedSchedulingPeriod.getID();
			Collection spEmpTemplateCollection = campaignMgr.getSPEmpTemplatesBySP(spid);
			spEmpTemplates.addAll(spEmpTemplateCollection);

			//load min//max phantoms
			SchedulingPeriod sp = campaignMgr.getSchedulingPeriodByID(spid);
			minPhantoms = sp.getMinPhantoms();
			maxPhantoms = sp.getMaxPhantoms();
		}
	}

	@Override
	protected void initHeader() {
	}

	@Override
	protected void initSelectableItems() {
		this.removeChildComponent(m_list);
		m_list = new StaffingProfilesListPC(m_context, spEmpTemplates, null,
				isByOrg, minPhantoms, maxPhantoms, orgMap, campaignMap, activityMap, empTypeMap) {
			@Override
			protected void setRowAttributes(DefaultMultiColumnNodeData rowData,
					SPEmpTemplate spEmpTemplate) {
				StaffingProfilesListPM.this.setRowAttributes(rowData, null);
				super.setRowAttributes(rowData, spEmpTemplate);

			}
		};
		m_list.setIsAutoSortEnabled(true);
		this.addChildComponent(m_list);
		m_list.extractParameters(m_context.getRequest());
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		m_contentTitlePC.setPageTitle(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES));
	}

	public static StaffingProfilesListPM getInstance(RequestContext context) {
		return (StaffingProfilesListPM) PageModel.getInstance(context, StaffingProfilesListPM.class);
	}

	@Override
	protected void initToolbar() {
		//Enable pagination
		m_toolbar.setPagination("test", this, true, JSUtil.REFRESH_PAGE);

		boolean canAdd = isAddActionEnabled();
		if (isByOrg && selectedOrg == null) {
			canAdd = false;
		} else if (isByOrg && !m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES_ID, new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE))) {
			canAdd = false;
		} else if (!isByOrg && !m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES_ID)) {
			canAdd = false;
		}
		ToolbarTextButton addButton = m_toolbar.createPopupWindowButton(StaffingProfilesListPopupPM.POPUP_ID,
				m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_ADDPROFILES), canAdd);

		if (selectedSchedulingPeriod != null) {
			addButton.setAttribute("ownerID", selectedSchedulingPeriod.getID().toString());
		}
		addButton.setAttribute("isByOrg", String.valueOf(true));
		addButton.setAttribute("isSelectionPopup", String.valueOf(true));
		addButton.setAttribute("onClick",
				"var idList=[]; " +
						"this.selectedIDs=idList.join(','); " +
						m_toolbar.getName() + ".onSelect('" + StaffingProfilesListPopupPM.POPUP_ID + "', this);");

		m_toolbar.addToolbarElement(addButton);

		addPopupArgs(StaffingProfilesListPopupPM.POPUP_ID, StaffingProfilesListPopupPM.FORM_ACTION, "",
				StaffingProfilesListPopupPM.POPUP_ARGS, "1000,600,ctr,ctr", true, 1);

		//Remove button will be disabled by default.  It will become enabled when a user selects a staffing profile to remove.
		m_toolbar.addSubmitTextButton(PageAction.DELETE,
				m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_REMOVEPROFILES), false);

		m_toolbar.addSubmitTextButton(PageAction.SAVE, m_toolbar.i18nCore(CoreWebBundleKey.TOOLBAR_SAVE), canAdd);
		ToolbarUtil.addResetBtn(m_toolbar, canAdd);
	}

	public Collection<SPEmpTemplate> getSPEmpTemplates() {
		Map minAgentRatio = ((StaffingProfilesListPC) m_list).getMinAgentRatiosMap();
		Map maxAgentRatio = ((StaffingProfilesListPC) m_list).getMaxAgentRatiosMap();
		Map minAgent = ((StaffingProfilesListPC) m_list).getMinAgentsMap();
		Map maxAgent = ((StaffingProfilesListPC) m_list).getMaxAgentsMap();

		for (SPEmpTemplate spEmpTemplate : spEmpTemplates) {
			ID id = spEmpTemplate.getID();

			String sMinAR = (String) (minAgentRatio.get(id));
			int nMinAR = new Integer(sMinAR).intValue();

			String sMaxAR = (String) (maxAgentRatio.get(id));
			int nMaxAR = new Integer(sMaxAR).intValue();

			String sMinNum = (String) (minAgent.get(id));
			int nMinNum = new Integer(sMinNum).intValue();

			String sMaxNum = (String) (maxAgent.get(id));
			int nMaxNum = new Integer(sMaxNum).intValue();

			spEmpTemplate.setMinAllocation(nMinAR);
			spEmpTemplate.setMaxAllocation(nMaxAR);
			spEmpTemplate.setMinNumEmployees(nMinNum);
			spEmpTemplate.setMaxNumEmployees(nMaxNum);
		}

		return spEmpTemplates;
	}

	public ID getSchedulingPeriodID() {
		return spid;
	}

	public int getSPMinPhantoms() {
		Map minAgent = ((StaffingProfilesListPC) m_list).getMinAgentsMap();
		ID id = new ID("test");
		String sMinP = (String) minAgent.get(id);
		int nMinP = new Integer(sMinP);
		return nMinP;
	}

	public int getSPMaxPhantoms() {
		Map maxAgent = ((StaffingProfilesListPC) m_list).getMaxAgentsMap();
		ID id = new ID("test");
		String sMaxP = (String) maxAgent.get(id);
		int nMaxP = new Integer(sMaxP);
		return nMaxP;
	}

	/**
	 * Retrieve request parameters from request object to the page model object.
	 */
	public boolean extractParameters(HttpServletRequest request) {
		if (!(m_list instanceof StaffingProfilesListPC)) {
			this.removeChildComponent(m_list);
			m_list = new StaffingProfilesListPC(m_context, spEmpTemplates, null,
					isByOrg, minPhantoms, maxPhantoms, orgMap, campaignMap, activityMap, empTypeMap) {
				@Override
				protected void setRowAttributes(DefaultMultiColumnNodeData rowData,
						SPEmpTemplate spEmpTemplate) {
					StaffingProfilesListPM.this.setRowAttributes(rowData, null);
					super.setRowAttributes(rowData, spEmpTemplate);
				}
			};
			m_list.setIsAutoSortEnabled(true);
			this.addChildComponent(m_list);
		}

		boolean success = super.extractParameters(request);
		return success && m_list.extractParameters(request);
	}

	public boolean validate() {
		boolean retVal = super.validate();

		int totalMinNo = 0;
		int totalMaxNo = 0;
		int totalMinAgents = getSPMinPhantoms();
		int totalMaxAgents = getSPMaxPhantoms();

		Map minAgentRatio = ((StaffingProfilesListPC) m_list).getMinAgentRatiosMap();
		Map maxAgentRatio = ((StaffingProfilesListPC) m_list).getMaxAgentRatiosMap();
		Map minAgent = ((StaffingProfilesListPC) m_list).getMinAgentsMap();
		Map maxAgent = ((StaffingProfilesListPC) m_list).getMaxAgentsMap();

		for (SPEmpTemplate spET : spEmpTemplates) {
			ID id = spET.getID();

			String sMinAR = (String) (minAgentRatio.get(id));
			int nMinAR = new Integer(sMinAR).intValue();

			String sMaxAR = (String) (maxAgentRatio.get(id));
			int nMaxAR = new Integer(sMaxAR).intValue();

			String sMinNum = (String) (minAgent.get(id));
			int nMinNum = new Integer(sMinNum).intValue();

			String sMaxNum = (String) (maxAgent.get(id));
			int nMaxNum = new Integer(sMaxNum).intValue();

			if (nMinAR > 100 || nMinAR < 0) {
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_MIN_PERCENT_BETWEEN_RANGE));
				retVal = false;
			}
			if (nMaxAR > 100 || nMaxAR < 0) {
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_MAX_PERCENT_BETWEEN_RANGE));
				retVal = false;
			}
			if (nMaxAR < nMinAR) {
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_MIN_PERCENT_GREATER_THAN_MAX_PERCENT));
				retVal = false;
			}
			if (nMinNum > 9999) {
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_MIN_NO_BETWEEN_RANGE));
				retVal = false;
			}
			if (nMaxNum > 9999) {
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_MAX_NO_BETWEEN_RANGE));
				retVal = false;
			}
			if (nMaxNum < nMinNum) {
				addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_MIN_NO_GREATER_THAN_MAX_NO));
				retVal = false;
			}
			totalMinNo += nMinNum;
			totalMaxNo += nMaxNum;
		}

		if (totalMinAgents > 9999) {
			addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_TOTAL_MIN_NO_BTN_RANGE));
			retVal = false;
		}
		if (totalMaxAgents > 9999) {
			addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_TOTAL_MAX_NO_BTN_RANGE));
			retVal = false;
		}
		if (totalMinAgents > totalMaxAgents) {
			addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_MIN_NO_GREATER_THAN_MAX_NO));
			retVal = false;
		}
		if (totalMaxNo < totalMinAgents) {
			addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_SUM_MAX_CANNOT_BE_LESS_THAN_TOTAL_MIN_NO));
			retVal = false;
		}
		if (totalMinNo > totalMaxAgents) {
			addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_SUM_MIN_CANNOT_BE_MORE_THAN_TOTAL_MAX_NO));
			retVal = false;
		}
		if (totalMaxAgents == 0) {
			addPageMessage(Message.ERROR_TYPE, m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_STAFFING_PROFILES_TOTAL_MAX_NO_CANNOT_BE_ZERO));
			retVal = false;
		}
		return retVal;
	}

	protected String isAuthorized() {
		if (m_context.getUser().isAuthorized(PrivilegeKeys.CORE_CONFIGUREORGANIZATION, new ScopeID(selectedCampaign.getID(), Scope.CAMPAIGN_SCOPE)) &&
				(m_context.getUser().isAuthorized(PrivilegeKeys.FS_EDITWORKRULES_ID, new ScopeID(selectedCampaign.getID(), Scope.CAMPAIGN_SCOPE))))

		{
			return "true";
		} else {
			return "false";
		}
	}

}
