package com.bluepumpkin.web.rm.requests.shiftbidding;

import com.witness.web.uif.pagemodel.*;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.listconfig.ListConfigSelectorPC;
import com.witness.web.uif.listconfig.ListConfigFP;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.data.wrapper.NumberWrapper;
import com.bluepumpkin.web.rm.requests.*;
import com.bluepumpkin.web.rm.keys.RmKeys;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestAggregate;
import com.bluepumpkin.ejb.rm.requests.common.model.Request;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidRequest;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidder;

/**
 * File:         AgentsShiftBiddingRequestsPM
 * Description:  Page model for the list of shift bidding requests 
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       pbosin, David Su
 * @version      1.0
 * Created on July 29, 2002, 2:38 PM
 */
public class AgentsShiftBiddingRequestsPM extends AgentsRequestsPM {

	/**
	 * Constructor
	 */
	public AgentsShiftBiddingRequestsPM(RequestContext context) {
		super(context, Request.REQUESTTYPE_SHIFTBID);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, AgentsShiftBiddingRequestsPM.class);
	}

	/**
	 * To be overwritten to load Page specific data
	 */
	protected void loadSpecifyData() throws Exception {
	}

	/**
	 * Initialize Request Collection
	 */
	protected void initRequestCollection() throws Exception {
		if (isAuthToViewSBRqts()) {
			super.initRequestCollection();
		}
	}

	/**
	 * Create Instanse of ListConfigSelectedPC
	 */
	protected ListConfigSelectorPC createListConfigSelectorPC() {
		return new ListConfigSelectorPC(m_context, AgentsShiftBiddingRequestsListConfig.class);
	}

	/**
	 * Add Cell Data for the specified List Config Field Property
	 */
	protected void addCellData(String rowID, RequestAggregate request, ListConfigFP fp,
			DefaultMultiColumnNodeData row_data) {
		if (fp.hasFieldName(RmWebBundleKey.BID_NAME)) {
			ShiftBidRequest sbRequest = (ShiftBidRequest)request;
			row_data.add( sbRequest.getName() );
		}	else if (fp.hasFieldName(Request.SORT_SHIFTBID_SENIORITY)) {
			ShiftBidRequest sbRequest = (ShiftBidRequest)request;
			Employee emp = sbRequest.getOptMethods().getEmployee();
			row_data.add( ShiftBidUtil.getSeniority(emp, m_view_tz, m_localizer) );
		}	else if (fp.hasFieldName(Request.SORT_EMP_RANK)) {
			ShiftBidRequest sbRequest = (ShiftBidRequest)request;
			Employee emp = sbRequest.getOptMethods().getEmployee();
			row_data.add( new NumberWrapper(emp.getRank(), m_localizer) );
		}	else if (fp.hasFieldName(Request.SORT_SHIFTBID_POINTS)) {
			ShiftBidRequest sbRequest = (ShiftBidRequest)request;
			ShiftBidder bidder = sbRequest.getOptMethods().getShiftBidder();
			row_data.add( new NumberWrapper(bidder.getOptMethods().getAccumulatedBonus(), m_localizer) );
		}	else if (fp.hasFieldName(Request.SORT_SHIFTBID_BIDDERBONUS)) {
			ShiftBidRequest sbRequest = (ShiftBidRequest)request;
			ShiftBidder bidder = sbRequest.getOptMethods().getShiftBidder();
			row_data.add( new NumberWrapper(bidder.getBonusThisAuction(), m_localizer) );
		}	else if (fp.hasFieldName(Request.SORT_SHIFTBID_SCORE)) {
			ShiftBidRequest sbRequest = (ShiftBidRequest)request;
			ShiftBidder bidder = sbRequest.getOptMethods().getShiftBidder();
			row_data.add( new NumberWrapper(bidder.getOptMethods().getScoreWithBonus(), m_localizer) );
		}	else if (fp.hasFieldName(Request.SORT_SHIFTBID_PREFERENCE)) {
			ShiftBidRequest sbRequest = (ShiftBidRequest)request;
			row_data.add( new NumberWrapper(sbRequest.getPreference(), m_localizer ) );
		} else if (fp.hasFieldName(Request.SORT_SHIFTBID_ISBONUSUSED)) {
			ShiftBidRequest sbRequest = (ShiftBidRequest)request;
			String isBonusUsed = m_localizer.formatYesNo(sbRequest.getIsBonusUsed());
			row_data.add( isBonusUsed );
		} else {
			super.addCellData(rowID, request, fp, row_data);
		}
	}

	/**
	 * Return Content Title Item Name
	 */
	protected String getContentTitleItemName() {
		return i18n(m_bundle, RmWebBundleKey.REQUEST_TYPE_SHIFT_BIDDING);
	}

	/**
	 * Add Create Request Button
	 */
	protected void addCreateRequestButton() {
		// Manager does not create Shift Bid Request here
	}

	/**
	 * Return true if Tentative Actions are enabled
	 */
	protected boolean isTentativeActionEnabled() {
		return false; // Not enabled for Shift Bid Requests
	}

	/** Return URL for HTML Form Action
	 */
	public String getFormAction() {
		if (isAuthToViewSBRqts()) {
			return RmKeys.RH_AG_REQUEST_SHIFTBID_LIBT;
		} else {
			return RmKeys.RH_AG_REQUEST_ALL_LIST;
		}
	}
	
	/**
	 * Get Selected Filter. To be overridden by subclasses.
	 * Subclasses should return m_rfilterBoxPC.getSelectedFilterName().
	 */
	public String getSelectedFilterName() {
		return m_rfilterBoxPC.getSelectedFilterName();
	}
}
