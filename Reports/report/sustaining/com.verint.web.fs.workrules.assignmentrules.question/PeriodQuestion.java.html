package com.verint.web.fs.workrules.assignmentrules.question;

import com.bluepumpkin.ejb.bbm.workrules.model.CalendarPeriod;
import com.bluepumpkin.ejb.bbm.workrules.model.CalendarPeriod.CalendarUnit;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.verint.web.fs.workrules.assignmentrules.*;

import java.util.*;

public class PeriodQuestion implements Question {

	private ComplexWorkRule data_assignmentRule;
	private List<PeriodResponse> data_responses = new ArrayList<PeriodQuestion.PeriodResponse>();
	private Response data_initialAnswer;
	private QuestionDataProvider data_provider;

	public PeriodQuestion(QuestionDataProvider dataProvider) throws Exception {
		super();
		data_provider = dataProvider;
		data_assignmentRule = dataProvider.getAssignmentRuleBeingEdited();
		constructResponses();
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.CalendarPeriod;
	}

	@Override
	public List<PeriodResponse> getResponses() {
		return data_responses;
	}

	@Override
	public Response getInitialAnswer() {
		return data_initialAnswer;
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		if (assignmentRule.getCalendarPeriod() == null) {
			assignmentRule.setCalendarPeriod(new CalendarPeriod());
		}

		CalendarPeriod cp = assignmentRule.getCalendarPeriod();
		cp.setCalendarUnit(((PeriodResponse) answer).getResponseType().getCalendarPeriodUnit());
		cp.setUnitCount(1);
		if (answer.getResponseParameters().size() > 0) {
			cp.setUnitCount((Integer) answer.getResponseParameters().iterator().next().getValue());
			cp.setCalendarUnit(((PeriodResponse) answer).getResponseType().getCalendarPeriodUnit());
		}
	}

	private void constructResponses() throws Exception {
		boolean monthlyEnabled = data_provider.isMonthlyRulesEnabled();
		boolean multiWeekEnabled = data_provider.isMultiWeekRulesEnabled();
		boolean isStartRule = RuleTypeQuestion.RuleTypeResponseType.Start.equals(data_provider.getRuleType());

		// Get a list of response types.
		PeriodResponseType[] responseTypes = PeriodResponseType.values();

		// X Month options are only visible to start rules or when multiWeek is enabled, so remove it for all other cases
		if (!isStartRule && !multiWeekEnabled) {
			List<PeriodResponseType> responseTypeList = new ArrayList<PeriodQuestion.PeriodResponseType>(Arrays.asList(responseTypes));
			responseTypeList.remove(PeriodResponseType.Month);
			responseTypes = responseTypeList.toArray(new PeriodResponseType[responseTypeList.size()]);
		}

		if (isStartRule) {
			// Not sure if there is a good reason, but the FS client doesn't show monthly options for start rules (QC-85568)
			monthlyEnabled = false;
		}

		for (PeriodResponseType type : responseTypes) {
			if (!monthlyEnabled && monthlyResponseTypes.contains(type)) {
				continue;
			}
			if (!multiWeekEnabled && multiWeekResponseTypes.contains(type)) {
				continue;
			}
			if (isStartRule && !startRuleResponseTypes.contains(type)) {
				continue;
			}

			// Construct parameter with constraints
			List<ResponseParam> params = new ArrayList<ResponseParam>();
			if (type.getParamTypes().size() > 0) {
				List<ResponseParamConstraint> constraints = new ArrayList<ResponseParamConstraint>();
				switch (type) {
					case Day:
						if (multiWeekEnabled) {
							constraints.add(new IntegerConstraint(1, 730));
						} else {
							constraints.add(new IntegerConstraint(1, 7));
						}
						break;
					case WeekX:
						if (multiWeekEnabled) {
							constraints.add(new IntegerConstraint(1, 104));
						} else {
							constraints.add(new IntegerConstraint(1, 8));
						}
						break;
					case Month:
						if (multiWeekEnabled && !isStartRule) { // Excluding start rules from this branch might be a bug in FS that we are copying here.
							constraints.add(new IntegerConstraint(1, 24));
						} else {
							constraints.add(new IntegerConstraint(1, 2));
						}
						break;
					case Year:
						constraints.add(new IntegerConstraint(1, 2));
						break;
					default:
						// Do nothing. Not all types have constraints.
				}
				ResponseParam param = new ResponseParam(
						ResponseParamType.Integer,
						constraints,
						(data_assignmentRule == null || data_assignmentRule.getCalendarPeriod() == null) ? 1
								: data_assignmentRule.getCalendarPeriod().getUnitCount());
				params.add(param);
			}
			data_responses.add(new PeriodResponse(type, params));
		}

		// find initial answer
		if (data_assignmentRule != null) {
			CalendarPeriod cp = data_assignmentRule.getCalendarPeriod();
			if (cp != null) {
				CalendarUnit cu = cp.getCalendarUnit();
				if (cu != null) {
					for (PeriodResponse response : data_responses) {
						if (cu.equals(response.getResponseType().getCalendarPeriodUnit())) {
							// There are two ResponseTypes that correspond to the CalendarUnit.Week value, so check them
							if (CalendarUnit.Week.equals(response.getResponseType().getCalendarPeriodUnit())) {
								// If the unit count is 1, we want the Week response type (every week). If not, keep looking.
								if (cp.getUnitCount() == 1 && !PeriodResponseType.Week.equals(response.getResponseType()) && !PeriodResponseType.WeekX.equals(response.getResponseType())) {
									continue;
								}
								// If the unit count is not 1, we want the WeekX response type (every x weeks). If not, keep looking.
								if (cp.getUnitCount() != 1 && !PeriodResponseType.WeekX.equals(response.getResponseType())) {
									continue;
								}
							}
							data_initialAnswer = response;
							break;
						}
					}
				}
			}
		}
	}

	private static Set<PeriodResponseType> multiWeekResponseTypes = new HashSet<PeriodResponseType>();

	static {
		multiWeekResponseTypes.add(PeriodResponseType.Year);
		multiWeekResponseTypes.add(PeriodResponseType.RollingWeek);
	}

	private static Set<PeriodResponseType> monthlyResponseTypes = new HashSet<PeriodQuestion.PeriodResponseType>();

	static {
		monthlyResponseTypes.add(PeriodResponseType.Month28Day);
		monthlyResponseTypes.add(PeriodResponseType.Month29Day);
		monthlyResponseTypes.add(PeriodResponseType.Month30Day);
		monthlyResponseTypes.add(PeriodResponseType.Month31Day);
	}

	private static Set<PeriodResponseType> startRuleResponseTypes = new HashSet<PeriodQuestion.PeriodResponseType>();

	static {
		startRuleResponseTypes.add(PeriodResponseType.Week);
		startRuleResponseTypes.add(PeriodResponseType.WeekX);
		startRuleResponseTypes.add(PeriodResponseType.Month);
	}

	public enum PeriodResponseType implements ResponseType {
		Day(CalendarUnit.Day, ResponseParamType.Integer),
		Week(CalendarUnit.Week),
		WeekX(CalendarUnit.Week, ResponseParamType.Integer),
		RollingWeek(CalendarUnit.RollingWeek, ResponseParamType.Integer),
		Month(CalendarUnit.Month, ResponseParamType.Integer),
		Month28Day(CalendarUnit.Month28Day),
		Month29Day(CalendarUnit.Month29Day),
		Month30Day(CalendarUnit.Month30Day),
		Month31Day(CalendarUnit.Month31Day),
		Year(CalendarUnit.Year, ResponseParamType.Integer);

		private CalendarUnit calendarPeriodUnit;
		private ResponseParamType[] paramTypes;

		PeriodResponseType(CalendarUnit calendarPeriodUnit, ResponseParamType... paramTypes) {
			this.calendarPeriodUnit = calendarPeriodUnit;
			this.paramTypes = paramTypes;
		}

		public CalendarUnit getCalendarPeriodUnit() {
			return calendarPeriodUnit;
		}

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Arrays.asList(paramTypes);
		}
	}

	public static class PeriodResponse extends ResponseImpl<PeriodResponseType> {
		public PeriodResponse(PeriodResponseType responseType, List<ResponseParam> params) {
			super(QuestionType.CalendarPeriod, responseType, params);
		}

	}

}
