package com.verint.web.fs.workrules.assignmentrules.question;

import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule.WorkUnitCountConstraint;
import com.verint.web.fs.workrules.assignmentrules.*;

import java.util.*;

public class CountQuestion implements Question {
	private ComplexWorkRule data_assignmentRule;
	private List<CountResponse> data_responses = new ArrayList<CountQuestion.CountResponse>();
	private CountResponse data_initialAnswer = null;

	public CountQuestion(ComplexWorkRule assignmentRule) {
		super();
		this.data_assignmentRule = assignmentRule;
		constructResponses();
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.Count;
	}

	@Override
	public List<? extends Response> getResponses() {
		return data_responses;
	}

	@Override
	public Response getInitialAnswer() {
		return data_initialAnswer;
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		CountResponseType responseType = (CountResponseType) answer.getResponseType();
		assignmentRule.setWorkUnitCountConstraint(responseType.getWorkUnitCountConstraint());
		switch (responseType) {
			case AtLeast:
				assignmentRule.setMinWorkUnitCount((Integer) answer.getResponseParameters().iterator().next().getValue());
				break;
			case AtMost:
				assignmentRule.setMaxWorkUnitCount((Integer) answer.getResponseParameters().iterator().next().getValue());
				break;
			case Exactly:
				Integer num = (Integer) answer.getResponseParameters().iterator().next().getValue();
				assignmentRule.setMinWorkUnitCount(num);
				assignmentRule.setMaxWorkUnitCount(num);
				break;
			case Between:
				Iterator<? extends ResponseParam> iter = answer.getResponseParameters().iterator();
				Integer min = (Integer) iter.next().getValue();
				Integer max = (Integer) iter.next().getValue();

				// If min is greater than max, swap them.
				if (min != null && max != null && min > max) {
					Integer temp = min;
					min = max;
					max = temp;
				}

				assignmentRule.setMinWorkUnitCount(min);
				assignmentRule.setMaxWorkUnitCount(max);
				break;
			case FairNumber:
				// This seems arbitrary, but it is what the old F&S client did
				assignmentRule.setMinWorkUnitCount(1);
				assignmentRule.setMaxWorkUnitCount(1);
				break;
		}
	}

	private void constructResponses() {
		WorkUnitCountConstraint wucc = null;
		if (data_assignmentRule != null) {
			wucc = data_assignmentRule.getWorkUnitCountConstraint();
		}

		ResponseParamConstraint constraint = new IntegerConstraint(0, 9999);
		ResponseParam atLeastNum = new ResponseParam(
				ResponseParamType.Integer,
				Collections.singletonList(constraint),
				(data_assignmentRule == null || !WorkUnitCountConstraint.AtLeast.equals(wucc)) ? 1 : data_assignmentRule.getMinWorkUnitCount());
		CountResponse atLeast = new CountResponse(CountResponseType.AtLeast, Collections.singletonList(atLeastNum));
		data_responses.add(atLeast);

		ResponseParam atMostNum = new ResponseParam(
				ResponseParamType.Integer,
				Collections.singletonList(constraint),
				(data_assignmentRule == null || !WorkUnitCountConstraint.NoMoreThan.equals(wucc) || data_assignmentRule.getMaxWorkUnitCount() < 0) ?
						1 : data_assignmentRule.getMaxWorkUnitCount());
		CountResponse atMost = new CountResponse(CountResponseType.AtMost, Collections.singletonList(atMostNum));
		data_responses.add(atMost);

		ResponseParam exactlyNum = new ResponseParam(
				ResponseParamType.Integer,
				Collections.singletonList(constraint),
				(data_assignmentRule == null || !WorkUnitCountConstraint.Exactly.equals(wucc)) ? 1 : data_assignmentRule.getMinWorkUnitCount());
		CountResponse exactly = new CountResponse(CountResponseType.Exactly, Collections.singletonList(exactlyNum));
		data_responses.add(exactly);

		ResponseParam betweenLow = new ResponseParam(
				ResponseParamType.Integer,
				Collections.singletonList(constraint),
				(data_assignmentRule == null || !WorkUnitCountConstraint.Between.equals(wucc)) ? 1 : data_assignmentRule.getMinWorkUnitCount());
		ResponseParam betweenHigh = new ResponseParam(
				ResponseParamType.Integer,
				Collections.singletonList(constraint),
				(data_assignmentRule == null || data_assignmentRule.getMaxWorkUnitCount() < 0 || !WorkUnitCountConstraint.Between.equals(wucc)) ?
						2 : data_assignmentRule.getMaxWorkUnitCount());
		CountResponse between = new CountResponse(CountResponseType.Between, Arrays.asList(new ResponseParam[]{
				betweenLow, betweenHigh}));
		data_responses.add(between);

		CountResponse fairNum = new CountResponse(
				CountResponseType.FairNumber,
				Collections.emptyList());
		data_responses.add(fairNum);

		// Find initial answer
		if (data_assignmentRule != null) {
			switch (data_assignmentRule.getWorkUnitCountConstraint()) {
				case AtLeast:
					data_initialAnswer = atLeast;
					break;
				case NoMoreThan:
					data_initialAnswer = atMost;
					break;
				case Exactly:
					data_initialAnswer = exactly;
					break;
				case Between:
					data_initialAnswer = between;
					break;
				case Fairly:
					data_initialAnswer = fairNum;
					break;
				case FairlyStartingBetween:
					data_initialAnswer = fairNum;
					break;
				case FairlyEndingBetween:
					data_initialAnswer = fairNum;
					break;
			}
		}
	}

	public enum CountResponseType implements ResponseType {
		AtLeast(WorkUnitCountConstraint.AtLeast, ResponseParamType.Integer),
		AtMost(WorkUnitCountConstraint.NoMoreThan, ResponseParamType.Integer),
		Exactly(WorkUnitCountConstraint.Exactly, ResponseParamType.Integer),
		Between(WorkUnitCountConstraint.Between, ResponseParamType.Integer, ResponseParamType.Integer),
		FairNumber(WorkUnitCountConstraint.Fairly);

		private ResponseParamType[] paramTypes;
		private WorkUnitCountConstraint wucc;

		CountResponseType(WorkUnitCountConstraint wucc, ResponseParamType... paramTypes) {
			this.wucc = wucc;
			this.paramTypes = paramTypes;
		}

		public WorkUnitCountConstraint getWorkUnitCountConstraint() {
			return wucc;
		}

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Arrays.asList(paramTypes);
		}
	}

	public class CountResponse extends ResponseImpl<CountResponseType> {

		public CountResponse(CountResponseType responseType, List<ResponseParam> parameters) {
			super(QuestionType.Count, responseType, parameters);
		}
	}
}
