package com.verint.web.fs.workrules.assignmentrules.question;

import java.util.Calendar;
import java.util.Collections;
import java.util.List;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.verint.web.fs.workrules.assignmentrules.DayOfWeekConstraint;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseImpl;
import com.verint.web.fs.workrules.assignmentrules.ResponseParam;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamConstraint;
import com.verint.web.fs.workrules.assignmentrules.ResponseParamType;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;
import com.verint.web.fs.workrules.assignmentrules.question.PeriodQuestion.PeriodResponseType;

public class EndDateQuestion implements Question {

	private ComplexWorkRule data_assignmentRule;
	private ResponseImpl<EndDateResponseType> response;

	public EndDateQuestion(QuestionDataProvider dataProvider) throws BPException {
		super();
		this.data_assignmentRule = dataProvider.getAssignmentRuleBeingEdited();

		DayOfWeekConstraint dowC = null;
		if (dataProvider.getPeriodType().equals(PeriodResponseType.Week))
			dowC = new DayOfWeekConstraint(dataProvider
					.getOrganization()
					.getWeekStartDay());

		LocalDate date = null;
		Calendar c = Calendar.getInstance(TimeZoneUtil.GMT_TIMEZONE);
		c.set(2078, Calendar.DECEMBER, 31, 0, 0, 0);
		c.set(Calendar.MILLISECOND, 0);
		date = new LocalDate(c.getTime(), TimeZoneUtil.GMT_TIMEZONE);
		
		if (data_assignmentRule != null) {
			date = data_assignmentRule.getEndDate();
		}
		ResponseParam rp = new ResponseParam(
				ResponseParamType.Date,
				dowC==null?Collections.<ResponseParamConstraint>emptyList():Collections.<ResponseParamConstraint> singletonList(dowC),
				date);
		this.response = new ResponseImpl<EndDateResponseType>(
				getQuestionType(),
				EndDateResponseType.EndDate,
				Collections.singletonList(rp));
	}

	@Override
	public QuestionType getQuestionType() {
		return QuestionType.EndDate;
	}

	@Override
	public List<? extends Response> getResponses() {
		return Collections.singletonList(response);
	}

	@Override
	public Response getInitialAnswer() {
		return response;
	}

	@Override
	public void applyAnswer(Response answer, ComplexWorkRule assignmentRule) {
		assignmentRule.setEndDate((LocalDate) answer.getResponseParameters().iterator().next().getValue());
	}

	public enum EndDateResponseType implements ResponseType {
		EndDate;

		@Override
		public List<ResponseParamType> getParamTypes() {
			return Collections.<ResponseParamType> singletonList(ResponseParamType.Date);
		}
	}
}
