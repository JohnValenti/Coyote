	/**
	 * Position picker
	 * 
	 * @param scenario
	 * @param businessUnits
	 * @return
	 */
	public List<StringsPair> getResourceClasses(String scenario, String businessUnits, boolean includeAllOption) {
		List<StringsPair> lResourceClasses = new ArrayList<StringsPair>();
		if (!businessUnits.equals("")) {
			try {
				JdmoRowset rs;

				StringBuilder pStmt = new StringBuilder();

				pStmt.append("SELECT  Id, Name FROM dbo.ResourceClasses WHERE Id IN (")
						.append("SELECT DISTINCT ResourceClass FROM dbo.EffectiveResourceTypes()")
						.append(" WHERE Branch IN (SELECT value FROM dbo.TokensBy(?,',')) AND Reportable = 1) ")
						.append(" ORDER BY DisplayOrder ");

				JdmoQuery query = dmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
				query.setParString(1, businessUnits);
				int i = 0;
				if (includeAllOption) {
					lResourceClasses.add(0, new StringsPair());
				} else {
					i = -1;
				}
				rs = dmo.createRowset(query, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
				while (rs.next()) {
					StringsPair resourceClass = new StringsPair();
					resourceClass.setKey(new ID(rs.getInt("Id")).toString());
					resourceClass.setValue(rs.getString("Name"));
					lResourceClasses.add(++i, resourceClass);
				}
			} catch (Exception se) {
				cat.error("Error in RfsReportParametersDAO.getResourceClasses() : ", se);
			} finally {
				cleanUp();
			}
		}
		return lResourceClasses;
	}

	/**
	 * Get Staff mix Filter Details
	 * 
	 * @param scenario
	 * @param staffMixFilter
	 * @return RfsStaffMixFilter
	 */
	public RfsStaffMixFilter getStaffMixFilterDetails(String scenario, String staffMixFilterId) {
		RfsStaffMixFilter staffMixFilter = new RfsStaffMixFilter();
		JdmoRowset rs;
		StringBuilder pStmt = new StringBuilder();
		try {
			pStmt.append(" SELECT Id, Name, MaxStaffMixesAfterLowestCostFilter, MaxOverstaffingRatioByAssignedVsServiceHours, ")
					.append("IncludeClosestToActualHeadCounts, IncludeClosestToBudgetedHeadCounts, IncludeClosestToCustomFteFractions ")
					.append("FROM StaffMixFilters WHERE Id = ? and Scenario = ? ");

			JdmoQuery query = dmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			query.setParInt(1, Integer.parseInt(staffMixFilterId));
			query.setParInt(2, Integer.parseInt(scenario));

			rs = dmo.createRowset(query, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			while (rs.next()) {
				staffMixFilter.setName(rs.getString("Name"));
				staffMixFilter.setMaxStaffMixesAfterLowCost(rs.getInt("MaxStaffMixesAfterLowestCostFilter"));
				staffMixFilter.setMaxOverstaffingRatio(rs.getInt("MaxOverstaffingRatioByAssignedVsServiceHours"));
				staffMixFilter.setIncludeClosestToActual(rs.getBoolean("IncludeClosestToActualHeadCounts"));
				staffMixFilter.setIncludeClosestToBudgeted(rs.getBoolean("IncludeClosestToBudgetedHeadCounts"));
				staffMixFilter.setIncludeClosestToFte(rs.getBoolean("IncludeClosestToCustomFteFractions"));
			}
			pStmt.setLength(0);
			pStmt.append("SELECT StaffMixFilter, EmployeeCategory, MinHeadCount, MaxHeadCount, MinHeadCountFraction, MaxHeadCountFraction,")
					.append("MinFteFraction, MaxFteFraction, MaxHeadCountBelowActual, MaxHeadCountAboveActual, MaxHeadCountBelowBudgeted, ")
					.append("MaxHeadCountAboveBudgeted, CustomFteFraction FROM StaffMixFilterEmployeeCategories WHERE StaffMixFilter = ?");
			query = dmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			query.setParInt(1, Integer.parseInt(staffMixFilterId));

			rs = dmo.createRowset(query, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			StringBuilder minHeadCounts = new StringBuilder();
			StringBuilder maxHeadCounts = new StringBuilder();
			StringBuilder minPercentHeadCounts = new StringBuilder();
			StringBuilder maxPercentHeadCounts = new StringBuilder();
			StringBuilder minPercentFte = new StringBuilder();
			StringBuilder maxPercentFte = new StringBuilder();
			StringBuilder headCountsBelowActual = new StringBuilder();
			StringBuilder headCountsAboveActual = new StringBuilder();
			StringBuilder headCountsBelowBudgeted = new StringBuilder();
			StringBuilder headCountsAboveBudgeted = new StringBuilder();
			StringBuilder customPercentFte = new StringBuilder();
			while (rs.next()) {
				minHeadCounts.append((minHeadCounts.length() == 0 ? "" : ",")).append(rs.getInt("MinHeadCount"));
				maxHeadCounts.append((maxHeadCounts.length() == 0 ? "" : ",")).append(rs.getInt("MaxHeadCount"));
				minPercentHeadCounts.append((minPercentHeadCounts.length() == 0 ? "" : ",")).append(rs.getInt("MinHeadCountFraction") * 100);
				maxPercentHeadCounts.append((maxPercentHeadCounts.length() == 0 ? "" : ",")).append(rs.getInt("MaxHeadCountFraction") * 100);
				minPercentFte.append((minPercentFte.length() == 0 ? "" : ",")).append(rs.getInt("MinFteFraction") * 100);
				maxPercentFte.append((maxPercentFte.length() == 0 ? "" : ",")).append(rs.getInt("MaxFteFraction") * 100);
				headCountsBelowActual.append((headCountsBelowActual.length() == 0 ? "" : ",")).append(rs.getInt("MaxHeadCountBelowActual"));
				headCountsAboveActual.append((headCountsAboveActual.length() == 0 ? "" : ",")).append(rs.getInt("MaxHeadCountAboveActual"));
				headCountsBelowBudgeted.append((headCountsBelowBudgeted.length() == 0 ? "" : ",")).append(rs.getInt("MaxHeadCountBelowBudgeted"));
				headCountsAboveBudgeted.append((headCountsAboveBudgeted.length() == 0 ? "" : ",")).append(rs.getInt("MaxHeadCountAboveBudgeted"));
				customPercentFte.append((customPercentFte.length() == 0 ? "" : ",")).append(rs.getInt("CustomFteFraction") * 100);
			}

			staffMixFilter.setMinHeadCounts(minHeadCounts.toString());
			staffMixFilter.setMaxHeadCounts(maxHeadCounts.toString());
			staffMixFilter.setMinPercentHeadCounts(minPercentHeadCounts.toString());
			staffMixFilter.setMaxPercentHeadCounts(maxPercentHeadCounts.toString());
			staffMixFilter.setMinPercentFte(minPercentFte.toString());
			staffMixFilter.setMaxPercentFte(maxPercentFte.toString());
			staffMixFilter.setHeadCountsBelowActual(headCountsBelowActual.toString());
			staffMixFilter.setHeadCountsAboveActual(headCountsAboveActual.toString());
			staffMixFilter.setHeadCountsBelowBudgeted(headCountsBelowBudgeted.toString());
			staffMixFilter.setHeadCountsAboveBudgeted(headCountsAboveBudgeted.toString());
			staffMixFilter.setCustomPercentFte(customPercentFte.toString());
		} catch (Exception se) {
			cat.error("Error in RfsReportParametersDAO.getStaffMixFilterDetails() : ", se);
		} finally {
			cleanUp();
		}
		return staffMixFilter;
	}

	/**
	 * Get all branches in current scenario for the branch picker when no view
	 * is selected
	 * 
	 * @param scenario
	 * @return
	 */
	public Collection<RfsBranch> getBranchesInCurrentScenario(String scenario) {

		String QUERY_BRANCHES_IN_SCENARIO = "SELECT Id,Branch, Label, Parent, BranchView FROM dbo.BranchViewNodes WHERE Id IN "
				+ "	(SELECT Id FROM (SELECT Id, Row_Number() over(partition by Branch order by Id) RowNum"
				+ "	FROM dbo.BranchViewNodes WHERE BranchView IN (SELECT Id FROM dbo.BranchViews where Scenario = ?) "
				+ "	AND Branch IS NOT NULL) branches WHERE branches.RowNum = 1) ";

		Collection<RfsBranch> branches = new ArrayList<RfsBranch>();
		JdmoRowset rs = null;
		try {
			cat.debug("getAllBranchesInCurrentScenario : query = " + QUERY_BRANCHES_IN_SCENARIO);
			JdmoQuery jq = dmo.createQuery(QUERY_BRANCHES_IN_SCENARIO, Jdmo.PARAM_QUERY);
			jq.setParInt(1, Integer.parseInt(scenario));
			rs = dmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			while (rs.next()) {
				RfsBranch bu = new RfsBranch();
				bu.setFieldValue(RfsBranchFieldInfo.BRANCH_ID, new ID(rs.getString(1)));
				bu.setFieldValue(RfsBranchFieldInfo.BRANCH_ID_NAME, rs.getString(2));
				bu.setFieldValue(RfsBranchFieldInfo.BRANCH_LABEL, rs.getString(3));
				bu.setFieldValue(RfsBranchFieldInfo.BRANCH_PARENTID, rs.getString(4) != null ? new ID(rs.getString(4)) : new ID(""));
				bu.setFieldValue(RfsBranchFieldInfo.BRANCH_VIEW, rs.getString(5));
				branches.add(bu);
			}
		} catch (Exception se) {
			cat.error("Error in RfsReportParametersDAO.getBranchesInCurrentScenario() : ", se);
		} finally {
			cleanUp();
		}
		return branches;
	}
