package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model;

import java.util.Comparator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.util.RmUtil;


public class UnsubmittedShiftBidPreference implements java.io.Serializable {

	private static final long serialVersionUID = 1L;

	public static final Comparator<UnsubmittedShiftBidPreference> PREFERENCE_COMPARATOR = new Comparator<UnsubmittedShiftBidPreference>() {
		@Override
		public int compare(UnsubmittedShiftBidPreference o1, UnsubmittedShiftBidPreference o2) {
			return RmUtil.integerCompare(o1.getUnSubmittedPreference(), o2.getUnSubmittedPreference());
		}
	};

	ID biddableScheduleID;
	ID employeeID;
	//The UNSUBMITTEDSHIFTBIDPREFERENCE table contain EmployeeID not BidderID. bidderID is here just to make the update easier. 
	ID bidderID;
	int unSubmittedPreference;
	//the number of available shifts for this biddable schedule
	int numberOfAvailableShifts;

	public int getNumberOfAvailableShifts() {
		return numberOfAvailableShifts;
	}

	public void setNumberOfAvailableShifts(int numberOfAvailableShifts) {
		this.numberOfAvailableShifts = numberOfAvailableShifts;
	}

	public ID getBidderID() {
		return bidderID;
	}

	public void setBidderID(ID bidderID) {
		this.bidderID = bidderID;
	}

	public ID getBiddableScheduleID() {
		return biddableScheduleID;
	}

	public void setBiddableScheduleID(ID biddableScheduleID) {
		this.biddableScheduleID = biddableScheduleID;
	}

	public ID getEmployeeID() {
		return employeeID;
	}

	public void setEmployeeID(ID employeeID) {
		this.employeeID = employeeID;
	}

	public int getUnSubmittedPreference() {
		return unSubmittedPreference;
	}

	public void setUnSubmittedPreference(int preference) {
		this.unSubmittedPreference = preference;
	}

}
