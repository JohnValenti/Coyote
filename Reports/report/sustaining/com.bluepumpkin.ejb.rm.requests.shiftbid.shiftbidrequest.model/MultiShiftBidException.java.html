package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model;

import java.util.TimeZone;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;

public class MultiShiftBidException extends RuntimeException {

	private static final long serialVersionUID = 1L;

	final String bidTemplateName;
	final RmHardValidationException hardValidationException;
	final Object[] params;
	final String bundleKey;



	private MultiShiftBidException(String templateName, RmHardValidationException hardValidationException, String bundleKey, Object[] params) {
		this.bidTemplateName = templateName;
		this.hardValidationException = hardValidationException;
		this.params = params;
		this.bundleKey = bundleKey;
	}

	public static MultiShiftBidException fromBundleKey(String bundleKey, Object... params) {
		return new MultiShiftBidException(null, null, bundleKey, params);
	}

	public static MultiShiftBidException fromHardValidation(String templateName, RmHardValidationException e) {
		return new MultiShiftBidException(templateName, e, null, null);
	}

	public static MultiShiftBidException fromTemplateName(String templateName) {
		return new MultiShiftBidException(templateName, null, null, null);
	}

	public String getAssociatedMessage(Localizer localizer, TimeZone tzView) {
		if (this.hardValidationException == null) {
			return "";
		}
		return hardValidationException.getLocalizedMessage(localizer, tzView);
	}

	public String getPageMessage(Localizer localizer, String defaultMessageBundleName, String defaultMesssageId) {

		if (this.bundleKey != null) {
			return localizer.i18n(RmEjbBundleKey.BUNDLE_NAME, this.bundleKey, this.params);
		}

		if (this.bidTemplateName == null || this.bidTemplateName.isEmpty()) {
			return localizer.i18n(defaultMessageBundleName, defaultMesssageId);
		}

		return localizer.i18n(RmEjbBundleKey.BUNDLE_NAME, RmEjbBundleKey.UNABLE_TO_SAVE_MULTI_BID_REQUEST, this.bidTemplateName);
	}

}
