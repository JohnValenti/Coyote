/**
 * An abstract class to encapsulate some of the common functionality between SkillsEmployeeAssignmentPM and
 * WorkRulesEmployeeAssignmentPM (and their campaign variants).
 */
public abstract class WfmEmployeeAssignmentPM extends WorkpaneFormPM {

	protected static final String START_DATE_FN = "startDate";
	protected static final String END_DATE_FN = "endDate";

	protected LocalDate viewStartDate;
	protected LocalDate viewEndDate;

	protected final ResourceBundle m_bundle;

	protected TimePeriodSelectorPC pc_timePeriod;

	//key = employee ID
	protected Map<ID, Employee> employeeMap = new HashMap<>();

	protected WfmEmployeeAssignmentPM(RequestContext context, int modeType) {
		super(context, modeType);
		m_bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);
	}

	public String getRequiredHTML() {
		try {
			StringBuilder sb = new StringBuilder();
			sb.append(super.getRequiredHTML());
			if (viewStartDate != null && viewEndDate != null) {
				sb.append(HtmlUtil.makeHiddenInput(START_DATE_FN,
						m_localizer.formatDate(viewStartDate, m_localizer.getDateInputFormat())));
				sb.append(HtmlUtil.makeHiddenInput(END_DATE_FN,
						m_localizer.formatDate(viewEndDate, m_localizer.getDateInputFormat())));
			}
			return sb.toString();
		} catch (Exception ex) {
			// Add additional logging for troubleshooting infrequent exception in automation here.
			log.debug(MessageFormat.format("m_localizer: {0}", m_localizer));
			log.debug(MessageFormat.format("viewStartDate: {0}", viewStartDate));
			log.debug(MessageFormat.format("viewEndDate: {0}", viewEndDate));
			throw ex;
		}
	}

	protected boolean userHasPermissionToEditAllSelectedEmployees(Collection<Employee> selectedEmployees) {
		return checkPrivilegesForEmployees(selectedEmployees, PrivilegeKeys.CORE_MODIFYPEOPLE);
	}

	protected boolean userHasPermissionToViewAllSelectedEmployees(Collection<Employee> selectedEmployees) {
		return checkPrivilegesForEmployees(selectedEmployees, PrivilegeKeys.CORE_VIEWPEOPLE);
	}

	private boolean checkPrivilegesForEmployees(Collection<Employee> selectedEmployees, String privilegeToCheck) {
		if (selectedEmployees == null || selectedEmployees.isEmpty()) {
			return false;
		}

		Collection<ID> employeeOrgIds = selectedEmployees.stream()
				.map(Employee::getOrganizationID)
				.collect(Collectors.toList());

		for (ID orgId : employeeOrgIds) {
			ScopeID scopeID = new ScopeID(orgId, Scope.ORG_SCOPE);
			if (!m_context.getUser().isAuthorized(privilegeToCheck, scopeID)) {
				return false;
			}
		}

		return true;
	}

	protected boolean isMultiEdit() {
		return getSelectedIDSize() > 1;
	}

	@Override
	protected void initToolbar() {
		m_toolbar.addValidateSubmitTextButton(PageAction.SAVE,
				i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_SAVE),
				!m_initFailure && userHasPermissionToEditAllSelectedEmployees(employeeMap.values()));
		m_toolbar.addResetTextButton(PageAction.RESET,
				i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_RESET),
				userHasPermissionToEditAllSelectedEmployees(employeeMap.values()));
	}

	protected abstract String getPageTitle();

	protected void initContentTitle() {
		super.initContentTitle();

		m_contentTitlePC.setPageTitle(getPageTitle());
		if (getSelectedIDSize() > 0) {
			try {
				if (isMultiEdit()) {
					m_contentTitlePC.setItemName(m_localizer.i18n(m_bundle, FsWebBundleKey.N_PEOPLE, getSelectedIDSize()));
				} else {
					Employee emp = EmployeeModelHandler.getWorkResourceManager(getRequestContext()).getEmployeeByID(getSelectedIDObject(),
							null, Employee.DETAIL_LEVEL_EMPLOYEE_BASIC |
									Employee.DETAIL_LEVEL_EMPLOYEE_TIMEBASEDPROPERTY_HISTORY);
					if (userHasPermissionToEditAllSelectedEmployees(Collections.singleton(emp))) {
						m_contentTitlePC.setItemName(m_localizer.formatName(emp.getPerson()));
					}
				}
			} catch (Exception ex) {
				handleUnableToLoadDataError(log, ex);
			}
		}
		TimePeriodSelectorController controller = getTimePeriodSelectorController();
		if (controller != null) {
			pc_timePeriod = new TimePeriodSelectorPC(m_context, controller);

			EmployeeAssignmentPMUtil.storeSelectedDateRangeInSession(pc_timePeriod, m_context);

			this.addChildComponent(pc_timePeriod);
			m_contentTitlePC.setActiveComponent(pc_timePeriod);
		}
	}

	protected TimePeriodSelectorController getTimePeriodSelectorController() {
		return new TimePeriodSelectorWeekController(getWeekStartForSelectedEmployee(), m_context);
	}

	protected DayOfWeek getWeekStartForSelectedEmployee() {
		Integer weekStart = null;
		if (getSelectedIDSize() > 0) {
			try {
				//Determine the week start day based on the selected employees.  If all the selected
				// employees' orgs have the same week start day, it will use that day.  Otherwise, it will
				// use the week start day of the root organization.
				PeopleFacade pf = BbmManagerFactory.getPeopleFacade();
				HashMap<ID, Integer> empWeekStarts =
						pf.getEmployeeWeekStarts(getSelectedIDsAsCollection(), new Date());
				for (Integer empWeekStart : empWeekStarts.values()) {
					if (weekStart == null) {
						weekStart = empWeekStart;
					} else if (!weekStart.equals(empWeekStart)) {
						WorkResourceManager wrm = BbmManagerFactory.getWorkResourceManager(m_context.isInWhatIfMode());
						weekStart = (int) wrm.getOrganizationByID(Organization.ROOT_ORG_ID_OBJ).getWeekStartDate();
						break;
					}
				}
			} catch (Exception e) {
				log.warn(m_localizer.i18n(FsWebLogBundleKey.BUNDLE_NAME, FsWebLogBundleKey.COULD_NOT_GET_WEEK_START_FOR_EMP,
						getSelectedIDObject()), e);
			}
		}
		if (weekStart == null) {
			//Default of 1 (sunday), but this shouldn't matter since we only set it here if there are no employees selected
			weekStart = 1;
		}
		// Org week start is a 1=sunday based number.
		return DayOfWeek.from0SundayCount(weekStart - 1);
	}
}
