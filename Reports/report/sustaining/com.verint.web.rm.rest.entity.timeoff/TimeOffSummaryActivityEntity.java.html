package com.verint.web.rm.rest.entity.timeoff;

import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeTimeOffAccrued;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TimeOffActivitySummary;
import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rest.util.FormatUtils;

@XmlRootElement(name = "timeOffSummaryActivity")
public class TimeOffSummaryActivityEntity extends AbstractWFOEntity {

	private String activityCategoryID;
	private String activityColor;
	private String activityName;
	private double balanceHours;
	private float carryOverHours;
	private String lastUpdated;
	private float pendingHours;
	private float remainingHours;
	private float scheduledHours;
	private float totalHours;
	private float usedHours;

	public TimeOffSummaryActivityEntity(TimeOffActivitySummary summary, boolean showingBalance) {
		ActivityCategory category = summary.getActivityCategory();
		if (category != null) {
			activityCategoryID = category.getID().toString();
		}

		// null activityID means it is an Activity Category
		ID activityID = summary.getActivityID();
		if (activityID != null) {
			this.setId(activityID.toString());
		}
		activityColor = summary.getActivityColor();
		activityName = summary.getActivityName();
		carryOverHours = summary.getCarryOver();
		pendingHours = summary.getPending();
		remainingHours = summary.getRemaining();
		scheduledHours = summary.getScheduled();
		totalHours = summary.getTotal();
		usedHours = summary.getUsed();

		EmployeeTimeOffAccrued etoa = summary.getEmployeeTimeOffAccrued();
		if (showingBalance && etoa != null) {
			balanceHours = etoa.getAccruedHours();
			lastUpdated = FormatUtils.dateToISO8601SecPrecisionString(etoa.getAccruedAtDate());
		}
	}

	public String getActivityCategoryID() {
		return activityCategoryID;
	}

	public void setActivityCategoryID(String activityCategoryID) {
		this.activityCategoryID = activityCategoryID;
	}

	public String getActivityColor() {
		return activityColor;
	}

	public void setActivityColor(String activityColor) {
		this.activityColor = activityColor;
	}

	public String getActivityName() {
		return activityName;
	}

	public void setActivityName(String activityName) {
		this.activityName = activityName;
	}

	public double getBalanceHours() {
		return balanceHours;
	}

	public void setBalanceHours(double balanceHours) {
		this.balanceHours = balanceHours;
	}

	public float getCarryOverHours() {
		return carryOverHours;
	}

	public void setCarryOverHours(float carryOverHours) {
		this.carryOverHours = carryOverHours;
	}

	public String getLastUpdated() {
		return lastUpdated;
	}

	public void setLastUpdated(String lastUpdated) {
		this.lastUpdated = lastUpdated;
	}

	public float getPendingHours() {
		return pendingHours;
	}

	public void setPendingHours(float pendingHours) {
		this.pendingHours = pendingHours;
	}

	public float getRemainingHours() {
		return remainingHours;
	}

	public void setRemainingHours(float remainingHours) {
		this.remainingHours = remainingHours;
	}

	public float getScheduledHours() {
		return scheduledHours;
	}

	public void setScheduledHours(float scheduledHours) {
		this.scheduledHours = scheduledHours;
	}

	public float getTotalHours() {
		return totalHours;
	}

	public void setTotalHours(float totalHours) {
		this.totalHours = totalHours;
	}

	public float getUsedHours() {
		return usedHours;
	}

	public void setUsedHours(float usedHours) {
		this.usedHours = usedHours;
	}
}
