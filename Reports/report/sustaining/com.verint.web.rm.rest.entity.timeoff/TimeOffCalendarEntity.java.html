package com.verint.web.rm.rest.entity.timeoff;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOCalendarDayData;
import com.verint.web.rest.entity.AbstractWFOEntity;
import com.verint.web.rest.util.FormatUtils;

public abstract class TimeOffCalendarEntity extends AbstractWFOEntity {

	private final int activityID;
	private final String endDate;
	private final boolean isTOPoolView;
	private final String startDate;
	private final String[] colors;

	public TimeOffCalendarEntity(Date startDate, Date endDate, boolean isTOPoolView, ID activityID) {
		colors = new String[TOCalendarDayData.NUMBER_DAY_TYPES];
		this.activityID = activityID.toInt();
		this.endDate = FormatUtils.dateToISO8601SecPrecisionString(endDate);
		this.startDate = FormatUtils.dateToISO8601SecPrecisionString(startDate);
		this.isTOPoolView = isTOPoolView;
	}

	public int getActivityID() {
		return activityID;
	}

	public String getEndDate() {
		return endDate;
	}

	public boolean isTOPoolView() {
		return isTOPoolView;
	}

	public String getStartDate() {
		return startDate;
	}

	public String[] getColors() {
		return colors;
	}
}
