package com.verint.web.fs.workrules.assignmentrules.ui;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;
import com.verint.web.fs.workrules.assignmentrules.question.Question;
import com.verint.web.fs.workrules.assignmentrules.question.TeamConstraintQuestion.TeamConstraintResponseType;

class TeamConstraintTypeQuestionI18N extends AbstractQuestionI18N {
	public String getResponseKey(Response response) {
		ResponseType responseType = response.getResponseType();
		if (!(responseType instanceof TeamConstraintResponseType)) {
			throw new IllegalArgumentException("Expecting an instance of TeamRuleTypeResponseType");
		}
		TeamConstraintResponseType rt = (TeamConstraintResponseType) responseType;
		switch (rt) {
			case TeamStart:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_SAME_START;
			case TeamStartSameDays:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_SAME_START_DAYS;
			case TeamStartAndEnd:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_SAME_START_END;
			case TeamStartAndEndSameDays:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_SAME_START_END_DAYS;
			case TeamStartEndBreakSameDays:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_SAME_START_END_BREAK_DAYS;
			case TeamOverlapEachDay:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_OVERLAP_DAY;
			case TeamOverlapEachWorkingDay:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_OVERLAP_WORK_DAY;
			case TeamOverlapEachWeek:
				return FsWebBundleKey.ASSIGNMENT_RULE_TEAM_OVERLAP_MINUTE_WEEK;
			default:
				throw new IllegalArgumentException("Unknown response type: " + rt.name());
		}
	}

	@Override
	public String getQuestionText(Localizer localizer, Question question) {
		return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.ASSIGNMENT_RULE_TEXT_TEAM_CONSTRAINT);
	}
}
