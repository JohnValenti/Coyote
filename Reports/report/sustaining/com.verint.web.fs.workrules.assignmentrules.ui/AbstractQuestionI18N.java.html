package com.verint.web.fs.workrules.assignmentrules.ui;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.question.Question;

/**
 * This is just a convenience class to help localize questions.
 * For the responses, subclasses must either override getResponseText, or they must
 * override getResponseKey.
 */
abstract class AbstractQuestionI18N implements QuestionI18N {
	/**
	 * Returns the bundle name containing the key returned by <code>getBundleKey</code>.
	 * The default is FsWebBundleKey.BUNDLE_NAME. This is used by the default implementation
	 * of the <code>getResponseText</code> method.
	 *
	 * @return
	 */
	protected String getBundleName() {
		return FsWebBundleKey.BUNDLE_NAME;
	}

	/**
	 * Returns the bundle key for the given response. The default implementation is to return null.
	 * This is used by the default implementation of the <code>getResponseText</code> method. Subclasses
	 * must either override this method or <code>getResponseText</code>.
	 *
	 * @return
	 */
	protected String getResponseKey(Response response) {
		return null;
	}

	@Override
	public String getResponseText(Localizer localizer, Response response, Object... params) {
		String bundleName = getBundleName();
		String bundleKey = getResponseKey(response);
		if (!StringUtil.isEmpty(bundleName) && !StringUtil.isEmpty(bundleKey)) {
			return localizer.i18n(bundleName, bundleKey, params);
		} else {
			throw new IllegalStateException("Either missing bundle name or bundle key.");
		}
	}

	@Override
	public abstract String getQuestionText(Localizer localizer, Question question);

}
