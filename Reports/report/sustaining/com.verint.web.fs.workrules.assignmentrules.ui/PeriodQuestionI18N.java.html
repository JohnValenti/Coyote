package com.verint.web.fs.workrules.assignmentrules.ui;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.ResponseType;
import com.verint.web.fs.workrules.assignmentrules.question.PeriodQuestion.PeriodResponseType;
import com.verint.web.fs.workrules.assignmentrules.question.Question;

class PeriodQuestionI18N extends AbstractQuestionI18N {
	@Override
	public String getResponseKey(Response response) {
		if (response == null) {
			return null;
		}
		ResponseType responseType = response.getResponseType();
		if (!(responseType instanceof PeriodResponseType)) {
			throw new IllegalArgumentException("Expected responseType to be of type "
					+ PeriodResponseType.class.getCanonicalName() + " instead of "
					+ responseType.getClass().getCanonicalName());
		}

		PeriodResponseType prt = (PeriodResponseType) responseType;
		switch (prt) {
			case Day:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_EACH_X_DAYS;
			case Month:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_EACH_X_MONTHS;
			case Month28Day:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_EACH_28_DAY_MONTH;
			case Month29Day:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_EACH_29_DAY_MONTH;
			case Month30Day:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_EACH_30_DAY_MONTH;
			case Month31Day:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_EACH_31_DAY_MONTH;
			case Week:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_EACH_WEEK;
			case WeekX:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_EACH_X_WEEKS;
			case RollingWeek:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_ROLLING_WEEK;
			case Year:
				return FsWebBundleKey.ASSIGNMENT_RULE_FREQUENCY_EACH_X_YEARS;
			default:
				throw new IllegalArgumentException("Unknown value of " + prt);
		}
	}

	@Override
	public String getQuestionText(Localizer localizer, Question question) {
		return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.ASSIGNMENT_RULE_TEXT_PERIOD);
	}

}
