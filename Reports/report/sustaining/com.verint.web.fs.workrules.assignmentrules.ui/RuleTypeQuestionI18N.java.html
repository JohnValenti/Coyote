package com.verint.web.fs.workrules.assignmentrules.ui;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.question.Question;
import com.verint.web.fs.workrules.assignmentrules.question.RuleTypeQuestion.RuleTypeResponseType;

class RuleTypeQuestionI18N extends AbstractQuestionI18N {
	@Override
	public String getResponseKey(Response response) {
		if (response == null) {
			return null;
		}
		if (!(response.getResponseType() instanceof RuleTypeResponseType)) {
			throw new IllegalArgumentException("ResponseType must be of type RuleTypeReponseType");
		}
		RuleTypeResponseType rtrt = (RuleTypeResponseType) response.getResponseType();
		switch (rtrt) {
			case MustWork:
				return FsWebBundleKey.ASSIGNMENT_RULE_TYPE_WORK;
			case Off:
				return FsWebBundleKey.ASSIGNMENT_RULE_TYPE_OFF;
			case Team:
				return FsWebBundleKey.ASSIGNMENT_RULE_TYPE_TEAM;
			case Start:
				return FsWebBundleKey.ASSIGNMENT_RULE_TYPE_START;
			default:
				throw new IllegalArgumentException("Unknown response type: " + rtrt);
		}
	}

	@Override
	public String getQuestionText(Localizer localizer, Question question) {
		return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.ASSIGNMENT_RULE_TEXT_RULE_TYPE);
	}
}
