package com.bluepumpkin.web.bbm.campaign.sp;

import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;

import java.util.*;

/**
 * Title:        SchedulingPeriodComparator
 * Description:  
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, DSu
 * @version      1.0

 * Created on Oct 14, 2003, 1:37:06 PM

public class SchedulingPeriodComparator implements Comparator<SchedulingPeriod> {
	public static short SORT_ASC = 1;
	public static short SORT_DESC = -1;

	private short m_sortBy = SORT_ASC;

	public SchedulingPeriodComparator(short sortBy) {
		m_sortBy = sortBy;
	}

	public int compare(SchedulingPeriod sp1, SchedulingPeriod sp2) {
		int result = 0;

		// Compare Start Date then End Date if necessary
		result = sp1.getStartTime().compareTo(sp2.getStartTime());
		if (result==0) {
			result = sp1.getEndTime().compareTo(sp2.getEndTime());
		}

		return m_sortBy * result;
	}
}
