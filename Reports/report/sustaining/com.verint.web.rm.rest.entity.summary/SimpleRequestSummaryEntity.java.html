package com.verint.web.rm.rest.entity.summary;

import com.verint.web.rest.entity.AbstractWFOEntity;

public abstract class SimpleRequestSummaryEntity extends AbstractWFOEntity {

	private int approved = 0;
	private int limit = 0;
	private int remaining = 0;

	private int period;
	private String startDate;
	private String endDate;

	public SimpleRequestSummaryEntity() { // NOSONAR
	}

	public int getApproved() {
		return approved;
	}

	public void setApproved(int approved) {
		this.approved = approved;
	}

	public int getLimit() {
		return limit;
	}

	public void setLimit(int limit) {
		this.limit = limit;
	}

	public int getRemaining() {
		return remaining;
	}

	public void setRemaining(int remaining) {
		this.remaining = remaining;
	}

	public int getPeriod() {
		return period;
	}

	public void setPeriod(int period) {
		this.period = period;
	}

	public String getStartDate() {
		return startDate;
	}

	public void setStartDate(String startDate) {
		this.startDate = startDate;
	}

	public String getEndDate() {
		return endDate;
	}

	public void setEndDate(String endDate) {
		this.endDate = endDate;
	}
}
