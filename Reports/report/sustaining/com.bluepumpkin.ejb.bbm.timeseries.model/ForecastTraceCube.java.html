package com.bluepumpkin.ejb.bbm.timeseries.model;

import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.base.BbmTimeSeriesException;

/**
 * Title:        Blue Pumpkin Software Basic Business Model
 * Description:  ForecastTraceCube object, mainly used by UI to load bunch of Trace Data for a given period
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 *
 * @author Sheng Song
 * @version 2.0
 */

public class ForecastTraceCube extends TraceCube {
	static final long serialVersionUID = 1179129116049343922L;

	//Trace.VAR in ForecastTraceCube is used in the aggregation formula for Required PCA. It will simply be a copy of CV.
	public static final short[] TYPES = {Trace.CV, Trace.AHT, Trace.CRATE, Trace.RPCAHT, Trace.RPCRATE, Trace.VAR, Trace.CV_VH};
	//array of all trace types that are computed for this trace cube
	public static final short[] COMPUTED_TYPES = {VAR, CV_VH};

	// Default constructor
	public ForecastTraceCube() {
		super();
	}

	public ForecastTraceCube(short[] types) {
		super(types);
	}

	public String getForecastInstanceID() {
		return forecastInstanceID;
	}

	public void setForecastInstanceID(String forecastInstanceID) {
		this.forecastInstanceID = forecastInstanceID;
	}

	private String forecastInstanceID =null;
	/**
	 * TraceCube is a container for types of Trace, given period
	 * This constructor creates all TYPES supported ForecastTraceCube
	 *
	 * @param ID,   queueID
	 * @param Date, startDate,
	 * @param Date, endDate,
	 */
	public ForecastTraceCube(ID queueID, Date startDate, Date endDate) throws BbmTimeSeriesException {
		this(queueID, startDate, endDate, TYPES, TimeZoneUtil.GMT_TIMEZONE);
	}

	/**
	 * TraceCube is a container for types of Trace, given period
	 *
	 * @param ID,      queueID
	 * @param Date,    startDate,
	 * @param Date,    endDate,
	 * @param short[], the Trace Types to be included
	 */
	public ForecastTraceCube(ID queueID, Date startDate, Date endDate, short[] types) throws BbmTimeSeriesException {
		this(queueID, startDate, endDate, types, TimeZoneUtil.GMT_TIMEZONE);
	}

	/**
	 * TraceCube is a container for types of Trace, given period
	 *
	 * @param ID,      queueID
	 * @param Date,    startDate,
	 * @param Date,    endDate,
	 * @param short[], the Trace Types to be included
	 * @param TimeZone
	 */
	public ForecastTraceCube(ID queueID, Date startDate, Date endDate, short[] types, TimeZone timeZone) throws BbmTimeSeriesException {
		super(queueID, startDate, endDate, types);
		checkTraceTypeSupport(types);
	}

	// Base TraceCube supports all types of Trace
	@Override
	protected short[] getSupportedTraceType() {
		return TYPES;
	}

	@Override
	protected String getCubeName() {
		return "FORECASTTIMESERIES";
	}

	// make a clone from given parameters
	@Override
	public TraceCube newInstance(ID queueID, Date startDate, Date endDate, short[] types) throws BbmTimeSeriesException {
		return new ForecastTraceCube(queueID, startDate, endDate, types);
	}

	@Override
	public void calcComputedTraceValues(short traceTypes[], float[] traceValues, ID mediaID) {
		for (int j = 0; j < traceTypes.length; j++) {
			short type = traceTypes[j];
			if (isTraceTypeComputed(type)) {
				if (type == Trace.VAR || type == Trace.CV_VH) {
					float val = getTraceValueForTraceTypeFromGivenArrayOfTraceValues(Trace.CV, traceTypes, traceValues);
					traceValues[j] = val;
				}
			}
		}
	}

	@Override
	public void calcComputedTraceValuesForCubes(TraceCube[] queCubes, ID mediaID) {
		short traceTypes[] = getTraceTypes();
		for (int j = 0; j < traceTypes.length; j++) {
			short type = traceTypes[j];
			if (isTraceTypeComputed(type)) {
				if (type == Trace.VAR || type == Trace.CV_VH) {
					if (queCubes[1] != null && queCubes[1].getTraceValuesD() != null) {   //Proceed only if Predicted Trace cube is available here
						double[][] traceValuesForecasted = queCubes[1].getTraceValuesD();
						double[] cArray = traceValuesForecasted[Trace.CV];
						double[] value = new double[cArray.length];
						System.arraycopy(cArray, 0, value, 0, cArray.length);
						setTraceValue(type, value);
					}
				} else {
					if (getQueueID().toInt() == -1) {
						continue;
					}
				}
			}
		}
	}

	@Override
	public void calcComputedTraceValuesForTraceChunk(TraceChunk tc, ID mediaID) {
		tc.setTraceValue(Trace.CV_VH, tc.getTraceValue(Trace.CV));
		tc.setTraceValue(Trace.VAR, tc.getTraceValue(Trace.CV));
	}

	@Override
	public short[] getComputedTraceTypes() {
		return COMPUTED_TYPES;
	}

	@Override
	public void calcComputedTraceValues(ID mediaID) {
		short traceTypes[] = getTraceTypes();
		for (int j = 0; j < traceTypes.length; j++) {
			short type = traceTypes[j];
			if (isTraceTypeComputed(type)) {
				if (type == Trace.CV_VH || type == Trace.VAR) {
					double[] cArray = getTraceValueD(Trace.CV);
					double[] value = new double[cArray.length];
					System.arraycopy(cArray, 0, value, 0, cArray.length);
					setTraceValue(type, value);
				}
			}
		}
	}

}
