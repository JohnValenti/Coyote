package com.bluepumpkin.web.rm.requests;

import com.bluepumpkin.web.rm.requests.RequestsRH;
import com.witness.web.uif.system.RequestContext;
import com.bluepumpkin.ejb.rm.requests.common.model.Request;

import java.util.*;

/**
 * Title:        MyRequestsRH
 * Description:  Decides which Screen to direct user to
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, DSu
 * @version      1.0
 *
 * Created on Dec 9, 2003, 6:37:34 PM
 */
public class MyRequestsRH extends RequestsRH {
	private static final Map SCREEN_MAP = new HashMap();
	static{
		SCREEN_MAP.put(Request.REQUESTTYPE_ALL, "MYSCHEDULE_REQUESTS");
		SCREEN_MAP.put(Request.REQUESTTYPE_TIMEOFF, "MYSCHEDULE_TIMEOFF_REQUESTS");
		SCREEN_MAP.put(Request.REQUESTTYPE_FLEXTIME, "MYSCHEDULE_FLEXTIME_REQUESTS");
		SCREEN_MAP.put(Request.REQUESTTYPE_SHIFTSWAP, "MYSCHEDULE_SHIFTSWAP_REQUESTS");
		SCREEN_MAP.put(Request.REQUESTTYPE_SHIFTBID, "MYSCHEDULE_SHIFTBIDDING_REQUESTS");
		SCREEN_MAP.put(Request.REQUESTTYPE_CUSTSHIFT, "MYSCHEDULE_CUSTSHIFT_REQUESTS");
	}

	/**
	 * Process the request.
	 */
	public void processRequest(RequestContext context) throws Exception {
		super.processRequest(context);

		String requestType = getRequestType(context);
		String nextScreen = (String)SCREEN_MAP.get(requestType);
		if (nextScreen!=null) {
			setNextScreenName(context, nextScreen);
		}
	}


	private static String getRequestType(RequestContext context) {
		String requestType = MyRequestsPM.getViewType(context);
		if (requestType==null) {
			requestType = RequestUtil.getCachedMyRequestType(context);
		}
		return requestType;
	}
}
