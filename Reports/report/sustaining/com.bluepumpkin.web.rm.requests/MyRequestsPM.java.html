				return new HtmlDivWSpan(getPCAriaLabelID(), i18n(m_bundle, RmWebBundleKey.RM_MYREQUESTS_GRID));
			}
			
			@Override
			protected boolean skipEncloseForKeyboardAccess(BaseNodeData nodeData) {
				return isDisabled() || !nodeData.isSelectable();
			}
			
			@Override
			protected String addAdditionalCellText(HtmlTableCell cell) {
				if (isFirstColumn(cell) && cell.getInnerHtml().toString().contains("<img")) {
					String cellText = cell.getInnerHtml().toString();
					return getNotCenteredData(cellText);
				}
				return super.addAdditionalCellText(cell);
			}
			
			@Override
			protected String getCellData(HtmlTableCell cell) {
				if (isFirstColumn(cell)) {
					String cellInnerText = cell.getInnerHtml().toString();
					if (cellInnerText.contains("<img") && !StringUtil.isEmpty(getInnerTextFromImg(cellInnerText))) {
						if (isAccessibilityComplianceMode()) {
							setFirstColumnWidth(ACCESSIBILITY_FIRST_COLUMN_SIZE);
						}
						return HtmlUtil.makeHiddenReadableText(getInnerTextFromImg(cellInnerText)) + HtmlUtil.NBSP;
					}
				}
				return super.getCellData(cell);
			}
			
			@Override
			protected String getCellDataWithAdditionalText(HtmlTableCell cell, String cellData) {
				String cellDataWithAdditionalText = super.getCellDataWithAdditionalText(cell, cellData);
				String cellInnerText = cell.getInnerHtml().toString();
				if (cellInnerText.contains(CENTER_OPEN_TAG) && cellInnerText.contains(CENTER_CLOSE_TAG)) {
					StringBuilder builder = new StringBuilder();
					builder.append(CENTER_OPEN_TAG).append(cellDataWithAdditionalText).append(CENTER_CLOSE_TAG);
					return builder.toString();
				} else {
					return cellDataWithAdditionalText;
				}
			}
		};
	}
	
	private boolean isFirstColumn(HtmlTableCell cell) {
		String firstColumnID = "c0";
		String idAttr = cell.getAttribute("id");
		if (!StringUtil.isEmpty(idAttr) && idAttr.contains(firstColumnID)) {
			return true;
		}
		return false;
	}
	
	private String getNotCenteredData(String data) {
		String centerOpenTag = "<center>";
		String centerCloseTag = "</center>";
		if (data.contains(centerOpenTag) && data.contains(centerCloseTag)) {
			int centerOpenTagIdx = data.indexOf(centerOpenTag);
			int centerCloseTagIdx = data.indexOf(centerCloseTag);
			StringBuilder builder = new StringBuilder();
			builder.append(data.substring(0, centerOpenTagIdx));
			String notCenteredData = data.substring(centerOpenTagIdx + centerOpenTag.length(), centerCloseTagIdx);
			builder.append(notCenteredData).append(data.substring(centerCloseTagIdx + centerCloseTag.length(), data.length()));
			return builder.toString();
		} else {
			return data;
		}
	}

	private void loadData2() throws Exception {

		if (m_empID != null) {
			try {
				m_orgSetting = RequestsMH.getOrgSettingForEmployee(m_context, m_empID);
				if (isAuthToViewAnyRqts()) {
					String requestIDStr = m_context.getRequest().getParameter("reqID");
					ID requestID = null;
					if (!StringUtil.isEmpty(requestIDStr)) {
						requestID = new ID(requestIDStr);
					}

					String requestTypeStr = m_context.getRequest().getParameter("requestType");

					if (requestID != null) {
						ArrayList reqList = new ArrayList(1);
						RequestAggregate reqAgg = RequestsMH.getRequest(requestID, requestTypeStr);
						if (reqAgg != null) {
							reqList.add(RequestsMH.getRequest(requestID, requestTypeStr));
						}
						m_requestList = reqList;
						if (m_requestList == null || m_requestList.isEmpty()) {
							addPageMessage(Message.INFO_TYPE, RmWebBundleKey.REQUEST_NOT_FOUND_ERROR, RmWebBundleKey.BUNDLE_NAME);
						} else {
							addPageMessage(Message.INFO_TYPE, RmWebBundleKey.REQUEST_FROM_DRILLTHROUGH_VIEW, RmWebBundleKey.BUNDLE_NAME);
						}
					} else {
						String rType = Request.REQUESTTYPE_ALL;
						m_requestList = RequestsMH.getRequestsForEmp(m_empID, rType, m_isIncludeExpired);
					}
					
				}
			} catch (BbmFinderException ex) {
				addPageMessage(Message.ERROR_TYPE, ex.getLocalizedMessage());
			} catch (RmHardValidationException ex) {
				Message msg = handleUnableToLoadDataError(log, ex);
				msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
			}
		}
	}

	/**
	 * Initialize Selectable items
	 */
	@Override
	protected void initSelectableItems() {
		if (isAccessibilityComplianceMode()) {
			m_list.setUseWrapper(true);
			m_list.addTableArrowNavigation();
			m_list.setReadableRowInfoEnabled(true);
			m_list.disableRepeatHeader();
		}		
		// Handle case where there are no requests found
		if (m_requestList == null || m_requestList.isEmpty()) {
			return;
		}

		ArrayList listModel = new ArrayList();
		for (Iterator it = m_requestList.iterator(); it.hasNext();) {
			// Create the row data
			RequestAggregate request = (RequestAggregate) it.next();
			//check for shift swap enable setting
			if (!isShiftSwapEnabled() && request.getRequestType().equals(Request.REQUESTTYPE_SHIFTSWAP)) {
				continue;
			}
			MultiColumnNodeData row_data = createRowData(request);
			listModel.add(row_data);
		}

		m_list.setListModel(listModel);
		if (isDateSort) {
			m_list.setIsAutoSortEnabled(false);
		} else {
			m_list.setIsAutoSortEnabled(true);
		}
	}

	/**
	 * Create Row Data From Request
	 */
	protected MultiColumnNodeData createRowData(RequestAggregate request) {
		String rowID = request.getID().toString();
		DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData(rowID);

		String status = createStatusIcon(request);
		row_data.add(status);

		String rType = getLocalizedRequestType(request);
		row_data.add(rType);

		DateWrapper submittedDT = createDataWrapper(request.getSubmittedOn());
		row_data.add(submittedDT);

		DateWrapper lastModifiedDT = createDataWrapper(request.getLastModifiedAt());
		row_data.add(lastModifiedDT);

		addCommonListData(request, row_data);
		return row_data;
	}

	/**
	 * Add the following common List Data
	 *
	 * - Comments
	 * - Alerts
	 * - Actions
	 *
	 * @param request - The Request Object with the data
	 * @param row_data - The DefaultMultiColumnNodeData to add data to
	 */
	protected void addCommonListData(RequestAggregate request, DefaultMultiColumnNodeData row_data) {
		String rowID = row_data.getID();

		String hasExpired = m_localizer.formatYesNo(request.hasExpired());
		row_data.add(hasExpired);

		String comments = renderComments(request);
		row_data.add(comments);

		String alerts = "";
		if (isAccessibilityComplianceMode()) {
			alerts = RequestUtil.createUniqueAlertIcons(request, m_localizer, true, true);
		} else {
			alerts=RequestUtil.createUniqueAlertIcons(request, m_localizer);
		}
		
		row_data.add(alerts);
		String itemActions = createItemActions(request, rowID);
		row_data.add(itemActions);
	}

	/**
	 * Create HTML for Drop down for View selection
	 */
	private String makeViewDropDown() {
		String onChangeJS = "pageMediator.handleURLLink(this.value, true);";

		ArrayList options = new ArrayList();
		options.add(createViewOption(RmKeys.RH_MY_REQUEST_ALL_LIST, RmWebBundleKey.REQUEST_TYPE_ALL));

		if (isAuthToViewTORqts()) {
			options.add(createViewOption(RmKeys.RH_MY_REQUEST_TIMEOFF_LIST, RmWebBundleKey.REQUEST_TYPE_TIME_OFF));
		}
		if (LicenseUtil.isAdvancedRMLicense() && isAuthToViewFTRqts()) {
			options.add(createViewOption(RmKeys.RH_MY_REQUEST_FLEXTIME_LIST, RmWebBundleKey.REQUEST_TYPE_FLEX_TIME));
		}
		if (isAuthToViewSSRqts()) {
			options.add(createViewOption(RmKeys.RH_MY_REQUEST_SHIFTSWAP_LIST, RmWebBundleKey.REQUEST_TYPE_SHIFT_SWAP));
		}
		if (isAuthToViewSBRqts()) {
			options.add(createViewOption(RmKeys.RH_MY_REQUEST_SHIFTBIDDING_LIST, RmWebBundleKey.REQUEST_TYPE_SHIFT_BIDDING));
		}
		if (isAuthToViewCSRqts()) {
			options.add(createViewOption(RmKeys.RH_MY_REQUEST_CUSTSHIFT_LIST, RmWebBundleKey.REQUEST_TYPE_CUSTOM_SHIFT));
		}

		String selectedOption = getRequestHandler(m_requestType);
		if (isAccessibilityComplianceMode()) {
			return HtmlUtil.makeSelectInput(VIEW_TYPE, "", new String[]{selectedOption}, onChangeJS, options,
					null, false, false, true, false, null, getHtmlSelectWithDescription(VIEW_TYPE));
		}
		return HtmlUtil.makeSelectInput(VIEW_TYPE, selectedOption, onChangeJS, options, false);
	}

		}

		return super.isValidAction(action, request)
				|| isEscalateActionValid(action, request)
				|| isShiftSwapActionsValid(action, request)
				|| isNormalTimeOffActionsValid(action, request)
				|| isTimeOffBidActionsValid(action, request);
		
	}
	
	
	/**
	 * Is the Submit to Manager action valid?
	 * @return true if the submit to manager action is valid.
	 */
	@Override
	protected boolean canSubmitToManager(String action, RequestAggregate request) {
		String rType = request.getRequestType();
		if (Request.REQUESTTYPE_SHIFTSWAP.equals(rType)) {
			try {
				ShiftSwapRequest ssRequest = ShiftSwapRequestsMH.getRequestByID(request.getID());
				if (ssRequest == null) {
					this.addPageMessage(Message.ERROR_TYPE, i18n(m_bundle, RmWebBundleKey.REQUEST_NOT_FOUND_ERROR));
					return false;
				}
				return SwapUtil.canUserSubmitShiftSwapToManager(m_empID, ssRequest);
			} catch (Exception e) {
				return false;
			}
		}
		return false;
	}

	/**
	 * Return true if User is authorized to escalate specified denied Request
	 */
	protected boolean isAuthToEscalateRqt(RequestAggregate request) {
		String rType = request.getRequestType();

		return isValidEmployee()

				&& ((Request.REQUESTTYPE_TIMEOFF.equals(rType) && m_user.isAuthorized(PrivilegeKeys.TOM_ESCALATEDENIEDREQUESTS_ID))
						|| (Request.REQUESTTYPE_SHIFTSWAP.equals(rType) && m_user.isAuthorized(PrivilegeKeys.SS_ESCALATEDENIEDREQUESTS_ID))
						|| (Request.REQUESTTYPE_SHIFTBID.equals(rType) && m_user.isAuthorized(PrivilegeKeys.SB_ESCALATEDENIEDBIDS_ID)) || (Request.REQUESTTYPE_CUSTSHIFT
						.equals(rType) && m_user.isAuthorized(PrivilegeKeys.CS_ESCALATEDENIEDREQUESTS_ID)));
	}

	//Return true if we are checking the withdraw action and the request is an approved shift bid  and the auction is closed.
	//False for all other cases. A return value of false does not mean the request is valid
	private boolean isShiftBidWithdrawAndIsInvalid(String action, RequestAggregate request) {
		String status = request.getRequestStatus();

		if (request.isShiftBidRequest() && status.equals(RequestAuditTrail.STATUS_APPROVED) && PageAction.WITHDRAW.equals(action)) {
			if (m_auctionMap != null && m_auctionMap.isEmpty()) {
				//get the auctionlist for approved requests to get auction
				try {
					m_auctionMap = RequestUtil.getAuctionmapForApprovedSBRequests(m_requestList);
				} catch (Exception ex) {
					handleUnableToLoadDataError(LOG, ex);
				}
			}
			return !m_auctionMap.get(((ShiftBidRequest) request).getShiftBidAuctionID()).booleanValue();
		}
		return false;

	}
	private boolean isNormalTimeOffActionsValid(String action, RequestAggregate request){
		String status = request.getRequestStatus();
		boolean result=false;
		if(request.isTimeOffRequest() && !request.isFlexTimeRequest()){
			TORequest toRequest=(TORequest) request;
				boolean isWaitlistActionValid=RmPageAction.ADD_TO_WAITLIST_ACTION.equals(action) && toRequest.isEligibleForWaitlist() && (RequestAuditTrail.STATUS_PENDING.equals(status) || RequestAuditTrail.STATUS_DENIED.equals(status));
				boolean isCancelActionValid=PageAction.CANCEL_WITHDRAW_ACTION.equals(action) && toRequest.isEligibleForCancelWithdrawAction();
				boolean isWithdrawActionValid=PageAction.WITHDRAW.equals(action) && toRequest.isEligibleForWithdraw();
				result=!request.hasExpired()&&(isWaitlistActionValid||isCancelActionValid||isWithdrawActionValid);
		}
		return result;
	}
	private boolean isEscalateActionValid(String action, RequestAggregate request){
		return !request.hasExpired()&&(RequestAuditTrail.STATUS_DENIED.equals(request.getRequestStatus()) && PageAction.ESCALATE.equals(action));
	}
	private boolean isShiftSwapActionsValid(String action, RequestAggregate request){
		boolean result=false;
		if(request.isShiftSwapRequest()){
			ShiftSwapRequest swRequest=(ShiftSwapRequest) request;
			boolean isRejecWithDrawActionValid=PageAction.REJECT_WITHDRAW_ACTION.equals(action) && swRequest.isEligibleForRejectWithdrawAction() ;
			boolean isAceptWithDrawActionValid=PageAction.ACCEPT_WITHDRAW_ACTION.equals(action) && swRequest.isEligibleForAcceptWithdrawAction();
			boolean isCancelWithDrawActionValid=PageAction.CANCEL_WITHDRAW_ACTION.equals(action) && swRequest.isEligibleForCancelWithdrawAction();
			boolean isWithDrawNegotiationActionValid=PageAction.WITHDRAW_NEGOTIATION.equals(action) && swRequest.isEligibleForWithdrawNegotiation();
			boolean isSubmitToMgrActionValid=PageAction.SUBMIT_TO_MGR.equals(action) && swRequest.isEligibleForConfirmWithdrawAction(m_context.getUser().getEmployeeID());
			result=!request.hasExpired()&&(isRejecWithDrawActionValid||isAceptWithDrawActionValid||isCancelWithDrawActionValid||isWithDrawNegotiationActionValid||isSubmitToMgrActionValid);
		}
		return result;
	}
	private boolean isTimeOffBidActionsValid(String action, RequestAggregate request){	
			return !request.hasExpired()&& !request.isInvalid() &&isBidClosedForRequest(request) && PageAction.COPY.equals(action);
	}
	/**
	 * Get the bid info of the TOB request from the time off bid map 
	 */
	private boolean isBidClosedForRequest(RequestAggregate request) {
		return request.isTimeOffRequest()&& getTimeOffBidMap().get(request.getID()) != null && getTimeOffBidMap().get(request.getID()).equals(Boolean.TRUE);
	}

	/**
	 * The map contains list of time off bid requests and the related bid info which is closed or not 
	 */
	private Map<ID, Boolean> getTimeOffBidMap() {
		if (timeOffBidMap != null && timeOffBidMap.isEmpty()) {
			try {
				//Assign time Off bid info which is closed or not to the request if have
				timeOffBidMap = RequestsMH.assignClosedTimeOffBidToRequests(m_empID, getRequestList());
			} catch (Exception ex) {
				handleUnableToLoadDataError(LOG, ex);
			}
		}
		return timeOffBidMap;
	}
