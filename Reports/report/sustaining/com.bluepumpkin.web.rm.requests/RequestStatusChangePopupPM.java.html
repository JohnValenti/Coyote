public class RequestStatusChangePopupPM extends RequestFormPopupPM
		implements ContainersInRowsPM {
	private static final String REQUEST_ACTION_FN = "requestAction";
	private static final String REQUEST_WL_EXPIRYDATE_FN = "wlExpiryDate";
	private final transient Collection containerList;
	private FormTwoColumnPC mainContainerPC;
	private MultiColListPC choiceListPC;

	private String requestAction;
	private Set<ID> toChoiceID;
	private DatePickerPC wlExpiryDatePickerPC;

	private boolean canSubmitAction = true; //can the user submit the action?
	private boolean isCloseAfterLoadCustomize = false;
	/**
	 * Constructor
	 */
	public RequestStatusChangePopupPM(RequestContext context) {
		super(context, RmKeys.RH_REQUEST_CHANGE_STATUS_FORM);

		containerList = new ContainerList(this, 2);

		initDatePickerPC();
	}

	/**
	 * Init waitlist date picker
	 */
	private void initDatePickerPC() {
		wlExpiryDatePickerPC = new DatePickerPC(m_context, REQUEST_WL_EXPIRYDATE_FN);
		wlExpiryDatePickerPC.setLowerLimit(new Date());
		wlExpiryDatePickerPC.setIsDateValueOptional(true);

		wlExpiryDatePickerPC.setIsTimeEnabled(true);
		addChildComponent(wlExpiryDatePickerPC);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static RequestStatusChangePopupPM getInstance(RequestContext context) {
		RequestStatusChangePopupPM model = null;

		PageModel cachedModel = context.getPageModel();
		if (cachedModel instanceof RequestStatusChangePopupPM) {
			model = (RequestStatusChangePopupPM)cachedModel;
		} else {
			model = new RequestStatusChangePopupPM(context);
			model.extractParameters(context.getRequest());
			context.setPageModel(model);
		}

		// Make sure model is populated
		model.populateModel();

		return model;
	}//getInstance

	/**
	 * Ensure the model is populated
	 */
	private void populateModel() {
		if (m_isInitialized) {
			return;
		}
		m_isInitialized = true;
        try {
            m_request = RequestsMH.getRequest(getRequestID(), getRequestType());
            if (m_request == null) {
				this.addPageMessage(Message.ERROR_TYPE, i18n(m_bundle, RmWebBundleKey.REQUEST_NOT_FOUND_ERROR));
			}

			calculateAdvancedVtoTimeOffChoices();

        } catch (RmHardValidationException ex) {
            Message msg = handleUnableToLoadDataError(log, ex);
            msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
        } catch (Exception ex) {
            handleUnableToLoadDataError(log, ex);
        } finally {
            initContentTitle();
            if (m_request != null) {
                initContent();
                initToolbar();
            } else {
                initToolbar(false);
            }
        }
    }

	private void calculateAdvancedVtoTimeOffChoices() {
		if (m_request == null || !m_request.isAdvancedVTORequest() || !isApproveAction()) {
			return;
		}

		TORequest toRequest = (TORequest) m_request;
		RequestsMH.calculateAdvancedVtoTimeOffChoices(toRequest);
	}

	/**
	 * Prepare content title for rendering
	 */
	protected void initContentTitle() {
		String pageTitle = i18n(m_bundle, RmWebBundleKey.REQUEST_STATUS_CHANGE_ACTION);
		String itemName = RequestUtil.getLocalizedAction(m_localizer, m_common_bundle, requestAction);
		m_contentTitlePC.setPageTitle(pageTitle);
		m_contentTitlePC.setItemNameText(itemName);
	}

	/**
	 * Initialize Content
	 */
	private void initContent() {
		//If manager tries to waitlist a request that was Denied and is not eligible to be waitlisted,
		//then a warning message will be displayed.
		try {
			if ((m_request instanceof TORequest) && isAddToManWaitlistAction()
					&& RequestAuditTrail.STATUS_DENIED.equals(m_request.getRequestStatus())) {
				SimpleTextPC warningTextPC = new SimpleTextPC(m_context, i18n(m_bundle,
						RmWebBundleKey.TIMEOFF_MANUAL_ADD_DENIED_REQUEST_TO_WL));
				warningTextPC.setTitle(i18n(m_bundle, RmWebBundleKey.TIMEOFF_WAITLIST_WARNING));
				containerList.add(warningTextPC);
			}

			//If agent tries to waitlist a request that was Pending/Denied and is not eligible to be waitlisted,
			//then a message will be displayed: "Request is not eligible to be waitlisted at this time. Please try again later."
			if ((m_request instanceof TORequest) && isAddToWaitlistAction() && !((TORequest) m_request).isEligibleForWaitlist()) {
				SimpleTextPC warningTextPC = new SimpleTextPC(m_context, i18n(m_bundle, RmWebBundleKey.TIMEOFF_REQUEST_CANNOT_ADD_TO_WL));
				warningTextPC.setTitle(i18n(m_bundle, RmWebBundleKey.TIMEOFF_WAITLIST_NOTE));
				containerList.add(warningTextPC);
				canSubmitAction = false;
			} else {
				initMainList();

				if (!hasError() && (m_request instanceof ShiftSwapRequest) && isWithdrawNegotiationAction()) {
					//If this is a withdraw of an approved swap request and not a manager (that is not required to confirm the withdrawal request),
					// show line with message about confirmation of swap employee
					ShiftSwapRequest ssRequest = (ShiftSwapRequest) m_request;
					String swapEmployee = RequestUtil.getSwapEmployeeName(m_request, m_context);
					SimpleTextPC confirmationText = new SimpleTextPC(m_context, m_localizer.i18n(m_bundle,
							RmWebBundleKey.SHIFTSWAP_REQUIRE_CONFIRMATION, swapEmployee));
					confirmationText.setTitle(i18n(m_bundle, RmWebBundleKey.SHIFTSWAP_CONFIRMATION_NOTICE));
					containerList.add(confirmationText);
				}

				if (m_request instanceof TORequest && !m_request.isFlexTimeRequest()) {
					TORequest toRequest = (TORequest) m_request;
					if (toRequest.isAdvancedVTORequest() && toRequest.getTOChoicesAdjustedForCalculatedVto().isEmpty()) {
						this.addPageMessage(Message.INFO_TYPE, i18n(m_bundle, RmWebBundleKey.VTO_OPTIONS_REQUESTS_NO_VALID_TO_CHOICE));
						canSubmitAction = false;
					}
					initChoiceList(toRequest);
				}

			}
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		}
	}

	/**
	 * Prepare toolbar for rendering
	 */
	@Override
	protected void initToolbar() {
		if (canSubmitAction) {
			m_toolbar.addValidateSubmitTextButton(requestAction, i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_CONTINUE), !hasError());
		}

		m_toolbar.addCloseWindowTextButton(PageAction.CANCEL, i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_CANCEL), true);
	}

	/**
	 * Initialize Main List
	 */
	private void initMainList() {
		if (hasError()) {
			return;
		}

		String addCommentLabel = i18n(m_common_bundle, UIFWebBundleKey.ADD_COMMENT);
		mainContainerPC = new FormTwoColumnPC(m_context);
		mainContainerPC.setWrapperCSSClass(CSSUtil.CLASS_BORDER_DEFAULT);
		mainContainerPC.setLastRowType(FormTwoColumnPC.LAST_ROW_NONE);
		mainContainerPC.setPCTitleSpan(HtmlUtil.makeHiddenReadableText(addCommentLabel));

		//add row for date picker if action is adding to waitlist
		if (requestAction != null && (isAddToWaitlistAction() || isAddToManWaitlistAction())) {
			mainContainerPC.addRow(i18n(m_bundle, RmWebBundleKey.TIMEOFF_WL_EXPIRY_DATE), wlExpiryDatePickerPC.getHtmlDisplay());
		}
		String commentUI = RequestUtil.createCommentUI(COMMENT_FN, getComment(), 100, 5, this, addCommentLabel);
		mainContainerPC.addRow(i18n(m_common_bundle, UIFWebBundleKey.ADD_COMMENT), commentUI);

		containerList.add(mainContainerPC);
	}


	/**
	 * Create New List PC
	 */
	private MultiColListPC createNewRequestChoiceList() {
		MultiColListPC listPC = new MultiColListPC(m_context);
		listPC.setEnabled(false);
		listPC.setIsBorderEnabled(true);
		return listPC;
	}

	/**
	 * Initialize Choise List
	 */
	public void initChoiceList(TORequest toRequest) {
		if (isAccessibilityComplianceMode()) {
			choiceListPC = createComplianceModeList(i18n(m_bundle, RmWebBundleKey.TIMEOFF_CHOICES_GRID));
		} else {
			choiceListPC = createNewRequestChoiceList();
		}
		containerList.add(choiceListPC);
		// Do not allow Change of Selected Choice when already tentatively approved
		boolean isSelectable = isApproveAction() &&
			!RequestAuditTrail.STATUS_TENTATIVE.equals(toRequest.getRequestStatus());

		if (toChoiceID==null) {
			toChoiceID = toRequest.getApprovedChoiceIDs();
		}
		TimeOffRequestUtil.TimeOffChoiceListPCConfig config = new TimeOffRequestUtil.TimeOffChoiceListPCConfig()
																.setIsSelectable(isSelectable)
																.setIsShowSelectedInfo(true)
																.setIsShowAlert(true);

		TimeOffRequestUtil.initTimeOffChoiceListPC(m_context, choiceListPC,
				toRequest, toChoiceID, m_common_bundle, m_bundle, config);
	}

	/** Return List of Containers
	 *
	 * NOTE: Items in the Collection will be treated either as Page Components or
	 *       just a basic Object.
	 */
	@Override
	public Collection getContainers() {
		return containerList;
	}

	/**
	 * Set Request Action
	 */
	public void setRequestAction(String action) {
		requestAction = action;
	}

	/**
	 * Return Request Action
	 */
	public String getRequestAction() {
		return requestAction;
	}

	/**
	 * Set TimeOff Choice ID
	 */
	public void setTOChoiceID(ID[] choiceIds) {

		if (choiceIds == null || choiceIds.length == 0) {
			toChoiceID = new HashSet<ID>();
			return;
		}

		toChoiceID = new HashSet<ID>(Arrays.asList(choiceIds));
	}

	/**
	 * Return Time Off Choice ID
	 */
	public Set<ID> getTOChoiceID() {
		return toChoiceID == null ? Collections.emptySet() : toChoiceID;
	}
	/**
	 * Return Expiry date
	 */
	public Date getExpiryDate() {
		return wlExpiryDatePickerPC.getDate();
	}

	/**
	 * Return true if Action is Approve or Tentatively Approve
	 */
	private boolean isApproveAction() {
		return PageAction.APPROVE.equals(requestAction) || PageAction.TENTATIVELY_APPROVE.equals(requestAction);
	}

	/**
	 * Return true if Action is Manually Add To Waitlist
	 */
	private boolean isAddToWaitlistAction() {
		return RmPageAction.ADD_TO_WAITLIST_ACTION.equals(requestAction);
	}

	/**
	 * Return true if Action is Manually Add To Waitlist
	 */
	private boolean isWithdrawNegotiationAction() {
		return PageAction.WITHDRAW_NEGOTIATION.equals(requestAction);
	}

	/**
	 * Return true if Action is Manually Add To Waitlist
	 */
	private boolean isAddToManWaitlistAction() {
		return RmPageAction.ADD_TO_MAN_WAITLIST_ACTION.equals(requestAction);
	}

	/**
	 * Return Required HTML
	 */
	@Override
	public String getRequiredHTML() {
		return super.getRequiredHTML() +
				HtmlUtil.makeHiddenInput(REQUEST_ACTION_FN, requestAction);
	}
	public boolean isCloseAfterLoadCustomize() {
		return isCloseAfterLoadCustomize;
	}

	public void setCloseAfterLoadCustomize(boolean isCloseAfterLoadCustomize) {
		this.isCloseAfterLoadCustomize = isCloseAfterLoadCustomize;
	}
	@Override
	public String getJavaScriptInitCode() {
		StringBuilder jsInit = new StringBuilder(1000);
		jsInit.append(super.getJavaScriptInitCode());
		if(isCloseAfterLoadCustomize()){
			jsInit.append(super.getJSBeforeClose() );
			jsInit.append("\n window.notifyOpener=true;\n");
			jsInit.append("if(is.chrome)\n");
			jsInit.append("handleUnload(window);\n");
			jsInit.append("window.close();\n");
		}
		return jsInit.toString();
	}

	private class SimpleTextPC extends CollapsibleContainerPC {
		private String m_text = "";

		public SimpleTextPC(RequestContext context, String text) {
			super(context);
			m_text = text;
		}

		/**
		 * Initialize Component For Display
		 */
		@Override
		public void initForDisplay() {
			setTemplate("{0}");
			setIsBorderEnabled(true);
			StringBuilder sb = new StringBuilder();
			sb.append("<table><tr><td class=\"formRowLightNoBorderBottom\">").append(m_text).append("</td></tr></table>");
			super.addComponent(sb.toString());
		}

		public void setText(String text) {
			m_text = text;
		}
	}

}
