package com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.listner;

import java.util.Collections;
import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.timeoffcalculator.TONotifyMessage;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.workresource.model.WorkResourceAssignment;
import com.bluepumpkin.ejb.bbm.workresource.model.WorkResourceAssignmentFieldInfo;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.ejb.TOHoursPerDayManager;
import com.witness.ejb.bbm.daonotifier.DAOChangeHook;
import com.witness.ejb.bbm.daonotifier.ValueObjectPairList;


/**
 * Employee Termination hook to capture the termination of an employee. This
 * will be called through a JMS thread which notifies an employee effectivity
 * change every one hour.
 */
public class EmpEffectivityChangeListner implements DAOChangeHook {

	private static Category LOG = Log.initCategory(EmpEffectivityChangeListner.class.getName());
	TimeZone tz = TimeZone.getTimeZone("GMT");

	/**
	 * Notifies RemoteCacheManager to send WorkResourceAssignment creation.
	 * Will be called when WorkResourceAssignment table has a new entry created as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.wra.model.WorkResourceAssignment objects that were created.
	 */
	@Override
	public void OnCreateObject(ValueObjectBase[] valueObjects) throws JdmoException {
		if (LOG.isDebugEnabled()){
			LOG.debug("WorkResourceAssignment Hook OnCreateObject valueObjects = " + valueObjects);
		}
		if (valueObjects != null) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			throw new IllegalArgumentException();
		}
	}

	/**
	 * Notifies RemoteCacheManager to send WorkResourceAssignment deletion.
	 * Will be called when WorkResourceAssignment table has an entry deleted as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.wra.model.WorkResourceAssignment objects that were deleted.
	 */
	@Override
	public void OnDeleteObject(ValueObjectBase[] valueObjects) throws JdmoException {
		// When an WorkResourceAssignment is deleted the send notification
		if (LOG.isDebugEnabled()){
			LOG.debug("WorkResourceAssignment Hook OnDeleteObject valueObjects = " + valueObjects);
		}
		if (valueObjects != null && valueObjects.length > 0) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			throw new IllegalArgumentException();
		}
	}

	/**
	 * Notifies RemoteCacheManager to send WorkResourceAssignment updates.
	 * Will be called when WorkResourceAssignment table has been updated as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param old and current value objects com.bluepumpkin.ejb.bbm.wra.model.WorkResourceAssignment
	 *            will be in oldAndNewObjects.
	 */
	@Override
	public void OnUpdateObject(ValueObjectPairList oldAndNewObjects) throws JdmoException {
			
		if (oldAndNewObjects == null || oldAndNewObjects.getCurrentValues() == null) {
			throw new IllegalArgumentException();
		}
		
		if (LOG.isDebugEnabled()) {
			LOG.debug("WorkResourceAssignment Hook OnUpdateObject sending update notification " + oldAndNewObjects.getCurrentValues());
					
		}

		ValueObjectBase[] newValueObjects = oldAndNewObjects.getCurrentValues();
		ValueObjectBase[] oldValueObjects = oldAndNewObjects.getOldValues();
		for (int i = 0; i < newValueObjects.length; i++) {
			WorkResourceAssignment newWRA = (WorkResourceAssignment) newValueObjects[i];
			WorkResourceAssignment oldWRA = (WorkResourceAssignment) oldValueObjects[i];
			if (newWRA.getOrganizationID().equals(oldWRA.getOrganizationID())) {
				Date startDate = newWRA.getStartTime().after(oldWRA.getStartTime()) ? oldWRA.getStartTime() : newWRA.getStartTime();
				Date endDate = null;
				if (newWRA.getEndTime() != null && oldWRA.getEndTime() != null) {
					endDate = newWRA.getEndTime().before(oldWRA.getEndTime()) ? oldWRA.getEndTime() : newWRA.getEndTime();
				}
				try {
						ID employeeId = newWRA.getFieldValueID(WorkResourceAssignmentFieldInfo.WRASSIGNMENT_WORKRESOURCEID);
						TONotifyMessage msg = new TONotifyMessage(Collections.singleton(employeeId)
						                        , startDate, endDate, TONotifyMessage.TO_NOTIFY_EMP_ORG_CHANGE);
						//getTOHoursPerDayManager().updateTOHoursPerDay(msg);
					msg.sendMessage();

				} catch (Exception e) {
					LOG.error("EmpEffectivityChangeListner.OnUpdateObject exception", e);
				}
			} else {
				updateTOHoursPerDay(oldWRA);
				updateTOHoursPerDay(newWRA);
			}
		}
	}

	public void updateTOHoursPerDayForValueObjects(ValueObjectBase[] valueObjects) {
		try {
			for (int i = 0; i < valueObjects.length; i++) {
				updateTOHoursPerDay((WorkResourceAssignment) valueObjects[i]);

			}
		} catch (Exception e) {
			LOG.error("EmpEffectivityChangeListner.updateTOHoursPerDayForValueObjects exception", e);
		}
	}

	void updateTOHoursPerDay(WorkResourceAssignment wra) {
		try {
			ID employeeID = wra.getFieldValueID(WorkResourceAssignmentFieldInfo.WRASSIGNMENT_WORKRESOURCEID);
			TONotifyMessage msg = new TONotifyMessage(Collections.singleton(employeeID)
			                                            , wra.getStartTime(), wra.getEndTime(), TONotifyMessage.TO_NOTIFY_EMP_ORG_CHANGE);
			//getTOHoursPerDayManager().updateTOHoursPerDay(msg);
			msg.sendMessage();
		} catch (Exception e) {
			LOG.error("EmpEffectivityChangeListner.updateTOHoursPerDay exception", e);
		}
	}

	public TOHoursPerDayManager getTOHoursPerDayManager() throws BbmCreateException {
		return RmManagerFactory.getInstance(true).getTOHoursPerDayManager(null, null);
	}
}
