/**
 * Sends EmployeeTimeOff updates to RM/IFs through remote cache manager. This class hooks
 * the EmployeeTimeOff table changes (CRUD) and sends the appropriate notifications through
 * PeopleCacheProcessor. PeopleCacheProcessor will be used by the RemoteCacheManager to send
 * EmployeeTimeOff updates to RM/IFs.
 */
public class EmpTODefaultChangeListner implements DAOChangeHook {

	private static Category LOG = Log.initCategory(EmpTODefaultChangeListner.class.getName());
	TimeZone tz = TimeZone.getTimeZone("GMT");

	/**
	 * Notifies RemoteCacheManager to send EmployeeTimeOff creation.
	 * Will be called when EmployeeTimeOff table has a new entry created as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.EmployeeTimeOff.model.EmployeeTimeOff objects that were created.
	 */
	public void OnCreateObject(ValueObjectBase[] valueObjects) throws JdmoException {
		if (LOG.isDebugEnabled()) LOG.debug("EmployeeTimeOff Hook OnCreateObject valueObjects = " + valueObjects);
		if (valueObjects != null) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			throw new IllegalArgumentException();
		}
	}

	/**
	 * Notifies RemoteCacheManager to send EmployeeTimeOff deletion.
	 * Will be called when EmployeeTimeOff table has an entry deleted as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.EmployeeTimeOff.model.EmployeeTimeOff objects that were deleted.
	 */
	public void OnDeleteObject(ValueObjectBase[] valueObjects) throws JdmoException {
		// When an EmployeeTimeOff is deleted the send notification
		if (LOG.isDebugEnabled()) LOG.debug("EmployeeTimeOff Hook OnDeleteObject valueObjects = " + valueObjects);
		if (valueObjects != null && valueObjects.length > 0) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			throw new IllegalArgumentException();
		}
	}

	/**
	 * Notifies RemoteCacheManager to send EmployeeTimeOff updates.
	 * Will be called when EmployeeTimeOff table has been updated as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param old and current value objects com.bluepumpkin.ejb.bbm.EmployeeTimeOff.model.EmployeeTimeOff
	 *            will be in oldAndNewObjects.
	 */
	public void OnUpdateObject(ValueObjectPairList oldAndNewObjects) throws JdmoException {
		if (oldAndNewObjects != null && oldAndNewObjects.getCurrentValues() != null) {
			if (LOG.isDebugEnabled()) LOG.debug("EmployeeTimeOff Hook OnUpdateObject sending update notification " + oldAndNewObjects.getCurrentValues());
			for (int i = 0; i < oldAndNewObjects.getCurrentValues().length; i++) {
				EmployeeTimeOffDefault empTODefaultNew = (EmployeeTimeOffDefault) oldAndNewObjects.getCurrentValues()[i];
				EmployeeTimeOffDefault empTODefaultOld = (EmployeeTimeOffDefault) oldAndNewObjects.getOldValues()[i];
				try {
					if (empTODefaultNew.getHoursPerDay() != empTODefaultOld.getHoursPerDay()
					        || empTODefaultNew.getHoursPerWeek() != empTODefaultOld.getHoursPerWeek()) {
						updateTOHoursPerDayForEmp(empTODefaultOld.getEmployeeID(), empTODefaultNew.getActivityID());
					}
				} catch (Exception e) {
					e.printStackTrace();
				}
			}
		} else {
			throw new IllegalArgumentException();
		}
	}

	public void updateTOHoursPerDayForValueObjects(ValueObjectBase[] valueObjects) {
		try {
			for (int i = 0; i < valueObjects.length; i++) {
				EmployeeTimeOffDefault empTODefault = (EmployeeTimeOffDefault) valueObjects[i];
				if(empTODefault.getHoursPerDay() >0 || empTODefault.getHoursPerWeek() >0) {  //send notification only for events that have values >0
					updateTOHoursPerDayForEmp(empTODefault.getEmployeeID(), empTODefault.getActivityID());
				}
			}
		} catch (Exception e) {
			e.printStackTrace();
		}
	}
	void updateTOHoursPerDayForEmp(ID empID, ID activityId){
		try {
			TONotifyMessage msg = new TONotifyMessage
			        ( Collections.singleton(empID),  activityId, TONotifyMessage.TO_NOTIFY_EMP_TO_DEF_HOURS_CHANGE);
			//getTOHoursPerDayManager().updateTOHoursPerDay(msg);
			msg.sendMessage();
		} catch  (Exception e) {
			e.printStackTrace(); 
		}
	}

	public TOHoursPerDayManager getTOHoursPerDayManager() throws BbmCreateException {
		return RmManagerFactory.getInstance(true).getTOHoursPerDayManager(null, null);
	}
}
