package com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.listner;

import java.util.TimeZone;
import java.util.Collections;
import java.util.Collection;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.workresource.model.Employee;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.ejb.TOHoursPerDayManager;
import com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.ejb.TOHoursPerDayDAO;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.witness.ejb.bbm.daonotifier.*;
import com.witness.ejb.bbm.daonotifier.ValueObjectPairList;


/**
 * Employee creation hook to capture the creation of an employee.
 */
public class EmployeeCreateListner implements DAOChangeHook, DAOChangeConnection {

	private static Category LOG = Log.initCategory(EmployeeCreateListner.class.getName());
	TimeZone tz = TimeZone.getTimeZone("GMT");

	private Jdmo m_jdmo;

	public void setJdmo(Jdmo dmo)
	{
		m_jdmo = dmo;
	}
	/**
	 * Notifies RemoteCacheManager to send Employee creation.
	 * Will be called when Employee table has a new entry created as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.wra.model.Employee objects that were created.
	 */
	public void OnCreateObject(ValueObjectBase[] valueObjects) throws JdmoException {
		if (LOG.isDebugEnabled()) LOG.debug("Employee Hook OnCreateObject valueObjects = " + valueObjects);
		if (valueObjects != null) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			throw new IllegalArgumentException();
		}
	}

	/**
	 * Notifies RemoteCacheManager to send Employee deletion.
	 * Will be called when Employee table has an entry deleted as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.wra.model.Employee objects that were deleted.
	 */
	public void OnDeleteObject(ValueObjectBase[] valueObjects) throws JdmoException {
	}

	/**
	 * Notifies RemoteCacheManager to send Employee updates.
	 * Will be called when Employee table has been updated as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param old and current value objects com.bluepumpkin.ejb.bbm.wra.model.Employee
	 *            will be in oldAndNewObjects.
	 */
	public void OnUpdateObject(ValueObjectPairList oldAndNewObjects) throws JdmoException {
	}

	public void updateTOHoursPerDayForValueObjects(ValueObjectBase[] valueObjects) {
		try {
			for (int i = 0; i < valueObjects.length; i++) {
				insertLastPublishedDate(Collections.singleton(((Employee) valueObjects[i]).getID()));
			}
		} catch (Exception e) {
			e.printStackTrace();
		}
	}
	public void insertLastPublishedDate(Collection empIDs) throws BbmUpdateException {
		TOHoursPerDayDAO toHoursPerDayDAO = null;
		try {
			toHoursPerDayDAO = new TOHoursPerDayDAO();
			toHoursPerDayDAO.insertLastPublishedDate(empIDs, m_jdmo);
		} catch (Exception e) {
			throw new BbmUpdateException(e);
		} finally {
			if (toHoursPerDayDAO != null) toHoursPerDayDAO.cleanUp();
		}
	}
	public TOHoursPerDayManager getTOHoursPerDayManager() throws BbmCreateException {
		return RmManagerFactory.getInstance(true).getTOHoursPerDayManager(null, null);
	}
}
