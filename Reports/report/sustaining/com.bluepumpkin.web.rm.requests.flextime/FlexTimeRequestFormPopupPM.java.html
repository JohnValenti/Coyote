	@Override
	protected TODetailData initializeModelTimeOffData(ID empID, TORequest aRequest)
			throws Exception {

		// Load data using FlexTimeRequestsMH instead TimeOffRequestsMH

		TimeOffRequestsMH.TODetailData toData;
		if (isManagerMode()) {
			toData = modelHandler.getFlexTimeDetailDataForManager(
				m_context, m_context.getUser(), empID, isNew(), aRequest, getIsFromNetStaffingRibbon());
		} else {
			toData = modelHandler.getFlexTimeDetailData(m_context, empID, aRequest, getIsFromNetStaffingRibbon());
		}

		mergeValidationResult(aRequest);

		// set default values for time off date range picker and time off type (and employee ID if manager mode)
		// note: using toData.getEmpID() instead provided empID because empID value will be null when manager creating new Flex Request
		setDefaultTimeOffAndMakeup(toData.getEmpID() , toData.getTOTypes(), toData.getOrgSetting(), new Date(), DEFAULT_FLEX_DURATION_IN_MINUTES);

		return toData;

	}

	/**
	 * Set default for time off date range picker to first 15' if this is a new flex request
	 * and also set default time off type and employee ID (so the Length and Accounted Hours can be calculated)
	 */
	private void setDefaultTimeOffAndMakeup(ID empID, Collection<Activity> toTypes, OrganizationSetting organizationSetting,
			Date shiftDate, int durationMins) throws Exception {
		if (isNew() && isTimeOffRangeEmpty(getTimeOffRequest()) && !toTypes.isEmpty() && empID != null) {
			// calculate date range for first 15 mins of shift event
			ShiftAssignmentDetailData db = modelHandler.getShiftAssignmentDetail(m_context, m_context.getUser().getEmployeeID(), empID, shiftDate);
			if (db.getShiftAssignment() != null) {

				TimeRange timeOffDuration = setDefaultTimeOff(empID, toTypes, durationMins, db);
				if (timeOffDuration != null) {
					setDefaultMakeup(empID, shiftDate, organizationSetting, (int) timeOffDuration.getDurationMin(), db);
				}
			}
		}
	}

	private TimeRange setDefaultTimeOff(ID empID, Collection<Activity> toTypes, int durationMins, ShiftAssignmentDetailData db)
		throws Exception {
		// first 15 minutes of shift event
		ShiftAssignment sa = db.getShiftAssignment();
		TimeRange duration = modelHandler.getDurationFromFirstAvailableBlock(empID, sa.getStartTime(), sa.getEndTime(), durationMins);

		if (duration != null) {
			// update TORequest with default values
			List<TOChoice> choiceList = new ArrayList<TOChoice>();
			TOChoice choice = new TOChoice();
			choice.setRank(1);
			choice.setUserRank(1);
			choice.setStartDate(duration.getStartDate());
			choice.setEndDate(duration.getEndDate());
			choiceList.add(choice);
			getTimeOffRequest().setRequestChoiceList(choiceList);

			// time off type must be set so Accounted Hours can be calculated
			getTimeOffRequest().setTimeOffType(toTypes.iterator().next().getID());

			if (getRequestedFor() == null) {
				// manager creating new Flex request, set default requestedFor to first option so Accounted Hours can be calculated
				getTimeOffRequest().setEmployeeID(empID);
			}

		}

		return duration;
	}

	private TimeRange getFirstGapFitDefaultMakeup(List<TimeRange> gaps, int durationMins) {
		Date currentTime = new Date();
		for (TimeRange gap : gaps) {
			if (currentTime.before(gap.getStartDate()) && gap.getDurationMin() >= durationMins) {
				return gap;
			}
		}

		return null;
	}

	private void setDefaultMakeup(ID empID, Date shiftDate, OrganizationSetting organizationSetting, int durationMins,
			ShiftAssignmentDetailData db) throws Exception {

		List<TimeRange> gaps = modelHandler.getAllGaps(db.getShiftAssignment());
		TimeRange firstGapFitDefaultMakeup = getFirstGapFitDefaultMakeup(gaps, durationMins);
		FlexRequestMakeup flexMakeup = null;

		// Set default 15' makeup in gap if there is a gap, and it is equal or
		// longer than 15 minutes
		if (organizationSetting.getAllowMakeupInGap() && firstGapFitDefaultMakeup != null) {
			FlexReqMakeupGap makeupGap = new FlexReqMakeupGap();
			makeupGap.setExtGapActivityID(db.getShiftAssignment().getActivityID());
			makeupGap.setExtGapStartTime(firstGapFitDefaultMakeup.getStartDate());
			makeupGap.setExtGapEndTime(firstGapFitDefaultMakeup.getEndDate());
			makeupGap.setExtGapDuration(durationMins);
			
			//the default makeup needs to be at the beginning of the gap
			makeupGap.setExtGapGap(0);
			flexMakeup = new FlexRequestMakeup();
			flexMakeup.addFlexReqMakeupGap(makeupGap);
		} else if (organizationSetting.getAllowMakeupAfterShift() && modelHandler.isAfterExtensionEditable(db.getShiftAssignment())) {
			// Set default 15' makeup if AFTER extension is available
			flexMakeup = new FlexRequestMakeup();
			flexMakeup.setExtAfterActivityID(db.getShiftAssignment().getActivityID());
			flexMakeup.setExtAfterGap(0);
			flexMakeup.setExtAfterDuration(durationMins);
		} else {
			//Set a empty  Makeup as a default value
			flexMakeup = new FlexRequestMakeup();
			flexMakeup.setExtAfterActivityID(db.getShiftAssignment().getActivityID());
			flexMakeup.setExtAfterGap(0);
			flexMakeup.setExtAfterDuration(durationMins);
		}

		if (flexMakeup != null) {
			getTimeOffRequest().setFlexRequestMakeupList(new ArrayList<FlexRequestMakeup>(Arrays.asList(flexMakeup)));

			//set shift assigment details
			flexMakeup.setShiftAssignmentID(db.getShiftAssignmentInfo().getShiftAssignmentId());
			flexMakeup.setSPID(modelHandler.getSPID(this.getRequestContext(), empID, shiftDate));
			TimeRange mainShiftStartEnd = db.getShiftAssignment().getMainShiftTimeRange();
			flexMakeup.setShiftStartTime(mainShiftStartEnd.getStartDate());
			flexMakeup.setShiftEndTime(mainShiftStartEnd.getEndDate());

			modelHandler.setMakeupStartAndEnd(flexMakeup, this.getRequestContext());
		}
	}

	private boolean isTimeOffRangeEmpty(TORequest toRequest) {
		boolean empty;
		List<TOChoice> toChoices = toRequest.getRequestChoiceList();
		if (toChoices.isEmpty()) {
			empty = true;
		} else {
			TOChoice toChoice = toChoices.get(0);
			empty = toChoice.getStartDate() == null || toChoice.getEndDate() == null;
		}
		return empty;
	}

	@Override
	public void initChoiceList() {

		// Below logic is modified from TimeOffRequestFormPopupPM.initChoiceList()

		timeOffChoiceListPC = new DynamicMultiColListPC(m_context) {
			private String rowID = null;
			private StringBuilder labelUpdatersJs = new StringBuilder("<script>");

			@Override
			protected String getCustomCellTemplate() {
				return "\n<a class=\"focusableNode\" href=\"#{1}\" onClick=\"return false;\" "
					+ "onKeyDown=\"{0}.onKeyDown(this);\">{2}</a>";
			}

			@Override
			protected void appendHtmlRowData(StringBuffer html, MultiColumnNodeData rowData, int nRow, int nRowDisplay) {
				rowID = this.getName() +"r"+nRow;
				super.appendHtmlRowData(html, rowData, nRow, nRowDisplay);
			}

			@Override
			public String getHtmlDisplay() {
//				can not use another method intended to JS, because they are called before labelUpdatersJs is created
				return super.getHtmlDisplay() + labelUpdatersJs.append("</script>").toString();
			}

			/**
			 * As far as <td> blocks in table are updated after date selections due to ajax requests.
			 * Row readable info must be updated. Here label is added in readable header for colums, which are with simple
			 * text inside (which updates) and jquery events are binded to this <td> columns to update
			 * readable info, when cell content changed
			 */
			@Override
			protected String addReadableRowInfo(int curentCol, MultiColumnNodeData data) {
				StringBuilder labels = new StringBuilder();
				int labelNum = 0;

				for (Object col : data.getColumnData()) {
//					if it is not tag content, bind label
					if (!(col!=null && col.toString().trim().startsWith("<th"))) {

						Object header = this.getListHeaders().get(labelNum);
						HtmlLabel hiddenReadableText = 
							HtmlUtil.createHiddenReadableText(col == null ? HtmlUtil.NBSP : (header.toString() + " " + col.toString()));
						String tdId = rowID + "c" + labelNum;
						String labelId = tdId + "_lbl";
						hiddenReadableText.setId(labelId);
						labelUpdatersJs.append("jQuery('#").append(tdId).append("').on('DOMSubtreeModified', function(e) {")
							.append("	jQuery('#").append(labelId).append("').html(' ").append(header).append(" ' + e.target.innerHTML);")
							.append("});");
						labels.append(hiddenReadableText.toString());
					}
					labelNum++;
				}
				return labels.toString();
			}
		};
//		navigation works, only if component have such name
		timeOffChoiceListPC.setName("dynList");
		if (isAccessibilityComplianceMode()) {
			timeOffChoiceListPC.setWrapperRegion(true);
			timeOffChoiceListPC.setReadableRowInfoEnabled(true);
			timeOffChoiceListPC.getWrapper().setAttribute("aria-label", i18n(m_bundle, RmWebBundleKey.FLEXTIME_CHOICES));
		}
		timeOffChoiceListPC.setEnabled( !isTOChoiceEditDisabled());
		timeOffChoiceListPC.setHeightInPixels(HEIGHT_IN_PIXELS);
		getContainers().add(timeOffChoiceListPC);

		// Init List
		timeOffChoiceListPC.setTableAttribute("summary", i18n(m_bundle, RmWebBundleKey.FT_REQUEST_POPUP_CHOICES_TABLE_SUMMARY));

		// Init Header
		TableHeaderPC headerPC = timeOffChoiceListPC.getHeader();
		headerPC.addColumnHeaderData("", i18n(m_bundle, RmWebBundleKey.FLEXTIME_CHOICES));
		headerPC.addColumnHeaderData("", i18n(m_common_bundle, UIFWebBundleKey.LENGTH));
		headerPC.addColumnHeaderData("", i18n(m_bundle, RmWebBundleKey.TIMEOFF_HOURS_ACCOUNTED_LABEL));

		// Init Template Node
		DefaultMultiColumnNodeData tmplNodeData = getTemplateData();
		timeOffChoiceListPC.setTemplateNodeData(tmplNodeData);

		// Populate List Model
		List<DefaultMultiColumnNodeData> listModel = getChoiceDateList();
		listModel.add(getDefaultLastRow());
		timeOffChoiceListPC.setListModel(listModel);
	}

	private String getPcValueReadableLabel(String value, String inputId, String title) {
		HtmlLabel valueLbl = HtmlUtil.createHiddenReadableText(value);
		
		String valueLblId = inputId + "_val_lbl";
		
		valueLbl.setId(valueLblId);

		String js = "$('#" + inputId + "').on('change', " +
			"	function(e) {$('#" + valueLblId + "').html(' " + (title == null ? HtmlUtil.NBSP : title) + " ' + e.target.value)" +
			"}); \n";
		appendJSToJSMain(js);

		return String.format("%s\n", valueLbl);
	}
	

	@Override
	protected String[] getTimeOffChoiceLenAndAlerts(
		TOChoice choice, Localizer localizer, TimeZone tz, ResourceBundle rmBundle, boolean isApproved, boolean printAccountedHours) {

		String[] result = TimeOffRequestUtil.getTimeOffChoiceLenAndAlerts(choice, localizer, tz, rmBundle, isApproved, true);

		// if valid data and length is not calculated, calculate and update length and accounted hours only
		if (choice != null
			&& getRequestedFor() != null
			&& getTimeOffType() != null
			&& choice.getStartDate() != null
			&& choice.getEndDate() != null
			&& !choice.isLengthSet()) {
			try {
				String[] data = modelHandler.calculateTOLengthAndAccountedHours(
					getRequestContext(), getRequestedFor(), getTimeOffType(),
					choice.getStartDate(), choice.getEndDate(), this.getTimeOffRequest().isApproved());
				result[0] = data[0];
				result[1] = data[1];
			} catch (Exception e) {
				LOG.error("Exception when calculating accounted hours", e);
			}
		}
		return new String[] {result[0], result[1]};
	}

	private void mergeValidationResult(TORequest toRequest) {
		if(toRequest != null) {
			Collection valResult = toRequest.getValidationResults(true);
			if(valResult != null && !valResult.isEmpty()) {
				List<ValidationResult> newList = new ArrayList<ValidationResult>();
				newList.addAll(valResult);
				m_valResults = newList;
			}
		}
	}
