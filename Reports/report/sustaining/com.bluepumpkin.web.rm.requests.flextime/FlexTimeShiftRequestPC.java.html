	// Shift Extension
	protected static final String EXTENSION_FN = "extension";

	//Extension Type Item IDs in Drop down list
	protected static final String EXTENSION_SELECTOR_OPTION_BEFORE_ID_PREFIX = "BEFORE";

	protected static final String EXTENSION_SELECTOR_OPTION_AFTER_ID_SUFFIX = "AFTER";

	protected static final String EXTENSION_SELECTOR_OPTION_NONE_ID = "NONE_NONE";

	private static final int DEFAULT_FLEX_DURATION_IN_MINUTES = 15;

	private final ResourceBundle bundle;
	private final FlexTimeRequestsMH modelHandler;

	private boolean paramValid;
	private boolean isNewRequest = false;

	private FlexTimeShiftRequestPCModel model = null;

	// Request Management Settings per organization
	private OrganizationSetting orgSettings;

	public FlexTimeShiftRequestPC(RequestContext context, String name, TORequest timeoffRequest, boolean viewOnlyMode) {
		super(context);
		setName(name);
		
		model = new FlexTimeShiftRequestPCModel(context);

		this.bundle = getLocalizer().getBundle(RmWebBundleKey.BUNDLE_NAME);
		modelHandler = new FlexTimeRequestsMH();

		this.model.setTimeoffRequest(timeoffRequest);
		this.model.setViewOnlyMode(viewOnlyMode);

		setTitle(i18n(bundle, RmWebBundleKey.FLEX_SHIFT_REQUEST));

		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
		addRequiredJavaScriptFile(RmJavaScriptFileID.CUSTOM_FLEX_TIME_SHIFT_REQUEST);

		//We need to add these JS variables for the field validators to work properly
		JSValidatorUtil.addVDJSVars(this);

	}

	private boolean isViewOnlyMode() {
		return this.model.isViewOnlyMode();
	}

	private DynamicMultiColListWithToolbarPC createShiftRequestTable(RequestContext context, String name, ID employeeID) throws Exception {
		DynamicMultiColListWithToolbarPC table = new DynamicMultiColListWithToolbarPC(context);

		table.setName(name);
		table.setJSMediator(RmJavaScriptFileID.MEDIATOR_FLEXTIME_MAKEUP_DYNAMICLIST);

		// buttons
		table.setIsMoveUpButtonShown(false);
		table.setIsMoveDownButtonShown(false);

		if (isViewOnlyMode()) {
			table.setIsCreateButtonShown(false);
			table.setIsDeleteButtonShown(false);
		} else {
			// template row
			table.getDynamicMultiColList().setTemplateNodeData(createTemplateRow(context, 0, true, null, null));
		}

		// header row
		new FlexTimeShiftRequestPCHeaderBuilder().createHeaderRow(table.getDynamicMultiColList().getHeader(), isViewOnlyMode(), m_localizer);

		return table;
	}

	private ShiftAssignmentInfo getShift(RequestContext context, FlexTimeShiftRequestPCModel.MakeupRow makeupRow, ShiftAssignmentDetailData db) {
		ShiftAssignmentInfo result;
		if (isViewOnlyMode()) {
			result = new ShiftAssignmentInfo();
			result.setShiftStartTime(modelHandler.formatShiftStartTime(context, makeupRow.getShiftStartTime()));
			result.setShiftDurationMinutes(modelHandler.getShiftDuration(makeupRow.getShiftStartTime(), makeupRow.getShiftEndTime()));
		} else if ((makeupRow != null && makeupRow.getShiftAssignmentID() == null) || db.getShiftAssignmentInfo() == null) {
			result = new ShiftAssignmentInfo();
		} else {
			result = db.getShiftAssignmentInfo();
		}
		return result;
	}

	private void updateOutputErrorMessages(FlexTimeShiftRequestPCModel.MakeupRow makeupRow, ShiftAssignmentDetailData db, Set<String> outputErrorMessages) {
		if (outputErrorMessages != null) {
			if (makeupRow != null && (makeupRow.getShiftAssignmentID() == null || db.getShiftAssignmentInfo() == null)) {
				// row input from user, but no shift assignment found for the date
				if (makeupRow.getShiftAssignmentID() != null && db.getShiftAssignmentInfo() == null) {
					outputErrorMessages.add(i18n(bundle, RmWebBundleKey.FLEX_REQUEST_NO_LONGER_VALID_ERROR_MESSAGE));
				}
			}
		}
	}

	@Override
	public boolean extractParameters(HttpServletRequest request) {
		boolean result = super.extractParameters(request);
		paramValid = true;

		// If the table is submitted, extract rows
		if (!new FlexTimeShiftRequestPCRowBuilder().createShiftDatePicker(getRequestContext(), CalendarPC.DEFAULT_FIRST_DAY_OF_WEEK, null, 0).getDates().isEmpty()) {
			boolean isManagerMode = RmKeys.RP_MODE_MGR.equals(ParameterUtil.getParameter(request, "pageMode"));
			ID requestedForID = !isManagerMode ? m_context.getUser().getEmployeeID() : this.model.getTimeoffRequest().getEmployeeID();

			FlexTimeShiftRequestPCParamExtractor extractor = new FlexTimeShiftRequestPCParamExtractor(requestedForID, this.getRequestContext());

			paramValid = extractor.extractParameters();

			// If the table is submitted, extract rows
			if (!paramValid) {
				addPageMessage(extractor.getMessage());
			} else {
				this.model.setMakeupRows(extractor.getMakeupRows(), false);
			}
		}

		return result && paramValid;
	}

	@Override
	public boolean validate() {
		return super.validate() && paramValid;
	}

	public void initializeModel(TORequest timeoffRequest, ID employeeID, OrganizationSetting orgSettings, boolean isNewRequest) {
		this.model.setTimeoffRequest(timeoffRequest);
		this.model.setEmployeeID(employeeID);
		if (RequestAuditTrail.STATUS_APPROVED.equals(timeoffRequest.getRequestStatus())) {
			this.model.setViewOnlyMode(true);
		}

		this.orgSettings = orgSettings;
		this.isNewRequest = isNewRequest;
	}

	// To be called by page model to simulate initializeContent event
	public void initializeContent() throws Exception {
		// initialize shiftRequestTable
		DynamicMultiColListWithToolbarPC shiftRequestTable = createShiftRequestTable(getRequestContext(), getName() + "_shifttable", model.getEmployeeID());

		// update shiftRequestTable rows
		//int[] outExtensionLength = new int[1];
		Set<String> outputErrorMessages = new LinkedHashSet<String>();

		shiftRequestTable.getDynamicMultiColList().setListModel(createRows(getRequestContext(), /*outExtensionLength, */outputErrorMessages));

		if (!isViewOnlyMode()) {
			int outExtensionLength = 0;
			for (FlexTimeShiftRequestPCModel.MakeupRow makeupRow : this.model.getMakeupRows()) {
				outExtensionLength += makeupRow.getDuration();
			}
			String lengthTotal = getLocalizer().i18n(bundle, RmWebBundleKey.FLEX_SHIFT_MAKEUP_TOTAL).replace("{0}", "<span id='makeupTotal'>" + modelHandler.formatMakeupLength(getRequestContext(), outExtensionLength) + "</span>");
			shiftRequestTable.getToolbar().addToolbarComponent(
					"<td><span id='makeupTotalMessage' style='font-family:arial;font-size:12px;font-weight: bold;white-space:nowrap;margin-right:30px;'>" + lengthTotal + "</span></td>");
		}

		// Show error messages found when displaying makeup rows for PENDING requests
		if (!isViewOnlyMode()) {
			for (String outputErrorMessage : outputErrorMessages) {
				addPageMessage(Message.ERROR_TYPE, outputErrorMessage);
			}
		}

		addComponent(shiftRequestTable);
	}

	private DefaultMultiColumnNodeData createTemplateRow(RequestContext context, int rowIndex, boolean isPendingRequest, int[] outExtensionLength, Set<String> outputErrorMessages) throws Exception {
		DefaultMultiColumnNodeData node;

		Date selectedDate = new Date();
		short weekStartDate = modelHandler.getWeekStartDateForEmpOrg(context, context.getUser().getEmployeeID());
		ShiftAssignmentDetailData db = modelHandler.getShiftAssignmentDetail(context, context.getUser().getEmployeeID(), this.model.getEmployeeID(), selectedDate);
		TimeZone tz = (this.model.getTimeoffRequest().getEmployeeID() != null) ? 
				modelHandler.getEmployeeOrgTimeZone(context, this.model.getTimeoffRequest().getEmployeeID()) :
				null;

		ShiftAssignmentInfo shiftAssignmentInfo = getShift(context, null, db);
		FlexTimeShiftRequestPCRowBuilder rowBuilder = new FlexTimeShiftRequestPCRowBuilder();
		// build the extension options list with gaps from shift assignment
		List<StringsPair> extensionOptions = rowBuilder.createExtensionOptions(context, modelHandler.isBeforeExtensionEditable(orgSettings, db.getShiftAssignment()),
				modelHandler.isAfterExtensionEditable(orgSettings, db.getShiftAssignment()), orgSettings.getAllowMakeupInGap(), db.getShiftAssignment());

		ID shiftAssignmentID = db.getShiftAssignment() != null? db.getShiftAssignment().getID(): null;
		ID spID = db.getShiftAssignment() != null? db.getShiftAssignment().getCampaignID(): null;

		FlexTimeShiftRequestPCModel.MakeupRow makeupRow = new FlexTimeShiftRequestPCModel.MakeupRow(spID, selectedDate, selectedDate, shiftAssignmentID, isPendingRequest, null, null, 0, 0, tz);

		node = rowBuilder.createDataRow(context, orgSettings, isViewOnlyMode(), rowIndex, weekStartDate, selectedDate, shiftAssignmentInfo, db, makeupRow, extensionOptions);
		return node;
	}

	private List<DefaultMultiColumnNodeData> createRows(RequestContext context, Set<String> outputErrorMessages) throws Exception {
		List<DefaultMultiColumnNodeData> list = new ArrayList<DefaultMultiColumnNodeData>();
		//the rowIndex starts from 1, the index of the rowTemplate is 0
		int rowIndex = 1;
		FlexTimeShiftRequestPCRowBuilder rowBuilder = new FlexTimeShiftRequestPCRowBuilder();

		// If it's a new request and no schedule, add a new default row.
		if (model.getMakeupRows().size() == 0 && isNewRequest) {
			Date selectedDate = new Date();
			TimeZone tz = (this.model.getTimeoffRequest().getEmployeeID() != null) ? 
					modelHandler.getEmployeeOrgTimeZone(context, this.model.getTimeoffRequest().getEmployeeID()) :
					null;
			FlexTimeShiftRequestPCModel.MakeupRow row = new FlexTimeShiftRequestPCModel.MakeupRow(null, selectedDate, selectedDate, null, false, 
					null, null, DEFAULT_FLEX_DURATION_IN_MINUTES, 0, tz);
			model.getMakeupRows().add(row);
		}
		
		for (FlexTimeShiftRequestPCModel.MakeupRow makeupRow : this.model.getMakeupRows()) {

			// shift assignment
			Date selectedDate = makeupRow.getShiftStartTime();
			short weekStartDate = modelHandler.getWeekStartDateForEmpOrg(context, context.getUser().getEmployeeID());
			ShiftAssignmentDetailData db = modelHandler.getShiftAssignmentDetail(context, context.getUser().getEmployeeID(),
					model.getEmployeeID(), selectedDate);
			ShiftAssignmentInfo shiftAssignmentInfo = getShift(context, makeupRow, db);

			boolean allowMakeupInGap = orgSettings != null ? orgSettings.getAllowMakeupInGap() : true;
			//rebuild the extension options list with gaps from shift assignment
			List<StringsPair> extensionOptions = rowBuilder.createExtensionOptions(context, modelHandler.isBeforeExtensionEditable(orgSettings, db.getShiftAssignment()),
					modelHandler.isAfterExtensionEditable(orgSettings, db.getShiftAssignment()), allowMakeupInGap, db.getShiftAssignment());

			list.add(rowBuilder.createDataRow(context, orgSettings, isViewOnlyMode(), rowIndex++, weekStartDate, selectedDate, shiftAssignmentInfo, db, makeupRow, extensionOptions));

			if (makeupRow.isRequestPending()) {
				updateOutputErrorMessages(makeupRow, db, outputErrorMessages);
			}
		}

		return list;
	}
