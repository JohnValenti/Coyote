package com.bluepumpkin.web.rm.requests.flextime;

import java.util.List;

import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;

@XmlRootElement(name = "shiftAssignmentDetail")
public class ShiftAssignmentDetailData {
	private List<Activity> extensionActivities;
	private ShiftAssignment shiftAssignment;

	private ShiftAssignmentInfo shiftAssignmentInfo;
	private ShiftExtension beforeExtension;
	private ShiftExtension afterExtension;

	public List<Activity> getExtensionActivities() {
		return extensionActivities;
	}

	public void setExtensionActivities(List<Activity> extensionActivities) {
		this.extensionActivities = extensionActivities;
	}

	public void setShiftAssignment(ShiftAssignment shiftAssignment) {
		this.shiftAssignment = shiftAssignment;
	}

	public ShiftAssignment getShiftAssignment() {
		return shiftAssignment;
	}

	public ShiftAssignmentInfo getShiftAssignmentInfo() {
		return shiftAssignmentInfo;
	}

	public void setShiftAssignmentInfo(ShiftAssignmentInfo shiftAssignmentInfo) {
		this.shiftAssignmentInfo = shiftAssignmentInfo;
	}

	public ShiftExtension getBeforeExtension() {
		return beforeExtension;
	}

	public void setBeforeExtension(ShiftExtension beforeExtension) {
		this.beforeExtension = beforeExtension;
	}

	public ShiftExtension getAfterExtension() {
		return afterExtension;
	}

	public void setAfterExtension(ShiftExtension afterExtension) {
		this.afterExtension = afterExtension;
	}

	public static class ShiftAssignmentInfo {
		private ID shiftAssignmentId;
		private String shiftStartTime;
		private String shiftName;
		private int shiftDurationMinutes;
		private String shiftActivityName;

		public ID getShiftAssignmentId() {
			return shiftAssignmentId;
		}
		public void setShiftAssignmentId(ID shiftAssignmentId) {
			this.shiftAssignmentId = shiftAssignmentId;
		}
		public String getShiftStartTime() {
			return shiftStartTime;
		}
		public void setShiftStartTime(String shiftStartTime) {
			this.shiftStartTime = shiftStartTime;
		}
		public String getShiftName() {
			return shiftName;
		}
		public void setShiftName(String shiftName) {
			this.shiftName = shiftName;
		}
		public int getShiftDurationMinutes() {
			return shiftDurationMinutes;
		}
		public void setShiftDurationMinutes(int shiftDurationMinutes) {
			this.shiftDurationMinutes = shiftDurationMinutes;
		}
		public String getShiftActivityName() {
			return shiftActivityName;
		}
		public void setShiftActivityName(String shiftActivityName) {
			this.shiftActivityName = shiftActivityName;
		}
	}

	public static class ShiftExtension {
		private int length;
		private int gap;
		private ID activityId;

		public ShiftExtension() {
		}
		public ShiftExtension(ID activityId, int length, int gap) {
			super();
			this.activityId = activityId;
			this.length = length;
			this.gap = gap;
		}

		public int getLength() {
			return length;
		}
		public void setLength(int length) {
			this.length = length;
		}
		public int getGap() {
			return gap;
		}
		public void setGap(int gap) {
			this.gap = gap;
		}
		public ID getActivityId() {
			return activityId;
		}
		public void setActivityId(ID activityId) {
			this.activityId = activityId;
		}
	}

}
