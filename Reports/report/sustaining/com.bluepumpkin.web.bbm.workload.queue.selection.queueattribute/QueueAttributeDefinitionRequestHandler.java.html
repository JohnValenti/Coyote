public class QueueAttributeDefinitionRequestHandler extends RequestHandler {
	private static final Category log = Log.initCategory(QueueAttributeDefinitionRequestHandler.class.getName());
	private ResourceBundle m_bbmBundle;
	private Localizer m_localizer;
	
	//private static final String DEFAULT_PAGE = "defaultPage";
	private static final String QUEUE_ATTRIB_DEF_LIST_PAGEFORM_PAGE = "queueAttributeDefForm";
	//private static final String QUEUE_SUMMARY_PAGE = "queueSummaryPage";
	private static final String QUEUE_ATTRIB_DEF_LIST_PAGE = "queueAttributeDefList";
	private static final String QUEUE_CONFIGURE_PRIVILEGE = PrivilegeKeys.FS_CONFIGUREQUEUES;
	
	//****** UDF Mode ******//
	private static final String UDF_DEF_DEFAULT_PAGE = "udfDefaultPage";
	private static final String UDF_DEF_FORM_PAGE = "udfDefForm";
	private static final String UDF_DEF_LIST_PAGE = "udfDefList";
	private static final String UDF_CONFIGURE_PRIVILEGE = PrivilegeKeys.CORE_CONFIGUREORGANIZATION;
	private boolean m_bIsInUDFMode = false;
	
	
	/**
	 * Process Requests for the queue area
	 */
	public void processRequest(RequestContext context) {
		
		m_localizer = context.getLocalizer();
		m_bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);
		m_bIsInUDFMode = isInUDFMode(context);
		
		// Process requested action
		if (isAction(context, PageAction.EDIT)) {
			processListEdit(context);
		} else if (isAction(context, PageAction.ADD)) {
			processListCreateNew(context);
		} else if (isAction(context, PageAction.DELETE)) {
			processListDelete(context);
		} else if (isAction(context, PageAction.CREATE_NEW)) {
			processFormCreateNew(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processFormSave(context);
		} else if (isAction(context, PageAction.CANCEL)) {
			processFormCancel(context);
		} else if (isAction(context, PageAction.NEW_ID)) {
			processDone(context); //gotoForm(context);
		} else {
			if(m_bIsInUDFMode)
				setNextPage(context, UDF_DEF_DEFAULT_PAGE);
			else 
				setNextPage(context, QUEUE_ATTRIB_DEF_LIST_PAGE);
		}
	}

	/**
	 * Process Create new Action from list
	 */
	private void processListCreateNew(RequestContext context) {
		if ((!m_bIsInUDFMode && !isAuthorized(context, QUEUE_CONFIGURE_PRIVILEGE))	||
				(m_bIsInUDFMode && !isAuthorized(context, UDF_CONFIGURE_PRIVILEGE))){
			gotoList(context);
			return;
		}
		
		/*
		QueueAttributeDefinitionFormPageModel formModel = new QueueAttributeDefinitionFormPageModel (context);
		formModel.extractParameters(context.getRequest());
		//formModel.setSelectedID("");
		//formModel.setPerformActionOnID("");
		context.setPageModel(formModel);*/
		
		gotoForm(context);
	}

	/**
	 * Process Edit Action from list
	 */
	private void processListEdit(RequestContext context) {
		if(m_bIsInUDFMode)
			setNextPage(context, UDF_DEF_FORM_PAGE);
		else 
			setNextPage(context, QUEUE_ATTRIB_DEF_LIST_PAGEFORM_PAGE);
	}

	/**
	 * Process delete Action from list
	 */
	private void processListDelete(RequestContext context) {
		if ((!m_bIsInUDFMode && !isAuthorized(context, QUEUE_CONFIGURE_PRIVILEGE))	||
				(m_bIsInUDFMode && !isAuthorized(context, UDF_CONFIGURE_PRIVILEGE)))	{
			gotoList(context);
			return;
		}
		
		
		// Extract Parameter
		QueueAttributeDefinitionListPageModel model = new QueueAttributeDefinitionListPageModel(context);
		model.extractParameters(context.getRequest());
		ID[] ids = model.getSelectedQueueIDs();
		String selectedRow = ids[0].toString();

		if(!StringUtil.isEmptyOrWhiteSpace(selectedRow)) {
			model.deleteQueueAttributeDefinition(new ID(selectedRow));
		}
		gotoList(context);
	}

	/**
	 * Process create new Action from form
	 */
	private void processFormCreateNew(RequestContext context) {
		//try {
			//QueueUtil.setSelectedQueueID(context,"");	    
			QueueAttributeDefinitionFormPageModel formModel = new QueueAttributeDefinitionFormPageModel (context);
			formModel.extractParameters(context.getRequest()); 
			context.setPageModel(formModel);
			gotoForm(context);
	}

 
	/**
	 * Process Save Action from form
	 */
	private void processFormSave(RequestContext context) { 
		QueueAttributeDefinitionFormPageModel formModel = new QueueAttributeDefinitionFormPageModel (context);
		String attributeName = "";
		try {
			formModel.extractParameters(context.getRequest());
			formModel.removeDeletedRows(context.getRequest());
			QueueAttributeDefintion attrib = formModel.getQueueAttribute(context.getRequest());
			if(attrib == null) {
				//Duplicate values found in drop down. Exiting Save. Page message added by the form model.
				formModel.extractParameters(context.getRequest());
				context.setPageModel(formModel);
				gotoForm(context);
				return;
			}
				
			attributeName = attrib.getName();
			new QueueAttributeDefinitionModelHandler().createQueueAttributes(context, attrib);
			if(m_bIsInUDFMode)
				setNextPage(context, UDF_DEF_LIST_PAGE);
			else
				setNextPage(context, QUEUE_ATTRIB_DEF_LIST_PAGE);
		} catch(BbmCreateDuplicateKeyException e) {
			log.l7dError( BbmWebLogBundleKey.QUEUE_ATTRIBUTE_DUPLICATE_DEFINTION, e);
			String message = verticaliseMsg( BbmWebBundleKey.QUEUE_ATTRIBUTE_DUPLICATE_DEFINTION, context) + " : " + attributeName;
			if(m_bIsInUDFMode){
				String udfLocalized = m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.UDF_MSG_ARG);
				udfLocalized = StringUtil.upperFirstCharOfEachWord(udfLocalized);
				message =  m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.QUEUE_ATTRIBUTE_DUPLICATE_DEFINTION, new Object[]{udfLocalized });
			}
			formModel.addPageMessage(Message.ERROR_TYPE, message);
			formModel.extractParameters(context.getRequest());
			context.setPageModel(formModel);
			gotoForm(context);
		} catch(BbmUpdateDuplicateKeyException e) {
			log.l7dError( BbmWebLogBundleKey.QUEUE_ATTRIBUTE_DUPLICATE_DEFINTION, e);
			String message = verticaliseMsg( BbmWebBundleKey.QUEUE_ATTRIBUTE_DUPLICATE_DEFINTION, context) + " : " + attributeName;
			if(m_bIsInUDFMode){
				String udfLocalized = m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.UDF_MSG_ARG);
				udfLocalized = StringUtil.upperFirstCharOfEachWord(udfLocalized);
				message =  m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.QUEUE_ATTRIBUTE_DUPLICATE_DEFINTION, new Object[]{ udfLocalized});
			}
			formModel.addPageMessage(Message.ERROR_TYPE, message);
			formModel.extractParameters(context.getRequest());
			context.setPageModel(formModel);
			gotoForm(context);	
		} catch(Exception e) {
			log.l7dError( BbmWebLogBundleKey.QUEUE_ATTRIBUTE_UNABLE_TO_SAVE, e);
			String message = verticaliseMsg( BbmWebBundleKey.QUEUE_ATTRIBUTE_UNABLE_TO_SAVE, context);
			if(m_bIsInUDFMode)
				message = m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.UDF_DUPLICATE_DEFINTION);
			formModel.addPageMessage(Message.ERROR_TYPE, message);
			context.setPageModel(formModel);
			gotoForm(context);
		}
	}
	
	
	

  private String verticaliseMsg (String key, RequestContext context) {
    // verticalise queue
    Object[] args = new Object[1];
    args[0] = context.getSystemManagerFactory().getVerticalizationManager().getVerticalizedString( context, DefaultVerticalizationManager.QUEUE);
  
    return MessageFormat.format( m_localizer.i18n(m_bbmBundle, key), args);
  
  }

	/**
	 * Process Cancel Action from form
	 */
	private void processFormCancel(RequestContext context) {
		gotoList(context);
	}


	/**
	 * Go to List
	 */
	private void gotoList(RequestContext context) {
		if(m_bIsInUDFMode)
			setNextPage(context, UDF_DEF_LIST_PAGE);
		else
			setNextPage(context, QUEUE_ATTRIB_DEF_LIST_PAGE);
	}


	/**
	 * Go to Form
	 */
	private void gotoForm(RequestContext context) {
		if(m_bIsInUDFMode)
			setNextPage(context, UDF_DEF_FORM_PAGE);
		else
			setNextPage(context, QUEUE_ATTRIB_DEF_LIST_PAGEFORM_PAGE);
	}
	
	/**
	 * Process Done Action
	 */
	private void processDone(RequestContext context) {
		if(m_bIsInUDFMode)
			setNextPage(context, UDF_DEF_LIST_PAGE);
		else
			setNextPage(context, QUEUE_ATTRIB_DEF_LIST_PAGE);
	}
	
	private boolean isInUDFMode(RequestContext context){
		HttpServletRequest aRequest = context.getRequest();
		String pathInfo = aRequest.getPathInfo();
		Screen screen = context.getScreen();
		String screenName = (screen == null)? "": screen.getName();
		return ("/udf_def".equals(pathInfo) ||
				"UDF_DEF_FORM".equals(screenName) ||
				"UDF_DEF_LIST".equals(screenName) );
	}

}
