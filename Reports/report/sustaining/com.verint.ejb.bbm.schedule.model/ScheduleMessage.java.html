package com.verint.ejb.bbm.schedule.model;

import java.util.Date;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectNode;

/**
 * Corresponds to a row in the SCHEDULEMESSAGE Table
 */
public class ScheduleMessage extends ValueObjectNode {

	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new ScheduleMessageFieldInfo();
	private static final int ACTION_ID_RECALC = 1;
	private static final int ACTION_ID_SCHEDULE = 2;
	private static final int ACTION_ID_ANALYZE = 3;
	
	public enum SchedulingAction {
		RECALC(ACTION_ID_RECALC, "R"), SCHEDULE(ACTION_ID_SCHEDULE, "S"), ANALYZE(ACTION_ID_ANALYZE, "A");
		private int actionID;
		private String actionCode;

		SchedulingAction(int actionID, String actionCode) {
			this.actionID = actionID;
			this.actionCode = actionCode;
		}

		public int getActionID() {
			return actionID;
		}

		public String getActionCode() {
			return actionCode;
		}
	}
	
	public ScheduleMessage() {
		super();
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}	

	/**
	 * Returns the scheduling action that needs to be performed. 1-Recalc,
	 * 2-Scheduling 3-Analyze
	 */
	public SchedulingAction getSchedulingAction() {
		int action = getFieldValueInt(ScheduleMessageFieldInfo.ACTION);
		try {
			return getSchedulingAction(action);
		} catch (IllegalArgumentException iae) {
			throw new IllegalStateException("The action type is invalid", iae);
		}
	}

	/**
	 * sets the scheduling action
	 */
	public void setSchedulingAction(SchedulingAction action) {
		if (action == null) {
			throw new IllegalArgumentException("The scheduling action cannot be null");
		}
		setFieldValue(ScheduleMessageFieldInfo.ACTION, action.getActionID());
	}

	/**
	 * Returns the data this is created
	 * 
	 * @return
	 */
	public Date getDateCreated() {
		return this.getFieldValueDate(ScheduleMessageFieldInfo.DATECREATED);
	}

	/**
	 * Returns the last updated time for this message.
	 * 
	 * @return
	 */
	public Date getLastUpdated() {
		return this.getFieldValueDate(ScheduleMessageFieldInfo.LASTUPDATED);
	}

	/**
	 * Returns true if the scheudles can be overridden.
	 */
	public boolean isOverrideSchedules() {
		return this.getFieldValueBoolean(ScheduleMessageFieldInfo.OVERRIDESCHEDULES);
	}

	/**
	 * Returns true if the scheduling periods are to be schedule in the
	 * chronological order.
	 */
	public boolean isScheduleInOrder() {
		return this.getFieldValueBoolean(ScheduleMessageFieldInfo.SCHEDULEINORDER);
	}

	public boolean isWhatIfMode() {
		return this.getFieldValueBoolean(ScheduleMessageFieldInfo.ISWHATIFMODE);
	}

	public boolean isEnableLQF() {
		return this.getFieldValueBoolean(ScheduleMessageFieldInfo.ENABLELQF);
	}

	public boolean isRecalcSubCampaigns() {
		return this.getFieldValueBoolean(ScheduleMessageFieldInfo.RECALCSUBCAMPAIGNS);
	}

	public String getUserSeal() {
		return this.getFormattedFieldValue(ScheduleMessageFieldInfo.USERSEAL);
	}

	public String getParameters() {
		return this.getFormattedFieldValue(ScheduleMessageFieldInfo.PARAMETERS);
	}
	
	/**
	 * Returns the scheduling action that needs to be performed. 1-Recalc,
	 * 2-Scheduling 3-Analyze
	 */
	public static SchedulingAction getSchedulingAction(int actionId) {
		switch (actionId) {
		case ACTION_ID_RECALC:
			return SchedulingAction.RECALC;
		case ACTION_ID_SCHEDULE:
			return SchedulingAction.SCHEDULE;
		case ACTION_ID_ANALYZE:
			return SchedulingAction.ANALYZE;
		default:
			throw new IllegalArgumentException("The action type is invalid");
		}
	}
}
