package com.verint.ejb.bbm.schedule.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class ScheduleMessageRequestMapping extends ValueObjectBase {
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new ScheduleMessageRequestMappingFieldInfo();
	
	public ScheduleMessageRequestMapping() {
		super();
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	/**
	 * Returns the message Id
	 */
	public ID getMessageID() {
		return getFieldValueID(ScheduleMessageRequestMappingFieldInfo.MESSAGEID);
	}

	/**
	 * Returns the request id
	 */
	public ID getRequestID() {
		return getFieldValueID(ScheduleMessageRequestMappingFieldInfo.REQUESTID);
	}
}
