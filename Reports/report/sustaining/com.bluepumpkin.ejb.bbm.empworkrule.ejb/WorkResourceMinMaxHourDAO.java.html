package com.bluepumpkin.ejb.bbm.empworkrule.ejb;

import java.util.Collection;
import java.util.Date;
import java.util.HashMap;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOEffectivity;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkResourceMinMaxHour;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkResourceMinMaxHourFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.core.base.MultiUserException;

public class WorkResourceMinMaxHourDAO extends DAOEffectivity<WorkResourceMinMaxHour> {

	private static FieldInfo m_fieldInfo = new WorkResourceMinMaxHourFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public WorkResourceMinMaxHourDAO() {
		super();
	}

	public WorkResourceMinMaxHourDAO(Jdmo dmo) {
		super(dmo);
	}

	protected WorkResourceMinMaxHour createValueObject() {
		return new WorkResourceMinMaxHour();
	}

	/**
	*   get a given work resource's min-max hour assignment
	*
	*   @idWorkResource work resource id
	*   @dtStart, @dtEnd: the time period, null date means a open period
	*
	*   @return a collection of WorkResourceMinMaxHour objects which hold the assignment information
	*/
	public Collection<WorkResourceMinMaxHour> getMinMaxHourAssignments(ID idWorkResource, Date dtStart, Date dtEnd)
			throws BbmFinderException {
		return super.getEffectivityAssignments(idWorkResource,
				WorkResourceMinMaxHourFieldInfo.WKRESMINMAXHOUR_WORKRESOURCEID, dtStart, dtEnd);
	}

	/**
	*   get min-max hour assignments for a list of work resources
	*
	*   @colWorkResourceIDs  a collection of work resource ids
	*   @dtStart, @dtEnd: the time period, null date means a open period
	*
	*   @return a HashMap of following pairs
	*            (idWorkResource, array list of WorkResourceMinMaxHour objects)
	*/
	public HashMap<ID, List<WorkResourceMinMaxHour>> getMinMaxHourAssignments(Collection<ID> colWorkResourceIDs,
			Date dtStart, Date dtEnd) throws BbmFinderException {
		return super.getEffectivityAssignments(colWorkResourceIDs,
				WorkResourceMinMaxHourFieldInfo.WKRESMINMAXHOUR_WORKRESOURCEID, dtStart, dtEnd);
	}

	/** create a min-max hour assignment */
	public ID createMinMaxHourAssignment(WorkResourceMinMaxHour objValue) throws BbmCreateException {
			return super.createNoOverlapAssignment(objValue,
				WorkResourceMinMaxHourFieldInfo.WKRESMINMAXHOUR_WORKRESOURCEID, -1);
	}

	/**
	*  save a min-max hour assignment
	*  @param bSaveAsNew where save as new assignment or replace existing one
	*
	*/
	public void updateMinMaxHourAssignment(WorkResourceMinMaxHour objValue, boolean bSaveAsNew)
			throws MultiUserException, BbmUpdateException {
		super.updateNoOverlapAssignment(objValue,
				WorkResourceMinMaxHourFieldInfo.WKRESMINMAXHOUR_WORKRESOURCEID, -1, bSaveAsNew);
	}

	/** delete skill assignments given their assignment ids */
	public void deleteMinMaxHourAssignments(Collection<ID> colIDAssignments) throws BbmRemoveException {
		deleteObjects(colIDAssignments);
	}

	public Collection<WorkResourceMinMaxHour> getMinMaxHourAssignmentsByIDs(Collection<ID> colIDSAs)
			throws BbmFinderException {
		return this.getObjectsByIDs(colIDSAs);
	}
}
