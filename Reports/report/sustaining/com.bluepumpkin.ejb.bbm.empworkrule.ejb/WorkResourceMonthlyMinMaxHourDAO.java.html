package com.bluepumpkin.ejb.bbm.empworkrule.ejb;


import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkResourceMonthlyMinMaxHour;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkResourceMonthlyMinMaxHourFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectUtil;

import java.util.*;

public class WorkResourceMonthlyMinMaxHourDAO extends DAOBase<WorkResourceMonthlyMinMaxHour> {

	@Override
	protected FieldInfo getFieldInfo() {
		return new WorkResourceMonthlyMinMaxHourFieldInfo();
	}

	@Override
	protected WorkResourceMonthlyMinMaxHour createValueObject() {
		return new WorkResourceMonthlyMinMaxHour();
	}

	public Map<ID, List<WorkResourceMonthlyMinMaxHour>> getObjectsByWorkResourceIdAndYear(Collection<ID> workResourceIDs,
			int year) throws JdmoException, BbmFinderException {
		Calendar startDate = Calendar.getInstance(TimeZoneUtil.GMT_TIMEZONE);
		Calendar endDate = Calendar.getInstance(TimeZoneUtil.GMT_TIMEZONE);
		startDate.set(year, Calendar.JANUARY, 1, 0, 0, 0);
		endDate.set(year, Calendar.DECEMBER, 1, 0, 0, 0);

		StringBuilder sql = new StringBuilder(200);
		sql.append(WorkResourceMonthlyMinMaxHourFieldInfo.Column.WORKRESOURCEID.name())
				.append(" IN ").append(getDMO().createInClause(workResourceIDs))
				.append(" AND ")
				.append(WorkResourceMonthlyMinMaxHourFieldInfo.Column.MONTHYEAR)
				.append(" BETWEEN ")
				.append(JdmoUtil.asSqlLiteral(startDate.getTime()))
				.append(" AND ")
				.append(JdmoUtil.asSqlLiteral(endDate.getTime()))
				.append(" ORDER BY ")
				.append(WorkResourceMonthlyMinMaxHourFieldInfo.Column.WORKRESOURCEID.name())
				.append(",")
				.append(WorkResourceMonthlyMinMaxHourFieldInfo.Column.MONTHYEAR);

		Collection<WorkResourceMonthlyMinMaxHour> objects = getObjects(sql.toString());
		return ValueObjectUtil.groupByField(WorkResourceMonthlyMinMaxHourFieldInfo.Column.WORKRESOURCEID.getIndex(),
				objects);
	}
}
