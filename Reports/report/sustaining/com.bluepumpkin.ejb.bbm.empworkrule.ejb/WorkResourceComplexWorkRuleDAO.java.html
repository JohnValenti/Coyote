package com.bluepumpkin.ejb.bbm.empworkrule.ejb;

import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.dao.OneToManyOneWayRelationship;
import com.bluepumpkin.ejb.bbm.dao.RichObjectDAO;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkResourceComplexWorkRule;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkResourceComplexWorkRuleFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.ejb.ComplexWorkRuleDAO;

public class WorkResourceComplexWorkRuleDAO extends RichObjectDAO<WorkResourceComplexWorkRule> {

	// meta data information
	private static FieldInfo m_fieldInfo = new WorkResourceComplexWorkRuleFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	ComplexWorkRuleDAO complexWorkRuleDAO;

	protected WorkResourceComplexWorkRule createValueObject() {
		return new WorkResourceComplexWorkRule();
	}
	
	public WorkResourceComplexWorkRuleDAO() {
		super();
	}
	
	public WorkResourceComplexWorkRuleDAO(Jdmo dmo) {
		super(dmo);
	}
	
	//over-written daobase
	protected boolean allowCreateWithId() {
		return true;
	}

	@Override
	protected void buildDBRelationships() {
		complexWorkRuleDAO = new ComplexWorkRuleDAO(m_dmo);

		addRelationship(new OneToManyOneWayRelationship(this, complexWorkRuleDAO,
				getFieldInfo().getFieldIndexByName("COMPLEXWORKRULEID"),
				complexWorkRuleDAO.getFieldInfo().getFieldIndexByName("ID"),
				WorkResourceComplexWorkRule.COMPLEXWORKRULE_RELATIONSHIP_KEY));
	}
	
	@Override
	//We only want to return records with complexworkrules that have a workunitcountconstraint <> rotation type
	public Collection<WorkResourceComplexWorkRule> getObjects(String where) throws BbmFinderException {
		return super.getObjects(where + " AND COMPLEXWORKRULEID IN (" +
				"SELECT ID FROM COMPLEXWORKRULE WHERE WORKUNITCOUNTCONSTRAINT <> " +
				Integer.toString(WorkResourceComplexWorkRuleFieldInfo.ROTATION_TYPE) + " AND ID IN (" +
				"SELECT COMPLEXWORKRULEID FROM COMPLEXWORKRULEWORKRESOURCE WHERE " + where + "))");
	}
	
	public void deleteWorkResourceComplexWorkRules(Collection<? extends ID> wrcwrIds)
	throws BbmRemoveException
	{
		if (wrcwrIds == null) return;
		try {
			getDMO().executeCommand("DELETE COMPLEXWORKRULEWORKRESOURCE WHERE SID IN " +
					getDMO().createInClause(wrcwrIds));
		} catch (Exception ex) {
			throw new BbmRemoveException(ex);
		}
	}
}
