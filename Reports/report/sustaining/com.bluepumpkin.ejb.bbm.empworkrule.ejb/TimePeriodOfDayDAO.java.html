package com.bluepumpkin.ejb.bbm.empworkrule.ejb;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.empworkrule.model.TimePeriodOfDay;
import com.bluepumpkin.ejb.bbm.empworkrule.model.TimePeriodOfDayFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

public class TimePeriodOfDayDAO extends DAOBase<TimePeriodOfDay> {

	public TimePeriodOfDayDAO() {
		super();
	}

	public TimePeriodOfDayDAO(Jdmo dmo, boolean isDmoCreated) {
		super(dmo, isDmoCreated);
	}

	public TimePeriodOfDayDAO(Jdmo dmo) {
		super(dmo);
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return new TimePeriodOfDayFieldInfo();
	}

	@Override
	protected TimePeriodOfDay createValueObject() {
		return new TimePeriodOfDay();
	}

}
