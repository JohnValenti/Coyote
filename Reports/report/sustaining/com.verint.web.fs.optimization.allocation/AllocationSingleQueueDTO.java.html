package com.verint.web.fs.optimization.allocation;

import java.util.Collection;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.verint.web.fs.optimization.SingleQueueDTO;

/**
 * A data transfer object for distributed forecast allocation data.  It may be created
 * with allocation data when creating a full response for a single queue, or without
 * data when creating a partial response for multiple queues.  It is the responsibility
 * of the client code to determine whether there is data and to lazily load the data if
 * needed.
 */
public class AllocationSingleQueueDTO extends SingleQueueDTO {

	private static final long serialVersionUID = -5974106157575145367L;

	/**
	 * Forecast volume for the parent queue.
	 */
	private TraceCube m_parentForecast = null;
	/**
	 * A map of allocations of each child queue for each time interval, keyed by
	 * child queue ID.
	 */
	private Map<ID, double[]> m_childQueueAllocations = null;
	/**
	 * A map of staffing for each child queue for each time interval, keyed by
	 * child queue ID.
	 */
	private Map<ID, double[]> m_childQueueStaffing = null;
	/**
	 * Hours of operation for each child queue, keyed by child SpQueue ID.  These are campaign HOOs
	 * that have been indexed for every SpQueue in the campaign with the current media type.
	 */
	private Map<ID, Collection<CampaignHOO>> m_childSpQueueHOOs = null;
	/**
	 * Child campaigns keyed by child SpQueue ID.
	 */
	private Map<ID, Campaign> m_childCampaigns;
	/**
	 * Child queue names, keyed by <em>SpQueue<em> ID.
	 */
	private Map<ID, String> m_childNames = null;

	/**
	 * Used to create a partial response.
	 * 
	 * @param queue
	 * @param spQueue
	 */
	public AllocationSingleQueueDTO(Queue queue, SPQueue spQueue) {
		super(queue, spQueue);
	}

	/**
	 * Used to create a full response.
	 * 
	 * It is required that childQueues, childQueueAllocation, and childQueueHOOs be
	 * ordered in the same sequence such that the corresponding elements of each
	 * list are associated with the same queue.
	 * 
	 * @param queue
	 * @param spQueue
	 * @param parentForecast
	 * @param childQueueAllocations
	 * @param childQueueHOOs
	 * @param childNames
	 */
	public AllocationSingleQueueDTO(
			Queue queue,
			SPQueue spQueue,
			TraceCube parentForecast,
			Map<ID, double[]> childQueueAllocations,
			Map<ID, double[]> childQueueStaffing,
			Map<ID, Collection<CampaignHOO>> childQueueHOOs,
			Map<ID, Campaign> childCampaigns,
			Map<ID, String> childNames) {
		super(queue, spQueue);
		m_parentForecast = parentForecast;
		m_childQueueAllocations = childQueueAllocations;
		m_childQueueStaffing = childQueueStaffing;
		m_childSpQueueHOOs = childQueueHOOs;
		m_childCampaigns = childCampaigns;
		m_childNames = childNames;
	}

	public TraceCube getParentForecast() {
		return m_parentForecast;
	}

	/**
	 * Returns the allocations of each child queue for each time interval,
	 * or {@code null} if this is a partial response.  If non-null,
	 * the array is indexed in 15-minute time intervals from the SPQueue's
	 * start date.
	 */
	public Map<ID, double[]> getChildQueueAllocation() {
		return m_childQueueAllocations;
	}

	/**
	 * Returns the paid staffing for each child queue for each time interval,
	 * or {@code null} if this is a partial response.  If non-null,
	 * the array is indexed in 15-minute time intervals from the SPQueue's
	 * start date.
	 */
	public Map<ID, double[]> getChildQueueStaffing() {
		return m_childQueueStaffing;
	}

	/**
	 * Returns the hours of operation for each child campaign, keyed by SpQueue ID,
	 * or {@code null} if this is a partial response.  Thus to retrieve
	 * HOOs for a campaign, pick any of that campaign's SP queues with the current
	 * media type and look it up with that.
	 * 
	 * @return
	 */
	public Map<ID, Collection<CampaignHOO>> getChildSpQueueHOOs() {
		return m_childSpQueueHOOs;
	}

	/**
	 * Returns the child campaigns keyed by child SpQueue ID, or {@code null} 
	 * if this is a partial response.
	 */
	public Map<ID, Campaign> getChildCampaigns() {
		return m_childCampaigns;
	}

	/**
	 * Returns the child display names, keyed by SpQueue ID,
	 * or {@code null} if this is a partial response.
	 */
	public Map<ID, String> getChildNames() {
		return m_childNames;
	}
}
