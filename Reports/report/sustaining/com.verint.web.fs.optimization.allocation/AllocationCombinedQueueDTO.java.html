package com.verint.web.fs.optimization.allocation;

import java.util.Collection;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.verint.web.fs.optimization.CombinedQueueDTO;

/**
 * Really a shell of a class, delegating all functions to an {@link AllocationMultiQueueDTO}
 * instance.  It's unclear what function this class really serves.
 */
public class AllocationCombinedQueueDTO extends CombinedQueueDTO<AllocationMultiQueueDTO> {

	public AllocationCombinedQueueDTO(
			SPQueue spQueue,
			AllocationMultiQueueDTO multiQueueDto) {
		super(spQueue, multiQueueDto);
	}

	public TraceCube getParentCombinedForecast() {
		return m_multiQueueDto.getParentForecast();
	}

	/**
	 * Returns the allocations of each child campaign's combined queue for
	 * the selected media type for each time interval,
	 * or {@code null} if this is a partial response.  If non-null,
	 * the array is indexed in 15-minute time intervals from the SPQueue's
	 * start date.
	 */
	public Map<ID, double[]> getChildCombinedQueueAllocation() {
		return m_multiQueueDto.getChildQueueAllocation();
	}

	/**
	 * Returns the paid staffing for each child campaign's combined queue for
	 * the selected media type for each time interval,
	 * or {@code null} if this is a partial response.  If non-null,
	 * the array is indexed in 15-minute time intervals from the SPQueue's
	 * start date.
	 */
	public Map<ID, double[]> getChildCombinedQueueStaffing() {
		return m_multiQueueDto.getChildQueueStaffing();
	}

	/**
	 * Returns the hours of operation for each child campaign, keyed by SpQueue ID.  Thus to retrieve
	 * HOOs for a campaign, pick any of that campaign's SP queues with the current
	 * media type and look it up with that.
	 * 
	 * @return
	 */
	public Map<ID, Collection<CampaignHOO>> getChildSpQueueHOOs() {
		return m_multiQueueDto.getChildSpQueueHOOs();
	}

	/**
	 * Returns the child campaigns keyed by child SpQueue ID, or {@code null} 
	 * if this is a partial response.
	 */
	public Map<ID, Campaign> getChildCampaigns() {
		return m_multiQueueDto.getChildCampaigns();
	}

	/**
	 * Returns the child display names, keyed by SpQueue ID.
	 */
	public Map<ID, String> getChildNames() {
		return m_multiQueueDto.getChildNames();
	}
}
