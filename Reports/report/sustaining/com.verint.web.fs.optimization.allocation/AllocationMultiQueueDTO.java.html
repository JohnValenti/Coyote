package com.verint.web.fs.optimization.allocation;

import java.util.Collection;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.verint.web.fs.optimization.MultiQueueDTO;

/**
 * A data transfer object for distributed forecast allocation data for a collection of
 * queues.  In practice this class is only used as a delegate to represent a single-media
 * combined queue.
 */
public class AllocationMultiQueueDTO extends MultiQueueDTO<AllocationSingleQueueDTO>{

	/**
	 * Forecast volume for the parent queue.
	 */
	private TraceCube m_aggregateForecast = null;
	/**
	 * A map of allocations of each child queue for each time interval, keyed by
	 * child queue ID.
	 */
	private Map<ID, double[]> m_aggregateAllocations = null;
	/**
	 * A map of staffing for each child queue for each time interval, keyed by
	 * child queue ID.
	 */
	private Map<ID, double[]> m_aggregateStaffing = null;
	/**
	 * Hours of operation for each child queue, keyed by child SpQueue ID.  These are campaign HOOs
	 * that have been indexed for every SpQueue in the campaign with the current media type.
	 */
	private Map<ID, Collection<CampaignHOO>> m_childSpQueueHOOs;
	/**
	 * Child campaigns keyed by child SpQueue ID.
	 */
	private Map<ID, Campaign> m_childCampaigns;
	/**
	 * Child queue names, keyed by <em>SpQueue<em> ID.
	 */
	private Map<ID, String> m_childNames;

	public AllocationMultiQueueDTO(
			List<AllocationSingleQueueDTO> queueDtos,
			TraceCube aggregateForecast,
			Map<ID, double[]> aggregateAllocations,
			Map<ID, double[]> aggregateStaffing,
			Map<ID, Collection<CampaignHOO>> childSpQueueHOOs,
			Map<ID, Campaign> childCampaigns,
			Map<ID, String> childNames) {
		super(queueDtos);
		m_aggregateForecast = aggregateForecast;
		m_aggregateAllocations = aggregateAllocations;
		m_aggregateStaffing = aggregateStaffing;
		m_childSpQueueHOOs = childSpQueueHOOs;
		m_childCampaigns = childCampaigns;
		m_childNames = childNames;
	}

	public TraceCube getParentForecast() {
		return m_aggregateForecast;
	}

	/**
	 * Returns the allocations of each child queue for each time interval,
	 * or {@code null} if this is a partial response.  If non-null,
	 * the array is indexed in 15-minute time intervals from the SPQueue's
	 * start date.
	 */
	public Map<ID, double[]> getChildQueueAllocation() {
		return m_aggregateAllocations;
	}

	/**
	 * Returns the paid staffing for each child queue for each time interval,
	 * or {@code null} if this is a partial response.  If non-null,
	 * the array is indexed in 15-minute time intervals from the SPQueue's
	 * start date.
	 */
	public Map<ID, double[]> getChildQueueStaffing() {
		return m_aggregateStaffing;
	}

	/**
	 * Returns the hours of operation for each child campaign, keyed by SpQueue ID.  Thus to retrieve
	 * HOOs for a campaign, pick any of that campaign's SP queues with the current
	 * media type and look it up with that.
	 * 
	 * @return
	 */
	public Map<ID, Collection<CampaignHOO>> getChildSpQueueHOOs() {
		return m_childSpQueueHOOs;
	}

	/**
	 * Returns the child campaigns keyed by child SpQueue ID, or {@code null} 
	 * if this is a partial response.
	 */
	public Map<ID, Campaign> getChildCampaigns() {
		return m_childCampaigns;
	}

	/**
	 * Returns the child display names, keyed by SpQueue ID.
	 */
	public Map<ID, String> getChildNames() {
		return m_childNames;
	}
}
