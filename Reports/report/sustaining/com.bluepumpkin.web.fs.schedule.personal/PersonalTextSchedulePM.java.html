	/**
	 * populate list component with data - prepare for rendering
	 */
	protected void populateList(Map<String, Object> theData) throws Exception
	{
		m_list.setEnabled(false);
		if (isAccessibilityComplianceMode()) {
			m_list.setRenderJavaScriptIfDisabled(true);
			//as result we will get div region as wrapper
			m_list.setUseWrapper(true);
			//add JavaScript files for ArrowNavigation
			m_list.addTableArrowNavigation();
			// add readable row info for reading whole row in grid
			m_list.setReadableRowInfoEnabled(true);
			//disable repeatHeader functionality
			m_list.disableRepeatHeader();
		}
		setListHeaders();

		if (theData==null || theData.isEmpty()) return;
		// check published periods
		Collection<TimeRange> publishedPeriods = (Collection<TimeRange>)theData.get(ScheduleViewMH.PUB_PERIODS_KEY);
		Collection<Collection<DailyScheduleSummary>> daySummaries = (Collection<Collection<DailyScheduleSummary>>) theData.get(ScheduleViewMH.DAY_SUMMARIES_KEY);
		Collection<SimpleEvent> schedule = (Collection<SimpleEvent>) theData.get(ScheduleViewMH.SCHEDULE_ITEMS_KEY);
		Collection<ShiftAssignment> shifts = (Collection<ShiftAssignment>) theData.get(ScheduleViewMH.SCHEDULE_SHIFTS_KEY);
		Collection<ShiftSwapPosting> postings = (Collection<ShiftSwapPosting>) theData.get(ScheduleViewMH.SWAP_BOARD_KEY);
		HashMap<ID, Activity> activities = (HashMap<ID, Activity>) theData.get(ScheduleViewMH.ACTIVITIES_KEY);
		Map<ID, String> campaignNamesBySpId = (Map<ID, String>) theData.get(ScheduleViewMH.CAMPAIGN_NAMES_BY_SP_ID_KEY);

		ID orgID = (ID)theData.get(ScheduleViewMH.USER_ORG_ID_KEY); //ScheduleViewMH.getOrgID(daySummaries);
		boolean bShowNote = ScheduleViewMH.getShowMyScheduleNote(m_context, orgID);

		boolean bSwapEnabled = isSwapEnabled(theData);
		boolean bShowUnavailable = showUnavailable(theData);
		boolean bShowTimeoff = showTimeoff(theData);

		ArrayList listModel = new ArrayList();

		TimeZone tz = m_context.getViewingTimeZone();
		CalendarRange aDay = DateTimeUtil.makeDayRange(m_timeRange.getStartDate(), tz);

		Collection<DailyScheduleSummary> empDaySummaries = daySummaries.iterator().next();

		//for each date in the viewing range create the row
		boolean dstAdjustment = false;
		while (!m_timeRange.isBefore(aDay)) {
			DefaultMultiColumnNodeData row_data = makeRow(aDay, publishedPeriods,
					schedule, shifts, postings, activities, empDaySummaries,
					bSwapEnabled, bShowUnavailable, bShowTimeoff, bShowNote,
					campaignNamesBySpId);
			listModel.add(row_data);

			//QC46598 java bug for Brazil DST transition 10/18/09
			//aDay.nextDay();
			dstAdjustment = ScheduleViewUtil.nextDay(aDay, dstAdjustment);
		}//endwhile
		m_list.setListModel(listModel);
	}//populateList

	/**
	 * set list headers for the multicolumn list
	 */
	private void setListHeaders(){
        VerticalizationManager verticalizationMgr = m_context.getSystemManagerFactory().getVerticalizationManager();

		TableHeaderPC header = m_list.getHeader();
        header.addColumnHeaderData(null, verticalize(m_bbm_bundle,BbmWebBundleKey.CAMPAIGN));
		header.addColumnHeaderData(null,i18n(m_bbm_bundle,BbmWebBundleKey.SHIFT));
		header.addColumnHeaderData(null,i18n(m_bbm_bundle,BbmWebBundleKey.ACTIVITIES));
		header.addColumnHeaderData(null,  verticalize(m_bbm_bundle,BbmWebBundleKey.ACTIVITYS_CAMPAIGN_LIST_HEADER));
	}//setListHeaders

	/**
	 * make one row for the schedule table
	 */
	private DefaultMultiColumnNodeData makeRow(CalendarRange aDay, Collection<TimeRange> publishedPeriods,
				Collection<SimpleEvent> schedule, Collection<ShiftAssignment> shifts, Collection<ShiftSwapPosting> postings,
				HashMap<ID, Activity> activities, Collection<DailyScheduleSummary> daySummaries, boolean bSwapEnabled,
				boolean bShowUnavailable, boolean bShowTimeoff, boolean bShowNote, Map<ID, String> campaignNamesBySpID)
	{
		if (aDay == null)
		{
			addPageMessage(Message.ERROR_TYPE, "Null aDay in PersonalTextSchedulePM.makeShiftString.");
			return null;
		}

		DefaultMultiColumnNodeData rowData = null;

		//  create the nodes
		if (!ScheduleViewPM.dayIsPublished(aDay,publishedPeriods))
		{
			rowData = makeUnpublishedRow(aDay, publishedPeriods,
                    schedule, shifts, postings, activities,
                    daySummaries, bSwapEnabled, bShowUnavailable,
                    bShowTimeoff, bShowNote);
		}
		else
		{
			rowData = new DefaultMultiColumnNodeData();

			StringsPair results = makeShiftColumnString(aDay, shifts, activities, postings, daySummaries,
	                    bSwapEnabled, bShowTimeoff, campaignNamesBySpID);

			String shiftString = results.getKey();
            String campaignName = results.getValue();

            rowData.add(campaignName, CSSUtil.STYLE_NORMAL_LEFT); //508); //Campaign column
            rowData.add(shiftString); //Shift column

            String eventsString;
			if (isAccessibilityComplianceMode()) {
				eventsString = makeEventsString(aDay, schedule, activities, bShowUnavailable, bShowTimeoff, bShowNote,
						true, true, m_localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.OVERTIME));
			} else {
				eventsString = makeEventsString(aDay, schedule, activities, bShowUnavailable, bShowTimeoff, bShowNote,
						true);
			}
			rowData.add(eventsString, "padding-right:0px;"); //Activities column
			//We set the padding for the campaign activity column (and the previous activity column) explicitly to zero because the html used to
			//render the campaign strings is already set to match the column header (7px).  If we do not set this here, there will be no padding.
			//We set the padding to 7px in the internal span because otherwise it was causing an issue with the bottom border not stretching to
			//the boundary of the cell.
			rowData.add(makeCampaignActivityString(aDay, daySummaries, shifts, schedule, activities,
					campaignNamesBySpID, bShowUnavailable, bShowTimeoff), "padding-left:0px;padding-right:0px;");
		}
		return rowData;
	}//makeRow

	/**
	 * make a row with "Unpublished" label on hte left and empty node on the right
	 */
	private DefaultMultiColumnNodeData makeUnpublishedRow(CalendarRange aDay, Collection<TimeRange> publishedPeriods,
            Collection<SimpleEvent> schedule, Collection<ShiftAssignment> shifts, Collection<ShiftSwapPosting> postings,
            HashMap<ID, Activity> activities, Collection<DailyScheduleSummary> daySummaries, boolean bSwapEnabled,
            boolean bShowUnavailable, boolean bShowTimeoff, boolean bShowNote) {
		DefaultMultiColumnNodeData rowData = new DefaultMultiColumnNodeData();
		StringBuffer sb = new StringBuffer();

        //leave Campaign column empty
        rowData.add("&nbsp;");

		// show date and NotPublished label in the Shift column
		sb.append(makeShiftDateString(aDay));
		sb.append("<br>");
		sb.append(this.i18n(m_bundle,FsWebBundleKey.SV_NOT_PUBLISHED));
		rowData.add(sb.toString(), CSSUtil.STYLE_NORMAL_LEFT); //508

		//leave Activities column empty
		//rowData.add("&nbsp;"); //QC#44831
        String right = makeEventsString(aDay, schedule, activities, bShowUnavailable, bShowTimeoff, bShowNote, true);
        rowData.add(right);

        //Activity campaign column should be blank for unpublished rows
        rowData.add(HtmlUtil.NBSP);

		return rowData;
	}//makeUnpublishedRow

	/**
	 * make date string for shifts column
	 */
	private String makeShiftDateString(CalendarRange aDay){

		return m_localizer.formatDate(aDay.getStartDate(),
				m_context.getViewingTimeZone(),
				RegionalFormatBundleKey.FULL_DATE_FORMAT);
	}//makeShiftDateString

	/**
	 * make html string for the shift column
	 */
	private StringsPair makeShiftColumnString(CalendarRange aDay, Collection<ShiftAssignment> shifts, HashMap<ID, Activity> activities,
			Collection<ShiftSwapPosting> postings, Collection<DailyScheduleSummary> daySummaries, boolean bSwapEnabled,
			boolean bShowTimeoff, Map<ID, String> campaignNamesBySpID) {
		StringBuffer sb = new StringBuffer();

		//--- make the date string ----
		sb.append(makeShiftDateString(aDay));
		sb.append("<br>");

		StringsPair results = makeShiftStringWithAccessibility(aDay, shifts, activities, postings, daySummaries,
				bSwapEnabled, true, m_myEmpID, bShowTimeoff, campaignNamesBySpID, isAccessibilityComplianceMode());
		
        String shiftString = results.getKey();
        String campaignName = results.getValue();

		//--- add Shift string ------
		sb.append(shiftString);

		//--- add Campaign string ------
		return new StringsPair(sb.toString(), campaignName);
	}//makeShiftColumnString

	@Override
	protected MultiColListPC getMultiColListPC(RequestContext context) {
		return new MultiColListPC(context) {
			protected HtmlDiv getNewHtmlDiv() {
				return new HtmlDivWSpan(getPCAriaLabelID(), i18n(m_bundle, FsWebBundleKey.SCHEDULE_GRID));
			}

			//list is disabled, but we allow to encloseForKeyboardAccess
			protected boolean skipEncloseForKeyboardAccess(BaseNodeData nodeData){
				return !this.isKeyboardAccessible() || !this.isRowSelectable() || !nodeData.isSelectable();
			}
		};
	}
}//class PersonalTextSchedulePM
