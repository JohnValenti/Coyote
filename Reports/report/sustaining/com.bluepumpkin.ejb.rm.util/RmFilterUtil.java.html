package com.bluepumpkin.ejb.rm.util;

import java.util.Calendar;
import java.util.Date;
import java.util.List;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.rm.requests.commonrequestmanager.model.RequestFilter;

public class RmFilterUtil {

	private RmFilterUtil() {

	}

	public static int dateToTime(Date date, int durInSecs, TimeZone tz) {

		Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
		cal.setTime(date);
		cal.setTimeZone(tz);

		cal.add(Calendar.SECOND, durInSecs);

		int retVal = cal.get(Calendar.HOUR_OF_DAY) * 3600;
		retVal += cal.get(Calendar.MINUTE) * 60;
		retVal += cal.get(Calendar.SECOND);

		return retVal;
	}

	public static StringsPair getStartAndEndDates(RequestFilter filter, String key) {
		StringsPair pair = null;

		@SuppressWarnings("unchecked")
		List<Date> dateRange = (List<Date>) filter.getValueForKey(key);
		Date start = null;
		Date end = null;
		if (dateRange != null) {
			start = dateRange.get(0);
			end = dateRange.get(1);
		}
		if (start == null) {
			start = new Date();
		}
		if (end == null) {
			end = new Date();
		}
		String rangeStart = JdmoUtil.formatDBString(start);
		String rangeEnd = JdmoUtil.formatDBString(end);

		pair = new StringsPair(rangeStart, rangeEnd);
		return pair;
	}

}
