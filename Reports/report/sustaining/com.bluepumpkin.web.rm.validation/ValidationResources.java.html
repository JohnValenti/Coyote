package com.bluepumpkin.web.rm.validation;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.witness.web.uif.keys.BaseStaticFileID;

/**
 * Title:        ValidationResources
 * Description:  Holds Validation Resources
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 *
 * @author David Su, dsu
 * @version 1.0
 *          <p/>
 *          Created on Mar 10, 2004, 10:04:23 PM
 */
public class ValidationResources {
	private String m_iconRBKey = null;
	private String m_msgRBKey = null;
	private String m_msgJSKey = null;
	private String m_errorMsgRBKey = null;

	public ValidationResources(String iconRBKey, String msgRBKey, String errorMsgRBKey, String msgJSKey) {
		m_iconRBKey = iconRBKey;
		m_msgRBKey = msgRBKey;
		m_errorMsgRBKey = errorMsgRBKey;
		m_msgJSKey = msgJSKey;
	}

	public ValidationResources(String iconRBKey, String msgRBKey, String errorMsgRBKey) {
		 this( iconRBKey,  msgRBKey,  errorMsgRBKey,msgRBKey);
	}

	public String getIconPath(Localizer localizer) {
		return BaseStaticFileID.createStaticFileURL(localizer.i18n(RmWebBundleKey.BUNDLE_NAME, m_iconRBKey));
	}

	public String getMessage(Localizer localizer) {
		return localizer.i18n(RmWebBundleKey.BUNDLE_NAME, m_msgRBKey);
	}

	public String getJSMessage(Localizer localizer) {
		return localizer.i18n(RmWebBundleKey.BUNDLE_NAME, m_msgJSKey);
	}

	public String getMessage(Localizer localizer, Object... args) {
		return localizer.i18n(RmWebBundleKey.BUNDLE_NAME, m_msgRBKey, args);
	}

	public String getErrorMessage(Localizer localizer) {
		return localizer.i18n(RmWebBundleKey.BUNDLE_NAME, m_errorMsgRBKey);
	}

	@Override
	public String toString() {
		return m_iconRBKey + "; " + m_msgRBKey+ "; " + m_msgJSKey  + "; " + m_errorMsgRBKey;
	}

	public String toString(Localizer localzier) {
		return getIconPath(localzier) + "; " + getMessage(localzier) + "; " + getJSMessage(localzier) + "; " + getErrorMessage(localzier);
	}
}
