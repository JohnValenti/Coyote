public class ShiftBidSummaryData extends RequestSummaryData {
	private int m_auctions;

	public ShiftBidSummaryData(
			int auctions,
			int requestsTotal,
			int requestsExpired,
			int requestsPending,
			int requestsApproved,
			int requestsDenied,
			int requestsInvalid) {
		super(requestsTotal, requestsExpired, requestsPending, requestsApproved,
				requestsDenied,	requestsInvalid);
		m_auctions = auctions;
	}

	public int getAuctions() {
		return m_auctions;
	}
}
