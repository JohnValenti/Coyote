package com.bluepumpkin.ejb.rm.scheduler.jobs;

import org.quartz.StatefulJob;
import org.quartz.JobExecutionContext;
import org.quartz.JobExecutionException;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.Lock;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidauction.ejb.ShiftBidAuctionManager;

/**
 * Title:        Blue Pumpkin Software Request Management Module
 * Copyright:    Copyright (c) 2001-2004
 * Company:      Blue Pumpkin Software, inc
 * @author       Raja Rajendran
 */
public class SerializedAuctionPollerJob implements StatefulJob {
    private static Category m_cat = Log.initCategory(SerializedAuctionPollerJob.class.getName());

    /**
     * lock to ensure that this job is not reexecuted while an earlier instance is still executing. 
     */    
    static Lock m_jobExecLock = new Lock();

    /*
     * @see org.quartz.Job#execute(org.quartz.JobExecutionContext)
     */
    public void execute(JobExecutionContext execCtxt) throws JobExecutionException {
		String methodName = "execute";
		
        try {
            // try to obtain the lock.
            if ( !m_jobExecLock.lock() ) {
                m_cat.debug("Quartz job not executed as an earlier job is still executing");
                return;
            };

            ShiftBidAuctionManager aucMgr = RequestUtil.getShiftBidAuctionMgr(null, null);   
			// continue to perform other polled tasks if an exception is thrown.                
			try {
				aucMgr.pollSerializedAuctions();
			} catch (Exception e) {
				m_cat.error("**** Exception caught and ignored ****: ", e);
			}

            // release lock
            m_jobExecLock.releaseLock();
        } catch (Exception e) {
            // release lock if it is being held.
            if (m_jobExecLock.isLocked()) {
                m_jobExecLock.releaseLock();
            }
    
            m_cat.error(methodName, e);

            throw new JobExecutionException(e, false);
        } 
    }
}
