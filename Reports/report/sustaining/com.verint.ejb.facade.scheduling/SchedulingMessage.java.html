package com.verint.ejb.facade.scheduling;

import javax.jms.ObjectMessage;
import javax.jms.Queue;
import javax.jms.QueueConnection;
import javax.jms.QueueConnectionFactory;
import javax.jms.QueueSender;
import javax.jms.QueueSession;
import javax.jms.Session;
import javax.naming.Context;
import javax.naming.InitialContext;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.JMSMessageBase;
import com.bluepumpkin.ejb.facade.JNDINames;

public class SchedulingMessage extends JMSMessageBase {

	/**
	 * 
	 */
	private static final long serialVersionUID = -8705797999778744125L;
	private final ID spID;
	private final String wfoUser;
	private final String wfoPassword;
	private final boolean isWhatIfMode;
	private final int schedAction;
	private final String[] schedParams;
	private final boolean isEnableLQF;
	private final boolean isRecalcSubCampaigns;

	public SchedulingMessage(ID spID, String wfoUser, String wfoPasswd,
			boolean isWhatIfMode, int action, String[] schedParams,
			boolean recalcSubCampaigns, boolean enableLQF) {
		this.spID = spID;
		this.wfoUser = wfoUser;
		this.wfoPassword = wfoPasswd;
		this.isWhatIfMode = isWhatIfMode;
		this.schedAction = action;
		this.schedParams = schedParams;
		this.isEnableLQF = enableLQF;
		this.isRecalcSubCampaigns = recalcSubCampaigns;
	}

	public void sendMessage() {
		try {
			Context ctx = new InitialContext();
			QueueConnectionFactory queueConnectionFactory = (QueueConnectionFactory) ctx
					.lookup(JNDINames.FACADE_JMS_QUEUE_FACTORY);
			QueueConnection queueConnection = queueConnectionFactory
					.createQueueConnection();
			QueueSession queueSession = queueConnection.createQueueSession(
					false, Session.AUTO_ACKNOWLEDGE);
			Queue queue = (Queue) ctx.lookup(JNDINames.FACADE_SCHEDULING_QUEUE);
			QueueSender queueSender = queueSession.createSender(queue);
			queueConnection.start();
			ObjectMessage objectMessage = queueSession.createObjectMessage();
			objectMessage.setObject(this);
			queueSender.send(objectMessage);
			queueSender.close();
			queueSession.close();
			queueConnection.close();
		} catch (Exception e) {
			//TODO
			e.printStackTrace();
			//throw new BbmUpdateException(e);
		}
	}

	public ID getSPID() {
		return spID;
	}

	/**
	 * @return the wfoUser
	 */
	public String getWfoUser() {
		return wfoUser;
	}

	/**
	 * @return the wfoPassword
	 */
	public String getWfoPassword() {
		return wfoPassword;
	}

	/**
	 * @return the isWhatIfMode
	 */
	public boolean isWhatIfMode() {
		return isWhatIfMode;
	}

	/**
	 * @return the schedAction
	 */
	public int getSchedulingAction() {
		return schedAction;
	}

	/**
	 * @return the schedParams
	 */
	public String[] getSchedulingParameters() {
		return schedParams;
	}

	public boolean isEnableLQF() {
		return isEnableLQF;
	}

	public boolean isRecalcSubCampaign() {
		return isRecalcSubCampaigns;
	}

}
