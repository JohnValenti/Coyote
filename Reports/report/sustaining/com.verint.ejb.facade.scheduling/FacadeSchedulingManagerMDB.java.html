package com.verint.ejb.facade.scheduling;

import javax.ejb.CreateException;
import javax.ejb.EJBException;
import javax.ejb.MessageDrivenBean;
import javax.ejb.MessageDrivenContext;
import javax.ejb.TimedObject;
import javax.ejb.Timer;
import javax.ejb.TimerService;
import javax.jms.Message;
import javax.jms.MessageListener;
import javax.jms.ObjectMessage;

import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.facade.FacadeManagerFactory;
import com.bluepumpkin.ejb.facade.l10n.FacadeEjbLogBundleKey;
import com.verint.ejb.bbm.schedule.model.WSAutoScheduleRequest;
import com.verint.ejb.bbm.schedule.model.WSAutoScheduleRequestDAO;

public class FacadeSchedulingManagerMDB implements MessageDrivenBean, MessageListener, TimedObject {

	private static final long serialVersionUID = 6672055979033740487L;

	MessageDrivenContext messageDrivenContext;
	TimerService timerService;
	ScheduleMessageScheduler scheduler;

	private static Category m_cat = Log.initCategory(FacadeSchedulingManagerMDB.class.getName());

	public void ejbCreate() throws CreateException {
		try {
			scheduler = FacadeManagerFactory.getScheduleMessageScheduler();
		} catch (Exception e) {
			Object[] param = new Object[] { ScheduleMessageScheduler.class.getName() };
			m_cat.l7dError(FacadeEjbLogBundleKey.EJB_CREATE_FAILURE, param, e);
			throw new CreateException();
		}
	}

	public void ejbRemove() throws EJBException {
		// Auto-generated method stub

	}

	public void setMessageDrivenContext(MessageDrivenContext messageDrivenContext) throws EJBException {
		this.messageDrivenContext = messageDrivenContext;
		this.timerService = messageDrivenContext.getTimerService();
	}

	/**
	 * Processes a ScheduleRequestsMessage .
	 * 
	 * ScheduleRequestsMessage contain a list of Scheduling Period IDs for which
	 * this class will persist in a table to schedule calculations.
	 */
	public void onMessage(Message msg) {
		try {
			if (msg instanceof ObjectMessage) {
				Object obj = ((ObjectMessage) msg).getObject();
				m_cat.debug(obj);
				if (obj instanceof ScheduleRequestsMessage) {
					ScheduleRequestsMessage message = (ScheduleRequestsMessage) obj;
					// persist the message, the job will schedule them
					createMessage(message);
				}
			} else {
				// If this is something other,
				// display a warning
				m_cat.warn("The message type " + msg.getClass().getName() + " is not a javax.jms.ObjectMessage");
			}
		} catch (Exception e) {
			Object[] param = new Object[] { "onMessage", FacadeSchedulingManagerMDB.class.getName() };
			m_cat.l7dError(FacadeEjbLogBundleKey.UNKNOWN_EXCEPTION_MF, param, e);
		}
	}

	/*
	 * (non-Javadoc)
	 * 
	 * @see javax.ejb.TimedObject#ejbTimeout(javax.ejb.Timer)
	 */
	@Override
	public void ejbTimeout(Timer timer) {
		// do nothing
	}
	
	
	/**
	 * Creates the message. If the message could not be created, the requests
	 * attached to the message will be marked with a status of error.
	 * 
	 * @param message
	 * @throws JdmoException
	 */
	private void createMessage(ScheduleRequestsMessage message) throws JdmoException {
		try {
			scheduler.createMessage(message);
		} catch (Exception any) {
			// if there is an error creating anything
			// we have to mark all the requests as error
			m_cat.error("Could not persist schedule message", any);
			// update the requests to error
			updateRequestsForFailedCreation(message);
		}
	}

	/**
	 * Updates the requests as failed as the jms message to process these
	 * requests could not be created.
	 */
	private void updateRequestsForFailedCreation(ScheduleRequestsMessage message) throws JdmoException {
		WSAutoScheduleRequestDAO dao = new WSAutoScheduleRequestDAO();
		try {
			dao.updateScheduleStatus(message.getRequestsToSchedule(), WSAutoScheduleRequest.SCHEDULE_STATUS_NEW,
					WSAutoScheduleRequest.SCHEDULE_STATUS_ERROR,
					"Could not schedule as the scheduling message could not be persisted");
		} finally {
			dao.cleanUp();
		}
	}
}
