package com.verint.ejb.facade.scheduling;

import java.io.Serializable;
import java.util.HashMap;
import java.util.Map;

public class ScheduleRequestParameters implements Serializable, Cloneable{

	/**
	 * 
	 */
	private static final long serialVersionUID = 6764550785178246011L;
	private final boolean m_isWhatIfMode;
	private final int m_schedAction;
	private final String[] m_schedParams;
	private final boolean m_isEnableLQF;
	private final boolean m_isRecalcSubCampaigns;
	private int m_retryAttempt;
	private final boolean m_overrideSchedules;
	private final boolean m_scheduleInOrder;
	private String m_userSeal = null;
	private Map<String, Object> m_secondarySchedParams = null;
	
	public ScheduleRequestParameters(boolean isWhatIfMode,
	int action, String[] schedParams, boolean recalcSubCampaigns,
	boolean enableLQF, boolean overrideSchedules, int retryAttempt, boolean scheduleInOrder)
	{
		this.m_isWhatIfMode = isWhatIfMode;
		this.m_schedAction = action;
		this.m_schedParams = schedParams;
		this.m_isEnableLQF = enableLQF;
		this.m_isRecalcSubCampaigns = recalcSubCampaigns;
		this.m_overrideSchedules = overrideSchedules;
		this.m_retryAttempt = retryAttempt;
		this.m_scheduleInOrder = scheduleInOrder;
	}

	public boolean isWhatIfMode() {
		return m_isWhatIfMode;
	}

	public int getSchedAction() {
		return m_schedAction;
	}

	public String[] getSchedParams() {
		return m_schedParams;
	}
	
	/**
	 * Returns the Map of Secondary Scheduling parameters. <code>null</code> by default.
	 * @return
	 */
	public Map<String, Object> getSecondarySchedParams()
	{
		return m_secondarySchedParams;
	}
	
	public void setSecondarySchedParams(Map<String, Object> secondaryParameters)
	{
		this.m_secondarySchedParams = secondaryParameters;
	}

	public boolean isEnableLQF() {
		return m_isEnableLQF;
	}

	public boolean isRecalcSubCampaigns() {
		return m_isRecalcSubCampaigns;
	}

	public int getRetryAttempt() {
		return m_retryAttempt;
	}

	public boolean isOverrideSchedules() {
		return m_overrideSchedules;
	}

	
	public void setRetryAttempt(int retryAttempt)
	{
		this.m_retryAttempt = retryAttempt;
	}
	
	public boolean isScheduleInOrder(){
		return this.m_scheduleInOrder;
	}
	
	public String getUserSeal() {
		return m_userSeal;
	}

	public void setUserSeal(String userSeal) {
		this.m_userSeal = userSeal;
	}

	@Override
	public Object clone() throws CloneNotSupportedException {
		ScheduleRequestParameters clone = (ScheduleRequestParameters)super.clone();
		//not necessary if using HashMap...but doing since Map is not cloneable
		if (m_secondarySchedParams != null) {
			clone.m_secondarySchedParams = new HashMap<String, Object>();
			clone.m_secondarySchedParams.putAll(m_secondarySchedParams);
		}
		return clone;
	}
}
