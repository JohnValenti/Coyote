package com.verint.web.fs.skills;

import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.web.core.pagecomponent.time.TimePeriodSelectorPC;
import com.witness.web.uif.keys.SessionKeys;
import com.witness.web.uif.system.RequestContext;

import java.util.Date;


/**
 * Title:        EmployeeAssignmentPMUtil
 * Description:  Provide convenient methods for employee management module
 */
public class EmployeeAssignmentPMUtil {
	/**
	 * To get/set cached date range for the user
	 *
	 * @param timePeriod The TimePeriodSelectorPC Object, to get the start/end dates in selected date range
	 * @param context    The Request Context Object, used to set session attributes
	 */
	public static void storeSelectedDateRangeInSession(TimePeriodSelectorPC timePeriod, RequestContext context) {
		LocalDate first = timePeriod.getSelectedDateRange().getFirst();
		LocalDate second = timePeriod.getSelectedDateRange().getSecond();
		Date current = new Date();
		if (context.getAttribute(RequestContext.SESSION_SCOPE, SessionKeys.START_DATE) == null) {
			if (!(current.compareTo(first.getTime()) >= 0 && current.compareTo(second.getTime()) <= 0)) {
				context.setAttribute(RequestContext.SESSION_SCOPE, SessionKeys.START_DATE, first);
				context.setAttribute(RequestContext.SESSION_SCOPE, SessionKeys.END_DATE, second);
			}
		} else if (context.getAttribute(RequestContext.SESSION_SCOPE, SessionKeys.START_DATE) != first) {
			context.setAttribute(RequestContext.SESSION_SCOPE, SessionKeys.START_DATE, first);
			context.setAttribute(RequestContext.SESSION_SCOPE, SessionKeys.END_DATE, second);
		}
	}
}
