package com.bluepumpkin.ejb.rm.requests.custshift.validation;

import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.shifts.model.ShiftsUtil;
import com.bluepumpkin.ejb.bbm.timeoffcalculator.TOCalcUtil;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;

/**
 * <li> Rule from UI: Both agents comply with Max consecutive working days org setting.
 * <p/>
 * For range with shift being given up and for the range with shift being received,
 * verify constraints, taking into account the shift being given up
 * and the shift being received. The range is determined as follows:
 * Assume sd is the swap day (i.e., the day being swapped), and n is the org setting
 * for max consecutive working days. Then the range that we need to check is: (sd-n, sd+n).
 * <p/>
 * <p/>
 * TODO: verify ontraint for week with shift being given up for each agent.  Presently validation
 * only for week with shift being received for each agent.
 * <p/>
 * <p/>
 * For one way swaps:<br>
 * for employee giving up shift, for week shift given up, verify constraint <br>
 * for employee giving up timeoff, for week shift being received, verify constraint. <br>
 * <p/>
 * <p/>
 * For tentative approvals: processing logic same as non tentative requests as
 * it uses only the published schedule for calculation of hours
 * <p/>
 * <p> Algorithm:
 * <p/>
 * <li> Validation rule to ensure that each employee, participating in the shift
 * swap, does not violate the "Max consecutive working days" org setting.
 * <ul>
 * <li>for the first employee find the start of the range which contains the shift
 * (or day off) this employee will get through the swap.
 * <li> get the schedule information for this employee for this range
 * <li> calculate the work hours for this range plus the shift hours gained
 * <li> compare the range work days with employee Max consecutive working days setting.
 * <li> for the second employee, find the start of the range which contains the shift
 * being given up.
 * <li> get the schedule information for this employee for this range.
 * <li> calculate the work hours for this range minus the shift hours lost
 * <li> compare the range work days with employee Max consecutive working days setting.
 * <li> repeat for each employee
 * <li> validation result in case of failure should have information about the
 * employee, number of work days, and Max consecutive days limit.
 * <ul>
 */
public class MaxConsecutiveWorkingDaysRule extends OrgWeekValidationRule {
	public static String m_className = MaxConsecutiveWorkingDaysRule.class.getName();
	public static Category m_cat = Log.initCategory(m_className);

	public Date extendStartDays(Date startDate, Organization org) {
		Date tempStartDate = new Date(startDate.getTime());
		//For MaxConsecutiveWorkingDaysRule only
		//This fix is for SP crossing validation
		for (int i = 1; i <= org.getMaxConsecutiveWorkingDay(); i++) {
			tempStartDate = TOCalcUtil.getDateBackwardOneDay(tempStartDate, org.getTimeZone());
		}
		return tempStartDate;
	}

	public Date extendEndDays(Date endDate, Organization org) {
		Date tempEndDate = new Date(endDate.getTime());
		//For MaxConsecutiveWorkingDaysRule only
		//This fix is for SP crossing validation
		for (int i = 1; i <= org.getMaxConsecutiveWorkingDay(); i++) {
			tempEndDate = TOCalcUtil.getDateForwardOneDay(tempEndDate, org.getTimeZone());
		}
		return tempEndDate;
	}

	protected ValidationResult checkRuleHours(ID empID, Date orgWeekStart, Date orgWeekEnd,
	                                          Collection shiftsDuringOrgWeek, Organization org, Validatable validatable) throws Exception {
		// now check the Max consecutive days rule for all shifts in the collection.
		Collection results = ShiftsUtil.getMaxConsecutiveDaysConflicts(shiftsDuringOrgWeek, orgWeekStart, orgWeekEnd, org);
		return RequestUtil.setValidationResult(empID,m_vc.getEmployeeNameByID(empID), validatable, results, m_className);
	}

}
