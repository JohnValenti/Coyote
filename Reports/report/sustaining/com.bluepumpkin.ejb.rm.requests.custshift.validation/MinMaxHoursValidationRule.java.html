/**
 * <li> Rule from UI: Both agents comply with min/max hours
 * <p/>
 * For week with shift being given up and for the week with shift being received,
 * verify min/max hour constraints, taking into account the shift being given up
 * and the shift being received.
 * <p/>
 * <p/>
 * TODO: verify min/max contraint for week with shift being given up.  Presently validation
 * only for week with shift being received.
 * <p/>
 * <p/>
 * For one way swaps:<br>
 * for employee giving up shift, for week shift given up, verify min/max constraint <br>
 * for employee giving up timeoff, for week shift being received, verify min/max constraint. <br>
 * <p/>
 * <p/>
 * For tentative approvals: processing logic same as non tentative requests as
 * it uses only the published schedule for calculation of min/max hours
 * <p/>
 * <p> Algorithm:
 * <p/>
 * <li> Validation rule to ensure that each employee, participating in the shift
 * swap, does not violate the min/max hours per week allocated to him/her.
 * <ul>
 * <li>for the first employee find the start of the week which contains the shift
 * (or day off) this employee will get through the swap.
 * <li> get the schedule information for this employee for this week
 * <li> calculate the work hours for this week plus the shift hours gained
 * <li> compare the week work hours with employee min/max hours per week
 * <li> for the first employee, find the start of the week which contains the shift
 * being given up.
 * <li> get the schedule information for this employee for this week.
 * <li> calculate the work hours for this week minus the shift hours lost
 * <li> compare the week work hours with employee min/max hours.
 * <li> repeat for each employee
 * <li> validation result in case of failure should have information about the
 * employee, work hours, min/max limits
 * <ul>
 */
public class MinMaxHoursValidationRule extends OrgWeekValidationRule {

	public MinMaxHoursValidationRule() {
		m_className = MinMaxHoursValidationRule.class.getName();
		m_cat = Log.initCategory(m_className);
	}

	protected ValidationResult checkRuleHours(ID empID, Date orgWeekStart, Date orgWeekEnd,
	                                          Collection shiftsDuringOrgWeek, Organization org, Validatable validatable) throws Exception {
		WorkResourceMinMaxHour minMaxAssn = null;
		try {
			//get min max hours assignments
			Collection minMaxAssns = m_ewrm.getMinMaxHourAssignments(empID, orgWeekStart, orgWeekEnd);
			if (minMaxAssns != null) minMaxAssn = RequestUtil.getMinMaxHourAssnOverlappingTimeRange(minMaxAssns, orgWeekStart, orgWeekEnd);
		} catch (BbmFinderException ex) {
			m_cat.warn("Exception getting minmaxhours for employeeid: " + empID + ". Using zero as default for validation.");
			minMaxAssn = new WorkResourceMinMaxHour();
			minMaxAssn.setMinMinutes(0);
			minMaxAssn.setMaxMinutes(0);
		}
		// now check the min/max hours rule for all shifts in the collection.
		try {
			if (minMaxAssn == null) {
				return ValidationUtil.setSoftValidationResult(validatable, RmEjbBundleKey.CS_MINMAX_HOURS_NOT_AVAILABLE, m_className);
			}
			Collection results = ShiftsUtil.getMinMaxHourConflicts(shiftsDuringOrgWeek, minMaxAssn, orgWeekStart, orgWeekEnd, org, null);
			return RequestUtil.setValidationResult(empID, m_vc.getEmployeeNameByID(empID), validatable, results, m_className);
		} catch (Exception ex) {
			m_cat.error(ex);
			throw ex;
		}
	}

}
