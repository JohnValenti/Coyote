/**
 * @author rrajendran
 *         <p/>
 *         Soft validation rule: Agent's schedule complies with organization hours of operation.
 */
public class CustShiftRequestWithinHOO implements Validator {
	private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

	private  static  final String CLASSNAME = CustShiftRequestWithinHOO.class.getName();

	private  static final Category LOG = Log.initCategory(CLASSNAME);

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		LOG.log(m_tracePriority, "Validation Rule: " + CLASSNAME + " started");

		try {
			CustShiftReq csr = (CustShiftReq) validatable;
			CustShiftValidationCache cache = (CustShiftValidationCache) csr.getValidationCache();
			// get the HOO time map
			TimeMap orgHooTimeMap = cache.getHOOAssignmentsTimeMap();
			ScheduleAccessManager sam = cache.getScheduleAccessManager();
			// obtain shift assignments associated with the csr.
			ShiftAssignment shiftAssnToCreate = cache.getNewShiftAssignToBeCreated(csr, sam);
			if (shiftAssnToCreate == null) {
				//Unable to create shift between {0} and {1} for employee {2} for the specified custom shift request.
				return ValidationUtil.setAndLogHardValidationResult(validatable,
				        RmEjbBundleKey.CS_CANNOT_CREATE_SHIFT,
				        RmEjbLogBundleKey.CS_CANNOT_CREATE_SHIFT,
				        new Serializable[]{csr.getStartTime(), csr.getEndTime(), cache.getEmployeeNameByID(csr.getEmployeeID())},
				        CLASSNAME);
			}
			// verify if shifts fall within HOO
			Event csrEvent = new TestEvent(csr.getStartTime(), csr.getEndTime());
			Event shift = RequestUtil.verifyOffBitsOverlapEvents(orgHooTimeMap, Collections.singletonList(csrEvent));
			if (shift != null) {
				return ValidationUtil.setSoftValidationResult(csr, RmEjbBundleKey.CS_NOT_WITHIN_HOO, shift.getStartTime(), shift.getEndTime(), CLASSNAME);
			}
			return null;
		} finally {
			LOG.log(m_tracePriority, "Validation Rule: " + CLASSNAME + " done");
		}
	}

}
