package com.bluepumpkin.ejb.rm.requests.custshift.validation;

import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.activity.model.Event;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.custshift.model.CustShiftReq;


//calendar events like unavailability
//check for calendar events, which can not overlap with a cust-shift employee is requesting.

/**
 * <p> Rule from UI: Request overlaps with  time off or unavailability..
 * <p/>
 * <p/>
 * Equally applies for tentative and non tentative requests as only the un published schedule is used.
 * <p/>
 * <p> calendar event assignments in the system are stored in the CalendarEventAssignment DB table.
 * The eventType column defines the type of Calendar Event:
 * <li> TimeOff: Represented by {@link com.bluepumpkin.ejb.bbm.activity.model.Event#EVENT_TYPE_TIME_OFF EVENT_TYPE_TIME_OFF}
 * <li> Unavailability: Represented by {@link com.bluepumpkin.ejb.bbm.activity.model.Event#EVENT_TYPE_UNAVAILABILITY EVENT_TYPE_UNAVAILABILITY }
 * <p/>
 * <p> Calendar event assignments also have an overlap attribute:
 * <li> {@link com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventTemplate#OVERLAP_TYPE_MUST_OVERLAP_SHIFT OVERLAP_TYPE_MUST_OVERLAP_SHIFT}
 * <li> {@link com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventTemplate#OVERLAP_TYPE_MAY_OVERLAP_SHIFT OVERLAP_TYPE_MAY_OVERLAP_SHIFT}
 */
public class TimeOffOverlapValidationRule implements Validator {
	private static final String m_className = TimeOffOverlapValidationRule.class.getName();

	// category for logging.
	private static final Category m_cat = Log.initCategory(m_className);

	public TimeOffOverlapValidationRule() {
	}

	public ValidationResult validate(Validatable validatable) throws Exception {
		CustShiftReq csr = (CustShiftReq) validatable;
		ID empID = csr.getEmployeeID();
		//always get time-off as well as unavailability because the collection is cached &
		// we need UNAVAILABILITY for the UNAVAILABILITY validation rule
		Collection pubOverlappingEvents = csr.getCache().getEventsDuringPeriodFromDB(csr, null);
		if (pubOverlappingEvents == null || pubOverlappingEvents.isEmpty())
			return null;
		for (Iterator it = pubOverlappingEvents.iterator(); it.hasNext();) {
			// CalendarEventAssignment is the base class for TOEvents and CalendarEvents.
			Event cal = (Event) it.next();
			// process only unavailability events and time off events.
			if (cal == null) continue;
			int eventType = cal.getEventType();
			if (eventType != Event.EVENT_TYPE_TIME_OFF) {
				continue;
			}
			if (csr.getStartTime().before(cal.getEndTime()) && csr.getEndTime().after(cal.getStartTime())) {
				return ValidationUtil.setSoftValidationResult(csr, RmEjbBundleKey.CS_TIMEOFF_OVERLAP, m_className);
			}
		}
		return null;
	}
}
