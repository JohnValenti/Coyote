package com.verint.web.rfs.reports.selection;

import java.util.List;

import javax.servlet.http.HttpServletRequest;

import org.w3c.dom.Node;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.core.reports.ReportControlsXml;
import com.bluepumpkin.web.core.reports.ReportParameterUtil;
import com.bluepumpkin.web.core.reports.ReportParametersPCInterface;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.bluepumpkin.web.ireport.framework.RffParameter;
import com.bluepumpkin.web.ireport.framework.RffRequest;
import com.verint.ejb.rfs.reports.model.RfsStaffMixFilter;
import com.verint.web.rfs.reports.util.RfsReportControlsXml;
import com.verint.web.rfs.reports.util.RfsReportParameterUtil;
import com.witness.web.uif.pagecomponent.form.FormInputPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

public class RfsReportParameterPlannerFilterPC extends FormInputPC implements ReportParametersPCInterface {

	private static final long serialVersionUID = 1L;

	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();
	private static RfsReportControlsXml rfsReportControls = RfsReportControlsXml.getInstance();
	private static RfsReportParameterUtil rfsReportUtil = RfsReportParameterUtil.getInstance();
	private static ReportParameterUtil reportUtil = ReportParameterUtil.getInstance();

	private Localizer m_localizer = null;
	private String mPickerName = null;

	public RfsReportParameterPlannerFilterPC(RequestContext context, ToolbarPC toolbar, Node controlNode) {
		super(context);
		m_localizer = m_context.getLocalizer();
		setName(reportControls.getControlSingleParameterName(controlNode));
		setIsRequired(reportControls.isControlRequired(controlNode));
	}

	public String getLabel(RffRequest rff, Node controlNode) {
		String label = reportControls.getControlLabel(controlNode);
		mPickerName = rfsReportUtil.getControlPickerName(m_context, controlNode);
		if (label == null || label.length() == 0) {
			label = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, mPickerName);
		}
		return label;
	}

	public String getHtmlDisplay() {
		String currentValue = m_context.getRequest().getParameter("StaffMixFilter");
		List<StringsPair> options = rfsReportUtil.getOptions(m_context, mPickerName, currentValue);
		return HtmlUtil.makeSelectInput(mPickerName, new String[] { currentValue }, rfsReportUtil.getJavaScript(mPickerName), options,
				false, true);
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode) {
		return getHtmlDisplay(rff, controlNode, false);
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode, boolean readOnly) {

		Node paramNode = rfsReportControls.getControlStaffMixFilter(controlNode);
		if (paramNode == null) {
			throw new IllegalArgumentException("A parameter was not specified for this control.");
		}

		String paramName = reportControls.getParameterName(paramNode);

		RffParameter param = rff.getRffParameter(paramName);
		String currentValue = null;
		currentValue = m_context.getRequest().getParameter(paramName);
		if (currentValue == null && param != null && param.getCurrentValue() != null) {
			currentValue = param.getCurrentValue().toString();
		}
		if (mPickerName == null) {
			mPickerName = rfsReportUtil.getControlPickerName(m_context, controlNode);
		}

		List<StringsPair> options = rfsReportUtil.getOptions(m_context, paramNode, currentValue, mPickerName, rff.getParentId());

		return HtmlUtil.makeSelectInput(paramName, new String[] { currentValue }, rfsReportUtil.getJavaScript(mPickerName), options,
				false, !readOnly);
	}

	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode) {
		RffParameter lowerHeadCountsParam = rff.getRffParameter("LowerHeadCounts");
		RffParameter upperHeadCountsParam = rff.getRffParameter("UpperHeadCounts");
		RffParameter lowerPercentHeadCountsParam = rff.getRffParameter("LowerPercentHeadCounts");
		RffParameter upperPercentHeadCountsParam = rff.getRffParameter("UpperPercentHeadCounts");
		RffParameter lowerPercentFTEsParam = rff.getRffParameter("LowerPercentFTEs");
		RffParameter upperPercentFTEsParam = rff.getRffParameter("UpperPercentFTEs");
		RffParameter lessActualHeadCountsParam = rff.getRffParameter("LessActualHeadCounts");
		RffParameter moreActualHeadCountsParam = rff.getRffParameter("MoreActualHeadCounts");
		RffParameter lessBudgetedHeadCountsParam = rff.getRffParameter("LessBudgetedHeadCounts");
		RffParameter moreBudgetedHeadCountsParam = rff.getRffParameter("MoreBudgetedHeadCounts");
		RffParameter upperPercentOverstaffedParam = rff.getRffParameter("UpperPercentOverstaffed");
		RffParameter upperCountStaffMixCostParam = rff.getRffParameter("UpperCountStaffMixCost");
		RffParameter closestToActualParam = rff.getRffParameter("ClosestToActual");
		RffParameter closestToBudgetedParam = rff.getRffParameter("ClosestToBudgeted");
		RffParameter closestToTargetedParam = rff.getRffParameter("ClosestToTargeted");
		RffParameter targetPercentFTEsParam = rff.getRffParameter("TargetPercentFTEs");

		RfsStaffMixFilter staffMixFilter = rfsReportUtil.getStaffMixFilter(m_context);
		lowerHeadCountsParam.setCurrentValue(staffMixFilter.getMinHeadCounts());
		upperHeadCountsParam.setCurrentValue(staffMixFilter.getMaxHeadCounts());
		lowerPercentHeadCountsParam.setCurrentValue(staffMixFilter.getMinPercentHeadCounts());
		upperPercentHeadCountsParam.setCurrentValue(staffMixFilter.getMaxPercentHeadCounts());
		lowerPercentFTEsParam.setCurrentValue(staffMixFilter.getMinPercentFte());
		upperPercentFTEsParam.setCurrentValue(staffMixFilter.getMaxPercentFte());
		lessActualHeadCountsParam.setCurrentValue(staffMixFilter.getHeadCountsBelowActual());
		moreActualHeadCountsParam.setCurrentValue(staffMixFilter.getHeadCountsAboveActual());
		lessBudgetedHeadCountsParam.setCurrentValue(staffMixFilter.getHeadCountsBelowBudgeted());
		moreBudgetedHeadCountsParam.setCurrentValue(staffMixFilter.getHeadCountsAboveBudgeted());
		upperPercentOverstaffedParam.setCurrentValue(staffMixFilter.getMaxOverstaffingRatio());
		upperCountStaffMixCostParam.setCurrentValue(staffMixFilter.getMaxStaffMixesAfterLowCost());
		closestToActualParam.setCurrentValue(staffMixFilter.getIncludeClosestToActual());
		closestToBudgetedParam.setCurrentValue(staffMixFilter.getIncludeClosestToBudgeted());
		closestToTargetedParam.setCurrentValue(staffMixFilter.getIncludeClosestToFte());
		targetPercentFTEsParam.setCurrentValue(staffMixFilter.getCustomPercentFte());
		mPickerName = rfsReportUtil.getControlPickerName(m_context, controlNode);
		Node staffMixFilterNode = rfsReportControls.getControlStaffMixFilter(controlNode);
		String paramName = reportControls.getParameterName(staffMixFilterNode);
		RffParameter param = rff.getRffParameter(paramName);
		reportUtil.setRffParameterValue(request, param, "STRING");
		String displayParamName = reportControls.getParameterDisplayName(staffMixFilterNode);
		RffParameter displayParam = rff.getRffParameter(displayParamName);

		if (displayParam != null) {
			String displayValue = getOptionValue(rff, staffMixFilterNode, mPickerName, (String) param.getCurrentValue());
			if (displayValue == null) {
				displayParam.setCurrentValue(param.getCurrentValue());
			} else {
				displayParam.setCurrentValue(displayValue);
			}
		}

	}

	private String getOptionValue(RffRequest rff, Node paramNode, String pickername, String currentKey) {
		List<StringsPair> options = rfsReportUtil.getOptions(m_context, paramNode, "", pickername, rff.getParentId());
		if (options == null || options.isEmpty()) {
			return "";
		}
		return reportUtil.getOptionValue(options, currentKey);
	}

	public String getCustomJavaScript(Node controlNode) {
		return null;
	}

	public boolean extractParameters(HttpServletRequest request) {
		super.extractParameters(request);
		return true;
	}
}
