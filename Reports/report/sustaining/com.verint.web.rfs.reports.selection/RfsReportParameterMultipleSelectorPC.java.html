package com.verint.web.rfs.reports.selection;

	private static final long serialVersionUID = 1L;

	private static final String DEFAULT_NOT_SET_VALUE = ReportParametersDesignPC.DEFAULT_NOT_SET_VALUE;
	
	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();
	private static RfsReportControlsXml rfsReportControls = RfsReportControlsXml.getInstance();
	private static RfsReportParameterUtil rfsReportUtil = RfsReportParameterUtil.getInstance();

	public RfsReportParameterMultipleSelectorPC(RequestContext context, ToolbarPC toolbar, Node controlNode) {
			super(context);
			m_localizer = m_context.getLocalizer();
			setName(reportControls.getControlSingleParameterName(controlNode));
			setIsRequired(reportControls.isControlRequired(controlNode));
		}
		
		public String getLabel(RffRequest rff, Node controlNode) {
			String label = reportControls.getControlLabel(controlNode);
			if (label == null || label.length() == 0) {
				label = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,rfsReportControls.getControlPickerName(controlNode));
			}
			return label;
		}
		
	public String getHtmlDisplay(RffRequest rff, Node controlNode) {
		return getHtmlDisplay(rff, controlNode, false);
	}
	
	public String getHtmlDisplay(RffRequest rff, Node controlNode,
				boolean readOnly) {
		if (reportControls.hasSingleParameter(controlNode) == false) {
			throw new IllegalArgumentException("Can only handle single report parameter.");
		}
		Node paramNode = reportControls.getParameter(controlNode);
		if (paramNode == null) {
			throw new IllegalArgumentException("A parameter was not specified for this control.");
		}

		String paramName = reportControls.getParameterName(paramNode);
		String optionType = rfsReportControls.getControlPickerName(controlNode);
		if (optionType == null || optionType.length() == 0) {
			optionType = paramName;
		}

		Object [] selected = m_context.getRequest().getParameterValues(paramName);
		
		String html = getHtmlDisplay(this, m_context, selected, paramNode, optionType, rff.getParentId(), readOnly);
		if (html == null) {
			return i18n(m_common_bundle, UIFWebBundleKey.NONE);
		}
		return html;
	}
	
	String getHtmlDisplay(PageComponent pc, RequestContext context, Object selected[], Node paramNode,
			String optionType, String parentId, boolean readOnly) {
		String paramName = reportControls.getParameterName(paramNode);
		if(RfsReportParameterConstants.RESOURCE_CLASS.equals(optionType)) {
			optionType = RfsReportParameterConstants.RESOURCE_CLASS_MULTIPLE;
		}
		List<StringsPair> options = rfsReportUtil.getOptions(context, paramNode, "",optionType, parentId);
		if (options == null || options.size() == 0) {
			return null;
		}
		int size = (selected != null) ? selected.length : 0;
		String[] selections = new String[size];
		for (int i = 0; i < size; i++) {
			selections[i] = selected[i].toString();
		}
		return HtmlUtil.makeSelectInput(paramName, "", selections, JSUtil.ON_CHANGE + JSUtil.REFRESH_PAGE_NO_CONFIRM,
								options, "", false, true, ! readOnly);
	}
	
	/**
	 * Assigns to the RFF request the values of the servlet request,
	 * these value will have been the user entered data. The RFF
	 * parameter will be those assigned thru the setRffParameter or the
	 * setRffRequest methods. Simply, extract the parameters from the
	 * HTTP request and put them in the report framework, for use by the
	 * reporting system.
	 * 
	 * @param request the HTTP request in which parameters are to be
	 *            retrieved.
	 * @param rff the report framework request which contains the RFF
	 *            parameters where the data is to be placed.
	 * @param controlNode the child with in the controls document which
	 *            contains the information regarding this PC component
	 *            (i.e. Report Control).
	 */
	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode) {
		if (reportControls.hasSingleParameter(controlNode) == false) {
			throw new IllegalArgumentException("Can only handle single report parameter.");
		}
		Node paramNode = reportControls.getParameter(controlNode);
		if (paramNode == null) {
			throw new IllegalArgumentException("A parameter was not specified for this control.");
		}
		String paramName = reportControls.getParameterName(paramNode);
		RffParameter param = rff.getRffParameter(paramName);
		if (param == null) {
			throw new IllegalArgumentException("Parameter in controls but not in report: " + paramName);
		}
		//String filtername = reportControls.getControlFilterMethod(controlNode);
		String filtername = rfsReportControls.getControlPickerName(controlNode);
		if (filtername == null || filtername.length() == 0) {
			filtername = paramName;
		}
		if(RfsReportParameterConstants.RESOURCE_CLASS.equals(filtername)) {
			filtername = RfsReportParameterConstants.RESOURCE_CLASS_MULTIPLE;
		}

		log.info("setRffParameterValue: " + paramName + ", " + request.getParameter(paramName));
		List<StringsPair> options = rfsReportUtil.getOptions(m_context, paramNode, "",filtername, rff.getParentId());

		String[] currentValues = request.getParameterValues(paramName);
		String[] selectedValues = null;
		
		if(currentValues != null && currentValues.length == 1 && currentValues[0].equals("-1")) {
			selectedValues = new String[options.size()-1];
			for(int i = 1; i< options.size(); i++) {
				selectedValues[i-1] = options.get(i).getKey();
			}
		}
		else {
			selectedValues = currentValues;
		}
		param.setCurrentValues(selectedValues);

		
		String displayParamName =  reportControls.getParameterDisplayName(paramNode);
		RffParameter displayParam = rff.getRffParameter(displayParamName);
		if (displayParam != null) {			
			if (selectedValues != null) {
				//String currentNames[] = new String[currentValues.length];
				String displayValue = null;
				for (int i = 0; i < selectedValues.length && i < 3; i++) {
					if(i == 0)						
						displayValue = getOptionValue(options, (String)selectedValues[i]);
					else
						displayValue = displayValue + "," + getOptionValue(options, (String)selectedValues[i]);
				}
				displayParam.setCurrentValue(displayValue);
			} else {
				displayParam.setCurrentValue(DEFAULT_NOT_SET_VALUE);
			}
		}
	}

	private String getOptionValue(List<StringsPair> options, String currentKey) {
		if (options != null && options.size() > 0) {
			for (int i = 0; i < options.size(); i++) {
				StringsPair pair = (StringsPair) options.get(i);
				if (currentKey.equals(pair.getKey())) {
					return pair.getValue();
				}
			}
		}
		return null;
	}
