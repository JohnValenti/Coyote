	private static final String DEFAULT_NOT_SET_VALUE = ReportParametersDesignPC.DEFAULT_NOT_SET_VALUE;
	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();
	private static RfsReportControlsXml rfsReportControls = RfsReportControlsXml.getInstance();
	private static RfsReportParametersModelHandler rfsReportParametersModelHandler = RfsReportParametersModelHandler.getInstance();
	private static Category mCat = Log.initCategory(RfsReportParameterBranchSelectorPC.class.getName());
	
	/**
	 * This constructor passes on most of it's work to super classes, however
	 * the name of the page component is set at this time. The component name
	 * often matches the parameter name.
	 * 
	 * @param context
	 *            the composite of the servlet information, the servlet request,
	 *            servlet response, session context, etc.
	 * @param toolbar
	 * @param controlNode
	 *            the child with in the controls document which contains the
	 *            information regarding this PC component (i.e. Report Control).
	 */
	public RfsReportParameterBranchSelectorPC(RequestContext context, ToolbarPC toolbar, Node controlNode) {
		super(context, toolbar);

		setIsRequired(reportControls.isControlRequired(controlNode));
		setIsMultiSelectEnabled(rfsReportControls.isControlMultiSelect(controlNode));
	}

	/**
	 * Returns the display-able label for the assigned report parameter. The
	 * label should be in the proper locale.
	 * 
	 * @param rff
	 *            the report framework request, which contains all the
	 *            parameters, is not needed in single parameter
	 *            ReportParametersPCIntereface's.
	 * @param controlNode
	 *            the child with in the controls document which contains the
	 *            information regarding this PC component (i.e. Report Control).
	 * @return the display-able report label.
	 */
	public String getLabel(RffRequest rff, Node controlNode) {
		String label = reportControls.getControlLabel(controlNode);
		if (label == null || label.length() == 0) {
			if (rfsReportControls.isControlMultiSelect(controlNode)) {
				label = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, "MULTIPLE_BRANCHES");
			} else {
				label = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, "SINGLE_BRANCH");
			}
		}
		return label;
	}

	/**
	 * Returns the HTML source for the implementation of the page component,
	 * corresponding to the specified frame work parameter and control data.
	 * <p>
	 * Special handling if the in-string attribute is set indicating the
	 * parameter is stored in HTTP as a single comma (i.e. DELIM) separated
	 * string.
	 * 
	 * @param rff
	 *            the report framework request, which contains all the
	 *            parameters.
	 * @param controlNode
	 *            the child with in the controls document which contains the
	 *            information regarding this PC component (i.e. Report Control).
	 * @return the HTML source for the input control

	public String getHtmlDisplay(RffRequest rff, Node controlNode) {
		return getHtmlDisplay(rff, controlNode, false);
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode, boolean readOnly) {
		if (reportControls.hasSingleParameter(controlNode) == false) {
			throw new IllegalArgumentException("Can only handle single report parameter.");
		}
		Node paramNode = reportControls.getParameter(controlNode);
		if (paramNode == null) {
			throw new IllegalArgumentException("A parameter was not specified for this control.");
		}
		String paramName = reportControls.getParameterName(paramNode);
		RffParameter param = rff.getRffParameter(paramName);

		Object values[] = null;
		if (param != null) {
			values = m_context.getRequest().getParameterValues("RfsReportParameterBranchSelectorPCHiddenValue");
		}
		if (values != null && values.length != 0) {
			Collection ids = null;
			if (reportControls.isParameterInString(paramNode)) {
				String strValue = values[0].toString();
				ids = StringUtil.parseIDStringToCollection(strValue);
			} else {
				int size = values.length;
				ids = new ArrayList(size);
				for (int i = 0; i < size; i++) {
					ids.add(new ID(values[i]));
				}
			}
			initIDs(ids);
		}
		this.setIsEditable(!readOnly);
		return super.getHtmlDisplay();
	}

	/**
	 * Assigns to the RFF request the values of the servlet request, these value
	 * will have been the user entered data. The RFF parameter will be those
	 * assigned thru the setRffParameter or the setRffRequest methods. Simply,
	 * extract the parameters from the HTTP request and put them in the report
	 * framework, for use by the reporting system.
	 * <p>
	 * Special handling if the in-string attribute is set indicating the
	 * parameter is stored in HTTP as a single comma (i.e. DELIM) separated
	 * string.
	 * 
	 * @param request
	 *            the HTTP request in which parameters are to be retrieved.
	 * @param rff
	 *            the report framework request which contains the RFF parameters
	 *            where the data is to be placed.
	 * @param controlNode
	 *            the child with in the controls document which contains the
	 *            information regarding this PC component (i.e. Report Control).
	 */
	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode) {
		if (reportControls.hasSingleParameter(controlNode) == false) {
			throw new IllegalArgumentException("Can only handle single report parameter.");
		}
		Node paramNode = reportControls.getParameter(controlNode);
		if (paramNode == null) {
			throw new IllegalArgumentException("A parameter was not specified for this control.");
		}
		String paramName = reportControls.getParameterName(paramNode);
		RffParameter param = rff.getRffParameter(paramName);
		if (param == null) {
			throw new IllegalArgumentException("Parameter in controls but not in report: " + paramName);
		}

		extractParameters(request);

		Object currentValues[] = getIDs().toArray();
		Collection<String> branchIds = Collections.emptyList();
		if (currentValues == null) {
			param.setCurrentValue(DEFAULT_NOT_SET_VALUE);
		} else {
			try {
				branchIds = rfsReportParametersModelHandler.getBranchesByIDs(m_context, getIDs());
				param.setCurrentValues(branchIds.toArray());
			} catch (Exception e) {
				mCat.error(" Error in setrffparam for Rfs Branch selector");
			}
		}

		try {
			String displayParamName = reportControls.getParameterDisplayName(paramNode);
			RffParameter displayParam = rff.getRffParameter(displayParamName);
			if (displayParam != null) {
				// Set the display value to be the name of the branch
				if (!param.getCurrentValue().equals(DEFAULT_NOT_SET_VALUE)) {
					displayParam.setCurrentValue(rfsReportParametersModelHandler.getBranchNames(m_context, branchIds.toArray()));
				} else {
					displayParam.setCurrentValue(param.getCurrentValue());
				}
			}
		} catch (Exception ex) {
			throw new IllegalArgumentException("Unable to retrieve branch names.");
		}
	}
