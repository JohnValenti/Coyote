package com.verint.web.rfs.reports.selection;


import com.bluepumpkin.web.bbm.keys.BbmJavaScriptFileID;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.rfs.reports.util.RfsReportParameterConstants;
import com.witness.web.uif.pagecomponent.container.ContainerPC;
import com.witness.web.uif.pagecomponent.form.FormTwoColumnPC;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.pagemodel.SelectorPopupWindowPM;
import com.witness.web.uif.system.RequestContext;

public class RfsReportParameterBranchSelectorPopupPM extends SelectorPopupWindowPM {
	
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private RfsBranchTreePC m_branchTreePC;
	private RfsReportParameterSingleSelectorPC m_branchViewPC;
	private ContainerPC m_displayPC;
	private FormTwoColumnPC m_formViewPC;

	public RfsReportParameterBranchSelectorPopupPM(RequestContext context) {
		super(context);		
		m_displayPC = new ContainerPC(m_context);
		m_branchViewPC = new RfsReportParameterSingleSelectorPC(m_context,RfsReportParameterConstants.BRANCH_VIEW);
		m_branchViewPC.initForDisplay();
		m_formViewPC = new FormTwoColumnPC(context);
		m_formViewPC.setTitle(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,FsWebBundleKey.BRANCH_PICKER_VIEW_TITLE));
		m_formViewPC.addRow(m_branchViewPC,"");
		m_displayPC.addComponent( m_formViewPC);		
		m_branchTreePC = new RfsBranchTreePC(m_context);		
		m_displayPC.addComponent(m_branchTreePC);
		addChildComponent(m_displayPC);
		init(RfsReportParameterBranchSelectorPC.REQUEST_HANDLER,BbmJavaScriptFileID.MEDIATOR_ORG_SELECTOR,FsWebBundleKey.BUNDLE_NAME,FsWebBundleKey.BRANCH_PICKER_PAGE_TITLE);
	}
	
	/**
	 * Return an instance of Page Model
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		//return getInstance(context,RfsReportParameterBranchSelectorPopupPM.class);
		RfsReportParameterBranchSelectorPopupPM model = new RfsReportParameterBranchSelectorPopupPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		model.populateModel();
		return model;
	}

	protected void initContentTitle(){
		super.initContentTitle();
		setIsContentItemNameRequired(false);
	}
	/**
	 * Initialize Model with Data
	 */
	protected void populateModel() {
		if (m_isInitialized)
			return;

		m_isInitialized = true;

		try {
			m_branchTreePC.init();

		} catch (Exception ex) {

		} finally {
			m_branchTreePC.setMultiRowSelectable(isMultiSelectEnabled());
			setDisplayPC(m_displayPC);
		}
	}
}
