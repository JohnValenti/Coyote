package com.verint.web.rfs.reports.selection;

import javax.servlet.http.HttpServletRequest;

import org.w3c.dom.Node;

import com.bluepumpkin.web.core.reports.ReportControlsXml;
import com.bluepumpkin.web.core.reports.ReportParametersPCInterface;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.bluepumpkin.web.ireport.framework.RffRequest;
import com.verint.web.rfs.reports.util.RfsReportControlsXml;
import com.witness.web.uif.pagecomponent.CompositePC;
import com.witness.web.uif.pagecomponent.form.FormTwoColumnPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;

public class RfsReportParameterStaffMixPickerPC extends CompositePC implements ReportParametersPCInterface {

	private static final long serialVersionUID = 1L;
	
	private static RfsReportControlsXml rfsReportControls = RfsReportControlsXml.getInstance();
	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();
	
	private RfsReportParameterSingleSelectorPC m_actualStaffMixPC = null;
	private RfsReportParameterSingleSelectorPC m_budgetStaffMixPC = null;
	private RfsReportParameterSingleSelectorPC m_plannedStaffMixPC = null;
	private RfsReportParameterSingleSelectorPC m_workgroupPC = null;
	private FormTwoColumnPC m_staffMixPC = null;
	
	public RfsReportParameterStaffMixPickerPC(RequestContext context) {
		super(context);
	}
	
	public RfsReportParameterStaffMixPickerPC(RequestContext context, ToolbarPC toolbar, Node controlNode) {
		super(context);
		setName(reportControls.getControlSingleParameterName(controlNode));
		m_staffMixPC = new FormTwoColumnPC(context);
		m_staffMixPC.setShowTitle(false);
		addChildComponent(m_staffMixPC);
		
		Node actualStaffMixNode = rfsReportControls.getControlActualStaffMix(controlNode);
		if(actualStaffMixNode != null)
			m_actualStaffMixPC = new RfsReportParameterSingleSelectorPC(context, toolbar, actualStaffMixNode);
			
		Node budgetStaffMixNode = rfsReportControls.getControlBudgetStaffMix(controlNode);
		if(budgetStaffMixNode != null)
			m_budgetStaffMixPC = new RfsReportParameterSingleSelectorPC(context, toolbar, budgetStaffMixNode);
	
		Node plannedStaffMixNode = rfsReportControls.getControlPlannedStaffMix(controlNode);
		m_plannedStaffMixPC = new RfsReportParameterSingleSelectorPC(context, toolbar, plannedStaffMixNode);
	
		Node workgroupNode = rfsReportControls.getControlWorkgroup(controlNode);
		if(workgroupNode != null)
			m_workgroupPC = new RfsReportParameterSingleSelectorPC(context, toolbar, workgroupNode);	
	}	
	
	public void initForDisplay() {
		if (m_isInitializedForDisplay)
			return;

		super.initForDisplay();
	}
	
	public String getLabel(RffRequest rff, Node controlNode) {
		String label = reportControls.getControlLabel(controlNode);
		if (label == null || label.length() == 0) {
			label = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,FsWebBundleKey.STAFFMIXES);//"Staff Mixes";
		}
		return label;
	}

	public String getHtmlDisplay() {
		// TODO Auto-generated method stub
		return null;
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode) {
		return getHtmlDisplay(rff, controlNode, false);
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode,boolean readOnly) {
		
		//StringBuffer content = new StringBuffer();
		Node actualStaffMixNode = rfsReportControls.getControlActualStaffMix(controlNode);		
		Node budgetStaffMixNode = rfsReportControls.getControlBudgetStaffMix(controlNode);
		Node plannedStaffMixNode = rfsReportControls.getControlPlannedStaffMix(controlNode);
		Node workgroupNode = rfsReportControls.getControlWorkgroup(controlNode);
		
		m_staffMixPC.addRow(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,FsWebBundleKey.PLANNED_STAFFMIX), m_plannedStaffMixPC.getHtmlDisplay(rff, plannedStaffMixNode, false, true));
		
		if(actualStaffMixNode != null) {
			m_staffMixPC.addRow(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,FsWebBundleKey.ACTUAL_STAFFMIX), m_actualStaffMixPC.getHtmlDisplay(rff, actualStaffMixNode, false, true));
		}
		
		if(budgetStaffMixNode!= null) {
			m_staffMixPC.addRow(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,FsWebBundleKey.BUDGET_STAFFMIX), m_budgetStaffMixPC.getHtmlDisplay(rff, budgetStaffMixNode, false, true));
		}
		
		if(workgroupNode != null) {
			m_staffMixPC.addRow(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,FsWebBundleKey.WORKGROUP), m_workgroupPC.getHtmlDisplay(rff, workgroupNode, false, true));
		}
		
		return m_staffMixPC.getHtmlDisplay();
	}

	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode) {

		if(m_actualStaffMixPC != null)
			m_actualStaffMixPC.setRffParameterValue(request, rff, rfsReportControls.getControlActualStaffMix(controlNode),true);	
		if(m_budgetStaffMixPC != null)
			m_budgetStaffMixPC.setRffParameterValue(request, rff, rfsReportControls.getControlBudgetStaffMix(controlNode),true);
		m_plannedStaffMixPC.setRffParameterValue(request, rff, rfsReportControls.getControlPlannedStaffMix(controlNode),true);
		if(m_workgroupPC != null)
			m_workgroupPC.setRffParameterValue(request, rff, rfsReportControls.getControlWorkgroup(controlNode),true);
		
	}
	
	public String getCustomJavaScript(Node controlNode) {
		return null;
	}
	
	public boolean extractParameters(HttpServletRequest request) {
		super.extractParameters(request);
		return true;
	}
}
