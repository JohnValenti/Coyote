package com.verint.ejb.bbm.servicegoals.model;

/**
 * Field info for Service Goals Time Series records (SPQUEUEGOALSERIES table).
**/
import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;


public class ServiceGoalsTimeSeriesDataItemFieldInfo extends FieldInfo {
	
	private static final long serialVersionUID = 407249147420362351L;

	public static final int SPQUEUEGOALSERIES_ID = 0; // Unique identifier for each service goal series item
	public static final int SPQUEUEGOALSERIES_SPQUEUEID = 1;// Identifier for SP queue
	public static final int SPQUEUEGOALSERIES_DATETIME = 2; // GMT date/time for which values are recorded. Values are typically recorded every 60 minutes in the case of goal series data, and this time represents the start of the interval
	public static final int SPQUEUEGOALSERIES_SLPERCENT = 3; // Service Level Percentage - when type of service goal is 0 in table SPQUEUE, this column stores the percentages for each time interval - goal for the interval is met if this percentage of calls are answered within the time specified in the SLANSWERWAITINGTIME column of the SPQUEUE table
	public static final int SPQUEUEGOALSERIES_ASA = 4; // Average Speed to Answer - When the service goal type is 1 in SPQUEUE, service goal for the time interval is met when average speed to answer is less than the time in this column
	public static final int SPQUEUEGOALSERIES_OCCUPANCY = 5; // Agent occupancy (percentage of agents in a call center that are at work) as a goal at the time
	public static final int SPQUEUEGOALSERIES_MAXDIALS = 6;// The maximum number of dials that can be made in this interval for an out-bound queue
	public static final int SPQUEUEGOALSERIES_CREATED = 7;// Date and time of instance creation
	public static final int SPQUEUEGOALSERIES_CREATEDBY = 8;// User who created the instance
	public static final int SPQUEUEGOALSERIES_MODIFIED = 9;// Date and time of instance modification
	public static final int SPQUEUEGOALSERIES_MODIFIEDBY = 10;// User who modified the instance

	public static final int NUMOFFIELDS = 11;
	private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[NUMOFFIELDS];
	private static String m_strTableName = "SPQUEUEGOALSERIES";
	static {
		m_arrFieldInfo[SPQUEUEGOALSERIES_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
		m_arrFieldInfo[SPQUEUEGOALSERIES_SPQUEUEID] = new FieldInfoEntry("SPQUEUEID", PrimitiveTypes.ID);
		m_arrFieldInfo[SPQUEUEGOALSERIES_DATETIME] = new FieldInfoEntry("DATETIME", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[SPQUEUEGOALSERIES_SLPERCENT] = new FieldInfoEntry("SLPERCENT", PrimitiveTypes.DOUBLE);
		m_arrFieldInfo[SPQUEUEGOALSERIES_ASA] = new FieldInfoEntry("ASA", PrimitiveTypes.FLOAT);
		m_arrFieldInfo[SPQUEUEGOALSERIES_OCCUPANCY] = new FieldInfoEntry("OCCUPANCY", PrimitiveTypes.FLOAT);
		m_arrFieldInfo[SPQUEUEGOALSERIES_MAXDIALS] = new FieldInfoEntry("MAXDIALS", PrimitiveTypes.FLOAT);
		m_arrFieldInfo[SPQUEUEGOALSERIES_CREATED] = new FieldInfoEntry("CREATED", PrimitiveTypes.DATE);
		m_arrFieldInfo[SPQUEUEGOALSERIES_CREATEDBY] = new FieldInfoEntry("CREATEDBY", PrimitiveTypes.STRING);
		m_arrFieldInfo[SPQUEUEGOALSERIES_MODIFIED] = new FieldInfoEntry("MODIFIED", PrimitiveTypes.DATE);
		m_arrFieldInfo[SPQUEUEGOALSERIES_MODIFIEDBY] = new FieldInfoEntry("MODIFIEDBY", PrimitiveTypes.STRING);
	}

	public ServiceGoalsTimeSeriesDataItemFieldInfo() {
		super();
	}

	public String getTableName() { return m_strTableName; }
	public int getIDFieldIndex() { return SPQUEUEGOALSERIES_ID; }
	public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }
	public int getParentIDFieldIndex() { return SPQUEUEGOALSERIES_SPQUEUEID; }
}
