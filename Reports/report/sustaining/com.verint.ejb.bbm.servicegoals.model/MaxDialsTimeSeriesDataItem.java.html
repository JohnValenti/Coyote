package com.verint.ejb.bbm.servicegoals.model;

import java.util.Date;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

public class MaxDialsTimeSeriesDataItem extends ServiceGoalsTimeSeriesDataItem {

	private static final long serialVersionUID = 7705208190663282995L;
	
	private static FieldInfo m_fieldInfo = new ServiceGoalsTimeSeriesDataItemFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	public MaxDialsTimeSeriesDataItem(Date time, Double value) {
		super(time, value);
	}

	@Override
	public double getValue() {
		return getFieldValueDouble(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_MAXDIALS);
	}

	@Override
	public void setValue(double value) {
		setFieldValue(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_MAXDIALS, value);
	}
}
