package com.verint.ejb.bbm.servicegoals.model;

import java.util.Date;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**
 * Value object representing a service goals time series record.
 * There are three known subclasses: ASATimeSeriesDataItem, MaxDialsTimeSeriesDataItem,
 * and ServiceLevelPercentTimeSeriesDataItem.  Each of these stores their values on different
 * columns of the same table, the logic of which is delegated to the subclasses.
 */
public abstract class ServiceGoalsTimeSeriesDataItem extends ValueObjectBase {

	private static final long serialVersionUID = -8736585620482340810L;

	public ServiceGoalsTimeSeriesDataItem(Date time, Double value) {
		setFieldValue(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_DATETIME, time);
		setValue(value);
	}

	private static FieldInfo m_fieldInfo = new ServiceGoalsTimeSeriesDataItemFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public Date getTime() {
		return getFieldValueDate(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_DATETIME);
	}

	public abstract double getValue();
	
	public abstract void setValue(double value);
}
