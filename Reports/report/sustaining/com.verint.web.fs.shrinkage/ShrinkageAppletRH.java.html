package com.verint.web.fs.shrinkage;

import java.util.Collection;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeContext;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.SPShrinkage;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.time.TimeContextFactory;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.core.applet.AppletRH;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.core.l10n.CoreWebLogBundleKey;
import com.witness.web.uif.system.RequestContext;


public class ShrinkageAppletRH extends AppletRH {

	@Override
	protected void processAppletRequest(RequestContext context,
			Map appletRequest, Map appletResponse) {
		if (isAppletAction(appletRequest, ShrinkageAppletKeys.GET_SHRINKAGE_ACTION)) {
			Object o = appletRequest.get(ShrinkageAppletKeys.GET_SHRINKAGE_PARAM_SPID);
			if (o != null) {
				// load shrinkage
				ID spid = new ID(o.toString());
				try {
					SchedulingPeriod sp = CampaignModelHandler.getSchedulingPeriodByID(context, spid);
					Campaign c = CampaignModelHandler.getCampaign(context, sp.getCampaignID());
					TimeContext tc = TimeContextFactory.getTimeContext(c);
					appletResponse.put(ShrinkageAppletKeys.GET_SHRINKAGE_RESPONSE_SP, sp);
					appletResponse.put(ShrinkageAppletKeys.GET_SHRINKAGE_RESPONSE_HOO_ASSIGNMENT, CampaignModelHandler.getCampaignHOOAssignments(context, sp.getCampaignID(), sp.getStartTime(), sp.getEndTime()));
					appletResponse.put(ShrinkageAppletKeys.GET_SHRINKAGE_RESPONSE_SPSHRINKAGE, CampaignModelHandler.getOrCreateSPShrinkageBySPID(context, spid));
					appletResponse.put(ShrinkageAppletKeys.GET_SHRINKAGE_RESPONSE_TIME_CONTEXT, tc);
				} catch (Exception ex) {
					handleError(context, ex,
							CoreWebLogBundleKey.APPLET_PROCESS_REQUEST_ERROR,
							CoreWebBundleKey.APPLET_PROCESS_REQUEST_ERROR,
							CoreWebBundleKey.BUNDLE_NAME);
				}
			}
		}
		else if (isAppletAction(appletRequest, ShrinkageAppletKeys.SAVE_SHRINKAGE_ACTION)) {
			try {
				Collection<SPShrinkage> collection = (Collection<SPShrinkage>) appletRequest.get(ShrinkageAppletKeys.SAVE_SHRINKAGE_PARAM_SPS_COL);
				if (collection != null && collection.size() > 0) {
					CampaignModelHandler.updateSPShrinkage(context, collection);
				}
			} catch (Exception ex) {
				handleError(context, ex,
						CoreWebLogBundleKey.APPLET_PROCESS_REQUEST_ERROR,
						CoreWebBundleKey.APPLET_PROCESS_REQUEST_ERROR,
						CoreWebBundleKey.BUNDLE_NAME);
			}
		}
		super.processAppletRequest(context, appletRequest, appletResponse);
	}
	
}
