package com.verint.web.fs.optimization.forecasting;

import java.io.Serializable;
import java.util.Collections;
import java.util.List;

import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.verint.ejb.bbm.forecast.model.ForecastProfile;
import com.verint.web.fs.optimization.SingleQueueDTO;

public class SingleForecastQueueDTO extends SingleQueueDTO implements Serializable {

	private static final long serialVersionUID = -6260243725486658336L;
	
	private final List<ForecastProfile> m_forecastProfiles;
	private TraceCube m_forecast; 
	
	public SingleForecastQueueDTO(Queue queue, SPQueue spQueue, List<ForecastProfile> forecastProfiles) {
		super(queue, spQueue);
		m_forecastProfiles = forecastProfiles;
	}
	
	public SingleForecastQueueDTO(Queue queue, SPQueue spQueue, List<ForecastProfile> forecastProfiles, 
			TraceCube forecast) {
		this(queue, spQueue, forecastProfiles);
		m_forecast = forecast;
	}
	
	public TraceCube getForecast() {
		return m_forecast;
	}
	
	public List<ForecastProfile> getForecastProfiles() {
		return Collections.unmodifiableList(m_forecastProfiles);
	}
}
