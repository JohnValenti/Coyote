package com.verint.web.fs.optimization.forecasting;

import java.util.Collection;

import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.workload.model.Media;
import com.verint.web.fs.optimization.Response;

public class ProjectForecastingResponse<T> extends ForecastResponse<T> {
	private boolean useUnknownVolumeInForecast;
	public ProjectForecastingResponse(Media media, Campaign campaign,
			SchedulingPeriod sp, T queueDto, Collection<CampaignHOO> hoo,
			boolean useUnknownVolumeInForecast,
			boolean useWeightedSmoothingInForecast) {
		super(media, campaign, sp, queueDto, hoo, useWeightedSmoothingInForecast);
		this.useUnknownVolumeInForecast = useUnknownVolumeInForecast;
	}

	public boolean isUseUnknownVolumeInForecast() {
		return useUnknownVolumeInForecast;
	}
}
