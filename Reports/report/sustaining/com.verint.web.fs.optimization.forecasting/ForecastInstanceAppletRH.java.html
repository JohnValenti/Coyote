package com.verint.web.fs.optimization.forecasting;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastInterimBacklog;
import com.bluepumpkin.ejb.bbm.workload.model.Media;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.core.l10n.CoreWebLogBundleKey;
import com.bluepumpkin.web.fs.Log;
import com.verint.ejb.bbm.forecast.ejb.ForecastTimeSeriesManager;
import com.verint.ejb.bbm.forecast.model.ForecastProfile;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.verint.web.fs.optimization.CombinedQueuePartialResponse;
import com.verint.web.fs.optimization.MultiQueuePartialResponse;
import com.verint.web.fs.optimization.SingleQueuePartialResponse;
import com.witness.web.uif.system.RequestContext;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Collections;
import java.util.Map;


public class ForecastInstanceAppletRH extends ForecastingAppletRequestHandler{

	/**
	 * Process Applet Request - to be overrided by derived classes
	 * @param context - The Request Context
	 * @param appletRequest - Map containing Applet Request Information
	 * @param appletResponse - Map containing Response to be send to Applet
	 */
	private static final Category cat = Log.initCategory(ForecastInstanceAppletRH.class.getName());

	protected void processAppletRequest(RequestContext context, Map request, Map response) {
		try {
			if (isAppletAction(request, ForecastingKeys.PROC_DELETE_FORECAST_INSTANCE)) {
				deleteForecastInstance(context, request, response);
			} else if (isAppletAction(request, ForecastingKeys.PROC_VALIDATE_SELECTED_FORECAST_INSTANCE)) {
				validateForecastInstanceIdForSpQueues(context, request, response);
			} else if (isAppletAction(request, ForecastingKeys.PROC_VALIDATE_SELECTED_FORECAST_INSTANCE_COMBINED)) {
				validateForecastInstanceIdForMedia(context, request, response);
			} else {
				super.processAppletRequest(context, request, response);
			}
		} catch (Exception e) {
			handleError(context, e,
					CoreWebLogBundleKey.APPLET_PROCESS_REQUEST_ERROR,
					CoreWebBundleKey.APPLET_PROCESS_REQUEST_ERROR,
					CoreWebBundleKey.BUNDLE_NAME);
		}
	}

	private void validateForecastInstanceIdForSpQueues(RequestContext context, Map request, Map response) {
		try {
			ForecastTimeSeriesManager ftsm = WfmManagerFactory.getForecastTimeSeriesManager(context.isInWhatIfMode());
			ID forecastInstanceId = (ID)request.get(ForecastingKeys.PARAM_FORECAST_INSTANCE_IDS);
			ID spId = (ID)request.get(ForecastingKeys.PARAM_SP_ID);
			Collection<ID> queueIds = (Collection<ID>)request.get(ForecastingKeys.PARAM_QUEUE_IDS);

			response.put(ForecastingKeys.RETVAL_BOOLEAN, ftsm.isForecastInstanceValidForQueuesInSp(forecastInstanceId,
					spId, queueIds));
		} catch (Exception ex) {
			notifyError(context, response, ex);
		}
	}

	private void validateForecastInstanceIdForMedia(RequestContext context, Map request, Map response) {
		try {
			ForecastTimeSeriesManager ftsm = WfmManagerFactory.getForecastTimeSeriesManager(context.isInWhatIfMode());
			ID forecastInstanceId = (ID)request.get(ForecastingKeys.PARAM_FORECAST_INSTANCE_IDS);
			ID spId = (ID)request.get(ForecastingKeys.PARAM_SP_ID);
			ID mediaId = (ID)request.get(ForecastingKeys.PARAM_MEDIA_ID);

			response.put(ForecastingKeys.RETVAL_BOOLEAN, ftsm.isForecastInstanceValidForMediaInSp(forecastInstanceId,
					spId, mediaId));
		} catch (Exception ex) {
			notifyError(context, response, ex);
		}
	}

	/**
	 * WARNING: this method also modified the SPQueues in partial response by updating their interim backlog
	 * information with the interim backlog belonging to the forecast instance!
	 */
	protected Collection<ForecastProfile> getForecastProfilesForSingleQueueModelResponse(RequestContext context,
			SingleQueuePartialResponse partialResponse, Map request) throws Exception {
		ForecastTimeSeriesManager ftsm = WfmManagerFactory.getForecastTimeSeriesManager(context.isInWhatIfMode());
		ID forecastInstanceId = (ID) request.get(ForecastingKeys.PARAM_FORECAST_INSTANCE_IDS);

		if (forecastInstanceId != null) {
			updateInterimBacklogOnSpQueues(forecastInstanceId, partialResponse.getSPQueue(), context);
			return ftsm.getForecastProfilesByForecastInstanceIDAndSPQueueIDs
				(forecastInstanceId, Collections.singleton(partialResponse.getSPQueue().getID()));
		} else {
			return super.getForecastProfilesForSingleQueueModelResponse(context, partialResponse, request);
		}
	}

	/**
	 * WARNING: this method also modified the SPQueues in partial response by updating their interim backlog
	 * information with the interim backlog belonging to the forecast instance!
	 */
	protected Collection<ForecastProfile> getForecastProfilesForCombinedQueueModelResponse(RequestContext context,
			CombinedQueuePartialResponse partialResponse, Map request, int queueIndex) throws Exception {
		ForecastTimeSeriesManager ftsm = WfmManagerFactory.getForecastTimeSeriesManager(context.isInWhatIfMode());
		ID forecastInstanceID = (ID) request.get(ForecastingKeys.PARAM_FORECAST_INSTANCE_IDS);

		if (forecastInstanceID != null) {
			updateInterimBacklogOnSpQueues(forecastInstanceID, partialResponse.getSPQueues().get(queueIndex), context);
			return ftsm.getForecastProfilesByForecastInstanceIDAndSPQueueIDs(forecastInstanceID,
					Collections.singleton(partialResponse.getSPQueues().get(queueIndex).getID()));
		} else {
			return super.getForecastProfilesForCombinedQueueModelResponse(context, partialResponse, request, queueIndex);
		}
	}

	/**
	 * WARNING: this method also modified the SPQueues in partial response by updating their interim backlog
	 * information with the interim backlog belonging to the forecast instance!
	 */
	protected Collection<ForecastProfile> getForecastProfilesForMultiQueueModelResponse(RequestContext context,
			MultiQueuePartialResponse partialResponse, Map request, int queueIndex) throws Exception {
		ForecastTimeSeriesManager ftsm = WfmManagerFactory.getForecastTimeSeriesManager(context.isInWhatIfMode());
		ID forecastInstanceId = (ID) request.get(ForecastingKeys.PARAM_FORECAST_INSTANCE_IDS);
		if (forecastInstanceId != null) {
			updateInterimBacklogOnSpQueues(forecastInstanceId, partialResponse.getSPQueues().get(queueIndex), context);
			return ftsm.getForecastProfilesByForecastInstanceIDAndSPQueueIDs(forecastInstanceId,
					Collections.singleton(partialResponse.getSPQueues().get(queueIndex).getID()));
		} else {
			return super.getForecastProfilesForMultiQueueModelResponse(context, partialResponse, request, queueIndex);
		}
	}


	protected void deleteForecastInstance(RequestContext context, Map request, Map response) throws BbmException {
		try {
			ForecastTimeSeriesManager ftsm = WfmManagerFactory.getForecastTimeSeriesManager(context.isInWhatIfMode());
			ID  instanceID = (ID)request.get(ForecastingKeys.PARAM_FORECAST_INSTANCE_IDS);
			Collection<ID> spQueueIDs = (Collection<ID>) request.get(ForecastingKeys.PARAM_SPQUEUE_IDS);
			ftsm.deleteForecastInstance(instanceID, spQueueIDs);

			response.put(ForecastingKeys.RETVAL, "Success!");
		} catch (Exception e) {
			throw new BbmException(e);
		}
	}

	/**
	 * Updates the interim backlog on the given SPQueue by looking up the interim backlog associated to the
	 * given forecast instance and then setting it on spQueue.
	 */
	private void updateInterimBacklogOnSpQueues(ID forecastInstanceId, SPQueue spQueue, RequestContext context)
			throws BbmEJBCreateException, BbmFinderException, RemoteException {

		if (!Media.isMediaDeferred(spQueue.getMediaID())) {
			return;
		}

		Collection<ForecastInterimBacklog> fibs =
				getForecastTimeSeriesManager(context).getInterimBacklogByInstanceIDnSPQID(forecastInstanceId,
						Collections.singleton(spQueue.getID()));

		if (fibs != null && !fibs.isEmpty()) {
			for (ForecastInterimBacklog fib : fibs) {
				if(fib.getSpQueueId().equals(spQueue.getID())) {
					cat.debug("ForecastInstanceAppletRH.updateInterimBacklogOnSpQueues: temporarily setting interim " +
							"backlog of " + fib.getInterimBacklog() + "for SpQueue " + spQueue.toString());
					spQueue.setInterimBacklogDate(fib.getInterimBacklogDate());
					spQueue.setInterimBacklog(fib.getInterimBacklog());
				}
			}
		}
	}

	private ForecastTimeSeriesManager getForecastTimeSeriesManager(RequestContext context) throws BbmEJBCreateException {
		return WfmManagerFactory.getForecastTimeSeriesManager(context.isInWhatIfMode());
	}
}
