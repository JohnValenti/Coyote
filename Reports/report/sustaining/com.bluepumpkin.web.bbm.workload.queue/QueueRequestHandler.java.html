	/**
	 * Process Requests for the queue area
	 */
	@Override
	public void processRequest(RequestContext context) {
		m_localizer = context.getLocalizer();
		m_bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);

		// Process requested action
		if (isAction(context, PageAction.NEXT)) {
			processNext(context);
		} else if (isAction(context, PageAction.PREVIOUS)) {
			processPrev(context);
		} else if (isAction(context, PageAction.EDIT)) {
			processListEdit(context);
		} else if (isAction(context, PageAction.ADD)) {
			processListCreateNew(context);
		} else if (isAction(context, PageAction.DELETE)) {
			processListDelete(context);
		} else if (isAction(context, PageAction.CREATE_NEW)) {
			processFormCreateNew(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processFormSave(context);
		} else if (isAction(context, PageAction.CANCEL)) {
			processFormCancel(context);
		} else if (isAction(context, PageAction.NEW_ID)) {
			processDone(context); // gotoForm(context);
		} else if (isAction(context, PageAction.QUEUE_DELETE_ACTION)) {
			processFormDelete(context);
		} else if (isAction(context, PageAction.COPY)) {
			processFormCopy(context);
		} else if (isAction(context, PageAction.QUEUE_COPY_ACTION)) {
			processListCopy(context);
		} else {
			// System.out.println("going to DEFAULT_PAGE");
			setNextPage(context, QUEUE_LIST_PAGE);
		}
	}

	/**
	 * Processes the next button action on the pagination toolbar.
	 *
	 * @param context
	 * @param pageModel
	 */
	protected void processNext(RequestContext context) {
		final QueueListPageModel listModel = new QueueListPageModel(context);
		context.setPageModel(listModel);
		listModel.adjustEditIndexForward();
		// Go back to the list
		this.gotoList(context);
	}

	/**
	 * Processes the previous button action on the pagination toolbar.
	 *
	 * @param context
	 * @param pageModel
	 */
	protected void processPrev(RequestContext context) {
		final QueueListPageModel listModel = new QueueListPageModel(context);
		context.setPageModel(listModel);
		listModel.adjustEditIndexBackward();
		// Go back to the list
		this.gotoList(context);
	}

	/**
	 * Process Create new Action from list
	 */
	private void processListCreateNew(RequestContext context) {

		if (!isAuthorized(context, CONFIGURE_PRIVILEGE)) {
			gotoList(context);
			return;
		}

		final QueueFormPageModel formModel = new QueueFormPageModel(context);
		formModel.extractParameters(context.getRequest());
		formModel.setIsAddMode(true);
		formModel.setIsListMode(true);
		formModel.setQueue();
		formModel.setSelectedID("");
		formModel.setPerformActionOnID("");
		context.setPageModel(formModel);
		gotoForm(context);
	}

	/**
	 * Process delete Action from list
	 */
	private void processListDelete(RequestContext context) {
		if (!isAuthorized(context, CONFIGURE_PRIVILEGE)) {
			gotoList(context);
			return;
		}

		// Extract Parameter
		final QueueListPageModel model = new QueueListPageModel(context);
		model.extractParameters(context.getRequest());
		final ID[] ids = model.getSelectedQueueIDs();
		final String selectedRow = ids[0].toString();

		final ArrayList list = new ArrayList();
		list.add(new ID(selectedRow));

		if (!StringUtil.isEmpty(selectedRow)) {
			try {
				// WorkloadModelHandler.deleteObject(new
				// ID(selectedRow),BPObjectInfo.BPOBJECT_QUEUE,context.getLocaleContext());
				WorkloadModelHandler.deleteQueues(context, list);

				model.setSelectedID(OrganizationUtil
						.getSelectedOrganizationID(context));
				model.setSelectedOrgID(OrganizationUtil
						.getSelectedOrganizationID(context));

				model.setPerformActionOnID("");
				QueueUtil.setSelectedQueueID(context, "");
			} catch (final Exception ex) {
				log.l7dError(BbmWebLogBundleKey.QUEUE_DELETE_FAILURE, ex);
				model.addPageMessage(
						Message.ERROR_TYPE,
						verticaliseMsg(BbmWebBundleKey.QUEUE_DELETE_FAILURE,
								context));
			}

		}

		model.setSelectedID(OrganizationUtil.getSelectedOrganizationID(context));
		context.setPageModel(model);
		// Go back to the list
		// System.out.println("going to DEFAULT_PAGE 2");
		setNextPage(context, QUEUE_LIST_PAGE);
	}

	/**
	 * Process delete Action from list
	 */
	private void processFormDelete(RequestContext context) {
		if (!isAuthorized(context, CONFIGURE_PRIVILEGE)) {
			gotoForm(context);
			return;
		}

		final QueueFormPageModel formModel = new QueueFormPageModel(context);
		formModel.extractParameters(context.getRequest());
		final String selectedRow = context.getRequest().getParameter(
				QueueFormPageModel.PERFORM_ACTION_ON_ID_FN);

		final ArrayList list = new ArrayList();
		list.add(new ID(selectedRow));
		try {
			WorkloadModelHandler.deleteQueues(context, list);
			// formModel.setSelectedID("");
			QueueUtil.setSelectedQueueID(context, "");
			formModel.setPerformActionOnID("");
			formModel.setQueue();
			formModel.setRefreshSelection(true);
		} catch (final Exception ex) {
			log.l7dError(BbmWebLogBundleKey.QUEUE_DELETE_FAILURE, ex);
			formModel.addPageMessage(
					Message.ERROR_TYPE,
					verticaliseMsg(BbmWebBundleKey.QUEUE_DELETE_FAILURE,
							context));
		}

		context.setPageModel(formModel);
		gotoForm(context);
	}

	/**
	 * Process copy Action from list
	 */
	private void processListCopy(RequestContext context) {
		if (!isAuthorized(context, CONFIGURE_PRIVILEGE)) {
			gotoForm(context);
			return;
		}

		final QueueListPageModel model = new QueueListPageModel(context);

		model.extractParameters(context.getRequest());
		final ID[] ids = model.getSelectedQueueIDs();
		final String selectedRow = ids[0].toString();

		try {
			final ID newQueueID = WorkloadModelHandler.copyQueue(context,
					new ID(selectedRow));
			// model.setSelectedID(context.getParameter(QueueFormPageModel.SELECTED_ID_FN));

			model.setSelectedID(OrganizationUtil
					.getSelectedOrganizationID(context));
			model.setPerformActionOnID(newQueueID.toString());
			model.setSelectedOrgID(OrganizationUtil
					.getSelectedOrganizationID(context));

		} catch (final Exception ex) {
			log.l7dError(BbmWebLogBundleKey.QUEUE_COPY_FAILURE, ex);
			model.addPageMessage(Message.ERROR_TYPE,
					verticaliseMsg(BbmWebBundleKey.QUEUE_COPY_FAILURE, context));

		}
		model.setSelectedID(OrganizationUtil.getSelectedOrganizationID(context));
		context.setPageModel(model);
		setNextPage(context, QUEUE_LIST_PAGE);
	}

	/**
	 * Process Save Action from form
	 */
	private void processFormSave(RequestContext context) {

		final QueueFormPageModel formModel = new QueueFormPageModel(context);
		ID queueID = null;
		boolean isUsedInCampaign = false;
		try {

			formModel.extractParameters(context.getRequest());
			final Queue queue = formModel.getQueue();
			// ESR#3243172-F&amp;S client in Operations module of Campaign mode
			// displays message:
			queue.setName(queue.getName().replaceAll("\\s+", " ").trim());

			if (formModel.getIsAddMode()) {
				if (!WorkloadModelHandler.isMultiSiteLicense()
						&& (queue.getQueueType() != Queue.QUEUE_TYPE_PROCESS)) {
					queue.setQueueType(Queue.QUEUE_TYPE_NORMAL);
				}
				queueID = WorkloadModelHandler.createQueue(context, queue);
				formModel.setIsAddMode(false);
			} else {// update
				// verify that queue is not associated with any campaign if
				// media needs to be changed
				queueID = queue.getID();
				final Queue dbQueue = WorkloadModelHandler.getQueueByID(
						context, queueID);

				if (!dbQueue.getMediaStrID().trim()
						.equals(queue.getMediaStrID().trim())) {
					// get campaign queue assignment
					final Collection queueCampaign = CampaignModelHandler
							.getQueueCampaignAssignments(context, queueID,
									null, null);
					if ((queueCampaign != null) && (queueCampaign.size() > 0)) {
						formModel
								.addPageMessage(
										Message.ERROR_TYPE,
										verticaliseMsg(
												BbmWebBundleKey.QUEUE_UPDATE_MEDIA_FAILURE,
												context));
						queue.setMediaStrID(dbQueue.getMediaStrID().trim());
						isUsedInCampaign = true;
					}
				}

				WorkloadModelHandler.updateQueue(context, queue);

				/*
				 * // update VCT org id, if necessary QueueVCTMH data = new
				 * QueueVCTMH(); VCTConfiguration activeVCTQueueConfig =
				 * data.getVCTQueueConfigDetails(queueID); if
				 * (activeVCTQueueConfig != null) {
				 * if(activeVCTQueueConfig.getQueueID() != null) {
				 * activeVCTQueueConfig
				 * .setOrganizationID(queue.getOrganizationID());
				 * data.saveQueueConfig(activeVCTQueueConfig); }
				 * 
				 * 
				 * }
				 */

			}

			formModel.setRefreshSelection(true);

			if (formModel.getIsListMode()) {
				// QueueSummaryPageModel model= new
				// QueueSummaryPageModel(context);
				final QueueListPageModel model = new QueueListPageModel(context);

				// TODO - change to org ID, right?
				// model.setSelectedID(/*queueID.toString()*/OrganizationUtil.getSelectedOrganizationID(context));
				model.setSelectedID(context
						.getParameter(QueueFormPageModel.SELECTED_ID_FN));

				model.setPerformActionOnID(queueID.toString());

				// no longer need to check, as queues used in campaign cannot
				// have their media type modified.
				/*
				 * if (isUsedInCampaign)
				 * model.addPageMessage(Message.ERROR_TYPE, verticaliseMsg(
				 * BbmWebBundleKey.QUEUE_UPDATE_MEDIA_FAILURE, context));
				 * context.setPageModel(model);
				 */

				setNextPage(context, QUEUE_LIST_PAGE);
			} else {

				formModel.setSelectedID(context
						.getParameter(QueueFormPageModel.SELECTED_ID_FN));
				formModel.setPerformActionOnID(queueID.toString());
				context.setPageModel(formModel);
				gotoForm(context);
			}

		} catch (final BbmUpdateDuplicateKeyException e) {
			log.l7dError(BbmWebLogBundleKey.QUEUE_UPDATE_FAILURE_DUPLICATE, e);
			formModel.addPageMessage(
					Message.ERROR_TYPE,
					verticaliseMsg(
							BbmWebBundleKey.QUEUE_UPDATE_FAILURE_DUPLICATE,
							context));
			context.setPageModel(formModel);
			gotoForm(context);
		} catch (final BbmCreateDuplicateKeyException e) {
			log.l7dError(BbmWebLogBundleKey.QUEUE_CREATE_NEW_FAILURE_DUPLICATE,
					e);
			formModel.addPageMessage(
					Message.ERROR_TYPE,
					verticaliseMsg(
							BbmWebBundleKey.QUEUE_CREATE_NEW_FAILURE_DUPLICATE,
							context));
			context.setPageModel(formModel);
			gotoForm(context);
		} catch (final MultiUserException e) {
			log.l7dError(BbmWebLogBundleKey.QUEUE_UPDATE_FAILURE, e);
			// "Failed to save {0} information, since it has been modified by another user."
			final Object[] arguments = { m_localizer.i18n(m_bbmBundle,
					BbmWebBundleKey.QUEUE) };
			final ResourceBundle m_coreEjbBundle = m_localizer
					.getBundle(CoreEjbBundleKey.BUNDLE);
			formModel.addPageMessage(Message.WARNING_TYPE, m_localizer.i18n(
					m_coreEjbBundle, CoreEjbBundleKey.MULTIUSER_UPDATE,
					arguments));
			context.setPageModel(formModel);
			gotoForm(context);
		} catch (final Exception e) {
			if (formModel.getIsAddMode()) {
				log.l7dError(BbmWebLogBundleKey.QUEUE_CREATE_NEW_FAILURE, e);
				formModel.addPageMessage(
						Message.ERROR_TYPE,
						verticaliseMsg(
								BbmWebBundleKey.QUEUE_CREATE_NEW_FAILURE,
								context));
			} else {
				log.l7dError(BbmWebLogBundleKey.QUEUE_UPDATE_FAILURE, e);
				formModel.addPageMessage(
						Message.ERROR_TYPE,
						verticaliseMsg(BbmWebBundleKey.QUEUE_UPDATE_FAILURE,
								context));
			}
			context.setPageModel(formModel);
			gotoForm(context);
		}

	}

	private String verticaliseMsg(String key, RequestContext context) {
		// verticalise queue
		final Object[] args = new Object[1];
		args[0] = context
				.getSystemManagerFactory()
				.getVerticalizationManager()
				.getVerticalizedString(context,
						DefaultVerticalizationManager.QUEUESMALL);

		return MessageFormat.format(m_localizer.i18n(m_bbmBundle, key), args);

	}
