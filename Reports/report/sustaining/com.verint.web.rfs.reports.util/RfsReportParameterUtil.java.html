	public List<StringsPair> getReportParamOptions(RequestContext context, String currentValue, String pickerName, String strReportID)
			throws BbmException, RemoteException {

		String currentScenario = rfsReportParametersModelHandler.getCurrentScenario(context);
		if (RfsReportParameterConstants.SCENARIO.equals(pickerName) && currentValue != null) {
			currentScenario = currentValue;
		}
		StringBuilder branchStr = new StringBuilder("");
		if (!RfsReportParameterConstants.SCENARIO.equals(pickerName) && !RfsReportParameterConstants.BUSINESS_UNIT.equals(pickerName)) {
			branchStr = getBranchIDs(context);
		}
		switch (pickerName) {
		case RfsReportParameterConstants.SCENARIO:
			// Scenario picker
			return rfsReportParametersModelHandler.getScenarios(context);
		case RfsReportParameterConstants.BRANCH_VIEW:
			// Branch view drop down in branch selector popup
			return rfsReportParametersModelHandler.getBranchViews(currentScenario);
		case RfsReportParameterConstants.BUSINESS_UNIT:
			// Branches list if used in a single or multiple selector pc
			return rfsReportParametersModelHandler.getBusinessUnits(currentScenario);
		case RfsReportParameterConstants.MONTH:
			// Month picker
			return rfsReportParametersModelHandler.getMonthsList(context);
		case RfsReportParameterConstants.YEAR:
			// Year picker
			return rfsReportParametersModelHandler.getYearsList();
		case RfsReportParameterConstants.NUMBER_OF_INTERVALS:
			// Interval spinner ( 1- 13)
			return rfsReportParametersModelHandler.getIntervalSpinner();
		case RfsReportParameterConstants.HIERARCHY_GROUPBY:
			// Hierarchy to Group By for branches
			return rfsReportParametersModelHandler.getHierarchyGroupByList(currentScenario);
		case RfsReportParameterConstants.ACTUAL_STAFFMIX:
			// Actual Staff Mix
			return rfsReportParametersModelHandler.getStaffMixes(currentScenario, branchStr.toString(),
					RfsReportParameterConstants.STAFFMIX_ACTUAL);
		case RfsReportParameterConstants.BUDGET_STAFFMIX:
			// Budget Staff Mix
			return rfsReportParametersModelHandler.getStaffMixes(currentScenario, branchStr.toString(),
					RfsReportParameterConstants.STAFFMIX_BUDGET);
		case RfsReportParameterConstants.PLANNED_STAFFMIX:
			// Planned Staff Mix
			return rfsReportParametersModelHandler.getStaffMixes(currentScenario, branchStr.toString(),
					RfsReportParameterConstants.STAFFMIX_PLANNED);
		case RfsReportParameterConstants.WORKGROUP:
			// Workgroup include all option
			return rfsReportParametersModelHandler.getWorkgroups(context, true);
		case RfsReportParameterConstants.WORKGROUP_SINGLESELECTION:
			// Workgroup don't include all option
			return rfsReportParametersModelHandler.getWorkgroups(context, false);
		case RfsReportParameterConstants.STAFF_MIX_FILTER:
			return rfsReportParametersModelHandler.getStaffMixFilters(currentScenario);
		case RfsReportParameterConstants.FORECAST_RESOURCES:
		case RfsReportParameterConstants.FORECAST_RESOURCES_FIRST:
		case RfsReportParameterConstants.FORECAST_RESOURCES_SECOND:
			// Forecast Resources
			return rfsReportParametersModelHandler.getResources(currentScenario, "Resource", false);
		case RfsReportParameterConstants.HISTORY_RESOURCES:
			// History Resources
			return rfsReportParametersModelHandler.getResources(currentScenario, "Actual", false);
		case RfsReportParameterConstants.VOLUME_FORECAST:
			// Volume Forecast
			return rfsReportParametersModelHandler.getResources(currentScenario, "Transaction", false);
		case RfsReportParameterConstants.DURATION_FORECAST:
			// Duration Forecast
			return rfsReportParametersModelHandler.getResources(currentScenario, "Duration", false);
		case RfsReportParameterConstants.RESOURCE_CLASS:
			// Position
			return rfsReportParametersModelHandler.getResourceClasses(currentScenario, branchStr.toString());
		case RfsReportParameterConstants.RESOURCE_CLASS_MULTIPLE:
			// Position to include option "All"
			return rfsReportParametersModelHandler.getResourceClassesForMultipleSelection(context, currentScenario, branchStr.toString());
		case RfsReportParameterConstants.DAY_OF_WEEK:
			// Day of Week
			return rfsReportParametersModelHandler.getDaysOfWeekList();
		case RfsReportParameterConstants.EMPLOYEE_CATEGORIES:
			// Employee Categories
			String staffMix = context.getRequest().getParameter(RfsReportParameterConstants.PARAM_PLANNED_STAFFMIX);
			return rfsReportParametersModelHandler.getEmployeeCategories(staffMix != null ? Integer.parseInt(staffMix) : -1);
		case RfsReportParameterConstants.RESOURCE_CATEGORY:
			// Resource Categories
			return rfsReportParametersModelHandler.getResourceCategories(currentScenario);
		case RfsReportParameterConstants.TRANSACTION_CATEGORY:
			// Transaction Categories
			return rfsReportParametersModelHandler.getTransactionCategories(currentScenario);
		default:
			return new ArrayList<StringsPair>();
		}
	}

	StringBuilder getBranchIDs(RequestContext context) throws BbmException, RemoteException {
		StringBuilder branchStr = new StringBuilder();
		String branchIDsString = context.getRequest().getParameter(RfsReportParameterConstants.PARAM_BRANCH_SELECTOR);
		branchStr = branchStr.append(StringUtil.join(
				rfsReportParametersModelHandler.getBranchesByIDs(context, StringUtil.parseIDStringToCollection(branchIDsString)), ","));

		if (branchStr.length() == 0) {
			branchIDsString = context.getRequest().getParameter(RfsReportParameterConstants.PARAM_BRANCHES);
			if (branchIDsString != null) {
				branchStr = branchStr.append(branchIDsString);
			}
		}
		if (branchStr.length() == 0) {
			branchIDsString = context.getRequest().getParameter(RfsReportParameterConstants.PARAM_BRANCH);
			if (branchIDsString != null) {
				branchStr = branchStr.append(branchIDsString);
			}
		}
		return branchStr;
	}

	@SuppressWarnings("unchecked")
	public List<StringsPair> getOptions(RequestContext context, Node paramNode, String currentValue, String pickerName, String parentId) {
		List<StringsPair> options = null;
		try {
			options = getReportParamOptions(context, currentValue, pickerName, parentId);
		} catch (Exception ex) {

		}
		if (options == null || options.isEmpty()) {
			options = reportControls.getReportParamListOptions(paramNode);
			if (options != null) {
				Localizer localizer = SessionUtil.getLocalizer(context.getLocaleContext());
				ResourceBundle bundle = ReportParametersDesignPC.getReportBundle();
				for (int i = 0; i < options.size(); i++) {
					StringsPair pair = (StringsPair) options.get(i);
					pair.setValue(localizer.i18n(bundle, pair.getValue()));
				}
			}
		}
		if (options == null || options.isEmpty()) {
			return null;
		}
		return options;
	}

	public List<StringsPair> getOptions(RequestContext context, String pickerName, String currentValue) {
		List<StringsPair> options = null;
		try {
			options = getReportParamOptions(context, currentValue, pickerName, "");
		} catch (Exception ex) {

		} finally {
			if (options != null) {
				Localizer localizer = SessionUtil.getLocalizer(context.getLocaleContext());
				ResourceBundle bundle = ReportParametersDesignPC.getReportBundle();
				for (int i = 0; i < options.size(); i++) {
					StringsPair pair = (StringsPair) options.get(i);
					pair.setValue(localizer.i18n(bundle, pair.getValue()));
				}
			}
		}

		return options;
	}

	public String getControlPickerName(RequestContext context, Node controlNode) {
		String pickerName = rfsReportControls.getControlPickerName(controlNode);
		String filterBy = reportControls.getControlFilterBy(controlNode);
		// Get picker used for resources, forecast/history/duration/volume
		if (RfsReportParameterConstants.RESOURCES.equals(pickerName)) {
			String resourcesType = context.getRequest().getParameter(RfsReportParameterConstants.PARAM_RESOURCETYPE);
			if (resourcesType == null || resourcesType.isEmpty()) {
				resourcesType = RfsReportParameterConstants.FORECAST_RESOURCES;
			}
			if (RfsReportParameterConstants.FILTER_DURATION_FORECAST.equals(filterBy)) {
				pickerName = RfsReportParameterConstants.DURATION_FORECAST;
			} else if (RfsReportParameterConstants.FORECAST_RESOURCES.equals(resourcesType)
					&& RfsReportParameterConstants.FILTER_VOLUME_FORECAST.equals(filterBy)) {
				pickerName = RfsReportParameterConstants.VOLUME_FORECAST;
			} else {
				pickerName = resourcesType;
			}
		} else if (RfsReportParameterConstants.CATEGORY.equals(pickerName)) {
			// Get picker used for category - transaction category/resource category
			String categoryType = context.getRequest().getParameter(RfsReportParameterConstants.PARAM_CATEGORYTYPE);
			if (categoryType == null || categoryType.isEmpty() || RfsReportParameterConstants.CATEGORYTYPE_RESOURCE.equals(categoryType)) {
				pickerName = RfsReportParameterConstants.RESOURCE_CATEGORY;
			} else if (RfsReportParameterConstants.CATEGORYTYPE_TRANSACTION.equals(categoryType)) {
				pickerName = RfsReportParameterConstants.TRANSACTION_CATEGORY;
			}
		} else if (RfsReportParameterConstants.FORECAST_RESOURCES.equals(pickerName) && !(filterBy == null || filterBy.isEmpty())) {
			// Get picker used for forecast resources
			pickerName = filterBy;
		}
		if (RfsReportParameterConstants.WORKGROUP.equals(pickerName) && !rfsReportControls.isIncludeAll(controlNode)) {
			pickerName = RfsReportParameterConstants.WORKGROUP_SINGLESELECTION;
		}
		return pickerName;
	}

	public String getJavaScript(String pickerName) {
		if (RfsReportParameterConstants.SCENARIO.equals(pickerName) || RfsReportParameterConstants.BUSINESS_UNIT.equals(pickerName)
				|| RfsReportParameterConstants.RESOURCE_CLASS.equals(pickerName)
				|| RfsReportParameterConstants.PLANNED_STAFFMIX.equals(pickerName)
				|| RfsReportParameterConstants.BRANCH_VIEW.equals(pickerName)) {
			return JSUtil.ON_CHANGE + JSUtil.REFRESH_PAGE_NO_CONFIRM;
		}
		return JSUtil.ON_CHANGE;
	}

	public RfsStaffMixFilter getStaffMixFilter(RequestContext context) {
		RfsStaffMixFilter staffMixFilter = new RfsStaffMixFilter();
		String staffMixFilterId = context.getRequest().getParameter("StaffMixFilter");
		try {
			if (staffMixFilterId != null && !staffMixFilterId.isEmpty()) {
				staffMixFilter = getStaffMixFilterInfo(context);
			} else {
				staffMixFilter = getDefaultStaffMixFilter(context);
			}
		} catch (Exception e) {
			mCat.error("Error getting staff mix filter details" + e);
		}
		return staffMixFilter;
	}

	public RfsStaffMixFilter getStaffMixFilterInfo(RequestContext context) throws BbmException, RemoteException {
		String staffMixFilter = context.getRequest().getParameter("StaffMixFilter");
		String currentScenario = rfsReportParametersModelHandler.getCurrentScenario(context);
		return rfsReportParametersModelHandler.getStaffMixFilterDetails(currentScenario, staffMixFilter);
	}

	public RfsStaffMixFilter getDefaultStaffMixFilter(RequestContext context) {
		RfsStaffMixFilter staffMixFilter = new RfsStaffMixFilter();
		List<StringsPair> empCategories = getOptions(context, RfsReportParameterConstants.EMPLOYEE_CATEGORIES, "");
		StringBuilder minHeadCounts = new StringBuilder("0");
		StringBuilder maxHeadCounts = new StringBuilder("1000");
		StringBuilder minPercentHeadCounts = new StringBuilder("0");
		StringBuilder maxPercentHeadCounts = new StringBuilder("100");
		StringBuilder minPercentFte = new StringBuilder("0");
		StringBuilder maxPercentFte = new StringBuilder("100");
		StringBuilder headCountsBelowActual = new StringBuilder("0");
		StringBuilder headCountsAboveActual = new StringBuilder("1000");
		StringBuilder headCountsBelowBudgeted = new StringBuilder("0");
		StringBuilder headCountsAboveBudgeted = new StringBuilder("1000");
		StringBuilder customPercentFte = new StringBuilder("0");

		if (empCategories != null) {
			for (int i = 1; i < empCategories.size(); i++) {
				minHeadCounts.append(",0");
				maxHeadCounts.append(",1000");
				minPercentHeadCounts.append(",0");
				maxPercentHeadCounts.append(",100");
				minPercentFte.append(",0");
				maxPercentFte.append(",100");
				headCountsBelowActual.append(",0");
				headCountsAboveActual.append(",1000");
				headCountsBelowBudgeted.append(",0");
				headCountsAboveBudgeted.append(",1000");
				customPercentFte.append(",0");
			}
		}
		staffMixFilter.setMinHeadCounts(minHeadCounts.toString());
		staffMixFilter.setMaxHeadCounts(maxHeadCounts.toString());
		staffMixFilter.setMinPercentHeadCounts(minPercentHeadCounts.toString());
		staffMixFilter.setMaxPercentHeadCounts(maxPercentHeadCounts.toString());
		staffMixFilter.setMinPercentFte(minPercentFte.toString());
		staffMixFilter.setMaxPercentFte(maxPercentFte.toString());
		staffMixFilter.setHeadCountsBelowActual(headCountsBelowActual.toString());
		staffMixFilter.setHeadCountsAboveActual(headCountsAboveActual.toString());
		staffMixFilter.setHeadCountsBelowBudgeted(headCountsBelowBudgeted.toString());
		staffMixFilter.setHeadCountsAboveBudgeted(headCountsAboveBudgeted.toString());
		staffMixFilter.setCustomPercentFte(customPercentFte.toString());
		staffMixFilter.setMaxOverstaffingRatio(1000);
		staffMixFilter.setMaxStaffMixesAfterLowCost(50);
		staffMixFilter.setIncludeClosestToActual(false);
		staffMixFilter.setIncludeClosestToBudgeted(false);
		staffMixFilter.setIncludeClosestToFte(false);
		return staffMixFilter;
	}
