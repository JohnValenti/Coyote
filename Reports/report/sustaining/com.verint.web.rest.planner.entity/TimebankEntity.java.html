package com.verint.web.rest.planner.entity;

import javax.xml.bind.annotation.XmlRootElement;

@XmlRootElement(name = "timebankdata")
public class TimebankEntity {

	private int orgId;
	private String name;
	private String startTime;
	private String endTime;
	private int dayOfWeek;
	private String weekNum;
	private String weekHour;


	public int getOrgId() {
		return orgId;
	}

	public void setOrgId() {
		this.orgId = orgId;
	}

	public String getName() {
		return name;
	}

	public void setName() {
		this.name = name;
	}

	public int getDayOfWeek() {
		return dayOfWeek;
	}

	public void setDayOfWeek() {
		this.dayOfWeek = dayOfWeek;
	}

	public String getWeekNum() {
		return weekNum;
	}

	public void setWeekNum() {
		this.weekNum = weekNum;
	}

	public String getWeekHour() {
		return weekHour;
	}

	public void setWeekHour() {
		this.weekHour = weekHour;
	}

	public String getStartTime() {
		return startTime;
	}

	/**
	 * @param startTime - Format yyyy-MM-dd-Thh:mm:ssZ
	 */
	public void setStartTime(String startTime) {
		this.startTime = startTime;
	}

	public String getEndTime() {
		return endTime;
	}

	/**
	 * @param endTime - Format yyyy-MM-dd-Thh:mm:ssZ
	 */
	public void setEndTime(String endTime) {
		this.endTime = endTime;
	}

}
