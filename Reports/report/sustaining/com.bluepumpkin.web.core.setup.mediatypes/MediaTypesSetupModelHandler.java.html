package com.bluepumpkin.web.core.setup.mediatypes;

import java.util.Collection;
import java.util.HashMap;

import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.ejb.core.mediatypes.ejb.MediaTypesManager;
import com.witness.web.uif.base.ModelHandler;

public class MediaTypesSetupModelHandler extends ModelHandler {
	
	public static Collection getAllMediaTypes() throws Exception {
		MediaTypesManager manager = WfmManagerFactory.getMediaTypesManager();
		return manager.getAllMediaTypes();   
	} 
    
    public static void updateTimeUnit(HashMap mediaUnits) throws Exception 
    {
        MediaTypesManager manager = WfmManagerFactory.getMediaTypesManager();
        manager.updateTimeUnit(mediaUnits);   
    } 
    
    public static void updateTimeUnitBySID(HashMap mediaUnits) throws Exception 
    {
        MediaTypesManager manager = WfmManagerFactory.getMediaTypesManager();
        manager.updateTimeUnitBySID(mediaUnits);   
    } 
}
