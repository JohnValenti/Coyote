package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectRich;

public class ShiftGroup extends ValueObjectRich {
	private static final long serialVersionUID = 1L;

	private static FieldInfo m_fieldInfo = new ShiftGroupFieldInfo();
	
	public static String SHIFTGROUPSHIFT_RELATIONSHIP_KEY = "SHIFTGROUPSHIFT|SHIFTGROUPID";
	
	public String getName() { return getFormattedFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_NAME); }
	public void setName(String name) { setFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_NAME, name); }
	
	public int getBlocksize() { return getFieldValueInt(ShiftGroupFieldInfo.SHIFTGROUP_BLOCKSIZE); }
	public void setBlocksize(int blocksize) { setFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_BLOCKSIZE, blocksize); }
	
	public Date getCreatedDate() { return getFieldValueDate(ShiftGroupFieldInfo.SHIFTGROUP_CREATED); }
	public void setCreatedDate(Date createdDate) { setFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_CREATED, createdDate); }
	
	public String getCreatedBy() { return getFormattedFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_CREATEDBY); }
	public void setCreatedBy(String createdBy) { setFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_CREATEDBY, createdBy); }

	public Date getModifiedDate() { return getFieldValueDate(ShiftGroupFieldInfo.SHIFTGROUP_MODIFIED); }
	public void setModifiedDate(Date modifiedDate) { setFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_MODIFIED, modifiedDate); }
	
	public String getModifiedBy() { return getFormattedFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_MODIFIEDBY); }
	public void setModifiedBy(String modifiedBy) { setFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_MODIFIEDBY, modifiedBy); }
	
	public ID getShiftGroupTypeID() { return getFieldValueID(ShiftGroupFieldInfo.SHIFTGROUP_SHIFTGROUPTYPEID); }
	public void setShiftGroupTypeID(ID shiftGroupTypeID) { setFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_SHIFTGROUPTYPEID, shiftGroupTypeID); }
	
	public ID getShiftPatternID() { return getFieldValueID(ShiftGroupFieldInfo.SHIFTGROUP_SHIFTPATTERNID); }
	public void setShiftPatternID(ID shiftPatternID) { setFieldValue(ShiftGroupFieldInfo.SHIFTGROUP_SHIFTPATTERNID, shiftPatternID); }
	
	public Collection<ShiftGroupShift> getShiftGroupShifts() { return getRelatedObjects(SHIFTGROUPSHIFT_RELATIONSHIP_KEY); }
	public void setShiftGroupShifts(Collection<ShiftGroupShift> groupShifts) { setRelatedObjects(SHIFTGROUPSHIFT_RELATIONSHIP_KEY, groupShifts); }

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}
}
