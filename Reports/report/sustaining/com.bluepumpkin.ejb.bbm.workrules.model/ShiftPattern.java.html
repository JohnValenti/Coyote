	public enum StartTime {
		StartOfShift(0),
		EndOfShift(1),
		StartOrEndOfShift(2);

		private final int intVal;
		private StartTime(int val) {
			intVal = val;
		}
		public int getIntVal() { return intVal; }
		public static StartTime fromIntVal(int intVal) {
			for (StartTime st : StartTime.values()) {
				if (intVal == st.intVal) return st;
			}
			System.err.println("Unknown " + StartTime.class.getName() + " value of " + intVal);
			return StartOrEndOfShift;
		}
	}

	public boolean isConsistentShiftActivities(ID shiftSID) {
		ShiftPatternShiftAttributes shiftAttributes = getShiftPatternShiftAttributesByShiftSID(shiftSID);
		if (shiftAttributes != null) {
			return shiftAttributes.isConsistentShiftActivities();
		}
		return false;		
	}
	
	public int getStartGroup(ID shiftSID, int nDay) {
		ShiftPatternShiftAttributes shiftAttributes = getShiftPatternShiftAttributesByShiftSID(shiftSID);
		if (shiftAttributes != null) {
			Integer startGroup = shiftAttributes.getConsistentShiftGroupMap().get(nDay); //QC57334
			if (startGroup != null) {
				return startGroup.intValue();
			}
		}
		return NOSTARTGROUP;
	}
	
	public Collection<ID> getPossibleShiftIDs(int nDay) {
		ArrayList<ID> possibleShiftIDs = new ArrayList<ID>();

		for(Map.Entry<Shift, ShiftPatternShiftAttributes> entry : getShiftPatternShiftAttributes().entrySet()) {
			if (entry.getValue().getOccurrenceDays().contains(nDay) || entry.getValue().getConsistentShiftGroupMap().containsKey(nDay)) {
				possibleShiftIDs.add(entry.getKey().getID());
			}
		}
		
		return possibleShiftIDs;
	}

	public HashMap<Shift, ShiftPatternShiftAttributes> getShiftPatternShiftAttributes() {
		//Lazy initialization of shiftPatternShiftAttributes
		if (shiftPatternShiftAttributes == null) {
			shiftPatternShiftAttributes = new HashMap<Shift, ShiftPatternShiftAttributes>();
			for (Shift shift : getShifts()) {
				ShiftPatternShiftAttributes attributes = new ShiftPatternShiftAttributes();
				attributes.setMaxConsecutiveDay(0);
				attributes.setMinConsecutiveDay(0);
				ArrayList<ShiftGroupShift> shiftGroupShifts = getShiftGroupShiftsForShift(shift.getDEID());
				
				for (ShiftGroupShift sgs : shiftGroupShifts) {
					if (getShiftGroupByID(sgs.getShiftGroupID()).getShiftGroupTypeID().toString().equals(
							ShiftGroupFieldInfo.SHIFT_GROUP_TYPE_ID_CONSISTENT_START)) {	//consistency
						if (sgs.getDaySequence() == 1000) {
							attributes.setisConsistentShiftActivities(true);
							continue;
						} else {
							attributes.getConsistentShiftGroupMap().put(sgs.getDaySequence(), sgs.getValue());
						}
					}
					if (getShiftGroupByID(sgs.getShiftGroupID()).getShiftGroupTypeID().toString().equals(
							ShiftGroupFieldInfo.SHIFT_GROUP_TYPE_ID_ON)) {	//work days
						if (sgs.getDaySequence() == 1000) {
							attributes.setMinConsecutiveDay(sgs.getValue());
							continue;
						} else if (sgs.getDaySequence() == 1001) {
							attributes.setMaxConsecutiveDay(sgs.getValue());
							continue;							
						} else {
							attributes.getOccurrenceDays().add(sgs.getDaySequence());
						}
					}
				}
				
				shiftPatternShiftAttributes.put(shift, attributes);
			}
		}
		
		return shiftPatternShiftAttributes;
	}
	
	public void setShiftPatternShiftAttributes(HashMap<Shift, ShiftPatternShiftAttributes> shiftAttributes) {

	private ArrayList<ShiftGroupShift> getShiftGroupShiftsForShift(ID shiftID) {
		ArrayList<ShiftGroupShift> shiftGroupShifts = new ArrayList<ShiftGroupShift>();
		
		for (ShiftGroup sg : getShiftGroups()) {
			for (ShiftGroupShift sgs : sg.getShiftGroupShifts()) {
				if (sgs.getShiftID().equals(shiftID)) {
					shiftGroupShifts.add(sgs);
				}
			}
		}
		return shiftGroupShifts;
	}
	
	private ShiftGroup getShiftGroupByID(ID shiftGroupID) {
		for (ShiftGroup sg : getShiftGroups()) {
			if (sg.getID().equals(shiftGroupID)) {
				return sg;
			}
		}
		return null;
	}
	
	private ShiftPatternShiftAttributes getShiftPatternShiftAttributesByShiftSID(ID shiftID) {
		for (Shift shift : getShiftPatternShiftAttributes().keySet()) {
			if (shift.getID().equals(shiftID)) {
				return getShiftPatternShiftAttributes().get(shift);
			}
		}
		return null;
	}
	
	public int getMinConsecutiveDays(ID shiftID) {
		return getMinMaxConsecutiveDays(shiftID, true);
	}
	
	public int getMaxConsecutiveDays(ID shiftID) {
		return getMinMaxConsecutiveDays(shiftID, false);
	}
	
	private int getMinMaxConsecutiveDays(ID shiftID, boolean isMin) {
		int nDays = -1; //no min/max consecutive days set
		Collection shiftGroupShifts = this.getShiftGroupShiftsForShift(shiftID);
		if (shiftGroupShifts != null) {
			ShiftGroupShift sgs = null;
			for (Iterator<ShiftGroupShift> i = shiftGroupShifts.iterator(); i.hasNext(); ) {
				sgs = (ShiftGroupShift)i.next();
				if (isMin && sgs.getDaySequence() == 1000 || !isMin && sgs.getDaySequence() == 1001) { //1000 is min consecutive days, 1001 IS MAX
					nDays = sgs.getValue();
					break;
				}
			}
		}
		return nDays;
	}
	
	/**
	 * Returns the period of the shift pattern.  Standard shift patterns are weekly, but if the
	 * monthly work rule license is active they could also have twenty eight, twenty nine, thirty, or
	 * thirty one day shift patterns.
	 */
	public ShiftPatternPeriod getPeriod() {
		int numOfWeeks = getNumOfWeeks();
		
		if (numOfWeeks <= 7) return ShiftPatternPeriod.Week;
		if (numOfWeeks == 28) return ShiftPatternPeriod.TwentyEightDayMonth;
		if (numOfWeeks == 29) return ShiftPatternPeriod.TwentyNineDayMonth;
		if (numOfWeeks == 30) return ShiftPatternPeriod.ThirtyDayMonth;
		if (numOfWeeks == 31) return ShiftPatternPeriod.ThirtyOneDayMonth;
		return ShiftPatternPeriod.Week;
	}

	public boolean isMonthlyShiftPattern() {
		return getPeriod() != ShiftPatternPeriod.Week;
	}
	
	/**
	 * Sets the period of the shift pattern.
	 */
	public void setPeriod(ShiftPatternPeriod period) {
		setNumOfWeeks(period.getDaysInPeriod().length);
	}
