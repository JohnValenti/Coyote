package com.bluepumpkin.ejb.bbm.workrules.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class ShiftFieldInfo extends FieldInfo {
	private static final long serialVersionUID = 1L;

	// field id constants
	public static final int SHIFT_ID = 0;
	public static final int SHIFT_SID = 1;
	public static final int SHIFT_CAMPAIGNID = 2; 	//STRING ID, SHOULD BE REPLACE WITH INTEGET ID LATER.
	public static final int SHIFT_ACTIVITYID = 3;	// The ID of the activity ( shift event ) as it appears at the SHIFTEVENT table.	
	public static final int SHIFT_NAME = 4; 		//THE NAME OF THE SHIFT EVENT
	public static final int SHIFT_DESCRIPTION = 5;
	public static final int SHIFT_SHIFTLENGTH = 6;
	public static final int SHIFT_MINSPACING = 7;
	public static final int SHIFT_MAXSPACING = 8;
	public static final int SHIFT_ORGANIZATIONID = 9;
	public static final int SHIFT_DELETEONDATE = 10;
	public static final int SHIFT_CREATED = 11;
	public static final int SHIFT_CREATEDBY = 12;
	public static final int SHIFT_MODIFIED = 13;
	public static final int SHIFT_MODIFIEDBY = 14;

	public static final int SHIFT_NUMOFFIELDS = 15;
	private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[SHIFT_NUMOFFIELDS];
	private static String m_strTableName = "SHIFT";

	public static final int SHIFT_NAME_MAX_LENGTH = 50;

	static {
		m_arrFieldInfo[SHIFT_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
		m_arrFieldInfo[SHIFT_SID] = new FieldInfoEntry("SID", PrimitiveTypes.ID, false);
		m_arrFieldInfo[SHIFT_CAMPAIGNID] = new FieldInfoEntry("CAMPAIGNID", PrimitiveTypes.ID);
		m_arrFieldInfo[SHIFT_ACTIVITYID] = new FieldInfoEntry("ACTIVITYID", PrimitiveTypes.ID);
		m_arrFieldInfo[SHIFT_NAME] = new FieldInfoEntry("NAME", PrimitiveTypes.STRING);
		m_arrFieldInfo[SHIFT_DESCRIPTION] = new FieldInfoEntry("DESCRIPTION", PrimitiveTypes.STRING);
		m_arrFieldInfo[SHIFT_SHIFTLENGTH] = new FieldInfoEntry("SHIFTLENGTH", PrimitiveTypes.INT);
		m_arrFieldInfo[SHIFT_MINSPACING] = new FieldInfoEntry("MINSPACING", PrimitiveTypes.INT);
		m_arrFieldInfo[SHIFT_MAXSPACING] = new FieldInfoEntry("MAXSPACING", PrimitiveTypes.INT);
		m_arrFieldInfo[SHIFT_ORGANIZATIONID] = new FieldInfoEntry("ORGANIZATIONID", PrimitiveTypes.ID);	
		m_arrFieldInfo[SHIFT_DELETEONDATE] = new FieldInfoEntry("DELETEONDATE", PrimitiveTypes.DATE);
		m_arrFieldInfo[SHIFT_CREATED] = new FieldInfoEntry("CREATED", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[SHIFT_CREATEDBY] = new FieldInfoEntry("CREATEDBY", PrimitiveTypes.STRING);
		m_arrFieldInfo[SHIFT_MODIFIED] = new FieldInfoEntry("MODIFIED", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[SHIFT_MODIFIEDBY] = new FieldInfoEntry("MODIFIEDBY", PrimitiveTypes.STRING);
	}
	
	// implement abstract methods in parent class
	public String getTableName() { return m_strTableName; }
	public int getIDFieldIndex() { return SHIFT_SID; }
	public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }

	public static int POSSIBLE_DAYS_OFF_SHIFT_ID = 1;
}
