package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.HashMap;
import java.util.HashSet;
import java.util.Set;

public class ShiftPatternShiftAttributes implements java.io.Serializable {
	private static final long serialVersionUID = 1L;
	
	private int m_minConsecutiveDay = 0;
	private boolean m_minConsecutiveDayMultiEdit = false;
	private int m_maxConsecutiveDay = 0;
	private boolean m_maxConsecutiveDayMultiEdit = false;
	private boolean m_isConsistentShiftActivities = false;
	private boolean m_consistentShiftActivityMultiEdit = false;

	//Set of integers representing the days of the time period (weekly or monthly) where this shift can occur.
	//If it is in the set then the shift is enabled (ON) for that particular day
	private HashSet<Integer> m_occurrenceDays = new HashSet<Integer>();
	//Hashmap key = Integer representing a day in the period, value = boolean that tracks which days of the pattern have occurrences of this shift
	//that differ from each other for the same day in the period.
	private HashMap<Integer, Boolean> m_occurrenceDaysMultiEdit = new HashMap<Integer, Boolean>();
	//Hashmap key = Integer representing a day in the period, value = integer representing shift group
	private HashMap<Integer, Integer> m_consistentShiftGroupMap = new HashMap<Integer, Integer>();
	//This map is used to track which days of the pattern have consistent shift groups that are being multi-edited
	//(i.e. the multi-selection has values that differ from each other for the same day in the period).
	private HashMap<Integer, Boolean> m_consistentShiftGroupMultiEdit = new HashMap<Integer, Boolean>();
	
	public ShiftPatternShiftAttributes() {}
	
	public int getMinConsecutiveDay() {
		return m_minConsecutiveDay;
	}

	public int getMaxConsecutiveDay() {
		return m_maxConsecutiveDay;
	}
	
	public boolean isConsistentShiftActivities(){
		return m_isConsistentShiftActivities;
	}
	
	public void setMinConsecutiveDay(int minConsecutiveDay) {
		m_minConsecutiveDay = minConsecutiveDay;
	}
	
	public void setMaxConsecutiveDay(int maxConsecutiveDay) {
		m_maxConsecutiveDay = maxConsecutiveDay;
	}
	
	public void setisConsistentShiftActivities(boolean isConsistentShiftActivities) {
		m_isConsistentShiftActivities = isConsistentShiftActivities;
	}
	
	public Set<Integer> getOccurrenceDays() {
		return m_occurrenceDays;
	}
	
	public HashMap<Integer, Integer> getConsistentShiftGroupMap() {
		return m_consistentShiftGroupMap;
	}
	
	public void setConsistentShiftGroupMap(HashMap<Integer, Integer> shiftGroupMap) {
		m_consistentShiftGroupMap = shiftGroupMap;
	}
	
	public void setIsConsistentShiftActivitesMultiEdit(boolean isMultiEdit) {
		m_consistentShiftActivityMultiEdit = isMultiEdit;
	}
	
	public boolean isConsistentShiftActivitiesMultiEdit() {
		return m_consistentShiftActivityMultiEdit;
	}
	
	public void setIsMinConsecutiveDaysMultiEdit(boolean isMultiEdit) {
		m_minConsecutiveDayMultiEdit = isMultiEdit;
	}
	
	public boolean isMinConsecutiveDaysMultiEdit() {
		return m_minConsecutiveDayMultiEdit;
	}
	
	public void setIsMaxConsecutiveDaysMultiEdit(boolean isMultiEdit) {
		m_maxConsecutiveDayMultiEdit = isMultiEdit;
	}
	
	public boolean isMaxConsecutiveDaysMultiEdit() {
		return m_maxConsecutiveDayMultiEdit;
	}
	
	public boolean isOccurrenceDayMultiEdit(Integer day) {
		if (m_occurrenceDaysMultiEdit.get(day) != null) {
			return m_occurrenceDaysMultiEdit.get(day);
		}
		return false;		
	}
	
	public void setOccurrenceDayMultiEdit(Integer day, boolean isMultiEdit) {
		m_occurrenceDaysMultiEdit.put(day, isMultiEdit);
	}
	
	public boolean isConsistentShiftGroupMultiEdit(Integer day) {
		if (m_consistentShiftGroupMultiEdit.get(day) != null) {
			return m_consistentShiftGroupMultiEdit.get(day);
		}
		return false;
	}
	
	public void setConsistentShiftGroupMultiEdit(Integer day, boolean isMultiEdit) {
		m_consistentShiftGroupMultiEdit.put(day, isMultiEdit);
	}
	
	/**
	 * Returns a cloned copy of this ShiftPatternShiftAttributes object.
	 */
	public ShiftPatternShiftAttributes clone() {
		ShiftPatternShiftAttributes clonedAttributes = new ShiftPatternShiftAttributes();
		clonedAttributes.setConsistentShiftGroupMap(
				new HashMap<Integer, Integer>(m_consistentShiftGroupMap));
		clonedAttributes.setConsistentShiftGroupMultiEditMap(
				new HashMap<Integer, Boolean>(m_consistentShiftGroupMultiEdit));
		clonedAttributes.setIsConsistentShiftActivitesMultiEdit(m_consistentShiftActivityMultiEdit);
		clonedAttributes.setisConsistentShiftActivities(m_isConsistentShiftActivities);
		clonedAttributes.setIsMaxConsecutiveDaysMultiEdit(m_maxConsecutiveDayMultiEdit);
		clonedAttributes.setIsMinConsecutiveDaysMultiEdit(m_minConsecutiveDayMultiEdit);
		clonedAttributes.setMaxConsecutiveDay(m_maxConsecutiveDay);
		clonedAttributes.setMinConsecutiveDay(m_minConsecutiveDay);
		clonedAttributes.setOccurrenceDayMultiEditMap(new HashMap<Integer, Boolean>(m_occurrenceDaysMultiEdit));
		clonedAttributes.setOccurrenceDays(new HashSet<Integer>(m_occurrenceDays));
		
		return clonedAttributes;
	}
	//The following 3 methods are used only for the clone method above
	private void setOccurrenceDayMultiEditMap(HashMap<Integer, Boolean> occurrenceDayMultiEditMap) {
		m_occurrenceDaysMultiEdit = occurrenceDayMultiEditMap;
	}
	private void setConsistentShiftGroupMultiEditMap(HashMap<Integer, Boolean> consistentShiftGroupMultiEditMap) {
		m_consistentShiftGroupMultiEdit = consistentShiftGroupMultiEditMap;
	}
	private void setOccurrenceDays(HashSet<Integer> occurrenceDays) {
		m_occurrenceDays = occurrenceDays;
	}
}
