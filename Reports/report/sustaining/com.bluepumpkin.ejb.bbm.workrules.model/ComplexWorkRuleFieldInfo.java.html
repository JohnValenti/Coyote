package com.bluepumpkin.ejb.bbm.workrules.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class ComplexWorkRuleFieldInfo extends FieldInfo {
	private static final long serialVersionUID = 2L;

	public static enum Column {
		ID("ID", PrimitiveTypes.ID),
		SID("SID", PrimitiveTypes.ID),
		OrganizationID("ORGANIZATIONID", PrimitiveTypes.ID),
		CalendarPeriodID("CALENDARPERIODID", PrimitiveTypes.ID),
		WorkUnitID("WORKUNITID", PrimitiveTypes.ID),
		Name("NAME", PrimitiveTypes.STRING),
		IsAnOffRule("ISANOFFRULE", PrimitiveTypes.BOOLEAN),
		Tolerance("TOLERANCE", PrimitiveTypes.FLOAT),
		Priority("PRIORITY", PrimitiveTypes.SHORT),
		
		//	 Start Date on COMPLEXWORKRULE is not used. Instead use the ALIGNMENTDATE on the CALENDARPERIOD table
		//StartDate("STARTDATE", PrimitiveTypes.DATETIME),
		
		EndDate("ENDDATE", PrimitiveTypes.DATETIME),
		WorkUnitCountConstraint("WORKUNITCOUNTCONSTRAINT", PrimitiveTypes.INT),
		MinWorkUnitCount("MINWORKUNITCOUNT", PrimitiveTypes.INT),
		MaxWorkUnitCount("MAXWORKUNITCOUNT", PrimitiveTypes.INT),
		CreatedBy("CREATEDBY", PrimitiveTypes.STRING),
		CreatedOn("CREATED", PrimitiveTypes.DATETIME),
		ModifiedBy("MODIFIEDBY", PrimitiveTypes.STRING),
		ModifiedOn("MODIFIED", PrimitiveTypes.DATETIME)
		;
		
		// Static block to set field array indexes
		static {
			int ctr = 0;
			for (Column fie : Column.values()) fie.index = ctr++;
		}

		// Column Properties
		private int index;
		private FieldInfoEntry fie;

		// Constructor
		private Column(String name, int type) {
			this.fie = new FieldInfoEntry(name, type);
		}
		
		public FieldInfoEntry getFieldInfoEntry() {
			return this.fie;
		}
		public int getIndex() {
			return index;
		}
	}

	public static final int NAME_MAX_LENGTH = 50;

	private static final FieldInfoEntry[] fieldInfoEntries;
	static {
		int ctr = 0;
		fieldInfoEntries = new FieldInfoEntry[Column.values().length];
		for (Column c : Column.values()) {
			fieldInfoEntries[ctr++] = c.getFieldInfoEntry();
		}
	}
	
	public String getTableName() { return "COMPLEXWORKRULE"; }
	public int getIDFieldIndex() { return Column.SID.getIndex(); }
	public FieldInfoEntry [] getFieldInfo() { return fieldInfoEntries; }

	@Override
	public int getParentIDFieldIndex() {
		return Column.OrganizationID.getIndex();
	}
	
}
