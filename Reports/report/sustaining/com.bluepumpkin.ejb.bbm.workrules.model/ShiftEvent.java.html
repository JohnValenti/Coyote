package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.Duration;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectRich;

public class ShiftEvent extends ValueObjectRich {
	private static final long serialVersionUID = 1L;
	
	// meta data information
	private static FieldInfo m_fieldInfo = new ShiftEventFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	public static String ACTIVITY_RELATIONSHIP_KEY = "ACTIVITY|ID";
	public static String SHIFT_EVENT_ACTIVITY_RELATIONSHIP_KEY = "SHIFTEVENTACTIVITY|SHIFTEVENTID";
	
	public ID getDEID() {
		return getFieldValueID(ShiftEventFieldInfo.SHIFTEVENT_ID);
	}
	
	public void setDEID(ID deid) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_ID, deid);
	}
	
	public ID getCampaignID() {
		return this.getFieldValueID(ShiftEventFieldInfo.SHIFTEVENT_CAMPAIGNID);
	}
	
	public void setCampaignID(ID campaignID) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_CAMPAIGNID, campaignID);
	}
	
	public ID getActivityID() {
		return this.getFieldValueID(ShiftEventFieldInfo.SHIFTEVENT_ACTIVITYID);
	}
	
	public void setActivityID(ID activityID) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_ACTIVITYID, activityID);
	}
	
	public ID getSID() {
		return this.getFieldValueID(ShiftEventFieldInfo.SHIFTEVENT_SID);
	}
	
	public void setSID(ID sID) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_SID, sID);
	}
	
	public String getName() {
		return this.getFormattedFieldValue(ShiftEventFieldInfo.SHIFTEVENT_NAME);
	}
	
	public void setName(String name) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_NAME, name);
	}
	
	/**
	 * If shift event is absolute - This value represents the minutes from the owner org/campaign's day boundary.
	 * If shift event is relative to shift start - this value represents the minutes from the shift start time.
	 * If shift event is anytime - this will return 0.
	 */
	public int getStart() {
		return this.getFieldValueInt(ShiftEventFieldInfo.SHIFTEVENT_WINDOWSTARTTIME);
	}

	public void setStart(int startMinutes) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_WINDOWSTARTTIME, startMinutes);
	}
	
	/**
	 * If shift event is absolute - This value represents the minutes from the owner org/campaign's day boundary.
	 * If shift event is relative to shift start - this value represents the minutes from the shift start time.
	 * If shift event is anytime - this will return 0.
	 */
	public int getEnd() {
		return this.getFieldValueInt(ShiftEventFieldInfo.SHIFTEVENT_WINDOWENDTIME);
	}

	public void setEnd(int endMinutes) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_WINDOWENDTIME, endMinutes);
	}
	
	public String getDescription() {
		return this.getFormattedFieldValue(ShiftEventFieldInfo.SHIFTEVENT_DESCRIPTION);
	}
	
	public void setDescription(String description) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_DESCRIPTION, description);
	}
	
	public boolean isPaid() {
		return this.getFieldValueBoolean(ShiftEventFieldInfo.SHIFTEVENT_PAID);
	}
	
	public void setIsPaid(boolean isPaid) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_PAID, isPaid);
	}
	
	public Duration getDuration() {
		return Duration.fromMinutes(this.getFieldValueInt(ShiftEventFieldInfo.SHIFTEVENT_LENGTH)) ;
	}
	
	public void setDuration(Duration duration) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_LENGTH, duration.getDurationInMinutes());
	}
	
	public boolean isConsistent() {
		return this.getFieldValueBoolean(ShiftEventFieldInfo.SHIFTEVENT_ISCONSISTENT);
	}
	
	public void setIsConsistent(boolean isConsistent) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_ISCONSISTENT, isConsistent);
	}
	
	public boolean isAnyTime() {
		return this.getFieldValueBoolean(ShiftEventFieldInfo.SHIFTEVENT_ISANYTIME);
	}
	
	public void setIsAnyTime(boolean isAnyTime) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_ISANYTIME, isAnyTime);
	}
	
	public boolean isCafeteria() {
		return this.getFieldValueBoolean(ShiftEventFieldInfo.SHIFTEVENT_ISCAFETERIA);
	}
	
	public void setIsCafeteria(boolean isCafeteria) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_ISCAFETERIA, isCafeteria);
	}
	
	public ID getOrganizationID() {
		return this.getFieldValueID(ShiftEventFieldInfo.SHIFTEVENT_ORGANIZATIONID);
	}
	
	public void setOrganizationID(ID orgID) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_ORGANIZATIONID, orgID);
	}

	public boolean isFlexible() {
		return this.getFieldValueBoolean(ShiftEventFieldInfo.SHIFTEVENT_ISFLEXIBLE);
	}
	
	public void setIsFlexible(boolean isFlexible) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_ISFLEXIBLE, isFlexible);
	}
	
	// TODO: convert this to an enum
	public int getWindowType() {
		return this.getFieldValueInt(ShiftEventFieldInfo.SHIFTEVENT_WINDOWTYPE);
	}
	
	public void setWindowType(int windowType) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_WINDOWTYPE, windowType);
	}
	
	public int getMinCount() {
		return this.getFieldValueInt(ShiftEventFieldInfo.SHIFTEVENT_MINCOUNT);
	}
	
	public void setMinCount(int minCount) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_MINCOUNT, minCount);
	}
	
	public int getMaxCount() {
		return this.getFieldValueInt(ShiftEventFieldInfo.SHIFTEVENT_MAXCOUNT);
	}
	
	public void setMaxCount(int maxCount) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_MAXCOUNT, maxCount);
	}
	
	public boolean isDeleted() {
		Date deleteOnDate = this.getFieldValueDate(ShiftEventFieldInfo.SHIFTEVENT_DELETEONDATE);
		if (deleteOnDate != null && (deleteOnDate.getTime() < System.currentTimeMillis()))
			return true;
		return false;
	}
	
	public Date getCreatedDate() {
		return getFieldValueDate(ShiftEventFieldInfo.SHIFTEVENT_CREATED);
	}
	public void setCreatedDate(Date createdDate) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_CREATED, createdDate);
	}
	
	public String getCreatedBy() {
		return getFormattedFieldValue(ShiftEventFieldInfo.SHIFTEVENT_CREATEDBY);
	}
	public void setCreatedBy(String createdBy) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_CREATEDBY, createdBy);
	}
	
	public Date getModifiedDate() {
		return getFieldValueDate(ShiftEventFieldInfo.SHIFTEVENT_MODIFIED);
	}
	
	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_MODIFIED, modifiedDate);
	}
	
	public String getModifiedBy() {
		return getFormattedFieldValue(ShiftEventFieldInfo.SHIFTEVENT_MODIFIEDBY);
	}
	public void setModifiedBy(String modifiedBy) {
		setFieldValue(ShiftEventFieldInfo.SHIFTEVENT_MODIFIEDBY, modifiedBy);
	}
	
	public Activity getActivity() {
		Collection<Activity> activity = (Collection<Activity>)getRelatedObjects(ACTIVITY_RELATIONSHIP_KEY);
		if (activity != null && activity.size() > 0) return activity.iterator().next();
		return null;
	}
	
	public Collection<Activity> getAdditionalActivities() {
		return getRelatedObjects(SHIFT_EVENT_ACTIVITY_RELATIONSHIP_KEY);
	}
	
	public void addAdditionalActivities(Collection<Activity> activitiesToAdd) {
		addRelatedObjects(SHIFT_EVENT_ACTIVITY_RELATIONSHIP_KEY, activitiesToAdd);
	}
	
	public void removeAdditionalActivities(Collection<Activity> activitiesToRemove) {
		removeRelatedObjects(SHIFT_EVENT_ACTIVITY_RELATIONSHIP_KEY, activitiesToRemove);
	}
	
	/** 
	 * Returns true if the shift event's activity has its "is used in shift" flag set.
	 * This type of activity is also called a "primary activity"
	 */
	public boolean isWorkType() {
		return getActivity().isUsedInShift();
	}
	
	public boolean isAdditionalActivity(ID activityID) {
		for (Activity activity: getAdditionalActivities()) {
			if (activity.getID().equals(activityID))
				return true;
		}
		return false;
	}
}
