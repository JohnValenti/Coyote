package com.bluepumpkin.ejb.bbm.workrules.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class ShiftEventActivityFieldInfo extends FieldInfo {
	private static final long serialVersionUID = 1L;

	public static enum ShiftEventActivityColumn {
		ID(new FieldInfoEntry("ID", PrimitiveTypes.ID)),
		ActivityID(new FieldInfoEntry("ACTIVITYID", PrimitiveTypes.ID)),
		ShiftEventID(new FieldInfoEntry("SHIFTEVENTID", PrimitiveTypes.ID)),
		Created(new FieldInfoEntry("CREATED", PrimitiveTypes.DATETIME)),
		CreatedBy(new FieldInfoEntry("CREATEDBY", PrimitiveTypes.STRING)),
		Modified(new FieldInfoEntry("MODIFIED", PrimitiveTypes.DATETIME)),
		ModifiedBy(new FieldInfoEntry("MODIFIEDBY", PrimitiveTypes.STRING));
		
		static {
			int ctr = 0;
			for (ShiftEventActivityColumn fie : ShiftEventActivityColumn.values()) fie.index = ctr++;
		}
		
		private int index;
		private FieldInfoEntry fie;
		private ShiftEventActivityColumn(FieldInfoEntry fie) {
			this.fie = fie;
		}
		
		public FieldInfoEntry getFieldInfoEntry() {
			return this.fie;
		}
		public int getIndex() {
			return index;
		}
	}
	
	private static final FieldInfoEntry[] fieldInfoEntries;
	static {
		int ctr = 0;
		fieldInfoEntries = new FieldInfoEntry[ShiftEventActivityColumn.values().length];
		for (ShiftEventActivityColumn c : ShiftEventActivityColumn.values()) {
			fieldInfoEntries[ctr++] = c.getFieldInfoEntry();
		}
	}
	
	@Override
	public FieldInfoEntry[] getFieldInfo() { return fieldInfoEntries; }

	@Override
	public int getIDFieldIndex() { return ShiftEventActivityColumn.ID.getIndex(); }

	@Override
	public String getTableName() { return "SHIFTEVENTACTIVITY"; }
}
