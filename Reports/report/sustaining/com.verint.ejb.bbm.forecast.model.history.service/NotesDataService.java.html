package com.verint.ejb.bbm.forecast.model.history.service;


import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.pulse.ejb.TrackingManager;
import com.bluepumpkin.ejb.bbm.pulse.model.PulseNoteBO;
import com.verint.ejb.bbm.forecast.model.history.service.INotesDataService;
import com.verint.ejb.wfm.WfmManagerFactory;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Date;

public class NotesDataService implements INotesDataService {
	@Override
	public Collection<PulseNoteBO> getNotes(Collection<ID> queueIDs, Date startDate, Date endDate) throws BbmException {
		try {
			TrackingManager tm = WfmManagerFactory.getPulseTrackingManager();
			return tm.getPulseNotes(queueIDs, startDate, endDate);
		} catch (RemoteException ex) {
			throw new BbmException(ex);
		}
	}
}
