import java.util.Date;
import java.util.HashMap;
import java.util.Map;
import java.util.StringTokenizer;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexRequestMakeup;


public class TimeOffIntervalPaidSegment extends ValueObjectBase implements Comparable<TimeOffIntervalPaidSegment> {

	private static final long serialVersionUID = 1980471623692331994L;
	private static FieldInfo fieldInfo = new TimeOffIntervalPaidSegmentFieldInfo();
	
	
	
	public TimeOffIntervalPaidSegment() {
		super();
	}
	
	public TimeOffIntervalPaidSegment(ID choiceID, Date start, Date end) {
		super();
		this.setTOChoiceID(choiceID);
		this.setStartTime(start);
		this.setEndTime(end);
	}
	
	public TimeOffIntervalPaidSegment(ID choiceID, TimeRange timeRange) {
		super();
		this.setTOChoiceID(choiceID);
		this.setStartTime(timeRange.getStartDate());
		this.setEndTime(timeRange.getEndDate());
		
	}
	
	@Override
	protected FieldInfo getFieldInfo() {
		return fieldInfo;
	}
	
	@Override
	public ID getID() {
		return getFieldValueID(TimeOffIntervalPaidSegmentFieldInfo.TIMEOFFINTERVALPAIDSEGMENT_I_ID);
	}

	@Override
	public void setID(ID id) {
		setFieldValue(TimeOffIntervalPaidSegmentFieldInfo.TIMEOFFINTERVALPAIDSEGMENT_I_ID, id);
	}

	public ID getTOChoiceID() {
		return getFieldValueID(TimeOffIntervalPaidSegmentFieldInfo.TIMEOFFINTERVALPAIDSEGMENT_I_TIMEOFFCHOICEID);
	}

	public void setTOChoiceID(ID toChoiceID) {
		setFieldValue(TimeOffIntervalPaidSegmentFieldInfo.TIMEOFFINTERVALPAIDSEGMENT_I_TIMEOFFCHOICEID, toChoiceID);
	}

	public void setStartTime(Date startDate) {
		setFieldValue(TimeOffIntervalPaidSegmentFieldInfo.TIMEOFFINTERVALPAIDSEGMENT_D_STARTTIME, startDate);
	}

	public Date getStartTime() {
		return getFieldValueDate(TimeOffIntervalPaidSegmentFieldInfo.TIMEOFFINTERVALPAIDSEGMENT_D_STARTTIME);
	}

	public void setEndTime(Date endDate) {
		setFieldValue(TimeOffIntervalPaidSegmentFieldInfo.TIMEOFFINTERVALPAIDSEGMENT_D_ENDTIME, endDate);
	}

	public Date getEndTime() {
		return getFieldValueDate(TimeOffIntervalPaidSegmentFieldInfo.TIMEOFFINTERVALPAIDSEGMENT_D_ENDTIME);
	}


	public int getDuration()
	{
		return (int) ((getEndTime().getTime() - getStartTime().getTime()) / (60 * 1000));
	}

	@Override
	public int compareTo(TimeOffIntervalPaidSegment obj) { // NOSONAR
				
		if( getTOChoiceID().toInt() != obj.getTOChoiceID().toInt()){
			return 1;
		}
		
		//Same choice. test the start/end times
		
		long diff = getStartTime().getTime() - obj.getStartTime().getTime();
		
		if (diff < 0) {
			return -1;
		} else if(diff > 0){
			return 1;
		}
		//start times equal at this point
		diff = getEndTime().getTime() - obj.getEndTime().getTime();
		
		if (diff < 0) {
			return -1;
		} else if (diff > 0){
			return 1;
		}
		//end times equal at this point
		
		// all things equal
		return 0;
	}


}
