public class MyTimeRequestHandler extends RequestHandler {

	public static final int SCHEDULE_EVENT = 1;
	public static final int HISTORY_EVENT = 0;
	public static final int STATUS_EVENT = 2;

	private static final int LAST_ACTIVITY_LIST_LIMIT = 20;

	private static Category cat  = Log.initCategory(MyTimeRequestHandler.class.getName());
	private MyTimeModel myTimeModel;
	private RequestContext context = null;
	private static IntegrationHook startShiftIntegration = null;
	private static IntegrationHook endShiftIntegration = null;

	static {
		startShiftIntegration = IntegrationHook.getConfigurationClass(IntegrationHook.MYTIME_START_SHIFT);
		endShiftIntegration = IntegrationHook.getConfigurationClass(IntegrationHook.MYTIME_END_SHIFT);
	}

	/**
	 *  Process the browser request from the my time screen.
	 *  Get a model handler for the model.
	 *  Extract the parameters from the request into the empty model.
	 *  Check if device_id exists in session
	 *  Set any entry default values.
	 *  Process the requested action from the page.
	 */
	public void processRequest(RequestContext requestContext)
	throws RemoteException, AmEJBCreateException, NamingException, CreateException, Exception{
		context = requestContext;
		String path = context.getRequest().getPathInfo();
        if (path.equals(MyTimeModel.CHANGE_REQUEST)
			&& !requestContext.getUser().isAuthorized(PrivilegeKeys.AM_MYTIMETIMERECORDS))
		{
			path = MyTimeModel.SCHEDULE_REQUEST;
			myTimeModel = MyTimeScheduleModel.getInstance(context);
		}
// test only
/*
			if ( startShiftIntegration != null ) {
				String str = startShiftIntegration.getHtmlString(context,IntegrationHook.MYTIME_START_SHIFT);
				context.addGenericHTML(str);
			}
			*/

		try {
			if (path.equals(MyTimeModel.SCHEDULE_REQUEST)){
				myTimeModel = MyTimeScheduleModel.getInstance(context);
			} else if (path.equals(MyTimeModel.CHANGE_REQUEST)){
				// Do not use getInstance here - we don't want to populate model
				// at this point - new transaction will be added by the
				// call to processAction.
				myTimeModel = new MyTimeRecordModel(context);
				context.setAttribute(RequestContext.REQUEST_SCOPE, MyTimeModel.MODEL_KEY, myTimeModel);
			} else if (path.equals(MyTimeModel.STATUS_REQUEST)){
				// Do not use getInstance here - we don't want to populate model
				// at this point - new transaction will be added by the
				// call to processAction.
				myTimeModel = new MyTimeAction(context);
				context.setAttribute(RequestContext.REQUEST_SCOPE, MyTimeModel.MODEL_KEY, myTimeModel);
			}
			processAction(context);
		} catch ( Exception e ) {
			cat.l7dError(AmWebLogBundleKey.UNABLE_TO_LOAD_DATA, e);
			myTimeModel.addPageMessage(Message.ERROR_TYPE,
															AmWebBundleKey.MSG_ERROR_UNABLETOLOADDATA,
															AmWebBundleKey.BUNDLE_NAME);
		}

		//if (cat.isInfoEnabled()) cat.info(myTimeModel.toString());
	}

	/**
	 * Process the action from the request: START_SHIFT/END_SHIFT
	 */
	private void processAction(RequestContext context) {
		if (isAction(context, MyTimeModel.START_ACTIVITY)) {
			processChangeShift();
		} else if (isAction(context, MyTimeModel.END_SHIFT)) {
			processEndShift(context);
		}else if (isAction(context, MyTimeRecordModel.CHANGE_REFRESH_RATE_ACTION)) {
			processChangeRefreshRate(context);
		}
		else if ( isAction(context,MyTimeModel.GET_STATUS)) {
			//processGetStatus();
		}
	}

	//This method will be invoked in case the user changes the refresh rate.
	// The new value is saved in the user profile and the page is processed as usual
	private void processChangeRefreshRate(RequestContext context) {
		// Get the new refresh rate from the form and det it to the user profile
		String refreshRate = context.getRequest().getParameter(MyTimeRecordModel.REFRESH_RATE_FN);
		context.setUserProperty(MyTimeRecordModel.REFRESH_RATE_USER_PROPERTY_KEY, refreshRate);
	}
	
    	/**
	 * deleteActivityFromProfile
	 * this method will be called if the user tries to select a deleted activity
	 * from the recent list
	 * The activity will be removed from the list
	 */
	private void deleteActivityFromProfile(ID activity){
		UserProfile aProfile = myTimeModel.getUserProfile();
		String addToProfileString = "";

		//check to see if there is a ACTIVITY_HISTORY key in the profile
		if (aProfile.getProperty("ACTIVITY_HISTORY")!=null){
			MyTimeProfileHelper aHelper = new MyTimeProfileHelper();
			//get string from profile
			ArrayList aFullList = aHelper.getArrListFromString(aProfile.getProperty("ACTIVITY_HISTORY"));

			//remove it if already exists
			for (int i=0; i<aFullList.size();i++){
				ArrayList aCheckList = (ArrayList) aFullList.get(i);

				if ((aCheckList.get(0)).equals(activity.toString())){
					aFullList.remove(aCheckList);
				}
			}

			addToProfileString = aHelper.getStrFromArrList(aFullList);
		}
		aProfile.setProperty("ACTIVITY_HISTORY", addToProfileString, true);
	}

	/**
	 * setLastActivity
	 * the system will save the recent 5 activities in the user profile
	 * this method will check if there is a change activity request, and
	 * if successful, it will overwrite the earliest activity with this one
	 * and save to the profile
	 */
	private void setLastActivity(){
		UserProfile aProfile = myTimeModel.getUserProfile();
		ArrayList aNewList= new ArrayList();
		//add the activity id, name, and order to the arraylist
		aNewList.add(0, (String.valueOf(myTimeModel.getActivity())));
		aNewList.add(1, myTimeModel.getActivityName());
		aNewList.add(2, String.valueOf(1));

		String addToProfileString = "";

		//check to see if there is a ACTIVITY_HISTORY key in the profile
		if (aProfile.getProperty("ACTIVITY_HISTORY")!=null){
			MyTimeProfileHelper aHelper = new MyTimeProfileHelper();
			//get string from profile
			ArrayList aFullList = aHelper.getArrListFromString(aProfile.getProperty("ACTIVITY_HISTORY"));

			//remove it if already exists
			for (int i=0; i<aFullList.size();i++){
				ArrayList aCheckList = (ArrayList) aFullList.get(i);
				if (aCheckList.get(0).equals(aNewList.get(0))){
					aFullList.remove(aCheckList);
				}
			}

			//add the new array list to the 0th position
			aFullList.add(0, aNewList);

			//remove last object, if length of ArrayList > LAST_ACTIVITY_LIST_LIMIT
			if (aFullList.size() > LAST_ACTIVITY_LIST_LIMIT){
				aFullList.remove(LAST_ACTIVITY_LIST_LIMIT);
			}
			addToProfileString = aHelper.getStrFromArrList(aFullList);
		} else {
			addToProfileString = String.valueOf(myTimeModel.getActivity()) + ","+
				myTimeModel.getActivityName()+","+ String.valueOf(1)+";";
		}
		aProfile.setProperty("ACTIVITY_HISTORY", addToProfileString, true);
        
        //Change starts for QA96514
		/*
		 * Following code checks if the request is not from the Time Record page (means it is from the My Schedule page)
		 * then it saves the userprofile with lates information (recent activities etc..) into session as well as database.
		 * In case of it is coming from Time record page this is not required as the full page is getting submitted and 
		 * during that process user profile is getting saved in database as well as session.    
		 */
		String sFromTimeRecord = context.getParameter("isFormTimeRecord");
		if(sFromTimeRecord != null && sFromTimeRecord.equalsIgnoreCase("false"))
			UserProfileManager.setUserProfile(context, aProfile);
		//Change ends for QA96514
	}

		/**
	 * Process change activity/shift
	 * This method takes care of start/change activity
	 * set appropriate page messages depending on the return value
	 */
	private void processChangeShift() {
		if ( myTimeModel.isValidSN() ) {
			if ( startShiftIntegration != null ) {
				String str = startShiftIntegration.generateHTML(context,IntegrationHook.MYTIME_START_SHIFT);
				context.addGenericHTML(str);
			}
			ID id = new ID(myTimeModel.getActivity());
			if ( id == Activity.ACTIVITY_NONE || id == Activity.ACTIVITY_UNKNOWN ) {
				myTimeModel.setErrorCode(MyTimeModel.NO_ACTIVITY);
			} else {
				boolean hasError = false;
				int errorCode = 0;
				try {
					ID punchID = MyTimeModelHandler.getChangeState(myTimeModel,true);
					if (punchID != null){
						setLastActivity();

						myTimeModel.setNotes("");
						myTimeModel.m_iTimeoutLength = MyTimeModel.initialTimeoutLength;
						myTimeModel.setPunchID(punchID.toString());
						myTimeModel.setPageAction(MyTimeModel.GET_STATUS);
						myTimeModel.m_iPullCount = 0;
						myTimeModel.setActivityDisplayName(myTimeModel.getActivityName());
					} else {
						hasError = true;
					}
					//if (cat.isInfoEnabled()) cat.info("Return Value from processChangeShift in class: "
							//	+ this.getClass().getName() +": "+ punchID);
					//set success message
					if (MyTimeModelHandler.getCurrentActivity(myTimeModel.getEmpId()).equals(Activity.ACTIVITY_NONE)){
						myTimeModel.setErrorCode(MyTimeModel.ACTIVITY_STARTED);
					} else
						myTimeModel.setErrorCode(MyTimeModel.ACTIVITY_CHANGED);
				} catch (AmInvalidActivityException e){
					errorCode = MyTimeModel.ACTIVITY_DELETED;
					deleteActivityFromProfile(id);
					hasError = true;
				} catch (Exception e ) {
					errorCode = MyTimeModel.ERROR_CHANGING_SHIFT;
					hasError = true;
				}
				if ( hasError ) {
					myTimeModel.setErrorCode(errorCode);
                    myTimeModel.setPageAction(MyTimeModel.GET_STATUS);
				}
			}
		}
	}

	/**
	 * Process end activity/shift
	 * set appropriate page messages depending on the return value
	 */
	private void processEndShift(RequestContext context) {
		boolean hasError = false;
		if ( myTimeModel.isValidSN() ) {
			if ( endShiftIntegration != null ) {
				String str = endShiftIntegration.generateHTML(context,IntegrationHook.MYTIME_END_SHIFT);
				context.addGenericHTML(str);
			}
			try {
				myTimeModel.setActivity(Activity.ACTIVITY_NONE);
				ID punchID = MyTimeModelHandler.getChangeState(myTimeModel,false);
				if (punchID != null){
					myTimeModel.setState(Activity.ACTIVITY_NONE);
					myTimeModel.setNotes("");
					myTimeModel.m_iTimeoutLength = MyTimeModel.initialTimeoutLength;
					myTimeModel.setPageAction(MyTimeModel.GET_STATUS);
					myTimeModel.setPunchID(punchID.toString());
					myTimeModel.setActivityDisplayName(context.getLocalizer().i18n(AmWebBundleKey.BUNDLE_NAME, AmWebBundleKey.ADHERENCE_SORT_NONE));
				} else {
					hasError = true;
				}
				//set success message
				myTimeModel.setErrorCode(MyTimeModel.SHIFT_ENDED);
				//if (cat.isInfoEnabled()) cat.info("Return Value from processEndShift in class: "
				//			+ this.getClass().getName() +": "+ punchID);
			} catch ( AmNoTimeRecordException ne ) {
                hasError = true;
				myTimeModel.setErrorCode(MyTimeModel.NO_START_SHIFT);
			} catch ( Exception e ) {
                myTimeModel.setErrorCode(MyTimeModel.ERROR_ENDING_SHIFT);
				hasError = true;
			}
		}
		if ( hasError ) {
                myTimeModel.setPageAction(MyTimeModel.GET_STATUS);
		} // if
	}

}
