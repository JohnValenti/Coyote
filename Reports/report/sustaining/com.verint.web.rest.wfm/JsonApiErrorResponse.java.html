package com.verint.web.rest.wfm;

import java.io.ByteArrayInputStream;
import java.io.ByteArrayOutputStream;
import java.io.IOException;

import javax.inject.Inject;
import javax.ws.rs.container.ContainerRequestContext;
import javax.ws.rs.container.ContainerResponseContext;
import javax.ws.rs.core.Response;

import com.fasterxml.jackson.databind.ObjectMapper;

import io.katharsis.errorhandling.ErrorResponse;
import io.katharsis.rs.type.JsonApiMediaType;

public abstract  class JsonApiErrorResponse {

    @Inject
    protected ObjectMapper objectMapper;

	protected void prepareResponse(ContainerRequestContext requestContext, ErrorResponse katharsisResponse,
			ContainerResponseContext responseContext) throws IOException {
        if (katharsisResponse == null) {
            return;
        }
        ByteArrayOutputStream os = new ByteArrayOutputStream();
        objectMapper.writeValue(os, katharsisResponse);
        ByteArrayInputStream is = new ByteArrayInputStream(os.toByteArray());

        // set the response on the ContainerResponseContext if its input param is not null
        if (responseContext != null) {
            responseContext.setStatus(katharsisResponse.getHttpStatus());
            responseContext.setEntity(is,
					responseContext.getEntityAnnotations(), JsonApiMediaType.APPLICATION_JSON_API_TYPE);
            return;
        }
        // set the response on the ContainerRequestContext otherwise.
        Response response = Response.status(katharsisResponse.getHttpStatus())
                                    .entity(is)
				.type(JsonApiMediaType.APPLICATION_JSON_API)
                                    .build();
        requestContext.abortWith(response);
    }
}
