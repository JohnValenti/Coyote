package com.verint.web.rest.wfm;

import java.util.Optional;

import org.aopalliance.intercept.MethodInterceptor;
import org.aopalliance.intercept.MethodInvocation;

import com.verint.web.rest.wfm.repository.WfoResourceRepository;

import io.katharsis.legacy.queryParams.QueryParams;

public class JsonApiFilterParamValidationInterceptor implements MethodInterceptor {

	@Override
	public Object invoke(MethodInvocation invocation) throws Throwable {
		if (!(invocation.getThis() instanceof WfoResourceRepository)) {
			return invocation.proceed();
		}

		Optional<QueryParams> queryParams = getQueryParamsMethodArgument(invocation);
		if (queryParams.isPresent()) {
			WfoResourceRepository repoInstance = (WfoResourceRepository) invocation.getThis();
			repoInstance.collectFilterParamsForResource(queryParams.get().getFilters());
		}

		return invocation.proceed();
	}

	private Optional<QueryParams> getQueryParamsMethodArgument(MethodInvocation invocation) {
		for (Object argument : invocation.getArguments()) {
			if (argument != null && argument instanceof QueryParams) {
				return Optional.of((QueryParams)argument);
			}
		}

		return Optional.empty();
	}
}
