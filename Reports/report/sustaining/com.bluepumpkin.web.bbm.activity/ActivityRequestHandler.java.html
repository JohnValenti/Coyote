public class ActivityRequestHandler extends RequestHandler {
	private static final Category log = Log.initCategory(ActivityRequestHandler.class.getName());
	private ResourceBundle m_bbmBundle;
	private static final String ACTIVITY_LIST_PAGE = "activityListPage";
	private static final String ACTIVITY_FORM_PAGE = "activityFormPage";
	private static final String MEDIA_SELECTOR_POPUP_PAGE = "mediaSelectorPopupPage";
	private static final String VIEW_ACTIVITIES_PRIVILEGE = PrivilegeKeys.CORE_VIEWACTIVITIES;
	private Localizer m_localizer;

	/**
	 * Process Requests for the activity area
	 */
	public void processRequest(RequestContext context) {
		m_localizer = context.getLocalizer();
		m_bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);

		// Process requested action
		if (isAction(context, PageAction.EDIT)) {
			processListEdit(context);
		} else if (isAction(context, PageAction.ADD)) {
			processListCreateNew(context);
		} else if (isAction(context, PageAction.DELETE)) {
			processListDelete(context);
		} else if (isAction(context, PageAction.CREATE_NEW)) {
			processFormCreateNew(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processFormSave(context);
		} else if (isAction(context, PageAction.CANCEL)) {
			processFormCancel(context);
		} else if (isAction(context, PageAction.RESET)) {
			processFormRevert(context);
		} else if (isAction(context, PageAction.REFRESH)) {
			StringBuffer requestURL = context.getRequest().getRequestURL();
			int requestURLindex = requestURL.indexOf("activity_form");
			if (requestURLindex > 0) {
				processFormRefresh(context);
			} else {
				processListRefresh(context);
			}
		} else if (isAction(context, PageAction.NEW_ID)) {
			gotoList(context);
		} else if (isAction(context, MediaSelectorPopupPageModel.MEDIA_SELECTOR_POPUP_ACTION)) {
			setNextPage(context, MEDIA_SELECTOR_POPUP_PAGE);
		} else {
			gotoList(context);
		}
	}

	/**
	 * Process Create new Action from list
	 */
	private void processListCreateNew(RequestContext context) {
		// validate request - check if organization was selected
		String selectedOrg = context.getRequest().getParameter(ActivityListPageModel.SELECTED_ID_FN);

		if (!StringUtil.isEmpty(selectedOrg)) {
			this.gotoForm(context);
		} else {
			// Oops... this should never happened. must have selected
			// organization
			// before creating new activity. Log and put a msg in page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_NO_ORG_CREATE_FAILURE);

			ActivityListPageModel model = ActivityListPageModel.getInstance(context);
			model.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_SELECT_ORG_BEFORE_CREATE));
			this.gotoList(context);
		}
	}

	/**
	 * Process Edit Action from list
	 */
	private void processListEdit(RequestContext context) {
		String selectedRow = context.getRequest().getParameter(ActivityListPageModel.PERFORM_ACTION_ON_ID_FN);

		if (!StringUtil.isEmpty(selectedRow)) {
			this.gotoForm(context);
		} else {
			// Oops... this should never happened. must have selected a row
			// before editing. Log and put a msg in page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_NO_SELECTION_EDIT_FAILURE);

			ActivityListPageModel model = ActivityListPageModel.getInstance(context);
			model.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_SELECT_BEFORE_EDIT));
			this.gotoList(context);
		}
	}

	/**
	 * Process delete Action from list
	 */
	private void processListDelete(RequestContext context) {
		String selectedRow = context.getRequest().getParameter(ActivityListPageModel.PERFORM_ACTION_ON_ID_FN);
		ActivityListPageModel listModel;

		try {
			if (!StringUtil.isEmpty(selectedRow)) {
				ID[] performActionOnID = StringUtil.parseIDString(selectedRow);

				//Verify that the user has permissions to delete the activity before proceeding
				Activity activityToDelete = getActivityById(context, performActionOnID[0]);
				if (!isAuthorized(context, PrivilegeKeys.CORE_DELETEACTIVITIES, new ScopeID(activityToDelete.getOrganizationId(),
						Scope.ORG_SCOPE))) {
					gotoList(context);
					return;
				}

				listModel = getNewListPageModel(context);
				context.setPageModel(listModel);

				// Process Delete
				processDeleteWrapper(context, performActionOnID[0], Activity.OBJECT_TYPE_ACTIVITY, listModel,
						BbmWebBundleKey.ACTIVITY_DELETE_FAILURE, BbmWebBundleKey.BUNDLE_NAME,
						BbmWebBundleKey.ACTIVITY_DELETE_FAILURE, log, true);
			} else {
				// Oops... this should never happen. must have selected a row
				// before deleting. Log and put a msg in page
				log.l7dError(BbmWebLogBundleKey.ACTIVITY_NO_SELECTION_DELETE_FAILURE);

				listModel = ActivityListPageModel.getInstance(context);
				listModel.addPageMessage(Message.ERROR_TYPE, BbmWebBundleKey.ACTIVITY_SELECT_BEFORE_DELETE,
						BbmWebBundleKey.BUNDLE_NAME);
			}

			// Go back to the list
			this.gotoList(context);
		} catch (Exception e) {
			// Log the message and add it to the page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_DELETE_FAILURE);

			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_DELETE_FAILURE));
			gotoList(context);
		}
	}

	/**
	 * Process create new Action from form
	 */
	private void processFormCreateNew(RequestContext context) {

		try {
			// Create new Activity
			Activity activity = extractForm(context);
			Collection mediaIds = extractMedia(context);
			Collection queueIds = extractQueues(context);
			ActivityProperties activityProperties = extractActivityProperties(context);

			if (!validateForm(context, mediaIds, queueIds))
				return;

			ID newID = ActivityModelHandler.createActivity(context, activity, mediaIds, activityProperties);
			ActivityModelHandler.createActivityQueue(context, newID, queueIds);

			// No exceptions - so prepare to go back to the list
			// Load the List model, Set the edited row to be pre-selected in the
			// list
			ActivityListPageModel listModel = new ActivityListPageModel(context);
			listModel.setPerformActionOnIDs(new ID[] { newID });

			// Cache the model and go back to list
			context.setPageModel(listModel);
			this.gotoList(context);

		} catch (ActivityValidationException e) {
			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					e.getLocalizedMessage(m_localizer.getLocaleContext().getLanguageLocale()));
			gotoForm(context);
		} catch (BbmCreateDuplicateKeyException e) {
			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_CREATE_NEW_FAILURE_DUPLICATE));
			gotoForm(context);
		} catch (Exception e) {
			// Log the message and add it to the page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_CREATE_NEW_FAILURE);

			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_CREATE_NEW_FAILURE));
			gotoForm(context);
		}
	}

	/**
	 * Do any custom validation that you need to do when saving or creating a
	 * new activity.
	 * 
	 * @param mediaIds
	 * @param queueIds
	 */
	private boolean validateForm(RequestContext context, Collection mediaIds, Collection queueIds) {
		// custom validation
		if (mediaIds != null && mediaIds.contains(Media.MEDIA_ID_PROJECT) && (queueIds == null || queueIds.size() != 1)) {
			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_PROJECT_MEDIA_NEEDS_ONE_QUEUE));
			gotoForm(context);
			return false;
		}
		return true;
	}

	/**
	 * Process Save Action from form
	 */
	private void processFormSave(RequestContext context) {
		try {
			// edit Activity after extracting from the form
			Activity activity = extractForm(context);
			Collection mediaIds = extractMedia(context);
			Collection queueIds = extractQueues(context);
			ActivityProperties activityProperties = extractActivityProperties(context);

			if (!validateForm(context, mediaIds, queueIds))
				return;

			ActivityModelHandler.updateActivity(context, activity, mediaIds, queueIds, activityProperties);
			
			// No exceptions - so prepare to go back to the list
			// Load the List model, Set the edited row to be pre-selected in the
			// list
			ActivityListPageModel listModel = new ActivityListPageModel(context);
			listModel.setPerformActionOnIDs(new ID[] { activity.getID() });

			// Cache the model and go back to list
			context.setPageModel(listModel);
			this.gotoList(context);

		} catch (ActivityValidationException e) {
			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					e.getLocalizedMessage(m_localizer.getLocaleContext().getLanguageLocale()));
			gotoForm(context);
		} catch (BbmUpdateDuplicateKeyException e) {
			// add message to the page
			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_UPDATE_FAILURE_DUPLICATE));
			gotoForm(context);
		} catch (Exception e) {
			// Log the message and add it to the page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_UPDATE_FAILURE);

			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_UPDATE_FAILURE));
			gotoForm(context);
		}

	}

	/**
	 * Process Cancel Action from form
	 */
	private void processFormCancel(RequestContext context) {
		ActivityListPageModel listModel = new ActivityListPageModel(context);
		String strID = context.getRequest().getParameter(ActivityFormPageModel.PERFORM_ACTION_ON_ID_FN);

		// Set the edited row to be pre-selected in the list
		if (!StringUtil.isEmpty(strID)) {
			listModel.setPerformActionOnID(strID);
		}

		// Cache the model and go back to list
		context.setPageModel(listModel);
		this.gotoList(context);
	}

	/**
	 * Process revert Action from form
	 */
	private void processFormRevert(RequestContext context) {
		this.gotoForm(context);
	}

	/**
	 * Process refresh Action from list
	 */
	private void processListRefresh(RequestContext context) {
		this.gotoList(context);
	}

	/**
	 * Process refresh Action from form
	 */
	private void processFormRefresh(RequestContext context) {
		this.gotoForm(context);
	}

	/**
	 * Go to List
	 */
	private void gotoList(RequestContext context) {
		setNextPage(context, ACTIVITY_LIST_PAGE);
	}

	/**
	 * Go to Form
	 */
	private void gotoForm(RequestContext context) {
		if (isAuthorized(context, VIEW_ACTIVITIES_PRIVILEGE)) {
			setNextPage(context, ACTIVITY_FORM_PAGE);
		} else {
			gotoList(context);
		}
	}

	/**
	 * extract parameters from submitted form
	 * 
	 * @return the activity value object created from the extracted form
	 */
	private Activity extractForm(RequestContext context) {
		ActivityFormPageModel formModel = new ActivityFormPageModel(context);
		formModel.extractParameters(context.getRequest());
		return formModel.getActivity();
	}

	/**
	 * extract media ids from a form
	 * 
	 * @return Collection of media Ids
	 */
	private Collection extractMedia(RequestContext context) {
		List al = null;
		String mediaIdsString = context.getRequest().getParameter(ActivityFields.MEDIA_FN);
		ID[] mediaIdsArray = StringUtil.parseIDString(mediaIdsString, ",");
		if (mediaIdsArray != null) {
			al = Arrays.asList(mediaIdsArray);
		}
		return al;
	}

	/**
	 * extract queue ids from a form
	 * 
	 * @return Collection of queue Ids
	 */
	private Collection extractQueues(RequestContext context) {
		List al = null;
		String queueIdsString = context.getRequest().getParameter(ActivityFields.QUEUES_FN);
		ID[] queueIdsArray = StringUtil.parseIDString(queueIdsString, ",");
		if (queueIdsArray != null) {
			al = Arrays.asList(queueIdsArray);
		}
		return al;
	}
	
	/**
	 * extract activity extract properties
	 * 
	 * @return the activityproperties value object created from the extracted form
	 */
	private ActivityProperties extractActivityProperties(RequestContext context) {
		ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
		formModel.extractActivityPropertiesParameters(context.getRequest());
		return formModel.getActivityProperties();
	}

	/**
	 * Wrapper to allow for mocking during unit test.
	 */
	boolean processDeleteWrapper(RequestContext context, ID id,
			String objType, WorkpanePageModel model, String deleteFailureKey, String bundleName,
			String logRBKey, Category log, boolean useContextWif) {
		return processDelete(context, id, objType, model, deleteFailureKey, bundleName, logRBKey, log, useContextWif);
	}

	Activity getActivityById(RequestContext context, ID activityId) throws BbmException, RemoteException {
		return ActivityModelHandler.getActivityByID(context, activityId);
	}

	/**
	 * Wrapper to allow for mocking during unit test.
	 */
	ActivityListPageModel getNewListPageModel(RequestContext context) {
		return new ActivityListPageModel(context);
	}
}
