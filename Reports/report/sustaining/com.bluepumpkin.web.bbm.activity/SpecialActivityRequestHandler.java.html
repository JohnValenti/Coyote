package com.bluepumpkin.web.bbm.activity;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.bbm.l10n.BbmWebLogBundleKey;
import com.witness.web.uif.base.Message;
import com.bluepumpkin.web.core.base.RequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;
import java.util.ResourceBundle;


/**
 * Title:        SpecialActivityRequestHandler
 * Description:  Handle requests from Special Activity form page
 * @author       Gonzalo Quintanar
 */
public class SpecialActivityRequestHandler extends RequestHandler {
	private static final Category log = Log.initCategory(SpecialActivityRequestHandler.class.getName());
	private ResourceBundle m_bbmBundle;
	//private static final String CONFIGURE_PRIVILEGE = PrivilegeKeys.CORE_VIEWACTIVITIES;
	private Localizer m_localizer;

	/**
	 * Process Requests for the activity area
	 */
	public void processRequest(RequestContext context) {
		m_localizer = context.getLocalizer();
		m_bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);

		// Process requested action
		if (isAction(context, PageAction.SAVE)) {
			processFormSave(context);
		}
		else if (isAction(context, PageAction.CANCEL)) {
			processFormCancel(context);
		}
	}
	
	/**
	 * Process Save Action from form
	 */
	private void processFormSave(RequestContext context) 
	{
		try 
		{
			SpecialActivityFormPageModel formModel = new SpecialActivityFormPageModel(context);
			formModel.extractParameters(context.getRequest());
			
			ID selectedOrgID = formModel.getSelectedIDObject();

			//Save the Vacation override
			ActivityModelHandler.setVacationOverride(context, selectedOrgID, formModel.getSelectedVacationActivityID());
			
			for (int i=SpecialActivityFormPageModel.PARTIALLY_OVERRIDABLE_SPECIAL_ACTIVITY_START_INDEX; i<SpecialActivityFormPageModel.NUM_SPECIAL_ACTIVITIES; i++)
			{
				//Save the attributes for this org (and all of its child orgs too, which happens automatically).				
				ID activityID = SpecialActivityFormPageModel.specialActivityIDs[i];
				ActivityModelHandler.setMaxDurationOverride(context, selectedOrgID, activityID, formModel.getDuration(activityID));
				ActivityModelHandler.setAdherenceToleranceOverride(context, selectedOrgID, activityID, formModel.getTolerance(activityID));				
			}			
		} 
		catch (Exception e) 
		{
			// Log the message and add it to the page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_UPDATE_FAILURE);

			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_UPDATE_FAILURE));
		}
	}

	/**
	 * Process Save Action from form
	 */
	private void processFormCancel(RequestContext context) 
	{
		try 
		{
			// edit Activity after extracting from the form
			SpecialActivityFormPageModel formModel = new SpecialActivityFormPageModel(context);
			formModel.extractParameters(context.getRequest());
			formModel.setCancelled(true);
			context.setPageModel(formModel);
		} 
		catch (Exception e) 
		{
			// Log the message and add it to the page
			log.l7dError(BbmWebLogBundleKey.ACTIVITY_UPDATE_FAILURE);

			ActivityFormPageModel formModel = ActivityFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					m_localizer.i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_UPDATE_FAILURE));
		}

	}

}
