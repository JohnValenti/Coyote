package com.bluepumpkin.web.bbm.activity;

import java.util.ResourceBundle;

import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityProperties;
import com.witness.web.uif.pagecomponent.container.Collapsible2ColFormPC;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        ActivityUsage
 * Description:  An abstract class for a collapsable container for the Activity usage attributes
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       Shimon Kakon
 * @version 1.0
 */

public abstract class ActivityUsage extends Collapsible2ColFormPC {

	protected static final String ACTIVITY_FIELD_ON_CHANGE = "activityOnChange(this);";

	protected Activity m_activity;
	protected ActivityProperties activityProperties;
	protected boolean m_visible = true;
	protected StringBuffer m_hiddenFields;
	protected StringBuffer m_fields;
	protected short[] m_displayMap;
	protected boolean m_hasMedia = false;
	protected boolean m_hasProjectMedia = false;

	public ActivityUsage(RequestContext context, ResourceBundle bundle, Activity activity, ActivityProperties activityProperties) {
		super(context, bundle);
		m_activity = activity; //needed here for extract parameters
		this.activityProperties = activityProperties; //needed here for extract parameters
	}

	public void initForDisplay() {
		if (m_isInitializedForDisplay) {
			return;
		} else {
			super.initForDisplay();
			if(m_visible) {
				setCollapsed(false);
			}
		}
	}

	public void initForm(Activity activity, ActivityProperties activityProperties, boolean visible, StringBuffer hiddenFields, short[] displayMap,
			             boolean hasMedia, boolean hasProjectMedia) {
		m_activity = activity;
		this.activityProperties = activityProperties;
		m_visible = visible;
		m_hiddenFields = hiddenFields;
		m_displayMap = displayMap;
		m_hasMedia = hasMedia;
		m_hasProjectMedia = hasProjectMedia;
		initFields();
	}

	protected abstract void initFields();
	public abstract void setDefaultValues();


	/**
	 * Get string of the HTML representing this component.
	 */
	public String getHtmlDisplay() {
		if(m_visible) {
			return super.getHtmlDisplay(m_fields.toString());
		} else {
			return "";
		}
	}
}
