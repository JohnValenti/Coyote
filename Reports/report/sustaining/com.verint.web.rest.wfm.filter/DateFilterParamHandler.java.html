package com.verint.web.rest.wfm.filter;


import java.time.ZonedDateTime;
import java.time.format.DateTimeFormatter;
import java.time.format.DateTimeParseException;
import java.util.Date;
import java.util.Optional;
import java.util.Set;

public abstract class DateFilterParamHandler extends FilterParamHandler {

	public final static String START_DATE_DEFAULT_FILTER_KEY = "startDate";
	public final static String END_DATE_DEFAULT_FILTER_KEY = "endDate";

	public static final DateTimeFormatter format = DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ss.SSSXXX");

	public static Optional<Date> getFilterValue(Set<String> params) {
		Optional<String> stringValue = getSingleFilterValue(params);
		if (stringValue.isPresent()) {
			try {
				return Optional.of(Date.from(ZonedDateTime.parse(stringValue.get(), format).toInstant()));
			} catch (DateTimeParseException e) {
				//TODO: Fix this
				//throw new QueryParamDateFormatException(null, stringValue.get());
				throw new RuntimeException(stringValue.get());
			}
		}
		return Optional.empty();
	}
}
