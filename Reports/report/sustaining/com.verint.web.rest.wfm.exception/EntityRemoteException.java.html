package com.verint.web.rest.wfm.exception;

import io.katharsis.response.HttpStatus;

public class EntityRemoteException extends EntityException {

	public EntityRemoteException(String entityName, String id) {
		super(entityName, id);
	}

	@Override
	public String getTitle() {
		return "EntityRemoteException";
	}

	@Override
	public String getDetail() {
		return "A RemoteException occurred on the server.  Please check the server logs for details.  Entity: " +
				getEntityName() + " ID: " + getId();
	}

	@Override
	public int getHttpStatusCode() {
		return HttpStatus.INTERNAL_SERVER_ERROR_500;
	}
}
