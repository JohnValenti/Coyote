package com.verint.web.rest.wfm.exception;

import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;

public class EntityRemoveException extends EntityException {

	private final BbmRemoveException removeException;

	public EntityRemoveException(String entityName, String id, BbmRemoveException exception) {
		super(entityName, id);

		this.removeException = exception;
	}

	public EntityRemoveException(String entityName, String id) {
		super(entityName, id);
		this.removeException = null;
	}

	@Override
	public String getTitle() {
		return "EntityRemoveException";
	}

	@Override
	public String getDetail() {
		if (removeException == null) {
			return "There was a problem removing " + entityName + " with ID " + id;
		}
		return removeException.getMessage();
	}

}
