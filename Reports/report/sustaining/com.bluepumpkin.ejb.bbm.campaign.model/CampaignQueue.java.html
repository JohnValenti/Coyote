package com.bluepumpkin.ejb.bbm.campaign.model;

/**

import java.util.*;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.effectivity.TimePeriod;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectEffectivity;

/**
 * This entity wraps a database table that no longer exists
 * <p>
 * In the current model, campaign does not have a direct relation to queue.
 * Indeed, Campaign has very direct properties... it really is a container for a
 * time series of scheduling periods (SP). Each SP in the series can have
 * relationship with Queue, embodied in SPQueue.
 * <p>
 * Code referencing this class should gradually be updated to reference SPQueue
 */
@Deprecated
public class CampaignQueue extends ValueObjectEffectivity {

	private static final long serialVersionUID = 7207492957439773508L;

	private static FieldInfo m_fieldInfo = new CampaignQueueFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public CampaignQueue() {
		super();
	}

	public ID getCampaignID() {
		return getFieldValueID(CampaignQueueFieldInfo.CAMPAIGNQUEUE_CAMPAIGNID);
	}

	public ID getQueueID() {
		return getFieldValueID(CampaignQueueFieldInfo.CAMPAIGNQUEUE_QUEUEID);
	}

	@Override
	public Date getStartTime() {
		return getFieldValueDate(CampaignQueueFieldInfo.CAMPAIGNQUEUE_STARTTIME);
	}

	@Override
	public Date getEndTime() {
		return getFieldValueDate(CampaignQueueFieldInfo.CAMPAIGNQUEUE_ENDTIME);
	}

	public String getCampaignName() {
		return this.getFormattedFieldValue(CampaignQueueFieldInfo.CAMPAIGNQUEUE_CAMPAIGNNAME);
	}

	public void setCampaignID(ID id) {
		setFieldValue(CampaignQueueFieldInfo.CAMPAIGNQUEUE_CAMPAIGNID, id);
	}

	public void setQueueID(ID id) {
		setFieldValue(CampaignQueueFieldInfo.CAMPAIGNQUEUE_QUEUEID, id);
	}

	@Override
	public void setStartTime(Date dt) {
		setFieldValue(CampaignQueueFieldInfo.CAMPAIGNQUEUE_STARTTIME, dt);
	}

	@Override
	public void setEndTime(Date dt) {
		setFieldValue(CampaignQueueFieldInfo.CAMPAIGNQUEUE_ENDTIME, dt);
	}

	private ID m_idSP = null;

	public void setSPID(ID idSP) {
		m_idSP = idSP;
	}

	public ID getSPID() {
		return m_idSP;
	}

	private ID m_idMedia = null;

	public void setMediaID(ID idMedia) {
		m_idMedia = idMedia;
	}

	public ID getMediaID() {
		return m_idMedia;
	}

	// clone the assignment object without the ID field
	@Override
	public TimePeriod cloneTimePeriod() {
		CampaignQueue item = new CampaignQueue();

		item.setCampaignID(getCampaignID());
		item.setQueueID(getQueueID());
		item.setStartTime(getStartTime());
		item.setEndTime(getEndTime());

		return item;
	}
}
