package com.bluepumpkin.ejb.bbm.campaign.model;

import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectNode;

public class SchedulingState extends ValueObjectNode {

	private static final long serialVersionUID = 4064410350942379418L;

	private static FieldInfo m_fieldInfo = new SchedulingStateFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public SchedulingState() {
		super();
	}

	public int getOverallStatus() {
		return this.getFieldValueInt(SchedulingStateFieldInfo.OVERALLSTATUS);
	}

	public int getCurrentStatus() {
		return this.getFieldValueInt(SchedulingStateFieldInfo.CURRSTATUS);
	}

	public boolean hasWarning() {
		return this.getFieldValueBoolean(SchedulingStateFieldInfo.WARNINGSFLAG);
	}

	public boolean isEnableStopSaveButton() {
		return this.getFieldValueBoolean(SchedulingStateFieldInfo.ENABLESTOPSAVE);
	}

	public Collection getMessages() {
		return this.getChildObjects(SchedulingStateFieldInfo.CHILD_MESSAGES);
	}

	public Collection getWarnings() {
		return this.getChildObjects(SchedulingStateFieldInfo.CHILD_WARNINGS);
	}

	public String getMode() {
		return this.getFormattedFieldValue(SchedulingStateFieldInfo.MODE);
	}

	/**
	 * Returns the time that this scheduling status was last updated.
	 */
	public Date getTimeLastUpdated() {
		return this.getFieldValueDate(SchedulingStateFieldInfo.TIMELASTUPDATED);
	}

	/**
	 * Returns the index of last status message, it is MSGINDEX of table SPSCHEDSTATUSMSGS  
	 */
	public int getStatusMsgIndex() {
		return this.getFieldValueInt(SchedulingStateFieldInfo.STATUSMSGINDEX);
	}
}
