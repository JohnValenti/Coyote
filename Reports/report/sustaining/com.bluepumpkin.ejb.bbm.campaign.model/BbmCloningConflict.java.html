package com.bluepumpkin.ejb.bbm.campaign.model;

/**
 * Title:        Blue Pumpkin Software Basic Business Model
 * Description:  BbmScheduleConflict is an object that represents information
 *               about a hard contraint violation in the schedule, such as
 *               a shift assignment that overlaps another shift assignment
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       Greg Fichtenholtz
 * @version 1.0
 */
import java.io.Externalizable;
import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;
import java.util.Collection;
import java.util.Locale;

import weblogic.utils.StringUtils;

import com.bluepumpkin.common.localization.LocaleContext;
import com.bluepumpkin.common.localization.LocalizationManager;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbBundleKey;

public class BbmCloningConflict implements Externalizable {
	// shift assignment overlaps with an already existing shift assignment
	public static final int CONFLICT_QUEUES_IN_OTHER_CAMPAIGN = 1;

	// one or more of the shift event assignments overlap
	public static final int CONFLICT_MULTIWEEK_ORGS_DONT_MATCH = 2;

	public static final int CONFLICT_MULTIWEEK_QUEUES_DONT_MATCH = 4;

	// more than 1 shift assignment on the same org day
	public static final int CONFLICT_MULTIWEEK_SKILLBASED_DOESNT_MATCH = 8;

	// calendar event assignment overlaps with another calendar event assignment
	public static final int CONFLICT_SHIFT_ASSIGNMENTS_EXIST = 16;

	private int m_conflictType;

	// these class members will be used to format the text of the conflict
	// message
	private Collection<String> m_names;

	public BbmCloningConflict() {
	}

	public BbmCloningConflict(int conflictType) {
		m_conflictType = conflictType;
	}

	public BbmCloningConflict(int conflictType, Collection<String> names) {
		m_conflictType = conflictType;
		m_names = names;
	}

	public int getConflictType() {
		return m_conflictType;
	}

	public Collection<String> getNames() {
		return m_names;
	}

	//
	// Returns the text of the cloning conflict-related message
	//
	public String getMessage() {

		String msgID = null;
		Object[] params = null;

		Locale locale = Locale.getDefault();
		String localeContextKey = LocaleContext.createLocaleContextKey(locale, locale, locale);
		Localizer localizer = LocalizationManager.getInstance().getLocalizer(localeContextKey);

		switch (m_conflictType) {
		case CONFLICT_QUEUES_IN_OTHER_CAMPAIGN:
			msgID = BbmEjbBundleKey.CLONE_CONFLICT_QUEUES_IN_OTHER_CAMPAIGN;
			params = new Object[] { StringUtils.join(m_names.toArray(new String[m_names.size()]), ", ") };
			break;
		case CONFLICT_MULTIWEEK_ORGS_DONT_MATCH:
			msgID = BbmEjbBundleKey.CLONE_CONFLICT_ORGS_DONT_MATCH;
			break;
		case CONFLICT_MULTIWEEK_QUEUES_DONT_MATCH:
			msgID = BbmEjbBundleKey.CLONE_CONFLICT_QUEUES_DONT_MATCH;
			break;
		case CONFLICT_MULTIWEEK_SKILLBASED_DOESNT_MATCH:
			msgID = BbmEjbBundleKey.CLONE_CONFLICT_SKILLBASED_DOESNT_MATCH;
			break;
		case CONFLICT_SHIFT_ASSIGNMENTS_EXIST:
			msgID = BbmEjbBundleKey.CLONE_CONFLICT_SHIFT_ASSIGNMENTS_EXIST;
			break;
		}

		return localizer.i18n(BbmEjbBundleKey.BUNDLE, msgID, params);
	}

	//
	// Reconstitute this object from a stream (i.e., deserialize it).
	//
	@Override
	public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
		m_conflictType = in.readInt();
		m_names = (Collection) in.readObject();
	}

	//
	// Save the state of this object to a stream (i.e., serialize it).
	//
	@Override
	public void writeExternal(ObjectOutput out) throws IOException {
		out.writeInt(m_conflictType);
		out.writeObject(m_names);
	}

}
