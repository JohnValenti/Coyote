import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public final class CampaignOrgFieldInfo extends FieldInfo {

	/**
	 *
	 */
	private static final long serialVersionUID = 1L;
	// field id constants
	public static final int CAMPAIGNORG_ID = 0;
	public static final int CAMPAIGNORG_CAMPAIGNID = 1;
	public static final int CAMPAIGNORG_ORGANIZATIONID = 2;
	public static final int CAMPAIGNORG_STARTTIME = 3;
	public static final int CAMPAIGNORG_ENDTIME = 4;

	private static final int CAMPAIGNORG_NUMOFFIELDS = 5;

	private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[CAMPAIGNORG_NUMOFFIELDS];
	private static String m_strTableName = "CAMPAIGNORGANIZATION";

	// static initialization
	static {
		m_arrFieldInfo[CAMPAIGNORG_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
		m_arrFieldInfo[CAMPAIGNORG_CAMPAIGNID] = new FieldInfoEntry("CAMPAIGNID", PrimitiveTypes.ID);
		m_arrFieldInfo[CAMPAIGNORG_ORGANIZATIONID] = new FieldInfoEntry("ORGANIZATIONID", PrimitiveTypes.ID);
		m_arrFieldInfo[CAMPAIGNORG_STARTTIME] = new FieldInfoEntry("STARTTIME", PrimitiveTypes.DATETIME);
		m_arrFieldInfo[CAMPAIGNORG_ENDTIME] = new FieldInfoEntry("ENDTIME", PrimitiveTypes.DATETIME);
	}

	public CampaignOrgFieldInfo() {
		super();
	}

	// implement abstract methods in parent class
	public String getTableName() {
		return m_strTableName;
	}

	public int getIDFieldIndex() {
		return CAMPAIGNORG_ID;
	}

	public FieldInfoEntry[] getFieldInfo() {
		return m_arrFieldInfo;
	}
}
