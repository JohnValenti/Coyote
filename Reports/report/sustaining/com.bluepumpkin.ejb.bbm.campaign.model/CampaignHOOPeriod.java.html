import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.util.NumberFactory;
import com.bluepumpkin.ejb.bbm.effectivity.TimePeriod;
import com.bluepumpkin.ejb.bbm.effectivity.TimePeriodImpl;
import com.bluepumpkin.ejb.bbm.effectivity.TimePeriodUtil;
import com.bluepumpkin.ejb.bbm.vo.HOOAssignment;
import com.bluepumpkin.ejb.bbm.vo.HOOPeriod;

import java.util.*;

public class CampaignHOOPeriod implements HOOPeriod, java.io.Serializable {

	private static final long serialVersionUID = -2009651363138497932L;

	private Date dtStart, dtEnd;
	private String tzCampaign;

	private Collection<CampaignHOO> colAssignments;
	private Map<Long, Short> mapStartTimes, mapEndTimes;

	public CampaignHOOPeriod(Date dtStart, Date dtEnd, TimeZone tzCampaign, Collection<CampaignHOO> colAssignments) {
		this.dtStart = dtStart;
		this.dtEnd = dtEnd;
		this.tzCampaign = tzCampaign.getID();
		this.colAssignments = colAssignments;
		mapStartTimes = new HashMap<Long, Short>();
		mapEndTimes = new HashMap<Long, Short>();
	}

	// Return Period start date
	@Override
	public Date getPeriodStartDate() {
		return dtStart;
	}

	// Return Period end date
	@Override
	public Date getPeriodEndDate() {
		return dtEnd;
	}

	// Return the TimeZone used to determine mid-night
	@Override
	public TimeZone getPeriodTimeZone() {
		return TimeZone.getTimeZone(tzCampaign);
	}

	// Return whether it's based on Multiple HOO assignments
	@Override
	public boolean isBasedOnMultiple() {
		return (colAssignments != null && colAssignments.size() > 1);
	}

	/**
	 * Return start/end points, this is snapped to 15 minutes interval For 24
	 * hours, it is [0, 95], if it is 8am to 9am, return [32, 35] If day falls
	 * outside of Period window, just return null
	 *
	 * @param Date , which day
	 * @return short[], two elements, first is open offset, second is the end
	 * offset
	 * @deprecated
	 */
	@Deprecated
	@Override
	public short[] getOpenOffset(Date day) {

		if (day == null) {
			return null;
		}

		if (TimePeriodUtil.compareStartTime(day, dtStart) == -1) {
			return null;
		}
		if (TimePeriodUtil.compareEndTime(dtEnd, day) == -1) {
			return null;
		}

		if (colAssignments == null || colAssignments.isEmpty()) {
			return new short[]{(short) 0, (short) 95};
		}

		// convert the day to local date
		LocalDate dtLocal = TimePeriodUtil.convertToLocalDateInTimeZone(day, getPeriodTimeZone());
		Long lDay = new Long(dtLocal.getTime().getTime());

		Date dtGMT = TimePeriodUtil.convertToGMTFromLocalDateInTimeZone(dtLocal, getPeriodTimeZone());

		Short sStart = this.mapStartTimes.get(lDay);
		Short sEnd = this.mapEndTimes.get(lDay);

		if (sStart != null && sEnd != null) {
			return new short[]{sStart.shortValue(), sEnd.shortValue()};
		}

		// search for the period
		Iterator<CampaignHOO> it = colAssignments.iterator();
		while (it.hasNext()) {
			CampaignHOO cpHOO = it.next();

			if (TimePeriodUtil.compareStartTime(cpHOO.getStartTime(), dtGMT) <= 0
					&& TimePeriodUtil.compareEndTime(dtGMT, cpHOO.getEndTime()) <= 0) {
				// set midnight day to calendar
				Calendar cal = Calendar.getInstance(getPeriodTimeZone());
				cal.set(dtLocal.get(Calendar.YEAR), dtLocal.get(Calendar.MONTH), dtLocal.get(Calendar.DATE), 0, 0, 0);
				short iDay = (short) cal.get(Calendar.DAY_OF_WEEK);

				short openOffset = cpHOO.getDayOpen(iDay);
				short closeOffset = cpHOO.getDayClose(iDay);

				sStart = NumberFactory.newShort((short) (openOffset / 15));
				sEnd = NumberFactory.newShort((short) (closeOffset / 15 - 1));

				mapStartTimes.put(lDay, sStart);
				mapEndTimes.put(lDay, sEnd);
				return new short[]{sStart.shortValue(), sEnd.shortValue()};
			}
		}

		return new short[]{(short) 0, (short) 95};
	}

	/**
	 * get HOO open period given a time range,
	 *
	 * @param dtStart , dtEnd: time period, both shouldn't be null
	 * @return a collection of objects which implemented TimePeriod interface
	 */
	// TODO: Sameet, March 2006, this method needs to change to consider HOO
	// differently for the DST transition day in OCT which is 25 hrs
	// Method arguments need to change to pass the campaign TIMEZONE to make a
	// DST decision
	@Override
	public Collection<TimePeriod> getOpenPeriods(Date start, Date end) {
		if (start == null || end == null) {
			throw new IllegalArgumentException("start, end time must not be null.");
		}

		ArrayList<TimePeriod> arrResult = new ArrayList<TimePeriod>();

		// compute the start date in campaign time zone
		// move forward one day
		Calendar cal = Calendar.getInstance(getPeriodTimeZone());
		cal.setTime(start);
		cal.set(Calendar.MILLISECOND, 0);
		cal.set(cal.get(Calendar.YEAR), cal.get(Calendar.MONTH), cal.get(Calendar.DATE), 0, 0, 0);
		cal.add(Calendar.DATE, -1);

		// get assignments overlapped with given period
		Collection<CampaignHOO> colAssignments = TimePeriodUtil.filter(this.colAssignments, cal.getTime(), end);
		if (colAssignments == null || colAssignments.isEmpty()) {
			return arrResult;
		}

		Calendar calTemp = Calendar.getInstance(getPeriodTimeZone());

		TimePeriod timePeriod = new TimePeriodImpl(start, end);

		Iterator<CampaignHOO> it = colAssignments.iterator();
		while (it.hasNext()) {
			HOOAssignment hoo = it.next();
			Date dtHOOStart = hoo.getStartTime();
			Date dtHOOEnd = hoo.getEndTime();

			TimePeriod tpHOO = new TimePeriodImpl(dtHOOStart, dtHOOEnd);
			TimePeriodUtil.retainOverlappedPeriod(tpHOO, timePeriod);
			dtHOOEnd = tpHOO.getEndTime();

			Date dtCurrent = cal.getTime();
			while (TimePeriodUtil.compareEndTime(dtCurrent, dtHOOEnd) < 0) {
				short iDayOfWeek = (short) cal.get(Calendar.DAY_OF_WEEK);
				int iStart = hoo.getDayOpen(iDayOfWeek);
				int iEnd = hoo.getDayClose(iDayOfWeek);

				if (iStart == -1 || iEnd == -1) {
					// not open, do nothing
				} else {
					calTemp.set(cal.get(Calendar.YEAR), cal.get(Calendar.MONTH), cal.get(Calendar.DATE), 0, 0, 0);
					calTemp.set(Calendar.MILLISECOND, 0);
					calTemp.getTime(); // force recalc
					// fix 93562: consider DTS
					// calTemp.add(Calendar.MINUTE, iStart);
					calTemp.set(Calendar.HOUR_OF_DAY, iStart / 60);
					calTemp.set(Calendar.MINUTE, iStart % 60);
					Date dtDayStart = calTemp.getTime();

					if (iStart < iEnd && iEnd <= 1440) { // open close in the
						// same day
						// calTemp.add(Calendar.MINUTE, iEnd - iStart);
						calTemp.set(Calendar.HOUR_OF_DAY, iEnd / 60);
						calTemp.set(Calendar.MINUTE, iEnd % 60);
					} else { // close in the second day
						calTemp.set(cal.get(Calendar.YEAR), cal.get(Calendar.MONTH), cal.get(Calendar.DATE), 0, 0, 0);
						calTemp.set(Calendar.MILLISECOND, 0);
						calTemp.add(Calendar.DATE, 1); // add one day
						// calTemp.add(Calendar.MINUTE, iEnd);
						calTemp.set(Calendar.HOUR_OF_DAY, (iEnd - 1440) / 60);
						calTemp.set(Calendar.MINUTE, (iEnd - 1440) % 60);
					}
					Date dtDayEnd = calTemp.getTime();

					// cut off by the larger period
					TimePeriod tpTmp = new TimePeriodImpl(dtDayStart, dtDayEnd);
					boolean bOverlap = TimePeriodUtil.retainOverlappedPeriod(tpTmp, tpHOO);
					if (bOverlap) {
						arrResult.add(tpTmp);
					}
				}

				// advance one day
				cal.add(Calendar.DATE, 1);
				dtCurrent = cal.getTime();
			} // end while in one HOO
			// back one day for next HOO since the last day is not in
			// previous hoo, should not add one day.
			cal.add(Calendar.DATE, -1);
			dtCurrent = cal.getTime();
		} // end while in all HOOs

		return arrResult;
	}

	/**
	 * Returns a collection of TimePeriods which represent times that
	 * the CampaignHOOPeriod is closed.
	 *
	 * @param dtStart, dtEnd: time period, both shouldn't be null
	 * @return a collection of objects which implemented TimePeriod interface
	 */
	@Override
	public Collection<TimePeriod> getClosePeriods(Date start, Date end) {
		ArrayList<TimePeriod> arrResult = new ArrayList<TimePeriod>();

		Collection<TimePeriod> colOpenPeriods = getOpenPeriods(start, end);
		if (colOpenPeriods == null || colOpenPeriods.isEmpty()) {
			arrResult.add(new TimePeriodImpl(start, end));
			return arrResult;
		}

		Collection<TimePeriod> colPeriods = TimePeriodUtil.merge(colOpenPeriods);
		Iterator<TimePeriod> it = colPeriods.iterator();
		TimePeriod tp = it.next();

		if (start.getTime() < tp.getStartTime().getTime()) {
			arrResult.add(new TimePeriodImpl(start, tp.getStartTime()));
		}

		while (it.hasNext()) {
			TimePeriod tp1 = it.next();
			TimePeriodImpl pTP = new TimePeriodImpl(tp.getEndTime(), tp1.getStartTime());
			if (!pTP.getStartTime().equals(pTP.getEndTime())) {
				arrResult.add(pTP);
			}
			tp = tp1;
		}

		if (tp.getEndTime().getTime() < end.getTime()) {
			arrResult.add(new TimePeriodImpl(tp.getEndTime(), end));
		}

		return arrResult;
	}
}
