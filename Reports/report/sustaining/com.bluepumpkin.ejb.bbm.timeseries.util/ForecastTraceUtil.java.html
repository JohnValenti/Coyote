package com.bluepumpkin.ejb.bbm.timeseries.util;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmTimeSeriesException;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastTraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.ejb.bbm.workload.model.MediaType;

/**
 * This is a utility class for Forecast-specific trace operations.
 */
public class ForecastTraceUtil {

	private ForecastTraceUtil() {}
	
	/**
	 * Creates and returns a ForecastTraceCube with the given trace types, initialized with zero values for every interval
	 * between the given start and end dates.
	 */
	public static ForecastTraceCube createZeroForecastTraceCube(ID queueId, Date start, Date end, short[] traceTypes) throws BbmTimeSeriesException {
		ForecastTraceCube cube = new ForecastTraceCube(queueId, start, end, traceTypes);
		TraceUtil.zeroOutTraceCubeValues(cube, traceTypes);
		return cube;
	}
	
	/**
	 * Returns the forecast trace types that are applicable to the given media type.
	 */
	public static short[] getApplicableForecastTraceTypes(MediaType mediaType) {
		switch (mediaType) {
		case IMMEDIATE:
			return new short[]{Trace.CV, Trace.AHT, Trace.CV_VH}; 
		case DEFERRED:
			return new short[]{Trace.CV, Trace.AHT, Trace.CV_VH}; 
		default:
			return new short[]{Trace.CV, Trace.AHT, Trace.CV_VH}; 
		}
	}
	
	/**
	 * Returns the forecast trace types that can be aggregated for the given media type.
	 */
	public static short[] getApplicableForecastTraceTypesForAggregation(MediaType mediaType) {
		switch (mediaType) {
		case IMMEDIATE:
			return new short[]{Trace.CV, Trace.VAR}; 
		case DEFERRED:
			return new short[]{Trace.CV, Trace.VAR}; 
		default:
			return new short[]{Trace.CV, Trace.VAR}; 
		}
	}
}
