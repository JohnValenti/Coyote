package com.bluepumpkin.ejb.rm.util.interval;

import java.text.SimpleDateFormat;
import java.util.Date;

import com.bluepumpkin.common.util.TimeZoneUtil;

public class IntervalUtil {


	private IntervalUtil() {

	}

	public static boolean intersects(Interval a, Interval b) {
		return a.start() < b.end() && a.end() > b.start();
	}

	/**
	 * Is instance is in the range [interval.start, interval.end)
	 */
	public static boolean isInRange(Interval interval, Date instance) {
		long at = instance.getTime();
		return interval.start() <= at && at < interval.end();
	}

	public static int compareTo(Interval a, Interval b) {
		if (a.start() > b.start()) {
			return 1;
		} else if (a.start() < b.start()) {
			return -1;
		} else if (a.end() > b.end()) { //NOSONAR
			return 1;
		} else if (a.end() < b.end()) { //NOSONAR
			return -1;
		} else {
			return 0;
		}
	}

	/** Creates a debug string representation of and interval assuming start() and end() return longs that can be converted to dates */
	public static String dateIntervalToDebugString(Interval interval) {
		SimpleDateFormat startFormat = new SimpleDateFormat("MM/dd h:mma");
		startFormat.setTimeZone(TimeZoneUtil.GMT_TIMEZONE);
		SimpleDateFormat endFormat = new SimpleDateFormat("h:mma");
		endFormat.setTimeZone(TimeZoneUtil.GMT_TIMEZONE);

		Date start = new Date(interval.start());
		Date end = new Date(interval.end());
		return String.format("[%s %s)", startFormat.format(start), endFormat.format(end));
	}



}
