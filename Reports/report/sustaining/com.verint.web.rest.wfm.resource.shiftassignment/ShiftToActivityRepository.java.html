package com.verint.web.rest.wfm.resource.shiftassignment;

import javax.inject.Inject;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import com.verint.web.rest.wfm.repository.WfoRelationshipRepository;
import com.verint.web.rest.wfm.repository.WfoResourceRepository;
import com.verint.web.rest.wfm.resource.activity.ActivityEntity;
import com.verint.web.rest.wfm.resource.activity.ActivityRepository;

import io.katharsis.legacy.repository.annotations.JsonApiRelationshipRepository;


@JsonApiRelationshipRepository(source = ShiftAssignmentEntity.class, target = ActivityEntity.class)
public class ShiftToActivityRepository
		extends WfoRelationshipRepository<Integer, Integer, ShiftAssignmentEntity, ActivityEntity> {

	private final static Logger log = LoggerFactory.getLogger(ShiftToActivityRepository.class.getName());

	@Inject
	private ActivityRepository activityRepository;

	@Inject
	private ShiftAssignmentRepository saRepository;

	@Override
	public WfoResourceRepository<Integer, ShiftAssignmentEntity> getParentRepository() {
		return saRepository;
	}

	@Override
	public WfoResourceRepository<Integer, ActivityEntity> getChildRepository() {
		return activityRepository;
	}

	@Override
	protected Logger logger() {
		return log;
	}
}
