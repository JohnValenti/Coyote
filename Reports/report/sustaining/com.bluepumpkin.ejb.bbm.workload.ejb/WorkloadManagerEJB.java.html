	public ID createQueue(Queue queue, SystemType externalSystemType, ID queueExternalID) throws BbmCreateException,
			BbmCreateDuplicateKeyException {
		methodStart("createQueue", queue, externalSystemType, queueExternalID);

		// validate
		if (externalSystemType == null) {
			throw new BbmCreateException("externalSystemType is null");
		}
		if (queueExternalID == null) {
			throw new BbmCreateException("external ID is null");
		}
		if ((queueExternalID.isInt() && !externalSystemType.isExtIdAnInt)) {
			throw new BbmCreateException("For this external system, external id must be a string not an integer");
		}
		if ((!queueExternalID.isInt() && externalSystemType.isExtIdAnInt)) {
			throw new BbmCreateException("For this external system, external id must be an integer not a string");
		}

		try {
			final ID queueSID = createQueue(queue);
			queue.setID(queueSID);
			final boolean insert = true;
			MapUtil.setWfoToExternalID(externalSystemType, ObjectType.Queue, queue.getID(), queueExternalID, new Jdmo(true),
					insert);
			return queueSID;
		} catch (final JdmoException e) { // from MapUtil
			handleException(e); // log and rollback transaction
			throw new BbmCreateException(e);
		} catch (final BbmCreateException e) { // createQueue(queue)
			handleException(e); // log and rollback transaction
			throw e;
		} finally {
			methodFinish();
		}
	}

	/**

	public ID copyQueue(ID idQueue, String copyOf) throws BbmCreateException {
		methodStart("copyQueue", idQueue);
		try {
			final Queue queue = getQueueByID(idQueue);
			String sNewName = copyOf + " " + queue.getName();
			int nName = 0;
			while (true) {
				if (isQueueNameUnique(sNewName)) {
					break;
				}
				nName++;
				sNewName = copyOf + " " + queue.getName() + " " + nName;
			}
			queue.setName(sNewName);
			final ID newQueueID = createQueue(queue);
			return newQueueID;
		} catch (final Exception e) {
			handleException(e, false);
			throw new BbmCreateException(e);
		} finally {
			methodFinish();
		}
	}

	/**
	 * Is queue name unique
	 */
	private boolean isQueueNameUnique(String queueName) {
		boolean isUnique = true;
		final Jdmo dmo = new Jdmo();
		try {
			final StringBuffer pStmt = new StringBuffer("select * from QUEUE where NAME=?");
			final JdmoQuery jq = dmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			jq.setParString(1, queueName);
			final JdmoRowset rs = dmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			if (rs.next()) {
				isUnique = false;
			}
		} catch (final JdmoException e) {
			handleException(e, false);
			isUnique = false;
		} finally {
			dmo.cleanUp();
		}
		return isUnique;
	}

	/**
	 * get all media types
	 */
	public Collection<Media> getAllMedia() throws BbmFinderException {
		methodStart("getAllMedia");
		final Jdmo jdmo = new Jdmo();
		final List<Media> mediaColl = new ArrayList<Media>();
		final boolean m_hasOutboundLicense = checkLicense(LicenseKeys.APP_MEDIA_OUTBOUND);
		final boolean m_hasProjectLicense = checkLicense(LicenseKeys.APP_PROJECTS);
		try {
			final StringBuffer pStmt = new StringBuffer("select SID, NAME, ID, MEDIAIMMEDIATE from MEDIA");
			final JdmoQuery jq = jdmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			final JdmoRowset rs = jdmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			while (rs.next()) {
				final Media media = new Media();
				media.setId(new ID(rs.getString("SID")));
				media.setName(rs.getString("NAME"));
				media.setStrID(rs.getString("ID"));
				media.setType(rs.getBoolean("MEDIAIMMEDIATE"));
				// If the Media is Outbound and we dont have license then do not
				// add it the available medias
				if (media.getId().equals(Media.MEDIA_ID_PHONE_OUTBOUND) && !m_hasOutboundLicense) {
					continue;
				}
				if (media.getId().equals(Media.MEDIA_ID_PROJECT) && !m_hasProjectLicense) {
					continue;
				}

				mediaColl.add(media);
			}
			return mediaColl;
		} catch (final JdmoException e) {
			handleException(e);
			throw new BbmFinderException(e);
		} finally {
			jdmo.cleanUp();
			methodFinish();
		}
	}

		QueueAttributeDropDownDAO ddDao = null;
		try {
			dao = new QueueAttributeDefinitionDAO();

			final Collection<QueueAttributeDropDown> dd = obj.getQueueAttributeDropDown();

				}
			}

		} finally {
			if (ddDao != null) {
				ddDao.cleanUp();
			}
			if (dao != null) {
				dao.cleanUp();

	public ID createOrUpdateQueueAttributeFilter(QueueAttributeFilter filter) throws BbmCreateException, BbmUpdateException,
			BbmRemoveException, MultiUserException, RemoteException {
		methodStart("createOrUpdateQueueAttributeFilter");
		ID filterID = null;
		QueueAttributeFilterDAO dao = null;
		QueueAttributeFilterSelectedOrganizationDAO oDao = null;
		QueueAttributeFilterSelectedTextValueDAO textDAO = null;
		QueueAttributeFilterSelectedNumericValueDAO numericDAO = null;
		QueueAttributeFilterSelectedDateValueDAO dateDAO = null;
		QueueAttributeFilterSelectedDropDownValueDAO dropDownDAO = null;
		QueueAttributeFilterMediaDAO mDao = null;
		try {
			dao = new QueueAttributeFilterDAO();
			// QueueAttributeFilterSelectedValueDAO sDao = new
			// QueueAttributeFilterSelectedValueDAO();
			oDao = new QueueAttributeFilterSelectedOrganizationDAO();
			mDao = new QueueAttributeFilterMediaDAO();

			textDAO = new QueueAttributeFilterSelectedTextValueDAO();
			numericDAO = new QueueAttributeFilterSelectedNumericValueDAO();
			// QueueAttributeFilterSelectedBooleanValueDAO booleanDAO = new
			// QueueAttributeFilterSelectedBooleanValueDAO();
			dateDAO = new QueueAttributeFilterSelectedDateValueDAO();
			dropDownDAO = new QueueAttributeFilterSelectedDropDownValueDAO();

			if (filter.getID() == null) {
				final ID pID = dao.createObject(filter);
				filterID = pID;

				if ((filter.getTextValues() != null) && (filter.getTextValues().size() > 0)) {
					final Collection<QueueAttributeFilterSelectedTextValue> values = filter.getTextValues();
					final Iterator<QueueAttributeFilterSelectedTextValue> iter = values.iterator();
					while (iter.hasNext()) {
						iter.next().setQueueAttributeFilterID(pID);
					}
					textDAO.createObjects(values);
				}

				if ((filter.getNumericValues() != null) && (filter.getNumericValues().size() > 0)) {
					final Collection<QueueAttributeFilterSelectedNumericValue> values = filter.getNumericValues();
					final Iterator<QueueAttributeFilterSelectedNumericValue> iter = values.iterator();
					while (iter.hasNext()) {
						iter.next().setQueueAttributeFilterID(pID);
					}
					numericDAO.createObjects(values);
				}

				if ((filter.getDateValues() != null) && (filter.getDateValues().size() > 0)) {
					final Collection<QueueAttributeFilterSelectedDateValue> values = filter.getDateValues();
					final Iterator<QueueAttributeFilterSelectedDateValue> iter = values.iterator();
					while (iter.hasNext()) {
						iter.next().setQueueAttributeFilterID(pID);
					}
					dateDAO.createObjects(values);
				}

				if ((filter.getDropDownValues() != null) && (filter.getDropDownValues().size() > 0)) {
					final Collection<QueueAttributeFilterSelectedDropDownValue> values = filter.getDropDownValues();
					final Iterator<QueueAttributeFilterSelectedDropDownValue> iter = values.iterator();
					while (iter.hasNext()) {
						iter.next().setQueueAttributeFilterID(pID);
					}
					dropDownDAO.createObjects(values);
				}
				// Orgs will always be selected
				final Collection<QueueAttributeFilterSelectedOrganization> orgs = filter.getSelectedOrgs();
				final Iterator<QueueAttributeFilterSelectedOrganization> iter = orgs.iterator();
				while (iter.hasNext()) {
					iter.next().setQueueAttributeFilterID(pID);
				}
				oDao.createObjects(orgs);
				// Creating Media
				if ((filter.getSelectedMedia() != null) && (filter.getSelectedMedia().size() > 0)) {
					final Collection<QueueAttributeFilterMedia> values = filter.getSelectedMedia();
					final Iterator<QueueAttributeFilterMedia> mIter = values.iterator();
					while (mIter.hasNext()) {
						mIter.next().setFilterID(pID);
					}
					mDao.createObjects(values);
				}
			} else {
				filterID = filter.getID();
				dao.updateObject(filter);

				textDAO.deleteObjectsWithParentID(filter.getID());
				numericDAO.deleteObjectsWithParentID(filter.getID());
				dateDAO.deleteObjectsWithParentID(filter.getID());
				dropDownDAO.deleteObjectsWithParentID(filter.getID());

				oDao.deleteObjectsWithParentID(filter.getID());
				mDao.deleteObjectsWithParentID(filter.getID());

				if ((filter.getTextValues() != null) && (filter.getTextValues().size() > 0)) {
					textDAO.createObjects(filter.getTextValues());
				}
				if ((filter.getNumericValues() != null) && (filter.getNumericValues().size() > 0)) {
					numericDAO.createObjects(filter.getNumericValues());
				}
				if ((filter.getDateValues() != null) && (filter.getDateValues().size() > 0)) {
					dateDAO.createObjects(filter.getDateValues());
				}
				if ((filter.getDropDownValues() != null) && (filter.getDropDownValues().size() > 0)) {
					dropDownDAO.createObjects(filter.getDropDownValues());
				}

				// Orgs will always be selected
				oDao.createObjects(filter.getSelectedOrgs());
				// Creating Media Association
				if ((filter.getSelectedMedia() != null) && (filter.getSelectedMedia().size() > 0)) {
					mDao.createObjects(filter.getSelectedMedia());
				}
			}

			return filterID;
		} finally {
			if (dao != null) {
				dao.cleanUp();
			}
			if (oDao != null) {
				oDao.cleanUp();
			}
			if (textDAO != null) {
				textDAO.cleanUp();
			}
			if (numericDAO != null) {
				numericDAO.cleanUp();
			}
			if (dateDAO != null) {
				dateDAO.cleanUp();
			}
			if (dropDownDAO != null) {
				dropDownDAO.cleanUp();
			}
			if (mDao != null) {
				mDao.cleanUp();
			}
			methodFinish();
		}
	}

	 *
	 * @param isExportFromPlanner specifies if queues are requested for export operation in Planner
	 * @param isLTFInPlanner      specifies if queues are requested for use in the LTF feature in Planner
	 * @return Map of Queue/string
	 * @throws BbmFinderException , RemoteException
	 */

	public Map<Queue, String> getQueuesLinkedToSPAndHasHistoricalData(boolean isExportFromPlanner, boolean isLTFInPlanner)
			throws BbmFinderException, RemoteException {
		methodStart("getQueuesLinkedToSPAndHasHitoricalData", isExportFromPlanner, isLTFInPlanner);
		final Jdmo jdmo = new Jdmo();
		final HashMap<Queue, String> queueHash = new HashMap<Queue, String>();

		try {

			StringBuilder pStmt = getQueueQuery(isExportFromPlanner, isLTFInPlanner);
			final JdmoQuery jq = jdmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			final JdmoRowset rs = jdmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			while (rs.next()) {
				Queue queue = QueueDAO.getQueueInfoForPlanner(rs);
				String tz = rs.getString("TZ");
				queueHash.put(queue, tz);
			}
			return queueHash;
		} catch (final JdmoException e) {
			handleException(e);
			throw new BbmFinderException(e);
		} finally {
			jdmo.cleanUp();
			methodFinish();
		}
