	/**
	 * get ids of skills linked to a given queue
	 *
	 * @return a collection of skill ids
	 */
	public Collection<QueueSkill> getLinkedSkillIDsForQueue(ID idQueue) throws BbmFinderException {
		if (idQueue == null) {
			return Collections.emptyList();
		}

		Jdmo jdmo = new Jdmo();
		try {
			List<QueueSkill> retVal = new ArrayList<>();

			StringBuilder pStmt = new StringBuilder("select distinct s.SID from SPQUEUESKILL q, SKILL s where q.SKILLID=s" 
				+ ".ID "
				+ "and q.SPQUEUEID in ( select ID from spqueue where queueid =(  select ID from QUEUE where sid = ");
			pStmt.append(idQueue);
			pStmt.append("))");

			JdmoQuery jq1 = jdmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			JdmoRowset rs = jdmo.createRowset(jq1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

			while (rs.next()) {
				QueueSkill objValue = new QueueSkill();
				objValue.setSkillID(rs.getID(1));
				objValue.setQueueID(idQueue);

				retVal.add(objValue);
			}

			return retVal;
		} catch (Exception ex) {
			throw new BbmFinderException(ex);
		}
	}

	/**
	 * get ids of skills linked to a list of queues
	 *
	 * @return a HashMap contains (queueID, array list of skills)
	 */
	public Map<ID, List<QueueSkill>> getLinkedSkillIDsForQueues(Collection colQueueIDs) throws BbmFinderException {
		if (colQueueIDs == null || colQueueIDs.isEmpty()) {
			return new HashMap();
		}

		Jdmo jdmo = new Jdmo();
		Map<ID, List<QueueSkill>> retVal = new HashMap<>();

		try {
			// build the list of Id's
			StringBuilder idBuilder = new StringBuilder();
			Iterator iter = colQueueIDs.iterator();
			while (true) {
				ID queID = (ID) iter.next();

				retVal.put(queID, new ArrayList<>());
				idBuilder.append(queID);

				if (iter.hasNext()) {
					idBuilder.append(",");
				} else {
					break;
				}
			}

			StringBuilder pStmt = new StringBuilder();
			pStmt.append(" select distinct s.SID, queue.sid QUEID from ");
			pStmt.append(" SPQUEUESKILL q,SKILL s,SPQUEUE,QUEUE ");
			pStmt.append(" where q.SKILLID=s.ID and ");
			pStmt.append(" q.SPQUEUEID=spqueue.id and ");
			pStmt.append(" spqueue.queueid = queue.id ");
			pStmt.append(" and queue.sid in (");
			pStmt.append(idBuilder.toString());
			pStmt.append(")");

			JdmoQuery jq1 = jdmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			JdmoRowset rs = jdmo.createRowset(jq1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

			while (rs.next()) {
				QueueSkill objValue = new QueueSkill();
				objValue.setSkillID(rs.getID("SID"));
				ID queID = rs.getID("QUEID");
				objValue.setQueueID(queID);

				List<QueueSkill> lst = retVal.get(queID);
				if (lst == null) {
					lst = new ArrayList<>();
					retVal.put(queID, lst);
				}

				lst.add(objValue);
			}

			rs.close();


			return retVal;
		} catch (Exception ex) {
			throw new BbmFinderException(ex);
		} finally {
			jdmo.cleanUp();
		}

	}

	public void linkSkillsToQueue(ID idQueue, Collection colSkillIDs) throws BbmCreateException {
		if (idQueue == null || colSkillIDs == null || colSkillIDs.isEmpty()) {
			return;
		}
		try {
			// in case of link skill to queue twice or by another user
			unLinkSkillsFromQueue(idQueue, colSkillIDs);

			Iterator it = colSkillIDs.iterator();
			while (it.hasNext()) {
				QueueSkill item = new QueueSkill();
				item.setQueueID(idQueue);
				item.setSkillID((ID) it.next());
				createObject(item);
			}
		} catch (Exception e) {
			throw new BbmCreateException(e);
		}
	}

	public void unLinkSkillsFromQueue(ID idQueue, Collection colSkillIDs) throws BbmRemoveException {
		if (idQueue == null || colSkillIDs == null || colSkillIDs.isEmpty()) {
			return;
		}

		try {
			StringBuilder strSQL = new StringBuilder(30);
			strSQL.append("delete ").append(getFieldInfo().getTableName());
			strSQL.append(" where QUEUEID = ").append(idQueue);
			strSQL.append(" and SKILLID in ").append(m_dmo.createInClause(colSkillIDs));

			m_dmo.executeCommand(new String(strSQL));
		} catch (JdmoException e) {
			throw new BbmRemoveException(e);
		}
	}

	public void unLinkAllSkillsFromQueue(ID idQueue) throws BbmRemoveException {
		if (idQueue == null) {
			return;
		}

		try {
			StringBuilder strSQL = new StringBuilder(30);
			strSQL.append("delete ").append(getFieldInfo().getTableName());
			strSQL.append(" where QUEUEID = ").append(idQueue);
			m_dmo.executeCommand(new String(strSQL));
		} catch (JdmoException e) {
			throw new BbmRemoveException(e);
		}
	}

}
