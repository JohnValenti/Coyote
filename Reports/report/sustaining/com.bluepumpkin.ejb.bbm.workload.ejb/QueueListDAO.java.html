package com.bluepumpkin.ejb.bbm.workload.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.workload.model.QueueList;
import com.bluepumpkin.ejb.bbm.workload.model.QueueListFieldInfo;
import com.bluepumpkin.ejb.core.base.MultiUserException;

import java.util.*;

public class QueueListDAO extends DAOBase<QueueList> {

	private static FieldInfo m_fieldInfo = new QueueListFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public QueueListDAO() {
		super();
	}

	public QueueListDAO(Jdmo dmo) {
		super(dmo);
	}

	public QueueListDAO(Jdmo dmo, boolean isDmoCreated) {
		super(dmo, isDmoCreated);
	}

	@Override
	protected QueueList createValueObject() {
		return new QueueList();
	}

	public Collection<QueueList> getQueueHistoryListForQueueID(ID queueSID) throws BbmFinderException {
		return getObjects(" QUEUEID=" + queueSID.toString());
	}

	public void deleteQueueHistoryByQueueID(ID queueSID) throws BbmRemoveException {
		deleteObjects(" QUEUEID=" + queueSID.toString());
	}

	public Map<QueueList, QueueList> getObjectsByTime(Collection<ID> queueColID, Date start, Date end)
		throws BbmFinderException {
		HashMap<QueueList, QueueList> retVal = new HashMap<QueueList, QueueList>();
		if (start.equals(end)) {
			return retVal;
		}
		StringBuilder sb = new StringBuilder();
		sb.append(" STARTTIME>='").append(JdmoUtil.formatDBString(start)).append("'");
		sb.append(" AND ENDTIME<='").append(JdmoUtil.formatDBString(end)).append("'");
		try {
			sb.append(" AND QUEUEID in ").append(m_dmo.createInClause(queueColID));
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
		Collection<QueueList> col = getObjects(sb.toString());
		for (Iterator<QueueList> iter = col.iterator(); iter.hasNext(); ) {
			QueueList obj = iter.next();
			retVal.put(obj, obj);
		}
		return retVal;
	}

	public void saveQueueHistoryList(Collection<QueueList> voCol) throws BbmFinderException, MultiUserException,
		BbmUpdateException, BbmCreateException {

		// saving queue history list include the following logic:
		// if there is already a queue history list for the same queue and
		// start/end time in the table, then update.
		// else create.
		// group all lists for the same queue and times
		Date tempStart = new Date();
		Date tempEnd = new Date(0);
		HashMap<QueueList, QueueList> map = new HashMap<QueueList, QueueList>();
		Collection<ID> queueCol = new HashSet<ID>();
		for (Iterator<QueueList> iter = voCol.iterator(); iter.hasNext(); ) {
			QueueList ql = iter.next();
			Date startTime = ql.getStartTime();
			Date endTime = ql.getEndTime();
			if (startTime.before(tempStart)) {
				tempStart = startTime;
			}
			if (endTime.after(tempEnd)) {
				tempEnd = endTime;
			}
			QueueList mapQL = (QueueList) map.get(ql);
			if (mapQL == null) {
				map.put(ql, ql);
			} else {
				ql.add(mapQL);
				map.put(ql, ql);
			}
			queueCol.add(ql.getQueueId());
		}
		Map<QueueList, QueueList> dbMap = getObjectsByTime(queueCol, tempStart, tempEnd);
		Collection<QueueList> updateObjs = new ArrayList<QueueList>();
		Collection<QueueList> createObjs = new ArrayList<QueueList>();
		for (Iterator<QueueList> iter = map.values().iterator(); iter.hasNext(); ) {
			QueueList ql = iter.next();
			QueueList dbql = dbMap.get(ql);
			if (dbql == null) {
				createObjs.add(ql);
			} else {
				dbql.setLength(ql.getLength());
				dbql.setRetries(ql.getRetries());
				updateObjs.add(dbql);
			}
		}
		updateObjects(updateObjs);
		createObjects(createObjs);
	}
}
