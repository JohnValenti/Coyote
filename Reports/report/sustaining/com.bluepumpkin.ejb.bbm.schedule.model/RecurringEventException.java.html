package com.bluepumpkin.ejb.bbm.schedule.model;

/**
 * Title:        Blue Pumpkin Software Basic Business Model
 * Description:  Value object class that models the exceptions to the
 *               regularly occuring calendar events. The events at these times
 *               are skipped when rendering recurring event instances for
 *               employees
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       Greg Fichtenholtz
 * @version 1.0
 */
import java.io.*;
import java.util.*;
import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.ejb.bbm.vo.*;

public class RecurringEventException extends ValueObjectBase implements Externalizable {
	// meta data info.
	private static FieldInfo m_fieldInfo = new RecurringEventExceptionFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ID getWorkResourceID() {
		return getFieldValueID(RecurringEventExceptionFields.WORKRESOURCEID);
	}

	public void setWorkResourceID(ID workResourceID) {
		setFieldValue(RecurringEventExceptionFields.WORKRESOURCEID, workResourceID);
	}

	public Date getExceptionDate() {
		return getFieldValueDate(RecurringEventExceptionFields.EXCEPTIONDATE);
	}

	public void setExceptionDate(Date exceptionDate) {
		setFieldValue(RecurringEventExceptionFields.EXCEPTIONDATE, exceptionDate);
	}

	/**
	 * Reconstitute this object from a stream (i.e., deserialize it).
	 */
	public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
		ValueObjectUtil.defaultReadExternal(this, in);
	}

	/**
	 * Save the state of this object to a stream (i.e., serialize it).
	 */
	public void writeExternal(ObjectOutput out) throws IOException {
		ValueObjectUtil.defaultWriteExternal(this, out);
	}
}
