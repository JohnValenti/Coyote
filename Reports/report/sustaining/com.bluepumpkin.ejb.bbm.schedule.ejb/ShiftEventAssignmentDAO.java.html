package com.bluepumpkin.ejb.bbm.schedule.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftEventAssignment;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftEventAssignmentFields;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.core.base.MultiUserException;

import java.util.*;

class ShiftEventAssignmentDAO extends DAOBase<ShiftEventAssignment> {
	// meta data info.
	private static FieldInfo m_fieldInfo = new ShiftEventAssignmentFields();

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	/**
	 * it is important to store the reference to the event precedence rule
	 * object, becase the event precedence DAO object will maintain a cache of
	 * activityID -> activity throughout the lifetime of this object. With out
	 * such activity cache, every create shift assignment and/or create of shift
	 * event assignment would be accompanied by a DB SQL roundtrip to find the
	 * activity given the event activity ID
	 */
	private EventPrecedenceRule m_rulePrecedence = new EventPrecedenceRule();

	ShiftEventAssignmentDAO() {
		super();
	}

	ShiftEventAssignmentDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ShiftEventAssignment createValueObject() {
		return new ShiftEventAssignment();
	}

	/** This is an override of the base class */
	public ID createObject(ShiftEventAssignment event, boolean bBatched) throws BbmCreateException {
		if (event.getSPID() != null && event.getSPID().isInt() && event.getSPID().toInt() == 0)
			event.setSPID(null);

		try {
			/* calculate the event precedence */
			int precedence = m_rulePrecedence.getEventOverlayPrecedence(event);
			event.setOverlayPrecedence(precedence);
		} catch (Exception e) {
			throw new BbmCreateException(e);
		}
		return super.createObject(event, bBatched);
	}

	/**
	 * This is an override of the base class functionality update an object and
	 * its child objects check if the activity changed and if so, re-calc the
	 * value of overlay precedence
	 */
	public void updateObject(ShiftEventAssignment event) throws MultiUserException, BbmUpdateException {
		if (event.isFieldValueSet(ShiftEventAssignmentFields.ACTIVITYID)) {
			try {
				int precedence = m_rulePrecedence.getEventOverlayPrecedence(event);
				event.setOverlayPrecedence(precedence);
			} catch (Exception e) {
				throw new BbmUpdateException(e);
			}
		}
		super.updateObject(event);
	}

	/**
	 * Locks or unlocks events of a given type for a given set of work resources
	 * and time period
	 */
	public void lockOrUnlockEvents(Collection workResourceIDs, Date dtStart, Date dtEnd, boolean bLock)
			throws BbmUpdateException {
		try {
			StringBuffer buf = new StringBuffer(200);
			buf.append("UPDATE SHIFTEVENTASSIGNMENT ");
			buf.append("SET ISLOCKED = ").append(bLock ? "1 " : "0 ");
			buf.append("WHERE ");
			buf.append("SHIFTASSIGNMENTID IN ");
			buf.append("(SELECT ID FROM SHIFTASSIGNMENT A ");
			buf.append("WHERE ");
			buf.append(m_dmo.dateAddMinutes("A.DURATION", "A.STARTTIME"));
			buf.append(" > '" + JdmoUtil.formatDBString(dtStart));
			buf.append("' AND A.STARTTIME <= '" + JdmoUtil.formatDBString(dtEnd));
			buf.append("' AND A.WORKRESOURCEID IN ");
			buf.append(m_dmo.createInClause(workResourceIDs)).append(")");

			m_dmo.executeCommand(buf.toString());
		} catch (JdmoException e) {
			throw new BbmUpdateException(e);
		}
	}

	/**
	 * Returns a collection of schedule events for multiple work resources
	 */
	Collection<ShiftEventAssignment> getShiftAssignmentsForWorkResources(
			Collection<ID> workResourceIDs, Date dtStart, Date dtEnd) throws BbmFinderException {
		try {
			StringBuffer strSQL = getWhereClauseForGetEventsQuery(workResourceIDs, dtStart, dtEnd);

			return getObjects(strSQL.toString());
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}

	/**
	 * Returns the contents of the SQL WHERE clause that is used to fetch the
	 * events for a given set of work resources in a given time period
	 */
	protected StringBuffer getWhereClauseForGetEventsQuery(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd)
			throws JdmoException {
		StringBuffer strSQL = getWhereClauseForGetEventsQueryByTimeRange(dtStart, dtEnd);
		if (workResourceIDs != null && !workResourceIDs.isEmpty()) {
			strSQL.append(" A.WORKRESOURCEID IN ");
			strSQL.append(m_dmo.createInClause(workResourceIDs));
		}
		return strSQL;
	}

	/**
	 * Returns the contents of the SQL WHERE clause that is used to fetch the
	 * events for a given workresource in a given time period
	 *
	 * TODO: largely copied from EventAssignmentDAO, this should be refactored to try and share the
	 * query logic.
	 */
	private StringBuffer getWhereClauseForGetEventsQueryByTimeRange(Date dtStart, Date dtEnd) throws JdmoException {
		StringBuffer strSQL = new StringBuffer(100);

		if (dtStart != null) {
			strSQL.append(m_dmo.dateAddMinutes("A.DURATION", "A.STARTTIME"));
			strSQL.append(" > '" + JdmoUtil.formatDBString(dtStart));
			strSQL.append("' AND ");

			// to enhance the sql statement performance.
			// add a.starttime > dtstart - 24 hours for querying shiftassignment
			// and shiftassignmentpub table to constraint the duration check range.
			Calendar aCalendar = new GregorianCalendar(TimeZone.getTimeZone("GMT"));
			aCalendar.setTime(dtStart);
			aCalendar.add(Calendar.DATE, -1);
			Date oneDayBack = aCalendar.getTime();

			strSQL.append(" A.STARTTIME > '" + JdmoUtil.formatDBString(oneDayBack));
			strSQL.append("' AND ");

			if (dtEnd != null) {
				strSQL.append(" A.STARTTIME < '" + JdmoUtil.formatDBString(dtEnd));
				strSQL.append("' AND ");
			}
		}
		return strSQL;
	}
}
