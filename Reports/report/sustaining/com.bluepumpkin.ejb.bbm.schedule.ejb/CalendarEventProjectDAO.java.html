package com.bluepumpkin.ejb.bbm.schedule.ejb;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventProject;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventProjectFields;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.core.base.MultiUserException;

public class CalendarEventProjectDAO extends DAOBase {

	// meta data info.
	private static FieldInfo m_fieldInfo = new CalendarEventProjectFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public CalendarEventProjectDAO() {
		super();
	}

	public CalendarEventProjectDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ValueObjectBase createValueObject() {
		return new CalendarEventProject();
	}

	/**
	 * update a value object, save its state to database this is an override of
	 * the base class
	 */
	public void updateObject(ValueObjectBase objValue) throws MultiUserException, BbmUpdateException {
		try {
			CalendarEventProject project = (CalendarEventProject) objValue;
			StringBuffer sqlBuf = new StringBuffer(100);
			sqlBuf.append("UPDATE CALENDAREVENTPROJECT ");
			sqlBuf.append("SET projectid = ").append(project.getProjectID());
			sqlBuf.append(" WHERE ");
			sqlBuf.append(" CALENDAREVENTASSIGNMENTID = ");
			sqlBuf.append(project.getParentID());
			m_dmo.executeCommand(sqlBuf.toString());
		} catch (JdmoException e) {
			throw new BbmUpdateException(e);
		}
	}
}
