import java.util.Date;
import java.util.HashSet;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.schedule.model.RecurringEventException;
import com.bluepumpkin.ejb.bbm.schedule.model.RecurringEventExceptionFields;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

class RecurringEventExceptionDAO extends DAOBase<RecurringEventException> {
	// meta data info.
	private static FieldInfo m_fieldInfo = new RecurringEventExceptionFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public RecurringEventExceptionDAO() {
		super();
	}

	public RecurringEventExceptionDAO(Jdmo dmo) {
		super(dmo);
	}

	protected RecurringEventException createValueObject() {
		return new RecurringEventException();
	}

	public HashSet<ID> getExceptionsOnDate(ID templateID, Date exceptionDate) throws JdmoException {
		StringBuffer buf = new StringBuffer(100);
		buf.append("SELECT WORKRESOURCEID ");
		buf.append("FROM RECURRINGEVENTEXCEPTION ");
		buf.append("WHERE RECURRINGEVENTTEMPLATEID = ").append(templateID);
		buf.append(" AND EXCEPTIONDATE = '");
		buf.append(JdmoUtil.formatDBString(exceptionDate)).append("'");

		HashSet<ID> workResourceIDs = new HashSet<ID>();
		JdmoRowset rs = m_dmo.createRowset(buf.toString());
		while (rs.next()) {
			workResourceIDs.add(rs.getID(1));
		}
		rs.close();
		return workResourceIDs;
	}

	/**
	 * delete objects given a list of their ids
	 */
	public void deleteRecurringException(ID workResourceID, ID templateID, Date exceptionDate) throws JdmoException {
		StringBuffer buf = new StringBuffer(100);
		buf.append("DELETE FROM RECURRINGEVENTEXCEPTION ");
		buf.append("WHERE WORKRESOURCEID = ").append(workResourceID);
		buf.append(" AND RECURRINGEVENTTEMPLATEID = ").append(templateID);
		buf.append(" AND EXCEPTIONDATE = '");
		buf.append(JdmoUtil.formatDBString(exceptionDate)).append("'");

		m_dmo.executeCommand(buf.toString());
	}
}
