package com.bluepumpkin.ejb.bbm.schedule.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.NumberFactory;
import com.bluepumpkin.ejb.bbm.activity.ejb.ActivityManager;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.Event;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbLogBundleKey;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventTemplate;
import com.verint.ejb.wfm.WfmManagerFactory;

/**
 * Determines schedule event overlay precedence based on event properties such
 * as type and activity.
 */
public class EventPrecedenceRule {
	public static final short PRECEDENCE_TYPE_SHIFT_ASSIGNMENT = 1;
	public static final short PRECEDENCE_TYPE_SHIFT_EVENT_ASSIGNMENT_USEDINSHIFT = 5;
	public static final short PRECEDENCE_TYPE_SHIFT_EVENT_ASSIGNMENT_NOTUSEDINSHIFT = 10;
	public static final short PRECEDENCE_TYPE_CALENDAR_EVENT_ASSIGNMENT = 20;
	public static final short PRECEDENCE_TYPE_TIME_OFF = Event.TIMEOFF_PRECEDENCE;
	public static final short PRECEDENCE_TYPE_TYPE_UNAVAILABILITY = 1;

	/**
	 * Returns a default overlay precedence of the event with a given type and
	 * activity.
	 *
	 * @param eventType
	 *            type of the event, as defined in an enumeration from class
	 *            com.bluepumpkin.ejb.bbm.activity.model.Event
	 * @param activityID
	 *            unique ID of the event's activity
	 * @return the integer number that represents the event overlay precedence
	 *         value
	 */
	private static short getEventOverlayPrecedence(int eventType, ID activityID) throws Exception {
		/* hardcode the precedence table for now ( F & S release ) */
		switch (eventType) {
		case Event.EVENT_TYPE_SHIFT_ASSIGNMENT:
			return PRECEDENCE_TYPE_SHIFT_ASSIGNMENT;

		case Event.EVENT_TYPE_SHIFT_EVENT_ASSIGNMENT:

			return (getActivityFromThreadLocalCache(activityID).isUsedInShift() ? PRECEDENCE_TYPE_SHIFT_EVENT_ASSIGNMENT_USEDINSHIFT
					: PRECEDENCE_TYPE_SHIFT_EVENT_ASSIGNMENT_NOTUSEDINSHIFT);

		case Event.EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT:
			return PRECEDENCE_TYPE_CALENDAR_EVENT_ASSIGNMENT;

		case Event.EVENT_TYPE_TIME_OFF:
			return PRECEDENCE_TYPE_TIME_OFF;

		case Event.EVENT_TYPE_UNAVAILABILITY:
			return PRECEDENCE_TYPE_TYPE_UNAVAILABILITY;
		}

		/* this should never happen */
		throw new BbmException(BbmEjbLogBundleKey.SCHEDULE_UNKNOWN_EVENT_TYPE,
				new Object[] { NumberFactory.newInteger(eventType) });
	}

	static Activity getActivityFromThreadLocalCache(ID activityID) {
		CachePerEJBMethod cache = CachePerEJBMethod.getCache();
		if (cache != null) {
			return cache.getActivity(activityID);
		} else {
			return getActivity(activityID);
		}
	}
	
	static Activity getActivity(ID activityID) {
		Activity activity = null;
		
		try {
			ActivityManager activityManager = WfmManagerFactory.getActivityManager(false);
			activity = activityManager.findActivityById(activityID);
		} catch (Exception e) {
			
		}
		
		return activity;
	}

	/**
	 * Returns a default overlay precedence of the event.
	 */
	public static short getEventOverlayPrecedence(Event event) throws Exception {
		return getEventOverlayPrecedence(event.getEventType(), event.getActivityID());
	}

	/**
	 * Returns a default overlay precedence of events created from the event
	 * template.
	 */
	public static short getEventOverlayPrecedence(CalendarEventTemplate eventTemplate) throws Exception {
		return getEventOverlayPrecedence(eventTemplate.getEventType(), eventTemplate.getActivityID());
	}
}
