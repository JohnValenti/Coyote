import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.Iterator;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;

import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventTemplateImpl;
import com.bluepumpkin.ejb.bbm.schedule.model.ValueObjectAggEventTemplate;
import com.bluepumpkin.ejb.core.base.MultiUserException;

abstract class AggEventTemplateDAO<T extends ValueObjectAggEventTemplate> extends DAONode<T> {
	// internal dao class for aggregated CALENDAREVENTTEMPLATE object
	private CalendarEventTemplateDAO m_aggDAO = null;

	public AggEventTemplateDAO() {
		super();
		/* m_dmo will be initialized after calling the super class constructor */
		m_aggDAO = new CalendarEventTemplateDAO(m_dmo);
	}

	public AggEventTemplateDAO(Jdmo dmo) {
		super(dmo);
		m_aggDAO = new CalendarEventTemplateDAO(dmo);
	}

	/**
	 * This method is an override from the base class
	 */
	protected boolean allowCreateWithId() {
		return true;
	}

	/**
	 * base class override Here we shall join the CALENDAREVENTTEMPLATE table
	 * and corresponding EVENTTEMPLATE table and use the result to initialize
	 * both the CalendarEventTemplate subclass and its aggregated
	 * CalendarEventTemplate "base" class so that we can initialize both
	 * instances as a result of a single SQL query. See also method
	 * readObjectFromDB.
	 */
	protected StringBuffer getSelectStatement() {
		StringBuffer strSQL = new StringBuffer(200);
		/* first append all the fields from the CALENDAREVENTTEMPLATE table */
		strSQL.append("SELECT ");
		for (int i = 0; i < m_aggDAO.getFieldInfo().getFieldCount(); i++) {
			if (i != 0)
				strSQL.append(", ");

			strSQL.append("A.");
			strSQL.append(m_aggDAO.getFieldInfo().getFieldName(i));
		}
		strSQL.append(", ");
		/* now append all the fields from the linked EVENTEMPLATE table */
		for (int i = 0; i < getFieldInfo().getFieldCount(); i++) {
			if (i != 0)
				strSQL.append(", ");

			strSQL.append("B.");
			strSQL.append(getFieldInfo().getFieldName(i));
		}
		strSQL.append(" ");
		return strSQL;
	}

	protected void appendSystemFieldsToSelect(StringBuffer strSelect) {
		strSelect.append(", A.CHANGECOUNTER, A.MODIFIEDBY ");
	}

	/** @return true means there is a 'where' in the select statement */
	protected boolean appendFromWhereStatement(StringBuffer strSQL) {
		strSQL.append("FROM ");
		strSQL.append(m_aggDAO.getFieldInfo().getTableName());
		strSQL.append(" A, ");
		strSQL.append(getFieldInfo().getTableName() + " B ");
		strSQL.append("WHERE A.ID = B.ID ");
		return true;
	}

	/** create an object including its child objects in batched mode */
	public ID createObject(T objValue, boolean bBatched) throws BbmCreateException {
		T eventTemplate = objValue;
		CalendarEventTemplateImpl aggObj = eventTemplate.getAggregatedTemplate();
		aggObj.setTemplateType(eventTemplate.getTemplateType());
		ID aggObjectID = m_aggDAO.createObject(aggObj, true);
		eventTemplate.setID(aggObjectID);
		super.createObject(eventTemplate, true);

		try {
			if (!bBatched)
				m_dmo.executeBatch();
		} catch (JdmoException e) {
			throw new BbmCreateException(e);
		}
		return aggObjectID;
	}

	/**
	 * update a value object, save its state to database
	 */
	@Override
	public void updateObject(T objValue) throws MultiUserException, BbmUpdateException {
		T eventTemplate = objValue;
		CalendarEventTemplateImpl aggObj = eventTemplate.getAggregatedTemplate();
		m_aggDAO.updateObject(aggObj);
		super.updateObject(eventTemplate);
	}

	/**
	 * delete objects given a list of their ids
	 */
	@Override
	public void deleteObjects(Collection<? extends ID> objID) throws BbmRemoveException {
		/**
		 * cascade delete procedure should take care of deleting the records
		 * from the linked tables
		 */
		m_aggDAO.deleteObjects(objID);
	}

	/**
	 * read object from database given its ID.
	 */
	public T getObjectByID(ID id) throws BbmObjectNotFoundException, BbmFinderException {
		T templateObj = super.getObjectByID(id);
		m_aggDAO.loadChildObjects(templateObj.getAggregatedTemplate());
		return templateObj;
	}

	/**
	 * get a collection of objects given a filter condition
	 */
	public Collection<T> getObjects(String strWhere) throws BbmFinderException {
		Collection<T> templates = super.getObjects(strWhere);
		List<CalendarEventTemplateImpl> arrParents = new ArrayList<CalendarEventTemplateImpl>(templates.size());
		for (Iterator<T> it = templates.iterator(); it.hasNext();) {
			T templateObj = it.next();
			arrParents.add(templateObj.getAggregatedTemplate());
		}
		m_aggDAO.loadChildObjects(arrParents);
		return templates;
	}

	/**
	 * This is an override from the base class create a value object and fill
	 * its state from database
	 * 
	 * @param rs
	 *            dmoRowset handle from where the object is intialized
	 */
	public T readObjectFromDB(JdmoRowset rs) throws Exception {
		T templateObj = super.readObjectFromDB(rs);

		CalendarEventTemplateImpl base = (CalendarEventTemplateImpl) m_aggDAO.readObjectFromDB(rs);

		if (templateObj != null) {
			templateObj.setAggregatedTemplate(base);
		}
		return templateObj;
	}

	// //////////// DK start //////////////
	/**
	 * Returns the contents of the SQL WHERE clause that is used to fetch the
	 * event templates for a given workresource in a given time period
	 */
	public String getWhereClauseForGetEventTemplatesQuery(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd)
			throws JdmoException {
		StringBuffer strSQL = new StringBuffer(100);
		strSQL.append("B.ENDDATE > '");
		strSQL.append(JdmoUtil.formatDBString(dtStart));
		strSQL.append("' AND ");
		if (dtEnd != null) {
			strSQL.append(" B.STARTDATE < '");
			strSQL.append(JdmoUtil.formatDBString(dtEnd));
			strSQL.append("' AND ");
		}
		strSQL.append(" EXISTS ");
		strSQL.append(" (SELECT 1 ");
		strSQL.append("  FROM CALENDAREVENTTEMPLATEATTENDEE ");
		strSQL.append("  WHERE WORKRESOURCEID IN ");
		strSQL.append(m_dmo.createInClause(workResourceIDs));
		strSQL.append("  AND CALENDAREVENTTEMPLATEID = B.ID ) ");
		return strSQL.toString();
	} /* end of getWhereClauseForGetEventTemplatesQuery */

	/**
	 * This method is common for FLOATINGEVENTTEMPLATE and
	 * RECURRINGEVENTTEMPLATE
	 */
	public Collection<T> getAllEventTemplatesForWorkResources(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd)
			throws BbmFinderException {
		try {
			return getObjects(getWhereClauseForGetEventTemplatesQuery(workResourceIDs, dtStart, dtEnd));
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}

	protected void calcAndSetEventPrecedence(T template) throws Exception {
		int precedence = EventPrecedenceRule.getEventOverlayPrecedence(template);
		template.setOverlayPrecedence(precedence);
	}
}
