public class PublishingPeriodDAO extends DAOBase<PublishingPeriod> {
	private static FieldInfo m_fieldInfo = new PublishingPeriodFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public PublishingPeriodDAO() {
		super();
	}

	public PublishingPeriodDAO(Jdmo dmo) {
		super(dmo);
	}

	public PublishingPeriod createValueObject() {
		return new PublishingPeriod();
	}

	/**
	 * For a given set of work resources, returns a collection of published
	 * periods that overlap with the given time window note: return all the
	 * periods overlapped with the time range. including both partial and full
	 * publishing
	 */
	public List<Collection<PublishingPeriod>> getPublishedPeriods(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd)
			throws BbmFinderException {
		try {
			StringBuffer buf = new StringBuffer(100);
			buf.append(" A.WORKRESOURCEID IN ");
			buf.append(m_dmo.createInClause(workResourceIDs));
			if (dtStart != null) {
				buf.append(" AND A.ENDTIME > '").append(JdmoUtil.formatDBString(dtStart)).append("'");
			}
			if (dtEnd != null) {
				buf.append(" AND A.STARTTIME < '").append(JdmoUtil.formatDBString(dtEnd)).append("'");
			}
			buf.append(" order by A.STARTTIME ASC ");// QA99660

			Collection<PublishingPeriod> periods = getObjects(buf.toString());

			HashMap<ID, List<PublishingPeriod>> mapPeriods = ValueObjectUtil.groupByField(PublishingPeriodFields.WORKRESOURCEID, periods);

			/* repackage as a list */
			List<Collection<PublishingPeriod>> listPeriods = new ArrayList<Collection<PublishingPeriod>>(workResourceIDs.size());
			for (Iterator<ID> it = workResourceIDs.iterator(); it.hasNext();) {
				ID workResourceID = it.next();
				listPeriods.add(mapPeriods.get(workResourceID));
			}
			return listPeriods;
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}

	/**
	 * Record the fact that the schedule has been published for this set of
	 * employees and this time window
	 */
	public void publishPeriods(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd)
			throws BbmSchedulePublishingException {
		publishPeriods(workResourceIDs, dtStart, dtEnd, false);
	}

	public void publishPeriods(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd, boolean isPartialPublish)
			throws BbmSchedulePublishingException {
		unPublishPeriods(workResourceIDs, dtStart, dtEnd, isPartialPublish);
		try {
			List<PublishingPeriod> listToCreate = new ArrayList<PublishingPeriod>(workResourceIDs.size());
			Date cur = new Date();
			for (Iterator<ID> it = workResourceIDs.iterator(); it.hasNext();) {
				ID workResourceID = it.next();
				PublishingPeriod newPeriod = new PublishingPeriod();
				newPeriod.setWorkResourceID(workResourceID);
				newPeriod.setStartTime(dtStart);
				newPeriod.setEndTime(dtEnd);
				newPeriod.setPublishTime(cur); // current time
				newPeriod.setIsPartialPublish(isPartialPublish);
				listToCreate.add(newPeriod);
			}
			if (!listToCreate.isEmpty())
				createObjects(listToCreate);
		} catch (Exception e) {
			throw new BbmSchedulePublishingException(e);
		}
	}

	/**
	 * Record the fact that the schedule has been un-published for this set of
	 * employees and this time window
	 */
	public void unPublishPeriods(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd)
			throws BbmSchedulePublishingException {
		unPublishPeriods(workResourceIDs, dtStart, dtEnd, false);
	}

	public void unPublishPeriods(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd, boolean isPartialPublish)
			throws BbmSchedulePublishingException {
		/*
		 * here we shall try to minimize the number of records that are stored
		 * in the PUBLISHINGPERIOD table, so we shall go through the existing
		 * records and remove records that are no longer needed and update the
		 * ones that need to be updated
		 */
		try {
			StringBuffer buf = new StringBuffer(100);
			buf.append(" A.WORKRESOURCEID IN ");
			buf.append(m_dmo.createInClause(workResourceIDs));
			if (isPartialPublish)
				// for partial unpublish, only remove the partial publish
				// period.
				buf.append(" AND A.ISPARTIALPUBLISH = ").append(JdmoUtil.formatBool(isPartialPublish));
			buf.append(" AND A.STARTTIME < '" + JdmoUtil.formatDBString(dtEnd));
			buf.append("' AND A.ENDTIME > '" + JdmoUtil.formatDBString(dtStart));
			buf.append("'");

			/* find the overlapping periods */
			HashMap<ID, Collection<PublishingPeriod>> mapOldPeriods = ValueObjectUtil.groupByField(PublishingPeriodFields.WORKRESOURCEID,
					getObjects(buf.toString()));

			List<ID> listToDelete = new ArrayList<ID>();
			List<PublishingPeriod> listToUpdate = new ArrayList<PublishingPeriod>();
			List<PublishingPeriod> listToCreate = new ArrayList<PublishingPeriod>();

			for (Iterator<ID> it = workResourceIDs.iterator(); it.hasNext();) {
				ID workResourceID = it.next();
				Collection<PublishingPeriod> oldPeriods = mapOldPeriods.get(workResourceID);
				if (oldPeriods != null && !oldPeriods.isEmpty()) {
					for (Iterator<PublishingPeriod> it2 = oldPeriods.iterator(); it2.hasNext();) {
						PublishingPeriod oldPeriod = it2.next();

						if (oldPeriod.getStartTime().before(dtStart)) {
							if (oldPeriod.getEndTime().after(dtEnd)) {
								/*
								 * if this is the case, we'll need to split this
								 * old period into two
								 */
								PublishingPeriod newPeriod = new PublishingPeriod();
								newPeriod.setWorkResourceID(workResourceID);
								newPeriod.setStartTime(oldPeriod.getStartTime());
								newPeriod.setEndTime(dtStart);
								newPeriod.setPublishTime(oldPeriod.getPublishTime());
								newPeriod.setIsPartialPublish(oldPeriod.isPartialPublish());
								listToCreate.add(newPeriod);

								oldPeriod.setStartTime(dtEnd);
								listToUpdate.add(oldPeriod);
							} else {
								oldPeriod.setEndTime(dtStart);
								listToUpdate.add(oldPeriod);
							}
						} else {
							if (oldPeriod.getEndTime().after(dtEnd)) {
								oldPeriod.setStartTime(dtEnd);
								listToUpdate.add(oldPeriod);
							} else {
								listToDelete.add(oldPeriod.getID());
							}
						}
					}
				}
			}

			if (!listToDelete.isEmpty()) {
				deleteObjects(listToDelete);
			}

			if (!listToUpdate.isEmpty()) {
				updateObjects(listToUpdate);
			}

			if (!listToCreate.isEmpty()) {
				createObjects(listToCreate);
			}
		} catch (Exception e) {
			throw new BbmSchedulePublishingException(e);
		}
	}

	// override the deleteObjects method defined in DAOBase
	public void deleteObjects(Collection colIDs) throws BbmRemoveException {
		if (colIDs == null || colIDs.isEmpty())
			return;
		try {
			// delete records in publishingperiod table
			StringBuffer stmt = new StringBuffer(100);
			stmt.append("DELETE PUBLISHINGPERIOD WHERE ID IN " + m_dmo.createInClause(colIDs));
			m_dmo.execute(stmt.toString());
		} catch (Exception ex) {
			new BbmRemoveException(ex);
		}
	}
}
