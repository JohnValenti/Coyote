package com.bluepumpkin.ejb.bbm.schedule.ejb;

import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventTemplateAttendee;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventTemplateAttendeeFields;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

class CalendarEventTemplateAttendeeDAO extends DAOBase {
	// meta data info.
	private static FieldInfo m_fieldInfo = new CalendarEventTemplateAttendeeFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public CalendarEventTemplateAttendeeDAO() {
		super();
	}

	public CalendarEventTemplateAttendeeDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ValueObjectBase createValueObject() {
		return new CalendarEventTemplateAttendee();
	}

	/**
	 * delete objects given either: a list of their ids or a list of all fields
	 * that uniquely identify each object. For calendar event template attendee
	 * this means workresourceID + calendar event template ID
	 */
	public void deleteObjects(Collection objs) throws BbmRemoveException {
		if (objs != null && !objs.isEmpty()) {
			Object obj = objs.iterator().next();
			if (obj instanceof ID) {
				// this is a collection of IDs. The super class knows how to
				// deal with that
				super.deleteObjects(objs);
			} else {
				// this is most likely a collection of CalendarEventAttendee
				// objects
				// with WORKRESOURCEID and CALENDAREVENTTEMPLATEID fields
				// initialized
				try {
					Iterator it = objs.iterator();
					while (it.hasNext()) {
						obj = it.next();
						if (obj instanceof CalendarEventTemplateAttendee) {
							CalendarEventTemplateAttendee attendee = (CalendarEventTemplateAttendee) obj;
							StringBuffer sqlBuf = new StringBuffer();
							sqlBuf.append("DELETE FROM CALENDAREVENTTEMPLATEATTENDEE ");
							sqlBuf.append("WHERE ");
							sqlBuf.append("WORKRESOURCEID = ");
							sqlBuf.append(attendee.getWorkResourceID());
							sqlBuf.append(" AND CALENDAREVENTTEMPLATEID = ");
							sqlBuf.append(attendee.getParentID());
							m_dmo.addBatch(sqlBuf.toString());
						}
					}
					m_dmo.executeBatch();
				} catch (JdmoException e) {
					throw new BbmRemoveException(e);
				}
			}
		}
	}
}
