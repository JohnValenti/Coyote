public class QueueServiceDelegate {
	private static final long serialVersionUID = 1L;
	
	private static Category CAT = Log.initCategory(QueueServiceDelegate.class.getName());

	private WorkloadManager workloadManager = null;
	private WorkResourceManager workResourceManager = null;

	public List<QueueNotPersistedInfo> createOrUpdate(Collection<QueueDTO> queueDTOs, String systemTypeName)
			throws ServiceApplicationException {

		SystemType systemType = SystemType.valueOf(systemTypeName);
		CAT.info("synchFrom(): systemType.name(): " + systemType.name());
		List<QueueNotPersistedInfo> queueNotPersistedInfos = new ArrayList<QueueNotPersistedInfo>();

		if(queueDTOs==null || queueDTOs.isEmpty()){
			CAT.debug("No queues to create/ update");
			return queueNotPersistedInfos;
		}

		// Convert From DTO's to entities so that the EJB layer can handle them
		Collection<QueueWithExternalIDWrapper> newQueueWrappers = new ArrayList<>();
		Collection<QueueWithExternalIDWrapper> oldQueueWrappers = new ArrayList<>();

		try {
			convertToEntity(queueDTOs, systemType, newQueueWrappers, oldQueueWrappers, queueNotPersistedInfos);
			CAT.info("Succeeded to convert DTOs!");
		} catch(RuntimeException e) {
			CAT.error("Unrecoverable system bug while converting DTO to entity Queue", e);
			throw new ServiceApplicationException(e);
		}

		CAT.info("createOrUpdate(): new and old queues size() resp.: " + newQueueWrappers.size() + ", "
				+ oldQueueWrappers.size());

		if (workloadManager == null) {
			CAT.error("In createOrUpdate method: ejb manager does not exist");
		}

		// Persist the new and old queues
		try {
			for (QueueWithExternalIDWrapper newQueueWrapper : newQueueWrappers) {
				workloadManager.createQueue(newQueueWrapper.getQueue(), systemType, newQueueWrapper.getQueueExernalID());
			}

			// persist update the old queues
			for (QueueWithExternalIDWrapper oldQueueWrapper : oldQueueWrappers) {
				workloadManager.updateQueue(oldQueueWrapper.getQueue());

				if( oldQueueWrapper.getQueueExernalID() != null) {
					ID wfoQueueSID = MapUtil.getExternalToWfoID(systemType,
							MapUtil.ObjectType.Queue,
							oldQueueWrapper.getQueueExernalID());
					if(wfoQueueSID == null) {
						workloadManager.matchQueueWithExternalID(
								oldQueueWrapper.getQueue(),
								systemType,
								oldQueueWrapper.getQueueExernalID());
					} 
				}
			}
			// Note: any problem with validity or database state should have been detected by now
			// So any exceptions thrown here are unrecoverable system bugs
			// They are captured separately in case invoked method's contract changes
		} catch(BbmCreateException e) {
			CAT.error("Failed to create the queue!", e);
			throw new ServiceApplicationException("Failed to create the queue!", e);
		} catch(BbmUpdateException e) {
			CAT.error("Failed to update the queue!", e);
			throw new ServiceApplicationException("Failed to update the queue!", e);
		} catch(RemoteException e) {
			CAT.error("There was a communication fault invoking workloadManager", e);
			throw new ServiceApplicationException("There was a communication fault invoking workloadManager", e);
		} catch(RuntimeException e) {
			CAT.error("Failed to persist the queue due to Runtime Exception", e);
			throw new ServiceApplicationException("Failed to persist the queue due to Runtime Exception", e);
		} catch(MultiUserException e) {
			CAT.error("Failed to persist the queue due to Database lock", e);
			throw new ServiceApplicationException("Failed to persist the queue due to Database lock", e);
		} catch(Exception any) {
			CAT.error("failed to Create/Update Queue!", any);
			throw new ServiceApplicationException(any);
		}
		return queueNotPersistedInfos;
	}

	@WebMethod
	public MapJAXBType areSaved(
			@WebParam(name = "queueExternalSourceIds") Collection<String> queueExternalSourceIds,
			@WebParam(name = "systemTypeName") String systemTypeName)
					throws ServiceApplicationException {
		return SyncUtil.areSaved(queueExternalSourceIds, systemTypeName, MapUtil.ObjectType.Queue);
	}

	// ---------------------------------------------------------------------- helpers

	@PostConstruct
	public void postConstruct() {
		try {
			workloadManager = WfmManagerFactory.getWorkloadManager();
			workResourceManager = BbmManagerFactory.getWorkResourceManager();
			CAT.info("Succeeded to obtain reference to managers :)");
		} catch(BbmEJBCreateException e) {
			CAT.error("Failed to obtain reference to managers :(", e);
		}
	}

	@PreDestroy
	public void preDestroy() {
		workloadManager = null;
		workResourceManager = null;
	}

	protected void convertToEntity(
			Collection<QueueDTO> queueDTOs,
			SystemType systemType,
			Collection<QueueWithExternalIDWrapper> newQueues,
			Collection<QueueWithExternalIDWrapper> oldQueues,
			List<QueueNotPersistedInfo> queueNotPersistedInfos)
					throws ServiceApplicationException {

		// Clear the external id cache in case its stale
		MapUtil.flushMap(systemType);
		// Refresh the org name cache too in case its stale
		SyncUtil.getOrganizationIDsByNamesCaseInsensitive(Collections.<String> emptyList(), workResourceManager, true);

		for (QueueDTO queueDTO : queueDTOs) {
			addQueueDTOToCollection(systemType, queueDTO, queueNotPersistedInfos, newQueues, oldQueues);
		}
	}

	private void addQueueDTOToCollection(
			SystemType systemType,
			QueueDTO queueDTO,
			Collection<QueueNotPersistedInfo> queueNotPersistedInfos,
			Collection<QueueWithExternalIDWrapper> newQueues,
			Collection<QueueWithExternalIDWrapper> oldQueues)
					throws ServiceApplicationException {
		// We will create exception info; only add to collection if there is an exception
		QueueNotPersistedInfo queueNotPersistedInfo = new QueueNotPersistedInfo(queueDTO.getExternalSourceId());
		// TODO verify name is unique
		validateQueueDTOExternalDAndName(queueDTO, queueNotPersistedInfo);

		ID orgID = getOrgIdFromQueueDTO(queueDTO, queueNotPersistedInfo);

		if (StringUtil.isEmptyOrWhiteSpace(queueDTO.getMediaId())) {
			queueNotPersistedInfo.addMediaIdMissingMessage();
		}


		if (!queueNotPersistedInfo.getMessages().isEmpty()) {
			CAT.error("User error: Activity entity will not be persisted");
			queueNotPersistedInfos.add(queueNotPersistedInfo);
		} else {
			QueueWithExternalIDWrapper queueWrapper = new QueueWithExternalIDWrapper();
			ID queueId = wrapQueue(systemType, orgID, queueDTO, queueWrapper);

			// sort into new and old
			if (queueId == null) {
				newQueues.add(queueWrapper);
			} else {
				oldQueues.add(queueWrapper);
			}
		}
	}

	private void validateQueueDTOExternalDAndName(QueueDTO queueDTO, QueueNotPersistedInfo queueNotPersistedInfo) {
		// Validation
		if (StringUtil.isEmptyOrWhiteSpace(queueDTO.getExternalSourceId())) {
			queueNotPersistedInfo.addExternalIdMissingMessage(queueDTO.getName());
		} else if (queueDTO.getExternalSourceId().length() > EntityImportService.EXTERNALID_MAX_LENGTH) {
			queueNotPersistedInfo.addExternalIdLengthExceededMessage(
					queueDTO.getExternalSourceId().length(),
					EntityImportService.EXTERNALID_MAX_LENGTH);
		}

		if (StringUtil.isEmptyOrWhiteSpace(queueDTO.getName())) {
			queueNotPersistedInfo.addNameMissingMessage();
		} else if(StringUtil.containsInvalidChars(queueDTO.getName(), ValidatorConstants.INVALID_NAME_CHARS)){
			queueNotPersistedInfo.addNameHasInvalidCharactersMessage(ValidatorConstants.INVALID_NAME_CHARS);
		} else if (queueDTO.getName().length() > EntityImportService.NAME_MAX_LENGTH) {
			queueNotPersistedInfo.addNameLengthExceededMessage(
					queueDTO.getName().length(),
					EntityImportService.NAME_MAX_LENGTH);
		}
	}

	private ID getOrgIdFromQueueDTO(QueueDTO queueDTO, QueueNotPersistedInfo queueNotPersistedInfo)
			throws ServiceApplicationException {
		ID orgID = null;
		if (StringUtil.isEmptyOrWhiteSpace(queueDTO.getOrganizationName())) {
			queueNotPersistedInfo.addOrganizationNameMissingMessage();
		} else {
			orgID = SyncUtil.getOrganizationIDByNameCaseInsensitive(
					queueDTO.getOrganizationName(),
					workResourceManager,
					false);
			CAT.info("Mapped org external id to: " + orgID);
			if (orgID == null) {
				queueNotPersistedInfo.addNoSuchOrganizatonNameMessage(queueDTO.getOrganizationName());
			}
		}
		return orgID;
	}

	private ID wrapQueue(SystemType systemType, ID orgID, QueueDTO queueDTO, QueueWithExternalIDWrapper queueWrapper)
			throws ServiceApplicationException {
		// Translate DTO properties to queue entity
		Queue queue = new Queue();
		queueWrapper.setQueue(queue);
		ID queueExternalID = ExternalIDUtil.buildExternalID(systemType, queueDTO.getExternalSourceId());
		queueWrapper.setQueueExernalID(queueExternalID);

		// lookup the WFO ID
		try {
			ID wfoQueueSID = MapUtil.getExternalToWfoID(
					systemType,
					MapUtil.ObjectType.Queue,
					queueWrapper.getQueueExernalID());
			if(wfoQueueSID == null) {
				if (workloadManager == null) {
					CAT.error("QueueServiceDelegate failed to obtain reference to WorkloadManager session bean!");
					throw new ServiceApplicationException(
							"QueueServiceDelegate failed to obtain reference to WorkloadManager session bean!");
				}
				wfoQueueSID = workloadManager.getQueueIDByName(queueDTO.getName());
			}
			queue.setID(wfoQueueSID);
			CAT.info("Mapped external id to: " + wfoQueueSID);
		} catch(RemoteException e) {
			throw new ServiceApplicationException(e);
		} catch(BbmObjectNotFoundException e) {
			throw new ServiceApplicationException(e);
		} catch(BbmFinderException e) {
			throw new ServiceApplicationException(e);
		} catch(JdmoException e) {
			throw new ServiceApplicationException(e);
		}

		queue.setExternalID(queueExternalID);
		queue.setOrganizationID(orgID);
		queue.setName(queueDTO.getName());
		queue.setDescription(queueDTO.getDescription());
		queue.setMediaStrID(queueDTO.getMediaId());

		return queue.getID();
	}

	public static class QueueWithExternalIDWrapper {
		protected Queue queue;
		protected ID queueExernalID;

		public Queue getQueue() {
			return queue;
		}

		public void setQueue(Queue queue) {
			this.queue = queue;
		}

		public ID getQueueExernalID() {
			return queueExernalID;
		}

		public void setQueueExernalID(ID queueExernalID) {
			this.queueExernalID = queueExernalID;
		}
	}
}
