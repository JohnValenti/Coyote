package com.bluepumpkin.web.rm.requests.timeoff;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.workresource.OrganizationScopeManagerProxy;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.setup.allocations.TimeOffPoolListPM;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarTextButton;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.MultiColPCFactory;

public class TimeOffPoolListCalPM extends TimeOffPoolListPM
{
	private static final Category log = Log.initCategory(TimeOffPoolListCalPM.class.getName());
	
	/**
	 * Constructor
	 */
	public TimeOffPoolListCalPM(RequestContext context) {
		super(context);
		setName("TimeOffPoolListCalPM");
		if (isAccessibilityComplianceMode()) {
			m_list.setUseWrapper(true);
		}
	}

	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static PageModel getInstance(RequestContext context) 
	{
		return getInstance(context, TimeOffPoolListCalPM.class);
	}
	
	@Override
	protected MultiColListPC getMultiColListPC(RequestContext context) {
		return MultiColPCFactory.getMultiColListPC(context, true, true,
				i18n(m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME), RmWebBundleKey.TIME_OFF_POOLS_GRID), false,
				true, true);
	}
	
	/**
	 * Initialize toolbar
	 */
	protected void initToolbar() 
	{
		String viewLabel = i18n(m_bbmBundle, BbmWebBundleKey.TOOLBAR_VIEW);
		ToolbarTextButton viewBtn = m_toolbar.addSubmitTextButton(PageAction.VIEW, viewLabel, isAuthorizedToView() && getPoolID()!=null);
		
		viewLabel = i18n(m_bbmBundle, BbmWebBundleKey.PEOPLE_TIMEOFF_POOL_VIEW_ORGANIZATION);		
		ToolbarTextButton viewOrgBtn = m_toolbar.addSubmitTextButton(PageAction.VIEW_DETAIL, viewLabel, isAuthorizedToView()); //VIEW_DETAIL will mean view org
		
		if (m_isSingleItemEdit) 
			viewBtn.setListButtonItems(1, 1);
	}
	
	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return "request_processing_topoollist_cal";
	}

	protected boolean isAuthorizedToView() {
		return isValidSelectedOrg() &&
		        (m_context.getUser().isAuthorized(PrivilegeKeys.TOM_VIEWTIMEOFFCALENDAR,OrganizationScopeManagerProxy.getScopeID(getSelectedIDObject()))
		        ||m_context.getUser().isAuthorized(PrivilegeKeys.TOM_VIEWAVAILABLETIMEOFF,OrganizationScopeManagerProxy.getScopeID(getSelectedIDObject())));
	}

	protected boolean checkViewPrivilege(ID idObj) {
		return isValidSelectedOrg() &&
		        (m_context.getUser().isAuthorized(PrivilegeKeys.TOM_VIEWTIMEOFFCALENDAR,OrganizationScopeManagerProxy.getScopeID(idObj))
				||m_context.getUser().isAuthorized(PrivilegeKeys.TOM_VIEWAVAILABLETIMEOFF,OrganizationScopeManagerProxy.getScopeID(getSelectedIDObject())));
	}

	/**
	 * Return true if current user is authorized to view time off for the specified Org
	 */
	protected boolean isAuthorizedToView(ID orgID) {
		return isAuthorizedToView() && checkViewPrivilege(orgID);
	}

	private String getIsAuthorizedToViewString() {
		return isAuthorizedToView() ? "true" : "false";
	}

	protected boolean isAuthorizedToConfigure() 
	{
		return false;
	}
	
	/**
	 * Is the list row editable? A row is editable if the row's organization
	 * is the same as the selected organization.
	 *
	 * @param rowOrgID the row's organization ID
	 * @return "true" if a row is editable, "false" otherwise
	 *
	 */
	protected String isRowEditable(ID rowOrgID) {
		return getIsAuthorizedToViewString();
	}
	
}
