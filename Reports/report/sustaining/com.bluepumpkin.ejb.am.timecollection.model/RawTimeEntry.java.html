	/**
	 * As in DB we lost Seconds, this is to get real StartTime precision
	 * 
	 * @param dateToRound
	 * @return
	 */
	public static Date roundSeconds(Date dateToRound) {
		long punchTimeInMillis = dateToRound.getTime() + 500;
		long punchTimeInMinutes = punchTimeInMillis - punchTimeInMillis % 1000;
		return new Date(punchTimeInMinutes);
	}
