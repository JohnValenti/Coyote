package com.bluepumpkin.ejb.am.payrollmanager.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.am.Log;
import com.bluepumpkin.ejb.am.payrollmanager.model.PayrollStatus;
import com.bluepumpkin.ejb.core.base.SessionEJBBase;

/**
 * Title:        Blue Pumpkin Software Activity Management
 * Description:  Simple Wrapper over PayrollStatusDAO&PayrollExportErrorDAO with TX_NEW
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author       Sheng Song
 * @version 1.0
 */

public class PayrollStatusManagerEJB extends SessionEJBBase {

	private static Category m_cat = Log.initCategory(PayrollStatusManagerEJB.class.getName());
	  /** override the base class to provide the appropriate logging category */
	protected Category getCategory() { return m_cat; }

	{
		super.init(PayrollStatusManagerEJB.class.getName());
	}

	public ID createPayrollStatus(PayrollStatus status) throws JdmoException
	{
		return PayrollStatusDAO.createPayrollStatus(status);
	}
	public void updateStatus(ID requestID, int status) throws JdmoException
	{
		PayrollStatusDAO.updateStatus(requestID, status);
	}
	public ID createPayrollExportError(ID statusID, String bundle, String key, Object[] param) throws JdmoException
	{
		return PayrollExportErrorDAO.createPayrollExportError(statusID, bundle, key, param);
	}
}
