package com.verint.ejb.bbm.fterequirements;

import java.util.Collection;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;

/**
 * A static class with methods for initiating the recalculation of
 * FTE Requirements for SPQueues.
 */
public class FteRequirementsUtil {
	private static int BATCH_SIZE = 50;		//The number of SPQueues that we will process (calculate FTE Requirements for) for each TimeSeriesMessage sent to the JMS queue.
	private FteRequirementsUtil() {};
	
	/**
	 * Asynchronously calculates FTE Requirements for the given set of SP Queues.  This method creates
	 * TimeSeriesMessages which batch the given spQueueIDs and then sends them to a JMS queue where they then
	 * have their FTE Requirements calculated (see TimeSeriesHandlerMDB).
	 * 
	 * SPQueues should have their FTE Requirements recalculated whenever their forecasts, service goals,
	 * or shrinkage values have been modified and persisted.
	 */
	public static void recalculateFteRequirementsForSPQueues(Collection<ID> spQueueIDs, boolean isWhatIf) throws BbmUpdateException {
		recalculateFteRequirementsForSPQueues(spQueueIDs, null, isWhatIf);
	}
	
	/**
	 * Asynchronously calculates FTE Requirements for the given set of SP Queues.  This method creates
	 * TimeSeriesMessages which batch the given spQueueIDs and then sends them to a JMS queue where they then
	 * have their FTE Requirements calculated (see TimeSeriesHandlerMDB).  Forecast data (in TraceCube format)
	 * may also be handed in as an optional parameter to speed up processing time, but is not required (null may
	 * be passed in, and if this is the case then the forecast data will be loaded dynamically as needed).
	 * 
	 * SPQueues should have their FTE Requirements recalculated whenever their forecasts, service goals,
	 * or shrinkage values have been modified and persisted.
	 */
	public static void recalculateFteRequirementsForSPQueues(Collection<ID> spQueueIDs,
			Map<? extends ID, ? extends TraceCube> forecastsBySPQueueID, boolean isWhatIf) throws BbmUpdateException {
		if (spQueueIDs != null && spQueueIDs.size() > 0) {
			HashSet<ID> spQueueIDBatch = new HashSet<ID>();		//Use a HashSet so we don't needlessly calculate the same sp queue multiple times in case the same ID was passed in twice
			for (ID spQueueID : spQueueIDs) {
				//For each sp queue ID in the list to process, we add it to the batch.  When the batch reaches the defined batch size,
				//we will send a TimeSeriesMessage to a JMS queue which will kick off the FTE Requirements calculations for the batched
				//SP queues asynchronously.
				spQueueIDBatch.add(spQueueID);
				if (spQueueIDBatch.size() >= BATCH_SIZE) {
					TimeSeriesMessage message = new TimeSeriesMessage(spQueueIDBatch,
							getForecastsForBatch(spQueueIDBatch, forecastsBySPQueueID), isWhatIf);
					message.sendMessage();
					spQueueIDBatch.clear();
				}
			}
			//Calculate the FTE Reqs for the remaining SPQueues in the list that were not large enough to make a full batch in
			//the previous loop.
			if (spQueueIDBatch.size() > 0) {
				TimeSeriesMessage message = new TimeSeriesMessage(spQueueIDBatch,
						getForecastsForBatch(spQueueIDBatch, forecastsBySPQueueID),
						isWhatIf);
				message.sendMessage();
			}
		}
	}
	
	/**
	 * Takes a larger set of mapped forecast data and returns a smaller set
	 * (i.e. a smaller map) which only includes data for SPQueues in the given
	 * list of IDs spQueueIDBatch.
	 * @param spQueueIDBatch - The SPQueue IDs for which we'd like to retrieve the mapped
	 * forecast data.
	 * @param forecastsBySPQueueID - The map of forecast data for the entire operation (not
	 * just the one batch).
	 */
	private static Map<ID, TraceCube> getForecastsForBatch(
			Collection<ID> spQueueIDBatch,
			Map<? extends ID, ? extends TraceCube> forecastsBySPQueueID) {
		if (forecastsBySPQueueID == null) return null;
		Map<ID, TraceCube> forecastsBySPQueueIDBatch = new HashMap<ID, TraceCube>();
		for (ID spQueueID : spQueueIDBatch) {
			TraceCube forecast = forecastsBySPQueueID.get(spQueueID);
			if (forecast != null) {
				forecastsBySPQueueIDBatch.put(spQueueID, forecast);
			}
		}
		return forecastsBySPQueueIDBatch;
	}
}
