package com.verint.web.rm.rest.facade.timeoff;

import java.rmi.RemoteException;
import java.text.MessageFormat;
import java.util.Calendar;
import java.util.Date;
import java.util.List;
import java.util.TimeZone;

import javax.ws.rs.core.Response;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOCalendarDayData;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendar;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.requests.RequestsMH;
import com.bluepumpkin.web.rm.requests.timeoff.TimeOffRequestsMH;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rm.rest.entity.timeoff.TimeOffDailyCalendarEntity;
import com.verint.web.rm.rest.entity.timeoff.TimeOffIntervalCalendarEntity;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.web.uif.system.RequestContext;

public class TimeOffCalendarFacadeImpl implements ITimeOffCalendarFacade {
	private static final Category LOG = Log.initCategory(TimeOffCalendarFacadeImpl.class.getName());

	public TimeOffCalendarFacadeImpl() { // NOSONAR
	}

	/**
	 * Gets the current employee's daily TOC for the given time range.
	 *
	 * @param isTOPoolView
	 *            True to get the TOP TOC. False to get the employee's TOC.
	 */
	@Override
	@SuppressWarnings("unchecked")
	public TimeOffDailyCalendarEntity getDailyCalendarForEmployee(RequestContext context, TimeRange timeRange, boolean isTOPoolView)
			throws WFOException {
		try {
			assertDateRangeLessThanOrEqualToYear(timeRange, TimeZoneUtil.GMT_TIMEZONE);

			Date startDate = timeRange.getStartDate();
			Date endDate = timeRange.getEndDate();
			ID employeeID = context.getUser().getEmployeeID();
			Organization empOrg = OrganizationModelHandler.getEmployeeOrg(context, employeeID);
			List<TOCalendarDayData> days = TimeOffRequestsMH.getTOCalendarDataForEmployee(employeeID, startDate, endDate, isTOPoolView, empOrg);

			TimeOffDailyCalendarEntity result = new TimeOffDailyCalendarEntity(startDate, endDate, isTOPoolView, days);
			setColors(context, result.getColors(), employeeID);
			return result;
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(ex, LOG);
		}
	}

	/**
	 * Gets the current employee's interval TOC for the given time range.
	 *
	 * @param isTOPoolView
	 *            True to get the TOP TOC. False to get the employee's TOC.
	 */
	@Override
	public TimeOffIntervalCalendarEntity getIntervalCalendarForEmployee(RequestContext context, TimeRange timeRange, boolean isTOPoolView)
			throws WFOException {
		try {
			TimeZone timeZone = context.getViewingTimeZone();
			assertDateRangeLessThanOrEqualToMonths(timeRange, 1, timeZone);
			Date startDate = timeRange.getStartDate();
			Date endDate = timeRange.getEndDate();

			ID employeeID = context.getUser().getEmployeeID();
			TOIntervalCalendar toic = TimeOffRequestsMH.getTOIntervalCalendarForEmp(employeeID, startDate, endDate, timeZone, isTOPoolView);
			TimeOffIntervalCalendarEntity result = new TimeOffIntervalCalendarEntity(toic, isTOPoolView);

			setColors(context, result.getColors(), employeeID);

			return result;
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(ex, LOG);
		}
	}

	private void assertDateRangeLessThanOrEqualToMonths(TimeRange timeRange, int numberOfMonths, TimeZone timeZone)
			throws BusinessWFOException {
		Calendar cal = Calendar.getInstance(timeZone);
		cal.setTime(timeRange.getStartDate());
		cal.add(Calendar.MONTH, numberOfMonths);
		Date maxEndTime = cal.getTime();
		if (maxEndTime.before(timeRange.getEndDate())) {
			throw new BusinessWFOException(MessageFormat.format("Time range cannot be greater than {0} month(s).", numberOfMonths),
					Response.Status.BAD_REQUEST);
		}
	}

	private void assertDateRangeLessThanOrEqualToYear(TimeRange timeRange, TimeZone timeZone) throws BusinessWFOException {
		Calendar cal = Calendar.getInstance(timeZone);
		cal.setTime(timeRange.getStartDate());
		cal.add(Calendar.YEAR, 1);
		Date maxEndTime = cal.getTime();
		if (maxEndTime.before(timeRange.getEndDate())) {
			throw new BusinessWFOException("Time range cannot be greater than 1 year.", Response.Status.BAD_REQUEST);
		}
	}

	public void setColors(RequestContext context, String[] colors, ID employeeID) throws RemoteException, BbmException {
		ID employeeOrgID = OrganizationModelHandler.getEmployeeOrgID(context, employeeID);
		OrganizationSetting orgSetting = RequestsMH.getOrgSetting(employeeOrgID);

		String[] keys = TOCalendarDayData.DAY_TYPE_TO_SETTINGS_KEY;
		colors[TOCalendarDayData.BLACKOUT_DAY_TYPE] = orgSetting.getValue(keys[TOCalendarDayData.BLACKOUT_DAY_TYPE]);
		colors[TOCalendarDayData.NO_TO_HOURS_AVAILABLE_DAY_TYPE] = orgSetting
				.getValue(keys[TOCalendarDayData.NO_TO_HOURS_AVAILABLE_DAY_TYPE]);
		colors[TOCalendarDayData.NON_OPERATION_DAY_TYPE] = orgSetting.getValue(keys[TOCalendarDayData.NON_OPERATION_DAY_TYPE]);
		colors[TOCalendarDayData.PAID_HOLIDAY_DAY_TYPE] = orgSetting.getValue(keys[TOCalendarDayData.PAID_HOLIDAY_DAY_TYPE]);
		colors[TOCalendarDayData.PENDING_DAY_TYPE] = orgSetting.getValue(keys[TOCalendarDayData.PENDING_DAY_TYPE]);
		colors[TOCalendarDayData.SCHEDULED_DAY_TYPE] = orgSetting.getValue(keys[TOCalendarDayData.SCHEDULED_DAY_TYPE]);
		colors[TOCalendarDayData.TO_HOURS_AVAILABLE_DAY_TYPE] = orgSetting.getValue(keys[TOCalendarDayData.TO_HOURS_AVAILABLE_DAY_TYPE]);
		colors[TOCalendarDayData.UNPAID_HOLIDAY_DAY_TYPE] = orgSetting.getValue(keys[TOCalendarDayData.UNPAID_HOLIDAY_DAY_TYPE]);
	}
}
