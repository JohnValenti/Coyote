package com.bluepumpkin.ejb.rm.servlet;

import javax.naming.Context;
import javax.naming.InitialContext;
import javax.transaction.SystemException;
import javax.transaction.UserTransaction;

import com.bluepumpkin.common.base.Log;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.core.CoreManagerFactory;
import com.bluepumpkin.ejb.core.servlet.ServletControl;
import com.bluepumpkin.ejb.rm.util.QuartzUtil;
import com.witness.ejb.core.licensing.LicenseKeys;
import com.witness.ejb.core.licensing.ejb.LicenseManager;

/**
 * @author swati tewari
 *         <p/>
 *         Helper class used to register AppServer during initilization
 *         It will register to DB, and if cache is used to the cache also
 */
public class RmQuartzServletControl implements ServletControl {
	private static final Category m_cat = Log.initCategory(RmQuartzServletControl.class.getName());

	/* (non-Javadoc)
		* @see com.bluepumpkin.ejb.core.servlet.ServletControl#startProcess()
		*/
	public void startProcess() {

		UserTransaction jta_userTX = null;

		try {
			// start a JTA transaction to encompass the database updates done by
			// StdSchedulerFactory.getScheduler() below.  These udpates are related to the recovery of
			// jobs and triggers stored in the DB.  The quartz scheduler is configured to use JTA
			// transactions (via JobStoreCMT) and if the transaction is not started before
			// StdSchedulerFactory.getScheduler() is called, any SQL updates done by this
			// method will be rolledback (JDBC driver dependent: rolledback for weblogic.mssql4 driver
			// but committed if the newer driver weblogic.sqlserver is used).
			Context ctx = new InitialContext();
			jta_userTX = (UserTransaction) ctx.lookup("java:comp/UserTransaction");
			jta_userTX.begin();

			LicenseManager licManager = CoreManagerFactory.getLicenseManager();
			//check for Shift bid license
			boolean hasShiftBidLicense = licManager.isLicensed(LicenseKeys.APP_RM_SB);
			if (hasShiftBidLicense) {
				QuartzUtil.addOrUpdateShiftBidJobsIfNecess();
			} else {
				QuartzUtil.removeShiftBidJobs();
			}
			//Need to add this job irrespective of TIme-off License. because it calculates TO Events as well.
			QuartzUtil.addOrUpdateTOwaitlistPollerBackgroudTasksJobIfNecess();
			QuartzUtil.addOrUpdateTOHoursPerDayUpdateJobIfNecess();
			// call context.close() since an open context affects the security principal associated with
			// this thread (security principals are maintained in the principal stack for each thread).
			ctx.close();

			jta_userTX.commit();
		} catch (Exception e) {
			try {
				if (jta_userTX != null) jta_userTX.rollback();
			} catch (SystemException se) {
				m_cat.error("Cannot rollback transaction", se);
			}

			m_cat.error(e.getMessage(), e);
		}

	}

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.core.servlet.ServletControl#stopProcess()
	 */
	public void stopProcess() {
	}
}
