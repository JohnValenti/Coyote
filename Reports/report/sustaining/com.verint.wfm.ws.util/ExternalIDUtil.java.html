package com.verint.wfm.ws.util;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.dao.MapUtil.SystemType;

public class ExternalIDUtil {

	public static ID buildExternalID(SystemType systemType, String externalIDAsString) throws NumberFormatException {
		ID externalID;
		if (StringUtil.isEmptyOrWhiteSpace(externalIDAsString)) {
			externalID = null;
		} else if (systemType.isExtIdAnInt) {
			externalID = ID.fromInt(Integer.parseInt(externalIDAsString));
		} else {
			externalID = ID.fromString(externalIDAsString);
		}
		return externalID;
	}
}
