package com.verint.wfm.ws.util;

import javax.xml.bind.annotation.XmlAttribute;

/**
 * 
 * @author ANuthi
 *
 *  externalID, name pair to uniquely identify organization
 */

public class OrganizationExternalKey {

		@XmlAttribute
		public String externalId;

		@XmlAttribute
		public String name;
		
		public OrganizationExternalKey() {
		}

		public OrganizationExternalKey(String id, String name) {
			this.externalId = id;
			this.name = name;
		}

}
