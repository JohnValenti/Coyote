package com.verint.web.fs.widgets;

import com.bluepumpkin.web.core.pagecomponent.DropDownSelectionPC;
import com.witness.web.uif.system.RequestContext;

import java.util.HashSet;

@SuppressWarnings("serial")
public class IntegerDropDownSelection extends DropDownSelectionPC<Integer> {

	public IntegerDropDownSelection(RequestContext context, String name, int minVal, int maxVal) {
		this(context, name, minVal, maxVal, new int[0]);
	}

	public IntegerDropDownSelection(RequestContext context, String name, int minVal, int maxVal, int[] selectedOptions) {
		super(context, name);
		if (maxVal - minVal > 100) {
			throw new IllegalArgumentException("A dropdown is not appropriate for such a large range of numbers(>100). Min: "
					+ minVal + ", Max: " + maxVal + ".");
		}
		HashSet<Integer> selected = new HashSet<Integer>();
		for (int i = 0; i < selectedOptions.length; i++) {
			selected.add(selectedOptions[i]);
		}
		for (int i = minVal; i <= maxVal; i++) {
			this.addOption("" + i, "" + i, i, selected.contains(i));
		}
	}

}
