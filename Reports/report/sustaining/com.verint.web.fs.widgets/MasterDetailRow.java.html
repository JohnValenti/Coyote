package com.verint.web.fs.widgets;

import com.witness.web.uif.pagecomponent.MultiColumnNodeData;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;

/**
 * This class represents a row in a table that contains a master record and an associated detail record.
 * It is sortable based on the master row.
 *
 * @author rfunderburg
 */
public class MasterDetailRow implements MultiColumnNodeData {
	private MultiColumnNodeData masterRow;
	private ArrayList<MultiColumnNodeData> detailRows;

	public MasterDetailRow(MultiColumnNodeData masterRow,
			ArrayList<MultiColumnNodeData> detailRows) {
		super();
		this.masterRow = masterRow;
		this.detailRows = detailRows;
	}

	public MasterDetailRow(MultiColumnNodeData masterRow,
			MultiColumnNodeData detailRow) {
		super();
		this.masterRow = masterRow;
		this.detailRows = new ArrayList<MultiColumnNodeData>();
		this.detailRows.add(detailRow);
	}

	public MultiColumnNodeData getMasterRow() {
		return masterRow;
	}

	public ArrayList<MultiColumnNodeData> getDetailRows() {
		return detailRows;
	}

	// ---- MultiColumnNodeData Methods Begin ----
