package com.verint.web.fs.widgets;

import com.bluepumpkin.common.util.StringUtil;
import com.verint.web.fs.workrules.WorkRulesJSFileID;
import com.witness.web.uif.keys.ImageFileID;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.pagecomponent.CompositePC;
import com.witness.web.uif.pagecomponent.PageComponent;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

import javax.servlet.http.HttpServletRequest;
import java.util.ArrayList;
import java.util.Collection;
import java.util.HashSet;

/**
 * Toggles the visibility of a named element
 * <p>
 * This toggle can also be used to toggle other toggles (a master toggle).
 * This is done by adding other ToggleVisiblePC objects as children
 *
 * @author rfunderburg
 */
public class ToggleVisiblePC extends CompositePC {
	private static final long serialVersionUID = 1L;

	private String expandedImg;
	private String collapseImg;
	private boolean expanded = false;
	private Collection<String> idsToBeToggled;
	private String onChangeJS;

	private Collection<ToggleVisiblePC> children = new HashSet<ToggleVisiblePC>();

	public ToggleVisiblePC(RequestContext context, String name) {
		super(context);
		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
		setJSMediator(WorkRulesJSFileID.TOGGLE_VISIBLE_PC);
		setName(name);
	}

	public ToggleVisiblePC(RequestContext context, String name, Collection<String> idsToBeToggled) {
		this(context, name);
		this.idsToBeToggled = idsToBeToggled;
	}

	public ToggleVisiblePC(RequestContext context, String name, Collection<String> idsToBeToggled, boolean expanded) {
		this(context, name, idsToBeToggled);
		this.expanded = expanded;
	}

	public ToggleVisiblePC(RequestContext context, String name, Collection<String> idsToBeToggled, boolean expanded, String expandedImgURL, String collapsedImgURL) {
		this(context, name, idsToBeToggled, expanded);
		this.expandedImg = expandedImgURL;
		this.collapseImg = collapsedImgURL;
	}

	@Override
	public String getHtmlDisplay() {
		if (children.size() == 0) {
			if (expandedImg == null) {
				expandedImg = ImageFileID.EXPAND_NODE_IMG;
			}
			if (collapseImg == null) {
				collapseImg = ImageFileID.COLLAPSE_NODE_IMG;
			}
		} else {
			if (expandedImg == null) {
				expandedImg = ImageFileID.NORGIE_ALL_OPEN;
			}
			if (collapseImg == null) {
				collapseImg = ImageFileID.NORGIE_ALL;
			}
		}
		StringBuffer html = new StringBuffer();
		html.append("<span style='cursor: pointer;' id='").append(getName()).append("__span'>");
		html.append("<img class='expanded_icon' src='").append(expandedImg).append("'").append(expanded ? "" : " style='display: none;'").append(">");
		html.append("<img class='collapsed_icon' src='").append(collapseImg).append("'").append(expanded ? " style='display: none;'" : "").append(">");
		html.append(HtmlUtil.makeHiddenInput(getExpandedFN(), isExpanded()));
		html.append("</span>");
		return html.toString();
	}

	@Override
	public String getJavaScriptInitCode() {
		// idToToggle, isExpanded, children
		Collection<String> childNames = new HashSet<>();
		for (ToggleVisiblePC child : new ArrayList<ToggleVisiblePC>(getChildComponentList())) {
			childNames.add(child.getName());
		}
		this.setJSMediatorArgs((idsToBeToggled != null ? "['" + StringUtil.createDelimitedString(idsToBeToggled, "','") + "']" : "null") + "," + expanded + ",[" + StringUtil.join(childNames, ",") + "], '" + getExpandedFN() + "'");
		StringBuffer code = new StringBuffer(super.getJavaScriptInitCode());
		if (onChangeJS != null) {
			code.append(getName()).append(".change(function() { ").append(onChangeJS).append(" } );\n");
		}
		return code.toString();
	}

	@Override
	public boolean extractParameters(HttpServletRequest request) {
		boolean retVal = super.extractParameters(request);
		String expanded = request.getParameter(getExpandedFN());
		if (expanded != null) {
			setExpanded(Boolean.parseBoolean(expanded));
		}
		return retVal;
	}

	private String getExpandedFN() {
		return getName() + "___expanded";
	}

	/**
	 * The parameter must be a ToggleVisiblePC
	 */
	@Override
	public void addChildComponent(PageComponent toggleVisiblePC) {
		if (!(toggleVisiblePC instanceof ToggleVisiblePC)) {
			throw new IllegalArgumentException("ToggleVisiblePC can only contain other ToggleVisiblePC objects.");
		}
		children.add((ToggleVisiblePC) toggleVisiblePC);
		super.addChildComponent(toggleVisiblePC);
	}

	public boolean isExpanded() {
		return expanded;
	}

	public void setExpanded(boolean expanded) {
		this.expanded = expanded;
	}

	public Collection<String> getIdsToBeToggled() {
		return idsToBeToggled;
	}

	public void setIdsToBeToggled(Collection<String> idsToBeToggled) {
		this.idsToBeToggled = idsToBeToggled;
	}

	/**
	 * Sets any custom logic that needs to be executed when the value changes
	 *
	 * @param onChangeJS
	 */
	public void setOnChangeJS(String onChangeJS) {
		this.onChangeJS = onChangeJS;
	}
}
