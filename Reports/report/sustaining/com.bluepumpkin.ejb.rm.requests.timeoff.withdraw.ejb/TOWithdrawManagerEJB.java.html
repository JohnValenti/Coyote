public class TOWithdrawManagerEJB extends SessionEJBBase {     //OUTSIDE_CONTAINER
    private static final String m_className = TOWithdrawManagerEJB.class.getName();

    //this hashmap caches the Validator instances
    protected HashMap s_validatorsMap = new HashMap();
    {
        super.init(TOWithdrawManagerEJB.class.getName());
    }

    /**
     * override the base class to provide the appropriate logging category
     */
    @Override
	protected Category getCategory() {
        return m_cat;
    }

    public void updateTOWithdrawRequest(TOWithdraw pTOWithdraw) throws MultiUserException, BbmUpdateException {
        TOWithdrawDAO TOWithdrawDAO = null;
        try {
            TOWithdrawDAO = new TOWithdrawDAO();
            TOWithdrawDAO.updateTOWithdrawRequest(pTOWithdraw);
        } finally {
            if (TOWithdrawDAO != null) TOWithdrawDAO.cleanUp();
        }
    }

    public void deleteTOWithdrawRequestSetting(ID pTOWithdrawID) throws BbmRemoveException {
        TOWithdrawDAO TOWithdrawDAO = null;
        try {
            TOWithdrawDAO = new TOWithdrawDAO();
            TOWithdrawDAO.deleteObject(pTOWithdrawID);
        } finally {

            if (TOWithdrawDAO != null) TOWithdrawDAO.cleanUp();
        }
    }

    public ID createTOWithdrawRequest(TOWithdraw pTOWithdraw) throws RmException, BbmCreateException {
        TOWithdrawDAO TOWithdrawDAO = null;
        try {
            TOWithdrawDAO = new TOWithdrawDAO();
            return TOWithdrawDAO.createTOWithdrawRequest(pTOWithdraw);
        } finally {
            if (TOWithdrawDAO != null) TOWithdrawDAO.cleanUp();
        }
    }

    public TOWithdraw getTOWithdrawByTimeOffRequestId(ID toRequestId) throws BbmFinderException {
        TOWithdrawDAO TOWithdrawDAO = null;
        try {
            TOWithdrawDAO = new TOWithdrawDAO();
            return TOWithdrawDAO.getWithdrawReqByTimeOffRequestId(toRequestId);
        } finally {
            if (TOWithdrawDAO != null) TOWithdrawDAO.cleanUp();
        }
    }

    public Collection getTOWithdrawsByTimeOffRequestIds(Collection toReqCol) throws BbmFinderException {
        TOWithdrawDAO TOWithdrawDAO = null;
        try {
            TOWithdrawDAO = new TOWithdrawDAO();
            return TOWithdrawDAO.getObjectsByParentIDs(toReqCol);
        } finally {
            if (TOWithdrawDAO != null) TOWithdrawDAO.cleanUp();
        }
    }

    public HashMap getTOWithdrawTORequestPairs(Collection toReqCol) throws BbmFinderException {
        HashMap hmap = new HashMap();
        Collection col = getTOWithdrawsByTimeOffRequestIds(toReqCol);
        for (Iterator iterator = col.iterator(); iterator.hasNext();) {
            TOWithdraw TOWithdraw = (TOWithdraw) iterator.next();
            hmap.put(TOWithdraw.getTORequestID(), TOWithdraw);
        }
        return hmap;
    }
    protected static CommonRequestManager getCommonRequestManager() throws BbmCreateException {
        return RmManagerFactory.getInstance(true).getCommonRequestManager();
    }

    protected static TORequestManager getTORequestManager() throws BbmCreateException {
        return RmManagerFactory.getInstance(true).getTimeOffRequestManager();
    }

    public static ActivityManager getActivityManager() throws BbmException {
        return WfmManagerFactory.getActivityManager(false);
    }

    public void runSoftValidationsForTOWithdrawRequest(TORequest toRequest) throws BbmException{
        String methodName = "runSoftValidationsForTOWithdrawRequest";
        m_cat.debug(RmUtil.dumpEnterMethod(methodName,toRequest ));
        try {

            if(!(toRequest.isEligibleForAcceptWithdrawAction()||toRequest.isEligibleForRejectWithdrawAction())){
                return;
            }

			TimeRange startEnd = toRequest.getApprovedChoiceDateRange();

            ValidationCache vc = toRequest.getValidationCache();
            ID orgId = vc.getOrgIDForEmployeeDuringPeriod(toRequest.getEmployeeID(),
					startEnd.getStartDate(),
					startEnd.getEndDate());
            String requestType = Request.REQUESTTYPE_TIMEOFF_WITHDRAW;
			Collection<String> validators = vc.getValidators(orgId, requestType);
			for (String validatorName : validators) {
            	// Run the validations
                ValidationUtil.doValidation(validatorName, toRequest, s_validatorsMap);
            }
        } catch (Exception e) {
            handleException(e);
            throw  new BbmException(e.getMessage());
        }finally{
            m_cat.debug(RmUtil.dumpExitMethod(methodName, toRequest));
        }
    }
}
