package com.bluepumpkin.ejb.rm.requests.timeoff.withdraw.ejb;

import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;

import com.bluepumpkin.ejb.core.base.MultiUserException;

import com.bluepumpkin.ejb.rm.base.RmException;

import com.bluepumpkin.ejb.rm.util.RmDAOBase;

/**
 * DAO class for TOWithdraw
 */
public class TOWithdrawDAO extends RmDAOBase<TOWithdraw> {
    // TOChoiceFieldInfo for the DAO class.
    private static FieldInfo m_fieldInfo = new TOWithdrawFieldInfo();
    private static Category m_cat = Log.initCategory(TOWithdrawDAO.class.getName());

    /**
     * Constructor
     */
    public TOWithdrawDAO() {
        super();
    }

    /**
     * Constructor
     *
     * @param - Jdmo - Jdmo object instance
     * @see
     */
    public TOWithdrawDAO(Jdmo dmo) {
        super(dmo);
    }

    @Override
	protected FieldInfo getFieldInfo() {
        return TOWithdrawDAO.m_fieldInfo;
    }

    protected Category getCategory() {
        return TOWithdrawDAO.m_cat;
    }

    /**
     * Creates the value object.
     *
     * @return - returns new Value object
     * @see com.bluepumpkin.ejb.bbm.vo.ValueObjectBase
     * @see com.bluepumpkin.ejb.rm.requests.timeoff.withdraw.model.TOWithdraw
     */
	@Override
	protected TOWithdraw createValueObject() {
		return new TOWithdraw();
    }
    /**
     * Finds TimeOff withdraw for the TimeOff withdraw id object passed as argument.
     *
     * @param - ID -  ID object for Time off Choice.
     * @return - TOWithdraw Object.
     * @throw BbmFinderException
     */
    public TOWithdraw getTOWithdrawRequestById(ID pTOWithdrawId) throws BbmFinderException {
        return getObjectByID(pTOWithdrawId);
    }

    public TOWithdraw getWithdrawReqByTimeOffRequestId(ID toRequestID) throws BbmFinderException {
        TOWithdraw toWithdraw = null;
        Collection withdrawCol = getObjectsByParentID(toRequestID);
        if (withdrawCol != null) {
            for (Iterator iterator = withdrawCol.iterator(); iterator.hasNext();) {
                toWithdraw = (TOWithdraw) iterator.next();
            }
        }
        return toWithdraw;
    }
    /**
     * Create new TimeOff withdraw for the TimeOff withdraw object passed as argument.
     *
     * @param - pTOWithdraw -  TOWithdraw object for Time off Choice.
     * @return - ID object for created TOWithdraw.
     * @throw BbmCreateException
     */
    public ID createTOWithdrawRequest(TOWithdraw pTOWithdraw) throws RmException, BbmCreateException {
        if (pTOWithdraw.getTORequestID() == null) {
            throw RequestUtil.createRmException(RmEjbLogBundleKey.TOWITHDRAW_REQUEST_ID_NULL, TOWithdrawDAO.m_cat);
        }
        if (pTOWithdraw.getCreationDate() == null) {
            throw RequestUtil.createRmException(RmEjbLogBundleKey.TOWITHDRAW_CREATION_DATE_NULL, TOWithdrawDAO.m_cat);
        }
        ID l_TOWithdrawId = createObject(pTOWithdraw);
        return l_TOWithdrawId;

    }

    /**
     * Update TimeOff Withdraw database for the TimeOff Withdraw object passed as argument.
     *
     * @param - pTOWithdraw -  TOWithdraw object for Time off Choice.
     * @throw BbmUpdateException
     */
    public void updateTOWithdrawRequest(TOWithdraw pTOWithdraw) throws BbmUpdateException, MultiUserException {
        updateObject(pTOWithdraw);
    }

    /**
     * Delete TimeOff Withdraw database for the TimeOff Withdraw id object passed as argument.
     *
     * @param - pTOWithdrawID -  ID object for Time off Choice.
     * @throw BbmRemoveException
     * @see
     */
    public void deleteTOWithdrawRequest(ID pTOWithdrawID) throws  BbmRemoveException {
        deleteObject(pTOWithdrawID);
    }
}
