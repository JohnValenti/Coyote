package com.verint.ejb.bbm.forecast.model;

import java.util.ArrayList;
import java.util.Collections;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**
 * Represents a week of history data in a forecast.  May be defined starting at
 * an absolute date, as a number of weeks or months in the past, or as a custom
 * collection of seven different independently-specified days.
 */
@SuppressWarnings("serial")
public class ProfileComponent extends ValueObjectBase {
	public enum Type {
		Relative(3),
		Absolute(7);
		private int dbIntVal;
		private Type(int dbIntVal) {
			this.dbIntVal = dbIntVal;
		}
		
		private static Type getTypeFromDBIntVal(int val) {
			for (Type t : Type.values()) {
				if (t.dbIntVal == val) return t;
			}
			throw new IllegalArgumentException();
		}
	}
	
	public static final String WEEKLY_PROJECT_FLAG = "1";

	public ProfileComponent() {
		super();
	}

	private static FieldInfo m_fieldInfo = new ProfileComponentFieldInfo();

	@Override
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	public ID getForecastProfileId() {
		return getFieldValueID(ProfileComponentFieldInfo.PROFILECOMPONENT_FORECASTPROFILEID);
	}

	public void setForecastProfileId(ID val) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_FORECASTPROFILEID, val);
	}

	public Type getType() {
		return Type.getTypeFromDBIntVal(getFieldValueInt(ProfileComponentFieldInfo.PROFILECOMPONENT_TYPE));
	}
	
	public void setType(Type type) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_TYPE, type.dbIntVal);
	}

	public String getName() {
		return getFormattedFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_NAME);
	}

	public void setName(String val) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_NAME, val);
	}
	
	public String getValue() {
		return getFormattedFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_VALUE);
	}

	public void setValue(String value) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_VALUE, value);
	}

	public Float getWeight() {
		return getFieldValueFloat(ProfileComponentFieldInfo.PROFILECOMPONENT_WEIGHT);
	}

	public void setWeight(Float val) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_WEIGHT, val);
	}

	public Date getCreated() {
		return getFieldValueDate(ProfileComponentFieldInfo.PROFILECOMPONENT_CREATED);
	}

	public void setCreated(Date val) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CREATED, val);
	}

	public String getCreatedBy() {
		return getFormattedFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CREATEDBY);
	}

	public void setCreatedBy(String val) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CREATEDBY, val);
	}

	public Date getModified() {
		return getFieldValueDate(ProfileComponentFieldInfo.PROFILECOMPONENT_MODIFIED);
	}

	public void setModified(Date val) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_MODIFIED, val);
	}

	public String getModifiedBy() {
		return getFormattedFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_MODIFIEDBY);
	}

	public void setModifiedBy(String val) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_MODIFIEDBY, val);
	}
	
	/**
	 * Returns true if this profile component represents a "monthly" history period.
	 */
	public boolean isMonthly() {
		return getFieldValueBoolean(ProfileComponentFieldInfo.PROFILECOMPONENT_ISMONTHYEAR);
	}
	/**
	 * Sets whether this is a monthly profile component.  If isMonthly is true, this
	 * will clear the isCustomWeek flag (the two flags are mutually exclusive).
	 */
	public void setIsMonthly(boolean isMonthly) {
		if (!isMonthly) {
			setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_INMONTH, new NullObject());
			setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_INYEAR, new NullObject());
		}
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_ISMONTHYEAR, isMonthly);
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_ISCUSTOMIZED, isMonthly);
	}
	
	/**
	 * Returns true if this profile component represents a custom history week.
	 */
	public boolean isCustomWeek() {
		return getFieldValueBoolean(ProfileComponentFieldInfo.PROFILECOMPONENT_ISCUSTOMIZED) && 
			(getFieldValueBoolean(ProfileComponentFieldInfo.PROFILECOMPONENT_ISMONTHYEAR) == false);
	}
	/**
	 * Sets whether this is a custom week profile component.  If isCustomWeek is true, this
	 * will clear the isMonthly flag (the two flags are mutually exclusive).
	 */
	public void setIsCustomWeek(boolean isCustomWeek) {
		if (isCustomWeek) {
			setIsMonthly(false);
		}
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_ISCUSTOMIZED, isCustomWeek);
	}
	
	/**
	 * Returns the custom date fields of this profile component as an ordered list of dates.
	 * Null dates are included in this list, they are inserted into the list as NULL.
	 * 
	 * NOTE: The FS client stores these custom dates at midnight GMT for Custom Weeks and
	 * at the day boundary of the campaign for Monthly "weeks", which is how they will be
	 * returned here.  Please be aware of this inconsistency.
	 */
	public List<Date> getCustomDates() {
		if (isMonthly() == false && isCustomWeek() == false) {
			return Collections.emptyList();
		}
		ArrayList<Date> retVal = new ArrayList<Date>();
		retVal.add(getFieldValueDate(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE1));
		retVal.add(getFieldValueDate(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE2));
		retVal.add(getFieldValueDate(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE3));
		retVal.add(getFieldValueDate(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE4));
		retVal.add(getFieldValueDate(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE5));
		retVal.add(getFieldValueDate(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE6));
		retVal.add(getFieldValueDate(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE7));
		return retVal;
	}
	
	/**
	 * Sets the custom date fields of this profile component with the dates given in the ordered list
	 * customDates.
	 * 
	 * The FS client currently stores these dates at midnight GMT for custom week periods, and
	 * day boundary GMT for monthly periods.  This is how the dates should be handed in to this method
	 * as well.
	 */
	public void setCustomDates(List<Date> customDates) {
		if (customDates != null) {
			if (customDates.size() > 0) {
				setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE1, customDates.get(0));
			}
			if (customDates.size() > 1) {
				setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE2, customDates.get(1));
			}
			if (customDates.size() > 2) {
				setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE3, customDates.get(2));
			}
			if (customDates.size() > 3) {
				setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE4, customDates.get(3));
			}
			if (customDates.size() > 4) {
				setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE5, customDates.get(4));
			}
			if (customDates.size() > 5) {
				setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE6, customDates.get(5));
			}
			if (customDates.size() > 6) {
				setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_CUSTOMDATE7, customDates.get(6));
			}
		}
	}
	
	/**
	 * Gets the month of this profile component.  Useful only for monthly profile components.
	 * Returns 0 if this is not a monthly component.
	 */
	public int getMonth() {
		return getFieldValueInt(ProfileComponentFieldInfo.PROFILECOMPONENT_INMONTH);
	}
	
	/**
	 * Sets the month of this profile component.  Useful only for monthly profile components.
	 */
	public void setMonth(int month) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_INMONTH, month);
	}
	
	/**
	 * Gets the year of this profile component.  Useful only for monthly profile components.
	 * Returns 0 if this is not a monthly component.
	 */
	public int getYear() {
		return getFieldValueInt(ProfileComponentFieldInfo.PROFILECOMPONENT_INYEAR);
	}
	
	/**
	 * Sets the year of this profile component.  Useful only for monthly profile components.
	 */
	public void setYear(int year) {
		setFieldValue(ProfileComponentFieldInfo.PROFILECOMPONENT_INYEAR, year);
	}
}
