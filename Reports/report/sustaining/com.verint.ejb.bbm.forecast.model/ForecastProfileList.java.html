package com.verint.ejb.bbm.forecast.model;

import java.util.Calendar;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

@SuppressWarnings("serial")
public class ForecastProfileList extends ValueObjectBase {

	private static FieldInfo m_fieldInfo = new ForecastProfileListFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }
    
    //This reference date is used to store the start/end dates
    //of forecast profile lists for manually modified ForecastProfiles
    //for outbound media, relative to the start date of the profile's SP week.
    private static final Calendar referenceDate;
    
    static {
    	//Initializes the reference date to January 1st, 1980 00:00:00 GMT
    	//This is an arbitrary date but it's what the FS client decided to use.
    	referenceDate = Calendar.getInstance(TimeZoneUtil.GMT_TIMEZONE);
    	referenceDate.set(1980, Calendar.JANUARY, 1, 0, 0, 0);
    }

	public ForecastProfileList() {
		super();
	}

	public ForecastProfileList(FieldInfo info) {
		super(info);
	}

	public Date getEndTime() {
		return getFieldValueDate(ForecastProfileListFieldInfo.FORECASTPROFILELIST_ENDTIME);
	}

	public void setEndTime(Date endTime) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_ENDTIME, endTime);
	}

	public ID getId() {
		return getFieldValueID(ForecastProfileListFieldInfo.FORECASTPROFILELIST_ID);
	}

	public void setId(ID id) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_ID, id);
	}

	public ID getQueueId() {
		return getFieldValueID(ForecastProfileListFieldInfo.FORECASTPROFILELIST_QUEUEID);
	}

	public void setQueueId(ID id) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_QUEUEID, id);
	}
	
	public ID geProfileId() {
		return getFieldValueID(ForecastProfileListFieldInfo.FORECASTPROFILELIST_FORECASTPROFILEID);
	}

	public void setProfileId(ID id) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_FORECASTPROFILEID, id);
	}

	public int getLength() {
		return getFieldValueInt(ForecastProfileListFieldInfo.FORECASTPROFILELIST_LENGTH);
	}

	public void setLength(int length) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_LENGTH, length);
	}

	public int getRetries() {
		return getFieldValueInt(ForecastProfileListFieldInfo.FORECASTPROFILELIST_RETRIES);
	}

	public void setRetries(int retries) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_RETRIES, retries);
	}

	public Date getStartTime() {
		return getFieldValueDate(ForecastProfileListFieldInfo.FORECASTPROFILELIST_STARTTIME);
	}

	public void setStartTime(Date startTime) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_STARTTIME, startTime);
	}
	
	public Date getCreatedDate() {
		return getFieldValueDate(ForecastProfileListFieldInfo.FORECASTPROFILELIST_CREATED);
	}
	public void setCreatedDate(Date createdDate) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_CREATED, createdDate);
	}
	
	public String getCreatedBy() {
		return getFormattedFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_CREATEDBY);
	}
	
	public void setCreatedBy(String createdBy) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_CREATEDBY, createdBy);
	}
	
	public Date getModifiedDate() {
		return getFieldValueDate(ForecastProfileListFieldInfo.FORECASTPROFILELIST_MODIFIED);
	}
	
	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_MODIFIED, modifiedDate);
	}
	
	public String getModifiedBy() {
		return getFormattedFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_MODIFIEDBY);
	}
	
	public void setModifiedBy(String modifiedBy) {
		setFieldValue(ForecastProfileListFieldInfo.FORECASTPROFILELIST_MODIFIEDBY, modifiedBy);
	}
	
	/**
	 * This method returns the reference date that is used to store the start/end dates
	 * of forecast profile lists for manually modified ForecastProfiles for outbound media.
	 * 
	 * When a manually modified ForecastProfile is saved for outbound media, its associated
	 * ForecastProfileLists are saved with a start and end date that is relative to the start
	 * date of the profiles SP week start date.  However, the SP week start date is not used as
	 * the reference point in the database, the date January 1st, 1980 00:00:00 GMT is used.
	 * Therefore, if the forecasted list starts two days before the start of its SP week, the start
	 * date of the list will saved in the DB as December 30th, 1980 00:00:00 (two days before the
	 * reference date).
	 */
	public static Calendar getReferenceDateForManuallyModifiedProfiles() {
		return (Calendar)referenceDate.clone();
	}
}
