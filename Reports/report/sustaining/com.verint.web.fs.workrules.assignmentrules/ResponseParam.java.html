package com.verint.web.fs.workrules.assignmentrules;

import java.util.List;

/**
 * Do not reuse ResponseParam instances in multiple responses. Create a new instance for each response.
 */
public class ResponseParam {
	private ResponseParamType parameterType;
	private List<ResponseParamConstraint> constraints;
	private Object value;

	public ResponseParam(ResponseParamType parameterType, List<ResponseParamConstraint> constraints, Object value) {
		super();
		this.parameterType = parameterType;
		this.constraints = constraints;
		setValue(value);
	}

	public ResponseParamType getParameterType() {
		return parameterType;
	}

	public Object getValue() {
		return value;
	}

	public void setValue(Object value) {
		if (value != null && !getParameterType().getJavaType().isAssignableFrom(value.getClass())) {
			throw new IllegalArgumentException("value must be of type: "
					+ getParameterType().getJavaType().getCanonicalName() + ", instead received: "
					+ value.getClass().getCanonicalName());
		}
		this.value = value;
	}

	public List<ResponseParamConstraint> getConstraints() {
		return constraints;
	}
}
