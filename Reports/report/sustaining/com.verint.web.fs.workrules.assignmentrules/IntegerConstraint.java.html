package com.verint.web.fs.workrules.assignmentrules;

public class IntegerConstraint implements ResponseParamConstraint {
	private int minVal;
	private int maxVal;

	public IntegerConstraint(int minVal, int maxVal) {
		this.minVal = minVal;
		this.maxVal = maxVal;
	}

	public int getMinVal() {
		return minVal;
	}

	public int getMaxVal() {
		return maxVal;
	}
}
