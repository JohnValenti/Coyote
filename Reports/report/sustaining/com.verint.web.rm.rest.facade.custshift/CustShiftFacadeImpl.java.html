	public static CustShiftRequestEntity createCustShiftRequestEntity(RequestContext context, CustShiftReq request,
			RequestsMetaData metaData) {
		CustShiftRequestEntity reqEntity = new CustShiftRequestEntity();
		reqEntity.populate(request, context);

		reqEntity.setActivityID(RmUtils.getInteger(request.getActivityID()));
		// durationInMinutes will throw an exception if start or end dates are null
		if (request.getStartTime() != null && request.getEndTime() != null) {
			reqEntity.setDurationInMinutes(request.getDurationinMinutes());
		}
		reqEntity.setEndTime(FormatUtils.dateToISO8601SecPrecisionString(request.getEndTime()));
		reqEntity.setOvertime(request.isOT());
		reqEntity.setShiftDurationInMinutes(request.getShiftDurationinMinutes());
		reqEntity.setShiftEndTime(FormatUtils.dateToISO8601SecPrecisionString(request.getShiftEndTime()));
		reqEntity.setShiftID(RmUtils.getInteger(request.getShiftID()));
		reqEntity.setShiftStartTime(FormatUtils.dateToISO8601SecPrecisionString(request.getShiftStartTime()));
		reqEntity.setStartTime(FormatUtils.dateToISO8601SecPrecisionString(request.getStartTime()));
		reqEntity.setSubType(request.getSubType());

		if (request.getExtAfterID() != null) {
			ShiftOTExtensionParametersEntity extensionAfter = new ShiftOTExtensionParametersEntity();
			reqEntity.setExtensionAfter(extensionAfter);
			extensionAfter.setActivityID(RmUtils.getInt(request.getExtAfterActivityID()));
			extensionAfter.setDuration(request.getExtAfterDuration());
			extensionAfter.setGap(request.getExtAfterGap());
			extensionAfter.setId(RmUtils.getString(request.getExtAfterID()));
			extensionAfter.setOvertime(request.getExtAfterIsOT());
		}

		if (request.getExtBeforeID() != null) {
			ShiftOTExtensionParametersEntity extensionBefore = new ShiftOTExtensionParametersEntity();
			reqEntity.setExtensionBefore(extensionBefore);
			extensionBefore.setActivityID(RmUtils.getInt(request.getExtBeforeActivityID()));
			extensionBefore.setDuration(request.getExtBeforeDuration());
			extensionBefore.setGap(request.getExtBeforeGap());
			extensionBefore.setId(RmUtils.getString(request.getExtBeforeID()));
			extensionBefore.setOvertime(request.getExtBeforeIsOT());
		}

		populateGaps(reqEntity, request);

		// metadata
		Set<ID> activityIDs = metaData.getActivityIDs();
		if (request.getActivityID() != null) {
			activityIDs.add(request.getActivityID());
		}
		if (request.getExtAfterActivityID() != null) {
			activityIDs.add(request.getExtAfterActivityID());
		}
		if (request.getExtBeforeActivityID() != null) {
			activityIDs.add(request.getExtBeforeActivityID());
		}
		Set<ID> shiftOTExtensionIDs = metaData.getShiftOTExtensionIDs();
		if (request.getExtAfterID() != null) {
			shiftOTExtensionIDs.add(request.getExtAfterID());
		}
		if (request.getExtBeforeID() != null) {
			shiftOTExtensionIDs.add(request.getExtBeforeID());
		}
		if (request.getShiftID() != null) {
			metaData.getShiftIDs().add(request.getShiftID());
		}
		return reqEntity;
	}

	@Override
	public CustShiftRequestResponseEntity createRequest(RequestContext context, CustShiftReq request, String comment)
			throws WFOException {
		try {
			if (!RequestUtil.isAuthorized(context, request, true)) {
				RmUtils.throwUnauthorizedRequest(context);
			}

			CustomShiftRequestFormPopupPMEx model = new CustomShiftRequestFormPopupPMEx(context);
			request = model.prepareForSavingFromREST(context, request);

			ID requestID = CustomShiftRequestsMH.createCSRequest(request, comment);
			request = CustomShiftRequestsMH.getCSRequest(requestID);

			return createResponse(context, request);
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(context, ex, LOG);
		}
	}

	@Override
	public CustShiftRequestResponseEntity editRequest(RequestContext context, CustShiftReq request, String comment)
			throws WFOException {
		try {
			if (!RequestUtil.isAuthorized(context, request, true)) {
				RmUtils.throwUnauthorizedRequest(context);
			}

			CustomShiftRequestFormPopupPMEx model = new CustomShiftRequestFormPopupPMEx(context);
			model.setRequestID(request.getID());
			request = model.prepareForSavingFromREST(context, request);

			CustomShiftRequestsMH.updateCSRequest(request, comment);
			request = CustomShiftRequestsMH.getCSRequest(request.getID());

			return createResponse(context, request);
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(context, ex, LOG);
		}
	}

	@Override
	public CustShiftDetailEntity getDetailData(RequestContext context, ID employeeID, Date date, Date ribbonEndDate)
			throws WFOException {
		CustShiftDetailEntity entity = new CustShiftDetailEntity();
		try {
			CustomShiftRequestFormPopupPMEx model = new CustomShiftRequestFormPopupPMEx(context);
			model.populateDetails(context, entity, employeeID, date, ribbonEndDate);
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(context, ex, LOG);
		}
		return entity;
	}

	/**
	 * Gets the request by the specified requestID.
	 */
	@Override
	public CustShiftRequestResponseEntity getRequest(RequestContext context, ID requestID)
			throws WFOException {
		try {
			CustShiftReq request = CustomShiftRequestsMH.getCSRequest(requestID);

			RmUtils.assertNotNull(context, request, requestID);
			if (!RequestUtil.isAuthToView(context, request)) {
				RmUtils.throwUnauthorizedRequest(context);
			}

			return createResponse(context, request);
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(context, ex, LOG);
		}
	}

	/**
	 * Performs an action on an existing shift bidding request.
	 */
	@Override
	public CustShiftRequestResponseEntity processRequest(RequestContext context, ID requestID, int actionCode, String comment)
			throws WFOException {
		try {
			CustShiftReq request = CustomShiftRequestsMH.getCSRequest(requestID);
			String type = Request.REQUESTTYPE_CUSTSHIFT;
			String version = request.getObjectVersionNumber();

			if (ActionCode.WITHDRAW.getActionCode() == actionCode) {
				CustomShiftRequestsMH.requestWithdrawOfApprovedRequest(requestID, type, version, RequestAuditTrail.STATUS_WITHDRAWN,
						comment);
			} else if (ActionCode.ESCALATE.getActionCode() == actionCode) {
				RequestsMH.changeRequestStatus(requestID, type, version,
						RequestAuditTrail.STATUS_ESCALATED, comment);
			} else {
				throw new BusinessWFOException("ActionCode not supported", Response.Status.CONFLICT);
			}

			request = CustomShiftRequestsMH.getCSRequest(requestID);

			return createResponse(context, request);
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(context, ex, LOG);
		}
	}

	public static void addActivities(List<Entity> dest, Collection<Activity> activities) {
		for (Activity activity : activities) {
			// lets just strip out the none items here instead of adding another parameter to the various getCSDetailData calls.
			if (activity.getID().equals(CustomShiftRequestFormPopupPM.NONE_ID)) {
				continue;
			}
			Entity entity = new Entity();
			entity.setId(activity.getID().toString());
			entity.setName(activity.getName());
			dest.add(entity);
		}
	}

	public static void addShifts(List<ShiftEntity> dest, Collection<Shift> shifts) {
		for (Shift shift : shifts) {
			// lets just strip out the<Shift Requests & Changes> item here because the activityID is now included in the shift assignment
			if (shift.getID().equals(CustomShiftRequestFormPopupPM.CUSTOM_ID)) {
				continue;
			}
			ShiftEntity entity = new ShiftEntity();
			entity.setId(shift.getID().toString());
			entity.setName(shift.getName());
			if (shift.getActivityID() != null) {
				entity.setActivityID(shift.getActivityID().toInt());
			}
			entity.setDuration(shift.getDuration());
			dest.add(entity);
		}
	}

	public static void addShiftOTExtensions(List<ShiftOTExtensionEntity> dest, Collection<ShiftOTExtension> items) {
		for (ShiftOTExtension item : items) {
			// lets just strip out the none items here instead of adding another parameter to the various populateCSDetailData calls.
			if (item.getID().equals(CustomShiftRequestFormPopupPM.NONE_ID)) {
				continue;
			}
			ShiftOTExtensionEntity entity = new ShiftOTExtensionEntity(item);
			dest.add(entity);
		}
	}

	public static CustShiftRequestResponseEntity createResponse(RequestContext context, CustShiftReq request)
			throws RemoteException, BbmException {
		CustShiftRequestResponseEntity result = new CustShiftRequestResponseEntity();
		ID employeeID = context.getUser().getEmployeeID();
		result.setId(context.getUser().getEmployeeID().toString());

		OrganizationSetting orgSettings = RequestsMH.getOrgSettingForEmployee(context, employeeID);
		RequestsMetaData metaData = new RequestsMetaData(orgSettings);
		CustShiftRequestEntity entity = createCustShiftRequestEntity(context, request, metaData);

		RequestFacadeImpl.addActions(context, request, entity, orgSettings, null);
		result.setRequest(entity);
		RmUtils.addActivities(context, result.getActivities(), metaData.getActivityIDs());
		RmUtils.addShifts(result.getShifts(), metaData.getShiftIDs());
		RmUtils.addShiftOTExtensions(result.getShiftOTExtensions(), metaData.getShiftOTExtensionIDs());
		return result;
	}

	private static void populateGaps(CustShiftRequestEntity reqEntity, CustShiftReq request) {
		Collection<CustShiftReqGap> gaps = request.getCustShiftReqGaps();
		if (gaps == null) {
			return;
		}
		List<CustShiftRequestGapEntity> dest = new ArrayList<CustShiftRequestGapEntity>();
		reqEntity.setGaps(dest);
		for (CustShiftReqGap gap : gaps) {
			CustShiftRequestGapEntity entity = new CustShiftRequestGapEntity();
			entity.setActivityID(RmUtils.getInteger(gap.getExtGapActivityID()));
			entity.setDuration(gap.getExtGapDuration());
			entity.setEndTime(FormatUtils.dateToISO8601SecPrecisionString(gap.getExtGapEndTime()));
			entity.setGap(gap.getExtGapGap());
			entity.setId(RmUtils.getString(gap.getID()));
			entity.setShiftOTGapID(RmUtils.getInteger(gap.getShiftOTGapID()));
			entity.setStartTime(FormatUtils.dateToISO8601SecPrecisionString(gap.getExtGapStartTime()));
			dest.add(entity);
		}
	}

	public static void populateShiftAssignment(CustShiftDetailEntity entity, ShiftAssignment shift) {
		if (shift == null) {
			return;
		}
		ShiftAssignmentEntity saEntity = new ShiftAssignmentEntity();
		saEntity.setId(shift.getID().toString());
		saEntity.setEndTime(FormatUtils.dateToISO8601SecPrecisionString(shift.getEndTime()));
		saEntity.setStartTime(FormatUtils.dateToISO8601SecPrecisionString(shift.getStartTime()));
		saEntity.setDuration(shift.getDuration());
		saEntity.setShiftID(RmUtils.getInteger(shift.getShiftID()));
		saEntity.setActivityID(RmUtils.getInteger(shift.getActivityID()));

		saEntity.setExtensionAfter(shift.getExtensionAfter());
		saEntity.setExtensionBefore(shift.getExtensionBefore());
		saEntity.setOTExtensionAfterActivityID(RmUtils.getInteger(shift.getOTExtensionAfterActivityID()));
		saEntity.setOTExtensionAfterID(RmUtils.getInteger(shift.getOTExtensionAfterID()));
		saEntity.setOTExtensionBeforeActivityID(RmUtils.getInteger(shift.getOTExtensionBeforeActivityID()));
		saEntity.setOTExtensionBeforeID(RmUtils.getInteger(shift.getOTExtensionBeforeID()));

		entity.setShiftAssignment(saEntity);
	}
