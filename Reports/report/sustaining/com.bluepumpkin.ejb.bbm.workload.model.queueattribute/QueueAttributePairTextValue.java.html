import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class QueueAttributePairTextValue extends ValueObjectBase {
	
	private static final long serialVersionUID = 1L;

	private static FieldInfo m_fieldInfo = new QueueAttributePairValueTextFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public QueueAttributePairTextValue() {
		super();
	}

	public ID getQueueAttributeID() {
		return getFieldValueID(QueueAttributePairValueTextFieldInfo.QAPV_QUEUEATTRIBUTEID);
	}

	public void setQueueAttributeID(ID id) {
		setFieldValue(QueueAttributePairValueTextFieldInfo.QAPV_QUEUEATTRIBUTEID, id);
	}

	public ID getQueueID() {
		return getFieldValueID(QueueAttributePairValueTextFieldInfo.QAPV_QUEUEID);
	}

	public void setQueueID(ID id) {
		setFieldValue(QueueAttributePairValueTextFieldInfo.QAPV_QUEUEID, id);
	}
	
	public String getValue() {
		return getFormattedFieldValue(QueueAttributePairValueTextFieldInfo.QAPV_VALUE);
	}

	public void setValue(String strValue) {
		setFieldValue(QueueAttributePairValueTextFieldInfo.QAPV_VALUE, strValue);
	}
}
