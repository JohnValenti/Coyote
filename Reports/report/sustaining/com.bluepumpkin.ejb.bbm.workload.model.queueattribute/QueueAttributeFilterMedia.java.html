import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class QueueAttributeFilterMedia extends ValueObjectBase {
	
	private static final long serialVersionUID = 1L;

	private static FieldInfo m_fieldInfo = new QueueAttributeFilterMediaFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public QueueAttributeFilterMedia() {
		super();
	}

	public ID getFilterID() {
		return getFieldValueID(QueueAttributeFilterMediaFieldInfo.QAFM_FILTERID);
	}

	public void setFilterID(ID filterID) {
		setFieldValue(QueueAttributeFilterMediaFieldInfo.QAFM_FILTERID, filterID);
	}

	public String getMediaID() {
		return (String)getFieldValue(QueueAttributeFilterMediaFieldInfo.QAFM_MEDIAID);
	}
	
	public void setMediaID(String mediaID) {
		setFieldValue(QueueAttributeFilterMediaFieldInfo.QAFM_MEDIAID, mediaID);
	}
}
