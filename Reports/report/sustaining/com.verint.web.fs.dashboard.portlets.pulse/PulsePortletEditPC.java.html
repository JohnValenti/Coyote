	public PulsePortletEditPC(RequestContext context) {
		super(context);
		m_bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);

		m_drPickerPC = new LastNPeriodPickerPC(m_context);
		addChildComponent(m_drPickerPC);
		m_trimToHOOCB = new CheckBoxPC(m_context, "trimToHOO", true);
		addChildComponent(m_trimToHOOCB);

		setName("pulsePortletEdit");
	}

	/**
	 * Initialize component for display.
	 */
	@Override
	public void initForDisplay() {
		if (!m_isInitializedForDisplay) {
			super.initForDisplay();
			initContent();
		}
	}

	/**
	 * Initialize component.
	 *
	 * @throws Exception
	 */
	public void init() throws Exception {
		m_MediaFilter = WorkloadModelHandler.getMediaMap(m_context);
	}

	/**
	 * Initialize content.
	 */
	private void initContent() {
		setLastRowType(LAST_ROW_NONE);
		setIsBorderEnabled(true);
		String selectedStatisticID = m_statisticType == null ? "" + StatisticType.FTE.getId() : "" + m_statisticType.getId();
		Collection statOptions = StatisticType.createDropDownValueList(m_localizer);
		String statisticTypeDD =
			HtmlUtil.makeSelectInput(Constants.STATISTIC_TYPE, selectedStatisticID, JSUtil.ON_CHANGE_NO_STMT, statOptions);
		addRow(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.STATISTIC_TYPE), statisticTypeDD);

		String selectedCampaignID = m_campaignID != null ? m_campaignID.toString() : "";

		String drLabel = i18n(m_common_bundle, UIFWebBundleKey.TIME_PERIOD);
		m_drPickerPC.getDateRangePickerPC().setOnChangeJS("submitForm();");
		addRow(drLabel + ": " + m_drPickerPC.getDateRangePickerPC().getTimeZone().getDisplayName(m_locale), m_drPickerPC.getHtmlDisplay());

		String cLabel = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_CAMPAIGN);

		ArrayList campOptions = new ArrayList();
		campOptions.addAll(CampaignModelHandler.createDropDownValueList(m_campaignList, selectedCampaignID, m_localizer, true));
		HtmlSelect htmlSelect = new HtmlSelect();
		htmlSelect.setMaxDisplayLen((short)5);
		String campDD = HtmlUtil.makeSelectInput("campaignID", "", new String[]{selectedCampaignID}, "submitForm();", campOptions, null, false, false, true, false, null, htmlSelect);
		addRow(cLabel, campDD);

		Collection queueList = Collections.emptyList();
		String selQueueID = m_queueID != null ? m_queueID.toString() : "";
		Date startDate = null;
		Date endDate = null;
		if (LastNPeriodPickerPC.PICKER_TYPE_DATE_RANGE.equals(getDatePickerType())) {
			startDate = getStartDate();
			endDate = getEndDate();
		}
		// we don't need to define dates for "last n period" since upon saving
		// and reopening (on a different day), the dates would always be
		// different, so for "last n period" always show all queues in the
		// dropdown, and in View mode the queue will just have no data if the
		// last n periods are not within the queue's period

		if (m_campaignID != null) {
			try {
				queueList = CampaignModelHandler.getQueuesAssignedToCampaign(m_context, m_campaignID, startDate, endDate, true);
			} catch(RemoteException e) {
				log.error("Exception getting queues assigned to Campaign " + m_campaignID, e);
			} catch(BbmException e) {
				log.error("Exception getting queues assigned to Campaign " + m_campaignID, e);
			}
		}

		// verticalise queue
		Object[] args = new Object[1];
		args[0] = m_context.getSystemManagerFactory().getVerticalizationManager().getVerticalizedString(m_context, DefaultVerticalizationManager.QUEUES);

		String qLabel = MessageFormat.format(m_localizer.i18n(ReportParameterBundleKey.BUNDLE_NAME, ReportParameterBundleKey.WORK_QUEUE_KEY), args);
		ArrayList queueOptions = new ArrayList();
		//queueOptions.add(new StringsPair("", ""));
		queueOptions.addAll(createQueueDropDownValueList(queueList, selQueueID, m_localizer, true));
		String queueDD = HtmlUtil.makeSelectInput("queueID", "", new String[]{selQueueID}, "", queueOptions, null, false, false, true, false, null, htmlSelect);
		addRow(qLabel, queueDD);

		addRow(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.TRIM_TO_CAMPAIGN_HOO), m_trimToHOOCB);
	}

	public Collection createQueueDropDownValueList(
		Collection queueList,
		String selectedID,
		Localizer localizer,
		boolean isCreateStringsPair) {
		// QC-99122: The media filter is null for the wsuperuser.
		if (m_MediaFilter == null || queueList == null || queueList.isEmpty()) {
			return Collections.emptyList();
		}

		queueList = SortUtil.sortBasicVO(queueList, localizer);
		QueueUtil.QueueTreeInCampaignModeNodeHelper treeNodeHelper = new QueueUtil.QueueTreeInCampaignModeNodeHelper(queueList, m_MediaFilter, m_campaignID, m_localizer, true, true, false, null);
		TreeModel treeModel = TreeModelBuilder.buildTreeModel(queueList, treeNodeHelper);
		DefaultMutableTreeNode root = (DefaultMutableTreeNode)treeModel.getRoot();
		return TreeModelBuilder.createDropDownValueList(root, selectedID, false, isCreateStringsPair);
	}
