package com.verint.web.fs.dashboard.portlets.pulse;

import java.text.DateFormat;
import java.util.TimeZone;

import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;
import com.bluepumpkin.web.core.base.RequestHandler;
import com.verint.web.core.dashboard.pluto.PlutoPreferencesUtil;
import com.verint.web.core.dashboard.pluto.Preferences;
import com.verint.web.fs.dashboard.portlets.pulse.util.Constants;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

public class PulsePortletEditRH
	extends RequestHandler
{
	@Override
	public void processRequest(RequestContext context)
		throws Exception
	{
		PulsePortletEditPM model = new PulsePortletEditPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);

		// Process Action
		if (model.isAction(PageAction.SAVE)) {
			model.setEditComplete(true);
			model.setSavePressed(true);
			// store any prefs that aren't null, which with defaults should only
			// happen in the case of user misuse or mistake
			Preferences prefs =
				PlutoPreferencesUtil.getPortletPreferences(context, model.getTemplateID(), model.getInstanceID());

			TimeZone tz = null;
			if (model.getStatisticType() != null) {
				prefs.setValue(Constants.STATISTIC_TYPE, "" + model.getStatisticType().getId());
			}

			if (model.getCampaignID() != null) {
				prefs.setValue(Constants.CAMPAIGN_ID, model.getCampaignID().toString());
				tz = PulsePortletUtil.getCampaignTimeZone(model.getCampaignID());
			}

			if (model.getQueueID() != null) {
				prefs.setValue(Constants.QUEUE_ID, model.getQueueID().toString());
			}

			if (model.getDatePickerType() != null) {
				prefs.setValue(Constants.DATE_PICKER_TYPE, model.getDatePickerType());
			}

			if (model.getLastNPeriods() != Constants.DATE_LAST_N_PERIOD_INVALID
				&& !model.getDatePickerType().equals(LastNPeriodPickerPC.PICKER_TYPE_DATE_RANGE)) {
				prefs.setValue(Constants.DATE_LAST_N_PERIOD, "" + model.getLastNPeriods());
			}

			if (model.getLastNPeriodType() != -1
				&& !model.getDatePickerType().equals(LastNPeriodPickerPC.PICKER_TYPE_DATE_RANGE)) {
				prefs.setValue(Constants.DATE_LAST_N_PERIOD_TYPE, "" + model.getLastNPeriodType());
			}

			if (model.getDateRangeStartDate() != null
				&& model.getDatePickerType().equals(LastNPeriodPickerPC.PICKER_TYPE_DATE_RANGE)) {
				DateFormat dateFormat = DateFormat.getDateTimeInstance(DateFormat.FULL, DateFormat.FULL);
				if (tz != null) {
					dateFormat.setTimeZone(tz);
				}
				prefs.setValue(Constants.DATE_DATE_RANGE_START_DATE, dateFormat.format(model.getDateRangeStartDate()));
			}

			if (model.getDateRangeEndDate() != null
				&& model.getDatePickerType().equals(LastNPeriodPickerPC.PICKER_TYPE_DATE_RANGE)) {
				DateFormat dateFormat = DateFormat.getDateTimeInstance(DateFormat.FULL, DateFormat.FULL);
				if (tz != null) {
					dateFormat.setTimeZone(tz);
				}
				prefs.setValue(Constants.DATE_DATE_RANGE_END_DATE, dateFormat.format(model.getDateRangeEndDate()));
			}

			if (model.getTrimToHOO() != null) {
				prefs.setValue(Constants.TRIM_TO_HOO, model.getTrimToHOO().toString());
			}

			try {
				prefs.savePortletPreferences();
			} catch(BbmObjectNotFoundException ex) {
				model.setSaveErrMsgID(ex.getMsgID());
			}
		} else if (model.isAction(PageAction.CANCEL)) {
			model.setEditComplete(true);
			model.setSavePressed(false);
		}
	}
}
