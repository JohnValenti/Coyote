package com.verint.web.fs.optimization;

import com.bluepumpkin.common.datatypes.TimeContext;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.workload.model.Media;

import java.util.Collection;

/**
 * The Class Response. This class represents the basic data
 * that is most likely in the returned response when a request for data is made to the server
 * in the context of optimization. Requesters use this to conveniently retrieve required pieces
 * of info from the response object.
 */
public class Response<TQueueDTO> implements java.io.Serializable {

	private static final long serialVersionUID = -4617395112511697538L;

	private final Campaign m_campaign;
	private final SchedulingPeriod m_sp;
	private final Collection<CampaignHOO> m_hoo;
	private final Media m_media;
	private TQueueDTO m_queueDto;
	private TimeContext m_timeContext;

	public Response(Media media, Campaign campaign, SchedulingPeriod sp, TQueueDTO queueDto,
			Collection<CampaignHOO> hoo) {
		m_queueDto = queueDto;
		m_media = media;
		m_campaign = campaign;
		m_sp = sp;
		m_hoo = hoo;
		m_timeContext = new TimeContext(m_campaign.getTimeZone(), m_campaign.getWeekStart());
	}

	public TimeContext getTimeContext() {
		return m_timeContext;
	}

	public Collection<CampaignHOO> getHoo() {
		return m_hoo;
	}

	public TQueueDTO getQueueDto() {
		return m_queueDto;
	}

	public Campaign getCampaign() {
		return m_campaign;
	}

	public SchedulingPeriod getSp() {
		return m_sp;
	}

	public Media getMedia() {
		return m_media;
	}
}
