package com.verint.web.fs.optimization;

import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;

import java.io.Serializable;

public class CombinedQueueDTO<TMultiQueueDTO extends MultiQueueDTO<? extends SingleQueueDTO>> implements Serializable {

	private static final long serialVersionUID = -3998120642696958476L;

	protected final SPQueue m_spQueue;
	protected final TMultiQueueDTO m_multiQueueDto;

	public CombinedQueueDTO(SPQueue spQueue, TMultiQueueDTO multiQueueDto) {
		m_spQueue = spQueue;
		m_multiQueueDto = multiQueueDto;
	}

	public SPQueue getSPQueue() {
		return m_spQueue;
	}

	public TMultiQueueDTO getMultiQueueDto() {
		return m_multiQueueDto;
	}
}
