package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeMonthlyMinMaxMinutes;

public class CFEmployeeMonthlyMinMax implements Serializable {
	
	private static final long serialVersionUID = 1L;
	
	public int minMinutes;
    public int maxMinutes;
    public int calendarMonth;
    public int year;
    
    public static CFEmployeeMonthlyMinMax convert(EmployeeMonthlyMinMaxMinutes from) {
    	CFEmployeeMonthlyMinMax to = new CFEmployeeMonthlyMinMax();
    	to.minMinutes = from.getMinMinutes();
    	to.maxMinutes = from.getMaxMinutes();
    	to.calendarMonth = from.getCalendarMonth();
    	to.year = from.getYear();
    	
    	return to;
    }
    
    public static CFEmployeeMonthlyMinMax[] convert(Collection from) {
    	if (from == null) {
			return null;
    	}

    	CFEmployeeMonthlyMinMax[] to = new CFEmployeeMonthlyMinMax[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++) {
			to[ix] = CFEmployeeMonthlyMinMax.convert((EmployeeMonthlyMinMaxMinutes)i.next());
		}
		return to;
    }
}
