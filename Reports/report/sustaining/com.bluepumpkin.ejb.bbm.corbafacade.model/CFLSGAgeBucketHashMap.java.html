package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.am.lsg.model.LSGAgeBucket;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;

public class CFLSGAgeBucketHashMap implements Serializable {

    private static final long serialVersionUID = -6911246295441378790L;
    public ID id;
    public CFLsgAgeBucket[] cfLsgAgeBucket;

    public static CFLSGAgeBucketHashMap[] convertToCFAgeBucketHashMap(HashMap<ID, Collection<LSGAgeBucket>> spQueueIdToAgeBuckets, Collection<SPQueue> spQueues) {
        CFLSGAgeBucketHashMap[] cflsgAgeBucketHashMaps = new CFLSGAgeBucketHashMap[spQueueIdToAgeBuckets.size()];
        int index = 0;
        Map<ID,SPQueue> spQueueToSpQueueId = new HashMap<ID, SPQueue>();
        for (SPQueue spQueue: spQueues) {
        	spQueueToSpQueueId.put(spQueue.getDEID(), spQueue);
        }
        for (Map.Entry<ID, Collection<LSGAgeBucket>> entry: spQueueIdToAgeBuckets.entrySet()) {
            CFLSGAgeBucketHashMap cflsgAgeBucketHashMap = new CFLSGAgeBucketHashMap();
            cflsgAgeBucketHashMap.id = spQueueToSpQueueId.get(entry.getKey()).getQueueID();
            cflsgAgeBucketHashMap.cfLsgAgeBucket = CFLsgAgeBucket.convertToCFLSGAgeBucket(entry.getValue());
            cflsgAgeBucketHashMaps[index] = cflsgAgeBucketHashMap;
            index++;
        }
        return cflsgAgeBucketHashMaps;
    }
}
