package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;

import java.util.Collection;
import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.skill.model.SkillAssignment;

public class CFEmployeeSkill implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id = null;
    public ID workResourceId = null;
    public ID skillID = null;
    public double proficiency = 0.0;
    public int priority = 0;
    public int reserveLevel = 0;
    public Date startDate = null;
    public Date endDate = null;
    public boolean isActive = true;
    
    public static CFEmployeeSkill convert(SkillAssignment from) {
    	CFEmployeeSkill to = new CFEmployeeSkill();
    	
    	to.id = from.getID();
    	to.workResourceId = from.getWorkResourceID();
    	to.skillID = from.getSkillID();
    	to.proficiency = from.getProficiency();
    	to.priority = from.getPriority();
    	to.reserveLevel = from.getClassification();
    	to.startDate = from.getStartTime();
    	to.endDate = from.getEndTime();
    	to.isActive = from.isActive();
    	
    	return to;
    }
    
    
    public static SkillAssignment convert(CFEmployeeSkill from) {
    	SkillAssignment to = new SkillAssignment();
    	
    	to.setID(from.id);
    	to.setWorkResourceID(from.workResourceId);
    	to.setSkillID(from.skillID);
    	to.setProficiency(from.proficiency);
    	to.setPriority(from.priority);
    	to.setClassification(from.reserveLevel);
    	to.setStartTime(from.startDate);
    	to.setEndTime(from.endDate);
    	to.setIsActive(from.isActive);
    	
    	return to;
    }
    
	public static CFEmployeeSkill[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeSkill[] to = new CFEmployeeSkill[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeeSkill.convert((SkillAssignment)i.next());
		return to;
	}
    
}
