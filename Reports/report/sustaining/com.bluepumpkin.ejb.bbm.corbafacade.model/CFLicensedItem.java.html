package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Date;
import java.util.Collection;
import java.util.Iterator;

import com.witness.common.licensing.LicensedItem;

public class CFLicensedItem implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public String parentName;
	public String licensee;
	public String name;
	public String displayName;
	public Date expiration;
	public int[] limitProperties;
	public String[] limitPropertiesXmlAttrPrefix;
	
	public static CFLicensedItem convert(LicensedItem from) {
		CFLicensedItem to = new CFLicensedItem();
		
		LicensedItem parent = (LicensedItem)from.getParent();
		if (parent != null) {
			to.parentName = parent.getName();
		}
		
		to.licensee = from.getLicensee();
		to.name = from.getName();
		to.displayName = from.getDisplayName();
		to.expiration = from.getExpiration();
		to.limitProperties = from.getLimitNumbers();
		to.limitPropertiesXmlAttrPrefix = from.getLimitNumbersXmlAttrPrefix();
		
		return to;		
	}
	
	public static CFLicensedItem[] convert(Collection col) {
		if( col == null || col.size() == 0 )
			return new CFLicensedItem[0];
		
		CFLicensedItem[] to = new CFLicensedItem[col.size()];
		LicensedItem lItem = null;
		int ix = 0;
		for (Iterator i = col.iterator(); i.hasNext(); ix++) {
			lItem = (LicensedItem)i.next();
			to[ix] = CFLicensedItem.convert(lItem);
		}
		return to;
	}
}
