package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;

import com.bluepumpkin.common.util.NumberFactory;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectNode;

/**
 * @author ayu
 *
 * To change the template for this generated type comment go to
 * Window - Preferences - Java - Code Generation - Code and Comments
 */
public class CFUtil {
	public static Collection getChildrenList(ValueObjectNode node, int iType) {
		HashMap wholeMap = node.getChildObjectMap();
		if (wholeMap == null || wholeMap.size() == 0)
			return null;
		
		Integer type = NumberFactory.newInteger(iType);
		HashMap childrenMap = (HashMap)wholeMap.get(type);
		
		if (childrenMap == null || childrenMap.size() == 0)
			return null;
		
		return childrenMap.values();
	}
	
	public static Collection sortDesc(Collection col) {
		if (col != null && !col.isEmpty()) {
			Collections.sort((ArrayList)col);
			Collections.reverse((ArrayList)col);
		}
		return col;
	}
}
