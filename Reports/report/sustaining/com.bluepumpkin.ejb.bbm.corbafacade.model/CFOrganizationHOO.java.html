package com.bluepumpkin.ejb.bbm.corbafacade.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.workresource.model.OrganizationHOO;

import java.io.Serializable;

public class CFOrganizationHOO implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id;
	public ID OrgID;
	public ID InheritOrgId;
	public int [] dayOpens;
	public int [] dayCloses;
	             	
	public static CFOrganizationHOO convert(OrganizationHOO from) {
		CFOrganizationHOO to = new CFOrganizationHOO();
		
		to.id = from.getID();
	//	to.id  = from.getOrganizationID();
		to.OrgID = from.getOrganizationID();
		to.InheritOrgId = from.getInheritOrgID();
	//	to.InheritOrgId = from.getOrganizationID();
		
		
		to.dayOpens = new int[7];
		to.dayCloses = new int [7];
		if (to.dayOpens != null && to.dayCloses != null)
		{
			for (short i = 0; i <= 6; i++) {
		        to.dayOpens[i] = from.getDayOpen((short)(i+1));
		        to.dayCloses[i] = from.getDayClose((short)(i+1));
		    }
		}
		 
		return to;
	}
	
	public static OrganizationHOO convert(CFOrganizationHOO fromHOO) {
		OrganizationHOO to = new OrganizationHOO();
		
		to.setID(fromHOO.id);
		to.setOrganizationID(fromHOO.OrgID);
		to.setInheritOrgID(fromHOO.InheritOrgId);
		
		if (fromHOO.dayOpens != null && fromHOO.dayCloses != null)
		{
			 for (short i = 1; i <= 7; i++) {
		           to.setDayOpen(i, (short)fromHOO.dayOpens[i-1]);
		           to.setDayClose(i, (short)fromHOO.dayCloses[i-1]);
		     } 
		}
		else 
		{
			 for (short i = 1; i <= 7; i++) {
		           to.setDayOpen(i, (short)-1);
		           to.setDayClose(i, (short)-1);
		     } 
			
		}

		return to;
	}
	
	/*  check if it's 24 * 7, i.e. time range is 0 to 1440 for each day
     */
    public boolean isOpen24Hours()
    {
        for (short i = 0; i < 7; i++)
            if (dayOpens[i] != 0 || dayCloses[i] != 1440) return false;
            
        return true;    
    }
    
    /** 
     *  set hour's of operation as 24 * 7, i.e. time ragne is 0 to 1440 for each day
     */
	    public void setOpen24Hours()
	    {
	        for (short i = 0; i < 7; i++)
	        {
	        	dayOpens[i] = 0;
	        	dayCloses[i] = 0;
	        }
	    }
}
