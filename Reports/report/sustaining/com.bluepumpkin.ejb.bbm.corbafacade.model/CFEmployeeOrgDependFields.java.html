package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeOrganizationDependFields;

public class CFEmployeeOrgDependFields implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID organizationID;
	public ID jobTitleID;
	public ID supervisorID;
	public ID teamLeadID;
	public Date startDate;
	public Date endDate;
	
	public static CFEmployeeOrgDependFields convert(EmployeeOrganizationDependFields from) {
		CFEmployeeOrgDependFields to = new CFEmployeeOrgDependFields();
		
		to.organizationID = from.getOrganizationID();
		to.jobTitleID = from.getJobTitleID();
		to.supervisorID = from.getSupervisorID();
		to.teamLeadID = from.getTeamLeadID();
		to.startDate = from.getStartTime();
		to.endDate = from.getEndTime();
		
		return to;
	}
	
	public static CFEmployeeOrgDependFields[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeOrgDependFields[] to = new CFEmployeeOrgDependFields[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeeOrgDependFields.convert((EmployeeOrganizationDependFields)i.next());
		return to;
	}
	
}
