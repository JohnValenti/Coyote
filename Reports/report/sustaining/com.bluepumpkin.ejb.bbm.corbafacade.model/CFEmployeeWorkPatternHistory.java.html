package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;

public class CFEmployeeWorkPatternHistory implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID workResourceID = null;
	public CFEmployeeWorkPattern[] workPatterns = null;
	
	public static CFEmployeeWorkPatternHistory[] convert(HashMap from) {
		if (from == null)
			return null;
		
		CFEmployeeWorkPatternHistory[] to = new CFEmployeeWorkPatternHistory[from.size()];
		
		Map.Entry entry = null;
		int ix = 0;
		for (Iterator i = from.entrySet().iterator(); i.hasNext(); ) {
			entry = (Map.Entry)i.next();
			CFEmployeeWorkPatternHistory item = new CFEmployeeWorkPatternHistory();
			item.workResourceID = (ID)entry.getKey();
			item.workPatterns = CFEmployeeWorkPattern.convert(CFUtil.sortDesc((Collection)entry.getValue()));
			to[ix++] = item;
		}
		
		return to;
	}
}
