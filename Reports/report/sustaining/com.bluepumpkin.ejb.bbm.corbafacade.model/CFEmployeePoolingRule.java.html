package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;

import java.util.Collection;
import java.util.Iterator;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.empworkrule.model.EmpPoolingRule;

public class CFEmployeePoolingRule implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public int minMinutes;
	public String orgids;
    public Date startDate;
    public Date endDate;
    
    public static CFEmployeePoolingRule convert(EmpPoolingRule from) {
    	CFEmployeePoolingRule to = new CFEmployeePoolingRule();
    	
    	to.minMinutes = from.getMinMinutes();
    	to.startDate = from.getStartTime();
    	to.endDate = from.getEndTime();
    	Collection orgs = from.getSecondaryOgs();
    	String strArray = "";;
		for(Object orgID: orgs){
			ID tempID = (ID)orgID;
			if (!strArray.isEmpty())
				strArray += "," + tempID.toString();
			else 
				strArray = tempID.toString();				
		}
    	to.orgids = strArray;
    	return to;
    }
    
	public static CFEmployeePoolingRule[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeePoolingRule[] to = new CFEmployeePoolingRule[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeePoolingRule.convert((EmpPoolingRule)i.next());
		return to;
	}

}
