	public static CFEmployee[] convert(Collection from, Collection<QueueAttributeDefintion> queueAttritDefs) {
		if (from == null)
			return null;
		HashMap<ID, QueueAttributeDefintion> map = new HashMap<ID, QueueAttributeDefintion>();
		for (QueueAttributeDefintion att: queueAttritDefs)
			map.put(att.getID(), att);
		
		CFEmployee[] to = new CFEmployee[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployee.convert((Employee)i.next(), map);
		return to;
	}
	
	public static Collection convert(CFEmployee[] from) {
		if (from == null)
			return null;
		
		ArrayList to = new ArrayList(from.length);
		for (int ix = 0; ix < from.length; ix++) {
			to.add(CFEmployee.convert(from[ix]));
		}
		return to;
	}
