package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;

public class CFEmployeeTimeBankHistory implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID employeeID = null;
	public CFEmployeeTimeBank[] timeBanks = null;
	
	public static CFEmployeeTimeBankHistory[] convert(HashMap from) {
		if (from == null)
			return null;
		
		CFEmployeeTimeBankHistory[] to = new CFEmployeeTimeBankHistory[from.size()];
		
		Map.Entry entry = null;
		int ix = 0;
		for (Iterator i = from.entrySet().iterator(); i.hasNext(); ) {
			entry = (Map.Entry)i.next();
			CFEmployeeTimeBankHistory item = new CFEmployeeTimeBankHistory();
			item.employeeID = (ID)entry.getKey();
			item.timeBanks = CFEmployeeTimeBank.convert(CFUtil.sortDesc((Collection)entry.getValue()));
			to[ix++] = item;
		}
		
		return to;
	}
}
