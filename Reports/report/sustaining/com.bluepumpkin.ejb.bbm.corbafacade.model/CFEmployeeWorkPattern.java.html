package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkResourceWorkPattern;

public class CFEmployeeWorkPattern implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id = null;
	public ID workResourceId = null;
	public ID workPatternId = null;
	public int preference = -1;
	public Date startDate = null;
	public Date endDate = null;
	public boolean isActive = true;
	
	public static CFEmployeeWorkPattern convert (WorkResourceWorkPattern from) {
		CFEmployeeWorkPattern to = new CFEmployeeWorkPattern();
		
		to.id = from.getID();
		to.workResourceId = from.getWorkResourceID();
		to.workPatternId = from.getWorkPatternSID();
		to.preference = from.getPreference();
		to.startDate = from.getStartTime();
		to.endDate = from.getEndTime();
		to.isActive = from.isActive();
		
		return to;
	}
	
	public static WorkResourceWorkPattern convert(CFEmployeeWorkPattern from) {
		WorkResourceWorkPattern to = new WorkResourceWorkPattern();
		
		to.setID(from.id);
		to.setWorkResourceID(from.workResourceId);
		to.setWorkPatternSID(from.workPatternId);
		to.setPreference(from.preference);
		to.setStartTime(from.startDate);
		to.setEndTime(from.endDate);
		to.setIsActive(from.isActive);
		
		return to;		
	}
	
	public static CFEmployeeWorkPattern[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeWorkPattern[] to = new CFEmployeeWorkPattern[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeeWorkPattern.convert((WorkResourceWorkPattern)i.next());
		return to;
	}
}
