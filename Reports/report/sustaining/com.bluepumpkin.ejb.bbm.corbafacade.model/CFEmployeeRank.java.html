package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeRank;

public class CFEmployeeRank implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public int rank;
	public Date startDate;
	public Date endDate;
	
	public static CFEmployeeRank convert(EmployeeRank from) {
		CFEmployeeRank to = new CFEmployeeRank();
		
		to.rank = from.getRank();
		to.startDate = from.getStartTime();
		to.endDate = from.getEndTime();
		
		return to;
	}
	
	public static CFEmployeeRank[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeRank[] to = new CFEmployeeRank[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeeRank.convert((EmployeeRank)i.next());
		
		return to;
	}
}
