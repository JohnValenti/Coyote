package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.empworkrule.model.EmpTimeBankAdjust;

public class CFEmployeeTimeBankAdjust implements Serializable {
	
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id;
	public ID employeeID;
	public int adjustMinute;
	public Date adjustDate;
	
	public static CFEmployeeTimeBankAdjust convert(EmpTimeBankAdjust from) {
		CFEmployeeTimeBankAdjust to = new CFEmployeeTimeBankAdjust();
		
		to.id = from.getID();
		to.employeeID = from.getEmployeeID();
		to.adjustMinute = from.getAdjustMinute();
		to.adjustDate = from.getAdjustDate();
		
		return to;
	}
	
	public static EmpTimeBankAdjust convert(CFEmployeeTimeBankAdjust from) {
		EmpTimeBankAdjust to = new EmpTimeBankAdjust();
		
		to.setID(from.id);
		to.setEmployeeID(from.employeeID);
		to.setAdjustMinute(from.adjustMinute);
		to.setAdjustDate(from.adjustDate);
		
		return to;
	}
	
	public static CFEmployeeTimeBankAdjust[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeTimeBankAdjust[] to = new CFEmployeeTimeBankAdjust[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeeTimeBankAdjust.convert((EmpTimeBankAdjust)i.next());
		return to;
	}

}
