package com.bluepumpkin.ejb.bbm.timeseries.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.*;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriodFieldInfo;
import com.bluepumpkin.ejb.bbm.dao.DAOBaseWithAutoIDMapping;
import com.bluepumpkin.ejb.bbm.dao.DAOUtil;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastInstance;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastInstanceFieldInfo;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;

public class ForecastInstanceDAO extends DAOBaseWithAutoIDMapping<ForecastInstance> {
	private static FieldInfo m_fieldInfo = new ForecastInstanceFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ForecastInstanceDAO() {
		super();
	}

	@Override
	protected Map<Integer, ? extends FieldInfo> getForeignKeyFieldInfo() {
		return Collections.singletonMap(ForecastInstanceFieldInfo.FORECASTINSTANCE_SPID, new SchedulingPeriodFieldInfo());
	}

	public static ID createForecastInstance(ForecastInstance objValue) throws JdmoException {
		Jdmo jdmo = new Jdmo();
		try {
			// get ID from DE
			String idStr = DAOUtil.getDEID(jdmo);

			// get the sp str id
			String spStmt = new String("select ID from SP where SID=?");
			JdmoQuery spQuery = jdmo.createQuery(spStmt, Jdmo.PARAM_QUERY);
			spQuery.setParString(1, objValue.getSpId().toString());
			JdmoRowset rsSp = jdmo.createRowset(spQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			String spID = null;
			if (rsSp.next()) {
				spID = rsSp.getString(1);
			}

			String pStmt1 = "insert into FORECASTINSTANCE (ID, SPID, ISBASE, DESCRIPTION, CREATED, CREATEDBY) values (?, ?, ?, ?, ?, ?)";

			Object[] param = new Object[6];
			param[0] = idStr;
			param[1] = spID;
			param[2] = new Boolean(objValue.getIsBase());
			param[3] = objValue.getDescription();
			param[4] = objValue.getCreated();
			param[5] = objValue.getCreatedBy();

			jdmo.executePCommand(pStmt1, param);

			StringBuffer pStmt = new StringBuffer("select SID from FORECASTINSTANCE where ID=?");
			JdmoQuery jq1 = jdmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			jq1.setParString(1, idStr);
			JdmoRowset rs = jdmo.createRowset(jq1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

			ID forecastInstanceID = null;
			if (rs.next()) {
				forecastInstanceID = rs.getID(1);
			}

			objValue.setId(new ID(idStr)); //set the real ID (the DE string ID)
			objValue.setID(forecastInstanceID); //set the SID

			return forecastInstanceID;
		} finally {
			jdmo.cleanUp();
		}
	}

	public static void deleteForecastInstanceBySpQueue(ID spId) throws JdmoException {
		if (spId == null) {
			return;
		}
		Jdmo jdmo = new Jdmo();
		try {

			StringBuffer strSQL = new StringBuffer(30);
			strSQL.append("delete from FORECASTINSTANCE ");
			strSQL.append("where SID=(");
			strSQL.append("select FORECASTINSTANCE.SID from FORECASTINSTANCE,SP");
			strSQL.append(" where FORECASTINSTANCE.SPID=SP.ID and SP.SID=?)");

			Object[] param = new Object[1];
			param[0] = spId;
			jdmo.executePCommand(strSQL.toString(), param);
		} finally {
			jdmo.cleanUp();
		}
	}

	/**
	 * Returns the forecast instance for a Scheduling Period with a given name.
	 * If none exists, returns null.
	 *
	 * @param name
	 * @param spID
	 * @return
	 * @throws BbmFinderException
	 */
	public ForecastInstance getObjectByNameAndSPID(String name, ID spID) throws BbmFinderException {
		spID = DAOUtil.mapIDToDEID(spID, new SchedulingPeriodFieldInfo());
		Collection<ForecastInstance> retCol = getObjects(" " + m_fieldInfo.getFieldName(ForecastInstanceFieldInfo.FORECASTINSTANCE_DESCRIPTION) + " = " + JdmoUtil.asSqlLiteral(name)
				+ " AND " + m_fieldInfo.getFieldName(ForecastInstanceFieldInfo.FORECASTINSTANCE_SPID) + " = " + JdmoUtil.asSqlLiteral(spID));
		if (retCol.size() > 0) {
			return retCol.iterator().next();
		}
		return null;
	}

	@Override
	protected ForecastInstance createValueObject() {
		return new ForecastInstance();
	}

	public ForecastInstance getBaseForecastInstance(ID spID) throws BbmFinderException {
		spID = DAOUtil.mapIDToDEID(spID, new SchedulingPeriodFieldInfo());
		Collection<ForecastInstance> retValCol = getObjects(" " + m_fieldInfo.getFieldName(ForecastInstanceFieldInfo.FORECASTINSTANCE_SPID) + " = " + JdmoUtil.asSqlLiteral(spID)
				+ " AND " + m_fieldInfo.getFieldName(ForecastInstanceFieldInfo.FORECASTINSTANCE_ISBASE) + " = 1");
		if (retValCol.isEmpty()) {
			return null;
		}
		return retValCol.iterator().next();
	}

	/**
	 * Returns a list of forecast instance IDs that both belong to the given SP and are also linked to the given
	 * queues.  Forecast instances are linked to SPs via the SPID column on the forecast instance table,
	 * and they are linked to queues via the SPQUEUE column on the Forecast Profile table, which has a
	 * forecast instance column id on it.
	 */
	public Collection<ID> getForecastInstanceIdsForSpsAndQueues(ID spId, Collection<ID> queueIds) throws JdmoException {
		Jdmo jdmo = new Jdmo();
		try {
			StringBuffer sql = new StringBuffer();
			sql.append("select fi.ID from FORECASTINSTANCE fi ")
					.append("INNER JOIN FORECASTPROFILE fp ")
					.append(" ON fi.ID = fp.FORECASTINSTANCEID ")
					.append(" INNER JOIN SPQUEUE spq ")
					.append(" ON fp.SPQUEUEID = spq.ID ")
					.append(" INNER JOIN QUEUE q ")
					.append(" ON q.ID = spq.QUEUEID ")
					.append(" INNER JOIN SP sp ")
					.append(" ON fi.SPID = sp.ID ")
					.append("where sp.SID = ? and q.SID IN ")
					.append(jdmo.createInClause(queueIds));

			JdmoQuery jq = jdmo.createQuery(sql.toString(), Jdmo.PARAM_QUERY);
			jq.setParID(1, spId);
			JdmoRowset rs = jdmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			Collection<ID> forecastInstanceIds = new ArrayList<ID>();
			if (rs.next()) {
				forecastInstanceIds.add(rs.getID(1));
			}
			return forecastInstanceIds;
		} finally {
			jdmo.cleanUp();
		}
	}
}
