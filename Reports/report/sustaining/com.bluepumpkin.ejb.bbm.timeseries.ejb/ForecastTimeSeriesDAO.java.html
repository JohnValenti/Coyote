	@SuppressWarnings("unchecked")
	public static ID createForecastTimeSeries(ForecastTimeSeries objValue) throws JdmoException {
		Jdmo jdmo = new Jdmo();
		ID forecastTimeSeriesID = null;
		try {
			// get ID from DE
			String idStr = DAOUtil.getDEID(jdmo);

			// get the spQueue str id
			String spQueueStmt = new String("select ID from SPQUEUE where SID=?");
			JdmoQuery spQueueQuery = jdmo.createQuery(spQueueStmt, Jdmo.PARAM_QUERY);
			spQueueQuery.setParString(1, objValue.getSpQueueId().toString());
			JdmoRowset rsSpQueue = jdmo.createRowset(spQueueQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			String spQueueID = null;
			if (rsSpQueue.next()) {
				spQueueID = rsSpQueue.getString(1);
			}

			// get the forecast instance str id
			String forecastInstanceStmt = new String("select ID from FORECASTINSTANCE where SID=?");
			JdmoQuery forecastInstanceQuery = jdmo.createQuery(forecastInstanceStmt, Jdmo.PARAM_QUERY);
			forecastInstanceQuery.setParString(1, objValue.getForecastInstanceId().toString());
			JdmoRowset rsForecastInstance = jdmo.createRowset(forecastInstanceQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			String forecastInstanceID = null;
			if (rsForecastInstance.next()) {
				forecastInstanceID = rsForecastInstance.getString(1);
			}


			StringBuffer buf = new StringBuffer();
			buf.append("insert into FORECASTTIMESERIES (ID, SPQUEUEID, DATETIME, FORECASTINSTANCEID, CREATED, CREATEDBY");
			Class forecastTimeSeriesClass = objValue.getClass();
			for (int i = 1; i < 97; i++) {
				buf.append(", AHT").append(i);
				buf.append(", CALLVOLUME").append(i);
			}
			buf.append(") values (?, ?, ?, ?, ?, ?");
			for (int i = 1; i < 97; i++) {
				buf.append(", ?");
				buf.append(", ?");
			}
			buf.append(")");

			int index = 0;
			Object[] param = new Object[198];
			param[index++] = idStr;
			param[index++] = spQueueID;
			param[index++] = objValue.getDateTime();
			param[index++] = forecastInstanceID;
			param[index++] = objValue.getCreated();
			param[index++] = objValue.getCreatedBy();

			for (int i = 1; i < 97; i++) {
				Method setAhtMethod = forecastTimeSeriesClass.getMethod("getAht" + i, (Class[]) null);
				Object ahtValue = setAhtMethod.invoke(objValue, (Object[]) null);
				param[index++] = ahtValue;

				Method setCvMethod = forecastTimeSeriesClass.getMethod("getCallVolume" + i, (Class[]) null);
				Object cvValue = setCvMethod.invoke(objValue, (Object[]) null);
				param[index++] = cvValue;
			}

			jdmo.executePCommand(buf.toString(), param);

			StringBuffer pStmt = new StringBuffer("select SID from FORECASTTIMESERIES where ID=?");
			JdmoQuery jq1 = jdmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			jq1.setParString(1, idStr);
			JdmoRowset rs = jdmo.createRowset(jq1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

			if (rs.next()) {
				forecastTimeSeriesID = rs.getID(1);
			}
		} catch (InvocationTargetException e) {
			// TODO - do something
		} catch (NoSuchMethodException e) {
			// TODO - do something
		} catch (IllegalAccessException e) {
			// TODO - do something
		} finally {
			jdmo.cleanUp();
		}
		return forecastTimeSeriesID;
	}

	public static void deleteForecastTimeSeriesBySpQueue(ID spQueueSId) throws BbmRemoveException {
		if (spQueueSId == null) {
			return;
		}
		Jdmo jdmo = new Jdmo();
		try {
			// get the spqueue str id
			String spQueueStmt = new String("select ID from SPQUEUE where SID=?");
			JdmoQuery spQueueQuery = jdmo.createQuery(spQueueStmt, Jdmo.PARAM_QUERY);
			spQueueQuery.setParString(1, spQueueSId.toString());
			JdmoRowset rsSp = jdmo.createRowset(spQueueQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			String spQueueID = null;
			if (rsSp.next()) {
				spQueueID = rsSp.getString(1);
			}

			StringBuffer strSQL = new StringBuffer(30);
			strSQL.append("delete from FORECASTTIMESERIES where SPQUEUEID=?");

			Object[] param = new Object[1];
			param[0] = spQueueID;
			jdmo.executePCommand(strSQL.toString(), param);
		} catch (JdmoException e) {
			throw new BbmRemoveException(e);
		} finally {
			jdmo.cleanUp();
		}
	}

	/**
	 * @param forecastInstanceID, may be null for active forecast
	 * @param spQueueIDs
	 * @throws BbmRemoveException
	 */
	public void deleteObjectsByInstanceAndSPQueueIDs(ID forecastInstanceID, Collection<? extends ID> spQueueIDs) throws BbmRemoveException {
		try {
			spQueueIDs = DAOUtil.mapIDsToDEIDs(spQueueIDs, new SPQueueFieldInfo()).values();
			forecastInstanceID = DAOUtil.mapIDToDEID(forecastInstanceID, new ForecastInstanceFieldInfo());
			StringBuffer sql = new StringBuffer(" SELECT ");
			sql.append(m_fieldInfo.getFieldName(ForecastTimeSeriesFieldInfo.FORECASTTIMESERIES_ID))
					.append(" FROM ")
					.append(m_fieldInfo.getTableName())
					.append(" WHERE ")
					.append(m_fieldInfo.getFieldName(ForecastTimeSeriesFieldInfo.FORECASTTIMESERIES_FORECASTINSTANCEID));
			if (forecastInstanceID == null) {
				sql.append(" IS NULL AND ");
			} else {
				sql.append("=").append(JdmoUtil.asSqlLiteral(forecastInstanceID)).append(" AND ");
			}
			sql.append(m_fieldInfo.getFieldName(ForecastTimeSeriesFieldInfo.FORECASTTIMESERIES_SPQUEUEID))
					.append(" IN ")
					.append(getDMO().createInClause(spQueueIDs));
			super.deleteObjects(sql.toString());
		} catch (Exception e) {
			if (e instanceof BbmRemoveException) {
				throw (BbmRemoveException) e;
			}
			throw new BbmRemoveException(e);
		}
	}

	/**
	 * @param forecastInstanceID, may be null for active forecast
	 * @param spQueueIDs
	 * @return
	 * @throws BbmFinderException
	 */
	public Map<ID, Boolean> isForecastSet(ID forecastInstanceID, Collection<? extends ID> spQueueIDs) throws BbmFinderException {
		try {
			HashMap<ID, Boolean> retVal = new HashMap<ID, Boolean>();
			for (ID id : spQueueIDs) {
				retVal.put(id, Boolean.FALSE);
			}

			spQueueIDs = DAOUtil.mapIDsToDEIDs(spQueueIDs, new SPQueueFieldInfo()).values();

			forecastInstanceID = DAOUtil.mapIDToDEID(forecastInstanceID, new ForecastInstanceFieldInfo());
			StringBuffer sql = new StringBuffer(" SELECT DISTINCT ");
			sql.append(m_fieldInfo.getFieldName(ForecastTimeSeriesFieldInfo.FORECASTTIMESERIES_SPQUEUEID))
					.append(" FROM ")
					.append(m_fieldInfo.getTableName())
					.append(" WHERE ")
					.append(m_fieldInfo.getFieldName(ForecastTimeSeriesFieldInfo.FORECASTTIMESERIES_FORECASTINSTANCEID));
			if (forecastInstanceID == null) {
				sql.append(" IS NULL AND ");
			} else {
				sql.append("=").append(JdmoUtil.asSqlLiteral(forecastInstanceID)).append(" AND ");
			}
			sql.append(m_fieldInfo.getFieldName(ForecastTimeSeriesFieldInfo.FORECASTTIMESERIES_SPQUEUEID))
					.append(" IN ")
					.append(getDMO().createInClause(spQueueIDs));
			JdmoRowset rs = getDMO().createRowset(sql.toString());
			Collection<ID> idsWithData = new HashSet<ID>();
			while (rs.next()) {
				idsWithData.add(rs.getID(1));
			}
			rs.close();
			idsWithData = DAOUtil.mapIDsToSIDs(idsWithData, new SPQueueFieldInfo()).values();
			for (ID id : idsWithData) {
				retVal.put(id, Boolean.TRUE);
			}
			return retVal;
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}
