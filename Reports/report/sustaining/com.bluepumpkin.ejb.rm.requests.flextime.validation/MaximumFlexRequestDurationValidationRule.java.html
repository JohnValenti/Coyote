package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Date;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEvent;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.DateUtil;

public class MaximumFlexRequestDurationValidationRule extends FlexRequestDurationValidationRule {
	public static final String MAX_FLEX_REQUEST_DURATION_VALUE = "MAX_FLEX_REQUEST_DURATION_VALUE";

	public static final String MAX_FLEX_REQUEST_DURATION_UNIT = "MAX_FLEX_REQUEST_DURATION_UNIT";

	public static final ID RULE_ID = new ID(-194063);


	@Override
	protected void init(TORequest toRequest) throws Exception {
		//only one TOC per flex request
		this.toChoice = toRequest.getRequestChoiceList().iterator().next();
		this.orgID = getEmployeeOrgId(toRequest.getEmployeeID(), toChoice);

		@SuppressWarnings("rawtypes")
		Map ruleParams = getValidationRuleManager().getValidationRuleParams(orgID, RULE_ID);
		this.duration = ((Integer) ruleParams.get(MAX_FLEX_REQUEST_DURATION_VALUE)).intValue();
		this.unit = ((Integer) ruleParams.get(MAX_FLEX_REQUEST_DURATION_UNIT)).intValue();
	}



	@Override
	protected TimeRange getOverLimitRange(long durationLimitMillis, long accountedMillis) throws Exception {

		// Use a really big date range to look for consecutive requests (7 days before and after the time off time)
		Date timeOffRangeStart = new Date(this.toChoice.getStartDate().getTime() - durationLimitMillis - 7 * TimeZoneUtil.DAY_IN_MILLISECONDS);
		Date timeOffRangeEnd = new Date(this.toChoice.getEndDate().getTime() + durationLimitMillis + 7 * TimeZoneUtil.DAY_IN_MILLISECONDS);
		Collection<TORequest> approvedRequests = getFTValidationCache(toRequest).getApprovedFlexRequests(toRequest.getEmployeeID(), new TimeRange(timeOffRangeStart, timeOffRangeEnd));
		Collection<TORequest> consecutiveRequests = getConsecutiveRequests(toRequest, approvedRequests);
		int totalFlexTimeInMinutes = 0;
		Collection<Date> timeOffDates = new ArrayList<Date>();
		for (TORequest request : consecutiveRequests) {
			// we use makeup time (instead to calculating time off time from calendar) because makeup time must be equal to time off time
			totalFlexTimeInMinutes += getFTValidationCache(toRequest).getTotalMakeupMinutes(request);
			timeOffDates.add(request.getFirstTOChoice().getStartDate());
			timeOffDates.add(request.getFirstTOChoice().getEndDate());
		}

		return (totalFlexTimeInMinutes * DateUtil.MINUTE_IN_MSECS > durationLimitMillis) ? new TimeRange(Collections.min(timeOffDates), Collections.max(timeOffDates)) : null;
	}

	private Collection<TORequest> getConsecutiveRequests(TORequest toRequest, Collection<TORequest> existingRequests) {
		Collection<TORequest> consecutiveRequests = new ArrayList<TORequest>();
		TimeRange consecutiveRange = new TimeRange(toRequest.getFirstTOChoice().getStartDate(), toRequest.getFirstTOChoice().getEndDate());
		Collection<TORequest> requests = new ArrayList<TORequest>(existingRequests);
		do {
			TORequest consecutiveRequest = getFirstConsecutiveRequest(requests, consecutiveRange);
			if (consecutiveRequest != null) {
				consecutiveRequests.add(consecutiveRequest);
				consecutiveRange = consecutiveRange.joinWithOverlap(
						new TimeRange(consecutiveRequest.getFirstTOChoice().getStartDate(), consecutiveRequest.getFirstTOChoice().getEndDate()), 0);
				requests.remove(consecutiveRequest);
			} else {
				break;
			}
		} while (true);
		consecutiveRequests.add(toRequest);
		return consecutiveRequests;
	}

	private TORequest getFirstConsecutiveRequest(Collection<TORequest> requests, TimeRange consecutiveRange) {
		for (TORequest request : requests) {
			TimeRange timeOffRange = new TimeRange(request.getFirstTOChoice().getStartDate(), request.getFirstTOChoice().getEndDate());
			int adjacency = consecutiveRange.getAdjacency(timeOffRange, 0);
			long overlapDuration = consecutiveRange.getOverlapDuration(timeOffRange);
			if (adjacency == TimeRange.ADJACENT_BEFORE || adjacency == TimeRange.ADJACENT_AFTER || overlapDuration > 0) {
				return request;
			}
		}
		return null;
	}

	protected boolean intersect(TOChoice toChoice, CalendarEvent calEvent) {
		return calEvent.getEndTime().after(toChoice.getStartDate()) && calEvent.getStartTime().before(toChoice.getEndDate());
	}

	@Override
	protected String getClassName() {
		return MaximumFlexRequestDurationValidationRule.class.getName();
	}

	@Override
	protected ValidationResult setSoftValidationResult(TimeRange overLimitRange) {
		String messageKey = overLimitRange.equals(new TimeRange(this.toChoice.getStartDate(), this.toChoice.getEndDate())) ?
				RmEjbBundleKey.FLEX_REQ_CHOICE_MAX_DURATION : RmEjbBundleKey.FLEX_REQ_CHOICE_WITH_CONSECUTIVE_REQS_MAX_DURATION;

		return ValidationUtil.setSoftValidationResult(this.toChoice,
				messageKey, overLimitRange.getStartDate(), overLimitRange.getEndDate(), getClassName());
	}
}
