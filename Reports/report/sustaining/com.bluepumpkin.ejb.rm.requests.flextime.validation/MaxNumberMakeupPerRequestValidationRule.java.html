package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexRequestMakeup;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.validation.ejb.ValidationRuleManager;
import com.bluepumpkin.ejb.rm.util.RmUtil;

public class MaxNumberMakeupPerRequestValidationRule implements Validator {
	public static final String MAX_FLEX_MAKEUP_PER_REQUEST = "MAX_FLEX_MAKEUP_PER_REQUEST";

	public static final ID RULE_ID = new ID(-194061);

	private static final String CLASS_NAME = MaxNumberMakeupPerRequestValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result;

		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod("validate", validatable));
		}

		TORequest tor = (TORequest) validatable;

		int maxMakeupBlocks = getMaxMakeupBlocksAllowed(getFTValidationCache(tor), RULE_ID);

		result = validateRequestMakeupBlockNumberNotOverMaxValue(tor, maxMakeupBlocks);

		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpExitMethod("validate", result));
		}
		return result;
	}

	private ValidationResult validateRequestMakeupBlockNumberNotOverMaxValue(TORequest tor, int maxMakeupBlocks) {
		ValidationResult result;
		int makeupBlockCount = getMakeupBlockCount(tor);
		if (makeupBlockCount > maxMakeupBlocks) {
			result = ValidationUtil.setSoftValidationResult(tor, RmEjbBundleKey.FLEX_REQ_MAKEUP_NUMBER_OVER_MAX, maxMakeupBlocks, CLASS_NAME);
		} else {
			result = null;
		}
		return result;
	}

	private int getMakeupBlockCount(TORequest tor) {
		int makeupBlockCount = 0;
		for (FlexRequestMakeup makeup : tor.getFlexRequestMakeupList()) {
			if (makeup.getExtBeforeActivityID() != null) {
				makeupBlockCount ++;
			}
			if (makeup.getExtAfterActivityID() != null) {
				makeupBlockCount ++;
			}
			if (makeup.getFlexReqMakeupGaps() != null || !makeup.getFlexReqMakeupGaps().isEmpty()) {
				makeupBlockCount += makeup.getFlexReqMakeupGaps().size();
			}

		}
		return makeupBlockCount;
	}

	private int getMaxMakeupBlocksAllowed(FTValidationCache cache, ID ruleID) throws Exception {
		ValidationRuleManager valMan = getValidationRuleManager();
		Map<String, Integer> ruleParams = valMan.getValidationRuleParams(cache.getOrg().getID(), ruleID);
		return ruleParams.get(MAX_FLEX_MAKEUP_PER_REQUEST);
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}

	protected ValidationRuleManager getValidationRuleManager() throws BbmCreateException {
		return RmManagerFactory.getInstance().getValidationRuleManager();
	}

}
