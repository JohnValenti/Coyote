package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.validation.ejb.ValidationRuleManager;
import com.bluepumpkin.ejb.rm.util.DateUtil;
import com.bluepumpkin.ejb.rm.util.RmUtil;

public abstract class FlexRequestDurationValidationRule implements Validator  {
	private static final Category LOG = Log.initCategory(FlexRequestDurationValidationRule.class.getName());

	private String methodName = "validate";

	protected TORequest toRequest;
	protected int duration;
	protected int unit;
	protected TOChoice toChoice;
	protected ID orgID;

	protected abstract void init(TORequest toRequest) throws Exception;

	protected abstract TimeRange getOverLimitRange(long durationLimit, long choiceDuration) throws Exception;

	protected abstract String getClassName();

	protected abstract ValidationResult setSoftValidationResult(TimeRange overlimitRange);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		LOG.debug(RmUtil.dumpEnterMethod(methodName, validatable));

		this.toRequest = (TORequest) validatable;

		init(this.toRequest);

		ValidationResult result = isRuleVilolated();

		LOG.debug(RmUtil.dumpExitMethod(methodName, result));
		return result;
	}

	/**Check if the rule is violated or not. It could be the TO duration is over or less than the max/min limit
	 *
	 * @return
	 * @throws Exception
	 */
	protected ValidationResult isRuleVilolated() throws Exception {

		long durationLimitMillis = getDurationLimit(this.duration, this.unit);
		long accountedMillis = DateUtil.MINUTE_IN_MSECS * getFTValidationCache(this.toRequest).getTotalMakeupMinutes(this.toRequest);

		ValidationResult result = null;
		TimeRange overLimitRange = getOverLimitRange(durationLimitMillis, accountedMillis);
		if (overLimitRange != null) {
			result = setSoftValidationResult(overLimitRange);
		}

		return result;
	}

	/**
	 * get duration from hours and minutes
	 * @param duration
	 * @param unit
	 * @return
	 */
	protected long getDurationLimit(int duration, int unit) {
		long valDuration = 0;
		if (unit == 1) {
			//HOUR
			valDuration = duration * DateUtil.HOUR_IN_MSECS;
		} else {
			//MINUTE
			valDuration = duration * DateUtil.MINUTE_IN_MSECS;
		}
		return valDuration;
	}

	/**Get the Org's timezone.
	 *
	 * @param toRequest
	 * @return
	 * @throws Exception
	 */
	protected TimeZone getOrgTimeZone(TORequest toRequest) throws Exception {
		return getFTValidationCache(toRequest).getOrg().getTimeZone();
	}

	/**Check if the validation rule is for the org or not
	 *
	 * @param toRequest
	 * @param orgID
	 * @return
	 * @throws Exception
	 */
    protected boolean isValidationRequiredForOrg(TORequest toRequest, ID orgID) throws Exception {
		return getFTValidationCache(toRequest).isValidationRequiredForOrg(orgID, "time-off", getClassName());
    }

    /**Return the orgID of the employee
     *
     * @param employeeID
     * @param toChoice
     * @return
     * @throws Exception
     */
	protected ID getEmployeeOrgId(ID employeeID, TOChoice toChoice) throws Exception {
		return ValidationUtil.getOrgIDForEmployeeDuringPeriod(employeeID, toChoice.getStartDate(), toChoice.getEndDate());
	}

	protected ValidationRuleManager getValidationRuleManager() throws BbmCreateException {
		return RmManagerFactory.getInstance().getValidationRuleManager();
	}
	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}
}
