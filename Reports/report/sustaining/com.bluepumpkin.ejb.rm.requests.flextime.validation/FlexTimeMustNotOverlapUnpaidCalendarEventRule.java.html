package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.activity.model.Event;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventAssignment;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.RmUtil;

/**
 * This class is an implementation of the following Hard Validation rule: <b>The
 * Flex Time Off request must not overlap unpaid calendar event.</b><br/>
 * The Flex Time Off must NOT be overlapping following items:
 * <li>Time Offs</li>
 * <li>Unavailabilities</li>
 * <li>Any calendar event with an unpaid activity</li>
 */
public class FlexTimeMustNotOverlapUnpaidCalendarEventRule implements Validator {
	private static final String CLASS_NAME = FlexTimeMustNotOverlapUnpaidCalendarEventRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		LOG.debug(RmUtil.dumpEnterMethod("validate", validatable));
		TORequest tor = (TORequest) validatable;

		TimeRange timeOffRange = new TimeRange(tor.getFirstTOChoice().getStartDate(),
				tor.getFirstTOChoice().getEndDate());
		Collection<CalendarEventAssignment> eventsInTimeOffRange = getPublishedShiftAssignments(tor, timeOffRange);
		Collection<CalendarEventAssignment> overlappedEvents = getOverlappedTimeOffAndUnavailabilitiesEvents(
				timeOffRange, eventsInTimeOffRange);

		// check the flex time off overlap the time off, unavailability
		// ==> don't pass the rule
		if (isTimeOffOrUnavailabilityEventsOverlapped(overlappedEvents)) {
			return createViolationResult(tor);
		}

		Collection<ID> activityIDsOfOverlapedEvent = getActivityIDFromCalendarEvents(overlappedEvents);
		Collection<Activity> activitiesOfOverlappedEvent = getActivitiesFromIDs(tor, activityIDsOfOverlapedEvent);
		Collection<ID> unpaidActivityIDsOfOverlapEvent = getUnpaidActivityIDs(activitiesOfOverlappedEvent);

		// check the flex time off overlap the unpaid activities
		// ==> don't pass the rule
		if (unpaidActivityIDsOfOverlapEvent != null && !unpaidActivityIDsOfOverlapEvent.isEmpty()) {
			return createViolationResult(tor);
		}

		// pass the rule, no violation
		return null;
	}

	private ValidationResult createViolationResult(TORequest tor) {
		return ValidationUtil.setHardValidationResult(tor,
				RmEjbBundleKey.FLEX_REQ_TIME_OFF_CANNOT_OVERLAP_UNAVAILABILITY_TIMEOFF_UNPAID_EVENT, CLASS_NAME);
	}

	/**
	 * Check if the Flex Time Off portion overlap following items:
	 * <li>Time Offs</li>
	 * <li>Unavailabilities</li>
	 * <li>Any calendar event with an unpaid activity</li>
	 * 
	 * @param timeOffRange
	 * @param timeOffAndUnvailaTimeRanges
	 * @return
	 */
	private boolean isTimeOffOrUnavailabilityEventsOverlapped(Collection<CalendarEventAssignment> overlappedEvents) {
		for (CalendarEventAssignment event : overlappedEvents) {
			if (event.getEventType() == Event.EVENT_TYPE_TIME_OFF
					|| event.getEventType() == Event.EVENT_TYPE_UNAVAILABILITY) {
				return true;
			}
		}

		return false;
	}

	private Collection<ID> getActivityIDFromCalendarEvents(Collection<CalendarEventAssignment> calendarEvents) {
		Collection<ID> activityIDs = new ArrayList<ID>();

		for (CalendarEventAssignment event : calendarEvents) {
			int eventType = event.getEventType();
			if (eventType == Event.EVENT_TYPE_CALENDAR_EVENT_ASSIGNMENT) {
				if (!activityIDs.contains(event.getActivityID())) {
					activityIDs.add(event.getActivityID());
				}
			}
		}

		return activityIDs;
	}

	private Collection<CalendarEventAssignment> getOverlappedTimeOffAndUnavailabilitiesEvents(TimeRange timeOffRange,
			Collection<CalendarEventAssignment> eventsInTimeOffRange) {
		List<CalendarEventAssignment> overlappedEvents = new ArrayList<CalendarEventAssignment>();

		TimeRange eventTimeRange = new TimeRange(new Date(), new Date());
		for (CalendarEventAssignment event : eventsInTimeOffRange) {
			eventTimeRange.setStartDate(event.getStartTime());
			eventTimeRange.setEndDate(event.getEndTime());
			if (timeOffRange.getOverlapInterval(eventTimeRange) != null && !eventTimeRange.isInstant()) {
				overlappedEvents.add(event);
			}
		}
		return overlappedEvents;
	}

	private Collection<ID> getUnpaidActivityIDs(Collection<Activity> activities) {
		Collection<ID> activityIDs = new ArrayList<ID>();

		if (activities == null || activities.isEmpty()) {
			return activityIDs;
		}

		for (Activity act : activities) {
			// collect unpaid activity
			if (!act.isPaid()) {
				activityIDs.add(act.getID());
			}
		}

		return activityIDs;
	}

	protected Collection<CalendarEventAssignment> getPublishedShiftAssignments(TORequest tor, TimeRange eventsDuration)
			throws Exception {
		return getFTValidationCache(tor).getPublishedCalendarEvents(eventsDuration);
	}

	protected Collection<Activity> getActivitiesFromIDs(TORequest tor, Collection<ID> activityIDs) throws Exception {
		return getFTValidationCache(tor).getActivitiesFromIDs(activityIDs);
	}

	private FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}
}
