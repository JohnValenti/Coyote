package com.bluepumpkin.web.bbm.campaign.selection;

import com.verint.web.wfm.keys.WfmJavaScriptFileID;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.pagemodel.SelectionpaneTreePM;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        CampaignSchedulingPeriodSelectionPM
 * Description:  Displays the Campaign/Scheduling Period Selection pane
 * Copyright:    Copyright (c) 2009
 * Company:      Verint Systems, Inc
 * @author       Dustin Stanley
 */
public class CampaignSchedulingPeriodSelectionPM extends SelectionpaneTreePM {
	
	public CampaignSchedulingPeriodSelectionPM(RequestContext context) {
		super(context, new CampaignSchedulingPeriodTreePC(context,
			! "false".equalsIgnoreCase(context.getScreenParameterValue("Campaign Selectable"))));
		
		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
		setJSMediator(WfmJavaScriptFileID.MEDIATOR_CAMPAIGN_SP_SELECTION);
	}

	public static PageModel getInstance(RequestContext context) {
		return getInstance(context, CampaignSchedulingPeriodSelectionPM.class);
	}
	
	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return "campaign_sp_selection";
	}
	
	protected void initTree() {
		super.initTree();
		((CampaignSchedulingPeriodTreePC)m_treePC).setExpandedNodes(m_selectedID);
	}
}
