package com.bluepumpkin.web.bbm.campaign.selection;

import java.util.Comparator;

import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;

public class CampaignSchedulingPeriodComparator implements Comparator {
	
	private Localizer m_localizer;
	
	public CampaignSchedulingPeriodComparator(Localizer localizer) {
		m_localizer = localizer;
	}

	public int compare(Object obj1, Object obj2) {
		if (obj1.getClass().equals(Campaign.class) && obj2.getClass().equals(SchedulingPeriod.class)) {
			return -1;
		}
		if (obj1.getClass().equals(SchedulingPeriod.class) && obj2.getClass().equals(Campaign.class)) {
			return 1;
		}
		if (obj1.getClass().equals(Campaign.class) && obj2.getClass().equals(Campaign.class)) {
			String name1 = ((Campaign)obj1).getName();
			String name2 = ((Campaign)obj2).getName();
			return m_localizer.getCollator().compare(name1,name2);
		}
		if (obj1.getClass().equals(SchedulingPeriod.class) && obj2.getClass().equals(SchedulingPeriod.class)) {
			LocalDate date1 = ((SchedulingPeriod)obj1).getStartTimeLocal();
			LocalDate date2 = ((SchedulingPeriod)obj2).getStartTimeLocal();
			return date2.compareTo(date1);
		} else {
			return 0;
		}
	}
}
