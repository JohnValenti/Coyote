package com.bluepumpkin.web.fs.shift;

import java.rmi.RemoteException;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.verint.ejb.wfm.WfmManagerFactory;

/**
 * Title:        ShiftMH
 * Description:  Shift Model Handler
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, DSu
 * @version      1.0
 *
 * Created on Aug 7, 2003, 3:07:52 PM
 */
public class ShiftMH {

	/**
	 * Return Shift Assignment for specified ID
	 */
	public static ShiftAssignment getShiftAssignment(ID saID) throws RemoteException, BbmException {
		ScheduleAccessManager saManager = WfmManagerFactory.getScheduleAccessManager();
		return saManager.getShiftAssignmentByID(saID);
	}
}
