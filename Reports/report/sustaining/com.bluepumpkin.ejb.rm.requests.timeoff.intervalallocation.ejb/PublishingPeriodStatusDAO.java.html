package com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb;

import java.util.Collection;
import java.util.Date;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.rm.util.DAOUtil;
import com.bluepumpkin.ejb.rm.util.JdmoBatchExecute;

public class PublishingPeriodStatusDAO {

	private Jdmo dmo = null;

	public PublishingPeriodStatusDAO() {
		dmo = new Jdmo(false);
	}

	public void cleanUp() {
		if (dmo != null) {
			dmo.cleanUp();
			dmo = null;
		}
	}

	public Map<IdDateRange, Integer> getPeriodStatus(Collection<IdDateRange> periods) throws JdmoException {

		insertIntoTempEmployeeDates(periods);

		JdmoQuery jdmoQuery = dmo.createQuery("RM_GET_PUBLISHING_STATUS", Jdmo.STORPROC_QUERY);

		JdmoRowset res = dmo.createRowset(jdmoQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

		Map<IdDateRange, Integer> result = new HashMap<IdDateRange, Integer>();
		while (res.next()) {
			int employeeId = res.getInt("EmployeeID");
			Date start = TimeZoneUtil.toDate(res.getTimestamp("FromDate"));
			Date end = TimeZoneUtil.toDate(res.getTimestamp("ToDate"));
			int status = res.getInt("FullyPublished");

			IdDateRange key = new IdDateRange(employeeId, start, end);
			result.put(key, status);
		}


		return result;
	}

	private void insertIntoTempEmployeeDates(Collection<IdDateRange> periods) throws JdmoException {
		DAOUtil.createTempTable(dmo, "EmployeeDates", "EmployeeId int, FromDate datetime, ToDate datetime");

		JdmoBatchExecute<IdDateRange> batchExecute = new JdmoBatchExecute<IdDateRange>(dmo) {
			@Override
			public void set(Map<Integer, Object> param, IdDateRange value) {
				param.put(1, value.id);
				param.put(2, value.start);
				param.put(3, value.end);
			}
		};

		batchExecute.executeBatch("insert into #EmployeeDates values (?,?,?)", periods);
	}

}
