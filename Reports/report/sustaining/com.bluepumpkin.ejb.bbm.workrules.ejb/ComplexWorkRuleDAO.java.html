package com.bluepumpkin.ejb.bbm.workrules.ejb;

import java.util.Collection;
import java.util.Date;
import java.util.LinkedList;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.common.localization.LocaleContext;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOUtil;
import com.bluepumpkin.ejb.bbm.dao.ManyToOneDependentRelationship;
import com.bluepumpkin.ejb.bbm.empworkrule.ejb.WorkUnitDAO;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit.WorkUnitType;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnitFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectUtil;
import com.bluepumpkin.ejb.bbm.workrules.model.CalendarPeriod;
import com.bluepumpkin.ejb.bbm.workrules.model.CalendarPeriodFieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRuleFieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRuleFieldInfo.Column;
import com.witness.ejb.bbm.copy.ejb.CopyDAO;

public class ComplexWorkRuleDAO extends WorkRuleBaseDAO<ComplexWorkRule> {
	
	CalendarPeriodDAO calendarPeriodDAO;
	WorkUnitDAO workUnitDAO;

	public ComplexWorkRuleDAO() {
		super();
	}
	
	public ComplexWorkRuleDAO(Jdmo dmo) {
		super(dmo);
	}
	
	protected ComplexWorkRule createValueObject() {
		return new ComplexWorkRule();
	}
	
	@Override
	protected void buildDBRelationships() {
		calendarPeriodDAO = new CalendarPeriodDAO(m_dmo);
		workUnitDAO = new WorkUnitDAO(m_dmo);
		
		addRelationship(new ManyToOneDependentRelationship(this, calendarPeriodDAO,
				getFieldInfo().getFieldIndexByName("CALENDARPERIODID"),
				calendarPeriodDAO.getFieldInfo().getFieldIndexByName("ID"),
				ComplexWorkRule.CALENDAR_PERIOD_RELATIONSHIP_KEY));		
		
		addRelationship(new ManyToOneDependentRelationship(this, workUnitDAO,
				getFieldInfo().getFieldIndexByName("WORKUNITID"),
				workUnitDAO.getFieldInfo().getFieldIndexByName("ID"),
				ComplexWorkRule.WORKUNIT_RELATIONSHIP_KEY));	
	}
	
	public Collection<ComplexWorkRule> getObjects(String where) throws BbmFinderException {
		if (!StringUtil.isEmpty(where)) {
			return super.getObjects(where + " AND " + getWhereClause());
		} else return super.getObjects(getWhereClause());
	}

	@Override
	protected String getWhereClause() { return "WORKUNITCOUNTCONSTRAINT <> 6"; }
	
	public Collection<ComplexWorkRule> getComplexWorkRulesForOrgs(Collection colOrgIDs) throws BbmFinderException
	{
			try {
				Collection<ID> orgsPlusParents = new LinkedList<ID>(colOrgIDs);
				
				//Return the work rules associated to the filtered org set - all common orgs or common parent orgs
				//  that also have the same start week offset as the employee's orgs
				orgsPlusParents.addAll(DAOUtil.getParentOrganizationsCollection(colOrgIDs, m_dmo));
				return getObjects(" ORGANIZATIONID IN " + getDMO().createInClause(orgsPlusParents));
			} catch (Exception ex) {
				throw new BbmFinderException(ex);
			}
	}

	@Override
	public void setChildFieldsOnCreate(String relationshipKey,
			ComplexWorkRule parentObject, ValueObjectBase childObject) {
		super.setChildFieldsOnCreate(relationshipKey, parentObject, childObject);
		if (relationshipKey.equals(ComplexWorkRule.CALENDAR_PERIOD_RELATIONSHIP_KEY)) {
			CalendarPeriod cp = (CalendarPeriod) childObject;
			cp.setFieldValue(CalendarPeriodFieldInfo.CALENDARPERIOD_CREATED, new Date());
			cp.setFieldValue(CalendarPeriodFieldInfo.CALENDARPERIOD_CREATEDBY, getCallerPrincipal());
		}
		if (relationshipKey.equals(ComplexWorkRule.WORKUNIT_RELATIONSHIP_KEY)) {
			WorkUnit wu = (WorkUnit) childObject;
			wu.setFieldValue(WorkUnitFieldInfo.WORKUNIT_CREATED, new Date());
			wu.setFieldValue(WorkUnitFieldInfo.WORKUNIT_CREATEDBY, getCallerPrincipal());
		}
	}

	@Override
	public void setChildFieldsOnUpdate(String relationshipKey,
			ComplexWorkRule parentObject, ValueObjectBase childObject) {
		super.setChildFieldsOnUpdate(relationshipKey, parentObject, childObject);
		if (relationshipKey.equals(ComplexWorkRule.CALENDAR_PERIOD_RELATIONSHIP_KEY)) {
			CalendarPeriod cp = (CalendarPeriod) childObject;
			cp.setFieldValue(CalendarPeriodFieldInfo.CALENDARPERIOD_MODIFIED, new Date());
			cp.setFieldValue(CalendarPeriodFieldInfo.CALENDARPERIOD_MODIFIEDBY, getCallerPrincipal());
		}
		if (relationshipKey.equals(ComplexWorkRule.WORKUNIT_RELATIONSHIP_KEY)) {
			WorkUnit wu = (WorkUnit) childObject;
			wu.setFieldValue(WorkUnitFieldInfo.WORKUNIT_MODIFIED, new Date());
			wu.setFieldValue(WorkUnitFieldInfo.WORKUNIT_MODIFIEDBY, getCallerPrincipal());
		}
	}

	public void copyToOrg(Collection<ID> assignmentRuleIDs, ID orgID,
			LocaleContext localeContext, String name) throws BbmCreateException {
		CopyDAO copyDAO = new CopyDAO();
		try {
			Collection<ComplexWorkRule> cwrs = getObjectsByIDs(assignmentRuleIDs);
			for (ComplexWorkRule cwr : cwrs) {
				cwr.setName(copyDAO.getNewNameForCopiedValueObjectWithLengthLimitation(cwr.getName(),
						getFieldInfo().getTableName(), localeContext, ComplexWorkRuleFieldInfo.NAME_MAX_LENGTH));
				cwr.setOrganizationID(orgID);
				cwr.setFieldValue(Column.ID.getIndex(), (Object)null);
				cwr.setFieldValue(Column.SID.getIndex(), (Object)null);
				cwr.setFieldValue(Column.CreatedBy.getIndex(), getCallerPrincipal());
				cwr.setFieldValue(Column.CreatedOn.getIndex(), new Date());
				cwr.setFieldValue(Column.ModifiedOn.getIndex(), (Object)null);
				cwr.setFieldValue(Column.ModifiedBy.getIndex(), (Object)null);
			}
			createObjects(cwrs);
		} catch (BbmFinderException e) {
			throw new BbmCreateException(e);
		}
	}

	public Collection<ID> getIDsForWorkUnitType(WorkUnitType workUnitType) throws BbmFinderException, JdmoException {
		WorkUnitDAO wuDao = new WorkUnitDAO(getDMO());
		FieldInfo wufi = wuDao.getFieldInfo();
		Collection<WorkUnit> wus = wuDao.getObjects(wufi.getFieldName(WorkUnitFieldInfo.WORKUNIT_WORKUNITTYPE) + " = " + JdmoUtil.asSqlLiteral(workUnitType.getDBValue()));
		Collection<ID> workUnitIDs = ValueObjectUtil.getIDFromObjects(wus);
		Collection<ComplexWorkRule> cwrs = getObjects(
				getFieldInfo().getFieldName(ComplexWorkRuleFieldInfo.Column.WorkUnitID.getIndex()) + " IN " + getDMO().createInClause(workUnitIDs));
		return ValueObjectUtil.getIDFromObjects(cwrs);
	}
}
