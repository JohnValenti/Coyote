package com.bluepumpkin.ejb.bbm.workrules.ejb;

import java.util.Collection;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEventShift;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEventShiftFieldInfo;
import com.bluepumpkin.common.datatypes.ID;

public class ShiftEventShiftDAO extends DAOBase<ShiftEventShift> {

	// meta data information
	private static FieldInfo m_fieldInfo = new ShiftEventShiftFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	
	public ShiftEventShiftDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ShiftEventShift createValueObject() {
		return new ShiftEventShift();
	}

	protected Collection<ShiftEventShift> getShiftEventShiftsByShifts(Collection<ID> shiftIDs) throws BbmFinderException {
		return getObjects("SHIFTID IN (" + JdmoUtil.createIDList(shiftIDs) + ")");
	}
}
