	{
		switch (type) {
			case TimeBankFieldInfo.CHILD_TARGETMINS:
				return new TimeBankTargetMinutesDAO(m_dmo);
			case TimeBankFieldInfo.CHILD_INTERPERIOD:
				return new TimeBankInterPeriodDAO(m_dmo);
		}
		return null;
	}
	
	public PaginationPair<TimeBank> getAvailableTimeBanksForEmployees(Collection<ID> orgIDsAndParents,
			Collection<ID> workResourceIDs,
			Date startDate,
			Date endDate,
			boolean isAscendingSort,
			int pageIDSize,
			int pageIndex,
			LinkedHashMap<DBSortField, Comparator> sortFieldMap,
			Collection<ID> filteredIDs) throws BbmFinderException {	
		try {
			StringBuffer whereClause = new StringBuffer(); //is the following query just a start before end and end after start? - Angela
			whereClause.append(" ORGANIZATIONID IN ").append(getDMO().createInClause(orgIDsAndParents)).append(" AND ((")
				.append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_STARTDATE)).append(" <= ").append(JdmoUtil.asSqlLiteral(startDate))
				.append(" AND ").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_ENDDATE) ).append(" >= ").append(JdmoUtil.asSqlLiteral(startDate))
				.append(") OR (").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_STARTDATE)).append(" <= ").append(JdmoUtil.asSqlLiteral(endDate))
				.append(" AND ").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_ENDDATE)).append(" >= ").append(JdmoUtil.asSqlLiteral(endDate))
				.append(") OR (").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_STARTDATE)).append(" >= ").append(JdmoUtil.asSqlLiteral(startDate))
				.append(" AND ").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_ENDDATE)).append(" <= ").append(JdmoUtil.asSqlLiteral(endDate))
				.append(") OR (").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_STARTDATE)).append(" <= ").append(JdmoUtil.asSqlLiteral(startDate))
				.append(" AND ").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_ENDDATE)).append(" >= ").append(JdmoUtil.asSqlLiteral(endDate))
				.append("))");
			//Return the work rules associated to the filtered org set - all common orgs or common parent orgs
			//  that also have the same start week offset as the employee's orgs
			return getCompletePaginationDataSet(whereClause.toString(), isAscendingSort, pageIDSize, pageIndex, sortFieldMap, filteredIDs);
		} catch (Exception ex) {
			throw new BbmFinderException(ex);
		}
	}
	
	@Override
	protected void buildSortQuery(List sortFields, StringBuffer selectQuery, StringBuffer fromQuery,
			StringBuffer whereQuery, StringBuffer groupByQuery, StringBuffer orderByQuery) {
		ArrayList<TimeBankSortField> timeBankSortFields = new ArrayList<TimeBankSortField>(sortFields);
		for (TimeBankSortField sortField : timeBankSortFields) {
			switch(sortField) {
				case ID:
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getIDFieldName());
					break;
				case NAME:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_NAME));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_NAME));
					break;
				case START_DATE:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_STARTDATE));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_STARTDATE));
					break;
				case END_DATE:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_ENDDATE));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_ENDDATE));
					break;
				case TOTAL_TARGET_HOURS:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_TOTALTARGETMINUTES));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_TOTALTARGETMINUTES));
					break;
				case ORGANIZATION:
					selectQuery.append(", (SELECT E.").append(ofi.getFieldName(OrganizationFieldInfo.ORGANIZATION_NAME)).append(
						" FROM ").append(ofi.getTableName()).append(" E WHERE E.").append(ofi.getIDFieldName()).append(
						" = A.").append(getFieldInfo().getFieldName(TimeBankFieldInfo.TIMEBANK_ORGANIZATIONID)).append(
						") AS ").append(organizationSortFieldKey);
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(organizationSortFieldKey);				
					break;
				default: break;
			}
		}
	}
	/**
	 * Returns a complete list of IDs and the first chunk of data to be displayed
	 * @param orgID
	 * @param sortFieldIndices
	 * @param isAscendingSort
	 * @return
	 * @throws BbmFinderException
	 */
	@SuppressWarnings("unchecked")
	public  PaginationPair<TimeBank>  getCompletePaginationDataSetForOrg(ID orgID, boolean isAscendingSort, int pageIDSize,
			int pageIndex, LinkedHashMap<DBSortField, Comparator> sortFieldMap,
			Collection<ID> filteredIDs, String filterByName) throws BbmFinderException {
		StringBuffer whereClause = new StringBuffer(200);
		whereClause.append(" A.ORGANIZATIONID IN (SELECT ID from FN_TF_CF_getParentOrgsIncludingChild('" +
				orgID.toString() + "') )");
		if (filterByName != null && filterByName.length() > 0) {
			whereClause.append(" AND A.NAME LIKE '%").
			append(filterByName).append("%'");
		}
		if(pageIDSize==-1||pageIndex ==-1)
			return getCompleteDataSet(whereClause.toString(), isAscendingSort, sortFieldMap, filteredIDs);
		else
			return getCompletePaginationDataSet(whereClause.toString(), isAscendingSort, pageIDSize, pageIndex, sortFieldMap, filteredIDs);
	}
	
	@SuppressWarnings({ "unchecked", "rawtypes" })
	public PaginationPair<TimeBank> getCompleteDataSet(String where, boolean isAscendingSort,
			LinkedHashMap<DBSortField, Comparator> sortFieldMap,
			Collection<ID> filteredIDs) throws BbmFinderException {
		try {
			ArrayList<ID> retIDs = new ArrayList<ID>();
			ArrayList<TimeBank> retObjects = new ArrayList<TimeBank>();
			StringBuffer selectClause = new StringBuffer("SELECT A." + getFieldInfo().getIDFieldName());
			StringBuffer fromClause = new StringBuffer(" FROM " + getFieldInfo().getTableName() + " A");
			StringBuffer whereClause = new StringBuffer();
			if (where != null && where.length() > 0) {
				whereClause.append(" WHERE " + where);
			}
			StringBuffer groupByClause = new StringBuffer();
			StringBuffer orderByClause = new StringBuffer();
			
			//Concrete DAO will generate the query to return back the data necessary to sort the object IDs based on
			//  the given sort columns
			buildSortQuery(new ArrayList<DBSortField>(sortFieldMap.keySet()), selectClause, fromClause, whereClause,
					groupByClause, orderByClause);
			
			if (orderByClause.length() > 0) {
				orderByClause.insert(0, " ORDER BY ");
			}
			
			boolean isDBSort = true;
			for (DBSortField sortField : sortFieldMap.keySet()) {
				if (sortField.isDBSortable() == false) {
					isDBSort = false;
					break;
				}
			}
			
			//Retrieve the list of sorted IDs
			if (isDBSort) {
				if (isAscendingSort == false) {
					orderByClause.append(" DESC");
				}
				retIDs = getPaginationIDs(selectClause.toString() +
						fromClause.toString() + whereClause.toString() + groupByClause.toString() +
						orderByClause.toString());
			}
			
			if (filteredIDs != null && filteredIDs.size() > 0) {
				for (ID idToFilter : filteredIDs) {
					retIDs.remove(idToFilter);
				}
			}

			if (retIDs.size() > 0) 
				retObjects = new ArrayList<TimeBank>(getObjectsByIDs(retIDs));
			Collections.sort(retObjects, new ValueObjectIDListComparator(retIDs));
			
			return new PaginationPair<TimeBank>(retIDs, retObjects);
		} catch (JdmoException ex) {
			throw new BbmFinderException(ex);
		}
	}
	/**
	 * DAOBase.getCompletePaginationDataSet() uses this when sorting IDs on the DB side
	 * @return
	 * @throws JdmoException 
	 */
	private ArrayList<ID> getPaginationIDs(String query) throws JdmoException {
		ArrayList<ID> retVal = new ArrayList<ID>();
		JdmoRowset r = getDMO().createRowset(query);

		while (r.next()) {
			retVal.add(r.getID("ID"));
		}
		
		return retVal;
	}
