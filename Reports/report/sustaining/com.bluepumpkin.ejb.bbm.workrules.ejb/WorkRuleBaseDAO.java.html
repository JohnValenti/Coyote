package com.bluepumpkin.ejb.bbm.workrules.ejb;

import java.util.Collection;
import java.util.Comparator;
import java.util.Date;
import java.util.LinkedHashMap;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.PaginationPair;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.RichObjectDAO;
import com.bluepumpkin.ejb.bbm.vo.DBSortField;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectRich;
import com.bluepumpkin.ejb.bbm.workresource.model.OrganizationFieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRuleFieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRuleSortField;
import com.bluepumpkin.ejb.bbm.workrules.model.WorkRuleUtil;

public abstract class WorkRuleBaseDAO<T extends ValueObjectRich> extends RichObjectDAO<T> {

	//Making this public (from protected) because the relationship with WorkResourceComplexWorkRuleDAO
	//  requires the field info to get the field indices of the fields in the relationship
	protected static FieldInfo m_fieldInfo = new ComplexWorkRuleFieldInfo();
    public FieldInfo getFieldInfo() { return m_fieldInfo; }
    
	private OrganizationFieldInfo ofi = new OrganizationFieldInfo();
	private final String organizationSortFieldKey = ofi.getTableName() + "_" + ofi.getFieldName(OrganizationFieldInfo.ORGANIZATION_NAME);
	
	public WorkRuleBaseDAO() {
		super();
	}

	public WorkRuleBaseDAO(Jdmo dmo) {
		super(dmo);
	}
	
	public PaginationPair<T> getAvailableWorkRulesForEmployees(Collection<ID> workResourceIDs,
			Date startDate, Date endDate, boolean isAscendingSort, int pageIDSize, int pageIndex,
			LinkedHashMap<DBSortField, Comparator> sortFieldMap, Collection<ID> filteredIDs,String filterByName)
			throws BbmFinderException {
		try {
			//Return the work rules associated to the filtered org set - all common orgs or common parent orgs
			//  that also have the same start week offset as the employee's orgs
			StringBuffer whereClause = new StringBuffer();
			whereClause.append(getWhereClause() + " AND A.ORGANIZATIONID IN " + getDMO().createInClause(
					ComplexWorkRuleUtil.getOwnerOrgIDsForEmployees(workResourceIDs, startDate, endDate, getDMO())));
			if (filterByName != null && filterByName.length() > 0) {
				whereClause.append(" AND A.NAME LIKE '%").
				append(filterByName.replaceAll("'", "''")).append("%'");
			}
			return getCompletePaginationDataSet(whereClause.toString(),
					isAscendingSort, pageIDSize, pageIndex, sortFieldMap, filteredIDs);
		} catch (Exception ex) {
			throw new BbmFinderException(ex);
		}
	}

	/**
	 * In 11SP1, we need to filter out assignment rules that are monthly (have a calendar period of 5,6,7 or 8 which means it has a 28,29,30, or 31 day period, respectively).
	 */
	protected String getMonthlyWorkRuleFilterClause() {
		//QC96068, Team assignment work rules (which has calendarperiodid = null) are not included in list of assignment rule 
		//return " CALENDARPERIODID IN (SELECT ID FROM CALENDARPERIOD WHERE CALENDARUNIT NOT IN (5,6,7,8))";
		return " (CALENDARPERIODID IS NULL OR CALENDARPERIODID IN (SELECT ID FROM CALENDARPERIOD WHERE CALENDARUNIT NOT IN (5,6,7,8)))";
	}
	@Override
	protected void buildSortQuery(List<DBSortField> sortFields, StringBuffer selectQuery,
			StringBuffer fromQuery, StringBuffer whereQuery, StringBuffer groupByQuery, StringBuffer orderByQuery) {
		for (DBSortField sortField : sortFields) {
			switch((ComplexWorkRuleSortField)sortField) {
				case ID:
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getIDFieldName());
					break;
				case NAME:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ComplexWorkRuleFieldInfo.Column.Name.getIndex()));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ComplexWorkRuleFieldInfo.Column.Name.getIndex()));
					break;
				case ORGANIZATION:
					selectQuery.append(", (SELECT E.").append(ofi.getFieldName(OrganizationFieldInfo.ORGANIZATION_NAME)).append(
						" FROM ").append(ofi.getTableName()).append(" E WHERE E.").append(ofi.getIDFieldName()).append(
						" = A.").append(getFieldInfo().getFieldName(ComplexWorkRuleFieldInfo.Column.OrganizationID.getIndex())).append(
						") AS ").append(organizationSortFieldKey);
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(organizationSortFieldKey);				
					break;
				case PRIORITY:
					selectQuery.append(", ").append(getFieldInfo().getFieldName(ComplexWorkRuleFieldInfo.Column.Priority.getIndex()));
					WorkRuleUtil.appendCommaIfPassStringIsNotEmpty(orderByQuery).append(getFieldInfo().getFieldName(ComplexWorkRuleFieldInfo.Column.Priority.getIndex()));						
					break;
				default: break;
			}
		}
		
		if (whereQuery.length() > 0) {
			int insertIdx = 0;
			String whereQueryStr = whereQuery.toString();
			String whereQueryStrTrimmed =  whereQueryStr.trim();
			if (whereQueryStrTrimmed.length() > 6 && "WHERE ".equalsIgnoreCase(whereQueryStrTrimmed.substring(0, 6))) {
				insertIdx = whereQueryStr.toLowerCase().indexOf("where ") + 6;
			}
			whereQuery.insert(insertIdx, "(");
			whereQuery.append(") AND ");
		} else {
			whereQuery.append(" WHERE ");
		}
		whereQuery.append(getWhereClause());
	}
	
	protected Object getPaginationDataFieldValue(JdmoRowset rowData, DBSortField sortField) throws JdmoException {
		ComplexWorkRuleSortField ssf = (ComplexWorkRuleSortField)sortField;
		switch(ssf) {
			case NAME:
				return rowData.getString(getFieldInfo().getFieldName(ComplexWorkRuleFieldInfo.Column.Name.getIndex()));
			case ORGANIZATION:
				return rowData.getString(organizationSortFieldKey);
			case PRIORITY:
				return rowData.getString(getFieldInfo().getFieldName(ComplexWorkRuleFieldInfo.Column.Priority.getIndex()));
			default: return null;
		}
	}
	
	protected abstract String getWhereClause();
}
