package com.bluepumpkin.web.am.adherence.testing;

import java.util.Date;
import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;

/**
 *  Defines the records used for scheduled/actual adherence events.
 */
public class SimpleEvent
	extends com.bluepumpkin.ejb.bbm.activity.model.SimpleEvent
	implements AdherenceEvent 
{
	private Date m_timestamp;

	public SimpleEvent(ID id, int nEventType, String strDesc, Date dateStart, Date dateEnd,
					   ID idActivity, boolean bPaid, boolean bIsWork, Collection colWorkResourceIDs,
					   Date timestamp)				
	{
		super(id, null, nEventType, strDesc, dateStart, dateEnd, idActivity, false, bPaid, bIsWork, colWorkResourceIDs);
		setTimestamp(timestamp);
	}

	public SimpleEvent(ID id, String strDesc, Date dateStart, Date dateEnd,	ID idActivity, 
					   boolean bPaid, boolean bIsWork, Collection colWorkResourceIDs, 
					   int nOverlayPrecedence, Date timestamp)
	{
		super(id, null, strDesc, dateStart, dateEnd, idActivity, false, bPaid, bIsWork, colWorkResourceIDs,
			  nOverlayPrecedence);
		setTimestamp(timestamp);
	}

	public Date getTimestamp() {
		return m_timestamp;
	}

	public void setTimestamp(Date timestamp) {
		m_timestamp = timestamp;
	}
}
