package com.bluepumpkin.ejb.rm.setup.timeoffbid.model;

import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.vo.*;
import com.bluepumpkin.ejb.bbm.workresource.model.Employee;
import com.bluepumpkin.ejb.bbm.cache.CacheUtilBBM;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.requests.common.model.RequestDetailLevel;

public class TimeOffBidder extends ValueObjectBase {
	private static Category LOG = Log.initCategory(TimeOffBidder.class.getName());
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new TimeOffBidderFieldInfo();
	private transient Employee cachedEmployee;
	private transient TimeOffBidderOptMethods timeOffBidderOptMethods;
	private static final String EMPLOYEE_CACHE_ERROR_MSG = "Unable to retrieve Employee Cache. ";

	public static final long DL_BASIC = RequestDetailLevel.DL_BASIC;
	public static final long DL_EMPLOYEE = RequestDetailLevel.DL_EMPLOYEE_SHIFTBIDDER_CHILD;

	public static final int STATUS_NOT_ADDED = 0;
	public static final int STATUS_NOT_IN_BID = -1;

	public static final int SORTBY_NONE = 1;
	public static final int SORTBY_STATUS = 2;
	public static final int SORTBY_SENIORITY = 3;
	public static final int SORTBY_RANK = 4;
	public static final int SORTBY_MAXNUMBEROFPERIODS = 5;
	public static final int SORTBY_NAME = 6;

	public TimeOffBidder() {
		super();
	}

	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ID getId() {
		return getFieldValueID(TimeOffBidderFieldInfo.TIMEOFFBIDDER_ID);
	}

	public void setId(ID val) {
		setFieldValue(TimeOffBidderFieldInfo.TIMEOFFBIDDER_ID, val);
	}

	public ID getTimeOffBidId() {
		return getFieldValueID(TimeOffBidderFieldInfo.TIMEOFFBIDDER_TIMEOFFBIDID);
	}

	public void setTimeOffBidId(ID val) {
		setFieldValue(TimeOffBidderFieldInfo.TIMEOFFBIDDER_TIMEOFFBIDID, val);
	}

	public ID getEmployeeId() {
		return getFieldValueID(TimeOffBidderFieldInfo.TIMEOFFBIDDER_EMPLOYEEID);
	}

	public void setEmployeeId(ID val) {
		setFieldValue(TimeOffBidderFieldInfo.TIMEOFFBIDDER_EMPLOYEEID, val);
	}

	public int getStatus() {
		return getFieldValueInt(TimeOffBidderFieldInfo.TIMEOFFBIDDER_STATUS);
	}

	public void setStatus(int val) {
		setFieldValue(TimeOffBidderFieldInfo.TIMEOFFBIDDER_STATUS, val);
	}

	public int getMaxNumOfPeriods() {
		return getFieldValueInt(TimeOffBidderFieldInfo.TIMEOFFBIDDER_MAXNUMOFPERIODS);
	}

	public void setMaxNumOfPeriods(int val) {
		setFieldValue(TimeOffBidderFieldInfo.TIMEOFFBIDDER_MAXNUMOFPERIODS, val);
	}

	public int getPeriodUnit() {
		return getFieldValueInt(TimeOffBidderFieldInfo.TIMEOFFBIDDER_PERIODUNIT);
	}

	public void setPeriodUnit(int val) {
		setFieldValue(TimeOffBidderFieldInfo.TIMEOFFBIDDER_PERIODUNIT, val);
	}

	public TimeOffBidderOptMethods getOptMethods() {
		if (timeOffBidderOptMethods == null) {
			timeOffBidderOptMethods = new TimeOffBidderOptMethods();
		}
		return timeOffBidderOptMethods;
	}

	/**
	 * 
	 * Helper Class to encapsulate the methods to get employee data
	 * 
	 */
	public class TimeOffBidderOptMethods {

		public Employee getEmployee() {

			if (cachedEmployee != null) {
				return cachedEmployee;
			}

			try {
				cachedEmployee = CacheUtilBBM.getEmployeeByID(getEmployeeId(), null, Employee.DETAIL_LEVEL_EMPLOYEE_BASIC
						| Employee.DETAIL_LEVEL_EMPLOYEE_TIMEBASEDPROPERTY);
			} catch (Exception e) {
				LOG.warn(EMPLOYEE_CACHE_ERROR_MSG + e.getMessage(), e);
			}
			return cachedEmployee;
		}
	}
}
