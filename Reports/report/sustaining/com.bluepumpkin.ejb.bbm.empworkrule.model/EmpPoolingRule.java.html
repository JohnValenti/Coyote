import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.effectivity.TimePeriod;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectEffectivity;

public class EmpPoolingRule extends ValueObjectEffectivity
{
    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new EmpPoolingRuleFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }
    
    public EmpPoolingRule() {
        super();
    }

    public ID getEmployeeID()
    {
        return getFieldValueID(EmpPoolingRuleFieldInfo.EMPPOOLINGRULE_EMPLOYEEID);
    }
    
    public Date getStartTime()
    {
        return getFieldValueDate(EmpPoolingRuleFieldInfo.EMPPOOLINGRULE_STARTTIME);
    }

    public Date getEndTime()
    {
        return getFieldValueDate(EmpPoolingRuleFieldInfo.EMPPOOLINGRULE_ENDTIME);
    }
    
    public int getMinMinutes()
    {
        return getFieldValueInt(EmpPoolingRuleFieldInfo.EMPPOOLINGRULE_MINMINUTES);  
    }
    
    public void setWorkResourceID(ID id)
    {
        setFieldValue(EmpPoolingRuleFieldInfo.EMPPOOLINGRULE_EMPLOYEEID, id);
    }
    
    public void setStartTime(Date dt)
    {
        setFieldValue(EmpPoolingRuleFieldInfo.EMPPOOLINGRULE_STARTTIME, dt);
    }

    public void setEndTime(Date dt)
    {
        setFieldValue(EmpPoolingRuleFieldInfo.EMPPOOLINGRULE_ENDTIME, dt);
    }
    
    public void setMinMinutes(int iValue)
    {
        setFieldValue(EmpPoolingRuleFieldInfo.EMPPOOLINGRULE_MINMINUTES, iValue);  
    }
    
    // clone the assignment object without the ID field
    public TimePeriod cloneTimePeriod()
    {          
    	EmpPoolingRule item = new EmpPoolingRule();
                    
        item.setWorkResourceID(getEmployeeID());
        item.setStartTime(getStartTime());
        item.setEndTime(getEndTime());
        item.setMinMinutes(getMinMinutes());
     
        return item;
    }
    
    ArrayList<ID> m_aOrgs = new ArrayList<ID>();
	public ArrayList<ID> getSecondaryOgs() {return m_aOrgs; }
	public void setSecondaryOgs(Collection<ID> orgs) {m_aOrgs = new ArrayList<ID>(orgs); }
    
}
