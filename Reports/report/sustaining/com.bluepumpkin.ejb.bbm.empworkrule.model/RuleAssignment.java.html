import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

public class RuleAssignment implements java.io.Serializable
{
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	ID m_idWorkRule;
	Date m_dtStart;
	Date m_dtEnd;

	public RuleAssignment(ID idRule, Date dtStart, Date dtEnd) {
		m_idWorkRule = idRule;
		m_dtStart = dtStart;
		m_dtEnd = dtEnd;
	}

	public ID getRuleID() { return m_idWorkRule; }
	public Date getStartDate() { return m_dtStart; }
	public Date getEndDate() { return m_dtEnd; }
}
