package com.bluepumpkin.ejb.bbm.empworkrule.model;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

import com.bluepumpkin.common.datatypes.DayOfWeek;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeOfDay;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPattern;

public class WorkUnit extends ValueObjectRich {
	private static final long serialVersionUID = 1L;

    private static FieldInfo m_fieldInfo = new WorkUnitFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }
    
    public static final String RELATIONSHIPKEY_SHIFTPATTERN = "RELATIONSHIPKEY_SHIFTPATTERN";
    public static final String RELATIONSHIPKEY_TIMEOFDAY = "RELATIONSHIPKEY_TIMEOFDAY";
    public static final String RELATIONSHIPKEY_ACTIVITY = "RELATIONSHIPKEY_WORKUNITACTIVITY";
    
    public enum WorkUnitType {
    	Shift(0), 
		AnyDay(1), 
		DayOfWeek(2), 
		Hours(3), 
		FullWeekend(4), 
		PartialWeekend(5), 
		Weekend(6), 
		Rotation(7), 
		ConsecutiveDay(8), 
		ConsecutiveDayOfWeek(9), 
		ConsecutiveShift(10), 
		ConsecutiveFullWeekend(11),	
		ConsecutivePartialWeekend(12), 
		ConsecutiveDayOff(13), 
		ConsecutiveDayOfWeekOff(14), 
		ConsecutiveFullWeekendOff(15),	
		ConsecutivePartialWeekendOff(16), 
		MustStart(17),
		ShiftEvent(18),
		MustStartBetween(19),
		MustEnd(20),
		Teams(21),
		SupervisorTeams(22),
		LeadTeams(23),
		MustEndBetween(24),
		NumSeats(25),
		MinShiftSpacing(26),
		MinShiftEventSpacing(27),
		Dollars(28),
		Activity(29),
		Invalid(30)
		;
    	
    	private int dbValue;
		private WorkUnitType(int dbValue) {
    		this.dbValue = dbValue;
    	}
		public int getDBValue() {
			return dbValue;
		}
		
		public static WorkUnitType fromDBValue(int dbValue) {
			for (WorkUnitType wut : WorkUnitType.values()) {
				if (wut.dbValue == dbValue) return wut;
			}
			throw new IllegalArgumentException("Unknown WorkUnitType DB Value of " + dbValue);
		}

		/**
		 * A Set of the WorkUnitTypes that do not have priority
		 */
		public static final Set<WorkUnitType> withoutPriority;
		static {
			HashSet<WorkUnitType> types = new HashSet<WorkUnitType>();
			types.add(Teams);
			types.add(SupervisorTeams);
			types.add(LeadTeams);
			types.add(MustStart);
			types.add(MustStartBetween);
			withoutPriority = Collections.unmodifiableSet(types);
		}
    	
    }
    public WorkUnitType getWorkUnitType() {
    	return WorkUnitType.fromDBValue(getFieldValueInt(WorkUnitFieldInfo.WORKUNIT_WORKUNITTYPE));
    }
    
    public void setWorkUnitType(WorkUnitType wut) {
    	this.setFieldValue(WorkUnitFieldInfo.WORKUNIT_WORKUNITTYPE, wut.getDBValue());
    }
    
    public TimePeriodOfDay getTimePeriodOfDay() {
    	@SuppressWarnings("unchecked")
		Collection<TimePeriodOfDay> retCol = getRelatedObjects(RELATIONSHIPKEY_TIMEOFDAY);
    	if (retCol.isEmpty()) return null;
    	return retCol.iterator().next();
    }
    
    public void setTimePeriodOfDay(TimePeriodOfDay timePeriod) {
   		setFieldValue(WorkUnitFieldInfo.WORKUNIT_TIMEOFDAYID, timePeriod != null ? timePeriod.getID() : null);
    	setRelatedObjects(RELATIONSHIPKEY_TIMEOFDAY, timePeriod == null ? Collections.emptyList() : Collections.singleton(timePeriod));
    }
    
    public TimeOfDay getFromTime() {
    	int increments = getFieldValueInt(WorkUnitFieldInfo.WORKUNIT_FROMTIME);
    	return new TimeOfDay(increments * Trace.INTERVAL);
    }
    
    public void setFromTime(TimeOfDay tod) {
    	setFieldValue(WorkUnitFieldInfo.WORKUNIT_FROMTIME, tod.getMinutesSinceMindnight() / Trace.INTERVAL);
    }
    
    public TimeOfDay getToTime() {
    	int increments = getFieldValueInt(WorkUnitFieldInfo.WORKUNIT_TOTIME);
    	return new TimeOfDay(increments * Trace.INTERVAL);
    }
    
    public void setToTime(TimeOfDay tod) {
    	setFieldValue(WorkUnitFieldInfo.WORKUNIT_TOTIME, tod.getMinutesSinceMindnight() / Trace.INTERVAL);
    }
    
    public enum TimeType {
    	DBTimePeriod,
    	StartBetween(0),
    	EndBetween(1);
    	
    	private Integer dbVal;
    	TimeType() {
    		this(null);
    	}
    	TimeType(Integer dbVal) {
    		this.dbVal = dbVal;
    	}
    	
    	public Integer getDBVal() {
    		return dbVal;
    	}
    }
    
    public TimeType getTimeType() {
    	if (getTimePeriodOfDay() != null) {
    		return TimeType.DBTimePeriod;
    	}
    	int timeTypeVal = getFieldValueInt(WorkUnitFieldInfo.WORKUNIT_TIMETYPE);
    	for (TimeType tt : TimeType.values()) {
    		Integer dbVal = tt.getDBVal();
    		if (dbVal != null) {
    			if (dbVal == timeTypeVal) { 
    				return tt;
    			}
    		}
    	}
    	throw new IllegalStateException("Unknown time type for assignment rule id(" + getID() + "): " + timeTypeVal);
    }
    
    public void setTimeType(TimeType timeType) {
    	Integer dbVal = timeType.getDBVal();
    	if (dbVal != null) setFieldValue(WorkUnitFieldInfo.WORKUNIT_TIMETYPE, dbVal);
    }
    
    /**
     * Legacy code used the following enumeration to identify the time of day:
     * <code>
     * public static final int PARTOFDAY_ANYTIME = 0;
	 * public static final int PARTOFDAY_MORNING = 1;
	 * public static final int PARTOFDAY_AFTERNOON = 2;
	 * public static final int PARTOFDAY_EVENING = 3;
	 * public static final int PARTOFDAY_NIGHT = 4;
	 * </code>
	 * If the value of the TimeOfDay does not match one of the legacy values,
	 * an oldID value of -1 should be used.
	 * 
	 * @Deprecated in favor of getTimePeriodOfDay
     */
    @Deprecated
    public int getTimeOfDayConstant() {
    	String value = this.getFormattedFieldValue(WorkUnitFieldInfo.WORKUNIT_TIMEOFDAYID);
        /* DB IDs of built in times
      	TIMEOFDAY_ID001	Morning
		TIMEOFDAY_ID002	Afternoon
		TIMEOFDAY_ID003	Evening
		TIMEOFDAY_ID004	Night
		TIMEOFDAY_ID005	Anytime
         */
    	/* map to workruleparameter 
			  public static final int PARTOFDAY_ANYTIME = 0;
			  public static final int PARTOFDAY_MORNING = 1;
			  public static final int PARTOFDAY_AFTERNOON = 2;
			  public static final int PARTOFDAY_EVENING = 3;
			  public static final int PARTOFDAY_NIGHT = 4;
    	 */
    	int num = -1;
    	if (value != null && value.length() > 0) {
    		if (value.equals("TIMEOFDAY_ID001")) num = 1;
    		if (value.equals("TIMEOFDAY_ID002")) num = 2;
    		if (value.equals("TIMEOFDAY_ID003")) num = 3;
    		if (value.equals("TIMEOFDAY_ID004")) num = 4;
    		if (value.equals("TIMEOFDAY_ID005")) num = 0;
    	}
    	return num;
    }

    //This value is meaningful only when work unit type is day of week - represents the day of the week with Monday 1 to Sunday 7
    public DayOfWeek getDayOfWeek() {
    	return DayOfWeek.from0SundayCount(this.getFieldValueInt(WorkUnitFieldInfo.WORKUNIT_DAYOFWEEK));
    }
    
    public void setDayOfWeek(DayOfWeek dow) {
    	this.setFieldValue(WorkUnitFieldInfo.WORKUNIT_DAYOFWEEK, dow.get0SundayCount());
    }
    
    public ID getShiftID() {
    	return this.getFieldValueID(WorkUnitFieldInfo.WORKUNIT_SHIFTID);
    }
    
    public void setShiftID(ID id) {
    	this.setFieldValue(WorkUnitFieldInfo.WORKUNIT_SHIFTID, id);
    }
    
    public ID getShiftEventID() {
    	return this.getFieldValueID(WorkUnitFieldInfo.WORKUNIT_SHIFTEVENTID);
    }
    
    public void setShiftEventID(ID id) {
    	this.setFieldValue(WorkUnitFieldInfo.WORKUNIT_SHIFTEVENTID, id);
    }
    
	
	@SuppressWarnings("unchecked")
	public List<ShiftPattern> getShiftPatterns() {
		return new ArrayList<ShiftPattern>(getRelatedObjects(RELATIONSHIPKEY_SHIFTPATTERN));
	}

	public void setShiftPatterns(List<ShiftPattern> shiftPatterns) {
		setRelatedObjects(RELATIONSHIPKEY_SHIFTPATTERN, shiftPatterns);
	}

	@SuppressWarnings("unchecked")
	public List<Activity> getActivities() {
		return new ArrayList<Activity>(getRelatedObjects(RELATIONSHIPKEY_ACTIVITY));
	}
	
	public void setActivities(List<Activity> activities) { 
		setRelatedObjects(RELATIONSHIPKEY_ACTIVITY, activities);
	}
}
