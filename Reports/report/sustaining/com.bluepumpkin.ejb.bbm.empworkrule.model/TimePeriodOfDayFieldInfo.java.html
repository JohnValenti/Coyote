package com.bluepumpkin.ejb.bbm.empworkrule.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class TimePeriodOfDayFieldInfo extends FieldInfo {
	private static final long serialVersionUID = 1L;

	public enum Column {
		ID("ID", PrimitiveTypes.ID),
		Name("NAME", PrimitiveTypes.STRING),
		FromTime("FROMTIME", PrimitiveTypes.INT),
		ToTime("TOTIME", PrimitiveTypes.INT),
		Created("CREATED", PrimitiveTypes.DATE),
		CreatedBy("CREATEDBY", PrimitiveTypes.STRING),
		Modified("MODIFIED", PrimitiveTypes.DATE),
		ModifiedBy("MODIFIEDBY", PrimitiveTypes.STRING)
		;
		
		// Static block to set field array indexes
		static {
			int ctr = 0;
			for (Column fie : Column.values()) fie.index = ctr++;
		}

		// Column Properties
		private int index;
		private FieldInfoEntry fie;

		// Constructor
		private Column(String name, int type) {
			this.fie = new FieldInfoEntry(name, type);
		}
		
		public FieldInfoEntry getFieldInfoEntry() {
			return this.fie;
		}
		public int getIndex() {
			return index;
		}
	}

	private static final FieldInfoEntry[] fieldInfoEntries;
	static {
		int ctr = 0;
		fieldInfoEntries = new FieldInfoEntry[Column.values().length];
		for (Column c : Column.values()) {
			fieldInfoEntries[ctr++] = c.getFieldInfoEntry();
		}
	}
	
	@Override
	public String getTableName() {
		return "TIMEOFDAY";
	}

	@Override
	public int getIDFieldIndex() {
		return Column.ID.getIndex();
	}

	@Override
	public FieldInfoEntry[] getFieldInfo() {
		return fieldInfoEntries; 
	}

}
