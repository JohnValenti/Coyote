package com.bluepumpkin.ejb.bbm.empworkrule.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class WorkUnitActivityFieldInfo extends FieldInfo {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;

	public static enum Column {
		ID(new FieldInfoEntry("ID", PrimitiveTypes.ID)),
		WorkUnitID(new FieldInfoEntry("WORKUNITID", PrimitiveTypes.ID)),
		ActivityID(new FieldInfoEntry("ACTIVITYID", PrimitiveTypes.ID))
		;
		private int index;
		private FieldInfoEntry fie;
		private Column(FieldInfoEntry fie) {
			this.fie = fie;
		}
		public FieldInfoEntry getFieldInfoEntry() {
			return this.fie;
		}
		public int getIndex() {
			return index;
		}
		static {
			int ctr = 0;
			for (Column fie : Column.values()) fie.index = ctr++;
		}
	};
	
	private static final FieldInfoEntry[] fieldInfoEntries = new FieldInfoEntry[Column.values().length];
	static {
		for (Column c : Column.values()) {
			fieldInfoEntries[c.getIndex()] = c.getFieldInfoEntry();
		}
	}
	
	@Override
	public FieldInfoEntry[] getFieldInfo() {
		return fieldInfoEntries;
	}

	@Override
	public int getIDFieldIndex() {
		return Column.ID.getIndex();
	}

	@Override
	public String getTableName() {
		return "WORKUNITACTIVITY";
	}

	@Override
	public int getParentIDFieldIndex() {
		return Column.WorkUnitID.getIndex();
	}

}
