package com.bluepumpkin.ejb.bbm.empworkrule.model;

import java.util.ResourceBundle;

import com.bluepumpkin.common.datatypes.TimeOfDay;
import com.bluepumpkin.ejb.bbm.empworkrule.model.TimePeriodOfDayFieldInfo.Column;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.common.localization.Localizer;


public class TimePeriodOfDay extends ValueObjectBase {
	private static final long serialVersionUID = 2L;

	public TimePeriodOfDay() {
		super();
	}

	public TimePeriodOfDay(TimeOfDay start, TimeOfDay end) {
		super();
		this.setStartTime(start);
		this.setEndTime(end);
	}

	private void setEndTime(TimeOfDay end) {
		setFieldValue(Column.ToTime.getIndex(), end.getMinutesSinceMindnight());
	}

	private void setStartTime(TimeOfDay start) {
		setFieldValue(Column.FromTime.getIndex(), start.getMinutesSinceMindnight());
	}

	public TimeOfDay getStartTime() {
		return new TimeOfDay(getFieldValueInt(Column.FromTime.getIndex()));
	}
	
	public TimeOfDay getEndTime() {
		return new TimeOfDay(getFieldValueInt(Column.ToTime.getIndex()));
	}
	
	@Override
	protected FieldInfo getFieldInfo() {
		return new TimePeriodOfDayFieldInfo();
	}
	


	public String getName(Localizer localizer) {
		int IDindex = Column.ID.getIndex();
		String key = "TIMEOFDAY_NAME_" + getFormattedFieldValue(IDindex);
		String bundleName = "com.bluepumpkin.ejb.bbm.l10n.TimeOfDay";
		ResourceBundle bundle = ResourceBundle.getBundle(bundleName, localizer.getLocaleContext().getLanguageLocale());
	   	return localizer.i18n(bundle, key);
	}

}
