package com.bluepumpkin.ejb.bbm.empworkrule.model;

/**
 * Title:        Blue Pumpkin Software Basic Business Model

 * Company:      Blue Pumpkin Software, Inc
 * @author       lixin zhang

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.ejb.bbm.effectivity.TimePeriod;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectEffectivity;
import com.bluepumpkin.ejb.bbm.workresource.model.WRTimeProperty;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPattern;

public class WorkResourceWorkPattern extends ValueObjectEffectivity implements WRTimeProperty {
	private static final long serialVersionUID = 1L;

	private static FieldInfo m_fieldInfo = new WorkResourceWorkPatternFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	private ID workPatternSID = null;
	private ShiftPattern shiftPattern;
	private LocalDate localStartTime;
	private LocalDate localEndTime;

	// This is used to keep track of if the preference field has modified
	private boolean isPreferenceModified = false;

	public WorkResourceWorkPattern() {
		super();
	}

	public ID getID() {
		// deid
		Object id = getFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_ID);
		if (id != null) {
			return new ID(id.toString());
		}
		return null;
	}

	public ID getWorkResourceID() {
		return getFieldValueID(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_WORKRESOURCEID);
	}

	public ID getWorkPatternID() {
		String str = getFormattedFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_WORKPATTERNID);
		if (str != null) {
			return new ID(str);
		}
		return null;
	}

	public ID getWorkPatternSID() {
		return workPatternSID;
	}

	public int getPreference() {
		return getFieldValueInt(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_PREFERENCE);
	}

	public Date getStartTime() {
		return getFieldValueDate(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_STARTTIME);
	}

	public Date getEndTime() {
		return getFieldValueDate(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_ENDTIME);
	}

	public void setWorkResourceID(ID id) {
		setFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_WORKRESOURCEID, id);
	}

	public void setWorkPatternID(ID id) {
		if (id != null)
			setFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_WORKPATTERNID, id.toString());
	}

	public void setWorkPatternSID(ID id) {
		workPatternSID = id;
	}

	public void setPreference(int value) {
		setFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_PREFERENCE, value);
	}

	public void setStartTime(Date dt) {
		setFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_STARTTIME, dt);
	}

	public void setEndTime(Date dt) {
		setFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_ENDTIME, dt);
	}

	public void setLocalStartTime(LocalDate startTime) {
		this.localStartTime = startTime;
	}

	public LocalDate getLocalStartTime() {
		return localStartTime;
	}

	public void setLocalEndTime(LocalDate endTime) {
		this.localEndTime = endTime;
	}

	public LocalDate getLocalEndTime() {
		return localEndTime;
	}

	public Date getCreatedDate() {
		return getFieldValueDate(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_CREATED);
	}

	public void setCreatedDate(Date createdDate) {
		setFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_CREATED, createdDate);
	}

	public String getCreatedBy() {
		return getFormattedFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_CREATEDBY);
	}

	public void setCreatedBy(String createdBy) {
		setFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_CREATEDBY, createdBy);
	}

	public Date getModifiedDate() {
		return getFieldValueDate(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_MODIFIED);
	}

	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_MODIFIED, modifiedDate);
	}

	public String getModifiedBy() {
		return getFormattedFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_MODIFIEDBY);
	}

	public void setModifiedBy(String modifiedBy) {
		setFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_MODIFIEDBY, modifiedBy);
	}

	// clone the assignment object without the ID field
	public TimePeriod cloneTimePeriod() {
		WorkResourceWorkPattern item = new WorkResourceWorkPattern();

		item.setWorkResourceID(getWorkResourceID());
		item.setWorkPatternID(getWorkPatternID());
		item.setStartTime(getStartTime());
		item.setEndTime(getEndTime());
		item.setPreference(getPreference());

		return item;
	}

	public boolean effectivityValueEqual(Object obj) {
		if (obj != null) {
			WorkResourceWorkPattern item = (WorkResourceWorkPattern) obj;

			// SID must be there is string id not there
			if (getFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_WORKPATTERNID) == null) {
				return getWorkPatternSID().equals(item.getWorkPatternSID()) && getPreference() == item.getPreference()
						&& isActive() == item.isActive();
			} else {
				return getFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_WORKPATTERNID).equals(
						item.getFieldValue(WorkResourceWorkPatternFieldInfo.WRKRESWORKPATTERN_WORKPATTERNID))
						&& getPreference() == item.getPreference() && isActive() == item.isActive();
			}
		} else {
			return false;
		}
	}

	public void setWorkPattern(ShiftPattern shiftPattern) {
		this.shiftPattern = shiftPattern;
	}

	public ShiftPattern getWorkPattern() {
		return shiftPattern;
	}

	public boolean isPreferenceModified() {
		return isPreferenceModified;
	}

	public void setPreferenceModified(boolean isPreferenceModified) {
		this.isPreferenceModified = isPreferenceModified;
	}
}
