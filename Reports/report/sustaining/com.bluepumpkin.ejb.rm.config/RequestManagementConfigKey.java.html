package com.bluepumpkin.ejb.rm.config;

import com.bluepumpkin.common.config.ConfigManager;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;

// This class is a slow re-factor away from foundation
// so there is less coupling to that code base.
public final class RequestManagementConfigKey {

	public static final String REQUEST_MANAGEMENT = "bluepumpkin/rm/";

	public static final String RM = "verint/rm/";

	public static final String RM_NETSTAFFING_VALIDATION_ENABLED = REQUEST_MANAGEMENT + "enableNetStaffingValidation";

	public static final String RM_MANUALLY_APPROVE_VTO = RM + "ManuallyApproveVTO";
	public static final String RM_VIEW_TENTATIVE_ALERT_APPROVE_TOB = RM + "ViewTentativeAlertTOB";
	public static final String RM_UPDATE_BIDDABLE_SCHEDULE = RM + "UpdateBiddableSchedule";
	public static final String RM_SHIFT_BIDDING_DISABLE_AUDIT_TRAIL_SCAN = RM + "ShiftBiddingDisableAuditTrailScan";

	private RequestManagementConfigKey() {

	}

	public static boolean getShiftBiddingDisableAuditTrailScan() {
		return tryGetBooleanValue(RM_SHIFT_BIDDING_DISABLE_AUDIT_TRAIL_SCAN, false);
	}

	public static boolean getUpdateBiddableSchedule() {
		return tryGetBooleanValue(RM_UPDATE_BIDDABLE_SCHEDULE, true);
	}

	public static boolean getAllowManualApprovalForAdvancedVto() {
		return tryGetBooleanValue(RM_MANUALLY_APPROVE_VTO, false);
	}

	public static boolean getAllowViewTentativeAlertForApproveMultiTOB() {
		return tryGetBooleanValue(RM_VIEW_TENTATIVE_ALERT_APPROVE_TOB, false);
	}

	//a value of zero should be treated as no limit. The default is to cap it at a maximum of 15 pages
	public static int getMaxPages() {
		return tryGetRmIntegerValue("maxPages", 15);
	}
	// 
	public static int getGroupApprovalBatchSize() {
		String key = RM + "groupApprovalBatchSize"; 
		int defaultValue = 5;
		int maxDefaultValue = 20;
		try {
			int val = BbmManagerFactory.getDBConfigManager().getIntValue(key);
			if (val == ConfigManager.NOVALUE) {
				return defaultValue;
			} else if (val >= maxDefaultValue) {
				return maxDefaultValue;
			} else {
				return val;
			}
		} catch (Exception ex) { // NOSONAR
			return defaultValue;
		}

	}

	public static boolean tryGetBooleanValue(String key, boolean defaultValue) {
		try {
			String value = BbmManagerFactory.getDBConfigManager().getValue(key);
			if (value == null || value.isEmpty()) {
				return defaultValue;
			}
			return "true".equals(value);
		} catch (Exception ex) { // NOSONAR
			return defaultValue;
		}
	}

	public static int tryGetRmIntegerValue(String key, int defaultValue) {
		return tryGetIntegerValue(REQUEST_MANAGEMENT + key, defaultValue);
	}

	public static int tryGetIntegerValue(String key, int defaultValue) {
		try {
			int val = BbmManagerFactory.getDBConfigManager().getIntValue(key);
			return val == ConfigManager.NOVALUE ? defaultValue : val;
		} catch (Exception ex) { // NOSONAR
			return defaultValue;
		}
	}

}
