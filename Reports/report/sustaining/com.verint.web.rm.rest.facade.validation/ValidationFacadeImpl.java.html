package com.verint.web.rm.rest.facade.validation;

import java.util.Collection;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.setup.validation.model.OrgValidation;
import com.bluepumpkin.ejb.rm.setup.validation.model.ValidationRule;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.setup.validation.ValidationModelHandler;
import com.bluepumpkin.web.rm.validation.ValidationResources;
import com.bluepumpkin.web.rm.validation.ValidationRulesMap;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rm.rest.entity.validation.ValidationRuleEntity;
import com.verint.web.rm.rest.entity.validation.ValidationRuleParameterEntity;
import com.verint.web.rm.rest.entity.validation.ValidationRulesResponseEntity;
import com.verint.web.rm.rest.facade.OrganizationSettingsFacadeImpl;
import com.verint.web.rm.rest.util.RmUtils;
import com.witness.web.uif.system.RequestContext;

public class ValidationFacadeImpl implements IValidationFacade {
	private static final Category LOG = Log.initCategory(OrganizationSettingsFacadeImpl.class.getName());

	/**
	 * Gets the RM validation settings for the employee.
	 */
	@Override
	public ValidationRulesResponseEntity getValidationSettingsForEmployee(RequestContext context, ID empID)
			throws WFOException {
		try {
			Localizer localizer = context.getLocalizer();
			ID orgID = OrganizationModelHandler.getEmployeeOrgID(context, empID);
			ValidationRulesMap rulesMap = ValidationRulesMap.getInstance();

			ValidationRulesResponseEntity result = new ValidationRulesResponseEntity();
			result.setId(RmUtils.getString(orgID));
			List<ValidationRuleEntity> resultRules = result.getRules();

			Collection<ValidationRule> rules = ValidationModelHandler.getEffectiveOrgValidationRules(orgID);
			for (ValidationRule rule : rules) {
				ValidationRuleEntity entity = new ValidationRuleEntity();
				entity.setId(RmUtils.getString(rule.getID()));
				entity.setKey(rule.getKey());
				entity.setRequestType(rule.getRequestType());

				OrgValidation orgValidation = rule.getOrgValidation();
				entity.setEnabled(orgValidation.getEnabled());
				entity.setActivityID(orgValidation.getActivityID());

				addRuleParameters(rule, entity.getParameters());

				ValidationResources resources = rulesMap.getValResource(rule.getKey());
				if (resources != null) {
					entity.setErrorMessage(resources.getErrorMessage(localizer));
					entity.setIconPath(resources.getIconPath(localizer));
					entity.setMessage(resources.getMessage(localizer));
				}
				resultRules.add(entity);
			}
			return result;
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(context, ex, LOG);
		}
	}

	@SuppressWarnings("rawtypes")
	private void addRuleParameters(ValidationRule rule, List<ValidationRuleParameterEntity> target) {
		Map parameters = rule.getParamMap();
		if (parameters == null) {
			return;
		}
		for (Object key : parameters.keySet()) {
			Object value = parameters.get(key);
			ValidationRuleParameterEntity entity = new ValidationRuleParameterEntity();
			entity.setKey(key.toString());
			if (value != null && value instanceof Integer) {
				entity.setValue((Integer) value);
			}
			target.add(entity);
		}
	}
}
