package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Comparator;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.ejb.bbm.timeoffcalculator.DailyHoursBuckets;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

/**
 * Represents a Time Off choice.
 */
public class TOChoice extends ValueObjectNode implements Validatable {


	
	public static final Comparator<TOChoice> USER_RANK_COMPARATOR = new Comparator<TOChoice>() {
	
		@Override
		public int compare(TOChoice o1, TOChoice o2) {
			return RmUtil.integerCompare(o1.getUserRank(), o2.getUserRank());
		}
	};
	
	
    private static FieldInfo m_fieldInfo = new TOChoiceFieldInfo();

	/**
	 * @see {@link DailyHoursBuckets#DAILYHOURSBUCKETS_PERIOD_SCHED DailyHoursBuckets}
	 * for valid states:
	 * 1=in scheduled period
	 * 2=overlaps scheduled and unscheduled periods.
	 * 3=in unscheduled period
	 */
    private int m_periodFlag = 0;

	private Collection<ValidationResult> m_validationResults = null;

    private float m_length = 0.0f;
	private final float m_request_duration =0.0f;

    /**
     * flag to remember if the lenght has been set.  If set, it need not be recomputed
     * again.
     */
    private boolean m_lengthSet;
    
    /*
     *  computed hours for reporting back to user how time was calculated
     */
   
	
	public TOChoice() {
		super();
		
	}

	@Override
	protected FieldInfo getFieldInfo() {
            return m_fieldInfo;
        }

	// getters
	public ID getTimeOffRequestID() {
            return getFieldValueID(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_I_TIMEOFFREQUESTID);
	}

	public int getRank() {
            return getFieldValueInt(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_N_RANK);
	}

	// TBD Post 15.1.0 RM_AA_PSR
	public int getGroupID() {
		return 0;
	}

	public Date getStartDate() {
            return getFieldValueDate(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_D_STARTTIME);
	}

	public Date getEndDate() {
            return getFieldValueDate(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_D_ENDTIME);
	}

    /**
     * @return length in hours (as a float).
     */
    public float getLength() {
        return m_length;
    }

	public boolean isZeroLength() {
		return 0 == (int) (m_length * 60);
	}

    public void setLength(float len) {
        m_length = len;
        m_lengthSet = true;
    }

    /**
     * @return
     */
    public boolean isLengthSet() {
        return m_lengthSet;
    }

	// setters
	public void setTimeOffRequestID(ID pTimeOffRequestID) {
            setFieldValue(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_I_TIMEOFFREQUESTID, pTimeOffRequestID);
	}

	public void setRank(int pSerialNumber) {
            setFieldValue(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_N_RANK, pSerialNumber);
	}



	public void setStartDate(Date pStartDate) {
            setFieldValue(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_D_STARTTIME, pStartDate);
	}

	public void setEndDate(Date pEndDate) {
            setFieldValue(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_D_ENDTIME, pEndDate);
	}

    @Override
	public Collection<ValidationResult> getValidationResults(boolean entireBranch) {
        if (m_validationResults == null) {
			m_validationResults = new ArrayList<ValidationResult>();
		}
        return m_validationResults;
    }

    @Override
	public void addValidationResult(ValidationResult pValidationResult) {
        getValidationResults(true).add(pValidationResult);
	}
  
    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validatable#clearValidationResults()
     */
    @Override
	public void clearValidationResults(boolean entireBranch) {
    	if (m_validationResults != null) {
			m_validationResults.clear();
		}
    }
       /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validatable#getValidationCache()
     */
    @Override
	public ValidationCache getValidationCache()
    {
        throw new UnsupportedOperationException();
    }

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validatable#getValidationTimeRanges()
     */
    @Override
	public Collection getValidationTimeRanges()
    {
        return Collections.singleton(new TimeRange(getStartDate(), getEndDate()));
    }

    /**
     * Set flag to denote which of 3 periods the time range falls into.
     * @see {@link com.bluepumpkin.ejb.bbm.timeoffcalculator.DailyHoursBuckets#DAILYHOURSBUCKETS_PERIOD_SCHED
     * DailyHoursBuckets} for valid states:
     * @param caseSchedUnsched
     */
    public void setPeriodFlag(int periodFlag) {
      m_periodFlag = periodFlag;
    }

    /**
     * @return a flag that denotes which of 3 periods the time range falls into.
     * @see {@link com.bluepumpkin.ejb.bbm.timeoffcalculator.DailyHoursBuckets#DAILYHOURSBUCKETS_PERIOD_SCHED
     * DailyHoursBuckets} for valid states:
     */
    public int getPeriodFlag() {
      return m_periodFlag;
    }


    public int getUserRank() {
        return getFieldValueInt(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_N_USERRANK);
    }

    public void setUserRank(int pSerialNumber) {
        setFieldValue(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_N_USERRANK, pSerialNumber);
    }

    public boolean isWaitlist() {
        return getFieldValueBoolean(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_B_ISWAITLIST);
    }

    public void setIsWaitlist(boolean bValue) {
        setFieldValue(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_B_ISWAITLIST, bValue);
    }

	public boolean isApproved() {
		return getFieldValueBoolean(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_B_APPROVED);
	}

	public void setIsApproved(boolean bValue) {
		setFieldValue(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_B_APPROVED, bValue);
	}
	
	public ID getReferenceScheduleID() {
		return getFieldValueID(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_I_REFERENCESPID);
	}
    
	public void setReferenceScheduleID(ID id) {
		setFieldValue(TOChoiceFieldInfo.TIMEOFFREQUESTCHOICE_I_REFERENCESPID, id);
	}

    /**
     * Returns HoursPerDay info associated with this TOChoice.
     *
     * Note:
     * @return
     */
    public TOHoursPerDay getHoursPerDay() {
        // if m_TOHoursPerDay Cache is uninitialized, initialize.
        if (m_TOHoursPerDayCache == null) {
            Collection toHoursPerDayCol = getChildObjects(TOChoiceFieldInfo.TOHOURSPERDAY_CHILD_TYPE);
            if (toHoursPerDayCol != null && !toHoursPerDayCol.isEmpty()) {
                m_TOHoursPerDayCache = (TOHoursPerDay) toHoursPerDayCol.iterator().next();
            }
        }
        return m_TOHoursPerDayCache;
    }
    /**
     * Replaces existing HoursPerDay with the new one.
     *
     * @param HoursPerDay
     */
    public void setHoursPerDay(TOHoursPerDay HoursPerDay) {
        // clear existing choices (in this valueObject and set flag for removal from database).
        deleteAllChildren(TOChoiceFieldInfo.TOHOURSPERDAY_CHILD_TYPE);
        if (HoursPerDay != null) {
            createChildObject(TOChoiceFieldInfo.TOHOURSPERDAY_CHILD_TYPE, HoursPerDay);
        }
        // invalidate cache as the m_TOHoursPerDay  has changed.
        m_TOHoursPerDayCache = null;
        //addDetailLevel(DL_TIMEOFF_HoursPerDay);
    }
       public void addDetailLevel(long detailLevel) {
        m_detailLevel |= detailLevel;
    }
	
	private long m_detailLevel =0;
	private transient TOHoursPerDay m_TOHoursPerDayCache;
		
	
	
	 /**
     * Boolean set if the period is unpublished BEFORE the last published period
     *
     * This request will need to be calculated using the defaults since it will have zero hours.
     */
	
	private boolean isInUnpublishedGap = false;
	
	/**
	 * @return the isInUnpublishedGap
	 */
	public boolean getIsInUnpublishedGap() {
		return isInUnpublishedGap;
	}

	/**
	 * @param isInUnpublishedGap the isInUnpublishedGap to set
	 */
	public void setInUnpublishedGap(boolean aIsInUnpublishedGap) {
		this.isInUnpublishedGap = aIsInUnpublishedGap;
	}


	
	

}
