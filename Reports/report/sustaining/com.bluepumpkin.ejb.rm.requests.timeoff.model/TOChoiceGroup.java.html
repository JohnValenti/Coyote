package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import java.util.Collection;
import java.util.List;

public class TOChoiceGroup {

	//The choices are sorted by start time
	final List<TOChoice> choices;

	//choices should be non-null, non-empty and sorted by start time.
	public TOChoiceGroup(List<TOChoice> choices) {

		if (choices == null) {
			throw new IllegalArgumentException("Choice group cannot be null");
		}

		if (choices.isEmpty()) {
			throw new IllegalArgumentException("Choice group cannot be empty");
		}

		this.choices = choices;
	}

	public Collection<TOChoice> getChoices() {
		return choices;
	}


	public int size() {
		return choices.size();
	}

	public int getUserRank() {
		return choices.get(0).getUserRank();
	}

}
