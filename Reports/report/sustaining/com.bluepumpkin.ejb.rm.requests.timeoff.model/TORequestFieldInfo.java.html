package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import com.bluepumpkin.ejb.bbm.vo.ChildInfoEntry;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;

/**
 * meta data information for Time off Request object, bet requests
 */
public class TORequestFieldInfo extends FieldInfo {
	public static final int TIMEOFFREQUEST_I_ID = 0;
	public static final int TIMEOFFREQUEST_I_ACTIVITYID = 1;
	public static final int TIMEOFFREQUEST_S_DEBITTYPE = 2;
	public static final int TIMEOFFREQUEST_N_FLEXTYPE = 3;
	public static final int TIMEOFFREQUEST_B_ISVTOFULLDAY = 4;
	public static final int TIMEOFFREQUEST_B_ISVTOPARTIALDAY = 5;
	public static final int TIMEOFFREQUEST_B_ISVTOLATESTART = 6;
	public static final int TIMEOFFREQUEST_B_ISVTOEARLYOFF = 7;
	public static final int TIMEOFFREQUEST_B_ISVTOANYTIMEOFF = 8;
	public static final int TIMEOFFREQUEST_D_ADVANCEDVTODATE = 9;
	public static final int TIMEOFFREQUEST_D_PARTIALDAYSTARTTIME = 10;
	public static final int TIMEOFFREQUEST_D_PARTIALDAYENDTIME = 11;
	private static final int TIMEOFFREQUEST_NUM_OF_FIELDS = 12;

	private static final String TABLE_NAME = "TIMEOFFREQUEST";

	// Time Off requests have choices as children.
	public static final int TO_CHOICE_CHILD_TYPE = 0;
	public static final int TO_WAITLIST_CHILD_TYPE = 1;
	public static final int TO_WITHDRAW_CHILD_TYPE = 2;
	public static final int TO_FLEXMAKEUP_CHILD_TYPE = 3;

	private static ChildInfoEntry[] m_childInfo = { ChildInfoEntry.oneToManyChild(), ChildInfoEntry.oneToManyChild(),
			ChildInfoEntry.oneToManyChild(), ChildInfoEntry.oneToManyChild() };

	private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[TIMEOFFREQUEST_NUM_OF_FIELDS];

	static {
		RequestUtil.initFieldInfoEntries(m_arrFieldInfo, TORequestFieldInfo.class, "TIMEOFFREQUEST_",
				TIMEOFFREQUEST_NUM_OF_FIELDS);
	}

	public TORequestFieldInfo() {
	}

	// implement abstract methods from parent class
	@Override
	public String getTableName() {
		return TABLE_NAME;
	}

	@Override
	public int getIDFieldIndex() {
		return TIMEOFFREQUEST_I_ID;
	}

	
	@Override
	public FieldInfoEntry[] getFieldInfo() {
		return m_arrFieldInfo;
	}

	protected int getNumFields() {
		return TIMEOFFREQUEST_NUM_OF_FIELDS;
	}

	// Return the child info array that contains necessary information about
	// the time off choice children entries and any entries from our superclass.
	@Override
	public ChildInfoEntry[] getChildInfo() {
		return m_childInfo;
	}
}
