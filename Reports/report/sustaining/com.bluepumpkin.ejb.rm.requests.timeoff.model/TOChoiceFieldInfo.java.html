package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import com.bluepumpkin.ejb.bbm.vo.ChildInfoEntry;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;


/**
 * meta data class for time off request choice entries
 */
public class TOChoiceFieldInfo extends FieldInfo {
    // field id constants
    public static final int TIMEOFFREQUESTCHOICE_I_ID = 0;
    public static final int TIMEOFFREQUESTCHOICE_I_TIMEOFFREQUESTID = 1;
    public static final int TIMEOFFREQUESTCHOICE_N_RANK = 2;
    public static final int TIMEOFFREQUESTCHOICE_D_STARTTIME = 3;
    public static final int TIMEOFFREQUESTCHOICE_D_ENDTIME = 4;
    public static final int TIMEOFFREQUESTCHOICE_N_USERRANK = 5;
    public static final int TIMEOFFREQUESTCHOICE_B_ISWAITLIST = 6;
	public static final int TIMEOFFREQUESTCHOICE_I_REFERENCESPID = 7;
	public static final int TIMEOFFREQUESTCHOICE_B_APPROVED = 8;


	protected static final int TIMEOFFREQUESTCHOICE_NUMOFFIELDS = 9;

    private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[TIMEOFFREQUESTCHOICE_NUMOFFIELDS];

    private static String m_strTableName = "TIMEOFFREQUESTCHOICE";

    static
    {
        RequestUtil.initFieldInfoEntries(m_arrFieldInfo, TOChoiceFieldInfo.class, "TIMEOFFREQUESTCHOICE_",
                TIMEOFFREQUESTCHOICE_NUMOFFIELDS);
    }
    // Time Off Choice has choices as  TOHOURSPERDAY as child.
    public static final int TOHOURSPERDAY_CHILD_TYPE = 0;
    public static final int TIMEOFFINTERVALPAIDSEGMENTS_CHILD_TYPE = 1;
    private static ChildInfoEntry[] m_childInfo = { ChildInfoEntry.oneToManyChild(), ChildInfoEntry.oneToManyChild()};


    public TOChoiceFieldInfo() {
        super();
    }

    @Override
	public String getTableName() {
        return m_strTableName;
    }

    @Override
	public int getIDFieldIndex() {
        return TIMEOFFREQUESTCHOICE_I_ID;
    }

    @Override
	public FieldInfoEntry[] getFieldInfo() {
        return m_arrFieldInfo;
    }

    protected int getNumFields() {
        return TIMEOFFREQUESTCHOICE_NUMOFFIELDS;
    }

    /**
     * Time off choice records have the request record as the parent
     */
    @Override
	public int getParentIDFieldIndex() {
        return TIMEOFFREQUESTCHOICE_I_TIMEOFFREQUESTID;
    }
    // Return the child info array that contains necessary information about
    // the time off choice children entries and any entries from our superclass.
    @Override
	public ChildInfoEntry[] getChildInfo() {
        return m_childInfo;
    }	
}
