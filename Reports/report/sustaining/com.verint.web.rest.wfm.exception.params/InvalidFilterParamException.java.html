package com.verint.web.rest.wfm.exception.params;


import java.util.Set;

import com.bluepumpkin.common.util.StringUtil;
import com.verint.web.rest.wfm.exception.EntityException;

import io.katharsis.repository.response.HttpStatus;

public class InvalidFilterParamException extends EntityException {

	private final String invalidFilterParamName;
	private final Set<String> validFilterParamNames;

	public InvalidFilterParamException(String entityName, String invalidFilterParamName,
			Set<String> validFilterParamNames) {
		super(entityName, "");

		this.invalidFilterParamName = invalidFilterParamName;
		this.validFilterParamNames = validFilterParamNames;
	}

	@Override
	public String getTitle() {
		return "InvalidFilterParamException";
	}

	@Override
	public String getDetail() {
		StringBuilder titleText = new StringBuilder(200);
		titleText.append("The filter parameter ")
				.append(invalidFilterParamName)
				.append(" is invalid for entity ")
				.append(entityName)
				.append(". Valid filter parameters are: ");

		if (validFilterParamNames == null || validFilterParamNames.isEmpty()) {
			titleText.append("<No valid filter parameters>");
		} else {
			titleText.append(StringUtil.createDelimitedString(validFilterParamNames, StringUtil.DELIM));
		}
		return titleText.toString();
	}

	@Override
	public int getHttpStatusCode() {
		return HttpStatus.BAD_REQUEST_400;
	}
}
