package com.bluepumpkin.web.fs.schedule.group.portlet;

import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;
import com.bluepumpkin.web.fs.schedule.group.GroupScheduleRH;
import com.verint.web.core.dashboard.pluto.PlutoPreferencesUtil;
import com.verint.web.core.dashboard.pluto.Preferences;
import com.witness.web.uif.base.WebException;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.pagemodel.WorkpanePageModel;
import com.witness.web.uif.system.RequestContext;

public class GroupMultidaySchedulePortletRH extends GroupScheduleRH {

	public static String EDIT_PARAM_ORG_ID = "orgId";
	public static String EDIT_PARAM_CAMPAIGN_ID = "campaignId";
	public static String EDIT_PARAM_NUM_DAYS = "numDays";
	public static String EDIT_PARAM_SORT_BY = "sortBy";

	/**
	 * Custom process Request
	 */
	public void processCustomRequest(RequestContext context) throws WebException {
		if(isAction(context, PageAction.SAVE)) {
			processSave(context);
		}  else if (isAction(context, PageAction.CANCEL)) {
			GroupMultidaySchedulePortletEditPM model = new GroupMultidaySchedulePortletEditPM(context);
			model.extractParameters(context.getRequest());
			context.setPageModel(model);
			model.setEditComplete(true);
			model.setSavePressed(false);
		}
	}

	protected void processSave(RequestContext context) {
		GroupMultidaySchedulePortletEditPM model = new GroupMultidaySchedulePortletEditPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		model.setEditComplete(true);
		model.setSavePressed(true);
		
		try {
			String orgIDStr = "";
			if (model.getOrgID() != null) {
				orgIDStr = model.getOrgID().toString();
			}
			Preferences prefs = PlutoPreferencesUtil.getPortletPreferences(context, model.getTemplateID(), model.getInstanceID());
			prefs.setValue(EDIT_PARAM_ORG_ID, orgIDStr);
		
			String campaignIDStr = "";
			if (model.getCampaignID() != null) {
				campaignIDStr = model.getCampaignID().toString();
			}
			prefs.setValue(EDIT_PARAM_CAMPAIGN_ID, campaignIDStr);
			prefs.setValue(EDIT_PARAM_NUM_DAYS, Integer.toString(model.getNumDays()));
			prefs.setValue(EDIT_PARAM_SORT_BY, model.getSortBy());
			
			try {
				prefs.savePortletPreferences();
			} catch(BbmObjectNotFoundException ex) {
				model.setSaveErrMsgID(ex.getMsgID());
			}
		} catch (Exception ex) {
			handleUnableToUpdate(model, ex);
			context.setPageModel(model);
		}
	}
	
	/**
	 * Return Page Model to perform generic action request on
	 */
	public WorkpanePageModel getPageModel(RequestContext context) {
		WorkpanePageModel model = new GroupMultidaySchedulePortletPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		return model;
	}//getPageModel

}
