package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.validation;

import java.util.List;

import org.apache.log4j.Priority;

import com.bluepumpkin.common.logging.Category;

import com.bluepumpkin.ejb.bbm.timeoffcalculator.TimeMap;

import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidRequest;

/**
 * @author rrajendran
 * 
 * Soft validation rule: Agent's schedule complies with organization hours of operation.
 */
public class BiddableScheduleWithinHOO implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

    private final static String m_className = BiddableScheduleWithinHOO.class.getName();

    private final static Category m_cat = Log.initCategory(m_className);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

        try {
            ShiftBidRequest sbr = (ShiftBidRequest) validatable;
            ShiftBidRequestValidationCache sbrCache = sbr.getValidationCacheSBR();

            // get the HOO time map
            TimeMap orgHooTimeMap = sbrCache.getHOOAssignmentsTimeMap();

            // obtain shift assignments associated with the sbr.
            List sbrShifts = sbr.getOptMethods().getShiftAssignments();

            // verify if shifts fall within HOO
            Event shift = RequestUtil.verifyOffBitsOverlapEvents(orgHooTimeMap, sbrShifts);
            if (shift != null)  {
                ValidationResult result = ValidationUtil.setSoftValidationResult(sbr,
										RmEjbBundleKey.SBR_SA_NOT_WITHIN_HOO,
										shift.getStartTime(), shift.getEndTime(), m_className);
                return result;
            }

            return null;
//      } c_atch (Exception e) {
//          t_hrow RequestUtil.createValidationExceptionWrapper(e, m_cat);
        } finally {
           m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
        }
    }

}
