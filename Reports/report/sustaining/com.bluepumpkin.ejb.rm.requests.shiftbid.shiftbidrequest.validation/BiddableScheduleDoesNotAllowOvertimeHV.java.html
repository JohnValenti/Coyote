package com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.validation;

import java.util.Collection;

import org.apache.log4j.Priority;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.schedule.model.BbmScheduleConflict;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignmentFields;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftEventAssignment;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidauction.model.ShiftBidAuction;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidRequest;
import com.bluepumpkin.ejb.rm.util.RmScheduleUtil;
import com.verint.ejb.wfm.WfmManagerFactory;

/**
 * In a multiple shifts per day bidding, the Before/After OTs are not allowed in
 * any shift block, nor existing shift assignment This hard rule is to enforce
 * that logic.
 */
public class BiddableScheduleDoesNotAllowOvertimeHV implements Validator {
	private static final Priority TRACEPRIORITY = RequestUtil.m_defaultTracePriority;

	private static final String CLASSNAME = BiddableScheduleDoesNotAllowOvertimeHV.class.getName();
	private static final Category CAT = Log.initCategory(CLASSNAME);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		CAT.log(TRACEPRIORITY, "Validation Rule: " + CLASSNAME + " started");
		ShiftBidRequest shiftBidderRequest = (ShiftBidRequest) validatable;
		ValidationResult result = null;

		try {
			boolean isMultiShiftBlocksBidding = getShiftBidAuction(shiftBidderRequest).getIsMultiShift();

			// the OTs are not allowed in multiple shifts per day bidding
			if (isMultiShiftBlocksBidding) {
				// get shift assignments of phantom employee
				Collection<ShiftAssignment> shiftAssignments = getShiftAssignmentsFromBiddableSchedules(
						shiftBidderRequest);
				result = validateShiftAssignmentsFromBiddableSchedules(shiftBidderRequest, shiftAssignments);

				if (result == null) {
					// replace phantom employee IDs by the bidder employee ID
					repacePhantomeIDByEmployeeID(shiftBidderRequest, shiftAssignments);
					result = validateConflictedShiftAssignments(shiftBidderRequest,
							getConflictsForNewScheduleObjects(shiftAssignments));
				}
			}
		} finally {
			CAT.log(TRACEPRIORITY, "Validation Rule: " + CLASSNAME + " done");
		}

		return result;
	}

	private ValidationResult validateShiftAssignmentsFromBiddableSchedules(ShiftBidRequest shiftBidderRequest,
			Collection<ShiftAssignment> shiftAssignments) throws Exception {
		ValidationResult result = null;
		for (ShiftAssignment shift : shiftAssignments) {
			result = validateShiftAssignments(shiftBidderRequest, shift);
			if (result != null) {
				break;
			}
		}

		return result;
	}

	private ValidationResult validateConflictedShiftAssignments(ShiftBidRequest shiftBidderRequest,
			Collection<BbmScheduleConflict> conflicts) throws Exception {
		ValidationResult result = null;

		if (conflicts == null || conflicts.isEmpty()) {
			return null;
		}

		for (BbmScheduleConflict conflict : conflicts) {

			if (conflict.getConflictType() == BbmScheduleConflict.CONFLICT_OVERLAP_SHIFT
					|| conflict.getConflictType() == BbmScheduleConflict.CONFLICT_MULTIPLE_SHIFTS_ON_SINGLE_DAY) {
				ShiftAssignment first = (ShiftAssignment) conflict.getFirstObject();
				ShiftAssignment second = (ShiftAssignment) conflict.getSecondObject();

				result = validateShiftAssignments(shiftBidderRequest, first);
				if (result == null) {
					result = validateShiftAssignments(shiftBidderRequest, second);
				}

				if (result != null) {
					break;
				}
			}

		}

		return result;
	}

	private ValidationResult validateShiftAssignments(ShiftBidRequest shiftBidderRequest, ShiftAssignment shift) {
		ValidationResult result = null;
		// before extension OR after extension exists
		if ((shift.getOTExtensionBeforeActivityID() != null && shift.getExtensionBefore() > 0)
				|| (shift.getOTExtensionAfterActivityID() != null && shift.getExtensionAfter() > 0)) {
			result = ValidationUtil.setHardValidationResult(shiftBidderRequest,
					RmEjbBundleKey.SBR_MULTI_SHIFTS_BIDDING_NOT_ALLOWED_OT, shift.getStartTime(), shift.getEndTime(),
					CLASSNAME);
		}
		return result;
	}

	private void repacePhantomeIDByEmployeeID(ShiftBidRequest shiftBidderRequest,
			Collection<ShiftAssignment> shiftAssignments) throws Exception {
		ID empID = shiftBidderRequest.getEmployeeID();
		for (ShiftAssignment shift : shiftAssignments) {
			shift.addWorkResourceID(empID);

			// get shift event assignments for this shift assignment
			Collection<ShiftEventAssignment> shiftEventAssns = shift.getChildren();
			// remove the shift event assignment children for this shift
			// assignment.
			shift.clearChildObjectMap(ShiftAssignmentFields.CHILD_SHIFT_EVENT);
			for (ShiftEventAssignment shiftEvent : shiftEventAssns) {
				shiftEvent.addWorkResourceID(empID);
				// add shift event to shift as a child to be created.
				shift.addShiftEventAssignment(shiftEvent);
			}
		}
	}

	protected ShiftBidAuction getShiftBidAuction(ShiftBidRequest sbr) {
		return sbr.getOptMethods().getShiftBidAuction();
	}

	@SuppressWarnings("unchecked")
	protected Collection<ShiftAssignment> getShiftAssignmentsFromBiddableSchedules(ShiftBidRequest sbr) {

		return RmScheduleUtil.deepCloneShiftAssignments(sbr.getOptMethods().getShiftAssignments());
	}

	protected Collection<BbmScheduleConflict> getConflictsForNewScheduleObjects(
			Collection<ShiftAssignment> shiftAssignments) throws BbmEJBCreateException, Exception {
		return WfmManagerFactory.getScheduleAccessManager().getConflictsForNewScheduleObjects(shiftAssignments);
	}
}
