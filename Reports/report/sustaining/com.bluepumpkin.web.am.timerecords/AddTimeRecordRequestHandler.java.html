package com.bluepumpkin.web.am.timerecords;

public class AddTimeRecordRequestHandler extends RequestHandler{

	 /** Log object used to log information to log file */
	private static Category cat  = Log.initCategory(AddTimeRecordRequestHandler.class.getName());
	private ResourceBundle m_amBundle;
	private Localizer m_localizer;
	private Employee m_emp;

	public void processRequest(RequestContext context){
		m_localizer = context.getLocalizer();

		m_amBundle = m_localizer.getBundle(AmWebBundleKey.BUNDLE_NAME);

		TimeRecord aRecord = null;

		AddTimeRecordModel model = new AddTimeRecordModel(context);
		model.extractParameters(context.getRequest(), context);

		setCurEmployee(model);

		setTimeZone(context, model);

		//get current employees org
		model.setOrgId(MyTimeModelHandler.getEmployeeOrgId(context, model.getEmpId()));
		if (isAction(context, AddTimeRecordModel.ADDTR_ACTION)){
			createTempRecord(model, context);
		} else if (isAction(context,AddTimeRecordModel.INSERTTE_ACTION)){
			insertTempEntry(context, model);
		} else if (isAction(context,AddTimeRecordModel.DELETETE_ACTION)){
			deleteTempEntry(context, model);
		} else if (isAction(context,AddTimeRecordModel.SAVETR_ACTION)){
			saveTimeRecord(context, model);
		} else if (isAction(context,AddTimeRecordModel.CLOSEANDSAVETR_ACTION)){
			closeAndSaveTimeRecord(context, model);
		} else if (isAction(context,AddTimeRecordModel.REFRESH_ACTION)){
			aRecord = TimeRecordModelHandler.populateTimeRecordFromModel(model);
			model.setTimeRecord(aRecord);
		}

		model.processActivityGroup();

		//cache myTimeModel
		context.setAttribute(RequestContext.REQUEST_SCOPE, AddTimeRecordModel.MODEL_KEY, model);
	}

	private void createTempRecord(AddTimeRecordModel model, RequestContext context){
		try{
			TimeRecord aRecord = TimeRecordModelHandler.createTempRecord(model, context);
			model.setTimeRecord(aRecord);
			//set defualt category to 0
			model.setActivityCategory(new String[]{"0", "0", "0","0"});
		} catch (Exception ex){

		}
	}
	/**
	 * setCurEmployee
	 * This method sets teh current employee based on pagination
	 */
	private void setCurEmployee(AddTimeRecordModel model){
		//set employee
		model.setEmpId(model.getSelectedIDs()[model.getCurIndex()]);
	}

	/**
	 * setTimeZone
	 */
	private void setTimeZone(RequestContext context, AddTimeRecordModel model){
		//if (model.getTimeZone() == null){
		Calendar aCal = Calendar.getInstance();
		if (model.getStartYear() == 0){
			//already have extracted once
			aCal.setTime((Date)model.getDateList().get(0));
		} else {//use start values
			aCal.set(model.getStartYear(), model.getStartMonth(), model.getStartDay(),0,0,0);
		}
		try {
			m_emp = WorkResourceModelHandler.getWorkResourceManager(context).getEmployeeByID(model.getEmpId(),
				aCal.getTime(), Employee.DETAIL_LEVEL_EMPLOYEE_TIMEBASEDPROPERTY);

			//get organization
			Organization org = OrganizationModelHandler.getOrganization(context, m_emp.getOrganizationID());
			model.setTimeZone(org.getTimeZone().getID());
		} catch (Exception ex){
			model.setTimeZone(context.getViewingTimeZone().getID());
		}
		//}
	}

	private void deleteTempEntry(RequestContext context, AddTimeRecordModel model){
		try {
			TimeRecord aRecord = TimeRecordModelHandler.populateTimeRecordFromModel(model);
			model.setTimeRecord(aRecord);
			aRecord = TimeRecordModelHandler.deleteTempEntry(model);

			//modify activityCategory variable of model to remove one entry
			ArrayList tempList = new ArrayList();
			for (int j=0;j<aRecord.getChildren().size(); j++){
				if (j != model.getPosition()-1){
					tempList.add(model.getActivityCategory()[j].toString());
				}
			}

			String [] catArray = new String[tempList.size()];
			for (int j=0;j<tempList.size(); j++){
				catArray[j] = (String)tempList.get(j);
			}

			model.setActivityCategory(catArray);
			model.setTimeRecord(aRecord);
		} catch (BbmTimeRecordException ex){
			String errorCode = EditTimeRecordRequestHandler.evaluateTimeRecordException(m_amBundle, ex, m_localizer);
			if (errorCode ==null){
				errorCode = m_localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_DELETINGTE);
			}
			model.addPageMessage(Message.ERROR_TYPE, errorCode);
			model.setPageAction(AddTimeRecordModel.INSERTTE_ACTION);
		}
	}

	private void insertTempEntry(RequestContext context, AddTimeRecordModel model){
		try {
			TimeRecord aRecord = TimeRecordModelHandler.populateTimeRecordFromModel(model);
			aRecord = TimeRecordModelHandler.insertTempEntry(model, aRecord, context);

			//modify activityCategory variable of model to add a new entry
			ArrayList tempList = new ArrayList();
			for (int j=0;j<aRecord.getChildren().size()-2; j++){
				tempList.add(model.getActivityCategory()[j].toString());
			}
			tempList.add(model.getPosition(), new String("0"));

			String [] catArray = new String[tempList.size()];
			for (int j=0;j<tempList.size(); j++){
				catArray[j] = (String)tempList.get(j);
			}

			model.setActivityCategory(catArray);

			model.setTimeRecord(aRecord);
		} catch (BbmTimeRecordException ex){
			String errorCode = EditTimeRecordRequestHandler.evaluateTimeRecordException(m_amBundle, ex, m_localizer);
			if (errorCode ==null){
				errorCode = m_localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_INSERTINGTE);
			}

			model.setTimeRecord(TimeRecordModelHandler.populateTimeRecordFromModel(model));

			model.addPageMessage(Message.ERROR_TYPE, errorCode);
			model.setPageAction(AddTimeRecordModel.INSERTTE_ACTION);
		}
	}

	private void saveTimeRecord(RequestContext context, AddTimeRecordModel model){
	    boolean isValidDateForEmployee = false;

		if (m_emp != null){
			//check that the record falls within the employee's time range
			if (m_emp.getStartTime().before((Date)model.getDateList().get(0))){
				if (m_emp.getEndTime() == null || (m_emp.getEndTime() !=null && m_emp.getEndTime().after((Date)model.getDateList().get(model.getDateList().size()-1))))
					isValidDateForEmployee = true;
			}
		}


		if (!isValidDateForEmployee){
       		model.setPageAction(AddTimeRecordModel.ADDTR_ACTION);
			model.addPageMessage(Message.ERROR_TYPE, AmWebBundleKey.MSG_ERRORCREATINGTR_NOEMPORG,
				AmWebBundleKey.BUNDLE_NAME);
			model.setTimeRecord(TimeRecordModelHandler.populateTimeRecordFromModel(model));
		} else {
			try {
				TimeRecordModelHandler.createTimeRecord(model, context);
				model.setPageAction(AddTimeRecordModel.DONE_ACTION);
				Message message = new Message(Message.RESPONSE_TYPE, AmWebBundleKey.MSG_SAVEDTR, AmWebBundleKey.BUNDLE_NAME);
				model.addPageMessage(message);
				TimeRecordUtil.setContextAttribute(context, RequestContext.APPLICATION_SCOPE,
					                               TimeRecordUtil.AM_WEB_CREATE_TIME_RECORD, message);
			} catch (BbmTimeRecordException ex) {
				//deal with open shift case
				if (ex.getErrCode() == BbmTimeRecordException.TR_PRV_OPEN_SHIFT){
					//get entry from thwe exception
					TimeRecordEntry entry = (TimeRecordEntry)ex.getContent();
					model.setTimeRecordID(entry.getParentID());

					String msgTemplate = m_localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_OPENSHIFT_MF);
					Object[] args = new Object[2];
					args[0] = m_localizer.formatDate(entry.getStartTime(), TimeZone.getTimeZone(model.getTimeZone()),
								RegionalFormatBundleKey.DATE_FORMAT);
					args[1] = m_localizer.formatDate(entry.getStartTime(), TimeZone.getTimeZone(model.getTimeZone()),
								RegionalFormatBundleKey.TIME_FORMAT);
					model.addPageMessage(Message.ERROR_TYPE, MessageFormat.format(msgTemplate, args));
					model.setPageAction(AddTimeRecordModel.CLOSEANDSAVETR_ACTION);
				} else {
					model.setPageAction(AddTimeRecordModel.ADDTR_ACTION);
					String errorCode = EditTimeRecordRequestHandler.evaluateTimeRecordException(m_amBundle, ex, m_localizer);
					if (errorCode ==null){
						errorCode = m_localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_SAVINGTR);
					}
					model.addPageMessage(Message.ERROR_TYPE, errorCode);
				}

				model.setTimeRecord(TimeRecordModelHandler.populateTimeRecordFromModel(model));
			} catch (Exception ex){
				model.setPageAction(AddTimeRecordModel.ADDTR_ACTION);

				cat.l7dError(AmWebLogBundleKey.MSG_ERROR_SAVINGTR, ex);
				model.addPageMessage(Message.ERROR_TYPE, AmWebBundleKey.MSG_ERROR_SAVINGTR,
					AmWebBundleKey.BUNDLE_NAME);
			}
		}
	}

	private void closeAndSaveTimeRecord (RequestContext context, AddTimeRecordModel model){
		boolean isException = false;
		try {
			//get the date to close the existing time record at
			Calendar aCal = Calendar.getInstance(TimeZone.getTimeZone(model.getTimeZone()));
			TimeRecord aRecord = null;

			//get time record from model
			aRecord = TimeRecordModelHandler.populateTimeRecordFromModel(model);

			ArrayList entryList = (ArrayList)aRecord.getChildren();
			TimeRecordEntry entry = (TimeRecordEntry) entryList.get(0);
			aCal.setTime(entry.getStartTime());
			aCal.add(Calendar.SECOND, -1);
			//close the open time record
			AmManagerFactory.getTimeCollectorManager().manualEndShift(model.getTimeRecordID(),
				model.getEmpId(), context.getUser().getID(), aCal.getTime(), null);
			//save the time record
			saveTimeRecord(context, model);
		} catch (AmRawEndShiftException ex){
			if (ex instanceof AmNotLastEndShiftException){
				model.addPageMessage(Message.ERROR_TYPE,
					AmWebBundleKey.MSG_ERROR_CREATINGENDSHIFT_NOTLASTENDSHIFT, AmWebBundleKey.BUNDLE_NAME);
			} else {
				model.addPageMessage(Message.ERROR_TYPE,
					AmWebBundleKey.MSG_ERROR_CREATINGENDSHIFT, AmWebBundleKey.BUNDLE_NAME);
			}
			cat.l7dInfo(AmWebLogBundleKey.MSG_ERROR_CREATINGENDSHIFT , ex);
			isException = true;
		} catch (AmTimeRecordEndShiftException ex){
			cat.l7dInfo(AmWebLogBundleKey.MSG_ERROR_CREATINGENDSHIFT , ex);
			isException = true;
			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_CREATINGENDSHIFT_FORTR, AmWebBundleKey.BUNDLE_NAME);
		} catch (Exception ex){
			cat.l7dInfo(AmWebLogBundleKey.MSG_ERROR_CREATINGENDSHIFT , ex);
			isException = true;
			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_CREATINGENDSHIFT, AmWebBundleKey.BUNDLE_NAME);
		}

		if (isException){
			model.setPageAction(AddTimeRecordModel.ADDTR_ACTION);
			model.setTimeRecord(TimeRecordModelHandler.populateTimeRecordFromModel(model));
		}
	}
}
