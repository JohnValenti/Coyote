package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;

import com.bluepumpkin.ejb.rm.Log;

import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;

import com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.TimeOffIntervalAllocationUtil;

import com.bluepumpkin.ejb.rm.util.RmUtil;
import com.bluepumpkin.ejb.rm.util.TORequestUtil;

/**
 * "TimeoffIntervalAllocationReferencesScheduleRule" validation rule for time off requests.
 * For activity types ONLY marked as "interval allocation" (i.e., and not also "daily") verify
 * that the employee has a reference schedule.
 * If no reference schedule exists for the employee, a hard validation exception is raised.
 **/

public class TimeoffIntervalAllocationReferenceScheduleRule implements Validator {

    private static final Category LOG = Log.initCategory( TimeoffIntervalAllocationReferenceScheduleRule.class.getName());


    @Override
    public ValidationResult validate(Validatable validatable) throws Exception {//NOSONAR
    	

        String methodName = "validate";
        if (LOG.isDebugEnabled()){
        	LOG.debug(RmUtil.dumpEnterMethod(methodName, validatable));
        }

        ValidationResult result = null;

        TORequest toRequest = (TORequest)validatable;

        // If we are not using the interval pool, no need to validate
        boolean useIntervalPool = isUseIntervalTOPool(toRequest);
		if (!useIntervalPool || toRequest.isAdvancedVTORequest()) {
        	return result;
        }


		ID empId = toRequest.getEmployeeID();

		// Check if published. If so, no need for reference schedules
		
		if(isRequestIsFullyPublished(toRequest)){
			return result;
		}
		

		if(!validateReferenceScheduleGood(toRequest)){
        	result = ValidationUtil
    			.setHardValidationResult(toRequest,
    				RmEjbBundleKey.TIMEOFFINTERVALALLOCATIONSEGMENT_NO_REFERENCE_SCHEDULE,
    				getEmployeeByID(toRequest), this.getClass().getName());

		}


        if (LOG.isDebugEnabled()) {
        	LOG.debug(RmUtil.dumpExitMethod(methodName, result));
        }
        return result;
    }

	protected Employee getEmployeeByID(TORequest tor) throws Exception {//NOSONAR
		return tor.getCache().getEmployeeBasedOnEmpDetailLevel(Employee.DETAIL_LEVEL_EMPLOYEE_BASIC);
	}
	

	protected boolean validateReferenceScheduleGood(TORequest toRequest) throws Exception {//NOSONAR
		return TimeOffIntervalAllocationUtil.validateReferenceScheduleGood(toRequest);
	}

	protected boolean isUseIntervalTOPool(TORequest toRequest) throws Exception {//NOSONAR
		return TORequestUtil.isUseIntervalTOPool(toRequest);
	}
	
	protected boolean isRequestIsFullyPublished(TORequest toRequest)
			throws Exception {//NOSONAR

		PublishingPeriodStatus publishingPeriodStatus = TimeOffIntervalAllocationUtil
				.initPublishedPeriodStatusForTimeoffRequest(toRequest);
		ID empID = toRequest.getEmployeeID();
		List<TOChoice> choiceList = toRequest.getRequestChoiceList();

		for (TOChoice toChoice : choiceList) {
			if (toChoice == null) {
				continue;
			}

			if (!TimeOffIntervalAllocationUtil.isPublishedStatusFullyPublished(
					publishingPeriodStatus, empID, toChoice)) {
				return false;
			}
		}
		return true;
	}
	
}
