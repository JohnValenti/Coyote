package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

/**
 * Title:        TimeOffBidRequestNotOverlapValidationRule
 * Description:  Validate a time off bid request to make sure it does not overlap with any approved time of bid request choice.
 * 				 The overlapping here is when the new request choice covers or overlapped partially with approved TO request choice.
 * 				 The case of new request choice within approved choice does not count here because in this case, it violates the 
 * 				 hard validation rule "WorkableTimeValidationRule" 
 * 				 For multi-mode: to make sure the considering choice does not overlap with approved choices
 * 				 This validation is invoked only if there is Advanced RM license
 * Copyright:    Copyright (c) 2016
 * Company:      Verint System, Inc
 * @author       Truc Pham
 * @version      1.0
**/
import java.util.Collection;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.Pair;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.localization.DefaultLocalizationManager;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidatorForMultiChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.ejb.TimeOffBidUtil;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBid;

public class TimeOffBidRequestNotOverlapValidationRule implements ValidatorForMultiChoice {
	private static final String clsName = TimeOffBidRequestNotOverlapValidationRule.class.getName();

	@Override
	public ValidationResult validateForMultiChoice(Validatable validatable) throws Exception {
		return validate(validatable, true);
	}

	private ValidationResult validate(Validatable validatable, boolean isForMultiChoice) throws Exception {
		ValidationResult result = null;
		TORequest toReq = (TORequest) validatable;
		//The rule is not applicable for Anytime VTO requests
		if (toReq.isAdvancedVTORequest()) {
			return null;
		}
		ID timeOffType = toReq.getTimeOffType();
		ID employeeId = toReq.getEmployeeID();
		Pair<TimeOffBid, Pair<Integer, Integer>> bidDetail = toReq.getTimeOffBidInfo();
		if (bidDetail == null) {
			bidDetail = getOpenTimeOffBidForActivityIdAndEmployeId(timeOffType, employeeId, toReq.getSubmittedOn());
			toReq.setTimeOffBidInfo(bidDetail);
		}
		if (bidDetail == null) {
			return null;
		}
		TimeOffBid bid = bidDetail.getFirst();
		if (bid == null) {
			return null;
		}
		long bidRequestStart = bid.getRequestStartTime().getTime();
		long bidRequestEnd = bid.getRequestEndTime().getTime();
		Collection<ID> activitiesIdInBid = getActivityIDsByBidId(bid.getId());
		Collection<TOChoice> approvedChoices =isForMultiChoice? toReq.getTOChoicesWillBeApproved(): getApprovedTimeOffRequestChoiceDuringOpenBid(bid.getId(), employeeId, activitiesIdInBid);
		//Validate choice compared to approved Choices  => soft validation rule
		for (TOChoice choice : toReq.getTOChoicesAdjustedForValidation()) {
			//For multi-choice,if the choice violates any validation rule ran before , do not run the rule and continue with next choice
			if(toReq.isRunForGroupTOBApproval()&&!choice.getValidationResults(true).isEmpty()){
				continue;
			}
			Date choiceStart = choice.getStartDate();
			Date choiceEnd = choice.getEndDate();
			boolean choicePeriodInBidRequestPeriod = bidRequestStart <= choiceStart.getTime() && choiceEnd.getTime() <= bidRequestEnd;
			TimeRange toChoiceRange = new TimeRange(choiceStart, choiceEnd);
			TimeRange choiceOverlapped = TimeOffBidUtil.getChoiceRangeOverlapWithGivenRange(toChoiceRange, approvedChoices,isForMultiChoice);
			if (choicePeriodInBidRequestPeriod && choiceOverlapped != null) {
				result = isForMultiChoice?
						 ValidationUtil.setSoftValidationResult(choice, RmEjbBundleKey.TIMEOFF_BID_VALIDATION_TOCHOICE_OVERLAP_ANOTHER_TOCHOICE, 
									choiceStart, choiceEnd, choiceOverlapped.getStartDate(), choiceOverlapped.getEndDate(), clsName):
					ValidationUtil.setSoftValidationResult(choice, RmEjbBundleKey.TIMEOFF_BID_VALIDATION_TOCHOICE_OVERLAP_APPROVED_REQUESTS, 
						getEmployeeName(toReq), choiceStart, choiceEnd, choiceOverlapped.getStartDate(), choiceOverlapped.getEndDate(), clsName);
				
			}
		}

		return result;
	}
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		return validate(validatable,false);
	}
	protected Pair<TimeOffBid, Pair<Integer, Integer>> getOpenTimeOffBidForActivityIdAndEmployeId(ID timeOffType, ID employeeId, Date submittedDate) throws Exception {
		return TimeOffBidUtil.getOpenTimeOffBidDetailForActivityIdAndEmployeId(timeOffType, employeeId, submittedDate);
	}

	protected Collection<TOChoice> getApprovedTimeOffRequestChoiceDuringOpenBid(ID bidId, ID employeeId, Collection<ID> activityIds) throws Exception {
		return TimeOffBidUtil.getApprovedTimeOffRequestChoiceDuringOpenBid(bidId, employeeId, activityIds);
	}

	public Collection<ID> getActivityIDsByBidId(ID bidId) throws Exception {
		return TimeOffBidUtil.getActivityIDsByBidId(bidId);
	}

	protected String getEmployeeName(TORequest toReq) throws Exception {
		return toReq.getCache().getEmployeeNameByID();
	}

	protected Localizer getLocalizer() {
		return DefaultLocalizationManager.getDefaultInstance().getLocalizer();
	}
}
