/**
 * Rule in UI: Requested time off is not being swapped
 * <p/>
 * Validate that there are no requests to get a shift during the requested time
 * off.
 * <p/>
 * <p/>
 * Applies equally for debit, debit-if and dont-debit types. For dont-debit
 * types, regardless of this rule, it is not posssible to obtain a shift during
 * the time off (as dont-debit maps to unavaialbility, and unavailabiliy can't
 * overlap a shift).
 * <p/>
 * <p/>
 * Applies equally for tentative and non tentative requests (does not use
 * unpublished calendar event assignments).
 * <p/>
 * <p/>
 * ValidationResults are associated with the choices.
 */
public class TimeOffNotBeingSwappedValidationRule implements Validator {
	private static final String m_className = TimeOffNotBeingSwappedValidationRule.class.getName();

	private static final Category m_cat = Log.initCategory(m_className);

	/**
	 * Validate a time off request.
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResultAggregator result = new ValidationResultAggregator();
		TORequest tor = (TORequest) validatable;
		String methodName = "validate";
		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpEnterMethod(methodName, tor));
		}
		// obtain shift swap requests
		Collection ssReqsForEmp = getRequestsByEmployee(tor.getEmployeeID());

		// For each choice in the request, verify it does not overlap with any
		// pending shift
		// swap requests. Assumption is that an employ may not have too many
		// unexpired ssReqs.
		// If not, then a more efficient implementation must be considered.
		for (TOChoice toChoice : tor.getTOChoicesAdjustedForValidation()) {
			if(tor.isRunForGroupTOBApproval()&& !toChoice.getValidationResults(true).isEmpty()){
				continue;
			}
			result.add(checkForIntersectionWithShiftSwaps(tor, toChoice, ssReqsForEmp));
		}


		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
		}
		return result.get(); // returns last result constructed or null
	}

	protected Collection getRequestsByEmployee(ID employeeID) throws Exception {
		ShiftSwapRequestDAO ssDao = new ShiftSwapRequestDAO(ShiftSwapRequest.getDetailLevelForValidation());
		try {
			// Note: The method ssDAO.getPendingRequestsByEmployeeInRange()
			// (used to
			// retrieve the list of pending requests for which the specified
			// employee receives a shift overlapping
			// the TOChoice) was replaced with the method below as it was too
			// slow.

			// get SS requests for the employee assoc with the TORequest.
			return ssDao.getRequestsByEmployee(employeeID, ShiftSwapRequest.getDetailLevelForValidation());
		} finally {
			ssDao.cleanUp();
		}
	}

	public ValidationResult checkForIntersectionWithShiftSwaps(TORequest tor, TOChoice choice, Collection ssReqsForEmp)
			throws Exception {
		ValidationResult result = null;
		for (Iterator ssReqsIter = ssReqsForEmp.iterator(); ssReqsIter.hasNext();) {
			ShiftSwapRequest ssReq = (ShiftSwapRequest) ssReqsIter.next();
			// if SSRequest status is not one of the below, continue.
			if (!(ssReq.isTentative() || ssReq.isNegotiation() || ssReq.isPending() || ssReq.isEscalated())) {
				continue;
			}
			// Find the shift swap item received by the TOReq empID.
			ShiftSwapItem ssItem = ssReq.getShiftSwapItemRcvdByEmpID(tor.getEmployeeID());
			/*
			 * Collection ssItemsList = ssReq.getShiftSwapItems(); for (Iterator
			 * iterator = ssItemsList.iterator(); iterator.hasNext();) {
			 * ShiftSwapItem ssItem = (ShiftSwapItem) iterator.next();
			 */
			result = setValidationResultForTimeOffNotBeingSwapped(tor, choice, ssItem);
			if (result != null) {
				return result;
			}

		}
		return result;
	}

	public ValidationResult setValidationResultForTimeOffNotBeingSwapped(TORequest tor, TOChoice choice,
			ShiftSwapItem ssItem) throws Exception {
		// Validate ssItem only if it belongs to the emp we are interested in,
		// if above is true then check if it intersects the TOChoice , else
		// return null.
		ValidationResult result = null;
		if (ssItem != null && ssItem.getEndDate().after(choice.getStartDate())
				&& ssItem.getStartDate().before(choice.getEndDate())) {
			result = ValidationUtil.setSoftValidationResult(choice,
					RmEjbBundleKey.TO_TIME_OFF_BEING_SWAPPED, new Serializable[] { getEmployeeNameByID(tor),
							choice.getStartDate(), choice.getEndDate(), ssItem.getStartDate(), ssItem.getEndDate() },
					m_className);
		}
		return result;
	}

	protected String getEmployeeNameByID(TORequest tor) throws Exception {
		return tor.getCache().getEmployeeNameByID();
	}
}
