package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

/**
 * Title:        TimeOffBidMaxWeeksValidationRule
 * Description:  Soft validation alert when approving a Time Off Bid request. The rule enforces the maximum approval per employee for TimeOffBids. Time off requests become Time Off Bid Requests 
 *                if it passes the Hard validation rule TimeOffBidPeriodPerChoiceHV. See description com.bluepumpkin.ejb.rm.requests.timeoff.validation.TimeOffBidPeriodPerChoiceHV for more details
 * 				 This validation is invoked only if there is Advanced RM license
 * Copyright:    Copyright (c) 2016
 * Company:      Verint System, Inc
 * @author       Truc Pham
 * @version      1.0
**/
import java.util.Collection;
import java.util.Date;
import java.util.TimeZone;
import java.util.concurrent.TimeUnit;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.Pair;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.localization.DefaultLocalizationManager;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidatorForMultiChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.ejb.TimeOffBidUtil;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBid;

public class TimeOffBidMaxWeeksValidationRule implements ValidatorForMultiChoice {
	private static final String clsName = TimeOffBidMaxWeeksValidationRule.class.getName();

	/**
	 * Validate a time off bid request.
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		TORequest toReq = (TORequest) validatable;
		ValidationResult result = null;
		//The rule is not applicable for Anytime VTO requests
		if (toReq.isAdvancedVTORequest()) {
			return null;
		}
	
		ID timeOffType = toReq.getTimeOffType();
		ID employeeId = toReq.getEmployeeID();
		Pair<TimeOffBid, Pair<Integer, Integer>> bidDetail = toReq.getTimeOffBidInfo();
		if (bidDetail == null) {
			bidDetail = getOpenTimeOffBidForActivityIdAndEmployeId(timeOffType, employeeId, toReq.getSubmittedOn());
			toReq.setTimeOffBidInfo(bidDetail);
		}
		if (bidDetail != null) {
			TimeOffBid bid = bidDetail.getFirst();
			if (bid != null) {
				long bidRequestStart = bid.getRequestStartTime().getTime();
				long bidRequestEnd = bid.getRequestEndTime().getTime();
				Pair<Integer, Integer> periodUnitValue = bidDetail.getSecond();
				int periodUnit = 0;
				int periodValue = 0;
				if (periodUnitValue != null) {
					periodUnit = periodUnitValue.getFirst().intValue();
					periodValue = periodUnitValue.getSecond().intValue();
				}
				Localizer localizer = getLocalizer();
				String periodText = TimeOffBidUtil.getPeriodText(localizer, periodValue, periodUnit);
				long daysInMinute = TimeUnit.DAYS.toMinutes(1);
				long minuteInMilisecond = TimeUnit.MINUTES.toMillis(1);
				TimeZone empTz = getTimeZoneOfEmployee(employeeId);
				long maxMinutesApprovalPerSetting = TimeOffBidUtil.getNumberOfUnits(periodUnit, periodValue) * daysInMinute;
				long maxMinuteApproved = getTotalMinutesBidRequestApprovedPerEmployee(bid.getId(), employeeId, timeOffType);
				//willBeApprovedChoices is for multi-choices approval-we need to consider to choices which will be approved in a same requests
				//when approving subsequent choices. It is empty list when comparing a choice in a request to approved choices in other requests 
				Collection<TOChoice> willBeApprovedChoices = toReq.getTOChoicesWillBeApproved();
				long durationAllChoicesInMinute =TimeOffBidUtil.getDurationChoices(willBeApprovedChoices, empTz)/minuteInMilisecond;
				//For multi-choices, toReq.getTOChoicesAdjustedForValidation() will return one choice due to 
				//toReq.setTOChoiceForValidation(toChoice) in getValidChoicesForMultiChoiceProcess of TORequestManagementEJB.
				//For normal request, toReq.getTOChoicesAdjustedForValidation() will be getRequestChoiceList()
				for (TOChoice choice : toReq.getTOChoicesAdjustedForValidation()) {
					
					long choiceStart = choice.getStartDate().getTime();
					long choiceEnd = choice.getEndDate().getTime();
					long realchoiceDuration = TimeOffBidUtil.getRealChoiceDuration(choice.getStartDate(), choice.getEndDate(), empTz);
					long choiceDurationInMinute = realchoiceDuration / minuteInMilisecond;
					boolean choicePeriodInBidRequestPeriod = bidRequestStart <= choiceStart && choiceEnd <= bidRequestEnd;
					long totalDurationIfApproved = maxMinuteApproved + durationAllChoicesInMinute+ choiceDurationInMinute;
					if (choicePeriodInBidRequestPeriod && totalDurationIfApproved > maxMinutesApprovalPerSetting) {
						result = ValidationUtil.setSoftValidationResult(choice, RmEjbBundleKey.TIMEOFF_BID_VALIDATION_MAXIMUM_APPROVAL_PER_EMPLOYEE, getEmployeeName(toReq), periodText, choice.getStartDate(), choice.getEndDate(), clsName);
					}
				}
			}
		}
		return result;
	}

	@Override
	public ValidationResult validateForMultiChoice(Validatable validatable) throws Exception {
		return validate(validatable);
	}

	protected Pair<TimeOffBid, Pair<Integer, Integer>> getOpenTimeOffBidForActivityIdAndEmployeId(ID timeOffType, ID employeeId, Date submittedDate) throws Exception {
		return TimeOffBidUtil.getOpenTimeOffBidDetailForActivityIdAndEmployeId(timeOffType, employeeId, submittedDate);
	}

	protected long getTotalMinutesBidRequestApprovedPerEmployee(ID bidId, ID employeeId, ID activityId) throws Exception {
		return TimeOffBidUtil.getTotalMinutesBidRequestApprovedPerEmployee(bidId, employeeId, activityId);
	}

	protected String getEmployeeName(TORequest toReq) throws Exception {
		return toReq.getCache().getEmployeeNameByID();
	}

	protected Localizer getLocalizer() {
		return DefaultLocalizationManager.getDefaultInstance().getLocalizer();
	}

	protected TimeZone getTimeZoneOfEmployee(ID employeeId) throws BbmFinderException {
		return TimeOffBidUtil.getTimeZoneOfEmployee(employeeId);
	}

}
