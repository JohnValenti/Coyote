package com.verint.ejb.bbm.fterequirements.util;

import com.bluepumpkin.common.datatypes.Duration;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeContext;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmTimeSeriesException;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.campaign.model.SPShrinkage;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.timeseries.model.RequireTraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.ejb.bbm.timeseries.util.DSTConversionUtil;
import com.verint.ejb.bbm.fterequirements.model.RequiredTimeSeries;

import java.util.Calendar;
import java.util.Collection;
import java.util.Date;
import java.util.LinkedList;

public class FteRequirementsTimeSeriesUtil {
	private static final Duration intervalLength = Duration.fromMinutes(Trace.INTERVAL);

	private FteRequirementsTimeSeriesUtil() {
		// Static class.
	}

	/**
	 * Constructs RequiredTimeSeries objects that contain the FTE trace values of the given RequireTraceCube.
	 * Only sets trace values, spqueue ID and date on the RequiredTimeSeries.
	 */
	public static Collection<RequiredTimeSeries> getFteRequirementsTimeSeriesCol(RequireTraceCube tc, SPQueue spQueue,
		TimeContext timeContext) throws BbmCreateException {
		Date startDate = tc.getRawStartDate();
		Date endDate = tc.getRawEndDate();
		Collection<RequiredTimeSeries> retVal = new LinkedList<>();
		Calendar timeCursor = Calendar.getInstance(timeContext.getTimeZone());
		timeCursor.setTime(startDate);
		int dayCounter = 0;
		while (timeCursor.getTime().before(endDate)) {
			timeCursor.setTime(startDate);
			timeCursor.add(Calendar.DATE, dayCounter);
			dayCounter++;
			Date currentDay = timeCursor.getTime();

			timeCursor.setTime(startDate);
			timeCursor.add(Calendar.DATE, dayCounter);
			Date nextDay = new Date(timeCursor.getTime().getTime() - 1);

			int[] vals = DSTConversionUtil.convertFromDSTAwareTimeToIdealTime(
				tc.getTraceValue(Trace.FTE, currentDay, nextDay),
				Trace.TRACENA,
				currentDay,
				intervalLength,
				timeContext);
			if (vals.length == 0) {
				continue;
			}
			if (vals.length != 96) {
				throw new IllegalArgumentException("TraceCube for queue " + tc.getQueueID() + " contains " + vals.length + " DST-corrected values for type " + Trace.getTraceName(Trace.FTE) + " for day " + currentDay + ". Expected 96.");
			}

			RequiredTimeSeries rts = new RequiredTimeSeries(spQueue, currentDay, vals);

			retVal.add(rts);
		}
		return retVal;
	}

	/**
	 * Converts the collection of RequiredTimeSeries into a single RequireTraceCube.
	 * The values are placed in the Trace.FTE trace.
	 * Assumes that all RequiredTimeSeries coming in represent data from the same queue.
	 */
	public static RequireTraceCube convertRequiredTimeSeriesToTraceCube(Collection<RequiredTimeSeries> timeSeriesCol,
		ID queueID, Date startDate, Date endDate) throws BbmTimeSeriesException {
		RequireTraceCube retVal = new RequireTraceCube(queueID, startDate, endDate, RequireTraceCube.TYPES);
		for (RequiredTimeSeries timeSeries : timeSeriesCol) {
			retVal.setTraceValue(Trace.FTE, timeSeries.getSeriesValues(), timeSeries.getDate());
		}
		return retVal;
	}

	/**
	 * Creates and returns a DST-adjusted array of shrinkage data extracted from the
	 * given spShrinkage objects.  Each value in the array represents a 15-minute interval of
	 * shrinkage data.  Shrinkage is only stored by hour, so we duplicate each hour value 3 times
	 * and store the same value 4 times in the returned array (once for each 15 minute interval).
	 * The DST-adjustment will add or remove values in the array if the SP week crosses a DST transition
	 * (see DSTConversionUtil for more info).
	 */
	public static float[] getShrinkageValuesFromSPShrinkage(Collection<SPShrinkage> spShrinkage, SchedulingPeriod sp,
		TimeContext timeContext) {
		//We need to create an array for shrinkage values
		//The array size is the number of 15 minute intervals present in the scheduling period
		float[] shrinkageValues = new float[spShrinkage.size() * 96];
		int dayIndex = 0;
		for (SPShrinkage sps : spShrinkage) {
			float[] dayShrinkageValues = sps.getShrinkageValuesFor15MinuteIntervals();
			System.arraycopy(dayShrinkageValues, 0, shrinkageValues, 96 * dayIndex, 96);
			dayIndex++;
		}

		//Shrinkage and modeling factors are stored in the DB in "ideal" time (24 hours each day)
		//This causes issues when we are in a period with a DST transition.  DSTConversionUtil will
		//add values (in the case where we fall back an hour) or remove the extra values when we jump forward
		//an hour, so that the number of intervals will match what is in the forecast/service goals trace data.
		shrinkageValues = DSTConversionUtil.convertFromIdealTimeToDSTAwareTime(
			shrinkageValues, 0, sp.getStartTime(), Duration.fromMinutes(Trace.INTERVAL), timeContext);

		return shrinkageValues;
	}

	/**
	 * Creates and returns a DST-adjusted array of modeling factor data extracted from the
	 * given spShrinkage objects.  Each value in the array represents a 15-minute interval of
	 * modeling factor data.  Modeling factors are only stored by hour, so we duplicate each hour value 3 times
	 * and store the same value 4 times in the returned array (once for each 15 minute interval).
	 * The DST-adjustment will add or remove values in the array if the sp week crosses a DST transition
	 * (see DSTConversionUtil for more info).
	 */
	public static float[] getModelingFactorsFromSPShrinkage(Collection<SPShrinkage> spShrinkage,
		SchedulingPeriod sp, TimeContext timeContext) {
		//We need to create an array for modeling factor values
		//The array size is the number of 15 minute intervals present in the scheduling period
		float[] modelingFactors = new float[spShrinkage.size() * 96];
		int dayIndex = 0;
		for (SPShrinkage sps : spShrinkage) {
			float[] dayModelingFactors = sps.getModelingFactorValuesFor15MinuteIntervals();
			System.arraycopy(dayModelingFactors, 0, modelingFactors, 96 * dayIndex, 96);
			dayIndex++;
		}

		//Shrinkage and modeling factors are stored in the DB in "ideal" time (24 hours each day)
		//This causes issues when we are in a period with a DST transition.  DSTConversionUtil will
		//add values (in the case where we fall back an hour) or remove the extra values when we jump forward
		//an hour, so that the number of intervals will match what is in the forecast/service goals trace data.
		modelingFactors = DSTConversionUtil.convertFromIdealTimeToDSTAwareTime(
			modelingFactors, 0, sp.getStartTime(), Duration.fromMinutes(Trace.INTERVAL), timeContext);

		return modelingFactors;
	}
}
