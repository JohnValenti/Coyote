package com.verint.web.rest.util;

import java.util.Date;

import javax.ws.rs.core.Response;

import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.witness.web.uif.system.RequestContext;

public final class FsRestUtil {

	private FsRestUtil() { // NOSONAR

	}

	/**
	 * Gets a date for the parameter. If one doesn't exist a BusinessWFOException is thrown with BAD_REQUEST status.
	 */
	public static Date getDate(RequestContext context, String paramName)
			throws BusinessWFOException {
		Date date = FormatUtils.iso8601SecPrecisionStringToDate(context.getParameter(paramName));
		if (date == null) {
			String msg = context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.PARAMETER_IS_REQUIRED, paramName);
			throw new BusinessWFOException(msg, Response.Status.BAD_REQUEST);
		}
		return date;
	}
}
