package com.verint.web.fs.optimization;

import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.workload.model.Media;

import java.util.Collection;

public abstract class PartialResponse {

	private final Campaign m_campaign;
	private final SchedulingPeriod m_sp;
	private final Collection<CampaignHOO> m_hoo;
	private final Media m_media;

	public PartialResponse(Campaign campaign, SchedulingPeriod sp, Media media,
			Collection<CampaignHOO> hoo) {
		m_campaign = campaign;
		m_sp = sp;
		m_media = media;
		m_hoo = hoo;
	}

	public Campaign getCampaign() {
		return m_campaign;
	}

	public SchedulingPeriod getSp() {
		return m_sp;
	}

	public Collection<CampaignHOO> getHoursOfOperation() {
		return m_hoo;
	}

	public Media getMedia() {
		return m_media;
	}
}
