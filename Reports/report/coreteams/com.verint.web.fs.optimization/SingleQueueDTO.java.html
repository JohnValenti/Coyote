package com.verint.web.fs.optimization;

import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;

import java.io.Serializable;

/**
 * The Class QueueDTO. This is a class that represents a basic composite
 * queue/sp queue pair that will be sent to clients from the server.
 */
public abstract class SingleQueueDTO implements Serializable {

	private static final long serialVersionUID = 5583853312723523404L;

	private final Queue m_queue;
	private final SPQueue m_spQueue;

	public SingleQueueDTO(Queue queue, SPQueue spQueue) {
		m_queue = queue;
		m_spQueue = spQueue;
	}

	public Queue getQueue() {
		return m_queue;
	}

	public SPQueue getSPQueue() {
		return m_spQueue;
	}
}
