package com.verint.web.fs.optimization;

import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.workload.model.Media;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;

import java.util.Collection;

public class SingleQueuePartialResponse extends PartialResponse {

	private final SPQueue m_spQueue;
	private final Queue m_queue;

	public SingleQueuePartialResponse(Campaign campaign, SchedulingPeriod sp,
			Media media, Collection<CampaignHOO> hoo, Queue queue, SPQueue spQueue) {
		super(campaign, sp, media, hoo);
		m_spQueue = spQueue;
		m_queue = queue;
	}

	public SPQueue getSPQueue() {
		return m_spQueue;
	}

	public Queue getQueue() {
		return m_queue;
	}
}
