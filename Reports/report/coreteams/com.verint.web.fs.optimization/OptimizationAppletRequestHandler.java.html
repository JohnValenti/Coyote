public class OptimizationAppletRequestHandler extends AppletRH {

	/**
	 * Process Applet Request - to be overridden by derived classes
	 *
	 * @param context - The Request Context
	 * @param appletRequest - Map containing Applet Request Information
	 * @param appletResponse - Map containing Response to be send to Applet
	 */
	private static final Category cat = Log.initCategory(OptimizationAppletRequestHandler.class.getName());

	@Override
	protected void processAppletRequest(RequestContext context, Map<? extends Object, ? extends Object> appletRequest, Map<Object, Object> appletResponse) {
		if (isAppletAction(appletRequest, OptimizationKeys.GET_MEDIA)) {
			prepareMedia(context, appletRequest, appletResponse, false);
		} else if (isAppletAction(appletRequest, OptimizationKeys.CHECK_CAMPAIGN_SP_SELECTION_VALIDITY)) {
			isSelectionValid(context, appletRequest, appletResponse);
		} else if (isAppletAction(appletRequest, OptimizationKeys.GET_IS_TIMEZONE_OVERRIDDEN)) {
			isTimeZoneOverridden(context, appletResponse);
		} else if (isAppletAction(appletRequest, OptimizationKeys.TOGGLE_TIMEZONE_OVERRIDE)) {
			toggleTimeZoneOverride(context, appletResponse);
		} else {
			super.processAppletRequest(context, appletRequest, appletResponse);
		}
	}

	protected void isSelectionValid(RequestContext context, Map appletRequest, Map appletResponse) {
		ID spId = (ID) appletRequest.get(OptimizationKeys.PARAM_SP_ID);
		ID campaignId = (ID) appletRequest.get(OptimizationKeys.PARAM_CAMPAIGN_ID);
		ArrayList<ID> queueIds = (ArrayList<ID>) appletRequest.get(OptimizationKeys.PARAM_QUEUE_IDS);
		boolean valid = true;
		try {
			WorkloadManager workloadManager = WfmManagerFactory.getWorkloadManager(context.isInWhatIfMode());
			SchedulingPeriod sp = CampaignModelHandler.getSchedulingPeriodByID(context, spId);
			if (sp == null) {
				appletResponse.put(OptimizationKeys.CAMPAIGN_SP_SELECTION_VALIDITY, false);
				return;
			}
			Campaign campaign = CampaignModelHandler.getCampaign(context, campaignId);
			if (campaign == null) {
				appletResponse.put(OptimizationKeys.CAMPAIGN_SP_SELECTION_VALIDITY, false);
				return;
			}
			for (ID queueId : queueIds) {
				Queue queue = workloadManager.getQueueByID(queueId);
				if (queue == null) {
					valid = false;
					break;
				}
				SPQueue spQueue = CampaignModelHandler.getSPQueue(context, spId, queueId);
				if (spQueue == null) {
					valid = false;
					break;
				}
			}
		} catch (Exception e) {
			valid = false;
			notifyError(context, appletResponse, e);
		}
		appletResponse.put(OptimizationKeys.CAMPAIGN_SP_SELECTION_VALIDITY, valid);
	}

	private void prepareMedia(RequestContext context, Map appletRequest, Map appletResponse, boolean storeResult) {
		try {
			ID mediaId = (ID) appletRequest.get(OptimizationKeys.PARAM_MEDIA_ID);
			Media media = OptimizationModelHandler.getMedia(context, mediaId);
			appletResponse.put(OptimizationKeys.MEDIA, media);
		} catch (Exception e) {
			notifyError(context, appletResponse, e);
		}
	}

	/**
	 * Indicates to the applet that a generic error has occurred, with a generic "unable to load data" message sent.
	 * If the applet wishes to use this message, the calling code (generally the server proxy class) should call the
	 * displayErrorMessagesFromServer method on the proxy class itself.
	 */
	protected void notifyError(RequestContext context, Map appletResponse, Exception ex) {
		cat.l7dError(FsWebBundleKey.UNABLE_TO_LOAD_DATA, ex);
		context.addPageMessage(new Message(Message.ERROR_TYPE, FsWebBundleKey.UNABLE_TO_LOAD_DATA, FsWebBundleKey.BUNDLE_NAME));
		notifyClientOfError(appletResponse);
	}

	/**
	 * Indicates to the applet that a generic error has occurred, with a specific error message sent.
	 * If the applet wishes to use this message, the calling code (generally the server proxy class) should call the
	 * displayErrorMessagesFromServer method on the proxy class itself.
	 */
	protected void notifyError(RequestContext context, Map appletResponse, Message errorMessage, Exception ex) {
		cat.l7dError(FsWebBundleKey.UNABLE_TO_LOAD_DATA, ex);
		context.addPageMessage(errorMessage);
		notifyClientOfError(appletResponse);
	}

	protected static SingleQueuePartialResponse createSingleQueuePartialResponse(RequestContext context, Map appletRequest)
			throws RemoteException, CoreFinderException, CoreEJBCreateException,
			BbmException {
		ID spId = (ID) appletRequest.get(OptimizationKeys.PARAM_SP_ID);
		ID mediaId = (ID) appletRequest.get(OptimizationKeys.PARAM_MEDIA_ID);
		ID campaignId = (ID) appletRequest.get(OptimizationKeys.PARAM_CAMPAIGN_ID);
		ID queueId = (ID) appletRequest.get(OptimizationKeys.PARAM_QUEUE_ID);

		Media media = OptimizationModelHandler.getMedia(context, mediaId);
		Queue queue = OptimizationModelHandler.getQueue(context, queueId);
		SchedulingPeriod sp = CampaignModelHandler.getSchedulingPeriodByID(context, spId);
		SPQueue spQueue = CampaignModelHandler.getSPQueue(context, spId, queueId);
		Campaign campaign = CampaignModelHandler.getCampaign(context, campaignId);
		Collection<CampaignHOO> hoo = CampaignModelHandler.getCampaignHOOAssignmentsForSp(context, sp.getID());
		return new SingleQueuePartialResponse(campaign, sp, media, hoo, queue, spQueue);
	}

	protected static CombinedQueuePartialResponse createCombinedQueuePartialResponse(RequestContext context, Map appletRequest) throws Exception {
		ID spId = (ID) appletRequest.get(OptimizationKeys.PARAM_SP_ID);
		ID mediaId = (ID) appletRequest.get(OptimizationKeys.PARAM_MEDIA_ID);
		ID campaignId = (ID) appletRequest.get(OptimizationKeys.PARAM_CAMPAIGN_ID);

		Media media = OptimizationModelHandler.getMedia(context, mediaId);
		SchedulingPeriod sp = CampaignModelHandler.getSchedulingPeriodByID(context, spId);
		CampaignManager campaignManager = WfmManagerFactory.getCampaignManager(context.isInWhatIfMode());
		SPQueue combinedSPQueue = campaignManager.getCombinedSPQueue(sp.getID(), mediaId);
		Campaign campaign = CampaignModelHandler.getCampaign(context, campaignId);
		Collection<CampaignHOO> hoo = CampaignModelHandler.getCampaignHOOAssignmentsForSp(context, sp.getID());

		WorkloadManager workLoadManager = WfmManagerFactory.getWorkloadManager(context.isInWhatIfMode());
		List<Queue> queues = new ArrayList<Queue>(workLoadManager.getNonCombinedQueuesByMediaAndSP(mediaId, sp.getID()));
		List<SPQueue> spQueues = new ArrayList<SPQueue>(campaignManager.getNonCombinedSPQueuesByMediaAndSP(sp.getID(), mediaId));
		List<SPQueue> orderedSpQueues = sortSpQueuesByQueueOrder(queues, spQueues);

		return new CombinedQueuePartialResponse(campaign, sp, media, hoo, queues, orderedSpQueues, combinedSPQueue);
	}

	protected static MultiQueuePartialResponse createMultiQueuePartialResponse(RequestContext context, Map appletRequest) throws Exception {
		ID spId = (ID) appletRequest.get(OptimizationKeys.PARAM_SP_ID);
		ID mediaId = (ID) appletRequest.get(OptimizationKeys.PARAM_MEDIA_ID);
		ID campaignId = (ID) appletRequest.get(OptimizationKeys.PARAM_CAMPAIGN_ID);
		Collection<ID> queueIds = (Collection<ID>) appletRequest.get(OptimizationKeys.PARAM_QUEUE_IDS);

		List<Queue> queues = new ArrayList<Queue>();
		List<SPQueue> spQueues = new ArrayList<SPQueue>();

		WorkloadManager workloadManager = WfmManagerFactory.getWorkloadManager(context.isInWhatIfMode());

		for (ID queueId : queueIds) {
			Queue queue = workloadManager.getQueueByID(queueId);
			SPQueue spQueue = CampaignModelHandler.getSPQueue(context, spId, queueId);
			queues.add(queue);
			spQueues.add(spQueue);
		}

		Media media = OptimizationModelHandler.getMedia(context, mediaId);
		SchedulingPeriod sp = CampaignModelHandler.getSchedulingPeriodByID(context, spId);
		Campaign campaign = CampaignModelHandler.getCampaign(context, campaignId);
		Collection<CampaignHOO> hoo = CampaignModelHandler.getCampaignHOOAssignmentsForSp(context, sp.getID());

		return new MultiQueuePartialResponse(campaign, sp, media, hoo, queues, spQueues);
	}

	/**
	 * This method operates similarly to {@link #setUndefinedValuesToZero(TraceCube)}
	 * except that an AHT value is set to undefined if the corresponding Volume value
	 * is nonpositive, and to zero if it was previously negative and the corresponding
	 * Volume value is positive.
	 *
	 * @param cube
	 */
	protected static void adjustVolumeAhtForecastTraceValues(TraceCube cube) {
		for (short type : cube.getTraceTypes()) {
			boolean isDirty = cube.isDirty(type);
			if (type == Trace.AHT) {
				double[] values = cube.getTraceValueD(type);
				double[] weights = cube.getTraceValueD(Trace.CV);
				if (weights == null) {
					weights = cube.getTraceValueD(Trace.CV_VH);
					if (weights == null) {
						// Can't adjust AHT without weights.
						continue;
					}
				}
				for (int i = 0; i <= values.length - 1; i++) {
					if (weights[i] <= 0 && values[i] >= 0) {
						values[i] = Trace.TRACENA;
					} else if (weights[i] > 0 && values[i] < 0) {
						values[i] = 0;
					}
				}
				cube.setTraceValue(type, values);
			} else {
				double[] values = cube.getTraceValueD(type);
				for (int i = 0; i <= values.length - 1; i++) {
					if (values[i] < 0) {
						values[i] = 0;
					}
				}
				cube.setTraceValue(type, values);
			}
			//This is done here because setting the undefined values to zero
			//  has the unfortunate side effect of setting the dirty flag.  So,
			//  we will clear out the dirty flag caused by this change unless
			//  the cube was already dirty beforehand.
			if (!isDirty) {
				cube.setTypeDirty(type, false);
			}
		}
	}

	/**
	 * Returns true if the viewing time zone is overridden (i.e. we are viewing
	 * data in the user's time zone defined in user prefs).  If it is not overridden,
	 * then we are viewing data in the campaign's time zone.
	 */
	protected void isTimeZoneOverridden(RequestContext context, Map appletResponse) {
		try {
			appletResponse.put(OptimizationKeys.IS_TIMEZONE_OVERRIDDEN_RETVAL,
					SelectionCacheUtil.isTimeZoneOverridden(context));
		} catch (Exception e) {
			notifyError(context, appletResponse, e);
		}
	}

	/**
	 * Toggles the cached viewing time zone override between on/off states.
	 * If it is on, that means the user will be viewing data in their time
	 * zone (defined in user preferences).  If it is off, then the user will
	 * be viewing data in the selected campaign's time zone.
	 */
	protected void toggleTimeZoneOverride(RequestContext context, Map appletResponse) {
		try {
			SelectionCacheUtil.toggleTimeZoneOverride(context);
			appletResponse.put(OptimizationKeys.DEFAULT_RESPONSE, "Success!");
		} catch (Exception e) {
			notifyError(context, appletResponse, e);
		}
	}

	/**
	 * Sorts the given list of SPQueues to match the order of the given list of queues.  For example,
	 * if the queues list has queues in the following order: [5, 10, 8], but spQueues list has queues in
	 * the following order: [10, 8, 5] then the returned list of spqueues will have the following order to
	 * match the queues list: [5, 10, 8].
	 */
	static List<SPQueue> sortSpQueuesByQueueOrder(List<Queue> queues, List<SPQueue> spQueues) {
		Map<ID, SPQueue> spQueueMap = ValueObjectUtil.getFieldObjectMap(SPQueueFieldInfo.SPQUEUE_QUEUEID, spQueues);
		List<SPQueue> retVal = new ArrayList<SPQueue>();

		for (Queue queue : queues) {
			retVal.add(spQueueMap.get(queue.getID()));
		}

		return retVal;
	}
