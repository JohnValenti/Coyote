	/**
	 * Gets next question that needs to be asked to create or edit an Assignment Rule, or
	 * null if no further questions are needed.
	 *
	 * @param dataProvider
	 * @return
	 * @throws BPException
	 */
	public static Question getNextQuestion(QuestionDataProvider dataProvider) throws BPException {
		QuestionFactoryState state = new QuestionFactoryState();
		Question nextQuestion = getQuestion(QuestionType.RuleType, dataProvider, state);
		List<Pair<QuestionType, Response>> answeredQuestions = dataProvider.getAnswers();
		if (!answeredQuestions.isEmpty()) {
			Iterator<Pair<QuestionType, Response>> iter = answeredQuestions.iterator();
			RuleTypeResponse response = (RuleTypeResponse) iter.next().getSecond();
			RuleTypeResponseType ruleType = response.getResponseType();

			QuestionType nextType = getNextQuestionType(ruleType, QuestionType.RuleType, ruleType, state);
			nextQuestion = getQuestion(nextType, dataProvider, state);

			// We have the first question. Now look through the passed in answers until we have a question that has not
			// been answered.
			while (iter.hasNext()) {
				Pair<QuestionType, Response> selectionPair = iter.next();
				QuestionType questionType = selectionPair.getFirst();
				if (questionType != nextQuestion.getQuestionType()) {
					// The answer is not supplied. Need to ask this question
					break;
				} else {
					// The answer is passed in, check next question
					nextType = getNextQuestionType(ruleType, nextQuestion.getQuestionType(), selectionPair
							.getSecond()
							.getResponseType(), state);
					nextQuestion = getQuestion(nextType, dataProvider, state);
				}
			}

		}
		return nextQuestion;
	}

	private static Question getQuestion(QuestionType type, QuestionDataProvider dataProvider, QuestionFactoryState state)
			throws BPException {
		if (type == null) {
			return null;
		}
		try {
			ComplexWorkRule assignmentRule = dataProvider.getAssignmentRuleBeingEdited();
			Question nextQuestion = null;
			switch (type) {
				case RuleType:
					nextQuestion = new RuleTypeQuestion(dataProvider);
					break;
				case TeamConstraintType:
					nextQuestion = new TeamConstraintQuestion(assignmentRule);
					break;
				case TeamMatchType:
					nextQuestion = new TeamMatchQuestion(assignmentRule);
					break;
				case Count:
					nextQuestion = new CountQuestion(assignmentRule);
					break;
				case TimeOfDay:
				case TimeOfDayAndBetween:
					nextQuestion = new TimePeriodOfDayQuestion(
							assignmentRule,
							dataProvider.getTimePeriods(),
							type == QuestionType.TimeOfDayAndBetween);
					break;
				case WorkUnit:
					nextQuestion = new WorkUnitQuestion(dataProvider, state);
					break;
				case StartUnit:
					nextQuestion = new StartUnitQuestion(dataProvider);
					break;
				case AlignmentDate:
					nextQuestion = new AlignmentDateQuestion(dataProvider);
					break;
				case CalendarPeriod:
					nextQuestion = new PeriodQuestion(dataProvider);
					break;
				case StartConstraint:
					nextQuestion = new StartConstraintQuestion();
					break;
				case EndDate:
					nextQuestion = new EndDateQuestion(dataProvider);
					break;
				case Tolerance:
				case TeamTolerance:
					nextQuestion = new ToleranceQuestion(dataProvider);
					break;
				default:
					throw new IllegalArgumentException("Unknown Question Type: " + type);
			}
			return nextQuestion;
		} catch (Exception e) {
			if (e instanceof BPException) {
				throw (BPException) e;
			} else {
				throw new BPException(e);
			}
		}
	}

	/**
	 * The getNextSelector must store some state from selection to selection. These are the pieces of data (flags) that
	 * get stored
	 */
	static enum StateCondition {
		Consecutive, RequiresTolerance
	}

	/**
	 * This object represents state that is passed from one invocation of the getNextSelector to the next.
	 *
	 * @author rfunderburg
	 */
	static final class QuestionFactoryState {
		private HashSet<StateCondition> state = new HashSet<StateCondition>();

		private boolean add(StateCondition arg0) {
			return state.add(arg0);
		}

		boolean contains(StateCondition arg0) {
			return state.contains(arg0);
		}

		private boolean remove(StateCondition arg0) {
			return state.remove(arg0);
		}

	}

	/**
	 * This method returns the next required selector type, or null if no further selectors are required.
	 * <p>
	 * This method was copied from CSentence::NextTerms in the C++ client and refactored.
	 *
	 * @param lastAnsweredQuestionType
	 * @param ruleType
	 * @param state                    The same State object must be passed in for each call to getNextSelector.
	 * @return
	 */
	private static QuestionType getNextQuestionType(RuleTypeResponseType ruleType,
			QuestionType lastAnsweredQuestionType, ResponseType lastAnsweredResponseType, QuestionFactoryState state) {
		QuestionType nextQuestionType = lastAnsweredQuestionType;

		switch (lastAnsweredQuestionType) {
			default:
				nextQuestionType = QuestionType.RuleType;
				break;
			case RuleType:
				if (lastAnsweredResponseType != null) {
					RuleTypeResponseType rtrt = (RuleTypeResponseType) lastAnsweredResponseType;
					switch (rtrt) {
						case MustWork:
							nextQuestionType = QuestionType.Count;
							break;
						case Off:
							nextQuestionType = QuestionType.Count;
							break;
						case Team:
							nextQuestionType = QuestionType.TeamConstraintType;
							break;
						case Start:
							nextQuestionType = QuestionType.StartUnit;
							break;
					}
				}
				break;

			case StartUnit: {
				if (ruleType == RuleTypeResponseType.Start) {
					nextQuestionType = QuestionType.StartConstraint;
				} else {
					nextQuestionType = QuestionType.TeamConstraintType;
				}
				break;
			}

			case StartConstraint: {
				state.remove(StateCondition.RequiresTolerance);
				switch ((StartConstraintResponseType) lastAnsweredResponseType) {
					case SameTime:
						nextQuestionType = QuestionType.CalendarPeriod;
						state.add(StateCondition.RequiresTolerance);
						break;
					default:
						throw new IllegalArgumentException("Unexpected response type: " + lastAnsweredResponseType);
				}
				break;
			}

			case TeamConstraintType: {
				if (lastAnsweredResponseType != null) {
					TeamConstraintResponseType trtrt = (TeamConstraintResponseType) lastAnsweredResponseType;
					nextQuestionType = QuestionType.TeamMatchType;
					state.remove(StateCondition.RequiresTolerance);
					switch (trtrt) {
						case TeamStart: // start_at_same_time:
						case TeamStartSameDays: // start_at_same_time_same_days:
						case TeamStartAndEnd: // start_end_at_same_time:
						case TeamStartAndEndSameDays: // start_end_at_same_time_same_days:
							state.add(StateCondition.RequiresTolerance);
							break;
					}
				}
				break;
			}
			case TeamMatchType: {
				if (lastAnsweredResponseType != null) {
					TeamMatchResponseType tmrt = (TeamMatchResponseType) lastAnsweredResponseType;
					switch (tmrt) {
						case AllOther: // as_everyone:
						case Supervisor: // as_supervisor:
						case TeamLead: // as_lead:
							if (state.contains(StateCondition.RequiresTolerance)) {
								nextQuestionType = QuestionType.TeamTolerance;
							} else {
								nextQuestionType = null;
							}
							break;
						default:
							nextQuestionType = null;
							break;
					}
				}
				break;
			}
			case Count: {
				if (lastAnsweredResponseType != null) {
					CountResponseType crt = (CountResponseType) lastAnsweredResponseType;
					switch (crt) {
						case FairNumber: // fairly:
							nextQuestionType = QuestionType.WorkUnit;
							state.add(StateCondition.RequiresTolerance);
							break;
						default:
							nextQuestionType = QuestionType.WorkUnit;
							state.remove(StateCondition.RequiresTolerance);
							break;
					}
				}
				break;
			}
			case WorkUnit: {
				if (lastAnsweredResponseType != null) {
					WorkUnitResponseType wurt = (WorkUnitResponseType) lastAnsweredResponseType;
					switch (wurt) {
						case MinShiftSpacing: // min_shift_spacing:
						case MinShiftEventSpacing: // min_shiftevent_spacing:
						case ConsecutiveDayOfWeekOff: // consecutive_dayofweekoff:
						case ConsecutiveDayOff: // consecutive_dayoff:
						case ConsecutiveFullWeekendOff: // consecutive_fullweekendoff:
						case ConsecutivePartialWeekendOff: // consecutive_partialweekendoff:
						case ConsecutiveDayOfWeek: // consecutive_dayofweek:
						case ConsecutiveDay: // consecutive_day:
						case ConsecutiveShift: // consecutive_shift:
						case ConsecutiveFullWeekend: // consecutive_fullweekend:
						case ConsecutivePartialWeekend: // consecutive_partialweekend:
							state.add(StateCondition.Consecutive);
							break;
						default:
							state.remove(StateCondition.Consecutive);
							break;
					}

					if (responsesRequiringTimeOfDay.contains(wurt)) {
						if (state.contains(StateCondition.RequiresTolerance)) // HACK (if fairness
						{
							nextQuestionType = QuestionType.TimeOfDayAndBetween;
						} else {
							nextQuestionType = QuestionType.TimeOfDay;
						}
					} else {
						if (state.contains(StateCondition.Consecutive)) {
							nextQuestionType = null;
						} else {
							nextQuestionType = QuestionType.CalendarPeriod;
						}
					}
				}
				break;
			}
			case TimeOfDay:
			case TimeOfDayAndBetween:
				if (state.contains(StateCondition.Consecutive)) {
					nextQuestionType = null;
				} else {
					nextQuestionType = QuestionType.CalendarPeriod;
				}
				break;
			case CalendarPeriod: {
				if (lastAnsweredResponseType != PeriodQuestion.PeriodResponseType.RollingWeek) {
					nextQuestionType = QuestionType.AlignmentDate;
					break;
				}
			}

			case AlignmentDate:
				nextQuestionType = QuestionType.EndDate;
				break;
			case EndDate:
				if (state.contains(StateCondition.RequiresTolerance)) {
					nextQuestionType = QuestionType.Tolerance;
				} else {
					nextQuestionType = null;
				}
				break;
			case Tolerance:
				nextQuestionType = null;
				break;
			case TeamTolerance:
				nextQuestionType = null;
				break;
		}

		return nextQuestionType;
	}
