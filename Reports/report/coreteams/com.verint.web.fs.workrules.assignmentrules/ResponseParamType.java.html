package com.verint.web.fs.workrules.assignmentrules;

import com.bluepumpkin.common.datatypes.DayOfWeek;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.datatypes.TimeOfDay;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;

import java.util.Collection;

/**
 * Represents the different types of data that can be collected as part of a response.
 */
public enum ResponseParamType {
	Shift(Shift.class), Integer(Integer.class), DayOfWeek(DayOfWeek.class), Activities(Collection.class), Date(
			LocalDate.class), TimeOfDay(TimeOfDay.class);
	private Class<?> javaType;

	ResponseParamType(Class<?> javaType) {
		this.javaType = javaType;
	}

	public Class<?> getJavaType() {
		return this.javaType;
	}
}
