	/**
	 * create skill
     * @return the DE string ID
	 */
	public ID createSkill(Skill objValue) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();
		try {
			//get ID from DE
		    String idStr = DAOUtil.getDEID(jdmo);

			String pStmt1 =
			    "insert into SKILL (ID, NAME, DESCRIPTION, SKILLACTIVE, MEDIAID, ORGANIZATIONID) values (?, ?, ?, ?, ?, ?)";

			Object[] param = new Object[6];
			param[0] = idStr;
			param[1] = objValue.getName();
			param[2] = objValue.getDescription();
			Integer val =  new Integer((objValue.getSkillActive() ? 1 : 0));
			param[3] = val;
			param[4] = Skill.convertMediaID2str(objValue.getMediaID());
            param[5] = objValue.getOrganizationID();

			jdmo.executePCommand(pStmt1, param);

            ID skillID = new ID(idStr);
			return skillID;
		} finally {
			if(jdmo!=null)jdmo.cleanUp();
		}
	}

    /**
	 * Remove a skills by SID's
	 * @param colIDs - a collection of Skill SID's to be deleted
	 */
	public void deleteSkills(Collection colIDs) throws JdmoException
	{
		if( colIDs == null || colIDs.size() <= 0 ) return;
		
		Jdmo jdmo = new Jdmo();
		try {
			StringBuffer pStmt = new StringBuffer("select ID from skill where SID in (");
			Iterator iter = colIDs.iterator();
			for( int i=0; i <  colIDs.size(); ++i  )
			{
				if( i > 0 ) pStmt.append(",");
				pStmt.append( iter.next() );
			}
			pStmt.append(")");
			
			JdmoQuery jq1 = m_dmo.createQuery(pStmt.toString(), Jdmo.STATIC_QUERY);
			JdmoRowset rs = m_dmo.createRowset(jq1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);		

			ArrayList skillIDs = new ArrayList(colIDs.size());
			if( rs != null )
			{
				while( rs.next() ) {
					skillIDs.add("'"+ rs.getString(1) + "'");
				}		
				rs.close();
			}
			jq1.close();

			if( skillIDs.size() > 0 )
				DAOUtil.deleteObjects(jdmo, "SKILL", skillIDs); //todo: shouldn't we be using DeleteSkilDAO for this?
			
		} finally {
			if(jdmo!=null)jdmo.cleanUp();
		}
	}

	/**
	 *  get skills in a list of given orgs and all their parent orgs
	 */
	public Collection getAvailableSkillsInOrg(Collection colOrgIDs)
		throws BbmFinderException
	{
		if (colOrgIDs == null || colOrgIDs.isEmpty()) return new ArrayList();

		HashSet setIDs = new HashSet(colOrgIDs);
		setIDs.addAll(DAOUtil.getParentOrganizationsCollection(colOrgIDs, m_dmo));

		StringBuffer strSQL = new StringBuffer();
		try
		{
			strSQL.append(" A.DELETEONDATE IS NULL AND A.ORGANIZATIONID IN ").append(m_dmo.createInClause(setIDs));
			return getObjects(new String(strSQL));
		}
		catch(Exception e)
		{
			throw new BbmFinderException(e);
		}
	}

	/**
	 *   see if a skill is assigned to a queue or an employee
	 */
	public boolean isSkillAssigned(ID idSkill) throws BbmFinderException
	{
		if (idSkill == null) return false;

		try
		{
			if (DAOUtil.dataExist("select 1 from SKILLQUEUE where SKILLID = " + idSkill, m_dmo))
				 return true;

			if (DAOUtil.dataExist("select 1 from WORKRESOURCEASSIGNMENT where SKILLID = " + idSkill, m_dmo))
				 return true;
		}
		catch(Exception e)
		{
			throw new BbmFinderException(e);
		}

		return false;
	}
	
	/**
	 * Returns a collection of skills with the given where clause.
	 * The skills returned have an integer based ID (SID).
	 * NOTE: the where clause should be prefixed with an "AND" as it will
	 * not form the first part of the where clause and will be appended to an
	 * existing where clause.
	 */
	public Collection<Skill> getSkills(String whereClause) throws BbmFinderException {
		try {
			//load all skills
			ArrayList<Skill> aSkills = new ArrayList<Skill>();
	
			String query = 
            "SELECT A.ID, A.SID, A.NAME, A.DESCRIPTION, A.SKILLACTIVE, A.MEDIAID, A.ORGANIZATIONID, B.NAME ORGANIZATIONNAME " +
            "FROM SKILL A, ORGANIZATION B " + 
            "WHERE A.ORGANIZATIONID = B.ID " +           
			"AND A.DELETEONDATE IS NULL ";
			if (whereClause != null) query = query + whereClause;
			JdmoRowset r = m_dmo.createRowset( query );
	
			//-------------------------------------------------
			// get data and create skills
			//-------------------------------------------------
	
			while(r.next()) {	
				String idStr = r.getString("ID");
				ID idSkill = r.getID("SID");
				String sName = r.getString("NAME");
				String sDescription = r.getString("DESCRIPTION");
                int iActive = r.getInt("SKILLACTIVE");
                String idMediaID = r.getString("MEDIAID");
                ID idOrgID = r.getID("ORGANIZATIONID");
                String sOrgName = r.getString("ORGANIZATIONNAME");
                
				Skill pSkill = new Skill();
                pSkill.setIDStr(idStr);
                pSkill.setID(idSkill);
				pSkill.setName(sName);
				pSkill.setDescription(sDescription);
                pSkill.setSkillActive(iActive==1);
                pSkill.setMediaID(Skill.convertStr2MediaID(idMediaID));
                pSkill.setOrganizationID(idOrgID);
                pSkill.setOrganizationName(sOrgName);

				aSkills.add(pSkill);
			}				
			return aSkills;	
		} catch(JdmoException e) {	
			throw new BbmFinderException(e);
		} finally {
			m_dmo.cleanUp();
		}		
	}

    /**
     *  get skill object given its name and mediaID
     */
    public Skill getSkillByNameAndMedia(String name, ID mediaID) throws BbmObjectNotFoundException, BbmFinderException
    {
        Skill skill = null;
        if (name == null || mediaID == null) return null;
        String whereClause = " A.NAME  = '" +  JdmoUtil.formatDBString(name) + "' AND A.MEDIAID = '" + Skill.convertMediaID2str(mediaID) + "' ";

        Collection skills = getObjects(whereClause);
        if (skills != null)
        {
            for (Iterator it=skills.iterator(); it.hasNext(); )
            {
                skill = (Skill)it.next();
                break;
            }
        }
        return skill;
    }
