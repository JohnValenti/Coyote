package com.bluepumpkin.ejb.am.rtaaexceptioncategory.ejb;

import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOEJBUtil;
import com.bluepumpkin.ejb.am.rtaaexceptioncategory.model.RTAAExceptionCategory;
import com.bluepumpkin.ejb.core.base.MultiUserException;
import com.bluepumpkin.ejb.core.base.SessionEJBBase;

/**
 * Title:        RTAAExceptionCategoryManagerEJB
 * Description:  ejb class to implement RTAAExceptionCategory interface
 * Copyright:    Copyright (c) 2006
 * Company:      Witness Systems, inc
 * @author       Swati Tewari
 * @version      1.0
 */

public class RTAAExceptionCategoryManagerEJB extends SessionEJBBase {

		private static Category m_cat = Log.initCategory( RTAAExceptionCategoryManagerEJB.class.getName());

		/** override the base class to provide the appropriate logging category */
		protected Category getCategory()
		{
			return m_cat;
		}

		{
			 super.init(RTAAExceptionCategoryManagerEJB.class.getName());
		}

		 /** get exception categories for given org collection */
		public Collection getRTAAExceptionCategories(Collection idOrgColl)
															 throws BbmFinderException
		{
				 methodStart("getRTAAExceptionCategories", idOrgColl);
				 RTAAExceptionCategoryDAO dao = new RTAAExceptionCategoryDAO();
				 try {
						return dao.getRTAAExceptionCategories(idOrgColl);
				 }
				 catch(BbmObjectNotFoundException e)
				 {
						handleException(e); // no rollback, since finder not support transaction
						throw e;
				 }
				 catch(BbmFinderException e)
				 {
						handleException(e);
						throw e;
				 }
				 finally
				 {
						dao.cleanUp();
						methodFinish();
				 }
		}

		/** get exception category of give id */
		public RTAAExceptionCategory getRTAAExceptionCategoryByID(ID idExceptionCategory)
												throws BbmFinderException, BbmObjectNotFoundException
		{
				 methodStart("getRTAAExceptionCategoryByID", idExceptionCategory);
				 RTAAExceptionCategoryDAO dao = new RTAAExceptionCategoryDAO();
				 try {
						return (RTAAExceptionCategory) dao.getObjectByID(idExceptionCategory);
				 }
				 catch(BbmObjectNotFoundException e)
				 {
						handleException(e); // no rollback, since finder not support transaction
						throw e;
				 }
				 catch(BbmFinderException e)
				 {
						handleException(e);
						throw e;
				 }
				 finally
				 {
						dao.cleanUp();
						methodFinish();
				 }
		}

		/** insert the exception category value object into database
		 *  @return id of the new object
		 */
		public ID createRTAAExceptioncategory(RTAAExceptionCategory objValue) throws BbmCreateException
		{
				 methodStart("createRTAAExceptioncategory", objValue);
				 RTAAExceptionCategoryDAO dao = new RTAAExceptionCategoryDAO();
				 try
				 {
						return DAOEJBUtil.createObject(dao, objValue, this, "createRTAAExceptioncategory");
				 }
				 finally
				 {
						methodFinish();
				 }
		}

		/** save the exception category information to database */
		public void updateRTAAExceptionCategory(RTAAExceptionCategory objValue) throws MultiUserException, BbmUpdateException
		{
				 methodStart("updateRTAAExceptionCategory", objValue);
				 RTAAExceptionCategoryDAO dao = new RTAAExceptionCategoryDAO();
				 try
				 {
						DAOEJBUtil.updateObject(dao, objValue, this, "updateRTAAExceptionCategory");
				 }
				 finally
				 {
						methodFinish();
				 }
		}

		/** delete exception categories given their ids */
		public void deleteRTAAExceptionCategories(Collection colRTAAExceptionCategoryIDs)
															 throws BbmRemoveException
		{
				 methodStart("deleteRTAAExceptionCategories", colRTAAExceptionCategoryIDs);
				 RTAAExceptionCategoryDAO dao = new RTAAExceptionCategoryDAO();
				 try
				 {
						DAOEJBUtil.deleteObjects(dao, colRTAAExceptionCategoryIDs, this, "deleteRTAAExceptionCategories");
				 }
				 finally
				 {
						methodFinish();
				 }

		}
}
