package com.bluepumpkin.web.fs.schedule.group;

import com.witness.web.uif.system.RequestContext;
import com.bluepumpkin.web.fs.schedule.ScheduleViewUtil;

import java.util.*;

/**
 * Title:        PeopleScheduleRH
 * 
 * Description:  Mainly used to decide which screen to show depending on viewType parameter, 
 *               then forward to that screen.
 *               
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, DSu
 * @version      1.0
 *
 * Created on Dec 9, 2003, 4:36:41 PM
 */
public class PeopleScheduleRH extends GroupScheduleRH {
	protected static final String SCREEN_MULTI_DAY = "PEOPLE_SCHEDULE_MULTIDAY";
	protected static final String SCREEN_TEXT = "PEOPLE_SCHEDULE_TEXT";
	protected static final String SCREEN_GRAPH = "PEOPLE_SCHEDULE_GRAPH";
	protected static final String SCREEN_MULTIDAY_GRAPH = "PEOPLE_SCHEDULE_MULTIDAY_GRAPH";

	protected static final HashMap FORM2SCREEN_MAP = new HashMap();
	static {
		FORM2SCREEN_MAP.put(GroupMultidaySchedulePM.FORM_ACTION, SCREEN_MULTI_DAY);
		FORM2SCREEN_MAP.put(GroupTextSchedulePM.FORM_ACTION, SCREEN_TEXT);
        FORM2SCREEN_MAP.put(GroupGraphSchedulePM.FORM_ACTION, SCREEN_GRAPH);
        FORM2SCREEN_MAP.put(GroupMultidayGraphSchedulePM.FORM_ACTION, SCREEN_MULTIDAY_GRAPH);
        FORM2SCREEN_MAP.put("people" + GroupMultidayGraphSchedulePM.FORM_ACTION, SCREEN_MULTIDAY_GRAPH);
	}

	/**
	 * Process the request.
	 */
	public void processRequest(RequestContext context) throws Exception {
		super.processRequest(context);

		// Determine which screen to use
		String formAction = ScheduleViewUtil.getCachedFormAction(context);
		String nextScreen = (String)FORM2SCREEN_MAP.get(formAction);
        
		if (nextScreen!=null)
        {
            if (nextScreen.lastIndexOf("GRAPH") != -1)
            {
                String selectedIDs = (String)context.getAttribute(RequestContext.SESSION_SCOPE, "PEOPLE_SELECTION_ID_KEY"); 
                if ((selectedIDs != null) && (selectedIDs.length() > 0) && (selectedIDs.lastIndexOf(",") == -1))
                    setNextScreenName(context, SCREEN_MULTIDAY_GRAPH); //exactly 1 employee selected
                else
                    setNextScreenName(context, SCREEN_GRAPH); //0 or many employees selected
            }
            else
            {
                setNextScreenName(context, nextScreen);
            }
		}
	}
}
