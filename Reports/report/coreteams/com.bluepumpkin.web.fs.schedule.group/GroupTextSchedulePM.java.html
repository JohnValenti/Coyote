	/** abstract method to prepare the m_list should be implemented by specific classes
	 *
	 */
	protected void populateList(HashMap theData) throws Exception
	{
		m_list.setEnabled(false);
		if (isAccessibilityComplianceMode()) {
			m_list.setRenderJavaScriptIfDisabled(true);
			//as result we will get div region as wrapper
			m_list.setUseWrapper(true);
			//add JavaScript files for ArrowNavigation
			m_list.addTableArrowNavigation();
			// add readable row info for reading whole row in grid
			m_list.setReadableRowInfoEnabled(true);
			//disable repeatHeader functionality
			m_list.disableRepeatHeader();
		}
		setListHeaders();

		if (theData==null || theData.isEmpty())
		    return;

		messageIfNotPublished(theData, m_dayRange.getStartDate(), m_context.getViewingTimeZone());
		Collection<Collection<TimeRange>> publishedPeriods = (Collection<Collection<TimeRange>>)theData.get(ScheduleViewMH.PUB_PERIODS_KEY);
		Collection<Collection<SimpleEvent>> schedule = (Collection<Collection<SimpleEvent>>) theData.get(ScheduleViewMH.SCHEDULE_ITEMS_KEY);
		Collection<Collection<DailyScheduleSummary>> cDaySummaries = (Collection<Collection<DailyScheduleSummary>>) theData.get(ScheduleViewMH.DAY_SUMMARIES_KEY);
		Collection<Collection<ShiftAssignment>> shifts = (Collection<Collection<ShiftAssignment>>) theData.get(ScheduleViewMH.SCHEDULE_SHIFTS_KEY);
		Collection<ShiftSwapPosting> postings = (Collection<ShiftSwapPosting>) theData.get(ScheduleViewMH.SWAP_BOARD_KEY);
		HashMap<ID, Activity> activities = (HashMap<ID, Activity>) theData.get(ScheduleViewMH.ACTIVITIES_KEY);
		HashMap<ID, EmployeeName> employeeNames = (HashMap<ID, EmployeeName>) theData.get(ScheduleViewMH.EMPLOYEE_NAMES_KEY);
		Collection<ID> employeeIDs = (Collection<ID>) theData.get(ScheduleViewMH.EMPLOYEE_IDS_KEY);

		Map<ID, String> campaignNamesBySpId = (Map<ID, String>) theData.get(ScheduleViewMH.CAMPAIGN_NAMES_BY_SP_ID_KEY);
		//Map campaignIdNames = (Map) theData.get(ScheduleViewMH.CAMPAIGN_NAMES_KEY);

		boolean bSwapEnabled = isSwapEnabled(theData);
		boolean bShowUnavailable = showUnavailable(theData);
		boolean bShowTimeoff = showTimeoff(theData);
		boolean bEnabledAbsentReasonToOnlySupervisor= isEnabledAbsentReasonToOnlySupervisor();
		ArrayList listModel = new ArrayList();
		//ID userEID = m_context.getUser().getEmployeeID();

		if (employeeIDs != null && schedule != null && shifts != null && publishedPeriods != null)
		{
			//--- for each employee create a row
			for (Iterator iEID = employeeIDs.iterator(),
					iSchedule = schedule.iterator(),
					iDaySummaries = cDaySummaries.iterator(),
					iShifts = shifts.iterator(),
					iPub = publishedPeriods.iterator(); iEID.hasNext();)
			{
				ID eID = (ID) iEID.next();
				boolean isMyRow = isMyID(eID);

				Collection<SimpleEvent> employeeSchedule = (Collection<SimpleEvent>) iSchedule.next();
				Collection<ShiftAssignment> employeeShifts = (Collection<ShiftAssignment>) iShifts.next();
				Collection<TimeRange> employeePubPeriods = (Collection<TimeRange>) iPub.next();
				Collection<DailyScheduleSummary> daySummaries = (Collection<DailyScheduleSummary>) iDaySummaries.next();
				EmployeeName employeeName = employeeNames.get(eID);

				DefaultMultiColumnNodeData row_data =
						makeRow(employeeName, employeePubPeriods, employeeSchedule, employeeShifts,
								postings, activities, daySummaries, bSwapEnabled, bShowUnavailable,
								bShowTimeoff, isMyRow, campaignNamesBySpId,bEnabledAbsentReasonToOnlySupervisor);

				listModel.add(row_data);
			}//for
		}

		m_list.setListModel(listModel);
	}//populateList

	/**
	 * set list headers for the multicolumn list
	 */
	private void setListHeaders(){

		TableHeaderPC header = m_list.getHeader();
        header.addColumnHeaderData(null,i18n(m_common_bundle,UIFWebBundleKey.NAME));
        header.addColumnHeaderData(null, verticalize(m_bbm_bundle,BbmWebBundleKey.CAMPAIGN));
		header.addColumnHeaderData(null,i18n(m_bbm_bundle,BbmWebBundleKey.SHIFT));
		header.addColumnHeaderData(null,i18n(m_bbm_bundle,BbmWebBundleKey.ACTIVITIES));
		header.addColumnHeaderData(null,  verticalize(m_bbm_bundle,BbmWebBundleKey.ACTIVITYS_CAMPAIGN_LIST_HEADER));
	}//setListHeaders

	/**
	 * make one row for the schedule table
	 */
	private DefaultMultiColumnNodeData makeRow(EmployeeName employeeName, Collection<TimeRange> employeePublishedPeriods,
		Collection<SimpleEvent> employeeSchedule, Collection<ShiftAssignment> employeeShifts, Collection<ShiftSwapPosting> postings,
		HashMap<ID, Activity> activities, Collection<DailyScheduleSummary> daySummaries, boolean bSwapEnabled,
		boolean bShowUnavailable, boolean bShowTimeoff, boolean isMyRow, Map<ID, String> campaignNamesBySpID, boolean isEnabledAbsentReasonToOnlySupervisor)
	{
		DefaultMultiColumnNodeData rowData = new DefaultMultiColumnNodeData();

		//--- create the Name column ---
		String name = (employeeName!=null)?m_localizer.formatFullName(employeeName):"";
		if (isMyRow)
            name = highlightName(name);

		rowData.add(name, CSSUtil.STYLE_NORMAL_LEFT); //508

		//-- make the Campaign, Shift, and Activities columns
		CalendarRange aDay = new CalendarRange(m_dayRange.getStartDate(),m_dayRange.getEndDate(),
				m_context.getViewingTimeZone(),m_locale);


		ID orgID = ScheduleViewMH.getOrgID(daySummaries, aDay);
		boolean bShowNote = (isMyRow)? ScheduleViewMH.getShowMyScheduleNote(m_context, orgID) :
			ScheduleViewMH.getShowOrgScheduleNote(m_context, orgID);



		if (!dayIsPublished(aDay,employeePublishedPeriods))
		{
            rowData.add(""); //no Campaign
			rowData.add(i18n(m_bundle,FsWebBundleKey.SV_NOT_PUBLISHED)); //Shift = "Not published"
            String eventString = this.makeEventsString(aDay, employeeSchedule, activities, bShowUnavailable, bShowTimeoff, bShowNote, isMyRow);
            rowData.add(eventString);
            //Activity campaign column should be blank for unpublished rows
            rowData.add(HtmlUtil.NBSP);
		}
		else
		{
			ID empID = employeeName.getID();
			StringsPair results = makeShiftStringWithAccessibility(aDay, employeeShifts, activities, postings, daySummaries, bSwapEnabled,
                    isMyRow, empID, bShowTimeoff, campaignNamesBySpID, isAccessibilityComplianceMode());
			String shiftString = results.getKey();
			String campaignName = results.getValue();

            rowData.add(campaignName); //Campaign column
			rowData.add(shiftString); //Shift column
			//For Story 34045 - 11.2 - PSR 6104 - Reason for Absence viewable only by Supervisor and Employee
			boolean isShowAbsentReason = true;
			if (isEnabledAbsentReasonToOnlySupervisor && !isSupervisor(empID) && !isMyRow)
				isShowAbsentReason = false;
			String eventString;
			if (isAccessibilityComplianceMode()) {
				eventString = this.makeEventsString(aDay, employeeSchedule, activities, bShowUnavailable, bShowTimeoff,
						bShowNote, isMyRow, isShowAbsentReason, m_localizer.i18n(BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.OVERTIME));
			} else {
				eventString = this.makeEventsString(aDay, employeeSchedule, activities, bShowUnavailable, bShowTimeoff,
						bShowNote, isMyRow, isShowAbsentReason);
			}
			
			rowData.add(eventString, "padding-right:0px;"); //Activities column
			//We set the padding for the campaign activity column (and the previous activity column) explicitly to zero because the html used to
			//render the campaign strings is already set to match the column header (7px).  If we do not set this here, there will be no padding.
			//We set the padding to 7px in the internal span because otherwise it was causing an issue with the bottom border not stretching to
			//the boundary of the cell.
			rowData.add(makeCampaignActivityString(aDay, daySummaries, employeeShifts, employeeSchedule, activities,
					campaignNamesBySpID, bShowUnavailable, bShowTimeoff), "padding-left:0px;padding-right:0px;");
		}
		return rowData;
	}//makeRow

	@Override
	protected MultiColListPC getMultiColListPC(RequestContext context) {
		return new MultiColListPC(context) {
			@Override
			protected HtmlDiv getNewHtmlDiv() {
				return new HtmlDivWSpan(getPCAriaLabelID(), i18n(m_bundle, FsWebBundleKey.SCHEDULE_GRID));
			}

			//list is disabled, but we allow to encloseForKeyboardAccess
			@Override
			protected boolean skipEncloseForKeyboardAccess(BaseNodeData nodeData) {
				return !this.isKeyboardAccessible() || !this.isRowSelectable() || !nodeData.isSelectable();
			}

			@Override
			protected String getCellData(HtmlTableCell cell) {
				if (getRowHeaderColNum() == getHeader().getColumnIndexByDisplayName(i18n(m_common_bundle, UIFWebBundleKey.NAME), true)
					&& super.getCellData(cell).contains("</b>")) {
					return super.getCellData(cell) + HtmlUtil.makeHiddenReadableText(i18n(m_bundle, FsWebBundleKey.CURRENT_USER));
				}
				return super.getCellData(cell);
			}

		};
	}
}//class GroupTextSchedulePM
