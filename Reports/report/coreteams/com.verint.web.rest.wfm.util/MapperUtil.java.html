package com.verint.web.rest.wfm.util;

import com.bluepumpkin.common.datatypes.ID;


public class MapperUtil {

	public Integer asInt(ID id) {
		if (id == null) {
			return null;
		}
		return id.toInt();
	}

	public ID fromInt(Integer id) {
		if (id == null) {
			return null;
		}
		return ID.fromInt(id);
	}

	public String asString(ID id) {
		if (id == null) {
			return null;
		}
		return id.toString();
	}
}
