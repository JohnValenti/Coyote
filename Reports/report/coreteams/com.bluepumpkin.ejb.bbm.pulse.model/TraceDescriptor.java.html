	private String targetFileLoc;
	private TimeZone timeZone;

	protected HashMap traceMap = new HashMap(10);

	private HashSet pickedPos = new HashSet(10);

	public TraceDescriptor(String targetFile, TimeZone tz, ID id, String name) {
		targetFileLoc = targetFile;
		timeZone = tz;
		queueID = id;
		queueName = name;

		itvChoice = itv;

	// If there has same colPos, return false
	public boolean pickColumn(Short colType, short colPos) {
		Short pos = NumberFactory.newShort(colPos);
		if (pickedPos.add(pos)) {
			traceMap.put(colType, pos);
			// If AHT is enabled, must enable ahtUnit by default
			if (colType.equals(AHT)) {
				if (ahtUnit == -1) {
					ahtUnit = (short)0;
				}
			}
			pickedCol.add(colType);
			return true;
		}
		return false;
	}

	public void removeColumn(Short colType) {
		if (colType.equals(AHT))
			ahtUnit = -1;
		pickedPos.remove(getColumnPos(colType));
		pickedCol.remove(colType);
	}

	// If null is returned, it means it is not picked up
	public Short getColumnPos(Short colType) {
		return (Short)traceMap.get(colType);
	}

	public ArrayList getSortedPos() {
		if (pickedPos != null) {
			ArrayList pos = new ArrayList(pickedPos);
			Collections.sort(pos);
			return pos;
		}
		return null;
	}

	// Return columns based on their position sequence
	public ArrayList getSortedColumn() {
		ArrayList posList = getSortedPos();
		if (posList == null)
			return null;
		ArrayList colList = new ArrayList(posList.size());
		Set colSet = traceMap.keySet();
		for (Iterator it = posList.iterator(); it.hasNext(); ) {
			Short pos = (Short)it.next();
			for (Iterator colIT = colSet.iterator(); colIT.hasNext(); ) {
				Short colType = (Short)colIT.next();
				if (pickedCol.contains(colType) && getColumnPos(colType).equals(pos)) {
					colList.add(colType);
					continue;
				}
			}
		}
		return colList;
	}

	public void setDayFormat(String format) { dayFormat = format; }
	public String getDayFormat() { return dayFormat; }

	public void setTimeFormat(String format) { timeFormat = format; }
	public String getTimeFormat() { return timeFormat; }
//	public void setNumberFormat(String format) { numberFormat = format; }
//	public String getNumberFormat() { return numberFormat; }

	public abstract TraceCube getTraceCube();
	public abstract boolean isSupportedTraceType(Short colType);

	public short[] getTraceTypes() {
		ArrayList colList = getSortedColumn();
		int num = 0;
		for (Iterator colIT = colList.iterator(); colIT.hasNext();) {
			Short col = (Short)colIT.next();
			if (isSupportedTraceType(col)) {
				num++;
			}
		}
		if (num == 0)
			return null;
		short[] types = new short[num];
		num = 0;
		for (Iterator colIT = colList.iterator(); colIT.hasNext();) {
			Short col = (Short)colIT.next();
			if (isSupportedTraceType(col)) {
				types[num++] = col.shortValue();
			}
		}
		return types;
	}
}
