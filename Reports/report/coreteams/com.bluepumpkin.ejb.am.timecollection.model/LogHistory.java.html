package com.bluepumpkin.ejb.am.timecollection.model;

import java.sql.Timestamp;
import java.util.*;
import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;
import java.io.Externalizable;

/**
 * Title:        Blue Pumpkin Software Activity Management
 * Description: LogHistory value object
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author Vincent Yu
 * @version 1.0
 */

public class LogHistory implements Externalizable {
	static final long serialVersionUID = 2140979709144414621L;

	/**
	 * Collection of log entries for activities
	 */
	private Collection logEntries;
	/**
	 * Time period for this collection of log entries
	 */
	private Timestamp start, end;

	public LogHistory() {}

	public LogHistory(Collection logEntries, Timestamp start, Timestamp end) {
		this.logEntries = logEntries;
		this.start = start;
		this.end = end;
	}
	/**
	 * accessor for logEntries
	 */
	public Collection getLogEntries() {
		return logEntries;
	}
	/**
	 * accessor for start
	 */
	public Timestamp getStart() {
		return start;
	}
	/**
	 * accessor for end
	 */
	public Timestamp getEnd() {
		return end;
	}
	public String toString() {
		StringBuffer out =
			new StringBuffer("Start: " + start + ", End: " + end + "\n");
		Iterator it = logEntries.iterator();
		while (it.hasNext()) {
			LogEntry le = (LogEntry) it.next();
			out.append("\n" + le);
		}
		return out.toString();
	}
	/* (non-Javadoc)
	 * @see java.io.Externalizable#readExternal(java.io.ObjectInput)
	 */
	public void readExternal(ObjectInput in)
		throws IOException, ClassNotFoundException {
		start = (Timestamp) in.readObject();
		end = (Timestamp) in.readObject();
		logEntries = (Collection) in.readObject();
	}

	/* (non-Javadoc)
	 * @see java.io.Externalizable#writeExternal(java.io.ObjectOutput)
	 */
	public void writeExternal(ObjectOutput out) throws IOException {
		out.writeObject(start);
		out.writeObject(end);
		out.writeObject(logEntries);
	}
}
