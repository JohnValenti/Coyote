package com.bluepumpkin.ejb.am.timecollection.model;

import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;
import java.io.Externalizable;

/**
 * Title:        Blue Pumpkin Software Activity Management
 * Description:
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */

public class PunchStatus implements Externalizable {
	static final long serialVersionUID = -1773965521308172626L;

	private int m_status;
	private int m_punchTypeCode;

	public PunchStatus() {}

	public PunchStatus(int status, int punchTypeCode) {
		this.m_status = status;
		this.m_punchTypeCode = punchTypeCode;
	}

	public int getStatus() { return m_status; }

	public int getPunchTypeCode() { return m_punchTypeCode; }

	/**
	* print object variables
	*/
	public String toString () {
		StringBuffer buffer = new StringBuffer();
		buffer.append(super.toString());
		buffer.append("\nPunchStatus: ");
		buffer.append("\nPunchStatus.m_status: "+m_status);
		buffer.append("\nPunchStatus.m_punchTypeCode: "+m_punchTypeCode);

		return buffer.toString();
	}
	/* (non-Javadoc)
	 * @see java.io.Externalizable#readExternal(java.io.ObjectInput)
	 */
	public void readExternal(ObjectInput in)
		throws IOException, ClassNotFoundException {
		m_status = in.readInt();
		m_punchTypeCode = in.readInt();
	}

	/* (non-Javadoc)
	 * @see java.io.Externalizable#writeExternal(java.io.ObjectOutput)
	 */
	public void writeExternal(ObjectOutput out) throws IOException {
		out.writeInt(m_status);
		out.writeInt(m_punchTypeCode);
	}
}
