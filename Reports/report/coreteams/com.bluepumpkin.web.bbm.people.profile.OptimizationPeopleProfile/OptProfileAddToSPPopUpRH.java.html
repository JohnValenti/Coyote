package com.bluepumpkin.web.bbm.people.profile.OptimizationPeopleProfile;

import java.rmi.RemoteException;
import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.bbm.workload.queue.selection.QueueFilterUtil;
import com.bluepumpkin.web.core.base.RequestHandler;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.l10n.CommonWebBundleKey;
import com.witness.web.uif.l10n.CommonWebLogBundleKey;
import com.witness.web.uif.system.RequestContext;

public class OptProfileAddToSPPopUpRH extends RequestHandler {

	protected OptProfileAddToSPPopUpPM getPageModel(RequestContext context) {
		OptProfileAddToSPPopUpPM pm = new OptProfileAddToSPPopUpPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
	}

	public void processRequest(RequestContext context) throws Exception {
		String action = getPageAction(context);
		//make page model
		OptProfileAddToSPPopUpPM model = new OptProfileAddToSPPopUpPM(context);

		// get data from the request
		model.extractParameters(context.getRequest());
		if ((action != null) && (isAction(context, PageAction.ADD) || action.isEmpty())) {
			processListAdd(context, model);
		} else{
			context.setPageModel(model);
		}
		
		//Cache the ProfilePageModel in the Request
		model.setPageDirty(false);
		context.setPageModel(model);
	}

	/**
	 * Process Edit Action from list
	 * 
	 * @param pm
	 */
	private void processListAdd(RequestContext context, OptProfileAddToSPPopUpPM model) {
		try {
			String selectedID = context.getRequest().getParameter("selectedID");
			Collection<ID> selectedEmployees = StringUtil.parseIDStringToCollection(selectedID);

			SchedulingPeriod sp = CampaignModelHandler.getSchedulingPeriodByID(
					context, QueueFilterUtil.retrieveSchedulingPeriodID(context));
			if (sp != null) {
				OptProfileAddToSPPopUpMH.addEmployeestoSP(context, sp.getDEID(),
						selectedEmployees);
			}
		} catch (RemoteException ex) {
			model.addPageMessage(Message.ERROR_TYPE,
					CommonWebBundleKey.UNABLE_TO_UPDATE_INFO,
					CommonWebBundleKey.BUNDLE_NAME);
			m_log.l7dError(CommonWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		} catch (BbmException ex) {
			model.addPageMessage(Message.ERROR_TYPE,
					CommonWebBundleKey.UNABLE_TO_UPDATE_INFO,
					CommonWebBundleKey.BUNDLE_NAME);
			m_log.l7dError(CommonWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}
	}
}
