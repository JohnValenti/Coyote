package com.bluepumpkin.ejb.rm.requests.swap.request.ejb;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.dao.DAONode;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.rm.requests.swap.request.model.ShiftSwapRequestItem;
import com.bluepumpkin.ejb.rm.requests.swap.request.model.ShiftSwapRequestItemFieldInfo;
import com.bluepumpkin.ejb.rm.requests.swap.shiftitem.ejb.ShiftSwapItemDAO;


/**
 * Title:        ShiftSwapRequestItemDAO
 * Description:  DAO class for ShiftSwapRequestItemDAO
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc.
 * @author       Zhibo Wang
 * @version 1.0
 */
public class ShiftSwapRequestItemDAO extends DAONode {
    private static FieldInfo m_fieldInfo = new ShiftSwapRequestItemFieldInfo();

    public ShiftSwapRequestItemDAO() {
        super();
    }

    public ShiftSwapRequestItemDAO(Jdmo dmo) {
        super(dmo);
    }

    protected FieldInfo getFieldInfo() {
        return m_fieldInfo;
    }

    protected ValueObjectBase createValueObject() {
        return new ShiftSwapRequestItem();
    }

    /** create a child DAO object given the type  */
    protected DAOBase createChildDAO(int iType) {
        if (iType == ShiftSwapRequestItemFieldInfo.ITEM_CHILD_TYPE) {
            return new ShiftSwapItemDAO(m_dmo);
        }
            
        return null;
    }
}
