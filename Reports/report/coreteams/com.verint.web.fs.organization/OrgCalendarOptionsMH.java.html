package com.verint.web.fs.organization;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;
import com.bluepumpkin.web.core.base.ModelHandler;
import com.bluepumpkin.web.fs.Log;
import com.witness.web.uif.system.RequestContext;

import java.rmi.RemoteException;

public class OrgCalendarOptionsMH extends ModelHandler {

	private static Category category = Log.initCategory(OrgCalendarOptionsMH.class.getName());

	/**
	 * Return a collection of Settings Property. Gets settings defined at this org, and
	 * inheritted settings if they are not set at this org level.
	 *
	 * @param orgID ID  Identifes the organizationId
	 * @return HashMap collection of key value pairs
	 * @throws BbmFinderException, RemoteException
	 */
	public static OrganizationSetting getSettings(RequestContext context, ID orgID)
			throws BbmFinderException, BbmCreateException, RemoteException {
		RmManagerFactory factory = RmManagerFactory.getInstance();
		// Invoke the facade method here
		return factory.getOrganizationConfigManager(context.isInWhatIfMode()).getConfiguration(orgID);
	}

	/**
	 * Configures the settings for an Organization.
	 *
	 * @param orgID        ID		Identifes the organizationId
	 * @param propertyList HashMap	Identifes the property list
	 * @param shallow      - true if you want to update only this org, false to update children too.
	 * @return HashMap collection of key value pairs
	 * @throws BbmCreateException,BbmUpdateException, RemoteException
	 */
	public static void setSettings(RequestContext context, ID orgID, OrganizationSetting settings, boolean shallow)
			throws BbmCreateException, BbmUpdateException, RemoteException {

		RmManagerFactory factory = RmManagerFactory.getInstance();
		// Invoke the facade method here
		factory.getOrganizationConfigManager(context.isInWhatIfMode()).setConfiguration(orgID, settings, shallow);
	}

	/**
	 * Tells us whether the specified organization has any child orgs.
	 *
	 * @param orgID ID  Identifes the organizationId
	 * @throws BbmFinderException, RemoteException
	 */
	public static boolean getHasChildren(RequestContext context, ID orgID)
			throws BbmFinderException, BbmEJBCreateException, RemoteException {
		return BbmManagerFactory.getWorkResourceManager(context.isInWhatIfMode()).organizationHasChildren(orgID);
	}

}
