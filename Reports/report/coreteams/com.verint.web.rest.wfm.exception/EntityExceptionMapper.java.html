package com.verint.web.rest.wfm.exception;

import io.katharsis.errorhandling.ErrorData;
import io.katharsis.errorhandling.ErrorResponse;
import io.katharsis.errorhandling.mapper.ExceptionMapper;


public class EntityExceptionMapper implements ExceptionMapper<EntityException> {
	@Override
	public ErrorResponse toErrorResponse(EntityException exception) {
		return ErrorResponse.builder()
				.setStatus(exception.getHttpStatusCode())
				.setSingleErrorData(ErrorData.builder()
						.setTitle(exception.getTitle())
						.setDetail(exception.getDetail())
						.setId(exception.getId() == null ? "<none>" : exception.getId())
						.build())
				.build();
	}

	@Override
	public EntityException fromErrorResponse(ErrorResponse errorResponse) {
		return null;
	}

	@Override
	public boolean accepts(ErrorResponse errorResponse) {
		return false;
	}
}
