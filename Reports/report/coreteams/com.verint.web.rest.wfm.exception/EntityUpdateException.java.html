package com.verint.web.rest.wfm.exception;


import com.verint.web.rest.wfm.entity.Entity;
import com.verint.web.rest.wfm.util.IdUtil;

public class EntityUpdateException extends EntityException {

	private final Entity entity;

	public EntityUpdateException(Entity entity) {
		super(entity.getClass().getSimpleName(), IdUtil.toString(entity.getID()));

		this.entity = entity;
	}

	@Override
	public String getTitle() {
		return "EntityUpdateException";
	}


	@Override
	public String getDetail() {
		return "There was a problem updating entity: " + entity.toString();
	}
}
