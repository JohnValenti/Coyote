	private Jdmo getJdmo() {
		// XXX:RC is there a test to check connection we can put here?

			m_jdmo = new Jdmo(true);
		}
		return m_jdmo;

	public Map<Object, Integer> getSchedulingParameters(ID idSP) throws BbmFinderException {

		Jdmo jdmo = getJdmo();
		HashMap<Object, Integer> mParameters = SPSchedulingParameters.getDefaults();

		try {
			StringBuffer sb = new StringBuffer(400);
			sb.append("SELECT SCHEDULINGWINDOW,SCHEDULESHIFTASSIGNMENTS, ADDPEOPLE,REMOVEPEOPLE,OFFPHONEONLY,TBDN0,");
			sb.append("OTSCHEDULING,PREFERENCEMODE,UNDERVSOVER,VARIANCE,SCHEDULEATLEAST,TIMETOSCHEDULE,ADDOTBEFORESHIFT,ADDOTAFTERSHIFT,");
			sb.append("ADDVTOBEFORESHIFT, ADDVTOAFTERSHIFT,MAXOTMINUTES,MAXVTOMINUTES,MINIMIZECLASSSESSIONS FROM SPSCHEDULERPARAMS ");
			sb.append("WHERE SPID IN (SELECT ID FROM SP WHERE SID=").append(idSP).append(")");

			JdmoRowset r = jdmo.createRowset(sb.toString());

			if (r.next()) { // only one record for one sp if there is.
				int i = 1;
				mParameters.put("SCHEDULINGWINDOW", new Integer(r.getInt(i++)));
				mParameters.put("SCHEDULESHIFTASSIGNMENTS", new Integer(r.getInt(i++)));
				mParameters.put("ADDPEOPLE", new Integer(r.getInt(i++)));
				mParameters.put("REMOVEPEOPLE", new Integer(r.getInt(i++)));
				mParameters.put("OFFPHONEONLY", new Integer(r.getInt(i++)));
				mParameters.put("TBDN0", new Integer(r.getInt(i++)));
				mParameters.put("OTSCHEDULING", new Integer(r.getInt(i++)));
				mParameters.put("PREFERENCEMODE", new Integer(r.getInt(i++)));
				mParameters.put("UNDERVSOVER", new Integer(r.getInt(i++)));
				mParameters.put("VARIANCE", new Integer(r.getInt(i++)));
				mParameters.put("SCHEDULEATLEAST", new Integer(r.getInt(i++)));
				mParameters.put("TIMETOSCHEDULE", new Integer(r.getInt(i++)));
				mParameters.put("ADDOTBEFORESHIFT", new Integer(r.getInt(i++)));
				mParameters.put("ADDOTAFTERSHIFT", new Integer(r.getInt(i++)));
				mParameters.put("ADDVTOBEFORESHIFT", new Integer(r.getInt(i++)));
				mParameters.put("ADDVTOAFTERSHIFT", new Integer(r.getInt(i++)));
				mParameters.put("MAXOTMINUTES", new Integer(r.getInt(i++)));
				mParameters.put("MAXVTOMINUTES", new Integer(r.getInt(i++)));
				mParameters.put("MINIMIZECLASSSESSIONS", new Integer(r.getInt(i++)));
			}
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		} finally {
			jdmo.cleanUp();
		}
		return mParameters;
	}

	public ID getMostRecentSPWithParameters(Date dtStart, ID idCampaign) throws BbmFinderException {

		Jdmo jdmo = getJdmo();
		ID idSP = null;

		try {

			StringBuffer sb = new StringBuffer(200);
			sb.append("	SELECT sid FROM SP WHERE todate =");
			sb.append("	(SELECT MAX(todate) FROM sp , campaign where sp.campaignid = campaign.id and CAMPAIGN.SID = ")
					.append(idCampaign);
			sb.append(" AND (SELECT COUNT(ID) FROM SPSCHEDULERPARAMS WHERE spid = sp.ID) > 0 )");

			JdmoRowset r = jdmo.createRowset(sb.toString());
			if (r.next()) {
				idSP = r.getID("ID");
			}
			r.close();
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		} finally {
			jdmo.cleanUp();
		}
		return idSP;
	}

	public void setSchedulingParameters(ID idSP, Map mParameters, String sWorkResourceIDs) throws BbmCreateException {

		Jdmo jdmo = getJdmo();

		try {
			JdmoRowset rs = jdmo.createRowset("select id from sp where sid = " + idSP);
			String spDEID = null;
			if (rs.next()) {
				spDEID = rs.getString(1);
			}
			if (spDEID == null) {
				return; // should never happen, should return.
			}

			String deleteExistingRecord = "delete from SPSCHEDULERPARAMS where spid ='" + spDEID + "'";
			jdmo.execute(deleteExistingRecord);

			String id = DAOUtil.getDEID(jdmo);

			mParameters.put("ID", new ID(id));
			mParameters.put("SPID", spDEID);
			HashMap insertHashMap = new HashMap(mParameters);
			jdmo.addBatchInsert("SPSCHEDULERPARAMS", insertHashMap);
			jdmo.executeBatch();

		} catch (JdmoException e) {
			throw new BbmCreateException(e);
		} finally {
			jdmo.cleanUp();
		}
	}

	public Pair<Date, Date> getDirtyStatus(ID idSchedulingPeriod) throws BbmFinderException {
		Jdmo jdmo = getJdmo();
		Pair<Date, Date> pStatus = new Pair<Date, Date>();

		try {
			String query = "SELECT LASTMODIFIEDTIME,LASTRECALCTIME FROM SCHEDULINGPERIOD WHERE ID =" + idSchedulingPeriod;

			JdmoRowset r = jdmo.createRowset(query);
			while (r.next()) {
				pStatus.setFirst(TimeZoneUtil.toDate(r.getTimestamp("LASTMODIFIEDTIME")));
				pStatus.setSecond(TimeZoneUtil.toDate(r.getTimestamp("LASTRECALCTIME")));
			}
			r.close();
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		} finally {
			jdmo.cleanUp();
		}
		return pStatus;
	}
}
