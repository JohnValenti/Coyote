package com.bluepumpkin.web.bbm.activity;

import java.util.ResourceBundle;

import javax.servlet.http.HttpServletRequest;

import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.verint.web.wfm.keys.WfmJavaScriptFileID;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagemodel.PopupWindowPM;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

/**
 * Title:        ActivityColorPickerPageModel
 * Description:  Activity Color Picker popup window PageModel
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author       Shimon kakon
 * @version 1.0
 */
@SuppressWarnings("serial")
public class ActivityColorPickerPageModel extends PopupWindowPM {
	protected boolean m_isInitialed = false;
	protected ResourceBundle m_bbmBundle;
	private ActivityColorPickerPC m_activityColorPickerPC;
	public static String COLOR_IN_PARENT_FN;
	public static final String COLOR_IN_PICKER_FN = "color_in_picker";
	// not really used. Will be override by JavaScript method (see getJavaScriptInitCode())
	private static final String INITIAL_COLOR = "#FFFFFF";

	/**
	 * Default constructor
	 */
	public ActivityColorPickerPageModel(RequestContext context) {
		super(context);
		m_bbmBundle = m_localizer.getBundle(BbmWebBundleKey.BUNDLE_NAME);
		extractParameters(context.getRequest());

		setJSMediator(WfmJavaScriptFileID.ACTIVITY_COLOR_PICKER);
		addRequiredJavaScriptFile(WfmJavaScriptFileID.ACTIVITY_ON_CHANGE);
		addRequiredJavaScriptFile(JavaScriptFileID.DIALOG_PICKERS);
		addRequiredJavaScriptFile(JavaScriptFileID.MEDIATOR_DIALOG);
		addRequiredJavaScriptFile(JavaScriptFileID.MODULE_DIALOGBOX);
		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);

		m_activityColorPickerPC = new ActivityColorPickerPC(context,
			COLOR_IN_PICKER_FN, INITIAL_COLOR, m_bbmBundle);

		initToolbar();
	}



	/**
	 * Return instance of Model which is ready to be rendered
	 *
	 * @param context the Request Context
	 */
	public static ActivityColorPickerPageModel getInstance(RequestContext context) {
		ActivityColorPickerPageModel model = null;

		// Check cache first
		if (context.getPageModel() instanceof ActivityColorPickerPageModel) {
			model = (ActivityColorPickerPageModel)context.getPageModel();
		} else {
			model = new ActivityColorPickerPageModel(context);
			model.extractParameters(context.getRequest());
			context.setPageModel(model);
		}
		return model;
	}

	protected void initToolbar(){
		m_toolbar.addCloseWindowTextButton(
				PageAction.CANCEL,
				i18n(m_common_bundle, UIFWebBundleKey.TOOLBAR_DONE),
				true);
	}

	/**
	 * Return the JavaScript init code required by Page model.
	 *
	 * Default behavior is to retrieve sub-component JavaScript Init code
	 */
	public String getJavaScriptInitCode() {
		// Add sub-component JavaScript code first
		return super.getJavaScriptInitCode() +
			"top.workpaneMediator.initColor();\n";
	}


	/**
	 * Return Required HTML
	 *
	 * Used by framework
	 */
	public String getRequiredHTML() {
		StringBuffer sb = new StringBuffer(1000);
		sb.append(this.getColorPicker());
		return sb.toString();
	}


	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return "";
	}


	public boolean extractParameters(HttpServletRequest request) {
		COLOR_IN_PARENT_FN = request.getParameter("name");
		return true;
	}

	// render HTML needed for color picker. Used from the JSP
	public String getColorPicker(){
		return HtmlUtil.makeColorPicker();
	}

	/**
	 * Return the Html code to render the display of the component
	 */
	public String getHtmlDisplay() {
		return this.m_activityColorPickerPC.getHtmlDisplay() + m_toolbar.getHtmlDisplay();
	}
}
