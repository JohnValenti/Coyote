package com.verint.ejb.bbm.fterequirements;

import java.util.Collection;
import java.util.Map;

import javax.jms.ObjectMessage;
import javax.jms.Queue;
import javax.jms.QueueConnection;
import javax.jms.QueueConnectionFactory;
import javax.jms.QueueSender;
import javax.jms.QueueSession;
import javax.jms.Session;
import javax.naming.Context;
import javax.naming.InitialContext;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.JMSMessageBase;
import com.bluepumpkin.ejb.bbm.JNDINames;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;

/**
 * TimeSeriesMessage is created and sent to a jms queue (where it is later handled
 * by TimeSeriesHandlerMDB).  These messages are created after a forecast, service goal,
 * or shrinkage is modified for a set of SPQueues.  When the message is received by the
 * handler, it indicates that FTE Requirements are to be recalculated for those SPQueues.
 */
@SuppressWarnings("serial")
public class TimeSeriesMessage extends JMSMessageBase {
	
	private final Collection<ID> spQueueIDs;
	private final Map<ID, ? extends TraceCube> forecastsBySPQueueID;
	private final boolean isWhatIfMode;
	
	/**
	 * Create a new time series message.
	 * @param spQueueIDs - The SPQueueIDs for which we'd like to calculate
	 * FTE Requirements.
	 * @param forecastsBySPQueueID - A map of forecast trace cube data for
	 * each SPQueue.  This optional parameter will speed up
	 * FTE Requirements calculations for certain media types.  If it is passed
	 * in as null, then the forecast data will be reloaded during the FTE
	 * calculation, if it is required for that media type.
	 */
	public TimeSeriesMessage(
			Collection<ID> spQueueIDs,
			Map<ID, ? extends TraceCube> forecastsBySPQueueID,
			boolean isWhatIfMode) {
		this.spQueueIDs = spQueueIDs;
		this.forecastsBySPQueueID = forecastsBySPQueueID;
		this.isWhatIfMode = isWhatIfMode;
	}

	public void sendMessage() throws BbmUpdateException {
		try {
			Context ctx = new InitialContext();
			QueueConnectionFactory queueConnectionFactory = (QueueConnectionFactory) ctx.lookup(JNDINames.BBM_JMS_QUEUE_FACTORY);
			QueueConnection queueConnection = queueConnectionFactory.createQueueConnection();
			QueueSession queueSession = queueConnection.createQueueSession(false, Session.AUTO_ACKNOWLEDGE);
			Queue queue = (Queue) ctx.lookup(JNDINames.BBM_TIMESERIES_QUEUE);
			QueueSender queueSender = queueSession.createSender(queue);
			queueConnection.start();
			ObjectMessage objectMessage = queueSession.createObjectMessage();
			objectMessage.setObject(this);
			queueSender.send(objectMessage);
			queueSender.close();
			queueSession.close();
			queueConnection.close();
		} catch (Exception e) {
			throw new BbmUpdateException(e);
		}
	}
	
	public Collection<ID> getSPQueueIDs() {
		return spQueueIDs;
	}
	
	/**
	 * Returns a map containing the forecast data for the spqueues in the message.
	 * This may not always be available in all cases (typically it is available
	 * after just saving the forecast, since we have the forecast data already in
	 * that case we just hand it in along with the message to prevent a reload
	 * of the same forecast data when the MDB handles the message).
	 */
	public Map<ID, ? extends TraceCube> getForecastsBySPQueueID() {
		return forecastsBySPQueueID;
	}
	
	public boolean isWhatIfMode() {
		return isWhatIfMode;
	}
}
