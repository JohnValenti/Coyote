	protected static final String ACTION_CACHE_SELECTED_ID = "CacheSelectedID";

	public void processRequest(RequestContext context) throws Exception {
		Map paramMap = ParameterUtil.getParameterMap(context.getRequest());
		String pageAction = ParameterUtil.getParameter(paramMap, "pageAction");

		if (ACTION_GET_EMP_IDs.equals(pageAction)) {
			getEmployeeIDs(context, paramMap);
		} else if (ACTION_GET_EMP_DATA.equals(pageAction)) {
			getEmployeeData(context, paramMap);
		} else if (ACTION_GET_FILTER_QUICK_INFO.equals(pageAction)) {
			getFilterQuickInfo(context, paramMap);
		} else if (ACTION_CACHE_SELECTED_ID.equals(pageAction)) {
			setSelectedEmployeeIDToCacheAfterFilterOutPhantom(context);
		}
		return;
	}
	
	private void setSelectedEmployeeIDToCacheAfterFilterOutPhantom(RequestContext context) throws BbmEJBCreateException, BbmFinderException, RemoteException {
		String selectedIDKey = context.getParameter(PageKeys.CACHE_SELECTED_ID_KEY);
		String selectedID = context.getParameter(PageKeys.CACHE_SELECTED_ID_FN);
		
		Collection<ID> idList = StringUtil.parseIDStringToCollection(selectedID);
		if (idList != null && !idList.isEmpty()) {
			ScheduleAccessManager sam = WfmManagerFactory.getScheduleAccessManager(context.isInWhatIfMode());
			idList = sam.filterPhantomIDsFromWorkResourceIDList(idList);
			if (idList != null && !idList.isEmpty()) {
				selectedID = StringUtil.createDelimitedString(idList, ",");
			} else {
				selectedID = "";
			}
		}
		String cacheMode = context.getParameter(PageKeys.CACHE_SELECTED_ID_MODE);
		if (! StringUtil.isEmpty(selectedIDKey)) {
			CacheUtil.setCachedValue(context, cacheMode, selectedIDKey, selectedID);
		}
	}

	protected void getEmployeeIDs(RequestContext context, Map paramMap) throws Exception {
		String filterName = ParameterUtil.getParameter(paramMap, "filterName");
		SelectionCacheUtil.setEmpFilterName(context, filterName);
		// forceRefresh in URL is used to force refresh of selection pane from client side irrespective of whether data changed.
		// Ex: when employee name is changed, IDs are same but employee name needs to be refreshed.
		String sForceRefresh = ParameterUtil.getParameter(paramMap, "forceRefresh");
		String selectedIDs = ParameterUtil.getParameter(paramMap, "selectedIDs");
		boolean bForceRefresh = false;
		if (!StringUtil.isEmptyOrWhiteSpace(sForceRefresh)) {
			bForceRefresh = Boolean.valueOf(sForceRefresh);
		}
		Date startDate = getStartDate(context);
		Date endDate = getEndDate(context);

		// Since this method getEmployeeIDs is equivalent to submitting selection pane in earlier version
		// Set selectedIDs from page to cache.
		//FIXME: we should not be caching selected IDs in a method called "getEmployeeIDs"
		String selIDStr = StringUtil.isEmpty(selectedIDs) ? "" : selectedIDs;
		CacheUtil.setCachedValue(context, CacheUtil.CACHE_MODE_SESSION,
				"PEOPLE_SELECTION_ID_KEY", selIDStr);

		ID spID = SelectionCacheUtil.getSchedulingPeriodID(context);
		boolean bShowPhantoms = CacheUtil.getShowPhantoms(context) && CacheUtil.getUseShowPhantoms(context);

		//Body of the request contains comma separated string of displayed employee ids.
		BufferedReader br = context.getRequest().getReader();
		String displayedEmpIdString = br.readLine();
		List<ID> displayedEmpIds = (List<ID>) StringUtil.parseIDStringToCollection(displayedEmpIdString);

		// Save the current user filter selection: because this is the only place that registers filter change.
		//FIXME: we should not be caching the selected filter in a method called "getEmployeeIDs"
		context.setUserProperty(UserPreferenceKeys.EMP_FILTER, filterName);

		Map<String, Collection<ID>> empIdsAndFilterInfo =
				getNewEmployeeIDsAndFilterInfo(context, new PeopleFilterParams(filterName, startDate, endDate));
		List<ID> newEmpIds = (List<ID>) empIdsAndFilterInfo.get(EmployeeModelHandler.FILTER_RESULT_EMPIDS);

		// start: put employee ids in cache
		StringBuffer idsInCache = new StringBuffer();
		if (newEmpIds != null) {
			for (Iterator<ID> it = newEmpIds.iterator(); it.hasNext(); ) {
				idsInCache.append(it.next());
				if (it.hasNext()) {
					idsInCache.append(",");
				}
			}
		}
		//FIXME: Can we refactor this method to maybe return the employee IDs instead of caching them (so it can be re-used in other places)?
		SelectionCacheUtil.setEmployeeIDs(context, idsInCache.toString());
		//end: put employee ids in cache
		String preFilter = CacheUtil.getPreFilter(context);
		List<ID> newPhantomEmpIds = getNewPhantomEmployeeIDs(context,
				new PeopleFilterParams(filterName, startDate, endDate), preFilter, bShowPhantoms, spID);
		List<ID> allIDs = new LinkedList<ID>();

		// If the show pooling option is enabled
		if (CacheUtil.getUseShowPoolers(context)) {
			boolean bShowPoolers = CacheUtil.getShowPoolers(context);

			List<ID> poolerIds = Collections.emptyList();
			if (startDate != null && endDate != null) {
				poolerIds = getNewPoolerIDs(context, spID, startDate, endDate);
			}

			if (!bShowPoolers) {
				if (newEmpIds != null) {
					newEmpIds.removeAll(poolerIds);
				}
			}
		}
		if (newEmpIds != null) {
			allIDs.addAll(newEmpIds);
		}
		allIDs.addAll(newPhantomEmpIds);


		PrintWriter prntWrtr = context.getResponse().getWriter();
		// Send page action as the first part of response. This is used to differentiate multiple responses (one for filter info, other for employee IDs)
		prntWrtr.print(ACTION_GET_EMP_IDs + ";");
		boolean bFilterIsContextBased = true;
		prntWrtr.print(FILTER_IS_CONTEXT_BASED + ":" + bFilterIsContextBased + ";");

		// One of the following 3 outcomes is possible:
		// 1) new set of IDs is same as the old one: return null as string.
		// 2) new set of IDs is different from old one and has non-empty set: return new data.
		// 3) new set of IDs is empty list (the filter resulted is no emps): return empty list (which translates to an empty string).
		//
		// bForceRefresh: overrides the comparison and forces server to send employee IDs. This is needed in cases where employees IDs might not change but name might have changed.
		// The method returns employee IDs and phantom IDs. Need to identify each type of ID.
		// The data returned will be list of objects where each object is ID followed by colon followed by ID type.
		// 0 for emp ID; 1 for phantom ID. Ex: 1:0,2:0,3:0,5:1,6:1
		if ((!bForceRefresh) && ((allIDs != null) && (allIDs.equals(displayedEmpIds)))) {
			prntWrtr.print((String) null);
		} else {
			StringBuffer empIDStr = new StringBuffer();
			// Employee IDs. Append ID and ID type as 0 separated by colon.
			if (newEmpIds != null) {
				for (Iterator<ID> it = newEmpIds.iterator(); it.hasNext(); ) {
					empIDStr.append((it.next()).toString()).append(":").append(EMP_ID_TYPE);
					if (it.hasNext()) {
						empIDStr.append(",");
					}
				}
			}

			if (!newPhantomEmpIds.isEmpty() && newPhantomEmpIds.size() > 0) {
				empIDStr.append(",");
			}

			// Phantom IDs. Append ID and ID type as 1 separated by colon.
			for (Iterator<ID> it = newPhantomEmpIds.iterator(); it.hasNext(); ) {
				empIDStr.append((it.next()).toString()).append(":").append(PHANTOM_ID_TYPE);
				if (it.hasNext()) {
					empIDStr.append(",");
				}
			}
			prntWrtr.print(empIDStr);
		}
		return;
	}

	/**
	 * Return employees that belong to the selected scheduling period AND are in the selected employee filter (if we are not in "no campaign" mode).
	 * If we are in "no campaign" mode we find the active employees within the start/end dates given and do not filter by campaign.
	 * Returns a map containing two entries (both keyed with the following String values):
	 * FILTER_INFO_IS_CONTEXT_BASED - a boolean indicating whether or not this filter is context based.
	 * FILTER_RESULT_EMPIDS - A Collection of the filtered employee IDs.
	 */
	protected Map<String, Collection<ID>> getNewEmployeeIDsAndFilterInfo(RequestContext context, PeopleFilterParams pfp) throws Exception {
		Map<String, Collection<ID>> empIDsAndFilterInfo = new HashMap<String, Collection<ID>>();

		ID schedulingPeriodID = SelectionCacheUtil.getSchedulingPeriodID(context);
		if (schedulingPeriodID != null) {
			// Create a "pre-filter" to retrieve employee IDs for the selected SP.
			// EmployeeModelHandler will (in a way that is completely impossible to decipher or debug) deserialize this
			// pre-filter and use it to query the employee IDs we need.


			SchedulingPeriod schedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
					context, schedulingPeriodID);

			// Warning: Filter constructor expects params to be a collection of collections
			Filter filter;
			//Don't need this check anymore. spid is passed at all cases, so EmployeeFilter will use
			//spid to figure out if the flag is turned on.
			if (!schedulingPeriod.isUsingAllEmployees()) {

				ArrayList<Collection<?>> filterParams = new ArrayList<Collection<?>>();
				// Filter constructor expects params to be a collection of collections
				filterParams.add(Collections.singletonList(SelectionCacheUtil.getCampaignID(context)));
				filter = new Filter(
						Filter.CAMPAIGNID, Filter.OPERATOR_IN, filterParams);

				schedulingPeriodID = null;
			} else {
				ArrayList<Collection<?>> filterParams = new ArrayList<Collection<?>>();

				Collection<ID> organizationIDs = CampaignModelHandler.getLinkedOrgsforSP(context, schedulingPeriodID);
				if (organizationIDs == null) {
					organizationIDs = new ArrayList<ID>();
				}

				// Filter constructor expects params to be a collection of collections
				filterParams.add(organizationIDs);
				filter = new Filter(Filter.ORGANIZATIONID, Filter.OPERATOR_IN, filterParams);
			}

			filter.setUserID(context.getUser().getID());
			filter.setStartTime(getStartDate(context));
			filter.setEndTime(getEndDate(context));
			filter.setTimePeriodType(Filter.TIMEPERIODTYPE_TIMEWINDOW);
			CacheUtil.setPreFilter(context, Filter.filterToString(filter));

			return EmployeeModelHandler.getEmployeeIDsAndFilterInfo(context, pfp, schedulingPeriodID);
		} else if (SelectionCacheUtil.isNoCampaignMode(context)) {
			//Clear the pre-filter in case it was previously set via campaign/sp mode selection above
			CacheUtil.removePreFilter(context);
			//Retrieve employee IDs based on the employee filter, and then filter those employees again based
			//on the selected date range.
			//FIXME:  It is inefficient to filter twice here.  It may be possible to create a pre filter to
			//achieve the same thing, similar to the above logic for campaign mode.
			//Look at ProfileModelHandler.getInactiveEmployees for an idea.
			empIDsAndFilterInfo = EmployeeModelHandler.getEmployeeIDsAndFilterInfo(context, pfp, null);
			//Pull the list of employees out of the map, filter those employees, and then put the filtered list
			//back in the map.  The map is what is returned from this method
			Collection<ID> newEmpIds = empIDsAndFilterInfo.get(EmployeeModelHandler.FILTER_RESULT_EMPIDS);
			Collection<ID> dateFilteredEmployeeIDs =
					EmployeeModelHandler.getEmployeeIDs(context, getStartDate(context), getEndDate(context));
			newEmpIds.retainAll(dateFilteredEmployeeIDs);
			empIDsAndFilterInfo.put(EmployeeModelHandler.FILTER_RESULT_EMPIDS, newEmpIds);
		}
		return empIDsAndFilterInfo;
	}

	protected List<ID> getNewPhantomEmployeeIDs(RequestContext context, PeopleFilterParams pfp, String preFilter,
			boolean bShowPhantoms, ID phantomSPID) {
		return (List<ID>) EmployeeModelHandler.getPhantomEmpIDs(context, pfp, preFilter, bShowPhantoms, phantomSPID);
	}

	protected Date getStartDate(RequestContext context) throws Exception {
		//If there is no scheduling period handed in, then we are in "no campaign" mode and we pull the start date from the start date param
		if (SelectionCacheUtil.getSchedulingPeriodID(context) == null) {
			return SelectionCacheUtil.getStartDate(context);
		} else if (selectedSchedulingPeriod == null) {
			selectedSchedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
					context, QueueFilterUtil.retrieveSchedulingPeriodID(context));
		}
		return selectedSchedulingPeriod.getStartTime();
	}

	protected Date getEndDate(RequestContext context) throws Exception {
		//If there is no scheduling period handed in, then we are in "no campaign" mode and we pull the end date from the end date param
		if (SelectionCacheUtil.getSchedulingPeriodID(context) == null) {
			return SelectionCacheUtil.getEndDate(context);
		} else if (selectedSchedulingPeriod == null) {
			selectedSchedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
					context, QueueFilterUtil.retrieveSchedulingPeriodID(context));
		}
		return selectedSchedulingPeriod.getEndTime();
	}
}
