/**
 * <p> <b> Soft Validation Rule:Request period is published </b>
 */
public class PublishedPeriodValidationRule implements Validator {

	private static Category m_cat = Log.initCategory(PublishedPeriodValidationRule.class.getName());

	public PublishedPeriodValidationRule() {
	};

	/* (non-Javadoc)
		* @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
		*/
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		CustShiftReq csr = (CustShiftReq) validatable;
		String methodName = "validate";
		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpEnterMethod(methodName, csr));
		}
		ValidationResult result = null;

		CustShiftValidationCache cache = getCustShiftValidationCache(csr);
		ID empID = csr.getEmployeeID();
		TimeRange range = new TimeRange(csr.getStartTime(), csr.getEndTime());
		List<Collection<PublishingPeriod>> pubPeriodsByEmp = getPublishedPeriods(cache.getScheduleAccessManager(),
				(Collections.singletonList(empID)), range.getStartDate(), range.getEndDate());
		//QC 192020:The rule works incorrectly if change start time of the request and approve it,
		//then change start time of the request again.
		//Before the fix, the code just compares the request range to the publishing period.
		//After the fix, it makes sure that the start of request range and the end of request range should be a publishing period
		//E.g: We have a published week, each day has shift from 9AM to 5PM:
		// 		1/23 |------------------------------------| 1/30
		//Submit a request on 1/27 to change start time from 9AM to 10PM. 
		//The code compares the range 1/27[10AM to 6PM] to the published range.
		//The publishing periods includes both start range(10AM) and end range(6PM) so the rule is valid
		//After approve it, the week is still published with 3 continuous publishing periods:
		//		[1/23 to 1/27 10AM],[ 1/27 10AM to 1/27 6PM],[1/27 6PM to 1/30] 
		//      1/23 |-------------|--------|-------------| 1/30
		//Submit a request on 1/27 to change start time to 10AM to 11AM
		//The request range have the start is inside the second publishing period and the end is in the the third publishing period
		//=> the rule is  still valid. The old code compared the request range to the whole publishing period and it always returns false since
		// now the publishing period is break in to small periods but the whole week is still published
		boolean startRangeIsInPublishedRange = false;
		boolean endRangeIsInPublishedRange = false;
		Collection<PublishingPeriod> pubPeriods = null;
		if (pubPeriodsByEmp != null && !pubPeriodsByEmp.isEmpty()) {
			pubPeriods = (Collection<PublishingPeriod>) pubPeriodsByEmp.iterator().next();
			if (pubPeriods != null) {
				for (Iterator<PublishingPeriod> i = pubPeriods.iterator(); i.hasNext();) {
					PublishingPeriod tp = (PublishingPeriod) i.next();
					TimeRange pubRange = new TimeRange(tp.getStartTime(), tp.getEndTime());
					if (!tp.isPartialPublish() && pubRange.includes(range.getStartDate())) {
						//no error Return here
						startRangeIsInPublishedRange = true;
						break;
					}
				}
				for (Iterator<PublishingPeriod> i = pubPeriods.iterator(); i.hasNext();) {
					PublishingPeriod tp = (PublishingPeriod) i.next();
					TimeRange pubRange = new TimeRange(tp.getStartTime(), tp.getEndTime());
					if (!tp.isPartialPublish() && pubRange.includes(range.getEndDate())) {
						//no error Return here
						endRangeIsInPublishedRange = true;
						break;
					}
				}
			}
		}
		if (!(startRangeIsInPublishedRange && endRangeIsInPublishedRange)) {
			result = ValidationUtil.setSoftValidationResult(csr, RmEjbBundleKey.CS_PUBLISHED_PERIOD_CHECK, null, this.getClass().getName());
		}
		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpExitMethod(methodName));
		}
		return result;
	}

	protected CustShiftValidationCache getCustShiftValidationCache(CustShiftReq csRequest) {
		return (CustShiftValidationCache) csRequest.getCache();
	}

	public List<Collection<PublishingPeriod>> getPublishedPeriods(ScheduleAccessManager sam, Collection<ID> workResourceIDs, Date dtStart,
			Date dtEnd) throws Exception {
		return sam.getPublishedPeriods(workResourceIDs, dtStart, dtEnd);
	}
}
