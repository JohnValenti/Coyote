package com.bluepumpkin.ejb.am.lsg.model;

/**
 * This ENUM is for defining the sources of LSG Proportion for auto populate
 * <p>
 * NONE - There is no specific source
 * WIT - The source from the WIT data
 *
 * @author Jinal Shah
 */
public enum LSGProportionSource {
  NONE(0), WIT(1);

  private final int value;

  LSGProportionSource(int value) {
    this.value = value;
  }

  public int getValue() {
    return value;
  }

  /**
   * @param source
   * @return the associated ENUM type for the given integer
   */
  public static LSGProportionSource getType(int source) {
    switch (source) {
      case 0:
        return NONE;
      case 1:
        return WIT;
    }
    return null;
  }
}
