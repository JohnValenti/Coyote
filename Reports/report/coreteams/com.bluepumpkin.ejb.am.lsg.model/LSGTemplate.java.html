package com.bluepumpkin.ejb.am.lsg.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Date;

import org.codehaus.jackson.annotate.JsonIgnoreProperties;
import org.codehaus.jackson.map.ObjectMapper;

import com.bluepumpkin.common.datatypes.ID;

/**
 * This class stores data about the LSG Template
 *
 * @author Jinal Shah
 */
@JsonIgnoreProperties({"id", "campaignDEID"})
public class LSGTemplate implements Serializable {

  private static final long serialVersionUID = 1515280185401277279L;

  private ID id;
  private String name;
  private boolean isCustom;
  private LSGTimeType timeType;
  private String createdBy;
  private Date creationDate;
  private String modifiedBy;
  private Date modifiedDate;
  private LSGProportionSource lsgProportionSource;
  private ID campaignDEID;

  private Collection<LSGAgeBucket> lsgAgeBuckets;

  /**
   * @return the ID of the template
   */
  public ID getId() {
    return id;
  }

  /**
   * Sets the ID of the template
   *
   * @param id
   */
  public void setId(ID id) {
    this.id = id;
  }

  /**
   * @return the name of the template
   */
  public String getName() {
    return name;
  }

  /**
   * Sets the name of the template
   *
   * @param name
   */
  public void setName(String name) {
    this.name = name;
  }

  /**
   * @return True if the template is custom or not
   */
  public boolean isCustom() {
    return isCustom;
  }

  /**
   * Sets whether the template is custom or not
   *
   * @param isCustom
   */
  public void setCustom(boolean isCustom) {
    this.isCustom = isCustom;
  }

  /**
   * @return the time type of template whether it is DAY or WEEK.
   */
  public LSGTimeType getTimeType() {
    return timeType;
  }

  /**
   * Sets the time type of template whether it is DAY or WEEK.
   *
   * @param timeType
   */
  public void setTimeType(LSGTimeType timeType) {
    this.timeType = timeType;
  }

  /**
   * @return the name of the user who created the template
   */
  public String getCreatedBy() {
    return createdBy;
  }

  /**
   * Sets the created by user for the template
   *
   * @param createdBy
   */
  public void setCreatedBy(String createdBy) {
    this.createdBy = createdBy;
  }

  /**
   * @return the date on which the template was created
   */
  public Date getCreationDate() {
    return creationDate;
  }

  /**
   * sets the date on which template was created
   *
   * @param creationDate
   */
  public void setCreationDate(Date creationDate) {
    this.creationDate = creationDate;
  }

  /**
   * @return the last modified user for the template
   */
  public String getModifiedBy() {
    return modifiedBy;
  }

  /**
   * Sets the last modified by user for the template
   *
   * @param modifiedBy
   */
  public void setModifiedBy(String modifiedBy) {
    this.modifiedBy = modifiedBy;
  }

  /**
   * @return the last modified date for the template
   */
  public Date getModifiedDate() {
    return modifiedDate;
  }

  /**
   * Sets the last modified date of the template
   *
   * @param modifiedDate
   */
  public void setModifiedDate(Date modifiedDate) {
    this.modifiedDate = modifiedDate;
  }

  /**
   * @return the campaign ID to which the template belongs
   */
  public ID getCampaignDEID() {
    return campaignDEID;
  }

  /**
   * Sets the campaign ID to which the template belongs
   *
   * @param campaignDEID
   */
  public void setCampaignDEID(ID campaignDEID) {
    this.campaignDEID = campaignDEID;
  }

  /**
   * @return the list of Age Buckets associated with the template
   */
  public Collection<LSGAgeBucket> getLsgAgeBuckets() {
    return lsgAgeBuckets;
  }

  /**
   * Sets the list of Age Buckets associated with the template
   *
   * @param lsgAgeBuckets
   */
  public void setLsgAgeBuckets(Collection<LSGAgeBucket> lsgAgeBuckets) {
    this.lsgAgeBuckets = lsgAgeBuckets;
  }

  /**
   * @return the source of proportion
   */
  public LSGProportionSource getLsgProportionSource() {
    return lsgProportionSource;
  }

  /**
   * Sets the proportion source from which to take the proportion
   *
   * @param lsgProportionSource
   */
  public void setLsgProportionSource(LSGProportionSource lsgProportionSource) {
    this.lsgProportionSource = lsgProportionSource;
  }

  /**
   * @return the JSON string representation of the current object
   */
  public String toJSON() {
    try {
      ObjectMapper mapper = new ObjectMapper();
      return mapper.writer().withDefaultPrettyPrinter().writeValueAsString(this);
    } catch (Exception e) {
      return "{}";
    }
  }

  /**
   * @param jsonString
   * @return
   */
  public static LSGTemplate convertJSONStringToLSGTemplateObject(
      String jsonString) {
    try {
      return (new ObjectMapper()).readValue(jsonString, LSGTemplate.class);
    } catch (Exception e) {
      return new LSGTemplate();
    }
  }

}
