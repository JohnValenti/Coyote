package com.bluepumpkin.ejb.am.lsg.model;

import java.io.Serializable;
import java.util.Date;

import org.codehaus.jackson.annotate.JsonIgnoreProperties;

import com.bluepumpkin.common.datatypes.ID;

/**
 * This model is used to store the Deadline Goal Bucket Ages for the respective
 * Queue
 *
 * @author jshah
 */
@JsonIgnoreProperties({"id", "lsgTemplateID"})
public class LSGAgeBucket implements Serializable {

  private static final long serialVersionUID = -463429851977326665L;

  private ID id;
  private int index;
  private int rangeLower;
  private int rangeUpper;
  private ID lsgTemplateID;
  private String name;
  private LSGAgeBucketProportion lsgAgeBucketProportion;
  private Date modifiedDate;

  /**
   * These 2 field are explicitly defined to send data to Algo team through
   * the Corba layer and this data will not be stored in database
   */
  private Date startTime;
  private Date endTime;

  /**
   * @return the primary key id
   */
  public ID getId() {
    return id;
  }

  /**
   * Sets the primary key Id for the given Age Bucket
   *
   * @param id
   */
  public void setId(ID id) {
    this.id = id;
  }

  /**
   * @return the index of the bucket
   */
  public int getIndex() {
    return index;
  }

  /**
   * Sets the index for this bucket
   *
   * @param index
   */
  public void setIndex(int index) {
    this.index = index;
  }

  public int getRangeLower() {
    return rangeLower;
  }

  public void setRangeLower(int rangeLower) {
    this.rangeLower = rangeLower;
  }

  public int getRangeUpper() {
    return rangeUpper;
  }

  public void setRangeUpper(int rangeUpper) {
    this.rangeUpper = rangeUpper;
  }

  /**
   * @return the Queue Id the bucket is associated with
   */
  public ID getLsgTemplateID() {
    return lsgTemplateID;
  }

  /**
   * sets the Queue Id the bucket is associated with
   *
   * @param queueID
   */
  public void setLsgTemplateID(ID queueID) {
    this.lsgTemplateID = queueID;
  }

  /**
   * @return the name of the bucket
   */
  public String getName() {
    return name;
  }

  /**
   * sets the name for the bucket
   *
   * @param name
   */
  public void setName(String name) {
    this.name = name;
  }

  /**
   * @return the modified date when the proportion was last modified
   */
  public Date getModifiedDate() {
    return modifiedDate;
  }

  /**
   * @param modifiedDate
   */
  public void setModifiedDate(Date modifiedDate) {
    this.modifiedDate = modifiedDate;
  }

  public LSGAgeBucketProportion getLsgAgeBucketProportion() {
    return lsgAgeBucketProportion;
  }

  public void setLsgAgeBucketProportion(LSGAgeBucketProportion lsgAgeBucketProportion) {
    this.lsgAgeBucketProportion = lsgAgeBucketProportion;
  }

  public Date getStartTime() {
    return startTime;
  }

  public void setStartTime(Date startTime) {
    this.startTime = startTime;
  }

  public Date getEndTime() {
    return endTime;
  }

  public void setEndTime(Date endTime) {
    this.endTime = endTime;
  }

}
