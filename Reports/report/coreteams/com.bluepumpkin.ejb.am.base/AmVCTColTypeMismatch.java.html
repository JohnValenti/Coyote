package com.bluepumpkin.ejb.am.base;

import com.bluepumpkin.ejb.am.l10n.AmEjbBundleKey;
/**
 * Title:       VCT Queue Collection Mismacth
 * Description: Source Event selected for chain is not valid as per collection type 
 * Copyright:   Copyright (c) 2007
 * Company:     Witness System
 * @author      Amit Kumar
 * @version     1.0
 */

public class AmVCTColTypeMismatch extends AmException {

	/**
	 *  Construct
	 *
	 *  @param msgID, key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 */
	public AmVCTColTypeMismatch(String msgID) {
		super(null, msgID, DEFAULT_SEVCODE, null);
	}

	public AmVCTColTypeMismatch(Exception e) {
		super(e, AmEjbBundleKey.QUEUE_VCT_ERROR_COLLECTION_TYPE_MISMATCH, DEFAULT_SEVCODE, null);
	}

}
