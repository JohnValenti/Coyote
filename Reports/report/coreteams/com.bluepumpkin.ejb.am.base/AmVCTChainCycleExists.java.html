package com.bluepumpkin.ejb.am.base;

import com.bluepumpkin.ejb.am.l10n.AmEjbBundleKey;
/**
 * Title:       VCT Queue config chains have cycles
 * Description: Creating a chain would result in cycles 
 * Copyright:   Copyright (c) 2007
 * Company:     Witness System
 * @author      Amit Kumar
 * @version     1.0
 */

public class AmVCTChainCycleExists extends AmException {

	/**
	 *  Construct
	 *
	 *  @param msgID, key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 */
	public AmVCTChainCycleExists(String msgID) {
		super(null, msgID, DEFAULT_SEVCODE, null);
	}

	public AmVCTChainCycleExists(Exception e) {
		super(e, AmEjbBundleKey.QUEUE_VCT_ERROR_CYCLES_EXIST, DEFAULT_SEVCODE, null);
	}

}
