	@Override
	@RepositoryLogger
	public <T extends ShiftAssignmentEntity> T save(T entity, ContainerRequestContext requestContext) {
		String entityID = entity.getID() == null ? "" : Integer.toString(entity.getID());
		try {
			ShiftAssignment sa = ShiftAssignmentMapper.INSTANCE.saEntityToSa(entity);
			//Create new shift assignment if ID is null
			if (sa.getID() == null) {
				ID saId = sam.createShiftAssignment(sa);
				entity.setID(saId.toInt());
				//Otherwise, update existing assignment
			} else {
				prepareShiftAssignmentForUpdate(sa);
				sam.updateShiftAssignment(sa);
			}
			return entity;
		} catch (BbmCreateException e) {
			throw handleBbmCreateException(entity, e);
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, entityID, e);
		} catch (BbmScheduleConflictException e) {
			log.error("ShiftAssignmentRepository.save: failed to save ShiftAssignment", e);
			throw new EntityScheduleConflictException(getLocalizer(requestContext), getUserTimeZone(requestContext), e.getConflicts());
		} catch (MultiUserException e) {
			throw handleMultiUserException(ENTITY_NAME, entityID, e);
		} catch (BbmFinderException e) {
			throw handleBbmFinderException(ENTITY_NAME, entityID, e);
		} catch (BbmUpdateException e) {
			throw handleBbmUpdateException(entity, e);
		}
	}

	@Override
	@RepositoryLogger
	@ValidateJsonApiFilterParams
	public ShiftAssignmentEntity findOne(Integer id, QueryParams requestParams, ContainerRequestContext requestContext) {
		try {
			ID idObj = ID.fromInt(id);
			ShiftAssignment sa = sam.getShiftAssignmentByID(idObj);
			return ShiftAssignmentMapper.INSTANCE.saToSaEntity(sa);
		} catch (BbmObjectNotFoundException e) {
			throw handleBbmObjectNotFoundException(ENTITY_NAME, Integer.toString(id), e);
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, Integer.toString(id), e);
		} catch (BbmFinderException e) {
			throw handleBbmFinderException(ENTITY_NAME, Integer.toString(id), e);
		}
	}

	/**
	 * Performs the following operations on a ShiftAssignment (intended to be invoked prior to an update operation):
	 * - Moves any child events if the shift assignment itself has been moved.
	 * - Truncates any child events if the duration of the shift assignment has been shortened enough such that the events
	 * no longer fit.
	 */
	private void prepareShiftAssignmentForUpdate(ShiftAssignment assignmentToUpdate)
			throws BbmFinderException, RemoteException {

		//Need to pull the original assignment to check if duration or start time was modified, if so, we need to potentially
		//update the set of child events
		ShiftAssignment originalShiftAssignment = sam.getShiftAssignmentByID(assignmentToUpdate.getID());

		moveAndTruncateChildEventsIfNeeded(originalShiftAssignment, assignmentToUpdate);
	}

	/**
	 * Compares the start/end times and durations of the old and new/recently updated assignments to see if the child
	 * events in the newly updated assignment need to be moved and or truncated from the shift depending on the new
	 * duration or start time adjustments made to the updated shift assignment.
	 */
	private void moveAndTruncateChildEventsIfNeeded(ShiftAssignment originalAssignment, ShiftAssignment updatedAssignment) {

		boolean durationChanged = originalAssignment.getDuration() != updatedAssignment.getDuration();
		// If start and end times both have changed, then we consider this a "move" operation and we need to also
		// move the child events the same amount
		if (ShiftAssignmentUtil.isMoveOperation(
				new Pair<>(originalAssignment.getStartTime(), originalAssignment.getEndTime()),
				new Pair<>(updatedAssignment.getStartTime(), updatedAssignment.getEndTime()))) {
			long timeDifference = getTimeDifferenceBetweenShiftAssignments(updatedAssignment, originalAssignment);
			//Adjust the updated assignment's start time back to the original, temporarily.  We do this because the
			//moveShiftAssignmentAndEventsIfNeeded method below will adjust it back to what it is supposed to be.
			updatedAssignment.setStartTime(new Date(updatedAssignment.getStartTime().getTime() - timeDifference));
			ShiftAssignmentUtil.moveShiftAssignmentAndEventsIfNeeded(updatedAssignment.getShiftID(), updatedAssignment,
					timeDifference, false);
		}

		// If the duration was changed, we need to truncate the shift events that may no longer fit inside the shift.
		if (durationChanged) {
			ShiftAssignmentUtil.adjustShiftEventsIfShiftAssignmentDurationHasChanged(updatedAssignment,
					new Pair<>(originalAssignment.getStartTime(), originalAssignment.getEndTime()),
					new Pair<>(updatedAssignment.getStartTime(), updatedAssignment.getEndTime()), false);
		}
	}

	private long getTimeDifferenceBetweenShiftAssignments(ShiftAssignment assignmentOne, ShiftAssignment assignmentTwo) {
		return assignmentOne.getStartTime().getTime() - assignmentTwo.getStartTime().getTime();
	}

		try {
			ID idObj = ID.fromInt(id);
			sam.deleteShiftAssignments(Collections.singleton(idObj));
		} catch (RemoteException e) {
			throw handleRemoteException(ENTITY_NAME, Integer.toString(id), e);
		} catch (MultiUserException e) {
			throw handleMultiUserException(ENTITY_NAME, Integer.toString(id), e);
