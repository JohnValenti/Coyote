package com.verint.web.rest.wfm.resource.shiftassignment;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignmentFields;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftEventAssignment;
import com.verint.web.rest.wfm.resource.shifteventassignment.ShiftEventAssignmentEntity;
import com.verint.web.rest.wfm.resource.shifteventassignment.ShiftEventAssignmentMapper;

import java.util.Collection;

/**
 * The purpose of this decorator is to handle the mapping to the unusually named
 * "addWorkResourceID" setter on the ShiftAssignment class, which is not handled by
 * the default mapping behavior.
 */
public abstract class ShiftAssignmentMapperDecorator implements ShiftAssignmentMapper {
	private final ShiftAssignmentMapper delegate;

	public ShiftAssignmentMapperDecorator(ShiftAssignmentMapper delegate) {
		this.delegate = delegate;
	}

	@Override
	public ShiftAssignment saEntityToSa(ShiftAssignmentEntity entity) {
		ShiftAssignment sa = delegate.saEntityToSa(entity);
		sa.addWorkResourceID(ID.fromInt(entity.getWorkResourceID()));
		if (entity.getShiftEvents() != null && !entity.getShiftEvents().isEmpty()) {
			for (ShiftEventAssignmentEntity event : entity.getShiftEvents()) {
				sa.fillChildObject(ShiftAssignmentFields.CHILD_SHIFT_EVENT,
						ShiftEventAssignmentMapper.INSTANCE.entityToSea(event));
			}
		}

		return sa;
	}
}
