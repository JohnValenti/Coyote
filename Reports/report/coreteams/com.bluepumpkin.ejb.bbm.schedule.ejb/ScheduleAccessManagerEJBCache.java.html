package com.bluepumpkin.ejb.bbm.schedule.ejb;

/**
 * <p>Title: Blue Pumpkin Software Basic Business Model</p>
 * <p>Description: This class manages the contents of Tangosol cache for the
 *                 purposes of Schedule Access Manager EJB</p>
 * <p>Copyright: Copyright (c) 2002</p>
 * <p>Company: Blue Pumpkin Software, inc</p>
 * @author Greg Fichtenholtz
 * @version 1.0
 */
import java.util.*;
import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.common.cache.*;
import com.bluepumpkin.ejb.bbm.*;
import com.bluepumpkin.ejb.bbm.activity.model.Event;

class ScheduleAccessManagerEJBCache {
	private Cache m_cache = null;
	private long m_cacheWindowSize;

	/**
	 * ctor
	 */
	public ScheduleAccessManagerEJBCache(long cacheWindowSize) throws Exception {
		this(cacheWindowSize, CacheFactory.getCache(JNDINames.BBM_SCHEDULEACCESS_EJBHOME,
				ScheduleAccessManagerEJB.class.getClassLoader()));
	}

	public ScheduleAccessManagerEJBCache(long cacheWindowSize, Cache cache) throws Exception {
		m_cacheWindowSize = cacheWindowSize;
		m_cache = cache;
	}

	public Date getCacheWindowStart() {
		return new Date(System.currentTimeMillis() - m_cacheWindowSize);
	}

	public Date getCacheWindowEnd() {
		return new Date(System.currentTimeMillis() + m_cacheWindowSize);
	}

	/**
	 * Adds the events for a given set of workresources to the schedule cache
	 * Events are only added if they overlap with the appropriate cache window
	 */
	public void put(Collection<ID> workResourceIDs, Collection listEvents, Date dtStart, Date dtEnd) {

		// for debugging
		long startTime = System.currentTimeMillis();

		long currentTime = System.currentTimeMillis();
		Date cacheWindowStart = new Date(currentTime - m_cacheWindowSize);
		Date cacheWindowEnd = new Date(currentTime + m_cacheWindowSize);

		/**
		 * eliminate the trivial case when the cache window does not even
		 * overlap with the given time interval
		 */
		if (dtStart.after(cacheWindowEnd) || dtEnd.before(cacheWindowStart))
			return;

		Date cachedEventsStart = cacheWindowStart.before(dtStart) ? dtStart : cacheWindowStart;
		Date cachedEventsEnd = cacheWindowEnd.after(dtEnd) ? dtEnd : cacheWindowEnd;

		/** this is for debugging */
		StringBuffer msg = new StringBuffer(100);
		if (isDebugEnabled()) {
			msg.append("Added events in time interval[");
			msg.append(cachedEventsStart).append(',').append(cachedEventsEnd);
			msg.append("] for work resources: ");
		}
		Iterator<ID> itWorkResource = workResourceIDs.iterator();
		Iterator itEvents = listEvents.iterator();
		while (itWorkResource.hasNext() && itEvents.hasNext()) {
			ID workResourceID = itWorkResource.next();
			Collection listWrkEvents = (Collection) itEvents.next();

			ArrayList listCachedEvents = null;
			if (listWrkEvents != null) {
				/**
				 * the cache should only store those events that overlap with
				 * the time window [cacheWindowStart to cacheWindowEnd], so go
				 * through this collection and filter out those events that fall
				 * outside this window
				 */
				for (Iterator itEvent = listWrkEvents.iterator(); itEvent.hasNext();) {
					Event event = (Event) itEvent.next();
					if (event.getStartTime().before(cacheWindowEnd) || event.getEndTime().after(cacheWindowStart)) {
						if (listCachedEvents == null)
							listCachedEvents = new ArrayList();
						listCachedEvents.add(event);
					}
				}
			}

			final CachedEventsPerWorkResource cachedEvents = new CachedEventsPerWorkResource(cachedEventsStart,
					cachedEventsEnd, listCachedEvents);

			String sDone = "false";
			if (m_cache != null) {
				sDone = (String) m_cache.invoke(workResourceID, new AbstractCacheProcessor() {

					private static final long serialVersionUID = 1L;

					@Override
					public Object processItem(ICacheEntry entry) {
						entry.setItemValue(cachedEvents);
						return "true";
					}
				});
			}

			/** this is for debugging */
			if (isDebugEnabled()) {
				CachedEventsPerWorkResource cachedEventsStored = (CachedEventsPerWorkResource) m_cache
						.get(workResourceID);
				msg.append(workResourceID);
				if (itEvents.hasNext())
					msg.append(",");
			}
		}

		if (isDebugEnabled()) {
			long endTime = System.currentTimeMillis();
			long duration = endTime - startTime;

			msg.append(" to schedule cache");
			ScheduleAccessManagerEJB.m_cat.debug(msg.toString());
			msg.append(" caching took " + duration + " milli seconds");
		}
	}

	/**
	 * Retrieves the set of events from the cache
	 */
	public HashMap getEventsForWorkResources(Collection<ID> workResourceIDs, Date dtStart, Date dtEnd) {
		/** this is where we shall store the events that we find in the cache */
		HashMap mapEvents = new HashMap();

		for (Iterator it = workResourceIDs.iterator(); it.hasNext();) {
			ID workResourceID = (ID) it.next();

			CachedEventsPerWorkResource cachedEvents = null;
			cachedEvents = (CachedEventsPerWorkResource) m_cache.get(workResourceID);
			if (cachedEvents == null) {
				continue;
			}
			/*
			 * check that the time range of cached events includes the requested
			 * time range
			 */
			else if (dtStart.before(cachedEvents.getStartDate()) || dtEnd.after(cachedEvents.getEndDate())) {
				/**
				 * this also is a miss because the time window of the cached
				 * events doesn't completely include the requested time window
				 */
				continue;
			} else {
				/**
				 * now we've got the collection of events that include our
				 * requested interval, but potentially lie outside of it:
				 * --------
				 * --------------------------------------------timeline--->
				 * cached events start dtStart dtEnd cached events end so let's
				 * filter out only those events that overlap with time window
				 * dtStart->dtEnd
				 */
				ArrayList listEventsToReturn = null;
				Collection listWrkEvents = cachedEvents.getEvents();
				if (listWrkEvents != null) {
					for (Iterator itEvent = listWrkEvents.iterator(); itEvent.hasNext();) {
						Event event = (Event) itEvent.next();
						if (event.getStartTime().before(dtEnd) && event.getEndTime().after(dtStart)) {
							if (listEventsToReturn == null)
								listEventsToReturn = new ArrayList();
							listEventsToReturn.add(event);
						}
					}
				}
				mapEvents.put(workResourceID, listEventsToReturn);
			}
		}
		return mapEvents;
	}

	/**
	 * Flush the contents of the schedule cache for the given set of work
	 * resources and the given start and end date
	 */
	public void flush(Collection workResourceIDs, Date dtStart, Date dtEnd) {
		/** this is for debugging purposes */
		StringBuffer msg = new StringBuffer(100);
		msg.append("Cleared schedule cache for the following work resources: ");
		boolean bCleared = false;

		for (Iterator it = workResourceIDs.iterator(); it.hasNext();) {
			ID workResourceID = (ID) it.next();
			CachedEventsPerWorkResource cachedEvents = (CachedEventsPerWorkResource) m_cache.get(workResourceID);
			if (cachedEvents != null && cachedEvents.getStartDate().before(dtEnd)
					&& cachedEvents.getEndDate().after(dtStart)) {
				m_cache.remove(workResourceID);
				/** this is for debugging purposes */
				if (bCleared)
					msg.append(",");
				else
					bCleared = true;
				msg.append(workResourceID);
			}
		}
		if (isDebugEnabled() && bCleared) {
			ScheduleAccessManagerEJB.m_cat.debug(msg.toString());
		}
	}

	boolean isDebugEnabled() {
		return ScheduleAccessManagerEJB.m_cat.isDebugEnabled();
	}
}
