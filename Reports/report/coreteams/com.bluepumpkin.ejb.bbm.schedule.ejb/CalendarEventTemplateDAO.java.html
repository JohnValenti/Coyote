package com.bluepumpkin.ejb.bbm.schedule.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmObjectNotFoundException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.dao.DAORoot;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventTemplateFields;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventTemplateImpl;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.core.base.MultiUserException;

class CalendarEventTemplateDAO extends DAORoot {
	// meta data info.
	private static FieldInfo m_fieldInfo = new CalendarEventTemplateFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public CalendarEventTemplateDAO() {
		super();
	}

	public CalendarEventTemplateDAO(Jdmo dmo) {
		super(dmo);
	}

	/* this method is an override from the base class */
	protected ValueObjectBase createValueObject() {
		return new CalendarEventTemplateImpl();
	}

	/** override from base class */
	protected DAOBase createChildDAO(int iType) {
		// only one child dao.
		return new CalendarEventTemplateAttendeeDAO(m_dmo);
	}

	/** This is an override of the base class to persist the event precedence */
	public ID createObject(ValueObjectBase objValue, boolean bBatched) throws BbmCreateException {
		CalendarEventTemplateImpl eventTemplate = (CalendarEventTemplateImpl) objValue;
		try {
			prepareToCreate(eventTemplate);
		} catch (Exception e) {
			throw new BbmCreateException(e);
		}
		return super.createObject(objValue, bBatched);
	}

	/* this is an override of the base class */
	public void updateObject(ValueObjectBase objValue) throws MultiUserException, BbmUpdateException {
		CalendarEventTemplateImpl eventTemplate = (CalendarEventTemplateImpl) objValue;
		try {
			prepareToUpdate(eventTemplate);
		} catch (Exception e) {
			throw new BbmUpdateException(e);
		}
		super.updateObject(objValue);
	}

	public short getTemplateTypeByID(ID templateID) throws BbmObjectNotFoundException, BbmFinderException {
		StringBuffer buf = new StringBuffer(100);
		buf.append("SELECT TEMPLATETYPE FROM CALENDAREVENTTEMPLATE ");
		buf.append("WHERE ID = " + templateID);

		try {
			JdmoRowset rs = m_dmo.createRowset(buf.toString());
			if (rs.next()) {
				return rs.getShort("TEMPLATETYPE");
			} else {
				throw new BbmObjectNotFoundException(templateID.toString());
			}
		} catch (JdmoException e) {
			throw new BbmFinderException(e);
		}
	}

	CalendarEventTemplateImpl prepareToCreate(CalendarEventTemplateImpl eventTemplate) throws Exception {
		eventTemplate.checkIsValid();
		int precedence = EventPrecedenceRule.getEventOverlayPrecedence(eventTemplate);
		eventTemplate.setOverlayPrecedence(precedence);
		return eventTemplate;
	}

	CalendarEventTemplateImpl prepareToUpdate(CalendarEventTemplateImpl eventTemplate) throws Exception {
		/* calendar event template precedence is decided by event type only */
		if (eventTemplate.isFieldValueSet(CalendarEventTemplateFields.EVENTTYPE)) {
			int precedence = EventPrecedenceRule.getEventOverlayPrecedence(eventTemplate);
			eventTemplate.setOverlayPrecedence(precedence);
		}
		return eventTemplate;
	}
}
