import java.util.*;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.*;
import com.bluepumpkin.ejb.core.base.*;
import com.bluepumpkin.ejb.bbm.base.*;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.schedule.model.*;
import com.bluepumpkin.ejb.bbm.vo.*;

class CalendarEventAttendeeDAO extends DAOBase<CalendarEventAttendee> {
	// meta data info.
	private static FieldInfo m_fieldInfo = new CalendarEventAttendeeFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public CalendarEventAttendeeDAO() {
		super();
	}

	public CalendarEventAttendeeDAO(Jdmo dmo) {
		super(dmo);
	}

	protected CalendarEventAttendee createValueObject() {
		return new CalendarEventAttendee();
	}

	/**
	 * update a value object, save its state to database this is an override of
	 * the base class
	 */
	public void updateObject(CalendarEventAttendee objValue) throws MultiUserException, BbmUpdateException {
		try {
			CalendarEventAttendee attendee = (CalendarEventAttendee) objValue;
			StringBuffer sqlBuf = new StringBuffer(100);
			sqlBuf.append("UPDATE CALENDAREVENTATTENDEE ");
			sqlBuf.append("SET ISLOCKED = ").append(attendee.isLocked() ? "1" : "0");
			sqlBuf.append(",ISATTENDING = ").append(attendee.isAttending() ? "1" : "0");
			sqlBuf.append(" WHERE ");
			sqlBuf.append(" WORKRESOURCEID = ");
			sqlBuf.append(attendee.getWorkResourceID());
			sqlBuf.append(" AND CALENDAREVENTASSIGNMENTID = ");
			sqlBuf.append(attendee.getParentID());
			m_dmo.executeCommand(sqlBuf.toString());
		} catch (JdmoException e) {
			throw new BbmUpdateException(e);
		}
	}

	/**
	 * delete objects given either: a list of their ids or a list of all fields
	 * that uniquely identify each object. For calendar event attendee this
	 * means workresourceID + calendar event assignment ID
	 */
	public void deleteObjects(Collection objs) throws BbmRemoveException {
		if (objs != null && !objs.isEmpty()) {
			Object obj = objs.iterator().next();
			if (obj instanceof ID) {
				// this is a collection of IDs. The super class knows how to
				// deal with that
				super.deleteObjects(objs);
			} else {
				// this is most likely a collection of CalendarEventAttendee
				// objects
				// with WORKRESOURCEID and CALENDAREVENTASSIGNMENTID fields
				// initialized
				try {
					Iterator it = objs.iterator();
					while (it.hasNext()) {
						obj = it.next();
						if (obj instanceof CalendarEventAttendee) {
							CalendarEventAttendee attendee = (CalendarEventAttendee) obj;
							StringBuffer sqlBuf = new StringBuffer(100);
							sqlBuf.append("DELETE FROM CALENDAREVENTATTENDEE ");
							sqlBuf.append("WHERE ");
							sqlBuf.append("WORKRESOURCEID = ");
							sqlBuf.append(attendee.getWorkResourceID());
							sqlBuf.append(" AND CALENDAREVENTASSIGNMENTID = ");
							sqlBuf.append(attendee.getParentID());
							m_dmo.addBatch(sqlBuf.toString());
						}
					}
					m_dmo.executeBatch();
				} catch (JdmoException e) {
					throw new BbmRemoveException(e);
				}
			}
		}
	}
}
