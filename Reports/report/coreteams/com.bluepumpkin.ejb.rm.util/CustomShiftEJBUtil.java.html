package com.bluepumpkin.ejb.rm.util;

import java.util.Collection;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.activity.model.Event;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.requests.custshift.model.CustShiftReq;

/**
 * Title:        CustomShiftUtil
 * Description:  Server Side utility class for My Custom Shift Requests
 * Copyright:    Copyright (c) 2011
 * Company:      Verint Systems, Inc
 *
 * @author Sameet
 */
public class CustomShiftEJBUtil {
	private static final Category LOG = Log.initCategory(CustomShiftEJBUtil.class.getName());

	private CustomShiftEJBUtil() {
		// Private constructor compliant with SonarQube
	}

	public static String getRequestSubTypeStringKey(int subType) {
		String key;

		if (subType == CustShiftReq.REQUEST_SUB_TYPE_OT_SHIFT_CHANGE.toInt()) {
			key = RmEjbBundleKey.CSR_SUBTYPE_OT_SHIFT_CHANGE;
		} else if (subType == CustShiftReq.REQUEST_SUB_TYPE_OT_NEW_SHIFT.toInt()) {
			key = RmEjbBundleKey.CSR_SUBTYPE_NEW_OT_SHIFT;
		} else if (subType == CustShiftReq.REQUEST_SUB_TYPE_REG_SHIFT_CHANGE.toInt()) {
			key = RmEjbBundleKey.CSR_SUBTYPE_REG_SHIFT_CHANGE;
		} else {
			key = RmEjbBundleKey.CSR_SUBTYPE_NEW_REG_SHIFT;
		}
		return key;
	}


	/*
	 *  The code here is similar to what we do for ShiftExistValidationRule when comparing, but there we compare different objects and do much more logic, so we did not refacture that.
	 */
	public static boolean isPublishedShiftInSync(ShiftAssignment sa, ScheduleAccessManager sam) {

		Collection events;
		try {
			events = RequestUtil.getEventsDuringPeriod(sa.getWorkResourceID(), Event.EVENT_TYPE_SHIFT_ASSIGNMENT, true, sa.getStartTime(),sa.getEndTime(), sam);
			for (Object event: events) {
				if (event instanceof ShiftAssignment){
					ShiftAssignment publishedSA = (ShiftAssignment) event;
					if (publishedSA != null && sa.getStartTime().equals(publishedSA.getStartTime())
							&& sa.getStartTime().equals(publishedSA.getStartTime())
							&& sa.getEndTime().equals(publishedSA.getEndTime())) {
						return true;
					}
				}
			}

		} catch (Exception ex) {
			ex.printStackTrace();
			LOG.error(ex);
		}
		return false;
	}
}
