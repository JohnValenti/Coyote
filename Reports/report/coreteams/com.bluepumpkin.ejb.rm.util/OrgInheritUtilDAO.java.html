package com.bluepumpkin.ejb.rm.util;

import java.util.ArrayList;
import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.dao.DAOUtil;
import com.bluepumpkin.ejb.rm.Log;

public abstract class OrgInheritUtilDAO extends DAOBase {

    protected static final Category LOG = Log.initCategory(OrgInheritUtilDAO.class.getName());

    /**
     * Default constructor for the creating object instance of the class.
    */
    public OrgInheritUtilDAO() {
        super();
    }

    /**
     * Constructor for the creating object instance of the class.
     * @param Jdmo  - Jdmo instance used for executing sql statements.
    */
    public OrgInheritUtilDAO(Jdmo dmo) {
        super(dmo);
    }
    
    /**
     * Returns the Category for the class
     * @return Category - Category object for the class
     */
    protected Category getCategory() {
        return LOG;
    }

   
    
    /**
     * Given an OrgID, return a Collection of that orgID, and (if shallow is false) its child orgs.
     * @param orgID - The orgID you want need. 
     * @param shallow - If true, also put the org's ancestors in the Collection, otherwise, just OrgID.
     * @param includeOrg - If true, we will include orgID in the collection. If false, we will not.
     * @return a Collection of that orgID, and (if shallow is false) its child orgs.
     * @throws Exception
     */
    @SuppressWarnings("unchecked")
	public Collection<ID> getChildOrgIdList(ID orgID, boolean shallow, boolean includeOrg)
        throws Exception // NOSONAR
    {
        Collection<ID> lChildOrgIDList;
        
        if (shallow) 
        {
            //We only want the settings that are defined at the given organization.
            lChildOrgIDList = new ArrayList<ID>(1);
        }
        else 
        {
            // we want settings that are inherited from parent organization
            // if they are missing from the ones defined at the organization
            lChildOrgIDList = DAOUtil.getChildOrganizations(orgID, m_dmo);
        }
        if (includeOrg)
            lChildOrgIDList.add(orgID);
        
        return lChildOrgIDList;
    }
    
 
  
  

	 
}
