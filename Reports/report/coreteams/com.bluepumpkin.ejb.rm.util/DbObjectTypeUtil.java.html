package com.bluepumpkin.ejb.rm.util;

import java.sql.Timestamp;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;

public class DbObjectTypeUtil {

	private DbObjectTypeUtil() {

	}

	public static int getSqlIntegerTypeFromJavaType(Class<?> javaType) { // NOSONAR

		if (javaType == null) {
			throw new RmRuntimeException("Null java type in getSqlTypeFromJavaType");
		} else if (javaType.equals(String.class)) {
			return java.sql.Types.VARCHAR;
		} else if (javaType.equals(Integer.class) || javaType.equals(int.class) || javaType.equals(ID.class)) {
			return java.sql.Types.INTEGER;
		} else if (javaType.equals(Timestamp.class) || javaType.equals(Date.class)) {
			return java.sql.Types.TIMESTAMP;
		} else if (javaType.equals(Long.class) || javaType.equals(long.class)) {
			return java.sql.Types.BIGINT;
		} else if (javaType.equals(Boolean.class) || javaType.equals(boolean.class)) {
			return java.sql.Types.BIT;
		} else if (javaType.equals(Short.class) || javaType.equals(short.class)) {
			return java.sql.Types.SMALLINT;
		} else if (javaType.equals(Byte.class) || javaType.equals(byte.class)) {
			return java.sql.Types.TINYINT;
		} else if (javaType.equals(Float.class) || javaType.equals(float.class)) {
			return java.sql.Types.REAL;
		} else if (javaType.equals(Double.class) || javaType.equals(double.class)) {
			return java.sql.Types.FLOAT;
		} else {
			throw new IllegalArgumentException(String.format("No mapping to sql server for type %s", javaType));
		}

	}

	public static String getSqlTypeFromJavaType(Class<?> javaType) { // NOSONAR
		if (javaType == null) {
			throw new RmRuntimeException("Null java type in getSqlTypeFromJavaType");
		} else if (javaType.equals(String.class)) {
			return "varchar (max)";
		} else if (javaType.equals(Integer.class) || javaType.equals(int.class) || javaType.equals(ID.class)) {
			return "int";
		} else if (javaType.equals(Timestamp.class) || javaType.equals(Date.class)) {
			return "datetime";
		} else if (javaType.equals(Long.class) || javaType.equals(long.class)) {
			return "bigint";
		} else if (javaType.equals(Boolean.class) || javaType.equals(boolean.class)) {
			return "bit";
		} else if (javaType.equals(Short.class) || javaType.equals(short.class)) {
			return "smallint";
		} else if (javaType.equals(Byte.class) || javaType.equals(byte.class)) {
			return "tinyint";
		} else if (javaType.equals(Float.class) || javaType.equals(float.class)) {
			return "real";
		} else if (javaType.equals(Double.class) || javaType.equals(double.class)) {
			return "float";
		} else {
			throw new IllegalArgumentException(String.format("No mapping to sql server for type %s", javaType));
		}

	}




}
