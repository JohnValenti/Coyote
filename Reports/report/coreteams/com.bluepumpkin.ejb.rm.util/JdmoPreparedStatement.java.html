/**
 * 
 * Creates JdmoRowset using prepared statements. 
 * Clients should use the static functions (constructor is package private for unit testing)
 * 
 * In clause parameter is specified using a single ? without parenthesis 
 * for example: Select name from Organization where ID in ?
 *  
 * The currently supported types for in clauses are ID (only integer type) and the integer types Integer,Long,Short
 * 
 * If number of parameters specified in the collection parameter corresponding to the in clause is less than MAX_IN_CLAUSE_PARAMS then
 * the parameters are expanded directly i.e. "ID in ?" is translated into "ID in (?,?,...)". 
 * If the number is greater than MAX_IN_CLAUSE_PARAMS  then a temp table is used automatically. 
 * Temp tables are called #JDMO_PREPARED_STATEMENT_IN_ID_LIST_0 for the first in clause parameter, ...LIST_1 for the  next and so on. 
 * The name is not random since that would mean that the SQL server plan cache would not be used. 
 * There is a slight limitation that the function might not be reentrant due to this.  
 * 
 * MAX_IN_CLAUSE_PARAMS must be a small enough value so that we do not pollute the SQL server plan cache with multiple versions of the 
 * same query. Currently this is 10 which means that there can be 11 distinct version of any given query in the plan cache (ten plans when 
 * the parameter count for the in clause varies from one to ten, plus the one that uses the temp table for counts > 10)
 *  
 *
 */
public class JdmoPreparedStatement {

	private static final int MAX_IN_CLAUSE_PARAMS = 10;
	public static final JdmoParam NullInteger = new JdmoParam(null, java.sql.Types.INTEGER);
	public static final JdmoParam NullVarChar = new JdmoParam(null, java.sql.Types.VARCHAR);
	public static final JdmoParam NullBit = new JdmoParam(null, java.sql.Types.BIT);
	public static final JdmoParam NullTimeStamp = new JdmoParam(null, java.sql.Types.TIMESTAMP);

	private final Jdmo dmo;
	private List<Object> tempTableValuesList;
	private StringBuilder tempTableCreateSql;
	private List<Object> adjustedParameterList;
	private String adjustedSql;


	//Should really be private, but package private just for unit testing. 
	JdmoPreparedStatement(Jdmo dmo) {
		this.dmo = dmo;
	}

	JdmoRowset createRowset(String sql, Object... parameters) throws JdmoException {
		adjustSqlAndParameters(sql, parameters);
		JdmoQuery query = dmo.createQuery(adjustedSql, Jdmo.PARAM_QUERY);
		setupParamaters(query);
		return dmo.createRowset(query, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
	}

	JdmoRowset createStoredProcedureRowset(String procedureName, Object... parameters) throws JdmoException {
		adjustedParameterList = adjustParametersForStoredProcedure(parameters);
		adjustedSql = procedureName;
		JdmoQuery query = dmo.createQuery(adjustedSql, Jdmo.STORPROC_QUERY);
		setupParamaters(query);
		return dmo.createRowset(query, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
	}
	
	void executeStoredProcedure(String procedureName, Object... parameters) throws JdmoException {
		adjustedParameterList = adjustParametersForStoredProcedure(parameters);
		adjustedSql = procedureName;
		JdmoQuery query = dmo.createQuery(adjustedSql, Jdmo.STORPROC_QUERY_NORS);
		setupParamaters(query);
		dmo.execute(query);
	}

	int executeSql(String sql, Object... parameters) throws JdmoException {

		adjustSqlAndParameters(sql, parameters);

		HashMap<Integer, Object> paramMap = new HashMap<Integer, Object>();
		int index = setupTempTableIfNeeded(paramMap);

		for (Object param : adjustedParameterList) {
			paramMap.put(index, param);
			index++;
		}

		JdmoPCommand pc = dmo.createPCommand(adjustedSql);
		return dmo.executePCommand(pc, paramMap);
	}

	private void setupParamaters(JdmoQuery query) throws JdmoException {
		int index = setupTempTableIfNeeded(query);
		for (Object param : adjustedParameterList) {
			index = setParamFromObject(query, index, param);
		}
	}

	private int setupTempTableIfNeeded(JdmoQuery query) {
		int index = 1;
		if (tempTableValuesList != null) {
			for (Object values : tempTableValuesList) {
				setJdmoCsvList(query, index, (Collection<?>) values);
				index++;
			}
		}
		return index;
	}

	private int setupTempTableIfNeeded(Map<Integer, Object> paramMap) {
		int index = 1;
		if (tempTableValuesList != null) {
			for (Object values : tempTableValuesList) {
				paramMap.put(index, convertCollectionToCsv((Collection<?>) values));
				index++;
			}
		}
		return index;
	}

	//Used only for unit testing
	String getAdjustedSql() {
		return adjustedSql;
	}

	//Used only for unit testing
	List<Object> getAdjustedParameters() {
		return adjustedParameterList;
	}

	//package private for unit testing
	void adjustSqlAndParameters(String sql, Object[] parameters) {

		adjustedParameterList = new ArrayList<Object>(parameters.length);
		adjustedSql = sql;

		int i = 0;
		for (Object param : parameters) {

			//If a parameter is a collection then it will not appear directly in adjustedParameterList
			//Either it is moved to  tempTableValuesList or the values in the collection are expanded into adjustedParameterList
			//See handleInClauseParameters for details
			if (param instanceof Collection<?>) {
				Collection<?> values = (Collection<?>) param;
				handleInClauseParameters(values);
				i = adjustInClauseSql(values.size(), i);
				continue;
			}
			adjustedParameterList.add(param);
		}

		if (tempTableCreateSql != null) {
			adjustedSql = tempTableCreateSql + adjustedSql;
		}
	}

	List<Object> adjustParametersForStoredProcedure(Object[] parameters) {

		List<Object> result = new ArrayList<Object>(parameters.length);

		for (Object param : parameters) {

			//If a parameter is a collection then it will not appear directly in adjustedParameterList
			//Either it is moved to  tempTableValuesList or the values in the collection are expanded into adjustedParameterList
			//See handleInClauseParameters for details
			if (param instanceof Collection<?>) {
				Collection<?> values = (Collection<?>) param;
				result.add(convertCollectionToCsv(values));
				continue;
			}
			result.add(param);
		}

		return result;
	}

	private int adjustInClauseSql(int count, int inClauseIndex) {
		if (count > MAX_IN_CLAUSE_PARAMS) {
			String tempTableName = String.format("#JDMO_PREPARED_STATEMENT_IN_ID_LIST_%d", inClauseIndex);
			addIDsTempTable(tempTableName);
			String replaceString = String.format(" in (select id from %s) ", tempTableName);
			replaceInClauseWith(replaceString);
			return inClauseIndex + 1;
		}

		String paramList = RmUtil.repeat("?", ",", count);

		String replaceString = String.format(" in (%s) ", paramList);
		replaceInClauseWith(replaceString);
		return inClauseIndex;
	}

	private void addIDsTempTable(String tempTableName) {

		String sql = tempTableSql(tempTableName);

		if (tempTableCreateSql == null) {
			tempTableCreateSql = new StringBuilder();
		}
		tempTableCreateSql.append(sql);

	}

	static String tempTableSql(String tempTableName) {
		return String.format("if Object_ID('tempdb..%s') is Not Null Drop Table %s  \n  "
				+ "select cast( case when item='null' then null else item end as int) id into %s from FN_DelimitedSplitMax(?,',') \n",
				tempTableName, tempTableName, tempTableName);
	}

	private void replaceInClauseWith(String replaceWith) {
		adjustedSql = adjustedSql.replaceFirst("(?i)\\s+IN\\s+\\?", replaceWith);
	}

	private void handleInClauseParameters(Collection<?> values) {
		if (values.size() > MAX_IN_CLAUSE_PARAMS) {
			//we need to use a temp table for these values
			if (tempTableValuesList == null) {
				tempTableValuesList = new ArrayList<Object>();
			}
			tempTableValuesList.add(values);
			return;
		}

		//Since the number of values is less than MAX_IN_CLAUSE_PARAMS we will expand the in clause to  in (?,?,...)
		//this means that we need to expand the in clause into the parameter list
		for (Object value : values) {
			if (value == null) {
				//since the in clause can only include integers we make any nulls be NullInteger
				adjustedParameterList.add(NullInteger);
			} else {
				adjustedParameterList.add(value);
			}
		}
	}

	private static int setParamFromObject(JdmoQuery query, int parIndex, Object obj) throws JdmoException { // NOSONAR

		if (obj == null) {
			throw new IllegalArgumentException(
					"JdmoPreparedStatement: Null values should be passed using an instance of JdmoParam. Use the nullable helpers");
		} else if (obj instanceof String) {
			query.setParString(parIndex, (String) obj);
		} else if (obj instanceof Integer) {
			query.setParInt(parIndex, (Integer) obj);
		} else if (obj instanceof Timestamp) {
			query.setParTimestamp(parIndex, (Timestamp) obj);
		} else if (obj instanceof Date) {
			query.setParTimestamp(parIndex, TimeZoneUtil.toTimestamp((Date) obj));
		} else if (obj instanceof ID) {
			query.setParID(parIndex, (ID) obj);
		} else if (obj instanceof Long) {
			query.setParLong(parIndex, (Long) obj);
		} else if (obj instanceof Boolean) {
			query.setParBoolean(parIndex, (Boolean) obj);
		} else if (obj instanceof JdmoParam) {
			JdmoParam jdmoParam = (JdmoParam) obj;
			setJdmoParam(query, parIndex, jdmoParam);
		} else if (obj instanceof Short) {
			query.setParShort(parIndex, (Short) obj);
		} else if (obj instanceof Byte) {
			query.setParByte(parIndex, (Byte) obj);
		} else if (obj instanceof Float) {
			query.setParFloat(parIndex, (Float) obj);
		} else if (obj instanceof Double) {
			query.setParDouble(parIndex, (Double) obj);
		} else if (obj instanceof ScopeID) {
			query.setParString(parIndex, obj.toString());
		} else if (obj instanceof BigDecimal) {
			query.setParBigDecimal(parIndex, (BigDecimal) obj);
		} else {
			throw new IllegalArgumentException(String.format("Parameter is of type %s which is not implemented by JdmoPreparedStatement",
					obj.getClass()));
		}
		return parIndex + 1;
	}

	private static void setJdmoCsvList(JdmoQuery query, int parIndex, Collection<?> values) {
		query.setParString(parIndex, convertCollectionToCsv(values));
	}

	static String convertCollectionToCsv(Collection<?> values) {
		StringBuilder csv = new StringBuilder();
		int i = 0;
		int count = values.size();
		for (Object value : values) {

			csv.append(toCsvStringValue(value));
			i++;
			if (i != count) {
				csv.append(",");
			}
		}
		return csv.toString();
	}

	static String toCsvStringValue(Object value) {

		if (value == null) {
			return "null";
		}

		if (value instanceof ID) {
			ID id = (ID) value;
			//do this so that id.toInt() will throw if the ID is not an integer
			return Integer.toString(id.toInt());
		} else if (value instanceof Integer || value instanceof Long || value instanceof Short) {
			return value.toString();
		} else {
			throw new IllegalArgumentException(String.format(
					"Unexpected type for in clause. %s. Currently supported types are ID (int) and Integer", value.getClass()));
		}
	}

	private static void setJdmoParam(JdmoQuery query, int parIndex, JdmoParam obj) {
		if (obj.getValue() == null) {
			query.setParNull(parIndex, obj.sqltype);
			return;
		}
		if (obj.sqltype == Types.SQLXML) {
			query.setParSqlXml(parIndex, (String) obj.value);
			return;
		}

		throw new IllegalArgumentException("JdmoParam object should be null or of type SQLXML");
	}

	public static Map<ID, Integer> getIDToIntegerMap(Jdmo dmo, String sql, Object... parameters) {
		try {
			Map<ID, Integer> result = new HashMap<ID, Integer>();
			JdmoRowset rs = createRowset(dmo, sql, parameters);
			while (rs.next()) {
				result.put(rs.getID(1), rs.getInt(2));
			}
			return result;
		} catch (JdmoException e) {
			throw new RmRuntimeException(e);
		}
	}

	public static int executeScalarInt(Jdmo dmo, int defaultValue, String sql, Object... parameters) {
		try {
			JdmoRowset rs = createRowset(dmo, sql, parameters);
			if (rs.next()) {
				Integer val = rs.getInt(1);
				return val == null ? defaultValue : val;
			} else {
				return defaultValue;
			}
		} catch (JdmoException e) {
			throw new RmRuntimeException(e);
		}
	}

	public static Object nullable(Integer value) {
		return value == null ? NullInteger : value;
	}

	public static Object nullable(ID value) {
		return value == null ? NullInteger : value;
	}

	public static Object nullable(String value) {
		return value == null ? NullVarChar : value;
	}

	public static Object getNullObjectForType(Class<?> propertyType) {
		if (propertyType.equals(Integer.class) || propertyType.equals(ID.class)) {
			return NullInteger;
		}

		if (propertyType.equals(String.class)) {
			return NullVarChar;
		}

		if (propertyType.equals(Boolean.class)) {
			return NullBit;
		}
		
		if (propertyType.equals(Date.class) || propertyType.equals(Timestamp.class)) {
			return NullTimeStamp;
		}

		throw new IllegalArgumentException("Nulls are not supported for the given type: " + propertyType.getName());

	}
