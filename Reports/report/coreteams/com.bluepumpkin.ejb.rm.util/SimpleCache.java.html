package com.bluepumpkin.ejb.rm.util;

import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.util.StringUtil;

public class SimpleCache {

	private final Map<String, Object> cache = new HashMap<String, Object>();

	/**
	 * Returns the map for the specified key.
	 * If the key does not exist an empty map is created for the key and returned.
	 */
	public <K, V> Map<K, V> getMap(String key) {
		checkKey(key);

		@SuppressWarnings("unchecked")
		Map<K, V> cachedMap = (Map<K, V>) cache.get(key);
		if (cachedMap != null) {
			return cachedMap;
		}

		cachedMap = new HashMap<K, V>();

		cache.put(key, cachedMap);
		return cachedMap;
	}

	private static void checkKey(String key) {
		if (StringUtil.isEmptyOrWhiteSpace(key)) {
			throw new IllegalArgumentException("The key to SimpleCache cannot be null or empty");
		}
	}
}
