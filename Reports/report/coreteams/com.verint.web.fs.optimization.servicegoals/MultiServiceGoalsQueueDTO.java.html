package com.verint.web.fs.optimization.servicegoals;

import java.io.Serializable;
import java.util.List;

import com.bluepumpkin.ejb.bbm.timeseries.model.ClientServiceGoalTraceCube;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastTraceCube;
import com.verint.web.fs.optimization.MultiQueueDTO;

/**
 * The Class MultiServiceGoalsQueueDTO. Instances of this class represent data transfer objects for multi queue selections
 * in the context of service goals.
 */
public class MultiServiceGoalsQueueDTO extends MultiQueueDTO<SingleServiceGoalsQueueDTO> implements Serializable {

	private static final long serialVersionUID = 1389921701456122502L;
	
	private ClientServiceGoalTraceCube m_aggregatedServiceGoals; 
	private ForecastTraceCube m_aggregatedForecast;

	/**
	 * Creates a new instance of MultiServiceGoalsQueueDTO
	 * 
	 * @param queueDtos the single queue dtos
	 * @param aggregatedServiceGoals the aggregated service goals
	 */
	public MultiServiceGoalsQueueDTO(List<SingleServiceGoalsQueueDTO> queueDtos, ClientServiceGoalTraceCube aggregatedServiceGoals, 
			ForecastTraceCube aggregatedForecast) {
		super(queueDtos);
		m_aggregatedServiceGoals = aggregatedServiceGoals;
		m_aggregatedForecast = aggregatedForecast;
	}
	
	/**
	 * Creates a new instance of MultiServiceGoalsQueueDTO
	 * 
	 * @param queueDtos the single queue dtos
	 */
	public MultiServiceGoalsQueueDTO(List<SingleServiceGoalsQueueDTO> queueDtos) {
		super(queueDtos);
	}
	
	/**
	 * Gets the aggregated service goals for the constituent queues.
	 * 
	 * @return the service goals
	 */
	public ClientServiceGoalTraceCube getServiceGoals() {
		return m_aggregatedServiceGoals;
	}
	
	/**
	 * Gets the aggregated forecast for the constituent queues.
	 * 
	 * @return the forecast
	 */
	public ForecastTraceCube getForecast() {
		return m_aggregatedForecast;
	}
}
