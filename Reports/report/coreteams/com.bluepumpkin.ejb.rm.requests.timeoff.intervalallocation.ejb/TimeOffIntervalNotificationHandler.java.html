package com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb;

import java.util.Collection;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.schedule.model.ScheduleChangeDetails;
import com.bluepumpkin.ejb.bbm.timeoffcalculator.TONotifyMessage;
import com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.ejb.TOHoursPerDayManager;
import com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.TimeOffIntervalAllocationUtil;
import com.bluepumpkin.ejb.rm.util.LicenseUtil;
import com.bluepumpkin.ejb.rm.util.PerformanceLog;

public class TimeOffIntervalNotificationHandler {

	private static final Category LOG = Log.initCategory(TimeOffIntervalNotificationHandler.class.getName());
	
	TOHoursPerDayManager toHourPerDayManager;
	
	
	public TimeOffIntervalNotificationHandler(TOHoursPerDayManager mgr){
		toHourPerDayManager = mgr;
	}
	
	
	

	private void onReferenceScheduleChange(TONotifyMessage message) throws Exception {
		PerformanceLog perf = new PerformanceLog();

		Date asOfDate = message.getStartDate();
		Collection<ID> employeeIDs = message.getEmployeeIDs();
		List<ID> choiceIDs = getAffectedRequestChoicesForRefScheduleChange(employeeIDs, asOfDate);

		if ( choiceIDs != null && !choiceIDs.isEmpty() ) {
			
			toHourPerDayManager.onReferenceScheduleChange(message, choiceIDs);
			reprocessTimeOffChoices(choiceIDs);
		}

		perf.logDuration("onReferenceScheduleChange");
	}


	private List<ID> getAffectedRequestChoicesForRefScheduleChange(Collection<ID> employeeIDs, Date asOfDate) throws BbmFinderException {
		ReferenceScheduleAssignmentsDAO dao = null;
		try {
			dao = new ReferenceScheduleAssignmentsDAO();
			return dao.getAffectedRequestChoices(employeeIDs, asOfDate);
		} finally {
			if (dao != null) {
				dao.cleanUp();
			}
		}

	}
	
	

	public void onTONotifyMessage(TONotifyMessage message) {
		

		try {

			if (message.isScheduleChangeMessage()) {
				
				ScheduleChangeDetails changeDetails = message.getScheduleChangeDetails();
				if (changeDetails == null) {
					return;
				}
				
				onScheduleChange(changeDetails);
				return;
			}

			if (message.isReferenceScheduleChangeMessage()) {
				onReferenceScheduleChange(message);
				return;
			}

		} catch (Exception e) {
			LOG.error("Error when handling time off notification", e);
		}

	}

	public void onScheduleChange(ScheduleChangeDetails changeDetails) throws BbmException {
	
		if (!LicenseUtil.isAdvancedRMLicense()){
			return;
		}
		
		PerformanceLog perf = new PerformanceLog();
		
		Date start = changeDetails.getStartDate();
		Date end = changeDetails.getEndDate();
		@SuppressWarnings("unchecked")
		Collection<ID> employeeIDs = changeDetails.getWorkResourceIDs();
		List<ID> choiceIDs = getAffectedRequestChoicesForScheduleChange(employeeIDs, start, end);
		reprocessTimeOffChoices(choiceIDs);
		reprocessCalendarEvents(changeDetails);
		perf.logDuration("onPublishUnPublishSchedules");

	}

	private void reprocessCalendarEvents(ScheduleChangeDetails changeDetails) {
		try {
			TimeOffIntervalAllocationUtil.updateTimeOffIntervalPaidSegmentsCalendarEventsForTimeoffChoices(changeDetails);
		} catch (Exception e) {
			LOG.error(e.getMessage(), e);
		}
	}
		


	private List<ID> getAffectedRequestChoicesForScheduleChange(Collection<ID> employeeIDs, Date start, Date end) throws BbmFinderException {
		TimeOffIntervalPaidSegmentDAO dao = null;
		try {
			dao = new TimeOffIntervalPaidSegmentDAO();
			return dao.getAffectedRequestChoices(employeeIDs, start, end);
		} finally {
			if (dao != null) {
				dao.cleanUp();
			}
		}

	}

	private void reprocessTimeOffChoices(List<ID> timeOffChoiceIDs) throws BbmUpdateException{
		if (timeOffChoiceIDs == null || timeOffChoiceIDs.isEmpty()) {
			return;
		}
		TimeOffIntervalAllocationUtil.updateTimeOffIntervalPaidSegmentsForTimeoffChoices(timeOffChoiceIDs);

	}

}
