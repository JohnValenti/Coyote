package com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb;

import java.util.Collections;
import java.util.Date;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.model.TimeOffIntervalAllocation;
import com.bluepumpkin.ejb.rm.util.orm.JdmoOrm;
import com.verint.ejb.wfm.util.DAOUtil;

public class TimeoffIntervalAllocationUtilDAO {

	private Jdmo dmo = null;

	public TimeoffIntervalAllocationUtilDAO() {
		dmo = new Jdmo(false);
	}

	public void cleanUp() {
		if (dmo != null) {
			dmo.cleanUp();
			dmo = null;
		}
	}
	
	/*
	 * checkAllocation
	 * Returns a map of Timeoff Request Choices that FAIL allocation validation, including the datetime of failure
	 */
	public Map<Integer, TimeOffRequestChoiceSegment> checkAllocation(List<ID> listOfRequestsIDs) throws JdmoException {

		if (listOfRequestsIDs == null || listOfRequestsIDs.isEmpty()) {
			return Collections.emptyMap();
		}
		
		
		String xml = DAOUtil.converIdsToXml(listOfRequestsIDs);
		
		Map<Integer, TimeOffRequestChoiceSegment> map = new HashMap<Integer, TimeOffRequestChoiceSegment>();
		
		JdmoQuery jdmoQuery = dmo.createQuery("RM_PendingRequest_AvailableTime", Jdmo.STORPROC_QUERY);
		jdmoQuery.setParSqlXml(1, xml);
		
		JdmoRowset res = dmo.createRowset(jdmoQuery, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

		
		while (res.next()) {
			boolean isPass = res.getBoolean("IsPass");
			if( isPass ){
				continue;
			}
			int toRequestID = res.getInt("TIMEOFFREQUESTID");
			int toChoiceID = res.getInt("TimeoffChoiceID");
			Date segmentFailTime = TimeZoneUtil.toDate(res.getTimestamp("PaidSegmentStartTimeFailed"));
			TimeOffRequestChoiceSegment segment = new TimeOffRequestChoiceSegment(toRequestID, toChoiceID, segmentFailTime, isPass);
			map.put(toChoiceID, segment);
		}

		return map;
	}
	
	
	public List<TimeOffIntervalAllocation> getAvailableMinutesForRequest(ID requestID, Date start, Date end) {
		return JdmoOrm.toObjectsFromStoredProcedure(TimeOffIntervalAllocation.class, dmo, "RM_AvailableTime_ForRequest", requestID, start,
				end);
	}

}
