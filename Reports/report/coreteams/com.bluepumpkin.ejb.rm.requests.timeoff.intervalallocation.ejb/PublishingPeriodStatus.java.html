package com.bluepumpkin.ejb.rm.requests.timeoff.intervalallocation.ejb;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Map;

import com.bluepumpkin.ejb.bbm.base.BbmFinderException;

public class PublishingPeriodStatus {

	public enum Status {
		/**
		 * Is the given date range for an employee completely covered by fully published periods? 
		 */
		FULLY_PUBLISHED,
		/**
		 * Is the given date range for an employee completely unpublished? 
		 */
		FULLY_UNPUBLISHED,
		/**
		 * Are some of the dates in the given date range for an employee published and others unpublished?
		 */
		HAS_GAPS_IN_PUBLISHING
	}

	Map<IdDateRange, Integer> periodStatusFromDB;

	public void initialize(Collection<IdDateRange> periods, TimeOffIntervalAllocationManager manager) throws BbmFinderException {
		try {
			periodStatusFromDB = manager.getPeriodStatus(periods);
		} catch (RemoteException e) {
			throw new BbmFinderException(e);
		}
	}

	public boolean hasGaps(IdDateRange period) {
		return getPeriodStatus(period) == Status.HAS_GAPS_IN_PUBLISHING;
	}

	public Status getPeriodStatus(IdDateRange period) {

		if (periodStatusFromDB == null) {
			throw new IllegalStateException("PublishingPeriodStatus.periodStatusFromDB is null. Was initialize called?");
		}

		if (!periodStatusFromDB.containsKey(period)) {
			return Status.FULLY_UNPUBLISHED;
		}

		int fullyPublished = periodStatusFromDB.get(period);
		return fullyPublished == 1 ? Status.FULLY_PUBLISHED : Status.HAS_GAPS_IN_PUBLISHING;
	}

}
