/**
 *
 * <b>Soft Validation: applies only for partial shift swap requests</b>
 * <p>
 *    Validates that the shift activities do not include the activities specified in the rule
 * <p>
 *    Checks for both the shifts for a two way swap
 */
public class PSSNonSwappableActivites  implements Validator
{
    private static final String m_className = PSSNonSwappableActivites.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);
    
    public static final String ACTIVITY = "ACTIVITY";
    public static final ID RULE_ID = new ID(-194035);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    @Override
	public ValidationResult validate(Validatable validatable) throws Exception
    {
        String methodName = "validate";
        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));

        ShiftSwapRequest ssr = (ShiftSwapRequest) validatable;
        ShiftSwapValidationCache valCache = ssr.getCache();
        ShiftSwapItem ssItem1 = ssr.getShiftSwapItems().get(0);
        ShiftSwapItem ssItem2 = ssr.getShiftSwapItems().get(1);
        ValidationResult result = null;
        if (ssItem1.getIsPartial() || ssItem2.getIsPartial()) {
        	//get the shifts corresponding to the items
        	ScheduleAccessManager sam = RequestUtil.getScheduleAccessManager();
        	ShiftAssignment[] sas = valCache.getShiftAssignsForReqSameOrgDay(true, sam);
        	
        	//make sure the orgIDs are not duplicate
			List<ID> orgList = new ArrayList<>();
			for (Iterator<ID> it = ssr.getValidationCache().getOrgIDs().iterator(); it.hasNext();) {
	        	ID orgID = it.next();
	        	if (!orgList.contains(orgID))
	        		orgList.add(orgID);
        	}
        	
        	//for each org, find out the validator- if beeded to run for both or not
        	for (int i = 0; i < orgList.size(); i++){
	        	ID orgID = orgList.get(i);
	        	if (valCache.isValidationRequiredForOrg(orgID, Request.REQUESTTYPE_SHIFTSWAP, m_className)){
	        		//get the rule params
		        	ValidationRuleManager valMan = RmManagerFactory.getInstance().getValidationRuleManager();
					Map<String, Integer> ruleParams = valMan.getValidationRuleParams(orgID, RULE_ID);
		        	if (ruleParams != null && ruleParams.isEmpty())
		        		continue;
		        	Collection activityColl = getActivityCollFromMap(ruleParams);
		        	
		        	for (int j=0; j<ssr.getShiftSwapItems().size(); j++){
		        		if (sas[j] == null)
		        			continue;
			        	result = verifyNonSwappableActivities(sas[j], activityColl, 
			        			validatable, ssr.getShiftSwapItems().get(j), valCache);
		        	}
	        	}
        	}
        }
        if (m_cat.isDebugEnabled()) m_cat.debug(RmUtil.dumpExitMethod(methodName));
        return result;
    }
    
	private Collection getActivityCollFromMap(Map<String, Integer> map) {
    	ArrayList activityColl = new ArrayList(map.size());
    	for (Iterator it = map.values().iterator(); it.hasNext();){
    		activityColl.add(new ID(((Integer)it.next()).intValue()));
    	}
    	return activityColl;
    }
    
    private ValidationResult verifyNonSwappableActivities(ShiftAssignment sa, Collection activityColl,
    		Validatable validatable, ShiftSwapItem ssItem, ShiftSwapValidationCache valCache) throws Exception{
    	//get shift activities overlapping with the swap duration
    	for (Iterator itr = sa.getChildren().iterator(); itr.hasNext(); ) {
            ShiftEventAssignment shiftEventAssn = (ShiftEventAssignment) itr.next();
            if (activityColl.contains(shiftEventAssn.getActivityID()) &&
            		TimePeriodUtil.overlap(shiftEventAssn.getStartTime(), shiftEventAssn.getEndTime(),
            		ssItem.getStartDate(), ssItem.getEndDate()))
            	return ValidationUtil.setSoftValidationResult(validatable, 
                        RmEjbBundleKey.PSS_NON_SWAPPABLE_ACTIVITY,
                        ssItem.getStartDate(), 
                        ssItem.getEndDate(),
                        valCache.getEmployeeNameByID(ssItem.getEmployeeID()),
                        m_className);
    		
    	}
    	return null;
    }
}
