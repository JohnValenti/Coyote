package com.bluepumpkin.ejb.rm.requests.swap.request.validation;

import java.util.ArrayList;
import java.util.Date;
import java.util.Map;
import java.util.TimeZone;

import com.bluepumpkin.common.base.Log;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.swap.request.model.ShiftSwapRequest;
import com.bluepumpkin.ejb.rm.requests.swap.shiftitem.model.ShiftSwapItem;
import com.bluepumpkin.ejb.rm.util.DateUtil;
import com.bluepumpkin.ejb.rm.util.RmUtil;
import com.bluepumpkin.ejb.rm.util.ShiftSwapRequestUtil;

public class MaxNumberOneWaySwapPerPersonPeriodValidationRule implements Validator {
	public static final String MAX_ONE_WAY_SWAP_PER_PERSON_PERIOD = "MAX_ONE_WAY_SWAP_PER_PERSON_PERIOD";
	public static final String MAX_ONE_WAY_SWAP_PER_PERSON_PERIOD_UNIT = "MAX_ONE_WAY_SWAP_PER_PERSON_PERIOD_UNIT";
	public static final String ENTERPRISESETTING_CAL_STRING_DAY_OF_YEAR = "EnterpriseSettings/CAL_STARTING_DATE_YEAR";

	public static final int PERIOD_MONTH_INDEX = 0;
	public static final int PERIOD_QUARTER_INDEX = 1;
	public static final int ERROR_CORRECTION = -1;

	private static final String CLASS_NAME = MaxNumberOneWaySwapPerPersonPeriodValidationRule.class.getName();
	public static final ID RULE_ID = new ID(-194066);
	private static final Category LOG = Log.initCategory(CLASS_NAME);
	private static final String METHOD_NAME = "validate";

	@SuppressWarnings("unchecked")
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		LOG.debug(RmUtil.dumpEnterMethod(METHOD_NAME, validatable));

		ShiftSwapRequest ssReq = (ShiftSwapRequest) validatable;

		if (ssReq.getSwapType().compareTo(ShiftSwapRequest.SHIFTSWAP_TYPE_ONEWAY) != 0) {
			return null;
		}

		Map<String, Integer> ruleParams = getValidationRuleParams(getOrg(ssReq).getID());
		int maxNumberOfSwaps = (ruleParams.get(MAX_ONE_WAY_SWAP_PER_PERSON_PERIOD) == null) ? 
				ERROR_CORRECTION : ruleParams.get(MAX_ONE_WAY_SWAP_PER_PERSON_PERIOD).intValue();

		int periodUnit = (ruleParams.get(MAX_ONE_WAY_SWAP_PER_PERSON_PERIOD_UNIT) == null) ? 
				ERROR_CORRECTION : ruleParams.get(MAX_ONE_WAY_SWAP_PER_PERSON_PERIOD_UNIT).intValue();
		
		Organization org = getOrg(ssReq);
		TimeZone orgTZ = org.getTimeZone();
		TimeRange periodTimeRange = getTimeRangeOfConfiguredPeriodUnit(ssReq, periodUnit, orgTZ);

		ValidationResult result = null;
		for (ShiftSwapItem ssItem : new ArrayList<ShiftSwapItem>(ssReq.getShiftSwapItems())) {
			if (ShiftSwapItem.SWAPITEMTYPE_SHIFT.equals(ssItem.getShiftType())) {
				int actualSwapsItem = getNumberOfOneWaySwapsForEmployee(periodTimeRange, ssItem.getEmployeeID(), org);
				if (actualSwapsItem >= maxNumberOfSwaps && maxNumberOfSwaps > ERROR_CORRECTION) {
					result = ValidationUtil.setSoftValidationResult(ssReq, RmEjbBundleKey.SS_MAX_NUMBER_OF_ONE_WAY_SWAPS,
							actualSwapsItem + 1, periodTimeRange.getStartDate(), new Date(periodTimeRange.getEndDate().getTime() - 1), maxNumberOfSwaps,
							CLASS_NAME);
					break;
				}
			}
		}

		LOG.debug(RmUtil.dumpExitMethod(METHOD_NAME, result));
		return result;
	}

	protected Organization getOrg(ShiftSwapRequest ssReq) throws Exception {
		return ssReq.getCache().getOrg();
	}

	protected Map<String, Integer> getValidationRuleParams(ID orgID) throws Exception {
		return RmManagerFactory.getInstance().getValidationRuleManager().getValidationRuleParams(orgID, RULE_ID);
	}

	protected TimeRange getTimeRangeOfConfiguredPeriodUnit(ShiftSwapRequest ssReq, int periodUnit, TimeZone orgTZ) throws Exception {
		ShiftSwapItem ssItem1 = (ShiftSwapItem) ssReq.getShiftSwapItems().get(0);
		Date shiftStart = ssItem1.getStartDate();

		String strFirstDateOfYearFromEM = BbmManagerFactory.getDBConfigManager().getValue(ENTERPRISESETTING_CAL_STRING_DAY_OF_YEAR);
		Date firstDateOfYear;
		if (strFirstDateOfYearFromEM == null || strFirstDateOfYearFromEM.isEmpty()) {
			firstDateOfYear = DateUtil.getFirstDateOfCurrentCalendarYear(orgTZ);
		} else {
			firstDateOfYear = DateUtil.iso8601SecPrecisionStringToDate(strFirstDateOfYearFromEM, orgTZ);
		}

		return DateUtil.getFiscalRange(firstDateOfYear, shiftStart, periodUnit == PERIOD_MONTH_INDEX ? 1 : 3, orgTZ);
	}

	protected int getNumberOfOneWaySwapsForEmployee(TimeRange periodTimeRange, ID employeeID, Organization org) throws Exception {
		return ShiftSwapRequestUtil.getNormalizedNumberOfOneWaySwapsForEmployees(employeeID, org, periodTimeRange.getStartDate(),
				periodTimeRange.getEndDate());
	}
}
