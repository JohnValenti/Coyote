package com.verint.web.rm.rest.entity;

import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.web.rm.keys.RmPageAction;
import com.witness.web.uif.keys.PageAction;

public enum ActionCode {
	EDIT(1),
	// 2 used to be submit, but SUBMIT_TO_MANAGER(14) was accidentally added before these were consolidated, 2 was never implemented though
	ADDTOWAITLIST(3),
	MANUALLYADDTOWAITLIST(4),
	DENY(5),
	ESCALATE(6),
	WITHDRAW(7),
	VIEWDETAIL(8),
	TENTATIVELYAPPROVE(9),
	APPROVE(10),
	CANCEL_WITHDRAWAL(11),
	ACCEPT_WITHDRAWAL(12),
	REJECT_WITHDRAWAL(13),
	SUBMIT_TO_MANAGER(14);

	private int code;
	private static Map<String, ActionCode> pageActionToActionCode = new HashMap<String, ActionCode>();

	ActionCode(int code) {
		this.code = code;
	}

	public int getActionCode() {
		return code;
	}

	public static Map<String, ActionCode> getPageActionToActionCode() {
		return pageActionToActionCode;
	}

	static {
		pageActionToActionCode.put(PageAction.EDIT, EDIT);
		pageActionToActionCode.put(RmPageAction.ADD_TO_WAITLIST_ACTION, ADDTOWAITLIST);
		pageActionToActionCode.put(RmPageAction.ADD_TO_MAN_WAITLIST_ACTION, MANUALLYADDTOWAITLIST);
		pageActionToActionCode.put(PageAction.DENY, DENY);
		pageActionToActionCode.put(PageAction.ESCALATE, ESCALATE);
		pageActionToActionCode.put(PageAction.WITHDRAW, WITHDRAW);
		pageActionToActionCode.put(PageAction.VIEW_DETAIL, VIEWDETAIL);
		pageActionToActionCode.put(PageAction.TENTATIVELY_APPROVE, TENTATIVELYAPPROVE);
		pageActionToActionCode.put(PageAction.APPROVE, APPROVE);
		pageActionToActionCode.put(PageAction.CANCEL_WITHDRAW_ACTION, CANCEL_WITHDRAWAL);
		pageActionToActionCode.put(PageAction.ACCEPT_WITHDRAW_ACTION, ACCEPT_WITHDRAWAL);
		pageActionToActionCode.put(PageAction.REJECT_WITHDRAW_ACTION, REJECT_WITHDRAWAL);
		pageActionToActionCode.put(PageAction.SUBMIT_TO_MGR, SUBMIT_TO_MANAGER);
	}
}
