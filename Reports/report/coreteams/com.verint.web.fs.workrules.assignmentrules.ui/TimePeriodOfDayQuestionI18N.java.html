package com.verint.web.fs.workrules.assignmentrules.ui;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.empworkrule.model.TimePeriodOfDay;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.workrules.assignmentrules.Response;
import com.verint.web.fs.workrules.assignmentrules.question.Question;
import com.verint.web.fs.workrules.assignmentrules.question.TimePeriodOfDayQuestion.TimePeriodOfDayResponse;
import com.verint.web.fs.workrules.assignmentrules.question.TimePeriodOfDayQuestion.TimePeriodOfDayResponseType;

class TimePeriodOfDayQuestionI18N extends AbstractQuestionI18N {
	@Override
	public String getResponseText(Localizer localizer, Response response, Object... params) {
		TimePeriodOfDayResponse tpodr = (TimePeriodOfDayResponse) response;
		TimePeriodOfDayResponseType tpodrt = (TimePeriodOfDayResponseType) response.getResponseType();
		switch (tpodrt.getTimeType()) {
			case DBTimePeriod:
				TimePeriodOfDay tpod = tpodr.getTimePeriodOfDay();
				String from = localizer.formatTimeInMins(tpod.getStartTime().getMinutesSinceMindnight());
				String to = localizer.formatTimeInMins(tpod.getEndTime().getMinutesSinceMindnight());

				String name = tpod.getName(localizer);


				return localizer.i18n(
						FsWebBundleKey.BUNDLE_NAME,
						FsWebBundleKey.ASSIGNMENT_RULE_TIME_PERIOD_FORMAT,
						name,
						from,
						to);
			case StartBetween:
				return localizer.i18n(
						FsWebBundleKey.BUNDLE_NAME,
						FsWebBundleKey.ASSIGNMENT_RULE_TIME_PERIOD_START_BETWEEN,
						params);
			case EndBetween:
				return localizer.i18n(
						FsWebBundleKey.BUNDLE_NAME,
						FsWebBundleKey.ASSIGNMENT_RULE_TIME_PERIOD_END_BETWEEN,
						params);
			default:
				throw new IllegalArgumentException("Unknown ResponseType: " + tpodrt);
		}
	}

	@Override
	public String getQuestionText(Localizer localizer, Question question) {
		return localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.ASSIGNMENT_RULE_TEXT_TIME_OF_DAY);
	}

}
