package com.verint.web.rm.rest.entity.shiftbidding;

import java.util.ArrayList;
import java.util.List;

import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.BidOptionRow;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.BiddableSchedule;
import com.verint.web.rest.entity.AbstractWFOEntity;

/**
 * Represents a Biddable Schedule.
 */
public class BiddableScheduleEntity extends AbstractWFOEntity {

	private final int available;
	private final int bonus;
	private String description;
	private int minutes;
	private final int submitted;
	private String name;
	private int preference;
	private int rank;
	private String rankDisplay;
	private final List<ShiftEntity> shifts = new ArrayList<ShiftEntity>();

	public BiddableScheduleEntity(BiddableSchedule schedule) {
		setId(schedule.getID().toString());
		bonus = schedule.getBonusPoints();
		available = schedule.getNumOfAvailableInstances();
		submitted = schedule.getNumberOfBids();
	}

	public BiddableScheduleEntity(BidOptionRow schedule) {
		setId(schedule.getID().toString());
		bonus = schedule.getBonusPoints();
		available = schedule.getNumOfAvailableInstances();
		submitted = schedule.getSubmittedBids();
	}

	public int getAvailable() {
		return available;
	}

	public int getBonus() {
		return bonus;
	}

	public String getDescription() {
		return description;
	}

	public void setDescription(String description) {
		this.description = description;
	}

	public int getMinutes() {
		return minutes;
	}

	public void setMinutes(int minutes) {
		this.minutes = minutes;
	}

	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public int getPreference() {
		return preference;
	}

	public void setPreference(int preference) {
		this.preference = preference;
	}

	public int getRank() {
		return rank;
	}

	public void setRank(int rank) {
		this.rank = rank;
	}

	public String getRankDisplay() {
		return rankDisplay;
	}

	public void setRankDisplay(String rankDisplay) {
		this.rankDisplay = rankDisplay;
	}

	public List<ShiftEntity> getShifts() {
		return shifts;
	}

	public int getSubmitted() {
		return submitted;
	}
}
