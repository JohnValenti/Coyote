package com.bluepumpkin.ejb.bbm.rtaamanager.model;

import com.bluepumpkin.common.datatypes.PrimitiveTypes;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;

public class OrganizationDayFieldInfo extends FieldInfo {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	//field id constants
    public static final int ORGANIZATIONDAY_ID = 0;
    public static final int ORGANIZATIONDAY_ORGANIZATIONID = 1;
    public static final int ORGANIZATIONDAY_SPID = 2;
    public static final int ORGANIZATIONDAY_STARTTIME = 3;
    public static final int ORGANIZATIONDAY_ENDTIME = 4;
    

    private static final int ORGANIZATIONDAY_NUMOFFIELDS = 5;

    private static FieldInfoEntry m_arrFieldInfo[] = new FieldInfoEntry[ORGANIZATIONDAY_NUMOFFIELDS];
    private static String m_strTableName = "ORGANIZATIONDAY";

    // static initialization
    static {
      m_arrFieldInfo[ORGANIZATIONDAY_ID] = new FieldInfoEntry("ID", PrimitiveTypes.ID);
      m_arrFieldInfo[ORGANIZATIONDAY_ORGANIZATIONID] = new FieldInfoEntry("ORGANIZATIONID", PrimitiveTypes.ID);
      m_arrFieldInfo[ORGANIZATIONDAY_SPID] = new FieldInfoEntry("SPID", PrimitiveTypes.ID);
      m_arrFieldInfo[ORGANIZATIONDAY_STARTTIME] = new FieldInfoEntry("STARTTIME", PrimitiveTypes.DATETIME);
      m_arrFieldInfo[ORGANIZATIONDAY_ENDTIME] = new FieldInfoEntry("ENDTIME", PrimitiveTypes.DATETIME);
    }

    public OrganizationDayFieldInfo() {
        super();
    }

    // implement abstract methods in parent class
    public String getTableName() { return m_strTableName; }
    public int getIDFieldIndex() { return ORGANIZATIONDAY_ID; }
    public FieldInfoEntry [] getFieldInfo() { return m_arrFieldInfo; }

}
