	public int compareTo(Object o) {
		PlannedEventTimeLine item2 = (PlannedEventTimeLine) o;

		long lDateDiff = getStartTime().getTime() - item2.getStartTime().getTime();

		if (lDateDiff < 0)
			return -1;
		if (lDateDiff > 0)
			return 1;
		return 0;
	}

	public boolean isBasicallyEqual(PlannedEventTimeLine event) {
		if (((getShiftStartTime() == null && event.getShiftStartTime() == null) || (getShiftStartTime() != null
				&& event.getShiftStartTime() != null && getShiftStartTime().equals(event.getShiftStartTime())))
				&& getActivityID().equals(event.getActivityID())
				&& isPaid() == event.isPaid()
				&& isWork() == event.isWork()
				&& isCalendar() == event.isCalendar()
				&& isUnavailability() == event.isUnavailability()
				&& isAbsence() == event.isAbsence()
				&& isBreak() == event.isBreak()
				&& isUnpublished() == event.isUnpublished()
				&& isOverTime() == event.isOverTime() && getEventType() == event.getEventType())
			return true;
		return false;
	}
