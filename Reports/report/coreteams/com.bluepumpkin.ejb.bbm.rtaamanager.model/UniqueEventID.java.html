package com.bluepumpkin.ejb.bbm.rtaamanager.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.Event;

public class UniqueEventID implements Comparable {

	private ID m_id;
	private int m_nEventType;
	private boolean b_isUnPublished = true;

	public UniqueEventID(ID id, int nEventType, boolean isUnPublished) {
		this(id, nEventType);
		b_isUnPublished = isUnPublished;
	}

	public UniqueEventID(Event event, boolean isUnPublished) {
		this(event);
		b_isUnPublished = isUnPublished;
	}

	public UniqueEventID(ID id, int nEventType) {
		m_id = id;
		m_nEventType = nEventType;
	}

	public UniqueEventID(Event event) {
		m_id = event.getID();
		m_nEventType = event.getEventType();
	}

	public boolean isUnPublished() {
		return b_isUnPublished;
	}

	/**
	 * 
	 * @return
	 */
	public int getEventType() {
		return m_nEventType;
	}

	/**
	 * 
	 */
	public int hashCode() {
		return (m_nEventType * 1000) + m_id.hashCode();
	}

	/** 
	 * compares this ID to another ID 
	 */
	public int compareTo(Object o) {
		if (o instanceof UniqueEventID) {

			UniqueEventID uniqueEventID = (UniqueEventID) o;

			if (m_nEventType == uniqueEventID.m_nEventType) {
				return m_id.compareTo(uniqueEventID.m_id);
			} else if (m_nEventType > uniqueEventID.m_nEventType) {
				return 1;
			} else {
				return -1;
			}
		} else {
			return -1;
		}
	}

	/**
	 * compares this ID to another ID
	 */
	public boolean equals(Object o) {
		if (o instanceof UniqueEventID) {

			UniqueEventID uniqueEventID = (UniqueEventID) o;

			if ((m_nEventType == uniqueEventID.m_nEventType)
					&& (b_isUnPublished == uniqueEventID.b_isUnPublished)) {
				return m_id.equals(uniqueEventID.m_id);
			} else {
				return false;
			}
		} else {
			return false;
		}
	}

}
