package com.bluepumpkin.ejb.bbm.activity.ejb;

import java.util.*;
import java.util.function.Function;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityQueue;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityQueueFieldInfo;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

public class ActivityQueueDAO extends DAOBase<ActivityQueue> {

	private static FieldInfo m_fieldInfo = new ActivityQueueFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ActivityQueueDAO() {
		super();
	}

	public ActivityQueueDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ActivityQueue createValueObject() {
		return new ActivityQueue();
	}

	protected Collection findQueueIdsForActivity(ID activityId)
		throws BbmFinderException {
		ArrayList<ID> list = new ArrayList<>(1);
		list.add(activityId);
		Object obj = findQueueForActivities(list).get(activityId);
		if (obj == null) {
			return Collections.emptyList();
		}
		return (Collection) obj;
	}

	/**
	 * <B>findQueueForActivities</B>
	 * <p>
	 * find Queue Ids per activity for a given collection of activities
	 * <p>
	 *
	 * @param activityIds: collection of activity Ids
	 * @return Map of activity ID (key) and collection of Queue Ids (value)
	 */
	public Map<ID, List<ID>> findQueueForActivities(Collection<ID> activityIds) throws BbmFinderException {
		return findPrimaryObjectForSecondaryObject(activityIds, "A.ACTIVITYID", ActivityQueue::getActivityId, ActivityQueue::getQueueId);
	}

	/**
	 * <B>findActivityForQueues</B>
	 * <p>
	 * find Activity Ids per Queue for a given collection of queue IDs
	 * <p>
	 *
	 * @param queueIds: collection of Queue Ids
	 * @return Map of Queue ID (key) and collection of Activity Ids (value)
	 */
	public Map<ID, List<ID>> findActivityForQueues(Collection<ID> queueIds) throws BbmFinderException {
		return findPrimaryObjectForSecondaryObject(queueIds, "A.QUEUEID", ActivityQueue::getQueueId, ActivityQueue::getActivityId);
	}

	private Map<ID, List<ID>> findPrimaryObjectForSecondaryObject(Collection<ID> ids, String primaryObjectColumn, Function<ActivityQueue, ID>
		primaryObjectIdFunction, Function<ActivityQueue, ID> secondaryObjectIdFunction) throws BbmFinderException {
		try {
			if (ids == null || ids.isEmpty()) {
				return Collections.emptyMap();
			}
			String whereClause = primaryObjectColumn + " in" + m_dmo.createInClause(ids) + " ORDER BY " + primaryObjectColumn;

			Collection<ActivityQueue> activityQueueVOCollection = getObjects(whereClause);
			Map<ID, List<ID>> activityQueueIdsMap = new HashMap<>(ids.size());
			List<ID> secondaryObjectIds;
			for (ActivityQueue activityQueue : activityQueueVOCollection) {
				ID primaryObjectId = primaryObjectIdFunction.apply(activityQueue);
				ID secondaryObjectId = secondaryObjectIdFunction.apply(activityQueue);
				if (activityQueueIdsMap.containsKey(primaryObjectId)) {
					secondaryObjectIds = activityQueueIdsMap.get(primaryObjectId);
					secondaryObjectIds.add(secondaryObjectId);
				} else {
					secondaryObjectIds = new ArrayList<>(1);
					secondaryObjectIds.add(secondaryObjectId);
					activityQueueIdsMap.put(primaryObjectId, secondaryObjectIds);
				}
			}
			return activityQueueIdsMap;
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
	}

	protected void unlinkAllActivityQueues(ID activityId) throws BbmRemoveException {
		if (activityId == null) {
			return;
		}
		try {
			m_dmo.executeCommand("delete " + getFieldInfo().getTableName() + " where ACTIVITYID = " + activityId.toString());
		} catch (JdmoException e) {
			throw new BbmRemoveException(e);
		}
	}

	public void unlinkQueueActivities(Collection<ID> queueIds) throws BbmRemoveException {
		if (!queueIds.isEmpty()) {
			try {
				m_dmo.executeCommand("delete " + getFieldInfo().getTableName() + " where " + "QUEUEID IN"
					+ m_dmo.createInClause(queueIds));
			} catch (JdmoException e) {
				throw new BbmRemoveException(e);
			}
		}
	}

}
