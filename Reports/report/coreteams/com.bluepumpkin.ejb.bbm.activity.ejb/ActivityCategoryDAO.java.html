package com.bluepumpkin.ejb.bbm.activity.ejb;

import java.util.Collection;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategory;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategoryFieldInfo;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategoryFilter;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityCategoryFilterDAO;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.dao.DAOUtil;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;

public class ActivityCategoryDAO extends DAOBase<ActivityCategory> {

  // meta data information
  private static final FieldInfo FIELD_INFO = new ActivityCategoryFieldInfo();

  protected FieldInfo getFieldInfo() {
    return FIELD_INFO;
  }

  public ActivityCategoryDAO() {
    super();
  }

  public ActivityCategoryDAO(Jdmo dmo) {
    super(dmo);
  }

  protected ActivityCategory createValueObject() {
    return new ActivityCategory();
  }

  protected Collection findActivityCategories(ActivityCategoryFilter acFilter) throws BbmFinderException {
    return getObjects(getFilterStr(acFilter), getDefaultOrderByStatement());
  }

  protected Collection findOrganizationActivityCategories(Collection organizationIdCol,
                                                          ActivityCategoryFilter filter) throws BbmFinderException {

    Collection parentOrgs = DAOUtil.getParentOrganizationsCollection(organizationIdCol, m_dmo);
    parentOrgs.addAll(organizationIdCol);
    StringBuilder strWhere = new StringBuilder();
    try {
      strWhere.append("A.ORGANIZATIONID IN ").append(m_dmo.createInClause(parentOrgs));
      String strFilter = getFilterStr(filter);
      if (!StringUtil.isEmpty(strFilter)) {
        strWhere.append(" AND ").append(strFilter);
      }
      return getObjects(strWhere.toString(), getDefaultOrderByStatement());
    } catch (JdmoException e) {
      throw new BbmFinderException(e);
    }
  }


  protected Collection findActivityCategories(Collection ids) throws BbmFinderException {
    try {
      StringBuilder strWhere = new StringBuilder();
      strWhere.append("A.ID in ").append(m_dmo.createInClause(ids));
      return getObjects(new String(strWhere), getDefaultOrderByStatement());
    } catch (JdmoException e) {
      throw new BbmFinderException(e);
    }
  }

  private String getDefaultOrderByStatement() {
    StringBuilder buffer = new StringBuilder();
    buffer.append(" ORDER BY A.").append(FIELD_INFO.getFieldName(ActivityCategoryFieldInfo.ACTIVITYCATEGORY_NAME)).append(" ASC ");
    return buffer.toString();
  }

  private String getFilterStr(ActivityCategoryFilter filter) {
    String result = "";
    if (filter != null) {
      ActivityCategoryFilterDAO acfDao = new ActivityCategoryFilterDAO(filter);
      String strFilter = new String(acfDao.getFilterText());
      if (!strFilter.isEmpty()) {
        result = strFilter;
      }
    }
    return result;
  }
}
