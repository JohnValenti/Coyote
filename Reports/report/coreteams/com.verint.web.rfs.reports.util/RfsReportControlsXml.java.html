package com.verint.web.rfs.reports.util;

import org.w3c.dom.Node;

import com.bluepumpkin.web.core.reports.ReportControlsXml;

public class RfsReportControlsXml {

	private static RfsReportControlsXml instance = new RfsReportControlsXml();
	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();

	public RfsReportControlsXml() {
	}

	public static RfsReportControlsXml getInstance() {
		return instance;
	}

	public String getControlPickerName(Node controlNode) {
		return reportControls.getNodeValue(controlNode, "@picker-name");
	}

	public Node getControlActualStaffMix(Node controlNode) {
		return reportControls.getNode(controlNode, "parameter[@actual-staffmix='true']");
	}

	public Node getControlBudgetStaffMix(Node controlNode) {
		return reportControls.getNode(controlNode, "parameter[@budget-staffmix='true']");
	}

	public Node getControlPlannedStaffMix(Node controlNode) {
		return reportControls.getNode(controlNode, "parameter[@planned-staffmix='true']");
	}

	public Node getControlWorkgroup(Node controlNode) {
		return reportControls.getNode(controlNode, "parameter[@workgroup='true']");
	}

	public Node getControlStaffMixFilter(Node controlNode) {
		return reportControls.getNode(controlNode, "parameter[@staff-mix-filter='true']");
	}

	public Node getControlMonth(Node controlNode) {
		return reportControls.getNode(controlNode, "parameter[@is-month='true']");
