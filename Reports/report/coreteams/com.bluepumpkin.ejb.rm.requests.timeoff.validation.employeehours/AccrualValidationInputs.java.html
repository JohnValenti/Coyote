package com.bluepumpkin.ejb.rm.requests.timeoff.validation.employeehours;

/*
 * The data that is common for all choices in a request
 */
public class AccrualValidationInputs {

	private Activity activity;
	private ActivityCategory activityCategory;

	int[] employeeTimeOffYearStart;
	//employee's organization at the current time
	Organization employeeOrganization;
	Set<ID> activityIDs = new TreeSet<ID>();
	Employee employee;
	ActivityTimeOffBalance activityBalance;
	boolean isInitialized;


	HashMap<Integer, Map<String, Map<String, EmployeeTimeOffYearly>>> employeeActivityYearly;

	public boolean isAccrualEnabled() {
		if (activityCategory == null && activity == null) {
			return false;
		}

		return activityCategory.isTOAllotment() || activity.isTimeoffWithAllotment();
	}

	public ID getActivityID() {
		return !activityCategory.isTOAllotment() && activity.isTimeoffWithAllotment() ? activity.getID() : null;
	}

	public ID getActivityCategoryID() {
		return activityCategory.isTOAllotment() ? activityCategory.getID() : null;
	}

	public boolean isZeroMaxCarryOver() {
		return (activityBalance != null) && activityBalance.isZeroMaxCarryOver();
	}

	public boolean isValid() {
		return isInitialized;
	}

	public ActivityCategory getActivityCategory() {
		return activityCategory.isTOAllotment() ? activityCategory : null;
	}
	
	public Activity getActivity() {
		return !activityCategory.isTOAllotment() && activity.isTimeoffWithAllotment() ? activity : null;
	}

	public boolean load(ID empID, ID activityID) throws Exception { // NOSONAR

		isInitialized = false;


		if (empID == null || activityID == null) {
			return false;
		}
	
		//this call is cached
		ID orgID = ValidationUtil.getOrgIDForEmployeeDuringPeriod(empID, new Date(), null);
		//this call is cached
		employeeOrganization = ValidationUtil.getOrganizationByID(orgID);	
		//this hits the database
		employee = CacheUtilBBM.getEmployeeByID(empID, null, Employee.DETAIL_LEVEL_EMPLOYEE_BASIC);
		//this is cached
		employeeTimeOffYearStart = RequestUtil.getEmployeeTimeOffYearStart(employee, employeeOrganization);
		
		
		//this is cached
		activity = TOAccrualCalculator.getActivity(activityID);
		ID activityCategoryID = activity.getActivityCategoryId();

		//hits the database
		activityCategory = TOAccrualCalculator.getActivityCategory(activityCategoryID);

		if (activityCategory.isTOAllotment()) {
			activityIDs.addAll(getTOActivityIDs(activityCategoryID));
		} else {
			activityIDs.add(activityID);
		}

		//get activity TO Balance - hits the database
		activityBalance = TOAccrualCalculator.getActivityTimeOffBalance(empID, getActivityID(), getActivityCategoryID());

		isInitialized = true;
		return true;
	}



	//Assumes the activity category passed in has isTOAllotment set to true
	private static Collection<ID> getTOActivityIDs(ID activityCategoryID) throws BbmEJBCreateException, RemoteException, BbmFinderException {
		ActivityFilter filter = new ActivityFilter();
		//QC:137557 setting filter.setTimeoffWithAllotment(true) will filter out activities that does not have ISTIMEOFFWITHALLOTMENT set to true
		// but this is not correct since it is set at the activity category level in this specific case 
		filter.setActivityCategoryId(activityCategoryID, Filter.OP_EQUAL);
		return WfmManagerFactory.getActivityManager().findActivitiesIds(filter);
	}

}
