package com.bluepumpkin.ejb.rm.requests.timeoff.validation.employeehours;

import java.util.Collection;
import java.util.Collections;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.workresource.ejb.EmployeeTimeOffYearlyDAO;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeTimeOffYearly;
import com.bluepumpkin.ejb.rm.base.RmRuntimeException;
import com.bluepumpkin.ejb.rm.util.ThreadLocalCache;

/*
 * Per thread cache for EmployeeTimeOffYearly by Employee, Activity/ActivityCategory, fromYear, toYear 
 */
public class EmployeeTimeOffYearlyCache {

	private static class EmployeeActivityYearly {
		//EmployeeTimeOffYearly is loaded from the table called EMPLOYEETIMEOFFALLOTMENT
		//Gives by employee, year, and activity or activity category the accrual rates,remaining,carry over, scheduled, used and pending
		//we seem to only use the accrual rate for the validation. 
		//The scheduled is calculated by loading the scheduled time off and iterating through it. 

		//Employee -> Activity/ActivityCat -> Year ->EmployeeTimeOffYearly
		//The key for the returned map is a string representation of an integer year, why?.
		//In EmployeeTimeOffYearlyDAO which is in Foundation so leaving it alone. 
		final Map<Integer, Map<String, Map<String, EmployeeTimeOffYearly>>> map;

		EmployeeActivityYearly(Map<Integer, Map<String, Map<String, EmployeeTimeOffYearly>>> map) {
			this.map = map;
		}
	}

	private EmployeeTimeOffYearlyCache() {

	}

	//Year ->EmployeeTimeOffYearly for the given employee,activity/activity category, years. 
	public static Map<String, EmployeeTimeOffYearly> getEmployeeTimeOffYearly(ID employeeID, ID activityID, ID activityCategoryID,
			int fromYear, int toYear) {

		String key = getLoadKey(employeeID, activityID, activityCategoryID, fromYear, toYear);
		Map<String, EmployeeActivityYearly> map = ThreadLocalCache.getMap(EmployeeTimeOffYearlyCache.class.getName());

		EmployeeActivityYearly activityYearly = map.get(key);
		if (!map.containsKey(key)) {
			activityYearly = load(employeeID, activityID, activityCategoryID, fromYear, toYear);
			map.put(key, activityYearly);
		}
		return getEmployeeTimeOffYearlyInternal(activityYearly, employeeID, activityID, activityCategoryID);
	}

	//The key for the returned map is a string representation of an integer year. 
	private static Map<String, EmployeeTimeOffYearly> getEmployeeTimeOffYearlyInternal(EmployeeActivityYearly employeeActivityYearly,
			ID employeeID, ID activityID, ID activityCategoryID) {

		if (employeeActivityYearly == null || employeeActivityYearly.map == null) {
			return new HashMap<String, EmployeeTimeOffYearly>();
		}

		Map<String, Map<String, EmployeeTimeOffYearly>> activityYearlyMap = employeeActivityYearly.map.get(employeeID);

		if (activityYearlyMap == null) {
			return new HashMap<String, EmployeeTimeOffYearly>();
		}

		Map<String, EmployeeTimeOffYearly> yearlyMap = activityYearlyMap.get(Activity.getActivity_ActCategoryComb(activityID,
				activityCategoryID));

		if (yearlyMap == null) {
			return new HashMap<String, EmployeeTimeOffYearly>();
		}
		return yearlyMap;

	}

	private static String getLoadKey(ID employeeID, ID activityID, ID activityCategoryID,
			int fromYear, int toYear) {
		return String.format("[%s]-[%s]-[%s]-[%d]-[%d]", employeeID, activityID, activityCategoryID, fromYear, toYear);
	}

	//The key for the returned map is a string representation of an integer year. 
	@SuppressWarnings("unchecked")
	private static EmployeeActivityYearly load(ID employeeID, ID activityID, ID activityCategoryID,
			int fromYear, int toYear) {

		Jdmo dmo = null;

		try {

			dmo = new Jdmo();

			EmployeeTimeOffYearlyDAO dao = new EmployeeTimeOffYearlyDAO(dmo);
			Collection<ID> actIDs = activityID != null ? Collections.singletonList(activityID) : null;
			Collection<ID> actCatIDs = activityCategoryID != null ? Collections.singletonList(activityCategoryID) : null;

			//  map is an Employee ID to Map<String, Map<String, EmployeeTimeOffYearly>> map
			//	Map<String, Map<String, EmployeeTimeOffYearly>> 
			//  the key is an activity ID, activity Category ID generated by Activity.getActivity_ActCategoryComb
			//  Map<String, EmployeeTimeOffYearly>, the key is a string with the calendar year - Why is this a string and not an integer?
			////Employee -> Activity/ActivityCat -> Year -> EmployeeTimeOffYearly

			return new EmployeeActivityYearly(dao.getEmployeeTimeOffYearlyMap(
					Collections.singletonList(employeeID), actIDs, actCatIDs, fromYear, toYear));

		} catch (Exception e) {
			throw new RmRuntimeException("Could not load EmployeeTimeOffYearly from the database", e);
		} finally {
			if (dmo != null) {
				dmo.cleanUp();
			}
		}

	}

}
