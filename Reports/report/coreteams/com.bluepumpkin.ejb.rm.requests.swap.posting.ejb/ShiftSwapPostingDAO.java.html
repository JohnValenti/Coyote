    /**
     * Finds all the shift swap postings for my campaign, starting now.  Parameter 'end' is ignored for
     * London.
     *
     * <p> Algorithm for London
     * <li> (givenEmpID) -->
     * <li> 1 or more SPs for the given employee in the future (givenEmpSPID, givenEmpSPStartTime, giveEmpSPEndTime) using SP & SPEmployee table-->
     * <li> 1 or more (otherEmpSPID, otherEmployeeID, otherEmpStartTime, otherEmpEndTime) for (givenEmpSPID) using SPEmployee table -->
     * <li> 1 (otherEmployeeOrg) for (otherEmployeeID) using WorkResourceOrganization table -->
     * <li> 1 or more Posting by (otherEmployeeID) between (otherEmpStarttime, otherEmpEndTime) and (otherEmployeeOrg) in swappablr org list.
     *
     * <p> Simplications for London:
     * <li> Only one org employee association.
     * <li> SPs are always for a week (scheduling period is a week).  And SPs for an employee do not overlap.
     *
     * <p> SQL:
     * <code>
     *
       SELECT
          SSP.ID
       FROM
          SP,
          SPEmployee SPEmpG,
          AMKeyMap AMKeyMapG,
           SPEmpO,
          AMKeyMap AMKeyMapO,
          ShiftSwapPosting SSP,
          ShiftSwapItem SSI,
          WorkResourceOrganization WRO
       WHERE
          SP.toDate > @currDate AND
          SPEmpG.SPID = SP.ID AND
          SPEmpG.EmployeeID = AMKeyMapG.DEID AND
          AMKeyMapG.AMTbl = 'EmployeeAM' AND
          AMKeyMapG.AMID = @givenEmpID AND
          SPEmpG.SPID = SPEmpO.SPID AND
          SPEmpO.EmployeeID <> SPEmpG.EmployeeID AND
          SPEmpO.EmployeeID = AMKeyMapO.DEID AND
          AmKeyMapO.AMTbl = 'EmployeeAM' AND
          AMKeyMapO.AMID = SSI.EmployeeID AND
          SSI.StartTime between SP.fromDate AND SP.toDate AND
          AmKeyMapO.AMID = WRO.WorkResourceID AND
          WRO.OrganizationID in @swappableOrgs AND
          SSI.StartTime between WRO.startTime and WRO.endTime AND
          SSP.ShiftSwapItemID = SSI.ID

     * </code>
     * <p> TODO: revise javadoc (not accurate).
     *
     * <p>Algorithm for 5.0 suite:
     * <li> Obtain all campaign associations for the given employee starting now.  Note that campaign
     * associations are time bound.
     * <li> For each given employee's campaign association,  obtain campaign associations of other
     * employees whose timerange overlaps given employee's campaign association timerange.
     * <li> For each employee campaign association,  obtain the org association for the employee, if the org association
     * timerange overlaps the campaign association timerange and the org is in the swappable list.
     * <li> retrieve all postings made by employees in the org associations, during the period of overlap
     * of the following 3 time ranges: <br>
     *    1. the given employee's campaign association time range,<br>
     *    2. the campaign employee association time range<br>
     *    3. and the org employee association timerange<br>
     *
     * Steps
     * Get all campaigns user is associated with starting now.
     * Find all orgs user is associated with starting now.
     * For each campaign, during the campaign's effectivity, find the org the user is associated with.
     * for (each campaign, org, effectivity) tuple
     *    get list of swappale orgs.
     *    get posting list during effectivity period: Restrict postings to employees from same campaign and
     *       belonging to one of the swappable orgs.
     * end for.
     *
     * @param empID
     * @param swappableOrgIDs
     * @return
     * @throws Exception
     */
    public List findSSPostingIDsForMyCampaign(ID empID, Date start,
        Date end, Collection swappableOrgIDs, boolean fetchPartial) throws Exception
    {
        deleteSSPostingsExpired();

        if (swappableOrgIDs.isEmpty()) {
			return Collections.emptyList();
		}

        String strQuery = MessageFormat.format(m_postingsForMyCampaignTmpl,
            new Object[] {JdmoUtil.formatDBString(start), empID,
                m_dmo.createInClause(swappableOrgIDs)
            }
        );
        
        if (!fetchPartial) {
			strQuery += " AND SSI.ISPARTIAL = 0";
		}


        return DAOUtil.getIDsUsingSQLQuery(m_dmo, strQuery);
    }

    /**
     * Find the ShiftSwap Posting which corresponds to the specified shiftswap items.
     * The startDate, endDate, swapType and empID of the ssItem are used.
     *
     * <p> Note: This method is used to locate any shiftswap postings associated with the
     * shiftswapitems of a shiftswaprequest (during request approval).  The ID of the ssItem is
     * not used to locate the ssPosting since the ssItem is not shared by the ssReq and ssPosting.
     * See {@link com.bluepumpkin.ejb.bbm.dao.DAOUtil#deleteObjects(Jdmo, String, Collection) deleteObjects}
     * for more information.
     *
     * @param ssItem
     * @return
     */
    public Collection findSSPostingsForSSItems(Collection givenSSItems, long detailLevel) throws Exception {
        saveAndSetDetailLevel(detailLevel);

        try {
            // get an array of start and end dates formatted as strings.
            String[] startDateStrArr = new String[givenSSItems.size()];
            String[] endDateStrArr = new String[givenSSItems.size()];
            String[] swapTypeArr = new String[givenSSItems.size()];
            Collection empIDs = new ArrayList(givenSSItems.size());
            int i = 0;
            for (Iterator iter = givenSSItems.iterator(); iter.hasNext();) {
                ShiftSwapItem ssItem = (ShiftSwapItem) iter.next();

                startDateStrArr[i] = JdmoUtil.formatDBString(ssItem.getStartDate());
                endDateStrArr[i] = JdmoUtil.formatDBString(ssItem.getEndDate());
                swapTypeArr[i] = ssItem.getShiftType();
                empIDs.add(ssItem.getEmployeeID());

                i++;
            }

            // conver the array to a SQL in clause
            String startDateInClause = DAOUtil.createInClause(startDateStrArr);
            String endDateInClause = DAOUtil.createInClause(endDateStrArr);
            Object empIDInClause = getDMO().createInClause(empIDs);
            String swapTypeInClause = DAOUtil.createInClause(swapTypeArr);

            // using a subquery (slower) instead of a join since the DAO framework does not support joins.
            // See DAOBase.getObjects() for a better implementation suggestion.
            //
            // Note: loaded postings is a superset of the postings we need.  This is due to loading
            // postings corresponding to multiple ssItems in a single query and the 'IN' clause in the query.
            // For instance assume ssItems has 2 ssItems: ssItem1 belongs to empID1 and ssItem2 belongs to empID2.
            // We need to identify postings with ssItems which match
            //    (ssItem1.empID, ssItem1.startTime, ssItem1.endTime, ssItem1.swapType)
            //       or
            //    (ssItem2.empID, ssItem2.startTime, ssItem2.endTime, ssItem2.swapType)
            // But this query might return any ssPosting which matches a combination of the above tuples: for
            // instance (ssItem2.empID, ssItem1.startTime, ssItem1.endTime, ssItem1.swapType).
            String whereClause = MessageFormat.format(m_whereClauseTmplForFindSSPostingsForSSItems,
                new Object[] {startDateInClause, endDateInClause, empIDInClause, swapTypeInClause});

            // fetch postings
            Collection ssPostings = getObjects(whereClause);

            // The above query condition loads a superset of the postings needed.  So filter the postings
            // and only retain the ones which match the givenSSItems.  see above for further details.
            // Note: not a very efficient algorithm for searching but since ssPostings will be a small
            // collection, it is acceptable.
            for (Iterator iter = ssPostings.iterator(); iter.hasNext();) {
                ShiftSwapPosting ssPosting = (ShiftSwapPosting) iter.next();
                ShiftSwapItem ssItemForSSPosting = ssPosting.getShiftSwapItem();
                // sr# 28260 protect against Null Pointer
                // don't throw exception: if this fails, the Request won't get approved.
                // by logging an error and continuing, the Posting won't get cleaned up,
                // which is not as bad as failing the approval.
                if (ssItemForSSPosting == null) {
                  m_cat.error("Cannot find ShiftSwapItem for posting: \n" + ssPosting.toString() +
                              "\n. Posting will not be removed.");
                  continue;
                }

                // does the ssItem for the posting match one of the givenSSItems
                boolean found = false;
                for (Iterator iterator = givenSSItems.iterator(); iterator.hasNext();) {
                    ShiftSwapItem givenSSItem = (ShiftSwapItem) iterator.next();

                    // does the ssItem for the posting match the givenSSItem
                    if (ssItemForSSPosting.equalsSwapItemUsingESES(givenSSItem)) {
                        found = true;
                        break;
                    }
                }

                // if ssItem for ssPosting does not match any of the given ssItems
                if (!found) {
                    iter.remove();
                }
            }

            return ssPostings;
        } finally {
            resetDetailLevel();
        }
    }
    private String getWhereClauseToFindDayOffPosting(List givenSSItems) throws Exception {
    	ShiftSwapItem ssItem1 = (ShiftSwapItem) givenSSItems.get(0);
    	ShiftSwapItem ssItem2 = (ShiftSwapItem) givenSSItems.get(1);
    	
    	ID  ssItemId1= ssItem1.getID();
    	ID  ssItemId2= ssItem2.getID();
    	Collection ssItemIDs = new ArrayList(givenSSItems.size());
    	Collection empIDs = new ArrayList(givenSSItems.size());
    	ssItemIDs.add(ssItemId1);
    	ssItemIDs.add(ssItemId2);
    	ID empID1 = ssItem1.getEmployeeID();
    	ID empID2 = ssItem2.getEmployeeID();
    	empIDs.add(empID1);
    	empIDs.add(empID2);
    	  
    	Object ssItemIdInClause = getDMO().createInClause(ssItemIDs);
        Object empIDInClause = getDMO().createInClause(empIDs);
        StringBuilder whereClauseTmp= new StringBuilder();
        whereClauseTmp.append(COLNAME_SHIFTSWAPITEMID).append(" IN  (SELECT I2.ID from SHIFTSWAPITEM I INNER JOIN SHIFTSWAPITEM I2 ON ");
        whereClauseTmp.append(" I.STARTTIME <= I2.ENDTIME AND I.ENDTIME >= I2.STARTTIME AND I2.SHIFTTYPE = '").append(ShiftSwapItem.SWAPITEMTYPE_TIMEOFF);
        whereClauseTmp.append("' AND I.SHIFTTYPE='shift' AND I.EMPLOYEEID IN").append(empIDInClause).append(" AND  I2.EMPLOYEEID IN").append(empIDInClause);
        whereClauseTmp.append(" WHERE I.ID IN").append(ssItemIdInClause).append(")");
         
        return whereClauseTmp.toString();
    }

    /**
     * Note: refer to {@link com.bluepumpkin.ejb.bbm.dao.DAOUtil#deleteObjects(Jdmo, String, Collection) deleteObjects}
     * for information about ssItems shared between ssReqs and ssPostings
     *
     * Deletes a SSPosting object and its child ShiftSwapItem object
     * from the database. To do this it only explicitly deletes the child
     * (ShiftSwapItem) object, which automatically causes the parent to be
     * deleted, because the underlying methods use cascading delete. (The
     * BP_CASCADEDELETE stored procedure relies on the DELETECONTROL table
     * in the database.)
     *
     * @param id The ID of the SSPosting object to be deleted.
     * @throws Exception
     */
    public void deleteSSPostings(Collection ssPostingIDs) throws Exception {

        DAOBase idao = null;

        try {
        	// the update will lock the rows to prevent deadlock
        	// execute a stored procedure to delete data
			JdmoQuery  query = m_dmo.createQuery("bp_updateSSItem", Jdmo.STORPROC_QUERY_NORS);
			query.setParString(1, StringUtil.createDelimitedString(ssPostingIDs, ","));
			m_dmo.execute(query);

            //Find posting valueobject to retrieve shiftswapItemID
            Collection ssPostings = getObjectsByIDs(ssPostingIDs);

            if (!ssPostings.isEmpty()) {
                idao = createChildDAO(ShiftSwapPostingFieldInfo.ITEM_CHILD_TYPE);
                List ssItemIDs = RequestUtil.getListOfFieldValuesFromVOBases(ssPostings,
                    ShiftSwapPostingFieldInfo.SHIFTSWAPPOSTING_I_SHIFTSWAPITEMID);

                //Delete ShiftSwapItem, which will cascade delete the ShiftSwapPosting
                idao.deleteObjects(ssItemIDs);
            }
            // this should never happen, but we can try to delete the posting
            // by ID, which will leave a dangling ShiftSwapItem
            else {
                this.deleteObjects(ssPostingIDs);
            }
        } finally {
			if (idao != null) {
				idao.cleanUp();
			}
        }
    }
