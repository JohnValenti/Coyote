package com.verint.ejb.bbm.forecast.ejb;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**
 * Represents a segment of trace data that is part of a longer time interval.
 * 
 * @param <VOType> the value object type underlying this trace segment
 */
public abstract class AbstractTraceSegment<VOType extends ValueObjectBase> {
	protected final VOType m_valueObject;
	private final int m_idFieldIndex;
	private final int m_dateTimeFieldIndex;

	/**
	 * Creates a new trace segment.
	 * 
	 * @param valueObject the underlying value object
	 * @param idFieldIndex the field index in the value object of the ID for the trace.
	 * All segments with the same ID will be combined into a single trace, sequenced by date. 
	 * @param dateTimeFieldIndex the field index in the value object of the segment's date
	 */
	public AbstractTraceSegment(VOType valueObject, int idFieldIndex, int dateTimeFieldIndex) {
		m_valueObject = valueObject;
		m_idFieldIndex = idFieldIndex;
		m_dateTimeFieldIndex = dateTimeFieldIndex;
	}

	/**
	 * Returns the appropriate trace data for this trace segment.  A single value object
	 * may contain more than one type of time series data and this segment provides
	 * access to just one such type of data.
	 * 
	 * @return
	 */
	public abstract double[] getTraceData();

	/**
	 * Returns the segment's starting date.
	 * 
	 * @return
	 */
	public Date getDate() {
		return m_valueObject.getFieldValueDate(m_dateTimeFieldIndex);
	}

	/**
	 * Returns the trace's ID.  Multiple segments having the same trace ID will
	 * be combined into a single trace.
	 * 
	 * @return
	 */
	public ID getID() {
		return m_valueObject.getFieldValueID(m_idFieldIndex);
	}
}
