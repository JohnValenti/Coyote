package com.bluepumpkin.ejb.rm.base;

import java.util.TimeZone;

import com.bluepumpkin.common.base.BPUserException;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;

/**
 * This class implements 'BPUserException' which makes this exception to be logged 
 * as a 'warning' and not an 'error' 
 * by {@link com.bluepumpkin.ejb.core.base.SessionEJBBase#handleException(String, Exception, boolean) handleException}
 *
 * Since RMHardValidationExceptions are mostly validation errors (which can relatively often), logging it with
 * a loglevel of 'warning' prevents cluttering of the error.log file with this exception.  
 * 
 * @author  RRajendran
 */
//TODO: replace this with ValidationException class.
public class RmHardValidationException extends RmBaseException implements BPUserException {
	private String m_uiBundleName;
	private String m_uiID;

	public RmHardValidationException(String msgBundleName, String msgID, Object[] args) {
		super(msgBundleName, msgID, args);
		m_uiBundleName = msgBundleName;
		m_uiID = msgID;
	}

	public RmHardValidationException(String msgBundleName, String msgID, String logBundleName, String logID, Object[] args) {
		super(logBundleName, logID, args);
		m_uiBundleName = msgBundleName;
		m_uiID = msgID;
	}

	public String getLocalizedMessage(Localizer localizer, TimeZone tzView) {
		return RequestUtil.getLocalizedMessage(m_uiBundleName, m_uiID, getParamObject(),
				localizer, tzView);
	}
}
