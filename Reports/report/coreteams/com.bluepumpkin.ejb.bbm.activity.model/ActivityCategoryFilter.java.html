package com.bluepumpkin.ejb.bbm.activity.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.Filter.Filter;
import com.bluepumpkin.ejb.bbm.Filter.FilterCondition;

/**
 * Title: Blue Pumpkin Software Foundation Description: filter for
 * ActivityCategory object Copyright: Copyright (c) 2001 Company: Blue Pumpkin
 * Software, Inc
 * 
 * @author
 * @version 1.0
 */

public class ActivityCategoryFilter extends Filter {

	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;

	// parentID condition
	FilterCondition m_fcParentID = null;

	// Activity Category states to filter out
	public static final int FILTER_NONE = 0;
	public static final int FILTER_ACTIVE = 1;
	public static final int FILTER_DELETED = 2;

	// Constructor. By the default, all deleted activities are filtered out.
	public ActivityCategoryFilter() {
		setDeleted(false);
	}

	/**
	 * Constructor
	 * 
	 * @param state
	 *            - the state of the activities to filter out. Possible values:
	 *            ActivityCategoryFilter.FILTER_NONE - do not filter any
	 *            activity ActivityCategoryFilter.FILTER_ACTIVE - filter out the
	 *            active (not deleted) activities
	 *            ActivityCategoryFilter.FILTER_DELETED - filter out the deleted
	 *            activites
	 */
	public ActivityCategoryFilter(int state) {
		switch (state) {
		case FILTER_NONE:
			break;
		case FILTER_ACTIVE:
			setDeleted(true);
			break;
		case FILTER_DELETED:
		default:
			setDeleted(false);
			break;
		}
	}

	public void setActivityCategoryId(ID id, int iOperator) {
		setUserDefinedProperty("ID", "ID", iOperator, id);
	}

	public void setOrganizationId(ID id, int iOperator) {
		setUserDefinedProperty("ORGANIZATIONID", "ORGANIZATIONID", iOperator, id);
	}

	public void setActivityCategoryName(String strValue, int iOperator) {
		setUserDefinedProperty("NAME", "NAME", iOperator, strValue);
	}

	public void setDeleted(boolean booleanValue) {
		setUserDefinedProperty("ISDELETED", "ISDELETED", OP_EQUAL, booleanValue ? "1" : "0");
	}

	public void setVisibleToToTimeCollection(boolean booleanValue) {
		setUserDefinedProperty("ISVISIBLETOTIMECOLLECTION", "ISVISIBLETOTIMECOLLECTION", OP_EQUAL, booleanValue ? "1"
				: "0");
	}

	public void setTimeoffWithAllotment(boolean booleanValue) {
		setUserDefinedProperty("ISTIMEOFFWITHALLOTMENT", "ISTIMEOFFWITHALLOTMENT", OP_EQUAL, booleanValue ? "1" : "0");
	}
}
