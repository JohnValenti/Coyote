/**
 * Title:        SchedulePreferenceRH

public class SchedulePreferenceRH extends PeopleRequestHandler 
{

	/**
	 * Result values mapped in the requestmappings.xml file.
	 */
	private static final String SIMPLE_PAGE = "simple";
	private static final String ADVANCED_PAGE = "advanced";
	private static final String PRINTABLE_PAGE = "printable";
	
	private static final String USER_MODEL_CACHED_BY_SSLMODE_KEY = "USER_MODEL_CACHED_BY_SSLMODE_KEY";


    /** 
     * Return Page Model to perform generic action request on
     */
    public WorkpanePageModel getPageModel(RequestContext context) {
        WorkpanePageModel model = null;
        //@todo: 5.x select model based on the view type
        //String pageAction = context.getRequest().getParameter(PageModel.PAGE_ACTION_FN);
        //...
        model = new SimpleSchedulePreferencePM(context);
        model.extractParameters(context.getRequest());
        
        String selectedIDs = context.getRequest().getParameter("selectedID");
		model.setSelectedID((selectedIDs!=null)?selectedIDs:"");
		
        context.setPageModel(model);
        return model;
    }//getPageModel
    

    /**
	 * Process Request
	 */
	public void processCustomRequest(RequestContext context)  {
		HttpServletRequest request = context.getRequest();

		SimpleSchedulePreferencePM model = new SimpleSchedulePreferencePM(context);
		model.extractParameters(request);
		
		String selectedIDs = context.getRequest().getParameter("selectedID");
		model.setSelectedID((selectedIDs!=null)?selectedIDs:"");
		
		// Process requested action
		if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		}
        else if (isAction(context, PageAction.RESET)) {
            processRevert(context);   
        }
        else if (isAction(context, PageAction.FILTER_CHANGE)) {
			processFilterChange(context, model);
		}
		
		//proceedToIframeSSLModelHandling(context, "people_schedpredorgmode", USER_MODEL_CACHED_BY_SSLMODE_KEY, model);
        
		// Set the next page depending on the view type
		//TODO 5.x setNextPageFromViewType(context);
	}

	/**
	 * Process Save Action
	 */
	private void processSave(RequestContext context) {
        //TODO 5.x get view type and construct appropriate PM for simple/advanced
        SimpleSchedulePreferencePM model = new SimpleSchedulePreferencePM(context);
        model.extractParameters(context.getRequest());
        
        String selectedIDs = context.getRequest().getParameter("selectedID");
		model.setSelectedID((selectedIDs!=null)?selectedIDs:"");
		
        model.save();
        context.setPageModel(model);        
	}//processSave

	/**
	 * Process Filter Change
	 */
	private void processFilterChange(RequestContext context, SimpleSchedulePreferencePM model) {
		preProcessFilterChange(context, model);
	}

    private void processRevert(RequestContext context){
    }

	/**
