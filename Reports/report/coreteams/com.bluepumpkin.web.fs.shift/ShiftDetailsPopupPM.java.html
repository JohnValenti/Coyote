package com.bluepumpkin.web.fs.shift;

/**
 * Title:        ShiftDetailsPopupPM
 * Description:  Shift Details Popup Page Model
 * Copyright:    Copyright (c) 2003
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su, DSu
 * @version      1.0
 *
 * Created on Aug 7, 2003, 1:04:38 PM
 */
public class ShiftDetailsPopupPM extends PopupWindowPM{
	private static final String RH_SHIFT_DETAILS_PU = "shift_details_pu";

	private static final String SHIFT_ASSIGN_ID_FN = "shiftAssignID";
	protected static final String TZ_ID_FN = "timeZoneID";
	protected static final String POUP_ARGS = SHIFT_ASSIGN_ID_FN + "," + TZ_ID_FN;

	private final ResourceBundle m_bundle;
	private final MultiColListPC m_listPC;
	private ID shiftAssignmentID;
	private TimeZone m_tz;
	private ShiftAssignment shiftAssignment;
	private Map<ID, Activity> actMap = Collections.emptyMap();

	/**
	 * Constructor
	 */
	public ShiftDetailsPopupPM(RequestContext context) {
		super(context);
		m_bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);

		m_listPC = MultiColPCFactory.getMultiColListPC(context, i18n(m_bundle, FsWebBundleKey.SHIFT_DETAILS_GRID));
		addChildComponent(m_listPC);

		m_tz = context.getViewingTimeZone();
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, ShiftDetailsPopupPM.class);
	}

	/**
	 * Add Popup Args for ShiftDetails Popup to specified component with associated
	 * page Action
	 */
	public static void addPopupArgs(PageComponent pc, String pageAction) {
		pc.addPopupArgs(pageAction, RH_SHIFT_DETAILS_PU, null, POUP_ARGS,
				"600,350,ctr,ctr", true, 1);
	}

	/**
	 * Initialize Model with Data
	 */
	@Override
	protected void initializeModel() {
		if (m_isInitialized) return; else m_isInitialized = true;
		try {
			if (shiftAssignmentID!=null) {
				shiftAssignment = ShiftMH.getShiftAssignment(shiftAssignmentID);
				actMap = ActivityModelHandler.getAllActivitiesMap(m_context);
			}
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		} finally {
			initContentTitle();
			initList();
			initToolbar();
		}
	}

	public ID getShiftAssignID() {
		return shiftAssignmentID;
	}

	public void setShiftAssignID(ID id) {
		shiftAssignmentID = id;
	}

	public void setTimeZoneID(String tzID) {
		m_tz = TimeZone.getTimeZone(tzID);
	}

	protected void setShiftAssignment(ShiftAssignment sa) {
		shiftAssignment = sa;
	}

	protected void setActivityMap(Map<ID, Activity> map) {
		actMap = map;
	}

	/**
	 * Prepare content title for rendering
	 */
	protected void initContentTitle() {
		String pageTitle = i18n(m_bundle, FsWebBundleKey.SHIFT_DETAILS);
		m_contentTitlePC.setPageTitle(pageTitle);

		if (shiftAssignment!=null) {
			Date sDate = shiftAssignment.getStartTime();
			String itemName = m_localizer.formatDate(sDate, m_tz,
				RegionalFormatBundleKey.FULL_DATE_FORMAT) +
				" - " + TimeZoneHelper.getDisplayTimeZone(m_tz.getID(), m_localizer);
			m_contentTitlePC.setItemName(itemName);
		}
	}

	/**
	 * Initialize List
	 */
	protected void initList() {
		TableHeaderPC header = m_listPC.getHeader();
		header.addColumnHeaderData(null,i18n(m_bundle,FsWebBundleKey.SHIFT));
		header.addColumnHeaderData(null,i18n(m_bundle,FsWebBundleKey.ACTIVITIES));

		m_listPC.setEnabled(false);
		m_listPC.setIsBorderEnabled(true);
		if (shiftAssignment == null) {
			return;
		}

		CalendarRange shiftCR = DateTimeUtil.makeDayRange(shiftAssignment.getStartTime(), m_tz);
		DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData();

		// Shift Info
		Activity activity = actMap.get(shiftAssignment.getActivityID());
		String shift = isAccessibilityComplianceMode() ? ShiftUtil.makeFocusableActivityBox(activity, m_localizer)
				: ShiftUtil.makeActivityBox(activity)
						+ " " + ShiftUtil.makeEventPeriodString(shiftCR, shiftAssignment, true, m_localizer, m_tz);
		node.add( shift, CSSUtil.STYLE_NOWRAP );

		// Activities Info
		List eventList = new ArrayList(shiftAssignment.getChildren());
		String activities = ShiftUtil.makeActivitiesString(shiftCR, eventList, false,
				true, true,	actMap, m_localizer, m_tz);
		node.add( activities, CSSUtil.STYLE_NOWRAP );

		ArrayList listModel = new ArrayList(1);
		listModel.add(node);
		m_listPC.setListModel(listModel);
	}
	
	
	
	@Override
	public String getWorkPaneDescription() {
		ContentTitlePC contentTitle = getContentTitle();
		if (contentTitle != null) {
			return m_localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.WORK_PANE_DESCRIPTION,
					contentTitle.getPageTitle(), getContentItemText());
		}
		return "";
	}
	
	public String getContentItemText() {
		return m_localizer.formatDate(shiftAssignment.getStartTime(), m_tz, RegionalFormatBundleKey.FULL_DATE_FORMAT)
				+ " - " + m_localizer.formatTimeZone(m_tz.getID());
}

	/**
	 * Return the Html code to render the display of the component
	 */
	@Override
	public String getHtmlDisplay() {
		return m_listPC.getHtmlDisplay();
	}

	/**
	 * Return URL for HTML Form Action
	 */
	@Override
	public String getFormAction() {
		return RH_SHIFT_DETAILS_PU;
	}

	/**
	 * Return Required HTML for this component
	 */
	@Override
	public String getRequiredHTML() {
		return super.getRequiredHTML() +
				HtmlUtil.makeHiddenInput(SHIFT_ASSIGN_ID_FN, shiftAssignmentID) +
			HtmlUtil.makeHiddenInput(TZ_ID_FN, m_tz.getID());
	}
}
