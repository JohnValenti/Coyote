	public static final String POPUP_ARGS = "currentPeriodStartDate,currentPeriodEndDate,viewEndDate,schedulingPeriodID,employeeIDs,workPatternID,preferenceDDFN,parentTableName";
	public static final String ARG_EMPLOYEE_IDS = "employeeIDs";

	public WorkPatternEffectivityPopupPM(RequestContext context) {	
		super(context, "");
		
		preferenceDDPC = new IntegerDropDownSelection(context, "integerDDPC", 1, 100);
		addChildComponent(preferenceDDPC);
		
		isAssignedInViewPeriodCheckBoxPC = new CheckBoxPC(m_context, "isAssignedCheckBoxPC", true,
				getCheckboxCheckedJS(), getCheckboxUncheckedJS());
		addChildComponent(isAssignedInViewPeriodCheckBoxPC);
		
		addRequiredJavaScriptFile(FsJavaScriptFileID.EFFECTIVITY_UTIL);
		addRequiredJavaScriptFile(FsJavaScriptFileID.WORK_PATTERN_EFFECTIVITY_UTIL);
		
		super.extractParameters(context.getRequest());
		m_onSetJS = "WorkPatternEffectivityPopupUtil.updateNumDataFromDropDownToParentWindow(integerDDPC,'" +
		WorkPatternInlineEditListPC.getPreferenceDDPCName(parentTableName, workPatternID) + "','" +
		WorkPatternInlineEditListPC.getStartDatePCName(parentTableName, workPatternID) + "','" +
		WorkPatternInlineEditListPC.getEndDatePCName(parentTableName, workPatternID) + "','" +
		WorkPatternInlineEditListPC.getPopupRadioOptionFN(parentTableName, workPatternID) + "','" +
		WorkPatternInlineEditListPC.getEditedStartDateFN(parentTableName, workPatternID) + "','" +
		WorkPatternInlineEditListPC.getEditedEndDateFN(parentTableName, workPatternID) + "',isAssignedCheckBoxPC," +
		"'" + parentTableName + "','" + workPatternID + "');";
	}
	
	public static WorkPatternEffectivityPopupPM getInstance(RequestContext context) {

	protected MyDataContainer createDataContainerPC() {
		dataContainerPC = new MyDataContainer(m_context);
		return dataContainerPC;
	}
	
	@Override
	protected void initializeModel() {
		super.initializeModel();
		if (m_isInitialized) return; else m_isInitialized = true;
		try {
			if (schedulingPeriodID != null) {
				dataContainerPC.setOnlyViewPeriodEditable(true);
			}
			if (employeeIDs.size() > 0) {
				setIsNewPerson(false);
				workPatterns = WorkRulesModelHandler.getShiftPatterns(m_context, Collections.singleton(workPatternID));
				if (employeeIDs.size() == 1) {
					m_empID = employeeIDs.iterator().next();
					workPatternAssignments = WorkRulesModelHandler.getWorkPatternAssignmentsForEmployee(m_context, 
							m_empID, null, null);
					ProfileModelHandler modelHandler = new ProfileModelHandler();
					m_employee = modelHandler.getEmpWithHistoryData(m_empID, dataContainerPC.getStartDate().getTime(), m_context.isInWhatIfMode());
					//We only care about the selected work resource work pattern
					if (workPatternAssignments != null) {
						for (Iterator<WorkResourceWorkPattern> iter = workPatternAssignments.iterator(); iter.hasNext();) {
							ID tempWPID = iter.next().getWorkPatternSID();
							if (tempWPID.equals(workPatternID) == false) {
								iter.remove();
							}
						}
					}
					populateHistoryRows();
				}
			}
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		} finally {
			initContentTitle();
			initToolbar();
		}
	}
	
	private void initContentTitle() {
		if (workPatterns != null && workPatterns.size() > 0) {
			m_contentTitlePC.setPageTitle(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_WORK_PATTERN_EFFECTIVITY_CONTENT_TITLE,
					workPatterns.iterator().next().getName()));
		}
		if (m_employee != null) {		
			m_contentTitlePC.setItemName(m_localizer.formatName(m_employee));
		} else if (employeeIDs.size() > 1) {
			m_contentTitlePC.setItemName(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.N_PEOPLE, employeeIDs.size()));
		}
	}
	
	private void populateHistoryRows() {
		m_historyPC.setHeightInPixels(160);
		TableHeaderPC headerPC = m_historyPC.getHeader();
		headerPC.addColumnHeaderData("", m_localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.FROM));
		headerPC.addColumnHeaderData("", m_localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.TO));
		headerPC.addColumnHeaderData("", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_WORK_PATTERN_PREFERENCES));
		
		ArrayList<MultiColumnNodeData> listModel = new ArrayList<MultiColumnNodeData>();
		
		if (workPatternAssignments != null) {
			for (WorkResourceWorkPattern wrwp : workPatternAssignments) {
				DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData();
				node.setSelectable(false);
				node.add(formatDate(wrwp.getStartTime()));
				node.add(formatEndDate(wrwp.getEndTime()));
				node.add(m_localizer.formatNumber(wrwp.getPreference()));
				listModel.add(node);
			}
		}
		m_historyPC.setListModel(listModel);
	}

	private class MyDataContainer extends DataContainerPC {
		private static final long serialVersionUID = 1L;
		private int preference;
		
		public MyDataContainer(RequestContext context) {
			super(context);
		}

		protected void addCustomDataRow() {
			addRow(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_IS_WORK_PATTERN_ASSIGNED),
					isAssignedInViewPeriodCheckBoxPC.getHtmlDisplay());
			preferenceDDPC.setSelectedIDs(Collections.singleton(Integer.toString(preference)));
			addRow(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_WORK_PATTERN_PREFERENCES),
					preferenceDDPC.getHtmlDisplay());
		}
		
		public boolean extractParameters(HttpServletRequest request) {
			boolean success = super.extractParameters(request);
			
			String radioOptionStr = getParameter(request,
					WorkPatternInlineEditListPC.getPopupRadioOptionFN(parentTableName, workPatternID));
			if (radioOptionStr != null && radioOptionStr.length() > 0) {
				int radioOption = Integer.parseInt(radioOptionStr);
				switch (radioOption) {
					case 0: this.setUpdateMode(DataContainerPC.UPDATE_MODE_1); break;
					case 1: this.setUpdateMode(DataContainerPC.UPDATE_MODE_2); break;
					case 2: this.setUpdateMode(DataContainerPC.UPDATE_MODE_3); break;
					default: this.setUpdateMode(DataContainerPC.UPDATE_MODE_1); break;
				}
			}
			this.startDate = new LocalDate(currentPeriodStartDate, TimeZoneUtil.GMT_TIMEZONE);
			if (currentPeriodEndDate != null) {
				this.endDate = new LocalDate(currentPeriodEndDate,TimeZoneUtil.GMT_TIMEZONE);
			}
			//When in campaign mode,  we need to force the end date to match the end date
			// of the scheduling period.
			if (schedulingPeriodID != null) {
				this.endDate = new LocalDate(viewEndDate, TimeZoneUtil.GMT_TIMEZONE);
			}
			
			LocalDate editedEndDate = 
				extractDate(request, WorkPatternInlineEditListPC.getEditedEndDateFN(parentTableName, workPatternID));
			if (editedEndDate != null) {
				m_endPeriodPC.setDate(editedEndDate);
			} else {
				m_endPeriodPC.setDate(viewEndDate);			
			}
			//We need to explicitly set the timezone to GMT (default it pulls the time zone from the user context)
			//  because the end date getting passed in is in GMT
			m_endPeriodPC.setTimeZone(TimeZoneUtil.GMT_TIMEZONE);
			
			if (employeeIDs.size() > 1) {
				m_isMultiEdit = true;
				dataContainerPC.setIsMultiEdit(true);				
			}
			
			String preferenceStr = getParameter(request, preferenceDDFN);
			if (preferenceStr != null && preferenceStr.length() > 0 && !preferenceStr.equals(MULTI_VALUE_DATA)) {
				this.preference = Integer.parseInt(preferenceStr);
			}

			return success;
		}
	}
