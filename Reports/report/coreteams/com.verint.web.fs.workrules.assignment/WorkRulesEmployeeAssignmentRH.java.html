package com.verint.web.fs.workrules.assignment;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.core.pagecomponent.time.TimePeriodSelectorPC;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

public class WorkRulesEmployeeAssignmentRH extends RequestHandler {

	protected WorkRulesEmployeeAssignmentPM getPageModel(RequestContext context) {
		return WorkRulesEmployeeAssignmentPM.getInstance(context);
	}

	@Override
	public void processRequest(RequestContext context) throws Exception {
		if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		}
	}

	protected void processSave(RequestContext context) throws BPException {
		WorkRulesEmployeeAssignmentPM pageModel = getPageModel(context);
		try {
			pageModel.extractParameters(context.getRequest());
			if (pageModel.validate()) {
				// Hours
				pageModel.saveMinMaxHours();
				// Required Pay Period Hour
				pageModel.savePayPeriodHours();

				//Pooling Rules
				pageModel.savePoolingRules();

				//Work Patterns
				pageModel.saveWorkPatterns();

				if (pageModel.isByOrg) {
					//Assignment rules
					pageModel.saveAssignmentRules();

					//Rotations
					pageModel.saveRotations();

					//Time Banks
					pageModel.saveTimeBanks();
				}
				//Success
				pageModel.addPageMessage(Message.INSTRUCTION_TYPE,
						BbmWebBundleKey.UPDATED_EMPS_STATUS, BbmWebBundleKey.BUNDLE_NAME,
						new String[]{String.valueOf(pageModel.getSelectedIDSize())});
			} else {
				setNextPage(context, "formPage");
			}
		} catch (Exception ex) {
			handleUnableToUpdate(pageModel, ex);
		}
	}

	public static boolean didTimePeriodChange(RequestContext context) {
		return isAction(context, TimePeriodSelectorPC.ACTION_TIME_PERIOD_CHANGED);
	}
}
