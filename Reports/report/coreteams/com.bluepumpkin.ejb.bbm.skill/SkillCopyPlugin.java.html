package com.bluepumpkin.ejb.bbm.skill;

import java.rmi.RemoteException;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbBundleKey;
import com.bluepumpkin.ejb.bbm.skill.ejb.SkillManager;
import com.bluepumpkin.ejb.bbm.skill.model.Skill;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.ejb.bbm.copy.CopyableObjectPlugin;
import com.witness.ejb.bbm.copy.ejb.CopyDAO;
import com.witness.ejb.bbm.copy.model.CopyResult;
import com.witness.ejb.core.security.PrivilegeKeys;

/**
 * @Title: SkillCopyPlugin
 * @Description: copy functionality for Skills
 * @Copyright: Copyright (c) June 18, 2008
 * @Company: Verint Systems
 * @author Gonzalo Quintanar
 */
public class SkillCopyPlugin extends CopyableObjectPlugin {

    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;

	/**
     * Constructor
     * 
     */
    public SkillCopyPlugin() {
        // TODO Auto-generated constructor stub
    }

    /*
     * (non-Javadoc)
     * 
     * @see com.witness.ejb.bbm.copy.CopyableObjectPlugin#copy(com.bluepumpkin.common.datatypes.ID,
     *      com.bluepumpkin.common.datatypes.ID, boolean,
     *      com.witness.ejb.bbm.copy.ejb.CopyDAO, java.lang.String)
     */
    public CopyResult copy(ID idSkill, ID idOrg, boolean bForce,
            CopyDAO pDAO, String sCopyPrefix) throws BbmFinderException,
            BbmCreateException {

        try {
            SkillManager pSkillManager = WfmManagerFactory.getSkillManager();
            
            // get the skill to be copied
            Skill pSkill = pSkillManager.getSkillByStringID(idSkill);

            // rename and copy the activitycat
            createCopySkill(pSkillManager, pSkill, idOrg, pDAO, sCopyPrefix);

            // return CopyResult COPIED
            return new CopyResult();
        } catch (BbmEJBCreateException e) {
            throw new BbmCreateException(e, "could not get Skills");
        } catch (RemoteException e) {
            throw new BbmCreateException(e, "could not get Skills");
        }
    }

    /*
     * (non-Javadoc)
     * 
     * @see com.witness.ejb.bbm.copy.CopyableObjectPlugin#isMultiTargetCopyAllowed()
     */
    public boolean isMultiTargetCopyAllowed() {
        return false;
    }

    /*
     * (non-Javadoc)
     * 
     * @see com.witness.ejb.bbm.GenericObjectPlugin#convertType(com.bluepumpkin.common.localization.Localizer,
     *      boolean)
     */
    public String convertType(Localizer localizer, boolean isPlural) {
        String rbKey = isPlural ? BbmEjbBundleKey.OBJECT_SKILL_PLURAL : BbmEjbBundleKey.OBJECT_SKILL; 
        return localizer.i18n(BbmEjbBundleKey.BUNDLE_NAME, rbKey);
    }

    /*
     * (non-Javadoc)
     * 
     * @see com.witness.ejb.bbm.GenericObjectPlugin#getModificationPrivilegeID()
     */
    public ID getModificationPrivilegeID() {
        return PrivilegeKeys.CORE_CONFIGURESKILLS_ID;
    }

    private ID createCopySkill(SkillManager pSkillManager,
            Skill pSkill, ID idOrg, CopyDAO pDAO, String sCopyPrefix)
            throws RemoteException, BbmCreateException, BbmFinderException {

        String sNewName = sCopyPrefix + pSkill.getName();
        int nName = 0;

        while (true) {
            boolean bNameIsUnique = pDAO.isNameUnique(sNewName, "SKILL");

            if (bNameIsUnique)
                break;
            nName++;
            sNewName = sCopyPrefix + pSkill.getName() + " " + nName;
        }

        pSkill.setName(sNewName);
        pSkill.setOrganizationID(idOrg);

        return pSkillManager.createSkill(pSkill);
    }
}
