	public static int getNumberOfUnits(int unit, int value) {
		if (unit == TimeOffBid.TIMEOFF_BID_DAYS) {
			return value;
		} else {
			return value * 7;
		}
	}

	public static int getMinimumDaysPerChoice(TimeOffBid bid) {
		return getNumberOfUnits(bid.getMinUnitPerChoice(), bid.getMinValuePerChoice());
	}

	public static int getMaximumsPerChoice(TimeOffBid bid) {
		return getNumberOfUnits(bid.getMaxUnitPerChoice(), bid.getMaxValuePerChoice());

	public static String getPeriodText(Localizer localizer, int value, int periodUnit) {
		String bundleName = RmEjbBundleKey.BUNDLE_NAME;
		if (periodUnit == TimeOffBid.TIMEOFF_BID_WEEKS) {
			return value == 1 ? localizer.i18n(bundleName, RmEjbBundleKey.PERIOD_WEEK) : localizer.i18n(bundleName,
					RmEjbBundleKey.PERIOD_WEEKS, String.valueOf(value));
		} else {
			return value == 1 ? localizer.i18n(bundleName, RmEjbBundleKey.PERIOD_DAY) : localizer.i18n(bundleName,
					RmEjbBundleKey.PERIOD_DAYS, String.valueOf(value));
		}
	}

	/*Correct the choice duration due to DST*/
	public static long getRealChoiceDuration(Date choiceStart, Date choiceEnd, TimeZone tz) {
		long cStart = choiceStart.getTime();
		long cEnd = choiceEnd.getTime();
		long realChoiceDuration = cEnd - cStart;
		DSTTransitionMode dst = DateUtil.getDSTTransitionMode(choiceStart, choiceEnd, tz);
		long hourInMiliSeconds = TimeUnit.HOURS.toMillis(1);
		if (dst == DSTTransitionMode.SPRINGFORWARD) {
			realChoiceDuration += hourInMiliSeconds;
		} else if (dst == DSTTransitionMode.FALLBACK) {
			realChoiceDuration -= hourInMiliSeconds;
		}
		return realChoiceDuration;
	}
	public static long getDurationChoices(Collection<TOChoice> choices, TimeZone tz) {
		long result=0;
		for(TOChoice choice:choices){
			result+= getRealChoiceDuration(choice.getStartDate(),choice.getEndDate(),tz);
		}
		return result;
	}

	public static boolean isIncrementsOfWeek(long choiceDuration) {
		long weekInMiliSeconds = TimeUnit.DAYS.toMillis(7);
		float week = (float) choiceDuration / weekInMiliSeconds;
		if (week == (int) week) {
			return true;
		}
		return false;
	}

	public static Collection<ID> getActivityIDsByBidId(ID bidId) throws BbmFinderException {

		if (!ThreadLocalCache.isEnabled()) {
			return getActivityIDsByBidIdFromDb(bidId);
		}

		Map<ID, Collection<ID>> cache = ThreadLocalCache.getMap("TimeOffBidutil.getActivityIDsByBidId");
		if (cache.containsKey(bidId)) {
			return new ArrayList<ID>(cache.get(bidId));
		}

		Collection<ID> result = getActivityIDsByBidIdFromDb(bidId);
		cache.put(bidId, result);
		return new ArrayList<ID>(result);
	}

	@SuppressWarnings("unchecked")
	public static TimeZone getTimeZoneOfEmployee(ID employeeId) throws BbmFinderException {
		TimeZone tz = TimeZone.getDefault();
		try {
			Collection<WorkResourceAssignment> wras = CacheUtilBBM.getWorkResourceAssnWithTZForWRID(employeeId);
			if (!wras.isEmpty()) {
				// now look for overlap with now
				for (Iterator<WorkResourceAssignment> it = wras.iterator(); it.hasNext();) {
					WorkResourceAssignment wra = it.next();
					if (TimePeriodUtil.overlap(wra.getStartTime(), wra.getEndTime(), new Date(), null)) {
						tz = wra.getStartTimeZone();
						break;
					}
				}
			}
		} catch (Exception e) {
			throw new BbmFinderException(e);
		}
		return tz;
	}
	/**Find the choice range from choice list which is overlapped with a given range*/
	public static TimeRange getChoiceRangeOverlapWithGivenRange(TimeRange givenRange, Collection<TOChoice> choices,boolean isForMultiMode) {
		TimeRange resultRange = null;
		boolean isOverlapped = false;
		for (Iterator<TOChoice> it = choices.iterator(); it.hasNext();) {
			TOChoice considerChoice = it.next();
			TimeRange considerChoiceRange = new TimeRange(considerChoice.getStartDate(), considerChoice.getEndDate());
			boolean isCover = considerChoiceRange.isInside(givenRange);
			boolean isInside = givenRange.isInside(considerChoiceRange);
			//We dont count givenRange.isInside(approveChoiceRange) 
			//when using this utility for soft validation rule TimeOffBidRequestNotOverlapValidationRule
			//since in this case, 
			//The given range is completely contained within a block of scheduled time off 
			boolean toChoiceOverlapStart = givenRange.getStartDate().before(considerChoice.getStartDate()) && considerChoice.getStartDate().before(givenRange.getEndDate());
			boolean toChoiceOverlapEnd = givenRange.getStartDate().before(considerChoice.getEndDate()) && considerChoice.getEndDate().before(givenRange.getEndDate());
			isOverlapped = isCover || toChoiceOverlapStart || toChoiceOverlapEnd||(isForMultiMode&&isInside);
			if (isOverlapped) {
				resultRange = considerChoiceRange;
				break;
			}
		}
		return resultRange;
	}
