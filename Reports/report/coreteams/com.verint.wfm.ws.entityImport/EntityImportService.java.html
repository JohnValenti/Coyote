/**
 * @author Eliot Clingman, Copyright (c) 2011 by Verint Systems, Inc. All Rights Reserved. This web
 *         service supports methods to synchronize entities between WFO and an external application
 */

@WebService(serviceName = "EntityImportService", targetNamespace = "http://verint.com/core/ws/EntityImportService")
public class EntityImportService {

	private static final long serialVersionUID = 1L;

	// For any entity type, this number of unknown system exception in a row will terminate server,
	// and a ServiceApplicationException will be thrown with stack trace of last exception instance
	public static int CONSECUTIVE_SERVER_EXCEPTION_MAX_COUNT = 5;
	public static int NAME_MAX_LENGTH = 350;
	public static int EXTERNALID_MAX_LENGTH = 350;

	private OrganizationServiceDelegate m_organizationServiceDelegate;
	private ActivityServiceDelegate m_activityServiceDelegate;
	private SkillServiceDelegate m_skillServiceDelegate;
	private CampaignServiceDelegate m_campaignServiceDelegate;
	private QueueServiceDelegate m_queueServiceDelegate;
	private SchedulingPeriodServiceDelegate m_schedulingPeriodServiceDelegate;
	private ResourceConstraintServiceDelegate m_resourceConstraintServiceDelegate;
	private ForecastServiceDelegate m_forecastServiceDelegate;
	private AutoScheduleDelegate m_autoScheduleDelegate;

	/**
	 * Export organizations from an external system into WFO. Whether an entity is new or update is
	 * based on name as unique identifier.
	 * 
	 * The organizationDTO class here is a simplified version of WFO's organization class.
	 * The organizationNotPersistedInfo is not used by any existing client app
	 *
	 * @throws ServiceApplicationException
	 *             when there is a application fault
	 */
	@WebMethod
	public List<OrganizationNotPersistedInfo> createOrUpdateOrganizations(
			@WebParam(name = "organizationDTOs") Collection<OrganizationDTO> organizationDTOs,
			@WebParam(name = "systemTypeName") String systemTypeName)
			throws ServiceApplicationException {
		return m_organizationServiceDelegate.createOrUpdate(organizationDTOs, systemTypeName);
	}

	/**
	 * Export activities from an external system into WFO. Whether an entity is new or update is
	 * based on name as unique identifier.
	 * 
	 * The activityDTO class here is a simplified version of WFO's organization class
	 * 
	 * @throws ServiceApplicationException
	 *             when there is a application fault
	 */
	@WebMethod
	public List<ActivityNotPersistedInfo> createOrUpdateActivities(
			@WebParam(name = "activityDTOs") Collection<ActivityDTO> activityDTOs,
			@WebParam(name = "systemTypeName") String systemTypeName) throws ServiceApplicationException {
		return m_activityServiceDelegate.createOrUpdate(activityDTOs, systemTypeName);
	}

	/**
	 * Export skills from an external system into WFO. New entities are determined by comparing the
	 * externalSourceId with what is persisted to WFO, and then created on WFO.
	 * 
	 * @throws ServiceApplicationException
	 *             when there is a application fault
	 */
	@WebMethod
	public List<SkillNotPersistedInfo> createOrUpdateSkills(Collection<SkillDTO> skillDTOs,
			String systemTypeName) throws ServiceApplicationException {
		return m_skillServiceDelegate.createOrUpdate(skillDTOs, systemTypeName);
	}

	/**
	 * Export campaigns from an external system into WFO. MNew entities are determined by comparing
	 * the externalSourceId with what's already persisted to WFO. Old entities are updated instead
	 * of recreated

	 * 
	 * @throws ServiceApplicationException
	 *             when there is a application fault

	/**
	 * Export project forecasts from an external system into WFO. New entities are determined by comparing
	 * the campaign external source id, scheduling period start date, and queue external source id
	 * with what's already persisted to WFO. Old entities are updated instead of recreated
	 * 
	 * The ProjectForecastDTO class here is a simplified version of WFOs TODO
	 * 
	 * @throws ServiceApplicationException
	 *             when there is a application fault
	 */
	@WebMethod
	public List<ForecastNotPersistedInfo> createOrUpdateProjectForecasts(
			@WebParam(name = "projectForecastDTOs") Collection<ProjectForecastDTO> projectForecastDTOs,
			@WebParam(name = "systemTypeName") String systemTypeName) throws ServiceApplicationException {
		return m_forecastServiceDelegate.createOrUpdateProjectForecasts(projectForecastDTOs, systemTypeName);
	}

	/**
	 * Export time series style forecasts (e.g. AHT and volume forecasts spaced at regular intervals over the day) from an external system into WFO. New entities are determined by comparing
	 * the campaign external source id, scheduling period start date, and queue external source id
	 * with what's already persisted to WFO. Old entities are updated instead of recreated.
	 * 
	 * The TimeSeriesForecastDTO class here is a simplified version of WFOs ForecastInstance and
	 * ForecastTraceCube classes
	 * 
	 * @throws ServiceApplicationException
	 *             when there is a application fault
	 */
	@WebMethod
	public List<ForecastNotPersistedInfo> createOrUpdateTimeSeriesForecasts(
			@WebParam(name = "timeSeriesForecastDTOs") Collection<TimeSeriesForecastDTO> timeSeriesForecastDTOs,
			@WebParam(name = "systemTypeName") String systemTypeName) throws ServiceApplicationException {
		return m_forecastServiceDelegate.createOrUpdateTimeSeriesForecasts(timeSeriesForecastDTOs, systemTypeName);
	}

	/**
	 * Creates the Schedules for the scheduling periods if the schedules are not already created.
	 * The scheduling Periods must be pushed in the WFO before calling this method.
	 * 
	 * @return Returns a list of SchedulingNotTriggeredInfo if there is any problem triggering the
	 *         scheduling
	 * @throws ServiceApplicationException
	 *             when there is a application fault
	 */
	@WebMethod
	public List<SchedulingNotTriggeredInfo> startScheduling(
			@WebParam(name = "schedulingPeriodIDs") Collection<SchedulingPeriodExternalNaturalKey> schedulingPeriodIDs,
			@WebParam(name = "systemTypeName") String systemTypeName) throws ServiceApplicationException {
		return m_autoScheduleDelegate.autoSchedule(schedulingPeriodIDs,false, systemTypeName);
	}

	/**
	 * Creates the Schedules for the scheduling periods. The scheduling Periods
	 * must be pushed in the WFO before calling this method.
	 * 
	 * @param schedulingPeriodIDs
	 * @param overrideSchedules boolean flag to decide if existing schedules can be overridden
	 * @param systemTypeName
	 * @return a list of SchedulingNotTriggeredInfo if there is any problem
	 *         triggering the scheduling
	 * @throws ServiceApplicationException
	 *             when there is a application fault
	 */
	@WebMethod
	public List<SchedulingNotTriggeredInfo> startSchedulingOverride(
			@WebParam(name = "schedulingPeriodIDs") Collection<SchedulingPeriodExternalNaturalKey> schedulingPeriodIDs,
			@WebParam(name = "overrideSchedules") boolean overrideSchedules,
			@WebParam(name = "systemTypeName") String systemTypeName)
					throws ServiceApplicationException {
		return m_autoScheduleDelegate.autoSchedule(schedulingPeriodIDs, overrideSchedules, systemTypeName);
	}

	@WebMethod
	public SchedulingPeriodExternalNaturalKeyMapJAXBType doesScheduleExists(
			@WebParam(name = "schedulingPeriodIDs") Collection<SchedulingPeriodExternalNaturalKey> schedulingPeriodIDs,
			@WebParam(name = "systemTypeName") String systemTypeName)
			throws ServiceApplicationException {
		return m_autoScheduleDelegate.doesScheduleExist(schedulingPeriodIDs, systemTypeName);
	}

	/**
	 * Returns the current status of the scheduling of the scheduling Periods
	 * 
	 * @return The status of the scheduling for Scheduling Periods
	 * @throws ServiceApplicationException
	 *             when there is a application fault
	 */
	@WebMethod
	public Collection<SchedulingStatus> getSchedulingStatus(
			@WebParam(name = "schedulingPeriodIDs") Collection<SchedulingPeriodExternalNaturalKey> schedulingPeriodIDs,
			@WebParam(name = "systemTypeName") String systemTypeName) throws ServiceApplicationException {
		return m_autoScheduleDelegate.getSchedulingStatus(schedulingPeriodIDs, systemTypeName);
	}
	
	
	/**
	 * Stops the scheduling of the scheduling of the scheduling periods if the scheduling is not
	 * already completed. This method does not undo the scheduling of already completed scheduling
	 * period.
	 * 
	 * @param schedulingKeys
	 * @throws ServiceApplicationException
	 *             when there is a application fault
	 */
	public void stopScheduling(
			@WebParam(name = "schedulingPeriodIDs") Collection<SchedulingPeriodExternalNaturalKey> schedulingPeriodIDs,
			@WebParam(name = "systemTypeName") String systemTypeName) throws ServiceApplicationException {
		m_autoScheduleDelegate.stopScheduling(schedulingPeriodIDs, systemTypeName);
	}
