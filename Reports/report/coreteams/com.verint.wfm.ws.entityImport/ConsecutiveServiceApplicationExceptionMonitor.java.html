package com.verint.wfm.ws.entityImport;

import com.verint.wfm.ws.ServiceApplicationException;

/**
 * This is used to prevent continuing after too many consecutive non-user exception After the limit is reached, the
 * exception is thrown rather than being treated as an info exception
 * 
 */
public class ConsecutiveServiceApplicationExceptionMonitor {

	int counter;

	public ConsecutiveServiceApplicationExceptionMonitor() {
		counter = 0;
	}

	public void reset() {
		counter = 0;
	}

	public void increment(Exception e) throws ServiceApplicationException {
		counter++;
		if (counter > EntityImportService.CONSECUTIVE_SERVER_EXCEPTION_MAX_COUNT) {
			throw (e instanceof ServiceApplicationException) ? (ServiceApplicationException) e : new ServiceApplicationException(e);
		}
	}
}
