package com.bluepumpkin.ejb.rm.requests.swap.shiftitem.model;


/**
 * Title:        ShiftSwapRequest

 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc.
 * @author       Zhibo Wang
 * @version 1.0
 */
import java.util.Calendar;
import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;


public class ShiftSwapItem extends ValueObjectBase {
    // Types of shifts
    public static final String SWAPITEMTYPE_SHIFT = "shift";
    public static final String SWAPITEMTYPE_TIMEOFF = "time-off";

    private static final FieldInfo m_fieldInfo = new ShiftSwapItemFieldInfo();

    public ShiftSwapItem() {
        super();
    }

    public FieldInfo getFieldInfo() {
        return m_fieldInfo;
    }

    //getters
    public ID getEmployeeID() {
        return getFieldValueID(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_I_EMPLOYEEID);
    }

    public Date getStartDate() {
    	Date startDate = this.getFieldValueDate(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_D_STARTTIME);
    	Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("GMT"));

    	cal.setTime(startDate);
    	int modVal = cal.get(Calendar.MINUTE)%15;
    	if ( modVal != 0){
    		cal.add(Calendar.MINUTE, -modVal);
    	}
        return cal.getTime();
    }

    public Date getDisplayStartDate() {
        return this.getFieldValueDate(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_D_STARTTIME);
    }

    public Date getEndDate() {
        return this.getFieldValueDate(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_D_ENDTIME);
    }
    
    /**
     * Duration of ShiftSwap item in minutes.
     * 
     * @return
     */
    public long getDuration() {
        return (getEndDate().getTime() - getStartDate().getTime())/(60*1000);
    }

    public Date getExpirationDate() {
        return getFieldValueDate(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_D_EXPIRATIONDATE);
    }
    
    public boolean getIsPartial() {
        return getFieldValueBoolean(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_B_ISPARTIAL);
    }

    public boolean getAllowPartial() {
        return getFieldValueBoolean(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_B_ISPARTIALSWAPALLOWED);
    }
    
    /**
     * Method name is misleading.  Fetches the swap type which is one of the following: 
     * {@link #SWAPITEMTYPE_SHIFT SWAPITEMTYPE_SHIFT} or {@link #SWAPITEMTYPE_TIMEOFF SWAPITEMTYPE_TIMEOFF}
     *  
     * @return
     */
    public String getShiftType() {
        return getFormattedFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_S_SHIFTTYPE);
    }

    public boolean getNegotiation() {
        return this.getFieldValueBoolean(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_B_ISNEGOTIATIONREQUIRED);
    }
    
    public int getHashCode() {
        return this.getFieldValueInt(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_N_HASHCODE);
    } 
    
    public ID getShiftID() {
        return this.getFieldValueID(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_I_SHIFTID);
    }

    // setters
    public void setEmployeeID(ID id) {
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_I_EMPLOYEEID, id);
    }

    public void setStartDate(Date date) {
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_D_STARTTIME, date);
    }

    public void setEndDate(Date date) {
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_D_ENDTIME, date);
    }

    public void setExpirationDate(Date date) {
        // todo: check with Pavel. presently assuming expirationdate == enddate. if
        //   expirationdate set using setExpirationDate(), do we throw an exception if
        //   expiration earlier than end date.
        // todo: rational behind having the expiration date in SSItem.
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_D_EXPIRATIONDATE, date);
    }
    
    public void setIsPartial(boolean boo) {
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_B_ISPARTIAL, boo);
    }
    
    public void setAllowPartial(boolean boo) {
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_B_ISPARTIALSWAPALLOWED, boo);
    }

    /**
     * Method name is misleading.  Sets the swap type which is one of the following: 
     * {@link #SWAPITEMTYPE_SHIFT SWAPITEMTYPE_SHIFT} or {@link #SWAPITEMTYPE_TIMEOFF SWAPITEMTYPE_TIMEOFF}
     *  
     * @return
     */
    public void setShiftType(String type) {
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_S_SHIFTTYPE, type);
    }

    public void setNegotiation(boolean boo) {
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_B_ISNEGOTIATIONREQUIRED, boo);
    }
    
    public void setHashCode(int hashCode) {
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_N_HASHCODE, hashCode);
    }    

    public void setShiftID(ID shiftID) {
        setFieldValue(ShiftSwapItemFieldInfo.SHIFTSWAPITEM_I_SHIFTID, shiftID);
    }    

    public boolean isSwapTypeShift() {
    	return getShiftType().equals(ShiftSwapItem.SWAPITEMTYPE_SHIFT);
    }
    
	/**
	 * returns true if the shift's start and end time matches the item's start and end time. False otherwise.
	 * 
	 * @param shift
	 * @return
	 */
	public boolean equalsShift(ShiftAssignment shift) {
		boolean equals = false; 
		// if shift assignment's start and end time matches SSItem's start and end time
		if ( shift.getStartTime().equals(getStartDate()) && shift.getEndTime().equals(getEndDate())) {
			equals = true;
		}
		
		return equals;
	}
    
    /**
     * compares for equality using empID, startTime, endTime and swapItemType
     */
    public boolean equalsSwapItemUsingESES(ShiftSwapItem ssi) {
        return getEmployeeID().equals(ssi.getEmployeeID()) &&
            getStartDate().equals(ssi.getStartDate()) &&
            getEndDate().equals(ssi.getEndDate()) &&
            getShiftType().equals(ssi.getShiftType());
    }
}
