public class TOWaitlistExipryPollerJob implements StatefulJob {
	private static Category m_cat = Log.initCategory(TOWaitlistExipryPollerJob.class.getName());
	/**
	 * lock to ensure that this job is not reexecuted while an earlier instance is still executing.
	 */
	static Lock m_jobExecLock = new Lock();
	/*
	 * @see org.quartz.Job#execute(org.quartz.JobExecutionContext)
	 */
	static long lastRunTime = 0;
	static int lastKnownRunHour = 0;

	public void execute(JobExecutionContext execCtxt) throws JobExecutionException {
		String methodName = "execute";
		try {
			// try to obtain the lock.
			if (!m_jobExecLock.lock()) {
				m_cat.debug("Quartz job not executed as an earlier job is still executing");
				return;
			}
			if (TORequestUtil.isTOWaitlistFeatureEnabled()) {
				// continue to perform other polled tasks if an exception is thrown.
				TOWaitlistManager toWaitlistManager = RequestUtil.getTOWaitlistManager();
				try {
					toWaitlistManager.updateStatusForExpiredTOWaitlistRequests();
				} catch (Exception e) {
					m_cat.error("**** Exception caught and ignored ****: ", e);
				}
				scanAllTOWaitlistedRequests(toWaitlistManager);
			}
			RequestUtil.getTOHoursPerDayManager(null,null).syncTOHoursPerDay();
			// release lock
			m_jobExecLock.releaseLock();
		} catch (Exception e) {
			// release lock if it is being held.
			if (m_jobExecLock.isLocked()) {
				m_jobExecLock.releaseLock();
			}
			m_cat.error(methodName, e);
			throw new JobExecutionException(e, false);
		}
	}

	public static int HOUR_IN_MSECS = 3600000;

	public boolean isElgibileToRunTOWaitlistScanNow(long now) {
		int runInterval = 24;
		int runHour = 20;
		try {
			String runHourStr = BbmManagerFactory.getDBConfigManager().getValue(ConfigKey.REQUEST_MANAGEMENT + "FULL_TO_WL_SCAN_RUN_TIME");
			if (runHourStr != null) runHour = Integer.parseInt(runHourStr);
		} catch (Exception e) {
		}
		if (runHour != lastKnownRunHour) {
			lastRunTime = 0;
			lastKnownRunHour = runHour;
		}
		int currentHour = Calendar.getInstance().get(Calendar.HOUR_OF_DAY);
		if (RequestUtil.isRMDebugEnabled()) m_cat.info("runHour=" + runHour + ": currentHour=" + currentHour + " : LAST Run time in Hrs= " + ((now - lastRunTime) / HOUR_IN_MSECS));
		if (runHour == currentHour) {
			return (lastRunTime == 0 || (now - lastRunTime) > (runInterval * HOUR_IN_MSECS));
		}
		return false;
	}

	public void scanAllTOWaitlistedRequests(TOWaitlistManager toWaitlistManager) throws Exception {
		long now = new Date().getTime();
		if (isElgibileToRunTOWaitlistScanNow(now)) {
			m_cat.info("running waitlist scan now to check for denials:");
			CalendarTimeOffDayFacade toCalFacade = RmManagerFactory.getInstance().getTimeOffDayFacade();
			Collection tobeScannedTOPoolIDs = toCalFacade.getTOPoolIDsWithWaitListScanEnabled();
			for (Iterator iterator = tobeScannedTOPoolIDs.iterator(); iterator.hasNext();) {
				ID pTOPoolID = (ID) iterator.next();
				try {
					TONotifyMessage msg = new TONotifyMessage(TONotifyMessage.TO_NOTIFY_TOPOOL_FULL_SCAN);
					msg.setTOPoolId(pTOPoolID);
					msg.sendMessage();
				} catch (Exception e) {
					m_cat.error("**** Exception caught and ignored ****: ", e);
				}
			}
			lastRunTime = now;
			m_cat.info("Waitlist scan to check for denials complete:" + (new Date().getTime() - now));
		}

	}
}
