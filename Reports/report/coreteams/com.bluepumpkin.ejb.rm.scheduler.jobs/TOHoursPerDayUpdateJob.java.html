/**
 * Title: Witness Systems Request Management Module
 *
 * @author Swati Tewari , Aug 2007
 */
public class TOHoursPerDayUpdateJob implements StatefulJob {
	private static Category m_cat = Log.initCategory(TOHoursPerDayUpdateJob.class.getName());

	/**
	 * lock to ensure that this job is not reexecuted while an earlier instance is still executing.
	 */
	static Lock m_jobExecLock = new Lock();

	/*
	 * @see org.quartz.Job#execute(org.quartz.JobExecutionContext)
	 */
	@Override
	public void execute(JobExecutionContext execCtxt) throws JobExecutionException {
		String methodName = "execute";
		long startTime = new Date().getTime();
		int CHUNK_SIZE = 10000;
		long countOfDirtyRows = 0;
		try {
			// try to obtain the lock.
			if (!m_jobExecLock.lock()) {
				m_cat.debug("Quartz job not executed as an earlier job is still executing");
				return;
			}
			try {
				WorkResourceManager wrm = BbmManagerFactory.getWorkResourceManager();
				TOHoursPerDayManager toHoursMgr = RmManagerFactory.getInstance().getTOHoursPerDayManager();
				if (RequestUtil.isRMDebugEnabled()) {
					m_cat.debug("TOHoursPerDay checkHealth=" + toHoursMgr.checkHealth());
				}
				m_cat.info("TOHoursPerDay checkHealth=" + toHoursMgr.checkHealth());
				//get org list
				List<ID> orgIDs = new ArrayList<ID>();
				long maxCountAllowed = 10000;

				countOfDirtyRows = toHoursMgr.getCountOfAllDirtyTOHoursPerDay();
				m_cat.info("Total number of dirty records in DB=" + countOfDirtyRows + ": Max Count Allowed=" + maxCountAllowed + " Running Cleanup Job=" + (countOfDirtyRows > maxCountAllowed));
				if (countOfDirtyRows > maxCountAllowed) {

					orgIDs.add(Organization.YOUR_COMPANY_ID_OBJ);

					Collection<ID> orgList = wrm.getOrganizationsChildrenByIDs(orgIDs);
					orgList.add(Organization.YOUR_COMPANY_ID_OBJ);
					int dirtyNum = 0;


					//iterate per organization

					for (ID orgId : orgList) {

						//for each organization, get empIDS
						@SuppressWarnings("unchecked")
						Collection<ID> empIds = wrm.getEmployeeIds(Collections.singleton(orgId), new Date(), new Date(),
								Employee.DETAIL_LEVEL_EMPLOYEE_BASIC);

						if (empIds.isEmpty())
							continue;

						//now get dirty events for all employees
						Collection<TOHoursPerDay> dirtyHourePerDayCol = toHoursMgr.getAllDirtyTOHoursPerDay(empIds);
						//group records by employees  
						Map<ID, List<TOHoursPerDay>> sortedMap = DAOUtil.groupByField(
								TOHoursPerDayFieldInfo.TIMEOFFHOURSPERDAY_I_EMPLOYEEID, dirtyHourePerDayCol);

						if (sortedMap == null || sortedMap.isEmpty())
							continue;


						Collection<TOHoursPerDay> eventsCol = new ArrayList<TOHoursPerDay>();
						for (Iterator<List<TOHoursPerDay>> itMap = sortedMap.values().iterator(); itMap.hasNext();) {
							eventsCol.addAll(itMap.next());
							if (eventsCol.size() > CHUNK_SIZE) {
								toHoursMgr.refreshAndUpdateTOHoursPerDay(eventsCol, true);
								eventsCol.clear();
							}
						}
						toHoursMgr.refreshAndUpdateTOHoursPerDay(eventsCol, true);

					}
				}
			} catch (Exception e) {
				m_cat.error("**** Exception caught and ignored ****: " + e);
			}
			// release lock
			m_jobExecLock.releaseLock();
		} catch (Exception e) {
			// release lock if it is being held.
			if (m_jobExecLock.isLocked()) {
				m_jobExecLock.releaseLock();
			}
			m_cat.error(methodName, e);
			throw new JobExecutionException(e, false);
		} finally {
			m_cat.info("Total number of dirty records=" + countOfDirtyRows + ": CHUNK_SIZE=" + CHUNK_SIZE + ": total time for update in mins=" + ((new Date().getTime() - startTime) / 60000));

		}
	}
}
