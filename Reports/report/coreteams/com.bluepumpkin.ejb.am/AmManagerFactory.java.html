  public static GTCCAdherenceImportManager getGTCCAdherenceImportManager()
      throws AmEJBCreateException {
    try {
      EJBHomeFactory factory = EJBHomeFactory.getFactory();
      GTCCAdherenceImportManagerHome mgrHome = (GTCCAdherenceImportManagerHome) factory
          .lookUpHome(JNDINames.AM_GTCCIMPORTMANAGER_EJBHOME,
              GTCCAdherenceImportManagerHome.class);
      return mgrHome.create();

    } catch (Exception e) {
      e.printStackTrace();
      throw new AmEJBCreateException("GTCCAdherenceImportManager", e);
    }
  }

    try {
      EJBHomeFactory factory = EJBHomeFactory.getFactory();
      StringBuilder homeJNDI = new StringBuilder(
          JNDINames.AM_RTAAMANAGER_EJBHOME);
      if (whatIf) {
        homeJNDI.append(WIF_SUFFIX);
      }
      RTAAManagerEJBHome mgrHome = (RTAAManagerEJBHome) factory
          .lookUpHome(homeJNDI.toString(), RTAAManagerEJBHome.class);
      return mgrHome.create();

      throw new AmEJBCreateException("RTAAManager in WhatIf Mode: "
          + whatIf, e);
