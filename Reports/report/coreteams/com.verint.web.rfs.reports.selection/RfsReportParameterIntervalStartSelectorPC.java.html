package com.verint.web.rfs.reports.selection;

import javax.servlet.http.HttpServletRequest;

import org.w3c.dom.Node;

import com.bluepumpkin.web.core.reports.ReportControlsXml;
import com.bluepumpkin.web.core.reports.ReportParameterUtil;
import com.bluepumpkin.web.core.reports.ReportParametersPCInterface;
import com.bluepumpkin.web.ireport.framework.RffParameter;
import com.bluepumpkin.web.ireport.framework.RffRequest;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.rfs.reports.util.RfsReportControlsXml;
import com.verint.web.rfs.reports.util.RfsReportParameterConstants;
import com.witness.web.uif.pagecomponent.CompositePC;
import com.witness.web.uif.pagecomponent.picker.date.DatePickerPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

/**
 * 
 * Interval start selector - will display either month and year or date based on the selected Interval (Monthly/Daily)
 *
 */
public class RfsReportParameterIntervalStartSelectorPC extends CompositePC implements ReportParametersPCInterface {

	private static final long serialVersionUID = 1L;
	
	private static RfsReportControlsXml rfsReportControls = RfsReportControlsXml.getInstance();
	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();
	private static ReportParameterUtil reportUtil = ReportParameterUtil.getInstance();
	
	private RfsReportParameterSingleSelectorPC m_monthPC = null;
	private RfsReportParameterSingleSelectorPC m_yearPC = null;
	private DatePickerPC m_datePC = null;
	private String m_periodType = null;
	private final int DEFAULT_MONTH = 1;
	private final int DEFAULT_YEAR = 1900;
	
	public RfsReportParameterIntervalStartSelectorPC(RequestContext context, ToolbarPC toolbar, Node controlNode) {
		super(context);
		setName(reportControls.getControlSingleParameterName(controlNode));
				
		reportControls.isControlRequired(controlNode);
	
		m_periodType = context.getRequest().getParameter(RfsReportParameterConstants.PARAM_PERIODTYPE);
		if(m_periodType == null) {
			m_periodType = RfsReportParameterConstants.PERIODTYPE_MONTHLY;
		}
		Node monthNode = rfsReportControls.getControlMonth(controlNode);
		Node yearNode = rfsReportControls.getControlYear(controlNode);
		
		if (RfsReportParameterConstants.PERIODTYPE_MONTHLY.equals(m_periodType)) {
			m_monthPC = new RfsReportParameterSingleSelectorPC(context, toolbar, monthNode);
			addChildComponent(m_monthPC);
			m_yearPC = new RfsReportParameterSingleSelectorPC(context, toolbar, yearNode);
			addChildComponent(m_yearPC);
		}
		else {
			m_datePC = new DatePickerPC(context, "StartDate");
			addChildComponent(m_datePC);
		}
	}
		
	public void initForDisplay() {
		if (m_isInitializedForDisplay)
			return;

		super.initForDisplay();
	}
	
	public String getLabel(RffRequest rff, Node controlNode) {
		String label = reportControls.getControlLabel(controlNode);
		if (label == null || label.length() == 0) {
			label = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,rfsReportControls.getControlPickerName(controlNode));
		}
		return label;
	}

	public String getHtmlDisplay() {
		// TODO Auto-generated method stub
		return null;
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode) {
		return getHtmlDisplay(rff, controlNode, false);
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode,boolean readOnly) {
		
		StringBuffer content = new StringBuffer();
		
		Node monthNode = rfsReportControls.getControlMonth(controlNode);
		Node yearNode = rfsReportControls.getControlYear(controlNode);
		
		if(RfsReportParameterConstants.PERIODTYPE_MONTHLY.equals(m_periodType)) {
			content.append(m_monthPC.getHtmlDisplay(rff, monthNode, false, true))
					.append(HtmlUtil.NBSP)
					.append(m_yearPC.getHtmlDisplay(rff, yearNode, false, true))
					.append(HtmlUtil.makeHiddenInput(RfsReportParameterConstants.PARAM_STARTDATE, ""));
		} else {			
			content.append(m_datePC.getHtmlDisplay())
					.append(HtmlUtil.makeHiddenInput(RfsReportParameterConstants.PARAM_MONTH, ""))
					.append(HtmlUtil.makeHiddenInput(RfsReportParameterConstants.PARAM_YEAR, ""));
		}
		
		return content.toString();
	}

	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode) {

		RffParameter periodTypeParam = rff.getRffParameter(RfsReportParameterConstants.PARAM_PERIODTYPE);
		if(periodTypeParam != null)
			periodTypeParam.setCurrentValue(request.getParameter(RfsReportParameterConstants.PARAM_PERIODTYPE));
		
		m_periodType = request.getParameter(RfsReportParameterConstants.PARAM_PERIODTYPE);
		if(m_periodType == null) {
			m_periodType = RfsReportParameterConstants.PERIODTYPE_MONTHLY;
		}
		if(RfsReportParameterConstants.PERIODTYPE_MONTHLY.equals(m_periodType)) {
			m_monthPC.setRffParameterValue(request, rff, rfsReportControls.getControlMonth(controlNode),true);	
			m_yearPC.setRffParameterValue(request, rff, rfsReportControls.getControlYear(controlNode),true);
		} else {
			Node paramNode = rfsReportControls.getControlDate(controlNode);
			if(paramNode != null) {
				String paramName = reportControls.getParameterName(paramNode);
				RffParameter param = rff.getRffParameter(paramName);
				if (param == null) {
					throw new IllegalArgumentException("Parameter in controls but not in report: " + paramName);
				}
				/*LocalDate selectedDate = (LocalDate) param.getCurrentValue();
				SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
				param.setCurrentValue(simpleDateFormat.format(DateTimeUtil.getDayStart(selectedDate.getTime())));
				param.setCurrentValue(request.getParameter("StartDate"));*/
				reportUtil.setRffParameterValue(rff,paramName,request.getParameter("StartDate"));
			}
			Node monthNode = rfsReportControls.getControlMonth(controlNode);
			if(monthNode != null) {
				String paramName = reportControls.getParameterName(monthNode);
				RffParameter param = rff.getRffParameter(paramName);
				if (param == null) {
					throw new IllegalArgumentException("Parameter in controls but not in report: " + paramName);
				}
				reportUtil.setRffParameterValue(rff, paramName,DEFAULT_MONTH);
			}
			Node yearNode = rfsReportControls.getControlYear(controlNode);
			if(yearNode != null) {
				String paramName = reportControls.getParameterName(yearNode);
				RffParameter param = rff.getRffParameter(paramName);
				if (param == null) {
					throw new IllegalArgumentException("Parameter in controls but not in report: " + paramName);
				}
				reportUtil.setRffParameterValue(rff, paramName,DEFAULT_YEAR);
			}
			
		}		
	}
	
	public String getCustomJavaScript(Node controlNode) {
		return null;
	}
	
	public boolean extractParameters(HttpServletRequest request) {
		super.extractParameters(request);
		return true;
	}
}
