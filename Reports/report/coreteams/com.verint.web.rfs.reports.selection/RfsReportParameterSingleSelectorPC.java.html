package com.verint.web.rfs.reports.selection;

import java.util.List;

import javax.servlet.http.HttpServletRequest;

import org.w3c.dom.Node;

import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.web.core.reports.ReportControlsXml;
import com.bluepumpkin.web.core.reports.ReportParameterUtil;
import com.bluepumpkin.web.core.reports.ReportParametersPCInterface;
import com.bluepumpkin.web.ireport.framework.RffParameter;
import com.bluepumpkin.web.ireport.framework.RffRequest;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.rfs.reports.util.RfsReportParameterConstants;
import com.verint.web.rfs.reports.util.RfsReportParameterUtil;
import com.witness.web.uif.pagecomponent.form.FormInputPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

/**
 * PageComponent for single selector drop down for RFS Forecaster report
 * parameter pickers
 * 
 */
public class RfsReportParameterSingleSelectorPC extends FormInputPC implements ReportParametersPCInterface {

	private static final long serialVersionUID = 1L;

	private static ReportParameterUtil reportUtil = ReportParameterUtil.getInstance();
	private static ReportControlsXml reportControls = ReportControlsXml.getInstance();
	private static RfsReportParameterUtil rfsReportUtil = RfsReportParameterUtil.getInstance();

	private Localizer mLocalizer = null;
	private String mPickerName = null;

	public RfsReportParameterSingleSelectorPC(RequestContext context, ToolbarPC toolbar, Node controlNode) {
		super(context);
		mLocalizer = m_context.getLocalizer();
		setName(reportControls.getControlSingleParameterName(controlNode));
		setIsRequired(reportControls.isControlRequired(controlNode));
	}

	public RfsReportParameterSingleSelectorPC(RequestContext context, String picker) {
		super(context);
		mLocalizer = m_context.getLocalizer();
		mPickerName = picker;
	}

	public String getLabel(RffRequest rff, Node controlNode) {
		String label = reportControls.getControlLabel(controlNode);
		mPickerName = rfsReportUtil.getControlPickerName(m_context, controlNode);

		if (label == null || label.length() == 0) {
			label = mLocalizer.i18n(FsWebBundleKey.BUNDLE_NAME, mPickerName);
		}
		return label;
	}

	public String getHtmlDisplay() {

		String currentValue = m_context.getRequest().getParameter(mPickerName);
		if (currentValue == null) {
			currentValue = rfsReportUtil.getDefaultValue(m_context, mPickerName);
		} else if (RfsReportParameterConstants.BRANCH_VIEW.equals(mPickerName)) {
			currentValue = rfsReportUtil.getCurrentBranchView(m_context);
		}
		List<StringsPair> options = rfsReportUtil.getOptions(m_context, mPickerName, currentValue);
		return HtmlUtil.makeSelectInput(mPickerName, new String[] { currentValue }, rfsReportUtil.getJavaScript(mPickerName), options,
				false, true);
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode) {
		return getHtmlDisplay(rff, controlNode, false);
	}

	/**
	 * Called from other rfs page components which have multiple objects within
	 * the same picker.
	 * 
	 * @param rff
	 * @param controlNode
	 * @param readOnly
	 * @param isParamNode
	 * @return
	 */
	public String getHtmlDisplay(RffRequest rff, Node controlNode, boolean readOnly, boolean isParamNode) {

		if (!isParamNode) {
			return getHtmlDisplay(rff, controlNode, readOnly);
		} else {
			String paramName = reportControls.getParameterName(controlNode);

			RffParameter param = rff.getRffParameter(paramName);
			String currentValue = null;
			currentValue = m_context.getRequest().getParameter(paramName);
			if (currentValue == null && param != null && param.getCurrentValue() != null) {
				currentValue = param.getCurrentValue().toString();
			}
			if (mPickerName == null) {
				mPickerName = rfsReportUtil.getControlPickerName(m_context, controlNode);
			}

			List<StringsPair> options = rfsReportUtil.getOptions(m_context, controlNode, currentValue, mPickerName, rff.getParentId());

			return HtmlUtil.makeSelectInput(paramName, new String[] { currentValue }, rfsReportUtil.getJavaScript(mPickerName), options,
					false, !readOnly);
		}
	}

	public String getHtmlDisplay(RffRequest rff, Node controlNode, boolean readOnly) {

		Node paramNode = reportControls.getParameter(controlNode);
		if (paramNode == null) {
			throw new IllegalArgumentException("A parameter was not specified for this control.");
		}

		String paramName = reportControls.getParameterName(paramNode);

		RffParameter param = rff.getRffParameter(paramName);
		String currentValue = null;
		currentValue = m_context.getRequest().getParameter(paramName);
		if (currentValue == null && param != null && param.getCurrentValue() != null) {
			currentValue = param.getCurrentValue().toString();
		}
		if (mPickerName == null) {
			mPickerName = rfsReportUtil.getControlPickerName(m_context, controlNode);
		}

		List<StringsPair> options = rfsReportUtil.getOptions(m_context, paramNode, currentValue, mPickerName, rff.getParentId());

		return HtmlUtil.makeSelectInput(paramName, new String[] { currentValue }, rfsReportUtil.getJavaScript(mPickerName), options, false,
				!readOnly);
	}

	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode, boolean isParamNode) {
		mPickerName = rfsReportUtil.getControlPickerName(m_context, controlNode);
		if (!isParamNode) {
			setRffParameterValue(request, rff, controlNode);
		} else {
			String paramName = reportControls.getParameterName(controlNode);
			RffParameter param = rff.getRffParameter(paramName);
			if (param == null) {
				throw new IllegalArgumentException("Parameter in controls but not in report: " + paramName);
			}

			reportUtil.setRffParameterValue(request, param, "STRING");

			String displayParamName = reportControls.getParameterDisplayName(controlNode);
			RffParameter displayParam = rff.getRffParameter(displayParamName);
			if (displayParam != null) {
				String displayValue = getOptionValue(rff, controlNode, mPickerName, (String) param.getCurrentValue());
				if (displayValue == null) {
					displayParam.setCurrentValue(param.getCurrentValue());
				} else {
					displayParam.setCurrentValue(displayValue);
				}
			}
		}
	}

	public void setRffParameterValue(HttpServletRequest request, RffRequest rff, Node controlNode) {
		mPickerName = rfsReportUtil.getControlPickerName(m_context, controlNode);
		if (reportControls.hasSingleParameter(controlNode) == false) {
			throw new IllegalArgumentException("Can only handle single report parameter.");
		}
		Node paramNode = reportControls.getParameter(controlNode);
		if (paramNode == null) {
			throw new IllegalArgumentException("A parameter was not specified for this control.");
		}
		String paramName = reportControls.getParameterName(paramNode);
		RffParameter param = rff.getRffParameter(paramName);
		if (param == null) {
			throw new IllegalArgumentException("Parameter in controls but not in report: " + paramName);
		}
		reportUtil.setRffParameterValue(request, param, "STRING");

		String displayParamName = reportControls.getParameterDisplayName(paramNode);
		RffParameter displayParam = rff.getRffParameter(displayParamName);
		if (displayParam != null) {
			String displayValue = getOptionValue(rff, paramNode, mPickerName, (String) param.getCurrentValue());
			displayParam.setCurrentValue(displayValue == null ? param.getCurrentValue() : displayValue);
		}
	}

	private String getOptionValue(RffRequest rff, Node paramNode, String pickername, String currentKey) {
		List<StringsPair> options = rfsReportUtil.getOptions(m_context, paramNode, "", pickername, rff.getParentId());
		if (options == null || options.size() == 0) {
			return "";
		}
		return reportUtil.getOptionValue(options, currentKey);
	}

	public String getCustomJavaScript(Node controlNode) {
		return null;
	}
}
