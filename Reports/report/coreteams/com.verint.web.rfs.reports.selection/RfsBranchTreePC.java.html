package com.verint.web.rfs.reports.selection;

import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.verint.ejb.rfs.reports.model.RfsBranch;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.rfs.reports.RfsReportParametersModelHandler;
import com.witness.web.uif.pagecomponent.tree.MultiColInputTreePC;
import com.witness.web.uif.system.RequestContext;

public class RfsBranchTreePC extends MultiColInputTreePC {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public static RfsBranch ROOT_OBJ = null;
	static { ROOT_OBJ = new RfsBranch(); ROOT_OBJ.setID(new ID(-1)); }
	public static final String CHECKED_ID_FN = "checkedOrgID";

	/**
	 * Constructor
	 */
	public RfsBranchTreePC(RequestContext context) {
		super(context, CHECKED_ID_FN, FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.MULTIPLE_BRANCHES);
		setName("orgTree");
		setOnSelectAction(HANDLE_ON_SELECT_SYNC_CHILD);
	}

	/**
	 * Subclass must implement this method to retrieve the tree objects.
	 */
	protected Collection retrieveTreeObjects() throws Exception {
		return RfsReportParametersModelHandler.getBranches(m_context);
	}

	/**
	 * Sets selected branch ID. It invokes setCheckedID internally and it is meant
	 * mainly for auto extract parameters
	 */
	public void setCheckedOrgID(ID[] checkedID) {
		setCheckedID(checkedID);
	}
	
	protected Object getTreeRootData() {
		return ROOT_OBJ;
	}

}
