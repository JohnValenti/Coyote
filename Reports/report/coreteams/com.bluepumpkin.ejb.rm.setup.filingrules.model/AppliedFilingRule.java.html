package com.bluepumpkin.ejb.rm.setup.filingrules.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;


/**
 * Title:        AppliedFilingRule
 * Description:  Represents a AppliedFilingRule
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc.
 * @author       Shailendra Sharma
 * @version 	 1.0
 */

public class AppliedFilingRule extends ValueObjectBase {

    // AppliedFilingRuleFieldInfo as FieldInfo for the class.
    private static FieldInfo m_fieldInfo = new AppliedFilingRuleFieldInfo();

    /**
     * The boolean value telling us whether the user can change the isApplied value for this rule.
     */
    private boolean m_canChangeApply = false; 

    
    /* Default constructor for the class.
     */
    public AppliedFilingRule() {
        super();
    }

    /* Returns the FieldInfo attribute for the class.
     * @return FieldInfo - FieldInfo attribute for the class.
     */
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }

    // getters

    /* Returns the ID object as OrganizationId.
     * @return ID - ID object as OrganizationId.
     */
    public ID getOrganizationId() {
        return getFieldValueID(AppliedFilingRuleFieldInfo.APPLIEDFILINGRULE_I_ORGANIZATIONID);
    }

    /* Returns the boolean value as OrganizationId.
     * @return ID - ID object as OrganizationId.
     */
    public boolean getApplied() {
        return getFieldValueBoolean(AppliedFilingRuleFieldInfo.APPLIEDFILINGRULE_B_ISAPPLIED);
    }

    /* Returns the ID object value for the Request Filing Rule.
     * @return ID - ID object value for the Request Filing Rule.
     */
    public ID getRequestFilingRuleId() {
        return getFieldValueID(AppliedFilingRuleFieldInfo.APPLIEDFILINGRULE_I_REQUESTFILINGRULEID);
    }

    /* Set ID object value as OrganizationId.
     * @param ID - ID object value as OrganizationId.
     */
    public void setOrganizationId(ID pOrganizationId) {
        setFieldValue(AppliedFilingRuleFieldInfo.APPLIEDFILINGRULE_I_ORGANIZATIONID, pOrganizationId);
    }

    /* Set boolean value passed as argument to Applied status class attribute.
     * @param boolean - boolean value for Applied  status for requestfiling rule.
     */
    public void setApplied(boolean  pIsApplied) {
        setFieldValue(AppliedFilingRuleFieldInfo.APPLIEDFILINGRULE_B_ISAPPLIED, pIsApplied);
    }

    /* Set ID value passed as argument to RequestFiling Rule ID  attribute.
     * @param ID - ID object RequestFiling Rule ID.
     */
    public void setRequestFilingRuleId(ID pRequestFilingRuleId) {
        setFieldValue(AppliedFilingRuleFieldInfo.APPLIEDFILINGRULE_I_REQUESTFILINGRULEID, pRequestFilingRuleId);
    }    
    
    /**
     * Returns the boolean value telling us whether the user can change the isApplied value for this rule.
     */
    public boolean getCanChangeApply() {
        return m_canChangeApply;
    }
    
    /** 
     * Sets the boolean value telling us whether the user can change the isApplied value for this rule.
     */
    public void setCanChangeApply(boolean  pCanChangeApply) {
        m_canChangeApply = pCanChangeApply;
    }

}
