package com.bluepumpkin.ejb.bbm.schedule.util.optimizeBreaks;

public class ObjectiveFunction
{
    public final void BindNetStaffingFTE(double d)
    {
        double oldNetStaffingFTE = netStaffingFTE;
        netStaffingFTE += d;
        Logger.Log((d > 0 ? "Binding " : "Unbinding ") + Math.abs(d) + (d > 0 ? " to" : " from") + " NetStaffing, old value = " + oldNetStaffingFTE + ", new value = " + netStaffingFTE);
    }
    public final void BindConstraint(double d)
    {
        constraints += d;
        Logger.Log((d > 0 ? "Binding " : "Unbinding ") + Math.abs(d) + (d > 0 ? " to" : " from") + " Constraints, new value = " + constraints);
        if (constraints < 0)
            Logger.Log("break");
    }
    public final double GetScore()
    {
        return Math.pow(constraints, 2) + netStaffingFTE;
    }
    @Override
    public String toString()
    {
        return "Score = " + GetScore() + ": NetStaffing = " + netStaffingFTE + ", Constraints = " + constraints;
    }
    private double netStaffingFTE;
    private double constraints;
}
