    public int GetPatternDayIndex(ShiftAssignment sa)
    {
        com.bluepumpkin.ejb.bbm.workresource.model.Organization shiftPatternOrganization = hmOrganizations.get(shiftPattern.getOrganizationID());
        Calendar calStart = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
        calStart.setTime(sa.getStartTime());
        calStart.setTimeZone(shiftPatternOrganization.getTimeZone());
        DayOfWeek shiftAssignmentStartDay = DayOfWeek.fromCalendarConstant(calStart.get(Calendar.DAY_OF_WEEK));
        com.bluepumpkin.ejb.bbm.workresource.model.Organization organization = hmOrganizations.get(new ID(employee.getOrganization()));
        DayOfWeek orgWeekStartDay = organization.getWeekStartDay();
        if (shiftPattern.getNumOfWeeks() == 1) {
            return ((shiftAssignmentStartDay.ordinal() - orgWeekStartDay.ordinal()) + 7) % 7;
        }
        return calStart.get(Calendar.DAY_OF_MONTH) - 1;
    }

    public boolean Run()
    {
        try
        {
            if (this.shift != null && this.shift.getID() != null)
            {
                Logger.Log("Optimize primary shift");
                Logger.Tab();

                Date startTime = shiftAssignment.getStartTime();
                Date endTime = shiftAssignment.getEndTime();
                if (shiftAssignment.getExtensionBefore() > 0 && shiftAssignment.getOTExtensionBeforeActivityID() != null)
                {
                    Calendar calStart = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
                    calStart.setTime(startTime);
                    calStart.add(Calendar.MINUTE, shiftAssignment.getExtensionBefore());
                    startTime = calStart.getTime();
                }
                if (shiftAssignment.getExtensionAfter() > 0 && shiftAssignment.getOTExtensionAfterActivityID() != null)
                {
                    Calendar calEnd = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
                    calEnd.setTime(endTime);
                    calEnd.add(Calendar.MINUTE, -shiftAssignment.getExtensionAfter());
                    endTime = calEnd.getTime();
                }
                OptimizeShift(startTime, endTime, this.shift.getID(), this.shift.getName(), this.shiftAssignment.getActivityID(), this.shift.getDuration(), this.shift.getShiftEvents());
                Logger.Untab();
            }

            if (this.shift == null && this.mainShift != null) {
                this.shift = this.mainShift; //GQ: We need the shift even if we are only optimizing the extensions
            }

            if (this.shift != null) {
	            if (this.otBefore != null && this.otBefore.getID() != null && shiftAssignment.getExtensionBefore() > 0) {
	                Logger.Log("Optimize OT before");
	                Logger.Tab();
	                Date otEnd = new TimeSpan(0, otBefore.getDuration(), 0).addTo(shiftAssignment.getStartTime());
	                OptimizeShift(shiftAssignment.getStartTime(), otEnd,
	                        this.otBefore.getID(), this.otBefore.getName(), this.shiftAssignment.getOTExtensionBeforeActivityID(), this.otBefore.getDuration(), this.otBefore.getShiftEvents()); //GQ WARNING: we pass extID as shiftID
	                //dealing with GAP
	                if (otBefore.getMinGap() > 0) {
	                	ShiftEventAssignment gap = new ShiftEventAssignment();
	                	gap.setActivityID(com.bluepumpkin.ejb.bbm.activity.model.Activity.ACTIVITY_SHIFT_OVERTIME_GAP);
	                	gap.setStartTime(otEnd);
	                	gap.setDuration(otBefore.getMinGap());
	                	gap.setLocked(true);
	                	gap.setOverTimeGapType((short)1);
	                	gap.addWorkResourceID(shiftAssignment.getWorkResourceID());
	                	shiftAssignment.addShiftEventAssignment(gap);
	                }
	                Logger.Untab();
	            }
	            
	            if (this.otAfter != null && this.otAfter.getID() != null && shiftAssignment.getExtensionAfter() > 0) {
	                Logger.Log("Optimize OT after");
	                Logger.Tab();
	                Date otStart = new TimeSpan(0, otAfter.getDuration(), 0).subFrom(shiftAssignment.getEndTime());
	                OptimizeShift(otStart, shiftAssignment.getEndTime(),
	                        this.otAfter.getID(), this.otAfter.getName(), this.shiftAssignment.getOTExtensionAfterActivityID(), this.otAfter.getDuration(), this.otAfter.getShiftEvents()); //GQ WARNING: we pass extID as shiftID
	                //dealing with GAP
	                if (otAfter.getMinGap() > 0) {
	                	ShiftEventAssignment gap = new ShiftEventAssignment();
	                	gap.setActivityID(com.bluepumpkin.ejb.bbm.activity.model.Activity.ACTIVITY_SHIFT_OVERTIME_GAP);
	                	gap.setStartTime(new TimeSpan(0, otAfter.getMinGap(), 0).subFrom(otStart));
	                	gap.setDuration(otAfter.getMinGap());
	                	gap.setLocked(true);
	                	gap.setOverTimeGapType((short)2);
	                	shiftAssignment.addShiftEventAssignment(gap);
	                	gap.addWorkResourceID(shiftAssignment.getWorkResourceID());
	                }
	                Logger.Untab();
	            }
            }
            Logger.enabled = true;
            Logger.Log("Outputs ******************************************************************* ");
            Logger.Log("Shift Assignment ********************************************************** ");
            Logger.Log(shiftAssignment.toString());
            Logger.Log(shiftAssignment.getChildObjects(ShiftAssignmentFields.CHILD_SHIFT_EVENT).toString());
        }
        finally
        {
            Logger.CloseLog();
        }
        return true;

    public boolean OptimizeShift(Date start, Date end, ID shiftID, String shiftName, ID shiftActivityID, int shiftDuration, Collection<com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent> shiftEvents)
    {
        if (!start.before(end)) {
            return false;
        }

		int origMinSpacing = this.shift.getMinSpacing();
        TimeSpan duration = new TimeSpan(0, shiftDuration, 0);

		int numShiftEventsToAdd = 0;
		int numShiftEventsAdded = 0; //the number of new shift events that are to be placed in the shift, as determined by the optimizer
		int highestNumShiftEventsAdded = 0; //the highest number of new shift events that are to be placed in the shift, for any value of minSpacing.
		int curMinSpacing = origMinSpacing >= shiftDuration ? 0 : origMinSpacing; //the value of minSpacing that is being used in the current iteration.
		int lastMinSpacing = origMinSpacing; //the value of minSpacing that was used the last time through the loop.
		int bestMinSpacing = 0; //the value of minSpacing that results in the highest number of shift events added.
		Shift newShift = null;

		//ESR#4237318/QC#125282: Shifts are being created without their shift activities when done manually through the web calendar.
		//While numShiftEventsAdded<numShiftEventsToAdd, we will decrement shiftEventSpacing and try ScheduleShiftEvents again.
		//This solution works, although we should consider refactoring the break optimization later. The better solution would be to
		//change the individual break placement algorithm so that it places the break even if it cannot satisfy the minSpacing hard constraint.
	   
		// ESR #4443662: Once all mandatory events are placed, do not loop through, since another pass through the loop with a decremented shiftEventSpacing
		// will violate the min space constraint of the shift template.
		Collection<ID> mandatoryShiftEventIDs = shift.getMandatoryShiftEventsID().keySet();
		
		do {
			//Create a new Scenario each time in the loop since its member variables depend on the curMinSpacing
			Logger.Log("Create new Scenario");
			scenario = new Scenario(start, end);
			double proficiency = employee.getProficiency() * 1.0 / 10;
			scenario.SetAgent(new Agent(scenario, employee.getFirstName() + " " + employee.getLastName(), proficiency));

			ArrayList<ShiftEvent> shiftEvents_ = new ArrayList<ShiftEvent>();
			Logger.Log("Add shift events");
			Logger.Tab();

			//put the must-have shift event first
			//shiftEvents = orderShiftEvents(shiftEvents);

			for (com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent shiftEvent : shiftEvents)
				AddShiftEvent(shiftEvents_, shiftEvent);
			Logger.Untab();
			numShiftEventsToAdd = shiftEvents_.size();

			Activity activity = GetActivity(hmActivities.get(shiftActivityID));
			Logger.Log("Shift Activity = " + activity.name);

			//create the shift using the curMinSpacing
			Pair<Shift, List<ID>> shiftEventIDsPair = createShiftAndScheduleEvents(start, end, shiftID, shiftName, activity, duration, shiftEvents_, curMinSpacing);
			newShift = shiftEventIDsPair.getFirst();
			numShiftEventsAdded = shiftEventIDsPair.getSecond().size();

			if (numShiftEventsAdded > highestNumShiftEventsAdded) {
				highestNumShiftEventsAdded = numShiftEventsAdded;
				bestMinSpacing = curMinSpacing;
			}

			Collection<ID> shiftEventIDsAdded = shiftEventIDsPair.getSecond();
			if (shiftEventIDsAdded.containsAll(mandatoryShiftEventIDs)) {
				break;
			}
			
			lastMinSpacing = curMinSpacing;
			curMinSpacing -= 15;

		} while ((numShiftEventsAdded < numShiftEventsToAdd) && (curMinSpacing >= 0));

		System.out.println(String.format("Optimizer.OptimizeShift(): numShiftEventsToAdd=%d, numShiftEventsAdded=%d, numOfMandatoryShiftEvents=%d, origMinSpacing=%d, curMinSpacing=%d, lastMinSpacing=%d, bestMinSpacing=%d",
					  numShiftEventsToAdd, numShiftEventsAdded, mandatoryShiftEventIDs.size(), origMinSpacing, curMinSpacing, lastMinSpacing, bestMinSpacing));
		
		Logger.Log("Saving shift event assignments");
        Logger.Tab();
        for (ShiftEvent shiftEvent : newShift.shiftEvents)
        {
            Collections.sort(shiftEvent.assignments, new ShiftEvent.Assignment.Compare());
            for (ShiftEvent.Assignment a : shiftEvent.assignments)
            {
                if (!a.locked && a.binding != null && a.binding.activity != null) {
                    AddShiftEventAssignment(hmShiftEvents.get(new ID(shiftEvent.id)), scenario.IntervalToDateTime(a.binding.start), new ID(a.binding.activity.id), a.binding.activity.layer + 1, shiftAssignment);
                }
            }
        }
        Logger.Untab();
        scenario.Log();
        return true;
    }

	private ArrayList<com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent> orderShiftEvents(Collection<com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent> shiftEvents)
	{
		ArrayList<com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent> orderedList = new ArrayList<com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent>(shiftEvents.size());
		ArrayList<com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent> remainingList = new ArrayList<com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent>();

		for (com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent shiftEvent : shiftEvents)
		{
			if (!shiftEvent.isFlexible() || (shiftEvent.isFlexible() && shiftEvent.getMinCount()>0))
			{
				orderedList.add(shiftEvent);
			}
			else
			{
				remainingList.add(shiftEvent); //this is the list of optional shift events. It will be added last to the orderedList
			}
		}
		orderedList.addAll(remainingList);
		return orderedList;
	}
	

	/**
	 * Create a Shift and schedule the shift events to it.
	 * NOTE: This doesn't actually write anything to the database. It's just an in-memory Shift object.
	 * @return The number of shift events added to the shift.
	 */
	private Pair<Shift, List<ID>> createShiftAndScheduleEvents(Date start, Date end, ID shiftID, String shiftName, Activity activity,
											 TimeSpan duration, ArrayList<ShiftEvent> shiftEvents_, int minSpacing)
	{
		Logger.Log("Creating shift");

		Pair<TimeSpan, TimeSpan> shiftEventSpacing = this.shift==null ? new Pair<TimeSpan, TimeSpan>(new TimeSpan(0, 0, 0), new TimeSpan(0, 0, 0)) :
				new Pair<TimeSpan, TimeSpan>(new TimeSpan(0, minSpacing, 0), new TimeSpan(0, this.shift.getMaxSpacing(), 0));

		Shift newShift = new Shift(scenario, shiftID.toString(), shiftName, activity, start, duration, shiftEvents_, shiftEventSpacing);
		newShift.Log();
		scenario.shift = newShift;

		scenario.Initialize();
		Logger.Log("Loading shift event assignments");
		Logger.Tab();
		for (ShiftEventAssignment sea : (Collection<ShiftEventAssignment>) shiftAssignment.getChildObjects(ShiftAssignmentFields.CHILD_SHIFT_EVENT))
		{
			if (sea.getStartTime().before(start) || sea.getEndTime().after(end))
			{
				continue;
			}
			if (!sea.isLocked())
			{
				if(sea.getID() == null ){
					shiftAssignment.clearCreatedChildObjectMap(ShiftAssignmentFields.CHILD_SHIFT_EVENT);
				}else{
					shiftAssignment.removeShiftEventAssignment(sea.getID());
				}
				continue;
			}
			if (sea.getShiftEventID() == null)
			{
				continue; //Gap events don't have shiftEventID
			}
			ShiftEvent shiftEvent = scenario.shiftEvents.get(sea.getShiftEventID().toString());
			if (shiftEvent != null)
			{
				Activity activity_ = scenario.activities.get(sea.getActivityID().toString());
				newShift.LoadShiftEvent(shiftEvent, activity_, sea.getStartTime());
			}
		}
		Logger.Untab();

		//int numShiftEventsAdded = 0; //the number of new shift events that are to be placed in the shift, as determined by the optimizer
		newShift.ScheduleShiftEvents();

		//count the number of shift events added for the current minSpacing value
		List<ID> shiftEventIDsAdded = new ArrayList<ID>();
		
		for (ShiftEvent shiftEvent : newShift.shiftEvents)
		{
			for (ShiftEvent.Assignment a : shiftEvent.assignments)
			{
				if (!a.locked && a.binding != null && a.binding.activity != null) {
					//numShiftEventsAdded++;
					shiftEventIDsAdded.add(new ID(shiftEvent.id));
				}
					
			}
		}

		return new Pair<Shift, List<ID>>(newShift, shiftEventIDsAdded);
	}

	private void AddQueueIfSkilled(ArrayList<Queue> skilledQueues, com.bluepumpkin.ejb.bbm.workload.model.Queue queue)
	    {

	        Queue queue_;
	        if (this.hmQueueSkills.containsKey(queue.getID())) {
	        	if (scenario.queues.containsKey(queue.getID().toString()))
	        	{
	        		queue_ = scenario.queues.get(queue.getID().toString());
	        		Logger.Log("Queue " + queue_.name + " already loaded");
	        	}
	        	else
	        	{
	        		double queueProficiency = 1.0;
	        		Collection<ID> queueSkills = (Collection<ID>) hmQueueSkills.get(queue.getID());
	        		if (queueSkills != null)
	        		{
	        			double totalProficiency = 0.0;
	        			for (ID idSkill : queueSkills)
	        			{
	        				if (!hmSkillAssignments.containsKey(idSkill)) {
	        					break;
	        				}
	        				com.bluepumpkin.ejb.bbm.skill.model.SkillAssignment skillAssignment = hmSkillAssignments.get(idSkill);
	        				double proficiency = skillAssignment.getProficiency() * 1.0;
	        				totalProficiency += proficiency;
	        			}
	        			queueProficiency = totalProficiency / queueSkills.size();
	        			if (queueProficiency == 0.0)
	        			{
	        				Logger.Log("Agent not skilled for " + queue.getName() + " so don't add");
	        						return;
	        			}
	        		} else {
	        			Logger.Log("No required skills");
	        		}
	        		queue_ = new Queue(scenario, queue.getID().toString(), queue.getName(), queueProficiency);
	        		Collection<NetStaffing> netStaffings = getNetStaffingForQueue(queue);
	        		if (netStaffings != null) //GQ
	        		{
	        			Logger.Log("Loading Net Staffing objects");
	        			Logger.Tab();
	        			for (NetStaffing netStaffing : netStaffings)
	        			{
	        				Logger.Log("Loading " + netStaffing);
	        				queue_.SetNetStaffing(netStaffing.from, netStaffing.to, new TimeSpan(0, netStaffing.minutes, 0));
	        			}
	        			Logger.Untab();
	        		}
	        		scenario.queues.put(queue_.id, queue_);
	        		Logger.Log("Loading queue: ");
	        		queue_.Log();
	        	}
	        	Logger.Log("Add skilled queue " + queue_.name);
	        	skilledQueues.add(queue_);
	        }
	    }

	    /**

	     * combined queue (ID -1), since the hmQueueNetStaffing map will only need to store the combined queue's data.

	    private Collection<NetStaffing> getNetStaffingForQueue(com.bluepumpkin.ejb.bbm.workload.model.Queue queue)

	        }
	        return hmQueueNetStaffing.get(queue.getID());
	    }


	    private void AddShiftEvent(ArrayList<ShiftEvent> shiftEvents, com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent shiftEvent)
	    {
	        ShiftEvent shiftEvent_;
	        if (scenario.shiftEvents.containsKey(shiftEvent.getID().toString())) {

	                Activity a_ = GetActivity(a);
	                if (activity_ != null) {
	                    activities.add(a_);
	                }
	            }
	            Pair<Date, Date> startWindow = new Pair<Date, Date>();
	            if (shiftEvent.isAnyTime())
	            {
	                startWindow.setFirst(scenario.start);
	                startWindow.setSecond((new TimeSpan(0, shiftEventLength, 0).subFrom(scenario.end))); //GQ

	                if ((new TimeSpan(0, shiftEventLength, 0).addTo(scenario.start)).after(scenario.end) ||
	                        (startWindow.getSecond().before(scenario.start))) {
	                    return; //GQ: It don't fit. We must quit.
	                }
	            }
	            else if (shiftEvent.isCafeteria()) // && shiftPattern != null)
	            {
	            	int dayStartOffSet = getDayStartOffSet(shiftEvent);
					Calendar dayStart = masterCalendar; //scenario.getCalendar();
					dayStart.setTime(shiftAssignment.getStartTime());
	                //dayStart.setTimeZone(shiftPatternOrganization.getTimeZone());
	                dayStart.set(Calendar.HOUR_OF_DAY, 0);
	                dayStart.set(Calendar.MINUTE, 0);
	                dayStart.set(Calendar.SECOND, 0);
	                dayStart.add(Calendar.MINUTE, dayStartOffSet);
	                Date startWindowDate = new TimeSpan(0, shiftEvent.getStart(), 0).addTo(dayStart.getTime());
	                Date endWindowDate = new TimeSpan(0, shiftEvent.getEnd(), 0).addTo(dayStart.getTime());
	                
	                if (isShiftEventStartBeforeShiftSinceShiftCrossDay(shiftAssignment.getStartTime().getTime(), dayStart.getTime().getTime() + shiftEvent.getEnd()*TimeZoneUtil.MINUTE_IN_MILLISECONDS)) {
	                	startWindowDate = TimeZoneUtil.addDay(startWindowDate);
	                }
	                if(endWindowDate.before(startWindowDate)){
	                	endWindowDate = TimeZoneUtil.addDay(endWindowDate);
	                }

	                startWindow.setFirst(startWindowDate);
	                startWindow.setSecond(endWindowDate);
	            }
	            else
	            {
	                startWindow.setFirst(new TimeSpan(0, shiftEvent.getStart(), 0).addTo(scenario.start)); //GQ: this+shiftEventLength should not be after scenario.end
	                startWindow.setSecond(new TimeSpan(0, shiftEvent.getEnd(), 0).addTo(scenario.start)); //GQ: this+shiftEventLength should not be after scenario.end

	                if (new TimeSpan(0, shiftEventLength, 0).addTo(startWindow.getSecond()).after(scenario.end))
	                {
	                    //try to reduce the start window
	                    startWindow.setSecond(new TimeSpan(0, shiftEventLength, 0).subFrom(scenario.end));
	                    if (startWindow.getSecond().before(startWindow.getFirst())) {
	                        return; //GQ: It still don't fit. We must quit.
	                    }
	                }
	            }

	            if (shift!=null && //GQ: When optimizing only extension(s), shift will be null!
	                    shiftPattern != null && hmShiftPatternShiftAttributes.get(shiftAssignment.getShiftID()) != null
	                    && hmShiftPatternShiftAttributes.get(shiftAssignment.getShiftID()).getConsistentShiftGroupMap() != null
	                    && shiftPattern.isConsistentShiftActivities(shift.getID()))
	            {
	                if (hmShiftPatternShiftAttributes.get(shiftAssignment.getShiftID()).getConsistentShiftGroupMap().containsKey(GetPatternDayIndex(shiftAssignment)))
	                {
	                    for (ShiftAssignment sa : hmSchedule.values())
	                    {
	                        if (sa.getID().equals(shiftAssignment.getID())) {
	                            continue;
	                        }
	                        HashMap<Integer, Integer> shiftAssignmentGroupMap = hmShiftPatternShiftAttributes.get(shiftAssignment.getShiftID()).getConsistentShiftGroupMap();
	                        Integer shiftAssignmentDayIndex = GetPatternDayIndex(shiftAssignment);

	                        Integer saDayIndex = GetPatternDayIndex(sa);
	                        HashMap<Integer, Integer> saGroupMap = hmShiftPatternShiftAttributes.get(sa.getShiftID()).getConsistentShiftGroupMap();

	                        if (shiftAssignmentGroupMap.get(shiftAssignmentDayIndex).intValue() == saGroupMap.get(saDayIndex).intValue())
	                        {
	                            for (ShiftEventAssignment sea : (Collection<ShiftEventAssignment>) sa.getChildObjects(ShiftAssignmentFields.CHILD_SHIFT_EVENT))
	                            {
	                                if (sea.getShiftEventID().equals(shiftEvent.getID()))
	                                {
	                                    int startOff = new TimeSpan(sa.getStartTime(), sea.getStartTime()).TotalMinutes;
	                                    Date left = new TimeSpan(0, Math.max(0, startOff - shiftPattern.getConsistentStartsTolerance()), 0).addTo(scenario.start);
	                                    Date right = new TimeSpan(0, Math.min(new TimeSpan(scenario.start, scenario.end).TotalMinutes, startOff + shiftPattern.getConsistentStartsTolerance()), 0).addTo(scenario.end);
	                                    if (startWindow.getFirst().before(left)) {
	                                        startWindow.setFirst(left);
	                                    }
	                                    if (startWindow.getSecond().after(right)) {
	                                        startWindow.setSecond(right);
	                                    }
	                                }
	                            }
	                        }
	                    }
	                }
	            }
	            Pair<Integer, Integer> minMaxCount = new Pair<Integer, Integer>(1, 1);
	            if (shiftEvent.isFlexible())
	            {
	                minMaxCount.setFirst(shiftEvent.getMinCount());
	                minMaxCount.setSecond(shiftEvent.getMaxCount());
	            }

	            shiftEvent_ = new ShiftEvent(scenario, shiftEvent.getID().toString(), shiftEvent.getName(), new TimeSpan(0, shiftEvent.getDuration().getDurationInMinutes(), 0),
	                    minMaxCount, activities, startWindow, shiftEvent.isWorkType());
	            scenario.shiftEvents.put(shiftEvent_.id, shiftEvent_);
	        }
	        shiftEvents.add(shiftEvent_);
	    }
	    
		private int getLatestShiftEventStart(
				com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent shiftEvent) {
			return shiftEvent.getEnd()-shiftEvent.getDuration().getDurationInMinutes();
		}
		
	    private boolean isShiftEventStartBeforeShiftSinceShiftCrossDay(long shiftStart, long shiftEventStart) {
	    	if (shiftStart > shiftEventStart) {
	    		return true;
	    	}
	    	return false;
	    }

	    private int getDayStartOffSet(com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent shiftEvent) {
			if (shiftEvent.getOrganizationID() != null) {
				Organization org = this.hmOrganizations.get(shiftEvent.getOrganizationID());
				if (org != null) {
					return org.getDayBoundaryOffset();
				}
			}
			return 0;
		}

		private Activity GetActivity(com.bluepumpkin.ejb.bbm.activity.model.Activity activity) {
	        Activity activity_;
	        boolean isSkillRequired = false;
	        if (scenario.activities.containsKey(activity.getID().toString()))  {
	            activity_ = scenario.activities.get(activity.getID().toString());
	            Logger.Log("Activity " + activity_.name + " already loaded");
	        } else {
	            Logger.Log("Loading activity " + activity.getName());
	            Logger.Tab();
	            ArrayList<Queue> skilledQueues = new ArrayList<Queue>();
	            if (activity.isQueueHopping()) {
	                Logger.Log("Is queue hopping so add linked queues if skilled");
	                Logger.Tab();
	                Collection<ID> activityQueues = (Collection<ID>) hmActivityQueues.get(activity.getID());
	                for (ID idQueue : activityQueues) {
	                	isSkillRequired = true;
	                    com.bluepumpkin.ejb.bbm.workload.model.Queue queue = (com.bluepumpkin.ejb.bbm.workload.model.Queue) hmQueues.get(idQueue);
	                    AddQueueIfSkilled(skilledQueues, queue);
	                }
	                Logger.Untab();
	            } else {
	                Collection<ID> activityMedias = (Collection<ID>) hmActivityMedias.get(activity.getID());
	                if (activityMedias != null)  {
	                    Logger.Log("Is media hopping so add queues of " + activityMedias.size() + " linked medias if skilled");
	                    Logger.Tab();
	                    for (ID idMedia : activityMedias)  {
	                        Logger.Log("Add queues for media " + idMedia);
	                        Logger.Tab();
	                        for (com.bluepumpkin.ejb.bbm.workload.model.Queue queue : hmQueues.values())  {
	                            if (queue.getMediaID() != null) {
	                                if (queue.getMediaID().equals(idMedia))  {
	                                    isSkillRequired = true;
	                                	Logger.Log("Consider " + queue.getName());
	                                    Logger.Tab();
	                                    AddQueueIfSkilled(skilledQueues, queue);
	                                    Logger.Untab();
	                                }
	                            }
	                        }
	                        Logger.Untab();
	                    }
	                    Logger.Untab();
	                } else {
	                    Logger.Log("There are no linked medias");
	                }
	            }
	            activity_ = new Activity(scenario, activity.getID().toString(), activity.getName(), activity.isUsedInShift() ? 0 : 1, skilledQueues, isSkillRequired);
	            activity_.Log();
	            scenario.activities.put(activity_.id, activity_);
	            Logger.Untab();
	        }
	        return activity_;
	    }
