package com.verint.test.utils;

import com.gargoylesoftware.htmlunit.BrowserVersion;
import com.verint.test.utils.http.URLBuilderUtils;
import org.json.JSONObject;

import com.gargoylesoftware.htmlunit.WebClient;
import com.verint.test.constants.PathKeys;

import java.util.HashMap;
import java.util.Map;

public class LoginHelperUtils {

	//Stores the pre-authed web clients for each user so they can be referenced later
	//without needing to log in again
	private static Map<String, WebClient> webClientsByUser = new HashMap<>();

	//Cached authentication token for applets
	private static String authentificationToken = null;

	public static final String SUPER_USER_NAME = "wsuperuser";
	public static final String SUPER_USER_PASSWORD = "pumpkin1";


	/**
	 * Gets an authentication token for WFO services.
	 * @return
	 * { "AuthToken":{"token" : "tokenValue"} }
	 */
	private static JSONObject getAuthenticationTokenForUser(String userName, String password) throws Exception {
		String authServiceUrl = URLBuilderUtils.getCoreRestURL() + PathKeys.AUTH_SERVICE_URL;
		JSONObject user = new JSONObject();
		user.put("user", userName);
		user.put("password", password);

		JSONObject authentication = URLBuilderUtils.postRestRequest(authServiceUrl, null, user);
		JSONObject tokenOnly = authentication.getJSONObject("AuthToken");
		return tokenOnly;
	}

	/**
	 * Return Singleton WebClient with disabled JavaScript Engine.
	 * Only few test required JS. We could enable JS before this tests and disable after.
	 * By disabling JS We increase test execution by 50%;
	 *
	 * @return
	 * @throws Exception
	 */
	public static WebClient getPreAuthedWebClientForSuperuser() throws Exception {
		return getPreAuthedWebClient(SUPER_USER_NAME, SUPER_USER_PASSWORD);
	}

	/**
	 * Return a pre-authed, singleton WebClient for the given username/password.
	 * @param username
	 * @param password
	 * @return
	 * @throws Exception
	 */
	public static WebClient getPreAuthedWebClient(String username, String password) throws Exception {
		//we already have web client
		if (webClientsByUser.containsKey(username)) {
			return webClientsByUser.get(username);
		}

		WebClient webClient = buildPreAuthedWebClient(username, password, false);
		webClientsByUser.put(username, webClient);
		return webClient;
	}

	/**
	 * Build New Client instead existing Singleton WebClient
	 *
	 * @param jsEnabled
	 * @return
	 * @throws Exception
	 */
	public static WebClient getNewPreAuthedWebClientForSuperuser(boolean jsEnabled) throws Exception {
		return buildPreAuthedWebClient(SUPER_USER_NAME, SUPER_USER_PASSWORD, jsEnabled);
	}

	public static WebClient getNewPreAuthedWebClient(String username, String password, boolean jsEnabled) throws Exception {
		return buildPreAuthedWebClient(username, password, jsEnabled);
	}


	/**
	 * Build Authentifed web client.
	 *
	 * @return
	 * @throws Exception
	 */
	private static WebClient buildPreAuthedWebClient(String username, String password, boolean jsEnabled) throws Exception {
		WebClient webClient;
		Host proxyHost = getProxyHost();
		//final WebClient webClient;
		if (proxyHost != null) {
			webClient = new WebClient(BrowserVersion.INTERNET_EXPLORER, proxyHost.getHost(), proxyHost.getPort());
		} else {
			webClient = new WebClient(BrowserVersion.INTERNET_EXPLORER);
		}
		webClient.getOptions().setThrowExceptionOnScriptError(false);
		webClient.getOptions().setJavaScriptEnabled(jsEnabled);
		webClient.getOptions().setHistorySizeLimit(0);
		addAuthTokenToRequestHeader(webClient, username, password);

		return webClient;
	}

	private static void addAuthTokenToRequestHeader(WebClient wc, String userName, String password) throws Exception {
		wc.addRequestHeader("Impact360AuthToken", getAuthenticationTokenStringForUser(userName, password));
	}

	/**
	 * Gets an authentication token string for user/password
	 * @return String auth token
	 * { "AuthToken":{"token" : "tokenValue"} }
	 */
	private static String getAuthenticationTokenStringForUser(String userName, String password) throws Exception {
		return getAuthenticationTokenForUser(userName, password).getString("token");
	}

	public static Host getProxyHost() {
		String proxyHost = System.getProperty("htmlUnitProxyHost");
		if(proxyHost != null){
			String[] hostAndPort = proxyHost.split(":");
			return new Host(hostAndPort[0], Integer.valueOf(hostAndPort[1]));
		}
		return null;
	}

	public static class Host {
		private String host;
		private int port;

		public Host(String host, int port) {
			this.host = host;
			this.port = port;
		}

		public String getHost() {
			return host;
		}

		public int getPort() {
			return port;
		}
	}

	/**
	 * Makes authentication call and returns AuthenticationToken for superuser
	 *
	 * @return
	 * @throws Exception
	 */
	public static String getAuthenticationTokenForSuperuser() throws Exception {
		if(authentificationToken == null) {
			authentificationToken = getAuthenticationTokenStringForUser(SUPER_USER_NAME, SUPER_USER_PASSWORD);
		}
		return authentificationToken;
	}
}
