package com.bluepumpkin.web.am.adherence.exceptioncategory;

import java.util.*;

import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.common.logging.Category;
import com.witness.web.uif.keys.*;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.pagecomponent.form.FormTwoColumnPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarUtil;
import com.witness.web.uif.validator.*;
import com.witness.web.uif.util.js.*;
import com.witness.web.uif.util.HtmlUtil;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagemodel.PageModel;
import com.bluepumpkin.web.bbm.Log;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.ejb.am.rtaaexceptioncategory.model.*;
import com.bluepumpkin.web.bbm.organization.OrgWorkpaneFormPageModel;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;

/**
 * Title:        ExceptionCategoryEditPageModel
 * Description:  Display the Edit page for Exception Categories
 * Copyright:    Copyright (c) 2006
 * Company:      Witness Systems, inc
 * @author       Swati Tewari
 * @version 1.0
 */
public class ExceptionCategoryEditPageModel extends OrgWorkpaneFormPageModel {
	private static final Category log = Log.initCategory(ExceptionCategoryEditPageModel.class.getName());
	private RTAAExceptionCategory m_exceptionCategory;
	private ResourceBundle m_uifBundle;

	// Page Components
	private FormTwoColumnPC m_formPC;

	public static final String MODEL_KEY   = "Exceptioncategory_Edit_Page_Model";
	private static final String EXCEPTIONCATEGORY_NAME_FN = "exceptionCategoryName";
	private static final String EXCEPTIONCATEGORY_DESCRIPTION_FN = "exceptionCategoryDescription";

	/**
	 * Constructor
	 */
	public ExceptionCategoryEditPageModel(RequestContext context) {
		super(context);

		// Initialize Form PC
		m_formPC = new FormTwoColumnPC(context);
		addChildComponent(m_formPC);

		// Component specific required JavaScript Files
		setJSMediator(JavaScriptFileID.MEDIATOR_WORKPANE_GENERIC);

		// Initialize Name
		m_exceptionCategory = new RTAAExceptionCategory();
		
		m_uifBundle = m_localizer.getBundle(UIFWebBundleKey.BUNDLE_NAME);
	}

	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, ExceptionCategoryEditPageModel.class);
	}

	/**
	 * Create Form Model
	 */
	protected void createFormModel() {
		try {
			if (getExceptionCategoryID()!=null && !hasError()) {
				ID orgID = getSelectedIDObject();
				m_exceptionCategory = ExceptionCategoryModelHandler.getExceptionCategory(m_context, getExceptionCategoryID());
			}
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		}
	}


	/**
	 * Initialize Form Display
	 */
	protected void initForm() {
		setForm(m_formPC);
		m_formPC.setTitle(getFormTitle());

		try {
			m_formPC.addRow(getOrganizationLabel(), getOrganizationName());
	
			String nameUI = HtmlUtil.makeTextInput(EXCEPTIONCATEGORY_NAME_FN, 50, 50,
					getExceptionCategoryName(), JSUtil.ON_CHANGE);
			String descriptionUI = HtmlUtil.makeTextInput(EXCEPTIONCATEGORY_DESCRIPTION_FN, 50, 50,
					getExceptionCategoryDescription(), JSUtil.ON_CHANGE);
			m_formPC.addRow(getExceptionCategoryNameLabel(), nameUI);
			m_formPC.addRow(getExceptionDescriptionNameLabel(), descriptionUI);
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
			m_initFailure = true;
		}
	}

	/**
	 * Initialize Toolbar
	 */
	protected void initToolbar() {
		ToolbarUtil.addValidateSaveBtn(m_toolbar, true);
		ToolbarUtil.addConfirmDoneBtn(m_toolbar, true);
		ToolbarUtil.addResetBtn(m_toolbar, true);
	}

	/**
	 * Set the validators to the form fields
	 */
	protected void initFieldValidators() {
		// Add Field Validators
		addValidator(new IsEnteredFieldValidator(this, EXCEPTIONCATEGORY_NAME_FN,
				BbmWebBundleKey.ORG_HOLIDAY_NAME_LABEL, BbmWebBundleKey.BUNDLE_NAME));
	}

	/**
	 * Set exception category Name
	 */
	public void setExceptionCategoryName(String name) {
		m_exceptionCategory.setName(name);
	}

	/**
	 * Return exception category Name
	 */
	public String getExceptionCategoryName() {
		return m_exceptionCategory.getName();
	}
	
	/**
	 * Set exception category description
	 */
	public void setExceptionCategoryDescription(String name) {
		m_exceptionCategory.setDescription(name);
	}

	/**
	 * Return exception category description
	 */
	public String getExceptionCategoryDescription() {
		return m_exceptionCategory.getDescription();
	}

	/**
	 * Return exception category ID
	 */
	public ID getExceptionCategoryID() {
		if (m_performActionOnIDs==null) return null;
		return m_performActionOnIDs[getEditIndex()];
	}
	
	/**
	 * Return the Associated exception category Object
	 */
	public RTAAExceptionCategory getExceptionCategory() {
		m_exceptionCategory.setID(getExceptionCategoryID());

		if (hasSelectedID()) {
			ID orgID = getSelectedIDObject();
			m_exceptionCategory.setOrganizationID(orgID);
		}

		return m_exceptionCategory;
	}

	/**
	 * Return it is new
	 */
	public boolean isNew() {
		return m_exceptionCategory.getID()==null;
	}


	/**
	 * Return Localized Organization Label
	 */
	public String getOrganizationLabel() {
		return i18n(m_bbmBundle, BbmWebBundleKey.ORGANIZATION_NAME);
	}

	/**
	 * Return Localized exception category Name Label
	 */
	public String getExceptionCategoryNameLabel() {
		return i18n(m_uifBundle, UIFWebBundleKey.NAME);
	}
	
	/**
	 * Return Localized exception category description Label
	 */
	public String getExceptionDescriptionNameLabel() {
		return i18n(m_uifBundle, UIFWebBundleKey.DESCRIPTION);
	}
	
	public String getCategoryOrganizationName() throws Exception{
		return OrganizationModelHandler.getOrganizationName(m_context, m_exceptionCategory.getOrganizationID());
	}

	/**
	 * Return the form title
	 *
	 * @return form title
	 */
	public String getFormTitle() {
		if (isNew()) {
			return "Create New Exception Category";//i18n(m_bbmBundle, BbmWebBundleKey.ORG_HOLIDAY_CREATING);
		} else {
			return HtmlUtil.NBSP;
		}
	}

	/**
	 * Get entity name. Used for title.
	 */
	protected String getTitleEntityType() {
		return "Exception Category";//i18n(m_bbmBundle, BbmWebBundleKey.ORG_HOLIDAY_CONTENT_TITLE);
	}

	/**
	 * Returns an entity name. Used for title.
	 */
	protected String getEntityName() {
		return getExceptionCategoryName();
	}

	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return "org_exceptioncategory";
	}
}
