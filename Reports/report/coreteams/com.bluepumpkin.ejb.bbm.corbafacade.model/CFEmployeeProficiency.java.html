package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;
import java.util.Date;

import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeProficiency;

public class CFEmployeeProficiency implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public int proficiency;
	public Date startDate;
	public Date endDate;
	
	public static CFEmployeeProficiency convert(EmployeeProficiency from) {
		CFEmployeeProficiency to = new CFEmployeeProficiency();
		
		to.proficiency = from.getProficiency();
		to.startDate = from.getStartTime();
		to.endDate = from.getEndTime();
		
		return to;
	}
	
	public static CFEmployeeProficiency[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeProficiency[] to = new CFEmployeeProficiency[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeeProficiency.convert((EmployeeProficiency)i.next());
		return to;
	}
}
