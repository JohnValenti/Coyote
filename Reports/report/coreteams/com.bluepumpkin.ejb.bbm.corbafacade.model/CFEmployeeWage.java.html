package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeWage;

public class CFEmployeeWage implements Serializable {
    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public double wage;
	public Date startDate;
	public Date endDate;
	
	public static CFEmployeeWage convert(EmployeeWage from) {
		CFEmployeeWage to = new CFEmployeeWage();
		
		to.wage = from.getWage();
		to.startDate = from.getStartTime();
		to.endDate = from.getEndTime();
		
		return to;
	}
	
	public static CFEmployeeWage[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeWage[] to = new CFEmployeeWage[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++)
			to[ix] = CFEmployeeWage.convert((EmployeeWage)i.next());
		return to;
	}
}
