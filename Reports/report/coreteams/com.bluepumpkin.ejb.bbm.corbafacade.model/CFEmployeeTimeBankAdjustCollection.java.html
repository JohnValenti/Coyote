package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;

public class CFEmployeeTimeBankAdjustCollection implements Serializable{
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public CFEmployeeTimeBankAdjust[] tbAdjust = null;
	
	private static CFEmployeeTimeBankAdjustCollection convertList(Collection from) {
		CFEmployeeTimeBankAdjustCollection to = new CFEmployeeTimeBankAdjustCollection();
		to.tbAdjust = CFEmployeeTimeBankAdjust.convert(from);
		return to;
	}
	
	//from is the list of the list
	public static CFEmployeeTimeBankAdjustCollection[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFEmployeeTimeBankAdjustCollection[] to = new CFEmployeeTimeBankAdjustCollection[from.size()];
		
		Collection tbaList = null;
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++) {
			tbaList = (Collection)i.next();
			if (tbaList != null && !tbaList.isEmpty()) {
				to[ix] = convertList(tbaList);
			}
		}
		return to;
	}
}
