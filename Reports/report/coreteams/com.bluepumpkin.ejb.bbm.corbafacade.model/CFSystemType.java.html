package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Collection;
import java.util.Iterator;

import com.bluepumpkin.ejb.bbm.dao.MapUtil.SystemType;


public class CFSystemType implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public  String m_systemTypeName;

	
	public static CFSystemType convert(SystemType systemType) {
		if (systemType == null)
			return null;
		
		CFSystemType cfSystemType = new CFSystemType();
		cfSystemType.m_systemTypeName = systemType.name();
				
		return cfSystemType;
	}
	
	public static SystemType convert(CFSystemType cfAct) {
		if (cfAct == null)
			return null;		
		SystemType systemType = SystemType.valueOf(cfAct.m_systemTypeName);
		return systemType;
	}
	
	public static CFSystemType[] convert(Collection from) {
		if (from == null)
			return null;
		
		CFSystemType[] to = new CFSystemType[from.size()];
		int ix = 0;
		for (Iterator i = from.iterator(); i.hasNext(); ix++) {
			to[ix] = CFSystemType.convert((SystemType)i.next());
		}
		
		return to;
	}
}
