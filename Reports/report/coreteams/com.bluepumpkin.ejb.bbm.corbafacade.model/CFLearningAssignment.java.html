package com.bluepumpkin.ejb.bbm.corbafacade.model;

import java.io.Serializable;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;

public class CFLearningAssignment implements Serializable
{
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	public ID id;
  /*
   *	eQNow: EXT_EMP_ID 
   *
   */
  public ID workResourceID; 		

  /*
   * eQNow: LESSON_ID
   *
   */
  public ID lessonID; 		

  /*
   * eQNow: LESSON_NAME
   *
   */
  public String lessonName;

  /*
   * eQNow: ACTIVITY
   *
   */
  public int status;	

  /*
   * eQNow: STATUS_DATE + STATUS_TIME
   *
   */
  public Date statusDate;		

  /*
   * eQNow: AVAILABLE_DATE
   *
   */
  public Date available;		

  /*
   * eQNow: DUE_DATE
   *
   */
  public Date due;

  /*
   * eQNow: PRIORITY
   *
   */
  public int priority;

  /*
   * eQNow: LESSON_DURATION
   *
   */
  public int length;

  /*
   * 
   *
   */
  public int maximumSessionLength;

  /*
   * 
   */ 
  public int timeToScheduleThisWeek;
  
  /*
   * this value object is used to hold both elearning and coaching assignment
   * the following flag can be used to differenciate the two objects.
   */
  public boolean isCoaching = false;

 
  private int changeCounter = 0;
  
  
  public int getchangeCounter()
  {
  	return this.changeCounter;
  }
  
  public void  setchangeCounter( int newValue )
  {
  	this.changeCounter = newValue;
  }
  
  public boolean isCoaching() {
	  return this.isCoaching;
  }
  
  public void setCoaching(boolean bValue) {
	  this.isCoaching = bValue;
  }
 
  public String toString()
  {
  	StringBuffer buf = new StringBuffer();
  	buf.append("[ ");
  	buf.append(" available=");
  	buf.append(this.available);
  	buf.append(" changeCounter=");
  	buf.append(this.changeCounter);
  	buf.append(" due=");
  	buf.append(this.due);
  	buf.append(" length=");
  	buf.append(this.length);
  	buf.append(" lessonID=");
  	buf.append(this.lessonID);
  	buf.append(" lessonName=");
  	buf.append(this.lessonName);
  	buf.append(" maximumSessionLength=");
  	buf.append(this.maximumSessionLength);
  	buf.append(" priority=");
  	buf.append(this.priority);
  	buf.append(" status=");
  	buf.append(this.status);
  	buf.append(" statusDate=");
  	buf.append(this.statusDate);
  	buf.append(" timeToScheduleThisWeek=");
  	buf.append(this.timeToScheduleThisWeek);
  	buf.append(" workResourceID=");
  	buf.append(this.workResourceID);
  	
  	
  	buf.append(" ]");
  	
  	return buf.toString();
  }
  
}
