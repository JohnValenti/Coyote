    /**
     * Get the display names for the primary and secondary actual lines. 
     * actualsNamepair.getFirst()
     * actualsNamepair.getSecond()
     * 
     * @param context RequestContext
     * @return A pair of strings.
     */ 
    public static Pair getActualDisplayNames(RequestContext context){
        Pair namePair = new Pair();
        String actualPrimaryStr = context.getLocalizer().i18n(UIFWebBundleKey.BUNDLE_NAME,UIFWebBundleKey.ACTUAL);
        String actualSecondaryStr = null;//
        try{
            LicenseManager licManager=CoreManagerFactory.getLicenseManager();
            if (licManager.isLicensed(LicenseKeys.APP_DESKTOPMONITORING)
                    || licManager.isLicensed(LicenseKeys.APP_ACDMONITORING)){
                actualPrimaryStr = BbmManagerFactory.getDBConfigManager().getValue(ConfigKey.DESKTOP_MONITORING_ACTUALS_PRIMARY);

                if (StringUtil.isEmpty(actualPrimaryStr))
                    actualPrimaryStr = context.getLocalizer().i18n(AmWebBundleKey.BUNDLE_NAME,AmWebBundleKey.ADHERENCE_TIMELINE_PRIMARY);
            }

            //add secondary string if DM license is available
            if (licManager.isLicensed(LicenseKeys.APP_DESKTOPMONITORING)){
                actualSecondaryStr = BbmManagerFactory.getDBConfigManager().getValue(ConfigKey.DESKTOP_MONITORING_ACTUALS_SECONDARY);

                if (StringUtil.isEmpty(actualSecondaryStr))
                    actualSecondaryStr = context.getLocalizer().i18n(AmWebBundleKey.BUNDLE_NAME,AmWebBundleKey.ADHERENCE_TIMELINE_SECONDARY);
            }
        } catch (Exception ex) {}
        
        namePair.setFirst(actualPrimaryStr);
        namePair.setSecond(actualSecondaryStr);

        return namePair;
    }
    
    public static HashMap getActivityMap(RequestContext context, HashMap trackingEventMap) throws Exception{
        Collection mapValues = trackingEventMap.values();
        HashMap activityMap = null;
        ArrayList activityList = new ArrayList();
        for (Iterator it = mapValues.iterator(); it.hasNext();){
            ArrayList eventList = (ArrayList)it.next();
            for (int i = 0; i < eventList.size(); i++){
                TimeTrackingEvent event = (TimeTrackingEvent)eventList.get(i);
                if (!activityList.contains(event.getActivityID()))
                    activityList.add(event.getActivityID());
            }
        }
        
        if (activityList !=  null && activityList.size() > 0){
            activityMap = new HashMap(activityList.size());
            Collection activities = ActivityModelHandler.getActivitiesByIDs(context, activityList);
            
            for (Iterator it = activities.iterator(); it.hasNext();){
                Activity activity = (Activity) it.next();
                
                activityMap.put(activity.getID(), activity);
            }
        }
        
        return activityMap;
    }
    
    public static boolean showUnavailable(RequestContext context) {
        UserProfile userProfile = UserProfileManager.getUserProfile(context);
        String show = userProfile.loadProperty(UserPreferenceKeys.USER_SHOW_UNAVAILABLE,"false");

        return !("false".equalsIgnoreCase(show));
    }

	static BalanceManager getQMManager() throws Exception {
		return new UltraQualityProvider();
	}
