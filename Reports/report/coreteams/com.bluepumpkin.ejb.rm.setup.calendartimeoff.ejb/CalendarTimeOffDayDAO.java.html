/**
 * DAO class for CalendarTimeOffDay
 */
public class CalendarTimeOffDayDAO extends DAOBase {
	private static FieldInfo m_fieldInfo = new CalendarTimeOffDayFieldInfo();

	private static TimeZone s_gmt = TimeZone.getTimeZone("GMT");

	@Override
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public CalendarTimeOffDayDAO() {
		super();
	}

	public CalendarTimeOffDayDAO(Jdmo dmo) {
		super(dmo);
	}

	@Override
	protected ValueObjectBase createValueObject() {
		return (new CalendarTimeOffDay());
	}

	/**
	 * Find the available time off hours for a given TO Pool and month range.
	 * If there are records defined at the given TO Pool then those will be used.
	 * If not this method will check the parent org and so on.
	 * All records for a month will be defined at a single TO Pool!
	 * <p/>
	 * This method returns an array with an element for each day of the month.
	 * Days without timeoff allocation are set to null.  Days with timeoff allocation
	 * are set to the CalendarTimeOffDay object.
	 */
	public CalendarTimeOffDay[] findWithinMonth(ID pTOPoolID, MonthRange monthRange) throws Exception {
		StringBuffer strQuery = new StringBuffer();
		strQuery.append(" A.TOPOOLID = ")
		        .append(JdmoUtil.asSqlLiteral(pTOPoolID))
		        .append(" AND A.DAYDATE >= '")
		        .append(JdmoUtil.formatDBString(monthRange.startDate))
		        .append("' AND A.DAYDATE < '")
		        .append(JdmoUtil.formatDBString(monthRange.endDate))
		        .append("' ORDER BY A.DAYDATE ASC");

		Collection items = null;
		items = getObjects(strQuery.toString());
		Calendar cal = new GregorianCalendar();
		CalendarTimeOffDay[] arr = new CalendarTimeOffDay[31];
		for (Iterator itr = items.iterator(); itr.hasNext();) {
			// obtain the next timeOffCalendarDay object
			CalendarTimeOffDay item = (CalendarTimeOffDay) itr.next();
			// convert the time in the timeOffCalendarDay to day of month
			cal.setTime(item.getTimeOffDate());
			int dayOfMonth = cal.get(Calendar.DAY_OF_MONTH) - cal.getMinimum(Calendar.DAY_OF_MONTH);
			// set the element at index 'day of month' to the timeOffCalendarDay object.
			arr[dayOfMonth] = item;
		}
		return arr;
	}

	/**
	 * Get an Collection of CalendarTimeOffDay objects in the given local date range,
	 * inclusively.
	 */
	public Collection findInRange(ID pTOPoolID, LocalDate start, LocalDate end) throws Exception {
		// Make sure that both start and end or not null and that end is after start
		if (start == null || end == null || !end.after(start)) {
			throw new IllegalArgumentException("invalid start or end date: " + start + " - " + end);
		}
		StringBuffer strQuery = new StringBuffer();
		strQuery.append(" A.TOPOOLID = ")
		        .append(JdmoUtil.asSqlLiteral(pTOPoolID))  //Fix QC196696
		        .append(" AND A.DAYDATE >= '")
		        .append(JdmoUtil.formatDBString(start.getTime(s_gmt)))
		        .append("' AND A.DAYDATE <= '")
		        .append(JdmoUtil.formatDBString(end.getTime(s_gmt)))
		        .append("' ORDER BY A.DAYDATE ASC");

		return getObjects(strQuery.toString());
	}


	/**
	 * Save a month's worth of time off days for a group of TO Pools.
	 * The array must reference a complete month's worth of data.
	 *
	 * @param pTOPoolID      - the  TO Pool id of the change
	 * @param monthRange     - the start and end time for the month
	 * @param newTimeOffDays - the time off hours per day to save.
	 */
	public void saveTimeOffDays(ID pTOPoolID, MonthRange monthRange, CalendarTimeOffDay newTimeOffDays[]) throws Exception {
		ArrayList updateList = new ArrayList(31);
		ArrayList createList = new ArrayList(31);
		HashMap oldTimeOffDaysMap = new HashMap(31);
		CalendarTimeOffDay oldTimeOffDays[] = findWithinMonth(pTOPoolID, monthRange);
		int l = oldTimeOffDays != null ? oldTimeOffDays.length : 0;
		for (int i = 0; i < l; i++) {
			CalendarTimeOffDay oldItem = oldTimeOffDays[i];
			if (oldItem != null && oldItem.getTimeOffLocalDate() != null) {
				oldTimeOffDaysMap.put(oldItem.getTimeOffLocalDate(), oldItem);
			}
		}
		l = newTimeOffDays != null ? newTimeOffDays.length : 0;
		for (int i = 0; i < l; i++) {
			CalendarTimeOffDay newTimeOffDay = newTimeOffDays[i];
			if (newTimeOffDay != null && newTimeOffDay.getTimeOffLocalDate() != null) {
				CalendarTimeOffDay oldCalendarTimeOffDay = (CalendarTimeOffDay) oldTimeOffDaysMap.get(newTimeOffDay.getTimeOffLocalDate());
				if (oldCalendarTimeOffDay == null) {//item exists only in new so create it
					newTimeOffDay.setTOPoolID(pTOPoolID);
					newTimeOffDay.setID(null);
					createList.add(newTimeOffDay);
				} else {
					//Item exists in new and old so the that day needs to be updated only
					//********** Be sure to remove it from the OLD List because we will use the old list that remains
					// after the newTimeOffDays iteration for deletion purposes.
					oldTimeOffDaysMap.remove(newTimeOffDay.getTimeOffLocalDate());
					oldCalendarTimeOffDay.setAllocatedHours(newTimeOffDay.getAllocatedHours());
					updateList.add(oldCalendarTimeOffDay);
				}
			}
		}
		if (!updateList.isEmpty()) {
			updateObjects(updateList);
		}
		if (!createList.isEmpty()) {
			createObjects(createList);
		}
		if (oldTimeOffDaysMap != null && !oldTimeOffDaysMap.isEmpty()) {
			deleteObjects(ValueObjectUtil.getIDObjectMap(oldTimeOffDaysMap.values()).keySet());
		}
	}

	public void deleteCalendarTimeOffDays(ID toPoolId, Date startDate, Date endDate)
			throws JdmoException, BbmFinderException
	{
		StringBuffer queryBuffer = new StringBuffer();
		queryBuffer.append("delete ");
		queryBuffer.append(" from ");
		queryBuffer.append(m_fieldInfo.getTableName());
		queryBuffer.append(" where ");
		queryBuffer.append("TOPOOLID");
		queryBuffer.append(" = ");
		queryBuffer.append(toPoolId);
		queryBuffer.append(" and ");
		queryBuffer.append("DAYDATE");
		queryBuffer.append(" < ");
		JdmoUtil.asSqlLiteral(endDate, queryBuffer);
		queryBuffer.append(" and not ");
		queryBuffer.append("DAYDATE");
		queryBuffer.append(" < ");
		JdmoUtil.asSqlLiteral(startDate, queryBuffer);
		m_dmo.execute(queryBuffer.toString());

	}

}
