package com.bluepumpkin.ejb.bbm.workrules.model;

/**

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeOfDay;

public class WorkRuleParameter implements java.io.Serializable, Comparable<WorkRuleParameter> {

	private static final long serialVersionUID = -8195704677440010853L;

	public static final int NOVALUE = -1;

	public static final int INT = 0;
	public static final int DOUBLE = 1;
	public static final int TIME = 2;
	public static final int CLAUSE = 3;
	public static final int ACTIVITY = 4;
	public static final int ACTIVITYCATEGORY = 5;
	public static final int WEEKDAY = 6;
	public static final int ELEMENT = 7;
	public static final int PARTOFDAY = 8;
	public static final int SHIFT = 9;

	public static final int PARTOFDAY_ANYTIME = 0;
	public static final int PARTOFDAY_MORNING = 1;
	public static final int PARTOFDAY_AFTERNOON = 2;
	public static final int PARTOFDAY_EVENING = 3;
	public static final int PARTOFDAY_NIGHT = 4;

	private int m_nOrder = NOVALUE;
	private int m_eType = NOVALUE;
	private int m_nValue = NOVALUE;
	private double m_dValue = NOVALUE;
	private TimeOfDay m_tValue = null;
	private ID m_idValue = null;

	public WorkRuleParameter() {
	}

	public WorkRuleParameter(int nOrder, int eType, int nValue) throws WorkRuleException {
		m_nOrder = nOrder;
		m_eType = eType;
		if ((eType != INT) && (eType != CLAUSE) && (eType != WEEKDAY) && (eType != ELEMENT) && (eType != PARTOFDAY)) {
			throw new WorkRuleException("illegal type");
		}
		m_nValue = nValue;
	}

	public WorkRuleParameter(int nOrder, int eType, double dValue) throws WorkRuleException {
		m_nOrder = nOrder;
		m_eType = eType;
		if (eType != DOUBLE) {
			throw new WorkRuleException("illegal type");
		}
		m_dValue = dValue;
	}

	public WorkRuleParameter(int nOrder, int eType, TimeOfDay tValue) throws WorkRuleException {
		m_nOrder = nOrder;
		m_eType = eType;
		if (eType != TIME) {
			throw new WorkRuleException("illegal type");
		}
		m_tValue = tValue;
	}

	public WorkRuleParameter(int nOrder, int eType, ID idValue) throws WorkRuleException {
		m_nOrder = nOrder;
		m_eType = eType;
		if (eType != ACTIVITY && eType != ACTIVITYCATEGORY && eType != WEEKDAY && eType != ELEMENT && eType != SHIFT
				&& eType != PARTOFDAY) {
			throw new WorkRuleException("illegal type");
		}
		if (eType == WEEKDAY || eType == ELEMENT || eType == PARTOFDAY) {
			m_nValue = Integer.parseInt(idValue.toString()); // HACK this gets the Java.Calendar Weekday int from the Selector ID
		} else {
			m_idValue = idValue;
		}
	}

	public int getType() {
		return m_eType;
	}

	public int getOrder() {
		return m_nOrder;
	}

	public int getIntValue() throws WorkRuleException {
		if (m_eType != INT && m_eType != ELEMENT && m_eType != PARTOFDAY) {
			throw new WorkRuleException("illegal type");
		}
		return m_nValue;
	}

	public int getWeekDayValue() throws WorkRuleException {
		if (m_eType != WEEKDAY) {
			throw new WorkRuleException("illegal type");
		}
		return m_nValue;
	}

	public int getClauseValue() throws WorkRuleException {
		if (m_eType != CLAUSE) {
			throw new WorkRuleException("illegal type");
		}
		return m_nValue;
	}

	public double getDoubleValue() throws WorkRuleException {
		if (m_eType != DOUBLE) {
			throw new WorkRuleException("illegal type");
		}
		return m_dValue;
	}

	public TimeOfDay getTimeValue() throws WorkRuleException {
		if (m_eType != TIME) {
			throw new WorkRuleException("illegal type");
		}
		return m_tValue;
	}

	public ID getActivityIDValue() throws WorkRuleException {
		if (m_eType != ACTIVITY) {
			throw new WorkRuleException("illegal type");
		}
		return m_idValue;
	}

	public ID getShiftIDValue() throws WorkRuleException {
		if (m_eType != SHIFT) {
			throw new WorkRuleException("illegal type");
		}
		return m_idValue;
	}

	public void setActivityIDValue(ID idActivity) throws WorkRuleException {
		if (m_eType != ACTIVITY) {
			throw new WorkRuleException("illegal type");
		}
		m_idValue = idActivity;
	}

	public ID getActivityCategoryIDValue() throws WorkRuleException {
		if (m_eType != ACTIVITYCATEGORY) {
			throw new WorkRuleException("illegal type");
		}
		return m_idValue;
	}

	public String getValueString() {
		if (m_eType == INT) {
			return "" + this.m_nValue;
		}
		if (m_eType == DOUBLE) {
			return "" + this.m_dValue;
		}
		if (m_eType == TIME) {
			return m_tValue.toString();
		}
		if (m_eType == CLAUSE) {
			return "" + this.m_nValue;
		}
		if (m_eType == ACTIVITY) {
			return this.m_idValue.toString();
		}
		if (m_eType == ACTIVITYCATEGORY) {
			return this.m_idValue.toString();
		}
		if (m_eType == WEEKDAY) {
			return "" + this.m_nValue;
		}
		if (m_eType == ELEMENT) {
			return "" + this.m_nValue;
		}
		if (m_eType == PARTOFDAY) {
			return "" + this.m_nValue;
		}
		if (m_eType == SHIFT) {
			return this.m_idValue.toString();
		}
		return "";
	}

	@Override
	public boolean equals(Object o) {
		if (this == o) {
			return true;
		}
		
		if (!(o instanceof WorkRuleParameter)) {
			return false;
		}

		WorkRuleParameter n = (WorkRuleParameter) o;
		if (m_nOrder == n.getOrder()) {
			return true;
		} else {
			return false;
		}
	}
	
	@Override
	public int hashCode() {
		return 31 * 17 + m_nOrder;
	}

	@Override
	public int compareTo(WorkRuleParameter o) {
		WorkRuleParameter n = o;
		if (m_nOrder < n.getOrder()) {
			return -1;
		} else if (m_nOrder > n.getOrder()) {
			return 1;
		} else {
			return 0;
		}
	}
}
