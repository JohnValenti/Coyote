package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.Collection;
import java.util.Date;
import java.util.HashSet;


import com.bluepumpkin.common.datatypes.Duration;
import com.bluepumpkin.common.datatypes.ID;

import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectRich;


public class ProjectRules extends ValueObjectRich {
	private static final long serialVersionUID = 1L;
	
	
	public static String PROJECT_START_RELATIONSHIP_KEY = "PROJECTSTART|PROJECTID";
	
	
	
	//CALENDAREVENTPROJECT
	//CALENDAREVENTPROJECTPUB
	//PROJECTSTART

	private static FieldInfo m_fieldInfo = new ProjectRulesFieldInfo();
	
	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}
	
	public ID getDEID(){
		return this.getFieldValueID(ProjectRulesFieldInfo.PROJECT_ID);
	}
	
	public void setDEID(ID deID){
		this.setFieldValue(ProjectRulesFieldInfo.PROJECT_ID, deID);
	}
	
	public ID getCampaignID(){	
		
		return this.getFieldValueID(ProjectRulesFieldInfo.PROJECT_CAMPAIGNID);
	}
	public void setCampaignID(ID campaignID){ 
		this.setFieldValue(ProjectRulesFieldInfo.PROJECT_CAMPAIGNID, campaignID);
	}
	public ID getActivityID(){ 
		return this.getFieldValueID(ProjectRulesFieldInfo.PROJECT_ACTIVITYID);
		
	}
	public void setActivityID(ID activityID){
		this.setFieldValue(ProjectRulesFieldInfo.PROJECT_ACTIVITYID, activityID);
		
	}
	public String getName() {
		return this.getFormattedFieldValue(ProjectRulesFieldInfo.PROJECT_NAME);		
	}
	public void setName(String name){
		this.setFieldValue(ProjectRulesFieldInfo.PROJECT_NAME, name);
	}
	public String getDescription(){ 	
		return this.getFormattedFieldValue(ProjectRulesFieldInfo.PROJECT_DESCRIPTION);
	}
	public void setDescription(String description){
		
		this.setFieldValue(ProjectRulesFieldInfo.PROJECT_DESCRIPTION, description);
	}
	
	public ID getOrganizationID() {
		return this.getFieldValueID(ProjectRulesFieldInfo.PROJECT_ORGANIZATIONID);
	}
	
	public void setOrganizationID(ID organizationID) { 
		this.setFieldValue(ProjectRulesFieldInfo.PROJECT_ORGANIZATIONID, organizationID);
	}
	
	public Duration getDuration() {
		return Duration.fromMinutes(this.getFieldValueInt(ProjectRulesFieldInfo.PROJECT_LENGTH));
		//return this.getFieldValueInt(ProjectRulesFieldInfo.PROJECT_LENGTH);
	}
	
	public void setDuration(int duration) {
		this.setFieldValue(ProjectRulesFieldInfo.PROJECT_LENGTH, duration);
	}
	
	public int getMaxSimultaneous() {
		return this.getFieldValueInt(ProjectRulesFieldInfo.PROJECT_MAXSIMULTANEOUS);
	}
	
	public void setMaxSimultaneous(int maxSimultaneous) {
		setFieldValue(ProjectRulesFieldInfo.PROJECT_MAXSIMULTANEOUS, maxSimultaneous);
	}
	
	public int getDailyMaxPerEmployee() {
		return this.getFieldValueInt(ProjectRulesFieldInfo.PROJECT_DAILYMAXPEREMPLOYEE);
	}
	
	public void setDailyMaxPerEmployee(int dailyMax) {
		setFieldValue(ProjectRulesFieldInfo.PROJECT_DAILYMAXPEREMPLOYEE, dailyMax);
	}
	
	public int getWeeklyMaxPerEmployee() {
		return this.getFieldValueInt(ProjectRulesFieldInfo.PROJECT_WEEKLYMAXPEREMPLOYEE);
	}
	
	public void setWeeklyMaxPerEmployee(int maxEmployee) {
		setFieldValue(ProjectRulesFieldInfo.PROJECT_WEEKLYMAXPEREMPLOYEE, maxEmployee);
	}
	public boolean isDeleted() {
		Date deleteOnDate = this.getFieldValueDate(ProjectRulesFieldInfo.PROJECT_DELETEONDATE);
		if (deleteOnDate != null && (deleteOnDate.getTime() < System.currentTimeMillis()))
			return true;
		return false;
	}
	public Date getCreatedDate() {
		return getFieldValueDate(ProjectRulesFieldInfo.PROJECT_CREATED);
	}
	public void setCreatedDate(Date createdDate) {
		setFieldValue(ProjectRulesFieldInfo.PROJECT_CREATED, createdDate);
	}
	
	public String getCreatedBy() {
		return getFormattedFieldValue(ProjectRulesFieldInfo.PROJECT_CREATEDBY);
	}
	public void setCreatedBy(String createdBy) {
		setFieldValue(ProjectRulesFieldInfo.PROJECT_CREATEDBY, createdBy);
	}
	
	public Date getModifiedDate() {
		return getFieldValueDate(ProjectRulesFieldInfo.PROJECT_MODIFIED);
	}
	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(ProjectRulesFieldInfo.PROJECT_MODIFIED, modifiedDate);
	}
	
	public String getModifiedBy() {
		return getFormattedFieldValue(ProjectRulesFieldInfo.PROJECT_MODIFIEDBY);
	}
	public void setModifiedBy(String modifiedBy) {
		setFieldValue(ProjectRulesFieldInfo.PROJECT_MODIFIEDBY, modifiedBy);
	}
	
	/**
	 * Returns an unmodifiable collection of ProjectStarts linked to this shift.
	 */
	public Collection<ProjectStart> getProjectStartTimes() {
		return getRelatedObjects(ProjectRules.PROJECT_START_RELATIONSHIP_KEY);
	}
	/**
	 * Sets the possible StartTimes along with Weekday for the Project Rule
	 * @param col
	 */
	
	public void setProjectStartTimes(Collection<ProjectStart> col) {
		setRelatedObjects(ProjectRules.PROJECT_START_RELATIONSHIP_KEY, new HashSet<ProjectStart>(col));
	}
	
	

}
