package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class VTOEvent extends ValueObjectBase {
	private static final long serialVersionUID = 1L;

	private static FieldInfo m_fieldInfo = new VTOEventFieldInfo();
	
	public ID getActivityID() { return getFieldValueID(VTOEventFieldInfo.VTOEVENT_ACTIVITYID); }
	public void setActivityID(ID activityID) { setFieldValue(VTOEventFieldInfo.VTOEVENT_ACTIVITYID, activityID); }
	
	public ID getCampaignID() { return getFieldValueID(VTOEventFieldInfo.VTOEVENT_CAMPAIGNID); }
	public void setCampaignID(ID campaignID) { setFieldValue(VTOEventFieldInfo.VTOEVENT_CAMPAIGNID, campaignID); }
	
	public ID getOrganizationID() { return getFieldValueID(VTOEventFieldInfo.VTOEVENT_ORGANIZATIONID); }
	public void setOrganizationID(ID organizationID) { setFieldValue(VTOEventFieldInfo.VTOEVENT_ORGANIZATIONID, organizationID); }
	
	public String getName() { return getFormattedFieldValue(VTOEventFieldInfo.VTOEVENT_NAME); }
	public void setName(String name) { setFieldValue(VTOEventFieldInfo.VTOEVENT_NAME, name); }
	
	public int getMinDuration() { return getFieldValueInt(VTOEventFieldInfo.VTOEVENT_MINDURATION); }
	public void setMinDuration(int minDuration) { setFieldValue(VTOEventFieldInfo.VTOEVENT_MINDURATION, minDuration); }
	
	public int getMaxDuration() { return getFieldValueInt(VTOEventFieldInfo.VTOEVENT_MAXDURATION); }
	public void setMaxDuration(int maxDuration) { setFieldValue(VTOEventFieldInfo.VTOEVENT_MAXDURATION, maxDuration); }
	
	public String getDescription() { return getFormattedFieldValue(VTOEventFieldInfo.VTOEVENT_DESCRIPTION); }
	public void setDescription(String description) { setFieldValue(VTOEventFieldInfo.VTOEVENT_DESCRIPTION, description); }
	
	public Date getCreatedDate() { return getFieldValueDate(VTOEventFieldInfo.VTOEVENT_CREATED); }
	public void setCreatedDate(Date createdDate) { setFieldValue(VTOEventFieldInfo.VTOEVENT_CREATED, createdDate); }
	
	public String getCreatedBy() { return getFormattedFieldValue(VTOEventFieldInfo.VTOEVENT_CREATEDBY); }
	public void setCreatedBy(String createdBy) { setFieldValue(VTOEventFieldInfo.VTOEVENT_CREATEDBY, createdBy); }

	public Date getModifiedDate() { return getFieldValueDate(VTOEventFieldInfo.VTOEVENT_MODIFIED); }
	public void setModifiedDate(Date modifiedDate) { setFieldValue(VTOEventFieldInfo.VTOEVENT_MODIFIED, modifiedDate); }
	
	public String getModifiedBy() { return getFormattedFieldValue(VTOEventFieldInfo.VTOEVENT_MODIFIEDBY); }
	public void setModifiedBy(String modifiedBy) { setFieldValue(VTOEventFieldInfo.VTOEVENT_MODIFIEDBY, modifiedBy); }

	public Date getDeleteOnDate() {
		return getFieldValueDate(VTOEventFieldInfo.VTOEVENT_DELETEONDATE);
	}
	
	@Override
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

}
