package com.bluepumpkin.ejb.bbm.time;

import java.util.Map;

import com.bluepumpkin.common.datatypes.DayOfWeek;
import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeContext;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.ProjectRules;
import com.bluepumpkin.ejb.bbm.workrules.model.Shift;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent;

/**
 * This class contains several convenience methods for getting a TimeContext appropriate for various objects.
 */
public class TimeContextFactory {
	
	public static TimeContext getTimeContext(Organization org) {
		TimeContext tc = new TimeContext(org.getTimeZone(), org.getWeekStartDay(), org.getDayBoundaryOffset());
		return tc;
	}
	
	public static TimeContext getTimeContext(Campaign c) {
		TimeContext tc = new TimeContext(c.getTimeZone(), DayOfWeek.from0SundayCount(c.getWeekStart() / 1440), c.getWeekStart() % 1440);
		return tc;
	}

	/**
	 * This method will get a TimeContext that is appropriate for a Shift. This method takes in maps of possible
	 * Orgs and Campaigns so there will be no trips to the database.
	 * <p>
	 * The intended use case for this method is the client will collect all owner Organization and Campaign IDs,
	 * load the objects up, then finally call this method.
	 * 
	 * @param s
	 * @param orgMap A Map containing the Organization, keyed by Organization ID, that owns the Shift, if the Shift is owned by an Organization
	 * @param campaignMap A Map containing the Campaign, keyed by Campaign ID, that owns the Shift, if the Shift is owned by a Campaign
	 * @return
	 * @throws IllegalArgumentException if the Shift does not belong to an org or campaign contained in the map parameters
	 */
	public static TimeContext getTimeContext(Shift s, Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap) {
		if (s.getOrganizationID() != null) {
			if (! orgMap.containsKey(s.getOrganizationID())) throw new IllegalArgumentException("Org map must contain org for id: " + s.getOrganizationID());
			return getTimeContext(orgMap.get(s.getOrganizationID()));
		}
		if (s.getCampaignID() != null) {
			if (! campaignMap.containsKey(s.getCampaignID())) throw new IllegalArgumentException("Campaign map must contain campaign for id: " + s.getCampaignID());
			return getTimeContext(campaignMap.get(s.getCampaignID()));
		}
		throw new IllegalArgumentException("Shift must contain either an org id or a campaign id.");
	}
	
	/**
	 * This method will get a TimeContext that is appropriate for a Shift Event. This method takes in maps of possible
	 * Orgs and Campaigns so there will be no trips to the database.
	 * <p>
	 * The intended use case for this method is the client will collect all owner Organization and Campaign IDs,
	 * load the objects up, then finally call this method.
	 * 
	 * @param s
	 * @param orgMap A Map containing the Organization, keyed by Organization ID, that owns the Shift Event, if the Shift Event is owned by an Organization
	 * @param campaignMap A Map containing the Campaign, keyed by Campaign ID, that owns the Shift Event, if the Shift Event is owned by a Campaign
	 * @return
	 * @throws IllegalArgumentException if the Shift Event does not belong to an org or campaign contained in the map parameters
	 */
	public static TimeContext getTimeContext(ShiftEvent se, Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap) {
		if (se.getOrganizationID() != null) {
			if (! orgMap.containsKey(se.getOrganizationID())) throw new IllegalArgumentException("Org map must contain org for id: " + se.getOrganizationID());
			return getTimeContext(orgMap.get(se.getOrganizationID()));
		}
		if (se.getCampaignID() != null) {
			if (! campaignMap.containsKey(se.getCampaignID())) throw new IllegalArgumentException("Campaign map must contain campaign for id: " + se.getCampaignID());
			return getTimeContext(campaignMap.get(se.getCampaignID()));
		}
		throw new IllegalArgumentException("Shift Event must contain either an org id or a campaign id.");
	}
	public static TimeContext getTimeContext(ProjectRules projRule, Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap) {
		if (projRule.getOrganizationID() != null) {
			if (! orgMap.containsKey(projRule.getOrganizationID())) throw new IllegalArgumentException("Org map must contain org for id: " + projRule.getOrganizationID());
			return getTimeContext(orgMap.get(projRule.getOrganizationID()));
		}
		if (projRule.getCampaignID() != null) {
			if (! campaignMap.containsKey(projRule.getCampaignID())) throw new IllegalArgumentException("Campaign map must contain campaign for id: " + projRule.getCampaignID());
			return getTimeContext(campaignMap.get(projRule.getCampaignID()));
		}
		throw new IllegalArgumentException("Project Rule must contain either an org id or a campaign id.");
	}
}
