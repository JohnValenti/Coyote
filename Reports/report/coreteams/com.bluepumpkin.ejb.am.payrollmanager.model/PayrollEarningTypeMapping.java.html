package com.bluepumpkin.ejb.am.payrollmanager.model;

import java.io.Serializable;

import com.bluepumpkin.common.datatypes.ID;
/**
 * Title:        Blue Pumpkin Software Foundation
 * Description:  PayrollProviderConfig
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */

public class PayrollEarningTypeMapping implements Serializable
{
	private ID m_ID;
	private ID m_DSPPID;
	private ID m_EarningTypeID;
	private String m_EarningType;
	private String m_PayrollEarningType;
	public PayrollEarningTypeMapping() {
		super();
	}
	// Constructor for backend
	public PayrollEarningTypeMapping(ID id,
									 ID dsppID,
									 ID earningTypeID,
									 String earningType,
									 String payrollEarningType) {
		m_ID = id;
		m_DSPPID = dsppID;
		m_EarningTypeID = earningTypeID;
		m_EarningType = earningType;
		m_PayrollEarningType = payrollEarningType;
	}
	// Constructor for UI
	public PayrollEarningTypeMapping(ID earningTypeID,
									 String earningType,
									 String payrollEarningType) {
		m_EarningTypeID = earningTypeID;
		m_EarningType = earningType;
		m_PayrollEarningType = payrollEarningType;
	}
	public ID getID() { return m_ID; }
	public void setID(ID id) { m_ID = id; }
	public ID getDataSourcePayrollProviderID() { return m_DSPPID; }
	public void setDataSourcePayrollProviderID(ID id) { m_DSPPID = id; }
	public void setEarningType(String name) { m_EarningType = name; }
	public void setPayrollEarningType(String name) { m_PayrollEarningType = name; }
	public String getEarningType() { return m_EarningType; }
	public ID getEarningTypeID() { return m_EarningTypeID; }
	public String getPayrollEarningType() { return m_PayrollEarningType; }
}
