package com.bluepumpkin.ejb.am.payrollmanager.model;

import java.io.Serializable;

import com.bluepumpkin.common.datatypes.ID;
/**
 * Title:        Blue Pumpkin Software Foundation
 * Description:  PayrollProviderConfig
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */

public class PayrollExportConfig implements Serializable, Comparable
{
	public static final short USERDEFINED = 0;
	public static final short CALCULATED = 1;
	public static final short CALCULATEDEARNINGS = 2;

	// Solely for UI to differentiate fields with multiple name
	private int m_UniqueNum;
	/**
	 * This object is a full capture of the configuration xml
	 */
	private ID m_ConfigID;
	//DataSourcePayrollProvider ID
	private ID m_DSPPID;
	private ID m_PayrollEarningTypeID;
	private String m_FieldName;
	private short m_FieldType;
	private String m_Value;
	private String m_DisplayName;
	private int m_Sequence = 0;
	private boolean m_IsMandatory = false;
	private boolean m_IsExported = false;
	private boolean m_IsPerExport = false;
	private String m_Version;
	public PayrollExportConfig() {
		super();
	}
	// Constructor for the backend load
	public PayrollExportConfig(ID configID,
								 ID datasourcePPID,
								 ID payrollEarningTypeID,
								 String fieldName,
								 short fieldType,
								 String value,
								 String displayName,
								 int sequence,
								 boolean isMandatory,
								 boolean isExported,
								 boolean isPerExport) {
		m_ConfigID = configID;
		m_DSPPID = datasourcePPID;
		m_PayrollEarningTypeID = payrollEarningTypeID;
		m_FieldName = fieldName;
		m_FieldType = fieldType;
		m_DisplayName = displayName;
		m_Sequence = sequence;
		m_IsMandatory = isMandatory;
		m_Value = value;
		// it implies if the field is mandatory, it must be exported
		m_IsExported = isMandatory?true:isExported;
		m_IsPerExport = isPerExport;
	}
	// Constructor for the frontend construct
	public PayrollExportConfig(ID datasourcePPID,
								 ID payrollEarningTypeID,
								 String fieldName,
								 short fieldType,
								 String value,
								 String displayName,
								 int sequence,
								 boolean isMandatory,
								 boolean isExported,
								 boolean isPerExport) {
		m_DSPPID = datasourcePPID;
		m_PayrollEarningTypeID = payrollEarningTypeID;
		m_FieldName = fieldName;
		m_FieldType = fieldType;
		m_DisplayName = displayName;
		m_Sequence = sequence;
		m_IsMandatory = isMandatory;
		m_Value = value;
		// it implies if the field is mandatory, it must be exported
		m_IsExported = isMandatory?true:isExported;
		m_IsPerExport = isPerExport;
	}
	public void setID(ID id) { m_ConfigID = id; }
	public ID getID() { return m_ConfigID; }
	public void setPayrollProviderDataSourceID(ID id) { m_DSPPID = id; }
	public ID getPayrollProviderDataSourceID() { return m_DSPPID; }
	public void setPayrollEarningTypeID(ID id) { m_PayrollEarningTypeID = id; }
	public ID getPayrollEarningTypeID() { return m_PayrollEarningTypeID; }

	// Doubt if the UI will ever use this
	public String getFieldName() { return m_FieldName; }
	// This is not to be used by the UI
	public void setFieldName(String name) { m_FieldName = name; }

	public short getFieldType() { return m_FieldType; }
	public void setFieldType(short type) { m_FieldType = type; }
	// If there is no given DisplayName, choose the FieldName
	// UI should always invoke getDisplayName/setDisplayName
	public String getDisplayName() {
		return m_DisplayName == null? m_FieldName:m_DisplayName;
	}
	public void setDisplayName(String name) { m_DisplayName = name; }

	public int getSequence() { return m_Sequence; }
	public int getUniqueNum() { return m_UniqueNum; }
	public void setUniqueNum(int num) { m_UniqueNum = num; }

	// If the field is not mandatory(which implies user can choose sequence
	public void setSequence(int sequence) {	m_Sequence = sequence; }

	public String getValue() { return m_Value; }
	public void setValue(String value) {
		// CALCULATED field is not to be modified by the end user
//		if (m_FieldType != CALCULATED)
		m_Value = value;
	}

	public boolean isMandatory() { return m_IsMandatory; }
	public void setIsMandatory(boolean value) { m_IsMandatory = value; }

	public boolean isPerExport() { return m_IsPerExport; }
	public void setIsPerExport(boolean value) { m_IsPerExport = value; }

	public boolean isExported() { return m_IsExported; }
	public void setIsExported(boolean value) {
		// it implies if the field is mandatory, it must be exported
		m_IsExported = m_IsMandatory?true:value;
	}
	// a cache to the DATASOURCEPAYROLLPROVIDER version
	public String getVersion() { return m_Version; }
	public void setVersion(String version) { m_Version = version; }
	// Implementation of compareTo for sorting on sequence
	public int compareTo(Object obj) {
		PayrollExportConfig pecObj = (PayrollExportConfig)obj;
		// if there is no sequence assigned for the config, it is put after
		if(pecObj.getSequence() == m_Sequence) {
			return getDisplayName().compareTo(pecObj.getDisplayName());
		}
		else if(pecObj.getSequence() == 0) {
			return -1;
		}
		else if(m_Sequence == 0) {
			return 1;
		}
		else if(pecObj.getSequence() > m_Sequence) {
			return -1;
		}
		else if(pecObj.getSequence() < m_Sequence) {
			return 1;
		}
		else {
			return 0;
		}
	}

}
