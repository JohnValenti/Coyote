package com.bluepumpkin.ejb.am.rtaaexceptioncategory.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

/**
 * Title:        RTAAExceptionCategory
 * Description:  RTAAExceptionCategory value object holds a record in adherenceexceptioncategory table
 * Copyright:    Copyright (c) 2006
 * Company:      Witness Systems, inc
 * @author       Swati Tewari
 * @version      1.0
 */

public class RTAAExceptionCategory extends ValueObjectBase {

    // meta data info.
    private static FieldInfo m_fieldInfo = new RTAAExceptionCategoryFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }

    public RTAAExceptionCategory() {
        super();
    }

    public String getName()
    {
       return getFormattedFieldValue(RTAAExceptionCategoryFieldInfo.RTAAEXCEPTIONCATEGORY_NAME);
    }
    
    public String getDescription()
    {
       return getFormattedFieldValue(RTAAExceptionCategoryFieldInfo.RTAAEXCEPTIONCATEGORY_DESCRIPTION);
    }

    public ID getOrganizationID()
    {
        return (ID) getFieldValue(RTAAExceptionCategoryFieldInfo.RTAAEXCEPTIONCATEGORY_ORGANIZATIONID);
    }

    public void setName(String strName)
    {
       setFieldValue(RTAAExceptionCategoryFieldInfo.RTAAEXCEPTIONCATEGORY_NAME, strName);
    }
    
    public void setDescription(String strName)
    {
       setFieldValue(RTAAExceptionCategoryFieldInfo.RTAAEXCEPTIONCATEGORY_DESCRIPTION, strName);
    }

    public void setOrganizationID(ID id)
    {
        setFieldValue(RTAAExceptionCategoryFieldInfo.RTAAEXCEPTIONCATEGORY_ORGANIZATIONID, id);
    }
}
