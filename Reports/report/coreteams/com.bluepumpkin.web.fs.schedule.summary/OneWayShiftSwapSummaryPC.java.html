package com.bluepumpkin.web.fs.schedule.summary;

import java.util.ArrayList;
import java.util.List;
import java.util.ResourceBundle;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.l10n.RegionalFormatBundleKey;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.BPMessageFormat;
import com.bluepumpkin.web.fs.Log;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.container.CollapsibleContainerPC;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.html.CSSUtil;

public class OneWayShiftSwapSummaryPC extends CollapsibleContainerPC {
	private static final long serialVersionUID = 1L;
	private static final Category LOG = Log.initCategory(OneWayShiftSwapSummaryPC.class.getName());

	private final ResourceBundle bundle;
	private final ID empID;
	private final String empName;
	private OneWayShiftSwapSummaryData summaryData;

	// column headers
	private static final String[] SIMPLE_HEADER_RB_KEYS = new String[] {
			FsWebBundleKey.SUMMARY_REQUESTS_APPROVED,
			FsWebBundleKey.SUMMARY_REQUESTS_LIMIT,
			FsWebBundleKey.SUMMARY_REQUESTS_REMAINING };

	public OneWayShiftSwapSummaryPC(RequestContext context) {
		this(context, context.getUser().getEmployeeID(), null, null);
	}

	public OneWayShiftSwapSummaryPC(RequestContext context, ID employeeID, String employeeName,
			OneWayShiftSwapSummaryData summary) {
		super(context);
		bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);
		empID = employeeID;
		empName = employeeName;
		summaryData = summary;
		setName("OneWayShiftSwapSummaryPC_" + empID);
	}

	/**
	 * Initialize with default Data
	 */
	public void init(OneWayShiftSwapSummaryData summary) {
		try {
			summaryData = summary;
		} catch (Exception ex) {
			LOG.l7dError(FsWebBundleKey.MSG_SCHED_SUMMARY_NO_ONE_WAY_SHIFT_SWAP_SUMMARY, ex);
			addPageMessage(Message.ERROR_TYPE, FsWebBundleKey.MSG_SCHED_SUMMARY_NO_ONE_WAY_SHIFT_SWAP_SUMMARY, FsWebBundleKey.BUNDLE_NAME);
		}
	}

	/**
	 * Initialize Component For Display
	 */
	@Override
	public void initForDisplay() {
		if (m_isInitializedForDisplay) {
			return;
		} else {
			super.initForDisplay();
			initContent();
		}
	}

	/**
	 * Initialize Content
	 */
	private void initContent() {
		MultiColListPC theListPC = initContianerAndCreateList(m_context, this);
		createOneWayShiftSwapListModel(theListPC, summaryData);
		String title;

		initHeader(theListPC, SIMPLE_HEADER_RB_KEYS);

		if (summaryData == null) {
			if (empName == null) {
				// do not show employee name in title
				title = i18n(bundle, FsWebBundleKey.SCHED_SUMMARY_ONE_WAY_SS_SUMMARY);
			} else {
				// show employee name in title
				String template = i18n(bundle, FsWebBundleKey.SCHED_SUMMARY_ONE_WAY_SS_SUMMARY_EMP);
				Object[] args = new Object[] { empName };
				title = BPMessageFormat.format(template, args);
			}
		} else {
			String template;
			if (empName == null) {
				// do not show employee name in title
				template = i18n(bundle, FsWebBundleKey.SCHED_SUMMARY_ONE_WAY_SS_SUMMARY);
			} else {
				// show employee name in title
				template = i18n(bundle, FsWebBundleKey.SCHED_SUMMARY_ONE_WAY_SS_SUMMARY_EMP_DATE);
			}

			TimeZone viewTimeZone = m_context.getViewingTimeZone();

			String start =
					m_localizer.formatDate(summaryData.getStartDate(), viewTimeZone, RegionalFormatBundleKey.DATE_FORMAT);
			String end = m_localizer.formatDate(summaryData.getEndDate(), viewTimeZone, RegionalFormatBundleKey.DATE_FORMAT);
			Object[] args = new Object[] { start, end, empName };
			title = BPMessageFormat.format(template, args);
		}

		setTitle(title);
	}

	/**
	 * set One Way Shift Swap Summary List model
	 */
	private void createOneWayShiftSwapListModel(MultiColListPC theListPC, SimpleRequestSummaryData summaryData) {
		if (summaryData == null) {
			return;
		}

		List<DefaultMultiColumnNodeData> listModel = new ArrayList<DefaultMultiColumnNodeData>(1);
		DefaultMultiColumnNodeData rowData = new DefaultMultiColumnNodeData();

		addNumData(summaryData.getApproved(), rowData);
		addNumData(summaryData.getLimit(), rowData);
		addNumData(summaryData.getRemaining(), rowData);

		//Add a spacer column to take up space on the right
		rowData.add("");

		listModel.add(rowData);
		theListPC.setListModel(listModel);
	}

	private void initHeader(MultiColListPC theListPC, String[] rbKeys) {
		TableHeaderPC header = theListPC.getHeader();
		for (int i = 0; i < rbKeys.length; i++) {
			String label = i18n(bundle, rbKeys[i]);
			if (FsWebBundleKey.SUMMARY_REQUESTS_REMAINING.equals(rbKeys[i])) {
				label = getRemainingColumTitle(label);
			}
			header.addColumnHeaderData(null, label, CSSUtil.CLASS_INNER_HEADER_BORDER_NUMERIC);
		}

		//Add a spacer column to take up space on the right
		header.addColumnHeaderData(null, "", "60%", CSSUtil.CLASS_INNER_HEADER_BORDER_NUMERIC, false);
	}

	private String getRemainingColumTitle(String label) {
		Object[] args = new Object[] { getPeriodName()};
		return BPMessageFormat.format(label, args);
	}

	private String getPeriodName() {
		if (summaryData == null) {
			return "";
		}

		return (summaryData.getPeriod() == ScheduleSummaryMH.PERIOD_MONTH_INDEX)
				? i18n(bundle, FsWebBundleKey.FLEX_TIME_SUMMARY_MONTH)
				: i18n(bundle, FsWebBundleKey.FLEX_TIME_SUMMARY_QUARTER);
	}

	public static final MultiColListPC initContianerAndCreateList(RequestContext context, CollapsibleContainerPC parentPC) {
		// List for Display purpose
		MultiColListPC theListPC = new MultiColListPC(context, false);
		theListPC.setEnabled(false);
		theListPC.setHasRowHeaders(false);

		// Initialize Container
		parentPC.setStyleClass(CSSUtil.CLASS_BORDER_DEFAULT);
		parentPC.setTemplate("{0}");
		parentPC.addComponent(theListPC);

		return theListPC;
	}

	private void addNumData(int num, DefaultMultiColumnNodeData rowData) {
		rowData.add( m_localizer.formatNumber(num), CSSUtil.STYLE_NUMERIC );
	}
}
