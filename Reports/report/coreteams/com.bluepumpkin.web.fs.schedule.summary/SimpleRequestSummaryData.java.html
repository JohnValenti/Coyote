package com.bluepumpkin.web.fs.schedule.summary;

import java.util.Date;

/**
 * Title: SimpleRequestSummaryData Description: Value Object for simple summary
 * data (Approved/Limit/Remaining) for one employee for UI display.
 */
public class SimpleRequestSummaryData {

	private int approved = 0;
	private int limit = 0;
	private int remaining = 0;

	private Date startDate;
	private Date endDate;

	public SimpleRequestSummaryData() {
	}

	public int getApproved() {
		return approved;
	}

	public SimpleRequestSummaryData setApproved(int approved) {
		this.approved = approved;
		return this;
	}

	public int getLimit() {
		return limit;
	}

	public SimpleRequestSummaryData setLimit(int limit) {
		this.limit = limit;
		return this;
	}

	public int getRemaining() {
		return remaining;
	}

	public SimpleRequestSummaryData setRemaining(int remaining) {
		this.remaining = remaining;
		return this;
	}

	public Date getStartDate() {
		return startDate;
	}

	public SimpleRequestSummaryData setStartDate(Date startDate) {
		this.startDate = startDate;
		return this;
	}

	public Date getEndDate() {
		return endDate;
	}

	public SimpleRequestSummaryData setEndDate(Date endDate) {
		this.endDate = endDate;
		return this;
	}
}
