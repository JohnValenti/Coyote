package com.bluepumpkin.web.fs.pulse;

import com.bluepumpkin.web.core.applet.AppletPC;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.system.config.Screen;

/**
 * Title:		PulseAppletPC.java
 * Description:	This compoment creates the applet tags needed for the HTML page. 
 * Copyright:	Copyright (c) 2004
 * Company:	    Blue Pumpkin Software, Inc
 * @author	    Alvin Cham, David Su
 * @version 1.0
 */
public class PulseAppletPC extends AppletPC {
	/**
	 * Constructor
	 *
	 * @param context The RequestContext
	 * @see RequestContext
	 */
	public PulseAppletPC(RequestContext context) {
		super(context);
		setName("PulseAppletPC");
	}

	/**
	 * Adds any parameters to the applet during each instantiation of
	 * the applet. Called from getHtmlDisplay().
	 */
	protected void addDynamicParameters() {
		super.addDynamicParameters();

		//Needs to be synchronized because the work frame (this class) and the queue  
		//selection frame (QueueSelectionPM) both load at the same time.
		synchronized (m_context.getRequest().getSession()) {
			Screen screen = m_context.getScreen();
			if (screen == null) {
				return;
			}

			//viewType is the Pulse mode ("view" or "history")
			String viewType = screen.getParameterValue("ViewType");
			addParameter("ViewType", viewType);
		}//synchronized
	}
}
