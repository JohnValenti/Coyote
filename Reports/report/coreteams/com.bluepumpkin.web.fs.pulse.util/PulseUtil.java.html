	public static Date getStartOfDay(Date oldDate, TimeZone oldTimeZone, TimeZone newTimeZone) {
		Calendar cal = Calendar.getInstance(oldTimeZone);
		cal.setTime(oldDate);
		int year = cal.get(Calendar.YEAR);
		int month = cal.get(Calendar.MONTH);
		int day = cal.get(Calendar.DATE);

		cal = Calendar.getInstance(newTimeZone);
		cal.set(year, month, day, 0, 0, 0);
		cal.set(Calendar.MILLISECOND, 0);
		return cal.getTime();
	}

	public static void dumpTrackingView(TrackingView view) {
		System.out.println("************* TrackingView ***********");
		if (view == null) {
			System.out.println(" null view ");
			return;
		}
		System.out.println("  id=" + view.getID());
		System.out.println("  userid=" + view.getUserID());
		if (!(view instanceof HistoryTrackingView)) {
			System.out.println("  name=" + view.getViewName());
			System.out.println("  desc=" + view.getDescription());
			System.out.println("  show trends=" + view.useTrending());
			System.out.println("  global=" + view.isGlobal());
		}
		System.out.println("  hoo=" + view.useHOO());
		System.out.println("  ***** charts ******");
		List charts = view.getChartDefinition();
		for (int i = 0; i < charts.size(); i++) {
			TraceChart tc = (TraceChart)charts.get(i);
			boolean[] lineTypes = tc.getLineType();
			System.out.println("     chart(" + i + ")=" + tc);
		}
	}

	public static void dumpCubeMap(HashMap cubeMap) {

		for (Iterator qIter = cubeMap.keySet().iterator(); qIter.hasNext();) {
			ID qID = (ID)qIter.next();

			System.out.println("********* qID = " + qID + "  ***********");
			TraceCube[] cubeArray = (TraceCube[])cubeMap.get(qID);
			dumpTraceCubeArray(cubeArray);
		}
	}

	public static void dumpTraceCubeArray(TraceCube[] cubeArray) {
		for (int i = 0; i < cubeArray.length; i++) {
			System.out.println("Cube#" + i + "::" + cubeArray[i]);
		}
	}
	
	/**
	 * returns an array of trace types which the given trace type depends on.
	 * @param traceType
     * @param outboundQueueSelected pass true if any outbound queues are selected (including when the 
     *        combined includes an outbound queue), false otherwise. If true, we will return any
     *        outbound statistics that non-outbound statistics depend on. If false, we will not
     *        return any outbound statistics that non-outbound statistics depend on.
     * @param immediateQueueSelected - pass true if any immediate queues are selected.
	 * @return
	 */
	public static short[] getDependantTraceTypes(short traceType, boolean outboundQueueSelected,
			boolean immediateQueueSelected) {
		switch (traceType) {
			case Trace.CV:
			case Trace.ABANDONMENT:
			case Trace.BACKLOG:
			case Trace.FTE:
			case Trace.AFTE:
				return new short[0];
			case Trace.ASTAFFING:
			case Trace.STAFFING:
				return new short[]{Trace.FTE, Trace.AFTE};
			case Trace.NETSTAFFING:
				return new short[]{Trace.STAFFING, Trace.ASTAFFING, Trace.FTE, Trace.AFTE};
			case Trace.AHT:
				if (outboundQueueSelected) {
					return new short[]{Trace.CV, Trace.CONNECTS, Trace.DIALS, Trace.CRATE, Trace.CV_VH, Trace.VH};
				} else {
					return new short[]{Trace.CV, Trace.VH, Trace.CV_VH};
				}
			case Trace.PCA:
			case Trace.SERVICE_GOALS_PCA_IMMEDIATE:
			case Trace.SERVICE_GOALS_PCA_DEFERRED:
				if (immediateQueueSelected) {
					return new short[]{Trace.CV, Trace.VH, Trace.ABANDONMENT, Trace.VAR};
				} else {
					return new short[]{Trace.CV, Trace.VH, Trace.VAR};
				}
			case Trace.ASA:
			case Trace.SERVICE_GOALS_ASA:
				if (immediateQueueSelected) {
					return new short[]{Trace.CV, Trace.VH, Trace.ABANDONMENT, Trace.VAR};
				} else {
					return new short[]{Trace.CV, Trace.ABANDONMENT};
				}
			case Trace.OCCUPANCY:
				return new short[]{Trace.STAFFING};
			case Trace.CONNECTS:
				return new short[]{Trace.DIALS, Trace.CRATE};
			case Trace.CRATE:
				return new short[]{Trace.CONNECTS, Trace.DIALS};
			case Trace.RPC:
				return new short[]{Trace.DIALS, Trace.RPCRATE};
			case Trace.RPCRATE:
				return new short[]{Trace.RPC, Trace.DIALS};
			case Trace.RPCAHT:
				return new short[]{Trace.RPC};
			case Trace.FORECASTING_AHT:
				if (outboundQueueSelected) {
					return new short[]{Trace.CV, Trace.CRATE, Trace.CV_VH, Trace.VH};
				} else {
					return new short[]{Trace.CV, Trace.VH, Trace.CV_VH};
				}
			default:
				return new short[0];
		}
	}
