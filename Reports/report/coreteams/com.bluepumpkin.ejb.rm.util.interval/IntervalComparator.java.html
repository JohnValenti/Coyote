package com.bluepumpkin.ejb.rm.util.interval;

import java.util.Comparator;

/**
 * Sorts intervals by start,end  ascending
 */
public class IntervalComparator<T extends Interval> implements Comparator<T> {

	private static final IntervalComparator<Interval> INSTANCE = new IntervalComparator<Interval>();

	@Override
	public int compare(T a, T b) {
		return IntervalUtil.compareTo(a, b);
	}

	@SuppressWarnings("unchecked")
	public static <T extends Interval> IntervalComparator<T> getStartEndAscending() {
		return (IntervalComparator<T>) INSTANCE;
	}

};
