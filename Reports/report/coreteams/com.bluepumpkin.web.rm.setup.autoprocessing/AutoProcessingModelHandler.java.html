package com.bluepumpkin.web.rm.setup.autoprocessing;

import java.rmi.RemoteException;
import java.util.Collection;
import java.util.Map;
import java.util.ResourceBundle;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.bluepumpkin.ejb.rm.setup.autoprocessing.model.AutoProcessingRule;
import com.bluepumpkin.web.core.base.ModelHandler;
import com.bluepumpkin.web.rm.setup.filingrules.FilingRuleModelHandler;

/**
 * Title          AutoProcessingModelHandler
 * Description    Model handler for Auto Processing Settings
 * Copyright      Copyright (c) 2002
 * Company        Blue Pumpkin Software, Inc
 * @author        Howard Mullings
 * @version       1.0
 */
public class AutoProcessingModelHandler extends ModelHandler { 
	/**
	 * Returns a list AutoProcessingRule
	 * @param   orgID	ID	Identifes the organizationId
	 * @return  Collection	list of AutoProcessingRule objects
	 * @throws BbmFinderException
	 * @throws RemoteException
	 */
	public static Collection getAutoProcessingRuleList(ID orgID)
		throws BbmFinderException, BbmCreateException, RemoteException {

		// Invoke the facade method here
		return RmManagerFactory.getInstance().
					getAutoProcessingRuleManager().
						findAutoProcessingRuleByOrgId(orgID);
	}

	/**
	 * Returns a list AutoProcessingRule
	 * @param   orgID	ID	Identifes the organizationId
	 * @param   requestType	String	Identifes the Request Type
	 * @return  Collection	list of AutoProcessingRule objects
	 * @throws BbmFinderException
	 * @throws RemoteException
	 */
	public static Collection getAutoProcessingRuleList(ID orgID, String requestType)
		throws BbmFinderException, BbmCreateException, RemoteException {

		// Invoke the facade method here
		return RmManagerFactory.getInstance().getAutoProcessingRuleManager().
					findAutoProcessingRuleByOrgId(orgID, requestType);
	}

	/**
	 * Find the AutoProcessingRule for the ID
	 * @param	autoProcessingId ID		Identifies the the AutoProcessingRule object.
	 * @return	AutoProcessingRule	A AutoProcessingRule Object
	 * @throws BbmCreateException
	 * @throws RemoteException
	 */
	public static AutoProcessingRule getAutoProcessingRule(ID autoProcessingId)
		throws BbmCreateException, RemoteException {

		try {
			return RmManagerFactory.getInstance().getAutoProcessingRuleManager().
				findAutoProcessingRuleById(autoProcessingId);
		}
		catch (BbmFinderException e) {
			// not likely to happen
		}
		return null;
	}


	/**
	 * Updates the  AutoProcessingRule
	 * @param autoProcessingRule - AutoProcessingRule object.
	 * @throws BbmUpdateException
	 * @throws RemoteException
	 */
	public static void updateAutoProcessingRule(AutoProcessingRule autoProcessingRule)
		throws BbmUpdateException, BbmCreateException, RemoteException {

		RmManagerFactory.getInstance().getAutoProcessingRuleManager().
			updateAutoProcessingRule(autoProcessingRule);
	}


	/**
	 * Deletes the AutoProcessingRule
	 * @param autoProcessingRuleId - Id for AutoProcessingRule.
	 * @throws BbmRemoveException
	 * @throws RemoteException
	 */
	public static void deleteAutoProcessingRule(ID autoProcessingRuleId)
		throws BbmRemoveException, BbmCreateException, RemoteException {
		RmManagerFactory.getInstance().getAutoProcessingRuleManager().
			deleteAutoProcessingRule(autoProcessingRuleId);
	}


	/**
	 * Creates a new AutoProcessingRule
	 * @param autoProcessingRule	AutoProcessingRule instance
	 * @throws BbmCreateException
	 * @throws RemoteException
	 */
	public static ID createAutoProcessingRule(AutoProcessingRule autoProcessingRule)
		throws BbmCreateException, RemoteException {
		return RmManagerFactory.getInstance().getAutoProcessingRuleManager().
			createAutoProcessingRule(autoProcessingRule);
	}

	/**
	 * Updates the InheritanceStatus for the AutoProcessingRule
	 * @param organizationId	ID	Identifies the organizationID
	 * @param autoProcessingRuleIDStatus	Collection contains the AutoProcessing Id
	 *                                  and the applied status.
	 * @throws BbmUpdateException
	 * @throws RemoteException
	 */
	public static void saveApplied(ID organizationId, 
								   Map autoProcessingRuleIDStatus)
		throws BbmUpdateException, BbmCreateException, RemoteException {

		RmManagerFactory.getInstance().getAutoProcessingRuleManager().
			saveApplied(organizationId,autoProcessingRuleIDStatus);
	}

	/**
	 * Copies an existing AutoProcessingRule to the organization specified
	 * @param autoProcessingRuleId	ID	Identifies the AutoProcessingRuleID
	 * @param organizationId		ID	Identifies the organizationID
	 * @throws BbmCreateException
	 * @throws BbmFinderException
	 * @throws RemoteException
	 */
	public ID copyAutoProcessingRule(ID autoProcessingRuleId, ID organizationId)
		throws BbmCreateException, BbmFinderException, RemoteException {
		return RmManagerFactory.getInstance().getAutoProcessingRuleManager().
			copyAutoProcessingRule(autoProcessingRuleId,organizationId);
	}

	/**
	 * Gets a collection of StringsPair objects that identify the TimeOff types
	 * @param   orgID	ID	Identifes the organizationId
	 * @return	Collection	list of StringsPair objects that contain the TimeOff types
	 * @throws BbmFinderException
	 * @throws BbmCreateException
	 * @throws RemoteException
	 */
	public static Collection getTimeOffTypes(ID orgID) throws BbmFinderException, BbmCreateException, RemoteException { // NOSONAR
		return FilingRuleModelHandler.getTimeOffTypes(orgID);
	}
	
	/**
	 * Gets a collection of StringsPair objects that identify the TimeOff types
	 * @param   orgID	ID	Identifes the organizationId
	 * @param   isFlex	true returns only flex time offs, false returns only non-flex time offs.
	 * @return	Collection	list of StringsPair objects that contain the TimeOff types
	 * @throws BbmFinderException
	 * @throws BbmCreateException
	 * @throws RemoteException
	 */
	@SuppressWarnings("rawtypes")
	public static Collection getTimeOffTypes(ID orgID, boolean isFlex) throws BbmFinderException, BbmCreateException, RemoteException { // NOSONAR
		return FilingRuleModelHandler.getTimeOffTypes(orgID, isFlex);
	}

	public static Collection<StringsPair> getTimeOffTypesExcludingVto(ID orgID, boolean isFlex) {
		return FilingRuleModelHandler.getTimeOffTypesExcludingVto(orgID, isFlex);
	}

	/**
	 * Get the day type list from the filing rule model handler
	 * so the lists are consistent across the screens.
	 */
	public static Collection getDayTypeList(ResourceBundle rmBundle, Localizer localizer) {
		return RequestUtil.getDayTypeList(rmBundle, localizer);
	}
}
