public class TimeOffBidEmployeesRH extends WorkpaneRequestHandler {
	private static final Category LOG = Log.initCategory(TimeOffBidEmployeesRH.class.getName());

	/**
	 * Process Request actions for the Time OFF Bid Employees List page
	 * 
	 * @param context The Request Context
	 */
	@Override
	public void processCustomRequest(RequestContext context) throws WebException { // NOSONAR
		if (isAction(context, RmPageAction.ADD_TO_BID)) {
			processAddToBid(context);
		} else if (isAction(context, RmPageAction.REMOVE_FROM_BID)) {
			processRemoveFromBid(context);
		} else if (isAction(context, PageAction.FILTER_CHANGE) || (isAction(context, PageAction.REFRESH))) {
			processFilterChange(context);
		}
	}

	@Override
	public WorkpanePageModel getPageModel(RequestContext context) {
		TimeOffBidEmployeesPM model = new TimeOffBidEmployeesPM(context);
		model.extractParameters(context.getRequest());
		context.setPageModel(model);
		return model;
	}

	/**
	 * Add Bidder(s) to a Bid.
	 * 
	 * @param context Context of Employees Page
	 */
	private void processAddToBid(RequestContext context) {
		TimeOffBidEmployeesPM model = null;
		Collection listData = Collections.emptyList();
		Collection<TimeOffBidder> alreadyAddToBid = new ArrayList<TimeOffBidder>();
		Collection<TimeOffBidder> canAddEmployees = Collections.emptyList();

		try {
			model = (TimeOffBidEmployeesPM) getPageModel(context);

			ID bidID = model.getTimeOffBidID();
			Collection<ID> employeeIDs = model.getEmployeeIDs();
			Collection<ID> orgIDs = model.getOrgAndChildOrgs();
			Collection<ID> selectedIDs = model.getPerfromActionOnIDsAsCollection();
			Collection<ID> waitingIDs = new ArrayList();

			if (employeeIDs == null || employeeIDs.isEmpty()) {
				listData = TimeOffBidModelHandler.getWaitingTimeOffBiddersForBid(bidID, orgIDs);
			} else {
				listData = TimeOffBidModelHandler.getWaitingTimeOffBiddersForBidIDAndEmployeeIDs(bidID, employeeIDs, orgIDs);
			}

			for (Iterator it = listData.iterator(); it.hasNext();) {
				TimeOffBidder timeOffBidder = (TimeOffBidder) it.next();
				if (timeOffBidder.getStatus() == TimeOffBidder.STATUS_NOT_ADDED) {
					ID bidderID = timeOffBidder.getEmployeeId();
					waitingIDs.add(bidderID);
				} else if (selectedIDs.contains(timeOffBidder.getEmployeeId())){
					alreadyAddToBid.add(timeOffBidder);
				}
			}

			Collection<ID> bidders = CollectionUtil.createIntersection(selectedIDs, waitingIDs);
			canAddEmployees = TimeOffBidModelHandler.addToBid(bidID, bidders);
			if (canAddEmployees.size() != 0) {
				processEmployeesAlreadyInBid(model, canAddEmployees, RmWebBundleKey.EMPLOYEES_ALREADY_IN_ANOTHER_BID, false, Message.ERROR_TYPE);
			}
			
			// Display a message for Bidder(s) that were added to the Bid previously
			if (!alreadyAddToBid.isEmpty()) {
				processEmployeesAlreadyInBid(model, alreadyAddToBid, RmWebBundleKey.EMPLOYEES_ALREADY_IN_BID, true, Message.INFO_TYPE);
			}
		} catch (Exception ex) {
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_ADD_EMPLOYEES_TO_BID, RmWebBundleKey.BUNDLE_NAME);
			LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}

		model.setFetchPage(true);
	}

	/**
	 * Removes Employee(s) from a Bid
	 * 
	 * @param context Context of Employees Page
	 */
	private void processRemoveFromBid(RequestContext context) {
		TimeOffBidEmployeesPM model = null;
		try {
			model = (TimeOffBidEmployeesPM) getPageModel(context);

			ID bidID = model.getTimeOffBidID();
			Collection<ID> selectedIDs = model.getPerfromActionOnIDsAsCollection();
			TimeOffBidModelHandler.removeFromBid(bidID, selectedIDs);
		} catch (Exception ex) {
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_REMOVE_EMPLOYEES_TO_BID, RmWebBundleKey.BUNDLE_NAME);
			LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
		}

		model.setFetchPage(true);
	}

	/**
	 * Sets User Preferences the Filter Name when a filter is selected
	 * 
	 * @param context Context of Employees Page
	 */
	private void processFilterChange(RequestContext context) {
		// Retrieve Filter Info Information
		ID filterID = getExtractedFilterID(context);
		String filterName = getFilterName(context, filterID);
		String filterIDName = filterID.toString();

		if (filterName != null && !"".equals(filterName)) {
			context.setUserProperty(UserPreferenceKeys.EMP_FILTER, filterName);
		} else {
			if (filterIDName != null && filterIDName.equals(PageKeys.DEFAULT_NONE)) {
				context.setUserProperty(UserPreferenceKeys.EMP_FILTER, filterIDName);
			} else if (filterIDName != null && filterIDName.equals(PageKeys.DEFAULT_ALL)) {
				context.setUserProperty(UserPreferenceKeys.EMP_FILTER, filterIDName);
			}
		}
	}

	/**
	 * Retrieves the name of the currently selected Filter.
	 * 
	 * @param context Context of Employees Page
	 * @param filterID ID of the currently selected Filter
	 * @return
	 */
	private String getFilterName(RequestContext context, ID filterID) {
		String filterName = "";
		
		try {
			if (filterID == null) {
				filterName = context.getUserProperty(UserPreferenceKeys.EMP_FILTER);
				if (filterName == null) {
					filterName = "";
				}
			} else {
				// this line is added because filterID is coming in as 'DEFAULT_ALL'.
				// employeeFilter.getFilterByID can be called only with the row id
				EmployeeFilter employeeFilter = EmployeeModelHandler.getEmployeeFilter(context);
				Filter filter = employeeFilter.getFilterByID(filterID);
				if (filter == null) {
					filterName = "";
				} else {
					filterName = filter.getFilterName();
				}
			}
		} catch (Exception e) {
			LOG.l7dError(RmWebLogBundleKey.RM_TIMEOFF_BID_RETRIEVE_EMPLOYEE_FILTER_FAILURE, e);
			filterName = "";
		}
		return filterName;
	}

	/**
	 * Retrieves selected Filter
	 * 
	 * @param context Context of Employees Page
	 * @return ID of selected Filter
	 */
	private ID getExtractedFilterID(RequestContext context) {
		// Retrieve Filter Info Information
		PeopleFilterSelectorPC fbPC = new PeopleFilterSelectorPC(context);
		return PeopleFilterSelectorPC.extractFilterID(context, fbPC.getName());
	}
	
	/**
	 * Process the employees that are already in the current bid and add them to an error message displayed to the user
	 * 
	 * @param model TimeOff Bid Employees Model 
	 * @param notInBid Collection of Time Off Bidders that are in the current bid
	 */
	protected void processEmployeesAlreadyInBid(TimeOffBidEmployeesPM model, Collection<TimeOffBidder> alreadyAddToBid, String messageKey, 
			                                    boolean addBidToMessage, int messageType) {
		String args[] = new String[2];
		String semiColon = "; ";

		int idx = 0;
		StringBuilder employeeNames = new StringBuilder();
		for (Iterator<TimeOffBidder> it = alreadyAddToBid.iterator(); it.hasNext();) {
			TimeOffBidder bidder = it.next();

			// Add a semicolon after the first name in the list if more than one.
			if (idx != 0) {
				employeeNames.append(semiColon);
			}

			// Format Name
			Employee emp = bidder.getOptMethods().getEmployee();
			employeeNames.append(model.getLocalizer().formatName(emp));

			idx++;
		}

		// Add elements to the args
		if (addBidToMessage) {
			args[0] = model.getTimeOffBidName();
			args[1] = employeeNames.toString();
		} else {
			args[0] = employeeNames.toString();
		}
		model.addPageMessage(messageType, messageKey, RmWebBundleKey.BUNDLE_NAME, args);
	}
}
