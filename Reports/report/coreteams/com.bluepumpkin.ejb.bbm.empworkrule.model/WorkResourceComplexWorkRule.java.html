package com.bluepumpkin.ejb.bbm.empworkrule.model;

import java.util.Collections;
import java.util.Date;
import java.util.HashSet;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectRich;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRuleFieldInfo;

public class WorkResourceComplexWorkRule extends ValueObjectRich {
	private static final long serialVersionUID = 1L;
	
	//These are used to keep track of fields that have been modified in a multi-edit scenario
	private boolean isStartWeekOffsetModified = false;
	private boolean isWeeksEnabledModified = false;
	private boolean isNumWeeksInRotationModified = false;
	
	// meta data information
	private static FieldInfo m_fieldInfo = new WorkResourceComplexWorkRuleFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	private HashSet<Integer> weeksEnabled;
	public final static String COMPLEXWORKRULE_RELATIONSHIP_KEY = "COMPLEXWORKRULE|ID";
	
	public WorkResourceComplexWorkRule() {
		super();
		setNumWeeksInRotation(1);  // Default cannot be 0 because of F&S Thick Client.
	}
	
	public ID getID() {
		return this.getFieldValueID(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_SID);
	}
	
	public ID getWorkResourceID() {
		return this.getFieldValueID(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_WORKRESOURCEID);
	}

	public ID getComplexWorkRuleID() {
		if (getComplexWorkRule() != null) return getComplexWorkRule().getID();
		return null;
	}
	
	//This is a DEID
	private void setComplexWorkRuleID(ID cwrID) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_COMPLEXWORKRULEID, cwrID);
	}
	
	public int getStartWeekOffSet() {
		return this.getFieldValueInt(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_STARTWEEKOFFSET);
	}
	
	public void setStartWeekOffSet(int startWeekOffSet) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_STARTWEEKOFFSET, startWeekOffSet);		
	}
	
	public int getNumWeeksInRotation() {
		return this.getFieldValueInt(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_NUMWEEKSINROTATION);
	}
	
	public void setNumWeeksInRotation(int numWeeksInRotation) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_NUMWEEKSINROTATION, numWeeksInRotation);
	}
	
	public int getWeekEnabledMask() {
		return this.getFieldValueInt(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_WEEKENABLEDMASK);
	}
	
	protected void setWeekEnabledMask(int mask) {
		this.setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_WEEKENABLEDMASK, mask);
	}
	
	public void setWorkResourceID(ID value) {
		this.setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_WORKRESOURCEID, value);
	}
	
	public Date getCreatedDate() {
		return getFieldValueDate(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_CREATED);
	}

	public void setCreatedDate(Date createdDate) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_CREATED, createdDate);
	}
	
	public String getCreatedBy() {
		return getFormattedFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_CREATEDBY);
	}

	public void setCreatedBy(String createdBy) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_CREATEDBY, createdBy);
	}

	public Date getModifiedDate() {
		return getFieldValueDate(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_MODIFIED);
	}

	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_MODIFIED, modifiedDate);
	}
	
	public String getModifiedBy() {
		return getFormattedFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_MODIFIEDBY);
	}

	public void setModifiedBy(String modifiedBy) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_MODIFIEDBY, modifiedBy);
	}
	
	public ComplexWorkRule getComplexWorkRule() {
		if (getRelatedObjects(COMPLEXWORKRULE_RELATIONSHIP_KEY).size() > 0) {
			return (ComplexWorkRule)getRelatedObjects(COMPLEXWORKRULE_RELATIONSHIP_KEY).iterator().next();
		}
		return null;
	}
	
	public void setComplexWorkRule(ComplexWorkRule cwr) {
		setComplexWorkRuleID(cwr.getFieldValueID(ComplexWorkRuleFieldInfo.Column.ID.getIndex()));
		setRelatedObjects(COMPLEXWORKRULE_RELATIONSHIP_KEY, Collections.singleton(cwr));
	}
	
	/**
	 * An array of integers from 0-25 representing the weeks that have been enabled for this work rule
	 * @return
	 */
	public HashSet<Integer> getWeeksEnabled() {
		if (weeksEnabled == null) {
			weeksEnabled = new HashSet<Integer>();
			
			//Convert weekenabledmask into a set of integers as weekenabledmask is a bit field
			int mask = getWeekEnabledMask();
			if (mask == -1) {	//The F&S client sets the mask to -1 for newly added assignments.  It treats this default value as having an active first week.
				weeksEnabled.add(0);
			} else {
				int i = 0;
				for (i = 0; i < 26; i++) {
					if (mask % 2 == 1) {
						weeksEnabled.add(i);
					}
					if (mask / 2 == 0) break;
					mask = mask / 2;
				}
			}
			
		}
		return weeksEnabled;
	}
	
	public void setWeeksEnabled(HashSet<Integer> weeksEnabled) {
		this.weeksEnabled = weeksEnabled;
		
		//Update the weekendabledmask integer that gets serialized to the DB
		int mask = 0;
		for (Integer week : weeksEnabled) {
			mask += Math.pow(2, week);
		}
		setWeekEnabledMask(mask);
	}

	public boolean isStartWeekOffsetModified() {
		return isStartWeekOffsetModified;
	}
	
	public void setStartWeekOffsetModified(boolean isStartWeekOffsetModified) {
		this.isStartWeekOffsetModified = isStartWeekOffsetModified;
	}

	public boolean isNumWeeksInRotationModified() {
		return isNumWeeksInRotationModified;
	}
	
	public void setNumWeeksInRotationModified(boolean isNumWeeksInRotationModified) {
		this.isNumWeeksInRotationModified = isNumWeeksInRotationModified;
	}

	public boolean isWeeksEnabledModified() {
		return isWeeksEnabledModified;
	}
	
	public void setWeeksEnabledModified(boolean isWeeksEnabledModified) {
		this.isWeeksEnabledModified = isWeeksEnabledModified;
	}
}
