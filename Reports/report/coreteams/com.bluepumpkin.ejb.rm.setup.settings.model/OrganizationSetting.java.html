	/**
	 * @param value : range 1 ~ MAX_TIME_OFF_CHOICES_COUNT.

	public void setTimeOffChoicesCount(int value) {
		if (value < 1) {

	public Date getRequiredPayPeriodHoursStartDate() {

		try {

			String results = (String) m_settings.get(RmSettingKey.ORG_RM_SETTINGS_REQUIRED_PAY_PERIOD_STARTDATE);
			if (results == null || results.isEmpty() || "0".equals(results)) {
				return new Date();
			}
			SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd");
			return simpleDateFormat.parse(results);

		} catch (Exception e) {
			throw new IllegalStateException(e);
		}

	}

	public void setRequiredPayPeriodHoursStartDate(Date dt) {
		SimpleDateFormat simpleDateFormat = new SimpleDateFormat("yyyy-MM-dd");
		simpleDateFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
		String formatted = simpleDateFormat.format(dt);
		m_settings.put(RmSettingKey.ORG_RM_SETTINGS_REQUIRED_PAY_PERIOD_STARTDATE, formatted);
	}

	private int getInt(String value, int defaultValue, String name) {
		try {
			if (!StringUtil.isEmptyOrWhiteSpace(value)) {
				return Integer.parseInt(value);
			}
		} catch (NumberFormatException nfe) {
			// This should be very rare as the QC was caused by defective automation and not a real world scenario.
			String msg = MessageFormat.format("Invalid RM setting value ''{0}'' for key ''{1}'', returning default ''{2}''.",
					value, name, defaultValue);
			LOG.debug(msg, nfe);
		}
		return defaultValue;
	}
