package com.bluepumpkin.web.core.setup.mediatypes;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.core.Log;
import com.bluepumpkin.web.core.base.RequestHandler;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

public class MediaTypesSetupRequestHandler extends RequestHandler {
	private static final Category LOG = Log.initCategory(MediaTypesSetupRequestHandler.class.getName());
    private static final String REQUIRED_PRIVILEGE = PrivilegeKeys.FS_CONFIGUREMEDIATYPES; 

	public void processRequest(RequestContext context) throws Exception {
		MediaTypesSetupPageModel model = new MediaTypesSetupPageModel(context);

		// Process requested action
		if (isAction(context, PageAction.SAVE)) {
			processSave(context, model);
		}

		context.setPageModel(model);

	}
	
	
	/**
	 * Process Save Action
	 */
	private void processSave(RequestContext context, MediaTypesSetupPageModel model) throws Exception 
    {        
        // Security check
        if (!isAuthorized(context, REQUIRED_PRIVILEGE)) 
        {
            model.addPageMessage(Message.ERROR_TYPE, CoreWebBundleKey.MEDIA_TYPES_SAVE_ERROR, CoreWebBundleKey.BUNDLE_NAME);
            LOG.error(CoreWebBundleKey.MEDIA_TYPES_SAVE_ERROR);
            return;
        }
        
		model.extractParameters(context.getRequest());
		if (model.validate()) 
        {
			try 
            {
                MediaTypesSetupModelHandler.updateTimeUnitBySID(model.getMediaUnits());
			} 
            catch (Exception ex) 
            {
				model.addPageMessage(Message.ERROR_TYPE, CoreWebBundleKey.MEDIA_TYPES_SAVE_ERROR, CoreWebBundleKey.BUNDLE_NAME);
				LOG.error(CoreWebBundleKey.MEDIA_TYPES_SAVE_ERROR, ex);
			}
		}
	}

}
