package com.bluepumpkin.ejb.rm.notification.notifyrules;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.IDStringPair;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.localization.DefaultLocalizationManager;
import com.bluepumpkin.ejb.bbm.notifyrules.rulecheckers.SelectorOptions;
import com.bluepumpkin.ejb.bbm.xmlrules.XmlRule;
import com.bluepumpkin.ejb.core.CoreManagerFactory;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.model.Request;
import com.witness.ejb.core.licensing.LicenseKeys;

/**
 * This class is used by the "Tracking->Alert Rules->Organization->create/edit" page to
 * display the selector options of the "Request Type" selector for the "Request Status Change
 * Rule" and the "Request Status Change Rule (batch) rule types. The possible options are
 * "TimeOff", "ShiftSwap", and "ShiftBid". We cannot hard-code these options in the
 * NOTIFYRULETEMPLATE.xml file because these options depend on which licenses are installed,
 * so we need to perform the check here.
 *
 * @author GQuintanar
 *
 */
public class RequestTypeSelector implements SelectorOptions {
	private static final Category m_cat = Log.initCategory(RequestTypeSelector.class.getName());
	private Localizer localizer;

	private static Map<Integer, String> REQUESTTYPE_VALIDVALUES_MAP = null;

	private enum RequestType {

		TIME_OFF(0, RmEjbBundleKey.TIME_OFF, Request.REQUESTTYPE_TIMEOFF, LicenseKeys.APP_RM_TOM),
		SHIFT_SWAP(1, RmEjbBundleKey.SHIFT_SWAP, Request.REQUESTTYPE_SHIFTSWAP, LicenseKeys.APP_RM_SS),
		SHIFT_BID(2, RmEjbBundleKey.SHIFT_BID, Request.REQUESTTYPE_SHIFTBID, LicenseKeys.APP_RM_SB),
		CUSTOM_SHIFT(3, RmEjbBundleKey.CUSTOM_SHIFT, Request.REQUESTTYPE_CUSTSHIFT, LicenseKeys.APP_RM_TOM),
		FLEX_TIME(4, RmEjbBundleKey.FLEX_TIME, Request.REQUESTTYPE_FLEXTIME, "");

		private final int optionValue;
		private final String bundleKey;
		private final String requestType;
		private final String licenseRequired;

		private RequestType(int value, String key, String requestType, String licenseRequired) {
			this.optionValue = value;
			this.bundleKey = key;
			this.requestType = requestType;
			this.licenseRequired = licenseRequired;
		}

		public int getOptionValue() {
			return optionValue;
		}

		public String getOptionName(Localizer localizer) {
			return localizer.i18n(RmEjbBundleKey.BUNDLE_NAME, bundleKey);
		}

		public String getRequestType() {
			return requestType;
		}

		String getLicenseRequired() {
			return licenseRequired;
		}
	}

	@Override
	public Collection getSelectorOptions(String selectorName, XmlRule rule) throws Exception {
		selectorName = selectorName.toLowerCase(); // NOSONAR
		ensureLocalizer();

		ArrayList<IDStringPair> options = new ArrayList<IDStringPair>();
		if ("rulerequesttype".equals(selectorName)) {
			for (RequestType aType : RequestType.values()) {
				if (checkLicense(aType.getLicenseRequired())) {
					options.add(new IDStringPair(new ID(aType.getOptionValue()), aType.getOptionName(localizer)));
				}
			}
		}

		return options;
	}

	/**
	 * Check whether a particular license is installed in the system.
	 *
	 * @param licenseKey
	 *            The name license key to check.
	 * @return true if the specified license is installed in the system, false otherwise.
	 */
	private static boolean checkLicense(String licenseKey) {
		if (licenseKey == null || licenseKey.isEmpty()) {
			//No license required (APP_ADVANCED_RM is now always enabled) 
			return true;
		}

		try {
			return CoreManagerFactory.getLicenseManager().isLicensed(licenseKey);
		} catch (Exception ex) {
			m_cat.error(ex);
		}
		return false;
	}

	public static Map<Integer, String> getRequestTypeMap() {

		if (REQUESTTYPE_VALIDVALUES_MAP == null) {
			REQUESTTYPE_VALIDVALUES_MAP = new HashMap<Integer, String>();
			for (RequestType aType : RequestType.values()) {
				if (checkLicense(aType.getLicenseRequired())) {
					REQUESTTYPE_VALIDVALUES_MAP.put(new Integer(aType.getOptionValue()), aType.getRequestType());
				}
			}
		}

		return REQUESTTYPE_VALIDVALUES_MAP;
	}

	/**
	 * Ensures there is a localizer. This was lazy loaded because DefaultLocalizationManager's initializer always ran
	 * and threw exceptions with powermock. The same code that works in another build environment fails here. We don't need it to run
	 * while unit testing, so just lazy load it.
	 */
	private void ensureLocalizer() {
		if (localizer == null) {
			localizer = DefaultLocalizationManager.getDefaultInstance().getLocalizer();
		}
	}
}
