	public void processRequest(RequestContext context){
		m_localizer = context.getLocalizer();

		m_amBundle = m_localizer.getBundle(AmWebBundleKey.BUNDLE_NAME);
		EditTimeRecordModel model = new EditTimeRecordModel(context);
		model.extractParameters(context.getRequest());

		setCurEmployee(model);

		if (isAction(context,EditTimeRecordModel.EDITTE_ACTION)
			|| isAction(context,EditTimeRecordModel.DELETETE_ACTION)){
			getTimeRecordEntry(model);
		} else if (isAction(context,EditTimeRecordModel.CREATE_END_SHIFT)){
			getTimeRecordEntry(model);
		} else if (isAction(context,EditTimeRecordModel.REFRESHEDIT_ACTION)
			|| isAction(context,EditTimeRecordModel.REFRESHINSERT_ACTION)){
			//create a tempEntry from parameters
			model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
		} else if (isAction(context,EditTimeRecordModel.INSERTTE_ACTION)){
			model.setActivityCategory("0");
			insertTempEntry(context, model);
		} else if (isAction(context,EditTimeRecordModel.OKDELETE_ACTION)){
			removeTimeRecordEntry(context, model);
		} else if (isAction(context,EditTimeRecordModel.OKEDIT_ACTION)){
			updateTimeRecordEntry(context, model);
		} else if (isAction(context,EditTimeRecordModel.OKINSERT_ACTION)){
			insertTimeRecordEntry(context, model);
		} else if (isAction(context,EditTimeRecordModel.EDITREMARK_ACTION)){
			editRemark(model);
		}  else if (isAction(context,EditTimeRecordModel.OKEDITREMARK_ACTION)){
			saveEditRemark(context, model);
		} else if (isAction(context,EditTimeRecordModel.OKENDSHIFT_ACTION)){
			saveEndShift(context, model);
		}
		//cache myTimeModel
		context.setAttribute(RequestContext.REQUEST_SCOPE, EditTimeRecordModel.MODEL_KEY, model);
	}

	/**
	 * setCurEmployee
	 * This method sets the current employee based on pagination
	 */
	private void setCurEmployee(EditTimeRecordModel model){
		//set employee
		model.setEmpId(model.getSelectedIDs()[model.getCurIndex()]);
	}

	/**
	 * getTimeRecordEntry()
	 */
	private void getTimeRecordEntry(EditTimeRecordModel model){
		try{
			TimeRecord aRecord = TimeRecordModelHandler.getTimeRecordById(model.getRecordID());
			ArrayList aList = (ArrayList)aRecord.getChildren();
			TimeRecordEntry anEntry = null;
			for (int i = 0; i < aList.size(); i++) {
				anEntry = (TimeRecordEntry) aList.get(i);
				if (this.compareIDWithDifferentDataTypes(anEntry,model.getEntryID())) {
					break;
				}
			}
			model.setAnEntry(anEntry);
			model.setActivityCategory(anEntry.getActivityCategoryID().toString());

			model.setActivityList(anEntry.getActivityID().toString());
			model.setHiddenMillisecond(anEntry.getStartTime().getTime());

			model.setIsOpenShift(aRecord.isOpenShift());

			Calendar aCal = Calendar.getInstance(TimeZone.getTimeZone(model.getTimeZone()));
			aCal.setTime(anEntry.getStartTime());
			model.setHiddenSecond(aCal.get(Calendar.SECOND));
		} catch (BbmObjectNotFoundException ex){
			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_FINDINGTE, AmWebBundleKey.BUNDLE_NAME);
		} catch (BbmFinderException ex){
			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_FINDINGTE, AmWebBundleKey.BUNDLE_NAME);
		} catch (Exception ex){
			cat.l7dError(AmWebLogBundleKey.MSG_ERROR_FINDINGTE, ex);
			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_FINDINGTE, AmWebBundleKey.BUNDLE_NAME);
		}
	}

	private boolean compareIDWithDifferentDataTypes(TimeRecordEntry entryFromDb, ID selectedEntryID) {
		return  (entryFromDb.getID().isLong() && entryFromDb.getID().equals(ID.fromLong(Long.parseLong(selectedEntryID + ""))))
				|| (entryFromDb.getID().isInt() && entryFromDb.getID().equals(selectedEntryID));
	}

	/**
	 * removeTimeRecordEntry()
	 */
	private void removeTimeRecordEntry(RequestContext context, EditTimeRecordModel model){
		try {
			TimeRecordModelHandler.removeTimeRecordEntry(new ID(model.getRecordID()),
					context.getUser().getID(), model.getPosition(), model.getRowToDelete(), model.getVersion());
			model.setPageAction(EditTimeRecordModel.DONE_ACTION);
			Message message = new Message(Message.RESPONSE_TYPE, AmWebBundleKey.MSG_DELETE_TIME_ENTRY, AmWebBundleKey.BUNDLE_NAME);
			TimeRecordUtil.setContextAttribute(context, RequestContext.APPLICATION_SCOPE,
				TimeRecordUtil.AM_WEB_SAVED_TIME_RECORD_COMMENT, message);
		} catch (BbmRemoveException ex){
			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_DELETINGTE, AmWebBundleKey.BUNDLE_NAME);
			model.setPageAction(EditTimeRecordModel.DELETETE_ACTION);
		} catch (BbmTimeRecordException ex){
			String errorCode = evaluateTimeRecordException(m_amBundle, ex, m_localizer);
			if (errorCode ==null){
				errorCode = m_localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_DELETINGTE);
			}
			model.addPageMessage(Message.ERROR_TYPE, errorCode);

			model.setPageAction(EditTimeRecordModel.DELETETE_ACTION);
		} catch (Exception ex){
			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_DELETINGTE, AmWebBundleKey.BUNDLE_NAME);
			model.setPageAction(EditTimeRecordModel.DELETETE_ACTION);
		}
	}

	/**
	 * updateTimeRecordEntry()
	 */
	private void updateTimeRecordEntry(RequestContext context, EditTimeRecordModel model){
		//UI validation: make sure that if it is an open record and last entry,
		//duration cannot be changed
		if (model.getIsLastEntry() && model.getIsOpenShift()
			&& (model.getDurationHour() != 0 || model.getDurationMinute() != 0)){
			//add UI message
			model.setPageAction(EditTimeRecordModel.EDITTE_ACTION);
			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_LASTENTRY_DURATION, AmWebBundleKey.BUNDLE_NAME);
			//set entry value
			model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
		} else {
			try {
				TimeRecord aRecord = TimeRecordModelHandler.updateTimeRecordEntry(context, model);
				model.setPageAction(EditTimeRecordModel.DONE_ACTION);
				Message message = new Message(Message.RESPONSE_TYPE, AmWebBundleKey.MSG_UPDATE_TIME_ENTRY, AmWebBundleKey.BUNDLE_NAME);
				TimeRecordUtil.setContextAttribute(context, RequestContext.APPLICATION_SCOPE,
					TimeRecordUtil.AM_WEB_SAVED_TIME_RECORD_COMMENT, message);
			} catch (BbmUpdateException ex){
				cat.l7dError(AmWebLogBundleKey.MSG_ERROR_UPDATINGTE, ex);

				model.addPageMessage(Message.ERROR_TYPE,
					AmWebBundleKey.MSG_ERROR_UPDATINGTE, AmWebBundleKey.BUNDLE_NAME);
				model.setPageAction(EditTimeRecordModel.EDITTE_ACTION);
				//set entry value
				model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
			} catch (BbmTimeRecordException ex){
				String errorCode = evaluateTimeRecordException(m_amBundle, ex, m_localizer);
				if (errorCode ==null){
					errorCode = m_localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_UPDATINGTE);
				}
				model.addPageMessage(Message.ERROR_TYPE, errorCode);

				model.setPageAction(EditTimeRecordModel.EDITTE_ACTION);
				//set entry value
				model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
			} catch (Exception ex){
				cat.l7dError(AmWebLogBundleKey.MSG_ERROR_UPDATINGTE, ex);

				model.setPageAction(EditTimeRecordModel.EDITTE_ACTION);
				model.addPageMessage(Message.ERROR_TYPE,
					AmWebBundleKey.MSG_ERROR_UPDATINGTE, AmWebBundleKey.BUNDLE_NAME);
				//set entry value
				model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
			}
		}
	}

	/**
	 * evaluateTimeRecordException()
	 */
	public static String evaluateTimeRecordException(ResourceBundle m_amBundle, BbmTimeRecordException ex, Localizer localizer){
		String errorCode = null;
		switch (ex.getErrCode())
		{
			case BbmTimeRecordException.TR_INFUTURE:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TEFUTURE);
			break;

			case BbmTimeRecordException.TR_OBSOLETE:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_MULTIUSER);
			break;

			case BbmTimeRecordException.TR_OVERLAP:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_OVERLAPSAVINGTE);
			break;

			case BbmTimeRecordException.TR_REPLACE_PRV:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TR_REPLACE_PRV);
			break;

			case BbmTimeRecordException.TR_REPLACE_NXT:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TR_REPLACE_NXT);
			break;

			case BbmTimeRecordException.TR_NOTBETWEEN:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TR_NOTBETWEEN);
			break;

			case BbmTimeRecordException.TR_END_BEFORE_START:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TR_END_BEFORE_START);
			break;

			case BbmTimeRecordException.TR_OVERLAP_RECORD:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TR_OVERLAP_RECORD);
			break;

			case BbmTimeRecordException.TR_INVALID:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TR_INVALID);
			break;

			case BbmTimeRecordException.TR_DUPLIC_SORTTIME:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TR_DUPLIC_SORTTIME);
			break;

			case BbmTimeRecordException.TR_REMOVED:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TR_REMOVED);
			break;

			case BbmTimeRecordException.TR_POSTED:
			errorCode = localizer.i18n(m_amBundle, AmWebBundleKey.MSG_ERROR_TR_POSTED);
			break;

		}
		//can add more error types here
		return errorCode;
	}

	/**
	 * saveEndShift()
	 */
	private void saveEndShift(RequestContext context, EditTimeRecordModel model){
		try {
			Calendar aCalendar = Calendar.getInstance(TimeZone.getTimeZone(model.getTimeZone()));
			aCalendar.setTime(model.getExtractedDate());
			Date aEndDate = aCalendar.getTime();

			if (aEndDate.after(new Date())){
				model.addPageMessage(Message.ERROR_TYPE,
					AmWebBundleKey.MSG_ERROR_ENDDATE_FUTURE, AmWebBundleKey.BUNDLE_NAME);
				model.setPageAction(EditTimeRecordModel.REFRESHCREATE_END_SHIFT);
				model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
			} else {
				TimeRecordModelHandler.createEndShift(model, context.getUser().getID());
				model.setPageAction(EditTimeRecordModel.DONE_ACTION);
			}
		} catch (AmRawEndShiftException ex){
			if (ex instanceof AmNotLastEndShiftException){
				model.addPageMessage(Message.ERROR_TYPE,
					AmWebBundleKey.MSG_ERROR_CREATINGENDSHIFT_NOTLASTENDSHIFT, AmWebBundleKey.BUNDLE_NAME);
			} else {
				model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_CREATINGENDSHIFT, AmWebBundleKey.BUNDLE_NAME);
			}

			model.setPageAction(EditTimeRecordModel.REFRESHCREATE_END_SHIFT);
			model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
		} catch (AmTimeRecordEndShiftException ex){

			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_CREATINGENDSHIFT_FORTR, AmWebBundleKey.BUNDLE_NAME);
			model.setPageAction(EditTimeRecordModel.REFRESHCREATE_END_SHIFT);
			model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
		} catch (AmException ex){

			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_CREATINGENDSHIFT, AmWebBundleKey.BUNDLE_NAME);
			model.setPageAction(EditTimeRecordModel.REFRESHCREATE_END_SHIFT);
			model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
		} catch (Exception ex){
			cat.l7dInfo(AmWebLogBundleKey.MSG_ERROR_CREATINGENDSHIFT , ex);

			model.addPageMessage(Message.ERROR_TYPE,
				AmWebBundleKey.MSG_ERROR_CREATINGENDSHIFT, AmWebBundleKey.BUNDLE_NAME);
			model.setPageAction(EditTimeRecordModel.REFRESHCREATE_END_SHIFT);
			model.setAnEntry(TimeRecordModelHandler.regenerateTempEntry(context, model));
		}
	}
