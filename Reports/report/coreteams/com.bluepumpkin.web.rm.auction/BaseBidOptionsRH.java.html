package com.bluepumpkin.web.rm.auction;

import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.base.RmAggregateException;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.MultiShiftBidException;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.ShiftBidRequest;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidrequest.model.UnsubmittedShiftBidPreference;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.keys.RmPageAction;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.l10n.RmWebLogBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.WebException;
import com.witness.web.uif.base.WorkpaneRequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.pagemodel.WorkpanePageModel;
import com.witness.web.uif.system.RequestContext;

/**
 * File:         BidOptionsRH
 * Description:  Request Handler for for the list of biddable schedules (Bid Options)
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       pbosin, David S.
 * @version      1.0
 * Created on July 29, 2002, 1:25 PM
 */
public class BaseBidOptionsRH extends WorkpaneRequestHandler {
	private static final Category LOG = Log.initCategory(BaseBidOptionsRH.class.getName());

	/**
	 * Process Request actions
	 */
	@Override
	public void processCustomRequest(RequestContext context) throws WebException {
		if (isAction(context, RmPageAction.POPUP_CREATE_BID_REQUEST_ACTION)) {
			processCreateMode(context);
		} else if (isAction(context, RmPageAction.POPUP_EDIT_BID_REQUEST_ACTION)) {
			processEditMode(context);
		} else if (isAction(context, RmPageAction.POPUP_VIEW_BID_REQUEST_ACTION)) {
			processViewMode(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processSaveRequest(context);
		} else if (isAction(context, RmPageAction.SAVE_UNSUBMITTED_BID_PREFERENCE)) {
			processSaveUnsubmittedBidPreference(context);
		} else if (isAction(context, RmPageAction.SUBMIT_MULTIPLE_BIDS)) {
			processSubmitMutlipleBids(context);
		}
	}

	private void processSubmitMutlipleBids(RequestContext context) {
		BaseBidOptionsPM model = (BaseBidOptionsPM) getPageModel(context);

		//save preferences first before processing 
		if (!processSaveUnsubmittedBidPreference(model)) {
			return;
		}

		try {
			ID bidderID = model.getBidderID();
			ID auctionID = model.getSelectedIDObject();
			ID employeeID = model.getEmployeeID();

			if (bidderID == null || auctionID == null || employeeID == null) {
				model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
				return;
			}

			AuctionMH.createRequestsFromUnsubmittedPreferences(context, auctionID, bidderID, employeeID);

		} catch (RmAggregateException aggregateException) {
			handleAggreageExceptionOnSave(context, model, aggregateException);
		} catch (Exception ex) {
			handleExceptionOnSave(model, ex);
		}
	}

	private void handleAggreageExceptionOnSave(RequestContext context, BaseBidOptionsPM model, RmAggregateException aggregateException) {
		for (Exception e : aggregateException.getExceptions()) {
			if (e instanceof MultiShiftBidException) {
				handleMultiBidExceptionOnSave(context, model, (MultiShiftBidException) e);
			} else {
				handleExceptionOnSave(model, e);
			}
		}
	}

	private static void handleMultiBidExceptionOnSave(RequestContext context, BaseBidOptionsPM model, MultiShiftBidException e) {
		Localizer lz = context.getLocalizer();
		String pageMessage = e.getPageMessage(lz, RmWebBundleKey.BUNDLE_NAME, RmWebBundleKey.UNABLE_TO_SAVE_REQUEST);
		Message msg = model.addPageMessage(Message.ERROR_TYPE, pageMessage);
		String associatedMessage = e.getAssociatedMessage(lz, context.getViewingTimeZone());
		msg.setAssociatedMessage(associatedMessage);
		LOG.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, e);
	}

	private void processSaveUnsubmittedBidPreference(RequestContext context) {
		BaseBidOptionsPM model = (BaseBidOptionsPM) getPageModel(context);
		processSaveUnsubmittedBidPreference(model);
	}	

	private boolean processSaveUnsubmittedBidPreference(BaseBidOptionsPM model) {
		ID bidderID = model.getBidderID();

		if (bidderID == null) {
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
			return false;
		}

		List<UnsubmittedShiftBidPreference> preferences = model.getUnsubmittedPreferencesToSave(bidderID);

		if (preferences == null) {
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
			return false;
		}


		try {
			AuctionMH.updateUnsubmittedSchedulePreference(preferences);
		} catch (Exception e) {
			handleExceptionOnSave(model, e);
			return false;
		}
		return true;
	}


	private static BidRequestFormPopupPM createModel(RequestContext context) {
		BidRequestFormPopupPM model = new BidRequestFormPopupPM(context);
		model.extractParameters(context.getRequest());
		model.setIsRefreshOpenerOnClose(true);
		context.setPageModel(model);
		return model;
	}

	/**
	 * Process create Action
	 */
	private static void processCreateMode(RequestContext context) {
		BidRequestFormPopupPM model = createModel(context);
		model.setMode(BidRequestFormPopupPM.MODE_CREATE);
	}

	/**
	 * Process Edit Action
	 */
	private static void processEditMode(RequestContext context) {
		BidRequestFormPopupPM model = createModel(context);
		model.setMode(BidRequestFormPopupPM.MODE_EDIT);
	}

	private static void processViewMode(RequestContext context) {
		BidRequestFormPopupPM model = createModel(context);
		model.setMode(BidRequestFormPopupPM.MODE_VIEW);
	}

	private static void processSaveRequest(RequestContext context) {
		BidRequestFormPopupPM model = createModel(context);

		ShiftBidRequest request = model.getShiftBidRequest();
		ID requestID = request.getID();
		String comment = model.getComment();

		try {
			if (requestID==null) {
				requestID = AuctionMH.createRequest(request, comment);
			} else {
				AuctionMH.updateRequest(request, comment);
			}
			model.setSavedRequestID(requestID);
		} catch(RmHardValidationException ex) {
			handleHardValidationOnSave(context, model, ex);
		} catch(Exception ex) {
			handleExceptionOnSave(model, ex);
		}
	}

	private static void handleExceptionOnSave(PageModel model, Exception ex) {
		model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
		LOG.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_DATA, ex);
	}

	private static void handleHardValidationOnSave(RequestContext context, PageModel model, RmHardValidationException ex) {
		Message msg = model.addPageMessage(Message.ERROR_TYPE,
				RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
		msg.setAssociatedMessage(ex.getLocalizedMessage(context.getLocalizer(), context.getViewingTimeZone()));
		LOG.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
	}

    /**
	 * Return Page Model to perform generic action request on
	 */
	@Override
	public WorkpanePageModel getPageModel(RequestContext context) {
		return new BidOptionsPM(context);
	}
}
