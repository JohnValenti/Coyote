	private void initAuction() {
		m_auction = new ShiftBidAuction(ShiftBidAuction.DL_BASIC);
		m_auction.setUsesScoring(false);
		m_auction.setStatus(ShiftBidAuction.STATUS_OPEN_FOR_BIDDING);

		ID empID = m_context.getUser().getEmployeeID();
		if (empID != null) {
			m_auction.setCreatorID(empID);
		}
		try{
			userScopeOrgIds =OrganizationModelHandler.getAllOrgIdsInUserScope(m_context, m_context.getUser());
		}catch (Exception e) {
			log.debug("Cannot get Org Ids in user Scope");
		}
	}

	/**
	 * Make sure that the model is populated
	 */
	@Override
	protected void loadData() throws Exception {
		try {
			if (m_context.getUser().getEmployeeID() == null) {
				addPageMessage(Message.INFO_TYPE, UIFWebBundleKey.SCREEN_DISABLED_FOR_NON_EMP, UIFWebBundleKey.BUNDLE_NAME);
				m_isPageEnabled = false;
				return;
			}
			viewableOrgsInSP = OrganizationModelHandler.getOrganizationsByIDs(m_context, userScopeOrgIds);

			if (hasSelectedID()) {
				setIsAddMode(false);
				ID auctionID = getSelectedIDObject();
				m_auction = AuctionUtil.getAuctionAndValidate(auctionID, m_context, this);
				ID spId = m_auction.getSPID();
				m_campaign = m_auction.getOptMethods().getCampaign();
				m_spSelectorPC.init(spId);
				viewableOrgsInSP=AuctionMH.getLinkedViewableOrgsBySid(m_context,spId,userScopeOrgIds);
			} else {
				setIsAddMode(true);
				//The SpId can be populated with same information in the case of selecting auction then click Create
				ID spId = m_auction.getSPID();
				if(spId!=null){
					viewableOrgsInSP=AuctionMH.getLinkedViewableOrgsBySid(m_context,spId,userScopeOrgIds);
				}
			}
		} catch (RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataErrorAndLogAsDebug(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
		}
	}

	/**
	 * Return true if current user is authorized to configure entities in a page
	 *
	 * NOTE: To be overwritten by subclasses.
	 *
	 * Default return true
	 */
	@Override
	protected boolean isAuthorizedToConfigure() {
		boolean result = false;

		if (m_isPageEnabled) {
			User user = m_context.getUser();
			if (isAddMode()) {
				result = user.isAuthorized(PrivilegeKeys.SB_MODIFYAUCTIONS);
			} else if (m_campaign != null) {
				result = user.isAuthorized(PrivilegeKeys.SB_MODIFYAUCTIONS, CampaignScopeManagerProxy.getScopeID(m_campaign.getID()));
			}
		}

		return result;
	}

	/**
	 * Override extract Parameters to handle date input
	 */
	@Override
	public boolean extractParameters(HttpServletRequest request) {
		boolean success = super.extractParameters(request);

		Date deadline = m_datePickerPC.extractDate(DEADLINE_FN);
		m_auction.setDeadlineDate(deadline);

		ID spID = m_spSelectorPC.getSelectedID();
		m_auction.setSPID(spID);

		if (isAdvancedLicense) {
			//minMaxSelected is set by the checkbox for hours per day. If it is checked, then set the
			//min max values. Otherwise, set the to 0.
			if (minMaxSelected) {
				m_auction.setMinHoursPerDay(this.minHours);
				m_auction.setMaxHoursPerDay(this.maxHours);
			} else {
				m_auction.setMinHoursPerDay(0);
				m_auction.setMaxHoursPerDay(0);
			}

			//maxBlocksSelected is set by the checkbox for Blocks Per Bid.
			// If it is checked, then set the max value. Otherwise, set it to 0.
			if (maxBlocksSelected) {
				m_auction.setMaxShiftBlocksPerBid(this.maxBlocks);
			} else {
				m_auction.setMaxShiftBlocksPerBid(0);
			}

			//maxConsecDaysSelected is set by the checkbox for Consecutive Days Per Bid.
			// If it is checked, then set the max value. Otherwise, set it to 0.
			if (maxConsecDaysSelected) {
				m_auction.setMaxConsecDaysPerBid(this.maxConsecDays);
			} else {
				m_auction.setMaxConsecDaysPerBid(0);
			}

			//The value of checkbox  is extracted as below: if the checkbox is unchecked, the parameter value is null
			//if the checkbox is checked, the parameter value is equals to the assigned value
			String isRequiredBidSubmission = m_context.getParameter(REQUIRED_BIDS_FN);
			String isDisplayBidApprovalOrder = m_context.getParameter(DISPLAY_BASED_ON_FN);

			m_auction.setIsRequiredBids(!StringUtil.isEmpty(isRequiredBidSubmission));
			m_auction.setIsDisplayBasedOn(!StringUtil.isEmpty(isDisplayBidApprovalOrder));
			if(isAddMode()){
				//Only extract in Add mode. No need to extract OrganizationId again since the field is disabled in Update mode
				String orgOption = m_context.getParameter(ORGANIZATION_OPTION_FN);
				String organizationId = m_context.getParameter(ORGANIZATION_ID_FN);
				if(!StringUtil.isEmpty(orgOption)&& !Boolean.valueOf(orgOption)&&!StringUtil.isEmpty(organizationId)){
					m_auction.setOrganizationId(Integer.valueOf(organizationId));
				} else {
					m_auction.setFieldNull(ShiftBidAuctionFieldInfo.SHIFTBIDAUCTION_N_ORGANIZATIONID);
				}
			}

		}
		return success;
	}

	/**
	 * Validates the Page Component
	 *
	 * NOTE: This method recurse through all sub-components
	 */
	public boolean validate(ShiftBidAuction existingAuction) {
		boolean success = super.validate();

		if (isAddMode() && m_auction.getSPID() == null) {
			addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.MISSING_SCHEDULING_PERIOD, RmWebBundleKey.BUNDLE_NAME);
			success = false;
		}

		if(isAdvancedLicense&&isAddMode()&&m_auction.getOrganizationId()==0){
			Collection<ID> orgIDs= Collections.emptyList();

			try {
				orgIDs = CampaignModelHandler.getLinkedOrgsforSP(m_context,m_auction.getSPID());

			} catch (BbmFinderException e) {
				log.debug("Cannot find orgIds linked by the scheduling period"+m_auction.getSPID());
			} catch (RemoteException e) {
				log.debug("Cannot find orgIds linked by the scheduling period"+m_auction.getSPID());
			} catch (BbmException e) {
				log.debug("Cannot find orgIds linked by the scheduling period"+m_auction.getSPID());
			}

			if(!userScopeOrgIds.containsAll(orgIDs)){
				addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.VALIDATE_ORGANIZATION_SP_CONCURRENT_AUCTION, RmWebBundleKey.BUNDLE_NAME);
				success = false;
			}
		}

		if(isAdvancedLicense &&
			 ((existingAuction==null && m_auction.getIsMultiShift()) ||(existingAuction!=null && existingAuction.getIsMultiShift()))) {
				if (m_auction.getMinHoursPerDay() != 0 && m_auction.getMaxHoursPerDay() != 0
						&& m_auction.getMinHoursPerDay() > m_auction.getMaxHoursPerDay()) {
					addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.MIN_HOURS_GREATER_THAN_MAX, RmWebBundleKey.BUNDLE_NAME);
					success = false;
				}

				if (minMaxSelected && this.maxHours == 0 && this.minHours == 0) {
					addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.MIN_MAX_HOURS_GREATER_THAN_ZERO, RmWebBundleKey.BUNDLE_NAME);
					success = false;
				}

				if (maxBlocksSelected && m_auction.getMaxShiftBlocksPerBid() == 0) {
					addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.MAX_BLOCKS_NOT_VALID, RmWebBundleKey.BUNDLE_NAME);
					success = false;
				}
		}
		if (isAdvancedLicense &&
				 ( (existingAuction==null && !m_auction.getIsFullPeriodSchedule()) ||(existingAuction!=null && !existingAuction.getIsFullPeriodSchedule()))) {
				if (maxConsecDaysSelected && ((0 >= m_auction.getMaxConsecDaysPerBid()) || (m_auction.getMaxConsecDaysPerBid() > 21))) {
					addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.MAX_CONSEC_DAYS_NOT_VALID, RmWebBundleKey.BUNDLE_NAME);
					success = false;
				}
		}

		return success;
	}

	private void addNameRow() {
		String rbKey = RmWebBundleKey.RM_SA_AUCTION_NAME;
		m_inputPC.reset();
		m_inputPC.setInputFieldDesc(rbKey, RmWebBundleKey.BUNDLE_NAME);
		m_inputPC.setInputFN(AUCTION_NAME_FN);
		m_inputPC.setIsReadOnly(!isAuthorizedToConfigure());
		m_inputPC.setInputValue(HtmlUtil.escapeAttributeValue(m_auction.getName()));
		m_inputPC.setIsRequired(true);
		m_inputPC.setMaxLength(DBInfo.SHIFTBIDAUCTION_NAME_MAXLENGTH);
		m_inputPC.initForDisplay();
		m_formPC.addRow(i18n(m_bundle, rbKey), m_inputPC.getHtmlDisplay());
	}

	private void addStatusRow() {
		String label = i18n(m_common_bundle, UIFWebBundleKey.STATUS);
		String status = i18n(m_common_bundle, UIFWebBundleKey.STATUS_OPEN);
		if (!m_auction.isStatusOpen() && !isAddMode()) {
			status = i18n(m_common_bundle, UIFWebBundleKey.STATUS_CLOSED);
		}
		status += HtmlUtil.makeHiddenInput(AUCTION_STATUS_FN, m_auction.getStatus());
		m_formPC.addRow(label, status);
	}

	private void addCampaingRows(boolean isEditable) {
		m_spSelectorPC.setIsRequired(true);
		m_spSelectorPC.setIsEditable(isEditable&&isAuthorizedToConfigure());
		if(isAdvancedLicense) {
			m_spSelectorPC.setOnChange(ON_CHANGE_SCHEDULING_PERIOD);
		}

	private void initChoiceInput(String fn, boolean isEditable, boolean isChecked, String jsString) {
		m_choiceInput.reset();
		m_choiceInput.setValue(isEditable ? "true" : String.valueOf(isChecked));
		m_choiceInput.setIsChecked(isChecked);
		m_choiceInput.setOnClickJS(StringUtil.isEmpty(jsString) ? JSUtil.ON_CHANGE : jsString);

	private String createDisplayDropdownList(String fieldName, boolean isEnabled) {

		int showOption =showOption(m_auction);
		Collection<StringsPair> displayOptions = AuctionUtil.getDisplayOptions(showOption, m_context);
		String selected =getSelecteValueForDropdown(displayOptions,String.valueOf(m_auction.getValueDisplayBasedOn()),String.valueOf(ShiftBidder.RANKBY_SENIORITY));
		String[] selectedValue = new String[] { selected };
		return HtmlUtil.makeSelectInput(fieldName, selectedValue, JSUtil.ON_CHANGE,
				AuctionUtil.getDisplayOptions(showOption, m_context), false, isEnabled);
	}
	private int showOption(ShiftBidAuction auction){
		if(auction.getIsRequiredBids()) {
			return AuctionUtil.SHOW_ONLY_RANK;
		} else if (auction.getUsesScoring()&&!isAddMode()) {
			return AuctionUtil.SHOW_ALL;
		} else {
			return AuctionUtil.SHOW_ALL_EXCEPT_SCORE;
		}
	}


	protected String getSelecteValueForDropdown(Collection<StringsPair> options, String selectedId,String defaultValue) {
		for (StringsPair sp: options) {
			if (sp.getKey().equals(selectedId)) {
				return sp.getKey();
			}
		}
		return defaultValue;
	}

	@SuppressWarnings("unchecked")
	private Collection<StringsPair> getEmpTypes(){
		Collection<StringsPair> empTypes =new ArrayList<StringsPair>();
		Collection<StringsPair> tempEmpTypes =new ArrayList<StringsPair>();
		ProfileModelHandler mh = new ProfileModelHandler();
		empTypes.add(new StringsPair(DEFAULT_EMPLOYEE_TYPE,getStringFromBundle(RmWebBundleKey.REQUEST_TYPE_ALL)));
		try{
			if(m_auction.getOrganizationId()!=DEFAULT_SELECTED_ORG_ID){
				empTypes.addAll(mh.getEmployeeTypesForOrg(new ID(m_auction.getOrganizationId()), m_context.isInWhatIfMode()));
			} else {//All
				for (Organization org: viewableOrgsInSP) {
					tempEmpTypes.addAll(mh.getEmployeeTypesForOrg(org.getID(), m_context.isInWhatIfMode()));
				}
				//Handle duplicated empTypes since sub organizations will inherit employee types from its parent organization
				for (StringsPair empType: tempEmpTypes) {
					if(!empTypes.contains(empType)) {
						empTypes.add(empType);
					}
				}
			}

		} catch (Exception e) {
			log.error("Cannot get Employee Type by selected Organization Id"+selectedOrgId);
		}

		return empTypes;
	}
