package com.bluepumpkin.web.rm.auction;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Iterator;
import java.util.ResourceBundle;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.base.RmHardValidationException;
import com.bluepumpkin.ejb.rm.requests.shiftbid.shiftbidauction.model.ShiftBidAuction;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.pagecomponent.DefaultNodeData;
import com.witness.web.uif.pagecomponent.list.SingleColListPC;
import com.witness.web.uif.pagecomponent.selection.SelectablePC;
import com.witness.web.uif.pagemodel.SelectionpanePageModel;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

/**
 * File:         AuctionSelectionPM
 * Description:  Page model for the selection pane with the list of shift bidding auctions
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       pbosin, David Su
 * @version      1.0
 * Created on July 29, 2002, 12:53 PM
 */
public class AuctionSelectionPM extends SelectionpanePageModel {
	private static final Category m_log = Log.initCategory(AuctionSelectionPM.class.getName());
	private static String FORM_ACTION = "auction_selection";

	private Collection m_auctions = Collections.emptyList();
	private SingleColListPC m_auctionListPC;
	private ResourceBundle m_bundle;

	/** Constructor */
	public AuctionSelectionPM(RequestContext context) {
		super(context);
		m_bundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);

		// Create List
		m_auctionListPC = new SingleColListPC(context);
		addChildComponent(m_auctionListPC);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static AuctionSelectionPM getInstance(RequestContext context) throws Exception {
		return (AuctionSelectionPM)getInstance(context, AuctionSelectionPM.class);
	}

	/**
	 * Initialize Model with Data
	 */
	@Override
	protected void initializeModel() {
		if (m_isInitialized) {
			return;
		} else {
			m_isInitialized = true;
		}
		try {
			m_auctions = AuctionMH.getAuctionsForManagerWithCampaign(m_context.getUser());
		} catch(RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataErrorAndLogAsDebug(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
		} catch (Exception ex) {
			handleUnableToLoadDataError(m_log, ex);
		} finally {
			setHeaderTitle(i18n(m_bundle, RmWebBundleKey.RM_SA_AUCTION_NAME));
			initToolbar();
			initList();
		}
	}

	protected void initList() {
		if (m_auctions.isEmpty()) {
			return;
		}

		int index = 0;
		ArrayList preselected = new ArrayList();
		Collection listModel = new ArrayList(m_auctions.size());
		for (Iterator it = m_auctions.iterator(); it.hasNext();) {
			ShiftBidAuction auction = (ShiftBidAuction) it.next();
			String rowID = auction.getID().toString();
			String name = auction.getName();
			DefaultNodeData node = new DefaultNodeData(rowID, HtmlUtil.escapeAttributeValue(name), "");
			listModel.add(node);

			if (rowID.equals(getSelectedID())) {
				preselected.add("r" + index);
			}
			index++;
		}
		if (!preselected.isEmpty()) {
			m_auctionListPC.setSelectedIndexes(preselected);
		}

		m_auctionListPC.setSubmitOnSelect(true);
		m_auctionListPC.setListModel(listModel);
	}

	/**
	 * Initialize Toolbar
	 */
	@Override
	protected void initToolbar() {
		useDefaultToolbar(false, false);
	}

	/**
	 * Return the Selectable Component
	 */
	@Override
	public SelectablePC getSelectablePC() {
		return m_auctionListPC;
	}

	/**
	 * Return URL for HTML Form Action
	 */
	@Override
	public String getFormAction() {
		return FORM_ACTION;
	}
}
