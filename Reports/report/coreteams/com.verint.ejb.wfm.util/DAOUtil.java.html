package com.verint.ejb.wfm.util;

import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;

public class DAOUtil {

	private DAOUtil() {

	}

	/**
	* 
	* @param ids A collection of integer IDs (Does not currently handle string IDs)
	* @return
	* A string that is of the form <IDlist><ID>678</ID><ID>666</ID></IDlist>
	* Note that only integer IDs are handled. 
	* Handling string will require escaping for XML and is currently not supported. 
	*/
	public static String converIdsToXml(Collection<ID> ids) {
		StringBuilder xml = new StringBuilder();
		xml.append("<IDlist>");
		for (ID id : ids) {
			xml.append("<ID>").append(id.toInt()).append("</ID>");
		}
		xml.append("</IDlist>");
		return xml.toString();
	}



}
