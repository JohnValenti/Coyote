public class TimeOffDataFacadeImpl implements ITimeOffDataFacade {
	private static final Category logger = Log.initCategory(TimeOffDataFacadeImpl.class.getName());

	private static TORequestManager toMgr = null;

	@Override
	public TimeOffDataEntity getTimeOffDataById(RequestContext context, ID empId, User user) throws WFOException {
		try {
			if (user == null) {
				throw new BusinessWFOException("null user passed to getTimeOffDataByID", Response.Status.CONFLICT);
			}

			ID orgID = OrganizationModelHandler.getEmployeeOrgID(context, empId);
			if (toMgr == null) {
				toMgr = RmUtils.getTORequestManager();
			}
			Collection<Activity> toTypes = toMgr.getTimeOffTypes(orgID);
			Set<ID> activityIDs = toTypes.stream().map(t -> t.getID()).collect(Collectors.toSet());

			Collection<ActivityProperties> activityProperties = ValidationModelHandler.findPropertiesForActivities(context, activityIDs);
			Map<ID, ActivityProperties> apLookup = activityProperties.stream().collect(Collectors.toMap(p -> p.getActivityId(), p -> p));

			// TOP by activity means waitlists are now by activity
			Map<ID, TOPool> toPoolsByActivityID = getTOPoolsByActivityID(empId);
			boolean isUserAuthToWaitlist = TORequestUtil.isTOWaitlistFeatureEnabled() && isAuthToWaitlist(user);

			TimeOffDataEntity toData = new TimeOffDataEntity();
			toData.setEmployeeId(empId.toInt());
			toData.setRequestableActivities(toTypes, apLookup, isUserAuthToWaitlist, toPoolsByActivityID);
			toData.setDebitType(getDebitTypes());

			return toData;
		} catch (BusinessWFOException ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw ex;
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}

	private int getDebitTypes() {
		// in DB: 0=use my timeoff; 1=schedule around; 2=both
		// for rest service: 1=use my timeoff; 2=schedule around; 3=both
		int reqTypeAllowedVal = 2;
		try {
			reqTypeAllowedVal = BbmManagerFactory.getDBConfigManager().getIntValue(ConfigKey.TIMEOFF_REQUEST_TYPES_ALLOWED);
		} catch (BbmEJBCreateException | RemoteException e) {
		}

		return ++reqTypeAllowedVal;
	}

	/**
	 * Gets a map of TOPool by activity ID for the employee's TOPool assignments.
	 * The default TOPool will have a null activity ID, thus the resulting map will have an entry for null.
	 */
	private Map<ID, TOPool> getTOPoolsByActivityID(ID empId) throws Exception {
		Date date = new Date();
		List<ID> employeeIDs = Collections.singletonList(empId);
		@SuppressWarnings("unchecked")
		Map<ID, List<EmpTOPoolAssignment>> empPoolsByEmpID = PeopleMH.getTOPoolsForWorkResources(employeeIDs, date, date);

		List<EmpTOPoolAssignment> empPoolAssignments = new ArrayList<>();
		if (empPoolsByEmpID.containsKey(empId)) {
			empPoolAssignments = empPoolsByEmpID.get(empId);
		}
		Collection<ID> toPoolIDs = empPoolAssignments.stream().map(e -> e.getTOPoolID()).collect(Collectors.toSet());
		Collection<TOPool> toPools = TOPoolUtil.getTOPoolsByIDs(toPoolIDs);
		Map<ID, TOPool> toPoolsByID = toPools.stream().collect(Collectors.toMap(p -> p.getID(), p -> p));
		Map<ID, TOPool> toPoolsByActivityID = new HashMap<>();
		for (EmpTOPoolAssignment assignment : empPoolAssignments) {
			if (toPoolsByID.containsKey(assignment.getTOPoolID())) {
				toPoolsByActivityID.put(assignment.getActivityID(), toPoolsByID.get(assignment.getTOPoolID()));
			}
		}
		return toPoolsByActivityID;
	}

	/**
	 * Return true if User is authorized to waitlist a Request.
	 */
	private boolean isAuthToWaitlist(User user) {
		return user.isAuthorized(PrivilegeKeys.TOM_ADDREQUESTTOWAITLIST_ID);
	}
}
