package com.verint.web.rest.wfm.resource.calendareventassignment;

import com.bluepumpkin.common.util.StringUtil;
import com.google.common.collect.ImmutableList;
import com.verint.web.rest.wfm.exception.EntityException;


public class InvalidCalendarEventTypeException extends EntityException {

	InvalidCalendarEventTypeException(String entityName, String invalidEventTypeFilterParam) {
		super(entityName, invalidEventTypeFilterParam);
	}

	public String getTitle() {
		return "InvalidCalendarEventTypeException";
	}

	@Override
	public String getDetail() {
		return "An error occurred when parsing the eventType filter parameter.  Valid eventType parameters are: " +
				StringUtil.createDelimitedString(ImmutableList.of(
						CalendarEventType.all.name(),
						CalendarEventType.calendar.name(),
						CalendarEventType.timeoff.name(),
						CalendarEventType.unavailability.name()), StringUtil.DELIM);
	}
}
