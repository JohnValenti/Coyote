package com.verint.wfm.ws.entityImport.activityschedulingperiodresourceconstraint.model;

import java.util.ArrayList;
import java.util.Collection;

import javax.xml.bind.annotation.XmlType;

import com.verint.wfm.ws.entityImport.activityschedulingperiodresourceconstraint.ResourceConstraintNotPersistedInfo;
import com.verint.wfm.ws.entityImport.schedulingPeriod.model.SchedulingPeriodDTO.SchedulingPeriodExternalNaturalKey;

@XmlType(name = "ResourceConstraintDTO", propOrder = {
		"resourceConstraintExternalKey", "minChair", "maxChair" })
public class ResourceConstraintDTO {

	protected ResourceConstraintExternalKey m_resourceConstraintExternalKey;
	protected Integer m_minChair;
	protected Integer m_maxChair;

	public ResourceConstraintDTO() {
	}

	public ResourceConstraintExternalKey getResourceConstraintExternalKey() {
		return m_resourceConstraintExternalKey;
	}

	public void setResourceConstraintExternalKey(ResourceConstraintExternalKey resourceConstraintExternalKey) {
		m_resourceConstraintExternalKey = resourceConstraintExternalKey;
	}

	public Integer getMinChair() {
		return m_minChair;
	}

	public void setMinChair(Integer minChair) {
		m_minChair = minChair;
	}

	public Integer getMaxChair() {
		return m_maxChair;
	}

	public void setMaxChair(Integer maxChair) {
		m_maxChair = maxChair;
	}

	public void validate(
			ResourceConstraintNotPersistedInfo activitySchedulingPeriodResourceConstraintNotPersistedInfo) {
		if (this.m_minChair != null && m_minChair.intValue() < 0) {
			activitySchedulingPeriodResourceConstraintNotPersistedInfo
					.addInvalidActivityMinSeatConstraintMessage(this.m_minChair);
		}
		if (this.m_maxChair != null && m_maxChair.intValue() < 0) {
			activitySchedulingPeriodResourceConstraintNotPersistedInfo
					.addInvalidActivityMaxSeatConstraintMessage(this.m_maxChair);
		}
	}

	static public Collection<SchedulingPeriodExternalNaturalKey> convertToSchedulingPeriodExternalNaturalKeys(
			Collection<ResourceConstraintDTO> constraintDTOs) {
		Collection<SchedulingPeriodExternalNaturalKey> result = new ArrayList<SchedulingPeriodExternalNaturalKey>();
		for (ResourceConstraintDTO constraintDTO : constraintDTOs) {
			result.add(constraintDTO.getResourceConstraintExternalKey()
					.getSchedulingPeriodExternalNaturalKey());
		}
		return result;
	}

	@XmlType(name = "ResourceConstraintExternalKey", propOrder = { "activityName",
			"schedulingPeriodExternalNaturalKey" })
	public static class ResourceConstraintExternalKey {

		protected String m_activityName;
		protected SchedulingPeriodExternalNaturalKey m_schedulingPeriodExternalNaturalKey;

		public ResourceConstraintExternalKey() {
		}

		public ResourceConstraintExternalKey(String activityName,
				SchedulingPeriodExternalNaturalKey schedulingPeriodExternalNaturalKey) {
			m_schedulingPeriodExternalNaturalKey = schedulingPeriodExternalNaturalKey;
			m_activityName = activityName;
		}

		public String getActivityName() {
			return m_activityName;
		}

		public void setActivityName(String activityName) {
			m_activityName = activityName;
		}

		public SchedulingPeriodExternalNaturalKey getSchedulingPeriodExternalNaturalKey() {
			return m_schedulingPeriodExternalNaturalKey;
		}

		public void setSchedulingPeriodExternalNaturalKey(
				SchedulingPeriodExternalNaturalKey schedulingPeriodExternalNaturalKey) {
			m_schedulingPeriodExternalNaturalKey = schedulingPeriodExternalNaturalKey;
		}

		@Override
		public String toString() {
			StringBuilder b = new StringBuilder();
			b.append("[schedulingPeriodExternalNaturalKey: ");
			b.append(m_schedulingPeriodExternalNaturalKey);
			b.append(", activity: ");
			b.append(m_activityName);
			b.append(']');
			return b.toString();
		}
	}
}
