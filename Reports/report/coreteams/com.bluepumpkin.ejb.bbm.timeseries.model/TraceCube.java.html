/**
 * An instance of this class stores one or more types of time-dependent data
 * and allows it to be set and queried at various intervals.  See Trace for a
 * list of the types of data supported.
 * <p/>
 * This class and related classes such as
 * com.bluepumpkin.ejb.bbm.timeseries.util.TraceUtil
 * and com.bluepumpkin.ejb.bbm.timeseries.util.TraceOperator
 * are intended to be solely responsible for business logic for time-based
 * business data related to media, such as call volume, average handling time,
 * connect rate, etc.
 * <p/>
 * TraceCube is an abstract class.  Concrete subclasses must minimally provide a
 * <code>newInstance()</code> method that creates a new instance of that subclass.
 * <p/>
 * Time intervals and index ranges in TraceCube and related classes use
 * inclusive end bounds.  This means that if you want to access all of a trace's values
 * on January 8th, your start time should be 00:00:00 on January 8th and your
 * end time should be 23:59:59 on January 8th, NOT 00:00:00 on January 9th.
 */
public abstract class TraceCube extends Trace {

	/**
	 * TraceCube is a container for types of Trace, given period
	 *
	 * @param ID,      queueID
	 * @param Date,    startDate,
	 * @param Date,    endDate,
	 * @param short[], the Trace Types to be included
	 * @param TimeZone
	 */
	protected TraceCube(ID queueID, Date startDate, Date endDate, short[] types) {
		m_QueueID = queueID;
		m_RawStartDate = startDate;
		m_RawEndDate = endDate;
		if (startDate == null || endDate == null) {
			m_Length = 1;
			m_rawStartOffset = 0;
			m_rawEndOffset = 0;
		} else {
			InsertionPoint result = TraceUtil.getInsertionPoint(startDate);
			m_StartDate = result.startDate;
			m_StartOffSet = result.offset;

			result = TraceUtil.getInsertionPoint(endDate);
			m_EndDate = result.startDate;
			m_EndOffSet = result.offset;
			int days = TimeZoneUtil.numberOfDays(m_StartDate, m_EndDate);
			m_Length = DAYPOINTS * (days - 1) - m_StartOffSet + m_EndOffSet + 1;
			m_rawStartOffset = TraceUtil.gapOffSet(m_StartDate, m_StartOffSet, m_RawStartDate);
			// End time parameter is exclusive but end offset is inclusive 
			// so we have to adjust the time slightly.
			m_rawEndOffset = TraceUtil.gapOffSet(m_StartDate, m_StartOffSet, new Date(m_RawEndDate.getTime() - 1000));
		}

		if (types != null) {
			m_Types = types;
			// If startDate and endDate are within 15 minutes, assign 1 slot
			for (int i = 0; i < types.length; i++) {
				m_TraceValue[types[i]] = new double[m_Length];
				//init the value to Trace.TRACENA;
				initIntArray(m_TraceValue[types[i]]);
			}
		}
	}

	public double[] getTraceValueD(short type, int start, int end) {
		if (m_Types == null || start > end || m_TraceValue[type] == null)
			return null;
		if (start == end) {
			double[] result = {getTraceValueD(type, start)};
			return result;
		}
		double[] result = new double[end - start + 1];
		System.arraycopy(m_TraceValue[type], start, result, 0, result.length);
		return result;
	}

	public double[] getTraceValueD(short type, Date startDate, Date endDate) {
		if (m_Types == null || startDate == null || endDate == null || startDate.after(endDate))
			return null;
		if (startDate.equals(endDate)) {
			double[] pointResult = {getTraceValueD(type, startDate)};
			return pointResult;
		}
		int start = TraceUtil.gapOffSet(m_StartDate, m_StartOffSet, startDate);
		int end = TraceUtil.gapOffSet(m_StartDate, m_StartOffSet, endDate);
		return getTraceValueD(type, start, end);
	}

	/**
	 * Get the maximum length of TraceValue for several Trace Types
	 *
	 * @param short[], types
	 * @return int, length
	 */
	public int getMaximumLength(short[] types) {
		int length = 0;
		if (types == null)
			return length;
		for (int i = 0; i < types.length; i++) {
			double[] value = getTraceValueD(types[i]);
			if (value == null)
				continue;
			length = (value.length > length) ? value.length : length;
		}
		return length;
	}

	public void setTraceValue(short type, double[] value, int offSet) {
		if (m_Types == null || m_TraceValue == null || value == null || m_TraceValue[type] == null)
			return;
		TraceUtil.adjustTraceValue(type, value);      // change -1 to value of TRACENA
		int length = value.length;
		// Check if expansion is needed
		if (length + offSet > m_TraceValue[type].length) {
			double[] result = new double[length + offSet];
			System.arraycopy(m_TraceValue[type], 0, result, 0, m_TraceValue[type].length);
			System.arraycopy(value, 0, result, offSet, length);
			m_TraceValue[type] = result;
		} else {
			System.arraycopy(value, 0, m_TraceValue[type], offSet, length);
		}
		setTypeDirty(type, true);
	}

			setTraceValue(type[i], value[i], offSet, true);
	}

	public int getTimeOffsetWithinCubeBounds(Date time) {

	// Convert TraceCube into a String
	public String toString() {
		StringBuffer sb = new StringBuffer(2048);
		sb.append(getCubeName());
		if (getQueueID() != null)
			sb.append("\t").append("Queue ID: " + getQueueID());
		if (getSPQueueSID() != null)sb.append("\t").append("SPQ ID: " + getSPQueueSID());
		if (getStartDate() != null)
			sb.append("\t").append("Starts at: " + getRawStartDate());
		if (getEndDate() != null)
			sb.append("\t").append("Ends at: " + getRawEndDate());
		if (getMaximumLength(getTraceTypes()) != 0)
			sb.append("\t").append("Contain points: " + getMaximumLength(getTraceTypes()));
		if (getTraceTypes() != null) {
			sb.append("\t").append("TraceTypes are: ");
			for (short i = 0; i < getTraceTypes().length; i++) {
				sb.append(getTraceName(getTraceTypes()[i])).append("\t");
			}
		}
		sb.append("\n");

		sb.append("TraceValues:");
		if (getTraceTypes() != null) {
			for (short i = 0; i < getTraceTypes().length; i++) {
				sb.append(getTraceName(getTraceTypes()[i])).append("::");
				for (int j = 0; j < getMaximumLength(getTraceTypes()); j++) {
					try {
						double traceVal = getTraceValueD(getTraceTypes()[i], j);
						if (isTraceValueValid(traceVal)) {
							sb.append(traceVal);
						} else {
							double traceValInt = (int) traceVal ;
							if (traceValInt == TRACENA) sb.append(PRINT_TRACENA);
							if (traceValInt == TRACEOFF) sb.append(PRINT_TRACEOFF);
							if (traceValInt == TRACEINCOMP) sb.append(PRINT_TRACEINCOMP);
						}
						sb.append(",");
					} catch (Exception ex) {
						// just ignore
					}
				}
				sb.append("\n");
			}
		}
		return sb.toString();
	}

	// Internal helper routine to check if Trace Type is supported
	protected void checkTraceTypeSupport(short[] types) throws BbmTimeSeriesException {
		if (types == null)
			return;
		for (int i = 0; i < types.length; i++) {
			short[] supported = getSupportedTraceType();
			boolean exist = false;
			for (int j = 0; j < supported.length; j++) {
				if (supported[j] == types[i]) {
					exist = true;
					break;
				}
			}
			if (!exist)
			{
				System.out.println("\nThe following trace type is not supported: " + types[i]);
				throw new BbmTimeSeriesException(BbmTimeSeriesException.TYPE_NOT_SUPPORTED);
			}
		}
	}

	// override default clone function, deep copy of the tracecube without queueID
	public Object clone() {
		try {
			TraceCube clone = cloneTraceCube();
			short[] types = clone.getTraceTypes();
			for (int i = 0; i < types.length; i++) {
				double[] value = getTraceValueD(types[i]);
				if (value == null)
					continue;
				double[] newValue = new double[value.length];
				System.arraycopy(value, 0, newValue, 0, value.length);
				clone.setTraceValue(types[i], newValue);
			}
			return clone;
		} catch (Exception e) {
			return null;
		}
	}

	public boolean fallsIn(Date point) {
		Date start = getRawStartDate();
		Date end = getRawEndDate();
		if (start == null) {
			if (end != null)
				return !point.after(end);
		} else {
			if (!point.before(start)) {
				if (end != null)
					return !point.after(end);
			} else {
				return false;
			}
		}
		return true;
	}

	private double[] convertIntToDouble(int[] result) {
		if (result == null) return null;
		double[] doubleValArr = new double[result.length];
		for (int y = 0; y < result.length; y++) {
			doubleValArr[y] = result[y];
		}
		return doubleValArr;
	}

	private int[] convertDoubleToInt(double[] result) {
		if (result == null) return null;
		int[] intValArr = new int[result.length];
		for (int y = 0; y < result.length; y++) {
			intValArr[y] = TraceUtil.roundDouble(result[y]);
		}
		return intValArr;
	}

	public static short[] toShortArray(ArrayList<Short> list)  {
		short[] ret = new short[list.size()];
		int i = 0;
		for (Short e : list)
			ret[i++] = e.shortValue();
		return ret;
	}
	
	/**
	 * Checks to see if all of the trace values in this TraceCube are invalid.
	 * @return true if all of the trace values in this TraceCube are invalid.
	 */
	public boolean isAllNull() {
		double[][] values = this.getTraceValuesD();
		if (values != null && values.length>0) {
			for (int traceType=0; traceType<values.length; traceType++) {
				if (values[traceType] != null && values[traceType].length>0) {
					for (int traceValue=0; traceValue<values[traceType].length; traceValue++) {
						double curVal = values[traceType][traceValue];
						if (Trace.isValidValue(curVal)) {
							return false;
						}
					}
				}
			}
		}
		return true;
	}
