		this(queueID, startDate, endDate, types, TimeZoneUtil.GMT_TIMEZONE);
	}

	/**
	 * TraceCube is a container for types of Trace, given period
	 *
	 * @param ID,      queueID
	 * @param Date,    startDate,
	 * @param Date,    endDate,
	 * @param short[], the Trace Types to be included
	 * @param TimeZone
	 */
	public AggrForecastedTraceCube(ID queueID, Date startDate, Date endDate, short[] types, TimeZone timeZone) throws BbmTimeSeriesException{
		super(queueID, startDate, endDate, null);
		checkTraceTypeSupport(types);
		m_Types = types;
		ArrayList<Short> forecastTypes = new ArrayList(types.length);
		ArrayList<Short> predictTypes = new ArrayList(types.length);
		for (int i=0; i<types.length; i++) {
			short type = types[i];
			if (TraceUtil.META_FORECAST.supportType(type)) {
				forecastTypes.add(type);
				if (type == Trace.VAR) {
					predictTypes.add(type);
				}
			} else {
				predictTypes.add(type);
			}
		}
		if (!forecastTypes.isEmpty()) {
			m_FTC =  new ForecastTraceCube(queueID, startDate, endDate, toShortArray(forecastTypes), timeZone);
		}
		if (!predictTypes.isEmpty()) {
			m_PTC = new PredictTraceCube(queueID, startDate, endDate, toShortArray(predictTypes), timeZone);
		} 
	}

	/**
	 * Get the maximum length of TraceValue for several Trace Types
	 *
	 * @param short[], types
	 * @return int, length
	 */
	public int getMaximumLength(short[] types) {
		int length = 0;
		if (m_PTC != null) {
			length = m_PTC.getMaximumLength(types);
		}
		if (m_FTC != null) {
			length = (m_FTC.getMaximumLength(types) > length) ? m_FTC.getMaximumLength(types) : length;
		}
		return length;
	}

	/**
	 * Override superclass method
	 *
	 * @param short, type
	 * @return boolean
	 */
	public boolean isDirty(short type) {
		if (m_FTC != null && m_FTC.getTraceValueD(type) != null)
			return m_FTC.isDirty(type);
		if (m_PTC != null && m_PTC.getTraceValueD(type) != null)
			return m_PTC.isDirty(type);
		return false;
	}

	protected String getCubeName() { return "AggrForecastedTraceCube"; }

	public TraceCube getTraceCubeForType(short type) {
		if (TraceUtil.META_PREDICT.supportType(type)) {
			if (m_PTC != null && m_PTC.getTraceValueD(type) != null) {
				return m_PTC;
			}
		}
	
		if (TraceUtil.META_FORECAST.supportType(type)) {
			if (m_FTC != null && m_FTC.getTraceValueD(type) != null) {
				return m_FTC;
			}
		}

		return null;
	}

	public TraceCube cloneTraceCube() throws BbmTimeSeriesException {
		AggrForecastedTraceCube clone = (AggrForecastedTraceCube) super.cloneTraceCube();
		if (getForecastTrace() != null) {
			clone.setForecastTrace((ForecastTraceCube) getForecastTrace().cloneTraceCube());
		}
		if (getPredictTrace() != null) {
			clone.setPredictTraceCube((PredictTraceCube) getPredictTrace().cloneTraceCube());
		}
		return clone;
	}
