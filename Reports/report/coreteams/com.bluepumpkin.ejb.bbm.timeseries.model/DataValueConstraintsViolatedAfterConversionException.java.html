package com.bluepumpkin.ejb.bbm.timeseries.model;

/**
 * This exception occurs when an attempt is made to set a value of
 * a trace cube to a value that is invalid for that particular trace
 * type.
 * 
 * If the value underwent a type transformation from double to int, this exception
 * should be used instead of DataValueConstraintsViolatedException, as it might be
 * desirable to include the unconverted double value as well as the converted int value
 * for additional information about the exception.
 */
public class DataValueConstraintsViolatedAfterConversionException extends DataValueConstraintsViolatedException {

	private static final long serialVersionUID = 5628654165413256327L;

	private int convertedValue;

	public DataValueConstraintsViolatedAfterConversionException(
			DataValueConstraints constraints, Double rawValue, int convertedValue) {
		super(constraints, rawValue);

		this.convertedValue = convertedValue;
	}
	
	@Override
	public String getMessage() {
		String min = "undefined";
		String max = "undefined";
		if (constraints != null) {
			min = "" + constraints.getMinValue();
			max = "" + constraints.getMaxValue();
		}
		return "Converted value of " + convertedValue + "violates constraints of min " + min +
			" and max of " + max + ". Raw (unconverted) value was: " + value; 
	}

	/**
	 * Returns the int value (after the double->int conversion) of the original
	 * double value that caused the data constraint violation.
	 */
	public int getConvertedValue() {
		return convertedValue;
	}
}
