	@Override
	public short[] getComputedTraceTypes() {
		return COMPUTED_TYPES;
	}
	public static boolean isPredictedTraceType(short traceType) {
		boolean retVal = false;
		for (int i = 0; i < PredictTraceCube.TYPES.length; i++) {
			if (traceType == PredictTraceCube.TYPES[i]) {
				retVal = true;
				break;
			}
		}
		return retVal;
	}

	 * time interval Separately (identified by the offset). This method is used to calculate the computed values

	/**
	 * the two dimensional array of Forcasted values is passed to the method to calculate the computed values
	 * @param queCubes [] First element is Actual, Second is Forecast, Third is Predict, Forth is ServiceGoal, Fifth is Required
	 */
	@Override
	public void calcComputedTraceValuesForCubes(TraceCube[] queCubes, ID mediaID){
	    double[][] traceValuesForecasted=queCubes[1]!=null?queCubes[1].getTraceValuesD():null;
		short traceTypes[] = getTraceTypes();
		
		for (int j = 0; j < traceTypes.length; j++) {
			short type = traceTypes[j];
			if (isTraceTypeComputed(type)) {
				if (traceValuesForecasted!=null&& (type == Trace.CONNECTS || type == Trace.RPC)) {
					if (getQueueID().toInt() == -1) {
						//System.out.println("no point in doing any calc for combined Queues");
						continue;
					}
					double[] cArray = null;
					if (type == Trace.CONNECTS) {
						cArray = traceValuesForecasted[Trace.CRATE];
					} else if (type == Trace.RPC) {
						cArray = traceValuesForecasted[Trace.RPCRATE];
					}
					double[] dialsArray = getTraceValueD(Trace.DIALS);
					if (dialsArray == null || cArray == null) {
						continue;//no point in doing any calc , skip this tracetype
					}
					int arrayLength = (dialsArray.length > cArray.length ? cArray.length : dialsArray.length);
					double[] value = new double[arrayLength];
					initIntArray(value);
					for (int i = 0; cArray != null && i < cArray.length; i++) {
						if (i < dialsArray.length) {
							if (cArray[i] < 0 || dialsArray[i] < 0)
								continue;
							
							//Calculates the Predicted Connects/RPC here. QC#129911: We round to int to be consistent with F&S Client and the TRACEREPORT table.
							value[i] = TraceUtil.roundDouble((cArray[i] * dialsArray[i]) / 100);							
						} else
							break;
					}
					setTraceValue(type, value);                             
				} else if (type == Trace.NETSTAFFING) {
					// (FFTE / RFTE)/ (FFTE / FSTAFF) or FSTAFF - (FSTAFF*RFTE)/FFTE
					//The code for computing net staffing exists in traceoperator.computeAndScaleNetStaffing()
				} else if (traceValuesForecasted != null && type == Trace.VH) {
					boolean isFirstInterval = true;
					// -999 is a little bit overkill; the idea is that neither the
					// interval -999 nor the interval after it, -998, will fall within
					// the bounds of the trace array.  -2 would probably have sufficed.
					int interimIndex = (getInterimBacklogDate() == null ? -999 : TraceUtil.gapOffSet(m_StartDate, m_StartOffSet, getInterimBacklogDate()));
					// run the index to include the value at the upper bounds i.e.,getRawEndOffset() value, 
					// else the trace will not be computed for the very last interval.
					for (int index = getRawStartOffset(); index <= getRawEndOffset(); index++) {
						calculatePredictedVolumeHandledAtIndex(mediaID, index, isFirstInterval, index == interimIndex, index == interimIndex + 1, queCubes[1]);
						isFirstInterval = false;
					}
				}
			}
		}

		if (traceValuesForecasted != null) {
			double[] cArray = traceValuesForecasted[Trace.CV];
			computeVARIfNeeded(mediaID, cArray, traceTypes);
		}

	}
	/**
	 * Compute VAR if needed. VAR contains the statistic (either CV or VH) to weight SL aggregations by.
	 * @param mediaID - MediaID is needed to determine whether to use VH instead of CV when aggregating Predicted SL.
	 * @param traceValuesForecasted
	 * @param traceTypes - Al trace types in the cube. If VAR is in there, we will populate it.
	 */
	public void computeVARIfNeeded(ID mediaID, double[] cArray, short[] traceTypes) {
		//
		for (int j = 0; j < traceTypes.length; j++) {
			short type = traceTypes[j];
			if (isTraceTypeComputed(type)) {
				if (type == Trace.VAR) {
					if (isComputeSLOnIntervalHandled(mediaID) || cArray==null) {
						cArray = getTraceValueD(Trace.VH);
					}
					if (cArray != null) {
						double[] value = new double[cArray.length];
						System.arraycopy(cArray, 0, value, 0, cArray.length);
						setTraceValue(type, value);
					}
				}
			}
		}
	}

	/**
	 * Does not populate predicted VH.  If predicted VH is needed here then this
	 * will have to be updated.  Since deferred predicted VH requires the previous
	 * interval's backlog this method's signature would have to be changed.
	 */
	@Override
	public void calcComputedTraceValues(short traceTypes[], float[] traceValues, ID mediaID) {
		float dials = getTraceValueForTraceTypeFromGivenArrayOfTraceValues(Trace.DIALS, traceTypes, traceValues);
		for (int j = 0; j < traceTypes.length; j++) {
			short type = traceTypes[j];
			if (isTraceTypeComputed(type)) {
				if (type == Trace.CONNECTS || type == Trace.RPC) {
					float cVal = 0;
					if (type == Trace.CONNECTS) {
						cVal = getTraceValueForTraceTypeFromGivenArrayOfTraceValues(Trace.CRATE, traceTypes, traceValues);
					} else if (type == Trace.RPC) {
						cVal = getTraceValueForTraceTypeFromGivenArrayOfTraceValues(Trace.RPCRATE, traceTypes, traceValues);
					}

					if (dials >= 0 && cVal >= 0) {
						traceValues[j] = (cVal * dials) / 100; //Calculates the Connects / RPC here
					}
				} else if (type == Trace.VAR){
					traceValues[j] = getTraceValueForTraceTypeFromGivenArrayOfTraceValues(Trace.CV, traceTypes, traceValues);
				}
			}
		}
	}

	/**
	 * Does not populate predicted VH.  If predicted VH is needed here then this
	 * will have to be updated.  Since deferred predicted VH requires the previous
	 * interval's backlog this method's signature would have to be changed.
	 */
	@Override
	public void calcComputedTraceValuesForTraceChunk(TraceChunk tc, ID mediaID) {
		float dials = tc.getTraceValue(Trace.DIALS);
		float cRate = tc.getTraceValue(Trace.CRATE);
		float rpcRate = tc.getTraceValue(Trace.RPCRATE);
		if (dials >= 0 && cRate >= 0) {
			tc.setTraceValue(Trace.CONNECTS, (cRate * dials) / 100); //Calculates the CONNECTS here
		} else if (dials == TRACENA && cRate == TRACENA) {
			tc.setTraceValue(Trace.CONNECTS, TRACENA);
		}
		if (dials >= 0 && rpcRate >= 0) {
			tc.setTraceValue(Trace.RPC, (rpcRate * dials) / 100); //Calculates the RPC here
		} else if (dials == TRACENA && rpcRate == TRACENA) {
			tc.setTraceValue(Trace.RPC, TRACENA);
		}
		tc.setTraceValue(Trace.VAR, tc.getTraceValue(Trace.CV));
	}

	/**
	 * Does not populate predicted VH.  If predicted VH is needed here then this
	 * will have to be updated.  Since deferred predicted VH requires the previous
	 * interval's backlog this method's signature would have to be changed.
	 */
	public static void calcComputedTraceValuesForTraceChunk(TraceChunk predictCube, TraceChunk forecastCube) {
		float dials = predictCube.getTraceValue(Trace.DIALS);
		float cRate = forecastCube.getTraceValue(Trace.CRATE);
		if (dials >= 0 && cRate >= 0) {
			predictCube.setTraceValue(Trace.CONNECTS, (cRate * dials) / 100); //Calculates the CONNECTS here
		} else if (dials == Trace.TRACENA && cRate == Trace.TRACENA) {
			predictCube.setTraceValue(Trace.CONNECTS, Trace.TRACENA);
		}
		float rpcRate = forecastCube.getTraceValue(Trace.RPCRATE);
		if (dials >= 0 && rpcRate >= 0) {
			predictCube.setTraceValue(Trace.RPC, (rpcRate * dials) / 100); //Calculates the RPC here
		} else if (dials == Trace.TRACENA && rpcRate == Trace.TRACENA) {
			predictCube.setTraceValue(Trace.RPC, Trace.TRACENA);
		}
		predictCube.setTraceValue(Trace.VAR, forecastCube.getTraceValue(Trace.CV));
	}

	/**
	 * Does not populate predicted VH. This implementation implicitly assumes that an unsupported trace type, CV, has been
	 * stored in this trace cube.
	 */
	@Override
	public void calcComputedTraceValues(ID mediaID) {
		short traceTypes[] = getTraceTypes();
		
		//first calculate VH, since it might be needed for VAR 
		for (int j = 0; j < traceTypes.length; j++) {
			short type = traceTypes[j];
			if (isTraceTypeComputed(type)) {
				if (type == Trace.VH) {
					boolean isFirstInterval = true;
					// -999 is a little bit overkill; the idea is that neither the
					// interval -999 nor the interval after it, -998, will fall within
					// the bounds of the trace array.  -2 would probably have sufficed.
					int interimIndex = (getInterimBacklogDate() == null ? -999 : TraceUtil.gapOffSet(m_StartDate, m_StartOffSet, getInterimBacklogDate()));
					for (int index = getRawStartOffset(); index < getRawEndOffset(); index++) {
						calculatePredictedVolumeHandledAtIndex(mediaID, index, isFirstInterval, index == interimIndex, index == interimIndex + 1, this);
						isFirstInterval = false;
					}
				}
			}
		}

		double[] cArray = getTraceValueD(Trace.CV);
		computeVARIfNeeded(mediaID, cArray, traceTypes);
	}
	
	/**
	 * Returns true if BPCONFIG key 'isComputeSLOnIntervalHandled' is true AND mediaID is a deferred type.
	 */
	public boolean isComputeSLOnIntervalHandled(ID mediaID) {
		return Media.isMediaDeferred(mediaID) && Trace.isComputeSLOnIntervalHandled();
	}

	/**
	 * Calculates the predicted Volume Handled at a particular index and stores
	 * it in that index of the {@link Trace#VH} trace of this trace cube.
	 * 
	 * @param mediaID the media ID of this trace cube's queue
	 * @param timeIndex the time index for which to calculate VH
	 * @param isFirstInterval whether this index is the first in the trace
	 * @param isInterimInterval whether this index is the one at which interim
	 * backlog is defined
	 * @param isIntervalAfterInterimInterval whether this index is the one immediately
	 * after the index at which interim backlog is defined
	 * @param forecastDataSource a trace cube containing volume (arrivals) in its
	 * {@link Trace#CV} trace.
	 */
	private void calculatePredictedVolumeHandledAtIndex(
			ID mediaID,
			int timeIndex,
			boolean isFirstInterval,
			boolean isInterimInterval,
			boolean isIntervalAfterInterimInterval,
			TraceCube forecastDataSource) {
		if (mediaID == null) {
			return;
		} else if (Media.isMediaImmediate(mediaID)) {
			setTraceValue(
					Trace.VH,
					getPredictedVolumeHandledImmediate(
							forecastDataSource.getTraceValueD(Trace.CV, timeIndex),
							getTraceValueD(Trace.ABANDONMENT, timeIndex)),
					timeIndex, true);
		} else if (Media.isMediaDeferred(mediaID)) {
			double previousBacklog;
			if (isFirstInterval) {
				previousBacklog = getStartingBacklog();
			} else if (isIntervalAfterInterimInterval) {
				previousBacklog = getInterimBacklog();
			} else {
				previousBacklog = getTraceValueD(Trace.BACKLOG, timeIndex - 1);
			}
			double currentBacklog;
			if (isInterimInterval) {
				currentBacklog = getInterimBacklog();
			} else {
				currentBacklog = getTraceValueD(Trace.BACKLOG, timeIndex);
			}
			setTraceValue(
					Trace.VH,
					getPredictedVolumeHandledDeferred(
							currentBacklog,
							previousBacklog,
							forecastDataSource.getTraceValueD(Trace.CV, timeIndex)),
					timeIndex, true);
		}
	}

	/**
	 * Calculates the predicted volume handled for an immediate media queue based on
	 * forecast volume (arrival) and abandonment.
	 * 
	 * @param volume
	 * @param abandonment FIXME: Is this actual abandons or abandonment rate???
	 * @return
	 */
	private double getPredictedVolumeHandledImmediate(double volume, double abandonment) {
		double vh = Trace.TRACENA;
		if (volume >= 0) {
			if (abandonment >= 0) {
				vh = volume - abandonment;
			} else {
				vh = volume;
			}
		} else {
			vh = volume;
		}
		return vh;
	}

	/**
	 * Calculates the predicted volume handled for a deferred media queue based on
	 * current and previous backlog and forecast volume (arrival).
	 * 
	 * @param currentBacklog
	 * @param previousBacklog
	 * @param forecastVolume
	 * @return
	 */
	private double getPredictedVolumeHandledDeferred(
			double currentBacklog,
			double previousBacklog,
			double forecastVolume) {
		double vh = Trace.TRACENA;
		if (previousBacklog >= 0 && currentBacklog >= 0 && forecastVolume >= 0) {
			vh = previousBacklog - currentBacklog + forecastVolume;
		}
		return vh;
	}
