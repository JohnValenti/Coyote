package com.bluepumpkin.ejb.bbm.timeseries.model;

import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.base.BbmTimeSeriesException;

/**
 * This flavor of trace cube is intended only to store FTE requirement data.
 */

public class  RequireTraceCube extends TraceCube {
//	static final long serialVersionUID = 8464384229870126467L;

	/**
	 *
	 */
	private static final long serialVersionUID = 1L;

	public static final short[] TYPES = {Trace.FTE};

	//array of all trace types that are computed for this tracecube
	private final short[] COMPUTED_TYPES = {	};
	public short[] getComputedTraceTypes() {
		return COMPUTED_TYPES;
	}

	// media id is used for combined-queue data, where spqueue id doesn't exist
	//This appears to be unused, leaving in to simplify port forward and commit comment refers to part 1 of a changd.
	private ID mediaID;

	public RequireTraceCube(short[] types) {
		super(types);
	}
	// Default constructor
	public RequireTraceCube() { super(); }
	/**
	 * TraceCube is a container for types of Trace, given period
	 *
	 *
	 * @param ID, queueID
	 * @param Date, startDate,
	 * @param Date, endDate,
	 * @param short[], the Trace Types to be included
	 */
	public RequireTraceCube(ID queueID, Date startDate, Date endDate, short[] types) throws BbmTimeSeriesException{
		this(queueID, startDate, endDate, types, TimeZoneUtil.GMT_TIMEZONE);
	}
	/**
	 * TraceCube is a container for types of Trace, given period
	 *
	 *
	 * @param ID, queueID
	 * @param Date, startDate,
	 * @param Date, endDate,
	 * @param short[], the Trace Types to be included
	 * @param TimeZone
	 */
	public RequireTraceCube(ID queueID, Date startDate, Date endDate, short[] types, TimeZone timeZone) throws BbmTimeSeriesException{
		super(queueID, startDate, endDate, types);
		checkTraceTypeSupport(types);
	}
	// Base TraceCube supports all types of Trace
	protected short[] getSupportedTraceType() {	return TYPES; }

	protected String getCubeName() { return "REQUIREDTIMESERIES"; }

	// make a clone from given parameters
	public TraceCube newInstance(ID queueID, Date startDate, Date endDate, short[] types) throws BbmTimeSeriesException{
		return new RequireTraceCube(queueID, startDate, endDate, types);
	}

	/**
	 * Returns whether this trace cube contains requirement data, as opposed to actual or
	 * forecast data.
	 *
	 * @return
	 */
	public boolean isRequiredData() {
		return true;
	}
}
