package com.bluepumpkin.ejb.bbm.timeseries.model;

/**
 * This enum lists the possible number types that can be used in a numeric model.
 * When dealing with a series of data values, the data type of the aggregate
 * value of an interval in the series may not be the same as the data type of
 * the individual values in the series.
 * <p>
 * For instance, if a series of integers is summed over an interval, the result
 * is an integer, but if the series is averaged over an interval the result is
 * generally a floating point value.
 * <p>
 * The ForcedInteger type is special in this sense; by definition the aggregate
 * of an interval of a series of ForcedInteger values is also a ForcedInteger,
 * regardless of the method of aggregation.
 */
public enum DataType
{
	/**
	 * Integer values.  Note that when Integer values are aggregated, some aggregation
	 * methods will give rise to FloatingPoint values.
	 * <p>
	 * If aggregate values of the data type you are defining will always be calculated 
	 * and presented as integers, even when averaging, then use ForcedInteger instead
	 * of Integer.
	 */
	Integer,
	/**
	 * Floating point values.  When FloatingPoint values are aggregated the result is
	 * always FloatingPoint.
	 */
	FloatingPoint,
	/**
	 * A nonstandard integer type.  When ForcedInteger values are aggregated the result
	 * will always be ForcedInteger, even when averaging.  This is the only data type
	 * currently supported in trace data. 
	 */
	ForcedInteger
}
