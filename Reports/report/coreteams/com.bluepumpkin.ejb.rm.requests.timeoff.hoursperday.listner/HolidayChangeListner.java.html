package com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.listner;

import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.timeoffcalculator.TONotifyMessage;
import com.bluepumpkin.ejb.bbm.base.BbmCreateException;
import com.bluepumpkin.ejb.bbm.holiday.model.Holiday;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.rm.requests.timeoff.hoursperday.ejb.TOHoursPerDayManager;
import com.bluepumpkin.ejb.rm.RmManagerFactory;
import com.witness.ejb.bbm.daonotifier.DAOChangeHook;
import com.witness.ejb.bbm.daonotifier.ValueObjectPairList;

import java.util.Date;
import java.util.TimeZone;

/**
 * Sends Holiday updates to RM/IFs through remote cache manager. This class hooks
 * the Holiday table changes (CRUD) and sends the appropriate notifications through
 * PeopleCacheProcessor. PeopleCacheProcessor will be used by the RemoteCacheManager to send
 * Holiday updates to RM/IFs.
 */
public class HolidayChangeListner implements DAOChangeHook {

	private static Category LOG = Log.initCategory(HolidayChangeListner.class.getName());
	TimeZone tz = TimeZone.getTimeZone("GMT");

	/**
	 * Notifies RemoteCacheManager to send Holiday creation.
	 * Will be called when Holiday table has a new entry created as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.holiday.model.Holiday objects that were created.
	 */
	public void OnCreateObject(ValueObjectBase[] valueObjects) throws JdmoException {
		if (LOG.isDebugEnabled()) {
			LOG.debug("Holiday Hook OnCreateObject valueObjects = " + valueObjects);
		}
		if (valueObjects != null) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			throw new IllegalArgumentException();
		}
	}

	/**
	 * Notifies RemoteCacheManager to send Holiday deletion.
	 * Will be called when Holiday table has an entry deleted as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param valueObjects com.bluepumpkin.ejb.bbm.holiday.model.Holiday objects that were deleted.
	 */
	public void OnDeleteObject(ValueObjectBase[] valueObjects) throws JdmoException {
		// When an Holiday is deleted the send notification
		if (LOG.isDebugEnabled()){
			LOG.debug("Holiday Hook OnDeleteObject valueObjects = " + valueObjects);
		}
		if (valueObjects != null && valueObjects.length > 0) {
			updateTOHoursPerDayForValueObjects(valueObjects);
		} else {
			throw new IllegalArgumentException();
		}
	}

	/**
	 * Notifies RemoteCacheManager to send Holiday updates.
	 * Will be called when Holiday table has been updated as this Class is
	 * registered in GLOBALCODEREGISTRY.
	 *
	 * @param old and current value objects com.bluepumpkin.ejb.bbm.holiday.model.Holiday
	 *            will be in oldAndNewObjects.
	 */
	public void OnUpdateObject(ValueObjectPairList oldAndNewObjects) throws JdmoException {
		if (oldAndNewObjects == null
				|| oldAndNewObjects.getCurrentValues() == null) {
			throw new IllegalArgumentException();
		}
		if (LOG.isDebugEnabled()) {
			LOG.debug("Holiday Hook OnUpdateObject sending update notification "
					+ oldAndNewObjects.getCurrentValues());
		}
		ValueObjectBase[] newValueObjects = oldAndNewObjects.getCurrentValues();
		ValueObjectBase[] oldValueObjects = oldAndNewObjects.getOldValues();
		for (int i = 0; i < newValueObjects.length; i++) {
			Holiday newHoliday = (Holiday) newValueObjects[i];
			Holiday oldHoliday = (Holiday) oldValueObjects[i];
			if (newHoliday.isPaidHoliday() || oldHoliday.isPaidHoliday()
					|| newHoliday.isBlackOutDay() || oldHoliday.isBlackOutDay()) {
				// Only send notification for Paid Holidays
				Date startDate = newHoliday.getStartDate().getTime()
						.after(oldHoliday.getStartDate().getTime()) ? oldHoliday
						.getStartDate().getTime(tz) : newHoliday.getStartDate()
						.getTime(tz);
				Date endDate = newHoliday.getEndDate().getTime()
						.before(oldHoliday.getEndDate().getTime()) ? oldHoliday
						.getEndDate().getTime(tz) : newHoliday.getEndDate()
						.getTime(tz);
				try {
					if (newHoliday.isPaidHoliday()
							|| oldHoliday.isPaidHoliday()) {
						TONotifyMessage msg = new TONotifyMessage(
								newHoliday.getOrganizationID(), startDate,
								endDate,
								TONotifyMessage.TO_NOTIFY_HOLIDAY_CHANGE);
						msg.sendMessage();
					}
					if (newHoliday.isBlackOutDay()
							|| oldHoliday.isBlackOutDay()) {
						TONotifyMessage msg = new TONotifyMessage(
								newHoliday.getOrganizationID(),
								startDate,
								endDate,
								TONotifyMessage.TO_NOTIFY_ORG_BLACKOUTDAY_CHANGE);
						msg.sendMessage();
					}
				} catch (Exception e) {
					LOG.error("HolidayChangeListner.OnUpdateObject exception", e);
				}
			}
		}

	}

	public void updateTOHoursPerDayForValueObjects(ValueObjectBase[] valueObjects) {
		try {
			for (int i = 0; i < valueObjects.length; i++) {
				Holiday holiday = (Holiday) valueObjects[i];
				if (holiday.isPaidHoliday()) {
					TONotifyMessage msg = new TONotifyMessage(holiday.getOrganizationID(), holiday.getStartDate().getTime(tz),
					        holiday.getEndDate().getTime(tz), TONotifyMessage.TO_NOTIFY_HOLIDAY_CHANGE);
					msg.sendMessage();
				}else  if (holiday.isBlackOutDay()) {
					TONotifyMessage msg = new TONotifyMessage(holiday.getOrganizationID(), holiday.getStartDate().getTime(tz),
					        holiday.getEndDate().getTime(tz), TONotifyMessage.TO_NOTIFY_ORG_BLACKOUTDAY_CHANGE );
					msg.sendMessage();
				}

			}
		} catch (Exception e) {
			LOG.error("HolidayChangeListner.updateTOHoursPerDayForValueObjects exception", e);
		}
	}


	public TOHoursPerDayManager getTOHoursPerDayManager() throws BbmCreateException {
		return RmManagerFactory.getInstance(true).getTOHoursPerDayManager(null, null);
	}
}
