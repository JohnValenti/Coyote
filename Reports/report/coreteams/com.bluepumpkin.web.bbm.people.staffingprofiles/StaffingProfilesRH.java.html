package com.bluepumpkin.web.bbm.people.staffingprofiles;

import java.rmi.RemoteException;
import java.util.Collection;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeTemplate;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.system.RequestContext;

public class StaffingProfilesRH extends WorkRulesRH<StaffingProfilesListPM, StaffingProfilesFormPM> {
	public static final String STAFFING_PROFILE_ADD= "STAFFING_PROFILE_ADD";
	public static final String STAFFING_PROFILE_REMOVE= "STAFFING_PROFILE_REMOVE";
	public static final String STAFFING_PROFILE_SAVE= "STAFFING_PROFILE_SAVE";
	public static final String STAFFING_PROFILE_REVERT= "STAFFING_PROFILE_REVERT";
	
	@Override
	protected StaffingProfilesFormPM getFormPageModel(RequestContext context) {
		return StaffingProfilesFormPM.getInstance(context);
	}

	@Override
	protected StaffingProfilesListPM getListPageModel(RequestContext context) {
		StaffingProfilesListPM pm = new StaffingProfilesListPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
		
		// Cannot getInstance because pagination needs to be set up before the page model is inited
		//return WorkPatternListPM.getInstance(context);
	}
	
	@Override
	protected void processCopy(RequestContext context, StaffingProfilesListPM pageModel)
	throws BPException {	
		try {
			if (pageModel.isByOrg) {
				ScheduleAccessManager saMgr = WfmManagerFactory.getScheduleAccessManager(context.isInWhatIfMode());
				saMgr.copyEmployeeTemplates(pageModel.getPerfromActionOnIDsAsCollection(), 
						pageModel.isByOrg, pageModel.getSelectedIDObject(), pageModel.getLocalizer().getLocaleContext());
			}
			setNextPage(context, LIST_PAGE);
		} catch(Exception ex) {
			pageModel.addPageMessage(Message.ERROR_TYPE,
					BbmWebBundleKey.UNABLE_TO_COPY_STAFFING_PROFILES, BbmWebBundleKey.BUNDLE_NAME);
			throw new BPException(ex);
		} finally {
			setNextPage(context, LIST_PAGE);
		}
	}

	@Override
	protected void processDelete(RequestContext context,
			StaffingProfilesListPM pageModel) throws BPException, RemoteException {
		ScheduleAccessManager saMgr = WfmManagerFactory.getScheduleAccessManager(context.isInWhatIfMode());
		saMgr.deleteEmployeeTemplates(pageModel.getPerfromActionOnIDsAsCollection());
		setNextPage(context, LIST_PAGE);
	}

	@Override
	protected void processExport(RequestContext context,
			StaffingProfilesListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processImport(RequestContext context,
			StaffingProfilesListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processSave(RequestContext context, StaffingProfilesFormPM pageModel)
			throws BPException {
		try {
			if (pageModel.validate(context)) {
				ScheduleAccessManager saMgr = WfmManagerFactory.getScheduleAccessManager(context.isInWhatIfMode());
				if (pageModel.isNew()) {
					saMgr.createEmployeeTemplate(pageModel.getEmployeeTemplates(context).iterator().next());
				} else {
					Collection<EmployeeTemplate> emps = pageModel.getEmployeeTemplates(context);
					for (EmployeeTemplate emp: emps){
						saMgr.updateEmployeeTemplate(emp);
					}
				}
				setNextPage(context, LIST_PAGE);
			} else {
				setNextPage(context, "staffingprofilesformPage");
			}
		} catch (BbmException e) {
			setNextPage(context, "staffingprofilesformPage");
			throw e;
		} catch (RemoteException e) {
			setNextPage(context, "staffingprofilesformPage");
		} 
		
	}

}
