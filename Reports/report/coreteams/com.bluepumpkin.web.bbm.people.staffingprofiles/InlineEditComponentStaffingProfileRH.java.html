package com.bluepumpkin.web.bbm.people.staffingprofiles;

import java.io.PrintWriter;
import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.HashSet;

import javax.servlet.http.HttpServletResponse;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.skill.model.Skill;
import com.bluepumpkin.ejb.bbm.skill.model.SkillAssignment;
import com.bluepumpkin.ejb.bbm.workresource.model.EmployeeTemplate;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.pagecomponent.form.FormInputPC;
import com.witness.web.uif.system.RequestContext;

public class InlineEditComponentStaffingProfileRH extends RequestHandler {

	public static String URL = "inline_edit_component";
	public static String PARAM_TYPE = "type";
	public static String PARAM_IDS = "ids";
	public static String PARAM_PROFICIENCY = "proficiency";
	public static String PARAM_PRIORITY = "priority";
	public static String PARAM_CLASSIFICATION = "classificationid";
//	public static String OT_EXTENSION_TYPE = "shiftOTExtension";
//	public static String SHIFT_TYPE = "shift";
//	public static String VTO_EVENT_TYPE = "vtoEvent";
	public static String SKILL = "skill";
	public static String ASSIGNMENTRULE = "assignmentRule";
	
	@Override
	public void processRequest(RequestContext context) throws Exception {	
		String ids = context.getParameter(PARAM_IDS);
		String type = context.getParameter(PARAM_TYPE);	
		String proficiency = context.getParameter(PARAM_PROFICIENCY);
		String priority = context.getParameter(PARAM_PRIORITY);
		String classificationid = context.getParameter(PARAM_CLASSIFICATION);
		
		StringBuffer response = new StringBuffer();
		try {
			if (SKILL.equals(type)) {
				HashMap<Skill, HashSet<SkillAssignment> > emptySkillAssignments = new HashMap<Skill, HashSet<SkillAssignment> >(0);
				Collection<ID> idColl = StringUtil.parseIDStringToCollection(ids);	
				Collection<ID> idCol = deleteUndefined(idColl);
				SkillsInlineEditPC pc_skillInlineEditPC =  new SkillsInlineEditPC(context, "skillsTable", emptySkillAssignments, false, null, null);
				Collection<Skill> skills = WfmManagerFactory.getSkillManager(context.isInWhatIfMode()).getSkillsByIDs(idCol);
				if (idCol.size() > 0) {
					for (Skill skill : skills) {
						SkillAssignment skillAssign = new SkillAssignment();
						skillAssign.setSkillID(skill.getID());
						skillAssign.setPriority(Integer.parseInt(priority));
						String dbTempStr = FormInputPC.convertLocalizedRealNumberAsString(proficiency, context.getLocalizer());
						skillAssign.setProficiency(Double.parseDouble(dbTempStr));
						skillAssign.setClassification(Integer.parseInt(classificationid));
						
						response.append(pc_skillInlineEditPC.getRowHtml(skillAssign));
					}
				}
			}
			if (ASSIGNMENTRULE.equals(type)) {
				HashMap<ID, HashSet<EmployeeTemplate> > emptySkillAssignments = new HashMap<ID, HashSet<EmployeeTemplate> >(0);
				Collection<ID> idColl = StringUtil.parseIDStringToCollection(ids);	
				Collection<ID> idCol = deleteUndefined(idColl);
				AssignmentRulesInlineEditPC pc_ruleInlineEditPC =  new AssignmentRulesInlineEditPC(context, "assignmentRuleTable", null, emptySkillAssignments, false, null);
				Collection<ComplexWorkRule> rules = WfmManagerFactory.getWorkRuleManager().getComplexWorkRulesByIDs(idCol);
				if (idCol.size() > 0) {
					for (ComplexWorkRule rule : rules) {
						response.append(pc_ruleInlineEditPC.getRowHtml(rule));
					}
				}
			}
		} catch (Exception ex) {
			context.getResponse().setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
		} finally {
			context.getResponse().setContentType(RequestContext.DEFAULT_RESPONSE_CONTENT_TYPE);
			PrintWriter responseWriter = context.getResponse().getWriter();
			
			responseWriter.write(response.toString());
			responseWriter.close();
		}
	}

	private  Collection<ID> deleteUndefined ( Collection<ID> col){
		Collection<ID> idCol = new ArrayList<ID>();
		ID undefined = new ID("undefined");
		for (ID id: col){
			if (!id.equals(undefined))
				idCol.add(id);
		}
		return idCol;
	}
}
