/**
 * Verfies the following:
 * ShiftAssignments associated with the shift bid request have the same activity between
 * themselves as well as with unpublished schedules for an org day.
 *
 * To change this generated comment go to
 * Window>Preferences>Java>Code Generation>Code and Comments
 */
public class BiddableScheduleSingleActivityShiftAssignmentsHV implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

    private static final String m_className = BiddableScheduleSingleActivityShiftAssignmentsHV.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
	public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

        try {
            ShiftBidRequest sbr = (ShiftBidRequest) validatable;
            ShiftBidAuction shiftBidAuction = sbr.getOptMethods().getShiftBidAuction();
            if (!shiftBidAuction.getIsMultiShift()){
            	return null;
            }

            ShiftBidRequestValidationCache sbrValidationCache = sbr.getValidationCacheSBR();

            // shift assignments associated with the shift bid request
            @SuppressWarnings("unchecked")
			List<ShiftAssignment> sbrShiftAssns = sbr.getOptMethods().getShiftAssignments();

            // get the schedule shift assignments for the request period
			@SuppressWarnings("unchecked")
			Collection<ShiftAssignment> schedShiftAssns = sbrValidationCache.getShiftsUnpubDuringPeriod();

            // get the organization for the request's employee
            Organization empOrg = sbrValidationCache.getOrg();

            Map<Date,ID> dateStrToActivityIdMap = new HashMap<Date, ID>();
            for (ShiftAssignment shift: schedShiftAssns) {
                // get the org day start time on which the shift falls.
                Date orgDayStartForShift = getDateForOrgDayStart(empOrg, shift.getStartTime());

                //Add shift assignments to map with date, activity id.
                dateStrToActivityIdMap.put(orgDayStartForShift, shift.getActivityID());
            }


        	for (ShiftAssignment shift:sbrShiftAssns) {
                 Date orgDayStartForShift = getDateForOrgDayStart(empOrg, shift.getStartTime());

                 if (dateStrToActivityIdMap.containsKey(orgDayStartForShift) && dateStrToActivityIdMap.get(orgDayStartForShift).compareTo(shift.getActivityID())!=0) {
                     // Error: a shift exists for the same org day in the schedule with a different activity.
                 	ValidationResult result = ValidationUtil.setHardValidationResult(sbr, RmEjbBundleKey.SBR_TWO_SHIFTASSNS_ACTIVITY_SAME_ORG_DAY,
                 			orgDayStartForShift,
                 			getActivity(shift.getActivityID()),
                 			getActivity(dateStrToActivityIdMap.get(orgDayStartForShift)), m_className);
                 	return result;
                 } else {
                	 //Add to map to compare with other shifts in bid
                	 dateStrToActivityIdMap.put(orgDayStartForShift, shift.getActivityID());
                 }
             }

            return null;
        } finally {
            m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
        }
    }


    public String getActivity(ID activityID) throws Exception{
    	return RequestUtil.getActivityNameByID(activityID);
    }

    public Date getDateForOrgDayStart(Organization o, Date d){
    	return TOCalcUtil.getDateForOrgDayStart(o, d);
    }
}
