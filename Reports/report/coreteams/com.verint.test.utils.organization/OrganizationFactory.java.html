package com.verint.test.utils.organization;


import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workresource.model.OrganizationHOO;
import com.verint.test.utils.DateUtils;
import com.verint.test.utils.GuidUtils;

import java.util.Calendar;
import java.util.TimeZone;

public class OrganizationFactory {

    /**
     * Create organization by name only all other fields have default values.
     *
     * @param name
     * @return
     */
    public static Organization createDefaultOrganization(String name) {
        return createDefaultOrganization(name, name.concat(" Description"));
    }

    public static Organization createDefaultOrganization(String name, String description) {
        Organization org = new Organization();
        org.setName(GuidUtils.createGuidName(name));
        org.setDescription(description);
        org.setTimeZone(TimeZone.getTimeZone("America/Los_Angeles"));
        org.setDayBoundaryOffset(0);
        org.setWeekStartDate((short) Calendar.SUNDAY);
        org.setParentID(Organization.YOUR_COMPANY_ID_OBJ);
        return org;
    }

    public static Organization createMinimalOrganization(TimeZone timeZone, ID parentOrgId, short weekStart,
                                                         String name) {
        Organization org = new Organization();

        org.setTimeZone(timeZone);
        org.setName(name);
        org.setParentID(parentOrgId);
        org.setWeekStartDate(weekStart);
        return org;
    }

    public static OrganizationHOO create8AmTo8PmHoo(ID orgId) {
        OrganizationHOO hoo = new OrganizationHOO();

        for (short i = 1; i <= 7; i++) {
            hoo.setDayOpen(i, (short) (8 * 60));
            hoo.setDayClose(i, (short) (20 * 60));
        }
        hoo.setOrganizationID(orgId);
        return hoo;
    }

    // startTime and endTime should be hours in 24 format (eg. startTime = 6, endTime = 18)
    public static OrganizationHOO createFixedHOO(ID orgId, int startTime, int endTime) {
        OrganizationHOO hoo = new OrganizationHOO();

        for (short i = 1; i <= 7; i++) {
            hoo.setDayOpen(i, (short) (startTime * 60));
            hoo.setDayClose(i, (short) (endTime * 60));
        }

        hoo.setOrganizationID(orgId);
        return hoo;
    }

    /**
     * Create organization operation hours by providing 12 hours string.
     *
     * @param orgId
     * @param startTime from like 6 AM.
     * @param endTime   to like 6 PM.
     * @return
     */
    public static OrganizationHOO createFixedHOO(ID orgId, String startTime, String endTime) {
        int startInMins = DateUtils.timeToMinutes12HoursStr(startTime);
        int endInMins = DateUtils.timeToMinutes12HoursStr(endTime);

        OrganizationHOO hoo = new OrganizationHOO();

        for (short i = 1; i <= 7; i++) {
            hoo.setDayOpen(i, (short) (startInMins));
            hoo.setDayClose(i, (short) (endInMins));
        }
        hoo.setOrganizationID(orgId);

        return hoo;
    }
}
