package com.bluepumpkin.ejb.bbm.rtaamanager.ejb;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.HashMap;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.rtaamanager.model.OrganizationDay;
import com.bluepumpkin.ejb.bbm.rtaamanager.model.OrganizationDayFieldInfo;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class OrganizationDayDAO extends DAOBase {

	// meta data info.
	private static FieldInfo m_fieldInfo = new OrganizationDayFieldInfo();
	
	public OrganizationDayDAO (Jdmo dmo) {
		super(dmo);
	}
	
	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	protected ValueObjectBase createValueObject() {
		return new OrganizationDay();
	}
	
	public Collection getSPDays(Date start, Date end) throws BbmFinderException {
		return getDays(start, end, false);
	}
	
	public HashMap getSPDaysPerWorkResource(Collection colEmployeeIDs, Date start, Date end) throws JdmoException {
		return getDaysPerWorkResource(colEmployeeIDs, start, end, false);
	}
	
	public HashMap getOrgDaysPerWorkResource(Collection colEmployeeIDs, Date start, Date end) throws JdmoException {
		return getDaysPerWorkResource(colEmployeeIDs, start, end, true);
	}
	
	private HashMap getDaysPerWorkResource(Collection colEmployeeIDs, Date start, Date end, boolean isOrgDay) throws JdmoException {
		HashMap map = new HashMap();
		StringBuffer sb = new StringBuffer(200);
		String startDBString = "'" + JdmoUtil.formatDBString(start) + "'";
		String endDBString = "'" + JdmoUtil.formatDBString(end) + "'";
		if (isOrgDay) {
			sb.append(" select WORKRESOURCEORGANIZATION.WORKRESOURCEID , ORGANIZATIONDAY.ID, ORGANIZATIONDAY.ORGANIZATIONID, ORGANIZATIONDAY.STARTTIME, ORGANIZATIONDAY.ENDTIME");
			sb.append("	from ORGANIZATIONDAY");
			sb.append("	join WORKRESOURCEORGANIZATION");
			sb.append("	on ORGANIZATIONDAY.ORGANIZATIONID = WORKRESOURCEORGANIZATION.ORGANIZATIONID");
			sb.append(" and WORKRESOURCEORGANIZATION.STARTTIME <= ORGANIZATIONDAY.STARTTIME");
			sb.append(" and (ORGANIZATIONDAY.STARTTIME < WORKRESOURCEORGANIZATION.ENDTIME or WORKRESOURCEORGANIZATION.ENDTIME is null)");
			sb.append(" where WORKRESOURCEORGANIZATION.WORKRESOURCEID IN ").append(m_dmo.createInClause(colEmployeeIDs));
			sb.append(" and ORGANIZATIONDAY.STARTTIME <  ").append(endDBString).append(" AND ORGANIZATIONDAY.ENDTIME > ").append(startDBString);
		} else {
			//sp day
			sb.append(" select SPWORKRESOURCE.WORKRESOURCEID, ORGANIZATIONDAY.ID, ORGANIZATIONDAY.SPID, ORGANIZATIONDAY.STARTTIME, ORGANIZATIONDAY.ENDTIME");
			sb.append(" from ORGANIZATIONDAY");
			sb.append(" join SP");
			sb.append(" on SP.SID = ORGANIZATIONDAY.SPID");
			sb.append(" join SPWORKRESOURCE");
			sb.append(" on SPWORKRESOURCE.SPID = SP.ID");
			sb.append(" where SPWORKRESOURCE.WORKRESOURCEID IN ").append(m_dmo.createInClause(colEmployeeIDs));
			sb.append(" and ORGANIZATIONDAY.STARTTIME <  ").append(endDBString).append(" AND ORGANIZATIONDAY.ENDTIME > ").append(startDBString);
		}
		
		JdmoRowset rs = m_dmo.createRowset(sb.toString());
		ID workresourceID = null;
		ArrayList list = null;
		
		while(rs.next()) {
			workresourceID = rs.getID(1);
			list = (ArrayList)map.get(workresourceID);
			if (list == null) {
				list = new ArrayList();
				map.put(workresourceID, list);
			}
			list.add(new OrganizationDay(rs.getID(2), rs.getID(3), rs.getTimestamp(4), rs.getTimestamp(5), isOrgDay));
		}
		return map;
	}
	
	private Collection getDays(Date start, Date end, boolean isOrgDay) throws BbmFinderException {
		StringBuffer sb = new StringBuffer(200);
		if (isOrgDay)
			sb.append(" ORGANIZATIONID IS NOT NULL ");
		else
			sb.append(" SPID IS NOT NULL ");
		if (start != null)
			sb.append(" AND ENDTIME > '").append(JdmoUtil.formatDBString(start)).append("'");
		if (end != null)
			sb.append(" AND STARTTIME < '").append(JdmoUtil.formatDBString(end)).append("'");
		
		return getObjects(sb.toString());
	}

}
