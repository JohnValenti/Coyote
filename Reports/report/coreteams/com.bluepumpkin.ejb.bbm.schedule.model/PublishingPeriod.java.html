package com.bluepumpkin.ejb.bbm.schedule.model;

/**
 * Title:        Blue Pumpkin Software Basic Business Model
 * Description:  Value object that models the the time periods for which
 *               employees' schedules have been published
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       Greg Fichtenholtz
 * @version 1.0
 */
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

public class PublishingPeriod extends ValueObjectBase {
	static final long serialVersionUID = -1410362669689034873L;

	private static FieldInfo m_fieldInfo = new PublishingPeriodFields();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public ID getWorkResourceID() {
		return getFieldValueID(PublishingPeriodFields.WORKRESOURCEID);
	}

	public Date getStartTime() {
		return getFieldValueDate(PublishingPeriodFields.STARTTIME);
	}

	public Date getEndTime() {
		return getFieldValueDate(PublishingPeriodFields.ENDTIME);
	}

	public Date getPublishTime() {
		return getFieldValueDate(PublishingPeriodFields.PUBLISHTIME);
	}

	public boolean isPartialPublish() {
		return getFieldValueBoolean(PublishingPeriodFields.ISPARTIALPUBLISH);
	}

	public void setWorkResourceID(ID id) {
		setFieldValue(PublishingPeriodFields.WORKRESOURCEID, id);
	}

	public void setStartTime(Date dt) {
		setFieldValue(PublishingPeriodFields.STARTTIME, dt);
	}

	public void setEndTime(Date dt) {
		setFieldValue(PublishingPeriodFields.ENDTIME, dt);
	}

	public void setPublishTime(Date dt) {
		setFieldValue(PublishingPeriodFields.PUBLISHTIME, dt);
	}

	public void setIsPartialPublish(boolean value) {
		setFieldValue(PublishingPeriodFields.ISPARTIALPUBLISH, value);
	}
}
