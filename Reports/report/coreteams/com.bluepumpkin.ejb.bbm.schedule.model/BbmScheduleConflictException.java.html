package com.bluepumpkin.ejb.bbm.schedule.model;

/**
 * Title:        Blue Pumpkin Software Basic Business Model
 * Description:  BbmScheduleConflictException is thrown by the schedule access
 *               manager when the client attempts to perform an operation that
 *               violates one or more of the schedule constraints
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, inc
 * @author       Greg Fichtenholtz
 * @version 1.0
 */
import java.util.Collection;
import java.util.Locale;
import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;

import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbBundleKey;

public class BbmScheduleConflictException extends BbmException {
	/**
	 *
	 */
	private static final long serialVersionUID = 1L;

	public BbmScheduleConflictException() {
		super();
	}

	public BbmScheduleConflictException(Collection<BbmScheduleConflict> conflicts) {
		super(BbmEjbBundleKey.SCHEDULE_CONFLICT_EXCEPTION);
		setContent(conflicts);
	}

	/**
	 * Returns a collection of conflicts which represent the hard constraints
	 * that were violated while attempting to perform the requested operation.
	 * Each object in the collection is of type BbmScheduleConflict. For each
	 * violation of a shcedule constraint there will be a separate
	 * BbmScheduleConflict object instance in the collection.
	 */
	@SuppressWarnings("unchecked")
	public Collection<BbmScheduleConflict> getConflicts() {
		return (Collection<BbmScheduleConflict>) getContent();
	}

	/**
	 * This is an override of BPException.getLocalizedMessage to provide a more
	 * descriptive message of that shows all conflicts
	 */
	@Override
	public String getLocalizedMessage(Locale locale) {
		StringBuilder msg = new StringBuilder(200);
		msg.append(super.getLocalizedMessage(locale));
		return msg.toString();
	}
	


	@Override
	public void writeExternal(ObjectOutput out) throws IOException {
		out.writeObject(getContent());
	}

	@Override
	public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
		this.setContent(in.readObject());
	}
	
}
