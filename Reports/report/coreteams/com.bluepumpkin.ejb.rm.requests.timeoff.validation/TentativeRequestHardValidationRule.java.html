/**
 * <b>Hard validation Rule (applies only to tentatively approved requests).</b>
 *
 * <p>
 * Ensures that created calendarEventAssignment in the unpublished schedule
 * <li>still exists
 * <li>hasn't been modified
 * <li>hasn't been published
 *
 * <p>
 * Applies equally for debit, debit-if and dont-debit TORequests
 *
 * <p>
 * Applies only for tentative requests.
 */
public class TentativeRequestHardValidationRule implements Validator {
	private static final String CLASS_NAME = TentativeRequestHardValidationRule.class.getName();
	private static final Category logger = Log.initCategory(CLASS_NAME);

	/**
	 * 
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		String methodName = "validate";
		if (logger.isDebugEnabled()) {
			logger.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		}

		ValidationResult result = null;

		// only process tentative time off requests.
		TORequest toReq = (TORequest) validatable;
		if (toReq.isTentative()) {
			ValidationResult r = checkCalEvents(toReq);
			result = r == null ? result : r;
		}

		if (logger.isDebugEnabled()) {
			logger.debug(RmUtil.dumpExitMethod("validate", result));
		}
		return result;
	}

	private ValidationResult checkCalEvents(TORequest toReq) throws Exception {

		ValidationResult finalResult = null;

		ScheduleAccessManager sam = getScheduleAccessManager();
		for (TOChoice apprChoice : toReq.getApprovedChoices()) {

			if (!checkIfCalEventAssignPublished(toReq, apprChoice, sam)) {
				ValidationResult result = checkIfCalEventAssignExistsInUnpub(toReq, apprChoice, sam);
				if (result != null) {
					finalResult = result;
				}
			}

		}

		return finalResult;
	}

	protected ValidationResult checkIfCalEventAssignExistsInUnpub(TORequest toReq, TOChoice apprChoice,
			ScheduleAccessManager sam) throws Exception {
		// verify if calendar event still exists in unpublished schedule
		// TODO: if event not found, exception thrown. Instead we need to add a
		// meaningful validation result
		CalendarEventAssignment event = getCalEventAssignForTOChoiceAligned(toReq, apprChoice, false, sam);

		// if event not found
		if (event == null) {
			return ValidationUtil.setHardValidationResult(toReq,
					RmEjbBundleKey.TO_CAL_EVENT_ASSN_NOT_FOUND_FOR_CHOICE, apprChoice.getStartDate(),
					apprChoice.getEndDate(), getEmployeeID(toReq), apprChoice.getID(), CLASS_NAME);

		}

		// verify that it hasn't been modified. The position and duration has
		// already been
		// verified by the call getCal....Aligned() call.
		if (!toReq.getTimeOffType().equals(event.getActivityID())) {
			// Calendar event (ID = {0}) found (startTime = {1,date}, endTime =
			// {2,date}) for the tentatively approved timeoff request (ID = {3})
			// choice (ID={4}) has been modified"
			return ValidationUtil.setHardValidationResult(toReq,
					RmEjbBundleKey.TO_CALEVENT_NOT_IN_SYNC_WITH_CHOICE, event.getID(), event.getStartTime(),
					event.getEndTime(), apprChoice.getTimeOffRequestID(), apprChoice.getID(), CLASS_NAME);

		}

		return null;
	}

	// verify that it hasn't been published yet.
	protected boolean checkIfCalEventAssignPublished(TORequest toReq, TOChoice apprChoice, ScheduleAccessManager sam)
			throws Exception {
		Collection events = getCalEventAssignsForTOChoiceOverlapped(toReq, apprChoice, true, sam);

		if (events.size() != 0) {
			String dispName = getDisplayEmployeeName(toReq);
			ValidationResult result = ValidationUtil.setHardValidationResult(toReq,
					RmEjbBundleKey.TO_TENT_APPR_REQ_HAS_PUBLISHED_CALEVENTS, dispName, apprChoice.getStartDate(),
					apprChoice.getEndDate(), CLASS_NAME);
			result.setApprove(true);

			return true;
		}

		return false;
	}
	
	protected ScheduleAccessManager getScheduleAccessManager() throws BbmEJBCreateException {
		return WfmManagerFactory.getScheduleAccessManager();
	}

	protected CalendarEventAssignment getCalEventAssignForTOChoiceAligned(TORequest toReq,
            TOChoice apprChoice, boolean published, ScheduleAccessManager sam) throws Exception {
    	return toReq.getCache().getCalEventAssignForTOChoiceAligned(toReq, apprChoice, false,
    			sam);
    }

	protected Collection getCalEventAssignsForTOChoiceOverlapped(TORequest toReq,
            TOChoice apprChoice, boolean published, ScheduleAccessManager sam) throws Exception {
    	return toReq.getCache().getCalEventAssignsForTOChoiceOverlapped(toReq, apprChoice, true, sam);
    }
	
	protected ID getEmployeeID(TORequest toReq) {
		return toReq.getEmployeeID();
    }

	protected String getDisplayEmployeeName(TORequest toReq) throws BbmEJBCreateException, BbmObjectNotFoundException, BbmFinderException, RemoteException {
		WorkResourceManager wrm = BbmManagerFactory.getWorkResourceManager();
		EmployeeName empName = wrm.getEmployeeNameByID(toReq.getEmployeeID());
		return empName.getDisplayName(RequestUtil.getLocalizerForAppDefLocale());		
	}
}
