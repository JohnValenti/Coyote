package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

/**
 * Title:        TimeOffChoiceHasZeroLengthHV
 * Description:  Hard validator for time off request of type DEBIT: since the request duration is static,
 * 				 create should be hard denied if the duration is 0
 * Copyright:    Copyright (c) 2006
 * Company:
 * @author:		 Swati Tewari
 * @version 1.0
 */

import com.bluepumpkin.ejb.bbm.timeoffcalculator.DailyHoursBuckets;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResultAggregator;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.TORequestUtil;

public class TimeOffChoiceHasZeroLengthHV implements Validator {

	private static String m_className = TimeOffChoiceHasZeroLengthHV.class.getName();

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		TORequest toReq = (TORequest) validatable;
		// only applies if request is of type 'debit' since length is static.
		// for debit-if request and its associated TOEvt length is dynamic
		//Only validate requests with Use my TO hours option
		if (!toReq.isUseFixedHours()) {
			return null;
		}
		ValidationResultAggregator result = new ValidationResultAggregator();
		toReq.addDetailLevel(TORequest.DL_TIMEOFF_CHOICES_LENGTH);
		TORequestUtil.fillTOChoicesLengthIfNecess(toReq, false);

		for (TOChoice toChoice : toReq.getTOChoicesAdjustedForValidation()) {
			result.add(setValidationResultAsNeeded(toChoice));
		}

		return result.get();
	}

	public ValidationResult setValidationResultAsNeeded(TOChoice toChoice) {
		if (toChoice.isZeroLength()) {
			String errorMsg;
			int periodFlag = toChoice.getPeriodFlag();
			switch (periodFlag) {
				case DailyHoursBuckets.DAILYHOURSBUCKETS_PERIOD_SCHED:
					errorMsg = RmEjbBundleKey.TO_CHOICE_HAS_ZERO_LENGTH_SCHED;
					break;
				case DailyHoursBuckets.DAILYHOURSBUCKETS_PERIOD_OVERLAP:
					errorMsg = RmEjbBundleKey.TO_CHOICE_HAS_ZERO_LENGTH_OVERLAP;
					break;
				case DailyHoursBuckets.DAILYHOURSBUCKETS_PERIOD_UNSCHED:
					errorMsg = RmEjbBundleKey.TO_CHOICE_HAS_ZERO_LENGTH_UNSCHED;
					break;
				default: // shouldn't get here, but if we do, use most generic message
					errorMsg = RmEjbBundleKey.TO_CHOICE_HAS_ZERO_LENGTH_OVERLAP;
			}
			return ValidationUtil.setHardValidationResult(toChoice, errorMsg,
			        toChoice.getStartDate(), toChoice.getEndDate(), m_className);
		}
		return null;
	}
}
