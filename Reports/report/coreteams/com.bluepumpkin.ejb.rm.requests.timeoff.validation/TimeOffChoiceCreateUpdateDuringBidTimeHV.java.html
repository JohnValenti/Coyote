package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

/**
 * Title:        TimeOffChoiceCreateUpdateDuringBidTimeHV
 * Description:  Hard validation alert when a Time Off Request entered during open bid time. It is normal time off bids so it should not be saved here -time off request dialog
 * It should be done in Time Off Bid request dialog
 * Copyright:    Copyright (c) 2016
 * Company:      Verint System, Inc
 * @author       Truc Pham
 * @version      1.0
**/
import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.Pair;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.ejb.TimeOffBidUtil;
import com.bluepumpkin.ejb.rm.setup.timeoffbid.model.TimeOffBid;

public class TimeOffChoiceCreateUpdateDuringBidTimeHV implements Validator {

	private static String clsName = TimeOffChoiceCreateUpdateDuringBidTimeHV.class.getName();

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	@SuppressWarnings("rawtypes")
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResult result = null;
		TORequest toReq = (TORequest) validatable;
		ID timeOffType = toReq.getTimeOffType();
		ID employeeId = toReq.getEmployeeID();
		//If new request: toReq.getSubmittedOn()
		//If modify an existing request: toReq.getLastModifiedAt()
		Date submittedRequestDate = toReq.getLastModifiedAt() == null ? toReq.getSubmittedOn() : toReq.getLastModifiedAt();
		Pair<TimeOffBid, Pair<Integer, Integer>> bidDetail = toReq.getTimeOffBidInfo();
		if (bidDetail == null) {
			bidDetail = getOpenTimeOffBidForActivityIdAndEmployeId(timeOffType, employeeId, toReq.getSubmittedOn());
			toReq.setTimeOffBidInfo(bidDetail);
		}
		if (bidDetail == null) {
			return null;
		}
		TimeOffBid bid = bidDetail.getFirst();
		if (bid == null) {
			return null;
		}
		long bidRequestStart = bid.getRequestStartTime().getTime();
		long bidRequestEnd = bid.getRequestEndTime().getTime();
		Date bidEnd = bid.getBidEndTime();
		for (Iterator it = toReq.getRequestChoiceList().iterator(); it.hasNext();) {
			TOChoice choice = (TOChoice) it.next();
			long choiceStart = choice.getStartDate().getTime();
			long choiceEnd = choice.getEndDate().getTime();
			boolean choicePeriodInBidRequestPeriod = bidRequestStart <= choiceStart && choiceEnd <= bidRequestEnd;
			//Only preventing creating time off bid if choice Period is in bid request period and the submitted Request date before bid End 
			if (choicePeriodInBidRequestPeriod && submittedRequestDate.before(bidEnd)) {
				result = ValidationUtil.setHardValidationResult(choice, RmEjbBundleKey.TIMEOFF_TOCHOICE_MUST_NOT_SAVE_DURING_BID_TIME,
						choice.getStartDate(), choice.getEndDate(), clsName);
				return result;
			}
		}

		return result;
	}

	protected Pair<TimeOffBid, Pair<Integer, Integer>> getOpenTimeOffBidForActivityIdAndEmployeId(ID timeOffType, ID employeeId,
			Date submittedDate) throws Exception {
		return TimeOffBidUtil.getOpenTimeOffBidDetailForActivityIdAndEmployeId(timeOffType, employeeId, submittedDate);
	}
}
