/**
 * Rule in UI: Requested Time Off is for the past date
 * <p/>
 * Check if request is for the past date. this rule will raise alert for only
 * those requests that are created for start time that falls before submitted
 * date of the request. I believe this prevents users from submitting requests
 * for timeoff taken in the past.
 * <p/>
 */
public class RequestInPastValidationRule extends TimeoffValidationRule {
	private static final String CLASSNAME = RequestInPastValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASSNAME);

	@Override
	protected void validateTimeOffChoiceGroup(TORequest request, TOChoiceGroup choiceGroup) throws Exception {
		String methodName = "validateTimeOffChoiceGroup";
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpEnterMethod(methodName, request));
		}
		  // Get the scheduled shifts
		ValidationResult result = null;
		if(request.isAdvancedVTORequest()&& !request.isVTOPartialDay()) {
		// QC 191594: When a VTO options request (Full day,Late Start, Early Off, Anytime Off, not Partial Day) is submitted for the same day,
		//the alert will check the schedule hours instead of 12am-12am 
			TOChoice toChoice = (TOChoice) choiceGroup.getChoices().iterator().next();
			Date startDate = toChoice.getStartDate();
			Date endDate = toChoice.getEndDate();
			boolean shiftAssignmentBeforeSubmitted = true;
			Collection<ShiftAssignment> shiftAssignments = getPublishedShiftAssignments(request.getEmployeeID(),startDate,endDate);
			//Please note for the case of shift across over night 
			// E.g 1/17: 9PM to 7AM; 1/18: 9PM to 7AM
			//In this case when getting published shift assignments for employee on the 1/18, there are 2 shift assignments ( from 0AM to 7AM and from 9PM to 12AM of 1/19)
			//Therefore, if submitting the request at 3PM,  the check here should make sure all shift assignments before the submitted day.
			//=> with this check, it should not be alerted since second assignment 9PM to 12 AM of 1/19 is after submitted day
			for (Iterator<ShiftAssignment> i = shiftAssignments.iterator(); i.hasNext(); ) {
				ShiftAssignment sa = (ShiftAssignment)i.next();
				if (!sa.getStartTime().before(request.getSubmittedOn())){
					shiftAssignmentBeforeSubmitted=false;
					break;
				}
			}
			if(shiftAssignmentBeforeSubmitted) {
				result = ValidationUtil.setSoftValidationResult(toChoice, RmEjbBundleKey.TO_REQUEST_FOR_PAST_DATE,
						getEmployeeNameByID(request),  startDate,endDate, CLASSNAME);
			}
		} else {
			for (TOChoice toChoice : choiceGroup.getChoices()) {
				//For TOB groupApproval,if the choice violates any validation rule ran before , do not run the rule and continue with next choice
				if(request.isRunForGroupTOBApproval()&& !toChoice.getValidationResults(true).isEmpty()){
					continue;
				}
				if (!validateTOChoice(request, toChoice)) {
					Date startDate = toChoice.getStartDate();
					Date endDate = toChoice.getEndDate();
					
					// Build validation result, store in choice.
					result = ValidationUtil.setSoftValidationResult(toChoice, RmEjbBundleKey.TO_REQUEST_FOR_PAST_DATE,
							getEmployeeNameByID(request), startDate, endDate, CLASSNAME);
				}
			}
		}
		if (LOG.isDebugEnabled()) {
			LOG.debug(RmUtil.dumpExitMethod(methodName, result));
		}
	}

	// Returns false if it fails validation
	boolean validateTOChoice(TORequest request, TOChoice choice) {
		return !TORequestUtil.isTimeOffChoiceForPastDate(request, choice);
	}

	protected String getEmployeeNameByID(TORequest request) throws Exception {
		return request.getCache().getEmployeeNameByID();
	}

	@SuppressWarnings("unchecked")
	protected Collection<ShiftAssignment> getPublishedShiftAssignments( ID empId, Date startTime, Date endTime) throws Exception {
		ScheduleAccessManager sam = RequestUtil.getScheduleAccessManager();
		return sam.getPublishedEventsForWorkResourceByType(Event.EVENT_TYPE_SHIFT_ASSIGNMENT, empId, startTime, endTime);
	}
}
