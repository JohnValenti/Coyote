package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.Date;
import java.util.Iterator;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.workresource.model.Employee;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.RmUtil;

/**
 * "Agent in good standing" validation rule for time off requests.  
 * Verify if each TOChoice falls within the employee's start and end time.
 *
 * <p>
 * Applies equally for debit, don't debit and debit-if requests
 *
 * <p>
 * Applies equally for tentative and non tentative requests (no schedule information is used).
 *
 * <p>
 * Validation result added to the request itself.
 */
public class AgentGoodStandingValidationRuleHV implements Validator {
	public static final String CLASS_NAME = AgentGoodStandingValidationRuleHV.class.getName();
    private static Category m_cat = Log.initCategory(CLASS_NAME);

    /**
     * Validate a time off request.  Ensure that the employee for whom the request
     * is for is in good standing for all the choices in the
     * requested time off.
     */
    @Override
	public ValidationResult validate(Validatable validatable) throws Exception {
        return validateAgentGoodStandingValidationRule(validatable,true);
    }
    protected ValidationResult validateAgentGoodStandingValidationRule(Validatable validatable,boolean isHard) throws Exception {
        ValidationResult result = null;
        
        TORequest tor = (TORequest)validatable;

        String methodName = "validate";
        if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		}

        // Get employee data including name, start and end dates
        Employee emp = getEmployee(tor, Employee.DETAIL_LEVEL_EMPLOYEE_BASIC);
        Date empStart = emp.getStartTime();
        Date empEnd   = emp.getEndTime();

		// for each TOChoice
        for (Iterator<TOChoice> it = tor.getRequestChoiceList().iterator(); it.hasNext(); ) {
            TOChoice toChoice = (TOChoice)it.next();
            if(tor.isRunForGroupTOBApproval()&& !toChoice.getValidationResults(true).isEmpty()){
				continue;
			}
            if (toChoice.getStartDate().before(empStart)) {
                // requesting time off before start date
            	if(isHard){
	                result = ValidationUtil.setHardValidationResult(tor,
	                    RmEjbBundleKey.TO_TIMEOFF_BEFORE_START, getEmployeeNameByID(tor),
	                    toChoice.getStartDate(), empStart, this.getClass().getName());
	
	                break;
                } else {
                	result = ValidationUtil.setSoftValidationResult(toChoice,
                            RmEjbBundleKey.TO_TIMEOFF_BEFORE_START, getEmployeeNameByID(tor),
                            toChoice.getStartDate(), empStart, this.getClass().getName());
                }
            }

			// if employee is or will be terminated
            if ( (empEnd != null) && (toChoice.getEndDate().after(empEnd))) { 
                // requesting time off after end date
            	if(isHard){
	                result = ValidationUtil.setHardValidationResult(tor,
	                    RmEjbBundleKey.TO_TIMEOFF_AFTER_TERM, getEmployeeNameByID(tor),
	                    toChoice.getEndDate(), empEnd, this.getClass().getName());
	
	                break;
                }else{
                	 result = ValidationUtil.setSoftValidationResult(toChoice,
                             RmEjbBundleKey.TO_TIMEOFF_AFTER_TERM, getEmployeeNameByID(tor),
                             toChoice.getEndDate(), empEnd, this.getClass().getName());
                }
            }
        }

        if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
		}
        return result;
    }
	protected String getEmployeeNameByID(TORequest tor) throws Exception {
		return tor.getCache().getEmployeeNameByID();
	}

    protected Employee getEmployee(TORequest tor, long empDetailLevel) throws Exception {
    	return tor.getCache().getEmployeeBasedOnEmpDetailLevel(empDetailLevel);
    }
}
