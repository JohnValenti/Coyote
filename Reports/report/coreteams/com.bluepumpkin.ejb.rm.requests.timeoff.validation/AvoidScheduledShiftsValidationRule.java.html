package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import java.util.Iterator;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.timeoffcalculator.TimeMap;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.RmUtil;
 
/**
 * Rule in UI: Request avoids scheduled shifts.
 *
 * Check that there is no shift already scheduled during the requested time off.
 * 
 * <p>
 * Applies for only debit and debit-if.  Does not apply for don't debit as created unavailability event
 * cannot overlap a shift (rule not skipped as no shift will be found (enforced by hard validation)).
 * 
 * <p>
 * Applies equally for tentative and non tentative requests as unpublished schedule not used.
 * 
 */
public class AvoidScheduledShiftsValidationRule implements Validator {

	private static final String m_className = AvoidScheduledShiftsValidationRule.class.getName();
    
    private static final Category m_cat = Log.initCategory( m_className );
    
   /**
    * Validate a time off request.  
    */
   @Override
public ValidationResult validate(Validatable validatable) throws Exception {
        ValidationResult result = null;
        TORequest tor = (TORequest)validatable;
        
		String methodName = "validate";
		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpEnterMethod(methodName, validatable));
		}

        // Check against scheduled shifts, sets results in choices and returns last result
        // set
        TimeMap shiftsTm = tor.getCache().getShiftsPubTimeMap();
      
        // Loop through the choices and check that each choice does not overlap a
        // scheduled shift.
        for (TOChoice c : tor.getRequestChoiceList()) {
			//For TOB groupApproval,if the choice violates any validation rule ran before , do not run the rule and continue with next choice
			if(tor.isRunForGroupTOBApproval()&& !c.getValidationResults(true).isEmpty()){
				continue;
			}
            if (shiftsTm.onBitsOverlapRange(c.getStartDate(), c.getEndDate() )) {
                // The given range overlaps scheduled shifts.  Validation fails
                result = ValidationUtil.setSoftValidationResult(c,
                	RmEjbBundleKey.TO_CHOICE_OVERLAPS_PUB_SHIFTS, 
                    tor.getCache().getEmployeeNameByID(), c.getStartDate(), c.getEndDate(),
                    tor.getCache().getOrgNameForMsg(), m_className);
            }
          } 

		if (m_cat.isDebugEnabled()) {
			m_cat.debug(RmUtil.dumpExitMethod(methodName, result));
		}
		
        return result;
   }

}
