package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

/**
 * Title:        TimeOffBidPeriodPerChoice
 * Description:  Soft mandatory validation alert when a Time Off Request of activity type specified for a bid is made during an open bid period by an employee in the bid,
 * 				 for a period in the request period of the bid, the rule for "Minimum of X Weeks/Days per Choice" and "Maximum of X Weeks/Days per Choice"
 * 				 is applied upon submission where appropriate. Besides that, if the minimum/maximum unit is week, the bid duration should be increments of weeks
 * 				 This validation rule is done for Group Approval for Time Off Bid choice,  if there is Advanced RM license
 * Copyright:    Copyright (c) 2016
 * Company:      Verint System, Inc
 * @author       Truc Pham
 * @version      1.0
**/
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;

/**
 *         <p/>
 *        When a Time Off Request of activity type specified for employee added to the bid while the bid is open,
 * 		  for a period in the request period of the bid, the hard validation rule for "Minimum of X Weeks/Days per Choice" and "Maximum of X Weeks/Days per Choice"
 * 			is applied upon submission where appropriate...
 * 		  Besides that, if the minimum/maximum uni is week, the bid duration should be increments of weeks
 *         <p/>
 *         <p> Note: This validation is invoked only if there is Advanced RM license.
 */
public class TimeOffBidPeriodPerChoice implements Validator {

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		TimeOffBidPeriodPerChoiceHV validator = new TimeOffBidPeriodPerChoiceHV();
		return validator.validateForTimeOffBidChoice(validatable,false);
	}
}
