package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.schedule.model.CalendarEventAssignment;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResultAggregator;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;

import com.bluepumpkin.ejb.rm.util.TORequestUtil;

/**
 * @author Sameet Joshi
 *         <p/>
 *         Raises an Hard validation alert if the approved  TO Choice has no underlying event.
 *         <p/>
 *         <p> Note: This validation is invoked only if the request is approved.
 */
public class TimeOffChoiceHasNoUnderlyingEvent implements Validator {

	private static String m_className = TimeOffChoiceHasNoUnderlyingEvent.class.getName();
	private static final Category m_cat = Log.initCategory(m_className);

	/* (non-Javadoc)
	 * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
	 */
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		ValidationResultAggregator result = new ValidationResultAggregator();
		TORequest toReq = (TORequest) validatable;
		if (toReq.isApproved()) {

			try {

				for (TOChoice apprChoice : toReq.getApprovedChoices()) {
					ValidationResult r = checkEvents(toReq, apprChoice);
					result.add(r);
				}

			} catch (Exception e) {
				result.add(ValidationUtil.setHardValidationResult(toReq, RmEjbBundleKey.TIMEOFF_WITHDRAW_UNDERLYING_EVENT_MISMATCH,
						m_className));
			}
		}
		return result.get();
	}

	private ValidationResult checkEvents(TORequest toReq, TOChoice apprChoice) throws Exception {
		ValidationResult result = null;
		CalendarEventAssignment eventPub = TORequestUtil.getCalEventAssignForTOChoiceAligned(toReq, apprChoice, true, null);
		if (eventPub == null) { //check if event exists in published world.
			result = ValidationUtil.setHardValidationResult(toReq, RmEjbBundleKey.TIMEOFF_WITHDRAW_UNDERLYING_EVENT_MISMATCH, m_className);
		} else { //check if event exists in un-published world.
			CalendarEventAssignment eventUnPub = TORequestUtil.getCalEventAssignForTOChoiceAligned(toReq, apprChoice, false, null);
			if (eventUnPub == null) {
				result = ValidationUtil.setHardValidationResult(toReq, RmEjbBundleKey.TIMEOFF_WITHDRAW_UNDERLYING_EVENT_NOT_EXIST_IN_UNPUB,
						m_className);
			}
		}
		return result;
	}


}
