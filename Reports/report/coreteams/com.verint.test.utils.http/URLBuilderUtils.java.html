package com.verint.test.utils.http;

import com.bluepumpkin.common.datatypes.ID;
import com.verint.test.constants.ParameterKeys;
import com.verint.test.constants.PathKeys;
import org.apache.http.HttpResponse;
import org.apache.http.client.HttpClient;
import org.apache.http.client.methods.HttpGet;
import org.apache.http.client.methods.HttpPost;
import org.apache.http.entity.StringEntity;
import org.apache.http.impl.client.HttpClientBuilder;
import org.apache.http.util.EntityUtils;
import org.json.JSONObject;

import java.net.URLEncoder;
import java.util.HashMap;
import java.util.Iterator;
import java.util.List;
import java.util.Map;

public class URLBuilderUtils {
	public static boolean isDevDomain = false;
	private static String providerURL = null;
	private static String providerRestURL = null;

	public final static String HTTP_CHARACTER_ENCODING = "UTF-8";

	/**
	 * Returns the host of the text fixture, generally handed in as a parameter to the JVM.  By convention,
	 * this is host + port.
	 * @return
	 */
	public static String getHostAndPort() {
		return System.getProperty("host");
	}

	/**
	 * Returns the host of the text fixture, generally handed in as a parameter to the JVM. This will return the host
	 * only and strip the port if one was passed in.
	 * @return
	 */
	public static String host() {
		if (System.getProperty("host").indexOf(":") > 0) {
			return System.getProperty("host").split(":")[0];
		}
		return System.getProperty("host");
	}

	public static String port() {
		return System.getProperty("host").split(":")[1];
	}
	
	/**
	 * Returns the host URL from the parameter -Dhost passed from gradle command line 
	 * like -Dhost=vm-sd-dev2:7001 - http//vm-sd-dev2:7001/wfo/control 
	 * if no parameter is passed then returns null
	 * @return 
	 * @throws Exception when the property is not found
	 */
	public static String getProviderURL() throws Exception{
		if (providerURL != null) {
			return providerURL;
		}
		
		String host = getHostAndPort();
		
		if (host == null) {
			throw new Exception("Parameter host not passed from gradle command line");
		}
		
		String extension = PathKeys.WFO_APPLICATION_URL;
		providerURL = "http://" + host + extension;
		return providerURL;
	}
	
	/**
	 * Returns the rest URL from the parameter -Dhost passed from gradle command line
	 * like -Dhost=vm-sd-dev2:7001 - http//vm-sd-dev2:7001/wfo/rest
	 * if no parameter is passed then returns null
	 * @return
	 * @throws Exception
	 */
	public static String getProviderRestURL() throws Exception{
		if (providerRestURL != null) {
			return providerRestURL;
		}

		String host = getHostAndPort();

		if (host == null) {
			throw new Exception("Parameter host not passed from gradle command line");
		}

		String extension = PathKeys.WFO_REST_URL;

		providerRestURL = "http://" + host + extension;
		return providerRestURL;
	}

	/**
	 * Returns the rest URL for calling core or foundation rest api's
	 * @return url string
	 * @throws Exception
	 */
	public static String getCoreRestURL() throws Exception {
		return getProviderRestURL() + PathKeys.CORE_REST_URL;
	}
	
	/**
	 * Builds url from map and path
	 * @param path
	 * @param parameters
	 * @return
	 * @throws Exception
	 */
	public static String buildURL(String path, Map<String, String> parameters) throws Exception {
		StringBuilder builder = new StringBuilder(); 
		builder.append(getProviderURL());
		builder.append(path);
		builder.append("?");

		builder.append(buildParameterString(parameters));

		return builder.toString();
	}

	public static String buildParameterString(Map<String, String> parameters) throws Exception {
		StringBuilder builder = new StringBuilder();
		Iterator<Map.Entry<String, String>> it = parameters.entrySet().iterator();
		while( it.hasNext() ){
			Map.Entry<String, String> entry = it.next();
			builder.append(entry.getKey());
			builder.append("=");
			builder.append(entry.getValue());
			builder.append("&");
		}
		return builder.toString();
	}
	
	/**
	 * Creates Url from path + parameters in map + static request string
	 * @param path
	 * @param parameters
	 * @param urlParams
	 * @return
	 * @throws Exception
	 */
	public static String buildURL(String path, Map<String, String> parameters, String urlParams) throws Exception{
		return buildURL(path,parameters) + urlParams;
	}
	
	/** 
	 * Build url from path and parameters passed as string
	 * @param path
	 * @param parameters
	 * @return
	 * @throws Exception
	 */
	public static String buildURL(String path, String parameters) throws Exception {
		StringBuilder builder = new StringBuilder();
		builder.append(getProviderURL());
		builder.append(path);
		if(parameters == null  || !parameters.isEmpty()){ 
			builder.append("?");
			builder.append(parameters);
		}
		
		return builder.toString();
	}
	
	/**
	 * builds url with path which does not need to send any parameters
	 * @param location
	 * @return
	 * @throws Exception
	 */
	public static String buildURL(String location) throws Exception {
		return buildURL(location, "");
	}
	
	/**
	 * Replaces white spaces with + for strings to append in request parameter
	 * @param param
	 * @return
	 */
	public static String buildParameterString(String param){
		param = param.trim();
		return param.replace(" ", "+");
	}

	/**
	 * Encode URL parameter in order to use for Http Request
	 *
	 * @param param
	 * @return
	 */
	public static String buildUrlParameter(String param) throws Exception {
		return URLEncoder.encode(param, HTTP_CHARACTER_ENCODING);
	}

	public static String createCommaSeparatedParameterStringFromList(List<String> list){
		StringBuilder builder = new StringBuilder();
		int count = 0;
		for (String elem : list) {
			if (count!=0 && count<list.size()) {
				builder.append(ParameterKeys.VALUE_COMMA);
			}
			count++;
			builder.append(elem);
		}
		return builder.toString();
	}
	
	public static Map<String, String> getParametersMapWithDefaultSet(String action){
		Map<String, String> parameters = new HashMap<String, String>();
		setupDefaultParametersSet(parameters);

		if (action != null) {
			parameters.put(ParameterKeys.KEY_PAGE_ACTION, action);
		}

		return parameters;
	}

	public static ParametersMap getParametersMapWithDefaultSet() {
		ParametersMap parameters = new ParametersMap();
		setupDefaultParametersSet(parameters);

		return parameters;
	}

	public static ParametersMap getDefaultParameters(String selectedId) {
		ParametersMap parameters = getParametersMapWithDefaultSet();
		parameters.put(ParameterKeys.KEY_SELECTED_ID, selectedId);

		return parameters;
	}

	/**
	 * Set Up common default form parameters.
	 *
	 * @param parameters
	 * @return
	 */
	public static void setupDefaultParametersSet(Map<String, String> parameters) {
		parameters.put(ParameterKeys.KEY_PAGE_DIRTY, ParameterKeys.VALUE_BOOLEAN_FALSE);
		parameters.put(ParameterKeys.KEY_PERFORM_ACTION_ON_ID, "" );
		parameters.put(ParameterKeys.KEY_EDIT_INDEX,  ParameterKeys.VALUE_STRING_ZERO);
		parameters.put(ParameterKeys.KEY_PAGE_MODEL_TYPE, ParameterKeys.VALUE_STRING_ONE);
		parameters.put(ParameterKeys.KEY_NEWNAV_UI, ParameterKeys.VALUE_STRING_ONE);
	}

	/**
	 * REST request for POST Verb.
	 * Attach the payload as JSONObject
	 * @param url url to rest api call
	 * @param authToken needs authToken to make a call or null if trying to get authorized token
	 * @param payload Json as payload
	 * @return
	 * @throws Exception
	 */
	public static JSONObject postRestRequest(String url, String authToken, JSONObject payload) throws Exception {
		HttpClient client = HttpClientBuilder.create().build();

		HttpPost postRequest = new HttpPost(url);
		StringEntity params = new StringEntity(payload.toString());

		postRequest.addHeader("content-type", "application/json");
		//if the request is to get a auth key
		if (authToken != null) {
			postRequest.addHeader("Impact360AuthToken", authToken);
		}

		postRequest.setEntity(params);
		HttpResponse response = client.execute(postRequest);
		String result = EntityUtils.toString(response.getEntity());
		JSONObject responseObject = new JSONObject(result);
		return responseObject;
	}

	/**
	 * Rest request for GET verb
	 * No payload all the parameters are present in the URL itself
	 * @param url url to rest api call
	 * @param authToken needs authToken to make a call
	 * @return
	 * @throws Exception
	 */
	public static JSONObject getRestRequest(String url, String authToken) throws Exception {
		if (authToken == null) {
			return new JSONObject();
		}

		HttpClient client = HttpClientBuilder.create().build();
		HttpGet getRequest = new HttpGet(url);

		getRequest.addHeader("content-type", "application/json");
		getRequest.addHeader("Impact360AuthToken", authToken);

		HttpResponse response = client.execute(getRequest);
		String result = EntityUtils.toString(response.getEntity());
		JSONObject responseObject = new JSONObject(result);
		return responseObject;
	}

}
