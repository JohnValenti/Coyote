package com.bluepumpkin.web.fs.calendar.data;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.NullablePair;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.pulse.model.TraceChart;
import com.bluepumpkin.ejb.bbm.pulse.model.TrackingView;
import com.bluepumpkin.ejb.bbm.timeseries.model.Trace;
import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectUtil;
import com.bluepumpkin.ejb.bbm.workload.model.Media;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;

import java.util.*;

public class SPQueueTraceCubeData {

	private static final int COMPLEX_SP_QUEUES = 25;
	private static final int TRACE_CHARTS_NUMBER = 18;
	private static final int TRACE_CHARTS_NO_PHANTOM_NUMBER = 4;

	private SchedulingPeriod sp;
	private Collection<SPQueue>spQueues;
	private boolean hasAppMediaOutboundLicense;
	
	private boolean isComplexSP = false;
	private boolean hasOutboundQueue = false;
	private boolean hasDeferredQueue = false;

	Collection<ID> queueIDCollection;
	Collection<ID> mediaIDCollection;
	Map<ID, Queue> idQueueMap;

	Map<NullablePair<ID, ID>, TraceCube> mapQueueRibbons;
	Map<NullablePair<ID, ID>, TraceCube> mapQueueRibbonsRequiredLine;
	Map<NullablePair<ID, ID>, TraceCube> mapQueueRibbonsNoPhantomLine;
	
	TrackingView tv;
	TrackingView tvNoPhantom;
	
	
	public SPQueueTraceCubeData(SchedulingPeriod sp, Collection<SPQueue> spQueues, boolean hasAppMediaOutboundLicense) {
		this.sp = sp;
		this.spQueues = spQueues;
		this.hasAppMediaOutboundLicense = hasAppMediaOutboundLicense;
		queueIDCollection = new ArrayList<ID>();
		mediaIDCollection = new ArrayList<ID>();
		idQueueMap = new HashMap<ID, Queue>();
		mapQueueRibbons = new HashMap<NullablePair<ID, ID>, TraceCube>();
		mapQueueRibbonsRequiredLine = new HashMap<NullablePair<ID, ID>, TraceCube>();
		mapQueueRibbonsNoPhantomLine = new HashMap<NullablePair<ID, ID>, TraceCube>();
	}
	
	public void init(Collection<Queue> queues) {
		idQueueMap.putAll(ValueObjectUtil.getIDObjectMap(queues));
		int numNonProjectQueues = 0;
		
		for (SPQueue spQ : spQueues) {
			ID qID = spQ.getQueueID();
			queueIDCollection.add(qID);
			Queue queue = idQueueMap.get(spQ.getQueueID());

			if (!mediaIDCollection.contains(queue.getMediaID())) {
				mediaIDCollection.add(queue.getMediaID());
			}

			if (Media.isMediaDeferred(queue.getMediaID())) {
				hasDeferredQueue = true;
				isComplexSP = true;
			}

			if (queue.getMediaID().equals(Media.MEDIA_ID_PHONE_OUTBOUND)) {
				hasOutboundQueue = true;
			}

			if (!queue.getMediaID().equals(Media.MEDIA_ID_PROJECT)) {
				numNonProjectQueues++;
			}
		}
		
		if (!isComplexSP && numNonProjectQueues > COMPLEX_SP_QUEUES) {
			isComplexSP = true;
		}
	}

	public SchedulingPeriod getSp() {
		return sp;
	}
	
	public Collection<ID> getQueueIDCollection() {
		return queueIDCollection;
	}

	public Collection<ID> getMediaIDCollection() {
		return mediaIDCollection;
	}

	public Map<ID, Queue> getIdQueueMap() {
		return idQueueMap;
	}

	public Map<NullablePair<ID, ID>, TraceCube> getMapQueueRibbons() {
		return mapQueueRibbons;
	}

	public Map<NullablePair<ID, ID>, TraceCube> getMapQueueRibbonsRequiredLine() {
		return mapQueueRibbonsRequiredLine;
	}

	public Map<NullablePair<ID, ID>, TraceCube> getMapQueueRibbonsNoPhantomLine() {
		return mapQueueRibbonsNoPhantomLine;
	}

	public boolean isComplexSP() {
		return isComplexSP;
	}
	
	public TrackingView getTv() {
		if (tv == null) {
			tv = new TrackingView();
			List list = new ArrayList(TRACE_CHARTS_NUMBER);
			list.add(new TraceChart(Trace.CV, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			list.add(new TraceChart(Trace.AHT, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			list.add(new TraceChart(Trace.PCA, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE, TraceChart.REQUIRE_LINE}, true));
			list.add(new TraceChart(Trace.ASA, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE, TraceChart.REQUIRE_LINE}, true));
			list.add(new TraceChart(Trace.ABANDONMENT, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE,
					TraceChart.REQUIRE_LINE},true));

			if (hasDeferredQueue || hasOutboundQueue) {
				list.add(new TraceChart(Trace.BACKLOG, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			}

			list.add(new TraceChart(Trace.STAFFING, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			list.add(new TraceChart(Trace.OCCUPANCY, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			list.add(new TraceChart(Trace.FTE, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE, TraceChart.REQUIRE_LINE}, true));
			list.add(new TraceChart(Trace.ASTAFFING, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			list.add(new TraceChart(Trace.AFTE, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			list.add(new TraceChart(Trace.NETSTAFFING, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));

			if (hasAppMediaOutboundLicense && hasOutboundQueue) {
				list.add(new TraceChart(Trace.DIALS, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
				list.add(new TraceChart(Trace.CONNECTS, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
				list.add(new TraceChart(Trace.CRATE, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
				list.add(new TraceChart(Trace.RPC, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
				list.add(new TraceChart(Trace.RPCRATE, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
				list.add(new TraceChart(Trace.RPCAHT, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			}

			tv.setChartDefinition(list);
			tv.setHOO(false);
			tv.setTrending(false);
			tv.setNoPhantoms(false);
		}
		return tv;
	}

	public TrackingView getTvNoPhantom() {
		if (tvNoPhantom == null) {
			//Create a separate view for the No Phantom FTE, PCA, ASA, Dials
			tvNoPhantom = new TrackingView();
			List noPhantomList = new ArrayList(TRACE_CHARTS_NO_PHANTOM_NUMBER);
			noPhantomList.add(new TraceChart(Trace.PCA, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			noPhantomList.add(new TraceChart(Trace.ASA, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			noPhantomList.add(new TraceChart(Trace.FTE, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			if (hasAppMediaOutboundLicense && hasOutboundQueue) {
				noPhantomList.add(new TraceChart(Trace.DIALS, TraceChart.ABSOLUTE_CHART, new short[]{TraceChart.FORECAST_LINE}, true));
			}
			tvNoPhantom.setChartDefinition(noPhantomList);
			tvNoPhantom.setHOO(false);
			tvNoPhantom.setTrending(false);
			tvNoPhantom.setNoPhantoms(true);
		}
		return tvNoPhantom;
	}
}
