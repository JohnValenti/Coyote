package com.bluepumpkin.ejb.bbm.timeseries.util;

import org.apfloat.Apfloat;

/**
 * A utility class for obtaining tables of factorial-related values at
 * various levels of precision.
 */
public class FactorialUtil {

	/**
	 * Creates a table of ln(i!), 0 <= i <= n
	 * 
	 * @param n
	 * @return an array of double where the ith element is ln(i!)
	 */
	public static double[] createLowPrecisionFactorialLogTable(int n) 
	{
		double[] array = new double [n + 1];
		array[0] = 0;
		for (int i = 1 ; i <= n ; i++)
		{	
			array[i] = array[i-1] + Math.log(i);
		}
		return array;
	}

	/**
	 * Creates a table of 1/(i!), 0 <= i <= n
	 * 
	 * We precalculate a table of factorial reciprocals to speed the
	 * calculation of (A^k)/(k!).  Rather than storing factorials and
	 * dividing, we store factorial reciprocals and multiply.
	 * 
	 * @param n
	 * @return an array of Apfloat where the ith element is 1/(i!)
	 */
	public static Apfloat[] createHighPrecisionFactorialReciprocalTable(int n, int precision)
	{
		Apfloat[] factorialReciprocals = new Apfloat[n + 1];
		// This precision setting cascades through to all subsequent calculations
		// with Apfloat objects; changing this value will affect all the
		// multi-precision calculations.
		Apfloat factorialReciprocal = new Apfloat(1, precision);
		factorialReciprocals[0] = factorialReciprocal;
		for (int i = 1; i <= n; i++)
		{
			factorialReciprocal = factorialReciprocal.divide(new Apfloat(i));
			factorialReciprocals[i] = factorialReciprocal;
		}
		return factorialReciprocals;
	}
}
