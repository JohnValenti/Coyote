package com.bluepumpkin.ejb.bbm.timeseries.util;

import java.util.Date;

import com.bluepumpkin.ejb.bbm.timeseries.model.TraceCube;

/**
 * A service level based on the average length of time a caller will have to wait,
 * modeling abandonment as a percentage of the total volume.
 */
public class AverageSecondsToAnswerWithAbandonment
	implements IServiceGoal
{
	private final TraceCube m_serviceLevelTraceCube;
	private final short m_serviceLevelTraceType;
	private final int m_percentAbandonment;

	/**
	 *  Creates a service level indicating that on average calls will be answered within
	 * a given number of seconds and assuming that <code>percentAbandonment</code>
	 * callers will abandon the queue before they are answered.
	 * 
	 * @param serviceLevelTraceCube the trace cube containing ASA data
	 * @param serviceLevelTraceType the trace type for the ASA data
	 * @param percentAbandonment the percentage of callers that will abandon before being served
	 */
	public AverageSecondsToAnswerWithAbandonment(TraceCube serviceLevelTraceCube, short serviceLevelTraceType, int percentAbandonment) {
		m_serviceLevelTraceCube = serviceLevelTraceCube;
		m_serviceLevelTraceType = serviceLevelTraceType;
		m_percentAbandonment = percentAbandonment;
	}

	/**
	 * Return the number of people required to satisfy this service level given an
	 * arrival rate and average handling time.
	 * 
	 * @param arrivalRate
	 * @param averageHandlingTime
	 * @return
	 */
	public int getFteRequirement(Date time, double arrivalRate, double averageHandlingTime) {
		double targetAsa = m_serviceLevelTraceCube.getTraceValue(m_serviceLevelTraceType, time);
		if (targetAsa <= 0) {
			return 0;
		}
		return ErlangC.getInstance().getAgentsAsaWithAbandonment(arrivalRate, averageHandlingTime, targetAsa, m_percentAbandonment);
	}
}
