package com.bluepumpkin.web.fs.schedule.publish;

import java.util.ArrayList;
import java.util.Calendar;
import java.util.Collection;
import java.util.Date;
import java.util.List;
import java.util.ResourceBundle;

import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.ejb.core.security.model.User;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.form.FormFourColumnPC;
import com.witness.web.uif.pagecomponent.form.FormTwoColumnPC;
import com.witness.web.uif.pagecomponent.picker.date.DatePickerPC;
import com.witness.web.uif.pagecomponent.picker.date.DateRangePickerPC;
import com.witness.web.uif.pagemodel.ContainersInRowsPM;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.pagemodel.WorkpanePageModel;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.html.HtmlBooleanCheckbox;
import com.witness.web.uif.util.js.JSUtil;

/**
 * Title:       SchedulePubPM
 * Description: Schedule Publish Page Model
 * Copyright:   Copyright (c) 2004
 * Company:     Blue Pumpkin Software, Inc
 *
 * @author David Su, dsu
 * @version 1.0
 *          Created on Aug 24, 2004, 9:21:44 AM
 */
public class SchedulePubPM extends WorkpanePageModel implements ContainersInRowsPM {
	private List m_containerList = new ArrayList(1);
	private boolean m_isAuthorized = false;
	private ResourceBundle m_bundle;
	private DateRangePickerPC m_dateRangePickerPC_1;
	private DateRangePickerPC m_dateRangePickerPC_2;
	private boolean m_publishPast = false;
	private boolean m_publishFuture = false;

	/**
	 * Constructor
	 */
	public SchedulePubPM(RequestContext context) {
		super(context, BASIC_MODEL);
		m_isAuthorized = isAuthorized(context.getUser());
		m_bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);
		setJSMediator(JavaScriptFileID.MEDIATOR_WORKPANE_GENERIC);

		m_dateRangePickerPC_1 = new DateRangePickerPC(context, "past");
		m_dateRangePickerPC_2 = new DateRangePickerPC(context, "future");
		addChildComponent(m_dateRangePickerPC_1);
		addChildComponent(m_dateRangePickerPC_2);
	}

	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, SchedulePubPM.class);
	}

	/**
	 * Initialize Model with Data
	 */
	protected void initializeModel() {
		if (m_isInitialized) return; else m_isInitialized = true;

		initContentTitle();
		initContainers();
		initToolbar();
	}

	/**
	 * Initialize Content Title
	 */
	protected void initContentTitle() {
		String pageTitle = i18n(m_bundle, FsWebBundleKey.SCHEDULE_PUB);
		m_contentTitlePC.setPageTitle(pageTitle);
	}

	/**
	 * Initialize Containers
	 */
	protected void initContainers() {
		FormTwoColumnPC descPC = new FormTwoColumnPC(m_context);
		descPC.setIsBorderEnabled(true);
		m_containerList.add(descPC);
		// Desc Row
		descPC.addRow(
				i18n(m_common_bundle, UIFWebBundleKey.DESCRIPTION),
				i18n(m_bundle, FsWebBundleKey.SCHEDULE_PUB_DESC) + "<br><br>"
		);

		FormFourColumnPC formPC = new FormFourColumnPC(m_context);
		formPC.setIsBorderEnabled(true);
		formPC.setCssClasses("formRowDark", "formRowLight");
		String chkbk_1 = HtmlBooleanCheckbox.makeInput("publishPast", null, JSUtil.ON_CHANGE, 
													   isPublishPast(), false);

		// Date Date Range Row
		Calendar aCal = Calendar.getInstance(m_context.getViewingTimeZone());
		aCal.add(Calendar.DATE, -1);
		Date yesterday = aCal.getTime();
		aCal.add(Calendar.YEAR, -1);
		Date lastYear = aCal.getTime();
		m_dateRangePickerPC_1.setIsAllDateButtonsEnabled(false);
		DatePickerPC startDatePC = m_dateRangePickerPC_1.getStartDatePickerPC();
		if (startDatePC.getDate()==null) startDatePC.setDate(lastYear);
		startDatePC.setUpperLimit(yesterday);
		DatePickerPC endDatePC = m_dateRangePickerPC_1.getEndDatePickerPC();
		if (endDatePC.getDate()==null) endDatePC.setDate(yesterday);
		endDatePC.setUpperLimit(yesterday);
		formPC.addRowData(i18n(m_bundle, FsWebBundleKey.SCHEDULE_PUB_PAST_SCHEDULES),
						  chkbk_1,
						  i18n(m_common_bundle, UIFWebBundleKey.DATE_RANGE),
						  m_dateRangePickerPC_1.getHtmlDisplay()
						  );

		String chkbk_2 = HtmlBooleanCheckbox.makeInput("publishFuture", null, JSUtil.ON_CHANGE, 
													   isPublishFuture(), false);

		// Date Date Range Row
		aCal = Calendar.getInstance(m_context.getViewingTimeZone());
		Date today = aCal.getTime();
		aCal.add(Calendar.YEAR, 1);
		Date nextYear = aCal.getTime();
		m_dateRangePickerPC_2.setIsAllDateButtonsEnabled(false);
		startDatePC = m_dateRangePickerPC_2.getStartDatePickerPC();
		if (startDatePC.getDate()==null) startDatePC.setDate(today);
		startDatePC.setLowerLimit(today);
		endDatePC = m_dateRangePickerPC_2.getEndDatePickerPC();
		if (endDatePC.getDate()==null) endDatePC.setDate(nextYear);
		endDatePC.setLowerLimit(today);
		formPC.addRowData(i18n(m_bundle, FsWebBundleKey.SCHEDULE_PUB_FUTURE_TIMEOFF),
						  chkbk_2,
						  i18n(m_common_bundle, UIFWebBundleKey.DATE_RANGE),
						  m_dateRangePickerPC_2.getHtmlDisplay()
						  );
		m_containerList.add(formPC);
	}

	/**
	 * Initialize toolbar
	 * This is the default toolbar, dervided classes should override this method
	 */
	protected void initToolbar() {
		String btnLabel = i18n(m_common_bundle, UIFWebBundleKey.PUBLISH);
		m_toolbar.addValidateSubmitTextButton(PageAction.PUBLISH, btnLabel, m_isAuthorized);
	}

	/**
	 * Return true if current user is authorized to page model
	 */
	public static boolean isAuthorized(User user) {
		return user!=null && user.isSuperUser();
	}

	public boolean isPublishPast() {
		return m_publishPast;
	}

	public boolean isPublishFuture() {
		return m_publishFuture;
	}

	public void setPublishPast(boolean bSwtch) {
		m_publishPast = bSwtch;
	}

	public void setPublishFuture(boolean bSwtch) {
		m_publishFuture = bSwtch;
	}

	public Date getFromDatePast() {
		return m_dateRangePickerPC_1.getStartDatePickerPC().getDate();
	}

	public Date getToDatePast() {
		return m_dateRangePickerPC_1.getEndDatePickerPC().getDate();
	}

	public Date getFromDateFuture() {
		return m_dateRangePickerPC_2.getStartDatePickerPC().getDate();
	}

	public Date getToDateFuture() {
		return m_dateRangePickerPC_2.getEndDatePickerPC().getDate();
	}

	/**
	 * Return List of Containers
	 * <p/>
	 * NOTE: Items in the Collection will be treated either as Page Components or
	 * just a basic Object.
	 */
	public Collection getContainers() {
		return m_containerList;
	}

	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return "schedule_pub";
	}
}
