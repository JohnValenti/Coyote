package com.verint.ejb.bbm.forecast.model;

import java.util.Date;

import com.bluepumpkin.ejb.bbm.workload.model.Queue;

/**
 * A wrapper for ForecastProfileList value objects.
 */
public class OutboundCallsList implements Comparable<OutboundCallsList> {

	private Queue m_queue;
	
	private ForecastProfileList m_wrappedList;
	
	
	public OutboundCallsList(Queue queue, ForecastProfileList list) {
		m_queue = queue;
		m_wrappedList = list;
	}
	
	public ForecastProfileList getForecastProfileList() {
		return m_wrappedList;
	}
	
	public Date getStartTime() {
		return m_wrappedList.getStartTime();
	}
	
	public Date getEndTime() {
		return m_wrappedList.getEndTime();
	}
	
	public void setStartTime(Date startTime) {
		m_wrappedList.setStartTime(startTime);
	}
	
	public void setEndTime(Date endTime) {
		m_wrappedList.setEndTime(endTime);
	}
	
	public int getNumberOfCallsToMake() {
		return m_wrappedList.getLength();
	}
	
	public void setNumberOfCallsToMake(int value) {
		m_wrappedList.setLength(value);
	}
	
	public int getAllowedNumberOfRetries() {
		return m_wrappedList.getRetries();
	}
	
	public void setAllowedNumberOfRetries(int value) {
		m_wrappedList.setRetries(value);
	}
	
	public Queue getQueue() {
		return m_queue;
	}

	/**
	 * Compares this OutboundCallsList to another based on the start and end times.
	 */
	@Override
	public int compareTo(OutboundCallsList other) {
		if (this.getStartTime().compareTo(other.getStartTime()) != 0) {
			return this.getStartTime().compareTo(other.getStartTime());
		} else {
			return this.getEndTime().compareTo(other.getEndTime());
		}
	}

	@Override
	public int hashCode() {
		final int prime = 31;
		int result = 1;
		result = prime * result + ((m_queue == null) ? 0 : m_queue.hashCode());
		result = prime * result
				+ ((m_wrappedList == null) ? 0 : m_wrappedList.hashCode());
		return result;
	}

	@Override
	public boolean equals(Object obj) {
		if (this == obj)
			return true;
		if (obj == null)
			return false;
		if (getClass() != obj.getClass())
			return false;
		OutboundCallsList other = (OutboundCallsList) obj;
		if (m_queue == null) {
			if (other.m_queue != null)
				return false;
		} else if (!m_queue.equals(other.m_queue))
			return false;
		if (m_wrappedList == null) {
			if (other.m_wrappedList != null)
				return false;
		} else if (!m_wrappedList.equals(other.m_wrappedList))
			return false;
		return true;
	}
}
