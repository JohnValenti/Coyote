	/**
	 * Used by the UI to create TimeInterval, which will creat a new TimeRecord
	 * @param PayrollStatus
	 * @return ID, TimeRecord
	 */
	public static ID createPayrollStatus(PayrollStatus status) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();
		try {
			HashMap map = new HashMap();
			// save basic request information
			map.put("BPUSERID", status.getUserID());
			map.put("DESTINATIONFILE", status.getDestination());
			map.put("PAYPERIODID", status.getPayPeriodID());
			map.put("REQUESTTIME", TimeZoneUtil.toTimestamp(status.getRequestTime()));
			map.put("STATUS", new Integer(status.getStatus()));
			map.put("ISFORPOST", JdmoParam.getObject(status.isForPost()));
			map.put("DATASOURCEID", status.getDataSourceID());
			ID id = jdmo.addBatchInsert("PAYROLLSTATUS", map);
			jdmo.executeBatch();
			status.setRequestID(id);
			// save the employees
			Collection col = status.getEmployee();
			for (Iterator it = col.iterator(); it.hasNext(); ) {
				map.clear();
				map.put("EMPLOYEEID", it.next());
				map.put("PAYROLLSTATUSID", id);
				jdmo.addBatchInsert("PAYROLLSTATUSEMPLOYEE", map);
			}
			jdmo.executeBatch();
			return id;
		} finally {
			if (jdmo != null)
				jdmo.cleanUp();
		}
	} //createPayrollStatus
	/**

	public static void updateStatus(ID requestID, int status) throws JdmoException

			jdmo.executePCommand(pStmt, param);
		} finally {

				jdmo.cleanUp();
		}
	}
	/**

	{
		Jdmo jdmo = new Jdmo();
		try {
			Object[] param = new Object[1];

				"delete PAYROLLSTATUS where ID=?";

	}
	/**
	 * poll latest request status, given a fixed return number
	 * @param int, number of requests
	 * @return Collection, PayrollStatus
	 */
	public static Collection pollStatus(int number) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();
		try {
			String pStmt1 =
"select c.NAME, b.STARTDATE, b.ENDDATE, d.NAME, a.ID, BPUSERID, e.USERNAME, DATASOURCEID, PAYPERIODID, REQUESTTIME, DESTINATIONFILE, STATUS, ISFORPOST from PAYROLLSTATUS a, PAYPERIOD b, PAYPERIODTYPE c, DATASOURCE d, BPUSER e where b.ID=a.PAYPERIODID and b.PAYPERIODTYPEID=c.ID and a.DATASOURCEID=d.ID and a.BPUSERID=e.ID order by a.REQUESTTIME desc";
			JdmoQuery jQuery1 = jdmo.createQuery(pStmt1, Jdmo.PARAM_QUERY);
			JdmoRowset rs = jdmo.createRowset(jQuery1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			HashMap result = new HashMap(number);
			ArrayList idCol = new ArrayList(number);
			while (rs.next() && --number>=0) {
				ID id = rs.getID("ID");
				PayrollStatus pStatus = new PayrollStatus(id,
											rs.getID("BPUSERID"),
											rs.getID("PAYPERIODID"),
											rs.getID("DATASOURCEID"),
											rs.getString("DESTINATIONFILE"),
											TimeZoneUtil.toDate(rs.getTimestamp("REQUESTTIME")),
											rs.getBoolean("ISFORPOST"),
											rs.getInt("STATUS"));
				pStatus.setUserName(rs.getString("USERNAME"));
				pStatus.setPayPeriodType(rs.getString(1));
				pStatus.setPayPeriodStartDate(TimeZoneUtil.toLocalDate(rs.getTimestamp(2)));
				pStatus.setPayPeriodEndDate(TimeZoneUtil.toLocalDate(rs.getTimestamp(3)));
				pStatus.setDataSourceName(rs.getString(4));
				pStatus.setErrorMessages(PayrollExportErrorDAO.getPayrollExportErrorForStatus(id));
				idCol.add(id);
				result.put(id, pStatus);
			}
			rs.close();
			if (!idCol.isEmpty()) {
				String pStmt2 =
				"select PAYROLLSTATUSID, count(*) from PAYROLLSTATUSEMPLOYEE where PAYROLLSTATUSID in ";
				pStmt2+=jdmo.createInClause(idCol);
				pStmt2+=" group by PAYROLLSTATUSID";
				JdmoQuery jQuery2 = jdmo.createQuery(pStmt2, Jdmo.PARAM_QUERY);
				JdmoRowset rs2 = jdmo.createRowset(jQuery2, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
				while (rs2.next()) {
					PayrollStatus pStatus = (PayrollStatus)result.get(rs2.getID(1));
					pStatus.setEmployeeNumber(rs2.getInt(2));
				}
				rs2.close();
				ArrayList statusCol = new ArrayList(idCol.size());
				for (Iterator it = idCol.iterator(); it.hasNext(); )
					statusCol.add(result.get(it.next()));
				return statusCol;
			} else {
				return Collections.emptyList();
			}
		} finally {
			if (jdmo != null)
				jdmo.cleanUp();
		}
	}
	/**
	 * load a request status
	 * @param ID, request ID
	 * @return PayrollStatus
	 */
	public static PayrollStatus getStatusByID(ID requestID) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();
		try {
			String pStmt1 =
"select c.NAME, b.STARTDATE, b.ENDDATE, d.NAME, BPUSERID, e.USERNAME, DATASOURCEID, PAYPERIODID, REQUESTTIME, DESTINATIONFILE, STATUS, ISFORPOST from PAYROLLSTATUS a, PAYPERIOD b, PAYPERIODTYPE c, DATASOURCE d, BPUSER e where b.ID=a.PAYPERIODID and b.PAYPERIODTYPEID=c.ID and a.DATASOURCEID=d.ID and a.BPUSERID=e.ID and a.ID=?";
			JdmoQuery jQuery1 = jdmo.createQuery(pStmt1, Jdmo.PARAM_QUERY);
			jQuery1.setParID(1, requestID);
			JdmoRowset rs = jdmo.createRowset(jQuery1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			PayrollStatus pStatus = null;
			if (rs.next()) {
				pStatus = new PayrollStatus(requestID,
											rs.getID("BPUSERID"),
											rs.getID("PAYPERIODID"),
											rs.getID("DATASOURCEID"),
											rs.getString("DESTINATIONFILE"),
											TimeZoneUtil.toDate(rs.getTimestamp("REQUESTTIME")),
											rs.getBoolean("ISFORPOST"),
											rs.getInt("STATUS"));
				pStatus.setUserName(rs.getString("USERNAME"));
				pStatus.setPayPeriodType(rs.getString(1));
				pStatus.setPayPeriodStartDate(TimeZoneUtil.toLocalDate(rs.getTimestamp(2)));
				pStatus.setPayPeriodEndDate(TimeZoneUtil.toLocalDate(rs.getTimestamp(3)));
				pStatus.setDataSourceName(rs.getString(4));
			}
			rs.close();
			if (pStatus != null) {
				String pStmt2 =
				"select count(*) from PAYROLLSTATUSEMPLOYEE where PAYROLLSTATUSID=?";
				JdmoQuery jQuery2 = jdmo.createQuery(pStmt2, Jdmo.PARAM_QUERY);
				jQuery2.setParID(1, requestID);
				JdmoRowset rs2 = jdmo.createRowset(jQuery2, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
				if (rs2.next()) {
					pStatus.setEmployeeNumber(rs2.getInt(1));
				}
				rs2.close();
				pStatus.setErrorMessages(PayrollExportErrorDAO.getPayrollExportErrorForStatus(requestID));
			}
			return pStatus;
		} finally {
			if (jdmo != null)
				jdmo.cleanUp();
		}
	}

	/**
	 * retrieve all requests status in the past given timeframe(in minutes)
	 * @param int, minutes
	 * @param boolean, is Post/Export
	 * @param ID, PayPeriod ID
	 * @return Set, Employee IDs
	 */
	public static Set getLatestRequestStatus(int minutes, boolean isForPost, ID payPeriodID) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();
		try {
			String pStmt1 =
"select distinct(b.EMPLOYEEID) from PAYROLLSTATUS a, PAYROLLSTATUSEMPLOYEE b where a.STATUS = 0 and a.REQUESTTIME>=? and a.ISFORPOST=? and a.PAYPERIODID=? and b.PAYROLLSTATUSID=a.ID";
			JdmoQuery jQuery1 = jdmo.createQuery(pStmt1, Jdmo.PARAM_QUERY);
			Date time = new Date(System.currentTimeMillis()-minutes*60000);
			jQuery1.setParTimestamp(1, TimeZoneUtil.toTimestamp(time));
			jQuery1.setParBoolean(2, isForPost);
			jQuery1.setParID(3, payPeriodID);
			JdmoRowset rs = jdmo.createRowset(jQuery1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
			Set empSet = new HashSet();
			while (rs.next()) {
				empSet.add(rs.getID(1));
			}
			rs.close();
			return empSet;
		} finally {
			jdmo.cleanUp();
		}
	}
