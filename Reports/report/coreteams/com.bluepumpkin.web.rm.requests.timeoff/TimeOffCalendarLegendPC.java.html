package com.bluepumpkin.web.rm.requests.timeoff;

import java.util.*;

import com.witness.web.uif.pagecomponent.dialog.DialogLegendPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;

import com.bluepumpkin.web.rm.l10n.*;
import com.bluepumpkin.ejb.rm.l10n.*;
import com.bluepumpkin.ejb.rm.setup.settings.model.OrganizationSetting;

/**
 * Title:        TimeOffCalendarLegendPC
 * Description:  Dialog Displaying Time Off Calendar Legend
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       dsu (David Su)
 * @version 1.0
 *
 * Created on August 22, 2002, 10:19 AM
 */
public class TimeOffCalendarLegendPC extends DialogLegendPC {
  private ResourceBundle m_bundle;
	private OrganizationSetting m_orgSetting;
  private boolean m_isForEmployee = true;
    
  /**
   * Constructor
   *
   * @param context The Request Context
   */
  public TimeOffCalendarLegendPC(RequestContext context, ToolbarPC toolbar) {
    super(context, toolbar);    
    
    m_bundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);
    
    String title = i18n(m_bundle, RmWebBundleKey.TIMEOFF_CALENDAR_LEGEND);
    setHeader(title);
    
    setItemWrapperStyle("width:400px; height:100px;");
  }
        
	/**
	 * Specify Organization Setting to use
	 */
	public void setOrgSetting(OrganizationSetting orgSetting) {
		m_orgSetting = orgSetting;
	}
	
	/**
	 * Return Organization Setting
	 */
	public OrganizationSetting getOrgSetting() {
		return m_orgSetting;
	}
  
  /**
   * Specify whether Legend is for Employee
   */
  public void setIsForEmployee(boolean bSwitch) {
    m_isForEmployee = bSwitch;
  }
  
  /**
   * Return true if Legend is for Employee
   */
  public boolean isForEmployee() { return m_isForEmployee; }
          
	/**
	 * Return Color
	 */
	private String getColor(String key) {
		if (m_orgSetting==null) return "";
		
		return m_orgSetting.getValue(key);
	}
  
  /**
   * Append Legend Item
   */
  private void initLegendItem() {        
    addLegendItem(
      getColor(RmSettingKey.ORG_RM_SETTINGS_TO_NON_OPERATION_COLOR),
      i18n(m_bundle, RmWebBundleKey.ORG_RM_SETTINGS_TO_NON_OPERATION_LABEL));

    addLegendItem(
      getColor(RmSettingKey.ORG_RM_SETTINGS_TO_BLACKOUT_COLOR),
      i18n(m_bundle, RmWebBundleKey.ORG_RM_SETTINGS_TO_BLACKOUT_LABEL));
    
    addLegendItem(
      getColor(RmSettingKey.ORG_RM_SETTINGS_TO_NON_WORKING_COLOR),
      i18n(m_bundle, RmWebBundleKey.ORG_RM_SETTINGS_TO_NON_WORKING_LABEL));
    
    addLegendItem(
      getColor(RmSettingKey.ORG_RM_SETTINGS_TO_WORKING_COLOR),
      i18n(m_bundle, RmWebBundleKey.ORG_RM_SETTINGS_TO_WORKING_LABEL));

    addLegendItem(
      getColor(RmSettingKey.ORG_RM_SETTINGS_TO_AVAILABLE_COLOR),
      i18n(m_bundle, RmWebBundleKey.ORG_RM_SETTINGS_TO_AVAILABLE_LABEL));
    
    addLegendItem(
      getColor(RmSettingKey.ORG_RM_SETTINGS_TO_NO_AVAILABLE_COLOR),
      i18n(m_bundle, RmWebBundleKey.ORG_RM_SETTINGS_TO_NO_AVAILABLE_LABEL));    
    
    if (m_isForEmployee) {
      addLegendItem(
        getColor(RmSettingKey.ORG_RM_SETTINGS_TO_PENDING_COLOR),
        i18n(m_bundle, RmWebBundleKey.ORG_RM_SETTINGS_TO_PENDING_LABEL));

      addLegendItem(
        getColor(RmSettingKey.ORG_RM_SETTINGS_TO_SCHEDULED_COLOR),
        i18n(m_bundle, RmWebBundleKey.ORG_RM_SETTINGS_TO_SCHEDULED_LABEL));      
    }
  }
	  
  /**
   * Return Dialog Content.
   */
  protected String getDialogContent() {
    initLegendItem();    
    return super.getDialogContent();
  }    
}
