package com.verint.web.rest.wfm;

import java.lang.reflect.Constructor;
import java.lang.reflect.Method;
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

import com.verint.web.rest.wfm.repository.RepositoryLogger;
import com.verint.web.rest.wfm.repository.RepositoryLoggingInterceptor;
import com.verint.web.rest.wfm.repository.ValidateJsonApiFilterParams;
import org.glassfish.hk2.api.Filter;

import org.glassfish.hk2.api.InterceptionService;
import org.glassfish.hk2.utilities.BuilderHelper;
import org.jvnet.hk2.annotations.Service;

import org.aopalliance.intercept.ConstructorInterceptor;
import org.aopalliance.intercept.MethodInterceptor;

@Service
public class HK2InterceptionService implements InterceptionService {
	private final static MethodInterceptor FILTER_PARAM_INTERCEPTOR = new JsonApiFilterParamValidationInterceptor();
	private final static MethodInterceptor LOGGING_INTERCEPTOR = new RepositoryLoggingInterceptor();

	private final static List<ConstructorInterceptor> CONSTRUCTOR_LIST = Collections.emptyList();

	public Filter getDescriptorFilter() {
		return BuilderHelper.allFilter();
	}

	public List<MethodInterceptor> getMethodInterceptors(Method method) {
		List<MethodInterceptor> interceptors = new ArrayList<>();
		if (method.isAnnotationPresent(ValidateJsonApiFilterParams.class)) {
			interceptors.add(FILTER_PARAM_INTERCEPTOR);
		}
		if (method.isAnnotationPresent(RepositoryLogger.class)) {
			interceptors.add(LOGGING_INTERCEPTOR);
		}
		return interceptors;
	}

	public List<ConstructorInterceptor> getConstructorInterceptors(
			Constructor<?> constructor) {
		if (constructor.isAnnotationPresent(ValidateJsonApiFilterParams.class)) {
			return CONSTRUCTOR_LIST;
		}
		return null;
	}
}
