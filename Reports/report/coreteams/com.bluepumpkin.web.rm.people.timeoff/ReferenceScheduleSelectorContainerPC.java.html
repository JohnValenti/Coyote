package com.bluepumpkin.web.rm.people.timeoff;

import java.util.ResourceBundle;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.web.bbm.campaign.sp.EmployeeSchedulingPeriodSelectorPC;
import com.bluepumpkin.web.bbm.people.profile.effectivedate.DataContainerPC;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;

public class ReferenceScheduleSelectorContainerPC extends DataContainerPC {

	private static final long serialVersionUID = 1L;

	/*
	 * The SchedulingPeriodSelectorPC populates two hidden fields, one for the display text and one for the ID
	 * The name is "spSelectorPC" (see the SchedulingPeriodSelectorPC constructor) + SelectorPC.DISPLAY_FN and 
	 * "spSelectoPC"  + SelectorPC.HIDDEN_FN
	 * These turn out to be "spSelectorPCDisplayValue" and "spSelectorPCHiddenValue" 
	 */
	private final EmployeeSchedulingPeriodSelectorPC schedulingPeriodSelector;

	private boolean hasSchedulingPeriod;

	public ReferenceScheduleSelectorContainerPC(RequestContext context, ToolbarPC toolbar) {
		super(context);
		setName("ReferenceScheduleSelectorContainerPC");
		schedulingPeriodSelector = new EmployeeSchedulingPeriodSelectorPC(context, toolbar);
		addChildComponent(schedulingPeriodSelector);
	}


	@Override
	protected void addCustomDataRow() {
		ResourceBundle rmBundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);
		String headerText = i18n(rmBundle, RmWebBundleKey.PEOPLE_REFERENCE_SCHEDULE_HEADER);
		//schedulingPeriodSelector
		if (this.hasSchedulingPeriod) {
			addRow(headerText, schedulingPeriodSelector);
		} else {
			addRow(headerText, i18n(m_common_bundle, UIFWebBundleKey.NO_OPTIONS_AVAILABLE) );
		}
	}

	public void setSchedulePeriodId(ID schedulePeriodId) {
		schedulingPeriodSelector.init(schedulePeriodId);
	}

	public void setHasSchedulingPeriod(boolean hasSchedulingPeriod) {
		this.hasSchedulingPeriod = hasSchedulingPeriod;
	}

	public void setSelectedEmployees(String selectedIDs) {
		schedulingPeriodSelector.setSelectedEmployees(selectedIDs);
	}

}
