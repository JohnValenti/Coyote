package com.bluepumpkin.web.rm.people.timeoff;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Calendar;
import java.util.List;
import java.util.ResourceBundle;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.workresource.model.Employee;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.EmployeeReferenceScheduleAssignment;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.bbm.people.profile.effectivedate.BasePageModel;
import com.bluepumpkin.web.bbm.people.profile.effectivedate.DataContainerPC;
import com.bluepumpkin.web.rm.keys.RmJavaScriptFileID;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.table.TableHeaderPC;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.pagemodel.WorkpaneFieldNames;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

/**
 * The popup dialog that is used to set an effective dated reference schedule for an employee
 */
public class ReferenceScheduleEffectivityPopupPM extends BasePageModel {

	private static final long serialVersionUID = 1L;

	public static final String FORM_ACTION = "ref_schedule_eff_popup";
	//The popup command needs to start with "POPUP_"
	public static final String POPUP_COMMAND = "POPUP_POOL_REF_SCHEDULE";

	//function is defined in ReferenceScheduleEffectivityPopup.js
	private static final String CUSTOM_ON_SET_JS = "referenceScheduleEffectivityPopup.Set()";

	private static final ID[] NO_IDS = {};

	private ID refScheduleId;
	private final transient ResourceBundle rmBundle;

	public ReferenceScheduleEffectivityPopupPM(RequestContext context) {
		super(context, CUSTOM_ON_SET_JS);
		addRequiredJavaScriptFile(RmJavaScriptFileID.REFERENCE_SCHEDULE_EFFECTIVITY_POPUP);
		rmBundle = m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME);
	}

	public static PageModel getInstance(RequestContext context) {
		return getInstance(context, ReferenceScheduleEffectivityPopupPM.class);
	}

	@Override
	protected DataContainerPC createDataContainerPC() {
		return new ReferenceScheduleSelectorContainerPC(m_context, m_toolbar);
	}


	@Override
	protected void initializeModel() {
		if (m_isInitialized) {
			return;
		}

		m_isInitialized = true;
		try {

			ID[] employeeIds = getEmployeeIds();
			if (employeeIds.length == 1) {
				//History is shown only if a single employee is selected. 
				initializeHistory(employeeIds[0]);
			}

			initializeContainer(employeeIds);
			initContentTitle();
			initToolbar();
		} catch (Exception e) {
			handleUnableToLoadDataError(log, e);
		}

	}

	private ID[] getEmployeeIds() {
		if (selectedID == null || selectedID.isEmpty()) {
			return NO_IDS;
		}
		return StringUtil.parseIDString(selectedID);
	}

	private void initializeHistory(ID employeeId) throws Exception {
		if (employeeId == null) {
			return;
		}

		Calendar minStart = Calendar.getInstance();
		minStart.set(Calendar.YEAR, 1900);
		Calendar maxEnd = Calendar.getInstance();
		maxEnd.set(Calendar.YEAR, 2100);
		List<EmployeeReferenceScheduleAssignment> history = TimeOffMH.getValidReferenceScheduleAssignments(employeeId, minStart.getTime(), maxEnd.getTime());
		populateHistory(history);

	}

	private void populateHistory(List<EmployeeReferenceScheduleAssignment> history) {
		m_historyPC.setHeightInPixels(160);
		TableHeaderPC headerPC = m_historyPC.getHeader();
		headerPC.addColumnHeaderData("", m_localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.FROM));
		headerPC.addColumnHeaderData("", m_localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.TO));
		headerPC.addColumnHeaderData("", i18n(rmBundle, RmWebBundleKey.PEOPLE_REFERENCE_SCHEDULE_HEADER));

		List<DefaultMultiColumnNodeData> listModel = new ArrayList<DefaultMultiColumnNodeData>(history.size());

		for (EmployeeReferenceScheduleAssignment data : history) {
			DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData();
			node.setSelectable(false);
			node.add(formatDate(data.getStartTime()));
			node.add(formatEndDate(data.getEndTime()));
			String name = ReferenceScheduleAssignmentViewModel.getScheduleNameFromID(m_context, data.getSchedulingPeriodDEID());
			node.add(name);
			listModel.add(node);
		}

		m_historyPC.setListModel(listModel);
	}


	private void initializeContainer(ID[] selectedEmployees) throws Exception {
		ReferenceScheduleSelectorContainerPC dataContainer = (ReferenceScheduleSelectorContainerPC) m_dataPC;
		dataContainer.setSchedulePeriodId(refScheduleId);
		boolean hasSchedulingPeriod = TimeOffMH.hasAtLeastOneCommonSchedulingPeriod(Arrays.asList(selectedEmployees));
		dataContainer.setHasSchedulingPeriod(hasSchedulingPeriod);
		dataContainer.setSelectedEmployees(selectedID);
	}

	public void setRefScheduleID(ID refScheduleId) {
		this.refScheduleId = refScheduleId;
	}

	public void setPerformActionOnID(String aVal) throws Exception {
		//let the parent class handle it
		setSelectedID(aVal);

		if (m_empID != null) {
				m_employee = ValidationUtil.getEmployeeByID(m_empID, null, Employee.DETAIL_LEVEL_EMPLOYEE_TIMEBASEDPROPERTY_HISTORY);
		}

	}

	private void initContentTitle() {
		m_contentTitlePC.setPageTitle(i18n(m_bundle, BbmWebBundleKey.EFFECTIVE_DATES));
		m_contentTitlePC.setItemNameText(i18n(rmBundle, RmWebBundleKey.PEOPLE_REFERENCE_SCHEDULE_HEADER));
	}

	@Override
	public String getRequiredHTML() {
		//store the empIDs
		return super.getRequiredHTML() + HtmlUtil.makeHiddenInput(WorkpaneFieldNames.PERFORM_ACTION_ON_ID_FN, selectedID);
	}

	@Override
	public String getFormAction() {
		return FORM_ACTION;
	}

}
