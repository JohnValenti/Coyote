package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import com.bluepumpkin.ejb.rm.cache.CacheUtilRM;

import java.util.*;

public class TOAccrualMonthly extends TOAccrual {
	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	Integer dayOfMonth = null;

	public int getAccrualDayOfMonth() {
		if (dayOfMonth == null) {
			int day = 1;
			try {
				//get accrual day of month from org setting
				day = CacheUtilRM.getOrgSetting(org.getID()).getAccrualDayOfMonth();
			} catch (Exception e) {
				e.printStackTrace();
			}
			dayOfMonth = new Integer(day);
		}
		return dayOfMonth.intValue();
	}

	public void incrementPeriod(Calendar calStart) {
		calStart.add(Calendar.MONTH, 1);
		adjustDayOfMonth(calStart);
	}

	public void decrementPeriod(Calendar calStart) {

		calStart.add(Calendar.MONTH, -1);
		adjustDayOfMonth(calStart);
	}

	/*
	 TO Yearly will init to start of the year based on anniversary date of the employee*/
	protected void initCalendarToStartOfPeriod(Calendar calStart) {
		super.initCalendarToStartOfPeriod(calStart);
		adjustDayOfMonth(calStart);
	}

	public void adjustDayOfMonth(Calendar calStart) {
		if (getAccrualDayOfMonth() > 28 && calStart.get(Calendar.DAY_OF_MONTH) < getAccrualDayOfMonth()) {
			int maxDayOfMonth = calStart.getActualMaximum(GregorianCalendar.DAY_OF_MONTH);
			calStart.set(Calendar.DAY_OF_MONTH, (maxDayOfMonth >= getAccrualDayOfMonth()) ? getAccrualDayOfMonth() : maxDayOfMonth);
		}else{
			calStart.set(Calendar.DAY_OF_MONTH, getAccrualDayOfMonth());
		}
	}
}
