package com.bluepumpkin.ejb.rm.requests.timeoff.model;

import com.bluepumpkin.ejb.bbm.vo.ChildInfoEntry;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfoEntry;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;

/**
 * Meta data information for the flex request makeup table, a child table of time off request.
 */
public class FlexRequestMakeupFieldInfo extends FieldInfo {
	public static final int FLEXREQUESTMAKEUP_I_ID = 0;
	public static final int FLEXREQUESTMAKEUP_I_TIMEOFFREQUESTID = 1;
	public static final int FLEXREQUESTMAKEUP_I_SPID = 2;
	public static final int FLEXREQUESTMAKEUP_D_STARTTIME = 3;
	public static final int FLEXREQUESTMAKEUP_D_ENDTIME = 4;
	public static final int FLEXREQUESTMAKEUP_I_EXTBEFOREACTIVITYID = 5;
	public static final int FLEXREQUESTMAKEUP_N_EXTBEFOREGAP = 6;
	public static final int FLEXREQUESTMAKEUP_N_EXTBEFOREDURATION = 7;
	public static final int FLEXREQUESTMAKEUP_I_EXTAFTERACTIVITYID = 8;
	public static final int FLEXREQUESTMAKEUP_N_EXTAFTERGAP = 9;
	public static final int FLEXREQUESTMAKEUP_N_EXTAFTERDURATION = 10;
	public static final int FLEXREQUESTMAKEUP_D_SHIFTSTARTTIME = 11;
	public static final int FLEXREQUESTMAKEUP_D_SHIFTENDTIME = 12;
	public static final int FLEXREQUESTMAKEUP_I_SHIFTASSIGNMENTID = 13;
	public static final int FLEX_NUM_OF_FIELDS = 14;

	private static FieldInfoEntry[] m_arrFieldInfo = new FieldInfoEntry[FLEX_NUM_OF_FIELDS];
	private static ChildInfoEntry[] m_childInfo = { ChildInfoEntry.oneToManyChild() };
	private static String m_strTableName = "FLEXREQUESTMAKEUP";

	// Flex Makeup requests have makeup gaps as children.
	public static final int FLEX_GAP_CHILD_TYPE = 0;

	private static final long serialVersionUID = 1L;

	static {
		RequestUtil.initFieldInfoEntries(m_arrFieldInfo, FlexRequestMakeupFieldInfo.class, m_strTableName + "_",
				FLEX_NUM_OF_FIELDS);
	}

	public static String getColumnName(int columnNo) {
		if (columnNo >= m_arrFieldInfo.length) {
			return "";
		}
		return m_arrFieldInfo[columnNo].getFieldName();
	}

	// implement abstract methods from parent class

	public String getTableName() {
		return m_strTableName;
	}

	public int getIDFieldIndex() {
		return FLEXREQUESTMAKEUP_I_ID;
	}

	public FieldInfoEntry[] getFieldInfo() {
		return m_arrFieldInfo;
	}

	protected int getNumFields() {
		return FLEX_NUM_OF_FIELDS;
	}

	public int getParentIDFieldIndex() {
		return FLEXREQUESTMAKEUP_I_TIMEOFFREQUESTID;
	}

	// Return the child info array that contains necessary information about
	// the time off choice children entries and any entries from our superclass.
	public ChildInfoEntry[] getChildInfo() {
		return m_childInfo;
	}
}
