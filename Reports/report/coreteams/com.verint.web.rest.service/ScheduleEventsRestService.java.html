/**

package com.verint.web.rest.service;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.List;

import javax.servlet.ServletContext;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.ws.rs.GET;
import javax.ws.rs.Path;
import javax.ws.rs.PathParam;
import javax.ws.rs.Produces;
import javax.ws.rs.core.Context;
import javax.ws.rs.core.Response;
import javax.xml.bind.annotation.XmlRootElement;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.web.fs.Log;
import com.verint.web.rest.entity.ScheduleEventsEntity;
import com.verint.web.rest.entity.ribbon.RibbonResponseEntity;
import com.verint.web.rest.exception.checked.BusinessWFOException;
import com.verint.web.rest.exception.checked.SystemWFOException;
import com.verint.web.rest.exception.checked.WFOException;
import com.verint.web.rest.facade.IScheduleEventsFacade;
import com.verint.web.rest.security.WFOSecurity;
import com.verint.web.rest.util.FormatUtils;
import com.verint.web.rest.util.FsRestUtil;
import com.witness.web.uif.system.RequestContext;

@Path("/schedule/")
@Produces("application/json")
@WFOSecurity
public class ScheduleEventsRestService extends AbstractWFOEntityService<ScheduleEventsEntity> {
  private static final Category logger = Log.initCategory(ScheduleEventsRestService.class.getName());

  @Context
  private HttpServletRequest request;
  @Context
  private HttpServletResponse response;
  @Context
  private ServletContext sc;

  public IScheduleEventsFacade schedEventsFacade;

  public ScheduleEventsRestService() {
  }

  public IScheduleEventsFacade getScheduleEventsFacade() {
    return schedEventsFacade;
  }

  public RequestContext getRequestContext() {
    return new RequestContext(request, response, sc);
  }

  public void setScheduleEventsFacade(IScheduleEventsFacade schedEventsFacade) {
    this.schedEventsFacade = schedEventsFacade;
  }

  @GET
  @Path("/get")
  @Override
  public ScheduleEventsList<ScheduleEventsEntity> getAllEntities() throws WFOException {
    try {
      Date startTime = null;
      Date endTime = null;
      String strEmpId = request.getParameter("employeeId");
      ID empId = strEmpId == null ? null : new ID(strEmpId);
      String strOrgId = request.getParameter("organizationId");
      ID orgId = strOrgId == null ? null : new ID(strOrgId);
      startTime = FormatUtils.iso8601SecPrecisionStringToDate(request.getParameter("startTime"));
      endTime = FormatUtils.iso8601SecPrecisionStringToDate(request.getParameter("endTime"));

      if (empId == null && orgId == null) {
        throw new BusinessWFOException("employeeId and organizationId both null", Response.Status.CONFLICT);
      } else if (empId != null) {
        orgId = null;
      }

      RequestContext context = getRequestContext();

      return (new ScheduleEventsList<ScheduleEventsEntity>(
          schedEventsFacade.getAllScheduleEvents(context, empId, orgId, startTime, endTime)
      ));

    } catch (WFOException ex) {
      throw ex; // should already have been logged
    } catch (Exception ex) {
      logger.error(ex.getLocalizedMessage(), ex);
      throw new SystemWFOException(ex.getLocalizedMessage());
    }
  }

  @Override
  public ScheduleEventsEntity getEntity(@PathParam("id") String id) throws WFOException {
    return new ScheduleEventsEntity();
  }

  @Override
  public ScheduleEventsEntity updateEntity(ScheduleEventsEntity employee) throws WFOException {
    return new ScheduleEventsEntity();
  }

  @Override
  public ScheduleEventsEntity addEntity(ScheduleEventsEntity employee) throws WFOException {
    return new ScheduleEventsEntity();
  }

  @Override
  public ScheduleEventsEntity deleteEntity(@PathParam("id") String id) throws WFOException {
    return new ScheduleEventsEntity();
  }

  @Override
  public boolean addEntities(Collection<ScheduleEventsEntity> entities) throws WFOException {
    return true;
  }

  @Override
  public boolean updateEntities(Collection<ScheduleEventsEntity> entities) throws WFOException {
    return true;
  }

  @Override
  public boolean deleteEntities(Collection<ScheduleEventsEntity> entities) throws WFOException {
    return true;
  }

  @XmlRootElement(name = "scheduleEvents")
  private class ScheduleEventsList<ScheduleEventsEntity> extends ArrayList<ScheduleEventsEntity> {
    public ScheduleEventsList(List<ScheduleEventsEntity> wrapped) {
      super(wrapped);
    }
  }

	@GET
	@Path("/getRibbon")
	public RibbonResponseEntity getRibbon() throws WFOException {
		try {
			RequestContext context = getRequestContext();
			ID employeeID = context.getUser().getEmployeeID();

			Date startTime = FsRestUtil.getDate(context, "startTime");
			Date endTime = FsRestUtil.getDate(context, "endTime");

			TimeRange timeRange = new TimeRange(startTime, endTime);

			return schedEventsFacade.getRibbon(context, employeeID, timeRange);

		} catch (WFOException ex) {
			// should already have been logged
			throw ex;
		} catch (Exception ex) {
			logger.error(ex.getLocalizedMessage(), ex);
			throw new SystemWFOException(ex.getLocalizedMessage());
		}
	}
}
