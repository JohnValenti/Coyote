public class RequestFacadeImpl implements IRequestFacade {
	private static final Category LOG = Log.initCategory(RequestFacadeImpl.class.getName());

	private static final String[] VALID_REQUEST_TYPES = {
			Request.REQUESTTYPE_ALL,
			Request.REQUESTTYPE_CUSTSHIFT,
			Request.REQUESTTYPE_SHIFTBID,
			Request.REQUESTTYPE_SHIFTSWAP,
			Request.REQUESTTYPE_TIMEOFF
	};

	public RequestFacadeImpl() { // NOSONAR
	}

	/**
	 * Gets the employee's requests.
	 *
	 * @param context
	 * @param employeeID
	 * @param type
	 *            The type of requests to return. Defaults to "all". See VALID_REQUEST_TYPES for possible values.
	 * @param includeExpired
	 *            True to include expired requests.
	 */
	@Override
	public RequestListEntity getRequestsForEmp(RequestContext context, ID employeeID, String type, boolean includeExpired,
			boolean includeWaitlistRank)
			throws WFOException {
		try {
			RequestListEntity result = new RequestListEntity();
			result.setId(employeeID.toString());
			List<RequestEntity> results = result.getRequests();
			if (StringUtil.isEmptyOrWhiteSpace(type)) {
				type = Request.REQUESTTYPE_ALL; // NOSONAR
			} else {
				validateRequestType(type);
			}
			OrganizationSetting orgSettings = RequestsMH.getOrgSettingForEmployee(context, employeeID);
			Collection<RequestAggregate> requests = RequestsMH.getRequestsForEmpFull(employeeID, type, includeExpired);
			Map<ID, EmployeeName> empNamesByID = getEmployeeNamesById(context, requests);
			RequestsMetaData metaData = new RequestsMetaData(orgSettings);
			Map<ID, String> activityIDNameMap = new HashMap<ID, String>();
			Map<ID, ShiftBidAuction> auctionsByID = getShiftBidAuctions(requests);
			Map<ID, Boolean> auctionsOpenByID = getShiftBidAuctionOpenMap(auctionsByID.values());
			for (RequestAggregate ra : requests) {
				RequestEntity reqEntity = null;
				if (ra instanceof TORequest) {
					TORequest tor = (TORequest) ra;
					reqEntity = TimeOffRequestFacadeImpl.createTORequestEntity(tor, activityIDNameMap, context, false,
							includeWaitlistRank);
				} else if (ra instanceof ShiftBidRequest) {
					ShiftBidRequest shiftBidReq = (ShiftBidRequest) ra;
					boolean usesScoring = getUsesScoring(shiftBidReq.getShiftBidAuctionID(), auctionsByID);
					reqEntity = ShiftBiddingFacadeImpl.createShiftBidRequestEntity(context, shiftBidReq, usesScoring, metaData);
				} else if (ra instanceof ShiftSwapRequest) {
					ShiftSwapRequest swapReq = (ShiftSwapRequest) ra;
					reqEntity = ShiftSwapFacadeImpl.createShiftSwapRequestEntity(context, swapReq, empNamesByID);
				} else if (ra instanceof CustShiftReq) {
					CustShiftReq custShiftReq = (CustShiftReq) ra;
					reqEntity = CustShiftFacadeImpl.createCustShiftRequestEntity(context, custShiftReq, metaData);
				}

				if (reqEntity != null) {
					results.add(reqEntity);
					addActions(context, ra, reqEntity, orgSettings, auctionsOpenByID);
				}
			}
			RmUtils.addActivities(context, result.getActivities(), metaData.getActivityIDs());
			RmUtils.addAuctions(result.getAuctions(), auctionsByID.values());
			RmUtils.addShifts(result.getShifts(), metaData.getShiftIDs());
			RmUtils.addShiftOTExtensions(result.getShiftOTExtensions(), metaData.getShiftOTExtensionIDs());
			return result;
		} catch (Exception ex) {
			throw RmUtils.handleFacadeException(context, ex, LOG);
		}
	}

	public static void addActions(RequestContext context, RequestAggregate request, RequestEntity entity, OrganizationSetting orgSettings,
			Map<ID, Boolean> auctionsOpenByID) {
		MyRequestsPMEx pm = new MyRequestsPMEx(context);
		pm.setupForRest(orgSettings, auctionsOpenByID);
		pm.createItemActions(request, "");

		List<Entity> actions = entity.getActions();
		Map<String, ActionCode> lookup = ActionCode.getPageActionToActionCode();
		for (String action : pm.actions) {
			if (lookup.containsKey(action)) {
				ActionCode code = lookup.get(action);
				Entity actionEntity = new Entity();
				actionEntity.setId(Integer.toString(code.getActionCode()));
				actionEntity.setName(action);
				actions.add(actionEntity);
			}
		}
	}

	public static Map<ID, EmployeeName> getEmployeeNamesById(RequestContext context, Collection<RequestAggregate> requests)
			throws BbmException, RemoteException {
		Set<ID> employeeIDs = new HashSet<ID>();
		for (RequestAggregate request : requests) {
			employeeIDs.add(request.getEmployeeID());
			if (request instanceof ShiftSwapRequest) {
				ShiftSwapRequest swapReq = (ShiftSwapRequest) request;
				for (ShiftSwapItem item : swapReq.getShiftSwapItems()) {
					employeeIDs.add(item.getEmployeeID());
				}
			}
		}
		return EmployeeModelHandler.getEmployeeNamesByIDs(context, employeeIDs);
	}

	@SuppressWarnings("unchecked")
	private Map<ID, ShiftBidAuction> getShiftBidAuctions(Collection<RequestAggregate> requests)
			throws RmHardValidationException, Exception {
		Set<ID> auctionIDs = new HashSet<ID>();
		for (RequestAggregate ra : requests) {
			if (ra instanceof ShiftBidRequest) {
				auctionIDs.add(((ShiftBidRequest) ra).getShiftBidAuctionID());
			}
		}
		Collection<ShiftBidAuction> auctions = AuctionMH.getAuctionsByIDs(auctionIDs);
		Map<ID, ShiftBidAuction> auctionsByID = new HashMap<ID, ShiftBidAuction>();
		for (ShiftBidAuction auction : auctions) {
			auctionsByID.put(auction.getID(), auction);
		}
		return auctionsByID;
	}

	private Map<ID, Boolean> getShiftBidAuctionOpenMap(Collection<ShiftBidAuction> auctions) {
		Map<ID, Boolean> map = new HashMap<ID, Boolean>();
		for (ShiftBidAuction auction : auctions) {
			ID id = auction.getID();
			if (!map.containsKey(id)) {
				map.put(id, auction.getIsAuctionOpen());
			}
		}
		return map;
	}

	private boolean getUsesScoring(ID auctionID, Map<ID, ShiftBidAuction> auctionsByID) {
		if (auctionsByID.containsKey(auctionID)) {
			return auctionsByID.get(auctionID).getUsesScoring();
		}
		return false;
	}

	private void validateRequestType(String type) throws BusinessWFOException {
		for (String validType : VALID_REQUEST_TYPES) {
			if (validType.equals(type)) {
				return;
			}
		}
		throw new BusinessWFOException("Invalid request type.", Response.Status.BAD_REQUEST);
	}
}
