package com.verint.wfm.ws.entityImport.schedulingPeriod;

import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignHOO;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;

public class ConvertSchedulingPeriodDTOMapParameters {
	private final Map<String, ID> organizationIDsByNameMap;
	private final Map<ID, Campaign> wfoIDCampaignMap;
	private final Map<ID, ID> queueExternalWfoIDMap;
	private final Map<ID, Queue> wfoIDQueueMap;
	private final Map<SchedulingPeriodServiceDelegate.SchedPeriodNaturalKey, SchedulingPeriod> naturalKeyOldSchedulingPeriodMap;
	private final Map<SchedulingPeriodServiceDelegate.SchedPeriodHOONaturalKey, CampaignHOO> naturalKeyOldSchedulingPeriodHooMap;
	private final Map<SchedulingPeriodServiceDelegate.SPQueueNaturalKey, SPQueue> naturalKeyOldSPQueueMap;

	public ConvertSchedulingPeriodDTOMapParameters(Map<String, ID> organizationIDsByNameMap,
			Map<ID, Campaign> wfoIDCampaignMap,
			Map<ID, ID> queueExternalWfoIDMap,
			Map<ID, Queue> wfoIDQueueMap,
			Map<SchedulingPeriodServiceDelegate.SchedPeriodNaturalKey, SchedulingPeriod> naturalKeyOldSchedulingPeriodMap,
			Map<SchedulingPeriodServiceDelegate.SchedPeriodHOONaturalKey, CampaignHOO> naturalKeyOldSchedulingPeriodHooMap,
			Map<SchedulingPeriodServiceDelegate.SPQueueNaturalKey, SPQueue> naturalKeyOldSPQueueMap) {
		this.organizationIDsByNameMap = organizationIDsByNameMap;
		this.wfoIDCampaignMap = wfoIDCampaignMap;
		this.queueExternalWfoIDMap = queueExternalWfoIDMap;
		this.wfoIDQueueMap = wfoIDQueueMap;
		this.naturalKeyOldSchedulingPeriodMap = naturalKeyOldSchedulingPeriodMap;
		this.naturalKeyOldSchedulingPeriodHooMap = naturalKeyOldSchedulingPeriodHooMap;
		this.naturalKeyOldSPQueueMap = naturalKeyOldSPQueueMap;
	}

	public Map<String, ID> getOrganizationIDsByNameMap() {
		return organizationIDsByNameMap;
	}

	public Map<ID, Campaign> getWfoIDCampaignMap() {
		return wfoIDCampaignMap;
	}

	public Map<ID, ID> getQueueExternalWfoIDMap() {
		return queueExternalWfoIDMap;
	}

	public Map<ID, Queue> getWfoIDQueueMap() {
		return wfoIDQueueMap;
	}

	public Map<SchedulingPeriodServiceDelegate.SchedPeriodNaturalKey, SchedulingPeriod> getNaturalKeyOldSchedulingPeriodMap() {
		return naturalKeyOldSchedulingPeriodMap;
	}

	public Map<SchedulingPeriodServiceDelegate.SchedPeriodHOONaturalKey, CampaignHOO> getNaturalKeyOldSchedulingPeriodHooMap() {
		return naturalKeyOldSchedulingPeriodHooMap;
	}

	public Map<SchedulingPeriodServiceDelegate.SPQueueNaturalKey, SPQueue> getNaturalKeyOldSPQueueMap() {
		return naturalKeyOldSPQueueMap;
	}
}
