	public Pair[] getTOWaitlistPriortyOrderSettings() {
		Pair pairArray[] = new Pair[RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_ORDER_COLUMN_MAX_COUNT];
		int offset=TOPoolFieldInfo.TOPOOL_WLPRIORITYORDER1;    //not very clear or might not make sense here, but this code was copied from somewhere else
		String sortCol0 = null;
		String sortCol1 = null;
		for (int i = 0; i < RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_ORDER_COLUMN_MAX_COUNT; i++) {
			Object value =  getFieldValue(offset + i);
			if (value == null || "".equals(value.toString()) || value.equals(RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_ORDER_COLUMN_NO_VALUE)) {
				break;
			}
			StringTokenizer st = new StringTokenizer(value.toString(), ":");
			Pair pair = new Pair(st.nextToken(), st.nextToken());
			try {
				if (pair != null && pair.getFirst() != null) {
					if (sortCol0 != null && pair.getFirst().equals(sortCol0)) {
						break;
					}
					if (sortCol1 != null && pair.getFirst().equals(sortCol1)) {
						break;
					}
					if (i == 0) sortCol0 = pair.getFirst().toString();
					if (i == 1) sortCol1 = pair.getFirst().toString();
				}
			} catch (Exception e) {
				e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
			}
			pairArray[i] = pair;
		}
		return pairArray;
	}
    public void setTOWaitlistPriortyOrderSettings(Pair[] pairArray) {
        int offset=TOPoolFieldInfo.TOPOOL_WLPRIORITYORDER1;    //not very clear or might not make sense here, but this code was copied from somewhere else
	    for (int i = 0; i < RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_ORDER_COLUMN_MAX_COUNT; i++) {
           setFieldValue (offset+ i,RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_ORDER_COLUMN_NO_VALUE);
        }
        for (int i = 0; pairArray != null && i < pairArray.length; i++) {
            Pair valuePair = pairArray[i];
            setFieldValue (offset+ i, valuePair.getFirst() + ":" + valuePair.getSecond());
        }
    }
