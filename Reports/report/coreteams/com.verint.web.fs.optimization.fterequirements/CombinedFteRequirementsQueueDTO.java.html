package com.verint.web.fs.optimization.fterequirements;

import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.timeseries.model.RequireTraceCube;
import com.verint.web.fs.optimization.CombinedQueueDTO;

import java.io.Serializable;

/**
 * This class contains a MultiFteRequirementsQueueDTO, which itself contains
 * a list of SingleFteRequirementsQueueDTOs.  This class also contains a
 * trace cube representing the aggregated FTE Requirements
 * data for these constituent queues.
 */
public class CombinedFteRequirementsQueueDTO extends
		CombinedQueueDTO<MultiFteRequirementsQueueDTO> implements Serializable {

	private static final long serialVersionUID = 5868453847688989262L;

	// This trace cube is used for non-skill based SPs only. Otherwise, the aggregation from the encapsulated
	// multi queue is used.
	private RequireTraceCube aggregatedFteRequirements;

	/**
	 * Creates a new instance of CombinedFteRequirementsQueueDTO. Use this constructor when it is skill based.
	 *
	 * @param spQueue       the sp queue
	 * @param multiQueueDto the multi queue dto
	 */
	public CombinedFteRequirementsQueueDTO(SPQueue spQueue, MultiFteRequirementsQueueDTO multiQueueDto) {
		super(spQueue, multiQueueDto);
	}

	/**
	 * Creates a new instance of CombinedFteRequirementsQueueDTO. Use this constructor when it is non-skill based.
	 *
	 * @param spQueue       the sp queue
	 * @param serviceGoals  the service goals
	 * @param forecast      the forecast
	 * @param multiQueueDto the multi queue dto
	 */
	public CombinedFteRequirementsQueueDTO(SPQueue spQueue, RequireTraceCube aggregatedFteRequirements,
			MultiFteRequirementsQueueDTO multiQueueDto) {
		super(spQueue, multiQueueDto);
		this.aggregatedFteRequirements = aggregatedFteRequirements;
	}

	/**
	 * Gets the aggregated FTE requirements for the combined queue
	 */
	public RequireTraceCube getFteRequirements() {
		if (aggregatedFteRequirements != null) {
			return aggregatedFteRequirements;
		}
		return m_multiQueueDto.getFteRequirements();
	}
}
