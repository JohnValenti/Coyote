package com.verint.web.fs.optimization.fterequirements;

import com.bluepumpkin.ejb.bbm.timeseries.model.RequireTraceCube;
import com.verint.web.fs.optimization.MultiQueueDTO;

import java.io.Serializable;
import java.util.List;

/**
 * This class represents a list of SingleFteRequirementsQueueDTOs,
 * along with a trace cube representing the aggregated FTE Requirements
 * data for these constituent queues.
 */
public class MultiFteRequirementsQueueDTO extends MultiQueueDTO<SingleFteRequirementsQueueDTO> implements Serializable {

	private static final long serialVersionUID = 6778055944288046308L;

	private RequireTraceCube aggregatedFteRequirements;

	/**
	 * Creates a new instance of MultiFteRequirementsQueueDTO
	 *
	 * @param queueDtos                 the single queue dtos
	 * @param aggregatedFteRequirements the aggregated FTE requirements
	 */
	public MultiFteRequirementsQueueDTO(List<SingleFteRequirementsQueueDTO> queueDtos, RequireTraceCube aggregatedFteRequirements) {
		super(queueDtos);
		this.aggregatedFteRequirements = aggregatedFteRequirements;
	}

	/**
	 * Creates a new instance of MultiFteRequirementsQueueDTO
	 *
	 * @param queueDtos the single queue dtos
	 */
	public MultiFteRequirementsQueueDTO(List<SingleFteRequirementsQueueDTO> queueDtos) {
		super(queueDtos);
	}

	/**
	 * Gets the aggregated fte requirements for the constituent queues.
	 */
	public RequireTraceCube getFteRequirements() {
		return aggregatedFteRequirements;
	}
}
