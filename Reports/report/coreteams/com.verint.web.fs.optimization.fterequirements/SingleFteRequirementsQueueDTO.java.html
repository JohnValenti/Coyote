package com.verint.web.fs.optimization.fterequirements;

import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.timeseries.model.RequireTraceCube;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.verint.web.fs.optimization.SingleQueueDTO;


/**
 * This class represents a basic composite queue/sp queue pair
 * (along with the FteRequirements trace data associated
 * with that spqueue) that will be sent to clients from the server.
 */
public class SingleFteRequirementsQueueDTO extends SingleQueueDTO {

	private static final long serialVersionUID = 2218479440747401426L;
	private RequireTraceCube fteRequirements;

	public SingleFteRequirementsQueueDTO(Queue queue, SPQueue spQueue) {
		super(queue, spQueue);
	}

	public SingleFteRequirementsQueueDTO(Queue queue, SPQueue spQueue, 
			RequireTraceCube fteRequirements) {
		super(queue, spQueue);
		this.fteRequirements = fteRequirements;
	}

	public RequireTraceCube getfteRequirements() {
		return fteRequirements;
	}
}
