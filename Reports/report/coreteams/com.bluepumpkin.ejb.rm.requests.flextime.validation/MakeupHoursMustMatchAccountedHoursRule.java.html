package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.List;

import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexReqMakeupGap;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexRequestMakeup;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.util.TORequestUtil;

/**
 * Validate that the total makeup hours for a flex time request matches the
 * accounted hours for the time off portion of the request.
 *
 */
public class MakeupHoursMustMatchAccountedHoursRule implements Validator{

	private static final String m_className = MakeupHoursMustMatchAccountedHoursRule.class.getName();

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		TORequest toRequest = (TORequest)validatable;

		int accountedMinutes = getAccountedMinutes(toRequest);
		int makeupMinutes = getMakeupMinutes(toRequest);

		if (accountedMinutes != makeupMinutes) {
			//Does violate the rule, use the setHardValidationResult to return the error message
			return ValidationUtil.setHardValidationResult(toRequest,
											RmEjbBundleKey.FLEX_REQ_MAKEUP_HOURS_DO_NOT_MATCH_ACCOUNTED_HOURS,
											m_className);
		}

		//The TORequest passes the rule, no violation
		return null;
	}

	protected int getAccountedMinutes(TORequest toRequest) throws Exception {
		return TORequestUtil.getAccountedMinutes(toRequest);
	}

	/**
	 * Get the total makeup hours for the request.
	 * @param toRequest - the flex time request
	 */
	protected int getMakeupMinutes(TORequest toRequest) {
		int hours = 0;

		if (toRequest != null) {
			List<FlexRequestMakeup> flexRequestMakeupList = toRequest.getFlexRequestMakeupList();

			if (flexRequestMakeupList == null) {
				return hours;
			}

			for (FlexRequestMakeup flex : flexRequestMakeupList) {
				hours += flex.getExtBeforeDuration() + flex.getExtAfterDuration();
				// Adding New Check to see if there are any makeup's within Shift Gaps - if yes, add those durations to the accounted hours.
				if (flex.getFlexReqMakeupGaps() != null) {
					for (FlexReqMakeupGap flexMakeupGap : flex.getFlexReqMakeupGaps()) {
						hours +=  flexMakeupGap.getExtGapDuration();
					}
				}
			}

		}
		return hours;
	}

}
