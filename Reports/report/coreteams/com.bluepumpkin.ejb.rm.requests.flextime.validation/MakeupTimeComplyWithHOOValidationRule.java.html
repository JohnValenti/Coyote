/**
 * This is a soft validation rule to validate all makeups of a Flex Time Request
 * must be inside the Hours Of Operation
 */
public class MakeupTimeComplyWithHOOValidationRule implements Validator {
	public static final String MAKEUP_TIME_COMPLY_HOO = "MAKEUP_TIME_COMPLY_HOO";

	public static final ID RULE_ID = new ID(-194069);
	private static final String CLASS_NAME = MakeupTimeComplyWithHOOValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		LOG.debug(RmUtil.dumpEnterMethod("validate", validatable));

		TORequest tor = (TORequest) validatable;

		ValidationResult result;

		if (!validate(tor)) {
			result = ValidationUtil.setSoftValidationResult(tor, RmEjbBundleKey.FLEX_REQ_MAKE_UP_MUST_BE_WITHIN_HOO, CLASS_NAME);
		} else {
			result = null;
		}

		LOG.debug(RmUtil.dumpExitMethod("validate", result));
		return result;
	}

	private boolean validate(TORequest toReq) throws Exception {

		boolean valid = true;

		List<TimeRange> makeupTimeRanges = getMakeupDurations(toReq);

		for (TimeRange makeupTimeRange : makeupTimeRanges) {

			Date makeupStartRange = makeupTimeRange.getStartDate();
			Date makeupEndRange = makeupTimeRange.getEndDate();

			// Find organization for shift duration
			Organization org = getOrganization(toReq, makeupStartRange, makeupEndRange);

			// Find organization HOO for for shift duration
			HOOPeriod orgHOOs = getHOOPeriod(org.getID(), makeupStartRange, makeupEndRange);


			if (orgHOOs == null) {
				valid = false;
			} else {
				Collection<TimePeriod> pClosedTimes = orgHOOs.getClosePeriods(makeupStartRange, makeupEndRange);
				for (TimePeriod closedTime: pClosedTimes) {
					if (makeupTimeRange.getOverlapDuration(new TimeRange(closedTime.getStartTime(), closedTime.getEndTime())) > 0) {
						valid = false;
						break;
					}
				}

			}

			if (!valid) {
				break;
			}
		}

		return valid;
	}

	protected List<TimeRange> getMakeupDurations(TORequest toReq) throws Exception {
		return this.getFTValidationCache(toReq).getMakeupDurations();
	}


	protected HOOPeriod getHOOPeriod(ID orgId, Date start, Date end) throws Exception {
		HOOPeriod result;
		WorkResourceManager wrm = getWorkResourceManager();

		@SuppressWarnings("unchecked")
		Map<ID, HOOPeriod> mapOrgHOOs = wrm.getHOOPeriod(Collections.singleton(orgId), start, end);
		if (mapOrgHOOs != null && mapOrgHOOs.containsKey(orgId) && mapOrgHOOs.get(orgId) != null) {
			result = mapOrgHOOs.get(orgId);
		} else {
			result = null;
		}
		return result;
	}


	protected Organization getOrganization(TORequest toReq, Date start, Date end) throws Exception {
		WorkResourceManager wrm = getWorkResourceManager();
		@SuppressWarnings("unchecked")
		Map<ID, List<WorkResourceAssignment>> empOrgAssignments = wrm.getValidWorkResourceAssignments(
				Collections.singleton(toReq.getEmployeeID()), new LocalDate(start, TimeZone.getTimeZone("GMT")), new LocalDate(end, TimeZone.getTimeZone("GMT")), false);

		ID orgId = Organization.ROOT_ORG_ID_OBJ;
		if (empOrgAssignments.containsKey(toReq.getEmployeeID())) {
			List<WorkResourceAssignment> assgns = empOrgAssignments.get(toReq.getEmployeeID());
			if (assgns != null && !assgns.isEmpty()) {
				orgId = assgns.get(0).getOrganizationID();
			}
		}

		return getFTValidationCache(toReq).getOrganizationByID(orgId);
	}

	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}

	protected CustShiftValidationCache getCustShiftValidationCache(CustShiftReq custShiftReq) {
		return new CustShiftValidationCache(custShiftReq);
	}

	protected WorkResourceManager getWorkResourceManager() throws BbmEJBCreateException {
		return BbmManagerFactory.getWorkResourceManager(false);
	}

	protected ScheduleAccessManager getScheduleAccessManager() throws BbmEJBCreateException {
		return WfmManagerFactory.getScheduleAccessManager();
	}

}
