package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;

public class MinimumFlexRequestDurationValidationRule extends FlexRequestDurationValidationRule {
	public static final String MIN_FLEX_REQUEST_DURATION_VALUE = "MIN_FLEX_REQUEST_DURATION_VALUE";

	public static final String MIN_FLEX_REQUEST_DURATION_UNIT = "MIN_FLEX_REQUEST_DURATION_UNIT";

	public static final ID RULE_ID = new ID(-194059);

	@Override
	protected void init(TORequest toRequest) throws Exception {
		//only one TOC per flex request
		this.toChoice = (TOChoice)toRequest.getRequestChoiceList().iterator().next();
		this.orgID = getEmployeeOrgId(toRequest.getEmployeeID(), toChoice);

		@SuppressWarnings("rawtypes")
		Map ruleParams = getValidationRuleManager().getValidationRuleParams(orgID, RULE_ID);
		this.duration = ((Integer) ruleParams.get(MIN_FLEX_REQUEST_DURATION_VALUE)).intValue();
		this.unit = ((Integer) ruleParams.get(MIN_FLEX_REQUEST_DURATION_UNIT)).intValue();
	}

	@Override
	protected TimeRange getOverLimitRange(long durationLimit, long accountedSeconds) {
		return (accountedSeconds < durationLimit) ? new TimeRange(this.toChoice.getStartDate(), this.toChoice.getEndDate()) : null;
	}

	@Override
	protected String getClassName() {
		return MinimumFlexRequestDurationValidationRule.class.getName();
	}

	@Override
	protected ValidationResult setSoftValidationResult(TimeRange overLimitRange) {
		return ValidationUtil.setSoftValidationResult(this.toChoice,
				RmEjbBundleKey.FLEX_REQ_CHOICE_MIN_DURATION, overLimitRange.getStartDate(), overLimitRange.getEndDate(), getClassName());
	}
}
