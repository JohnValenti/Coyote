package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.io.Serializable;
import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.rm.Log;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.FilingRuleChecker;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResultAggregator;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.FlexRequestMakeup;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOChoice;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.bluepumpkin.ejb.rm.setup.filingrules.model.RequestFilingRule;
import com.bluepumpkin.ejb.rm.util.RmUtil;

public class FlexTimeComplyFilingRulesValidationRule implements Validator {
	public static final String FLEX_TIME_COMPLY_FILING_RULES = "FLEX_TIME_COMPLY_FILING_RULES";

	public static final ID RULE_ID = new ID(-194065);

	private static final String CLASS_NAME = FlexTimeComplyFilingRulesValidationRule.class.getName();
	private static final Category LOG = Log.initCategory(CLASS_NAME);
	private static final String METHOD_NAME = "validate";

	@SuppressWarnings({ "rawtypes" })
	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {
		LOG.debug(RmUtil.dumpEnterMethod(METHOD_NAME, validatable));
		ValidationResultAggregator result = new ValidationResultAggregator();
		TORequest tor = (TORequest) validatable;

		// Build a filing rule validation checker for this request
		FilingRuleChecker ckr = tor.getCache().getFilingRuleChecker(tor);
		Date earliestMakeUp = getEarliestMakeUpStart(tor);

		// Loop through the time off choices and check each one against the filing rules
		for (TOChoice toChoice : tor.getTOChoicesAdjustedForValidation()) {
			ckr.resetLastFailingRule();
			result.add(setValidationResultForFilingRuleCompliance(tor, toChoice, ckr, earliestMakeUp));
		}

		LOG.debug(RmUtil.dumpExitMethod(METHOD_NAME, result));
		return result.get();
	}

	private static Date getEarliestMakeUpStart(TORequest tor) {
		Date earliestStart = null;
		for (FlexRequestMakeup flexMakeup : tor.getFlexRequestMakeupList()) {
			Date makeupStart = flexMakeup.getStartTime();
			if (earliestStart == null || makeupStart.before(earliestStart)) {
				earliestStart = makeupStart;
			}
		}
		return earliestStart;
	}

	/**
	 * Validate one time off choice for this Flex Time request.
	 */
	public ValidationResult setValidationResultForFilingRuleCompliance(TORequest tor, TOChoice choice, FilingRuleChecker ckr,
			Date earliestMakeUp)
			throws Exception {
		ValidationResult result = null;
		Date fileDate = tor.getSubmittedOn();
		Date dateToValidate = (earliestMakeUp == null || choice.getStartDate().before(earliestMakeUp)) ? choice.getStartDate()
				: earliestMakeUp;

		if (!ckr.requestComplies(fileDate, choice.getStartDate(), choice.getEndDate(), dateToValidate)) {
			// log the rule that failed the request
			RequestFilingRule rule = ckr.getLastFailingRule();
			LOG.debug("flex time request id:" + tor.getID() + " failed because " + "choice id:" + choice.getID()
					+ " did not pass filing rule with id:" + rule.getID());

			Serializable[] params = new Serializable[] {tor.getCache().getEmployeeNameByID(),
					choice.getStartDate(), choice.getEndDate(), tor.getCache().getOrgNameForMsg(),
					tor.getCache().getTimeZoneForOrg(), rule};

			result = ValidationUtil.setSoftValidationResult(choice, RmEjbBundleKey.FT_FAILED_FILING_RULE, params, CLASS_NAME);
		}
		return result;
	}
}
