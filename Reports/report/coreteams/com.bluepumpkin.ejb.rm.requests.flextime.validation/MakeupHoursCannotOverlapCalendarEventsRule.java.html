package com.bluepumpkin.ejb.rm.requests.flextime.validation;

import java.util.Collection;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.activity.model.Event;
import com.bluepumpkin.ejb.bbm.schedule.ejb.ScheduleAccessManager;
import com.bluepumpkin.ejb.bbm.schedule.model.PlannedEvent;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TORequest;
import com.verint.ejb.wfm.WfmManagerFactory;

/**
 * Validate that the makeup time for a flex time request doesn't overlap
 * calendar events (including time off events, unavailability events, and normal calendar events) for the employee.
 *
 */
public class MakeupHoursCannotOverlapCalendarEventsRule implements Validator {

	private static final String CLASS_NAME = MakeupHoursCannotOverlapCalendarEventsRule.class.getName();

	@Override
	public ValidationResult validate(Validatable validatable) throws Exception {

		ValidationResult result = null;

		TORequest toRequest = (TORequest) validatable;

		// Find makeup durations
		List<TimeRange> makeupDurations = getFTValidationCache(toRequest).getMakeupDurations();
		Date firstDate = makeupDurations.get(0).getStartDate();
		Date lastDate = makeupDurations.get(makeupDurations.size() - 1).getEndDate();

		// Find calendar events
		Collection<PlannedEvent> calEvents = getCalEventsDuringPeriod(toRequest.getEmployeeID(), firstDate, lastDate);

		// Check for conflicts
		for (TimeRange makeupDuration : makeupDurations) {
			for (PlannedEvent calEvent : calEvents) {
				if (intersect(makeupDuration.getStartDate(), makeupDuration.getEndDate(), calEvent.getStartTime(), calEvent.getEndTime())) {
					return ValidationUtil.setHardValidationResult(toRequest, RmEjbBundleKey.FLEX_REQ_MAKEUP_TIME_OVERLAP_CALENDAR_EVENT,
							getFTValidationCache(toRequest).getEmployeeNameByID(), makeupDuration.getStartDate(), makeupDuration.getEndDate(),  CLASS_NAME);
				}
			}
		}

		return result;
	}

	private static boolean intersect(Date start1, Date end1, Date start2, Date end2) {
		return end2.after(start1) && start2.before(end1);
	}

	@SuppressWarnings("unchecked")
	protected Collection<PlannedEvent> getCalEventsDuringPeriod(ID empID, Date start, Date end) throws Exception {
		return getScheduleAccessManager().getPublishedEventsForWorkResourceByType(Event.EVENT_TYPE_ALL_CALENDAR_EVENTS, empID, start, end);
	}

	private ScheduleAccessManager getScheduleAccessManager() throws BbmEJBCreateException {
		return WfmManagerFactory.getScheduleAccessManager();
	}
	protected FTValidationCache getFTValidationCache(TORequest toRequest) {
		return (FTValidationCache) toRequest.getCache();
	}
}
