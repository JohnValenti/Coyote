package com.bluepumpkin.web.bbm.workload.queue.selection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.workload.ejb.WorkloadManager;
import com.bluepumpkin.ejb.bbm.workload.model.Media;
import com.bluepumpkin.web.bbm.workload.WorkloadModelHandler;
import com.bluepumpkin.web.bbm.workload.queue.selection.queueattribute.SPQueueAttributeFilterBoxPC;
import com.verint.ejb.wfm.WfmManagerFactory;
import com.witness.web.uif.pagecomponent.tree.util.TreeModelBuilder;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.pagemodel.SelectionpaneTreePM;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;
import com.witness.web.uif.util.SortUtil;

/**
 * Accepts Request Parameters:
 * <ul>
 * <li>ignorePageDirty</li>
 * </ul>
 */
@SuppressWarnings("serial")
public class SPQueueSelectionPM extends SelectionpaneTreePM {

	private static final String QUEUE_CAMPAIGN_SP_FORM_ACTION = "queue_campaign_sp_selection";
	private static final String PARAM_EXCLUDE_MEDIA = "excludeMedia";
	private static final String PARAM_IS_DISTRIBUTED_CAMPAIGN_ENABLED = "isDistributedCampaignEnabled";
	private static final String PARAM_IS_DISTRIBUTED_CAMPAIGN_ONLY = "isDistributedCampaignOnly";
	private static final String PARAM_SELECTION_BEHAVIOR_TYPE = "selectionBehaviorType";
	private static final String LAST_CAMPAIGN_ID_FN = "lastCampaignID";
	private static final String LAST_SCHEDULING_PERIOD_ID_FN = "lastSchedulingPeriodID";
	private static final String IS_INFO_SUBMITTED_FROM_SELF_FN = "isInfoSubmittedFromSelf";

	protected SPQueueAttributeFilterBoxPC spQueueAttributeFilterBoxPC;
	private ID campaignID = null;
	private ID schedulingPeriodID = null;
	private String excludeMediaRawParam = null;
	private boolean isDistributedCampaignEnabled = true;
	private String selectionBehaviorType = null;
	private boolean isDistributedCampaignOnly = false;

	public SPQueueSelectionPM(RequestContext context) {
		super(context, new SPQueueSelectionPC(context));

		((SPQueueSelectionPC) m_treePC).setSelectionBehaviorType(context.getParameter(PARAM_SELECTION_BEHAVIOR_TYPE));

		spQueueAttributeFilterBoxPC = new SPQueueAttributeFilterBoxPC(m_context, "spQueueFilterBox");
		addChildComponent(spQueueAttributeFilterBoxPC);
		m_filterBoxPC = spQueueAttributeFilterBoxPC;
		m_filterBoxPC.setUseWrapper(true);
	}

	/**
	 * Called via reflection from the request
	 *
	 * @param ignorePageDirty ignorePageDirty
	 */
	public void setIgnorePageDirty(boolean ignorePageDirty) {
		((SPQueueSelectionPC) m_treePC).setIgnorePageDirty(ignorePageDirty);
	}

	private Map<ID, Media> filterMediaByExcludedMediaTypes(Map<ID, Media> media) {
		excludeMediaRawParam = m_context.getParameter(PARAM_EXCLUDE_MEDIA);
		if (excludeMediaRawParam == null) {
			return media;
		}

		String[] mediaTypes = excludeMediaRawParam.split(",");
		List<ID> lookup = new ArrayList<>(mediaTypes.length);
		for (String mediaType : mediaTypes) {
			lookup.add(new ID(mediaType.trim()));
		}

		HashMap<ID, Media> filteredMedia = new HashMap<>();
		for (ID key : media.keySet()) {
			if (!lookup.contains(key)) {
				filteredMedia.put(key, media.get(key));
			}
		}

		return filteredMedia;
	}

	protected void preLoadData() throws Exception {
		//Needs to be synchronized because the work frame (Applet) and the queue  
		//selection frame (this class) both load at the same time.
		synchronized (m_context.getRequest().getSession()) {
			// Retrieve the defined media types.
			Map<ID, Media> mediaMap = WorkloadModelHandler.getMediaMap(m_context);
			mediaMap = filterMediaByExcludedMediaTypes(mediaMap);

			isDistributedCampaignEnabled = Boolean.parseBoolean(m_context.getParameter(PARAM_IS_DISTRIBUTED_CAMPAIGN_ENABLED));

			Collection<Media> mediaList = SortUtil.sortBasicVO(mediaMap.values(), m_localizer);
			spQueueAttributeFilterBoxPC.init(mediaList, isDistributedCampaignEnabled, isDistributedCampaignOnly);

			// Get the filter values.
			campaignID = spQueueAttributeFilterBoxPC.getCampaignID();
			ID mediaID = spQueueAttributeFilterBoxPC.getMediaID();
			schedulingPeriodID = spQueueAttributeFilterBoxPC.getSchedulingPeriodID();

			ID queueAttribID = spQueueAttributeFilterBoxPC.getQueueAttributeID();

			if (mediaID != null) {
				Media media = mediaMap.get(mediaID);
				mediaMap = new HashMap<>();
				mediaMap.put(mediaID, media);
			}

			// Configure the selection page component based on the filter.
			SPQueueSelectionPC spQueuePC = (SPQueueSelectionPC) m_treePC;
			spQueuePC.setMultiRowSelectable(false);
			spQueuePC.setIsComplexIDs(true);

			WorkloadManager wlm = WfmManagerFactory.getWorkloadManager(m_context.isInWhatIfMode());
			Collection<ID> rootQueues = java.util.Collections.emptyList();
			if (schedulingPeriodID != null) {
				rootQueues = wlm.getRootLinkQueueIDs(schedulingPeriodID);
			}

			spQueuePC.bind(campaignID, schedulingPeriodID, queueAttribID, mediaMap, rootQueues);
		}
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) {
		return getInstance(context, SPQueueSelectionPM.class);
	}

	public String getRequiredHTML() {
		return super.getRequiredHTML() + HtmlUtil.makeHiddenInput(LAST_CAMPAIGN_ID_FN, campaignID) + HtmlUtil.makeHiddenInput
			(LAST_SCHEDULING_PERIOD_ID_FN, schedulingPeriodID) + HtmlUtil.makeHiddenInput(IS_INFO_SUBMITTED_FROM_SELF_FN, true) + HtmlUtil
			.makeHiddenInput(PARAM_IS_DISTRIBUTED_CAMPAIGN_ONLY, isDistributedCampaignOnly);
	}

	/**
	 * Initialize Toolbar, override for selection because multi
	 * select restricted to groups doesn't have a good
	 * implementation for select all.
	 */
	protected void initToolbar() {
		useDefaultToolbar(false, true);
	}

	/**
	 * Called via reflection from request
	 *
	 * @param val val
	 */
	public void setIsDistributedCampaignOnly(boolean val) {
		isDistributedCampaignOnly = val;
	}

	public void setIsDistributedCampaignEnabled(boolean bValue) {
		isDistributedCampaignEnabled = bValue;
	}

	/**
	 * Called via reflection from the request
	 *
	 * @param selectionBehaviorType selectionBehaviorType
	 */
	public void setSelectionBehaviorType(String selectionBehaviorType) {
		((SPQueueSelectionPC) m_treePC).setSelectionBehaviorType(selectionBehaviorType);
	}

	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return QUEUE_CAMPAIGN_SP_FORM_ACTION + "?" + PARAM_IS_DISTRIBUTED_CAMPAIGN_ENABLED + "=" + isDistributedCampaignEnabled + 
			(excludeMediaRawParam == null ? "" : "&" + PARAM_EXCLUDE_MEDIA + "=" + excludeMediaRawParam) + (selectionBehaviorType == null 
			? "" : "&" + PARAM_SELECTION_BEHAVIOR_TYPE + "=" + selectionBehaviorType);
	}

	@Override
	protected String[] getPreSelectedRows() {
		Collection<String> idList = StringUtil.parseStringToCollection(m_selectedID, StringUtil.DELIM);
		return getSelectedRowIndex(idList);
	}

	protected String[] getSelectedRowIndex(Collection<String> idList) {
		return TreeModelBuilder.getSelectedRowIndex(m_treePC, new HashSet<>(idList));
	}
}
