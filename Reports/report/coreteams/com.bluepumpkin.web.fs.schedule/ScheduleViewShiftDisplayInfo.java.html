package com.bluepumpkin.web.fs.schedule;

import com.bluepumpkin.common.util.StringUtil;

/**
 * A helper class which contains the information needed to display
 * shifts in the various schedule view screens.
 */
public class ScheduleViewShiftDisplayInfo {
	
	private final String m_previousDayIcon;
	private final String m_shiftString;
	private final String m_nextDayIcon;
	private final String m_poolingActivityIcon;
	private final String m_overtimeIcon;
	private final String m_calendarEventIcon;

	public ScheduleViewShiftDisplayInfo(String previousDayIcon,
			String shiftString,
			String nextDayIcon,
			String poolingActivityIcon,
			String overtimeIcon,
			String calendarEventIcon) {
		if (StringUtil.isEmpty(previousDayIcon)) {
			m_previousDayIcon = "";
		} else {
			m_previousDayIcon = previousDayIcon;
		}
		
		if (StringUtil.isEmpty(shiftString)) {
			m_shiftString = "";
		} else {
			m_shiftString = shiftString;
		}
		
		if (StringUtil.isEmpty(nextDayIcon)) {
			m_nextDayIcon = "";
		} else {
			m_nextDayIcon = nextDayIcon;
		}
		
		if (StringUtil.isEmpty(poolingActivityIcon)) {
			m_poolingActivityIcon = "";
		} else {
			m_poolingActivityIcon = poolingActivityIcon;
		}
		
		if (StringUtil.isEmpty(overtimeIcon)) {
			m_overtimeIcon = "";
		} else {
			m_overtimeIcon = overtimeIcon;
		}
		
		if (StringUtil.isEmpty(calendarEventIcon)) {
			m_calendarEventIcon = "";
		} else {
			m_calendarEventIcon = calendarEventIcon;
		}
	}
	
	public ScheduleViewShiftDisplayInfo(String previousDayIcon,
			String shiftString,
			String nextDayIcon) {

		if (StringUtil.isEmpty(previousDayIcon)) {
			m_previousDayIcon = "";
		} else {
			m_previousDayIcon = previousDayIcon;
		}
		
		if (StringUtil.isEmpty(shiftString)) {
			m_shiftString = "";
		} else {
			m_shiftString = shiftString;
		}
		
		if (StringUtil.isEmpty(nextDayIcon)) {
			m_nextDayIcon = "";
		} else {
			m_nextDayIcon = nextDayIcon;
		}
		m_poolingActivityIcon = "";
		m_overtimeIcon = "";
		m_calendarEventIcon = "";
	}
	
	public String getPreviousDayIcon() {
		return m_previousDayIcon;
	}
	public String getShiftString() {
		return m_shiftString;
	}
	public String getNextDayIcon() {
		return m_nextDayIcon;
	}
	public String getPoolingActivityIcon() {
		return m_poolingActivityIcon;
	}
	public String getOvertimeIcon() {
		return m_overtimeIcon;
	}
	public String getCalendarEventIcon() {
		return m_calendarEventIcon;
	}
}
