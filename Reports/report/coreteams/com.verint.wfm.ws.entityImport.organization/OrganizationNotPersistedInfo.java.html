package com.verint.wfm.ws.entityImport.organization;

import java.util.List;

import com.verint.wfm.ws.entityImport.EntityNotPersistedInfo;

/**
 * For each organization not persisted by createOrUpdateOrganizations(), one of these will be in the list returned This
 * is returned in lieu of an exception, because we always want to continue on to all the organizations
 *
 * @see EntityImportService.createOrUpdateOrganizations()
 * 
 */
public class OrganizationNotPersistedInfo extends EntityNotPersistedInfo<String> {
	private static final long serialVersionUID = 1L;

	private boolean m_isValidButNotCreated;
	private String organizationName;

	public OrganizationNotPersistedInfo() {
		super();
	}

	public OrganizationNotPersistedInfo(String externalId, String organizationName) {
		super(externalId);
		this.organizationName = organizationName;
	}

	public String getOrganizationName() {
		return organizationName;
	}

	// -----------------------------------------
	public String getOrganizationIdentifier() {
		return super.getEntityExternalSourceId();
	}

	/**
	 * Dummy here on server, but needed for SOAP stub
	 */
	public void setOrganizationIdentifier(String organizationIdentifier) {
		// do nothing
	}

	public void addOrganizationServerExceptionMessage(String exception) {
		getMessages().add(String.format(
				"Exception is %s. The server could not persist organization %s.", exception, getOrganizationName()));
	}

	public void addNameMissingMessage(String organizationDescription) {
		getMessages().add(String.format("Name was missing for organization with description %s.", organizationDescription));
	}

	public void addNameLengthExceededMessage(int actualLength, int lengthLimit) {
		getMessages().add(
				String.format("For organization with name %s the name length %d exceeded the limit of %d.",
				getOrganizationName(), actualLength, lengthLimit));
	}
	
	public void addNameHasInvalidCharactersMessage(String invalidChars) {
		getMessages().add(String.format(
				"The characters %s are not allowed in the organization name. The organization %s has invalid characters in the name.",
				invalidChars, getOrganizationName()));
	}
	
	public void addNameSameAsRoot() {
		getMessages().add(String.format(
				"For organization name %s should not be same as the Root Organization name in WFM",
				getOrganizationName()));
	}

	public void addParentNameSameAsRoot() {
		getMessages().add(String.format(
				"For organization with name %s, the parent organization name should not be same as the Root Organization name in WFM",
				getOrganizationName()));
	}

	public String getDuplicateNameMessage() {
		return String.format(
				"Multiple organizations with the same name [%s] can not be imported.",
				getOrganizationName());
	}

	public void addWeekStartMessage(int weekStart) {
		getMessages().add(String.format(
				"For organization with name %s, week start %d is not between 1 and 7.", getOrganizationName(), weekStart));
	}

	public void addWeekDayOpenLengthMessage(int openingTimeCount) {
		getMessages().add(String.format(
				"For organization with name %s, there were only %d opening times, but one for each day of the week is required (plus one dummy).",
				getOrganizationName(), openingTimeCount));
	}

	public void addWeekDayCloseLengthMessage(int closingTimeCount) {
		getMessages().add(String.format(
				"For organization with name %s, there were only %d opening times, but one for each day of the week is required (plus one dummy).",
				getOrganizationName(), closingTimeCount));
	}

	public String getNoSuchParentOrganizationNameMessage(String parentOrganizationName) {
		return String.format(
				"For organization with name %s, parent organization specified to have name %s does not exist.",
				getOrganizationName(), parentOrganizationName);
	}

	public String getCircularOrganizationHierarchyMessage(String currentOrganizationName, String parentOrganizationName) {
		return String.format(
				"A circular dependency was found within the organization hierarchy verifying [%s], detected during the link between [%s]"
				+ "[%s]. No organizations will be modified.",
				getOrganizationName(),
				currentOrganizationName,
				parentOrganizationName);
	}

	public void addOrganizatonAlreadyExistsMessage() {
		getMessages().add(String.format("Organization with name %s already exists.",getOrganizationName()));
	}

	public String getRenameOrganizationToExistingMessage() {
		return String.format(
				"Failed to rename organization with the external id [%s], another existing organization has the name [%s].",
				getEntityExternalSourceId(),
				getOrganizationName());
	}

	public boolean isValidButNotCreated() {
		return m_isValidButNotCreated;
	}

	public void setValidButNotCreated(boolean isValidButNotCreated) {
		if (!this.getMessages().isEmpty()) {
			throw new RuntimeException("Not valid entity; application logic error");
		}
		m_isValidButNotCreated = isValidButNotCreated;
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public List<String> getMessages() {
		return super.getMessages();
	}

	/**
	 * Axis bug work around
	 */
	@Override
	public void setMessages(List<String> messages) {
		// Do nothing
	}
}
