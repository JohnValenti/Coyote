package com.bluepumpkin.ejb.rm.requests.swap.posting.validation;

import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.rm.l10n.RmEjbBundleKey;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validatable;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationResult;
import com.bluepumpkin.ejb.rm.requests.common.validation.ValidationUtil;
import com.bluepumpkin.ejb.rm.requests.common.validation.Validator;
import com.bluepumpkin.ejb.rm.requests.swap.posting.ejb.ShiftSwapPostingDAO;
import com.bluepumpkin.ejb.rm.requests.swap.posting.model.ShiftSwapPosting;

/**
 * @author Raja
 *
 * Hard validation rule to check if the specified posting is a duplicate of an
 * already posted posting.
 * 
 */
public class SameShiftNotAlreadyPostedHV implements Validator {
    private static String m_className = SameShiftNotAlreadyPostedHV.class.getName();
    
    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable)
        throws Exception {

        ShiftSwapPostingDAO sspDAO = null;        
        try {
            ShiftSwapPosting ssPosting = (ShiftSwapPosting) validatable;
            
            sspDAO = new ShiftSwapPostingDAO(ShiftSwapPosting.DL_BASIC);
            
            Date startDate = ssPosting.getStartDate();
            Date endDate = ssPosting.getEndDate();
            ID employeeID = ssPosting.getEmployeeID();
            
            // get list of posting IDs for the specified empID, start and end time.
            //for partial shift swap, check for any overlap
            List idList = sspDAO.findOverlappingSSPostingIDsByStartEndDate(employeeID, 
                startDate, endDate);
            
            // if one or more postings were found
            if ( !idList.isEmpty() ) {
                // return a hard validation error.
                return ValidationUtil.setHardValidationResult(ssPosting, 
                    RmEjbBundleKey.SSP_DUPLICATE_POSTING, startDate, endDate, m_className);
            }
            
            return null;            
        } finally {
            if (sspDAO != null) sspDAO.cleanUp();
        }
    }

}
