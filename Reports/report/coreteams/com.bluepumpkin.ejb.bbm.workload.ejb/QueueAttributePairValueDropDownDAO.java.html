package com.bluepumpkin.ejb.bbm.workload.ejb;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.ejb.bbm.base.BbmRemoveException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.workload.model.queueattribute.QueueAttributePairDropDownValue;
import com.bluepumpkin.ejb.bbm.workload.model.queueattribute.QueueAttributePairValueDropDownFieldInfo;

import java.util.Collection;

public class QueueAttributePairValueDropDownDAO extends DAOBase {

	private static FieldInfo m_fieldInfo = new QueueAttributePairValueDropDownFieldInfo();

	protected FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	public QueueAttributePairValueDropDownDAO() {
		super();
	}

	public QueueAttributePairValueDropDownDAO(Jdmo dmo) {
		super(dmo);
	}

	protected ValueObjectBase createValueObject() {
		return new QueueAttributePairDropDownValue();
	}

	/**
	 * delete objects with parent ids
	 * NOTE: if the parent table has both ID and SID columns,
	 * you must pass in the ID value
	 */
	public void deleteObjectsWithParentIDsAndNonSpecifiedAttributes(Collection<ID> parentIDs, Collection<ID> 
		attributeIDsToBeRetained)
		throws BbmRemoveException {
		try {
			StringBuilder statement = new StringBuilder(100);
			statement.append("SELECT ");
			statement.append(m_fieldInfo.getIDFieldName());
			statement.append(" FROM ");
			statement.append(m_fieldInfo.getTableName());
			statement.append(" WHERE ");
			statement.append(m_fieldInfo.getParentIDFieldName());
			statement.append(" IN ");
			statement.append(getDMO().createInClause(parentIDs));
			if (attributeIDsToBeRetained != null && !attributeIDsToBeRetained.isEmpty()) {
				statement.append(" AND QUEUEATTRIBUTEID NOT IN ");
				statement.append(getDMO().createInClause(attributeIDsToBeRetained));
			}

			deleteObjects(statement.toString());
		} catch (JdmoException ex) {
			throw new BbmRemoveException(ex);
		}
	}
}
