package com.bluepumpkin.ejb.bbm.workload.ejb;

import java.util.ArrayList;
import java.util.Collection;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.ejb.bbm.base.BbmFinderException;
import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.workload.model.queueattribute.QueueAttributeDefintion;
import com.bluepumpkin.ejb.bbm.workload.model.queueattribute.QueueAttributeFieldInfo;
import com.verint.ejb.wfm.IQueueAttributeDefinitionDAO;

public class QueueAttributeDefinitionDAO extends DAOBase implements IQueueAttributeDefinitionDAO {

		private static FieldInfo m_fieldInfo = new QueueAttributeFieldInfo();
		protected FieldInfo getFieldInfo() { return m_fieldInfo; }

		public QueueAttributeDefinitionDAO() {
            super();
		}

		public QueueAttributeDefinitionDAO(Jdmo dmo) {
            super(dmo);
		}

		protected ValueObjectBase createValueObject()
		{
            return (new QueueAttributeDefintion());
		}
		
		public Collection<QueueAttributeDefintion> getQueueAttributeDefinition(Collection<ID> orgIDs, ID moduleID) throws BbmFinderException {
			Jdmo jdmo = new Jdmo();
			try {

				JdmoQuery jq = jdmo.createQuery("SELECT Q.ID, Q.NAME, Q.DESCRIPTION, Q.ORGANIZATIONID, Q.QUEUEATTRIBUTETYPEID, O.NAME AS ORGANIZATIONNAME FROM QUEUEATTRIBUTEDEFINITION Q INNER JOIN ORGANIZATION O ON Q.ORGANIZATIONID = O.ID WHERE Q.MODULEID = " + moduleID + " AND Q.ORGANIZATIONID IN " + jdmo.createInClause(orgIDs), Jdmo.PARAM_QUERY);
				JdmoRowset rs = jdmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
				Collection<QueueAttributeDefintion> attributeDefinition = new ArrayList<>();
				ArrayList<ID> attribIDs = new ArrayList<>();
				QueueAttributeDefintion definition;
				while (rs.next()) {
				    definition = new QueueAttributeDefintion();
				    definition.setID(rs.getID("ID"));
				    attribIDs.add(rs.getID("ID"));
				    definition.setName(rs.getString("NAME"));
				    definition.setDescription(rs.getString("DESCRIPTION"));
				    definition.setOrganizationId(rs.getID("ORGANIZATIONID"));
				    definition.setType(rs.getID("QUEUEATTRIBUTETYPEID"));
				    definition.setOrganizationName(rs.getString("ORGANIZATIONNAME"));
				    attributeDefinition.add(definition);
				}
				
				if(!attribIDs.isEmpty() && !attributeDefinition.isEmpty()) {
					Collection<ID> attribs = getQueueAttributesLinkedToFilter(attribIDs);
					for (QueueAttributeDefintion def : attributeDefinition) {
						if (attribs.contains(def.getID())) {
							def.setIsFiltered(true);
						}
					}
				}
				return attributeDefinition;
			} catch(JdmoException e) {
				throw new BbmFinderException(e);
			} finally {
				jdmo.cleanUp();
			}
		}
		
		private Collection<ID> getQueueAttributesLinkedToFilter(Collection<ID> queueAttributeIDs) throws BbmFinderException {
			Jdmo jdmo = new Jdmo();
			try {
				String inClause = jdmo.createInClause(queueAttributeIDs);
				String pStmt = "SELECT DISTINCT QUEUEATTRIBUTEID FROM QUEUEATTRIBFILTERSELDATE WHERE QUEUEATTRIBUTEID IN " + inClause +
					" UNION ALL " +
					"SELECT DISTINCT QUEUEATTRIBUTEID FROM QUEUEATTRIBFILTERSELDD WHERE QUEUEATTRIBUTEID IN " +
					inClause +
					" UNION ALL " +
					"SELECT DISTINCT QUEUEATTRIBUTEID FROM QUEUEATTRIBFILTERSELNUMERIC WHERE QUEUEATTRIBUTEID IN " +
					inClause +
					" UNION ALL " +
					"SELECT DISTINCT QUEUEATTRIBUTEID FROM QUEUEATTRIBFILTERSELTEXT WHERE QUEUEATTRIBUTEID IN " +
					inClause;

				JdmoQuery jq = jdmo.createQuery(pStmt, Jdmo.PARAM_QUERY);
				JdmoRowset rs = jdmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);
				ArrayList<ID> attribIDs = new ArrayList<>();
				while (rs.next()) {
				    attribIDs.add(rs.getID("QUEUEATTRIBUTEID"));
				}
				return attribIDs;
			} catch(JdmoException e) {
				throw new BbmFinderException(e);
			} finally {
				jdmo.cleanUp();
			}
		}

		public Collection<String> getQueueAttributeDefinitionNames() throws BbmFinderException {
			ArrayList<String> filterNames = new ArrayList<>();
			StringBuilder query = new StringBuilder("");
			query.append("SELECT NAME FROM QUEUEATTRIBUTEDEFINITION");
			Jdmo jdmo = new Jdmo();
			try {
				JdmoRowset r = jdmo.createRowset(query.toString());
				while (r.next()) {
					filterNames.add(r.getString("NAME"));
				}
			} catch (JdmoException e) {
				e.printStackTrace();
				throw new BbmFinderException(e);
			} finally {
				jdmo.cleanUp();
			}
			
			return filterNames;
		}
}
