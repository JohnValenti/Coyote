package com.bluepumpkin.ejb.bbm.empworkrule.ejb;

import java.util.Date;
import java.util.HashMap;
import java.util.Map;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.ejb.bbm.activity.ejb.ActivityDAO;
import com.bluepumpkin.ejb.bbm.activity.model.ActivityFieldInfo;
import com.bluepumpkin.ejb.bbm.dao.JunctionTableRelationship;
import com.bluepumpkin.ejb.bbm.dao.OneToManyOneWayRelationship;
import com.bluepumpkin.ejb.bbm.dao.RichObjectDAO;
import com.bluepumpkin.ejb.bbm.empworkrule.model.TimePeriodOfDayFieldInfo;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnit;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnitActivityFieldInfo;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnitFieldInfo;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnitShiftPattern;
import com.bluepumpkin.ejb.bbm.empworkrule.model.WorkUnitShiftPatternFieldInfo;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.bbm.workrules.ejb.ShiftPatternDAO;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEventFieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftFieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPatternFieldInfo;

public class WorkUnitDAO extends RichObjectDAO<WorkUnit> {

	private static FieldInfo m_fieldInfo = new WorkUnitFieldInfo();

	public WorkUnitDAO(Jdmo dmo) {
		super(dmo);
	}

	public FieldInfo getFieldInfo() {
		return m_fieldInfo;
	}

	@Override
	protected WorkUnit createValueObject() {
		return new WorkUnit();
	}
	
	@Override
	protected Map<Integer, ? extends FieldInfo> getForeignKeyFieldInfo() {
		Map<Integer, FieldInfo> retVal = new HashMap<Integer, FieldInfo>();
		retVal.put(WorkUnitFieldInfo.WORKUNIT_SHIFTID, new ShiftFieldInfo());
		retVal.put(WorkUnitFieldInfo.WORKUNIT_SHIFTEVENTID, new ShiftEventFieldInfo());
		// Commented out because this constant doesn't exist yet.
		//retVal.put(WorkUnitFieldInfo.WORKUNIT_ACTIVITYID, new ActivityFieldInfo());
		return retVal;
	}

	@Override
	protected void buildDBRelationships() {
		addRelationship(new JunctionTableRelationship(
				this,
				new ShiftPatternDAO(getDMO()),
				new WorkUnitShiftPatternDAO(getDMO()),
				WorkUnitFieldInfo.WORKUNIT_ID,
				WorkUnitShiftPatternFieldInfo.WORKUNITSHIFTPATTERN_WORKUNITID,
				WorkUnitShiftPatternFieldInfo.WORKUNITSHIFTPATTERN_SHIFTPATTERNID,
				ShiftPatternFieldInfo.SHIFTPATTERN_ID,
				WorkUnit.RELATIONSHIPKEY_SHIFTPATTERN));
		addRelationship(new OneToManyOneWayRelationship(
				this,
				new TimePeriodOfDayDAO(getDMO()),
				WorkUnitFieldInfo.WORKUNIT_TIMEOFDAYID,
				TimePeriodOfDayFieldInfo.Column.ID.getIndex(),
				WorkUnit.RELATIONSHIPKEY_TIMEOFDAY));
		addRelationship(new JunctionTableRelationship(
				this, //parentDAO 
				new ActivityDAO(getDMO()), //childDAO 
				new WorkUnitActivityDAO(getDMO()),  //junctionDAO
				WorkUnitFieldInfo.WORKUNIT_ID, //parentFieldIndex 
				WorkUnitActivityFieldInfo.Column.WorkUnitID.getIndex(), //junctionParentFieldIndex
				WorkUnitActivityFieldInfo.Column.ActivityID.getIndex(), //junctionChildFieldIndex
				ActivityFieldInfo.ACTIVITY_ID, //childFieldIndex
				WorkUnit.RELATIONSHIPKEY_ACTIVITY //relationshipKey
				));
		
	}

	@Override
	public void setJunctionFieldsOnCreate(
		String relationshipKey,
		WorkUnit workUnit,
		ValueObjectBase junctionObject,
		ValueObjectBase relatedObject,
		int index)
	{
		if (WorkUnit.RELATIONSHIPKEY_SHIFTPATTERN.equals(relationshipKey)) {
			WorkUnitShiftPattern wusp = (WorkUnitShiftPattern) junctionObject;
			wusp.setWeek(index);
			wusp.setCreatedBy(getCallerPrincipal());
			wusp.setCreatedDate(new Date());
			wusp.setModifiedBy(null);
			wusp.setModifiedDate(null);
		}

		if (WorkUnit.RELATIONSHIPKEY_ACTIVITY.equals(relationshipKey)) {
			// Do nothing. No extra fields to set
		}
	}

}
