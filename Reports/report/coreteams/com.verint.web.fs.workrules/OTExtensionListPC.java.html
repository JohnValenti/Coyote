	private boolean isShiftPatternScreen = false;

	public OTExtensionListPC(RequestContext context, String name, TimeZone optimizationTargetTimeZone,
			Date timeZoneEffectiveDate, Collection<ShiftOTExtension> otExtensions, 
			Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap,
			Map<ID, Activity> activityMap) {
		super(context);
		setName(name);
		setFirstColumnWidth("2.5%");
		this.orgMap = orgMap;
		this.campaignMap = campaignMap;
		this.activityMap = activityMap;
		this.optimizationTargetTimeZone = optimizationTargetTimeZone;
		this.timeZoneEffectiveDate = timeZoneEffectiveDate;
		
		if (otExtensions != null) {
			for (ShiftOTExtension sote : otExtensions) {
				otExtensionStartTimeMap.put(sote, null);
			}
		}
		
		toggleAll = new ToggleVisiblePC(m_context, "toggleAll");
		this.addChildComponent(toggleAll);
		
		initRows(otExtensionStartTimeMap);
		initHeader();
		
		addRequiredJavaScriptFile(WorkRulesJSFileID.WORK_RULES_UTIL);
		addPopupArgs(ActivityPopupPM.POPUP_ID, ActivityPopupPM.FORM_ACTION, "",
				ActivityPopupPM.POPUP_ARGS, "1000,600,ctr,ctr", true, 1);
		setIsCancelSelectEventPropagation(false);
	}
	
	public OTExtensionListPC(RequestContext context, String name, ShiftPattern shiftPattern, 
			TimeZone optimizationTargetTimezone, Date timeZoneEffectiveDate,
			Map<ID, Organization> orgMap, Map<ID, Campaign> campaignMap,
			Map<ID, Activity> activityMap, boolean isByOrg, boolean isChildTable) {
		super(context);
		setName(name);
		isShiftPatternScreen = true;
		this.isChildTable = isChildTable;
		
		this.orgMap = orgMap;
		this.campaignMap = campaignMap;
		this.activityMap = activityMap;
		this.optimizationTargetTimeZone = optimizationTargetTimezone;
		this.timeZoneEffectiveDate = timeZoneEffectiveDate;
		this.isByOrg = isByOrg;

		otExtensionStartTimeMap = new LinkedHashMap<ShiftOTExtension, ShiftPattern.StartTime>(shiftPattern.getOTExtensionStartTimes());
		// Filter out objects events deleted before view date
		for (Iterator<Map.Entry<ShiftOTExtension, ShiftPattern.StartTime>> iter = otExtensionStartTimeMap.entrySet().iterator(); iter.hasNext(); ) {
			ShiftOTExtension sote = iter.next().getKey();
			if (sote.getDeleteOnDate() != null && sote.getDeleteOnDate().before(timeZoneEffectiveDate)) iter.remove();
		}
		
		initRows(otExtensionStartTimeMap);
		initHeader();
		
		addRequiredJavaScriptFile(WorkRulesJSFileID.WORK_RULES_UTIL);
		addPopupArgs(OTExtensionListPopupPM.POPUP_ID, OTExtensionListPopupPM.FORM_ACTION, "",
				OTExtensionListPopupPM.POPUP_ARGS, "1000,600,ctr,ctr", true, 1);
		addPopupArgs(ActivityPopupPM.POPUP_ID, ActivityPopupPM.FORM_ACTION, "",
				ActivityPopupPM.POPUP_ARGS, "1000,600,ctr,ctr", true, 1);
		setIsCancelSelectEventPropagation(false);
	}
	
	private void initHeader() {	
		TableHeaderPC header = getHeader();
		header.setPartialSortable(true);
		Localizer localizer = m_context.getLocalizer();
		if (isShiftPatternScreen) {
			DefaultHeaderData dhd = new DefaultHeaderData(FsWebBundleKey.FS_OT_EXTENSION,
					getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_OT_EXTENSION));
			dhd.addImageButton(getOTExtensionPopupImage(), getOTExtensionPopupCaption(), getOTExtensionPopupJS());
			dhd.setSortable(true);
			header.addColumnHeaderData(dhd);
		} else {
			header.addColumnHeaderData("toggleAll", toggleAll.getHtmlDisplay(), false);
			header.addColumnHeaderData(FsWebBundleKey.FS_OT_EXTENSION, localizer.i18n(
					FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_OT_EXTENSION), true);
		}
		DefaultHeaderData dhd = new DefaultHeaderData(FsWebBundleKey.FS_ACTIVITY,
				getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ACTIVITY));
		dhd.setSortable(true);
		dhd.addImageButton(getActivityPopupImage(), getActivityPopupCaption(), getActivityPopupJS());

		header.addColumnHeaderData(dhd);
		header.addColumnHeaderData(FsWebBundleKey.FS_LENGTH, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_LENGTH), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_MIN_GAP, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MIN_GAP), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_MAX_GAP, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_MAX_GAP), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_ORGANIZATION, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ORGANIZATION), true);
		header.addColumnHeaderData(FsWebBundleKey.FS_DESCRIPTION, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_DESCRIPTION), true);
	
		if (isShiftPatternScreen) {
			header.addColumnHeaderData(FsWebBundleKey.FS_START_TIMES, localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_START_TIMES));
		}
	}
	
	public HtmlTag getWrapper() {
		if (isChildTable) {
			if (m_wrapper == null) m_wrapper = new HtmlDiv();
			m_wrapper.setStyle("margin:20px");
			return m_wrapper;
		} else {
			return super.getWrapper();
		}
	}

	private void initRows(HashMap<ShiftOTExtension, ShiftPattern.StartTime> otExtensionMap) {
		List<MultiColumnNodeData> rows = super.getListModel();
		
		String optimizationTargetName = "";
		
		for (ShiftOTExtension otExtension : otExtensionMap.keySet()) {
			String detailRowID = getName() + "__detail" + otExtension.getID();
			DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(
					otExtension.getID());
			
			if (otExtension.getOrganizationID() != null) {
				Organization o = orgMap.get(otExtension.getOrganizationID());
				if (o != null) {
					optimizationTargetName = o.getName();
				}
			} else {
				Campaign c = campaignMap.get(otExtension.getCampaignID());
				if (c != null) {
					optimizationTargetName = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.LOCAL);
				}
			}
			
			boolean hasShiftEvents = otExtension.getShiftEvents().size() > 0;
			if (!isShiftPatternScreen) {
				if (! hasShiftEvents) {
					nodeData.add("");
				} else {
					// Toggle
					ToggleVisiblePC toggle = new ToggleVisiblePC(m_context, "toggle" + otExtension.getID(), Collections.singleton(detailRowID));
					toggleAll.addChildComponent(toggle);
					nodeData.add(toggle.getHtmlDisplay());
				}
			}

			//Name
			if (otExtension.getDeleteOnDate() != null) {
				HtmlDel delTag = new HtmlDel(optimizationTargetTimeZone, otExtension.getDeleteOnDate(), otExtension.getName());
				nodeData.add(delTag);
			}
			else nodeData.add(otExtension.getName());
			
			//Activity
			Activity a = activityMap.get(otExtension.getActivityID());
			if (a != null) {
				nodeData.add(a.getName());
			} else {
				nodeData.add("");
			}
			
			//Length
			nodeData.add(getLocalizer().formatDurationInMins(otExtension.getDuration()));
			
			//Min Gap
			nodeData.add(getLocalizer().formatDurationInMins(otExtension.getMinGap()));
			
			//Max Gap
			nodeData.add(getLocalizer().formatDurationInMins(otExtension.getMaxGap()));
			
			//Organization / campaign
			nodeData.add(optimizationTargetName);
			
			//Description
			nodeData.add(otExtension.getDescription());
			
			//Start Times
			if (isShiftPatternScreen) {
				nodeData.add(WorkRulesUtil.getOTExtensionStartTimeText(getLocalizer(),
						otExtensionMap.get(otExtension)));
			}
			
			setRowAttributes(nodeData, otExtension);
			
			if (!hasShiftEvents || isShiftPatternScreen) {
				rows.add(nodeData);
			} else {
				// Shift Events
				DefaultMultiColumnNodeData mcndDetail = new DefaultMultiColumnNodeData(otExtension.getID());
				setDetailRowAttributes(nodeData, mcndDetail, otExtension);				
				ShiftEventListPC mclpc = new ShiftEventListPC(m_context, "shift" + otExtension.getID() + "_events",
						otExtension.getShiftEvents(), optimizationTargetTimeZone, timeZoneEffectiveDate,
						orgMap, campaignMap, activityMap, true);
				mclpc.setIsUserSortable(false);
				mclpc.setName("shift" + otExtension.getID() + "_events");
				mclpc.setIsBorderEnabled(true);
				mclpc.setRowSelectable(false);
				addChildComponent(mclpc);
				HtmlTableRow row = new HtmlTableRow();
				row.add("<td colspan='9'>" + mclpc + "</td>"); // Hard code the colspan for now, since we init selectable items before the header because of the toggles
				mcndDetail.setNodeAttribute("id", detailRowID);
				mcndDetail.setNodeAttribute("masterID", otExtension.getID().toString());
				mcndDetail.setNodeAttribute("style", "display: none;");
				mcndDetail.add(row);
	
				MasterDetailRow mdr = new MasterDetailRow(nodeData, mcndDetail);
				rows.add(mdr);
			}
		}
	}

			rowData.setNodeAttribute(ActivityPopupPM.POPUP_ARG_VO_ID, otExtension.getActivityID().toString());
		}
		rowData.setNodeAttribute(OTExtensionListPopupPM.POPUP_ARG_VO_ID, otExtension.getID().toString());
	}

	private String getOTExtensionPopupJS() {
		String optimizationTargetTimeZoneStr = optimizationTargetTimeZone != null ? optimizationTargetTimeZone.getID() : "";
		long tzDateMillis = timeZoneEffectiveDate != null ? timeZoneEffectiveDate.getTime() : new Date().getTime();
		String popupWindowArgNames = "isByOrg,isSelectionPopup,optimizationTargetTimeZoneID,timeZoneEffectiveDate";
		String popupWindowArgValues = Boolean.toString(isByOrg) + ",false," + optimizationTargetTimeZoneStr + "," +
			tzDateMillis;
		return "WorkRulesUtil.handleInfoPopup(this, " + getName() + "," + getHeader().getName() + ",'" +
			OTExtensionListPopupPM.POPUP_ID + "','" + OTExtensionListPopupPM.POPUP_ARG_VO_ID + "','" +
			OTExtensionListPopupPM.POPUP_ARG_VO_IDS + "','" + popupWindowArgNames + "','" +
			popupWindowArgValues + "');";	
	}

	public LinkedHashMap<DBSortField, Comparator> getSortComparators() {
		LinkedHashMap<DBSortField, Comparator> comparators =
			new LinkedHashMap<DBSortField, Comparator>();
		for (ShiftOTExtensionSortField sortField : getDBSortFieldIndices()) {
			switch (sortField) {
				case ID:
					comparators.put(ShiftOTExtensionSortField.ID, null);
					break;
				case NAME:
					comparators.put(ShiftOTExtensionSortField.NAME, null);
					break;
				case ACTIVITY:
					comparators.put(ShiftOTExtensionSortField.ACTIVITY, null);
					break;
				case LENGTH:
					comparators.put(ShiftOTExtensionSortField.LENGTH, null);
					break;
				case MIN_GAP:
					comparators.put(ShiftOTExtensionSortField.MIN_GAP, null);
					break;
				case MAX_GAP:
					comparators.put(ShiftOTExtensionSortField.MAX_GAP, null);
					break;
				case ORGANIZATION:
					comparators.put(ShiftOTExtensionSortField.ORGANIZATION,
							new OrganizationCampaignNameComparator(getLocalizer()));
					break;
				case DESCRIPTION:
					comparators.put(ShiftOTExtensionSortField.DESCRIPTION, null);
					break;
			}
		}
		
		return comparators;
	}
	
	public ArrayList<ShiftOTExtensionSortField> getDBSortFieldIndices() {
		ArrayList<ShiftOTExtensionSortField> retVal = new ArrayList<ShiftOTExtensionSortField>();
		if (getHeader().getSortColumnIndexes() == null) {
			//Default sort index is by ID (chronological order in which object was created)
			retVal.add(ShiftOTExtensionSortField.ID);
		} else {
			for (int sortField : getHeader().getSortColumnIndexes()) {
				switch (sortField) {
					case 1: 
						retVal.add(ShiftOTExtensionSortField.NAME);
						break;
					case 2: 
						retVal.add(ShiftOTExtensionSortField.ACTIVITY);
						break;
					case 3: 
						retVal.add(ShiftOTExtensionSortField.LENGTH);
						break;
					case 4: 
						retVal.add(ShiftOTExtensionSortField.MIN_GAP);
						break;
					case 5: 
						retVal.add(ShiftOTExtensionSortField.MAX_GAP);
						break;
					case 6: 
						retVal.add(ShiftOTExtensionSortField.ORGANIZATION);
						break;
					case 7: 
						retVal.add(ShiftOTExtensionSortField.DESCRIPTION);
						break;
					default: break;
				}
			}
		}
		return retVal;
	}
	
	/**
	 * Setting this to false will prevent the user from being able to change
	 * the sort order by clicking on the column header.  This behavior may be
	 * desirable if this list is used as an assignment list for another master
	 * list.  If this value is set to false, the list will be sorted by shift event
	 * name.  Default value is true.
	 */
	public void setIsUserSortable(boolean isUserSortable) {
		if (this.getHeader() != null) {
			this.getHeader().setSortable(isUserSortable);
			this.getHeader().setPartialSortable(isUserSortable);
		}
		
		//If user sorting is disabled, we will have a default sort which is by item name.  We are sorting here because
		//the data model (retrieved using getListModel()) has already been initialized in the constructor.
		if (isUserSortable == false) SortUtil.sort(getListModel(), getHeader().getColumnIndexByName(FsWebBundleKey.FS_OT_EXTENSION),
				true, m_context.getLocalizer());
	}
