package com.verint.web.fs.workrules;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.HashSet;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeOfDay;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.time.TimeContextFactory;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.data.DataSet;
import com.witness.web.uif.data.DataTable;
import com.witness.web.uif.data.DataTableColumn;
import com.witness.web.uif.data.DataTableRow;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.TimeZoneOverrideUtil;

public class ShiftEventExportPopupPM extends WorkRulesExportPopupPM {
	private static final long serialVersionUID = 1L;
	
	public static final String VIEW_ACTION = "shift_event_export";
	public static final String FORM_ACTION = "shift_event_export_download";
	
	public ShiftEventExportPopupPM(RequestContext context) {
		super(context);
		
		// Disable comma delimiter because shift event additional activities field comma delimits multiple assignments.
		setIsCommaDelimiterEnabled(false);
	}
	
	public static ShiftEventExportPopupPM getInstance(RequestContext context) {
		return (ShiftEventExportPopupPM)getInstance(context, ShiftEventExportPopupPM.class);
	}

	@Override
	public DataSet getDataSet() throws Exception {
		DataSet ds = new DataSet();
		
		// Only have 1 table to export.
		DataTable table = new DataTable("Shift Events");
		ds.add(table);
		
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_SHIFT_EVENT)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_ACTIVITY)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_LENGTH)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_START_TIME_TYPE)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_START_WINDOW)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_EVENT_PAID)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_FLEXIBLE)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_MIN_COUNT)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_MAX_COUNT)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_ADDITIONAL_ACTIVITIES)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_ORGANIZATION)));
		table.addColumn(new DataTableColumn(i18n(m_fsResourceBundle, FsWebBundleKey.FS_DESCRIPTION)));

		Collection<ShiftEvent> shiftEvents = null;
		if(m_isByOrganization) {
			Map<String, ShiftEvent> shiftEventsByName = WorkRulesModelHandler.getShiftEventsByOrg(m_context, new ID(m_organizationId));
			shiftEvents = new ArrayList<ShiftEvent>();
			for(ShiftEvent se : shiftEventsByName.values()) {
				shiftEvents.add(se);
			}
		} else {
			ID spID = new ID(m_schedulingPeriodId); // not sure this is the way to create an ID.
			shiftEvents = WorkRulesModelHandler.getShiftEventsBySchedulingPeriodCampaignOnly(m_context, spID);
		}
		
		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		Collection<ID> campaignIDs = new HashSet<ID>();
		Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
		Map<ID, Campaign> campaignMap = new HashMap<ID, Campaign>();
		
		if(shiftEvents != null) {
			// Collect the distinct organization and campaign ids so we can populate the maps
			for(ShiftEvent se : shiftEvents) {
				if (se.getOrganizationID() != null) orgIDs.add(se.getOrganizationID());
				if (se.getCampaignID() != null) campaignIDs.add(se.getCampaignID());
			}
			orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
			campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		
			for(ShiftEvent se : shiftEvents) {
				DataTableRow row = new DataTableRow(table, false);
				
				// Shift Name
				row.add(se.getName());
			
				// Activity (Should never be null but if a license is missing where it once existed,
				// activities get stripped out on the return associated to the missing license and
				// can cause nulls).
				row.add((se.getActivity() != null ? se.getActivity().getName() : ""));
				
				// Length (format into 00:00)			
				row.add(m_localizer.formatDurationInMins(se.getDuration().getDurationInMinutes()));
				
				// Start Time Type				
				if (se.isAnyTime()) row.add(i18n(m_fsResourceBundle, FsWebBundleKey.FS_ANY_TIME));
				else if (se.isCafeteria()) row.add(i18n(m_fsResourceBundle, FsWebBundleKey.FS_ABSOLUTE));
				else row.add(i18n(m_fsResourceBundle, FsWebBundleKey.FS_RELATIVE_TO_SHIFT_START));
				
				// Start Window
				if (se.isAnyTime()) row.add("");
				else {
					int timeZoneOffsetInMins = TimeZoneOverrideUtil.getTimeZoneOffset(m_context, m_optimizationTargetTimezone, m_timeZoneEffectiveDate); 
					if (se.isCafeteria()) {
						int dayBoundary = TimeContextFactory.getTimeContext(se, orgMap, campaignMap).getDayBoundary();
						row.add(m_localizer.formatTimeInMins(new TimeOfDay(se.getStart() + dayBoundary).addMinutes(timeZoneOffsetInMins).getMinutesSinceMindnight())
								+ "-" + m_localizer.formatTimeInMins(new TimeOfDay(se.getEnd() + dayBoundary).addMinutes(timeZoneOffsetInMins).getMinutesSinceMindnight()));
					} else {
						row.add(m_localizer.formatDurationInMins(se.getStart())
								+ "-" + m_localizer.formatDurationInMins(se.getEnd()));	
					}
				}
				
				// Event Paid
				row.add((se.isPaid() ? i18n(m_fsResourceBundle, FsWebBundleKey.FS_YES) : i18n(m_fsResourceBundle, FsWebBundleKey.FS_NO)));
				
				// Flexible
				row.add((se.isFlexible() ? i18n(m_fsResourceBundle, FsWebBundleKey.FS_YES) : i18n(m_fsResourceBundle, FsWebBundleKey.FS_NO)));
				
				// Min Count
				row.add(String.valueOf(se.getMinCount()));
				
				// Max Count
				row.add(String.valueOf(se.getMaxCount()));
				
				// Additional Activities
				Collection<Activity> additionalActivities = se.getAdditionalActivities();
				StringBuilder sb = new StringBuilder();
				for(Activity activity : additionalActivities) {
					if(sb.length() > 0) sb.append(", ");
					sb.append(activity.getName());
				}
				row.add(sb.toString());
				
				// Organization
				Organization o = orgMap.get(se.getOrganizationID()); 
				row.add(o == null ? m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.LOCAL) : String.valueOf(o.getName()));
				
				// Description
				row.add(se.getDescription());
				
				table.addRow(row);
			}
			
		}
		
		return ds;
	}
	
	/**
	 * Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return FORM_ACTION;
	}

	@Override
	public String getDefaultFileName() {
		return m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_SHIFT_EVENT_FILE_NAME);
	}
}
