package com.verint.web.fs.workrules;

import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.config.ConfigKey;
import com.verint.web.fs.widgets.JumpToPagePaginationPC;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagecomponent.picker.time.TimeZoneOverrideTogglePC;
import com.witness.web.uif.pagemodel.PopupWindowListPM;
import com.witness.web.uif.system.RequestContext;

import java.util.Date;
import java.util.TimeZone;

public abstract class WorkRulesSelectionPopupPM extends PopupWindowListPM {
	private static final long serialVersionUID = 1L;

	protected static final String FIND_ITEM_FN = "itemToFind";

	protected boolean option_showTimeZoneToggle = true;
	protected boolean isFindMode = false;
	protected TimeZone optimizationTargetTimeZone;
	protected Date timeZoneEffectiveDate = new Date();
	protected String itemToFind;

	private JumpToPagePaginationPC paginationPC;

	public WorkRulesSelectionPopupPM(RequestContext context) {
		super(context);

		m_list = initList();
		//This needs to be done here so that we can get the sort columns/order
		m_list.extractParameters(context.getRequest());

		paginationPC = new JumpToPagePaginationPC(getToolbar(), getLocalizer());
		addChildComponent(paginationPC);

		setIsPaginatingSelectedIDs(false);
		addRequiredJavaScriptFile(WorkRulesJSFileID.WORK_RULES_UTIL);
		addRequiredJavaScriptFile(WorkRulesJSFileID.ROW_INSERTION_UTIL);
		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
	}

	protected void initializeModel() {
		if (m_isInitialized) {
			return;
		} else {
			m_isInitialized = true;
		}
		try {
			loadData();
			initContentTitle();
			initScreenContent();
			//Enable pagination
			enableToolbarPagination(getPaginationControl());
			initToolbar();
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		}
	}

	@Override
	protected void initHeader() {
	}

	@Override
	protected void initToolbar() {
		int pageIDSizeLimit = 60;
		try {
			String pageIDSizeFetched = BbmManagerFactory.getDBConfigManager().getValue(ConfigKey.FS_WEB_ID_SIZE_LIMIT);
			if (pageIDSizeFetched != null) {
				pageIDSizeLimit = Integer.parseInt(pageIDSizeFetched);
			}
		} catch (Exception e) {
			handleUnableToLoadDataErrorAndLogAsDebug(log, e);
		}
		setLimitingPageSize(pageIDSizeLimit);
	}

	@Override
	protected void initSelectableItems() {
		m_list = initList();
		addChildComponent(m_list);
		m_list.getHeader().setPartialSortable(true);
		m_list.extractParameters(m_context.getRequest());
	}

	protected abstract MultiColListPC initList();

	public String getHtmlDisplay() {
		return m_list.getHtmlDisplay();
	}

	protected void initContentTitle() {
		super.initContentTitle();

		// Add read-only time zone info to the header
		if (optimizationTargetTimeZone != null && option_showTimeZoneToggle) {
			m_contentTitlePC.addActiveComponent(
					new TimeZoneOverrideTogglePC(m_context, m_toolbar, optimizationTargetTimeZone, true));
		}
	}

	public void setOptimizationTargetTimeZoneID(String optimizationTargetTimeZoneID) {
		this.optimizationTargetTimeZone = TimeZone.getTimeZone(optimizationTargetTimeZoneID);
	}

	public void setTimeZoneEffectiveDate(long timeZoneEffectiveDate) {
		this.timeZoneEffectiveDate = new Date(timeZoneEffectiveDate);
	}

	public JumpToPagePaginationPC getPaginationControl() {
		return paginationPC;
	}

	public void setFindMode(boolean isFindMode) {
		this.isFindMode = isFindMode;
	}

	public void setItemToFind(String itemToFind) {
		this.itemToFind = itemToFind;
	}

	/* The Find functionality will find all names with the word typed occurs anywhere in the name.
	 * In SQL statement, the syntax will be LIKE '%word%'. However, SQL have some wildcard characters SQL
	 *  So this method is used to replaced some wildcard characters.
	 * [], [^] : Any single character within/not within the specified range ([a-f]) or set ([abcdef]). =>replace the [ by [[]
	 * %:AnyString of zero or more charaters=> replaced by  [%]
	 * _(underscore) : Any single character ==> replaced by [_]
	 *
	 * */
	public String getTextToFind(String textFind) {
		String textReplaced = textFind;
		if (textFind != null) {
			//Must handle the [ in [] first
			int lastOpenIndex = textFind.lastIndexOf("[");
			int firstCloseIndex = textFind.indexOf("]");
			if (lastOpenIndex > -1 && firstCloseIndex > -1 && lastOpenIndex + 1 < firstCloseIndex) {
				String firstPart = textFind.substring(0, lastOpenIndex);
				String replacepart = textFind.substring(lastOpenIndex + 1, firstCloseIndex);
				String lastPart = textFind.substring(firstCloseIndex);
				textFind = firstPart.concat("[[]").concat(replacepart).concat(lastPart);
			}
			//Then handle % and _
			textReplaced = textFind.replaceAll("%", "[%]");
			textReplaced = textReplaced.replaceAll("_", "[_]");
		}
		return textReplaced;
	}
}
