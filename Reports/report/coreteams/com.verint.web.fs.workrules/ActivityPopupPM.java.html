package com.verint.web.fs.workrules;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashSet;
import java.util.Iterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.CampaignOrg;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.web.bbm.activity.ActivityListPC;
import com.bluepumpkin.web.bbm.activity.ActivityModelHandler;
import com.bluepumpkin.web.bbm.activity.ActivityModelHandler.ActivitySelectionProperty;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.bbm.l10n.BbmWebBundleKey;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarTextButton;
import com.witness.web.uif.system.RequestContext;

/**
 * Title:        ActivityPopupPM
 * Description:  Popup page model intended to display a list of activities.
 * Copyright:    Copyright (c) 2009
 * Company:      Verint Systems Inc.
 * @author       Dustin Stanley
 */

public class ActivityPopupPM extends WorkRulesSelectionPopupPM {
	private static final long serialVersionUID = 1L;

	public static final String FORM_ACTION = "activity_list_popup";
	public static final String POPUP_ID = "POPUP_ACTIVITY";
	public static final String POPUP_ARGS = "activityIDs,showAddButton,ownerID,isByOrg,activitySelection,filteredActivityIDs";
	public static final String POPUP_ARG_VO_ID = "activityID";
	public static final String POPUP_ARG_VO_IDS = "activityIDs";
	private HashSet<ID> activityIDs = new HashSet<ID>();
	private HashSet<ID> filteredActivityIDs = new HashSet<ID>();
	private Collection<Activity> activities;
	private boolean showAddButton = false;
	private ID ownerID;
	private boolean isByOrg;
	private ActivitySelectionProperty activitySelection;
	
	public ActivityPopupPM(RequestContext context) {
		super(context);
	}
	
	public static ActivityPopupPM getInstance(RequestContext context) {
		return (ActivityPopupPM)getInstance(context, ActivityPopupPM.class);
	}
	
	protected void initializeModel() {
		if (m_isInitialized) return;
		super.initializeModel();

		if (showAddButton) {
			ToolbarTextButton addButton = m_toolbar.createCloseWindowButton(
					"addActivities", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_ADD_SELECTED_ACTIVITIES), true);
			StringBuffer addButtonJS = new StringBuffer();
			addButtonJS.append(m_toolbar.getName())
			.append(".disableButton('addActivities'); ")
			.append("RowInsertionUtil.addNewDynamicRow('activity', eval('window.opener.activityTable'), ")
			.append("'" + InlineEditComponentRH.URL + "', ")
			.append("'" + InlineEditComponentRH.PARAM_TYPE + "=" +
					InlineEditComponentRH.ACTIVITY_TYPE)
			.append("');");
			
			addButton.setCustomJS(addButtonJS.toString());
			m_toolbar.addToolbarElement(addButton);
			m_toolbar.addCloseWindowTextButton("cancel", m_localizer.i18n(
					CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_CANCEL), true);
		}
	}
	
	protected void initUtilityPane() {
		getUtilityPane().setIsHelpEnabled(false);
	}
	
	protected void initContentTitle() {
		m_contentTitlePC.setPageTitle(getLocalizer().i18n(
			BbmWebBundleKey.BUNDLE_NAME, BbmWebBundleKey.ACTIVITY_LIST_TITLE));
	}
	
	protected void loadData() throws Exception {
		if (activityIDs != null && activityIDs.size() > 0)  {
			activities = ActivityModelHandler.getActivitiesByIDs(getRequestContext(), activityIDs);
		} else if (ownerID != null && activitySelection != null) {
			Collection<ID> orgIDs = new ArrayList<ID>();
			if (isByOrg) {
				orgIDs.add(ownerID);
			} else {
				SchedulingPeriod schedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
						m_context, ownerID);
				Campaign campaign = CampaignModelHandler.getCampaign(m_context,
						schedulingPeriod.getCampaignID());
				CampaignModelHandler cmh = new CampaignModelHandler();
				Collection<CampaignOrg> orgAssigned = cmh.getCampaignOrgAssignments(m_context,
						campaign.getID(), schedulingPeriod.getStartTime(),
						schedulingPeriod.getEndTime());
				for(CampaignOrg co : orgAssigned)
				{
					orgIDs.add(co.getOrganizationID());
				}
			}
			activities = ActivityModelHandler.getActivitiesByActivitySelectionType(getRequestContext(), activitySelection,
					orgIDs);
			if (filteredActivityIDs != null) {
				for (ID filteredID : filteredActivityIDs) {
					for (Iterator<Activity> iter = activities.iterator(); iter.hasNext();) {
						if (iter.next().getID().equals(filteredID)) {
							iter.remove();
							break;
						}
					}
				}
			}
		}
	}

	protected ActivityListPC initList() {
		return new ActivityListPC(getRequestContext(), "activityTable", activities);
	}
	
	public void setActivityIDs(String activityIDStr) {
		activityIDs = new HashSet<ID>(StringUtil.parseIDStringToCollection(activityIDStr));
	}
	
	public String getHtmlDisplay() {
		if (m_list != null) {
			return m_list.getHtmlDisplay();
		}
		return "";
	}
	
	public void setShowAddButton(boolean showAddButton) {
		this.showAddButton = showAddButton;
	}
	
	public void setOwnerID(ID ownerID) {
		this.ownerID = ownerID;
	}
	
	public void setIsByOrg(boolean isByOrg) {
		this.isByOrg = isByOrg;
	}
	
	public void setActivitySelection(String activitySelection) {
		this.activitySelection = ActivitySelectionProperty.valueOf(activitySelection);
	}
	
	public void setFilteredActivityIDs(String filteredIDs) {
		filteredActivityIDs = new HashSet<ID>(StringUtil.parseIDStringToCollection(filteredIDs));
	}
}
