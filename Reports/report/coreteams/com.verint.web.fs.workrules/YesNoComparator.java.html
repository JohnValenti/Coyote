package com.verint.web.fs.workrules;

import com.bluepumpkin.common.localization.Localizer;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.HashMap;

public class YesNoComparator implements Comparator<Boolean>, Serializable {
	private static final long serialVersionUID = 1L;

	HashMap<Boolean, Integer> comparisonMap = new HashMap<Boolean, Integer>();
	String yesLocalizedString;
	String noLocalizedString;

	public YesNoComparator(Localizer localizer) {
		HashMap<String, Boolean> tempMap = new HashMap<String, Boolean>();
		yesLocalizedString = localizer.formatYesNo(true);
		noLocalizedString = localizer.formatYesNo(false);

		tempMap.put(yesLocalizedString, true);
		tempMap.put(noLocalizedString, false);

		ArrayList<String> localizedValues = new ArrayList<String>();
		localizedValues.add(yesLocalizedString);
		localizedValues.add(noLocalizedString);

		Collections.sort(localizedValues, localizer.getCollator());

		for (int i = 0; i < localizedValues.size(); i++) {
			comparisonMap.put(tempMap.get(localizedValues.get(i)), i);
		}
	}

	public int compare(Boolean o1, Boolean o2) {
		if (o1 == null && o2 == null) {
			return 0;
		}
		if (o1 == null && o2 != null) {
			return -1;
		}
		if (o2 == null && o1 != null) {
			return 1;
		}
		if (comparisonMap.get(o1) > comparisonMap.get(o2)) {
			return 1;
		}
		if (comparisonMap.get(o1) == comparisonMap.get(o2)) {
			return 0;
		}
		return -1;
	}
}
