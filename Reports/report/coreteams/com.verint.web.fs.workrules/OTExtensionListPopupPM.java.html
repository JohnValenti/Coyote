package com.verint.web.fs.workrules;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.PaginationPair;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectIDListComparator;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftOTExtension;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarTextButton;
import com.witness.web.uif.system.RequestContext;

import java.util.*;

public class OTExtensionListPopupPM extends WorkRulesSelectionPopupPM {
	private static final long serialVersionUID = 1L;

	public static final String FORM_ACTION = "otextension_list_popup";
	public static final String POPUP_ID = "POPUP_OT_EXTENSION";
	public static final String POPUP_ARGS = "otExtensionIDs,isByOrg,isSelectionPopup,ownerID,selectedIDs,optimizationTargetTimeZoneID,timeZoneEffectiveDate";
	public static final String POPUP_ARG_VO_ID = "otExtensionID";
	public static final String POPUP_ARG_VO_IDS = "otExtensionIDs";
	public static final String ARG_TIME_ZONE = "optimizationTargetTimeZoneID";
	public static final String ARG_TIME_ZONE_DATE = "timeZoneEffectiveDate";

	private ArrayList<ShiftOTExtension> otExtensions;
	private String otExtensionIDStr;
	private ID ownerID;
	private boolean isByOrg;
	private boolean isSelectionPopup = false;
	private HashSet<ID> selectedIDs = new HashSet<>();

	private Map<ID, Organization> orgMap = new HashMap<>();
	private Map<ID, Campaign> campaignMap = new HashMap<>();
	private Map<ID, Activity> activityMap = new HashMap<>();

	public OTExtensionListPopupPM(RequestContext context) {
		super(context);
	}

	public static OTExtensionListPopupPM getInstance(RequestContext context) {
		return (OTExtensionListPopupPM) getInstance(context, OTExtensionListPopupPM.class);
	}

	@Override
	protected void initializeModel() {
		if (m_isInitialized) {
			return;
		}
		super.initializeModel();
		if (isSelectionPopup) {
			ToolbarTextButton addButton = m_toolbar.createCloseWindowButton(
					"addShiftOTExtensions", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
							FsWebBundleKey.FS_ADD_OT_EXTENSIONS), true);
			addButton.setCustomJS(m_toolbar.getName() + ".disableButton('addShiftOTExtensions'); WorkRulesUtil.addShiftOTExtensions();");
			m_toolbar.addToolbarElement(addButton);
			m_toolbar.addCloseWindowTextButton("cancel", m_localizer.i18n(
					CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_CANCEL), true);
		}
	}

	@Override
	protected void initUtilityPane() {
		getUtilityPane().setIsHelpEnabled(false);
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		if (isSelectionPopup) {
			m_contentTitlePC.setPageTitle(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_OT_EXTS));
		} else {
			m_contentTitlePC.setPageTitle(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_OT_EXT_DETAILS));
		}
	}

	@Override
	protected void loadData() throws Exception {
		if (ownerID != null) {
			if (isByOrg) {
				if (isPaginationNavPressed()) {
					otExtensions = new ArrayList<>(WorkRulesModelHandler.getShiftOTExtensionsByIDs(m_context,
							getPageIDs()));
				} else {
					PaginationPair<ShiftOTExtension> otExtensionData =
							WorkRulesModelHandler.getShiftOTExtensionsByOrgHierarchy(m_context,
									ownerID, m_list.getHeader().isAscending(), getPageIDSize(),
									getPagePaginationIndex(), m_list.getSortComparators(), selectedIDs);
					otExtensions = new ArrayList<>(otExtensionData.getValues());
					initPaginationIDs(otExtensionData.getIDs());
				}
			} else {
				SchedulingPeriod selectedSchedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
						m_context, ownerID);
				if (isPaginationNavPressed()) {
					otExtensions = new ArrayList<>(
							WorkRulesModelHandler.getShiftOTExtensionsByIDs(m_context, getPageIDs()));
				} else {
					PaginationPair<ShiftOTExtension> otExtensionData =
							WorkRulesModelHandler.getShiftOTExtensionsBySchedulingPeriod(m_context,
									selectedSchedulingPeriod, m_list.getHeader().isAscending(), getPageIDSize(),
									getPagePaginationIndex(), m_list.getSortComparators(), selectedIDs);
					otExtensions = new ArrayList<>(otExtensionData.getValues());
					initPaginationIDs(otExtensionData.getIDs());
				}
			}
		} else {
			otExtensions = new ArrayList<>(
					WorkRulesModelHandler.getShiftOTExtensionsByIDs(m_context, StringUtil.parseIDStringToCollection(
							otExtensionIDStr)));
		}

		Collections.sort(otExtensions, new ValueObjectIDListComparator(new ArrayList<>(getPageIDs())));

		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<>();
		Collection<ID> campaignIDs = new HashSet<>();
		Collection<ID> activityIDs = new HashSet<>();
		for (ShiftOTExtension sote : otExtensions) {
			if (sote.getOrganizationID() != null) {
				orgIDs.add(sote.getOrganizationID());
			}
			if (sote.getCampaignID() != null) {
				campaignIDs.add(sote.getCampaignID());
			}
			if (sote.getActivityID() != null) {
				activityIDs.add(sote.getActivityID());
			}
			for (ShiftEvent se : sote.getShiftEvents()) {
				if (se.getOrganizationID() != null) {
					orgIDs.add(se.getOrganizationID());
				}
				if (se.getCampaignID() != null) {
					campaignIDs.add(se.getCampaignID());
				}
				if (se.getActivityID() != null) {
					activityIDs.add(se.getActivityID());
				}
			}
		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
		if (!isByOrg) {
			campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
		}
		activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);
	}

	public void setotExtensionIDs(String otExtensionIDs) {
		otExtensionIDStr = otExtensionIDs;
	}

	public void setIsByOrg(boolean isByOrg) {
		this.isByOrg = isByOrg;
	}

	public void setIsSelectionPopup(boolean isSelectionPopup) {
		this.isSelectionPopup = isSelectionPopup;
	}

	public void setOwnerID(String ownerID) {
		this.ownerID = ID.fromInt(Integer.parseInt(ownerID));
	}

	public void setSelectedIDs(String selectedIDs) {
		this.selectedIDs = new HashSet<>(StringUtil.parseIDStringToCollection(selectedIDs));
	}

	@Override
	protected OTExtensionListPC initList() {
		return new OTExtensionListPC(getRequestContext(), "otExtensionTable", optimizationTargetTimeZone,
				timeZoneEffectiveDate, otExtensions, orgMap, campaignMap, activityMap);
	}
}
