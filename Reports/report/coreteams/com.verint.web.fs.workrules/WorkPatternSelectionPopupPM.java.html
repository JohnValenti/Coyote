@SuppressWarnings("serial")
public class WorkPatternSelectionPopupPM extends WorkRulesSelectionPopupPM {
	// DynamicListMediator.js needs popup ids to start with POPUP_
	public static final String POPUP_ID = "POPUP_WorkPatternSelection";
	public static final String POPUP_ARG_PARENT_TABLE = "parentTable";
	public static final String POPUP_ARG_ORG_ID = "orgID";
	public static final String POPUP_ARGS = POPUP_ARG_PARENT_TABLE + "," + POPUP_ARG_ORG_ID;
	public static final String POPUP_URL = "workpattern_selection_popup";
	public static final String  WORKPATTERN_SELECTIONPOPUP_SORTCOLUMN = WorkPatternMasterDetailListPC.WORKPATTERN_SELECTIONPOPUP_SORTCOLUMN;
	public static final String  WORKPATTERN_SELECTIONPOPUP_SORTORDER = WorkPatternMasterDetailListPC.WORKPATTERN_SELECTIONPOPUP_SORTORDER;

	private Collection<ShiftPattern> data_workPatterns = new ArrayList<ShiftPattern>();
	private Map<ID, Organization> data_orgMap = new HashMap<ID, Organization>();
	private Map<ID, Activity> data_activityMap = new HashMap<ID, Activity>();
	private Map<ID, EmployeeType> data_empTypeMap = new HashMap<ID, EmployeeType>();
	private Map<ID, Campaign> data_campaignMap = new HashMap<ID, Campaign>();
	private ID data_orgID = null;
	private String data_parentTable = null;

	private WorkPatternMasterDetailListPC pc_workPatternList;

	private Category m_log = Log.initCategory(WorkPatternMasterDetailListPC.class.getName());

	public WorkPatternSelectionPopupPM(RequestContext context) {
		super(context);
	}

	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		ContentTitlePC pc = getContentTitle();
		pc.setPageTitle(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.WORK_PATTERNS));
	}

	@Override
	public boolean extractParameters(HttpServletRequest request) {
		boolean retVal = super.extractParameters(request);

		String orgIDStr = request.getParameter(POPUP_ARG_ORG_ID);
		if (!StringUtil.isEmpty(orgIDStr)) {
			data_orgID = ID.fromInt(Integer.parseInt(orgIDStr));
		} else {
			retVal = false;
		}

		String parentTableStr = request.getParameter(POPUP_ARG_PARENT_TABLE);
		if (!StringUtil.isEmpty(parentTableStr)) {
			data_parentTable = parentTableStr;
		} else {
			retVal = false;
		}

		return retVal;
	}

	private boolean dataLoaded = false;

	@Override
	protected void loadData() throws Exception {
		if (dataLoaded) {
			return;
		}
		dataLoaded = true;

		super.loadData();
		if (isPaginationNavPressed()) {
			data_workPatterns.addAll(WorkRulesModelHandler.getShiftPatterns(m_context, getPageIDs()));
		} else {
			PaginationPair<ShiftPattern> shiftPatternData = WorkRulesModelHandler.getWeeklyShiftPatternsByOrgHierarchy(m_context,
					data_orgID, pc_workPatternList.getHeader().isAscending(), getPageIDSize(),
					getPagePaginationIndex(), pc_workPatternList.getSortComparators(), new HashSet<ID>(),getTextToFind(itemToFind));
			data_workPatterns.addAll(shiftPatternData.getValues());
			initPaginationIDs(shiftPatternData.getIDs());
		}

		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		orgIDs.add(data_orgID);
		Collection<ID> campaignIDs = new HashSet<ID>();
		Collection<ID> activityIDs = new HashSet<ID>();
		Collection<ID> empTypeIDs = new HashSet<ID>();

		for (ShiftPattern sp : data_workPatterns) {
			if (sp.getOrganizationID() != null) {
				orgIDs.add(sp.getOrganizationID());
			}
			if (sp.getCampaignID() != null) {
				campaignIDs.add(sp.getCampaignID());
			}
			if (sp.getEmployeeTypeID() != null) {
				empTypeIDs.add(sp.getEmployeeTypeID());
			}
			for (Shift shift : sp.getShiftPatternShiftAttributes().keySet()) {
				if (shift.getOrganizationID() != null) {
					orgIDs.add(shift.getOrganizationID());
				}
				if (shift.getCampaignID() != null) {
					campaignIDs.add(shift.getCampaignID());
				}
				if (shift.getActivityID() != null) {
					activityIDs.add(shift.getActivityID());
				}
				for (ShiftEvent se : shift.getShiftEvents()) {
					if (se.getOrganizationID() != null) {
						orgIDs.add(se.getOrganizationID());
					}
					if (se.getCampaignID() != null) {
						campaignIDs.add(se.getCampaignID());
					}
					if (se.getActivityID() != null) {
						activityIDs.add(se.getActivityID());
					}
				}
			}
			for (VTOEvent vtoe : sp.getVTOEventStartTimes().keySet()) {
				if (vtoe.getOrganizationID() != null) {
					orgIDs.add(vtoe.getOrganizationID());
				}
				if (vtoe.getCampaignID() != null) {
					campaignIDs.add(vtoe.getCampaignID());
				}
				if (vtoe.getActivityID() != null) {
					activityIDs.add(vtoe.getActivityID());
				}
			}
			for (ShiftOTExtension sote : sp.getOTExtensionStartTimes().keySet()) {
				if (sote.getOrganizationID() != null) {
					orgIDs.add(sote.getOrganizationID());
				}
				if (sote.getCampaignID() != null) {
					campaignIDs.add(sote.getCampaignID());
				}
				if (sote.getActivityID() != null) {
					activityIDs.add(sote.getActivityID());
				}
			}
		}

		data_orgMap.putAll(WorkRulesModelHandler.getOrgs(m_context, orgIDs));
		data_activityMap.putAll(WorkRulesModelHandler.getActivities(m_context, activityIDs));
		data_empTypeMap.putAll(EmployeeTypeModelHandler.getEmployeeTypesByIDs(m_context, empTypeIDs));

		pc_workPatternList.setData(
			data_workPatterns,
			data_orgMap.get(data_orgID).getTimeZone(),
			new Date(),
			true, //isByOrg
			data_orgMap,
			data_campaignMap,
			data_activityMap,
			data_empTypeMap
		);

	}

	private void setRowAttributes(DefaultMultiColumnNodeData rowData,
			ShiftPattern shiftPattern) {
		rowData.setNodeAttribute(WorkpaneMultiColPM.IS_ADDABLE, String.valueOf(true));
	}
	@Override
	protected void initToolbar() {
		FindToolbarPageComponent findToolBar = new FindToolbarPageComponent(m_context, getToolbar(),
				FIND_ITEM_FN, WorkRulesSelectionPopupRH.FIND_ACTION,
				getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_FIND_WORKPATTERNS));	
		getToolbar().addToolbarComponent(findToolBar);
		getToolbar().addDivider(); 
		ToolbarPC toolbarPC = getToolbar();
		ToolbarTextButton addButton = toolbarPC.addCloseWindowTextButton("ADD", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ADD_SELECTED_WORK_PATTERN), true);
		addButton.setCustomJS(
				m_toolbar.getName() + ".disableButton('ADD'); " +
				"jQuery('tr.tblRowSelect').each(function(idx, val) {var data = []; jQuery(val).find('td').each(function(tdidx, td) {data[data.length]=td.innerHTML;}); window.opener." + data_parentTable + ".appendRow(jQuery(val).attr('uid'), data);});");

		//ESR 4361448
		int pageIDSizeLimit = 60;
		try {
			String pageIDSizeFetched = BbmManagerFactory.getDBConfigManager().getValue(ConfigKey.FS_WEB_ID_SIZE_LIMIT);
			if ( pageIDSizeFetched != null ){
				pageIDSizeLimit = Integer.parseInt(pageIDSizeFetched);
			}
		} catch (Exception e) {
			handleUnableToLoadDataErrorAndLogAsDebug(log, e);
		}
		setLimitingPageSize(pageIDSizeLimit);
	}
	@Override
	protected MultiColListPC initList() {
		try {
			if (pc_workPatternList == null) {
				pc_workPatternList = new WorkPatternMasterDetailListPC(m_context,WorkPatternMasterDetailListPC.USED_IN_SELECTIONPOPUP){
					@Override
					protected void setRowAttributes(
							DefaultMultiColumnNodeData rowData,
							ShiftPattern shiftPattern) {
						super.setRowAttributes(rowData, shiftPattern);
						WorkPatternSelectionPopupPM.this.setRowAttributes(rowData, shiftPattern);
					}};
			}
			return pc_workPatternList;
		} catch (Exception e) {
			this.handleUnableToLoadDataError(m_log, e);
		}
		return null;
	}

	public static WorkPatternSelectionPopupPM getInstance(RequestContext context) {
		return (WorkPatternSelectionPopupPM) PageModel.getInstance(context, WorkPatternSelectionPopupPM.class);
	}
}
