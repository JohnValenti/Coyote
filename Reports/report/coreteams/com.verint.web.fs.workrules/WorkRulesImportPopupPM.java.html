package com.verint.web.fs.workrules;

import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.pagemodel.ImportPopupPM;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;

import java.util.Date;
import java.util.ResourceBundle;
import java.util.TimeZone;

public abstract class WorkRulesImportPopupPM extends ImportPopupPM {

	private static final long serialVersionUID = 1L;

	protected ResourceBundle m_fsResourceBundle;

	public static final String IS_BY_ORG_ARGUMENT = "isByOrg";
	public static final String CAMPAIGN_ID_ARGUMENT = "campaignId";
	public static final String SCHEDULING_PERIOD_ID_ARGUMENT = "schedulingPeriodId";
	public static final String ORGANIZATION_ID_ARGUMENT = "organizationId";
	public static final String OPTIMIZATION_TARGET_TIMEZONE = "optimizationTargetTimeZone";
	public static final String TIMEZONE_EFFECTIVE_DATE = "timezoneEffectiveDate";

	protected boolean m_isByOrganization = false;
	protected int m_campaignId = -1;
	protected int m_schedulingPeriodId = -1;
	protected int m_organizationId = -1;
	protected TimeZone m_optimizationTargetTimezone = null;
	protected Date m_timeZoneEffectiveDate = new Date();

	public WorkRulesImportPopupPM(RequestContext context) {
		super(context);
		m_fsResourceBundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);
	}

	// Auto-extract parameter helper.
	public void setIsByOrg(boolean val) {
		m_isByOrganization = val;
	}

	// Auto-extract parameter helper.
	public void setCampaignId(String val) {
		m_campaignId = Integer.parseInt(val);
	}

	// Auto-extract parameter helper.
	public void setSchedulingPeriodId(String val) {
		m_schedulingPeriodId = Integer.parseInt(val);
	}

	// Auto-extract parameter helper.
	public void setOrganizationId(String val) {
		m_organizationId = Integer.parseInt(val);
	}

	// Auto-extract parameter helper.
	public void setOptimizationTargetTimeZone(String val) {
		m_optimizationTargetTimezone = TimeZone.getTimeZone(val);
	}

	@SuppressWarnings("deprecation")
	public void setTimezoneEffectiveDate(String val) {
		m_timeZoneEffectiveDate = new Date(val);
	}

	public String getHtmlDisplay() {
		return super.getHtmlDisplay() +
				HtmlUtil.makeHiddenInput(IS_BY_ORG_ARGUMENT, String.valueOf(m_isByOrganization)) +
				HtmlUtil.makeHiddenInput(CAMPAIGN_ID_ARGUMENT, String.valueOf(m_campaignId)) +
				HtmlUtil.makeHiddenInput(SCHEDULING_PERIOD_ID_ARGUMENT, String.valueOf(m_schedulingPeriodId)) +
				HtmlUtil.makeHiddenInput(ORGANIZATION_ID_ARGUMENT, String.valueOf(m_organizationId)) +
				HtmlUtil.makeHiddenInput(OPTIMIZATION_TARGET_TIMEZONE, String.valueOf(m_optimizationTargetTimezone.getID())) +
				HtmlUtil.makeHiddenInput(TIMEZONE_EFFECTIVE_DATE, String.valueOf(m_timeZoneEffectiveDate));
	}

	/**
	 * Parses minutes from time display format like 14:15, 08:10 or 9:15
	 *
	 * @param value time display (ex: 18:00, 9:00, 10:05)
	 * @return int minutes
	 */
	protected int parseDurationMinutesFromHHMM(String value) {
		if (value == null) {
			return 0;
		}

		int hours;
		int minutes;
		String[] splitValues = value.split(":");
		if (splitValues.length < 2) {
			return 0;
		}

		hours = Integer.parseInt(splitValues[0]);
		minutes = Integer.parseInt(splitValues[1]);

		return hours * 60 + minutes;
	}

	protected enum StartType {
		AnyTime,
		Relative,
		Absolute
	}

}
