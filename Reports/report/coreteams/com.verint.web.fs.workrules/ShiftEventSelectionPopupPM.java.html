package com.verint.web.fs.workrules;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.PaginationPair;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectIDListComparator;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEvent;
import com.bluepumpkin.web.bbm.campaign.CampaignModelHandler;
import com.bluepumpkin.web.bbm.organization.OrganizationModelHandler;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.verint.web.fs.widgets.FindToolbarPageComponent;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagecomponent.list.MultiColListPC;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarTextButton;
import com.witness.web.uif.pagemodel.WorkpaneMultiColPM;
import com.witness.web.uif.system.RequestContext;

import javax.servlet.http.HttpServletRequest;
import java.util.*;

public class ShiftEventSelectionPopupPM extends WorkRulesSelectionPopupPM {
	private static final long serialVersionUID = 1L;

	public static final String FORM_ACTION = "shift_event_selection_popup";

	private ArrayList<ShiftEvent> m_shiftEvents = new ArrayList<ShiftEvent>();
	private ID m_ownerID;    //This is either an org ID or a scheduling period ID depending on the mode (isByOrg)
	private HashSet<ID> m_selectedIDs = new HashSet<ID>();
	private boolean isByOrg;
	protected Organization m_selectedOrg = null;
	protected Campaign m_selectedCampaign = null;
	protected SchedulingPeriod m_selectedSchedulingPeriod = null;

	private Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
	private Map<ID, Campaign> campaignMap = new HashMap<ID, Campaign>();
	private Map<ID, Activity> activityMap = new HashMap<ID, Activity>();

	public ShiftEventSelectionPopupPM(RequestContext context) {
		super(context);
	}

	public static ShiftEventSelectionPopupPM getInstance(RequestContext context) {
		return (ShiftEventSelectionPopupPM) getInstance(context, ShiftEventSelectionPopupPM.class);
	}

	public boolean extractParameters(HttpServletRequest request) {
		boolean success = super.extractParameters(request);
		extractParameters(request, "ownerID");
		extractParameters(request, "isByOrg");
		return success;
	}

	protected void initToolbar() {
		FindToolbarPageComponent findToolBar = new FindToolbarPageComponent(m_context, getToolbar(), FIND_ITEM_FN,
				WorkRulesSelectionPopupRH.FIND_ACTION,
				getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_FIND_SHIFT_EVENTS_PROMPT));
		getToolbar().addToolbarComponent(findToolBar);
		getToolbar().addDivider();
		ToolbarTextButton addButton = this.getToolbar().createCloseWindowButton(
				"addShiftEvents", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME,
						FsWebBundleKey.FS_ADD_SHIFT_EVENTS), true);
		addButton.setCustomJS("WorkRulesUtil.addSelectedShiftEvents();");
		this.getToolbar().addToolbarElement(addButton);
		this.getToolbar().addCloseWindowTextButton("cancel", m_localizer.i18n(
				CoreWebBundleKey.BUNDLE_NAME, CoreWebBundleKey.TOOLBAR_CANCEL), true);
	}

	@Override
	protected void initScreenContent() {
		initSelectableItems();
		initHeader();
	}

	protected void initUtilityPane() {
		getUtilityPane().setIsHelpEnabled(false);
	}

	protected void initContentTitle() {
		super.initContentTitle();

		m_contentTitlePC.setPageTitle(getLocalizer().i18n(
				FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_AVAILABLE_SHIFT_EVENTS));
	}

	protected void loadData() throws Exception {
		if (m_ownerID != null) {
			if (isByOrg) {
				m_selectedOrg = OrganizationModelHandler.getOrganization(m_context, m_ownerID);
				optimizationTargetTimeZone = m_selectedOrg.getTimeZone();
				if (isPaginationNavPressed()) {
					m_shiftEvents = new ArrayList<ShiftEvent>(WorkRulesModelHandler.getShiftEvents(m_context, getPageIDs()));
				} else {
					PaginationPair<ShiftEvent> shiftEventData = WorkRulesModelHandler.getShiftEventsByOrgHierarchy(m_context,
							m_selectedOrg.getID(), m_list.getHeader().isAscending(), getPageIDSize(),
							getPagePaginationIndex(), m_list.getSortComparators(), m_selectedIDs, getTextToFind(itemToFind));
					m_shiftEvents = new ArrayList<ShiftEvent>(new HashSet<ShiftEvent>(shiftEventData.getValues()));
					initPaginationIDs(shiftEventData.getIDs());
				}
			} else {
				m_selectedSchedulingPeriod = CampaignModelHandler.getSchedulingPeriodByID(
						m_context, m_ownerID);
				m_selectedCampaign = CampaignModelHandler.getCampaign(m_context,
						m_selectedSchedulingPeriod.getCampaignID());
				optimizationTargetTimeZone = m_selectedCampaign.getTimeZone();
				if (isPaginationNavPressed()) {
					m_shiftEvents = new ArrayList<ShiftEvent>(WorkRulesModelHandler.getShiftEvents(m_context, getPageIDs()));
				} else {
					PaginationPair<ShiftEvent> shiftEventData =
							WorkRulesModelHandler.getShiftEventsBySchedulingPeriod(m_context,
									m_selectedSchedulingPeriod, m_list.getHeader().isAscending(), getPageIDSize(),
									getPagePaginationIndex(), m_list.getSortComparators(), m_selectedIDs, getTextToFind(itemToFind));
					m_shiftEvents = new ArrayList<ShiftEvent>(new HashSet<ShiftEvent>(shiftEventData.getValues()));
					initPaginationIDs(shiftEventData.getIDs());
				}
			}

			Collections.sort(m_shiftEvents, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));

			// Collections of IDs of objects we will need for display
			HashSet<ID> orgIDs = new HashSet<ID>();
			HashSet<ID> campaignIDs = new HashSet<ID>();
			HashSet<ID> activityIDs = new HashSet<ID>();
			for (ShiftEvent se : m_shiftEvents) {
				if (se.getOrganizationID() != null) {
					orgIDs.add(se.getOrganizationID());
				}
				if (se.getCampaignID() != null) {
					campaignIDs.add(se.getCampaignID());
				}
				if (se.getActivityID() != null) {
					activityIDs.add(se.getActivityID());
				}
			}

			orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
			if (!isByOrg) {
				campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
			}
			activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);
		}
	}

	public void setOwnerID(ID ownerID) {
		m_ownerID = ownerID;
	}

	public void setIsByOrg(boolean isByOrg) {
		this.isByOrg = isByOrg;
	}

	public void setSelectedIDs(String selectedIDs) {
		m_selectedIDs = new HashSet<ID>(StringUtil.parseIDStringToCollection(selectedIDs));
	}

	@Override
	protected void initHeader() {
	}

	@Override
	protected MultiColListPC initList() {
		return new ShiftEventListPC(getRequestContext(), "shiftEventTable", m_shiftEvents,
				optimizationTargetTimeZone,
				m_selectedSchedulingPeriod == null ? new Date() : m_selectedSchedulingPeriod.getStartTime(),
				orgMap, campaignMap, activityMap, false) {
			@Override
			protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, ShiftEvent shiftEvent) {
				super.setRowAttributes(mcnd, shiftEvent);
				mcnd.setNodeAttribute(WorkpaneMultiColPM.IS_DELETEABLE, String.valueOf(true));
			}
		};
	}
}
