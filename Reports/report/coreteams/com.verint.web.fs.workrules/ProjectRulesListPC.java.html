	public ProjectRulesListPC(
			RequestContext context,
			String name,
			Collection<ProjectRules> projRules,
			TimeZone optimizationTargetTimezone,
			Date timeZoneEffectiveDate,
			Map<ID, Organization> orgMap,
			Map<ID, Campaign> campaignMap,
			Map<ID, Activity> activityMap) {
		super(context);
		setName(name);

		if (optimizationTargetTimezone != null) {
			this.optimizationTargetTimeZone = optimizationTargetTimezone;
		} else {
			this.optimizationTargetTimeZone = getRequestContext().getViewingTimeZone();
		}
		this.timeZoneEffectiveDate = timeZoneEffectiveDate;

		this.projectRules = projRules;
		this.orgMap = orgMap;
		this.campaignMap = campaignMap;
		this.activityMap = activityMap;

		initHeader();

		addRequiredJavaScriptFile(WorkRulesJSFileID.WORK_RULES_UTIL);
		addPopupArgs(ActivityPopupPM.POPUP_ID, ActivityPopupPM.FORM_ACTION, "",
				ActivityPopupPM.POPUP_ARGS, "1000,600,ctr,ctr", true, 1);
	}

	private void initRows() {
		if (isRowsInited) {
			return;
		}
		isRowsInited = true;

		Collection<MultiColumnNodeData> rows = super.getListModel();

		Localizer localizer = m_context.getLocalizer();

		String optimizationTargetName = "";
		StringBuffer projectRuleID = new StringBuffer();
		for (ProjectRules projRules : this.projectRules) {
			int timeZoneOffsetInMins = TimeZoneOverrideUtil.getTimeZoneOffset(m_context, optimizationTargetTimeZone, timeZoneEffectiveDate);
			if (projectRuleID.length() > 0) {
				projectRuleID.append(",");
			}
			projectRuleID.append(projRules.getID().toString());

			if (projRules.getOrganizationID() != null) {
				Organization o = orgMap.get(projRules.getOrganizationID());
				if (o != null) {
					if (timeZoneEffectiveDate != null) {
						timeZoneOffsetInMins += (optimizationTargetTimeZone.getOffset(timeZoneEffectiveDate.getTime())
								- o.getTimeZone().getOffset(timeZoneEffectiveDate.getTime())) / (60 * 1000L);
					}
					optimizationTargetName = o.getName();
				}
			} else {
				Campaign c = campaignMap.get(projRules.getCampaignID());
				if (c != null) {
					if (timeZoneEffectiveDate != null) {
						timeZoneOffsetInMins += (optimizationTargetTimeZone.getOffset(timeZoneEffectiveDate.getTime())
								- c.getTimeZone().getOffset(timeZoneEffectiveDate.getTime())) / (60 * 1000L);
					}
					optimizationTargetName = m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.LOCAL);
				}
			}
			DefaultMultiColumnNodeData nodeData = new DefaultMultiColumnNodeData(projRules.getID());

			//Name
			if (projRules.isMultiEditCommon() == false) {
				nodeData.add(MULTI_VALUE_DATA + projRules.getName());
			} else {
				nodeData.add(projRules.getName());
			}

			// Description
			nodeData.add(projRules.getDescription());
			// Activity
			Activity a = activityMap.get(projRules.getActivityID());
			//The activity should never be null, but if it is it's preferable to display a blank value for
			//  activity name instead of having the app crash
			if (a != null) {
				nodeData.add(a.getName());
			} else {
				nodeData.add("");
			}

			//Duration
			nodeData.add(localizer.formatDurationInMins(projRules.getDuration().getDurationInMinutes()));

			//Organization or Campaign name
			nodeData.add(optimizationTargetName);
			nodeData.add(projRules.getDailyMaxPerEmployee() == ProjectRulesFormPM.EmployeeCountPC.MAX_EMPLOYEE_COUNT ? "" : projRules.getDailyMaxPerEmployee());
			nodeData.add(projRules.getWeeklyMaxPerEmployee() == ProjectRulesFormPM.EmployeeCountPC.MAX_EMPLOYEE_COUNT ? "" : projRules.getWeeklyMaxPerEmployee());
			nodeData.add(projRules.getMaxSimultaneous() == ProjectRulesFormPM.EmployeeCountPC.MAX_EMPLOYEE_COUNT ? "" : projRules.getMaxSimultaneous());

			Map<ProjectStart.WeekDay, StartTimes> m = getStartTimes(TimeContextFactory.getTimeContext(projRules, orgMap, campaignMap),
					projRules.getProjectStartTimes());

			//As the header is starting from Sunday the values in the table rows should also display in the same way
			for (WeekDay wd : WeekDay.values()) {
				StartTimes value = m.get(wd);

				if (value != null) {
					nodeData.add("<table><tr><td style=" + TIME_RANGE_STYLE + " nowrap>"
							+ value.toString().replaceAll(",", ",</td></tr><tr><td style=" + TIME_RANGE_STYLE + ">") + "</td></tr></table>");
				} else {
					nodeData.add(" ");
				}
			}
			setRowAttributes(nodeData, projRules);

			rows.add(nodeData);
		}
		setRowIDs(projectRuleID.toString());
	}

	@Override
	public LinkedHashMap<DBSortField, Comparator> getSortComparators() {
		LinkedHashMap<DBSortField, Comparator> comparators
				= new LinkedHashMap<DBSortField, Comparator>();
		for (ProjectRulesSortField sortField : getDBSortFieldIndices()) {
			switch (sortField) {
				case ID:
					comparators.put(ProjectRulesSortField.ID, null);
					break;
				case NAME:
					comparators.put(ProjectRulesSortField.NAME, null);
					break;
				case ACTIVITY:
					comparators.put(ProjectRulesSortField.ACTIVITY, null);
					break;
				case LENGTH:
					comparators.put(ProjectRulesSortField.LENGTH, null);
					break;
				case DAILYMAXPEREMPLOYEE:
					comparators.put(ProjectRulesSortField.DAILYMAXPEREMPLOYEE, null);
					break;
				case DESCRIPTION:
					comparators.put(ProjectRulesSortField.DESCRIPTION, null);
					break;
				case MAXSIMULTANEOUS:
					comparators.put(ProjectRulesSortField.MAXSIMULTANEOUS, null);
					break;
				case ORGANIZATION:
					comparators.put(ProjectRulesSortField.ORGANIZATION, new OrganizationCampaignNameComparator(getLocalizer()));
					break;
				case WEEKLYMAXPEREMPLOYEE:
					comparators.put(ProjectRulesSortField.WEEKLYMAXPEREMPLOYEE, null);
					break;
				case START_TIME:
					comparators.put(ProjectRulesSortField.START_TIME, null);
					break;
			}
		}
		return comparators;
	}

	public ArrayList<ProjectRulesSortField> getDBSortFieldIndices() {
		ArrayList<ProjectRulesSortField> retVal = new ArrayList<ProjectRulesSortField>();
		if (getHeader().getSortColumnIndexes() == null) {
			//Default sort index is by ID (chronological order in which object was created)
			retVal.add(ProjectRulesSortField.ID);
		} else {
			String sortFieldName = getHeader().getSortColumn();
			if (sortFieldName.equals(FsWebBundleKey.FS_PROJECT_RULES_NAME)) {
				retVal.add(ProjectRulesSortField.NAME);
			} else if (sortFieldName.equals(FsWebBundleKey.FS_PROJECT_RULES_DESCRIPTION)) {
				retVal.add(ProjectRulesSortField.DESCRIPTION);
			} else if (sortFieldName.equals(FsWebBundleKey.FS_PROJECT_RULES_ACTIVITY)) {
				retVal.add(ProjectRulesSortField.ACTIVITY);
			} else if (sortFieldName.equals(FsWebBundleKey.FS_PROJECT_RULES_LENGTH)) {
				retVal.add(ProjectRulesSortField.LENGTH);
			} else if (sortFieldName.equals(FsWebBundleKey.FS_PROJECT_RULES_ORGANIZATION)) {
				retVal.add(ProjectRulesSortField.ORGANIZATION);
			} else if (sortFieldName.equals(FsWebBundleKey.FS_PROJECT_RULES_DAILYMAXPEREMPLOYEE)) {
				retVal.add(ProjectRulesSortField.DAILYMAXPEREMPLOYEE);
			} else if (sortFieldName.equals(FsWebBundleKey.FS_PROJECT_RULES_WEEKLYMAXPEREMPLOYEE)) {
				retVal.add(ProjectRulesSortField.WEEKLYMAXPEREMPLOYEE);
			} else if (sortFieldName.equals(FsWebBundleKey.FS_PROJECT_RULES_MAXSIMULTANEOUS)) {
				retVal.add(ProjectRulesSortField.MAXSIMULTANEOUS);
			} else if (sortFieldName.equals(FsWebBundleKey.SUNDAY)
					|| sortFieldName.equals(FsWebBundleKey.MONDAY)
					|| sortFieldName.equals(FsWebBundleKey.TUESDAY)
					|| sortFieldName.equals(FsWebBundleKey.WEDNESDAY)
					|| sortFieldName.equals(FsWebBundleKey.THURSDAY)
					|| sortFieldName.equals(FsWebBundleKey.FRIDAY)
					|| sortFieldName.equals(FsWebBundleKey.SATURDAY)) {
				retVal.add(ProjectRulesSortField.START_TIME);
			}
		}
		return retVal;
	}

	/**
	 * @param tc
	 * @param projStartTimes <p>
	 * @return Utility method to get StartTimes Object which take care of merging contiguous time intervals and making periods
	 * and also showing discrete values of time. These objects even take care of the Timezone toggling.
	 */
	public Map<WeekDay, StartTimes> getStartTimes(TimeContext tc, Collection<ProjectStart> projStartTimes) {
		Map<WeekDay, StartTimes> weeklyMap = new HashMap(WeekDay.WEEK_LENGTH);
		weeklyMap.put(WeekDay.SUNDAY, null);
		weeklyMap.put(WeekDay.MONDAY, null);
		weeklyMap.put(WeekDay.TUESDAY, null);
		weeklyMap.put(WeekDay.WEDNESDAY, null);
		weeklyMap.put(WeekDay.THURSDAY, null);
		weeklyMap.put(WeekDay.FRIDAY, null);
		weeklyMap.put(WeekDay.SATURDAY, null);

		if (projStartTimes != null) {

			for (ProjectStart lObjPst : projStartTimes) {
				StartTimes lObjSt;

				if (weeklyMap.containsKey(lObjPst.getWeekDay())
						&& weeklyMap.get(lObjPst.getWeekDay()) != null) {
					lObjSt = weeklyMap.get(lObjPst.getWeekDay());
					TimeOfDay tod = new TimeOfDay(lObjPst.getPossibleStartTime() + tc.getDayBoundary());
					lObjSt.add(tod);
				} else {
					lObjSt = new StartTimes(
							m_context.getLocalizer(),
							TimeZoneOverrideUtil.getTimeZoneOffset(m_context, optimizationTargetTimeZone, timeZoneEffectiveDate));
					TimeOfDay tod = new TimeOfDay(lObjPst.getPossibleStartTime() + tc.getDayBoundary());
					lObjSt.add(tod);
					weeklyMap.put(lObjPst.getWeekDay(), lObjSt);
				}
			}
		}
		return weeklyMap;
	}
