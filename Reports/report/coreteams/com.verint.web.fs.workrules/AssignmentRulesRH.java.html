package com.verint.web.fs.workrules;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.common.datatypes.Pair;
import com.bluepumpkin.ejb.bbm.workrules.ejb.WorkRuleManager.AssignmentRuleValidationError;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRule;
import com.witness.web.uif.system.RequestContext;

import java.util.Collection;
import java.util.List;

public class AssignmentRulesRH extends WorkRulesRH<AssignmentRulesListPM, AssignmentRulesFormPM> {

	/**
	 * This page action is submitted when the user makes a selection
	 * for part of the assignment rule. The assignment rule is built
	 * in a step-wise, wizard-like fashion.
	 * <p>
	 * When this action is submitted, the next question to present
	 * to the user must be determined based on previous answers.
	 */
	public static final String PAGE_ACTION_SUBMIT_STEP = "STEP";

	@Override
	protected AssignmentRulesListPM getListPageModel(RequestContext context) {
		// Don't getInstance yet because the pagination needs to be set before the data is loaded
		// Instead of calling getInstance, instantiate and extract params.
		AssignmentRulesListPM pm = new AssignmentRulesListPM(context);
		pm.extractParameters(context.getRequest());
		return pm;
	}

	@Override
	protected AssignmentRulesFormPM getFormPageModel(RequestContext context) {
		return AssignmentRulesFormPM.getInstance(context);
	}

	@Override
	protected void processImport(RequestContext context, AssignmentRulesListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processExport(RequestContext context, AssignmentRulesListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processDownload(RequestContext context, AssignmentRulesListPM pageModel) throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processCopy(RequestContext context, AssignmentRulesListPM pageModel) throws BPException {
		WorkRulesModelHandler.copyAssignmentRulesToOrg(
				context,
				pageModel.getPerfromActionOnIDsAsCollection(),
				pageModel.getSelectedIDObject(),
				pageModel.getLocalizer().getLocaleContext());
		context.setPageModel(null); // reset the page model to refresh the list
	}

	@Override
	protected void processDelete(RequestContext context, AssignmentRulesListPM pageModel) throws BPException {
		WorkRulesModelHandler.deleteComplexWorkRule(context, pageModel.getPerfromActionOnIDsAsCollection());
		context.setPageModel(null); // reset the page model to refresh the list

	}

	@Override
	protected void processSave(RequestContext context, AssignmentRulesFormPM pageModel) throws BPException {
		if (pageModel.validate()) {
			ComplexWorkRule cwr = pageModel.getAssignmentRule();
			if (cwr != null) {
				Collection<Pair<AssignmentRuleValidationError, List<?>>> errors = WorkRulesModelHandler
						.saveAssignmentRule(context, cwr);
				if (errors.size() > 0) {
					pageModel.addErrorMessages(errors);
					setNextPage(context, FORM_PAGE);
				}
			} else {
				setNextPage(context, FORM_PAGE);
			}
		} else {
			setNextPage(context, FORM_PAGE);
		}
	}

	@Override
	public void processRequest(RequestContext context) throws Exception {
		if (isAction(context, PAGE_ACTION_SUBMIT_STEP)) {
			setNextPage(context, FORM_PAGE);
		} else {
			super.processRequest(context);
		}
	}

	@Override
	protected void processTableSort(RequestContext context, AssignmentRulesListPM pageModel) {
		String sortColumn = context.getRequest().getParameter("sortColumn");
		String sortOrder = context.getRequest().getParameter("sortOrder");
		context.setUserProperty(AssignmentRulesListPM.ASSIGNMENTRULE_LIST_SORTCOLUMN, sortColumn, true);
		context.setUserProperty(AssignmentRulesListPM.ASSIGNMENTRULE_LIST_SORTORDER, sortOrder, true);
	}

}
