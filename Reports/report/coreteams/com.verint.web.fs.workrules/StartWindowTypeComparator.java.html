package com.verint.web.fs.workrules;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.ejb.bbm.workrules.model.ShiftEventStartTimeType;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.HashMap;

public class StartWindowTypeComparator implements Comparator<ShiftEventStartTimeType>, Serializable {
	private static final long serialVersionUID = 1L;

	HashMap<ShiftEventStartTimeType, Integer> comparisonMap = new HashMap<ShiftEventStartTimeType, Integer>();
	private String anyTimeTypeLocalizedString;
	private String absoluteTypeLocalizedString;
	private String relativeTypeLocalizedString;

	public StartWindowTypeComparator(Localizer localizer) {
		HashMap<String, ShiftEventStartTimeType> tempMap = new HashMap<String, ShiftEventStartTimeType>();
		anyTimeTypeLocalizedString = localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ANY_TIME);
		absoluteTypeLocalizedString = localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_ABSOLUTE);
		relativeTypeLocalizedString = localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_RELATIVE_TO_SHIFT_START);

		tempMap.put(anyTimeTypeLocalizedString, ShiftEventStartTimeType.ANY_TIME);
		tempMap.put(absoluteTypeLocalizedString, ShiftEventStartTimeType.ABSOLUTE);
		tempMap.put(relativeTypeLocalizedString, ShiftEventStartTimeType.RELATIVE_TO_SHIFT_START);

		ArrayList<String> localizedValues = new ArrayList<String>();
		localizedValues.add(anyTimeTypeLocalizedString);
		localizedValues.add(absoluteTypeLocalizedString);
		localizedValues.add(relativeTypeLocalizedString);

		Collections.sort(localizedValues, localizer.getCollator());

		for (int i = 0; i < localizedValues.size(); i++) {
			comparisonMap.put(tempMap.get(localizedValues.get(i)), i);
		}
	}

	public int compare(ShiftEventStartTimeType o1, ShiftEventStartTimeType o2) {
		if (comparisonMap.get(o1) > comparisonMap.get(o2)) {
			return 1;
		}
		if (comparisonMap.get(o1) == comparisonMap.get(o2)) {
			return 0;
		}
		return -1;
	}
}
