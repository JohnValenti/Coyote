package com.verint.web.fs.workrules;

import java.rmi.RemoteException;
import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.system.RequestContext;


public class OTExtensionRH extends WorkRulesRH<OTExtensionListPM, OTExtensionFormPM> {
	private static final long serialVersionUID = 1L;



	@Override
	protected OTExtensionListPM getListPageModel(RequestContext context) {

		OTExtensionListPM pm = new OTExtensionListPM(context);
		pm.extractParameters(context.getRequest());
		return pm;

		// Can't getInstance yet because the pagination needs to be set up before the data is loaded
		//return OTExtensionListPM.getInstance(context);
	}

	@Override
	protected OTExtensionFormPM getFormPageModel(RequestContext context) {
		return OTExtensionFormPM.getInstance(context);
	}

	@Override
	protected void processAdd(RequestContext context, OTExtensionListPM pageModel)
	throws BPException {
		setNextPage(context, "formPage");
	}

	@Override
	protected void processCopy(RequestContext context, OTExtensionListPM pageModel)
	throws BPException {	
		try {
			if (pageModel.isByOrg) {
				WorkRulesModelHandler.copyOTExtensions(context, pageModel.getPerfromActionOnIDsAsCollection(), 
						pageModel.isByOrg, pageModel.getSelectedIDObject(), pageModel.getLocalizer().getLocaleContext());
			} else {
				pageModel.loadData();			//we need to do this to initialize the campaign/sp value objects
				WorkRulesModelHandler.copyOTExtensions(context, pageModel.getPerfromActionOnIDsAsCollection(), 
						pageModel.isByOrg, pageModel.selectedCampaign.getDEID(),
						pageModel.getLocalizer().getLocaleContext());
			}
			context.setPageModel(null);
			pageModel = getListPageModel(context);		//reload the new shifts
			pageModel.addPageMessage(Message.RESPONSE_TYPE,
					FsWebBundleKey.FS_COPY_OTEXTENSION_SUCCESS, FsWebBundleKey.BUNDLE_NAME);
		} catch(Exception ex) {
			pageModel.addPageMessage(Message.ERROR_TYPE,
					FsWebBundleKey.FS_UNABLE_TO_COPY_OTEXTENSION, FsWebBundleKey.BUNDLE_NAME);
			throw new BPException(ex);
		} finally {
			context.setPageModel(pageModel);
		}
	}

	@Override
	protected void processDelete(RequestContext context, OTExtensionListPM pageModel)
	throws BPException {
		try{
			WorkRulesModelHandler.deleteOTExtensions(context, pageModel.getPerfromActionOnIDsAsCollection());
		}catch(Exception e)
		{
			throw new BPException(e);
		}finally{
			context.setPageModel(null); // reset the page model to remove the deleted entries
		}
	}

	@Override
	protected void processEdit(RequestContext context, OTExtensionListPM pageModel)
	throws BPException {
		ShiftFormPM formPM = new ShiftFormPM(context);
		formPM.extractParameters(context.getRequest());
		context.setPageModel(formPM);
		setNextPage(context, "formPage");
	}

	@Override
	protected void processExport(RequestContext context, OTExtensionListPM pageModel)
	throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processImport(RequestContext context, OTExtensionListPM pageModel)
	throws BPException {
		// TODO Auto-generated method stub

	}

	@Override
	protected void processSave(RequestContext context, OTExtensionFormPM pageModel)
	throws BPException {
		try {
			if (pageModel.validate()) {
				if (pageModel.isNew()) {
					WorkRulesModelHandler.createOTExtension(context, pageModel.getOTExtensions().iterator().next());
				} else {
					WorkRulesModelHandler.updateOTExtensions(context, pageModel.getOTExtensions());
				}
			} else {
				setNextPage(context, "formPage");
			}
		} catch (BbmException e) {
			setNextPage(context, "formPage");
			throw e;
		} catch(RemoteException ex){
			setNextPage(context, "formPage");
			throw new BPException(ex);
		}
	}

	@Override
	protected void processDownload(RequestContext context,
			OTExtensionListPM pageModel) throws BPException {
		// TODO Auto-generated method stub
		
	}

}
