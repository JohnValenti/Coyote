	@Override
	protected void initializeModel() {
		super.initializeModel();
		if (m_isInitialized) {
			return;
		} else {
			m_isInitialized = true;
		}
		try {
			if (schedulingPeriodID != null) {
				dataContainerPC.setOnlyViewPeriodEditable(true);
			}
			if (employeeIDs.size() > 0) {
				setIsNewPerson(false);
				skill = SkillModelHandler.getSkillByID(skillID);
				if (employeeIDs.size() == 1) {
					m_empID = employeeIDs.iterator().next();
					skillAssignments = SkillsAssignmentMH.getSkillAssignmentsForEmployee(m_context,
							m_empID, null, null);
					ProfileModelHandler modelHandler = new ProfileModelHandler();
					m_employee = modelHandler.getEmpWithHistoryData(m_empID,
							dataContainerPC.getStartDate().getTime(TimeZoneUtil.GMT_TIMEZONE),
							m_context.isInWhatIfMode());
					//We only care about the selected work resource work pattern
					if (skillAssignments != null) {
						for (Iterator<SkillAssignment> iter = skillAssignments.iterator(); iter.hasNext(); ) {
							ID tempSkillID = iter.next().getSkillID();
							if (tempSkillID.equals(skillID) == false) {
								iter.remove();
							}
						}
					}
					populateHistoryRows();
				}
			}
		} catch (Exception ex) {
			handleUnableToLoadDataError(log, ex);
		} finally {
			initContentTitle();
			initToolbar();
		}
	}

	private void initContentTitle() {
		if (skill != null) {
			m_contentTitlePC.setPageTitle(getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME,
					FsWebBundleKey.FS_SKILL_EFFECTIVITY_CONTENT_TITLE, skill.getName()));
		}
		if (m_employee != null) {
			m_contentTitlePC.setItemName(m_localizer.formatName(m_employee));
		} else if (employeeIDs.size() > 1) {
			m_contentTitlePC.setItemName(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.N_PEOPLE, employeeIDs.size()));
		}
	}

	private void populateHistoryRows() {
		m_historyPC.setHeightInPixels(160);
		TableHeaderPC headerPC = m_historyPC.getHeader();
		headerPC.addColumnHeaderData("", m_localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.FROM));
		headerPC.addColumnHeaderData("", m_localizer.i18n(UIFWebBundleKey.BUNDLE_NAME, UIFWebBundleKey.TO));
		headerPC.addColumnHeaderData("", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.EMPLOYEE_SKILL_PROFICIENCY));
		headerPC.addColumnHeaderData("", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.EMPLOYEE_SKILL_PRIORITY));
		headerPC.addColumnHeaderData("", m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.EMPLOYEE_SKILL_CLASSIFICATION));

		ArrayList<MultiColumnNodeData> listModel = new ArrayList<>();

		if (skillAssignments != null) {
			for (SkillAssignment assignment : skillAssignments) {
				DefaultMultiColumnNodeData node = new DefaultMultiColumnNodeData();
				node.setSelectable(false);
				node.add(formatDate(assignment.getStartTime()));
				node.add(formatEndDate(assignment.getEndTime()));
				node.add(m_localizer.formatNumber(assignment.getProficiency(), 2));
				node.add(m_localizer.formatNumber(assignment.getPriority()));
				node.add(SkillsAssignmentUtil.getClassificationString(m_localizer, assignment.getClassification()));
				listModel.add(node);
			}
		}
		m_historyPC.setListModel(listModel);
	}

					priorityPC.getHtmlDisplay());

					classificationDDPC.getHtmlDisplay());
		}

		public boolean extractParameters(HttpServletRequest request) {
			boolean success = super.extractParameters(request);

			String radioOptionStr = getParameter(request,
					SkillsInlineEditListPC.getPopupRadioOptionFN(parentTableName, skillID));
			if (radioOptionStr != null && radioOptionStr.length() > 0) {
				int radioOption = Integer.parseInt(radioOptionStr);
				switch (radioOption) {
					case 0:
						this.setUpdateMode(DataContainerPC.UPDATE_MODE_1);
						break;
					case 1:
						this.setUpdateMode(DataContainerPC.UPDATE_MODE_2);
						break;
					case 2:
						this.setUpdateMode(DataContainerPC.UPDATE_MODE_3);
						break;
					default:
						this.setUpdateMode(DataContainerPC.UPDATE_MODE_1);
						break;
				}
			}
			this.startDate = new LocalDate(currentPeriodStartDate, TimeZoneUtil.GMT_TIMEZONE);
			if (currentPeriodEndDate != null) {
				this.endDate = new LocalDate(currentPeriodEndDate, TimeZoneUtil.GMT_TIMEZONE);
			}
			//When in campaign mode,  we need to force the end date to match the end date
			// of the scheduling period.
			if (schedulingPeriodID != null) {
				this.endDate = new LocalDate(viewEndDate, TimeZoneUtil.GMT_TIMEZONE);
			}

			LocalDate editedEndDate =
					extractDate(request, SkillsInlineEditListPC.getEditedEndDateFN(parentTableName, skillID));
			if (editedEndDate != null) {
				m_endPeriodPC.setDate(editedEndDate);
			} else {
				m_endPeriodPC.setDate(viewEndDate);
			}
			//We need to explicitly set the timezone to GMT (default it pulls the time zone from the user context)
			//  because the end date getting passed in is in GMT
			m_endPeriodPC.setTimeZone(TimeZoneUtil.GMT_TIMEZONE);

			if (employeeIDs.size() > 1) {
				m_isMultiEdit = true;
				dataContainerPC.setIsMultiEdit(true);
			}

			if (isProficiencyVisible) {
				String proficiencyStr = getParameter(request, proficiencyFN);
				if (!StringUtil.isEmpty(proficiencyStr)) {
					if (proficiencyStr.equals(MULTI_VALUE_DATA)) {
						proficiencyPC.setInputValue(MULTI_VALUE_DATA);
					} else {
						try {
							NumberFormat numberFormat = m_localizer.getNumberFormatter(1);
							double skillProficiency = numberFormat.parse(proficiencyStr).doubleValue();
							proficiencyPC.setInputValue(m_localizer.formatNumber(skillProficiency, 1));
							proficiencyPC.setIsReadOnly(!isProficiencyEditable);
						} catch (ParseException ex) {
							handleExtractParamException("SkillsEffectivityPopupPM.extractParameters()", "proficiencyFN", proficiencyStr, ex);
						}
					}
				}
			}

			String priorityStr = getParameter(request, priorityFN);
			if (!StringUtil.isEmpty(priorityStr)) {
				if (priorityStr.equals(MULTI_VALUE_DATA)) {
					priorityPC.setInputValue(MULTI_VALUE_DATA);
				} else {
					priorityPC.setInputValue(m_localizer.formatNumber(Integer.parseInt(priorityStr)));
				}
			}

			String preferenceStr = getParameter(request, classificationDDFN);
			if (!StringUtil.isEmpty(preferenceStr)) {
				if (preferenceStr.equals(MULTI_VALUE_DATA)) {
					classificationDDPC.setMultiEdit(true);
				} else {
					classificationDDPC.setSelectedIDs(Collections.singleton(preferenceStr));
				}
			}

			return success;
		}
	}

	private String getCheckboxCheckedJS() {
		StringBuilder js = new StringBuilder();

		if (isProficiencyVisible) {
			js.append("var proficiencyInputElement = jQuery('input[name=proficiencyPC]');");
			js.append("proficiencyInputElement.removeClass('" + FormInputPC.INPUT_TEXT_READ_ONLY_STYLE + "');");
			js.append("proficiencyInputElement.addClass('" + FormInputPC.DEFAULT_STYLE + "');");
			js.append("proficiencyInputElement.removeAttr('readOnly');");
		}

		js.append("var priorityInputElement = jQuery('input[name=priorityPC]');");
		js.append("priorityInputElement.removeClass('" + FormInputPC.INPUT_TEXT_READ_ONLY_STYLE + "');");
		js.append("priorityInputElement.addClass('" + FormInputPC.DEFAULT_STYLE + "');");
		js.append("priorityInputElement.removeAttr('readOnly');");
		js.append(classificationDDPC.getName() + ".setEnabled(true);");
		return js.toString();
	}

	private String getCheckboxUncheckedJS() {
		StringBuilder js = new StringBuilder();

		if (isProficiencyVisible) {
			js.append("var proficiencyInputElement = jQuery('input[name=proficiencyPC]');");
			js.append("proficiencyInputElement.removeClass('" + FormInputPC.DEFAULT_STYLE + "');");
			js.append("proficiencyInputElement.addClass('" + FormInputPC.INPUT_TEXT_READ_ONLY_STYLE + "');");
			js.append("proficiencyInputElement.attr('readOnly', 'true');");
		}

		js.append("var priorityInputElement = jQuery('input[name=priorityPC]');");
		js.append("priorityInputElement.removeClass('" + FormInputPC.DEFAULT_STYLE + "');");
		js.append("priorityInputElement.addClass('" + FormInputPC.INPUT_TEXT_READ_ONLY_STYLE + "');");
		js.append("priorityInputElement.attr('readOnly', 'true');");
		js.append(classificationDDPC.getName() + ".setEnabled(false);");
		return js.toString();
	}
