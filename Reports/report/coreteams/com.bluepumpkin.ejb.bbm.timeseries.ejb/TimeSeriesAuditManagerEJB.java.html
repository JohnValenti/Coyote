package com.bluepumpkin.ejb.bbm.timeseries.ejb;

/**

 * Description:  TimeSeriesAuditManager EJB implementation

public class TimeSeriesAuditManagerEJB extends SessionEJBBase {
  /**
   *
   */
  private static final long serialVersionUID = 1L;

  private static Category m_cat = Log.initCategory(TimeSeriesAuditManagerEJB.class.getName());

  // override the base class to provide the appropriate logging category
  protected Category getCategory() {
    return m_cat;
  }

  {
    super.init(TimeSeriesAuditManagerEJB.class.getName());
  }

  private Cache m_TimeSeriesAuditCache = null;

  private boolean WhatIfMode = false;

  public void ejbCreate() {
    try {
      // First query environment to get WIF setting from DD
      Context initialContext = new InitialContext();
      Boolean WIF = (Boolean) initialContext.lookup("java:comp/env/WIF");
      if (WIF != null)
        WhatIfMode = WIF.booleanValue();
      if (!WhatIfMode && ConfigCacheUtil.isCacheEnabled(BbmManagerFactory.getDBConfigManager(), ConfigKey.TIMESERIES_CACHE_USAGE)) {
        m_TimeSeriesAuditCache = CacheFactory.getCache(JNDINames.BBM_TIMESERIESAUDIT_EJBHOME, TimeSeriesAuditManagerEJB.class.getClassLoader());
      }
    } catch (Exception e) {
      handleException("ejbCreate", e);
    }
  }

  /**
   * For backend usage, to create TimeSeries based on a TraceChunk
   *
   * @param TraceChunk
   */
  public void createTimeSeriesAudit(ID campaignID, ID mediaID, ID queueID, short areaType, Date start, Date end) throws BbmCreateException {
    if (queueID == null) {
      methodStart("createTimeSeriesAudit", campaignID, mediaID, start, end, NumberFactory.newInteger(areaType));
    } else {
      methodStart("createTimeSeriesAudit", queueID, start, end, NumberFactory.newInteger(areaType));
    }
    try {
      Date[] cache = TimeSeriesCacheUtil.getCacheWindow();
      if (TimeZoneUtil.periodsOverlap(start, end, cache[0], cache[1])) {
        TimeSeriesDAO.createAudit(campaignID, mediaID, queueID, areaType, start, end);
        if (m_TimeSeriesAuditCache != null) {
          CompoundQueueKey key = new CompoundQueueKey(campaignID, mediaID, queueID, areaType);
          if (m_TimeSeriesAuditCache.lock(key, -1)) {
            // lock the key, retrieve content, append
            ArrayList timeRangeList = (ArrayList) m_TimeSeriesAuditCache.get(key);
            boolean merged = false;
            if (timeRangeList == null) {
              timeRangeList = new ArrayList();
            } else {
              // remove old time range obj
              for (Iterator it = timeRangeList.iterator(); it.hasNext(); ) {
                Date[] tr = (Date[]) it.next();
                if (!TimeZoneUtil.periodsOverlap(tr[0], tr[1], cache[0], cache[1])) {
                  // no longer needed, free some mem
                  it.remove();
                }
              }
            }
            if (!merged)
              timeRangeList.add(new Date[]{start, end, new Date()});
            m_TimeSeriesAuditCache.put(key, timeRangeList);
            m_TimeSeriesAuditCache.unLock(key);
          }
        }
      }
    } catch (JdmoException e) {
      handleException(e);
      throw new BbmCreateException(e);
    } finally {
      methodFinish();
    }
  }

  public boolean hasChange(ID campaignID, ID mediaID, ID queueID, short areaType, Date start, Date end, Date lastCheck) throws BbmFinderException {
    if (queueID == null) {
      methodStart("hasChange", campaignID, mediaID, start, end, NumberFactory.newInteger(areaType));
    } else {
      methodStart("hasChange", queueID, start, end, lastCheck, NumberFactory.newInteger(areaType));
    }
    try {
      if (m_TimeSeriesAuditCache != null) {
        CompoundQueueKey key = new CompoundQueueKey(campaignID, mediaID, queueID, areaType);
        // lock the key, retrieve content, append
        ArrayList timeRangeList = (ArrayList) m_TimeSeriesAuditCache.get(key);
        if (timeRangeList != null) {
          if (m_cat.isDebugEnabled()) {
            m_cat.debug("In TimeSeries Audit Cache, timeRanges are: " + timeRangeList);
          }
          for (Iterator it = timeRangeList.iterator(); it.hasNext(); ) {
            Date[] tr = (Date[]) it.next();
            if (lastCheck != null && tr[2] != null && !tr[2].before(lastCheck)) {
              if (TimeZoneUtil.periodsOverlap(tr[0], tr[1], start, end)) {
                return true;
              }
            }
          }
        }
      } else {
        // now need hit DB
        return TimeSeriesDAO.hasAudit(campaignID, mediaID, queueID, start, end, lastCheck);
      }
      return false;
    } catch (JdmoException e) {
      handleException(e);
      throw new BbmFinderException(e);
    } finally {
      methodFinish();
    }
  }

}
