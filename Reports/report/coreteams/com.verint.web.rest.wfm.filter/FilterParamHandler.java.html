package com.verint.web.rest.wfm.filter;


import io.katharsis.queryspec.FilterSpec;
import io.katharsis.queryspec.QuerySpec;

import java.util.Collections;
import java.util.Optional;
import java.util.Set;

public abstract class FilterParamHandler {

	/**
	 * Useful if you are only expecting a single value for the current filter parameter.
	 */
	protected static Optional<String> getSingleFilterValue(Set<String> params) {
		String firstValue = null;
		if (params != null && !params.isEmpty()) {
			firstValue = params.iterator().next();
		}
		return Optional.ofNullable(firstValue);
	}

	public static Optional<?> getSingleFilterParamValueFromSpec(String filterParamName, QuerySpec querySpec) {
		Optional<FilterSpec> matchingFilterSpec = getFilterSpecMatchingParamValue(filterParamName, querySpec);
		return matchingFilterSpec.map(FilterSpec::getValue);
	}

	@SuppressWarnings("unchecked")
	public static Optional<Set<?>> getFilterParamMultiValueFromSpec(String filterParamName, QuerySpec querySpec) {
		Optional<?> rawFilterParamValue = getSingleFilterParamValueFromSpec(filterParamName, querySpec);

		if (rawFilterParamValue.isPresent()) {
			//Katharsis will attempt to serialize multiple values associated to a single parameter (e.g. comma delimited list of IDs)
			//as a Set.
			if (rawFilterParamValue.get() instanceof Set) {
				return Optional.of((Set) rawFilterParamValue.get());
			} else {
				return Optional.of(Collections.singleton(rawFilterParamValue.get()));
			}
		} else {
			return Optional.empty();
		}
	}

	private static Optional<FilterSpec> getFilterSpecMatchingParamValue(String filterParamName, QuerySpec querySpec) {
		return querySpec.getFilters().stream().filter(filterSpec -> filterParamName.equals(filterSpec.getAttributePath().get(0)))
			.findFirst();
	}
}
