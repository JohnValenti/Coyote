package com.verint.web.rest.wfm.filter;

import com.bluepumpkin.common.datatypes.ID;

import java.util.*;
import java.util.stream.Collectors;


public abstract class IdsFilterParamHandler extends FilterParamHandler {

	public static Optional<Collection<ID>> getFilterValue(Set<String> filterParams) {
		if (filterParams != null && !filterParams.isEmpty()) {

			return Optional.of(filterParams.stream()
					.map(id -> Integer.parseInt(id))
					.map(id -> ID.fromInt(id))
					.collect(Collectors.toSet()));
		}
		return Optional.of(Collections.emptyList());
	}
}
