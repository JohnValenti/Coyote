package com.verint.web.rest.wfm.filter;

import com.verint.web.rest.wfm.filter.FilterParamHandler;

import java.util.Optional;
import java.util.Set;

public class BooleanFilterParamHandler extends FilterParamHandler {

	public static Optional<Boolean> getFilterValue(Set<String> params) {
		Optional<String> stringValue = getSingleFilterValue(params);
		if (stringValue.isPresent()) {
			return Optional.of(Boolean.parseBoolean(stringValue.get()));
		}
		return Optional.empty();
	}
}
