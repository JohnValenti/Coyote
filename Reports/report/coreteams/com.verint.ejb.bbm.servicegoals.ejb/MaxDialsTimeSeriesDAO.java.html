package com.verint.ejb.bbm.servicegoals.ejb;

import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoPKGenerator;
import com.bluepumpkin.common.jdmo.JdmoPKGeneratorFactory;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.ejb.bbm.base.BbmUpdateException;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.verint.ejb.bbm.servicegoals.model.MaxDialsTimeSeriesDataItem;

public class MaxDialsTimeSeriesDAO extends ServiceGoalsTimeSeriesDAO<MaxDialsTimeSeriesDataItem> {

	@Override
	protected MaxDialsTimeSeriesDataItem createValueObject() {
		return new MaxDialsTimeSeriesDataItem(new Date(System.currentTimeMillis()), 0.0);
	}

	public void saveMaxDialsTimeSeries(
			List<MaxDialsTimeSeriesDataItem> timeSeries, SPQueue spQueue,
			String user) throws BbmUpdateException {
		
		StringBuilder statementBuilder = new StringBuilder();
		statementBuilder.append("IF EXISTS (SELECT [MAXDIALS] FROM [SPQUEUEGOALSERIES] WHERE [SPQUEUEID] = %s AND [DATETIME] = %s)");
		statementBuilder.append(" UPDATE [SPQUEUEGOALSERIES] SET [MAXDIALS] = %s, [MODIFIED] = %s, [MODIFIEDBY] = %s WHERE [SPQUEUEID] = %s AND [DATETIME] = %s");
		statementBuilder.append(" ELSE INSERT INTO [SPQUEUEGOALSERIES] ([ID], [SPQUEUEID], [DATETIME], [MAXDIALS], [CREATED], [CREATEDBY])");
		statementBuilder.append(" VALUES(%s, %s, %s, %s, %s, %s)");
		JdmoPKGenerator idGenerator = JdmoPKGeneratorFactory.create("SPQUEUEGOALSERIES", false);
		String statement = statementBuilder.toString();
		Jdmo jdmo = new Jdmo();
		try {
			for (MaxDialsTimeSeriesDataItem dataItem : timeSeries) {
				String query = String.format(statement, 
						JdmoUtil.asSqlLiteral(spQueue.getDEID()),
						JdmoUtil.asSqlLiteral(dataItem.getTime()),
						Double.toString(dataItem.getValue()), 
						JdmoUtil.asSqlLiteral(new Date(System.currentTimeMillis())),
						JdmoUtil.asSqlLiteral(user),
						JdmoUtil.asSqlLiteral(spQueue.getDEID()),
						JdmoUtil.asSqlLiteral(dataItem.getTime()),
						
						JdmoUtil.asSqlLiteral(idGenerator.generate(jdmo.getConnection(), "SPQUEUEGOALSERIES").toString()),
						JdmoUtil.asSqlLiteral(spQueue.getDEID()),
						JdmoUtil.asSqlLiteral(dataItem.getTime()),
						Double.toString(dataItem.getValue()), 
						JdmoUtil.asSqlLiteral(new Date(System.currentTimeMillis())),
						JdmoUtil.asSqlLiteral(user));

					jdmo.executeCommand(query);
			} 
		} catch (Exception e) {
			e.printStackTrace();
			throw new BbmUpdateException(e);
		} finally {
			jdmo.cleanUp();
		}
	}
}
