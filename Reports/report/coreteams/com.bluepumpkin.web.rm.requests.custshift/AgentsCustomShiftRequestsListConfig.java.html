package com.bluepumpkin.web.rm.requests.custshift;

import java.util.*;
import com.witness.web.uif.l10n.*;
import com.bluepumpkin.web.rm.l10n.*;
import com.bluepumpkin.web.rm.requests.AgentsRequestsListConfig;
import com.bluepumpkin.ejb.rm.requests.common.model.Request;
import com.bluepumpkin.ejb.rm.util.LicenseUtil;

/**
 * Title:        AgentsCustomShiftRequestsListConfig
 * Description:  List Configuration for Agents Custom Shift Requests
 * Copyright:    Copyright (c) 2011
 * Company:      Verint Systems, Inc.
 * @author       Gonzalo Quintanar
 */
public class AgentsCustomShiftRequestsListConfig extends AgentsRequestsListConfig
{
	private List m_allFPs;
	private Map m_fpMap;

	/**
	 * Creates a new instance of AgentsCustomShiftRequestsListConfig
	 */
	public AgentsCustomShiftRequestsListConfig()
	{
		initFPs();
	}

	/**
	 * Initialize Field Properties
	 */
    private void initFPs()
    {
	    if (m_allFPs==null)
	    {
			m_allFPs = new ArrayList(19);

			m_allFPs.add( createFP(Request.SORT_LAST_NAME_FIRST, UIFWebBundleKey.NAME, true) );
			m_allFPs.add( createFP(Request.SORT_STATUS, UIFWebBundleKey.STATUS, true) );

			m_allFPs.add( createFP(Request.SORT_CS_SUBTYPE, UIFWebBundleKey.TYPE, true) );
			m_allFPs.add( createFP(Request.SORT_CS_CAMPAIGN, RmWebBundleKey.CAMPAIGN, RmWebBundleKey.BUNDLE_NAME, true));

			m_allFPs.add( createFP(Request.SORT_CREATED, UIFWebBundleKey.SUBMITTED, true) );
			m_allFPs.add( createFP(Request.SORT_MODIFIED, UIFWebBundleKey.LAST_MODIFIED, true) );
			m_allFPs.add( createFP(Request.SORT_TO_SENIORITY, UIFWebBundleKey.SENIORITY, false) );
			m_allFPs.add( createFP(Request.SORT_EMP_RANK, UIFWebBundleKey.RANK, false) );

			m_allFPs.add( createFP(Request.SORT_CS_STARTTIME, UIFWebBundleKey.START_DATE, true) );
			m_allFPs.add( createFP(Request.SORT_CS_ENDTIME, UIFWebBundleKey.END_DATE, true) );

			m_allFPs.add( createFP(Request.SORT_CS_SHIFTID, RmWebBundleKey.RM_SA_FILTER_SHIFT, RmWebBundleKey.BUNDLE_NAME, true));
			m_allFPs.add( createFP(Request.SORT_CS_ACTIVITYID, RmWebBundleKey.RM_SA_FILTER_ACTIVITY, RmWebBundleKey.BUNDLE_NAME, true));
			m_allFPs.add( createFP(Request.SORT_CS_ISOT, RmWebBundleKey.ORG_RM_FILING_RULE_OVERTIME, RmWebBundleKey.BUNDLE_NAME, true));

			m_allFPs.add( createFP(Request.SORT_CS_EXTBEFOREID, RmWebBundleKey.CSR_EXTENSION_BEFORE_SHIFT_ABBREVIATED, RmWebBundleKey.BUNDLE_NAME, true));
			m_allFPs.add( createFP(Request.SORT_CS_EXTBEFOREACTIVITYID, RmWebBundleKey.CSR_EXTENSION_BEFORE_SHIFT_ACTIVITY_ABBREVIATED, RmWebBundleKey.BUNDLE_NAME, true));
			m_allFPs.add( createFP(Request.SORT_CS_EXTBEFOREISOT, RmWebBundleKey.CSR_EXTENSION_BEFORE_SHIFT_OVERTIME_ABBREVIATED, RmWebBundleKey.BUNDLE_NAME, true));
			if (LicenseUtil.isAdvancedRMLicense()) {
				m_allFPs.add( createFP(Request.SORT_CS_EXTINGAPSID, RmWebBundleKey.CSR_EXTENSION_IN_GAPS_ABBREVIATED, RmWebBundleKey.BUNDLE_NAME, true));
				m_allFPs.add( createFP(Request.SORT_CS_EXTINGAPSACTIVITYID, RmWebBundleKey.CSR_EXTENSION_IN_GAPS_ACTIVITY_ABBREVIATED, RmWebBundleKey.BUNDLE_NAME, true));
			}
			m_allFPs.add( createFP(Request.SORT_CS_EXTAFTERID, RmWebBundleKey.CSR_EXTENSION_AFTER_SHIFT_ABBREVIATED, RmWebBundleKey.BUNDLE_NAME, true));
			m_allFPs.add( createFP(Request.SORT_CS_EXTAFTERACTIVITYID, RmWebBundleKey.CSR_EXTENSION_AFTER_SHIFT_ACTIVITY_ABBREVIATED, RmWebBundleKey.BUNDLE_NAME, true));
			m_allFPs.add( createFP(Request.SORT_CS_EXTAFTERISOT, RmWebBundleKey.CSR_EXTENSION_AFTER_SHIFT_OVERTIME_ABBREVIATED, RmWebBundleKey.BUNDLE_NAME, true));

			addCommonFPs(m_allFPs);
	    }

    	if (m_fpMap==null) m_fpMap = createFPMap(m_allFPs);
    }

	/**
	 * Return All List Config Field Properties (ListConfigFP)
	 */
	public List getAllFPs()
	{
		return m_allFPs;
	}

	/**
 	 * Return a Map of All List Config Field Properties (ListConfigFP)
 	 * where key is ListConfigFP.getFieldName();
	 */
	public Map getFPMap()
	{
	    return m_fpMap;
	}

}
