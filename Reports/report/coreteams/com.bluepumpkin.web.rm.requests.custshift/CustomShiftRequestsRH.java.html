/**
 * Title:		 CustomShiftRequestsRH
 * Description:  Custom Shift Request Request Handler
 * Copyright:    Copyright (c) 2011
 * Company:      Verint Systems, Inc.
 * @author       Gonzalo Quintanar
 */
public class CustomShiftRequestsRH extends RequestsRH {
	private static Category m_log = Log.initCategory(CustomShiftRequestsRH.class.getName()); 

	/**
	 * Process Request for the User page
	 *
	 * @param context The Request Context
	 */  
	@Override
	public void processCustomRequest(RequestContext context) throws WebException 
	{
		// Process requested action
		if (isAction(context, PageAction.SAVE)) 
		{
			processSave(context);
		} 
		else 
		{
			super.processCustomRequest(context);
		}
	}

	/**
	 * Process Save
	 */
	public void processSave(RequestContext context) 
	{
		CustomShiftRequestFormPopupPM model = new CustomShiftRequestFormPopupPM(context);

		try 
		{				
			// Extract Form Parameter
			model.extractParameters(context.getRequest());
			
			model.setReloadAction(CustomShiftRequestFormPopupPM.RELOAD_ACTION_SAVE);
			
			//we must initialize the model because some form element values are not submitted when they are 
			//disabled, so extractParameters() is insufficient. This will ensure the values are properly set.
			model.initializeModelForSaving();
	
			if (model.validate()) 
			{
				// Security check
				CustShiftReq request = model.getCustShiftRequestForSaving();
				if (request != null)
				{
					ID requestID = request.getID();
					
					if (!RequestUtil.isAuthToUpdate(context, request)) {
						return;
					}
					
					if (requestID == null)
					{
						requestID = CustomShiftRequestsMH.createCSRequest(request, model.getComment());
					} 
					else 
					{
						CustomShiftRequestsMH.updateCSRequest(request, model.getComment());
						model.setOpenerValue(AgentsRequestsPM.IS_ON_SAME_PAGE_FN, "true");
					}
					
					model.setSavedRequestID(requestID);
				}
			}
		} 
		catch(MultiUserException ex) 
		{
			model.addPageMessage(Message.WARNING_TYPE,
					RmWebBundleKey.UNABLE_TO_UPDATE_REQUEST_DUE_TO_MULTIUSER_EXCEPTION,
					RmWebBundleKey.BUNDLE_NAME);
		} 
		catch(RmHardValidationException ex) 
		{
			m_log.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
			Message msg = new Message(Message.ERROR_TYPE,	RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
			msg.setAssociatedMessage( getLocMsgFromException(context, ex) );
			model.addPageMessage(msg);
		} 
		 catch(BbmUpdateException ex) { 
			m_log.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
			Message msg = new Message(Message.ERROR_TYPE,	RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
			msg.setAssociatedMessage( ex.getLocalizedMessage(context.getLocaleContext().getLanguageLocale()));
			model.addPageMessage(msg);			
		}
		 catch(BbmCreateException ex) {
				m_log.l7dDebug(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
				Message msg = new Message(Message.ERROR_TYPE,	RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
				msg.setAssociatedMessage( ex.getLocalizedMessage(context.getLocaleContext().getLanguageLocale()));
				model.addPageMessage(msg);			
		}
		catch(Exception ex) 
		{
			m_log.l7dError(RmWebLogBundleKey.UNABLE_TO_SAVE_REQUEST, ex);
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.UNABLE_TO_SAVE_REQUEST, RmWebBundleKey.BUNDLE_NAME);
		}

		context.setPageModel(model);
	}

}
