package com.bluepumpkin.ejb.facade.hosted.billing.ejb;

import java.rmi.RemoteException;
import java.util.Date;
import java.util.TimeZone;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.core.base.SessionEJBBase;
import com.bluepumpkin.ejb.bbm.BbmManagerFactory;
import com.bluepumpkin.ejb.bbm.workresource.ejb.WorkResourceManager;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.facade.Log;
import com.bluepumpkin.ejb.facade.base.FacadeException;

public class DailyBillingUsageManagerEJB
		extends SessionEJBBase {
	private static Category m_cat = Log.initCategory(DailyBillingUsageManagerEJB.class.getName());

	// override the base class to provide the appropriate logging category
	@Override
	protected Category getCategory() {
		return m_cat;
	}

	public void dumpDailyUsage(Date startTime, Date endTime)
			throws FacadeException, RemoteException {
		DailyBillingUsageDAO dao = new DailyBillingUsageDAO();
		try {
			WorkResourceManager wrMgr = BbmManagerFactory.getWorkResourceManager();
			Organization rootOrg = wrMgr.getOrganizationByID(Organization.YOUR_COMPANY_ID_OBJ);
			TimeZone rootTz = rootOrg.getTimeZone();

			dao.dumpDailyUsage(startTime, endTime, rootTz);
		} catch (Exception ex) {
			this.handleException(ex);
			throw new FacadeException(ex);
		} finally {
			dao.cleanUp();
		}
	}
}
