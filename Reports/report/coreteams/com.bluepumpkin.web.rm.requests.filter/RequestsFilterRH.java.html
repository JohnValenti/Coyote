package com.bluepumpkin.web.rm.requests.filter;

import java.util.Collection;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.IDStringPair;
import com.bluepumpkin.common.datatypes.StringsPair;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.rm.requests.commonrequestmanager.model.RequestFilter;
import com.bluepumpkin.web.rm.Log;
import com.bluepumpkin.web.rm.keys.RmPageAction;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;
import com.bluepumpkin.web.rm.l10n.RmWebLogBundleKey;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.base.RequestHandler;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.system.RequestContext;

/**
 * Title: RequestsFilterRH Description: Requests Filter Request Handler Copyright: Copyright (c) 2001 Company: Blue Pumpkin
 * Software, inc
 *
 * @author David Su
 * @version 1.0
 */
public class RequestsFilterRH extends RequestHandler {
	private static Category m_log = Log.initCategory(RequestsFilterRH.class.getName());

	/**
	 * Process Request for the User page
	 *
	 * @param context The Request Context
	 */
	@Override
	public void processRequest(RequestContext context) {
		RequestsFilterPopupPM model = new RequestsFilterPopupPM(context);
		model.extractParameters(context.getRequest());

		if (isAction(context, PageAction.SAVE_AS)) {
			if (model.validate()) {
				processSaveAs(context, model);
			} else {
				model.setHasError(true);
			}
		} else if (isAction(context, PageAction.SAVE)) {
			if (model.validate()) {
				processSave(context, model);
			} else {
				model.setHasError(true);
			}
		} else if (isAction(context, PageAction.DELETE)) {
			processDelete(context, model);
		} else if (isAction(context, RmPageAction.REQUEST_TYPE_CHANGE_ACTION)) {
			processRequestTypeChange(context, model);
		} else {
			initModel(context, model);
		}

		// Handle Model with Error
		if (model.hasError()) {
			// Need to load support Data to redisplay page properly
			model.setIsCloseAfterLoad(false);
			loadSupportData(context, model);
		}

		context.setPageModel(model);
	}

	/**
	 * Process Save As
	 */
	public void processSaveAs(RequestContext context, RequestsFilterPopupPM model) {
		try {
			String saveAsName = model.getSaveAsName();
			RequestFilter filter = model.getRequestFilter();
			filter.setName(saveAsName);
			ID rID = RequestsFilterMH.createRequestFilter(filter);
			model.setIsRefreshOpenerOnClose(true);
			//Use the new variable isCloseAfterLoadCustomize instead of using parent's variable isCloseAfterLoad
			//to handle for Chrome
			model.setCloseAfterLoadCustomize(true);
			model.setSelectedConfigDisplay(saveAsName);
			model.setSelectedConfig(rID.toString());
		} catch (Exception ex) {
			handleException(model, ex, RmWebBundleKey.REQUEST_FILTER_SAVE_ERROR_MSG,
					RmWebLogBundleKey.REQUEST_FILTER_SAVE_ERROR_MSG);
		}
	}

	/**
	 * Process Save
	 */
	public void processSave(RequestContext context, RequestsFilterPopupPM model) {
		try {
			String filterName = model.getFilterName();
			RequestFilter filter = model.getRequestFilter();
			filter.setName(filterName);

			ID filterID = model.getSelectedFilterID();
			filter.setId(filterID);
			RequestsFilterMH.updateRequestFilter(filter);
			model.setIsRefreshOpenerOnClose(true);
			//Use the new variable isCloseAfterLoadCustomize instead of using parent's variable isCloseAfterLoad
			//to handle for Chrome
			model.setCloseAfterLoadCustomize(true);
		} catch (Exception ex) {
			handleException(model, ex, RmWebBundleKey.REQUEST_FILTER_SAVE_ERROR_MSG,
					RmWebLogBundleKey.REQUEST_FILTER_SAVE_ERROR_MSG);
		}
	}

	/**
	 * Process Delete
	 */
	public void processDelete(RequestContext context, RequestsFilterPopupPM model) {
		try {
			ID filterID = model.getSelectedFilterID();
			RequestsFilterMH.deleteRequestFilter(filterID);
			model.setIsCloseAfterLoad(true);
			model.setSelectedConfig(model.getDefaultConfig());
		} catch (Exception ex) {
			handleException(model, ex, RmWebBundleKey.REQUEST_FILTER_DELETE_ERROR_MSG,
					RmWebLogBundleKey.REQUEST_FILTER_DELETE_ERROR_MSG);
		}
	}

	/**
	 * Process Request Type Change
	 */
	public void processRequestTypeChange(RequestContext context, RequestsFilterPopupPM model) {
		// Load Support Data using new Request Type
		loadSupportData(context, model);

		// Then Load Request Filter
		loadRequestFilter(model);

		// Need to retain new user options which overwrites above retrieved info
		model.extractParameters(context.getRequest());
	}

	/**
	 * Load Support Date To Model
	 */
	private void loadSupportData(RequestContext context, RequestsFilterPopupPM model) {
		try {
			String reqType = model.getRequestType();
			Map dataMap = RequestsFilterMH.getRequestFilterData(context, model.getUser(), reqType);

			Collection orgList = (Collection) dataMap.get(RequestsFilterMH.ORGANIZATION_KEY);
			IDStringPair[] filterNames = (IDStringPair[]) dataMap.get(RequestsFilterMH.FILTER_NAME_KEY);
			Collection supervisorList = (Collection) dataMap.get(RequestsFilterMH.SUPERVISOR_KEY);
			List reqStatusList = (List) dataMap.get(RequestsFilterMH.REQUEST_STATUS_KEY);
			Collection skillList = (Collection) dataMap.get(RequestsFilterMH.SKILL_KEY);
			Collection poolList = (Collection) dataMap.get(RequestsFilterMH.POOL_KEY);
			Collection bidList = (Collection) dataMap.get(RequestsFilterMH.BID_KEY);
			Collection campaignList = (Collection) dataMap.get(RequestsFilterMH.CAMPAIGN_KEY);
			HashMap<ID, String> shiftMap = (HashMap<ID, String>) dataMap.get(RequestsFilterMH.SHIFT_KEY);
			Collection activityList = (Collection) dataMap.get(RequestsFilterMH.ACTIVITY_KEY);
			HashMap<ID, String> extensionMap = (HashMap<ID, String>) dataMap.get(RequestsFilterMH.EXTENSION_KEY);
			Collection<Activity> timeoffActivitiesList = (Collection<Activity>) dataMap.get(RequestsFilterMH.TO_ACTIVITIES_KEY);
			List<StringsPair> shiftBidAuctionList = (List<StringsPair>) dataMap.get(RequestsFilterMH.SHIFT_BID_AUCTION_KEY);

			model.setSupportData(filterNames, orgList, supervisorList, reqStatusList, skillList, poolList, bidList, campaignList,
					shiftMap, activityList, extensionMap,timeoffActivitiesList,shiftBidAuctionList);
		} catch (Exception ex) {
			model.handleUnableToLoadDataError(m_log, ex);
		}
	}

	/**
	 * Return Request Filter
	 */
	private void loadRequestFilter(RequestsFilterPopupPM model) {
		try {
			ID filterID = model.getSelectedFilterID();
			if (filterID != null) {
				RequestFilter rFilter = RequestsFilterMH.getRequestFilter(filterID);
				if (rFilter != null) {
					model.setRequestFilter(rFilter);
				}
			} else {
				model.resetRequestFilter();
			}
		} catch (Exception ex) {
			model.handleUnableToLoadDataError(m_log, ex);
		}
	}

	/**
	 * Initialize Model
	 */
	private void initModel(RequestContext context, RequestsFilterPopupPM model) {
		loadRequestFilter(model);
		loadSupportData(context, model);
	}

	/**
	 * Handle Exception
	 */
	private void handleException(RequestsFilterPopupPM model, Exception ex, String rbKey, String logRBKey) {
		model.addPageMessage(Message.ERROR_TYPE, rbKey, RmWebBundleKey.BUNDLE_NAME);
		m_log.l7dError(logRBKey, ex);
	}
}
