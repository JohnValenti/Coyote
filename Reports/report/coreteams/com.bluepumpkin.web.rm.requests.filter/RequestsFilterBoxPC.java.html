package com.bluepumpkin.web.rm.requests.filter;

import java.util.*;

import com.bluepumpkin.common.util.*;
import com.bluepumpkin.common.datatypes.*;
import com.bluepumpkin.ejb.rm.requests.common.ejb.RequestUtil;
import com.witness.web.uif.system.*;
import com.witness.web.uif.keys.*;
import com.witness.web.uif.l10n.UIFWebBundleKey;
import com.witness.web.uif.pagecomponent.list.OptionData;
import com.witness.web.uif.pagecomponent.filterbox.FilterBoxPC;
import com.bluepumpkin.web.rm.keys.RmKeys;
import com.bluepumpkin.web.rm.l10n.RmWebBundleKey;

/**
 * Title:        AgentsRequestsFilterBoxPC
 * Description:  This generate the Agents Requests Filter Option UI
 * Copyright:    Copyright (c) 2002
 * Company:      Blue Pumpkin Software, Inc
 * @author       David Su
 * @version      1.0
 */
public class RequestsFilterBoxPC extends FilterBoxPC {
	private static final String FILTER_POPUP_SIZE = "700,600,ctr,ctr";
	private static final String FILTER_RH   = "request_filter_config";
	private static final String FILTER_NAME = "agentsRequestsFilter";
	private static final String EDIT_FILTER_ACTION = "POPUP" + UIFWebBundleKey.FILTER_EDIT;
	private static final String CREATE_FILTER_ACTION = "POPUP" + UIFWebBundleKey.FILTER_CREATE;

	private IDStringPair[] m_fNames;

	/**
	 * Constructor
	 */
	public RequestsFilterBoxPC(RequestContext context) {
		super(context);

		setName(FILTER_NAME);
		setUseWrapper(false);
	}

	/**
	 * Initialize Component For Display
	 */
	public void initForDisplay() {
		if (m_isInitializedForDisplay) return;
		else {   		
			super.initForDisplay();

			addPopupArgs(EDIT_FILTER_ACTION, FILTER_RH, getPopupArgs(false),
					null, FILTER_POPUP_SIZE, true, 1);
			addPopupArgs(CREATE_FILTER_ACTION, FILTER_RH, getPopupArgs(true),
					null, FILTER_POPUP_SIZE, true, 1);
		}
	}

	/**
	 * Return Popup Arguments
	 */
	private String getPopupArgs(boolean isForCreateNew) {
		StringBuffer sb = new StringBuffer(100);
		sb.append(PageKeys.FORM_ACTION_FN).append("=").append(FILTER_RH)
				.append("&").append(PageKeys.SELECTOR_NAME_FN).append("=").append(getName());

		if (!isForCreateNew) {
			sb.append("&").append(PageKeys.SELECTED_CONFIG_FN)
					.append("=").append(getSelectedFilterName());
		}
		sb.append("&").append("isNewFilter").append("=").append(isForCreateNew);

		return sb.toString();
	}

	/**
	 * Set Selected Filter
	 */
	public void setSelectedFilterName(String filterName) {
		// Ignore Selection Option Refresh
		if (REFRESH.equals(filterName)) return;

		super.setSelectedFilterName(filterName);
		m_context.setUserProperty(RmKeys.REQUEST_FILTER_NAME, filterName);
	}

	/**
	 * Get Selected Filter
	 */
	public String getSelectedFilterName() {
		String filterName = super.getSelectedFilterName();
		if (filterName == null)
			filterName = m_context.getUserProperty(RmKeys.REQUEST_FILTER_NAME);

		if (filterName == null) filterName = PageKeys.DEFAULT_ALL;

		return filterName;
	}

	/**
	 * Initialize Filters
	 */
	public void initFilter(IDStringPair[] fNames, String selectedFilterName, boolean isBlankSelected) {
		m_fNames = fNames;

		// Enable Blank Filter to be selected
		if (isBlankSelected) {
			selectedFilterName = "";
			setSelectedFilterName("");
		}

		ArrayList filters = RequestsFilterUtil.createFilterOptions(fNames, selectedFilterName);
		String sAllDisplay = i18n(m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME), RmWebBundleKey.REQUEST_FILTER_ALL_ACTIVE);
		
		OptionData data1 = new OptionData(sAllDisplay, PageKeys.DEFAULT_ALL);
		if (selectedFilterName.equals(PageKeys.DEFAULT_ALL))
			data1.setSelected(true);
		filters.add(0, data1 );
		
		OptionData data2 = new OptionData(i18n(m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME), RmWebBundleKey.REQUEST_TYPE_NO_FILTER), 
				RequestUtil.ALL_FILTER);
		if (selectedFilterName.equals(RequestUtil.ALL_FILTER))
			data2.setSelected(true);
		filters.add(1, data2 );
	
		if (isBlankSelected) filters.add(0, new OptionData("", PageKeys.DEFAULT_ALL) );

		if (m_isActionOptionsEanbled) addDefaultFilterOptions(filters);
		setFilters(filters);
	}

	/**
	 * Return Selected Filter ID
	 */
	public ID getSelectedFilterID() {
		ID result=null;

		String selectedID = getSelectedFilterName();
		if (m_fNames!=null && m_fNames.length>0) {
			for(int i=0; i<m_fNames.length; i++) {
				IDStringPair idStr = m_fNames[i];
				ID fID = idStr.getID();
				if (fID.toString().equals(selectedID)) {
					result = fID;
					break;
				}
			}
		}	else if (!RequestUtil.ALL_FILTER.equals(selectedID) &&
				!PageKeys.DEFAULT_ALL.equals(selectedID) && !StringUtil.isEmpty(selectedID)) {
			return new ID(selectedID);
		}

		return result;
	}

	/**
	 * Return Html Display
	 */
	public String getHtmlDisplay() {
		StringBuffer buff = new StringBuffer(1024);
		buff.append("\n<div style=\"margin-top:5px;\">");
		buff.append(super.getHtmlDisplay());
		buff.append("\n</div>");
		return buff.toString();
	}
	
	public String getFilterRegionAriaLabel() {
		return i18n(m_localizer.getBundle(RmWebBundleKey.BUNDLE_NAME), RmWebBundleKey.RM_MYREQUESTS_FILTER);
	}
}
