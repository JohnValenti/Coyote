package com.bluepumpkin.ejb.bbm.timerecord.ejb;


import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.HashMap;
import java.util.Iterator;

import com.bluepumpkin.ejb.bbm.timerecord.model.TimeTrackingEvent;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.common.datatypes.ID;

/**
 * TimeRecordDAO is not to be accessed by other objects directly
 * Use default package level restriction
 */
public class TimeTrackingDAO 
{
	private static String strTableName = "TIMETRACKINGEVENT";
	private static String vstrTableName = "V_TIMETRACKINGEVENT";

	public HashMap getEventsForWorkResource(Collection workResourceID,
			Date dtStart,
			Date dtEnd ) throws JdmoException
	{
		HashMap retVal = new HashMap();
		if( workResourceID == null || workResourceID.size() <= 0 )
			return retVal;
		
		Iterator iter = workResourceID.iterator();
		ArrayList idList = new ArrayList();
		int chunkSize = 250;

		long count = 0;
		while( iter.hasNext() )
		{
			idList.add( iter.next() );
			++count; 
			
			if( idList.size() >= chunkSize || count >= workResourceID.size() )
			{
				retVal.putAll( this.getEventsForWorkResourceInChunk(idList, dtStart, dtEnd) );
				idList.clear();
			}
		}
		
		return retVal;
	}

	
	private HashMap getEventsForWorkResourceInChunk(Collection workResourceID,
													Date dtStart,
													Date dtEnd ) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();

		try
		{
			StringBuffer sql = new StringBuffer();
			sql.append( "SELECT ID, EMPLOYEEID, DATASOURCEID, ACTIVITYID, STARTTIME, ENDTIME FROM ");
			sql.append( vstrTableName );
			sql.append( " WHERE " );
			
			if( workResourceID != null && workResourceID.size() > 0 )
			{
				String empInClause = JdmoUtil.createIDList(workResourceID);
				sql.append(" EMPLOYEEID IN (");
				sql.append(empInClause);
				sql.append(") AND ");
			}

			
			sql.append(" STARTTIME <= '");
			sql.append( JdmoUtil.formatDBString(TimeZoneUtil.toTimestamp(dtEnd)) );
			sql.append("' AND (ENDTIME > '");
			sql.append( JdmoUtil.formatDBString(TimeZoneUtil.toTimestamp(dtStart)) );
			sql.append("'" );
			sql.append(" OR ENDTIME IS NULL) ");
			
			HashMap retVal = new HashMap();

			JdmoQuery jq1 = jdmo.createQuery(sql.toString(), Jdmo.PARAM_QUERY);
			JdmoRowset rs = jdmo.createRowset(jq1, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

			while( rs.next() ) 
			{
				TimeTrackingEvent obj = new TimeTrackingEvent();

				obj.setID( new ID( rs.getInt("ID")) );
				obj.setEmployeeID( new ID( rs.getInt("EMPLOYEEID")) );
				obj.setDataSourceID( new ID( rs.getInt("DATASOURCEID")) );
				obj.setActivityID( new ID( rs.getInt("ACTIVITYID")) );
				obj.setStartTime( TimeZoneUtil.toDate(rs.getTimestamp("STARTTIME")) );
				obj.setEndTime( TimeZoneUtil.toDate(rs.getTimestamp("ENDTIME")) );
				
				ID empID = obj.getEmployeeID();
				
				Collection empColl = (Collection)retVal.get(empID);
				if( empColl == null )
				{
					empColl = new ArrayList();
					retVal.put( empID, empColl);
				}
				
				empColl.add(obj);
			}
			
			return retVal;
		}
		finally
		{
			jdmo.cleanUp();
		}

	}
    

	public void insertEvents(Collection events ) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();

		try
		{
			if( events == null || events.size() <= 0 )
				return;
			
			Iterator iter = events.iterator();
			while( iter.hasNext() )
			{
				TimeTrackingEvent obj = (TimeTrackingEvent)iter.next();

				HashMap map = new HashMap();
				
				map.put("EMPLOYEEID", obj.getEmployeeID());
				map.put("DATASOURCEID", obj.getDataSourceID());
				map.put("ACTIVITYID", obj.getActivityID());
				map.put("STARTTIME", obj.getStartTime());

				jdmo.addBatchInsert(strTableName, map);
			}
			
			jdmo.executeBatch();
			
		}
		finally
		{
			jdmo.cleanUp();
		}
	}

}
