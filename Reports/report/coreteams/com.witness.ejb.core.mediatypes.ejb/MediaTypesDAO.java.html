package com.witness.ejb.core.mediatypes.ejb;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.Iterator;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.witness.ejb.core.base.CoreFinderException;
import com.witness.ejb.core.base.CoreUpdateException;
import com.witness.ejb.core.mediatypes.model.MediaType;

public class MediaTypesDAO {
	private Jdmo m_dmo = null;

	public MediaTypesDAO(Jdmo dmo) {
		m_dmo = dmo;
	}

	public Collection getObjects() throws CoreFinderException {
		ArrayList list = new ArrayList();
		try {
			JdmoRowset rs = m_dmo.createRowset("select id, sid, name, timeunit from media");
			MediaType media;
			while (rs.next()) {
				media = new MediaType(rs.getString(1), rs.getID(2), rs.getString(3), rs.getInt(4));
				list.add(media);
			}
		} catch (Exception ex) {
			throw new CoreFinderException(ex);
		}
		return list;
	}

	public MediaType getObjectByID(ID id) throws CoreFinderException {
		try {
			JdmoRowset rs = m_dmo.createRowset("select id, sid, name, timeunit from media where id = " +
					JdmoUtil.asSqlLiteral(id));
			while (rs.next()) {
				return new MediaType(rs.getString(1), rs.getID(2), rs.getString(3), rs.getInt(4));
			}
		} catch (Exception ex) {
			throw new CoreFinderException(ex);
		}
		return null;
	}

	public void updateTimeUnit(HashMap idTimeUnitMap) throws CoreUpdateException {
		try {
			Map.Entry entry = null;
			for (Iterator i = idTimeUnitMap.entrySet().iterator(); i.hasNext();) {
				entry = (Map.Entry) i.next();
				m_dmo.execute("update media set timeunit = " +
						JdmoUtil.asSqlLiteral(entry.getValue()) + " where id = " +
						JdmoUtil.asSqlLiteral(entry.getKey()));
			}
		} catch (Exception ex) {
			throw new CoreUpdateException(ex);
		}
	}

	public void updateTimeUnitBYSID(HashMap idTimeUnitMap) throws CoreUpdateException {
		try {
			Map.Entry entry = null;
			for (Iterator i = idTimeUnitMap.entrySet().iterator(); i.hasNext();) {
				entry = (Map.Entry) i.next();
				m_dmo.execute("update media set timeunit = " +
						JdmoUtil.asSqlLiteral(entry.getValue()) + " where sid = " +
						JdmoUtil.asSqlLiteral(entry.getKey()));
			}
		} catch (Exception ex) {
			throw new CoreUpdateException(ex);
		}
	}

}
