package com.verint.web.fs.widgets;

import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.core.pagecomponent.DropDownSelectionPC;
import com.verint.web.fs.workrules.WorkRulesRH;
import com.witness.web.uif.keys.PageAction;
import com.witness.web.uif.pagecomponent.toolbar.Pagination;
import com.witness.web.uif.pagecomponent.toolbar.ToolbarPC;
import com.witness.web.uif.system.RequestContext;

import javax.servlet.http.HttpServletRequest;

import static com.witness.web.uif.base.RequestHandler.getPageAction;

public class JumpToPagePaginationPC extends Pagination {
	private static final long serialVersionUID = 1L;

	private DropDownSelectionPC<Integer> jumpToPageDDPC;
	public static int USED_IN_NORMAL_PAGE = 0; // No persist the page index. otherwise, the page index is persisted at specific pages as customer request in 11.1SP1
	public static int USED_IN_SHIFT_LIST_PAGE = 1; // the page index is persisted in Shift List page
	public static int USED_IN_SHIFTEVENT_LIST_PAGE = 2; // the page index is persisted in Shift Event List page
	public static int USED_IN_WORKPATTERN_LIST_PAGE = 3; // the page index is persisted in Work Pattern List page
	public static int USED_IN_ROTATION_LIST_PAGE = 4; // the page index is persisted in Rotations List page

	public static final String SHIFTLIST_ITEMINDEX = "SHIFTLIST_ITEMINDEX";
	public static final String SHIFTEVENTLIST_ITEMINDEX = "SHIFTEVENTLIST_ITEMINDEX";
	public static final String WORKPATTERNLIST_ITEMINDEX = "WORKPATTERNLIST_ITEMINDEX";
	public static final String ROTATIONLIST_ITEMINDEX = "ROTATIONLIST_ITEMINDEX";
	private int m_usedInPage = USED_IN_NORMAL_PAGE;

	public JumpToPagePaginationPC(ToolbarPC toolbar, Localizer localizer) {
		super(toolbar, "", localizer);

		jumpToPageDDPC = new DropDownSelectionPC<Integer>(m_toolbar.getRequestContext(), "jumpToPageDDPC");
		jumpToPageDDPC.setOnChangeJS("if (typeof(" + m_toolbar.getName() +
				")!='undefined') " + m_toolbar.getName() + ".onSelect('" + PageAction.SELECT_PAGE + "', this)");
	}

	public JumpToPagePaginationPC(ToolbarPC toolbar, Localizer localizer, int usedInPage) {
		this(toolbar, localizer);
		m_usedInPage = usedInPage;
	}

	/**
	 * Get string of the HTML representing this component.
	 * this can live only inside <tr>...</tr> tags
	 */
	public String getHtmlDisplay() {
		if (m_nItemTotal <= 1) {
			return "";
		}

		StringBuffer sb = new StringBuffer(500);

		// Text Label
		if (m_isTextLabelEnabled) {
			sb.append("<td class=\"tbarLbl\" nowrap>").append(getTextLabel()).append("</td>");
		}

		// Left Arrow
		sb.append("<td>").append(getLeftArrow().getHtmlDisplay()).append("</td>");

		// Jump to Page Control
		sb.append("<td>").append(getJumpToPageControl()).append("</td>");

		// Right Arrow
		sb.append("<td>").append(getRightArrow().getHtmlDisplay()).append("</td>");

		// Add Increment Control
		if (m_isIncrementControlEnabled) {
			sb.append("<td>").append(getIncrementControl()).append("</td>");
		}

		return sb.toString();
	}

	public void setProperties(int itemIndex, int itemIncrement, int itemTotal) {
		String pageAction = getPageAction(m_context);
		int newItemIndex = getPersistedItemIndex(itemIndex, pageAction);
		super.setProperties(newItemIndex, itemIncrement, itemTotal);

		//As soon as the pagination properties are set we need to initialize the drop down pagination control.
		int selectedPage = 1;
		if (m_nItemIncrement != 0 && jumpToPageDDPC.isEdited() == false) {
			if (m_nItemIndex < m_nItemIncrement) {
				selectedPage = 1;
			} else {
				selectedPage = (m_nItemIndex / m_nItemIncrement) + (m_nItemIndex % m_nItemIncrement > 0 ? 1 : 0);
			}
		} else if (jumpToPageDDPC.getSelectedIDs().size() > 0) {
			selectedPage = Integer.parseInt(jumpToPageDDPC.getSelectedIDs().iterator().next());
		}
		jumpToPageDDPC.clearSelections();
		jumpToPageDDPC.setIsEdited(false);
		for (int page = 0; page * m_nItemIncrement < m_nItemTotal; page++) {
			jumpToPageDDPC.addOption(Integer.toString(page + 1), getLocalizer().i18n(CoreWebBundleKey.BUNDLE_NAME,
					CoreWebBundleKey.PAGE_LABEL, new Object[]{Integer.toString(page + 1)}),
					page, selectedPage == page + 1);
		}
	}

	/**
	 * Return the Increment Control
	 */
	protected String getJumpToPageControl() {
		return jumpToPageDDPC.getHtmlDisplay();
	}

	public int getSelectedPaginationPageIndex() {
		if (jumpToPageDDPC.getSelectedIDs() == null || jumpToPageDDPC.getSelectedIDs().size() == 0) {
			return 1;
		}
		return Integer.parseInt(jumpToPageDDPC.getSelectedIDs().iterator().next());
	}

	public boolean extractParameters(HttpServletRequest request) {
		boolean success = super.extractParameters(request);

		jumpToPageDDPC.extractParameters(request);

		return success;
	}

	public void initForDisplay() {
		if (m_nItemTotal > 1) {
			addChildComponent(jumpToPageDDPC);
		}
		super.initForDisplay();
	}

	//Get the item index from the session to return the page where users clicked to make the edit
	private int getPersistedItemIndex(int itemIndex, String pageAction) {
		String strItemIndex = "";
		int newItemIndex = itemIndex;
		if (m_usedInPage == USED_IN_SHIFT_LIST_PAGE) {
			strItemIndex = (String) m_context.getAttribute(RequestContext.SESSION_SCOPE, SHIFTLIST_ITEMINDEX);
		}
		if (m_usedInPage == USED_IN_SHIFTEVENT_LIST_PAGE) {
			strItemIndex = (String) m_context.getAttribute(RequestContext.SESSION_SCOPE, SHIFTEVENTLIST_ITEMINDEX);
		}
		if (m_usedInPage == USED_IN_WORKPATTERN_LIST_PAGE) {
			strItemIndex = (String) m_context.getAttribute(RequestContext.SESSION_SCOPE, WORKPATTERNLIST_ITEMINDEX);
		}
		if (m_usedInPage == USED_IN_ROTATION_LIST_PAGE) {
			strItemIndex = (String) m_context.getAttribute(RequestContext.SESSION_SCOPE, ROTATIONLIST_ITEMINDEX);
		}
		if (!StringUtil.isEmpty(strItemIndex)) {
			newItemIndex = Integer.parseInt(strItemIndex);
		}

		// When the users uses the "Find" feature to filter work rules, the contents of the list can change, resetting
		// the results of the page. This code allows for the reset on the pagination toolbar display for items and
		// page number.
		if (pageAction != null && WorkRulesRH.FIND_ACTION.equals(pageAction)) {
			newItemIndex = 0;
		}

		return newItemIndex;
	}

}
