package com.verint.web.fs.widgets;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Comparator;
import java.util.Iterator;
import java.util.SortedSet;
import java.util.TreeSet;

import com.bluepumpkin.common.datatypes.TimeOfDay;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.util.StringUtil;

/**
 * Translates any times added by offset.
 * @author rfunderburg
 *
 */
public class StartTimes implements SortedSet<TimeOfDay> {

	public static final int ADJACENT_OFFSET = 15;
	private TreeSet<TimeOfDay> data = new TreeSet<TimeOfDay>();
	private Localizer localizer = null;
	private int offset;
	public static final String RANGE_DELIMMITER = " - ";
	
	public StartTimes(Localizer localizer, int timeZoneOffset) {
		super();
		this.localizer = localizer;
		this.offset = timeZoneOffset;
	}

	public StartTimes(Localizer localizer, int timeZoneOffset, Collection<TimeOfDay> times) {
		this(localizer, timeZoneOffset);
		addAll(times);
	}

	public boolean add(TimeOfDay o) {
		return data.add(o.addMinutes(offset));
	}

	public boolean addAll(Collection<? extends TimeOfDay> c) {
		boolean success = true;
		for (TimeOfDay t : c) success = add(t) && success;
		return success;
	}

	public void clear() {
		data.clear();
	}

	public Object clone() {
		return data.clone();
	}

	public Comparator<? super TimeOfDay> comparator() {
		return data.comparator();
	}

	public boolean contains(Object o) {
		return data.contains(o);
	}

	public TimeOfDay first() {
		return data.first();
	}

	public SortedSet<TimeOfDay> headSet(TimeOfDay toElement) {
		return data.headSet(toElement);
	}

	public boolean isEmpty() {
		return data.isEmpty();
	}

	public Iterator<TimeOfDay> iterator() {
		return data.iterator();
	}

	public TimeOfDay last() {
		return data.last();
	}

	public boolean remove(Object o) {
		return data.remove(o);
	}

	public int size() {
		return data.size();
	}

	public SortedSet<TimeOfDay> subSet(TimeOfDay fromElement, TimeOfDay toElement) {
		return data.subSet(fromElement, toElement);
	}

	public SortedSet<TimeOfDay> tailSet(TimeOfDay fromElement) {
		return data.tailSet(fromElement);
	}

	public boolean equals(Object o) {
		if (! (o instanceof StartTimes)) return false;
		StartTimes object = (StartTimes) o;
		return data.equals(object.data);
	}

	public int hashCode() {
		return data.hashCode();
	}

	public boolean removeAll(Collection<?> c) {
		return data.removeAll(c);
	}

	public boolean containsAll(Collection<?> c) {
		return data.containsAll(c);
	}

	public boolean retainAll(Collection<?> c) {
		return data.retainAll(c);
	}

	public Object[] toArray() {
		return data.toArray();
	}

	public <T> T[] toArray(T[] a) {
		return data.toArray(a);
	}

	public String toString() {
		ArrayList<String> times = new ArrayList<String>();
		TimeOfDay first=null, last=null;
		for (TimeOfDay t : data) {
			if (first == null) first = t;
			else {
				if (last != null) {
					if (t.getMinutesSinceMindnight() - last.getMinutesSinceMindnight() == ADJACENT_OFFSET) {
						last = t;
					} else {
						times.add(localizer.formatTimeInMins(first.getMinutesSinceMindnight()) + RANGE_DELIMMITER + localizer.formatTimeInMins(last.getMinutesSinceMindnight()));
						first = t;
						last = null;
					}
				} else {
					if (t.getMinutesSinceMindnight() - first.getMinutesSinceMindnight() == ADJACENT_OFFSET) {
						last = t;
					} else {
						times.add(localizer.formatTimeInMins(first.getMinutesSinceMindnight()));
						first = t;
						last = null;
					}
				}
			}
		}
		if (first != null) {
			if (last == null) times.add(localizer.formatTimeInMins(first.getMinutesSinceMindnight()));
			else times.add(localizer.formatTimeInMins(first.getMinutesSinceMindnight()) + RANGE_DELIMMITER + localizer.formatTimeInMins(last.getMinutesSinceMindnight()));
		}
		return StringUtil.join(times, ", ");
	}
	
}
