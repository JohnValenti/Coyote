package com.verint.web.fs.widgets;

import com.bluepumpkin.common.datatypes.DayOfWeek;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.datatypes.Pair;
import com.bluepumpkin.common.l10n.RegionalFormatBundleKey;
import com.bluepumpkin.common.localization.Localizer;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.web.core.l10n.CoreWebBundleKey;
import com.bluepumpkin.web.core.pagecomponent.DropDownSelectionPC;
import com.witness.web.uif.system.RequestContext;

import javax.servlet.http.HttpServletRequest;
import java.util.Calendar;

public class WeekDropDownSelectionPC extends DropDownSelectionPC<Pair<LocalDate, LocalDate>> {

	private static final long serialVersionUID = 1L;

	public WeekDropDownSelectionPC(RequestContext context, String name, DayOfWeek weekStartDay, LocalDate startDate, LocalDate endDate) {
		super(context, name);
		// Make defensive copies
		LocalDate start = new LocalDate(startDate);
		LocalDate end = new LocalDate(endDate);

		// expand the date range to fall on week boundaries
		while (start.get(Calendar.DAY_OF_WEEK) != weekStartDay.getCalendarConstant()) {
			start.add(Calendar.DAY_OF_YEAR, -1);
		}
		while (end.get(Calendar.DAY_OF_WEEK) != weekStartDay.getCalendarConstant()) {
			end.add(Calendar.DAY_OF_YEAR, 1);
		}

		// Add Options
		LocalDate temp = new LocalDate(start);
		while (temp.before(end)) {
			LocalDate weekStart = new LocalDate(temp);
			LocalDate weekEnd = new LocalDate(temp);
			weekEnd.add(Calendar.WEEK_OF_YEAR, 1);
			LocalDate weekEndDisplay = new LocalDate(weekEnd);
			weekEndDisplay.add(Calendar.DAY_OF_YEAR, -1);
			Localizer l = context.getLocalizer();
			String displayValue = l.i18n(CoreWebBundleKey.BUNDLE_NAME,
					CoreWebBundleKey.WEEK_OF_X_TO_Y,
					l.formatDate(weekStart, RegionalFormatBundleKey.DATE_FORMAT),
					l.formatDate(weekEndDisplay, RegionalFormatBundleKey.DATE_FORMAT));
			addOption(Long.toString(temp.getTime(TimeZoneUtil.GMT_TIMEZONE).getTime()), displayValue, new Pair<LocalDate, LocalDate>(weekStart, weekEnd), false);
			temp.add(Calendar.WEEK_OF_YEAR, 1);
		}
	}

	boolean isExtracted = false;

	@Override
	public boolean extractParameters(HttpServletRequest request) {
		if (isExtracted) {
			return true;
		}
		isExtracted = true;
		return super.extractParameters(request);
	}


}
