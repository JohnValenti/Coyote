package com.verint.web.fs.widgets;

import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.web.bbm.keys.BbmJavaScriptFileID;
import com.bluepumpkin.web.fs.keys.FsJavaScriptFileID;
import com.witness.web.uif.keys.JavaScriptFileID;
import com.witness.web.uif.pagecomponent.PageComponent;
import com.witness.web.uif.system.RequestContext;
import com.witness.web.uif.util.HtmlUtil;
import com.witness.web.uif.util.js.JSUtil;

import javax.servlet.http.HttpServletRequest;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.Map;

/**
 * This widget is designed to work as follows.
 * <p>
 * <ol>
 * <li>Instantiate an instance
 * <li>Call addRadioOption to add options to the widget
 * <li>Call extractParams to read any previously made selection
 * </ol>
 *
 * @param <T> An object representing an option in the selection list
 * @author rfunderburg
 */
@SuppressWarnings("serial")
public class RadioSelectionGroupPC<T> extends PageComponent {

	private boolean isMultiEdit = false;
	private ArrayList<RadioEntry> radioEntries = new ArrayList<RadioEntry>();
	private Map<String, T> selections = new HashMap<String, T>();
	private String selectedKey;
	private boolean isEdited = false;

	public RadioSelectionGroupPC(RequestContext context, String name) {
		super(context);
		setName(name);
		addRequiredJavaScriptFile(JavaScriptFileID.JQUERY);
		addRequiredJavaScriptFile(FsJavaScriptFileID.MODULE_RADIO_SELECTION_GROUP);
		addRequiredJavaScriptFile(BbmJavaScriptFileID.MODULE_EDITABLE);
		addRequiredJavaScriptFile(JavaScriptFileID.MEDIATOR_DEFAULT);
	}

	public void addRadioOption(String uniqueID, String labelText, String extraHtml, T object, boolean isSelected, String onClickJS) {
		selections.put(uniqueID, object);
		if (onClickJS == null) {
			onClickJS = "";
		}
		if (isSelected) {
			selectedKey = uniqueID;
		}
		radioEntries.add(new RadioEntry(uniqueID, labelText, extraHtml, onClickJS));
	}

	@Override
	public String getHtmlDisplay() {
		StringBuffer html = new StringBuffer();
		if (isMultiEdit && !isEdited) {
			html.append("<span id='").append(getMultiEditIndicatorID()).append("' class='multiEditIndicator' style='padding-right: 2px;'>*</span>");
		}
		html.append("<span id='").append(getName()).append("__wrapper'><div>");
		String inputName = getHTMLInputName();
		int counter = 0;
		for (RadioEntry re : radioEntries) {
			String inputID = inputName + "__option_" + counter++;
			html.append("<input class='vAlignMiddle' ").append(re.uniqueID.equals(selectedKey) ? "checked " : "").append("type='radio' id='").append(inputID).append("' name='").append(inputName).append("' value='").append(re.uniqueID).append("' onClick=\"").append(HtmlUtil.escapeAttributeValue(JSUtil.ON_CHANGE)).append(HtmlUtil.escapeAttributeValue(re.onClickJS)).append("\">");
			html.append("<label class='vAlignMiddle' for='").append(inputID).append("'>").append(re.labelText).append("</label>");
			html.append(re.extraHtml);
			html.append("</input><br>\n");
		}
		html.append("<input type='hidden' id='" + getEditedID() + "' name='" + getEditedName() + "' value='false'></input>");
		html.append("</div></span>");
		return html.toString();
	}

	@Override
	public String getJavaScriptInitCode() {
		return "\n//RadioSelectionGroup\n"
				+ getName() + " = new RadioSelectionGroupPC('" + getName() + "', '" + getHTMLInputName() + "', '" + getEditedID() + "', '" + getMultiEditIndicatorID() + "');\n"
				+ getName() + ".bind();\n";
	}


	@Override
	public boolean extractParameters(HttpServletRequest request) {
		if (!isAutoExtractParamEnabled()) {
			return true;
		}
		String editedStr = getParameter(request, getEditedName());
		if (editedStr != null) {
			this.isEdited = "true".equalsIgnoreCase(editedStr);
		}
		String key = getParameter(request, getHTMLInputName());
		if (!StringUtil.isEmpty(key)) {
			selectedKey = key;
		}
		return true;
	}

	public T getSelected() {
		return selections.get(selectedKey);
	}

	public void setSelectedID(String id) {
		selectedKey = id;
	}

	public String getSelectedID() {
		return selectedKey;
	}

	private class RadioEntry {
		String uniqueID;
		String labelText;
		String extraHtml;
		String onClickJS;

		public RadioEntry(String uniqueID, String labelText, String extraHtml,
				String onClickJS) {
			super();
			this.uniqueID = uniqueID;
			this.labelText = labelText;
			this.extraHtml = extraHtml;
			this.onClickJS = onClickJS;
		}
	}

	private String getMultiEditIndicatorID() {
		return getName() + "__me";
	}

	private String getEditedName() {
		return getName() + "__edited";
	}

	private String getEditedID() {
		return getName() + "__edited_id";
	}

	private String getHTMLInputName() {
		return getName() + "__input";
	}

	public boolean isEdited() {
		return isEdited;
	}

	public boolean isMultiEdit() {
		return isMultiEdit;
	}

	public void setMultiEdit(boolean multiEdit) {
		isMultiEdit = multiEdit;
	}
}
