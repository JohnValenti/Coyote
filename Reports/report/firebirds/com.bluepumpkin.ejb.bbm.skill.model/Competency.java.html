package com.bluepumpkin.ejb.bbm.skill.model;

/**

import java.util.Date;

import com.bluepumpkin.ejb.bbm.vo.*;

public class Competency extends ValueObjectBase
{
    /**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new CompetencyFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }

    public Competency()
    {
       super();
    }

    public String getExternalIdent()
    {
       return getFormattedFieldValue(CompetencyFieldInfo.COMPETENCY_EXTERNALIDENT);
    }

    public String getName()
    {
       return getFormattedFieldValue(CompetencyFieldInfo.COMPETENCY_NAME);
    }

    public String getDescription()
    {
       return getFormattedFieldValue(CompetencyFieldInfo.COMPETENCY_DESCRIPTION);
    }

    public Date getCreationTime()
    {
       return this.getFieldValueDate(CompetencyFieldInfo.COMPETENCY_CREATIONTIME);
    }

    public Date getModificationTime()
    {
       return this.getFieldValueDate(CompetencyFieldInfo.COMPETENCY_MODIFICATIONTIME);
    }

    public boolean getIsDeleted()
    {
       return this.getFieldValueBoolean(CompetencyFieldInfo.COMPETENCY_ISDELETED);
    }

    public void  setExternalIdent( String strValue)
    {
    	setFieldValue(CompetencyFieldInfo.COMPETENCY_EXTERNALIDENT, strValue);
    }

    public void setName(String strValue)
    {
       setFieldValue(CompetencyFieldInfo.COMPETENCY_NAME, strValue);
    }

    public void setDescription(String strValue)
    {
       setFieldValue(CompetencyFieldInfo.COMPETENCY_DESCRIPTION, strValue);
    }

    public void  setCreationTime(Date date)
    {
       setFieldValue(CompetencyFieldInfo.COMPETENCY_CREATIONTIME, date);
    }

    public void setModificationTime(Date date)
    {
        setFieldValue(CompetencyFieldInfo.COMPETENCY_MODIFICATIONTIME, date);
    }

    public void  setIsDeleted(boolean value )
    {
       setFieldValue(CompetencyFieldInfo.COMPETENCY_ISDELETED, value);
    }

	public String toString()
	{
			StringBuffer buf = new StringBuffer();
			buf.append( "[" );
			buf.append( " External Ident=" );
			buf.append( this.getExternalIdent() );
			buf.append( " Name=" );
			buf.append( this.getName() );
			buf.append( " Description=" );
			buf.append( this.getDescription() );
			buf.append( " Deleted=" );
			buf.append( this.getIsDeleted() );
			
			buf.append( "]" );

			return buf.toString();
	}

}
