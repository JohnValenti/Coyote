package com.verint.web.rest.wfm.resource.shiftassignment;

import com.bluepumpkin.ejb.bbm.schedule.model.ShiftAssignment;
import com.verint.web.rest.wfm.resource.shifteventassignment.ShiftEventAssignmentMapper;
import com.verint.web.rest.wfm.util.MapperUtil;
import org.mapstruct.DecoratedWith;
import org.mapstruct.Mapper;
import org.mapstruct.Mapping;
import org.mapstruct.factory.Mappers;

import java.util.Collection;

@Mapper(uses = { MapperUtil.class, ShiftEventAssignmentMapper.class })
@DecoratedWith(ShiftAssignmentMapperDecorator.class)
public interface ShiftAssignmentMapper {

	ShiftAssignmentMapper INSTANCE = Mappers.getMapper(ShiftAssignmentMapper.class);

	@Mapping(source = "children", target = "shiftEvents")
	@Mapping(source = "campaignID", target = "schedulingPeriodID")
	@Mapping(source = "OTExtensionAfterActivityID", target = "otExtensionAfterActivityID")
	@Mapping(source = "OTExtensionBeforeActivityID", target = "otExtensionBeforeActivityID")
	@Mapping(source = "OTExtensionBeforeID", target = "otExtensionBeforeID")
	@Mapping(source = "OTExtensionAfterID", target = "otExtensionAfterID")
	ShiftAssignmentEntity saToSaEntity(ShiftAssignment source);

	@Mapping(source = "schedulingPeriodID", target = "campaignID")
	@Mapping(source = "otExtensionAfterActivityID", target="OTExtensionAfterActivityID")
	@Mapping(source = "otExtensionBeforeActivityID", target="OTExtensionBeforeActivityID")
	@Mapping(source = "otExtensionBeforeID", target = "OTExtensionBeforeID")
	@Mapping(source = "otExtensionAfterID", target = "OTExtensionAfterID")
	ShiftAssignment saEntityToSa(ShiftAssignmentEntity source);

	Collection<ShiftAssignmentEntity> sasToSaEntities(Collection<ShiftAssignment> source);
}
