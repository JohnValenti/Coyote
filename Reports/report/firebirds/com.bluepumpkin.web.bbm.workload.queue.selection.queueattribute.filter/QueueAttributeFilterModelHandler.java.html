public class QueueAttributeFilterModelHandler extends ModelHandler {
	private static Category log = Log.initCategory( QueueAttributeFilterModelHandler.class.getName());

	private static WorkloadManager workLoadManager = null;

	private static WorkloadManager getWorkloadManager()throws Exception {
		if (workLoadManager == null)
			workLoadManager = WfmManagerFactory.getWorkloadManager();
		return workLoadManager;
	}
	
	public static Collection getOrganizations(RequestContext m_context) throws BbmException{
		try {
			return OrganizationModelHandler.getOrganizations(m_context);
		} catch (RemoteException e) {
			throw new BbmException(e);
		}
	}
	
	public static Collection getQueueAttributes(RequestContext context, Collection<ID> orgIDs) throws Exception {
		
		HashSet orgsToFetch = new HashSet(orgIDs);
		try {
			
			orgsToFetch.addAll(
					OrganizationModelHandler
					.getWorkResourceManager(context)
					.getOrganizationsParentsByIDs(orgIDs));
			
		} catch (RemoteException e) {
			throw new BbmException(e);
		}
	
		return getWorkloadManager().getQueueAttributeDefintionsByOrgID(orgsToFetch);
	}
	
	public static ID createOrUpdateQueueAttributeFilter(RequestContext context, QueueAttributeFilter filter) throws Exception {
		return getWorkloadManager().createOrUpdateQueueAttributeFilter(filter);
	}
	
	public static Collection<QueueAttributeFilter> getQueueAttributeFilterForUser(ID userID) throws Exception {
		return getWorkloadManager().getQueueAttributeFilters(userID);
	}
	
	public static Collection<QueueAttributeFilter> getPublicQueueAttributeFilterForUser(ID userID, Collection<ID> orgIDs) throws Exception {
		return getWorkloadManager().getPublicQueueAttributeFilters(userID, orgIDs);
	}
	
	public static QueueAttributeFilter getQueueAttributeFilter(ID queueAttributeFilterID) throws Exception {
		return getWorkloadManager().getQueueAttributeFilterByID(queueAttributeFilterID);
	}
	
	public static void deleteQueueAttributeFilter(ID queueAtributeFilterID) throws Exception {
		getWorkloadManager().deleteQueueAttributeFilter(queueAtributeFilterID);
	}
	
	public static Collection<Queue> getQueuesAssociatedToFilter(RequestContext context, ID queueAttributeFilterID, ID campaignID, ID spID, 
			boolean showSubQueues, Date startDate, Date endDate) throws Exception {
		Collection<ID> queueIDs = getWorkloadManager().getQueuesAssociatedToQueueAttributeFilter(queueAttributeFilterID, campaignID, spID);

		if(showSubQueues) {
			Collection subQueueIDs = WorkloadModelHandler.getSubQueues(context, queueIDs);
			if (!subQueueIDs.isEmpty()) {
				ArrayList tmpQIDs = new ArrayList(queueIDs.size() + subQueueIDs.size());
				tmpQIDs.addAll(queueIDs);
				tmpQIDs.addAll(subQueueIDs);
				queueIDs = tmpQIDs;
			}
		}
		Collection<Queue> retrievedQueues = getWorkloadManager().getQueuesByIDs(queueIDs);
		
    	//The below is done to ensure the queue attribute selection returns the subset of the queues for the current date
    	//range defined by startDate and endDate.
    	//This is because there is no SP id passed to getQueuesAssociatedToFilter from Pulse and so it will return all the queues that were linked to 
    	//the campaign irrespective of the date range.
    	ArrayList filteredQueues = new ArrayList();
    	if(campaignID != null && spID == null && startDate != null && endDate != null) {
    		Collection campQueues = CampaignModelHandler.getQueuesAssignedToCampaign(context, campaignID, startDate, endDate,
    				showSubQueues);
    		for(Object q : retrievedQueues) {
    			if(campQueues.contains((Queue)q)) {
    				filteredQueues.add(q);
    			}
    		}
    		return filteredQueues;
    	}
    	
    	return retrievedQueues;
    	
    	

	}
	
	public static Collection<String> getExistingQueueAttributeFilterNames() throws Exception {
		return getWorkloadManager().getExistingQueueAttributeFilterNames();
	}
}
