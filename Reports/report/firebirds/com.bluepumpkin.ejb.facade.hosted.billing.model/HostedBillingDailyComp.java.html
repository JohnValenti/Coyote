package com.bluepumpkin.ejb.facade.hosted.billing.model;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;

import java.util.Date;

public class HostedBillingDailyComp extends ValueObjectBase {
	// meta data info.
	private static FieldInfo m_fieldInfo = new HostedBillingDailyCompFieldInfo();

	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public HostedBillingDailyComp() {
		super();
	}

	public Date getADate() {
		return getFieldValueDate(HostedBillingDailyCompFieldInfo.HOSTEDBILLINGDAILYCOMP_DAYDATE);
	}
	
	public ID getOrganizationID() {
		return getFieldValueID(HostedBillingDailyCompFieldInfo.HOSTEDBILLINGDAILYCOMP_ORGANIZATIONID);
	}
	
	public ID getHostedBillingComponentID() {
		return getFieldValueID(HostedBillingDailyCompFieldInfo.HOSTEDBILLINGDAILYCOMP_HOSTEDBILLINGCOMPONENTID);
	}
	
	public boolean getXists() {
		return getFieldValueBoolean(HostedBillingDailyCompFieldInfo.HOSTEDBILLINGDAILYCOMP_XISTS);
	}
	
	public boolean privilegeExists() {
		return getXists()==true;	
	}
	
	public void setADate(Date day) {
		setFieldValue(HostedBillingDailyCompFieldInfo.HOSTEDBILLINGDAILYCOMP_DAYDATE, day);
	}
	
	public void setOrganizationID(ID orgID) {
		setFieldValue(HostedBillingDailyCompFieldInfo.HOSTEDBILLINGDAILYCOMP_ORGANIZATIONID, orgID);
	}
	
	public void setHostedBillingComponentID(ID hostedBillingComponentsID) {
		setFieldValue(HostedBillingDailyCompFieldInfo.HOSTEDBILLINGDAILYCOMP_HOSTEDBILLINGCOMPONENTID, hostedBillingComponentsID);
	}

	public void setXists(boolean exists) {
		setFieldValue(HostedBillingDailyCompFieldInfo.HOSTEDBILLINGDAILYCOMP_XISTS, exists);
	}
}
