package com.bluepumpkin.ejb.rm.util.interval;


/**
 * Simplification of MergeIntervalOperations if the input lists and the output lists are all of the same type
 */
public abstract class SingleTypeMergeIntervalOperations<T extends Interval> implements MergeIntervalOperations<T, T, T> {

	@Override
	public T select1(T a) {
		return a;
	}

	@Override
	public T select2(T b) {
		return b;
	}

}
