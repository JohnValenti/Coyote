import java.util.Collections;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectRich;
import com.bluepumpkin.ejb.bbm.workrules.model.ComplexWorkRuleFieldInfo;
import com.bluepumpkin.ejb.bbm.workrules.model.Rotation;

public class WorkResourceRotation extends ValueObjectRich
{
	private static final long serialVersionUID = 1L;

	private static FieldInfo m_fieldInfo = new WorkResourceComplexWorkRuleFieldInfo();
    protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	public ID getID() {
		return this.getFieldValueID(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_SID);
	}
	
	public ID getWorkResourceID() {
		return this.getFieldValueID(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_WORKRESOURCEID);
	}

	public ID getComplexWorkRuleID() {
		if (getRotation() != null) return getRotation().getID();
		return null;
	}
	
	//This is a DEID
	public void setComplexWorkRuleID(ID cwrID) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_COMPLEXWORKRULEID, cwrID);
	}
	
	public int getStartWeekOffSet() {
		return this.getFieldValueInt(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_STARTWEEKOFFSET);
	}
	
	public void setStartWeekOffSet(int startWeekOffSet) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_STARTWEEKOFFSET, startWeekOffSet);		
	}
	
	public int getNumWeeksInRotation() {
		int numWeeksInRotation = 0;
		Rotation rotationRule = getRotation();
		if ( rotationRule != null) {
			numWeeksInRotation = rotationRule.getShiftPatterns().size();
		}
		return numWeeksInRotation;
	}
	
	@Deprecated
	public void setNumWeeksInRotation(int numWeeks) {
		setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_NUMWEEKSINROTATION, numWeeks);
	}
	
	protected int getWeekEnabledMask() {
		return this.getFieldValueInt(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_WEEKENABLEDMASK);
	}
	
	public void setWeekEnabledMask(int mask) {
		this.setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_WEEKENABLEDMASK, mask);
	}
	
	public void setWorkResourceID(ID value) {
		this.setFieldValue(WorkResourceComplexWorkRuleFieldInfo.COMPLEXWORKRULEWORKRESOURCE_WORKRESOURCEID, value);
	}
    
	public Rotation getRotation() {
		if (getRelatedObjects(WorkResourceComplexWorkRule.COMPLEXWORKRULE_RELATIONSHIP_KEY).size() > 0) {
			return (Rotation)getRelatedObjects(WorkResourceComplexWorkRule.COMPLEXWORKRULE_RELATIONSHIP_KEY).iterator().next();
		}
		return null;
	}
	
	public void setRotation(Rotation rotation) {
		setComplexWorkRuleID(rotation.getFieldValueID(ComplexWorkRuleFieldInfo.Column.ID.getIndex()));
		setRelatedObjects(WorkResourceComplexWorkRule.COMPLEXWORKRULE_RELATIONSHIP_KEY, Collections.singleton(rotation));
	}  
}
