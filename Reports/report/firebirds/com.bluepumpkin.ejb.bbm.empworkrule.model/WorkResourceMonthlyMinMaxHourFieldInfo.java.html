package com.bluepumpkin.ejb.bbm.empworkrule.model;

public class WorkResourceMonthlyMinMaxHourFieldInfo extends FieldInfo {
	private static final long serialVersionUID = 1L;

	public enum Column {
		ID("ID", PrimitiveTypes.ID),
		WORKRESOURCEID("WORKRESOURCEID", PrimitiveTypes.ID),
		MONTHYEAR("MONTHYEAR", PrimitiveTypes.DATE),
		MINMINUTES("MINMINUTES", PrimitiveTypes.INT),
		MAXMINUTES("MAXMINUTES", PrimitiveTypes.INT);

		// Static block to set field array indexes
		static {
			int ctr = 0;
			for (Column fie : Column.values()) {
				fie.index = ctr;
				ctr++;
			}
		}

		// Column Properties
		private int index;
		private FieldInfoEntry fie;

		Column(String name, int type) {
			this.fie = new FieldInfoEntry(name, type);
		}

			return this.fie;
		}

		}
	}

	private static final FieldInfoEntry[] FIELD_INFO_ENTRIES;
	static {
		int ctr = 0;
		FIELD_INFO_ENTRIES = new FieldInfoEntry[Column.values().length];
		for (Column c : Column.values()) {
			FIELD_INFO_ENTRIES[ctr] = c.getFieldInfoEntry();
			ctr++;
		}
	}

	@Override

		return "WORKRESOURCEMONTHLYMINMAXMINUTES";
