/**
 * For single shift auctions, ensure that two or more shift assignments associated with the shift
 * bid request do not start in the same org day.
 *
 * @author rrajendran
 *
 * To change this generated comment go to
 * Window>Preferences>Java>Code Generation>Code and Comments
 */
public class NoMultipleShiftAssignmentsForADayHV implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

    private final static String m_className = NoMultipleShiftAssignmentsForADayHV.class.getName();
    private static final Category m_cat = Log.initCategory(m_className);

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

        try {
            ShiftBidRequest sbReq = (ShiftBidRequest) validatable;
            ShiftBidRequestValidationCache sbReqValCache = sbReq.getValidationCacheSBR();

            // rule only applies to single shift auctions
            ShiftBidAuction shiftBidAuction = sbReq.getOptMethods().getShiftBidAuction();
            if (shiftBidAuction.getIsFullPeriodSchedule()) {
            	return null;
            }

            TimeZone empOrgTZ = sbReqValCache.getTimeZoneForOrg();
            DateFormat dfShortInOrgTZ = DateFormat.getDateInstance(DateFormat.SHORT);
            dfShortInOrgTZ.setTimeZone(empOrgTZ);

            Map dateStrToShiftAssnMap = new HashMap(5);

            List shiftAssns = sbReq.getOptMethods().getShiftAssignments();
            ValidationResult valResult = null;

            TimeMap aggregatedTM = RequestUtil.buildTimeMap(null, sbReqValCache.getDateRange(), TimeMap.ONE_MINUTE * 1L, false);
            for (Iterator shiftAssnsIter = shiftAssns.iterator(); shiftAssnsIter.hasNext();) {
                ShiftAssignment shiftAssn = (ShiftAssignment) shiftAssnsIter.next();

                // Get the shift starttime formatted as a date string (example: mm/dd/yy; this format depends on the locale)
                // in the organization timeZone.
                String dateStr = dfShortInOrgTZ.format(shiftAssn.getStartTime());

            	//Check overlap of requests in the bid.
                if (shiftBidAuction.getIsMultiShift()) {
                    valResult = checkMultiShift(shiftAssn, aggregatedTM, dateStr, sbReq, sbReqValCache);
                	//Multi shift will skip checking same org day.
                	continue;
                };

                // See notes.txt for details about HOO, OrgDayOffset and OrgWeekOffset calculations.
                // check map to see if another shift assignment already exists for the same date.
                if (dateStrToShiftAssnMap.containsKey(dateStr)) {
                    ShiftAssignment sameDayShiftAssn = (ShiftAssignment) dateStrToShiftAssnMap.get(dateStr);

                    valResult = ValidationUtil.setHardValidationResult(sbReq, RmEjbBundleKey.SBR_TWO_SBR_SHIFTASSNS_SAME_ORG_DAY,
                        sameDayShiftAssn.getStartTime(), sameDayShiftAssn.getEndTime(),
                        shiftAssn.getStartTime(), shiftAssn.getEndTime(), m_className);

                    //return valResult; //defer returning until all shift assignments are processed.
                } else { // add start date string for the shift assignment to the map.
                    dateStrToShiftAssnMap.put(dateStr, shiftAssn);
                }
            }

            return valResult;
        }finally {
           m_cat.debug("Validation Rule: " + m_className + " done");
        }
    }

    /**
     * checkMultiShift - checks if on same day shifts don't overlap
     * @param shiftAssn
     * @param aggregatedTM
     * @param dateStr
     * @param sbReq
     * @param sbReqValCache
     * @return ValidationResult
     * @throws Exception
     */
    public ValidationResult checkMultiShift(ShiftAssignment shiftAssn,  TimeMap aggregatedTM, String dateStr,
    		ShiftBidRequest sbReq, ShiftBidRequestValidationCache sbReqValCache) throws Exception{
    	ValidationResult valResult = null;
    	TimeMap tm = RequestUtil.buildTimeMap(Collections.singleton(shiftAssn), sbReqValCache.getDateRange(), TimeMap.ONE_MINUTE * 1L, false);
        if (aggregatedTM!=null){
        	if (aggregatedTM.onBitsOverlapRange(shiftAssn.getStartTime(), shiftAssn.getEndTime())){
        		valResult = ValidationUtil.setHardValidationResult(sbReq, RmEjbBundleKey.SBR_OVERLAPS_IN_BID,
        				shiftAssn.getStartTime(), shiftAssn.getEndTime(), m_className);
        	}
        	//add the shift to the timemap. If each shift doesn't overlap, then there are no overlaps.
    		aggregatedTM.or(tm);
        }

        return valResult;
    }
}
