/**
 * Verifies if another ShiftBidRequest has the same set of biddable schedules as this.
 *
 * @author rrajendran
 *
 * To change this generated comment go to
 * Window>Preferences>Java>Code Generation>Code and Comments
 */
public class ShiftBidRequestNoDuplicatesHV implements Validator {
    private static final Priority m_tracePriority = RequestUtil.m_defaultTracePriority;

    private static final String m_className = ShiftBidRequestNoDuplicatesHV.class.getName();

    private static Category m_cat = Log.initCategory(m_className);

    public static Set m_statusesForDuplicateSBReqCheck = new HashSet();
    static {
        m_statusesForDuplicateSBReqCheck.add(RequestAuditTrail.STATUS_PENDING);
        m_statusesForDuplicateSBReqCheck.add(RequestAuditTrail.STATUS_APPROVED);
        m_statusesForDuplicateSBReqCheck.add(RequestAuditTrail.STATUS_ESCALATED);
    }

    /* (non-Javadoc)
     * @see com.bluepumpkin.ejb.rm.requests.common.validation.Validator#validate(com.bluepumpkin.ejb.rm.requests.common.validation.Validatable)
     */
    public ValidationResult validate(Validatable validatable) throws Exception {
        m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " started");

        ShiftBidRequestDAO sbrDAO = null;
        try  {
            ShiftBidRequest sbr = (ShiftBidRequest) validatable;

            //TODO: refactor Validator interface to add a DAO as an argument to this method.
            sbrDAO = new ShiftBidRequestDAO(ShiftBidRequest.DL_BASIC);

            Collection otherSBRs = sbrDAO.getRequestsByShiftBidderID(sbr.getShiftBidderID(),
                ShiftBidRequest.DL_BASIC | ShiftBidRequest.DL_BIDDABLESCHEDULE_IDS);
            if (otherSBRs.isEmpty())   return null; //no other shift bid requests for this shift bidder.

            // get biddable schedule IDs for the the given SBR
            List sbrBiddableScheduleIDs = sbr.getOptMethods().getBiddableScheduleIDs();

            // scan the other SBRs for shift bidder
            for (Iterator otherSBRsIter = otherSBRs.iterator(); otherSBRsIter.hasNext();) {
                ShiftBidRequest otherSBR = (ShiftBidRequest) otherSBRsIter.next();

                // if request's status is not one of the allowed values for duplicate request processing.
                if (!m_statusesForDuplicateSBReqCheck.contains(otherSBR.getRequestStatus())) continue;

                // get biddable schedule IDs for the other SBR.
                List otherSBRBidSchedIDs = otherSBR.getOptMethods().getBiddableScheduleIDs();

                // if biddable scheduleID collection sizes don't match, then not a duplicate request.
                if (otherSBRBidSchedIDs.size() != sbrBiddableScheduleIDs.size())  continue;

                // now see if the biddable schedule IDs match.  Note that the biddable schedule IDs are
                // sorted, so we can do a one to one comparison.
                int idx = 0;
                for (idx = 0; idx < sbrBiddableScheduleIDs.size(); idx++)  {
                    // if IDs do not match, stop comparsion.
                    if (!sbrBiddableScheduleIDs.get(idx).equals(otherSBRBidSchedIDs.get(idx)))
                        break;
                }

                // if all biddable IDs were scanned and they matched, then duplicate request.
                if (idx == sbrBiddableScheduleIDs.size())  {
                    ValidationResult result = ValidationUtil.setHardValidationResult(sbr, 
                    	RmEjbBundleKey.SBR_NO_DUPLICATE_REQS, otherSBR.getName(), m_className);
                    return result;
                }
            }

            return null;
//        } c_atch (Exception e)   {
//            t_hrow RequestUtil.createValidationExceptionWrapper(e, m_cat);
        } finally  {
            if (sbrDAO != null) sbrDAO.cleanUp();
            m_cat.log(m_tracePriority, "Validation Rule: " + m_className + " done");
        }
    }
}
