package com.verint.ejb.bbm.forecast.model.history.service;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.BbmEJBCreateException;
import com.bluepumpkin.ejb.bbm.base.BbmException;
import com.bluepumpkin.ejb.bbm.base.BbmTimeSeriesException;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.time.TimeIntervalAtTime;
import com.bluepumpkin.ejb.bbm.timeseries.model.ActualTraceCube;
import com.bluepumpkin.ejb.bbm.workload.model.Queue;
import com.verint.ejb.bbm.forecast.ejb.ForecastTimeSeriesManager;
import com.verint.ejb.wfm.WfmManagerFactory;

import java.util.Collection;
import java.util.Date;
import java.util.List;
import java.util.Map;

abstract class AbstractHistoryDataService implements IHistoryDataService {

	protected final ForecastTimeSeriesManager forecastTimeSeriesManager;
	protected final Campaign campaign;

	AbstractHistoryDataService(Campaign campaign, boolean isWhatIf) throws BbmEJBCreateException {
		this.campaign = campaign;
		this.forecastTimeSeriesManager = WfmManagerFactory.getForecastTimeSeriesManager(isWhatIf);
	}

	@Override
	public abstract Map<ID, ActualTraceCube> getHistory(Collection<Queue> queue, TimeIntervalAtTime period)
			throws BbmTimeSeriesException;

	Date getAdjustedEndDate(TimeIntervalAtTime period) {
		return new Date(period.getEndTime().getTime() - 1);
	}

	@Override
	public List<Date> getDaysWithHistory(Collection<Queue> queues, Date start, Date end) throws BbmException {
		throw new UnsupportedOperationException("Not implemented: AbstractHistoryDataService.getDaysWithHistory");
	}
}
