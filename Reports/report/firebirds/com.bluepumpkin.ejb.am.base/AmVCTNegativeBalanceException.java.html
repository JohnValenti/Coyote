package com.bluepumpkin.ejb.am.base;

import com.bluepumpkin.ejb.am.l10n.AmEjbBundleKey;
/**
 * Title:        Blue Pumpkin Software Activity Management
 * Description:  Base Exception for all AM exceptions
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, inc
 * @version 1.0
 */

public class AmVCTNegativeBalanceException extends AmException {

	/**
	 *  Construct
	 *
	 *  @param msgID, key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 */
	public AmVCTNegativeBalanceException(String msgID) {
		super(null, msgID, DEFAULT_SEVCODE, null);
	}
	
	public AmVCTNegativeBalanceException(String msgID, Object[] args) {
		super(null, msgID, DEFAULT_SEVCODE, args);
	}

	public AmVCTNegativeBalanceException(Exception e) {
		super(e, AmEjbBundleKey.VCT_NEGATIVE_BALANCE, DEFAULT_SEVCODE, null);
	}
	
	// this constructor allows parms to bubble up as the exception gets passed around
	public AmVCTNegativeBalanceException(AmVCTNegativeBalanceException e) {
		super(e, AmEjbBundleKey.VCT_NEGATIVE_BALANCE, DEFAULT_SEVCODE, e.getParamObject());
	}

}
