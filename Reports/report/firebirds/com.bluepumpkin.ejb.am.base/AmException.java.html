package com.bluepumpkin.ejb.am.base;

import com.bluepumpkin.common.base.BPException;
import com.bluepumpkin.ejb.am.l10n.AmEjbLogBundleKey;
/**
 * Title:        Blue Pumpkin Software Activity Management
 * Description:  Base Exception for all AM exceptions
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, inc
 * @author       Sheng Song
 * @version 1.0
 */

public class AmException extends BPException {
//	private static Category m_cat = Log.initCategory(AmException.class.getName());

	// override the base class to provide the appropriate logging category
//	protected Category getCategory() { return m_cat; }
	// In some cases, the EJB will return an object back to the controller via Exception
	// So the controller doesn't store it in the session
	private Object m_content = null;
	// Override BPException's default resource bundle value, must provide the getBundleName() method
	protected String bundle = "com.bluepumpkin.ejb.am.l10n.AmEjbBundle";
	/**
	 *  Construct
	 *
	 *  @param e any embbed bbm Exception, usually wrapps a non-BPException exception
	 *  @param msgID key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 *  @param sevCode user can provide a specific severity code, but the constructor will pick up the greatest value from passed-in exception, existing default exception severity code and this user specied value. This is to ensure the severity code is always increased.
	 *  @param args provide parameters for the localized string, specified by msgID in the resource bundle file.
	 */
	public AmException (Exception e, String msgID, int sevCode, Object[] args) {
		super(e, msgID, sevCode, args);
	}
	/**
	 *  Construct
	 *
	 *  @param e any embbed bbm Exception, usually wrapps a non-BPException exception
	 *  @param msgID key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 *  @param sevCode user can provide a specific severity code, but the constructor will pick up the greatest value from passed-in exception, existing default exception severity code and this user specied value. This is to ensure the severity code is always increased.
	 */
	public AmException(Exception e, String msgID, int sevCode) {
		this(e, msgID, sevCode, null);
	}
	/**
	 *  Construct
	 *
	 *  @param e any embbed bbm Exception, usually wrapps a non-BPException exception
	 *  @param msgID key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 *  @param args provide parameters for the localized string, specified by msgID in the resource bundle file.
	 */
	public AmException(Exception e, String msgID, Object[] args) {
		this(e, msgID, DEFAULT_SEVCODE, args);
	}

	/**
	 *  Construct
	 *
	 *  @param msgID key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 *  @param args provide parameters for the localized string, specified by msgID in the resource bundle file.
	 */
	public AmException(String msgID, Object[] args) {
		this(null, msgID, DEFAULT_SEVCODE, args);
	}
	/**
	 *  Construct
	 *
	 *  @param msgID key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 *  @param sevCode user can provide a specific severity code, but the constructor will pick up the greatest value from passed-in exception, existing default exception severity code and this user specied value. This is to ensure the severity code is always increased.
	 */
	public AmException(String msgID, int sevCode) {
		this(null, msgID, sevCode, null);
	}
	/**
	 *  Construct
	 *
	 *  @param msgID key to locate the localized message information from the resource bundle file, with default value as "DEFAULT"
	 */
	public AmException(String msgID) {
		this(null, msgID, DEFAULT_SEVCODE, null);
	}

	public AmException(Exception e) {
		this(e, AmEjbLogBundleKey.DEFAULT, DEFAULT_SEVCODE, null);
	}

	/**
	 *  getBundleName
	 *
	 *  Override the default resource bundle name defined in the BPException, so have to provide the override method to get the runtime value
	 *  @return the default resource bundle name
	 */
	public String getBundleName() {
		return bundle;
	}
	public Object getContent() { return m_content; }
	public void setContent(Object obj) { m_content = obj; }
}
