import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.Calendar;
import java.util.Iterator;
import java.util.TimeZone;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.LocalDate;
import com.bluepumpkin.common.jdmo.Jdmo;
import com.bluepumpkin.common.jdmo.JdmoException;
import com.bluepumpkin.common.jdmo.JdmoQuery;
import com.bluepumpkin.common.jdmo.JdmoRowset;
import com.bluepumpkin.common.jdmo.JdmoUtil;
import com.bluepumpkin.common.util.TimeZoneUtil;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.workload.model.CombinedQueueStaffing;
import com.bluepumpkin.ejb.bbm.workload.model.CombinedQueueStaffingFieldInfo;

public class CombinedQueueStaffingDAO
{
	private static FieldInfo m_fieldInfo = new CombinedQueueStaffingFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }

	/**
	* default constructor
	*/
	public CombinedQueueStaffingDAO() {}

	/**
	 *  get CombinedQueueStaffing objects given a collection spqueue id's
	 */
	static public Collection getCombinedQueueStaffing(Collection spQueueIDColl, Date interval) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();
		try
		{
			StringBuffer pStmt = new StringBuffer("select A.SPQUEUEID, A.TIMEINTERVAL, A.STAFFING from SPCOMBINEDQUEUESTAFFING A where A.SPQUEUEID in (");
			Iterator iter = spQueueIDColl.iterator();
			while( iter.hasNext() )
			{
				Object obj = iter.next();
				pStmt.append("'");
				pStmt.append(obj);
				pStmt.append("'");

				if( iter.hasNext() )
					pStmt.append(",");
			}
			pStmt.append(")");

			pStmt.append(" and A.TIMEINTERVAL = ");
			pStmt.append("'");
			pStmt.append(JdmoUtil.formatDBString(TimeZoneUtil.toTimestamp(interval)));
			pStmt.append("'");

			JdmoQuery jq = jdmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			JdmoRowset rs = jdmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

			ArrayList retVal = new ArrayList();
			while(rs.next())
			{
			    CombinedQueueStaffing que = new CombinedQueueStaffing();
			    que.setSPQueueID(rs.getString("SPQUEUEID").trim());
			    que.setStaffing(rs.getFloat("STAFFING"));
			    que.setTimeInterval(TimeZoneUtil.toDate(rs.getTimestamp("TIMEINTERVAL")));

			    retVal.add(que);
			}
			return retVal;
		}
		finally
		{
			jdmo.cleanUp();
		}
	}

	static public void updateCombinedQueueStaffing( ID spID, 
			ID mediaID, float staffing, Date timeInterval ) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();
		try
		{

			// get the spqueueid for the given media
			StringBuffer pStmt = new StringBuffer();
			pStmt.append("select ID, MEDIAID from spqueue q where q.spid in ( select ID from sp where SID = ");
			pStmt.append(spID);
			pStmt.append(" ) and queueid is null");

			JdmoQuery jq = jdmo.createQuery(pStmt.toString(), Jdmo.PARAM_QUERY);
			JdmoRowset rs = jdmo.createRowset(jq, Jdmo.FORWARD_ONLY, Jdmo.READ_ONLY);

			ArrayList retVal = new ArrayList();
			CombinedQueueStaffing staff = new CombinedQueueStaffing();

			while(rs.next())
			{
				staff.setSPQueueID( rs.getString("ID").trim() );
				staff.setTimeInterval(timeInterval);

				ID media = null;
				String idStr = rs.getString("MEDIAID");
				if( idStr != null )
				{
					media = new ID(idStr.trim().hashCode());
					staff.setMediaID(media);
				}

				if( (media == null && mediaID == null)
						|| (media != null && mediaID != null && media.equals(mediaID))	)
				{
					staff.setStaffing(staffing);
					retVal.add( staff.getSPQueueID());
					break;
				}
			}

			// could not find matching spqueue/media for update
			if( retVal.size() <= 0 )
			{
				return;
			}

			// check if this exists in the database
			Collection cqColl = getCombinedQueueStaffing(retVal, timeInterval);
			if( cqColl != null && cqColl.size() > 0 )
			{
				updateCombinedQueueStaffing(staff);
			}
			else
			{
				createCombinedQueueStaffing(staff);
			}

		}
		finally
		{
			jdmo.cleanUp();
		}

	}

	static private String getCurrentDBTime()
	{
		Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("GMT"));
		LocalDate date = new LocalDate(cal);
		return JdmoUtil.formatDBString(TimeZoneUtil.toTimestamp(date));	
	}

	/**
	 * create record
	 */
	static public void createCombinedQueueStaffing(CombinedQueueStaffing obj) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();
		try
		{
			StringBuffer pStmt = new StringBuffer();

			pStmt.append( "insert into spcombinedqueuestaffing (SPQUEUEID,TIMEINTERVAL,STAFFING,LASTMODIFIEDAT)  values (" );
			pStmt.append("'");
			pStmt.append( obj.getSPQueueID() );
			pStmt.append("'");
			pStmt.append(",");
			pStmt.append("'");
			pStmt.append(JdmoUtil.formatDBString(TimeZoneUtil.toTimestamp(obj.getTimeInterval())));
			pStmt.append("'");
			pStmt.append(",");
			pStmt.append( obj.getStaffing() );
			pStmt.append(",");
			pStmt.append("'");
			pStmt.append(getCurrentDBTime());
			pStmt.append("'");
			pStmt.append(")");

			
			jdmo.executeCommand(pStmt.toString());
		}
		finally
		{
			jdmo.cleanUp();
		}
	}


	/**
	 * update record
	 */
	static public void updateCombinedQueueStaffing(CombinedQueueStaffing obj) throws JdmoException
	{
		Jdmo jdmo = new Jdmo();
		try
		{
			StringBuffer pStmt = new StringBuffer();
			pStmt.append( "update spcombinedqueuestaffing set staffing = " );
			pStmt.append( obj.getStaffing() );
			pStmt.append( ", lastmodifiedat = " );
			pStmt.append( "'");
			pStmt.append( getCurrentDBTime() );
			pStmt.append( "'");
			pStmt.append(" where spqueueid = '");
			pStmt.append( obj.getSPQueueID() );
			pStmt.append("'");
			pStmt.append(" and timeinterval = ");
			pStmt.append("'");
			pStmt.append(JdmoUtil.formatDBString(TimeZoneUtil.toTimestamp(obj.getTimeInterval())));
			pStmt.append("'");

			jdmo.executeCommand(pStmt.toString());
		}
		finally
		{
			jdmo.cleanUp();
		}
	}

}
