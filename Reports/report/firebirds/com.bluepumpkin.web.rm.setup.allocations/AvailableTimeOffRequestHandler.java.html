public class AvailableTimeOffRequestHandler extends OrganizationRequestHandler {
	private static final Category log =
		Log.initCategory(AvailableTimeOffRequestHandler.class.getName());

	/**
	 * Process Request for the User page
	 *
	 * @param context The Request Context
	 */
	@Override
	public void processRequest(RequestContext context) throws Exception
	{
		// Process requested action
		if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		} else if (isAction(context, RmPageAction.TOIC_INTERVAL_ACTION)) {
			processIntervalAction(context);
		} else if (isAction(context, PageAction.IMPORT)) {
			processImport(context);
		} else if (isAction(context, PageAction.REFRESH) ||isAction(context,RmPageAction.TIMEOFF_ALLOCATION_TYPE_CHANGE_ACTION)) {
			setNextPage(context, TimeOffPoolListRH.EDIT_PAGE);
        } else if (isAction(context, PageAction.CANCEL)) {
			setNextPage(context, TimeOffPoolListRH.LIST_PAGE);
        } else {
			//return to time off pools list page
			setNextPage(context, TimeOffPoolListRH.LIST_PAGE);
		}
	}

	/**
	 * Processes the Save action
	 * @param context RequestContext        The Request context object
	 */
	private void processSave(RequestContext context)
	{
		AvailableTimeOffPageModel model = new AvailableTimeOffPageModel(context);
		model.extractParameters(context.getRequest());

		if (!isAuthorizedForOrg(context, PrivilegeKeys.TOM_EDITAVAILABLETIMEOFF, true))
		{
			return;
		}

		// Store page model in request object
		context.setPageModel(model);

		if (model.validate())
		{
			try
			{
				ID poolID = model.getPoolID();
				TOPool pool = model.getPool();

				// get the pool's waitlist settings from the request
				getWaitlistSettings(context, pool);

				if (poolID == null)
				{
					// create a new pool
					poolID = AvailableTimeOffModelHandler.createTOPool(pool);
					model.setPoolID(poolID);
				} else {
					// update an existing pool
					AvailableTimeOffModelHandler.updateTOPool(pool);
				}

				if (poolID != null) {
					saveAllocations(context, model, pool);
				}

				// trigger auto scan if user selected the option and waitlist is NEWLY enabled
				if (pool.isWaitlistEnabled() && pool.isIsWaitlistscan() && !model.getWlScanEnabled()) {
					triggerWaitlistScan(model.getSelectedIDObject());
				}
			} catch (Exception ex) {
				log.l7dError(RmWebLogBundleKey.ORG_RM_TIME_OFF_CREATE_FAILURE, ex);
				String msgID = RmWebBundleKey.ORG_RM_TIME_OFF_CREATE_FAILURE;
				model.addPageMessage(Message.ERROR_TYPE, msgID, RmWebBundleKey.BUNDLE_NAME);
			}
		}
	}

	private void saveAllocations(RequestContext context, AvailableTimeOffPageModel model, TOPool pool) throws MultiUserException,
			RemoteException, ParseException, BbmException {

		if (model.isIntervalAllocationType()) {
			saveIntervalAllocations(context, model, pool);
		} else {
			// now save the time off allocations for the pool
			CalendarTimeOffDay[] calendarTimeOffDays = model.getCalendarTimeOffDays();
			AvailableTimeOffModelHandler.saveCalendarTimeOffDays(calendarTimeOffDays, pool.getId());
		}
	}

	private void saveIntervalAllocations(RequestContext context, AvailableTimeOffPageModel model, TOPool pool) throws ParseException,
			BbmException, RemoteException, MultiUserException {

		AvailableTimeOffIntervalCalendarPC toicPC = model.getIntervalAllocationPC();
		AvailableTimeOffIntervalCalendarViewModel vm = toicPC.getIntervalAllocationViewModel();
		Map<LocalDate, List<TOIntervalAllocation>> intervalMap = new HashMap<LocalDate, List<TOIntervalAllocation>>();
		Organization org = OrganizationModelHandler.getOrganization(context, pool.getOrganizationId());
		if (vm != null) {
			NumberFormat numberFormat = toicPC.getNumberFormat();
			SimpleDateFormat dateTimeAttributeFormat = AvailableTimeOffIntervalCalendarPC.getDateTimeFormatGmt();
			List<TOIntervalAllocation> intervals = vm.getIntervalAllocations(pool.getID(), dateTimeAttributeFormat, numberFormat);
			intervalMap = TOIntervalCalendarUtil.toLocalDateMap(intervals, org.getTimeZone());
		}
		MonthRange monthRange = toicPC.getMonthRange();
		BlackoutDay[] blackoutDays = toicPC.getBlackoutDays();
		AvailableTimeOffModelHandler.saveTimeOffIntervalCalendar(intervalMap, pool.getID(), org.getTimeZone(), monthRange, blackoutDays);
	}

	private void getWaitlistSettings(RequestContext context, TOPool pool)
	{
		HttpServletRequest request = context.getRequest();
        String key = RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_FEATURE_ENABLE;
        boolean isWaitlistEnabled = request.getParameter(AvailableTimeOffPageModel.getNameKey(key)) != null;
        pool.setWaitlistEnabled(isWaitlistEnabled);

		if (isWaitlistEnabled) {
			boolean isScanEnabled = request.getParameter(AvailableTimeOffPageModel.getNameKey(RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_SCAN_ENABLE)) != null;
			//waitlist settings: save only when enabled
			pool.setIsWaitlistscan(isScanEnabled);
			Pair[] pairArray = new Pair[3];
			pairArray[0] = getWLSortOrderPair(request, "0");
			pairArray[1] = getWLSortOrderPair(request, "1");
			pairArray[2] = getWLSortOrderPair(request, "2");
			pool.setTOWaitlistPriortyOrderSettings(pairArray);
		}
	}

	private Pair<String, String> getWLSortOrderPair(HttpServletRequest request, String index) {
		String column = getParameter(request, AvailableTimeOffPageModel.getNameKey(RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_ORDER_PREFIX + index) + AvailableTimeOffPageModel.COLUMN);
		String order = getParameter(request, AvailableTimeOffPageModel.getNameKey(RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_ORDER_PREFIX + index) + AvailableTimeOffPageModel.ORDER);

		if (!StringUtil.isEmpty(column) && !StringUtil.isEmpty(order)) {
			return new Pair<String, String>(column, order);
		} else {
			return new Pair<String, String>(RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_ORDER_COLUMN_NO_VALUE, RmSettingKey.ORG_RM_SETTINGS_TO_WAITLIST_ORDER_COLUMN_NO_VALUE);
		}
	}

	public void triggerWaitlistScan(ID orgID) throws BbmUpdateException {
        TONotifyMessageClient.scanTOWaitlist(orgID,true, TONotifyMessage.TO_NOTIFY_WAITLIST_SETTING_CHANGE);
	}

	/**
	 * Get a normal html paramater.
	 *
	 * @param request - request
	 * @param key - the parameter name.
	 *
	 * @return String - the html parameter or null.
	 */
	private String getParameter(HttpServletRequest request, String key) {
		key = AvailableTimeOffPageModel.getNameKey(key);
		return request.getParameter(key);
	}

	/**
	 * AvailableTimeOffIntervalCalendarPC.getBlackoutDays returns an array for each day of the month. Remove the null items.
	 *
	 * @param blackoutDays
	 * @return
	 */
	private List<BlackoutDay> getNonNullBlackoutDays(BlackoutDay[] blackoutDays) {
		List<BlackoutDay> results = new ArrayList<BlackoutDay>(blackoutDays.length);
		for (BlackoutDay blackoutDay : blackoutDays) {
			if (blackoutDay != null) {
				results.add(blackoutDay);
			}
		}
		return results;
	}

	/**
	 * Processes the Import action
	 * @param context RequestContext        The Request context object
	 */
	private void processImport(RequestContext context)
	{
		AvailableTimeOffImportPageModel model = new AvailableTimeOffImportPageModel(context);
		model.extractParameters(context.getRequest());

		if (!isAuthorizedForOrg(context, PrivilegeKeys.TOM_EDITAVAILABLETIMEOFF, true))
		{
			return;
		}

		// Store page model in request object
		context.setPageModel(model);
		try
		{
			model.processUpload(context);
		}
		catch (Exception e)
		{
			log.error("", e);
		}
	}

	private void processIntervalAction(RequestContext context) {
		AvailableTimeOffPageModel model = new AvailableTimeOffPageModel(context);
		model.extractParameters(context.getRequest());

		if (!isAuthorizedForOrg(context, PrivilegeKeys.TOM_EDITAVAILABLETIMEOFF, true)) {
			return;
		}

		// Store page model in request object
		context.setPageModel(model);
		// always set it to dirty because submitting from the popup clears this value.
		model.setPageDirty(true);

		if (!model.validate()) {
			return;
		}
		try {
			processIntervalAction(context, model);
		} catch (Exception ex) {
			log.l7dError(RmWebLogBundleKey.ORG_RM_TIME_OFF_CREATE_FAILURE, ex);
			String msgID = RmWebBundleKey.TIMEOFF_POOL_FAILED_TO_SET_INTERVAL;
			model.addPageMessage(Message.ERROR_TYPE, msgID, RmWebBundleKey.BUNDLE_NAME);
		}
	}

	private void processIntervalAction(RequestContext context, AvailableTimeOffPageModel model) throws BbmException, RemoteException,
			ParseException, CloneNotSupportedException {

		TOPool pool = model.getPool();
		Organization org = OrganizationModelHandler.getOrganization(context, pool.getOrganizationId());
		AvailableTimeOffIntervalCalendarPC toicPC = model.getIntervalAllocationPC();
		SimpleDateFormat dateTimeAttributeFormat = AvailableTimeOffIntervalCalendarPC.getDateTimeFormatGmt();

		// view model -> model
		AvailableTimeOffIntervalCalendarViewModel viewModel = toicPC.getIntervalAllocationViewModel();
		if (viewModel == null) {
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.TIMEOFF_POOL_FAILED_TO_SET_INTERVAL, RmWebBundleKey.BUNDLE_NAME);
			return;
		}
		List<TOIntervalAllocation> intervals = viewModel.getIntervalAllocations(model.getPoolID(), dateTimeAttributeFormat,
				toicPC.getNumberFormat());
		if (intervals == null) {
			model.addPageMessage(Message.ERROR_TYPE, RmWebBundleKey.TIMEOFF_POOL_FAILED_TO_SET_INTERVAL, RmWebBundleKey.BUNDLE_NAME);
			return;
		}

		List<BlackoutDay> blackoutDays = getNonNullBlackoutDays(toicPC.getBlackoutDays());

		String action = toicPC.getIntervalAllocationMode();
		log.debug("processIntervalAction: " + action);
		if (PageAction.CREATE_NEW.equals(action) || PageAction.EDIT.equals(action)) {
			// popup params -> model
			TOIntervalAllocation interval = toicPC.getIntervalForAction(org, pool.getID());
			if (interval != null) {
				if (PageAction.CREATE_NEW.equals(action)) {
					TOIntervalCalendarUtil.addInterval(intervals, interval);
				} else {
					log.debug("processIntervalAction: IntervalAllocationEditIntervalStartTime: "
							+ toicPC.getIntervalAllocationEditIntervalStartTime());

					Date editStartTime = dateTimeAttributeFormat.parse(toicPC.getIntervalAllocationEditIntervalStartTime());
					TOIntervalCalendarUtil.editInterval(intervals, interval, editStartTime, org.getTimeZone(),
							org.getDayBoundaryOffset());
				}
			}
		} else if (PageAction.DELETE.equals(action)) {
			Date startTime = dateTimeAttributeFormat.parse(toicPC.getIntervalAllocationStartTime());
			log.debug("processIntervalAction: Delete at IntervalAllocationStartTime: " + toicPC.getIntervalAllocationStartTime());
			TOIntervalCalendarUtil.removeInterval(intervals, startTime, org.getTimeZone(), org.getDayBoundaryOffset());
		}

		TOIntervalCalendar intervalCalendar = viewModel.getTOIntervalCalendar(pool.getID(), org, intervals, blackoutDays,
				dateTimeAttributeFormat);
		toicPC.setTimeOffIntervalCalendar(intervalCalendar);
	}
