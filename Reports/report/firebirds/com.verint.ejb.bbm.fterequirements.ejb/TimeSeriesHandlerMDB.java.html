package com.verint.ejb.bbm.fterequirements.ejb;

import java.util.Collection;

import javax.ejb.CreateException;
import javax.ejb.EJBException;
import javax.ejb.MessageDrivenBean;
import javax.ejb.MessageDrivenContext;
import javax.jms.Message;
import javax.jms.MessageListener;
import javax.jms.ObjectMessage;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.common.util.StringUtil;
import com.bluepumpkin.ejb.bbm.Log;
import com.bluepumpkin.ejb.bbm.l10n.BbmEjbLogBundleKey;
import com.verint.ejb.bbm.fterequirements.TimeSeriesMessage;
import com.verint.ejb.bbm.fterequirements.model.RequiredTimeSeries;
import com.verint.ejb.wfm.WfmManagerFactory;

/**
 * This message driven bean handles TimeSeriesMessages.  These messages are created and
 * sent to a jms queue after forecasts, service goals, or shrinkage is recalculated.  When
 * a message is received, it kicks off a recalculation of FTE Requirements for the set of SPQueues
 * that had their forecast, service goals, or shrinkage modified.
 */
@SuppressWarnings("serial")
public class TimeSeriesHandlerMDB implements MessageDrivenBean, MessageListener {
	MessageDrivenContext m_messageDrivenContext;
	private static Category m_cat = Log.initCategory(TimeSeriesHandlerMDB.class.getName());
	private FteRequirementsManager m_fteRequirementsManager;
	private FteRequirementsManager m_fteRequirementsManagerWIF;
	
	public void ejbCreate() throws CreateException {
		try {
			m_fteRequirementsManager = WfmManagerFactory.getFteRequirementsManager(false);
			m_fteRequirementsManagerWIF = WfmManagerFactory.getFteRequirementsManager(true);
		}
		catch (Exception e) {
			Object[] param = new Object[] { "ejbCreate", TimeSeriesHandlerMDB.class.getName() };
			m_cat.l7dError(BbmEjbLogBundleKey.UNKNOWN_EXCEPTION_MF, param, e);
			throw new CreateException();
		}
	}
	
	public void ejbRemove() throws EJBException {
		// TODO Auto-generated method stub
	}

	public void setMessageDrivenContext(MessageDrivenContext messageDrivenContext)
			throws EJBException {
		m_messageDrivenContext = messageDrivenContext;
	}

	public void onMessage(Message msg) {
		TimeSeriesMessage msgObj = null;
		try {
			if (msg instanceof ObjectMessage) {
				Object obj = ((ObjectMessage) msg).getObject();
				m_cat.debug(obj);
				if (obj instanceof TimeSeriesMessage) {
					msgObj = (TimeSeriesMessage) (obj);
					Collection<RequiredTimeSeries> fteTimeSeries = 
						getFteRequirementsManager(msgObj.isWhatIfMode()).calculateFteRequirementsForSPQueues(msgObj.getSPQueueIDs(),
								msgObj.getForecastsBySPQueueID());
					getFteRequirementsManager(msgObj.isWhatIfMode()).createAndUpdateRequiredTimeSeries(fteTimeSeries);
					//Debug log message to indicate that the given sp queues were processed
					m_cat.l7dDebug(BbmEjbLogBundleKey.TIME_SERIES_HANDLER_FINISHED_PROCESSING_SPQUEUES,
							new Object[] { StringUtil.createDelimitedString(msgObj.getSPQueueIDs(), StringUtil.DELIM) });
				}
			} else {
				//If this is something other than a TimeSeriesMessage object, display a warning
				m_cat.l7dWarn(BbmEjbLogBundleKey.TIME_SERIES_HANDLER_EXPECTED_TIME_SERIES_MSG,
						new Object[] { msg.getClass().getName() }); 
			}
		} catch(Exception e) {
			Object[] param = new Object[]{"onMessage", TimeSeriesHandlerMDB.class.getName()};
			m_cat.l7dError(BbmEjbLogBundleKey.UNKNOWN_EXCEPTION_MF, param, e);
		}
	}
	
	private FteRequirementsManager getFteRequirementsManager(boolean isWhatIfMode) {
		if (isWhatIfMode) return m_fteRequirementsManagerWIF;
		return m_fteRequirementsManager;
	}
}
