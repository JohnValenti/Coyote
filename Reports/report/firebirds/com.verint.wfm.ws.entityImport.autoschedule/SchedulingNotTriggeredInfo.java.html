package com.verint.wfm.ws.entityImport.autoschedule;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

import javax.xml.bind.annotation.XmlType;

import com.verint.wfm.ws.entityImport.schedulingPeriod.model.SchedulingPeriodDTO.SchedulingPeriodExternalNaturalKey;

@XmlType(propOrder = { "messages", "SPExternalNaturalKey" })
public class SchedulingNotTriggeredInfo implements Serializable {
	/**
	 * 
	 */
	private static final long serialVersionUID = 5285462381778361193L;
	private SchedulingPeriodExternalNaturalKey m_spExternalNaturalKey = null;
	private List<String> m_messages = new ArrayList<String>();

	public SchedulingNotTriggeredInfo() {

	}

	public SchedulingNotTriggeredInfo(
			SchedulingPeriodExternalNaturalKey spExternalNaturalKey) {
		m_spExternalNaturalKey = spExternalNaturalKey;
	}

	public SchedulingPeriodExternalNaturalKey getSPExternalNaturalKey() {
		return m_spExternalNaturalKey;
	}

	public void addNoSuchSPExternalIdMessage() {
		getMessages().add(
				String.format("The scheduling period   %s does not exist.",
						m_spExternalNaturalKey.toString()));
	}

	public void addReqestNotCreatedMessage() {
		getMessages()
				.add(String
						.format(" The Scheduling Request for the scheduling period  %s could not be created. Could not auto schedule the scheduling period ",
								m_spExternalNaturalKey.toString()));
	}
	
	public void addAutoSchedulingNotTriggeredMessage() {
		getMessages()
		.add(String
				.format(" Could not schedule request due to an infrastructure failure for the scheduling period  %s.",
						m_spExternalNaturalKey.toString()));
	}

	public List<String> getMessages() {
		return m_messages;
	}

	public void setMessages(List<String> messages) {
		// Do nothing
	}

	/**
	 * 
	 */
	public void setSPExternalNaturalKey(
			SchedulingPeriodExternalNaturalKey schedulingPeriodExternalNaturalKey) {
	}

}
