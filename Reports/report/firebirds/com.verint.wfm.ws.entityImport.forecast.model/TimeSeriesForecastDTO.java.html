package com.verint.wfm.ws.entityImport.forecast.model;

import java.util.List;

import javax.xml.bind.annotation.XmlType;

@XmlType(name = "TimeSeriesForecastDTO", propOrder = { "timeSeriesTypes", "timeSeriesEncoded" })
public class TimeSeriesForecastDTO extends AbstractForecastDTO {

	public static final String TIME_SERIES_TYPE_AHT = "AHT";
	public static final String TIME_SERIES_TYPE_CV = "CV";
	public static final String TIME_SERIES_TYPE_FTE = "FTE";

	protected List<String> m_timeSeriesTypes;
	protected List<String> m_timeSeriesEncoded;

	public TimeSeriesForecastDTO() {
		super();
	}

	public List<String> getTimeSeriesTypes() {
		return m_timeSeriesTypes;
	}

	public void setTimeSeriesTypes(List<String> timeSeriesTypes) {
		m_timeSeriesTypes = timeSeriesTypes;
	}

	public List<String> getTimeSeriesEncoded() {
		return m_timeSeriesEncoded;
	}

	public void setTimeSeriesEncoded(List<String> timeSeriesEncoded) {
		m_timeSeriesEncoded = timeSeriesEncoded;
	}

	// override of superclass methods needed because of flaw in jax-ws
	public ForecastExternalNaturalKey getForecastExternalNaturalKey() {
		return m_forecastExternalNaturalKey;
	}

	public void setForecastExternalNaturalKey(ForecastExternalNaturalKey forecastExternalNaturalKey) {
		m_forecastExternalNaturalKey = forecastExternalNaturalKey;
	}

	public String getDescription() {
		return m_description;
	}

	public void setDescription(String description) {
		m_description = description;
	}
}
