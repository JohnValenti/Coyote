package com.bluepumpkin.web.rm.setup.allocations.interval;

import java.text.NumberFormat;
import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Collection;
import java.util.Date;
import java.util.List;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.TimeRange;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendar;
import com.bluepumpkin.ejb.rm.requests.timeoff.model.TOIntervalCalendarInterval;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.BlackoutDay;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOIntervalAllocation;
import com.bluepumpkin.ejb.rm.setup.calendartimeoff.model.TOIntervalAllocationFactory;
import com.bluepumpkin.ejb.rm.util.TOIntervalCalendarUtil;

public class AvailableTimeOffIntervalCalendarViewModel {
	private List<AvailableTimeOffIntervalCalendarInterval> intervals = new ArrayList<AvailableTimeOffIntervalCalendarInterval>();
	private String startTime;
	private String endTime;

	// ================================================================================
	// CONSTRUCTION
	// ================================================================================

	/**
	 * Serialization Constructor.
	 */
	public AvailableTimeOffIntervalCalendarViewModel() { // NOSONAR

	}

	public AvailableTimeOffIntervalCalendarViewModel(TOIntervalCalendar toic, SimpleDateFormat dateTimeAttributeFormat,
			NumberFormat numberFormat) {
		startTime = dateTimeAttributeFormat.format(toic.getStartTime());
		endTime = dateTimeAttributeFormat.format(toic.getEndTime());

		for (TOIntervalCalendarInterval interval : toic.getIntervals()) {
			intervals.add(new AvailableTimeOffIntervalCalendarInterval(interval, dateTimeAttributeFormat, numberFormat));
		}
	}

	// ================================================================================
	// PROPERTIES
	// ================================================================================

	public String getStartTime() {
		return startTime;
	}

	public void setStartTime(String startTime) {
		this.startTime = startTime;
	}

	public String getEndTime() {
		return endTime;
	}

	public void setEndTime(String endTime) {
		this.endTime = endTime;
	}

	public List<AvailableTimeOffIntervalCalendarInterval> getIntervals() {
		return intervals;
	}

	public void setIntervals(List<AvailableTimeOffIntervalCalendarInterval> intervals) {
		this.intervals = intervals;
	}

	// ================================================================================
	// INTERFACE
	// ================================================================================

	/**
	 * Gets a list of interval allocations built from the view model.
	 *
	 * @throws ParseException
	 */
	public List<TOIntervalAllocation> getIntervalAllocations(ID poolID, SimpleDateFormat dateTimeAttributeFormat, NumberFormat numberFormat)
			throws ParseException {
		List<TOIntervalAllocation> toIntervals = new ArrayList<TOIntervalAllocation>(intervals.size());
		for (AvailableTimeOffIntervalCalendarInterval interval : intervals) {
			toIntervals.add(interval.toIntervalAllocation(poolID, dateTimeAttributeFormat, numberFormat));
		}
		return toIntervals;
	}

	public TOIntervalCalendar getTOIntervalCalendar(ID toPoolID, Organization org, List<TOIntervalAllocation> dbIntervals,
			Collection<BlackoutDay> blackoutDays, SimpleDateFormat dateTimeAttributeFormat) throws ParseException {

		Date startDateTime = dateTimeAttributeFormat.parse(startTime);
		Date endDateTime = dateTimeAttributeFormat.parse(endTime);

		TOIntervalCalendar intervalCalendar = new TOIntervalCalendar(startDateTime, endDateTime, org.getTimeZone());

		// Fill in the Gaps.
		List<TOIntervalAllocation> continuousIntervals = TOIntervalCalendarUtil.getIntervals(startDateTime, endDateTime, dbIntervals,
				org.getTimeZone(), org.getDayBoundaryOffset(), new TOIntervalAllocationFactory());

		// Wrap the value object in our model.
		TimeRange toCalRange = new TimeRange(startDateTime, endDateTime);
		intervalCalendar.addIntervals(TOIntervalCalendarUtil.wrapIntervals(continuousIntervals, toCalRange, toPoolID));

		intervalCalendar.splitIntoDays();

		intervalCalendar.addBlackoutDays(blackoutDays, org.getTimeZone(), org.getDayBoundaryOffset());

		return intervalCalendar;
	}

	@Override
	public String toString() {
		return String.format("Intervals: %s", intervals);
	}

}
