public class ExceptionCategoryRequestHandler extends RequestHandler {
	private static final Category log = Log.initCategory(ExceptionCategoryRequestHandler.class.getName());

	private static final String REQUIRED_PRIVILEGE = PrivilegeKeys.CORE_CONFIGUREHOLIDAYS;
	private static final String LIST_PAGE = "listPage";
	private static final String FORM_PAGE = "editPage";

	/**
	 * Process Request for the User page
	 *
	 * @param context The Request Context
	 */
	public void processRequest(RequestContext context) throws Exception {
		// Process requested action
		if (isAction(context, PageAction.ADD) ||
				isAction(context, PageAction.EDIT)) {
			goToFormPage(context);
		} else if (isAction(context, PageAction.DELETE)) {
			processDelete(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processSave(context);
		} else if (isAction(context, PageAction.DONE) ||
				isAction(context, PageAction.NEW_ID)) {
			setNextPage(context, LIST_PAGE);
		} else {
			setNextPage(context, LIST_PAGE);
		}
	}

	/**
	 * Go To Form Page
	 */
	private void goToFormPage(RequestContext context) {
		if (isAuthorized(context, REQUIRED_PRIVILEGE)) {
			setNextPage(context, FORM_PAGE);
		} else {
			setNextPage(context, LIST_PAGE);
		}
	}

	/**
	 * Process Delete Action
	 */
	private void processDelete(RequestContext context) {
		setNextPage(context, LIST_PAGE);

		// Security check
		if (!isAuthorized(context, REQUIRED_PRIVILEGE))	return;

		// Extract Parameter
		ExceptionCategoryListPageModel model = new ExceptionCategoryListPageModel(context);
		model.extractParameters(context.getRequest());

		// Process Delete
		ID[] ids = model.getSelectedHolidayIDs();

		// Catch Model
		context.setPageModel(model);
	}


	/**
	 * Process Save Action
	 */
	private void processSave(RequestContext context) throws RemoteException {
		setNextPage(context, FORM_PAGE);

		// Security check
		if (!isAuthorized(context, REQUIRED_PRIVILEGE))	return;

		ExceptionCategoryEditPageModel model = new ExceptionCategoryEditPageModel(context);

		// Retrieve Information and validate it
		model.extractParameters(context.getRequest());
		model.validate();

		if (!model.hasError()) {
			RTAAExceptionCategory exceptionCategory = model.getExceptionCategory();

			try {
				if (model.isNew()) {
					// Create New Job Title
					ID newID = ExceptionCategoryModelHandler.createExceptionCategory(context, exceptionCategory);

					// Prepare List Model for List page so that newly created Record is selected
					ExceptionCategoryListPageModel listModel = new ExceptionCategoryListPageModel(context);
					listModel.setPerformActionOnIDs(new ID[] {newID});
					context.setPageModel(listModel);
				} else {
					// Update Information to back end
					ExceptionCategoryModelHandler.saveExceptionCategory(context, exceptionCategory);
				}
			} catch (BbmCreateDuplicateKeyException ex) {
				String msgID = BbmWebBundleKey.ORG_HOLIDAY_ERROR_SAVE_DUPLICATED_KEY;
				log.l7dError(msgID, ex);
				model.addPageMessage(Message.ERROR_TYPE, msgID, BbmWebBundleKey.BUNDLE_NAME);
			} catch (BbmUpdateDuplicateKeyException ex) {
				String msgID = BbmWebBundleKey.ORG_HOLIDAY_ERROR_SAVE_DUPLICATED_KEY;
				log.l7dError(msgID, ex);
				model.addPageMessage(Message.ERROR_TYPE, msgID, BbmWebBundleKey.BUNDLE_NAME);
			} catch (BPException ex) {
				String msgID = BbmWebBundleKey.ORG_HOLIDAY_ERROR_SAVE;
				log.l7dError(msgID, ex);
				model.addPageMessage(Message.ERROR_TYPE, msgID,	BbmWebBundleKey.BUNDLE_NAME);
			}
		}

		// Only proceed to List page if page model does not have any errors
		if (model.hasError()) {
			context.setPageModel(model);
			setNextPage(context, FORM_PAGE);
		} else {
			setNextPage(context, LIST_PAGE);
		}
	}
}
