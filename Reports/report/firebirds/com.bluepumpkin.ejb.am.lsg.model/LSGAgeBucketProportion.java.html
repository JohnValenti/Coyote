package com.bluepumpkin.ejb.am.lsg.model;

import java.io.Serializable;
import java.util.Date;

import org.codehaus.jackson.annotate.JsonIgnoreProperties;

import com.bluepumpkin.common.datatypes.ID;

/**
 * This class stores the proportion for the LSGAgeBucket for the given SPQueueID
 *
 * @author Jinal Shah
 */
@JsonIgnoreProperties({"id","lsgAgeBucketId","lsgTempateSpQueueConfigId"})
public class LSGAgeBucketProportion implements Serializable {

    private static final long serialVersionUID = -967355946852598859L;

    private ID id;
    private ID lsgAgeBucketId;
    private ID lsgTempateSpQueueConfigId;
    private float proportion;
    private Date modifiedDate;

    /**
     * @return the ID for the LSGAgeBucketProportion
     */
    public ID getId() {
        return id;
    }

    /**
     * @param id
     */
    public void setId(ID id) {
        this.id = id;
    }

    /**
     * @return LSGAgeBucketId
     */
    public ID getLsgAgeBucketId() {
        return lsgAgeBucketId;
    }

    /**
     * @param lsgAgeBucketId
     */
    public void setLsgAgeBucketId(ID lsgAgeBucketId) {
        this.lsgAgeBucketId = lsgAgeBucketId;
    }

    /**
     * @return the ID for the LSG Template and SPQueueConfig
     */
    public ID getLsgTempateSpQueueConfigId() {
        return lsgTempateSpQueueConfigId;
    }

    /**
     * @param lsgTempateSpQueueConfigId
     */
    public void setLsgTempateSpQueueConfigId(ID lsgTempateSpQueueConfigId) {
        this.lsgTempateSpQueueConfigId = lsgTempateSpQueueConfigId;
    }

    /**
     * @return the proportion of the LSGAgeBucket
     */
    public float getProportion() {
        return proportion;
    }

    /**
     * @param proportion
     */
    public void setProportion(float proportion) {
        this.proportion = proportion;
    }

    /**
     * @return the modified date when the proportion was last modified
     */
    public Date getModifiedDate() {
        return modifiedDate;
    }

    /**
     * @param modifiedDate
     */
    public void setModifiedDate(Date modifiedDate) {
        this.modifiedDate = modifiedDate;
    }
}
