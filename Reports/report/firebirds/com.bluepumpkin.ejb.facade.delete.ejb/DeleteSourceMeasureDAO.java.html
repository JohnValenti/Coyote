import java.util.Collection;

import java.util.Date;

import com.bluepumpkin.common.datatypes.*;

import com.bluepumpkin.common.localization.Localizer;
import com.witness.ejb.bbm.delete.DeleteException;
import com.witness.ejb.bbm.delete.DeleteableObjectPlugin;
import com.witness.ejb.bbm.delete.ejb.DAOUtil;
import com.witness.ejb.bbm.delete.model.*;
import com.bluepumpkin.ejb.facade.l10n.FacadeEjbBundleKey;


public class DeleteSourceMeasureDAO implements DeleteableObjectPlugin {
  
  /**
	 * 
	 */
	private static final long serialVersionUID = 1L;

public DeleteSourceMeasureDAO() {
  }
  
  /**
   * check if sourceMeasure is linked to kpi formula by checking following link tables
   *       KpiSourceMeasure
   *
   * @return DeletionResult
   *    if sourceMeasure is linked to KpiSourceMeasure tables
   *       set ResultCode to CONFIRM_DELETE, and fill DeleteResult.m_colLinkedKpis
   *       with names of linked kpi sourcemeasures
   *    else
   *       set ResultCode to DELETED
   *
   */
  private DeletionResult checkLinkedObjects(ID idSourceMeasure, Jdmo dmo) throws Exception {
    DeletionResult dResult = new DeletionResult();
    Collection colLinkedKpis = new ArrayList();
    
    // check if link to kpi objects
    StringBuffer strSQL = new StringBuffer();
    strSQL.append("select distinct A.KPIID AS ID, B.NAME from KPISOURCEMEASURE A, KPI B where A.KPIID=B.ID and ");
    strSQL.append("A.SOURCEMEASURESYSTEMID in (select distinct ID from SOURCEMEASURESYSTEM where SOURCEMEASUREID = ");
    strSQL.append(idSourceMeasure).append(")");
    
    DAOUtil.loadLinkedObjects(colLinkedKpis, new String(strSQL), BPObjectInfo.BPOBJECT_KPI,
			FacadeEjbBundleKey.BUNDLE_NAME, FacadeEjbBundleKey.OBJECT_KPI, dmo);
    
    if (colLinkedKpis.size() > 0) {
      dResult.setResultCode(DeletionResult.CONFIRM_DELETE);
      dResult.setLinkedObjects(colLinkedKpis);
      return dResult;
    }
    
    // not linked to any object, marked it as deleted
    dResult.setResultCode(DeletionResult.DELETED);
    return dResult;
  }
  
  /**
   *  delete sourceMeasure implements the following algorithms
   *  1) call checkLinkedObjects() to check if sourceMeasure is linked to any other objects
   *  2) if sourceMeasure is linked to kpi
   *        just return DeletionResult, since the deletion need user confirmation
   *     else, delete it
   *
   */
  public DeletionResult deleteObject(ID idSourceMeasure,
		  Date dtDeleteOnDate, Localizer localizer, Object objCustomDefined)
		  throws DeleteException {
    Jdmo dmo = new Jdmo(false);
    try {
      DeletionResult dResult = checkLinkedObjects(idSourceMeasure, dmo);
      switch (dResult.getResultCode()) {
        case DeletionResult.DELETED:
          // delete the object
          DAOUtil.deleteObjectCascade(idSourceMeasure, "SOURCEMEASURE", dmo);
          break;
          
        default:
          // do nothing if prohibit delete or need substitute
          break;
      }
      
      return dResult;
    }
    catch(Exception e) {
      throw new DeleteException(e);
    }
    finally {
      dmo.cleanUp();
    }
  }
  
  
  /**
   *  force delete sourceMeasure implements the following algorithms
   *  Cascade delete it.
   *
   */
  public void forceDeleteObject(ID idSourceMeasure) throws DeleteException {
    Jdmo dmo = new Jdmo(false);
    try {
      DAOUtil.deleteObjectCascade(idSourceMeasure, "SOURCEMEASURE", dmo);
    }
    catch(Exception e) {
      throw new DeleteException(e);
    }
    finally {
      dmo.cleanUp();
    }
  }

	/**
	 * substitute the object with another object in object relationship
	 * 
	 * @param objID
	 *            the id of the object being substituted
	 * @param substituteObjID
	 *            the id of the substitute object
	 * @param dtSubstituteDate
	 *            the date where the substitution occurred
	 */
	public void substituteObject(ID objID, ID substituteObjID,
			Date dtSubstituteDate) throws DeleteException {
	}

	/**
	 * Converts an object type to a localized string
	 * 
	 * @param localizer The localizer
	 * @param isPlural Specify whether display is for multiple object 
	 * @return a localized object type string
	 */
	public String convertType(Localizer localizer, boolean isPlural) {
		String rbKey = isPlural ? FacadeEjbBundleKey.OBJECT_SOURCE_MEASURE
				: FacadeEjbBundleKey.OBJECT_SOURCE_MEASURE;
		return localizer.i18n(FacadeEjbBundleKey.BUNDLE_NAME, rbKey);
	}

}
