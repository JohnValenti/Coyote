package com.bluepumpkin.ejb.rm.requests.timeoff.validation;

import com.bluepumpkin.common.logging.Category;

/**
 * "Timeoff Interval Allocation Published Status Rule" validation rule for time off requests.
 * For activity types of type "interval allocation," verify if each TOChoice date range has:
 * 1) Completely published schedule period
 * 2) Completely unpublished schedule period
 * 3) Partially published/unpublished (i.e., at least one day is unpublished)
 *
 * If the result is 3, a hard validation error is raised.
 *
 **/

public class TimeoffIntervalAllocationPublishedStatusRule implements Validator {

    private static final Category LOG = Log.initCategory( TimeoffIntervalAllocationPublishedStatusRule.class.getName());


    @Override
    public ValidationResult validate(Validatable validatable) throws Exception {

        String methodName = "validate";
        if (LOG.isDebugEnabled()){
        	LOG.debug(RmUtil.dumpEnterMethod(methodName, validatable));
        }

        ValidationResult result = null;

        TORequest toRequest = (TORequest)validatable;

        // If we are not using the interval pool, no need to validate
        boolean checkInvervalPool = isUseIntervalTOPool(toRequest);
        if(!checkInvervalPool){
        	return result;
        }

		PublishingPeriodStatus publishingPeriodStatus = initPublishedPeriodStatusForTimeoffRequest(toRequest);
		ID empId = toRequest.getEmployeeID();

		// for each TOChoice
        @SuppressWarnings("unchecked")
		List<TOChoice> choiceList = toRequest.getRequestChoiceList();
        for(TOChoice toChoice : choiceList){
            if( toChoice== null){
            	continue;
            }

            if(!validatePublishedStatusGood(publishingPeriodStatus, empId, toChoice)){
            	result = ValidationUtil
	    			.setHardValidationResult(toChoice,
	    				RmEjbBundleKey.TIMEOFFINTERVALALLOCATIONSEGMENT_CHOICE_SCHEDULES_PARTLY_PUBLISHED,
	    				toChoice.getStartDate(), this.getClass().getName());

            }
        }

        if (LOG.isDebugEnabled()) {
        	LOG.debug(RmUtil.dumpExitMethod(methodName, result));
        }
        return result;
    }

	protected boolean validatePublishedStatusGood(PublishingPeriodStatus publishingPeriodStatus, ID empId,
			TOChoice toChoice) {
		return TimeOffIntervalAllocationUtil.validatePublishedStatusGood(publishingPeriodStatus, empId, toChoice);
	}

	protected PublishingPeriodStatus initPublishedPeriodStatusForTimeoffRequest(TORequest toRequest)
			throws BbmFinderException {
		return TimeOffIntervalAllocationUtil
				.initPublishedPeriodStatusForTimeoffRequest(toRequest);
	}

    protected boolean isUseIntervalTOPool(TORequest toRequest) throws Exception {
        return TORequestUtil.isUseIntervalTOPool(toRequest);
    }

}
