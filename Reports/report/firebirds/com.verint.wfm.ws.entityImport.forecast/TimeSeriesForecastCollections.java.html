package com.verint.wfm.ws.entityImport.forecast;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.campaign.model.SPQueue;
import com.bluepumpkin.ejb.bbm.campaign.model.SchedulingPeriod;
import com.bluepumpkin.ejb.bbm.timeseries.model.ForecastTraceCube;
import com.verint.ejb.bbm.forecast.model.ForecastProfile;
import com.verint.wfm.ws.entityImport.forecast.ForecastServiceDelegate.CampaignSPForecastWrapperData;
import com.verint.wfm.ws.entityImport.forecast.ForecastServiceDelegate.TimeSeriesForecastWrapper;
import com.verint.wfm.ws.entityImport.forecast.model.TimeSeriesForecastDTO;

class TimeSeriesForecastCollections {

	private Collection<TimeSeriesForecastDTO> timeSeriesForecastDTOs;
	private List<ForecastNotPersistedInfo> forecastNotPersistedInfos;

	private Map<ID, Map<ID, Map<ID, ForecastTraceCube>>> immediateCubeByIDsTimezoneToSPToSPQueue;
	// immediateForecastProfilesBySP are the immediate forecasts use the trace cube to represent time
	// series of AHT and CV, typically every 15 minutes (96 instances per forecast day)
	private Map<ID, List<ForecastProfile>> immediateForecastProfilesBySPID;

	private Map<ID, Map<ID, Map<ID, ForecastTraceCube>>> deferredCubeByIDsTimezoneToSPToSPQueue;
	// deferredForecastProfilesBySP are the Deferred forecasts have the same data structure as immediate ones
	private Map<ID, List<ForecastProfile>> deferredForecastProfilesBySPID;
	private Map<ID, List<SPQueue>> deferredSPQueuesBySPID;

	private Map<ID, SchedulingPeriod> spById;
	private CampaignSPForecastWrapperData<TimeSeriesForecastWrapper> campaignSPForecastWrapperData;

	TimeSeriesForecastCollections(Collection<TimeSeriesForecastDTO> pTimeSeriesForecastDTOs) {
		this.timeSeriesForecastDTOs = pTimeSeriesForecastDTOs;
		this.forecastNotPersistedInfos = new ArrayList<ForecastNotPersistedInfo>();
		this.immediateCubeByIDsTimezoneToSPToSPQueue = new HashMap<ID, Map<ID, Map<ID, ForecastTraceCube>>>();
		this.immediateForecastProfilesBySPID = new HashMap<ID, List<ForecastProfile>>();
		this.deferredCubeByIDsTimezoneToSPToSPQueue = new HashMap<ID, Map<ID, Map<ID, ForecastTraceCube>>>();
		this.deferredForecastProfilesBySPID = new HashMap<ID, List<ForecastProfile>>();
		this.deferredSPQueuesBySPID = new HashMap<ID, List<SPQueue>>();
		this.spById = new HashMap<ID, SchedulingPeriod>();
	}

	Collection<TimeSeriesForecastDTO> getTimeSeriesForecastDTOs() {
		return timeSeriesForecastDTOs;
	}

	Map<ID, Map<ID, Map<ID, ForecastTraceCube>>> getImmediateCubeByIDsTimezoneToSPToSPQueue() {
		return immediateCubeByIDsTimezoneToSPToSPQueue;
	}

	Map<ID, List<ForecastProfile>> getImmediateForecastProfilesBySPID() {
		return immediateForecastProfilesBySPID;
	}

	Map<ID, Map<ID, Map<ID, ForecastTraceCube>>> getDeferredCubeByIDsTimezoneToSPToSPQueue() {
		return deferredCubeByIDsTimezoneToSPToSPQueue;
	}

	Map<ID, List<ForecastProfile>> getDeferredForecastProfilesBySPID() {
		return deferredForecastProfilesBySPID;
	}

	Map<ID, List<SPQueue>> getDeferredSPQueuesBySPID() {
		return deferredSPQueuesBySPID;
	}

	Map<ID, SchedulingPeriod> getSpById() {
		return spById;
	}

	List<ForecastNotPersistedInfo> getForecastNotPersistedInfos() {
		return forecastNotPersistedInfos;
	}

	CampaignSPForecastWrapperData<TimeSeriesForecastWrapper> getCampaignSPForecastWrapperData() { return campaignSPForecastWrapperData; }

	void setCampaignSPForecastWrapperData(CampaignSPForecastWrapperData<TimeSeriesForecastWrapper> campaignSPForecastWrapperData) {
		this.campaignSPForecastWrapperData = campaignSPForecastWrapperData;
	}
}
