package com.bluepumpkin.ejb.rm.requests.common.model;

import com.bluepumpkin.common.datatypes.ID;

/**
 * Holds the (field index, field value) pair to describe
 * the value of a field in {@link com.bluepumpkin.ejb.bbm.vo.ValueObjectBase ValueObjectBase}  
 *
 */
public class FieldDescriptor {
    private int m_fieldID;
    
//    private int m_valInt;
    private Object m_valObj;
    
    public FieldDescriptor(int fieldID, int value)  {
        m_fieldID = fieldID;
        //TODO: could be made more efficient by using int m_valInt, instead of Object m_valObj. 
        m_valObj = new Integer(value);
    }
    
    public FieldDescriptor(int fieldID, String value)  {
        m_fieldID = fieldID;
        m_valObj = value;
    }
    
    public FieldDescriptor(int fieldID, ID value)  {
        m_fieldID = fieldID;
        m_valObj = value;
    }
    
    /**
     * @return
     */
    public int getFieldID() {
        return m_fieldID;
    }

    /**
     * @return
     */
    public ID getValID() {
        return (ID) m_valObj;
    }

    /**
     * @return
     */
    public int getValInt() {
        if (m_valObj == null) throw new IllegalArgumentException();
        
        return ((Integer)m_valObj).intValue();
    }

    /**
     * @return
     */
    public String getValString() {
        return (String) m_valObj;
    }
}
