	/**
	 * Store Selected DataSource ID
	 *
	 * NOTE: Store accross session
	 */
	public static void setSelectedQueueID(RequestContext context, String dsID) {
		boolean persist = !getSelectedQueueID(context).equals(dsID);
		if (!persist) return;

		if (dsID==null) dsID="";
		context.setUserProperty(SELECTED_QUEUE_ID, dsID, persist);
	}

	/**
	 * Return Selected DataSource ID
	 *
	 * NOTE: Store accross session
	 */
	public static String getSelectedQueueID(RequestContext context) {
		String dsID = context.getUserProperty(SELECTED_QUEUE_ID);
		return (dsID==null)?"":dsID;
	}

	/**
	 * Return Queues filtered by Media ID
	 */
	public static Collection filterQueuesByMediaID(Collection queues, ID mediaID) {
		if (mediaID == null) return queues;

		List newQs = new ArrayList(queues.size());
		for (Iterator it = queues.iterator(); it.hasNext();) {
			Queue q = (Queue) it.next();
			if (mediaID.equals(q.getMediaID())) newQs.add(q);
		}

		return newQs;
	}
	
	/**
	 * Return Queues filtered by Media ID
	 */
	public static Collection filterQueuesByMediaMap(Collection queues, Map mediaIDs) {
		if (mediaIDs == null || mediaIDs.size() == 0) return queues;

		List newQs = new ArrayList(queues.size());
		for (Iterator it = queues.iterator(); it.hasNext();) {
			Queue q = (Queue) it.next();
			if (mediaIDs.containsKey(q.getMediaID())) {
				newQs.add(q);
			}
		}

		return newQs;
	}
	
	/**
	 * Return Queues filtered by Media ID
	 */
	public static Collection filterQueuesByProcessType(Collection queues) {
		

		List newQs = new ArrayList(queues.size());
		for (Iterator it = queues.iterator(); it.hasNext();) {
			Queue q = (Queue) it.next();
			if (q.getQueueType() != Queue.QUEUE_TYPE_PROCESS)
				newQs.add(q);
		}

		return newQs;
	}

	/**
	 * Helper to create Tree Node based on Queue information
	 */
	public static class QueueTreeInCampaignModeNodeHelper implements TreeNodeHelper {
		private DefaultMutableTreeNode _rootNode;
		private Map _nodeMap = Collections.EMPTY_MAP;
		private Map _parentMap = new HashMap(10);
		private Map _mediaMap = Collections.EMPTY_MAP;
		private boolean _isEmptyRoot = false;
		private boolean _isForSingleCol = true;
		private boolean _isCombinedIDs = false;
		private boolean _isNodeDisplayEnabled = false;
		private boolean _isCombinedQueueSelectable = true;
		private int _combQueueCount = 0;
		private ID _campaignID = null;
		private Collection _dataList = Collections.EMPTY_LIST;
		private Localizer _localizer;
		
		private ID _queueAttributeFilterID = null;

		/**
		 * Constructor
		 */
		public QueueTreeInCampaignModeNodeHelper(Collection nodeList, Map mediaMap, ID campaignID, Localizer localizer,
				boolean isNodeDisplayEnabled, boolean isForSingleCol, boolean isCombinedIDs) {
			this(nodeList, mediaMap, campaignID, localizer, isNodeDisplayEnabled, isForSingleCol, isCombinedIDs, null);
		}
		
		public QueueTreeInCampaignModeNodeHelper(Collection nodeList, Map mediaMap, ID campaignID, Localizer localizer,
				boolean isNodeDisplayEnabled, boolean isForSingleCol, boolean isCombinedIDs, ID queueAttribFilterID) {
			_dataList = nodeList;
			_mediaMap = mediaMap;
			_campaignID = campaignID;
			_localizer = localizer;
			_isNodeDisplayEnabled = isNodeDisplayEnabled;
			_isForSingleCol = isForSingleCol;
			_isCombinedIDs = isCombinedIDs;
			_isCombinedQueueSelectable = (_campaignID!=null);
			_queueAttributeFilterID = queueAttribFilterID;
			buildTreeNodeMap();
		}
		

		/**
		 * Return true if Root is empty
		 */
		public boolean isEmptyRoot() {
			if (_rootNode == null) initRootNode();
			return _isEmptyRoot;
		}

		private BaseNodeData createNodeData(String nodeID, String name, String desc, boolean isSelectable, boolean isSingleSelect) {
			BaseNodeData nodeData;
			if (_isForSingleCol) {
				nodeData = new DefaultNodeData(nodeID, name, desc);
			} else {
				DefaultMultiColumnNodeData mNodeData = new DefaultMultiColumnNodeData(nodeID);
				mNodeData.add(name);
				nodeData = mNodeData;
			}

			if (_isNodeDisplayEnabled) {
				nodeData.getNodeAttributes().put(NODE_DISPLAY, name);
			}

			nodeData.setSelectable(isSelectable);
			nodeData.getNodeAttributes().put(SelectablePC.IS_SINGLE_SELECT, isSingleSelect ? "true" : "false");

			return nodeData;
		}

		/**
		 * Return the number of Combined Queues found
		 */
		public int getCombinedQueueCount() {
			return _combQueueCount;
		}

		/**
		 * Initialize Root Node
		 */
		private void initRootNode() {
			_rootNode = (DefaultMutableTreeNode) _parentMap.get(null);
			int count = 0;
			if (_rootNode == null) {
				// create Media ID Set for Parent Nodes
				HashSet pNodeMediaIDSet = createNonCombQMediaIDSet(_dataList);
				Set keySet = _mediaMap.keySet();
				for (Iterator keys = keySet.iterator(); keys.hasNext();) {
					Media media = (Media) _mediaMap.get(keys.next());
					if (pNodeMediaIDSet.contains(media.getId())) {
						count++;
					}
				}

				String nodeID = _isCombinedIDs ? createCombinedID(_campaignID, null, null, _queueAttributeFilterID, _dataList) : "";
				BaseNodeData nodeData = null;
				if (count > 1){
					String name = MediaUtil.getMediaNodeName(null, _localizer);
					nodeData = createNodeData(nodeID, name, "", _isCombinedQueueSelectable, true);
					nodeData.getNodeAttributes().put("MEDIA_NAME", name);
					_isEmptyRoot = false;
				} else {
					nodeData = createNodeData(nodeID, "", "", _isCombinedQueueSelectable, true);
					_isEmptyRoot = true;
				}

				_rootNode = new DefaultMutableTreeNode(nodeData);
				_parentMap.put(null, _rootNode);
			}
		}

		/**
		 * Return Root Node of the Tree
		 */
		public DefaultMutableTreeNode getRootNode() {
			if (_rootNode == null) initRootNode();
			return _rootNode;
		}

		/**
		 * Return Parent Tree Node for the given Data
		 */
		public DefaultMutableTreeNode getParentNode(Object data) {
			Object node=null;

			Queue queue = (Queue) data;
			ID parentID = queue.getParentID();
			if (parentID!=null && !parentID.equals(queue.getID())) {
				node = _nodeMap.get(parentID);
			}

			// Try to use Media as Parent, but only if there are > 1 queues total to be shown
			if ((node==null) && (_dataList.size() > 1)) 
			{
				Object parentKey = queue.getMediaID();
				node = _parentMap.get(parentKey);

				// Create Parent Media Node Node on the fly
				if (node == null && parentKey != null) 
				{
					Media media = (Media) _mediaMap.get(parentKey);
					node = toTreeNode(media);
                    if (node != null)
                    {
    					_parentMap.put(parentKey, node);
    					getRootNode().add((DefaultMutableTreeNode) node);
                    }
				}
			}

			// Finally just use root node as parent
			if (node == null) node = getRootNode();
			return (DefaultMutableTreeNode) node;
		}

		/**
		 * Return Tree Node for the given Data
		 */
		public DefaultMutableTreeNode getNode(Object data) {
			Queue queue = (Queue) data;
			return (DefaultMutableTreeNode) _nodeMap.get(queue.getID());
		}

		/**
		 * Build Map of Organization Tree Nodes
		 */
		private void buildTreeNodeMap() {
			if (_dataList.isEmpty()) return;

			// create Media ID Set for Parent Nodes
			HashSet pNodeMediaIDSet = createNonCombQMediaIDSet(_dataList);

			initRootNode();

			//put all media ids into the parent map, but only if there are >1 total queues to be shown.
			if (_dataList.size() > 1)
			{
				Collection medias = SortUtil.sortBasicVO(_mediaMap.values(), _localizer);
				for (Iterator it = medias.iterator(); it.hasNext(); ) 
				{
					Media media = (Media)it.next();
					if (pNodeMediaIDSet.contains(media.getId())) 
					{
						_combQueueCount++;
						Object node = toTreeNode(media);
						_parentMap.put(media.getId(), node);
						getRootNode().add((DefaultMutableTreeNode) node);
					}
				}
			}
			
			// Build Tree
			_nodeMap = new HashMap(_dataList.size());
			for (Iterator it = _dataList.iterator(); it.hasNext();) {
				Queue queue = (Queue) it.next();

				// Update Node Map
				DefaultMutableTreeNode node = toTreeNode(queue);
				_nodeMap.put(queue.getID(), node);
			}
		}

		/**
		 * Return HashSet of Media IDs for non combined queues
		 */
		private HashSet createNonCombQMediaIDSet(Collection dataList) {
			HashSet mSet = new HashSet(10);
			mSet.add(null); // Indicate Combined-Combined is a parent

			for (Iterator it = dataList.iterator(); it.hasNext();) {
				Queue queue = (Queue) it.next();

				ID mediaID = queue.getMediaID();
				if (mediaID != null && !mSet.contains(mediaID)) mSet.add(mediaID);
			}

			return mSet;
		}

		private static String createCombinedID(ID campaingID, ID mediaID, ID queueID, ID queueAttribFilterID, Collection dataList)
		{
			if(queueID != null || queueAttribFilterID == null || QueueAttributeFilter.BLANK_ID.equals(queueAttribFilterID))
			{
				return QueueSelection.createCombinedID(campaingID, mediaID, queueID);
			}
			else
			{
				HashSet parentIDs = new HashSet();
				if (dataList != null)
				{
					Iterator qIt = dataList.iterator();
					while (qIt.hasNext())
					{
						Queue q = (Queue) qIt.next();
						if (q.getQueueType()==Queue.QUEUE_TYPE_DISTRIBUTED || q.getQueueType()==Queue.QUEUE_TYPE_VIRTUAL)
							parentIDs.add(q.getID());
					}
				}

				if (mediaID != null)
				{
					if(dataList != null)
					{
						try {
							QueueSelection selection = new QueueSelection();
							selection.setCampaignID(campaingID);
							selection.setMediaID(mediaID);
							Collection<ID> ids = new ArrayList<ID>();
							for(Object q : dataList) {
								ID parentID = ((Queue)q).getParentID();
								if(((Queue)q).getMediaID().equals(mediaID) && (parentID==null || !parentIDs.contains(parentID)))
									ids.add(((Queue)q).getID());
							}
							selection.setQueueIDs(ids);
							return selection.toString();
						} catch (Exception e) {
							// TODO Auto-generated catch block
							e.printStackTrace();
							return QueueSelection.createCombinedID(campaingID, mediaID, queueID);
						}
					}
					else
					{
						return QueueSelection.createCombinedID(campaingID, mediaID, queueID);
					}
				}
				else
				{
					if(dataList != null)
					{
						try
						{
							QueueSelection selection = new QueueSelection();
							selection.setCampaignID(campaingID);
							Collection<ID> ids = new ArrayList<ID>();
							for(Object q : dataList) {
								ID parentID = ((Queue)q).getParentID();
								if (parentID==null || !parentIDs.contains(parentID))
									ids.add(((Queue)q).getID());
							}
							selection.setQueueIDs(ids);
							return selection.toString();
						} catch (Exception e) {
							// TODO Auto-generated catch block
							e.printStackTrace();
							return QueueSelection.createCombinedID(campaingID, mediaID, queueID);
						}
					}
					else
					{
						return QueueSelection.createCombinedID(campaingID, mediaID, queueID);
					}
				}
			}
		}

		/**
		 * Convert Queue Object to MutableTreeNode
		 */
		private DefaultMutableTreeNode toTreeNode(Queue queue) {
			if (queue == null) return null;

			// Generate Node ID
			ID qID = queue.getID();
			String nodeID = null;
			if (_isCombinedIDs) {
				nodeID = createCombinedID(_campaignID, queue.getMediaID(), qID, _queueAttributeFilterID, _dataList);
			} else {
				nodeID = (qID != null) ? qID.toString() : "";
			}

			// Generate Name
			String name = queue.getName();

			// Check for partial assignment
			// if (m_paqSet.contains(qID)) name += m_partiallyAssignImg;

			// Make sure parent queues do support multi select
            boolean isSingleSelect = false;
            //if (queue.getQueueType() == Queue.QUEUE_TYPE_DISTRIBUTED || queue.getQueueType() == Queue.QUEUE_TYPE_VIRTUAL)
            //    isSingleSelect = true;
            
			BaseNodeData nodeData = createNodeData(nodeID, name, queue.getDescription(), true, isSingleSelect);
			nodeData.getNodeAttributes().put("MEDIA_ID", queue.getMediaID().toString());
			return new DefaultMutableTreeNode(nodeData);
		}

		/**
		 * Convert Queue Object to MutableTreeNode
		 */
		private DefaultMutableTreeNode toTreeNode(Media media) {
			if (media == null) return null;

			// Generate Node ID
			String nodeID = null;
			if (_isCombinedIDs) {
				nodeID = createCombinedID(_campaignID, media.getID(), null, _queueAttributeFilterID, _dataList);
			} else {
				nodeID = media.getID().toString();
			}

			String name = MediaUtil.getMediaNodeName(media.getName(), _localizer);
			BaseNodeData nodeData = createNodeData(nodeID, name, media.getDescription(),_isCombinedQueueSelectable, true);
			nodeData.getNodeAttributes().put("MEDIA_NAME", name);

			return new DefaultMutableTreeNode(nodeData);
		}
	}
