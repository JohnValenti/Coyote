package com.verint.test.utils.http;

import com.gargoylesoftware.htmlunit.HttpMethod;
import com.gargoylesoftware.htmlunit.Page;
import com.gargoylesoftware.htmlunit.WebClient;
import com.gargoylesoftware.htmlunit.WebRequest;
import com.gargoylesoftware.htmlunit.html.HtmlPage;
import com.gargoylesoftware.htmlunit.util.NameValuePair;
import com.gargoylesoftware.htmlunit.util.UrlUtils;
import com.verint.test.constants.ParameterKeys;
import com.verint.test.constants.StaticValues;
import com.verint.test.utils.parser.*;
import org.apache.commons.lang3.StringUtils;

import java.util.List;
import java.util.Map;

/**
 * Abstract class which encapsulates request and parse functionality.
 */
public abstract class AbstractHttpRequest implements HttpRequest {

	public abstract PageResult doRequest(WebClient webClient, String requestPath, Map<String, String> parameters, Map<String, Object> parserConfig) throws Exception;

	public abstract PageResult doGetRequest(WebClient webClient, String requestPath, List<NameValuePair> parameters, Map<String, Object> parserConfig) throws Exception;

	public abstract PageResult doPostRequest(WebClient webClient, String requestPath, List<NameValuePair> parameters, Map<String, Object> parserConfig) throws Exception;

	/**
	 * There are limitation for support export operations.
	 * Not all screens support Export, in case when screen request support export operation,
	 * we need to override doExportRequest method for extended class.
	 *
	 * Otherwise in order to prevent call to this method exception is thrown.
	 *
	 * @param webClient
	 * @param requestPath
	 * @param parameters
	 * @return
	 * @throws Exception when screen is not support export operation.
	 */
	public FileResult doExportRequest(WebClient webClient, String requestPath, List<NameValuePair> parameters) throws Exception {
		throw new UnsupportedOperationException(String.format("Export operation is not supported for provided path (%s).", requestPath));
	}

	/**
	 * Make request in old common way through Parameters Map.
	 *
	 *
	 * @param webClient
	 * @param requestPath
	 * @param parameters
	 * @param parserType
	 * @return
	 * @throws Exception
	 */
	protected PageResult doRequest(WebClient webClient, final String requestPath, Map<String, String> parameters, final String parserType) throws Exception {

		//make HTTP request
		HtmlPage page = webClient.getPage(URLBuilderUtils.buildURL(requestPath, parameters));

		//return result
		return this.parsePage(page, parserType);
	}

	/**
	 * Make Post request and parse page in more HttpUnit common way.
	 *
	 *
	 * @param webClient
	 * @param requestPath
	 * @param parameters
	 * @param parserType
	 * @return
	 * @throws Exception
	 */
	public PageResult doGetRequest(WebClient webClient, final String requestPath, List<NameValuePair> parameters, final String parserType) throws Exception {

		//make HTTP request
		HtmlPage page = (HtmlPage)doRequest(webClient, requestPath, parameters, HttpMethod.GET);

		//return result
		return this.parsePage(page, parserType);
	}

	/**
	 * Make Post request and parse page in more HttpUnit common way.
	 *
	 * @param webClient
	 * @param requestPath
	 * @param parameters
	 * @param parserType
	 * @return
	 * @throws Exception
	 */
	public PageResult doPostRequest(WebClient webClient, final String requestPath, List<NameValuePair> parameters, final String parserType) throws Exception {

		//make HTTP request
		HtmlPage page = (HtmlPage)doRequest(webClient, requestPath, parameters, HttpMethod.POST);

		//return result
		return this.parsePage(page, parserType);
	}


	/**
	 * Make Post request and parse page in more HttpUnit common way.
	 *
	 * @param webClient
	 * @param requestPath
	 * @param parameters
	 * @param parserType
	 * @param parserConfig
	 * @return
	 * @throws Exception
	 */
	public PageResult doPostRequest(WebClient webClient, final String requestPath, List<NameValuePair> parameters, final String parserType, Map<String, Object> parserConfig) throws Exception {

		//make HTTP request
		HtmlPage page = (HtmlPage)doRequest(webClient, requestPath, parameters, HttpMethod.POST);

		//return result
		return this.parsePage(page, parserType, parserConfig);
	}

	/**
	 * Make Export request and parse page in more HttpUnit common way.
	 *
	 *
	 * @param webClient
	 * @param requestPath
	 * @param parameters
	 * @param parserType
	 * @return
	 * @throws Exception
	 */
	public FileResult doExportRequest(WebClient webClient, final String requestPath, List<NameValuePair> parameters,
			final String parserType) throws Exception {

		//Export operations must have delimiter in parameters
		NameValuePair delimeterParam = parameters.stream()
				.filter(e-> StringUtils.equals(e.getName(), ParameterKeys.PARAM_DELIMETER))
				.findFirst().orElse(null);
		if (null == delimeterParam) {
			throw new UnsupportedOperationException(String.format("Delimiter have to be provided for Export operation.", requestPath));
		}
		String delimiter = StringUtils.equals(delimeterParam.getValue(), StaticValues.DELIMITER_TAB_STRING) ?
				StaticValues.DELIMITER_TAB_VALUE : delimeterParam.getValue();


		//make HTTP request
		Page page = doRequest(webClient, requestPath, parameters, HttpMethod.POST);

		//Parse File
		ExportFileParser parser =  ExportFileParserFactory.getInstance(parserType);
		FileResult fileResult =  parser.parseFile(page, delimiter);

		return fileResult;
	}

	/**
	 * Make web request through HttpUnit common way.
	 *
	 * @param webClient
	 * @param requestPath
	 * @param parameters
	 * @param method
	 * @return
	 * @throws Exception
	 */
	public Page doRequest(WebClient webClient, final String requestPath, List<NameValuePair> parameters, final HttpMethod method) throws Exception {
		String requestUrl = URLBuilderUtils.buildURL(requestPath);

		WebRequest webRequest = new WebRequest(UrlUtils.toUrlUnsafe(requestUrl), method);
		webRequest.setRequestParameters(parameters);


		//make HTTP request
		return webClient.getPage(webRequest);
	}

	/**
	 * Parse HTML Page with required parser with default configuration.
	 *
	 * @param page
	 * @param parserType
	 * @return
	 * @throws Exception
	 */
	private PageResult parsePage(HtmlPage page, final String parserType) throws Exception {
		HtmlPageParser parser =  HtmlPageParserFactory.getInstance(parserType);
		return parser.parsePage(page);
	}

	/**
	 * Parse HTML Page with required parser with custom configuration.
	 *
	 * @param page
	 * @param parserType
	 * @param parserConfig
	 * @return
	 * @throws Exception
	 */
	private PageResult parsePage(HtmlPage page, final String parserType, Map<String, Object> parserConfig) throws Exception {
		HtmlPageParser parser =  HtmlPageParserFactory.getInstance(parserType, parserConfig);
		return parser.parsePage(page);
	}
}
