package com.bluepumpkin.ejb.am;
import java.util.ResourceBundle;

import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.am.l10n.AmEjbBundleKey;
/**
 * Title:        Log
 * Description:
 * Copyright:    Copyright (c) 2001
 * Company:
 * @author: fbourgeois
 * @version 1.0
 */

public class Log
{
	public static final String VERSION =
			new com.bluepumpkin.common.version.Version().toString();
	public static final String NAME = "ejb.am";

	// Initialize a default resource bundle, just in case localized versions can't be found.
	// This also forces JBuilder to include the resource bundle in the JAR file
	//private static ResourceBundle resourceBundle =new com.bluepumpkin.ejb.am.l10n.AmEjbBundle();
	
	
	
	private static	ResourceBundle resourceBundle = ResourceBundle.getBundle(AmEjbBundleKey.BUNDLE);

	

	public static com.bluepumpkin.common.logging.Category initCategory(String categoryName)
	{
	   ResourceBundle res=ResourceBundle.getBundle(AmEjbBundleKey.BUNDLE);
		//ResourceBundle res =ResourceBundle.getBundle( com.bluepumpkin.ejb.am.l10n.AmEjbBundle.class.getName() );
		if ( res != null )
		{
			Log.resourceBundle = res;
		}
		// Create and configure the category for this log
		com.bluepumpkin.common.logging.Category cat =
											Category.getInstance(categoryName,Log.NAME,Log.VERSION);

		// Set resource bundle for the category
		cat.setResourceBundle(Log.resourceBundle);
		return cat;
	}

	// No instances of this class are allowed.  This class only exists so that we can
	// call its static functions.
	private Log()
	{

	}
}
