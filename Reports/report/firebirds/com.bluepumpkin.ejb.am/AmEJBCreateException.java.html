package com.bluepumpkin.ejb.am;

/**
 * Title:        Blue Pumpkin Software Basic Business Model
 * Description:  AmEJBCreateException is to encapsulate the NamingException, CreateException for AmManagerFactory
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, inc
 * @author       Sheng Song
 * @version 1.0
 */
import com.bluepumpkin.ejb.am.base.*;
import com.bluepumpkin.ejb.am.l10n.*;

public class AmEJBCreateException extends AmException
{
  public AmEJBCreateException(String ejbName, Exception e)
  {
	super(e, AmEjbBundleKey.EJB_CREATE_FAILURE, DEFAULT_SEVCODE);
	Object args[] = new Object[1];
	args[0] = ejbName;
	// initialize AmException's parameter list
	param = args;
  }
}
