	private static final BPMessageFormat JS_MF = new BPMessageFormat(
			"// AppletPC \n" +
			"{0} = new {1}(\"{0}\"); \n" +
			"{0}.bind();\n", 4);
	private static final String JRE_CODEBASE = "jreCodeBase";
	private static final String JRE_MSG = "jreMSG";
	private static final String APPLET_PARAMMAP = "appletParamMap";
	private static final int DEFAULT_WIDTH = 900;
	private static final int DEFAULT_HEIGHT = 500;
	private static final String FALSE_STRING = "false";
	private static final int STATIC_VERSION = 0;
	private static final int DYNAMIC_VERSION = 1;
	
	private static Set<String> RESERVED_PARAM_SET = new HashSet<String>();
	private static Object[] m_jreVersion = new Object[4];
	private static String[] m_classIDs = new String[2];
	private static String[] m_javaPlugins = new String[2];
	private static String m_codebaseAttr;
	static {
		RESERVED_PARAM_SET.add("code");
		RESERVED_PARAM_SET.add("codebase");
		RESERVED_PARAM_SET.add("cache_option");
		RESERVED_PARAM_SET.add("cache_archive");
		RESERVED_PARAM_SET.add("cache_version");
		RESERVED_PARAM_SET.add("type");
		RESERVED_PARAM_SET.add("MAYSCRIPT");
		RESERVED_PARAM_SET.add("scriptable");
		RESERVED_PARAM_SET.add("APPLET_REQUEST");
		RESERVED_PARAM_SET.add("APPLET_CONTEXT_PATH");
		RESERVED_PARAM_SET.add("id");
		RESERVED_PARAM_SET.add("separate_jvm");
		
		// initialize JRE information
		initJREInfo();
	}

	private String m_classID;
	private String m_javaPlugin;
	private ResourceBundle m_core_bundle;

	// List of jar file to use.
	private String m_jarFiles = null;

	// List of jar file versions to use.
	private String m_jarFileVersions = null;

	// Default width and height of the applet.
	// Usually, this is just the default size and reset by the
	// page resize.
	private int m_width = DEFAULT_WIDTH;
	private int m_height = DEFAULT_HEIGHT;

	// Name of the java class to call for the applet
	private String m_appletClass = null;
	private String m_appletJSName;

	// paramaters that must be added to the component HTML tag.
	private HashMap<String, String> m_parameters = new HashMap<String, String>(10);

	/**
	 * Constructor
	 *
	 * @param context The RequestContext
	 * @see RequestContext
	 */
	public AppletPC(RequestContext context) {
		super(context);
		m_core_bundle = m_localizer.getBundle(CoreWebBundleKey.BUNDLE_NAME);
		setName("appletPC");
		addRequiredJavaScriptFile(JavaScriptFileID.UTIL_MESSAGE);
		setAppletJS(WfmJavaScriptFileID.MODULE_APPLET);

		addParameter(PageKeys.LOCALE_ID_FN, m_locale.toString());
		addParameter(PageKeys.TIMEZONE_ID_FN, context.getViewingTimeZone().getID());
		addParameter(PageKeys.APPLET_CONTEXT_PATH, context.getContextPath());
		addParameter(PageKeys.APPLET_JAVA_ARGUMENTS, "-Xmx900m");

		if (useStaticVersioning()) { 
			m_classID = m_classIDs[STATIC_VERSION];
			m_javaPlugin = m_javaPlugins[STATIC_VERSION];			
		} else {
			m_classID = m_classIDs[DYNAMIC_VERSION];
			m_javaPlugin = m_javaPlugins[DYNAMIC_VERSION];
		}
	}

		if ( !StringUtil.isEmpty(name) ) {

	/**
	 * Specify Applet JavaScript File
	 * @param jsFile
	 */
	public void setAppletJS(String jsFile) {
		m_appletJSName = JSUtil.extractJavaScriptFileName(jsFile);
		addRequiredJavaScriptFile(jsFile);
	}

	/**
	 * Return JavaScript Init Code
	 */
	@Override
	public String getJavaScriptInitCode() {
		String[] args = new String[]{getName(), m_appletJSName};
		return JS_MF.format(args);
	}

	/**
	 * Get the Html display - for use by JSP to construct page.
	 */
	@Override
	public String getHtmlDisplay() {
		// http://java.sun.com/j2se/1.4/pdf/plugin-dev-guide-1.4.2.pdf
		addDynamicParameters();

		// need to add this so that applet can get component name - object tag
		// does not allow acces to the attributes.
		addParameter("id",getName());

		LinkedHashMap<String, String> paramMap = new LinkedHashMap<String, String>();
		
		// set the class name
		paramMap.put("code", m_appletClass);
		
		// set context
		String contextPath = m_context.getContextPath();
		paramMap.put("codebase", contextPath);

		// set jar fields
		if ( m_jarFiles != null) {
			paramMap.put("cache_option", "Plugin");
			paramMap.put("cache_archive", m_jarFiles);
			if ( m_jarFileVersions != null) {
				paramMap.put("cache_version", m_jarFileVersions);
			}
		}
		paramMap.put("type", m_javaPlugin);
		paramMap.put("MAYSCRIPT", "true");
		paramMap.put("scriptable", "true");
		paramMap.put("separate_jvm", "true");
		
		// add dynamic params
		paramMap.putAll(m_parameters);

		// jre message
		String jreMsg = m_localizer.i18n(m_core_bundle, CoreWebBundleKey.MSG_INVALID_APPLET_VERSION, m_jreVersion);
		
		// param map size
		int paramMapSize = paramMap.size();
		
		// create necessary js code
		String name = getName();
		StringBuffer sb = new StringBuffer(512);
		sb.append("var ").append(JRE_CODEBASE).append("='").append(contextPath).append(m_codebaseAttr).append("';\n");
		sb.append("var ").append(JRE_MSG).append("=\"").append(jreMsg).append("\";\n");
		
		// create js array for param map
		sb.append("var ").append(APPLET_PARAMMAP).append("= new Array(").append(paramMapSize).append(");\n");		
		int i=0;
		for(Iterator<Map.Entry<String,String>> it=paramMap.entrySet().iterator(); it.hasNext();) {
			Map.Entry<String,String> entry = it.next();
			sb.append(APPLET_PARAMMAP).append("[").append(i++).append("]=new Array('")
				.append(JSUtil.escapeStringLiteral("" + entry.getKey())).append("','").append(JSUtil.escapeStringLiteral("" + entry.getValue())).append("');\n");
		}
		
		// js to create object tag
		sb.append("createObjControl('").append(name)
			.append("','").append(m_classID).append("',").append("'100%'").append(",")
			.append("'100%'").append(",").append(JRE_CODEBASE).append(",")
			.append(JRE_MSG).append(",").append(APPLET_PARAMMAP).append(");");
		
		return HtmlUtil.jsTag(sb.toString(), false);		
	}

	/**
	 *	Adds any parameters to the applet during each instantiation of
	 *	of the applet.
	 */
	protected void addDynamicParameters() {
		boolean b = AppletDebugManager.getInstance().isEnabled(m_context);
		if (b) {
			addParameter(PageKeys.APPLET_DEFAULT_CLIENT_REQUEST, FALSE_STRING);
		}

		if (!log.isDebugEnabled()) {
			addParameter(PageKeys.APPLET_DEBUG, FALSE_STRING);
		}

		String editIndex = (String)m_context.getAttribute(RequestContext.SESSION_SCOPE,
				PageKeys.APPLET_PARAM_EDIT_INDEX);
		if (editIndex != null) {
			addParameter(PageKeys.APPLET_PARAM_EDIT_INDEX, editIndex);
			m_context.removeAttribute(RequestContext.SESSION_SCOPE,
					PageKeys.APPLET_PARAM_EDIT_INDEX);
		}
				
		// Pass URL Parameters to applet as PARAM tags (For example, used by framework forward)
		Map<String, ?> paramMap = ParameterUtil.getParameterMap(m_context.getRequest());
		for(Iterator<String> it=paramMap.keySet().iterator(); it.hasNext(); ) {			
			String key = it.next(); 			
			if (key==null || RESERVED_PARAM_SET.contains(key)) continue;
			
			String value = ParameterUtil.getParameter(paramMap, key);
			addParameter(key, value);
		}

		// The following comment was copied from Pulse.  I don't actually know whether
		// this code needs to be synchronized because it does not contain all the
		// processing that the synchronized Pulse code contains.
		//Needs to be synchronized because the work frame (this class) and the queue  
		//selection frame (QueueSelectionPM) both load at the same time.
		synchronized (m_context.getRequest().getSession())
		{
			// Copy certain parameters from the screen definition to the applet.
			// In the future it might make sense to define a screen parameter that
			// itself lists which screen parameters to copy.
			Screen screen = m_context.getScreen();
			if (screen != null) {
				// ViewType is used to select one of several possible flavors of the applet.
				String viewType = screen.getParameterValue("ViewType");
				if (viewType != null) {
					addParameter("ViewType", viewType);
				}
			}
		}

		//508 Accessibility compliance
        //addParameter("TABINDEX", "1");
	}

	/**
	 *  whether to use static versioning or dynamic versioning.
	 */
	private boolean useStaticVersioning() {
		try {
			DBConfigManager configMgr = BbmManagerFactory.getDBConfigManager();
			String str = configMgr.getValue(ConfigKey.APPLET_AUTOINSTALLJRE);
			if (FALSE_STRING.equalsIgnoreCase(str)) {
				return false;
			}
		} catch (Exception ex) {}
		return true;
	}
	
	// process JRE version form System.properties
	private static void initJREInfo() {
		String jreInfo = SystemProperties.get("JRE");
		log.debug("JRE=" + jreInfo);
		
		if (StringUtil.isEmpty(jreInfo)) {
			jreInfo = "1.5.0.4;jre-1_5_0_04-windows-i586-p.exe"; // NOSONAR
		}
		
		// Initialize Version Information
		StringTokenizer stk = new StringTokenizer(jreInfo, ";");
		String tmpJREversion = stk.nextToken();
		StringTokenizer stk2 = new StringTokenizer(tmpJREversion, ".");
		for(int i=0; i<m_jreVersion.length; i++) {
			m_jreVersion[i] = stk2.nextToken();
		}
		
		// Initialize Code Base Attribute
		String tmpJreFilename = stk.nextToken();
		String codeBaseTemplate = null;
		if (tmpJreFilename.startsWith("http")) {
			// Enable JRE file download from any where in the Web
			codeBaseTemplate = tmpJreFilename;
		} else {
			// Enable JRE file download from within the EAR file
			codeBaseTemplate = "/../clientExeFiles/" + tmpJreFilename;
		}
		codeBaseTemplate+="#Version={0},{1},{2},{3}";
		m_codebaseAttr = BPMessageFormat.format(codeBaseTemplate, m_jreVersion);
		
		// JRE VERSION: 
		// http://java.sun.com/products/plugin/versions.html
		// http://java.sun.com/j2se/1.4.2/docs/guide/plugin/developer_guide/using_tags.html
		
		// m_classID = "clsid:CAFEEFAC-$$$$-$$$$-$$$$-ABCDEFFEDCBA";
		//    where the "$$$$-$$$$-$$$$" segment will need to be replaced by
		//    "0014-0002-0006" => 1.4.2._06 or "0015-0000-0001" => 1.5.0._01
		// m_javaPlugin = "application/x-java-applet;jpi-version=$.$.$.$";
		//    where the "$.$.$.$" should be replaced by the version we are providing.
		// m_codebaseAttr = <the path where we package the installation>
		m_classIDs[STATIC_VERSION] = BPMessageFormat.format("clsid:CAFEEFAC-{0}-{1}-{2}-ABCDEFFEDCBA", getStaticClassIDVersions(m_jreVersion));
		m_javaPlugins[STATIC_VERSION] = BPMessageFormat.format("application/x-java-applet;jpi-version={0}.{1}.{2}.{3}", m_jreVersion);			

		// for dynamic versioning.. the classID NEVER changes
		m_classIDs[DYNAMIC_VERSION] = "clsid:8AD9C840-044E-11D1-B3E9-00805F499D93";
		m_javaPlugins[DYNAMIC_VERSION]= BPMessageFormat.format("application/x-java-applet;version={0}.{1}.{2}.{3}", m_jreVersion);		
	}
	
	private static String[] getStaticClassIDVersions(Object[] jreVersion) {
		String[] staticJREVersions = { jreVersion[0].toString() + jreVersion[1].toString(), jreVersion[2].toString(), jreVersion[3].toString() };
		for(int i=0; i<staticJREVersions.length; i++ ) {
			switch (staticJREVersions[i].length()) {
				case 1:
					staticJREVersions[i] = "000" + staticJREVersions[i];
					break;
				case 2:
					staticJREVersions[i] = "00" + staticJREVersions[i];
					break;
				case 3:
					staticJREVersions[i] = "0" + staticJREVersions[i];
					break;					
				default:
					break;
			}
		}
		return staticJREVersions;
	}
