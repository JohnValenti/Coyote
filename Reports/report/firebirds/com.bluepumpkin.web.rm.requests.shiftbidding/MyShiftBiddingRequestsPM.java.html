package com.bluepumpkin.web.rm.requests.shiftbidding;

import com.witness.web.uif.system.RequestContext;

public class MyShiftBiddingRequestsPM extends MyRequestsPM {
	public static final String FORM_ACTION = "my_shiftbidding_requests";

	/**
	 * Constructor
	 */
	public MyShiftBiddingRequestsPM(RequestContext context) {
		super(context, Request.REQUESTTYPE_SHIFTBID);
	}

	/**
	 * Return an instance of the model that is ready for rendering
	 */
	public static PageModel getInstance(RequestContext context) throws Exception {
		return getInstance(context, MyShiftBiddingRequestsPM.class);
	}

	/**
	 * Populate model with data
	 */
	protected void loadData() throws Exception {
		if (m_empID!=null) {
			m_orgSetting = RequestsMH.getOrgSettingForEmployee(m_context, m_empID);
			// Silk 80212
			// QA39167
			// Modified to invoke the right authorization method
			if (isAuthToViewSBRqts()) {
				m_requestList = ShiftBiddingRequestsMH.getRequestsByEmployee(m_empID, m_isIncludeExpired);
			}
		}
	}

	/**
	 * Initialize Header
	 */
	protected void initHeader() {
		TableHeaderPC header = m_list.getHeader();
		header.setPartialSortable(true);

		addHeaderData(header, m_bundle, RmWebBundleKey.BID_NAME, true);
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.STATUS, true);
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.PREFERENCE, true);
		addHeaderData(header, m_bundle, RmWebBundleKey.USE_POINTS, true);
		addHeaderData(header, m_common_bundle, UIFWebBundleKey.SUBMITTED, true);
		addCommonHeaderData();
	}

	/**
	 * Create Row Data From Request
	 */
	protected MultiColumnNodeData createRowData(RequestAggregate request) {
		ShiftBidRequest sbRequest = (ShiftBidRequest)request;

		String rowID = request.getID().toString();
		DefaultMultiColumnNodeData row_data = new DefaultMultiColumnNodeData(rowID);

		row_data.add( sbRequest.getName() );

		String status = createStatusIcon(request);
		row_data.add(status);

		row_data.add( new NumberWrapper(sbRequest.getPreference(), m_localizer) );
		row_data.add( m_localizer.formatYesNo(sbRequest.getIsBonusUsed()) );
		row_data.add( createDataWrapper(sbRequest.getSubmittedOn()) );

		addCommonListData(request, row_data);
		return row_data;
	}

	/**
	 * Initialize Toolbar
	 */
	protected void initToolbar() {
		String viewLbl = i18n(m_bundle, RmWebBundleKey.VIEW_BIDDING_OPTIONS);
		m_toolbar.addSubmitTextButton(RmPageAction.GO_TO_MY_BID_OPTION, viewLbl, isAuthToEditSBRqts());
	}

	/** Return URL for HTML Form Action
	 */
	public String getFormAction() {
		return FORM_ACTION;
	}
}
