package com.bluepumpkin.ejb.bbm.workrules.model;

import java.util.Date;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;


public class ProjectStart extends ValueObjectBase {


	/**
	 * 
	 */
	private static final long serialVersionUID = 1L;
	private static FieldInfo m_fieldInfo = new ProjectStartFieldInfo();
	protected FieldInfo getFieldInfo() { return m_fieldInfo; }
	
	public enum WeekDay{
		SUNDAY(7),		
		MONDAY(1),
		TUESDAY(2),
		WEDNESDAY(3),
		THURSDAY(4),
		FRIDAY(5),
		SATURDAY(6);
		
		
		private int dayCount =0; 
		public static int WEEK_LENGTH = WeekDay.values().length;
		
		WeekDay(int count){
			this.dayCount = count;
			
		}
		public  void setCount(int count){
			this.dayCount = count;
		}
		public  int  getCount(){
			return this.dayCount ;
		}
	}
	
	public ID getID() {
		return getFieldValueID(ProjectStartFieldInfo.PROJECTSTART_ID);
	}

	public ID getProjectID() {
		return getFieldValueID(ProjectStartFieldInfo.PROJECTSTART_PROJECTID);
	}

	public void setProjectID(ID projectID) { setFieldValue(ProjectStartFieldInfo.PROJECTSTART_PROJECTID, projectID); }
	
	
	/**
	 * Returns the Sequence of the day in a week, 
	 * as the start times for the project can be defined at each week level
	 * @return
	 */
	public int getDaySeq() {
		return getFieldValueInt(ProjectStartFieldInfo.PROJECTSTART_DAYSEQ);
	}
	
	public WeekDay getWeekDay(){
		switch(this.getFieldValueInt(ProjectStartFieldInfo.PROJECTSTART_DAYSEQ)){
			 
			case(1): return WeekDay.MONDAY;
			case(2): return WeekDay.TUESDAY;
			case(3): return WeekDay.WEDNESDAY;
			case(4): return WeekDay.THURSDAY;
			case(5): return WeekDay.FRIDAY;
			case(6): return WeekDay.SATURDAY;
			case(7): return WeekDay.SUNDAY;
			default: return WeekDay.SUNDAY;
		}
		
	}
	
	public void setDaySeq(int daySeq) { 
		setFieldValue(ProjectStartFieldInfo.PROJECTSTART_DAYSEQ, daySeq);
	}
	
	/**
	 * @return minutes since day boundary for the shift's org or campaign
	 */
	public int getPossibleStartTime() {
		return getFieldValueInt(ProjectStartFieldInfo.PROJECTSTART_POSSIBLESTSTARTTIME);
	}
	
	public void setPossibleStartTime(int startTime) { 
		setFieldValue(ProjectStartFieldInfo.PROJECTSTART_POSSIBLESTSTARTTIME, startTime);
	}
	
	public Date getCreatedDate() {
		return getFieldValueDate(ProjectStartFieldInfo.PROJECTSTART_CREATED);
	}
	public void setCreatedDate(Date createdDate) {
		setFieldValue(ProjectStartFieldInfo.PROJECTSTART_CREATED, createdDate);
	}
	
	public String getCreatedBy() {
		return getFormattedFieldValue(ProjectStartFieldInfo.PROJECTSTART_CREATEDBY);
	}
	public void setCreatedBy(String createdBy) {
		setFieldValue(ProjectStartFieldInfo.PROJECTSTART_CREATEDBY, createdBy);
	}
	
	public Date getModifiedDate() {
		return getFieldValueDate(ProjectStartFieldInfo.PROJECTSTART_MODIFIED);
	}
	public void setModifiedDate(Date modifiedDate) {
		setFieldValue(ProjectStartFieldInfo.PROJECTSTART_MODIFIED, modifiedDate);
	}
	
	public String getModifiedBy() {
		return getFormattedFieldValue(ProjectStartFieldInfo.PROJECTSTART_MODIFIEDBY);
	}
	public void setModifiedBy(String modifiedBy) {
		setFieldValue(ProjectStartFieldInfo.PROJECTSTART_MODIFIEDBY, modifiedBy);
	}
}
