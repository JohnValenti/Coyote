package com.bluepumpkin.web.fs.schedule.preferences;

 * Description:  Schedule preference advanced entry.
 * Copyright:    Copyright (c) 2001

 * @author       Chris King
 * @version      1.0
 */
public class AdvancedSchedulePreferencePM extends SchedulePreferencePM {

	/**
	 * Constructs a schedule view page model shell.
	 */
	public AdvancedSchedulePreferencePM(RequestContext context)	{
		super(context, SchedulePreferencePM.VIEW_TYPE_ADVANCED);
	}//constructor


	/**
	 * Return instance of Model which is ready to be rendered
	 */
	public static AdvancedSchedulePreferencePM getInstance(RequestContext context) {
		AdvancedSchedulePreferencePM model = null;

		if (context.getPageModel() instanceof AdvancedSchedulePreferencePM) {
			model = (AdvancedSchedulePreferencePM)context.getPageModel();
		} else {
			model = new AdvancedSchedulePreferencePM(context);
			model.extractParameters(context.getRequest());
			context.setPageModel(model);
		}
		model.populateModel(context);
		return model;
	}//getInstance
