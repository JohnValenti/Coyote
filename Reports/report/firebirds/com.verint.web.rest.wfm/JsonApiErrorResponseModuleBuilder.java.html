package com.verint.web.rest.wfm;

import com.fasterxml.jackson.core.Version;
import com.fasterxml.jackson.databind.module.SimpleModule;
import io.katharsis.jackson.JsonApiModuleBuilder;
import io.katharsis.jackson.serializer.*;

public class JsonApiErrorResponseModuleBuilder extends JsonApiModuleBuilder {

	public SimpleModule buildWithErrorResponseSerializer() {
		SimpleModule simpleModule = new SimpleModule(JSON_API_MODULE_NAME,
				new Version(1, 0, 0, null, null, null));

		simpleModule.addSerializer(new ErrorResponseSerializer());
		return simpleModule;
	}
}
