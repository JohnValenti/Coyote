package com.verint.web.rest.entity.ribbon;

import java.util.ArrayList;
import java.util.List;

import javax.xml.bind.annotation.XmlRootElement;

@XmlRootElement(name = "ribbon")
public class RibbonResponseEntity {
	private final List<RibbonDayEntity> days = new ArrayList<RibbonDayEntity>();
	private boolean overMaxDays;
	private String overMaxDaysMessage;

	public List<RibbonDayEntity> getDays() {
		return days;
	}

	public boolean isOverMaxDays() {
		return overMaxDays;
	}

	public void setOverMaxDays(boolean overMaxDays) {
		this.overMaxDays = overMaxDays;
	}

	public String getOverMaxDaysMessage() {
		return overMaxDaysMessage;
	}

	public void setOverMaxDaysMessage(String overMaxDaysMessage) {
		this.overMaxDaysMessage = overMaxDaysMessage;
	}
}
