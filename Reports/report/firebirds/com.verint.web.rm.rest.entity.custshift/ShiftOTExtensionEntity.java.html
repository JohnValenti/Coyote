package com.verint.web.rm.rest.entity.custshift;

import com.bluepumpkin.ejb.bbm.workrules.model.ShiftOTExtension;
import com.verint.web.rm.rest.entity.Entity;

public class ShiftOTExtensionEntity extends Entity {

	private int activityID;
	private int duration;
	private int minGap;
	private int maxGap;

	public ShiftOTExtensionEntity() {
		super();
	}

	public ShiftOTExtensionEntity(ShiftOTExtension shiftExtension) {
		super();
		setId(shiftExtension.getID().toString());
		setName(shiftExtension.getName());
		if (shiftExtension.getActivityID() != null) {
			activityID = shiftExtension.getActivityID().toInt();
		}
		duration = shiftExtension.getDuration();
		minGap = shiftExtension.getMinGap();
		maxGap = shiftExtension.getMaxGap();
	}

	public int getActivityID() {
		return activityID;
	}

	public void setActivityID(int activityID) {
		this.activityID = activityID;
	}

	public int getDuration() {
		return duration;
	}

	public void setDuration(int duration) {
		this.duration = duration;
	}

	public int getMinGap() {
		return minGap;
	}

	public void setMinGap(int minGap) {
		this.minGap = minGap;
	}

	public int getMaxGap() {
		return maxGap;
	}

	public void setMaxGap(int maxGap) {
		this.maxGap = maxGap;
	}

}
