package com.bluepumpkin.ejb.am.timecollection.model;

import java.util.ArrayList;
import java.util.Collection;
import java.util.Collections;
import java.util.Iterator;
import java.util.List;
import java.util.ListIterator;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.timerecord.model.TimeEntrySourceCode;

/**
 * Title:        Blue Pumpkin Software Activity Management
 *
 * Description:  handler object for RawTimeEntry value objects
 *               sorts and removes them by employee and shift
 *
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */

public class RawTimeEntryHandler {

	List m_rawTimeEntries = null;

	boolean m_needMerge = false;
	int m_mergeWindow = 0;
	/**
	* <B>RawTimeEntryHandler</B>
	* <P>
	*
	*  main constructor
	*
	*  @param rawTimeEntires       collection of RawTimeEntires to handle
	* <P>
	*
	*/
	public RawTimeEntryHandler(Collection rawTimeEntries) {
		if (rawTimeEntries != null && !rawTimeEntries.isEmpty()) {
			m_rawTimeEntries = new ArrayList(rawTimeEntries);
			Collections.sort(m_rawTimeEntries, RawTimeEntrySort.EMPLOYEEID_ORDER);
		}
	}

	public RawTimeEntryHandler(Collection rawTimeEntries, boolean needMerge, int mergeWindow) {
		if (rawTimeEntries != null && !rawTimeEntries.isEmpty()) {
			m_needMerge = needMerge;
			m_mergeWindow = mergeWindow;
			m_rawTimeEntries = new ArrayList(rawTimeEntries);
			Collections.sort(m_rawTimeEntries, RawTimeEntrySort.EMPLOYEEID_ORDER);
		}
	}


	/**
	* <B>nextEmployeeShift</B>
	* <P>
	*
	*  get the next list of RawTimeEntrys for one employee
	*
	* <P>
	*
	*/
	public ArrayList nextEmployeeShift() {
		//-----------------------------------------------------
		//  create return collection
		//  sort list by employees
		//-----------------------------------------------------
		ArrayList employeeShiftData = new ArrayList();

		//-------------------------------------------------------------------------
		//  while we have RawTimeEntry data for this employee
		//  and until we hit the end shift. or a start shift after first entry
		//  remove data and put in return list
		//-------------------------------------------------------------------------
		ID nextEmployeeId = nextEmployeeId();

		if (nextEmployeeId != null) {
			
			for (Iterator i = m_rawTimeEntries.iterator(); i.hasNext(); ) {
				RawTimeEntry rawTimeEntry = (RawTimeEntry)i.next();
				//------------------------------------------------------------------
				//  if this belongs to our employee
				//    add them to our list and remove them from overall list
				//  else we're done
				//------------------------------------------------------------------
				if (rawTimeEntry.getEmployeeID().equals(nextEmployeeId)) {
					employeeShiftData.add(rawTimeEntry);
					i.remove();
				} else {
					break;
				}

				//---------------------------------------------------------------------
				// if we hit a end shift, we're done (after we've added it to our list)
				// if it meets auto merge rule, just continue
				//---------------------------------------------------------------------
				if (rawTimeEntry.isEndShift()) {
					if (!i.hasNext())
						break;
					RawTimeEntry nextEntry = (RawTimeEntry)i.next();
					if (nextEntry.getEmployeeID().equals(nextEmployeeId) && m_needMerge) {
						if ((nextEntry.getStartTime().getTime() - rawTimeEntry.getStartTime().getTime()) < m_mergeWindow) {
							// need perform auto merge
							rawTimeEntry.setActivityId(Activity.ACTIVITY_MERGE);
							rawTimeEntry.setTimeSourceCode(TimeEntrySourceCode.AUTO_MERGE);
							rawTimeEntry.setPunchTimeCode(RawTimeEntry.START_ACTIVITY);
							employeeShiftData.add(nextEntry);
							i.remove();
							continue;
						}
					}					
					break;
				}				
			}
		}
		return employeeShiftData;
	}

	/**
	* <B>nextEmployeeId</B>
	* <P>
	*
	*  gets ID of next employee in handler
	*
	* <P>
	*
	*/
	public ID nextEmployeeId() {
		ID nextEmployeeId = null;

		if (m_rawTimeEntries != null) {
			if (m_rawTimeEntries.size() > 0) {
				RawTimeEntry rawTimeEntry = (RawTimeEntry)m_rawTimeEntries.get(0);
				nextEmployeeId = rawTimeEntry.getEmployeeID();
			}
		}
		return nextEmployeeId;
	}

	/**
	* <B>RawTimeEntryHandler</B>
	* <P>
	*
	*  sets data to be handled
	*
	*  @param rawTimeEntires       collection of RawTimeEntires to handle
	* <P>
	*
	*/
	public void setRawTimeEntries(Collection rawTimeEntries) {
		m_rawTimeEntries = new ArrayList(rawTimeEntries);
		Collections.sort(m_rawTimeEntries, RawTimeEntrySort.EMPLOYEEID_ORDER);
	}
	
	/**
	 * Helper function to remove dup activies (adjacent) and entries within 1ms
	 * @param nextShiftOfRawTimeEntries (sorted ArrayList)
	 * @param FixZeroDuration
	 * @return ArrayList
	 */
	public static ArrayList normalizeEntries(ArrayList nextShiftOfRawTimeEntries, boolean FixZeroDuration) {
		// normalize it, eliminate zero duration, dup activities
		ListIterator rit = nextShiftOfRawTimeEntries.listIterator();
		if (rit.hasNext()) {
			RawTimeEntry pEntry = (RawTimeEntry)rit.next();
			while (rit.hasNext()) {
				RawTimeEntry nEntry = (RawTimeEntry)rit.next();
				// if adjacent entries have same activity, remove the next one
				if (pEntry.getActivityID().equals(nEntry.getActivityID())) {
					rit.remove();
					continue;
				}
				// if next entry is within 1ms, remove it
				if (FixZeroDuration && (nEntry.getSortTime().getTime() - pEntry.getSortTime().getTime() < 1000)) {
					rit.previous();
					rit.previous();
					rit.remove();
					rit.next();
				} 					
				pEntry = nEntry;
			}
		}
		return nextShiftOfRawTimeEntries;
	}
}
