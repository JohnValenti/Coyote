package com.bluepumpkin.ejb.am.timecollection.model;

import java.util.ArrayList;
import java.util.Collection;
import com.bluepumpkin.common.datatypes.ID;
import java.io.IOException;
import java.io.ObjectInput;
import java.io.ObjectOutput;
import java.io.Externalizable;


/**
 * Title:        Blue Pumpkin Software Activity Management
 *
 * Description:  value object representing entry in PunchError table
 *
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, Inc
 * @author
 * @version 1.0
 */

public class PunchError implements Externalizable {
	static final long serialVersionUID = -5102870909869480848L;

	// attributes
	ID m_rawTimeEntryId = null;
	String m_messageKey = null;
	ArrayList m_arguments = null;

	public PunchError() {}
	/**
	 *  constructor - PunchError
	 *
	 *  @param rawTimeEntryId    ID of associated rawtimeentry
	 *  @param messageKey        bundle key of error message
	 *  @param arguments         array list of arguments further defining error
	 */
	public PunchError(ID rawTimeEntryId, String messageKey, ArrayList arguments) {
		this.m_rawTimeEntryId = rawTimeEntryId;
		this.m_messageKey = messageKey;
		this.m_arguments = arguments;
	}

	/**
	 *  constructor - PunchError
	 *
	 *  @param rawTimeEntryId    ID of associated rawtimeentry
	 *  @param messageKey        bundle key of error message
	 */
	public PunchError(ID rawTimeEntryId, String messageKey) {
		this.m_rawTimeEntryId = rawTimeEntryId;
		this.m_messageKey = messageKey;
	}

	/*
	* getters/setters
	*/
	public ID getRawTimeEntryId() { return m_rawTimeEntryId; }
	public void setRawTimeEntryId( ID rawTimeEntryId ) { this.m_rawTimeEntryId = rawTimeEntryId; }

	public String getMessageKey() { return m_messageKey; }
	public void setMessageKey( String messageKey ) { this.m_messageKey = messageKey; }

	public Collection getArguments() { return m_arguments; }
	public void setArguments( ArrayList arguments ) { this.m_arguments = arguments; }
	public void addArgument( String argument ) {
		if( this.m_arguments == null ) {
			this.m_arguments = new ArrayList();
		}
		this.m_arguments.add(argument);
	}

	/**
	* print object variables
	*/
	public String toString () {
		StringBuffer buffer = new StringBuffer();
		buffer.append(super.toString());
		buffer.append("\nPunchError: ");
		buffer.append("\nPunchError.m_rawTimeEntryId: "+m_rawTimeEntryId);
		buffer.append("\nPunchError.m_messageKey: "+m_messageKey);
		buffer.append("\nPunchError.m_arguments: "+m_arguments);

		return buffer.toString();
	}
	/* (non-Javadoc)
	 * @see java.io.Externalizable#readExternal(java.io.ObjectInput)
	 */
	public void readExternal(ObjectInput in)
		throws IOException, ClassNotFoundException {
		m_rawTimeEntryId = (ID) in.readObject();
		m_messageKey = (String) in.readObject();
		m_arguments = (ArrayList) in.readObject();
	}

	/* (non-Javadoc)
	 * @see java.io.Externalizable#writeExternal(java.io.ObjectOutput)
	 */
	public void writeExternal(ObjectOutput out) throws IOException {
		out.writeObject(m_rawTimeEntryId);
		out.writeObject(m_messageKey);
		out.writeObject(m_arguments);
	}
}
