package com.verint.web.rest.wfm.resource.activity;

import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.verint.web.rest.wfm.resource.shiftassignment.ShiftAssignmentMapperDecorator;
import com.verint.web.rest.wfm.util.MapperUtil;
import org.mapstruct.*;
import org.mapstruct.factory.Mappers;

import java.util.Collection;

@Mapper(uses = { MapperUtil.class })
@DecoratedWith(ActivityMapperDecorator.class)
public interface ActivityMapper {

	ActivityMapper INSTANCE = Mappers.getMapper(ActivityMapper.class);

	@Mapping(source = "organizationId", target = "organizationID")
	@Mapping(source = "activityCategoryId", target = "activityCategoryID")
	@Mapping(source = "TOAccrualSchedule", target = "toAccrualSchedule")
	@Mapping(source = "TOAccrualPolicy", target = "toAccrualPolicy")
	ActivityEntity activityToEntity(Activity source);

	@InheritInverseConfiguration
	Activity entityToActivity(ActivityEntity source);

	Collection<ActivityEntity> activitiesToEntities(Collection<Activity> source);
}
