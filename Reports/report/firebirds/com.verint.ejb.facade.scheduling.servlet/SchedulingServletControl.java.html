package com.verint.ejb.facade.scheduling.servlet;

import javax.naming.Context;
import javax.naming.InitialContext;
import javax.transaction.SystemException;
import javax.transaction.UserTransaction;

import com.bluepumpkin.common.base.Log;
import com.bluepumpkin.common.logging.Category;
import com.bluepumpkin.ejb.core.servlet.ServletControl;
import com.verint.ejb.facade.scheduling.util.SchedulingJobUtil;

/**
 * Helper class to register the Quartz job to the application server
 */
public class SchedulingServletControl implements ServletControl {
	private static final Category m_cat = Log.initCategory(SchedulingServletControl.class.getName());

	/*
	 * (non-Javadoc)
	 * 
	 * @see com.bluepumpkin.ejb.core.servlet.ServletControl#startProcess()
	 */
	public void startProcess() {

		UserTransaction jtaUserTX = null;

		try {
			// start a JTA transaction to encompass the database updates done by
			// StdSchedulerFactory.getScheduler() below. These udpates are
			// related to the recovery of
			// jobs and triggers stored in the DB. The quartz scheduler is
			// configured to use JTA
			// transactions (via JobStoreCMT) and if the transaction is not
			// started before
			// StdSchedulerFactory.getScheduler() is called, any SQL updates
			// done by this
			// method will be rolledback (JDBC driver dependent: rolledback for
			// weblogic.mssql4 driver
			// but committed if the newer driver weblogic.sqlserver is used).
			Context ctx = new InitialContext();
			jtaUserTX = (UserTransaction) ctx.lookup("java:comp/UserTransaction");
			jtaUserTX.begin();

			SchedulingJobUtil.addOrUpdateSchedulingJobIfNecessary();
			// call context.close() since an open context affects the security
			// principal associated with
			// this thread (security principals are maintained in the principal
			// stack for each thread).
			ctx.close();

			jtaUserTX.commit();
		} catch (Exception e) {
			try {
				if (jtaUserTX != null) {
					jtaUserTX.rollback();
				}
			} catch (SystemException se) {
				m_cat.error("Cannot rollback transaction", se);
			}

			m_cat.error(e.getMessage(), e);
		}

	}

	/*
	 * (non-Javadoc)
	 * 
	 * @see com.bluepumpkin.ejb.core.servlet.ServletControl#stopProcess()
	 */
	public void stopProcess() {
		// there is no porcessing needed. Have to implement interface method
	}
}
