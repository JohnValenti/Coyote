package com.verint.ejb.bbm.servicegoals.model;

import java.util.Date;

/**
 *  A data item representing deadline goals and deadline time for a specific point
 *  of time for an spqueue.
 */
public class DeadlineGoalsTimeSeriesDataItem extends ServiceGoalsTimeSeriesDataItem {

	private static final long serialVersionUID = -4231139415966940115L;

	/**
	 * Constructs a time series data item reprsenting two points of data at a point of time:
	 * deadline goals and deadline time.
	 * FIXME:  This class was created to fix an issue where deadline time data was not getting saved.
	 * It was discovered that the abstract super class is insufficient in modeling the service goals
	 * data types (specifically, it assumes that the data items only have one point of data but deadline
	 * goals actually has two).  This needs to be refactored.
	 */
	public DeadlineGoalsTimeSeriesDataItem(Date time, double deadlineGoal, double deadlineTime) {
		super(time, deadlineGoal);	//Deadline goal value is set in super constructor, we will set the
									//deadline time manually in the next line.
		setFieldValue(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_ASA, deadlineTime);
	}

	/**
	 * This method returns the deadline goal value for the data item.
	 * FIXME: Note that though this data item actually contains two data points, only one is returned
	 * here as the abstract class definition for this method returns a single double value.  This is a limitation
	 * of the abstract class that needs to be refactored.
	 */
	@Override
	public double getValue() {
		return getFieldValueDouble(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_SLPERCENT);
	}

	/**
	 * Sets the deadline goals value for this data item.  Note that this method does the same thing as
	 * setDeadlineGoal, it is here because of a limitation in the superclass.  FIXME: refactor/remove this
	 */
	@Override
	public void setValue(double value) {
		setFieldValue(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_SLPERCENT, value);	
	}
	
	public void setDeadlineGoal(double value) {
		setFieldValue(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_SLPERCENT, value);
	}
	
	public void setDeadlineTime(double value) {
		setFieldValue(ServiceGoalsTimeSeriesDataItemFieldInfo.SPQUEUEGOALSERIES_ASA, value);
	}
}
