package com.bluepumpkin.ejb.facade.export.ejb;

import com.bluepumpkin.ejb.bbm.dao.DAOBase;
import com.bluepumpkin.ejb.bbm.vo.FieldInfo;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectBase;
import com.bluepumpkin.ejb.facade.export.model.ExportValueObject;

public class ExportDAO extends DAOBase 
{
	private FieldInfo m_info;
	private String m_fromWhereClause = "";
	private String m_selectStmt = "";
	
	protected FieldInfo getFieldInfo() { return this.m_info; }

	public ExportDAO( FieldInfo info,
						String selectStmt,
						String fromWhereClause ) 
	{
		this.m_info = info;
		this.m_fromWhereClause = fromWhereClause;
		this.m_selectStmt = selectStmt;
		this.setFieldInfo(info);
	}

	protected StringBuffer getSelectStatement() 
	{
		if( this.m_selectStmt == null || this.m_selectStmt.length() <= 0)
		{
			return super.getSelectStatement();
		}
		else
		{
			return new StringBuffer(this.m_selectStmt);
		}
		
	}
	
	protected boolean appendFromWhereStatement(StringBuffer strSQL) 
	{
		if( this.m_fromWhereClause == null || this.m_fromWhereClause.length() <= 0)
		{
			return super.appendFromWhereStatement(strSQL);
		}
		else
		{
			strSQL.append(this.m_fromWhereClause);
		}
		
		return true;
	}
	
	protected ValueObjectBase createValueObject()
	{
		return new ExportValueObject(this.m_info);
	}

}
