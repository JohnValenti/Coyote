package com.verint.web.fs.workrules;

import java.util.*;

import com.bluepumpkin.common.datatypes.ID;
import com.bluepumpkin.common.datatypes.PaginationPair;
import com.bluepumpkin.common.datatypes.ScopeID;
import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.ejb.bbm.campaign.model.Campaign;
import com.bluepumpkin.ejb.bbm.vo.ValueObjectIDListComparator;
import com.bluepumpkin.ejb.bbm.workresource.model.Organization;
import com.bluepumpkin.ejb.bbm.workrules.model.ProjectRules;

import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.ejb.core.security.PrivilegeKeys;
import com.witness.ejb.core.security.Scope;
import com.witness.web.uif.base.Message;
import com.witness.web.uif.pagecomponent.DefaultMultiColumnNodeData;
import com.witness.web.uif.pagemodel.PageModel;
import com.witness.web.uif.system.RequestContext;

public class ProjectRulesListPM extends WorkRulesListPM
{
	private static final long serialVersionUID = 1L;
	
	private ArrayList<ProjectRules> projRules = new ArrayList<ProjectRules>();
	private Map<ID, Organization> orgMap = new HashMap<ID, Organization>();
	private Map<ID, Campaign> campaignMap = new HashMap<ID, Campaign>();
	private Map<ID, Activity> activityMap = new HashMap<ID, Activity>();
	
	public ProjectRulesListPM(RequestContext context) {
		super(context);
		removeChildComponent(m_list);
		m_list = new ProjectRulesListPC(
				m_context,
				"projectTable",
				projRules, 
				optimizationTargetTimeZone, 
				selectedSchedulingPeriod == null ? new Date() : selectedSchedulingPeriod.getStartTime(), 
				orgMap,
				campaignMap, 
				activityMap);
		//This needs to be done here so that we can get the sort columns/order
		m_list.extractParameters(context.getRequest());
		this.addChildComponent(m_list);
		this.setIsMultiRowSelectable(true);
	}

	
	@Override
	protected void loadData() throws Exception {
		super.loadData();
		
		if (!hasSelectedID()) return;
		
		// Collections of IDs of objects we will need for display
		Collection<ID> orgIDs = new HashSet<ID>();
		Collection<ID> campaignIDs = new HashSet<ID>();
		Collection<ID> activityIDs = new HashSet<ID>();
		
		// Retrieve projects and add the selected org or campaign into the collection of objects to retrieve 
		if (isByOrg) {
			if (selectedOrg != null) {
				if ( m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID, new ScopeID(selectedOrg.getID(), Scope.ORG_SCOPE))) {
					if (isPaginationNavPressed()) {
						projRules = new ArrayList<ProjectRules>(WorkRulesModelHandler.getProjectRulesByID(m_context, getPageIDs()));
					} else {
						PaginationPair<ProjectRules> projRulesDO = WorkRulesModelHandler.getProjectRulesListByOrg(m_context, getSelectedIDObject(),
								m_list.getHeader().isAscending(), getPageIDSize(), getPagePaginationIndex(),
								m_list.getSortComparators(), null, null);
						projRules = new ArrayList<ProjectRules>(projRulesDO.getValues());
						initPaginationIDs(projRulesDO.getIDs());
					}
				} else {
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE, FsWebBundleKey.BUNDLE_NAME);
				}
			}
		} else {
			if (selectedSchedulingPeriod != null) {
				if (m_context.getUser().isAuthorized(PrivilegeKeys.FS_VIEWWORKRULES_ID)) {
					if (isPaginationNavPressed()) {
						projRules = new ArrayList<ProjectRules>(WorkRulesModelHandler.getProjectRulesByID(m_context, getPageIDs()));
					} else {
						PaginationPair<ProjectRules> projRulesDO = WorkRulesModelHandler.getProjectListBySchedulingPeriod(m_context, 
								selectedSchedulingPeriod, m_list.getHeader().isAscending(), getPageIDSize(),
								getPagePaginationIndex(), m_list.getSortComparators(), null, null);
						
						projRules = new ArrayList<ProjectRules>(projRulesDO.getValues());
						initPaginationIDs(projRulesDO.getIDs());
					}
				} else {
					addPageMessage(Message.INFO_TYPE, FsWebBundleKey.FS_NOT_AUTHORIZED_TO_VIEW_WORK_RULES_IN_THIS_SCOPE, FsWebBundleKey.BUNDLE_NAME);
				}
			}
		}
		
		Collections.sort(projRules, new ValueObjectIDListComparator(new ArrayList<ID>(getPageIDs())));

		// Retrieve the needed org, campaign, and activity IDs from the project rules
		for (ProjectRules projRulesDO : projRules) {
			if (projRulesDO.getOrganizationID() != null) orgIDs.add(projRulesDO.getOrganizationID());
			if (projRulesDO.getCampaignID() != null) campaignIDs.add(projRulesDO.getCampaignID());
			if (projRulesDO.getActivityID() != null) activityIDs.add(projRulesDO.getActivityID());

		}

		orgMap = WorkRulesModelHandler.getOrgs(m_context, orgIDs);
			if (!isByOrg) {
				campaignMap = WorkRulesModelHandler.getCampaigns(m_context, campaignIDs);
			}
			activityMap = WorkRulesModelHandler.getActivities(m_context, activityIDs);
	}

	@Override
	protected void initScreenContent() {
		// Init selectable items first to initialize all of the toggles
		initSelectableItems();
		initHeader();
	}

	@Override
	protected void initHeader() {
	}

	@Override
	protected void initSelectableItems() {
		this.removeChildComponent(m_list);
		m_list = new ProjectRulesListPC(
				m_context,
				"projectTable",
				projRules, 
				optimizationTargetTimeZone, 
				selectedSchedulingPeriod == null ? new Date() : selectedSchedulingPeriod.getStartTime(), 
				orgMap,
				campaignMap, 
				activityMap){
			
			@Override
			protected void setRowAttributes(DefaultMultiColumnNodeData mcnd, ProjectRules projRule) {
				super.setRowAttributes(mcnd, projRule);
				ProjectRulesListPM.this.setRowAttributes(mcnd, projRule.getOrganizationID());
			}
		};

		m_list.extractParameters(m_context.getRequest());
		this.addChildComponent(m_list);
	}
	
	@Override
	protected void initContentTitle() {
		super.initContentTitle();
		
		ID selectedID = getSelectedIDObject();
		if (selectedID != null) {
			if (isByOrg) {
				if (selectedOrg != null) m_contentTitlePC.setItemName(str(selectedOrg.getName()));
				
			} else {
				Campaign c = campaignMap.get(selectedID);
				if (c != null) m_contentTitlePC.setItemName(str(c.getName()));
			}
		}

        ResourceBundle bundle = m_localizer.getBundle(FsWebBundleKey.BUNDLE_NAME);
        String pageTitle = verticalize(bundle, FsWebBundleKey.PROJECT_RULES_TITLE);

		//String pageTitle = str(m_localizer.i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.PROJECT_RULES_TITLE));
		m_contentTitlePC.setPageTitle(pageTitle);
	}

	private String str(Object v) {
		return v == null ? "" : v.toString();
	}
	
	public static ProjectRulesListPM getInstance(RequestContext context) {
		return (ProjectRulesListPM) PageModel.getInstance(context, ProjectRulesListPM.class);
	}
	
	@Override
	protected void initToolbar() {
		super.initToolbar();
		//Setting max page size to 100 to avoid unresponsive script warnings in IE
		//setLimitingPageSize(100);  [Lin Lu commented these out in a chnage for ESR 4203226]
	}
}
