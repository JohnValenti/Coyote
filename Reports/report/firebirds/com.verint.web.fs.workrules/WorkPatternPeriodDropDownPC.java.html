package com.verint.web.fs.workrules;

import com.bluepumpkin.ejb.bbm.workrules.model.ShiftPatternPeriod;
import com.bluepumpkin.web.core.pagecomponent.DropDownSelectionPC;
import com.bluepumpkin.web.fs.l10n.FsWebBundleKey;
import com.witness.web.uif.system.RequestContext;

/**
 * This drop down allows a user to select from one of the possible types of
 * shift pattern periods.  Should only be present on screen if the monthly work
 * rules license is present.
 */
@SuppressWarnings("serial")
public class WorkPatternPeriodDropDownPC extends DropDownSelectionPC<ShiftPatternPeriod> {
	
	public static String WORK_PATTERN_PERIOD_CHANGED_ACTION = "WORK_PATTERN_PERIOD_CHANGED_ACTION";

	public WorkPatternPeriodDropDownPC(RequestContext context, String name) {
		super(context, name);

		this.addOption("7", context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_WORK_PATTERN_PERIOD_WEEK), ShiftPatternPeriod.Week, false);
		this.addOption("28", context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_WORK_PATTERN_PERIOD_28_DAY_MONTH), ShiftPatternPeriod.TwentyEightDayMonth, false);
		this.addOption("29", context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_WORK_PATTERN_PERIOD_29_DAY_MONTH), ShiftPatternPeriod.TwentyNineDayMonth, false);
		this.addOption("30", context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_WORK_PATTERN_PERIOD_30_DAY_MONTH), ShiftPatternPeriod.ThirtyDayMonth, false);
		this.addOption("31", context.getLocalizer().i18n(FsWebBundleKey.BUNDLE_NAME, FsWebBundleKey.FS_WORK_PATTERN_PERIOD_31_DAY_MONTH), ShiftPatternPeriod.ThirtyOneDayMonth, false);
		setUseDefaultOnChange(false);
		setOnChangeJS(getFormSubmitJS());
	}

	public void setSelectedWorkPatternPeriod(ShiftPatternPeriod period) {
		switch (period) {
		case Week:
			setSelectedID("7");
			break;
		case TwentyEightDayMonth:
			setSelectedID("28");
			break;
		case TwentyNineDayMonth:
			setSelectedID("29");
			break;
		case ThirtyDayMonth:
			setSelectedID("30");
			break;
		case ThirtyOneDayMonth:
			setSelectedID("31");
			break;
		}
	}
	
	/**
	 * Right now this component is only used on the WorkPatternFormPM page.  When a selection is made on this drop down,
	 * the page needs to refresh so that the shift container can re-render based on the period that was selected.
	 * We also prompt the user that they will be discarding any unsaved changes.
	 */
	private String getFormSubmitJS() {
		return getName() + ".setInputValue( " + getName() + ".INPUT_PAGE_ACTION, '" + WORK_PATTERN_PERIOD_CHANGED_ACTION + "'); " + 
			getName() + ".setPageDirty(false); " +
			getName() + ".submitForm(); ";
	}
}
