	/**
	 * Process Requests for the Adherence mapping area
	 *
	 * @param the Request Context
	 */
	public void processRequest(RequestContext context) {
		m_amBundle = context.getLocalizer().getBundle(AmWebBundleKey.BUNDLE_NAME);
		m_bbmBundle = context.getLocalizer().getBundle(BbmWebBundleKey.BUNDLE_NAME);

		// Process requested action
		if (isAction(context, PageAction.EDIT)) {
			processListEdit(context);
		//} else if (isAction(context, PageAction.DELETE)) {
		//	processListDelete(context);
		} else if (isAction(context, PageAction.SAVE)) {
			processFormSave(context);
		} else if (isAction(context, PageAction.CANCEL)) {
			processFormCancel(context);
		} else if (isAction(context, PageAction.RESET)) {
			processFormRevert(context);
		} else {
			gotoList(context);
		}
	}



	/**
	 * Process Edit Action from list
	 *
	 * @param the Request Context
	 */
	private void processListEdit(RequestContext context) {
		String selectedRow = (String)context.getRequest()
			.getParameter(AdherenceMappingListPageModel.PERFORM_ACTION_ON_ID_FN);

		if (!StringUtil.isEmpty(selectedRow)) {
			this.gotoForm(context);
		}
		else {
			// Oops... this should never happened. must have selected a row
			// before editing. Log and put a msg in page
			log.l7dError( AmWebLogBundleKey.ADHERENCE_MAPPING_NO_SELECTION_EDIT_FAILURE);
			AdherenceMappingListPageModel listModel =
				AdherenceMappingListPageModel.getInstance(context);
			listModel.addPageMessage(Message.ERROR_TYPE,
				context.getLocalizer().i18n(m_bbmBundle, BbmWebBundleKey.ACTIVITY_SELECT_BEFORE_EDIT));
			this.gotoList(context);
		}
	}

	/**
	 * Process Save Action from form
	 *
	 * @param the Request Context
	 */
	private void processFormSave(RequestContext context) {
		try {
			// update after extracting from the form
			AdherenceMappingFormPageModel formModel =
				AdherenceMappingFormPageModel.getInstance(context );
			String activityIDString = formModel.getActivityID();
			ID activityID = new ID(activityIDString);
			String mapped = formModel.getMappedActivities();
			String nonEditableMapped = formModel.getNotEditableMappedActivities();
			mapped = mapped + "," + nonEditableMapped;
			ID[] mappedIDs = StringUtil.parseIDString(mapped);
			AdherenceMappingModelHandler.updateActivityMapping( activityID, Arrays.asList(mappedIDs) );

			// No exceptions - so prepare to go back to the list
			// Load the List model, Set the edited row to be pre-selected in the list
			AdherenceMappingListPageModel listModel =
				new AdherenceMappingListPageModel(context);
			listModel.setPerformActionOnIDs(new ID[] {activityID});
			// Cache the model and go back to list
			context.setPageModel(listModel);
			this.gotoList(context);

		} catch (Exception e) {
			// Log the message and add it to the page
			log.l7dError( AmWebLogBundleKey.ADHERENCE_MAPPING_UPDATE_FAILURE, e);
			AdherenceMappingFormPageModel formModel =
				AdherenceMappingFormPageModel.getInstance(context);
			formModel.addPageMessage(Message.ERROR_TYPE,
					context.getLocalizer().i18n(m_amBundle,
						AmWebBundleKey.ADHERENCE_MAPPING_UPDATE_FAILURE));
			// Go back to the form
			this.gotoForm(context);
		}

	}







	/**
	 * Process Cancel Action from form
	 *
	 * @param the Request Context
	 */
	private void processFormCancel(RequestContext context) {
		String strID = (String)context.getRequest()
			.getParameter(AdherenceMappingFormPageModel.ID_FN);
		ID id = null;
		AdherenceMappingListPageModel listModel = new AdherenceMappingListPageModel(context);

		// Set the edited row to be pre-selected in the list
		if(!StringUtil.isEmpty(strID)) {
			id = new ID(strID);
			listModel.setPerformActionOnIDs(new ID[] {id});
		}

		// Cache the model and go back to list
		context.setPageModel(listModel);
		this.gotoList(context);
	}



	/**
	 * Process revert Action from form
	 *
	 * @param the Request Context
	 */
	private void processFormRevert(RequestContext context) {
		this.gotoForm(context);
	}




	/**
	 * Go to List
	 *
	 * @param the Request Context
	 */
	private void gotoList(RequestContext context) {
		setNextPage(context, ADHERENCE_MAPPING_LIST_PAGE);
	}


	/**
	 * Go to Form
	 *
	 * @param the Request Context
	 */
	private void gotoForm(RequestContext context) {
		if (isAuthorized(context, CONFIGURE_PRIVILEGE)) {
			setNextPage(context, ADHERENCE_MAPPING_FORM_PAGE);
		}
		else {
			gotoList(context);
		}
	}



}
