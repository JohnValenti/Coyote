package com.bluepumpkin.web.fs.schedule.personal;

/**
 * Title:        PersonalSchedulePM 
 * Description:  Base class for Personal Schedule viewing pages.
 *   Left after refactoring as placeholder for personal schedule specific code 
 * Copyright:    Copyright (c) 2001
 * Company:      Blue Pumpkin Software, inc
 * @author       Chris King, Pavel Bosin
 * @version      1.0
 */
public abstract class PersonalSchedulePM extends ScheduleViewPM {

    protected boolean m_isFullPageMode = true;
    
	/**
	 * Constructs a schedule view base class.
	 */
	public PersonalSchedulePM(RequestContext context, String formAction, String viewType, boolean isDualListUsed){
		super(context, 
                (formAction.equals(GroupMultidayGraphSchedulePM.FORM_ACTION)) ? ("people" + formAction) : formAction, 
                viewType, isDualListUsed);
	}//constructor

    /**
     * Initialize Model with Data
     */
    protected void initializeModel() {
        if (m_isInitialized) return;

        adjustModelTypeFromScreenDef();
        m_isFullPageMode = (BASIC_MODEL == getPageModelType());

        // only the group multiday graph "personal" page would have a selection pane 
        if (!m_isFullPageMode)
            m_formAction = adjustFormActionForMode(GroupMultidayGraphSchedulePM.FORM_ACTION);

        super.initializeModel();
    }
    
    /**
     * Get the ID of the employee whose schedule will be viewed.
     * This will be the logged-in employee ID for normal personal pages, but, for the 
     * GroupMultidayGraphSchedulePM page, it will be the selected employeeID.
     * @return the ID of the employee whose schedule will be viewed.
     */
    protected ID getEmployeeID()
    {
        ID employeeID = null;
        
        if (m_isFullPageMode)
        {
        	employeeID = m_context.getUser().getEmployeeID();
        }
        else
        {
	        Collection employeeIDs = getPageIDs();
	        if (employeeIDs != null)
	        {
	            for (Iterator it = employeeIDs.iterator(); it.hasNext();)
	            {
	                employeeID = (ID) it.next();
	                break;
	            }	            
	        }
        }
        return employeeID;
    }
    
    /**
	 * get time range as this week for the user org.
	 */
	private TimeRange setThisWeekTimeRange()
			throws RemoteException, BbmException 
	{		
		TimeRange timeRange = getThisWeekTimeRange();
		
		m_context.setAttribute(RequestContext.SESSION_SCOPE,
				UserPreferenceKeys.USER_SCHEDULE_VIEW_PERIOD,timeRange);
		return timeRange;
	}
	
	/**
	 * this is called from the base class
	 */
	protected void loadAndInitData() {
		try {
            ID employeeID = getEmployeeID();
            
            Date fromDate = new Date();
            Date toDate = new Date();

            if(this instanceof PersonalTextSchedulePM)//QA100526 Adding -1 and +1 day range to cover the crossing events 
            {
        		//changed for QA101247
            	if (m_timeRange == null)
        		{
        			m_timeRange = setThisWeekTimeRange();
        		}
				fromDate = m_timeRange.getStartDate();
				toDate = m_timeRange.getEndDate();
				LocalDate fromLDate = new LocalDate(getFromDate());
				fromLDate.add(Calendar.DATE, -1);
				LocalDate toLDate = new LocalDate(getToDate());
				toLDate.add(Calendar.DATE, 1);
				m_timeRange.setStartDate(fromLDate.getTime());
				m_timeRange.setEndDate(toLDate.getTime());		
			
			}

            HashMap theData = null;
            if (employeeID != null) //employeeID is null in GroupMultidayGraphSchedulePM if no agent is selected
            	theData = ScheduleViewMH.getPersonalData(m_context, m_viewType, m_timeRange, employeeID, m_context.getUser().getEmployeeID());

			if(this instanceof PersonalTextSchedulePM)//QA100526 removing -1 and +1 day range
			{
				m_timeRange.setStartDate(fromDate);
				m_timeRange.setEndDate(toDate);		
			}
			
			// Make sure Time Range is not null - Use Value specified from MH
			if (m_timeRange == null) refreshTimeRangeFromSession();

			if (employeeID != null)
			{
				populateList(theData);
				initLegend(theData);
			}
		} catch (RmHardValidationException ex) {
			Message msg = handleUnableToLoadDataError(log, ex);
			msg.setAssociatedMessage(ex.getLocalizedMessage(m_localizer, m_context.getViewingTimeZone()));
			ex.printStackTrace();
		} catch(Exception ex) {
			handleUnableToLoadDataError(log, ex);
			ex.printStackTrace();
		}
	}//loadAndInitData

    /**
     * Return Collection of View Selector Options
     */
    protected Collection getViewSelectorOptions() {
        ArrayList options;
        if (m_isFullPageMode)
        {
            options = new ArrayList(2);
            options.add(new StringsPair(
                    PersonalTextSchedulePM.FORM_ACTION,
                    i18n(m_bundle, FsWebBundleKey.SV_TYPE_TEXT)));
            options.add(new StringsPair(
                    PersonalGraphSchedulePM.FORM_ACTION,
                    i18n(m_bundle, FsWebBundleKey.SV_TYPE_GRAPH)));
        }
        else
        {
            options = new ArrayList(3);
            
            options.add(new StringsPair(
                    adjustFormActionForMode(GroupMultidaySchedulePM.FORM_ACTION),
                    i18n(m_bundle, FsWebBundleKey.SV_TYPE_MULTIDAY)));
            options.add(new StringsPair(
                    adjustFormActionForMode(GroupTextSchedulePM.FORM_ACTION),
                    i18n(m_bundle, FsWebBundleKey.SV_TYPE_TEXT)));
            options.add(new StringsPair(
                    adjustFormActionForMode(GroupMultidayGraphSchedulePM.FORM_ACTION),
                    i18n(m_bundle, FsWebBundleKey.SV_TYPE_GRAPH)));
        }
        return options;
    }    
    
    protected String adjustFormActionForMode(String formAction) 
    {
        return (m_isFullPageMode ? formAction : ("people" + formAction));
    }
    
    
	/**
	 * Initialize Content Title
	 */
	protected void initContentTitle()	{
		String pageTitle = i18n(m_bundle, 
                m_isFullPageMode ? FsWebBundleKey.MYSCHEDULE_PERSONAL_PAGETITLE : FsWebBundleKey.PEOPLE_SCHEDULE_SINGLE);
		m_contentTitlePC.setPageTitle(pageTitle);

		if (m_isPageEnabled) {
			//add group and dates selector as an item
			String employeeName = "";
			try {			
                ID employeeID = getEmployeeID();
				employeeName = (employeeID == null) ? "" : ScheduleViewMH.getEmployeeDisplayName(m_context, employeeID, m_localizer);
			} catch (Exception ex) {
				handleUnableToLoadDataError(log, ex);
				ex.printStackTrace();
			} 
			String item = employeeName; // + makeDatesSelector(200);
			m_contentTitlePC.setItemName(item);
            
            //add dates selector as an active component
            m_contentTitlePC.addActiveComponent(makeDatesSelector(10));
            
			//add view selector as an active component
            //add extra space for ESR#4216564
			m_contentTitlePC.addActiveComponent("&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;" + makeViewSelector());
		}
	}//initContentTitle

	/**
	 * Make the Time Interval selection html string
	 */
	private String makeDatesSelector(int leftMargin){
		StringBuffer sb = new StringBuffer();
		sb.append("<span class=\"ctCompLbl\" style=\"margin-left:").append(leftMargin).append("px;\">")
				.append(i18n(m_bundle, FsWebBundleKey.SV_DATES_SELECTOR)).append(": ")
				.append("</span>");
		m_dateRangePicker.getStartDatePickerPC().setDate(m_timeRange.getStartDate());
		m_dateRangePicker.getEndDatePickerPC().setDate(m_timeRange.getEndDate());
		m_dateRangePicker.setOnChangeJS(JSUtil.REFRESH_IGNORE_CHANGES);
		limitDateRangePicker(); //make selectable +/- 30years
		m_dateRangePicker.initForDisplay();
		m_dateRangePicker.getStartDatePickerPC().setInputType(DatePickerPC.INPUT_TYPE_NORMAL);
		m_dateRangePicker.getEndDatePickerPC().setInputType(DatePickerPC.INPUT_TYPE_NORMAL);
        
        m_dateRangePicker.setInputTitles(i18n(m_bundle, FsWebBundleKey.FS_EVENT_STARTDATE), 
                i18n(m_bundle, FsWebBundleKey.FS_EVENT_ENDDATE)); //508
        
		sb.append(m_dateRangePicker.toString());
		return sb.toString();
	}//makeDatesSelector

	//================ abstract ======================
	/** prepare the m_list should be implemented by inheriting classes  */
	protected abstract void populateList(HashMap theData) throws Exception;

}//class PersonalSchedulePM
