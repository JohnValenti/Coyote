package com.verint.web.fs.widgets;

import java.util.Collection;

import com.bluepumpkin.ejb.bbm.activity.model.Activity;
import com.bluepumpkin.web.core.pagecomponent.DropDownSelectionPC;
import com.witness.web.uif.system.RequestContext;

public class ActivitySelectorDDPC extends DropDownSelectionPC<Activity> {
	private static final long serialVersionUID = 1L;

	public ActivitySelectorDDPC(RequestContext context, String name,  Collection<Activity> activities) {
		this(context, name, activities, "");
	}
	
	public ActivitySelectorDDPC(RequestContext context, String name, Collection<Activity> activities, String selectedID) {
		super(context, name);
		for (Activity a : activities) {
			addOption(a.getID().toString(), a.getName(), a, a.getID().toString().equals(selectedID));
		}
	}
}
